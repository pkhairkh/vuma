  2715→impl Backend for PPC64Backend {
  2716→    fn target_info(&self) -> &dyn TargetInfo {
  2717→        &self.target_info
  2718→    }
  2719→
  2720→    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError> {
  2721→        let func_name = func.name.clone();
  2722→
  2723→        // ── Phase 1: Collect all vreg IDs and compute stack layout ──
  2724→        let mut all_vreg_ids: std::collections::HashSet<u32> = std::collections::HashSet::new();
  2725→        for &id in func.vregs.keys() { all_vreg_ids.insert(id); }
  2726→        for param in &func.params {
  2727→            if let Some(id) = param.as_register() { all_vreg_ids.insert(id); }
  2728→        }
  2729→        for block in &func.blocks {
  2730→            for instr in &block.instructions {
  2731→                for id in instr.defined_regs() { all_vreg_ids.insert(id); }
  2732→                for id in instr.used_regs() { all_vreg_ids.insert(id); }
  2733→            }
  2734→        }
  2735→        for val in &func.results {
  2736→            if let Some(id) = val.as_register() { all_vreg_ids.insert(id); }
  2737→        }
  2738→
  2739→        let mut stack_alloc_vregs: std::collections::HashSet<u32> = std::collections::HashSet::new();
  2740→        let mut alloc_sizes: std::collections::HashMap<u32, i32> = std::collections::HashMap::new();
  2741→        for block in &func.blocks {
  2742→            for instr in &block.instructions {
  2743→                if let IRInstr::Alloc { dst, size } = instr {
  2744→                    if let Some(id) = dst.as_register() {
  2745→                        stack_alloc_vregs.insert(id);
  2746→                        let aligned_size = ((*size as i32 + 15) & !15) as i32;
  2747→                        alloc_sizes.insert(id, aligned_size);
  2748→                    }
  2749→                }
  2750→            }
  2751→        }
  2752→
  2753→        // Stack Layout (relative to R31, which = R1 + frame_size):
  2754→        // [R31 + 8]  = saved LR
  2755→        // [R31]      = back chain
  2756→        // [R31 - 16] = saved R31
  2757→        // Then alloc regions, then vreg slots at [R31 - offset]
  2758→
  2759→        let mut alloc_offsets: std::collections::HashMap<u32, i32> = std::collections::HashMap::new();
  2760→        let mut current_offset: i32 = 24; // after saved R31 at -16
  2761→
  2762→        let mut alloc_vreg_ids: Vec<u32> = stack_alloc_vregs.iter().copied().collect();
  2763→        alloc_vreg_ids.sort();
  2764→        for &id in &alloc_vreg_ids {
  2765→            let size = alloc_sizes[&id];
  2766→            current_offset += size;
  2767→            alloc_offsets.insert(id, current_offset);
  2768→        }
  2769→
  2770→        let mut vreg_stack_slots: std::collections::HashMap<u32, i32> = std::collections::HashMap::new();
  2771→        let mut all_vreg_ids_sorted: Vec<u32> = all_vreg_ids.iter().copied().collect();
  2772→        all_vreg_ids_sorted.sort();
  2773→        for &id in &all_vreg_ids_sorted {
  2774→            current_offset += 8;
  2775→            vreg_stack_slots.insert(id, current_offset);
  2776→        }
  2777→
  2778→        let frame_size = ((current_offset + 15) & !15) as usize;
  2779→        let fs = frame_size as i32;
  2780→
  2781→        // ── Phase 2: Emit prologue ──
  2782→        let mut instructions: Vec<AllocatedInstruction> = Vec::new();
  2783→        let mut relocations: Vec<crate::backend::RelocationEntry> = Vec::new();
  2784→
  2785→        // STDU R1, -frame_size(R1)
  2786→        instructions.push(AllocatedInstruction {
  2787→            opcode: "stdu".into(),
  2788→            reads: vec![PhysicalReg::new(RegClass::Gpr, 1)],
  2789→            writes: vec![PhysicalReg::new(RegClass::Gpr, 1)],
  2790→            encoded: Instruction::Stdu { rs: Gpr::R1, ra: Gpr::R1, ds: -fs }.encode().to_vec(),
  2791→        });
  2792→        // MFLR R0
  2793→        let mflr_word: u32 = (31u32 << 26) | (0u32 << 21) | (8u32 << 16) | (339 << 1);
  2794→        instructions.push(AllocatedInstruction {
  2795→            opcode: "mflr".into(), reads: vec![], writes: vec![PhysicalReg::new(RegClass::Gpr, 0)],
  2796→            encoded: encode_word(mflr_word).to_vec(),
  2797→        });
  2798→        // STD R0, fs+8(R1) - save LR
  2799→        instructions.push(AllocatedInstruction {
  2800→            opcode: "std".into(),
  2801→            reads: vec![PhysicalReg::new(RegClass::Gpr, 0), PhysicalReg::new(RegClass::Gpr, 1)],
  2802→            writes: vec![],
  2803→            encoded: Instruction::Std { rs: Gpr::R0, ra: Gpr::R1, ds: fs + 8 }.encode().to_vec(),
  2804→        });
  2805→        // STD R31, fs-16(R1) - save R31
  2806→        instructions.push(AllocatedInstruction {
  2807→            opcode: "std".into(),
  2808→            reads: vec![PhysicalReg::new(RegClass::Gpr, 31), PhysicalReg::new(RegClass::Gpr, 1)],
  2809→            writes: vec![],
  2810→            encoded: Instruction::Std { rs: Gpr::R31, ra: Gpr::R1, ds: fs - 16 }.encode().to_vec(),
  2811→        });
  2812→        // ADDI R31, R1, frame_size
  2813→        instructions.push(AllocatedInstruction {
  2814→            opcode: "addi".into(),
  2815→            reads: vec![PhysicalReg::new(RegClass::Gpr, 1)],
  2816→            writes: vec![PhysicalReg::new(RegClass::Gpr, 31)],
  2817→            encoded: Instruction::Addi { rt: Gpr::R31, ra: Gpr::R1, simm: fs }.encode().to_vec(),
  2818→        });
  2819→
  2820→        // Store function params from R3-R10 to stack slots
  2821→        let arg_regs = [Gpr::R3, Gpr::R4, Gpr::R5, Gpr::R6, Gpr::R7, Gpr::R8, Gpr::R9, Gpr::R10];
  2822→        for (i, param) in func.params.iter().enumerate() {
  2823→            if let Some(id) = param.as_register() {
  2824→                if i < 8 {
  2825→                    let offset = vreg_stack_slots.get(&id).copied().unwrap_or(0);
  2826→                    let store_code = ss_store_to_slot(arg_regs[i], offset);
  2827→                    instructions.push(AllocatedInstruction {
  2828→                        opcode: "std".into(),
  2829→                        reads: vec![PhysicalReg::new(RegClass::Gpr, arg_regs[i].encoding())],
  2830→                        writes: vec![], encoded: store_code,
  2831→                    });
  2832→                }
  2833→            }
  2834→        }
  2835→
  2836→        // ── Phase 3: Emit body ──
  2837→        let mut current_byte_offset: u64 = instructions.iter().map(|i| i.encoded.len() as u64).sum();
  2838→        let mut label_offsets: std::collections::HashMap<String, u64> = std::collections::HashMap::new();
  2839→
  2840→        struct BranchFixup { instr_idx: usize, offset_in_encoded: usize, abs_byte_offset: u64, target_label: String, is_unconditional: bool, bc_bo: u32, bc_bi: u32 }
  2841→        let mut branch_fixups: Vec<BranchFixup> = Vec::new();
  2842→
  2843→        for block in &func.blocks {
  2844→            label_offsets.insert(block.label.clone(), current_byte_offset);
  2845→            for instr in &block.instructions {
  2846→                let encoded: Vec<u8> = match instr {
  2847→                    IRInstr::BinOp { op, dst, lhs, rhs, .. } => {
  2848→                        let dst_id = dst.as_register().unwrap_or(0);
  2849→                        let dst_offset = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
  2850→                        let mut code = Vec::new();
  2851→                        match op {
  2852→                            BinOpKind::Ror | BinOpKind::Rol => {
  2853→                                code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::R4));
  2854→                                code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::R5));
  2855→                                if *op == BinOpKind::Ror {
  2856→                                    code.extend_from_slice(&Instruction::Neg { rt: Gpr::R5, ra: Gpr::R5 }.encode());
  2857→                                    code.extend_from_slice(&Instruction::Addi { rt: Gpr::R5, ra: Gpr::R5, simm: 64 }.encode());
  2858→                                    code.extend_from_slice(&Instruction::Rldcl { ra: Gpr::R3, rs: Gpr::R4, rb: Gpr::R5, mb: 0 }.encode());
  2859→                                } else {
  2860→                                    code.extend_from_slice(&Instruction::Rldcl { ra: Gpr::R3, rs: Gpr::R4, rb: Gpr::R5, mb: 0 }.encode());
  2861→                                }
  2862→                            }
  2863→                            BinOpKind::Shl | BinOpKind::ShrL | BinOpKind::ShrA => {
  2864→                                code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::R4));
  2865→                                code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::R5));
  2866→                                match op {
  2867→                                    BinOpKind::Shl => { code.extend_from_slice(&Instruction::Sld { ra: Gpr::R3, rs: Gpr::R4, rb: Gpr::R5 }.encode()); }
  2868→                                    BinOpKind::ShrL => { code.extend_from_slice(&Instruction::Srd { ra: Gpr::R3, rs: Gpr::R4, rb: Gpr::R5 }.encode()); }
  2869→                                    BinOpKind::ShrA => { code.extend_from_slice(&Instruction::Srad { ra: Gpr::R3, rs: Gpr::R4, rb: Gpr::R5 }.encode()); }
  2870→                                    _ => unreachable!(),
  2871→                                }
  2872→                            }
  2873→                            BinOpKind::SRem | BinOpKind::URem => {
  2874→                                code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::R3));
  2875→                                code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::R4));
  2876→                                code.extend_from_slice(&Instruction::Divd { rt: Gpr::R5, ra: Gpr::R3, rb: Gpr::R4 }.encode());
  2877→                                code.extend_from_slice(&Instruction::Mulld { rt: Gpr::R5, ra: Gpr::R5, rb: Gpr::R4 }.encode());
  2878→                                code.extend_from_slice(&Instruction::Subf { rt: Gpr::R3, ra: Gpr::R5, rb: Gpr::R3 }.encode());
  2879→                            }
  2880→                            _ => {
  2881→                                code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::R3));
  2882→                                code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::R4));
  2883→                                match op {
  2884→                                    BinOpKind::Add => { code.extend_from_slice(&Instruction::Add { rt: Gpr::R3, ra: Gpr::R3, rb: Gpr::R4 }.encode()); }
  2885→                                    BinOpKind::Sub => { code.extend_from_slice(&Instruction::Subf { rt: Gpr::R3, ra: Gpr::R4, rb: Gpr::R3 }.encode()); }
  2886→                                    BinOpKind::Mul => { code.extend_from_slice(&Instruction::Mulld { rt: Gpr::R3, ra: Gpr::R3, rb: Gpr::R4 }.encode()); }
  2887→                                    BinOpKind::SDiv | BinOpKind::UDiv => { code.extend_from_slice(&Instruction::Divd { rt: Gpr::R3, ra: Gpr::R3, rb: Gpr::R4 }.encode()); }
  2888→                                    BinOpKind::And => { code.extend_from_slice(&Instruction::And { ra: Gpr::R3, rs: Gpr::R3, rb: Gpr::R4 }.encode()); }
  2889→                                    BinOpKind::Or => { code.extend_from_slice(&Instruction::Or { ra: Gpr::R3, rs: Gpr::R3, rb: Gpr::R4 }.encode()); }
  2890→                                    BinOpKind::Xor => { code.extend_from_slice(&Instruction::Xor { ra: Gpr::R3, rs: Gpr::R3, rb: Gpr::R4 }.encode()); }
  2891→                                    BinOpKind::Eq|BinOpKind::Ne|BinOpKind::SLt|BinOpKind::SLe|BinOpKind::SGt|BinOpKind::SGe|BinOpKind::ULt|BinOpKind::ULe|BinOpKind::UGt|BinOpKind::UGe => {
  2892→                                        let cmp_kind = match op {
  2893→                                            BinOpKind::Eq => CmpKind::Eq,
  2894→                                            BinOpKind::Ne => CmpKind::Ne,
  2895→                                            BinOpKind::SLt => CmpKind::SLt,
  2896→                                            BinOpKind::SLe => CmpKind::SLe,
  2897→                                            BinOpKind::SGt => CmpKind::SGt,
  2898→                                            BinOpKind::SGe => CmpKind::SGe,
  2899→                                            BinOpKind::ULt => CmpKind::ULt,
  2900→                                            BinOpKind::ULe => CmpKind::ULe,
  2901→                                            BinOpKind::UGt => CmpKind::UGt,
  2902→                                            BinOpKind::UGe => CmpKind::UGe,
  2903→                                            _ => CmpKind::Eq,
  2904→                                        };
  2905→                                        code.extend(ss_emit_cmp(&cmp_kind, Gpr::R3, Gpr::R3, Gpr::R4));
  2906→                                    }
  2907→                                    _ => { code.extend_from_slice(&Instruction::Add { rt: Gpr::R3, ra: Gpr::R3, rb: Gpr::R4 }.encode()); }
  2908→                                }
  2909→                            }
  2910→                        }
  2911→                        code.extend(ss_store_to_slot(Gpr::R3, dst_offset));
  2912→                        code
  2913→                    }
  2914→                    IRInstr::Add { dst, lhs, rhs, .. } => {
  2915→                        let dst_id = dst.as_register().unwrap_or(0);
  2916→                        let dst_offset = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
  2917→                        let mut code = Vec::new();
  2918→                        code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::R3));
  2919→                        if let IRValue::Immediate(imm) = rhs {
  2920→                            let i = *imm as i32;
  2921→                            if (-32768..=32767).contains(&i) {
  2922→                                code.extend_from_slice(&Instruction::Addi { rt: Gpr::R3, ra: Gpr::R3, simm: i }.encode());
  2923→                            } else {
  2924→                                code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::R4));
  2925→                                code.extend_from_slice(&Instruction::Add { rt: Gpr::R3, ra: Gpr::R3, rb: Gpr::R4 }.encode());
  2926→                            }
  2927→                        } else {
  2928→                            code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::R4));
  2929→                            code.extend_from_slice(&Instruction::Add { rt: Gpr::R3, ra: Gpr::R3, rb: Gpr::R4 }.encode());
  2930→                        }
  2931→                        code.extend(ss_store_to_slot(Gpr::R3, dst_offset));
  2932→                        code
  2933→                    }
  2934→                    IRInstr::Sub { dst, lhs, rhs, .. } => {
  2935→                        let dst_id = dst.as_register().unwrap_or(0);
  2936→                        let dst_offset = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
  2937→                        let mut code = Vec::new();
  2938→                        code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::R3));
  2939→                        code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::R4));
  2940→                        code.extend_from_slice(&Instruction::Subf { rt: Gpr::R3, ra: Gpr::R4, rb: Gpr::R3 }.encode());
  2941→                        code.extend(ss_store_to_slot(Gpr::R3, dst_offset));
  2942→                        code
  2943→                    }
  2944→                    IRInstr::Mul { dst, lhs, rhs, .. } => {
  2945→                        let dst_id = dst.as_register().unwrap_or(0);
  2946→                        let dst_offset = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
  2947→                        let mut code = Vec::new();
  2948→                        code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::R3));
  2949→                        code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::R4));
  2950→                        code.extend_from_slice(&Instruction::Mulld { rt: Gpr::R3, ra: Gpr::R3, rb: Gpr::R4 }.encode());
  2951→                        code.extend(ss_store_to_slot(Gpr::R3, dst_offset));
  2952→                        code
  2953→                    }
  2954→                    IRInstr::Div { dst, lhs, rhs, .. } => {
  2955→                        let dst_id = dst.as_register().unwrap_or(0);
  2956→                        let dst_offset = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
  2957→                        let mut code = Vec::new();
  2958→                        code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::R3));
  2959→                        code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::R4));
  2960→                        code.extend_from_slice(&Instruction::Divd { rt: Gpr::R3, ra: Gpr::R3, rb: Gpr::R4 }.encode());
  2961→                        code.extend(ss_store_to_slot(Gpr::R3, dst_offset));
  2962→                        code
  2963→                    }
  2964→                    IRInstr::UnaryOp { op, dst, operand, .. } => {

  2966→                        let dst_offset = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);

  2982→                        code.extend(ss_store_to_slot(Gpr::R3, dst_offset));

  2984→                    }

  2987→                        let dst_offset = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);

  2989→                        code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::R3));

  3025→                    }

  3029→                    }

  3033→                        let mut code = Vec::new();
  3034→                        code.extend(ss_load_value(src, &vreg_stack_slots, Gpr::R3));
  3035→                        if let CastKind::SExt = kind {

  3037→                        }
  3038→                        code.extend(ss_store_to_slot(Gpr::R3, dst_offset));
  3039→                        code
  3040→                    }
  3041→                    IRInstr::Select { dst, cond, true_val, false_val, .. } => {

  3043→                        let dst_offset = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
