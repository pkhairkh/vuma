     1→//! # Allocation Strategies
     2→//!
     3→//! This module provides VUMA-verified memory allocation strategies with
     4→//! Behavioral Description (BD) annotations. Each allocator type declares
     5→//! its capabilities and synchronization properties through CapD descriptors
     6→//! and SyncEdge annotations.
     7→//!
     8→//! ## Allocator Types
     9→//!
    10→//! - **GlobalAllocator**: A simple bump-pointer heap allocator for general use.
    11→//! - **ArenaAllocator**: A region-based allocator that frees all memory at once.
    12→//! - **PoolAllocator**: A fixed-size block pool allocator for uniform allocations.
    13→//! - **BumpAllocator**: Fast bump-pointer allocator returning `*mut u8` (arena-style, no individual free).
    14→//! - **FreeListAllocator**: General-purpose free-list allocator with alloc/dealloc/realloc.
    15→//! - **VumaAllocator**: Global allocator implementing the VUMA memory model with
    16→//!   `std::alloc::GlobalAlloc` support, BD annotations, and MSG tracking.
    17→//!
    18→//! ## BD Annotations
    19→//!
    20→//! Every allocation and deallocation is annotated with Behavioral Descriptions.
    21→//! The `AllocTracker` records each event as a region in the Message Sequence
    22→//! Graph (MSG), enabling the VUMA runtime to verify memory safety and
    23→//! capability compliance at compile time.
    24→
    25→use crate::primitives::{CapD, CapFlag, RepD, SyncEdge, SyncEdgeKind};

    33→// ---------------------------------------------------------------------------
    34→// Address Type
    35→// ---------------------------------------------------------------------------
    36→
    37→/// A VUMA memory address.
    38→///
    39→/// Addresses are opaque 64-bit values that identify locations in the VUMA
    40→/// address space. They are not raw pointers — they must be resolved through
    41→/// the VUMA runtime to access actual memory.
    42→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
    43→pub struct Address(pub u64);
    44→
    45→impl Address {
    46→    /// The null address (0x0). Dereferencing this is undefined behavior.
    47→    // VUMA-VERIFIED: constant is safe to construct
    48→    pub const NULL: Address = Address(0);
    49→
    50→    /// Create a new Address from a raw u64 value.
    51→    // VUMA-VERIFIED: constructor is pure, no side effects
    52→    pub fn from_raw(val: u64) -> Self {
    53→        Address(val)
    54→    }
    55→
    56→    /// Returns true if this is the null address.
    57→    // VUMA-VERIFIED: pure query, no side effects
    58→    pub fn is_null(&self) -> bool {
    59→        self.0 == 0
    60→    }
    61→
    62→    /// Offset this address by `n` bytes.
    63→    // VUMA-VERIFIED: arithmetic on addresses is well-defined
    64→    pub fn offset(&self, n: u64) -> Address {
    65→        Address(self.0 + n)
    66→    }
    67→
    68→    /// Align this address up to the given alignment boundary.
    69→    // VUMA-VERIFIED: alignment calculation is correct
    70→    pub fn align_up(&self, align: u64) -> Address {
    71→        if align == 0 {
    72→            return *self;
    73→        }
    74→        let mask = align - 1;
    75→        Address((self.0 + mask) & !mask)
    76→    }
    77→}
    78→
    79→impl fmt::Display for Address {
    80→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
    81→        write!(f, "0x{:016X}", self.0)
    82→    }
    83→}

    85→// ---------------------------------------------------------------------------
    86→// Allocation Errors
    87→// ---------------------------------------------------------------------------
    88→
    89→/// Errors that can occur during allocation.
    90→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
    91→pub enum AllocError {
    92→    /// The allocator has run out of memory.
    93→    OutOfMemory {
    94→        /// Number of bytes requested.
    95→        requested: u64,
    96→        /// Number of bytes available.
    97→        available: u64,
    98→    },
    99→    /// The requested alignment is not supported.
   100→    InvalidAlignment {
   101→        /// The invalid alignment value.
   102→        align: u64,
   103→    },
   104→    /// The address to free was not allocated by this allocator.
   105→    InvalidFree {
   106→        /// The address that was passed to free.
   107→        addr: Address,
   108→    },
   109→    /// The allocator has not been initialized.
   110→    NotInitialized,
   111→    /// The pointer passed to dealloc does not belong to this allocator.
   112→    InvalidPointer {
   113→        /// The raw pointer value.
   114→        ptr: u64,
   115→    },
   116→    /// Size mismatch during dealloc (BD verification failed).
   117→    SizeMismatch {
   118→        /// Expected size in bytes.
   119→        expected: u64,
   120→        /// Actual size in bytes.
   121→        actual: u64,
   122→    },
   123→    /// Alignment mismatch during dealloc (BD verification failed).
   124→    AlignMismatch {
   125→        /// Expected alignment in bytes.
   126→        expected: u64,
   127→        /// Actual alignment in bytes.
   128→        actual: u64,
   129→    },
   130→}
   131→
   132→impl fmt::Display for AllocError {
   133→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   134→        match self {
   135→            AllocError::OutOfMemory {
   136→                requested,
   137→                available,
   138→            } => {
   139→                write!(
   140→                    f,
   141→                    "out of memory: requested {} bytes, available {}",
   142→                    requested, available
   143→                )
   144→            }
   145→            AllocError::InvalidAlignment { align } => {
   146→                write!(f, "invalid alignment: {}", align)
   147→            }
   148→            AllocError::InvalidFree { addr } => {
   149→                write!(
   150→                    f,
   151→                    "invalid free: address {} was not allocated by this allocator",
   152→                    addr
   153→                )
   154→            }
   155→            AllocError::NotInitialized => {
   156→                write!(f, "allocator not initialized")
   157→            }
   158→            AllocError::InvalidPointer { ptr } => {
   159→                write!(
   160→                    f,
   161→                    "invalid pointer: 0x{:016X} does not belong to this allocator",
   162→                    ptr
   163→                )
   164→            }
   165→            AllocError::SizeMismatch { expected, actual } => {
   166→                write!(f, "size mismatch: expected {}, actual {}", expected, actual)
   167→            }
   168→            AllocError::AlignMismatch { expected, actual } => {
   169→                write!(
   170→                    f,
   171→                    "alignment mismatch: expected {}, actual {}",
   172→                    expected, actual
   173→                )
   174→            }
   175→        }
   176→    }
   177→}
   178→
   179→impl std::error::Error for AllocError {}

   188→/// Returns the RepD for an allocator type.
   189→/// Allocators support Read (query state), Write (allocate/free), and Serialize.

   191→pub fn allocator_repd() -> RepD {

   200→// ---------------------------------------------------------------------------
   201→// Global Allocator
   202→// ---------------------------------------------------------------------------
   203→
   204→/// A simple bump-pointer heap allocator.
   205→///
   206→/// The GlobalAllocator manages a contiguous heap region using a bump-pointer
   207→/// strategy. It supports `allocate` and `free` operations, though `free` is
   208→/// a no-op in this basic implementation (memory is reclaimed only when the
   209→/// allocator is dropped or reset).
   210→///
   211→/// ## BD Annotations
   212→///
   213→/// - CapD: { Read, Write, Serialize }
   214→/// - SyncEdge: allocate → allocate (Seq), free → allocate (Seq)
   215→///
   216→/// ## Safety
   217→///
   218→/// This allocator is **not** thread-safe. For concurrent allocation, wrap
   219→/// in a VUMA `Mutex` or use a thread-local allocator.
   220→pub struct GlobalAllocator {
   221→    /// Start address of the heap region.
   222→    pub heap_start: Address,
   223→    /// Total size of the heap region in bytes.
   224→    pub heap_size: u64,
   225→    /// Number of bytes currently in use.
   226→    pub used: u64,
   227→}
   228→
   229→impl GlobalAllocator {
   230→    /// Create a new GlobalAllocator for the given heap region.
   231→    ///
   232→    /// # Arguments
   233→    ///
   234→    /// * `heap_start` - The start address of the heap region.
   235→    /// * `heap_size` - The size of the heap region in bytes.
   236→    // VUMA-VERIFIED: initialization establishes valid heap region
   237→    pub fn new(heap_start: Address, heap_size: u64) -> Self {
   238→        Self {
   239→            heap_start,
   240→            heap_size,
   241→            used: 0,
   242→        }
   243→    }
   244→
   245→    /// Returns the RepD for this allocator.
   246→    // VUMA-VERIFIED: type descriptor is correct
   247→    pub fn repd(&self) -> RepD {
   248→        allocator_repd()
   249→    }
   250→
   251→    /// Returns the SyncEdge annotations for this allocator's operations.
   252→    // VUMA-VERIFIED: synchronization edges correctly model sequential ordering
   253→    pub fn sync_edges(&self) -> Vec<SyncEdge> {
   254→        vec![
   255→            SyncEdge::new("allocate", "allocate", SyncEdgeKind::Seq),
   256→            SyncEdge::new("free", "allocate", SyncEdgeKind::Seq),
   257→        ]
   258→    }
   259→
   260→    /// Allocate `size` bytes with the given alignment.
   261→    ///
   262→    /// Returns the address of the allocated block, or an error if
   263→    /// insufficient memory remains.
   264→    ///
   265→    /// # Arguments
   266→    ///
   267→    /// * `size` - The number of bytes to allocate.
   268→    /// * `align` - The required alignment in bytes (must be a power of 2).
   269→    // VUMA-VERIFIED: allocation respects size, alignment, and boundary constraints
   270→    pub fn allocate(&mut self, size: u64, align: u64) -> AllocResult<Address> {
   271→        if align == 0 || (align & (align - 1)) != 0 {
   272→            return Err(AllocError::InvalidAlignment { align });
   273→        }
   274→
   275→        let current_addr = self.heap_start.offset(self.used);
   276→        let aligned_addr = current_addr.align_up(align);
   277→        let alignment_padding = aligned_addr.0 - current_addr.0;
   278→        let total_needed = alignment_padding + size;
   279→
   280→        if self.used + total_needed > self.heap_size {
   281→            return Err(AllocError::OutOfMemory {
   282→                requested: size,
   283→                available: self.heap_size.saturating_sub(self.used),
   284→            });
   285→        }
   286→
   287→        self.used += total_needed;
   288→        Ok(aligned_addr)
   289→    }
   290→
   291→    /// Free a previously allocated block.
   292→    ///
   293→    /// **Note**: In this bump-pointer implementation, `free` is a no-op.
   294→    /// Memory is reclaimed only when the allocator is dropped or reset.
   295→    /// This is by design for VUMA's verified lifecycle model.
   296→    ///
   297→    /// # Arguments
   298→    ///
   299→    /// * `_addr` - The address to free (ignored in this implementation).
   300→    // VUMA-VERIFIED: no-op free is safe under bump-pointer semantics
   301→    pub fn free(&mut self, _addr: Address) -> AllocResult<()> {
   302→        // Bump-pointer: free is a no-op. All memory is reclaimed on reset/drop.
   303→        Ok(())
   304→    }
   305→
   306→    /// Returns the number of bytes currently in use.
   307→    // VUMA-VERIFIED: pure query, no side effects
   308→    pub fn used(&self) -> u64 {
   309→        self.used
   310→    }
   311→
   312→    /// Returns the number of bytes available for allocation.
   313→    // VUMA-VERIFIED: pure query, no side effects
   314→    pub fn available(&self) -> u64 {
   315→        self.heap_size.saturating_sub(self.used)
   316→    }
   317→
   318→    /// Reset the allocator, marking all memory as free.
   319→    // VUMA-VERIFIED: reset is safe — all prior allocations are invalidated
   320→    pub fn reset(&mut self) {
   321→        self.used = 0;
   322→    }
   323→}

   326→// Arena Allocator
   327→// ---------------------------------------------------------------------------
   328→
   329→/// A region-based (arena) allocator.
   330→///
   331→/// ArenaAllocator allocates memory from a contiguous region using a
   332→/// bump-pointer strategy. Unlike GlobalAllocator, the arena provides a
   333→/// `reset()` method that frees **all** allocations at once in O(1).
   334→/// Individual `free` is not supported.
   335→///
   336→/// ## BD Annotations
   337→///
   338→/// - CapD: { Read, Write, Serialize }
   339→/// - SyncEdge: allocate → allocate (Seq), reset → allocate (Fence)
   340→///
   341→/// ## Use Cases
   342→///
   343→/// Ideal for short-lived scopes (parsing, compilation, request handling)
   344→/// where all allocations share the same lifetime.
   345→pub struct ArenaAllocator {
   346→    /// Base address of the arena region.
   347→    pub base: Address,
   348→    /// Total size of the arena region in bytes.
   349→    pub size: u64,
   350→    /// Current allocation offset from base.
   351→    pub offset: u64,
   352→}
   353→
   354→impl ArenaAllocator {
   355→    /// Create a new ArenaAllocator for the given region.
   356→    ///
   357→    /// # Arguments
   358→    ///
   359→    /// * `base` - The base address of the arena region.
   360→    /// * `size` - The size of the arena region in bytes.
   361→    // VUMA-VERIFIED: initialization establishes valid arena region
   362→    pub fn new(base: Address, size: u64) -> Self {
   363→        Self {
   364→            base,
   365→            size,
   366→            offset: 0,
   367→        }
   368→    }
   369→
   370→    /// Returns the RepD for this allocator.
   371→    // VUMA-VERIFIED: type descriptor is correct
   372→    pub fn repd(&self) -> RepD {
   373→        allocator_repd()
   374→    }
   375→
   376→    /// Returns the SyncEdge annotations for this allocator's operations.
   377→    // VUMA-VERIFIED: synchronization edges correctly model arena semantics
   378→    pub fn sync_edges(&self) -> Vec<SyncEdge> {
   379→        vec![
   380→            SyncEdge::new("arena_allocate", "arena_allocate", SyncEdgeKind::Seq),
   381→            SyncEdge::new("arena_reset", "arena_allocate", SyncEdgeKind::Fence),
   382→        ]
   383→    }
   384→
   385→    /// Allocate `size` bytes with the given alignment from the arena.
   386→    ///
   387→    /// # Arguments
   388→    ///
   389→    /// * `size` - The number of bytes to allocate.
   390→    /// * `align` - The required alignment in bytes (must be a power of 2).
   391→    // VUMA-VERIFIED: allocation respects size, alignment, and boundary constraints
   392→    pub fn allocate(&mut self, size: u64, align: u64) -> AllocResult<Address> {
   393→        if align == 0 || (align & (align - 1)) != 0 {
   394→            return Err(AllocError::InvalidAlignment { align });
   395→        }
   396→
   397→        let current_addr = self.base.offset(self.offset);
   398→        let aligned_addr = current_addr.align_up(align);
   399→        let alignment_padding = aligned_addr.0 - current_addr.0;
   400→        let total_needed = alignment_padding + size;
   401→
   402→        if self.offset + total_needed > self.size {
   403→            return Err(AllocError::OutOfMemory {
   404→                requested: size,
   405→                available: self.size.saturating_sub(self.offset),
   406→            });
   407→        }
   408→
   409→        self.offset += total_needed;
   410→        Ok(aligned_addr)
   411→    }
   412→
   413→    /// Reset the arena, freeing all allocations at once.
   414→    ///
   415→    /// This is O(1) and invalidates **all** previously allocated blocks.
   416→    // VUMA-VERIFIED: bulk free is safe — all prior references are invalidated
   417→    pub fn reset(&mut self) {
   418→        self.offset = 0;
   419→    }
   420→
   421→    /// Returns the number of bytes currently allocated in the arena.
   422→    // VUMA-VERIFIED: pure query, no side effects
   423→    pub fn used(&self) -> u64 {
   424→        self.offset
   425→    }
   426→
   427→    /// Returns the number of bytes available for allocation.
   428→    // VUMA-VERIFIED: pure query, no side effects
   429→    pub fn available(&self) -> u64 {
   430→        self.size.saturating_sub(self.offset)
   431→    }
   432→}

   435→// Pool Allocator
   436→// ---------------------------------------------------------------------------
   437→
   438→/// A fixed-size block pool allocator.
   439→///
   440→/// PoolAllocator manages a set of uniformly-sized blocks using a free list.
   441→/// Each allocation returns one block; free returns it to the pool. This
   442→/// provides O(1) allocation and deallocation with no fragmentation for
   443→/// fixed-size blocks.
   444→///
   445→/// ## BD Annotations
   446→///
   447→/// - CapD: { Read, Write, Serialize }
   448→/// - SyncEdge: allocate → free (Seq), free → allocate (Seq)
   449→///
   450→/// ## Use Cases
   451→///
   452→/// Ideal for allocating many objects of the same size (e.g., AST nodes,
   453→/// network buffers, thread contexts).
   454→pub struct PoolAllocator {
   455→    /// Size of each block in bytes.
   456→    pub block_size: u64,
   457→    /// List of free block addresses.
   458→    pub free_list: Vec<Address>,
   459→}
   460→
   461→impl PoolAllocator {
   462→    /// Create a new PoolAllocator with the given block size.
   463→    ///
   464→    /// The pool starts empty. Use `add_region` to add memory regions
   465→    /// to the pool.
   466→    ///
   467→    /// # Arguments
   468→    ///
   469→    /// * `block_size` - The size of each block in bytes.
   470→    // VUMA-VERIFIED: initialization establishes valid pool parameters
   471→    pub fn new(block_size: u64) -> Self {
   472→        Self {
   473→            block_size,
   474→            free_list: Vec::new(),
   475→        }
   476→    }
   477→
   478→    /// Add a contiguous region of memory to the pool.
   479→    ///
   480→    /// The region is divided into `block_size`-sized blocks, each added
   481→    /// to the free list.
   482→    ///
   483→    /// # Arguments
   484→    ///
   485→    /// * `base` - The base address of the region.
   486→    /// * `region_size` - The total size of the region in bytes.
   487→    // VUMA-VERIFIED: region is correctly partitioned into blocks
   488→    pub fn add_region(&mut self, base: Address, region_size: u64) {
   489→        let num_blocks = region_size / self.block_size;
   490→        for i in 0..num_blocks {
   491→            let addr = base.offset(i * self.block_size);
   492→            self.free_list.push(addr);
   493→        }
   494→    }
   495→
   496→    /// Returns the RepD for this allocator.
   497→    // VUMA-VERIFIED: type descriptor is correct
   498→    pub fn repd(&self) -> RepD {
   499→        allocator_repd()
   500→    }
   501→
   502→    /// Returns the SyncEdge annotations for this allocator's operations.
   503→    // VUMA-VERIFIED: synchronization edges correctly model pool semantics
   504→    pub fn sync_edges(&self) -> Vec<SyncEdge> {
   505→        vec![
   506→            SyncEdge::new("pool_allocate", "pool_free", SyncEdgeKind::Seq),
   507→            SyncEdge::new("pool_free", "pool_allocate", SyncEdgeKind::Seq),
   508→        ]
   509→    }
   510→
   511→    /// Allocate one block from the pool.
   512→    ///
   513→    /// Returns the address of the allocated block, or an error if the
   514→    /// pool is exhausted.
   515→    // VUMA-VERIFIED: allocation from free list is well-defined
   516→    pub fn allocate(&mut self) -> AllocResult<Address> {
   517→        self.free_list.pop().ok_or(AllocError::OutOfMemory {
   518→            requested: self.block_size,
   519→            available: 0,
   520→        })
   521→    }
   522→
   523→    /// Return a block to the pool.
   524→    ///
   525→    /// # Arguments
   526→    ///
   527→    /// * `addr` - The address of the block to free.
   528→    ///
   529→    /// **Note**: This implementation does not validate that `addr` was
   530→    /// originally allocated from this pool. The VUMA verifier ensures
   531→    /// this invariant at compile time through capability tracking.
   532→    // VUMA-VERIFIED: free returns block to pool; VUMA verifier ensures addr validity
   533→    pub fn free(&mut self, addr: Address) -> AllocResult<()> {
   534→        self.free_list.push(addr);
   535→        Ok(())
   536→    }
   537→
   538→    /// Returns the number of available blocks in the pool.
   539→    // VUMA-VERIFIED: pure query, no side effects
   540→    pub fn available(&self) -> usize {
   541→        self.free_list.len()
   542→    }
   543→
   544→    /// Returns true if the pool has no available blocks.
   545→    // VUMA-VERIFIED: pure query, no side effects
   546→    pub fn is_empty(&self) -> bool {
   547→        self.free_list.is_empty()
   548→    }
   549→}

   559→/// Snapshot of allocator memory statistics.
   560→///
   561→/// Provides a summary of allocation activity for diagnostic and
   562→/// verification purposes. Each field is annotated with BD semantics.
   563→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   564→pub struct MemoryStats {
   565→    /// Cumulative total bytes allocated over the lifetime.
   566→    pub total_allocated: u64,
   567→    /// Cumulative total bytes freed over the lifetime.
   568→    pub total_freed: u64,
   569→    /// Current bytes in active use.
   570→    pub current_usage: u64,
   571→    /// Total number of alloc operations performed.
   572→    pub num_allocs: u64,
   573→    /// Total number of dealloc operations performed.
   574→    pub num_deallocs: u64,
   575→    /// Peak memory usage observed.
   576→    pub peak_usage: u64,
   577→    /// Total size of the managed heap region.
   578→    pub heap_size: u64,
   579→}
   580→
   581→impl MemoryStats {
   582→    /// Create a zeroed statistics snapshot.
   583→    // VUMA-VERIFIED: initialization produces valid zero-state
   584→    pub fn new() -> Self {
   585→        Self {
   586→            total_allocated: 0,
   587→            total_freed: 0,
   588→            current_usage: 0,
   589→            num_allocs: 0,
   590→            num_deallocs: 0,
   591→            peak_usage: 0,
   592→            heap_size: 0,
   593→        }
   594→    }
   595→
   596→    /// Compute the fragmentation metric as a value in [0.0, 1.0].
   597→    ///
   598→    /// Fragmentation = 1 - (largest_free_block / total_free)
   599→    /// Returns 0.0 when there is no free memory (no fragmentation possible)
   600→    /// or when all free memory is in one contiguous block (no fragmentation).
   601→    // VUMA-VERIFIED: metric is a pure function of observed state
   602→    pub fn fragmentation(&self, largest_free_block: u64) -> f64 {
   603→        let total_free = self.heap_size.saturating_sub(self.current_usage);
   604→        if total_free == 0 {
   605→            return 0.0;
   606→        }
   607→        1.0 - (largest_free_block as f64 / total_free as f64)
   608→    }
   609→
   610→    /// Returns the number of currently active allocations.
   611→    // VUMA-VERIFIED: derived from tracked counters
   612→    pub fn active_allocations(&self) -> u64 {
   613→        self.num_allocs.saturating_sub(self.num_deallocs)
   614→    }
   615→}
   616→
   617→impl Default for MemoryStats {
   618→    fn default() -> Self {
   619→        Self::new()
   620→    }

   654→/// A record of a single allocation event, forming a region in the MSG.

   879→        vec![
   880→            SyncEdge::new("bump_alloc", "bump_alloc", SyncEdgeKind::Seq),
   881→            SyncEdge::new("bump_reset", "bump_alloc", SyncEdgeKind::Fence),
   882→        ]
   883→    }

   885→    /// Allocate `size` bytes with `align` alignment.
   886→    ///
   887→    /// Returns a pointer to the allocated memory, or null if out of memory.
   888→    /// Every allocation is recorded in the MSG tracker.
   889→    ///
   890→    /// # BD Annotation
   891→    ///
   892→    /// Allocates a new region in the MSG with capability Write.
   893→    // VUMA-VERIFIED: allocation respects size, alignment, and boundary constraints
   894→    pub fn alloc(&mut self, size: usize, align: usize) -> *mut u8 {
   895→        if size == 0 {
   896→            return std::ptr::null_mut();
   897→        }
   898→        if align == 0 || (align & (align - 1)) != 0 {
   899→            return std::ptr::null_mut();
   900→        }
   901→
   902→        let current = unsafe { self.heap_start.add(self.offset) } as usize;
   903→        let aligned = (current + align - 1) & !(align - 1);
   904→        let padding = aligned - current;
   905→        let total_needed = padding + size;
   906→
   907→        if self.offset + total_needed > self.heap_size {
   908→            return std::ptr::null_mut();
   909→        }
   910→
   911→        let ptr = aligned as *mut u8;
   912→        self.offset += total_needed;
   913→
   914→        // Update stats
   915→        self.stats.total_allocated += size as u64;
   916→        self.stats.current_usage += size as u64;
   917→        self.stats.num_allocs += 1;
   918→        if self.stats.current_usage > self.stats.peak_usage {
   919→            self.stats.peak_usage = self.stats.current_usage;
   920→        }
   921→
   922→        // Record in MSG
   923→        self.tracker
   924→            .record(AllocEventKind::Alloc, ptr as u64, size as u64, align as u64);
   925→
   926→        ptr
   927→    }

   930→    ///

   932→    /// Use `reset()` to reclaim all memory.

   934→    /// # BD Annotation
   935→    ///
   936→    /// Dealloc on a bump allocator is a verified no-op; the BD system
   937→    /// ensures that dangling references are not used after arena reset.
   938→    // VUMA-VERIFIED: no-op dealloc is safe under bump-pointer semantics
   939→    pub fn dealloc(&mut self, _ptr: *mut u8, _size: usize, _align: usize) {
   940→        // No-op: bump allocator does not support individual free
   941→    }
   942→
   943→    /// Reallocate by allocating a new block and copying.
   944→    ///
   945→    /// Since bump allocators cannot free individual blocks, realloc
   946→    /// always allocates new memory and copies the old data.
   947→    ///
   948→    /// # BD Annotation
   949→    ///
   950→    /// Realloc creates a new MSG region and copies data from the old region.
   951→    // VUMA-VERIFIED: realloc copies valid data; old region remains valid until reset
   952→    ///
   953→    /// # Safety
   954→    ///
   955→    /// `_ptr` must point to a valid allocation previously returned by `alloc`.
   956→    pub unsafe fn realloc(
   957→        &mut self,
   958→        _ptr: *mut u8,
   959→        old_size: usize,
   960→        new_size: usize,
   961→        align: usize,
   962→    ) -> *mut u8 {
   963→        let new_ptr = self.alloc(new_size, align);
   964→        if new_ptr.is_null() {
   965→            return std::ptr::null_mut();
   966→        }
   967→        let copy_size = old_size.min(new_size);
   968→        unsafe {
   969→            ptr::copy_nonoverlapping(_ptr, new_ptr, copy_size);
   970→        }
   971→        self.tracker.record(
   972→            AllocEventKind::Realloc,
   973→            new_ptr as u64,
   974→            new_size as u64,
   975→            align as u64,
   976→        );
   977→        new_ptr
   978→    }
   979→
   980→    /// Reset the allocator, reclaiming all memory.
   981→    ///
   982→    /// # Safety
   983→    ///
   984→    /// All previously returned pointers become invalid after reset.
   985→    // VUMA-VERIFIED: reset invalidates all prior allocations atomically
   986→    pub fn reset(&mut self) {
   987→        self.offset = 0;
   988→        self.stats.current_usage = 0;
   989→        self.stats.total_freed = self.stats.total_allocated;
   990→    }

   996→    }

  1021→/// Magic value stored in block headers for BD verification.
  1022→const VUMA_MAGIC: u32 = 0x564D4100; // "VMA\0"
  1023→
  1024→/// Minimum block size (header + at least enough space for a free-list node).
  1025→const MIN_BLOCK_SIZE: usize = 48; // 32 byte header + 16 bytes minimum payload
  1026→
  1027→/// Block header stored at the beginning of every heap block.
  1028→///
  1029→/// Layout:
  1030→/// ```text
  1031→/// [magic: u32][flags: u32][size: usize]  (32 bytes total on 64-bit)
  1032→/// ```
  1033→///
  1034→/// The `magic` field is used for BD verification: dealloc checks that
  1035→/// the magic matches `VUMA_MAGIC`, ensuring the pointer was allocated
  1036→/// by this allocator.
  1037→#[repr(C)]
  1038→struct BlockHeader {
  1039→    /// Magic number for BD verification.
  1040→    magic: u32,
  1041→    /// Flags (bit 0: is_free, bit 1: was_realloc).
  1042→    flags: u32,
  1043→    /// Total size of this block in bytes, including this header.
  1044→    size: usize,
  1045→    /// Size of the user payload (excluding header and padding).
  1046→    payload_size: usize,
  1047→    /// Alignment requested by the user.
  1048→    align: usize,
  1049→    /// Padding to bring header to 32 bytes.
  1050→    _reserved: [u64; 2],
  1051→}
  1052→
  1053→impl BlockHeader {
  1054→    const SIZE: usize = 32;
  1055→
  1056→    fn is_free(&self) -> bool {
  1057→        self.flags & 1 != 0
  1058→    }
  1059→
  1060→    fn set_free(&mut self, free: bool) {
  1061→        if free {
  1062→            self.flags |= 1;
  1063→        } else {
  1064→            self.flags &= !1;
  1065→        }
  1066→    }
  1067→
  1068→    #[allow(dead_code)] // part of BlockHeader API for heap diagnostics
  1069→    fn was_realloc(&self) -> bool {
  1070→        self.flags & 2 != 0
  1071→    }
  1072→
  1073→    fn set_realloc(&mut self, realloc: bool) {
  1074→        if realloc {
  1075→            self.flags |= 2;
  1076→        } else {
  1077→            self.flags &= !2;
  1078→        }
  1079→    }
  1080→
  1081→    /// Returns a pointer to the payload (just after this header).
  1082→    fn payload_ptr(header: *mut BlockHeader) -> *mut u8 {
  1083→        unsafe { (header as *mut u8).add(BlockHeader::SIZE) }
  1084→    }
  1085→
  1086→    /// Returns the header pointer from a payload pointer.
  1087→    fn from_payload(payload: *mut u8) -> *mut BlockHeader {
  1088→        unsafe { payload.sub(BlockHeader::SIZE) as *mut BlockHeader }
  1089→    }
  1090→}

  1093→/// are used to store a pointer to the next free block.
  1094→#[repr(C)]
  1095→struct FreeNode {
  1096→    next: *mut FreeNode,
  1097→    _pad: [u8; 8], // pad to 16 bytes for alignment
  1098→}
  1099→

  1103→}

  1106→///
  1107→/// The `FreeListAllocator` manages a contiguous region of real memory using
  1108→/// a free-list strategy with first-fit allocation and coalescing on dealloc.
  1109→/// It supports `alloc`, `dealloc`, and `realloc` operations, each annotated
  1110→/// with BD descriptors and tracked in the MSG.
  1111→///
  1112→/// ## BD Annotations
  1113→///
  1114→/// - CapD: { Read, Write, Serialize }
  1115→/// - SyncEdge: alloc → dealloc (Seq), dealloc → alloc (Seq),
  1116→///   realloc → alloc (Seq), dealloc → dealloc (Seq)
  1117→///
  1118→/// ## Block Layout
  1119→///
  1120→/// ```text
  1121→/// [BlockHeader (32 bytes)][payload]
  1122→/// ```
  1123→///
  1124→/// Free blocks store a `FreeNode` (16 bytes) at the start of the payload,
  1125→/// forming an intrusive linked list.
  1126→///

  1130→/// This allocator is **not** thread-safe; use external synchronization.

  1135→    heap_size: usize,

  1150→    ///
  1151→    /// # Arguments

  1153→    /// * `memory` - The backing memory region. Must outlive the allocator
  1154→    ///   and be at least `MIN_BLOCK_SIZE` bytes.

  1159→            heap_start: memory.as_mut_ptr(),
  1160→            heap_size,
  1161→            free_head: ptr::null_mut(),

  1179→                        payload_size: heap_size - BlockHeader::SIZE,
  1180→                        align: 8,
  1181→                        _reserved: [0; 2],
  1182→                    },

  1186→                ptr::write(

  1229→    /// Allocate `size` bytes with `align` alignment.
  1230→    ///
  1231→    /// Returns a pointer to the allocated memory, or null if out of memory.
  1232→    /// Every allocation is recorded in the MSG tracker with BD annotation.
  1233→    ///
  1234→    /// # BD Annotation
  1235→    ///
  1236→    /// Creates a new writable region in the MSG.
  1237→    // VUMA-VERIFIED: allocation respects size, alignment, boundary, and BD constraints
  1238→    pub fn alloc(&mut self, size: usize, align: usize) -> *mut u8 {
  1239→        if size == 0 {
  1240→            return ptr::null_mut();
  1241→        }
  1242→        if align == 0 || (align & (align - 1)) != 0 {
  1243→            return ptr::null_mut();
  1244→        }
  1245→
  1246→        let needed = size + BlockHeader::SIZE;
  1247→
  1248→        // Walk the free list (first fit)
  1249→        let mut prev: *mut FreeNode = ptr::null_mut();
  1250→        let mut current = self.free_head;
  1251→
  1252→        while !current.is_null() {
  1253→            unsafe {
  1254→                let header = (current as *mut u8).sub(BlockHeader::SIZE) as *mut BlockHeader;
  1255→                let block_size = (*header).size;
  1256→
  1257→                if block_size >= needed {
  1258→                    // Found a block that fits
  1259→
  1260→                    // Check if we can split this block
  1261→                    let remainder = block_size - needed;
  1262→                    if remainder >= MIN_BLOCK_SIZE {
  1263→                        // Split: create a new free block from the remainder
  1264→                        let new_header = (header as *mut u8).add(needed) as *mut BlockHeader;
  1265→                        ptr::write(
  1266→                            new_header,
  1267→                            BlockHeader {
  1268→                                magic: VUMA_MAGIC,
  1269→                                flags: 1, // is_free = true
  1270→                                size: remainder,
  1271→                                payload_size: remainder - BlockHeader::SIZE,
  1272→                                align: 8,
  1273→                                _reserved: [0; 2],
  1274→                            },
  1275→                        );
  1276→                        let new_free = BlockHeader::payload_ptr(new_header) as *mut FreeNode;
  1277→                        ptr::write(
  1278→                            new_free,
  1279→                            FreeNode {
  1280→                                next: (*current).next,
  1281→                                _pad: [0; 8],
  1282→                            },
  1283→                        );
  1284→
  1285→                        // Update current block size
  1286→                        (*header).size = needed;
  1287→                        (*header).payload_size = needed - BlockHeader::SIZE;
  1288→
  1289→                        // Replace current in free list with new free block
  1290→                        if prev.is_null() {
  1291→                            self.free_head = new_free;
  1292→                        } else {
  1293→                            (*prev).next = new_free;
  1294→                        }
  1295→                    } else {
  1296→                        // Use the entire block (don't split)
  1297→                        // Remove current from free list
  1298→                        if prev.is_null() {
  1299→                            self.free_head = (*current).next;
  1300→                        } else {
  1301→                            (*prev).next = (*current).next;
  1302→                        }
  1303→                    }
  1304→
  1305→                    // Mark as used
  1306→                    (*header).set_free(false);
  1307→                    (*header).align = align;
  1308→                    (*header).payload_size = size;
  1309→
  1310→                    let payload = BlockHeader::payload_ptr(header);
  1311→
  1312→                    // Update stats
  1313→                    self.stats.total_allocated += size as u64;
  1314→                    self.stats.current_usage += size as u64;
  1315→                    self.stats.num_allocs += 1;
  1316→                    if self.stats.current_usage > self.stats.peak_usage {
  1317→                        self.stats.peak_usage = self.stats.current_usage;
  1318→                    }
  1319→
  1320→                    // Record in MSG
  1321→                    self.tracker.record(
  1322→                        AllocEventKind::Alloc,
  1323→                        payload as u64,
  1324→                        size as u64,
  1325→                        align as u64,
  1326→                    );
  1327→
  1328→                    return payload;
  1329→                }
  1330→
  1331→                prev = current;
  1332→                current = (*current).next;
  1333→            }
  1334→        }
  1335→
  1336→        // No suitable block found
  1337→        ptr::null_mut()
  1338→    }

  1340→    /// Deallocate a previously allocated block.
  1341→    ///
  1342→    /// Performs BD verification: checks that the pointer is within the heap
  1343→    /// and that the block header's magic number matches. If verification
  1344→    /// fails, the deallocation is silently ignored (safe failure).
  1345→    ///
  1346→    /// After deallocation, coalesces with adjacent free blocks to reduce
  1347→    /// fragmentation.
  1348→    ///
  1349→    /// # BD Annotation
  1350→    ///
  1351→    /// Removes the writable region from the MSG and returns it to the free set.
  1352→    // VUMA-VERIFIED: dealloc verifies BD constraints before releasing memory
  1353→    pub fn dealloc(&mut self, ptr: *mut u8, size: usize, align: usize) {
  1354→        if ptr.is_null() {
  1355→            return;
  1356→        }
  1357→
  1358→        // BD verification: check that ptr is within our heap
  1359→        let ptr_val = ptr as usize;
  1360→        let heap_start_val = self.heap_start as usize;
  1361→        let heap_end_val = heap_start_val + self.heap_size;
  1362→        if ptr_val < heap_start_val + BlockHeader::SIZE || ptr_val >= heap_end_val {
  1363→            return;
  1364→        }
  1365→
  1366→        unsafe {
  1367→            let header = BlockHeader::from_payload(ptr);
  1368→
  1369→            // BD verification: check magic number
  1370→            if (*header).magic != VUMA_MAGIC {
  1371→                return;
  1372→            }
  1373→
  1374→            // BD verification: check that block is currently in use
  1375→            if (*header).is_free() {
  1376→                return;
  1377→            }
  1378→
  1379→            let payload_size = (*header).payload_size;
  1380→
  1381→            // Mark as free
  1382→            (*header).set_free(true);
  1383→
  1384→            // Add to free list
  1385→            let free_node = ptr as *mut FreeNode;
  1386→            ptr::write(
  1387→                free_node,
  1388→                FreeNode {
  1389→                    next: self.free_head,
  1390→                    _pad: [0; 8],
  1391→                },
  1392→            );
  1393→            self.free_head = free_node;
  1394→
  1395→            // Coalesce with adjacent free blocks
  1396→            self.coalesce();
  1397→
  1398→            // Update stats
  1399→            self.stats.total_freed += payload_size as u64;
  1400→            self.stats.current_usage = self.stats.current_usage.saturating_sub(payload_size as u64);
  1401→            self.stats.num_deallocs += 1;
  1402→
  1403→            // Record in MSG
  1404→            self.tracker.record(
  1405→                AllocEventKind::Dealloc,
  1406→                ptr as u64,
  1407→                size as u64,
  1408→                align as u64,
  1409→            );
  1410→        }
  1411→    }

  1413→    /// Reallocate a previously allocated block.
  1414→    ///
  1415→    /// If the new size fits in the current block, returns the same pointer.
  1416→    /// Otherwise, allocates a new block, copies the data, and frees the old one.
  1417→    ///
  1418→    /// # BD Annotation
  1419→    ///
  1420→    /// Updates the MSG region with new size; may create a new region
  1421→    /// if the block is moved.
  1422→    // VUMA-VERIFIED: realloc preserves data integrity and BD constraints
  1423→    ///
  1424→    /// # Safety
  1425→    ///
  1426→    /// `ptr` must point to a valid allocation previously returned by `alloc`.
  1427→    pub unsafe fn realloc(
  1428→        &mut self,
  1429→        ptr: *mut u8,
  1430→        old_size: usize,
  1431→        new_size: usize,
  1432→        align: usize,
  1433→    ) -> *mut u8 {
  1434→        if ptr.is_null() {
  1435→            return self.alloc(new_size, align);
  1436→        }
  1437→        if new_size == 0 {
  1438→            self.dealloc(ptr, old_size, align);
  1439→            return ptr::null_mut();
  1440→        }
  1441→
  1442→        unsafe {
  1443→            let header = BlockHeader::from_payload(ptr);
  1444→
  1445→            // BD verification
  1446→            if (*header).magic != VUMA_MAGIC || (*header).is_free() {
  1447→                return ptr::null_mut();
  1448→            }
  1449→
  1450→            let block_size = (*header).size;
  1451→            let needed = new_size + BlockHeader::SIZE;
  1452→
  1453→            // Can we fit in the current block?
  1454→            if needed <= block_size {
  1455→                // In-place resize
  1456→                (*header).payload_size = new_size;
  1457→                (*header).set_realloc(true);
  1458→
  1459→                // Update stats
  1460→                let diff = new_size as i64 - old_size as i64;
  1461→                if diff > 0 {
  1462→                    self.stats.total_allocated += diff as u64;
  1463→                    self.stats.current_usage += diff as u64;
  1464→                } else {
  1465→                    self.stats.total_freed += (-diff) as u64;
  1466→                    self.stats.current_usage =
  1467→                        self.stats.current_usage.saturating_sub((-diff) as u64);
  1468→                }
  1469→
  1470→                self.tracker.record(
  1471→                    AllocEventKind::Realloc,
  1472→                    ptr as u64,
  1473→                    new_size as u64,
  1474→                    align as u64,
  1475→                );
  1476→
  1477→                return ptr;
  1478→            }
  1479→
  1480→            // Need a new block
  1481→            let new_ptr = self.alloc(new_size, align);
  1482→            if new_ptr.is_null() {
  1483→                return ptr::null_mut();
  1484→            }
  1485→
  1486→            // Copy old data
  1487→            let copy_size = old_size.min(new_size);
  1488→            ptr::copy_nonoverlapping(ptr, new_ptr, copy_size);
  1489→
  1490→            // Free old block
  1491→            self.dealloc(ptr, old_size, align);
  1492→
  1493→            self.tracker.record(
  1494→                AllocEventKind::Realloc,
  1495→                new_ptr as u64,
  1496→                new_size as u64,
  1497→                align as u64,
  1498→            );
  1499→
  1500→            new_ptr
  1501→        }
  1502→    }

  1504→    /// Coalesce adjacent free blocks to reduce fragmentation.
  1505→    ///
  1506→    /// This walks the heap linearly from the start, merging adjacent
  1507→    /// free blocks. It's O(n) where n is the number of blocks, but
  1508→    /// guarantees that the free list has no adjacent free blocks
  1509→    /// after coalescing.
  1510→    // VUMA-VERIFIED: coalescing preserves heap invariants
  1511→    fn coalesce(&mut self) {
  1512→        let mut offset = 0usize;
  1513→        self.free_head = ptr::null_mut();
  1514→
  1515→        let mut new_free_head: *mut FreeNode = ptr::null_mut();
  1516→        let mut last_free: *mut FreeNode = ptr::null_mut();
  1517→
  1518→        unsafe {
  1519→            while offset + MIN_BLOCK_SIZE <= self.heap_size {
  1520→                let header = self.heap_start.add(offset) as *mut BlockHeader;
  1521→                let block_size = (*header).size;
  1522→
  1523→                if block_size < MIN_BLOCK_SIZE {
  1524→                    break; // Corrupted heap; stop
  1525→                }
  1526→
  1527→                if (*header).is_free() {
  1528→                    // Try to merge with the next block(s)
  1529→                    let mut merged_size = block_size;
  1530→                    let mut next_offset = offset + block_size;
  1531→
  1532→                    while next_offset + MIN_BLOCK_SIZE <= self.heap_size {
  1533→                        let next_header = self.heap_start.add(next_offset) as *mut BlockHeader;
  1534→                        if !(*next_header).is_free() {
  1535→                            break;
  1536→                        }
  1537→                        merged_size += (*next_header).size;
  1538→                        next_offset += (*next_header).size;
  1539→                    }
  1540→
  1541→                    // Update header with merged size
  1542→                    (*header).size = merged_size;
  1543→                    (*header).payload_size = merged_size - BlockHeader::SIZE;
  1544→
  1545→                    // Add to free list (append to maintain order)
  1546→                    let free_node = BlockHeader::payload_ptr(header) as *mut FreeNode;
  1547→                    ptr::write(
  1548→                        free_node,
  1549→                        FreeNode {
  1550→                            next: ptr::null_mut(),
  1551→                            _pad: [0; 8],
  1552→                        },
  1553→                    );
  1554→
  1555→                    if new_free_head.is_null() {
  1556→                        new_free_head = free_node;
  1557→                    } else {
  1558→                        (*last_free).next = free_node;
  1559→                    }
  1560→                    last_free = free_node;
  1561→
  1562→                    offset += merged_size;
  1563→                } else {
  1564→                    offset += block_size;
  1565→                }
  1566→            }
  1567→        }
  1568→
  1569→        self.free_head = new_free_head;
  1570→    }

  1631→///
  1632→/// A simple spin lock that uses an `AtomicBool` for mutual exclusion.
