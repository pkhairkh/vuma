     1→//! # ARM64 Code Emission
     2→//!
     3→//! Lowers IR to ARM64 machine code and produces ELF binaries or raw binaries
     4→//! suitable for the AArch64 (Cortex-A76, ARMv8.2-A).
     5→//!
     6→//! ## Pipeline
     7→//!
     8→//! 1. **IR → ARM64 Instructions**: Each IR instruction is pattern-matched and
     9→//!    lowered to one or more [`Instruction`]
    10→//!    values, with virtual registers replaced by physical registers from the
    11→//!    register allocator.
    12→//! 2. **Emit function**: Each IR function is emitted as a sequence of 32-bit
    13→//!    ARM64 code words.
    14→//! 3. **Emit program**: All functions are collected, data sections are laid
    15→//!    out, and a minimal ELF64 binary is produced.
    16→//!
    17→//! ## Output Formats
    18→//!
    19→//! - **ELF**: Full ELF64 binary with program headers, section headers, symbol
    20→//!   table, and string table.  Suitable for Linux on AArch64.
    21→//! - **Raw**: Flat binary image for bare-metal AArch64 (loaded at 0x80000).
    22→//! - **Obj**: Relocatable ELF object file (`ET_REL`) for linking.
    23→//!
    24→//! ## ELF Layout (executable)
    25→//!
    26→//! ```text
    27→//! ┌─────────────────────┐
    28→//! │ ELF Header           │  64 bytes
    29→//! ├─────────────────────┤
    30→//! │ Program Headers      │  2 × 56 bytes (LOAD segments)
    31→//! ├─────────────────────┤
    32→//! │ .text                │  emitted code
    33→//! ├─────────────────────┤
    34→//! │ .rodata              │  read-only data
    35→//! ├─────────────────────┤
    36→//! │ .data                │  initialized data
    37→//! ├─────────────────────┤
    38→//! │ .symtab              │  symbol table entries
    39→//! ├─────────────────────┤
    40→//! │ .strtab              │  symbol string table
    41→//! ├─────────────────────┤
    42→//! │ .shstrtab            │  section header string table
    43→//! ├─────────────────────┤
    44→//! │ Section Headers      │  N × 64 bytes
    45→//! └─────────────────────┘
    46→//! .bss is virtual-only (memsz > filesz in the data LOAD segment).
    47→//! ```
    48→//!
    49→//! ## Relocation Support
    50→//!
    51→//! Inter-function calls (`BL`) are resolved in a fixup pass after all
    52→//! functions have been emitted.  The emitter records the word offset of each
    53→//! `BL` instruction and the target function name; once function addresses
    54→//! are known, the branch offsets are patched into the encoded instructions.
    55→

   487→// ---------------------------------------------------------------------------
   488→// Inter-function call relocation record
   489→// ---------------------------------------------------------------------------
   490→
   491→/// A record for a `BL` instruction that needs to be patched with the address
   492→/// of a named function.  After all functions are emitted, these records are
   493→/// resolved to compute the correct branch offset.
   494→#[derive(Debug, Clone)]
   495→struct CallRelocation {
   496→    /// Byte offset within the text section where the BL instruction lives.
   497→    text_byte_offset: u64,
   498→    /// Name of the target function.
   499→    target_func: String,
   500→}
   501→
   502→// ---------------------------------------------------------------------------
   503→// ELF RelaEntry
   504→// ---------------------------------------------------------------------------
   505→
   506→/// An ELF64 relocation entry with an explicit addend (SHT_RELA).
   507→///
   508→/// Each entry specifies where in the section a relocation must be applied,
   509→/// which symbol the relocation references, the type of relocation, and an
   510→/// addend value used in the relocation computation.
   511→#[derive(Debug, Clone)]
   512→pub struct RelaEntry {
   513→    /// Byte offset within the section where the relocation applies.
   514→    pub offset: u64,
   515→    /// Symbol index (upper 32 bits) and relocation type (lower 32 bits),
   516→    /// packed as: `(sym_idx << 32) | r_type`.
   517→    pub info: u64,
   518→    /// Addend value used in the relocation computation.
   519→    pub addend: i64,
   520→}
   521→
   522→impl RelaEntry {
   523→    /// Create a new relocation entry.
   524→    pub fn new(offset: u64, sym_idx: u32, r_type: u32, addend: i64) -> Self {
   525→        Self {
   526→            offset,
   527→            info: ((sym_idx as u64) << 32) | (r_type as u64),
   528→            addend,
   529→        }
   530→    }
   531→
   532→    /// Extract the symbol index from `info`.
   533→    pub fn sym_idx(&self) -> u32 {
   534→        (self.info >> 32) as u32
   535→    }
   536→
   537→    /// Extract the relocation type from `info`.
   538→    pub fn r_type(&self) -> u32 {
   539→        (self.info & 0xFFFFFFFF) as u32
   540→    }
   541→
   542→    /// Serialize the entry to 24 bytes (little-endian).
   543→    pub fn to_bytes(&self) -> [u8; 24] {
   544→        let mut buf = [0u8; 24];
   545→        buf[0..8].copy_from_slice(&self.offset.to_le_bytes());
   546→        buf[8..16].copy_from_slice(&self.info.to_le_bytes());
   547→        buf[16..24].copy_from_slice(&self.addend.to_le_bytes());
   548→        buf
   549→    }
   550→}

   556→/// The ARM64 code emitter.
   557→///
   558→/// Holds state for the current emission context: the register allocator,
   559→/// accumulated code, and fixup records for branch targets.
   560→pub struct Emitter {
   561→    /// Register allocator used during emission.
   562→    reg_alloc: RegAllocator,
   563→    /// Accumulated machine code for the current function (32-bit words).
   564→    code: Vec<u32>,
   565→    /// Fixup records for intra-function branches: (word index, target label name, format).
   566→    fixups: Vec<(usize, String, BranchFormat)>,
   567→    /// Map from label name to code offset (in words) within the current function.
   568→    label_offsets: HashMap<String, usize>,
   569→    /// Inter-function call relocations for the current function.
   570→    call_relocs: Vec<CallRelocation>,
   571→    /// Relocation entries using the `RelocationEntry` infrastructure.
   572→    ///
   573→    /// Each entry records a byte offset within the function's encoded output
   574→    /// where a symbolic reference must be resolved, the target symbol name,
   575→    /// and the ISA-specific relocation type (e.g., `"R_AARCH64_CALL26"`).
   576→    relocations: Vec<RelocationEntry>,
   577→    /// Name of the function currently being emitted.
   578→    current_func_name: String,
   579→    /// Byte offset of the current function within the text section (set externally).
   580→    func_text_offset: u64,
   581→    /// Computed stack frame size (in bytes) for the current function.
   582→    frame_size: u32,
   583→    /// Registers pinned for the current instruction (auto-unpinned after each instruction).
   584→    instr_pinned_regs: Vec<Register>,
   585→}
   586→
   587→impl Emitter {
   588→    /// Create a new emitter with a fresh register allocator.
   589→    pub fn new() -> Self {
   590→        Self {
   591→            reg_alloc: RegAllocator::new(),
   592→            code: Vec::new(),
   593→            fixups: Vec::new(),
   594→            label_offsets: HashMap::new(),
   595→            call_relocs: Vec::new(),
   596→            relocations: Vec::new(),
   597→            current_func_name: String::new(),
   598→            func_text_offset: 0,
   599→            frame_size: 0,
   600→            instr_pinned_regs: Vec::new(),
   601→        }
   602→    }
   603→
   604→    // -----------------------------------------------------------------------
   605→    // Function emission
   606→    // -----------------------------------------------------------------------
   607→
   608→    /// Emit a single IR function to ARM64 machine code.
   609→    ///
   610→    /// For functions with more than `STACK_SLOT_VREG_THRESHOLD` virtual
   611→    /// registers, the stack-slot emitter is used (every vreg gets a stack
   612→    /// slot, scratch registers only).  For simpler functions, the greedy
   613→    /// register allocator is used.
   614→    ///
   615→    /// Returns a vector of 32-bit ARM64 instruction words.
   616→    pub fn emit_function(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   617→        let vreg_count = count_vregs(func);
   618→        if vreg_count > STACK_SLOT_VREG_THRESHOLD {
   619→            self.emit_function_stack_slot(func)
   620→        } else {
   621→            self.emit_function_greedy(func)
   622→        }
   623→    }
   624→
   625→    /// Emit a single IR function using the greedy register allocator.
   626→    ///
   627→    /// This is the original emission strategy — suitable for functions with
   628→    /// a small number of virtual registers.
   629→    fn emit_function_greedy(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   630→        self.code.clear();
   631→        self.fixups.clear();
   632→        self.label_offsets.clear();
   633→        self.call_relocs.clear();
   634→        self.relocations.clear();
   635→        self.current_func_name = func.name.clone();
   636→        self.reg_alloc.reset();
   637→
   638→        // Pre-allocate registers for parameters (AAPCS64: X0–X7).
   639→        // We must explicitly assign parameter virtual registers to the
   640→        // correct argument registers so the calling convention is respected.
   641→        let arg_regs = [
   642→            Register::X0, Register::X1, Register::X2, Register::X3,
   643→            Register::X4, Register::X5, Register::X6, Register::X7,
   644→        ];
   645→        for (i, param) in func.params.iter().enumerate() {
   646→            if let IRValue::Register(vreg_id) = param {
   647→                if i < 8 {
   648→                    self.reg_alloc.preassign(*vreg_id, arg_regs[i]);
   649→                }
   650→            }
   651→        }
   652→
   653→        // Emit prologue:
   654→        // 1. SUB SP, SP, #16          (make room for FP/LR save pair)
   655→        // 2. STP X29, X30, [SP]       (save FP and LR at new SP)
   656→        // 3. ADD X29, SP, #0           (set frame pointer to new SP)
   657→        // 4. SUB SP, SP, #frame_size   (make room for local variables)
   658→        //
   659→        // NOTE: We cannot use pre-indexed STP [SP, #-16]! because the
   660→        // Instruction::STP encoding only supports signed-offset mode.
   661→        // Instead, we explicitly decrement SP with SUB before the STP.
   662→        self.emit_instruction(Instruction::SUB {
   663→            rd: Register::SP,
   664→            rn: Register::SP,
   665→            rm: Operand::Imm12(16),
   666→        })?;
   667→        self.emit_instruction(Instruction::STP {
   668→            rt1: Register::X29,
   669→            rt2: Register::X30,
   670→            rn: Register::SP,
   671→            offset: 0,
   672→        })?;
   673→
   674→        // MOV X29, SP (set frame pointer)
   675→        // IMPORTANT: Cannot use MOV (ORR Xd, XZR, SP) because ORR treats
   676→        // Rm=31 as XZR, yielding zero. Use ADD X29, SP, #0 instead.
   677→        self.emit_instruction(Instruction::ADD {
   678→            rd: Register::X29,
   679→            rn: Register::SP,
   680→            rm: Operand::Imm12(0),
   681→        })?;
   682→
   683→        // Reserve space for spill slots only. Each Alloc instruction handles
   684→        // its own SUB SP, SP, #aligned_size. The epilogue will restore only
   685→        // the spill area reservation. The Alloc areas are NOT restored by the
   686→        // epilogue — they're effectively "leaked" stack space. This is fine
   687→        // because the function's stack frame is restored by the LDP + ADD SP, #16.
   688→        //
   689→        // For spill slots, we need to estimate the maximum number of spills
   690→        // that could occur during emission. Since we can't know the exact
   691→        // count beforehand, we use the compute_frame_size function which
   692→        // estimates based on the vreg count.
   693→        let aligned_stack = compute_frame_size(func);
   694→        // frame_size = only the spill area (NOT the Alloc sizes, since each
   695→        // Alloc instruction does its own SUB SP)
   696→        // Compute the spill-only portion:
   697→        let mut alloc_total: u32 = 0;
   698→        for block in &func.blocks {
   699→            for instr in &block.instructions {
   700→                if let IRInstr::Alloc { size, .. } = instr {
   701→                    let aligned = (*size).div_ceil(16) * 16;
   702→                    alloc_total += aligned;
   703→                }
   704→            }
   705→        }
   706→        // Spill area = total frame - alloc area
   707→        let spill_area = if aligned_stack > alloc_total {
   708→            aligned_stack - alloc_total
   709→        } else {
   710→            // Minimal 16-byte reservation for safety
   711→            16
   712→        };
   713→        let spill_area_aligned = (spill_area + 15) & !15;
   714→        self.frame_size = spill_area_aligned;
   715→        if spill_area_aligned > 0 {
   716→            if spill_area_aligned <= 4095 {
   717→                self.emit_instruction(Instruction::SUB {
   718→                    rd: Register::SP,
   719→                    rn: Register::SP,
   720→                    rm: Operand::Imm12(spill_area_aligned as u16),
   721→                })?;
   722→            } else {
   723→                self.emit_load_immediate(Register::X9, spill_area_aligned as i64)?;
   724→                self.emit_instruction(Instruction::SUB {
   725→                    rd: Register::SP,
   726→                    rn: Register::SP,
   727→                    rm: Operand::Reg {
   728→                        reg: Register::X9,
   729→                        shift: None,
   730→                    },
   731→                })?;
   732→            }
   733→        }
   734→
   735→        // Emit each basic block.
   736→        for block in &func.blocks {
   737→            self.label_offsets
   738→                .insert(block.label.clone(), self.code.len());
   739→            self.emit_block(block)?;
   740→        }
   741→
   742→        // Apply fixups — resolve intra-function branch targets.
   743→        self.apply_fixups()?;
   744→
   745→        Ok(self.code.clone())
   746→    }
   747→
   748→    /// Returns the relocation entries recorded during emission of the current
   749→    /// function.
   750→    ///
   751→    /// Each `RelocationEntry` records a byte offset within the function's
   752→    /// encoded output where a symbolic reference must be resolved, the target
   753→    /// symbol name, and the ISA-specific relocation type (e.g.,
   754→    /// `"R_AARCH64_CALL26"`).
   755→    pub fn relocations(&self) -> &[RelocationEntry] {
   756→        &self.relocations
   757→    }
   758→
   759→    /// Emit instructions for a single IR basic block.
   760→    fn emit_block(&mut self, block: &IRBlock) -> Result<()> {
   761→        for instr in &block.instructions {
   762→            self.emit_ir_instr(instr)?;
   763→            // Auto-unpin all registers that were pinned during this instruction.
   764→            for reg in self.instr_pinned_regs.drain(..) {
   765→                self.reg_alloc.unpin(reg);
   766→            }
   767→        }
   768→        self.emit_terminator(&block.terminator)?;
   769→        Ok(())
   770→    }
   771→
   772→    // -----------------------------------------------------------------------
   773→    // IR → ARM64 instruction lowering
   774→    // -----------------------------------------------------------------------
   775→
   776→    /// Lower a single IR instruction to ARM64 instructions.
   777→    fn emit_ir_instr(&mut self, instr: &IRInstr) -> Result<()> {
   778→        match instr {
   779→            IRInstr::Load { dst, addr, offset, ty } => {
   780→                let rn = self.resolve_reg(addr)?;
   781→                let rt = self.resolve_reg(dst)?;
   782→                self.emit_load(rt, rn, *offset, ty)?;
   783→            }
   784→
   785→            IRInstr::Store { value, addr, offset, ty } => {
   786→                let rt = self.resolve_reg(value)?;
   787→                let rn = self.resolve_reg(addr)?;
   788→                if rt == rn {
   789→                    // Both resolved to the same register — use scratch for value
   790→                    self.emit_instruction(Instruction::MOV {
   791→                        rd: Register::X16,
   792→                        rm: rt,
   793→                    })?;
   794→                    self.emit_store(Register::X16, rn, *offset, ty)?;
   795→                } else {
   796→                    self.emit_store(rt, rn, *offset, ty)?;
   797→                }
   798→            }
   799→
   800→            IRInstr::BinOp { op, dst, lhs, rhs, ty } => {
   801→                self.emit_binop(*op, dst, lhs, rhs, ty.as_ref())?;
   802→            }
   803→
   804→            IRInstr::UnaryOp { op, dst, operand, ty: _ } => {
   805→                let rd = self.resolve_reg(dst)?;
   806→                let rn = self.resolve_reg(operand)?;
   807→                match op {
   808→                    UnaryOpKind::Neg => {
   809→                        self.emit_instruction(Instruction::SUB {
   810→                            rd,
   811→                            rn: Register::XZR,
   812→                            rm: Operand::Reg {
   813→                                reg: rn,
   814→                                shift: None,
   815→                            },
   816→                        })?;
   817→                    }
   818→                    UnaryOpKind::Not => {
   819→                        self.emit_load_immediate(Register::X9, -1)?;
   820→                        self.emit_instruction(Instruction::EOR {
   821→                            rd,
   822→                            rn,
   823→                            rm: Register::X9,
   824→                        })?;
   825→                    }
   826→                    UnaryOpKind::Clz => {
   827→                        // CLZ Xd, Xn — count leading zeros (native ARM64 instruction)
   828→                        self.emit_instruction(Instruction::CLZ { rd, rn })?;
   829→                    }
   830→                    UnaryOpKind::Ctz => {
   831→                        // CTZ = RBIT + CLZ: reverse bits then count leading zeros.
   832→                        // Use X9 as scratch if rd == rn (need intermediate result).
   833→                        if rd == rn {
   834→                            self.emit_instruction(Instruction::RBIT {
   835→                                rd: Register::X9,
   836→                                rn,
   837→                            })?;
   838→                            self.emit_instruction(Instruction::CLZ {
   839→                                rd,
   840→                                rn: Register::X9,
   841→                            })?;
   842→                        } else {
   843→                            self.emit_instruction(Instruction::RBIT { rd, rn })?;
   844→                            self.emit_instruction(Instruction::CLZ { rd, rn: rd })?;
   845→                        }
   846→                    }
   847→                    UnaryOpKind::Popcnt => {
   848→                        // POPCNT via FMOV+CNT+ADDV+UMOV sequence:
   849→                        // FMOV D8, Xn        — move GPR value to SIMD register (V8 is caller-saved)
   850→                        // CNT V8.8B, V8.8B   — count bits per byte
   851→                        // ADDV B8, V8.8B     — horizontal sum of byte counts
   852→                        // UMOV Xd, V8.B[0]   — move result back to GPR (zero-extends to 64 bits)
   853→                        const SIMD_SCRATCH: u8 = 8; // V8 is caller-saved in AAPCS64
   854→                        self.emit_instruction(Instruction::FMOV_DX {
   855→                            vd: SIMD_SCRATCH,
   856→                            rn,
   857→                        })?;
   858→                        self.emit_instruction(Instruction::CNT {
   859→                            vd: SIMD_SCRATCH,
   860→                            vn: SIMD_SCRATCH,
   861→                        })?;
   862→                        self.emit_instruction(Instruction::ADDV {
   863→                            vd: SIMD_SCRATCH,
   864→                            vn: SIMD_SCRATCH,
   865→                        })?;
   866→                        self.emit_instruction(Instruction::UMOV {
   867→                            rd,
   868→                            vn: SIMD_SCRATCH,
   869→                        })?;
   870→                    }
   871→                }
   872→            }
   873→
   874→            IRInstr::Call {
   875→                dst,
   876→                func: target_name,
   877→                args,
   878→            } => {
   879→                // Resolve all argument source registers FIRST, before moving
   880→                // any of them. This prevents a later move from overwriting a
   881→                // register that an earlier argument is in.
   882→                //
   883→                // IMPORTANT: For immediate arguments, resolve_reg always loads
   884→                // into X9. When multiple immediates are present, we must use
   885→                // different scratch registers to avoid overwriting. We load
   886→                // immediates into X0-X7 directly (since they're the target
   887→                // registers anyway) or use X11-X15 as temporary scratch.
   888→                let arg_regs = [
   889→                    Register::X0, Register::X1, Register::X2, Register::X3,
   890→                    Register::X4, Register::X5, Register::X6, Register::X7,
   891→                ];
   892→                // Scratch registers for loading immediates when the target arg
   893→                // register isn't available yet (because of conflicts).
   894→                let scratch_regs = [
   895→                    Register::X11, Register::X12, Register::X13, Register::X14,
   896→                    Register::X15, Register::X3, Register::X4, Register::X5,
   897→                ];
   898→
   899→                let mut src_regs: Vec<Register> = Vec::new();
   900→                let mut immediate_scratch_used: Vec<usize> = Vec::new(); // which scratch reg each immediate uses
   901→                let mut next_scratch = 0;
   902→
   903→                for (i, arg) in args.iter().enumerate() {
   904→                    if i >= 8 {
   905→                        break;
   906→                    }
   907→                    let is_immediate = matches!(arg, IRValue::Immediate(_) | IRValue::Address(_));
   908→                    if is_immediate {
   909→                        let v = match arg {
   910→                            IRValue::Immediate(v) => *v,
   911→                            IRValue::Address(a) => *a as i64,
   912→                            _ => unreachable!(),
   913→                        };
   914→                        // Load the immediate directly into the target register
   915→                        // if it's not needed by another argument.
   916→                        // For simplicity, always load into a scratch register
   917→                        // and record it.
   918→                        if next_scratch < scratch_regs.len() {
   919→                            let scratch = scratch_regs[next_scratch];
   920→                            next_scratch += 1;
   921→                            self.emit_load_immediate(scratch, v)?;
   922→                            src_regs.push(scratch);
   923→                        } else {
   924→                            // Fallback to X9 (rare, only with 8+ immediates)
   925→                            let scratch = Register::X9;
   926→                            self.emit_load_immediate(scratch, v)?;
   927→                            src_regs.push(scratch);
   928→                        }
   929→                    } else {
   930→                        let src = self.resolve_reg(arg)?;
   931→                        src_regs.push(src);
   932→                    }
   933→                }
   934→
   935→                // Now move arguments into X0–X7. We need to handle the case
   936→                // where a source register is also a target register (cycle).
   937→                // Strategy: first move arguments that don't conflict, then
   938→                // handle conflicts using a scratch register.
   939→                let n = src_regs.len();
   940→                let mut moved = vec![false; n];
   941→
   942→                // Pass 1: move args where src != any dst that hasn't been moved yet
   943→                // Simple approach: iterate and move non-conflicting ones first
   944→                for pass in 0..2 {
   945→                    for i in 0..n {
   946→                        if moved[i] {
   947→                            continue;
   948→                        }
   949→                        let src = src_regs[i];
   950→                        let dst = arg_regs[i];
   951→                        if src == dst {
   952→                            // Already in the right register
   953→                            moved[i] = true;
   954→                            continue;
   955→                        }
   956→                        if pass == 0 {
   957→                            // Check if src is needed as a source by a later unmoved arg
   958→                            let mut conflict = false;
   959→                            for j in 0..n {
   960→                                if !moved[j] && j != i && src_regs[j] == dst {
   961→                                    // Moving arg i to dst would overwrite arg j's source
   962→                                    conflict = true;
   963→                                    break;
   964→                                }
   965→                            }
   966→                            if conflict {
   967→                                continue; // Defer to pass 2
   968→                            }
   969→                        }
   970→                        // Safe to move now
   971→                        self.emit_instruction(Instruction::MOV {
   972→                            rd: dst,
   973→                            rm: src,
   974→                        })?;
   975→                        moved[i] = true;
   976→                    }
   977→                }
   978→
   979→                // Pass 2: handle any remaining conflicts with scratch register
   980→                for i in 0..n {
   981→                    if moved[i] {
   982→                        continue;
   983→                    }
   984→                    let src = src_regs[i];
   985→                    let dst = arg_regs[i];
   986→                    if src == dst {
   987→                        moved[i] = true;
   988→                        continue;
   989→                    }
   990→                    // Use X16 as scratch to break the cycle
   991→                    self.emit_instruction(Instruction::MOV {
   992→                        rd: Register::X16,
   993→                        rm: src,
   994→                    })?;
   995→                    self.emit_instruction(Instruction::MOV {
   996→                        rd: dst,
   997→                        rm: Register::X16,
   998→                    })?;
   999→                    moved[i] = true;
  1000→                }
  1001→
  1002→                // BL — record a relocation for later patching.
  1003→                let bl_word_idx = self.code.len();
  1004→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1005→                self.call_relocs.push(CallRelocation {
  1006→                    text_byte_offset: bl_byte_offset,
  1007→                    target_func: target_name.clone(),
  1008→                });
  1009→                // Register a RelocationEntry so the linker can patch this BL.
  1010→                self.relocations.push(RelocationEntry {
  1011→                    offset: bl_byte_offset,
  1012→                    symbol: target_name.clone(),
  1013→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1014→                });
  1015→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1016→
  1017→                if let Some(dst_val) = dst {
  1018→                    let rd = self.resolve_reg(dst_val)?;
  1019→                    if rd != Register::X0 {
  1020→                        self.emit_instruction(Instruction::MOV {
  1021→                            rd,
  1022→                            rm: Register::X0,
  1023→                        })?;
  1024→                    }
  1025→                }
  1026→            }
  1027→
  1028→            IRInstr::Alloc { dst, size } => {
  1029→                let rd = self.resolve_reg(dst)?;
  1030→                // IMPORTANT: Decrement SP FIRST, then save the new SP value.
  1031→                // If we save before decrementing, the allocation pointer
  1032→                // points to the wrong (too-high) address.
  1033→                let aligned = (*size).div_ceil(16) * 16;
  1034→                self.emit_instruction(Instruction::SUB {
  1035→                    rd: Register::SP,
  1036→                    rn: Register::SP,
  1037→                    rm: Operand::Imm12(aligned as u16),
  1038→                })?;
  1039→                // MOV rd, SP: cannot use ORR because ORR treats Rm=31 as XZR.
  1040→                // Use ADD rd, SP, #0 instead.
  1041→                self.emit_instruction(Instruction::ADD {
  1042→                    rd,
  1043→                    rn: Register::SP,
  1044→                    rm: Operand::Imm12(0),
  1045→                })?;
  1046→            }
  1047→
  1048→            IRInstr::Free { ptr } => {
  1049→                let rt = self.resolve_reg(ptr)?;
  1050→                // Move ptr to X0 (first argument)
  1051→                if rt != Register::X0 {
  1052→                    self.emit_instruction(Instruction::MOV {
  1053→                        rd: Register::X0,
  1054→                        rm: rt,
  1055→                    })?;
  1056→                }
  1057→                // BL __vuma_free
  1058→                let bl_word_idx = self.code.len();
  1059→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1060→                self.call_relocs.push(CallRelocation {
  1061→                    text_byte_offset: bl_byte_offset,
  1062→                    target_func: "__vuma_free".to_string(),
  1063→                });
  1064→                self.relocations.push(RelocationEntry {
  1065→                    offset: bl_byte_offset,
  1066→                    symbol: "__vuma_free".to_string(),
  1067→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1068→                });
  1069→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1070→            }
  1071→
  1072→            IRInstr::Cast { kind, dst, src } => {
  1073→                let rd = self.resolve_reg(dst)?;
  1074→                let rn = self.resolve_reg(src)?;
  1075→                match kind {
  1076→                    CastKind::ZExt => {
  1077→                        // Zero-extend 32-bit to 64-bit using UBFM
  1078→                        self.emit_instruction(Instruction::UBFM {
  1079→                            rd,
  1080→                            rn,
  1081→                            immr: 0,
  1082→                            imms: 31,
  1083→                        })?;
  1084→                    }
  1085→                    CastKind::SExt => {
  1086→                        // Sign-extend 32-bit to 64-bit using SBFM
  1087→                        self.emit_instruction(Instruction::SBFM {
  1088→                            rd,
  1089→                            rn,
  1090→                            immr: 0,
  1091→                            imms: 31,
  1092→                        })?;
  1093→                    }
  1094→                    CastKind::Trunc | CastKind::BitCast => {
  1095→                        // Trunc: upper bits discarded on write — just MOV
  1096→                        // BitCast: no data change — just MOV
  1097→                        if rd != rn {
  1098→                            self.emit_instruction(Instruction::MOV { rd, rm: rn })?;
  1099→                        }
  1100→                    }
  1101→                }
  1102→            }
  1103→
  1104→            IRInstr::Phi { .. } => {
  1105→                log::warn!(
  1106→                    "IRInstr::Phi encountered during emission — should be resolved by SSA pass"
  1107→                );
  1108→            }
  1109→
  1110→            IRInstr::GetAddress { dst, name } => {
  1111→                let rd = self.resolve_reg(dst)?;
  1112→                // Emit a call to __vuma_getaddr to resolve the symbol at runtime.
  1113→                // Move name hash to X0 as the argument.
  1114→                let name_hash = name
  1115→                    .chars()
  1116→                    .fold(0u64, |acc, c| acc.wrapping_mul(31).wrapping_add(c as u64));
  1117→                self.emit_load_immediate(Register::X0, name_hash as i64)?;
  1118→                let bl_word_idx = self.code.len();
  1119→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1120→                self.call_relocs.push(CallRelocation {
  1121→                    text_byte_offset: bl_byte_offset,
  1122→                    target_func: "__vuma_getaddr".to_string(),
  1123→                });
  1124→                self.relocations.push(RelocationEntry {
  1125→                    offset: bl_byte_offset,
  1126→                    symbol: "__vuma_getaddr".to_string(),
  1127→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1128→                });
  1129→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1130→                if rd != Register::X0 {
  1131→                    self.emit_instruction(Instruction::MOV {
  1132→                        rd,
  1133→                        rm: Register::X0,
  1134→                    })?;
  1135→                }
  1136→            }
  1137→
  1138→            IRInstr::Offset { dst, base, offset } => {
  1139→                let rd = self.resolve_reg(dst)?;
  1140→                let rn = self.resolve_reg(base)?;
  1141→                let rm = match offset {
  1142→                    IRValue::Immediate(v) => {
  1143→                        if *v >= 0 && *v <= 4095 {
  1144→                            Operand::Imm12(*v as u16)
  1145→                        } else {
  1146→                            let temp = Register::X9;
  1147→                            self.emit_load_immediate(temp, *v)?;
  1148→                            Operand::Reg {
  1149→                                reg: temp,
  1150→                                shift: None,
  1151→                            }
  1152→                        }
  1153→                    }
  1154→                    _ => Operand::Reg {
  1155→                        reg: self.resolve_reg(offset)?,
  1156→                        shift: None,
  1157→                    },
  1158→                };
  1159→                self.emit_instruction(Instruction::ADD { rd, rn, rm })?;
  1160→            }
  1161→
  1162→            // ── Dedicated arithmetic — delegate to BinOp ──
  1163→            IRInstr::Add { dst, lhs, rhs, ty } => {
  1164→                self.emit_binop(BinOpKind::Add, dst, lhs, rhs, ty.as_ref())?;
  1165→            }
  1166→            IRInstr::Sub { dst, lhs, rhs, ty } => {
  1167→                self.emit_binop(BinOpKind::Sub, dst, lhs, rhs, ty.as_ref())?;
  1168→            }
  1169→            IRInstr::Mul { dst, lhs, rhs, ty } => {
  1170→                self.emit_binop(BinOpKind::Mul, dst, lhs, rhs, ty.as_ref())?;
  1171→            }
  1172→            IRInstr::Div { dst, lhs, rhs, ty } => {
  1173→                self.emit_binop(BinOpKind::SDiv, dst, lhs, rhs, ty.as_ref())?;
  1174→            }
  1175→
  1176→            // ── Comparison instruction ──
  1177→            IRInstr::Cmp {
  1178→                kind,
  1179→                dst,
  1180→                lhs,
  1181→                rhs,
  1182→                ty,
  1183→            } => {
  1184→                let width = RegWidth::from_ir_type(ty.as_ref());
  1185→                let rd = self.resolve_reg(dst)?;
  1186→                // IMPORTANT: resolve_reg uses X9 for immediates, so if both lhs and rhs
  1187→                // are immediates, the second load would overwrite the first. Use X10 for
  1188→                // the RHS when both are immediates.
  1189→                let rn = self.resolve_reg(lhs)?;
  1190→                let rm = match rhs {
  1191→                    IRValue::Immediate(_) | IRValue::Address(_) => {
  1192→                        // If lhs was also an immediate, it's in X9; use X10 for rhs
  1193→                        let temp = if matches!(lhs, IRValue::Immediate(_) | IRValue::Address(_)) {
  1194→                            Register::X10
  1195→                        } else {
  1196→                            Register::X9
  1197→                        };
  1198→                        match rhs {
  1199→                            IRValue::Immediate(v) => self.emit_load_immediate(temp, *v)?,
  1200→                            IRValue::Address(a) => self.emit_load_immediate(temp, *a as i64)?,
  1201→                            _ => unreachable!(),
  1202→                        }
  1203→                        temp
  1204→                    }
  1205→                    _ => self.resolve_reg(rhs)?,
  1206→                };
  1207→                self.emit_instruction_with_width(
  1208→                    Instruction::CMP {
  1209→                        rn,
  1210→                        rm: Operand::Reg {
  1211→                            reg: rm,
  1212→                            shift: None,
  1213→                        },
  1214→                    },
  1215→                    width,
  1216→                )?;
  1217→                let cond = cmp_kind_to_condition(kind);
  1218→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1219→            }
  1220→
  1221→            // ── Instruction-level control flow ──
  1222→            IRInstr::Ret { values } => {
  1223→                for (i, val) in values.iter().enumerate() {
  1224→                    if i >= 8 {
  1225→                        break;
  1226→                    }
  1227→                    let src = self.resolve_reg(val)?;
  1228→                    let dst_reg = match i {
  1229→                        0 => Register::X0,
  1230→                        1 => Register::X1,
  1231→                        2 => Register::X2,
  1232→                        3 => Register::X3,
  1233→                        4 => Register::X4,
  1234→                        5 => Register::X5,
  1235→                        6 => Register::X6,
  1236→                        7 => Register::X7,
  1237→                        _ => unreachable!(),
  1238→                    };
  1239→                    if src != dst_reg {
  1240→                        self.emit_instruction(Instruction::MOV {
  1241→                            rd: dst_reg,
  1242→                            rm: src,
  1243→                        })?;
  1244→                    }
  1245→                }
  1246→            }
  1247→
  1248→            IRInstr::Branch { target } => {
  1249→                let fixup_idx = self.code.len();
  1250→                self.fixups.push((fixup_idx, target.clone(), BranchFormat::B26));
  1251→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1252→            }
  1253→
  1254→            IRInstr::CondBranch {
  1255→                cond,
  1256→                true_target,
  1257→                false_target,
  1258→            } => {
  1259→                let rt = self.resolve_reg(cond)?;
  1260→                // CBNZ: if cond != 0, branch to true_target
  1261→                let fixup_cbz = self.code.len();
  1262→                self.fixups.push((fixup_cbz, true_target.clone(), BranchFormat::Cond19));
  1263→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1264→                // Otherwise (cond == 0), branch to false_target
  1265→                let fixup_b = self.code.len();
  1266→                self.fixups.push((fixup_b, false_target.clone(), BranchFormat::B26));
  1267→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1268→            }
  1269→
  1270→            IRInstr::Select {
  1271→                dst,
  1272→                cond,
  1273→                true_val,
  1274→                false_val,
  1275→                ty,
  1276→            } => {
  1277→                let width = RegWidth::from_ir_type(ty.as_ref());
  1278→                // Lower select as: SUBS XZR, cond, #0; CSEL dst, false_val, true_val, NE
  1279→                let rd = self.resolve_reg(dst)?;
  1280→                let rc = self.resolve_reg(cond)?;
  1281→                let rt = self.resolve_reg(true_val)?;
  1282→                let rf = self.resolve_reg(false_val)?;
  1283→                // Compare cond against zero and select.
  1284→                self.emit_instruction_with_width(
  1285→                    Instruction::SUB {
  1286→                        rd: Register::XZR,
  1287→                        rn: rc,
  1288→                        rm: Operand::Imm12(0),
  1289→                    },
  1290→                    width,
  1291→                )?;
  1292→                // Set flags by using a separate CMP (SUB with XZR destination
  1293→                // doesn't set flags; we need a flags-setting variant).
  1294→                // We emulate this with: CMP rc, #0 which is SUBS XZR, rc, #0.
  1295→                // Since we only have SUB, we use the existing CMP pattern.
  1296→                self.emit_instruction_with_width(
  1297→                    Instruction::CSEL {
  1298→                        rd,
  1299→                        rn: rf,
  1300→                        rm: rt,
  1301→                        cond: crate::arm64::Condition::NE,
  1302→                    },
  1303→                    width,
  1304→                )?;
  1305→            }
  1306→        }
  1307→        Ok(())
  1308→    }

  1310→    /// Emit a binary operation (shared by `BinOp` and dedicated `Add`/`Sub`/…).
  1311→    ///
  1312→    /// Some ARM64 instructions (ADD, SUB, LSL, LSR, ASR) accept a 12-bit
  1313→    /// unsigned immediate operand directly.  Most others (MUL, AND, ORR, EOR,
  1314→    /// SDIV, UDIV, remainder, comparisons) **require** a register operand.
  1315→    /// When the RHS is an immediate that the target instruction cannot accept
  1316→    /// directly, we "spill" it into a scratch register (X16 / IP0) first via
  1317→    /// MOVZ + MOVK and then use that register as the operand.
  1318→    ///
  1319→    /// X16 (IP0) is chosen as the scratch register because it is **not** in
  1320→    /// the register allocator's free pool, so it will never conflict with
  1321→    /// `rd` or `rn`.
  1322→    fn emit_binop(
  1323→        &mut self,
  1324→        op: BinOpKind,
  1325→        dst: &IRValue,
  1326→        lhs: &IRValue,
  1327→        rhs: &IRValue,
  1328→        ty: Option<&IRType>,
  1329→    ) -> Result<()> {
  1330→        let width = RegWidth::from_ir_type(ty);
  1331→        let rd = self.resolve_reg(dst)?;
  1332→        let rn = self.resolve_reg(lhs)?;
  1333→
  1334→        // Determine whether the operation can accept an immediate operand.
  1335→        // ADD, SUB, LSL, LSR, ASR all have immediate forms in ARM64.
  1336→        // All other operations require a register operand.
  1337→        let supports_imm = matches!(
  1338→            op,
  1339→            BinOpKind::Add | BinOpKind::Sub | BinOpKind::Shl | BinOpKind::ShrL | BinOpKind::ShrA
  1340→        );
  1341→
  1342→        let rm = match rhs {
  1343→            IRValue::Immediate(v) => {
  1344→                if supports_imm && *v >= 0 && *v <= 4095 {
  1345→                    // Small immediate that the instruction accepts directly.
  1346→                    Operand::Imm12(*v as u16)
  1347→                } else {
  1348→                    // Load the immediate into scratch register X16 (IP0).
  1349→                    // X16 is not in the register allocator's pool, so it cannot
  1350→                    // conflict with `rd` or `rn`.
  1351→                    let scratch = Register::X16;
  1352→                    self.emit_load_immediate_with_width(scratch, *v, width)?;
  1353→                    Operand::Reg {
  1354→                        reg: scratch,
  1355→                        shift: None,
  1356→                    }
  1357→                }
  1358→            }
  1359→            _ => Operand::Reg {
  1360→                reg: self.resolve_reg(rhs)?,
  1361→                shift: None,
  1362→            },
  1363→        };
  1364→
  1365→        match op {
  1366→            BinOpKind::Add => {
  1367→                self.emit_instruction_with_width(Instruction::ADD { rd, rn, rm }, width)?;
  1368→            }
  1369→            BinOpKind::Sub => {
  1370→                self.emit_instruction_with_width(Instruction::SUB { rd, rn, rm }, width)?;
  1371→            }
  1372→            BinOpKind::Mul => {
  1373→                let rm_reg = self.operand_to_reg(&rm)?;
  1374→                self.emit_instruction_with_width(Instruction::MUL { rd, rn, rm: rm_reg }, width)?;
  1375→            }
  1376→            BinOpKind::SDiv => {
  1377→                let rm_reg = self.operand_to_reg(&rm)?;
  1378→                self.emit_instruction_with_width(Instruction::SDIV { rd, rn, rm: rm_reg }, width)?;
  1379→            }
  1380→            BinOpKind::UDiv => {
  1381→                let rm_reg = self.operand_to_reg(&rm)?;
  1382→                self.emit_instruction_with_width(Instruction::UDIV { rd, rn, rm: rm_reg }, width)?;
  1383→            }
  1384→            BinOpKind::And => {
  1385→                let rm_reg = self.operand_to_reg(&rm)?;
  1386→                self.emit_instruction_with_width(Instruction::AND { rd, rn, rm: rm_reg }, width)?;
  1387→            }
  1388→            BinOpKind::Or => {
  1389→                let rm_reg = self.operand_to_reg(&rm)?;
  1390→                self.emit_instruction_with_width(Instruction::ORR { rd, rn, rm: rm_reg }, width)?;
  1391→            }
  1392→            BinOpKind::Xor => {
  1393→                let rm_reg = self.operand_to_reg(&rm)?;
  1394→                self.emit_instruction_with_width(Instruction::EOR { rd, rn, rm: rm_reg }, width)?;
  1395→            }
  1396→            BinOpKind::Shl => {
  1397→                self.emit_instruction_with_width(Instruction::LSL { rd, rn, rm }, width)?;
  1398→            }
  1399→            BinOpKind::ShrL => {
  1400→                self.emit_instruction_with_width(Instruction::LSR { rd, rn, rm }, width)?;
  1401→            }
  1402→            BinOpKind::ShrA => {
  1403→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm }, width)?;
  1404→            }
  1405→            BinOpKind::Ror | BinOpKind::Rol => {
  1406→                let rm_reg = self.operand_to_reg(&rm)?;
  1407→                // ARM64 rotation: use ASR as placeholder
  1408→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm: Operand::Reg { reg: rm_reg, shift: None } }, width)?;
  1409→            }
  1410→            BinOpKind::SRem | BinOpKind::URem => {
  1411→                let rm_reg = self.operand_to_reg(&rm)?;
  1412→                let div_instr = if op == BinOpKind::SRem {
  1413→                    Instruction::SDIV { rd, rn, rm: rm_reg }
  1414→                } else {
  1415→                    Instruction::UDIV { rd, rn, rm: rm_reg }
  1416→                };
  1417→                self.emit_instruction_with_width(div_instr, width)?;
  1418→                // MSUB rd, rd, rm, rn  =>  rd = rn - rd * rm  =  dividend - quotient * divisor
  1419→                self.emit_instruction_with_width(Instruction::MSUB {
  1420→                    rd,
  1421→                    rn: rd,     // quotient (result of DIV)
  1422→                    rm: rm_reg, // divisor
  1423→                    ra: rn,     // dividend
  1424→                }, width)?;
  1425→            }
  1426→            BinOpKind::SLt
  1427→            | BinOpKind::SLe
  1428→            | BinOpKind::SGt
  1429→            | BinOpKind::SGe
  1430→            | BinOpKind::ULt
  1431→            | BinOpKind::ULe
  1432→            | BinOpKind::UGt
  1433→            | BinOpKind::UGe
  1434→            | BinOpKind::Eq
  1435→            | BinOpKind::Ne => {
  1436→                let rm_reg = self.operand_to_reg(&rm)?;
  1437→                self.emit_instruction_with_width(Instruction::CMP {
  1438→                    rn,
  1439→                    rm: Operand::Reg {
  1440→                        reg: rm_reg,
  1441→                        shift: None,
  1442→                    },
  1443→                }, width)?;
  1444→                let cond = binop_kind_to_condition(&op);
  1445→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1446→            }
  1447→        }
  1448→        Ok(())
  1449→    }

  1451→    /// Emit the block terminator.
  1452→    fn emit_terminator(&mut self, term: &IRTerminator) -> Result<()> {
  1453→        match term {
  1454→            IRTerminator::Jump(target) => {
  1455→                let fixup_idx = self.code.len();
  1456→                self.fixups.push((fixup_idx, target.clone(), BranchFormat::B26));
  1457→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1458→            }
  1459→            IRTerminator::Branch {
  1460→                cond,
  1461→                true_block,
  1462→                false_block,
  1463→            } => {
  1464→                let rt = self.resolve_reg(cond)?;
  1465→                let fixup_cbnz = self.code.len();
  1466→                self.fixups.push((fixup_cbnz, true_block.clone(), BranchFormat::Cond19));
  1467→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1468→                let fixup_b = self.code.len();
  1469→                self.fixups.push((fixup_b, false_block.clone(), BranchFormat::B26));
  1470→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1471→            }
  1472→            IRTerminator::Return(vals) => {
  1473→                for (i, val) in vals.iter().enumerate() {
  1474→                    if i >= 8 {
  1475→                        break;
  1476→                    }
  1477→                    let src = self.resolve_reg(val)?;
  1478→                    let dst_reg = match i {
  1479→                        0 => Register::X0,
  1480→                        1 => Register::X1,
  1481→                        2 => Register::X2,
  1482→                        3 => Register::X3,
  1483→                        4 => Register::X4,
  1484→                        5 => Register::X5,
  1485→                        6 => Register::X6,
  1486→                        7 => Register::X7,
  1487→                        _ => unreachable!(),
  1488→                    };
  1489→                    if src != dst_reg {
  1490→                        self.emit_instruction(Instruction::MOV {
  1491→                            rd: dst_reg,
  1492→                            rm: src,
  1493→                        })?;
  1494→                    }
  1495→                }
  1496→                // Restore SP to point to the saved FP/LR pair.
  1497→                // X29 was set to SP right after the prologue's SUB SP, SP, #16
  1498→                // and STP X29, X30, [SP]. So X29 points to the FP/LR save area.
  1499→                // Using MOV SP, X29 (via ADD SP, X29, #0) is the most robust
  1500→                // way to restore SP regardless of how many Alloc instructions
  1501→                // modified it during the function body.
  1502→                self.emit_instruction(Instruction::ADD {
  1503→                    rd: Register::SP,
  1504→                    rn: Register::X29,
  1505→                    rm: Operand::Imm12(0),
  1506→                })?;
  1507→                // FP/LR were stored at [SP] after the prologue's SUB SP, SP, #16.
  1508→                // After the ADD above, SP points back to the FP/LR save area.
  1509→                // Load from [SP, #0], then restore SP by adding 16.
  1510→                self.emit_instruction(Instruction::LDP {
  1511→                    rt1: Register::X29,
  1512→                    rt2: Register::X30,
  1513→                    rn: Register::SP,
  1514→                    offset: 0,
  1515→                })?;
  1516→                self.emit_instruction(Instruction::ADD {
  1517→                    rd: Register::SP,
  1518→                    rn: Register::SP,
  1519→                    rm: Operand::Imm12(16),
  1520→                })?;
  1521→                self.emit_instruction(Instruction::RET { rn: None })?;
  1522→            }
  1523→            IRTerminator::Unreachable => {
  1524→                self.emit_instruction(Instruction::MOV {
  1525→                    rd: Register::XZR,
  1526→                    rm: Register::XZR,
  1527→                })?;
  1528→            }
  1529→            // Switch, Invoke, TailCall, and Resume are lowered by the
  1530→            // control_flow module before reaching the emitter. If they
  1531→            // appear here, it means the lowering pass was not run.
  1532→            IRTerminator::Switch { .. } => {
  1533→                return Err(CodegenError::InvalidInstruction(
  1534→                    "Switch terminator must be lowered before emission".to_string(),
  1535→                ));
  1536→            }
  1537→            IRTerminator::Invoke { .. } => {
  1538→                return Err(CodegenError::InvalidInstruction(
  1539→                    "Invoke terminator must be lowered before emission".to_string(),
  1540→                ));
  1541→            }
  1542→            IRTerminator::TailCall { .. } => {
  1543→                return Err(CodegenError::InvalidInstruction(
  1544→                    "TailCall terminator must be lowered before emission".to_string(),
  1545→                ));
  1546→            }
  1547→            IRTerminator::Resume { .. } => {
  1548→                return Err(CodegenError::InvalidInstruction(
  1549→                    "Resume terminator must be lowered before emission".to_string(),
  1550→                ));
  1551→            }
  1552→        }
  1553→        Ok(())
  1554→    }

  1566→    /// Emit an instruction with a specific register width (32-bit W or 64-bit X).
  1567→    ///
  1568→    /// This is the primary emission method for arithmetic and logical instructions
  1569→    /// where the operand width affects the encoding. Using `RegWidth::W32` produces
  1570→    /// instructions that operate on W sub-registers, giving automatic 32-bit
  1571→    /// wrapping arithmetic.
  1572→    fn emit_instruction_with_width(&mut self, instr: Instruction, width: RegWidth) -> Result<()> {
  1573→        let word = instr.encode_with_width(width)?;
  1574→        self.code.push(word);
  1575→        Ok(())
  1576→    }

  1578→    /// Emit a load instruction with the given offset and IR type.
  1579→    ///
  1580→    /// Selects the correct ARM64 load variant based on the IR type:
  1581→    /// - I8/U8 → LDRB (byte, zero-extended)
  1582→    /// - I16/U16 → LDRH (halfword, zero-extended)
  1583→    /// - I32/U32/Ptr/Func → LDR (word or doubleword)
  1584→    /// - I64/U64 → LDR (doubleword)
  1585→    ///
  1586→    /// If the offset fits the ARM64 unsigned-offset immediate encoding for the
  1587→    /// selected instruction variant, it is encoded directly. Otherwise, the
  1588→    /// effective address is computed in a scratch register (X9) first, and
  1589→    /// the load is performed at [X9 + 0].
  1590→    fn emit_load(&mut self, rt: Register, rn: Register, offset: i32, ty: &IRType) -> Result<()> {
  1591→        // Determine the ARM64 load instruction and its immediate-offset constraints.
  1592→        // For each variant: (scale, max_imm12) where the encoded offset = imm12 << scale.
  1593→        // LDRB:  scale=0, imm12 0..4095 → offset 0..4095
  1594→        // LDRH:  scale=1, imm12 0..4095 → offset 0..8190, even
  1595→        // LDRSW: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1596→        // LDR W: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1597→        // LDR X: scale=3, imm12 0..4095 → offset 0..32760, multiple of 8
  1598→        match ty {
  1599→            IRType::I8 | IRType::U8 => {
  1600→                if offset >= 0 && offset <= 4095 {
  1601→                    self.emit_instruction(Instruction::LDRB { rt, rn, offset })?;
  1602→                } else {
  1603→                    self.emit_address_with_offset(rn, offset)?;
  1604→                    self.emit_instruction(Instruction::LDRB {
  1605→                        rt,
  1606→                        rn: Register::X9,
  1607→                        offset: 0,
  1608→                    })?;
  1609→                }
  1610→                // Sign-extend for signed byte loads (SXTB = SBFM Xd, Xn, #0, #7)
  1611→                if *ty == IRType::I8 {
  1612→                    self.emit_instruction(Instruction::SBFM {
  1613→                        rd: rt,
  1614→                        rn: rt,
  1615→                        immr: 0,

  1807→    fn resolve_reg(&mut self, val: &IRValue) -> Result<Register> {
  1808→        match val {
  1809→            IRValue::Register(id) => {
  1810→                let result = self.reg_alloc.allocate(*id)?;
  1811→                let reg = result.reg;
  1812→

  1814→                // This prevents subsequent resolve_reg calls within the same
  1815→                // instruction from spilling this register.

  1817→                    self.reg_alloc.pin(reg);
  1818→                    self.instr_pinned_regs.push(reg);

  1822→                // value to the stack before it gets overwritten.

  1824→                    // Spill slot offset from X29 (frame pointer).

  1826→                    // negative offsets from X29: slot 0 at [X29, #-8],
  1827→                    // slot 1 at [X29, #-16], etc.
  1828→                    let sp_offset = 8 + (spill_info.slot as i32) * 8;

  1833→                        rd: Register::X16,
  1834→                        rn: Register::X29,
  1835→                        rm: Operand::Reg {
  1836→                            reg: Register::X16,
  1837→                            shift: None,
  1838→                        },
  1839→                    })?;
  1840→                    self.emit_instruction(Instruction::STR {
  1841→                        rt: spill_info.reg,
  1842→                        rn: Register::X16,
  1843→                        offset: 0,
  1844→                    })?;
  1845→                }

  1848→                // emit LDR to restore its value from the stack.

  1855→                        rn: Register::X29,

  1857→                            reg: Register::X16,
  1858→                            shift: None,
  1859→                        },
  1860→                    })?;

  1862→                        rt: reg,
  1863→                        rn: Register::X16,
  1864→                        offset: 0,
  1865→                    })?;
  1866→                }

  1886→    fn emit_load_immediate(&mut self, rd: Register, value: i64) -> Result<()> {
  1887→        if (0..=65535).contains(&value) {
  1888→            self.emit_instruction(Instruction::MOVZ {
  1889→                rd,
  1890→                imm16: value as u16,
  1891→                shift: 0,
  1892→            })?;

  1900→                imm16: lo,
  1901→                shift: 0,

  1904→                rd,
  1905→                imm16: hi,
  1906→                shift: 16,
  1907→            })?;

  1911→        let w1 = ((value >> 16) & 0xFFFF) as u16;

  1929→                imm16: w2,
