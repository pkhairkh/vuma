     1→//! # VUMA Compilation Pipeline
     2→//!
     3→//! The full compilation pipeline that wires together every workspace crate:
     4→//!
     5→//! ```text
     6→//! Source → Parse → AST → SCG → BD Inference → MSG Construction
     7→//!        → IVE Verification → SCG Transforms → IR Lowering
     8→//!        → Register Allocation → ARM64 Codegen → ELF Emission
     9→//! ```
    10→//!
    11→//! ## Quick Start
    12→//!
    13→//! ```rust
    14→//! use vuma::pipeline::{compile, CompileConfig, CompileTarget, OptLevel, VerificationLevel};
    15→//!
    16→//! let source = r#"
    17→//!     region buf = allocate(256);
    18→//!     fn main() {
    19→//!         ptr = buf + 64;
    20→//!         header = ptr as *NodeHeader;
    21→//!     }
    22→//! "#;
    23→//!
    24→//! let config = CompileConfig::default();
    25→//! let output = compile(source, &config);
    26→//! match output {
    27→//!     Ok(out) => println!("Compiled {} bytes, {} SCG nodes", out.binary.len(), out.scg.node_count()),
    28→//!     Err(errors) => {
    29→//!         for err in &errors {
    30→//!             eprintln!("{}", err);
    31→//!         }
    32→//!     }
    33→//! }
    34→//! ```

    91→/// Optimization level.

    93→    Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize, Default,

    95→pub enum OptLevel {

   105→}

   109→        match self {
   110→            OptLevel::O0 => write!(f, "O0"),
   111→            OptLevel::O1 => write!(f, "O1"),
   112→            OptLevel::O2 => write!(f, "O2"),
   113→            OptLevel::O3 => write!(f, "O3"),
   114→        }
   115→    }

   170→            verification_level: VerificationLevel::Quick,

   210→/// A unified error type for the VUMA compilation pipeline.
   211→///
   212→/// Each variant captures the pipeline stage where the error occurred
   213→/// and the underlying cause.
   214→#[derive(Debug, Clone)]
   215→pub enum VumaError {
   216→    /// Error during lexing or parsing.
   217→    Parse {
   218→        /// The parse errors.
   219→        errors: Vec<ParseError>,
   220→    },
   221→    /// Error converting AST to SCG.
   222→    AstToScg {
   223→        /// Error message.
   224→        message: String,
   225→    },
   226→    /// SCG validation failed.
   227→    ScgValidation {
   228→        /// Validation error messages.
   229→        errors: Vec<String>,
   230→    },
   231→    /// SCG → MSG conversion error.
   232→    ScgToMsg {
   233→        /// The conversion error.
   234→        error: ConversionError,
   235→    },
   236→    /// BD inference error.
   237→    BdInference {
   238→        /// Node ID where inference failed, if known.
   239→        node_id: Option<u64>,
   240→        /// Error message.
   241→        message: String,
   242→    },
   243→    /// IVE verification failure (one or more invariants violated).
   244→    Verification {
   245→        /// The aggregated verification result.
   246→        result: AggregatedResult,
   247→    },
   248→    /// SCG transformation pass error.
   249→    Transform {
   250→        /// Name of the pass that failed.
   251→        pass_name: String,
   252→        /// Error messages from the pass.
   253→        errors: Vec<String>,
   254→    },
   255→    /// IR lowering / codegen error.
   256→    Codegen {
   257→        /// The codegen error.
   258→        error: CodegenError,
   259→    },
   260→    /// Register allocation failure.
   261→    RegisterAlloc {
   262→        /// Error message.
   263→        message: String,
   264→    },
   265→    /// ELF emission failure.
   266→    Emission {
   267→        /// Error message.
   268→        message: String,
   269→    },
   270→    /// COR initialization failure.
   271→    CorInit {
   272→        /// Error message.
   273→        message: String,
   274→    },
   275→    /// A collection of errors accumulated across stages.
   276→    Multi {
   277→        /// The collected errors.
   278→        errors: Vec<VumaError>,
   279→    },
   280→}
   281→
   282→impl VumaError {
   283→    /// Returns the pipeline stage that produced this error.
   284→    pub fn stage(&self) -> &'static str {
   285→        match self {
   286→            VumaError::Parse { .. } => "parse",
   287→            VumaError::AstToScg { .. } => "ast-to-scg",
   288→            VumaError::ScgValidation { .. } => "scg-validation",
   289→            VumaError::ScgToMsg { .. } => "scg-to-msg",
   290→            VumaError::BdInference { .. } => "bd-inference",
   291→            VumaError::Verification { .. } => "ive-verification",
   292→            VumaError::Transform { .. } => "scg-transform",
   293→            VumaError::Codegen { .. } => "codegen",
   294→            VumaError::RegisterAlloc { .. } => "register-alloc",
   295→            VumaError::Emission { .. } => "elf-emission",
   296→            VumaError::CorInit { .. } => "cor-init",
   297→            VumaError::Multi { .. } => "multi",
   298→        }
   299→    }
   300→}
   301→
   302→impl fmt::Display for VumaError {
   303→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   304→        match self {
   305→            VumaError::Parse { errors } => {
   306→                write!(f, "[parse] {} error(s):", errors.len())?;
   307→                for e in errors {
   308→                    write!(f, "\n  - {}", e)?;
   309→                }
   310→                Ok(())
   311→            }
   312→            VumaError::AstToScg { message } => write!(f, "[ast-to-scg] {}", message),
   313→            VumaError::ScgValidation { errors } => {
   314→                write!(f, "[scg-validation] {} error(s):", errors.len())?;
   315→                for e in errors {
   316→                    write!(f, "\n  - {}", e)?;
   317→                }
   318→                Ok(())
   319→            }
   320→            VumaError::ScgToMsg { error } => write!(f, "[scg-to-msg] {}", error),
   321→            VumaError::BdInference { node_id, message } => {
   322→                write!(f, "[bd-inference] {}", message)?;
   323→                if let Some(id) = node_id {
   324→                    write!(f, " (node {})", id)?;
   325→                }
   326→                Ok(())
   327→            }
   328→            VumaError::Verification { result } => {
   329→                write!(f, "[ive-verification] verdict: {}", result.overall)
   330→            }
   331→            VumaError::Transform { pass_name, errors } => {
   332→                write!(f, "[scg-transform:{}] {} error(s)", pass_name, errors.len())
   333→            }
   334→            VumaError::Codegen { error } => write!(f, "[codegen] {}", error),
   335→            VumaError::RegisterAlloc { message } => write!(f, "[register-alloc] {}", message),
   336→            VumaError::Emission { message } => write!(f, "[elf-emission] {}", message),
   337→            VumaError::CorInit { message } => write!(f, "[cor-init] {}", message),
   338→            VumaError::Multi { errors } => {
   339→                write!(f, "multiple errors ({}):", errors.len())?;
   340→                for (i, e) in errors.iter().enumerate() {
   341→                    write!(f, "\n{}. {}", i + 1, e)?;
   342→                }
   343→                Ok(())
   344→            }
   345→        }
   346→    }
   347→}
   348→
   349→impl std::error::Error for VumaError {}

   398→/// A fingerprint of a source file, used to detect changes for
   399→/// incremental compilation.
   400→#[derive(Debug, Clone, PartialEq, Eq)]
   401→pub struct SourceFingerprint {
   402→    /// A hash of the source text.
   403→    pub hash: u64,
   404→    /// Byte length of the source.
   405→    pub len: usize,
   406→}
   407→
   408→impl SourceFingerprint {
   409→    /// Compute a fingerprint from source text.
   410→    pub fn from_source(source: &str) -> Self {
   411→        // Simple FNV-1a hash — sufficient for change detection.
   412→        let mut hash: u64 = 0xcbf29ce484222325;
   413→        for byte in source.bytes() {
   414→            hash ^= byte as u64;
   415→            hash = hash.wrapping_mul(0x100000001b3);
   416→        }
   417→        Self {
   418→            hash,
   419→            len: source.len(),
   420→        }
   421→    }
   422→}

   444→/// Identifies a pipeline stage for incremental invalidation.
   445→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
   446→pub enum PipelineStage {
   447→    /// Lexing + parsing.
   448→    Parse,
   449→    /// AST → SCG conversion.
   450→    AstToScg,
   451→    /// SCG validation.
   452→    ScgValidation,
   453→    /// BD inference.
   454→    BdInference,
   455→    /// SCG → MSG construction.
   456→    MsgConstruction,
   457→    /// IVE verification.
   458→    IveVerification,
   459→    /// SCG transformation passes.
   460→    ScgTransforms,
   461→    /// IR lowering (SCG → IR).
   462→    IrLowering,
   463→    /// Register allocation.
   464→    RegisterAlloc,
   465→    /// ARM64 code emission.
   466→    CodeEmission,
   467→    /// COR (Continuous Optimization Runtime) initialization.
   468→    CorInit,
   469→}

   471→impl PipelineStage {
   472→    /// All stages in order.
   473→    pub fn all() -> &'static [PipelineStage; 11] {
   474→        &[
   475→            PipelineStage::Parse,
   476→            PipelineStage::AstToScg,
   477→            PipelineStage::ScgValidation,
   478→            PipelineStage::BdInference,
   479→            PipelineStage::MsgConstruction,
   480→            PipelineStage::IveVerification,
   481→            PipelineStage::ScgTransforms,
   482→            PipelineStage::IrLowering,
   483→            PipelineStage::RegisterAlloc,
   484→            PipelineStage::CodeEmission,
   485→            PipelineStage::CorInit,
   486→        ]
   487→    }

   489→    /// Returns all stages from (and including) the given stage onwards.
   490→    pub fn from(stage: PipelineStage) -> Vec<PipelineStage> {
   491→        PipelineStage::all()
   492→            .iter()
   493→            .filter(|&&s| s >= stage)
   494→            .copied()
   495→            .collect()
   496→    }
   497→}

   521→// ── Edge Index ─────────────────────────────────────────────────────────
   522→
   523→/// Pre-computed edge index for efficient graph traversal during bridge
   524→/// conversion. Built once from all edges in the SCG and then queried
   525→/// by node ID and edge kind.
   526→struct EdgeIndex {
   527→    /// Outgoing edges keyed by source node.
   528→    outgoing: HashMap<NodeId, Vec<EdgeData>>,
   529→    /// Incoming edges keyed by target node.
   530→    incoming: HashMap<NodeId, Vec<EdgeData>>,
   531→}
   532→
   533→impl EdgeIndex {
   534→    /// Build the edge index from all edges in the SCG.
   535→    fn build(scg: &SCG) -> Self {
   536→        let mut outgoing: HashMap<NodeId, Vec<EdgeData>> = HashMap::new();
   537→        let mut incoming: HashMap<NodeId, Vec<EdgeData>> = HashMap::new();
   538→        for edge in scg.edges() {
   539→            outgoing.entry(edge.source).or_default().push(edge.clone());
   540→            incoming.entry(edge.target).or_default().push(edge.clone());
   541→        }
   542→        Self { outgoing, incoming }
   543→    }
   544→
   545→    /// Get outgoing ControlFlow edges from a node.
   546→    fn outgoing_cf(&self, id: NodeId) -> Vec<&EdgeData> {
   547→        self.outgoing
   548→            .get(&id)
   549→            .map(|edges| {
   550→                edges
   551→                    .iter()
   552→                    .filter(|e| e.kind == EdgeKind::ControlFlow)
   553→                    .collect()
   554→            })
   555→            .unwrap_or_default()
   556→    }
   557→
   558→    /// Get incoming DataFlow edges to a node.
   559→    fn incoming_df(&self, id: NodeId) -> Vec<&EdgeData> {
   560→        self.incoming
   561→            .get(&id)
   562→            .map(|edges| {
   563→                edges
   564→                    .iter()
   565→                    .filter(|e| e.kind == EdgeKind::DataFlow)
   566→                    .collect()
   567→            })
   568→            .unwrap_or_default()
   569→    }
   570→
   571→    /// Get outgoing DataFlow edges from a node.
   572→    fn outgoing_df(&self, id: NodeId) -> Vec<&EdgeData> {
   573→        self.outgoing
   574→            .get(&id)
   575→            .map(|edges| {
   576→                edges
   577→                    .iter()
   578→                    .filter(|e| e.kind == EdgeKind::DataFlow)
   579→                    .collect()
   580→            })
   581→            .unwrap_or_default()
   582→    }
   583→}

   587→/// Generate a variable name for a node with a given prefix.
   588→fn node_var(id: NodeId, _prefix: &str) -> String {
   589→    // Must match the naming convention in resolve_df_input so that
   590→    // source references (v_{source_id}) resolve correctly in the
   591→    // codegen IR builder's names map.
   592→    format!("v_{}", id.as_u64())
   593→}
   594→
   595→/// Resolve a DataFlow input for a node, returning a `ScgExpr` referencing
   596→/// the variable produced by the source node of the DataFlow edge at the
   597→/// given position.
   598→///
   599→/// If the source node is a Control node (FunctionEntry, etc.) that does not
   600→/// produce a named variable, falls back to `ScgExpr::Int(0)` to avoid
   601→/// referencing a non-existent variable in the codegen IR.
   602→fn resolve_df_input(
   603→    node_id: NodeId,
   604→    position: usize,
   605→    edge_idx: &EdgeIndex,
   606→    scg: &SCG,
   607→) -> ScgExpr {
   608→    let df_inputs = edge_idx.incoming_df(node_id);
   609→    if position < df_inputs.len() {
   610→        let source = df_inputs[position].source;
   611→        // Check if the source node is a productive node that defines a variable.
   612→        // Control nodes (FunctionEntry, Branch, etc.) and Phantom nodes do not
   613→        // produce named variables — skip them to avoid UnknownVariable errors.
   614→        if let Some(src_data) = scg.get_node(source) {
   615→            match &src_data.payload {
   616→                // These node types do NOT produce named variables in the
   617→                // codegen SCG.  Control/Phantom are structural; Deallocation
   618→                // and Effect are lowered to runtime calls (no dst vreg).
   619→                NodePayload::Control(_)
   620→                | NodePayload::Phantom(_)
   621→                | NodePayload::Deallocation(_)
   622→                | NodePayload::Effect(_)
   623→                | NodePayload::VTable(_)
   624→                | NodePayload::ClosureEnv(_) => {
   625→                    // Non-productive source — use 0 as placeholder
   626→                    ScgExpr::Int(0)
   627→                }
   628→                _ => ScgExpr::Var(format!("v_{}", source.as_u64())),
   629→            }
   630→        } else {
   631→            ScgExpr::Int(0)
   632→        }
   633→    } else {
   634→        // No DataFlow edge at this position — use 0 as placeholder
   635→        ScgExpr::Int(0)
   636→    }
   637→}

   639→/// Resolve the condition expression for a Branch node by looking at its
   640→/// incoming DataFlow edges.
   641→fn resolve_branch_cond(branch_id: NodeId, edge_idx: &EdgeIndex, scg: &SCG) -> ScgExpr {
   642→    resolve_df_input(branch_id, 0, edge_idx, scg)
   643→}

   647→/// Find the `FunctionReturn` node reachable from a `FunctionEntry` via
   648→/// ControlFlow edges, using BFS.
   649→fn find_function_return(entry_id: NodeId, scg: &SCG, edge_idx: &EdgeIndex) -> Option<NodeId> {
   650→    let mut visited = HashSet::new();
   651→    let mut queue = VecDeque::new();
   652→    queue.push_back(entry_id);
   653→    visited.insert(entry_id);
   654→
   655→    while let Some(current) = queue.pop_front() {
   656→        for edge in edge_idx.outgoing_cf(current) {
   657→            let target = edge.target;
   658→            if visited.contains(&target) {
   659→                continue;
   660→            }
   661→            visited.insert(target);
   662→            if let Some(node) = scg.get_node(target) {
   663→                if let NodePayload::Control(c) = &node.payload {
   664→                    if c.kind == ControlKind::FunctionReturn {
   665→                        return Some(target);
   666→                    }
   667→                }
   668→            }
   669→            queue.push_back(target);
   670→        }
   671→    }
   672→
   673→    None
   674→}
   675→
   676→/// Find all `Join` nodes reachable from `start` via ControlFlow edges,
   677→/// stopping at the first Join encountered on each path (Joins are
   678→/// convergence points, not passed through during search).
   679→fn find_reachable_joins(start: NodeId, scg: &SCG, edge_idx: &EdgeIndex) -> Vec<NodeId> {
   680→    let mut joins = Vec::new();
   681→    let mut visited = HashSet::new();
   682→    let mut queue = VecDeque::new();
   683→    queue.push_back(start);
   684→    visited.insert(start);
   685→
   686→    let max_steps = 500;
   687→    let mut steps = 0;
   688→
   689→    while let Some(current) = queue.pop_front() {
   690→        steps += 1;
   691→        if steps > max_steps {
   692→            break;
   693→        }
   694→
   695→        if let Some(node) = scg.get_node(current) {
   696→            if let NodePayload::Control(c) = &node.payload {
   697→                if c.kind == ControlKind::Join {
   698→                    joins.push(current);
   699→                    continue; // Don't walk past Join
   700→                }
   701→            }
   702→        }
   703→
   704→        for edge in edge_idx.outgoing_cf(current) {
   705→            let target = edge.target;
   706→            if !visited.contains(&target) {
   707→                visited.insert(target);
   708→                queue.push_back(target);
   709→            }
   710→        }
   711→    }
   712→
   713→    joins
   714→}

   716→/// Find the `Join` node where a Branch's then and else arms converge.
   717→fn find_join_for_branch(
   718→    then_start: NodeId,
   719→    else_start: Option<NodeId>,
   720→    scg: &SCG,
   721→    edge_idx: &EdgeIndex,
   722→) -> Option<NodeId> {
   723→    let then_joins = find_reachable_joins(then_start, scg, edge_idx);
   724→
   725→    if let Some(else_start) = else_start {
   726→        let else_joins = find_reachable_joins(else_start, scg, edge_idx);
   727→        // Find the first Join reachable from both arms
   728→        for jid in &then_joins {
   729→            if else_joins.contains(jid) {
   730→                return Some(*jid);
   731→            }
   732→        }
   733→    }
   734→
   735→    // Fallback: first Join reachable from then_start
   736→    then_joins.into_iter().next()
   737→}

   739→/// Resolve a Branch node's then/else targets and Join convergence point.
   740→///
   741→/// Looks for labeled ControlFlow edges ("then", "else", "else_fallthrough")
   742→/// and falls back to positional ordering if labels are missing.
   743→fn resolve_branch(
   744→    branch_id: NodeId,
   745→    scg: &SCG,
   746→    edge_idx: &EdgeIndex,
   747→) -> (NodeId, Option<NodeId>, Option<NodeId>) {
   748→    let cf_edges = edge_idx.outgoing_cf(branch_id);
   749→
   750→    // Look for labeled edges
   751→    let then_target = cf_edges
   752→        .iter()
   753→        .find(|e| e.label.as_deref() == Some("then"))
   754→        .map(|e| e.target)
   755→        .or_else(|| cf_edges.first().map(|e| e.target));
   756→
   757→    let else_target = cf_edges
   758→        .iter()
   759→        .find(|e| {
   760→            e.label.as_deref() == Some("else") || e.label.as_deref() == Some("else_fallthrough")
   761→        })
   762→        .map(|e| e.target)
   763→        .or_else(|| {
   764→            // If there are exactly 2 CF edges and one is "then", the other is "else"
   765→            if cf_edges.len() == 2 {
   766→                let then = then_target?;
   767→                cf_edges.iter().find(|e| e.target != then).map(|e| e.target)
   768→            } else {
   769→                None
   770→            }
   771→        });
   772→
   773→    let then_tgt = then_target.unwrap_or(branch_id);
   774→    let join = find_join_for_branch(then_tgt, else_target, scg, edge_idx);
   775→
   776→    (then_tgt, else_target, join)
   777→}

   779→/// Resolve a LoopHeader node's body and exit targets.
   780→///
   781→/// Classifies outgoing ControlFlow edges: edges targeting a `LoopExit`
   782→/// node are the exit; all other edges are the loop body.
   783→fn resolve_loop(header_id: NodeId, scg: &SCG, edge_idx: &EdgeIndex) -> (NodeId, Option<NodeId>) {
   784→    let cf_edges = edge_idx.outgoing_cf(header_id);
   785→
   786→    let mut body_target = None;
   787→    let mut exit_target = None;
   788→
   789→    for edge in &cf_edges {
   790→        if let Some(target_node) = scg.get_node(edge.target) {
   791→            if let NodePayload::Control(c) = &target_node.payload {
   792→                if c.kind == ControlKind::LoopExit {
   793→                    exit_target = Some(edge.target);
   794→                    continue;
   795→                }
   796→            }
   797→        }
   798→        if body_target.is_none() {
   799→            body_target = Some(edge.target);
   800→        }
   801→    }
   802→
   803→    // Fallbacks
   804→    if body_target.is_none() {
   805→        body_target = cf_edges.first().map(|e| e.target);
   806→    }
   807→    if exit_target.is_none() && cf_edges.len() > 1 {
   808→        exit_target = cf_edges.get(1).map(|e| e.target);
   809→    }
   810→
   811→    (body_target.unwrap_or(header_id), exit_target)
   812→}

   814→// ── Match/switch case-value extraction ──────────────────────────────────
   815→
   816→/// Extract the integer case value for a match/switch arm from the branch
   817→/// condition and surrounding SCG context.
   818→///
   819→/// A match arm `42 => body` produces a branch whose condition is typically
   820→/// the result of an equality comparison `disc == 42`. This function traces
   821→/// back through the DataFlow edges to find the constant operand, which is
   822→/// the case value from the AST's MatchArm pattern.
   823→///
   824→/// Extraction strategy (in priority order):
   825→///
   826→/// 1. If `cond` is already an `ScgExpr::Int(n)`, return `n`.
   827→/// 2. Trace the first DataFlow edge of the Branch back to its source node.
   828→///    If the source is a Computation node with an equality operation
   829→///    (`eq` / `==`), inspect its second DataFlow input — the RHS of the
   830→///    comparison is the case value. If that RHS source node is a
   831→///    Computation whose `operation` string parses as an integer, use it.
   832→/// 3. Try to parse an integer from the control node's label string.
   833→///    Recognised formats: `"match disc == 42"`, `"case 2: 42"`.
   834→/// 4. Fall back to `arm_index` — each arm in a match expression receives
   835→///    a distinct fallback value so that unknown case values don't collide.
   836→fn extract_case_value(
   837→    branch_id: NodeId,
   838→    cond: &ScgExpr,
   839→    ctrl_label: Option<&str>,
   840→    scg: &SCG,
   841→    edge_idx: &EdgeIndex,
   842→    arm_index: usize,
   843→) -> i64 {
   844→    // Strategy 1: direct integer condition.
   845→    if let ScgExpr::Int(n) = cond {
   846→        return *n;
   847→    }
   848→
   849→    // Strategy 2: trace back through the equality comparison node.
   850→    let df_inputs = edge_idx.incoming_df(branch_id);
   851→    if let Some(df_edge) = df_inputs.first() {
   852→        let cond_source = df_edge.source;
   853→        if let Some(source_node) = scg.get_node(cond_source) {
   854→            if let NodePayload::Computation(comp) = &source_node.payload {
   855→                let is_eq = comp.operation == "eq" || comp.operation == "==";
   856→                if is_eq {
   857→                    // The RHS of the equality is the case value.
   858→                    let rhs_inputs = edge_idx.incoming_df(cond_source);
   859→                    if rhs_inputs.len() >= 2 {
   860→                        let rhs_source = rhs_inputs[1].source;
   861→                        if let Some(rhs_node) = scg.get_node(rhs_source) {
   862→                            // The RHS node might be a Computation whose
   863→                            // operation string is a literal integer.
   864→                            if let NodePayload::Computation(rhs_comp) = &rhs_node.payload {
   865→                                if let Ok(val) = rhs_comp.operation.parse::<i64>() {
   866→                                    return val;
   867→                                }
   868→                            }
   869→                        }
   870→                    }
   871→                }
   872→            }
   873→        }
   874→    }
   875→
   876→    // Strategy 3: parse from the control node label.
   877→    if let Some(label) = ctrl_label {
   878→        // Format: "match <disc> == <value>"
   879→        if let Some(idx) = label.find("==") {
   880→            let after_eq = label[idx + 2..].trim();
   881→            // Take the first token (stop at whitespace / punctuation)
   882→            let token = after_eq
   883→                .split(|c: char| c.is_whitespace() || c == ')')
   884→                .next()
   885→                .unwrap_or(after_eq);
   886→            if let Ok(val) = token.parse::<i64>() {
   887→                return val;
   888→            }
   889→        }
   890→        // Format: "case <idx>: <value>"
   891→        if label.starts_with("case ") {
   892→            let parts: Vec<&str> = label.splitn(2, ':').collect();
   893→            if parts.len() == 2 {
   894→                let value_str = parts[1].trim();
   895→                let token = value_str
   896→                    .split(|c: char| c.is_whitespace() || c == ')')
   897→                    .next()
   898→                    .unwrap_or(value_str);
   899→                if let Ok(val) = token.parse::<i64>() {
   900→                    return val;
   901→                }
   902→            }
   903→        }
   904→    }
   905→
   906→    // Strategy 4: fallback to arm_index so each arm gets a distinct value.
   907→    arm_index as i64
   908→}

   910→// ── Control flow walk ──────────────────────────────────────────────────
   911→
   912→/// Walk control flow starting from `start`, producing `ScgStatement`s.
   913→///
   914→/// Stops when reaching a node in `stop_at` (does NOT consume that node;
   915→/// the caller is responsible for handling it). Adds processed nodes to
   916→/// `consumed`.
   917→///
   918→/// # Control Flow Reconstruction
   919→///
   920→/// - **Branch → If**: Follows "then"/"else" labeled CF edges, walks each
   921→///   arm until reaching a Join, produces `ControlNode::If`.
   922→/// - **LoopHeader → Loop**: Follows body/exit CF edges, walks the body
   923→///   until back-edge or LoopExit, produces `ControlNode::Loop`.
   924→/// - **Jump("break")**: Produces `ControlNode::Break`.
   925→/// - **Jump("continue")**: Produces `ControlNode::Continue`.
   926→/// - **FunctionReturn**: Produces `ScgStatement::Return`.
   927→fn walk_control_flow(
   928→    start: NodeId,
   929→    scg: &SCG,
   930→    edge_idx: &EdgeIndex,
   931→    consumed: &mut HashSet<NodeId>,
   932→    stop_at: &HashSet<NodeId>,
   933→) -> Vec<ScgStatement> {
   934→    let mut stmts = Vec::new();
   935→    let mut current = Some(start);
   936→
   937→    while let Some(node_id) = current {
   938→        // Stop if we've reached a merge point
   939→        if stop_at.contains(&node_id) {
   940→            break;
   941→        }
   942→        // Skip already-consumed nodes
   943→        if consumed.contains(&node_id) {
   944→            break;
   945→        }
   946→        consumed.insert(node_id);
   947→
   948→        let node_data = match scg.get_node(node_id) {
   949→            Some(n) => n,
   950→            None => break,
   951→        };
   952→
   953→        match &node_data.payload {
   954→            // ── Control nodes ──────────────────────────────────────
   955→            NodePayload::Control(ctrl) => match ctrl.kind {
   956→                ControlKind::Branch => {
   957→                    let (then_tgt, else_tgt, join_node) = resolve_branch(node_id, scg, edge_idx);
   958→                    let cond = resolve_branch_cond(node_id, edge_idx, scg);
   959→
   960→                    // Check if this is a match/switch branch (label starts
   961→                    // with "match") vs a simple if/else. For match branches,
   962→                    // we look for multiple Branch→Join diamonds that share
   963→                    // the same Join and collapse them into a Switch node.
   964→                    let is_match = ctrl
   965→                        .label
   966→                        .as_ref()
   967→                        .map(|l| l.starts_with("match"))
   968→                        .unwrap_or(false);
   969→
   970→                    if is_match {
   971→                        // For match/switch, collect all arms that lead to
   972→                        // the same Join node. Each arm is a then/else pair
   973→                        // where the then branch is the matched case.
   974→                        // We walk the then arm to find the case value and body.
   975→                        let mut arms = Vec::new();
   976→                        let mut default_body = Vec::new();
   977→
   978→                        // Build stop-at for both arms (includes Join)
   979→                        let mut arm_stop = stop_at.clone();
   980→                        if let Some(join) = join_node {
   981→                            arm_stop.insert(join);
   982→                        }
   983→
   984→                        // Generate a simple switch from the then arm
   985→                        // with a discriminant expression.
   986→                        let then_body_stmts =
   987→                            walk_control_flow(then_tgt, scg, edge_idx, consumed, &arm_stop);
   988→
   989→                        // Extract the case value from the AST's MatchArm pattern.
   990→                        // The branch condition for a match arm is typically
   991→                        // `disc == value`. We trace back through the DataFlow
   992→                        // edges to find the constant being compared against.
   993→                        let case_value = extract_case_value(
   994→                            node_id,
   995→                            &cond,
   996→                            ctrl.label.as_deref(),
   997→                            scg,
   998→                            edge_idx,
   999→                            arms.len(),
  1000→                        );
  1001→                        arms.push(SwitchArm {
  1002→                            value: case_value,
  1003→                            body: then_body_stmts,
  1004→                        });
  1005→
  1006→                        if let Some(tgt) = else_tgt {
  1007→                            let else_stmts =
  1008→                                walk_control_flow(tgt, scg, edge_idx, consumed, &arm_stop);
  1009→                            default_body = else_stmts;
  1010→                        }
  1011→
  1012→                        // Use the first operand of the condition as discriminant
  1013→                        let disc = if let ScgExpr::Var(_) = &cond {
  1014→                            cond.clone()
  1015→                        } else {
  1016→                            ScgExpr::Var("disc".to_string())
  1017→                        };
  1018→
  1019→                        stmts.push(ScgStatement::Control(ControlNode::Switch {
  1020→                            discriminant: disc,
  1021→                            arms,
  1022→                            default_body,
  1023→                        }));
  1024→                    } else {
  1025→                        // Standard if/else
  1026→                        let mut arm_stop = stop_at.clone();
  1027→                        if let Some(join) = join_node {
  1028→                            arm_stop.insert(join);
  1029→                        }
  1030→
  1031→                        let then_body =
  1032→                            walk_control_flow(then_tgt, scg, edge_idx, consumed, &arm_stop);
  1033→
  1034→                        let else_body = else_tgt
  1035→                            .map(|tgt| walk_control_flow(tgt, scg, edge_idx, consumed, &arm_stop));
  1036→
  1037→                        stmts.push(ScgStatement::Control(ControlNode::If {
  1038→                            cond,
  1039→                            then_body,
  1040→                            else_body,
  1041→                        }));
  1042→                    }
  1043→
  1044→                    // Continue from the Join
  1045→                    if let Some(join) = join_node {
  1046→                        consumed.insert(join);
  1047→                        current = edge_idx.outgoing_cf(join).first().map(|e| e.target);
  1048→                    } else {
  1049→                        current = None;
  1050→                    }
  1051→                    continue;
  1052→                }
  1053→
  1054→                ControlKind::LoopHeader => {
  1055→                    let (body_tgt, exit_tgt) = resolve_loop(node_id, scg, edge_idx);
  1056→
  1057→                    // Stop the body walk at back-edges (LoopHeader) and LoopExit
  1058→                    let mut loop_stop = stop_at.clone();
  1059→                    loop_stop.insert(node_id); // back-edge target
  1060→                    if let Some(exit) = exit_tgt {
  1061→                        loop_stop.insert(exit);
  1062→                    }
  1063→
  1064→                    let body = walk_control_flow(body_tgt, scg, edge_idx, consumed, &loop_stop);
  1065→
  1066→                    stmts.push(ScgStatement::Control(ControlNode::Loop { body }));
  1067→
  1068→                    // Continue from the LoopExit
  1069→                    if let Some(exit) = exit_tgt {
  1070→                        consumed.insert(exit);
  1071→                        current = edge_idx.outgoing_cf(exit).first().map(|e| e.target);
  1072→                    } else {
  1073→                        current = None;
  1074→                    }
  1075→                    continue;
  1076→                }
  1077→
  1078→                ControlKind::Jump => match ctrl.label.as_deref() {
  1079→                    Some("break") => {
  1080→                        stmts.push(ScgStatement::Control(ControlNode::Break));
  1081→                        current = None;
  1082→                        continue;
  1083→                    }
  1084→                    Some("continue") => {
  1085→                        stmts.push(ScgStatement::Control(ControlNode::Continue));
  1086→                        current = None;
  1087→                        continue;
  1088→                    }
  1089→                    _ => {
  1090→                        // Unconditional jump — follow the CF edge
  1091→                        let target = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1092→                        if let Some(tgt) = target {
  1093→                            if !consumed.contains(&tgt) && !stop_at.contains(&tgt) {
  1094→                                current = Some(tgt);
  1095→                                continue;
  1096→                            }
  1097→                        }
  1098→                        current = None;
  1099→                        continue;
  1100→                    }
  1101→                },
  1102→
  1103→                ControlKind::FunctionReturn => {
  1104→                    stmts.push(ScgStatement::Return(vec![]));
  1105→                    current = None;
  1106→                    continue;
  1107→                }
  1108→
  1109→                ControlKind::Join | ControlKind::LoopExit => {
  1110→                    // Structural nodes handled by Branch/LoopHeader.
  1111→                    // Pass through to the next node.
  1112→                    current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1113→                    continue;
  1114→                }
  1115→
  1116→                ControlKind::FunctionEntry => {
  1117→                    // Shouldn't appear inside a function body; pass through
  1118→                    current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1119→                    continue;
  1120→                }
  1121→
  1122→                ControlKind::Switch | ControlKind::SwitchCase => {
  1123→                    // Switch/switch-case nodes are handled like Branch
  1124→                    current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1125→                    continue;
  1126→                }
  1127→
  1128→                ControlKind::ClosureEntry | ControlKind::ClosureReturn => {
  1129→                    // Closure entry/return handled like function entry/return
  1130→                    current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1131→                    continue;
  1132→                }
  1133→
  1134→                ControlKind::FuturePoll
  1135→                | ControlKind::WakerRegistration
  1136→                | ControlKind::StateTransition => {
  1137→                    // Async state machine nodes: pass through
  1138→                    current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1139→                    continue;
  1140→                }
  1141→            },
  1142→
  1143→            // ── Non-control nodes: convert to statements ───────────
  1144→            _ => {
  1145→                if let Some(stmt) = convert_node_to_statement(node_id, node_data, edge_idx, scg) {
  1146→                    stmts.push(stmt);
  1147→                }
  1148→
  1149→                // Continue to the next node via ControlFlow
  1150→                current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1151→            }
  1152→        }
  1153→    }
  1154→
  1155→    stmts
  1156→}

  1158→// ── Node-to-statement conversion ───────────────────────────────────────
  1159→
  1160→/// Convert a non-control SCG node into an `ScgStatement`.
  1161→///
  1162→/// Handles all node types except `Control` (which is handled by
  1163→/// `walk_control_flow`) and `Phantom` (which is skipped).
  1164→fn convert_node_to_statement(
  1165→    node_id: NodeId,
  1166→    node_data: &NodeData,
  1167→    edge_idx: &EdgeIndex,
  1168→    scg: &SCG,
  1169→) -> Option<ScgStatement> {
  1170→    match &node_data.payload {
  1171→        NodePayload::Allocation(alloc) => {
  1172→            let ty = alloc
  1173→                .type_name
  1174→                .as_deref()
  1175→                .and_then(parse_scg_type)
  1176→                .unwrap_or(ScgType::U8);
  1177→            Some(ScgStatement::Allocation(AllocationNode::Stack {
  1178→                name: node_var(node_id, "alloc"),
  1179→                size: alloc.size as u32,
  1180→                ty,
  1181→            }))
  1182→        }
  1183→
  1184→        NodePayload::Access(access) => match access.mode {
  1185→            AccessMode::Read => Some(ScgStatement::Access(AccessNode::Load {
  1186→                dst: node_var(node_id, "val"),
  1187→                ptr: resolve_df_input(node_id, 0, edge_idx, scg),
  1188→                offset: access.offset.map(|o| ScgExpr::Int(o as i64)),
  1189→            })),
  1190→            AccessMode::Write | AccessMode::ReadWrite => {
  1191→                Some(ScgStatement::Access(AccessNode::Store {
  1192→                    ptr: resolve_df_input(node_id, 0, edge_idx, scg),
  1193→                    offset: access.offset.map(|o| ScgExpr::Int(o as i64)),
  1194→                    value: resolve_df_input(node_id, 1, edge_idx, scg),
  1195→                }))
  1196→            }
  1197→        },
  1198→
  1199→        NodePayload::Computation(comp) => {
  1200→            let op = parse_binop(&comp.operation).unwrap_or(IrBinOpKind::Add);
  1201→            Some(ScgStatement::Computation(ComputationNode {
  1202→                dst: node_var(node_id, "comp"),
  1203→                op,
  1204→                lhs: resolve_df_input(node_id, 0, edge_idx, scg),
  1205→                rhs: resolve_df_input(node_id, 1, edge_idx, scg),
  1206→                tail_call: false,
  1207→            }))
  1208→        }
  1209→
  1210→        NodePayload::Cast(cast) => {
  1211→            let to_ty = parse_scg_type(&cast.to_type).unwrap_or(ScgType::Ptr);
  1212→            let from_ty = parse_scg_type(&cast.from_type).unwrap_or(ScgType::Ptr);
  1213→            Some(ScgStatement::Cast(CastNode {
  1214→                dst: node_var(node_id, "cast"),
  1215→                src: resolve_df_input(node_id, 0, edge_idx, scg),
  1216→                kind: if cast.is_lossless {
  1217→                    CodegenCastKind::ZExt
  1218→                } else {
  1219→                    CodegenCastKind::BitCast
  1220→                },
  1221→                from_ty,
  1222→                to_ty,
  1223→            }))
  1224→        }
  1225→
  1226→        NodePayload::Deallocation(_dealloc) => {
  1227→            // Lower deallocation as a proper runtime call rather than
  1228→            // a semantic no-op (`*ptr = 0`).  This ensures the memory
  1229→            // is actually freed at runtime and is more correct than
  1230→            // simply zeroing the pointer.
  1231→            Some(ScgStatement::Call(CallNode {
  1232→                dst: None,
  1233→                func: "__vuma_dealloc".to_string(),
  1234→                args: vec![resolve_df_input(node_id, 0, edge_idx, scg)],
  1235→            }))
  1236→        }
  1237→
  1238→        NodePayload::Effect(eff) => Some(ScgStatement::Call(CallNode {
  1239→            dst: Some(node_var(node_id, "eff")),
  1240→            func: eff.effect_kind.clone(),
  1241→            args: vec![],
  1242→        })),
  1243→
  1244→        NodePayload::Phantom(_) => None,
  1245→
  1246→        NodePayload::Control(_) => {
  1247→            // Control nodes are handled by walk_control_flow
  1248→            None
  1249→        }
  1250→
  1251→        NodePayload::VTable(_) | NodePayload::ClosureEnv(_) => {
  1252→            // VTable and ClosureEnv are structural nodes; no IR statement
  1253→            None
  1254→        }
  1255→    }
  1256→}
  1257→
  1258→// ── Function parameter extraction ──────────────────────────────────────
  1259→
  1260→/// Extract function parameters from DataFlow edges leaving the
  1261→/// FunctionEntry node.
  1262→///
  1263→/// Parameter types are inferred from the target node's payload, which
  1264→/// has been refined by BD inference (via `refine_scg_types_with_bd`).
  1265→/// If no type info is available, defaults to `ScgType::I64`.
  1266→fn extract_function_params(entry_id: NodeId, scg: &SCG, edge_idx: &EdgeIndex) -> Vec<ScgParam> {
  1267→    let df_edges = edge_idx.outgoing_df(entry_id);
  1268→    let mut params = Vec::new();
  1269→
  1270→    for (i, edge) in df_edges.iter().enumerate() {
  1271→        let (name, ty) = if let Some(target_node) = scg.get_node(edge.target) {
  1272→            match &target_node.payload {
  1273→                NodePayload::Allocation(alloc) => {
  1274→                    let name = alloc
  1275→                        .type_name
  1276→                        .clone()
  1277→                        .unwrap_or_else(|| format!("param_{}", i));
  1278→                    let ty = alloc
  1279→                        .type_name
  1280→                        .as_deref()
  1281→                        .and_then(parse_scg_type)
  1282→                        .unwrap_or(ScgType::I64);
  1283→                    (name, ty)
  1284→                }
  1285→                NodePayload::Computation(comp) => {
  1286→                    let name = comp
  1287→                        .result_type
  1288→                        .clone()
  1289→                        .unwrap_or_else(|| format!("param_{}", i));
  1290→                    let ty = comp
  1291→                        .result_type
  1292→                        .as_deref()
  1293→                        .and_then(parse_scg_type)
  1294→                        .unwrap_or(ScgType::I64);
  1295→                    (name, ty)
  1296→                }
  1297→                NodePayload::Cast(cast) => {
  1298→                    let name = format!("param_{}", i);
  1299→                    let ty = parse_scg_type(&cast.to_type).unwrap_or(ScgType::I64);
  1300→                    (name, ty)
  1301→                }
  1302→                _ => (format!("param_{}", i), ScgType::I64),
  1303→            }
  1304→        } else {
  1305→            (format!("param_{}", i), ScgType::I64)
  1306→        };
  1307→
  1308→        params.push(ScgParam { name, ty });
  1309→    }
  1310→
  1311→    params
  1312→}
  1313→
  1314→// ── Type parsing helper ────────────────────────────────────────────────
  1315→
  1316→/// Parse a type string into a `ScgType`.
  1317→fn parse_scg_type(type_str: &str) -> Option<ScgType> {
  1318→    match type_str {
  1319→        "i8" | "I8" => Some(ScgType::I8),
  1320→        "i16" | "I16" => Some(ScgType::I16),
  1321→        "i32" | "I32" => Some(ScgType::I32),
  1322→        "i64" | "I64" => Some(ScgType::I64),
  1323→        "u8" | "U8" => Some(ScgType::U8),
  1324→        "u16" | "U16" => Some(ScgType::U16),
  1325→        "u32" | "U32" => Some(ScgType::U32),
  1326→        "u64" | "U64" => Some(ScgType::U64),
  1327→        "ptr" | "*void" | "*u8" | "*i8" => Some(ScgType::Ptr),
  1328→        "void" => Some(ScgType::Void),
  1329→        _ => None,
  1330→    }
  1331→}
  1332→
  1333→// ── BD type refinement ─────────────────────────────────────────────────
  1334→
  1335→/// Map a BD `RepD` to the codegen's `ScgType`.
  1336→///
  1337→/// Uses the RepD's size and kind to pick the most specific `ScgType`:
  1338→/// - Pointer RepDs → `ScgType::Ptr`
  1339→/// - Byte RepDs → integer types by size (u8, u16, u32, u64)
  1340→/// - Struct/Array/Enum/Union → `ScgType::Ptr` (passed by reference)
  1341→/// - Generic → `ScgType::I64` (fallback)
  1342→fn repd_to_scg_type(repd: &RepD) -> ScgType {
  1343→    match repd {
  1344→        RepD::Ptr(_) | RepD::Func(_) => ScgType::Ptr,
  1345→        RepD::Byte(byte_rep) => match byte_rep.size {
  1346→            1 => ScgType::U8,
  1347→            2 => ScgType::U16,
  1348→            4 => ScgType::U32,
  1349→            _ => ScgType::U64,
  1350→        },
  1351→        RepD::Struct(_) | RepD::Array(_) | RepD::Enum(_) | RepD::Union(_) => ScgType::Ptr,
  1352→        RepD::Generic { .. } => ScgType::I64,
  1353→    }
  1354→}
  1355→
  1356→/// Convert a `ScgType` to its canonical string name for storing in SCG
  1357→/// node payloads (e.g., `AllocationNode.type_name`, `CastNode.from_type`).
  1358→fn scg_type_to_name(ty: &ScgType) -> &'static str {
  1359→    match ty {
  1360→        ScgType::I8 => "i8",
  1361→        ScgType::I16 => "i16",
  1362→        ScgType::I32 => "i32",
  1363→        ScgType::I64 => "i64",
  1364→        ScgType::U8 => "u8",
  1365→        ScgType::U16 => "u16",
  1366→        ScgType::U32 => "u32",
  1367→        ScgType::U64 => "u64",
  1368→        ScgType::Ptr => "ptr",
  1369→        ScgType::Void => "void",
  1370→    }
  1371→}
  1372→
  1373→/// Refine SCG node type metadata using BD inference results.
  1374→///
  1375→/// After BD inference, each node's `RepD` describes the actual memory
  1376→/// representation.  This function maps those RepDs back to `ScgType`s
  1377→/// and stores the result in the SCG node payloads so that downstream
  1378→/// bridge code (`convert_node_to_statement`, `extract_function_params`)
  1379→/// can pick up the refined types instead of using defaults.
  1380→///
  1381→/// # What is refined
  1382→///
  1383→/// - **Allocation nodes**: `type_name` is set if it was previously `None`.
  1384→/// - **Cast nodes**: `from_type` / `to_type` are updated if they couldn't
  1385→///   previously be parsed by `parse_scg_type`.
  1386→/// - **Computation nodes**: `result_type` is set if it was previously `None`.
  1387→fn refine_scg_types_with_bd(scg: &mut SCG, bd_results: &[(NodeId, BD)]) {
  1388→    let bd_map: HashMap<NodeId, &BD> = bd_results.iter().map(|(id, bd)| (*id, bd)).collect();
  1389→
  1390→    let node_ids: Vec<_> = scg.node_ids().collect();
  1391→    for node_id in node_ids {
  1392→        let Some(bd) = bd_map.get(&node_id) else {
  1393→            continue;
  1394→        };
  1395→        let inferred_type = repd_to_scg_type(&bd.repd);
  1396→        let type_name = scg_type_to_name(&inferred_type);
  1397→
  1398→        if let Some(node) = scg.get_node_mut(node_id) {
  1399→            match &mut node.payload {
  1400→                NodePayload::Allocation(alloc)
  1401→                    // Update type_name if it was previously unset.
  1402→                    if alloc.type_name.is_none() => {
  1403→                        alloc.type_name = Some(type_name.to_string());
  1404→                    }
  1405→                NodePayload::Cast(cast) => {
  1406→                    // Update from_type / to_type if they couldn't previously
  1407→                    // be parsed by `parse_scg_type` (i.e., they were opaque
  1408→                    // type names from the AST that don't map directly).
  1409→                    if parse_scg_type(&cast.from_type).is_none() {
  1410→                        cast.from_type = type_name.to_string();
  1411→                    }
  1412→                    if parse_scg_type(&cast.to_type).is_none() {
  1413→                        cast.to_type = type_name.to_string();
  1414→                    }
  1415→                }
  1416→                NodePayload::Computation(comp)
  1417→                    if comp.result_type.is_none() => {
  1418→                        comp.result_type = Some(type_name.to_string());
  1419→                    }
  1420→                _ => {}
  1421→            }
  1422→        }
  1423→    }
  1424→}
  1425→
  1426→// ── Entry-point detection ──────────────────────────────────────────────
  1427→
  1428→/// Find entry-point nodes (no incoming ControlFlow edges) for a function
  1429→/// that lacks an explicit FunctionEntry node.
  1430→fn find_entry_points(scg: &SCG, edge_idx: &EdgeIndex) -> Vec<NodeId> {
  1431→    let mut entry_points = Vec::new();
  1432→
  1433→    for node_id in scg.node_ids() {
  1434→        let has_incoming_cf = edge_idx
  1435→            .incoming
  1436→            .get(&node_id)
  1437→            .map(|edges| edges.iter().any(|e| e.kind == EdgeKind::ControlFlow))
  1438→            .unwrap_or(false);
  1439→
  1440→        if !has_incoming_cf {
  1441→            if let Some(node_data) = scg.get_node(node_id) {
  1442→                // Skip Phantom nodes
  1443→                if matches!(node_data.node_type, NodeType::Phantom) {
  1444→                    continue;
  1445→                }
  1446→                entry_points.push(node_id);
  1447→            }
  1448→        }
  1449→    }
  1450→
  1451→    // If no entry points found, use the first node
  1452→    if entry_points.is_empty() {
  1453→        if let Some(first_id) = scg.node_ids().next() {
  1454→            entry_points.push(first_id);
  1455→        }
  1456→    }
  1457→
  1458→    entry_points
  1459→}

  1461→// ── Main bridge function ───────────────────────────────────────────────
  1462→
  1463→/// Convert a `vuma_scg::SCG` into the codegen's stub `Scg` type.
  1464→///
  1465→/// This function reconstructs real control flow (if/else, loops, function
  1466→/// boundaries, break/continue) from the SCG's graph structure, instead of
  1467→/// just flattening everything into a single linear "main" function.
  1468→///
  1469→/// # Algorithm
  1470→///
  1471→/// 1. **Phase 1: Function boundary detection** — Group nodes by
  1472→///    FunctionEntry→FunctionReturn regions.
  1473→/// 2. **Phase 2: Control flow reconstruction** — Within each function,
  1474→///    detect Branch+Join diamonds (if/else) and LoopHeader+LoopExit
  1475→///    patterns (loops).
  1476→/// 3. **Phase 3: Statement generation** — Convert non-control nodes into
  1477→///    ScgStatements with DataFlow-based variable naming.
  1478→fn bridge_scg_to_codegen(scg: &SCG) -> Scg {
  1479→    let edge_idx = EdgeIndex::build(scg);
  1480→    let mut consumed: HashSet<NodeId> = HashSet::new();
  1481→    let mut scg_nodes: Vec<ScgNode> = Vec::new();
  1482→
  1483→    // ── Phase 1: Function boundary detection ─────────────────────
  1484→    let function_entries: Vec<(NodeId, String)> = scg
  1485→        .nodes()
  1486→        .filter_map(|n| {
  1487→            if let NodePayload::Control(c) = &n.payload {
  1488→                if c.kind == ControlKind::FunctionEntry {
  1489→                    let name = c.label.clone().unwrap_or_else(|| "unknown".to_string());
  1490→                    return Some((n.id, name));
  1491→                }
  1492→            }
  1493→            None
  1494→        })
  1495→        .collect();
  1496→
  1497→    if !function_entries.is_empty() {
  1498→        // Process each function defined by a FunctionEntry node
  1499→        for (entry_id, func_name) in &function_entries {
  1500→            consumed.insert(*entry_id);
  1501→
  1502→            let return_node = find_function_return(*entry_id, scg, &edge_idx);
  1503→            let params = extract_function_params(*entry_id, scg, &edge_idx);
  1504→
  1505→            let mut body = if let Some(first_cf) = edge_idx.outgoing_cf(*entry_id).first() {
  1506→                let mut stop_at = HashSet::new();
  1507→                if let Some(ret) = return_node {
  1508→                    stop_at.insert(ret);
  1509→                }
  1510→                walk_control_flow(first_cf.target, scg, &edge_idx, &mut consumed, &stop_at)
  1511→            } else {
  1512→                vec![]
  1513→            };
  1514→
  1515→            // Add return statement if the function has a FunctionReturn
  1516→            if let Some(ret) = return_node {
  1517→                consumed.insert(ret);
  1518→            }
  1519→            if !body.iter().any(|s| matches!(s, ScgStatement::Return(_))) {
  1520→                body.push(ScgStatement::Return(vec![]));
  1521→            }
  1522→
  1523→            scg_nodes.push(ScgNode::Function(ScgFunction {
  1524→                name: func_name.clone(),
  1525→                params,
  1526→                results: vec![],
  1527→                body,
  1528→            }));
  1529→        }
  1530→    } else {
  1531→        // No FunctionEntry nodes — find entry points and walk control flow
  1532→        let entry_points = find_entry_points(scg, &edge_idx);
  1533→
  1534→        let mut body = Vec::new();
  1535→        for start in &entry_points {
  1536→            let stop_at = HashSet::new();
  1537→            let mut partial = walk_control_flow(*start, scg, &edge_idx, &mut consumed, &stop_at);
  1538→            body.append(&mut partial);
  1539→        }
  1540→
  1541→        // Process any remaining unconsumed nodes (connected only via DataFlow)
  1542→        let remaining: Vec<NodeId> = scg.node_ids().filter(|id| !consumed.contains(id)).collect();
  1543→        for nid in &remaining {
  1544→            if consumed.contains(nid) {
  1545→                continue;
  1546→            }
  1547→            consumed.insert(*nid);
  1548→            if let Some(node_data) = scg.get_node(*nid) {
  1549→                if let Some(stmt) = convert_node_to_statement(*nid, node_data, &edge_idx, scg) {
  1550→                    body.push(stmt);
  1551→                }
  1552→            }
  1553→        }
  1554→
  1555→        if !body.iter().any(|s| matches!(s, ScgStatement::Return(_))) {
  1556→            body.push(ScgStatement::Return(vec![]));
  1557→        }
  1558→
  1559→        scg_nodes.push(ScgNode::Function(ScgFunction {
  1560→            name: "main".to_string(),
  1561→            params: vec![],
  1562→            results: vec![],
  1563→            body,
  1564→        }));
  1565→    }
  1566→
  1567→    // Handle remaining nodes not consumed by any function (multi-function case)
  1568→    let remaining: Vec<NodeId> = scg.node_ids().filter(|id| !consumed.contains(id)).collect();
  1569→    if !remaining.is_empty() {
  1570→        let mut stmts = Vec::new();
  1571→        for nid in &remaining {
  1572→            if consumed.contains(nid) {
  1573→                continue;
  1574→            }
  1575→            consumed.insert(*nid);
  1576→            if let Some(node_data) = scg.get_node(*nid) {
  1577→                if let Some(stmt) = convert_node_to_statement(*nid, node_data, &edge_idx, scg) {
  1578→                    stmts.push(stmt);
  1579→                }
  1580→            }
  1581→        }
  1582→        if !stmts.is_empty() {
  1583→            if !stmts.iter().any(|s| matches!(s, ScgStatement::Return(_))) {
  1584→                stmts.push(ScgStatement::Return(vec![]));
  1585→            }
  1586→            scg_nodes.push(ScgNode::Function(ScgFunction {
  1587→                name: "__remaining".to_string(),
  1588→                params: vec![],
  1589→                results: vec![],
  1590→                body: stmts,
  1591→            }));
  1592→        }
  1593→    }
  1594→
  1595→    // Ensure at least one function exists
  1596→    if scg_nodes.is_empty() {
  1597→        scg_nodes.push(ScgNode::Function(ScgFunction {
  1598→            name: "main".to_string(),
  1599→            params: vec![],
  1600→            results: vec![],
  1601→            body: vec![ScgStatement::Return(vec![])],
  1602→        }));
  1603→    }
  1604→
  1605→    Scg { nodes: scg_nodes }
  1606→}

  1608→/// Try to parse an operation string into a BinOpKind.
  1609→fn parse_binop(op: &str) -> Option<IrBinOpKind> {

  1612→        "sub" | "-" => Some(IrBinOpKind::Sub),
  1613→        "mul" | "*" => Some(IrBinOpKind::Mul),
  1614→        "sdiv" | "/" => Some(IrBinOpKind::SDiv),
  1615→        "udiv" => Some(IrBinOpKind::UDiv),
  1616→        "srem" | "%" => Some(IrBinOpKind::SRem),
  1617→        "urem" => Some(IrBinOpKind::URem),
  1618→        "and" | "&" => Some(IrBinOpKind::And),
  1619→        "or" | "|" => Some(IrBinOpKind::Or),
  1620→        "xor" | "^" => Some(IrBinOpKind::Xor),
  1621→        "shl" | "<<" => Some(IrBinOpKind::Shl),
  1622→        "shr.l" | ">>" => Some(IrBinOpKind::ShrL),
  1623→        "shr.a" => Some(IrBinOpKind::ShrA),
  1624→        "slt" | "<" => Some(IrBinOpKind::SLt),
  1625→        "sle" | "<=" => Some(IrBinOpKind::SLe),
  1626→        "sgt" | ">" => Some(IrBinOpKind::SGt),
  1627→        "sge" | ">=" => Some(IrBinOpKind::SGe),
  1628→        "ult" => Some(IrBinOpKind::ULt),
  1629→        "ule" => Some(IrBinOpKind::ULe),
  1630→        "ugt" => Some(IrBinOpKind::UGt),
  1631→        "uge" => Some(IrBinOpKind::UGe),
  1632→        "eq" | "==" => Some(IrBinOpKind::Eq),
  1633→        "ne" | "!=" => Some(IrBinOpKind::Ne),

  1639→// Compile pipeline
  1640→// ═══════════════════════════════════════════════════════════════════════════
  1641→
  1642→/// Compile VUMA source code with the given configuration.
  1643→///
  1644→/// This is the main entry point for the VUMA compilation pipeline.
  1645→/// It runs all stages in order, collecting errors and producing a
  1646→/// [`CompilationOutput`] on success.
  1647→///
  1648→/// # Pipeline Stages
  1649→///
  1650→/// 1. **Parse** — lex and parse source into an AST
  1651→/// 2. **AST → SCG** — convert the AST into a Semantic Computation Graph
  1652→/// 3. **SCG Validation** — verify the SCG is well-formed
  1653→/// 4. **BD Inference** — infer behavioral descriptions from the SCG
  1654→/// 5. **MSG Construction** — build the Memory State Graph from the SCG
  1655→/// 6. **IVE Verification** — verify the five core VUMA invariants
