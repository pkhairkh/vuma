  3406→    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError> {
  3407→        // ── PPC64LE Linux static executable ──
  3408→        //
  3409→        // Layout:
  3410→        //   _start:  BL main           ; call main (result in R3)
  3411→        //            LI R0, 1          ; sys_exit = 1
  3412→        //            SC                ; syscall: exit(R3)
  3413→        //   <functions...>
  3414→        //
  3415→        // The _start stub is 3 instructions = 12 bytes.
  3416→        // After that come all user functions.
