     1→//! # ARM64 Code Emission
     2→//!
     3→//! Lowers IR to ARM64 machine code and produces ELF binaries or raw binaries
     4→//! suitable for the Raspberry Pi 5 (Cortex-A76, ARMv8.2-A).
     5→//!
     6→//! ## Pipeline
     7→//!
     8→//! 1. **IR → ARM64 Instructions**: Each IR instruction is pattern-matched and
     9→//!    lowered to one or more [`Instruction`]
    10→//!    values, with virtual registers replaced by physical registers from the
    11→//!    register allocator.
    12→//! 2. **Emit function**: Each IR function is emitted as a sequence of 32-bit
    13→//!    ARM64 code words.
    14→//! 3. **Emit program**: All functions are collected, data sections are laid
    15→//!    out, and a minimal ELF64 binary is produced.
    16→//!
    17→//! ## Output Formats
    18→//!
    19→//! - **ELF**: Full ELF64 binary with program headers, section headers, symbol
    20→//!   table, and string table.  Suitable for Linux on Pi 5.
    21→//! - **Raw**: Flat binary image for bare-metal Pi 5 (loaded at 0x80000).
    22→//! - **Obj**: Relocatable ELF object file (`ET_REL`) for linking.
    23→//!
    24→//! ## ELF Layout (executable)
    25→//!
    26→//! ```text
    27→//! ┌─────────────────────┐
    28→//! │ ELF Header           │  64 bytes
    29→//! ├─────────────────────┤
    30→//! │ Program Headers      │  2 × 56 bytes (LOAD segments)
    31→//! ├─────────────────────┤
    32→//! │ .text                │  emitted code
    33→//! ├─────────────────────┤
    34→//! │ .rodata              │  read-only data
    35→//! ├─────────────────────┤
    36→//! │ .data                │  initialized data
    37→//! ├─────────────────────┤
    38→//! │ .symtab              │  symbol table entries
    39→//! ├─────────────────────┤
    40→//! │ .strtab              │  symbol string table
    41→//! ├─────────────────────┤
    42→//! │ .shstrtab            │  section header string table
    43→//! ├─────────────────────┤
    44→//! │ Section Headers      │  N × 64 bytes
    45→//! └─────────────────────┘
    46→//! .bss is virtual-only (memsz > filesz in the data LOAD segment).
    47→//! ```
    48→//!
    49→//! ## Relocation Support
    50→//!
    51→//! Inter-function calls (`BL`) are resolved in a fixup pass after all
    52→//! functions have been emitted.  The emitter records the word offset of each
    53→//! `BL` instruction and the target function name; once function addresses
    54→//! are known, the branch offsets are patched into the encoded instructions.

   311→/// Output format for the code emitter.

   314→    /// Full ELF64 executable (Linux) or object file.

   316→    /// Flat raw binary image (bare-metal).
   317→    Raw,
   318→    /// Relocatable ELF object file (`.o`).

   324→        match self {
   325→            OutputFormat::ELF => write!(f, "elf"),
   326→            OutputFormat::Raw => write!(f, "raw"),
   327→            OutputFormat::Obj => write!(f, "obj"),

   468→/// A record for a `BL` instruction that needs to be patched with the address
   469→/// of a named function.  After all functions are emitted, these records are
   470→/// resolved to compute the correct branch offset.
   471→#[derive(Debug, Clone)]
   472→struct CallRelocation {
   473→    /// Byte offset within the text section where the BL instruction lives.
   474→    text_byte_offset: u64,
   475→    /// Name of the target function.
   476→    target_func: String,
   477→}
   478→
   479→// ---------------------------------------------------------------------------
   480→// ELF RelaEntry
   481→// ---------------------------------------------------------------------------
   482→
   483→/// An ELF64 relocation entry with an explicit addend (SHT_RELA).
   484→///
   485→/// Each entry specifies where in the section a relocation must be applied,
   486→/// which symbol the relocation references, the type of relocation, and an
   487→/// addend value used in the relocation computation.
   488→#[derive(Debug, Clone)]
   489→pub struct RelaEntry {
   490→    /// Byte offset within the section where the relocation applies.
   491→    pub offset: u64,
   492→    /// Symbol index (upper 32 bits) and relocation type (lower 32 bits),
   493→    /// packed as: `(sym_idx << 32) | r_type`.
   494→    pub info: u64,
   495→    /// Addend value used in the relocation computation.
   496→    pub addend: i64,
   497→}
   498→
   499→impl RelaEntry {
   500→    /// Create a new relocation entry.
   501→    pub fn new(offset: u64, sym_idx: u32, r_type: u32, addend: i64) -> Self {
   502→        Self {
   503→            offset,
   504→            info: ((sym_idx as u64) << 32) | (r_type as u64),
   505→            addend,
   506→        }
   507→    }
   508→
   509→    /// Extract the symbol index from `info`.
   510→    pub fn sym_idx(&self) -> u32 {
   511→        (self.info >> 32) as u32
   512→    }
   513→
   514→    /// Extract the relocation type from `info`.
   515→    pub fn r_type(&self) -> u32 {
   516→        (self.info & 0xFFFFFFFF) as u32
   517→    }
   518→
   519→    /// Serialize the entry to 24 bytes (little-endian).
   520→    pub fn to_bytes(&self) -> [u8; 24] {
   521→        let mut buf = [0u8; 24];
   522→        buf[0..8].copy_from_slice(&self.offset.to_le_bytes());
   523→        buf[8..16].copy_from_slice(&self.info.to_le_bytes());
   524→        buf[16..24].copy_from_slice(&self.addend.to_le_bytes());
   525→        buf
   526→    }
   527→}

   533→/// The ARM64 code emitter.
   534→///
   535→/// Holds state for the current emission context: the register allocator,
   536→/// accumulated code, and fixup records for branch targets.
   537→pub struct Emitter {
   538→    /// Register allocator used during emission.
   539→    reg_alloc: RegAllocator,
   540→    /// Accumulated machine code for the current function (32-bit words).
   541→    code: Vec<u32>,
   542→    /// Fixup records for intra-function branches: (word index, target label name).
   543→    fixups: Vec<(usize, String)>,
   544→    /// Map from label name to code offset (in words) within the current function.
   545→    label_offsets: HashMap<String, usize>,
   546→    /// Inter-function call relocations for the current function.
   547→    call_relocs: Vec<CallRelocation>,
   548→    /// Relocation entries using the `RelocationEntry` infrastructure.
   549→    ///
   550→    /// Each entry records a byte offset within the function's encoded output
   551→    /// where a symbolic reference must be resolved, the target symbol name,
   552→    /// and the ISA-specific relocation type (e.g., `"R_AARCH64_CALL26"`).
   553→    relocations: Vec<RelocationEntry>,
   554→    /// Name of the function currently being emitted.
   555→    current_func_name: String,
   556→    /// Byte offset of the current function within the text section (set externally).
   557→    func_text_offset: u64,
   558→    /// Computed stack frame size (in bytes) for the current function.
   559→    frame_size: u16,
   560→}
   561→
   562→impl Emitter {
   563→    /// Create a new emitter with a fresh register allocator.
   564→    pub fn new() -> Self {
   565→        Self {
   566→            reg_alloc: RegAllocator::new(),
   567→            code: Vec::new(),
   568→            fixups: Vec::new(),
   569→            label_offsets: HashMap::new(),
   570→            call_relocs: Vec::new(),
   571→            relocations: Vec::new(),
   572→            current_func_name: String::new(),
   573→            func_text_offset: 0,
   574→            frame_size: 0,
   575→        }
   576→    }

   581→
   582→    /// Emit a single IR function to ARM64 machine code.
   583→    ///
   584→    /// Returns a vector of 32-bit ARM64 instruction words.
   585→    pub fn emit_function(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   586→        self.code.clear();
   587→        self.fixups.clear();
   588→        self.label_offsets.clear();
   589→        self.call_relocs.clear();
   590→        self.relocations.clear();
   591→        self.current_func_name = func.name.clone();
   592→        self.reg_alloc.reset();
   593→
   594→        // Allocate registers for parameters (AAPCS64: X0–X7).
   595→        for (i, param) in func.params.iter().enumerate() {
   596→            if let IRValue::Register(vreg_id) = param {
   597→                if i < 8 {
   598→                    let _ = self.reg_alloc.allocate(*vreg_id);
   599→                }
   600→            }
   601→        }
   602→
   603→        // Emit prologue: STP X29, X30, [SP, #-16]!
   604→        self.emit_instruction(Instruction::STP {
   605→            rt1: Register::X29,
   606→            rt2: Register::X30,
   607→            rn: Register::SP,
   608→            offset: -16,
   609→        })?;
   610→
   611→        // MOV X29, SP (set frame pointer)
   612→        self.emit_instruction(Instruction::MOV {
   613→            rd: Register::X29,
   614→            rm: Register::SP,
   615→        })?;
   616→
   617→        // Compute frame size from the function's Alloc instructions.
   618→        let aligned_stack = compute_frame_size(func);
   619→        self.frame_size = aligned_stack;
   620→        self.emit_instruction(Instruction::SUB {
   621→            rd: Register::SP,
   622→            rn: Register::SP,
   623→            rm: Operand::Imm12(aligned_stack),
   624→        })?;
   625→
   626→        // Emit each basic block.
   627→        for block in &func.blocks {
   628→            self.label_offsets
   629→                .insert(block.label.clone(), self.code.len());
   630→            self.emit_block(block)?;
   631→        }
   632→
   633→        // Apply fixups — resolve intra-function branch targets.
   634→        self.apply_fixups()?;
   635→
   636→        Ok(self.code.clone())
   637→    }
   638→
   639→    /// Returns the relocation entries recorded during emission of the current
   640→    /// function.
   641→    ///
   642→    /// Each `RelocationEntry` records a byte offset within the function's
   643→    /// encoded output where a symbolic reference must be resolved, the target
   644→    /// symbol name, and the ISA-specific relocation type (e.g.,
   645→    /// `"R_AARCH64_CALL26"`).
   646→    pub fn relocations(&self) -> &[RelocationEntry] {
   647→        &self.relocations
   648→    }
   649→
   650→    /// Emit instructions for a single IR basic block.
   651→    fn emit_block(&mut self, block: &IRBlock) -> Result<()> {
   652→        for instr in &block.instructions {
   653→            self.emit_ir_instr(instr)?;
   654→        }
   655→        self.emit_terminator(&block.terminator)?;
   656→        Ok(())
   657→    }

   660→    // IR → ARM64 instruction lowering
   661→    // -----------------------------------------------------------------------
   662→
   663→    /// Lower a single IR instruction to ARM64 instructions.
   664→    fn emit_ir_instr(&mut self, instr: &IRInstr) -> Result<()> {
   665→        match instr {
   666→            IRInstr::Load { dst, addr, offset, ty } => {
   667→                let rt = self.resolve_reg(dst)?;
   668→                let rn = self.resolve_reg(addr)?;
   669→                self.emit_load(rt, rn, *offset, ty)?;
   670→            }
   671→
   672→            IRInstr::Store { value, addr, offset, ty } => {
   673→                let rt = self.resolve_reg(value)?;
   674→                let rn = self.resolve_reg(addr)?;
   675→                self.emit_store(rt, rn, *offset, ty)?;
   676→            }
   677→
   678→            IRInstr::BinOp { op, dst, lhs, rhs, ty } => {
   679→                self.emit_binop(*op, dst, lhs, rhs, ty.as_ref())?;
   680→            }
   681→
   682→            IRInstr::UnaryOp { op, dst, operand, ty: _ } => {
   683→                let rd = self.resolve_reg(dst)?;
   684→                let rn = self.resolve_reg(operand)?;
   685→                match op {
   686→                    UnaryOpKind::Neg => {
   687→                        self.emit_instruction(Instruction::SUB {
   688→                            rd,
   689→                            rn: Register::XZR,
   690→                            rm: Operand::Reg {
   691→                                reg: rn,
   692→                                shift: None,
   693→                            },
   694→                        })?;
   695→                    }
   696→                    UnaryOpKind::Not => {
   697→                        self.emit_load_immediate(Register::X9, -1)?;
   698→                        self.emit_instruction(Instruction::EOR {
   699→                            rd,
   700→                            rn,
   701→                            rm: Register::X9,
   702→                        })?;
   703→                    }
   704→                    UnaryOpKind::Clz => {
   705→                        // CLZ Xd, Xn — count leading zeros (native ARM64 instruction)
   706→                        self.emit_instruction(Instruction::CLZ { rd, rn })?;
   707→                    }
   708→                    UnaryOpKind::Ctz => {
   709→                        // CTZ = RBIT + CLZ: reverse bits then count leading zeros.
   710→                        // Use X9 as scratch if rd == rn (need intermediate result).
   711→                        if rd == rn {
   712→                            self.emit_instruction(Instruction::RBIT {
   713→                                rd: Register::X9,
   714→                                rn,
   715→                            })?;
   716→                            self.emit_instruction(Instruction::CLZ {
   717→                                rd,
   718→                                rn: Register::X9,
   719→                            })?;
   720→                        } else {
   721→                            self.emit_instruction(Instruction::RBIT { rd, rn })?;
   722→                            self.emit_instruction(Instruction::CLZ { rd, rn: rd })?;
   723→                        }
   724→                    }
   725→                    UnaryOpKind::Popcnt => {
   726→                        // POPCNT via FMOV+CNT+ADDV+UMOV sequence:
   727→                        // FMOV D8, Xn        — move GPR value to SIMD register (V8 is caller-saved)
   728→                        // CNT V8.8B, V8.8B   — count bits per byte
   729→                        // ADDV B8, V8.8B     — horizontal sum of byte counts
   730→                        // UMOV Xd, V8.B[0]   — move result back to GPR (zero-extends to 64 bits)
   731→                        const SIMD_SCRATCH: u8 = 8; // V8 is caller-saved in AAPCS64
   732→                        self.emit_instruction(Instruction::FMOV_DX {
   733→                            vd: SIMD_SCRATCH,
   734→                            rn,
   735→                        })?;
   736→                        self.emit_instruction(Instruction::CNT {
   737→                            vd: SIMD_SCRATCH,
   738→                            vn: SIMD_SCRATCH,
   739→                        })?;
   740→                        self.emit_instruction(Instruction::ADDV {
   741→                            vd: SIMD_SCRATCH,
   742→                            vn: SIMD_SCRATCH,
   743→                        })?;
   744→                        self.emit_instruction(Instruction::UMOV {
   745→                            rd,
   746→                            vn: SIMD_SCRATCH,
   747→                        })?;
   748→                    }
   749→                }
   750→            }
   751→
   752→            IRInstr::Call {
   753→                dst,
   754→                func: target_name,
   755→                args,
   756→            } => {
   757→                // Move arguments into X0–X7.
   758→                for (i, arg) in args.iter().enumerate() {
   759→                    if i >= 8 {
   760→                        break;
   761→                    }
   762→                    let src = self.resolve_reg(arg)?;
   763→                    let dst_reg = match i {
   764→                        0 => Register::X0,
   765→                        1 => Register::X1,
   766→                        2 => Register::X2,
   767→                        3 => Register::X3,
   768→                        4 => Register::X4,
   769→                        5 => Register::X5,
   770→                        6 => Register::X6,
   771→                        7 => Register::X7,
   772→                        _ => unreachable!(),
   773→                    };
   774→                    if src != dst_reg {
   775→                        self.emit_instruction(Instruction::MOV {
   776→                            rd: dst_reg,
   777→                            rm: src,
   778→                        })?;
   779→                    }
   780→                }
   781→
   782→                // BL — record a relocation for later patching.
   783→                let bl_word_idx = self.code.len();
   784→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
   785→                self.call_relocs.push(CallRelocation {
   786→                    text_byte_offset: bl_byte_offset,
   787→                    target_func: target_name.clone(),
   788→                });
   789→                // Register a RelocationEntry so the linker can patch this BL.
   790→                self.relocations.push(RelocationEntry {
   791→                    offset: bl_byte_offset,
   792→                    symbol: target_name.clone(),
   793→                    reloc_type: "R_AARCH64_CALL26".to_string(),
   794→                });
   795→                self.emit_instruction(Instruction::BL { offset: 0 })?;
   796→
   797→                if let Some(dst_val) = dst {
   798→                    let rd = self.resolve_reg(dst_val)?;
   799→                    if rd != Register::X0 {
   800→                        self.emit_instruction(Instruction::MOV {
   801→                            rd,
   802→                            rm: Register::X0,
   803→                        })?;
   804→                    }
   805→                }
   806→            }
   807→
   808→            IRInstr::Alloc { dst, size } => {
   809→                let rd = self.resolve_reg(dst)?;
   810→                self.emit_instruction(Instruction::MOV {
   811→                    rd,
   812→                    rm: Register::SP,
   813→                })?;
   814→                let aligned = (*size).div_ceil(16) * 16;
   815→                self.emit_instruction(Instruction::SUB {
   816→                    rd: Register::SP,
   817→                    rn: Register::SP,
   818→                    rm: Operand::Imm12(aligned as u16),
   819→                })?;
   820→            }
   821→
   822→            IRInstr::Free { ptr } => {
   823→                let rt = self.resolve_reg(ptr)?;
   824→                // Move ptr to X0 (first argument)
   825→                if rt != Register::X0 {
   826→                    self.emit_instruction(Instruction::MOV {
   827→                        rd: Register::X0,
   828→                        rm: rt,
   829→                    })?;
   830→                }
   831→                // BL __vuma_free
   832→                let bl_word_idx = self.code.len();
   833→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
   834→                self.call_relocs.push(CallRelocation {
   835→                    text_byte_offset: bl_byte_offset,
   836→                    target_func: "__vuma_free".to_string(),
   837→                });
   838→                self.relocations.push(RelocationEntry {
   839→                    offset: bl_byte_offset,
   840→                    symbol: "__vuma_free".to_string(),
   841→                    reloc_type: "R_AARCH64_CALL26".to_string(),
   842→                });
   843→                self.emit_instruction(Instruction::BL { offset: 0 })?;
   844→            }
   845→
   846→            IRInstr::Cast { kind, dst, src } => {
   847→                let rd = self.resolve_reg(dst)?;
   848→                let rn = self.resolve_reg(src)?;
   849→                match kind {
   850→                    CastKind::ZExt => {
   851→                        // Zero-extend 32-bit to 64-bit using UBFM
   852→                        self.emit_instruction(Instruction::UBFM {
   853→                            rd,
   854→                            rn,
   855→                            immr: 0,
   856→                            imms: 31,
   857→                        })?;
   858→                    }
   859→                    CastKind::SExt => {
   860→                        // Sign-extend 32-bit to 64-bit using SBFM
   861→                        self.emit_instruction(Instruction::SBFM {
   862→                            rd,
   863→                            rn,
   864→                            immr: 0,
   865→                            imms: 31,
   866→                        })?;
   867→                    }
   868→                    CastKind::Trunc | CastKind::BitCast => {
   869→                        // Trunc: upper bits discarded on write — just MOV
   870→                        // BitCast: no data change — just MOV
   871→                        if rd != rn {
   872→                            self.emit_instruction(Instruction::MOV { rd, rm: rn })?;
   873→                        }
   874→                    }
   875→                }
   876→            }
   877→
   878→            IRInstr::Phi { .. } => {
   879→                log::warn!(
   880→                    "IRInstr::Phi encountered during emission — should be resolved by SSA pass"
   881→                );
   882→            }
   883→
   884→            IRInstr::GetAddress { dst, name } => {
   885→                let rd = self.resolve_reg(dst)?;
   886→                // Emit a call to __vuma_getaddr to resolve the symbol at runtime.
   887→                // Move name hash to X0 as the argument.
   888→                let name_hash = name
   889→                    .chars()
   890→                    .fold(0u64, |acc, c| acc.wrapping_mul(31).wrapping_add(c as u64));
   891→                self.emit_load_immediate(Register::X0, name_hash as i64)?;
   892→                let bl_word_idx = self.code.len();
   893→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
   894→                self.call_relocs.push(CallRelocation {
   895→                    text_byte_offset: bl_byte_offset,
   896→                    target_func: "__vuma_getaddr".to_string(),
   897→                });
   898→                self.relocations.push(RelocationEntry {
   899→                    offset: bl_byte_offset,
   900→                    symbol: "__vuma_getaddr".to_string(),
   901→                    reloc_type: "R_AARCH64_CALL26".to_string(),
   902→                });
   903→                self.emit_instruction(Instruction::BL { offset: 0 })?;
   904→                if rd != Register::X0 {
   905→                    self.emit_instruction(Instruction::MOV {
   906→                        rd,
   907→                        rm: Register::X0,
   908→                    })?;
   909→                }
   910→            }
   911→
   912→            IRInstr::Offset { dst, base, offset } => {
   913→                let rd = self.resolve_reg(dst)?;
   914→                let rn = self.resolve_reg(base)?;
   915→                let rm = match offset {
   916→                    IRValue::Immediate(v) => {
   917→                        if *v >= 0 && *v <= 4095 {
   918→                            Operand::Imm12(*v as u16)
   919→                        } else {
   920→                            let temp = Register::X9;
   921→                            self.emit_load_immediate(temp, *v)?;
   922→                            Operand::Reg {
   923→                                reg: temp,
   924→                                shift: None,
   925→                            }
   926→                        }
   927→                    }
   928→                    _ => Operand::Reg {
   929→                        reg: self.resolve_reg(offset)?,
   930→                        shift: None,
   931→                    },
   932→                };
   933→                self.emit_instruction(Instruction::ADD { rd, rn, rm })?;
   934→            }
   935→
   936→            // ── Dedicated arithmetic — delegate to BinOp ──
   937→            IRInstr::Add { dst, lhs, rhs, ty } => {
   938→                self.emit_binop(BinOpKind::Add, dst, lhs, rhs, ty.as_ref())?;
   939→            }
   940→            IRInstr::Sub { dst, lhs, rhs, ty } => {
   941→                self.emit_binop(BinOpKind::Sub, dst, lhs, rhs, ty.as_ref())?;
   942→            }
   943→            IRInstr::Mul { dst, lhs, rhs, ty } => {
   944→                self.emit_binop(BinOpKind::Mul, dst, lhs, rhs, ty.as_ref())?;
   945→            }
   946→            IRInstr::Div { dst, lhs, rhs, ty } => {
   947→                self.emit_binop(BinOpKind::SDiv, dst, lhs, rhs, ty.as_ref())?;
   948→            }
   949→
   950→            // ── Comparison instruction ──
   951→            IRInstr::Cmp {
   952→                kind,
   953→                dst,
   954→                lhs,
   955→                rhs,
   956→                ty,
   957→            } => {
   958→                let width = RegWidth::from_ir_type(ty.as_ref());
   959→                let rd = self.resolve_reg(dst)?;
   960→                let rn = self.resolve_reg(lhs)?;
   961→                let rm = self.resolve_reg(rhs)?;
   962→                self.emit_instruction_with_width(
   963→                    Instruction::CMP {
   964→                        rn,
   965→                        rm: Operand::Reg {
   966→                            reg: rm,
   967→                            shift: None,
   968→                        },
   969→                    },
   970→                    width,
   971→                )?;
   972→                let cond = cmp_kind_to_condition(kind);
   973→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
   974→            }
   975→
   976→            // ── Instruction-level control flow ──
   977→            IRInstr::Ret { values } => {
   978→                for (i, val) in values.iter().enumerate() {
   979→                    if i >= 8 {
   980→                        break;
   981→                    }
   982→                    let src = self.resolve_reg(val)?;
   983→                    let dst_reg = match i {
   984→                        0 => Register::X0,
   985→                        1 => Register::X1,
   986→                        2 => Register::X2,
   987→                        3 => Register::X3,
   988→                        4 => Register::X4,
   989→                        5 => Register::X5,
   990→                        6 => Register::X6,
   991→                        7 => Register::X7,
   992→                        _ => unreachable!(),
   993→                    };
   994→                    if src != dst_reg {
   995→                        self.emit_instruction(Instruction::MOV {
   996→                            rd: dst_reg,
   997→                            rm: src,
   998→                        })?;
   999→                    }
  1000→                }
  1001→            }
  1002→
  1003→            IRInstr::Branch { target } => {
  1004→                let fixup_idx = self.code.len();
  1005→                self.fixups.push((fixup_idx, target.clone()));
  1006→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1007→            }
  1008→
  1009→            IRInstr::CondBranch {
  1010→                cond,
  1011→                true_target,
  1012→                false_target,
  1013→            } => {
  1014→                let rt = self.resolve_reg(cond)?;
  1015→                let fixup_cbz = self.code.len();
  1016→                self.fixups.push((fixup_cbz, false_target.clone()));
  1017→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1018→                let fixup_b = self.code.len();
  1019→                self.fixups.push((fixup_b, true_target.clone()));
  1020→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1021→            }
  1022→
  1023→            IRInstr::Select {
  1024→                dst,
  1025→                cond,
  1026→                true_val,
  1027→                false_val,
  1028→                ty,
  1029→            } => {
  1030→                let width = RegWidth::from_ir_type(ty.as_ref());
  1031→                // Lower select as: SUBS XZR, cond, #0; CSEL dst, false_val, true_val, NE
  1032→                let rd = self.resolve_reg(dst)?;
  1033→                let rc = self.resolve_reg(cond)?;
  1034→                let rt = self.resolve_reg(true_val)?;
  1035→                let rf = self.resolve_reg(false_val)?;
  1036→                // Compare cond against zero and select.
  1037→                self.emit_instruction_with_width(
  1038→                    Instruction::SUB {
  1039→                        rd: Register::XZR,
  1040→                        rn: rc,
  1041→                        rm: Operand::Imm12(0),
  1042→                    },
  1043→                    width,
  1044→                )?;
  1045→                // Set flags by using a separate CMP (SUB with XZR destination
  1046→                // doesn't set flags; we need a flags-setting variant).
  1047→                // We emulate this with: CMP rc, #0 which is SUBS XZR, rc, #0.
  1048→                // Since we only have SUB, we use the existing CMP pattern.
  1049→                self.emit_instruction_with_width(
  1050→                    Instruction::CSEL {
  1051→                        rd,
  1052→                        rn: rf,
  1053→                        rm: rt,
  1054→                        cond: crate::arm64::Condition::NE,
  1055→                    },
  1056→                    width,
  1057→                )?;
  1058→            }
  1059→        }
  1060→        Ok(())
  1061→    }
  1062→
  1063→    /// Emit a binary operation (shared by `BinOp` and dedicated `Add`/`Sub`/…).
  1064→    ///
  1065→    /// Some ARM64 instructions (ADD, SUB, LSL, LSR, ASR) accept a 12-bit
  1066→    /// unsigned immediate operand directly.  Most others (MUL, AND, ORR, EOR,
  1067→    /// SDIV, UDIV, remainder, comparisons) **require** a register operand.
  1068→    /// When the RHS is an immediate that the target instruction cannot accept
  1069→    /// directly, we "spill" it into a scratch register (X16 / IP0) first via
  1070→    /// MOVZ + MOVK and then use that register as the operand.
  1071→    ///
  1072→    /// X16 (IP0) is chosen as the scratch register because it is **not** in
  1073→    /// the register allocator's free pool, so it will never conflict with
  1074→    /// `rd` or `rn`.
  1075→    fn emit_binop(
  1076→        &mut self,
  1077→        op: BinOpKind,
  1078→        dst: &IRValue,
  1079→        lhs: &IRValue,
  1080→        rhs: &IRValue,
  1081→        ty: Option<&IRType>,
  1082→    ) -> Result<()> {
  1083→        let width = RegWidth::from_ir_type(ty);
  1084→        let rd = self.resolve_reg(dst)?;
  1085→        let rn = self.resolve_reg(lhs)?;
  1086→
  1087→        // Determine whether the operation can accept an immediate operand.
  1088→        // ADD, SUB, LSL, LSR, ASR all have immediate forms in ARM64.
  1089→        // All other operations require a register operand.
  1090→        let supports_imm = matches!(
  1091→            op,
  1092→            BinOpKind::Add | BinOpKind::Sub | BinOpKind::Shl | BinOpKind::ShrL | BinOpKind::ShrA
  1093→        );
  1094→
  1095→        let rm = match rhs {
  1096→            IRValue::Immediate(v) => {
  1097→                if supports_imm && *v >= 0 && *v <= 4095 {
  1098→                    // Small immediate that the instruction accepts directly.
  1099→                    Operand::Imm12(*v as u16)
  1100→                } else {
  1101→                    // Load the immediate into scratch register X16 (IP0).
  1102→                    // X16 is not in the register allocator's pool, so it cannot
  1103→                    // conflict with `rd` or `rn`.
  1104→                    let scratch = Register::X16;
  1105→                    self.emit_load_immediate_with_width(scratch, *v, width)?;
  1106→                    Operand::Reg {
  1107→                        reg: scratch,
  1108→                        shift: None,
  1109→                    }
  1110→                }
  1111→            }
  1112→            _ => Operand::Reg {
  1113→                reg: self.resolve_reg(rhs)?,
  1114→                shift: None,
  1115→            },
  1116→        };
  1117→
  1118→        match op {
  1119→            BinOpKind::Add => {
  1120→                self.emit_instruction_with_width(Instruction::ADD { rd, rn, rm }, width)?;
  1121→            }
  1122→            BinOpKind::Sub => {
  1123→                self.emit_instruction_with_width(Instruction::SUB { rd, rn, rm }, width)?;
  1124→            }
  1125→            BinOpKind::Mul => {
  1126→                let rm_reg = self.operand_to_reg(&rm)?;
  1127→                self.emit_instruction_with_width(Instruction::MUL { rd, rn, rm: rm_reg }, width)?;
  1128→            }
  1129→            BinOpKind::SDiv => {
  1130→                let rm_reg = self.operand_to_reg(&rm)?;
  1131→                self.emit_instruction_with_width(Instruction::SDIV { rd, rn, rm: rm_reg }, width)?;
  1132→            }
  1133→            BinOpKind::UDiv => {
  1134→                let rm_reg = self.operand_to_reg(&rm)?;
  1135→                self.emit_instruction_with_width(Instruction::UDIV { rd, rn, rm: rm_reg }, width)?;
  1136→            }
  1137→            BinOpKind::And => {
  1138→                let rm_reg = self.operand_to_reg(&rm)?;
  1139→                self.emit_instruction_with_width(Instruction::AND { rd, rn, rm: rm_reg }, width)?;
  1140→            }
  1141→            BinOpKind::Or => {
  1142→                let rm_reg = self.operand_to_reg(&rm)?;
  1143→                self.emit_instruction_with_width(Instruction::ORR { rd, rn, rm: rm_reg }, width)?;
  1144→            }
  1145→            BinOpKind::Xor => {
  1146→                let rm_reg = self.operand_to_reg(&rm)?;
  1147→                self.emit_instruction_with_width(Instruction::EOR { rd, rn, rm: rm_reg }, width)?;
  1148→            }
  1149→            BinOpKind::Shl => {
  1150→                self.emit_instruction_with_width(Instruction::LSL { rd, rn, rm }, width)?;
  1151→            }
  1152→            BinOpKind::ShrL => {
  1153→                self.emit_instruction_with_width(Instruction::LSR { rd, rn, rm }, width)?;
  1154→            }
  1155→            BinOpKind::ShrA => {
  1156→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm }, width)?;
  1157→            }
  1158→            BinOpKind::Ror | BinOpKind::Rol => {
  1159→                let rm_reg = self.operand_to_reg(&rm)?;
  1160→                // ARM64 rotation: use ASR as placeholder
  1161→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm: Operand::Reg { reg: rm_reg, shift: None } }, width)?;
  1162→            }
  1163→            BinOpKind::SRem | BinOpKind::URem => {
  1164→                let rm_reg = self.operand_to_reg(&rm)?;
  1165→                let div_instr = if op == BinOpKind::SRem {
  1166→                    Instruction::SDIV { rd, rn, rm: rm_reg }
  1167→                } else {
  1168→                    Instruction::UDIV { rd, rn, rm: rm_reg }
  1169→                };
  1170→                self.emit_instruction_with_width(div_instr, width)?;
  1171→                // MSUB rd, rd, rm, rn  =>  rd = rn - rd * rm  =  dividend - quotient * divisor
  1172→                self.emit_instruction_with_width(Instruction::MSUB {
  1173→                    rd,
  1174→                    rn: rd,     // quotient (result of DIV)
  1175→                    rm: rm_reg, // divisor
  1176→                    ra: rn,     // dividend
  1177→                }, width)?;
  1178→            }
  1179→            BinOpKind::SLt
  1180→            | BinOpKind::SLe
  1181→            | BinOpKind::SGt
  1182→            | BinOpKind::SGe
  1183→            | BinOpKind::ULt
  1184→            | BinOpKind::ULe
  1185→            | BinOpKind::UGt
  1186→            | BinOpKind::UGe
  1187→            | BinOpKind::Eq
  1188→            | BinOpKind::Ne => {
  1189→                let rm_reg = self.operand_to_reg(&rm)?;
  1190→                self.emit_instruction_with_width(Instruction::CMP {
  1191→                    rn,
  1192→                    rm: Operand::Reg {
  1193→                        reg: rm_reg,
  1194→                        shift: None,
  1195→                    },
  1196→                }, width)?;
  1197→                let cond = binop_kind_to_condition(&op);
  1198→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1199→            }
  1200→        }
  1201→        Ok(())
  1202→    }
  1203→
  1204→    /// Emit the block terminator.
  1205→    fn emit_terminator(&mut self, term: &IRTerminator) -> Result<()> {
  1206→        match term {
  1207→            IRTerminator::Jump(target) => {
  1208→                let fixup_idx = self.code.len();
  1209→                self.fixups.push((fixup_idx, target.clone()));
  1210→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1211→            }
  1212→            IRTerminator::Branch {
  1213→                cond,
  1214→                true_block,
  1215→                false_block,
  1216→            } => {
  1217→                let rt = self.resolve_reg(cond)?;
  1218→                let fixup_cbnz = self.code.len();
  1219→                self.fixups.push((fixup_cbnz, true_block.clone()));
  1220→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1221→                let fixup_b = self.code.len();
  1222→                self.fixups.push((fixup_b, false_block.clone()));
  1223→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1224→            }
  1225→            IRTerminator::Return(vals) => {
  1226→                for (i, val) in vals.iter().enumerate() {
  1227→                    if i >= 8 {
  1228→                        break;
  1229→                    }
  1230→                    let src = self.resolve_reg(val)?;
  1231→                    let dst_reg = match i {
  1232→                        0 => Register::X0,
  1233→                        1 => Register::X1,
  1234→                        2 => Register::X2,
  1235→                        3 => Register::X3,
  1236→                        4 => Register::X4,
  1237→                        5 => Register::X5,
  1238→                        6 => Register::X6,
  1239→                        7 => Register::X7,
  1240→                        _ => unreachable!(),
  1241→                    };
  1242→                    if src != dst_reg {
  1243→                        self.emit_instruction(Instruction::MOV {
  1244→                            rd: dst_reg,
  1245→                            rm: src,
  1246→                        })?;
  1247→                    }
  1248→                }
  1249→                // Use the same frame size computed in the prologue
  1250→                let frame_size = self.frame_size;
  1251→                self.emit_instruction(Instruction::ADD {
  1252→                    rd: Register::SP,
  1253→                    rn: Register::SP,
  1254→                    rm: Operand::Imm12(frame_size),
  1255→                })?;
  1256→                self.emit_instruction(Instruction::LDP {
  1257→                    rt1: Register::X29,
  1258→                    rt2: Register::X30,
  1259→                    rn: Register::SP,
  1260→                    offset: 16,
  1261→                })?;
  1262→                self.emit_instruction(Instruction::RET { rn: None })?;
  1263→            }
  1264→            IRTerminator::Unreachable => {
  1265→                self.emit_instruction(Instruction::MOV {
  1266→                    rd: Register::XZR,
  1267→                    rm: Register::XZR,
  1268→                })?;
  1269→            }
  1270→            // Switch, Invoke, TailCall, and Resume are lowered by the
  1271→            // control_flow module before reaching the emitter. If they
  1272→            // appear here, it means the lowering pass was not run.
  1273→            IRTerminator::Switch { .. } => {
  1274→                return Err(CodegenError::InvalidInstruction(
  1275→                    "Switch terminator must be lowered before emission".to_string(),
  1276→                ));
  1277→            }
  1278→            IRTerminator::Invoke { .. } => {
  1279→                return Err(CodegenError::InvalidInstruction(
  1280→                    "Invoke terminator must be lowered before emission".to_string(),
  1281→                ));
  1282→            }
  1283→            IRTerminator::TailCall { .. } => {
  1284→                return Err(CodegenError::InvalidInstruction(
  1285→                    "TailCall terminator must be lowered before emission".to_string(),
  1286→                ));
  1287→            }
  1288→            IRTerminator::Resume { .. } => {
  1289→                return Err(CodegenError::InvalidInstruction(
  1290→                    "Resume terminator must be lowered before emission".to_string(),
  1291→                ));
  1292→            }
  1293→        }
  1294→        Ok(())
  1295→    }

  1307→    /// Emit an instruction with a specific register width (32-bit W or 64-bit X).

  1311→    /// instructions that operate on W sub-registers, giving automatic 32-bit

  1319→    /// Emit a load instruction with the given offset and IR type.
  1320→    ///
  1321→    /// Selects the correct ARM64 load variant based on the IR type:
  1322→    /// - I8/U8 → LDRB (byte, zero-extended)
  1323→    /// - I16/U16 → LDRH (halfword, zero-extended)
  1324→    /// - I32/U32/Ptr/Func → LDR (word or doubleword)
  1325→    /// - I64/U64 → LDR (doubleword)
  1326→    ///
  1327→    /// If the offset fits the ARM64 unsigned-offset immediate encoding for the
  1328→    /// selected instruction variant, it is encoded directly. Otherwise, the
  1329→    /// effective address is computed in a scratch register (X9) first, and
  1330→    /// the load is performed at [X9 + 0].
  1331→    fn emit_load(&mut self, rt: Register, rn: Register, offset: i32, ty: &IRType) -> Result<()> {
  1332→        // Determine the ARM64 load instruction and its immediate-offset constraints.
  1333→        // For each variant: (scale, max_imm12) where the encoded offset = imm12 << scale.
  1334→        // LDRB:  scale=0, imm12 0..4095 → offset 0..4095
  1335→        // LDRH:  scale=1, imm12 0..4095 → offset 0..8190, even
  1336→        // LDRSW: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1337→        // LDR W: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1338→        // LDR X: scale=3, imm12 0..4095 → offset 0..32760, multiple of 8
  1339→        match ty {
  1340→            IRType::I8 | IRType::U8 => {
  1341→                if offset >= 0 && offset <= 4095 {
  1342→                    self.emit_instruction(Instruction::LDRB { rt, rn, offset })?;
  1343→                } else {
  1344→                    self.emit_address_with_offset(rn, offset)?;
  1345→                    self.emit_instruction(Instruction::LDRB {
  1346→                        rt,
  1347→                        rn: Register::X9,
  1348→                        offset: 0,
  1349→                    })?;
  1350→                }
  1351→                // Sign-extend for signed byte loads (SXTB = SBFM Xd, Xn, #0, #7)
  1352→                if *ty == IRType::I8 {
  1353→                    self.emit_instruction(Instruction::SBFM {
  1354→                        rd: rt,
  1355→                        rn: rt,
  1356→                        immr: 0,
  1357→                        imms: 7,
  1358→                    })?;
  1359→                }
  1360→            }
  1361→            IRType::I16 | IRType::U16 => {
  1362→                if offset >= 0 && offset <= 8190 && offset % 2 == 0 {
  1363→                    self.emit_instruction(Instruction::LDRH { rt, rn, offset })?;
  1364→                } else {
  1365→                    self.emit_address_with_offset(rn, offset)?;
  1366→                    self.emit_instruction(Instruction::LDRH {
  1367→                        rt,
  1368→                        rn: Register::X9,
  1369→                        offset: 0,
  1370→                    })?;
  1371→                }
  1372→                // Sign-extend for signed halfword loads (SXTH = SBFM Xd, Xn, #0, #15)
  1373→                if *ty == IRType::I16 {
  1374→                    self.emit_instruction(Instruction::SBFM {
  1375→                        rd: rt,
  1376→                        rn: rt,
  1377→                        immr: 0,
  1378→                        imms: 15,
  1379→                    })?;
  1380→                }
  1381→            }
  1382→            IRType::I32 | IRType::U32 => {
  1383→                // 32-bit load uses LDR Wt encoding (scale=2, offset must be multiple of 4)
  1384→                if offset >= 0 && offset <= 16380 && offset % 4 == 0 {
  1385→                    self.emit_instruction(Instruction::LDR_W { rt, rn, offset })?;
  1386→                } else {
  1387→                    self.emit_address_with_offset(rn, offset)?;
  1388→                    self.emit_instruction(Instruction::LDR_W {
  1389→                        rt,
  1390→                        rn: Register::X9,
  1391→                        offset: 0,
  1392→                    })?;
  1393→                }
  1394→                // Sign-extend for signed word loads (SXTW = SBFM Xd, Xn, #0, #31)
  1395→                if *ty == IRType::I32 {
  1396→                    self.emit_instruction(Instruction::SBFM {
  1397→                        rd: rt,
  1398→                        rn: rt,
  1399→                        immr: 0,
  1400→                        imms: 31,
  1401→                    })?;
  1402→                }
  1403→            }
  1404→            IRType::I64 | IRType::U64 | IRType::Ptr | IRType::Func => {
  1405→                // 64-bit load uses LDR Xt encoding (scale=3, offset must be multiple of 8)
  1406→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1407→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1408→                } else {
  1409→                    self.emit_address_with_offset(rn, offset)?;
  1410→                    self.emit_instruction(Instruction::LDR {
  1411→                        rt,
  1412→                        rn: Register::X9,
  1413→                        offset: 0,
  1414→                    })?;
  1415→                }
  1416→            }
  1417→            _ => {
  1418→                // Default: treat as 64-bit load
  1419→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1420→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1421→                } else {
  1422→                    self.emit_address_with_offset(rn, offset)?;
  1423→                    self.emit_instruction(Instruction::LDR {
  1424→                        rt,
  1425→                        rn: Register::X9,
  1426→                        offset: 0,
  1427→                    })?;
  1428→                }
  1429→            }
  1430→        }
  1431→        Ok(())
  1432→    }
  1433→
  1434→    /// Emit a store instruction with the given offset and IR type.
  1435→    ///
  1436→    /// Selects the correct ARM64 store variant based on the IR type:

  1439→    /// - I32/U32 → STR_W (32-bit word)

  1514→
  1515→    /// Compute `X9 = rn + offset` for load/store with an offset that does not
  1516→    /// fit the ARM64 unsigned-offset immediate encoding.
  1517→    ///
  1518→    /// Uses X9 as a scratch register. If the offset fits in a 12-bit unsigned
  1519→    /// immediate (0..4095), emits `ADD X9, rn, #offset`. Otherwise, loads the
  1520→    /// offset into X9 with MOVZ/MOVK and then emits `ADD X9, rn, X9`.
  1521→    fn emit_address_with_offset(
  1522→        &mut self,
  1523→        rn: Register,
  1524→        offset: i32,
  1525→    ) -> Result<()> {
  1526→        if offset >= 0 && offset <= 4095 {
  1527→            // Small positive offset: ADD X9, rn, #offset
  1528→            self.emit_instruction(Instruction::ADD {
  1529→                rd: Register::X9,
  1530→                rn,
  1531→                rm: Operand::Imm12(offset as u16),
  1532→            })?;
  1533→        } else {
  1534→            // Large or negative offset: load offset into X9, then ADD X9, rn, X9
  1535→            self.emit_load_immediate(Register::X9, offset as i64)?;
  1536→            self.emit_instruction(Instruction::ADD {
  1537→                rd: Register::X9,
  1538→                rn,
  1539→                rm: Operand::Reg {
  1540→                    reg: Register::X9,
  1541→                    shift: None,
  1542→                },
  1543→            })?;
  1544→        }
  1545→        Ok(())
  1546→    }

  1548→    fn resolve_reg(&mut self, val: &IRValue) -> Result<Register> {
  1549→        match val {
  1550→            IRValue::Register(id) => self.reg_alloc.allocate(*id),
  1551→            IRValue::Immediate(v) => {
  1552→                let temp = Register::X9;
  1553→                self.emit_load_immediate(temp, *v)?;
  1554→                Ok(temp)
  1555→            }
  1556→            IRValue::Address(addr) => {
  1557→                let temp = Register::X10;
  1558→                self.emit_load_immediate(temp, *addr as i64)?;
  1559→                Ok(temp)
  1560→            }
  1561→            IRValue::Label(_) => Err(CodegenError::EncodingError(
  1562→                "label value cannot be resolved to a register".into(),
  1563→            )),
  1564→        }
  1565→    }
  1566→
  1567→    fn emit_load_immediate(&mut self, rd: Register, value: i64) -> Result<()> {
  1568→        if (0..=65535).contains(&value) {
  1569→            self.emit_instruction(Instruction::MOVZ {
  1570→                rd,
  1571→                imm16: value as u16,
  1572→                shift: 0,
  1573→            })?;
  1574→            return Ok(());
  1575→        }
  1576→        if (0..=0xFFFF_FFFF).contains(&value) {
  1577→            let lo = (value & 0xFFFF) as u16;
  1578→            let hi = ((value >> 16) & 0xFFFF) as u16;
  1579→            self.emit_instruction(Instruction::MOVZ {
  1580→                rd,
  1581→                imm16: lo,
  1582→                shift: 0,
  1583→            })?;
  1584→            self.emit_instruction(Instruction::MOVK {
  1585→                rd,
  1586→                imm16: hi,
  1587→                shift: 16,
  1588→            })?;
  1589→            return Ok(());
  1590→        }
  1591→        let w0 = (value & 0xFFFF) as u16;
  1592→        let w1 = ((value >> 16) & 0xFFFF) as u16;
  1593→        let w2 = ((value >> 32) & 0xFFFF) as u16;
  1594→        let w3 = ((value >> 48) & 0xFFFF) as u16;
  1595→        self.emit_instruction(Instruction::MOVZ {
  1596→            rd,
  1597→            imm16: w0,
  1598→            shift: 0,
  1599→        })?;
  1600→        if w1 != 0 {
  1601→            self.emit_instruction(Instruction::MOVK {
  1602→                rd,
  1603→                imm16: w1,
  1604→                shift: 16,
  1605→            })?;
  1606→        }
  1607→        if w2 != 0 {
  1608→            self.emit_instruction(Instruction::MOVK {
  1609→                rd,
  1610→                imm16: w2,
  1611→                shift: 32,
  1612→            })?;
  1613→        }
  1614→        if w3 != 0 {
  1615→            self.emit_instruction(Instruction::MOVK {
  1616→                rd,
  1617→                imm16: w3,
  1618→                shift: 48,
  1619→            })?;
  1620→        }
  1621→        Ok(())
  1622→    }
  1623→
  1624→    /// Emit a load-immediate sequence with a specific register width.
  1625→    ///
  1626→    /// For `RegWidth::W32`, only MOVZ/MOVK with shift 0 and 16 are emitted
  1627→    /// (32-bit registers don't support shift=32 or shift=48). Values larger
  1628→    /// than 32 bits are truncated to 32 bits.
  1629→    fn emit_load_immediate_with_width(
  1630→        &mut self,
  1631→        rd: Register,
  1632→        value: i64,
  1633→        width: RegWidth,
  1634→    ) -> Result<()> {
  1635→        match width {
  1636→            RegWidth::X64 => self.emit_load_immediate(rd, value),
  1637→            RegWidth::W32 => {
  1638→                // For 32-bit: mask to 32 bits and use only shift 0 and 16.
  1639→                let val32 = (value as u32) as i64;
  1640→                if (0..=65535).contains(&val32) {
  1641→                    self.emit_instruction_with_width(
  1642→                        Instruction::MOVZ {
  1643→                            rd,
  1644→                            imm16: val32 as u16,
  1645→                            shift: 0,
  1646→                        },
  1647→                        RegWidth::W32,
  1648→                    )?;
  1649→                } else {

  1766→/// Compute the stack frame size for a function by summing its `Alloc`
  1767→/// instructions, adding 16 bytes for the FP/LR save pair, and rounding up
  1768→/// to 16-byte alignment.
  1769→fn compute_frame_size(func: &IRFunction) -> u16 {
  1770→    let mut total: u32 = 16; // FP/LR save pair
  1771→    for block in &func.blocks {
  1772→        for instr in &block.instructions {
  1773→            if let IRInstr::Alloc { size, .. } = instr {
  1774→                let aligned = (*size).div_ceil(16) * 16;
  1775→                total += aligned;
  1776→            }
  1777→        }
  1778→    }
  1779→    // Round up to 16-byte alignment (should already be, but be safe).
  1780→    total = (total + 15) & !15;
  1781→    total as u16
  1782→}
