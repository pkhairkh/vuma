     1→//! Cleanup Invariant Verifier for the IVE module.
     2→//!
     3→//! This module implements VUMA's **cleanup invariant**: "Every acquired resource
     4→//! is eventually released." It operates on a simplified control-flow/resource
     5→//! graph derived from the SCG and verifies:
     6→//!
     7→//! 1. **No resource leaks** — every allocation/acquisition reaches a matching
     8→//!    deallocation/release on ALL execution paths.
     9→//! 2. **No double-free** — the same resource is never freed more than once
    10→//!    on any execution path.
    11→//! 3. **No use-after-free** — no access occurs after a deallocation on any
    12→//!    execution path.
    13→//!
    14→//! # Architecture
    15→//!
    16→//! The verifier works on a [`CleanupGraph`] — a directed graph whose nodes
    17→//! represent resource operations (acquire, release, access) and control-flow
    18→//! points (branch, join, return, error). Edges represent possible execution
    19→//! transfer. The verifier performs path-sensitive analysis by enumerating
    20→//! paths through the graph and checking the cleanup invariant on each.
    21→//!
    22→//! # Resource Model
    23→//!
    24→//! Resources include heap allocations, locks, file handles, and any other
    25→//! acquire/release pair. Each resource is identified by a [`ResourceId`]
    26→//! and tracked across all paths through the graph.
    27→
    28→use crate::result::{CounterExample, VerificationResult, VerificationStatus};
    29→use std::collections::{BTreeMap, BTreeSet, VecDeque};
    30→use std::fmt;

    87→/// The kind of operation represented by a node in the cleanup graph.
    88→#[derive(Debug, Clone, PartialEq, Eq, Hash)]
    89→pub enum OperationKind {
    90→    /// Acquire a resource (allocate, lock, open, …).
    91→    Acquire {
    92→        /// The resource being acquired.
    93→        resource: ResourceId,
    94→        /// The kind of resource.
    95→        kind: ResourceKind,
    96→    },
    97→    /// Release a resource (free, unlock, close, …).
    98→    Release {
    99→        /// The resource being released.
   100→        resource: ResourceId,
   101→        /// The kind of resource.
   102→        kind: ResourceKind,
   103→    },
   104→    /// Access a resource (read/write after acquisition).
   105→    Access {
   106→        /// The resource being accessed.
   107→        resource: ResourceId,
   108→    },
   109→    /// A conditional branch point (e.g. if-else).
   110→    Branch {
   111→        /// A label for the branch condition.
   112→        condition: String,
   113→    },
   114→    /// A join point where branches merge.
   115→    Join,
   116→    /// A normal function return / exit.
   117→    Return,
   118→    /// An early return due to error.
   119→    ErrorReturn {
   120→        /// Optional description of the error.
   121→        description: String,
   122→    },
   123→    /// A no-op / passthrough node (useful for graph construction).
   124→    Passthrough,
   125→}
   126→
   127→impl OperationKind {
   128→    /// Returns the resource referenced by this operation, if any.
   129→    pub fn resource(&self) -> Option<ResourceId> {
   130→        match self {
   131→            OperationKind::Acquire { resource, .. } => Some(*resource),
   132→            OperationKind::Release { resource, .. } => Some(*resource),
   133→            OperationKind::Access { resource } => Some(*resource),
   134→            _ => None,
   135→        }
   136→    }
   137→
   138→    /// Returns `true` if this is an acquire operation.
   139→    pub fn is_acquire(&self) -> bool {
   140→        matches!(self, OperationKind::Acquire { .. })
   141→    }
   142→
   143→    /// Returns `true` if this is a release operation.
   144→    pub fn is_release(&self) -> bool {
   145→        matches!(self, OperationKind::Release { .. })
   146→    }
   147→
   148→    /// Returns `true` if this is an access operation.
   149→    pub fn is_access(&self) -> bool {
   150→        matches!(self, OperationKind::Access { .. })
   151→    }
   152→}

   165→// ---------------------------------------------------------------------------
   166→// Cleanup Graph
   167→// ---------------------------------------------------------------------------
   168→
   169→/// A directed graph representing resource operations and control flow.
   170→///
   171→/// Nodes are [`CleanupNode`]s (acquire, release, access, branch, join, etc.)
   172→/// and edges represent possible execution transfer. The graph is built
   173→/// incrementally and then verified by [`CleanupVerifier`].
   174→#[derive(Debug, Clone, Default)]
   175→pub struct CleanupGraph {
   176→    /// Nodes indexed by [`NodeId`].
   177→    nodes: BTreeMap<NodeId, CleanupNode>,
   178→    /// Adjacency list: node → set of successor node IDs.
   179→    successors: BTreeMap<NodeId, BTreeSet<NodeId>>,
   180→    /// Reverse adjacency list: node → set of predecessor node IDs.
   181→    predecessors: BTreeMap<NodeId, BTreeSet<NodeId>>,
   182→    /// Counter for generating the next `NodeId`.
   183→    next_node_id: u64,
   184→    /// Entry node of the graph (set when verification begins).
   185→    entry: Option<NodeId>,
   186→}
   187→
   188→impl CleanupGraph {
   189→    /// Create a new, empty cleanup graph.
   190→    pub fn new() -> Self {
   191→        Self::default()
   192→    }
   193→
   194→    /// Add a node to the graph, returning its `NodeId`.
   195→    pub fn add_node(&mut self, operation: OperationKind, label: impl Into<String>) -> NodeId {
   196→        let id = NodeId(self.next_node_id);
   197→        self.next_node_id += 1;
   198→        self.nodes.insert(
   199→            id,
   200→            CleanupNode {
   201→                id,
   202→                operation,
   203→                label: label.into(),
   204→            },
   205→        );
   206→        self.successors.insert(id, BTreeSet::new());
   207→        self.predecessors.insert(id, BTreeSet::new());
   208→        id
   209→    }
   210→
   211→    /// Add a directed edge from `source` to `target`.
   212→    ///
   213→    /// Returns `Ok(())` if both nodes exist, or `Err` with a description
   214→    /// if either node is missing.
   215→    pub fn add_edge(&mut self, source: NodeId, target: NodeId) -> Result<(), String> {
   216→        if !self.nodes.contains_key(&source) {
   217→            return Err(format!("source node {source} does not exist"));
   218→        }
   219→        if !self.nodes.contains_key(&target) {
   220→            return Err(format!("target node {target} does not exist"));
   221→        }
   222→        self.successors.get_mut(&source).unwrap().insert(target);
   223→        self.predecessors.get_mut(&target).unwrap().insert(source);
   224→        Ok(())
   225→    }
   226→
   227→    /// Set the entry node for verification.
   228→    pub fn set_entry(&mut self, id: NodeId) -> Result<(), String> {
   229→        if !self.nodes.contains_key(&id) {
   230→            return Err(format!("entry node {id} does not exist"));
   231→        }
   232→        self.entry = Some(id);
   233→        Ok(())
   234→    }
   235→
   236→    /// Get a reference to a node by ID.
   237→    pub fn get_node(&self, id: NodeId) -> Option<&CleanupNode> {
   238→        self.nodes.get(&id)
   239→    }
   240→
   241→    /// Get the successor set of a node.
   242→    pub fn successors_of(&self, id: NodeId) -> Option<&BTreeSet<NodeId>> {
   243→        self.successors.get(&id)
   244→    }
   245→
   246→    /// Get the predecessor set of a node.
   247→    pub fn predecessors_of(&self, id: NodeId) -> Option<&BTreeSet<NodeId>> {
   248→        self.predecessors.get(&id)
   249→    }
   250→
   251→    /// Return the number of nodes in the graph.
   252→    pub fn node_count(&self) -> usize {
   253→        self.nodes.len()
   254→    }
   255→
   256→    /// Return an iterator over all node IDs.
   257→    pub fn node_ids(&self) -> impl Iterator<Item = NodeId> + '_ {
   258→        self.nodes.keys().copied()
   259→    }
   260→
   261→    /// Return all acquire nodes in the graph.
   262→    pub fn acquire_nodes(&self) -> Vec<NodeId> {
   263→        self.nodes
   264→            .iter()
   265→            .filter(|(_, n)| n.operation.is_acquire())
   266→            .map(|(id, _)| *id)
   267→            .collect()
   268→    }
   269→
   270→    /// Return all release nodes in the graph.
   271→    pub fn release_nodes(&self) -> Vec<NodeId> {
   272→        self.nodes
   273→            .iter()
   274→            .filter(|(_, n)| n.operation.is_release())
   275→            .map(|(id, _)| *id)
   276→            .collect()
   277→    }
   278→
   279→    /// Return all access nodes in the graph.
   280→    pub fn access_nodes(&self) -> Vec<NodeId> {
   281→        self.nodes
   282→            .iter()
   283→            .filter(|(_, n)| n.operation.is_access())
   284→            .map(|(id, _)| *id)
   285→            .collect()
   286→    }
   287→
   288→    /// Find all nodes where a specific resource is acquired.
   289→    pub fn acquire_nodes_for(&self, resource: ResourceId) -> Vec<NodeId> {
   290→        self.nodes
   291→            .iter()
   292→            .filter(|(_, n)| match &n.operation {
   293→                OperationKind::Acquire { resource: r, .. } => *r == resource,
   294→                _ => false,
   295→            })
   296→            .map(|(id, _)| *id)
   297→            .collect()
   298→    }
   299→
   300→    /// Find all nodes where a specific resource is released.
   301→    pub fn release_nodes_for(&self, resource: ResourceId) -> Vec<NodeId> {
   302→        self.nodes
   303→            .iter()
   304→            .filter(|(_, n)| match &n.operation {
   305→                OperationKind::Release { resource: r, .. } => *r == resource,
   306→                _ => false,
   307→            })
   308→            .map(|(id, _)| *id)
   309→            .collect()
   310→    }
   311→
   312→    /// Find all nodes that access a specific resource.
   313→    pub fn access_nodes_for(&self, resource: ResourceId) -> Vec<NodeId> {
   314→        self.nodes
   315→            .iter()
   316→            .filter(|(_, n)| match &n.operation {
   317→                OperationKind::Access { resource: r } => *r == resource,
   318→                _ => false,
   319→            })
   320→            .map(|(id, _)| *id)
   321→            .collect()
   322→    }
   323→
   324→    /// Check whether a path exists from `source` to `target` using BFS.
   325→    pub fn has_path(&self, source: NodeId, target: NodeId) -> bool {
   326→        if source == target {
   327→            return true;
   328→        }
   329→        let mut visited = BTreeSet::new();
   330→        let mut queue = VecDeque::new();
   331→        visited.insert(source);
   332→        queue.push_back(source);
   333→        while let Some(current) = queue.pop_front() {
   334→            if let Some(succs) = self.successors_of(current) {
   335→                for &succ in succs {
   336→                    if succ == target {
   337→                        return true;
   338→                    }
   339→                    if visited.insert(succ) {
   340→                        queue.push_back(succ);
   341→                    }
   342→                }
   343→            }
   344→        }
   345→        false
   346→    }
   347→
   348→    /// Return all terminal nodes (nodes with no successors) — these are
   349→    /// the exit points of the graph (Return, ErrorReturn, or dead ends).
   350→    pub fn terminal_nodes(&self) -> Vec<NodeId> {
   351→        self.nodes
   352→            .keys()
   353→            .filter(|&&id| self.successors.get(&id).is_none_or(|s| s.is_empty()))
   354→            .copied()
   355→            .collect()
   356→    }
   357→}

   363→/// The kind of cleanup invariant violation detected.
   364→#[derive(Debug, Clone, PartialEq, Eq, Hash, PartialOrd, Ord)]
   365→pub enum ViolationKind {
   366→    /// A resource was acquired but never released on some path (leak).
   367→    Leak,
   368→    /// A resource was released more than once on some path (double-free).
   369→    DoubleFree,
   370→    /// A resource was accessed after it was released on some path
   371→    /// (use-after-free).
   372→    UseAfterFree,
   373→}
   374→
   375→impl fmt::Display for ViolationKind {
   376→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   377→        match self {
   378→            ViolationKind::Leak => write!(f, "resource leak"),
   379→            ViolationKind::DoubleFree => write!(f, "double free"),
   380→            ViolationKind::UseAfterFree => write!(f, "use after free"),
   381→        }
   382→    }
   383→}

   385→/// A single cleanup invariant violation, with trace information.
   386→#[derive(Debug, Clone)]
   387→pub struct CleanupViolation {
   388→    /// What kind of violation.
   389→    pub kind: ViolationKind,
   390→    /// The resource involved.
   391→    pub resource: ResourceId,
   392→    /// The kind of the resource involved.
   393→    pub resource_kind: ResourceKind,
   394→    /// The execution path leading to the violation (sequence of node labels).
   395→    pub path: Vec<String>,
   396→    /// The node at which the violation is detected.
   397→    pub violation_node: NodeId,
   398→    /// Human-readable description.
   399→    pub description: String,
   400→}
   401→
   402→impl fmt::Display for CleanupViolation {
   403→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   404→        let path_str = self.path.join(" → ");
   405→        write!(
   406→            f,
   407→            "{}: {} ({} at {}): {}",
   408→            self.kind, self.resource, self.resource_kind, path_str, self.description
   409→        )
   410→    }
   411→}

   417→/// Tracks the state of resources along a single execution path during
   418→/// graph traversal.
   419→#[derive(Debug, Clone, Default)]
   420→struct PathState {
   421→    /// Resources currently acquired (not yet released) on this path.
   422→    /// Maps ResourceId → (AcquireNodeId, ResourceKind).
   423→    live_resources: BTreeMap<ResourceId, (NodeId, ResourceKind)>,
   424→    /// Resources that have been released on this path.
   425→    /// Maps ResourceId → (ReleaseNodeId, ResourceKind).
   426→    released_resources: BTreeMap<ResourceId, (NodeId, ResourceKind)>,
   427→    /// Number of times each resource has been released on this path.
   428→    release_count: BTreeMap<ResourceId, usize>,
   429→    /// The path so far (node labels for diagnostics).
   430→    path_labels: Vec<String>,
   431→    /// The path so far (node IDs for structural reasoning).
   432→    path_nodes: Vec<NodeId>,
   433→}
   434→
   435→impl PathState {
   436→    /// Process a node, updating the path state. Returns a list of any
   437→    /// violations detected at this node.
   438→    fn process_node(&mut self, node: &CleanupNode) -> Vec<CleanupViolation> {
   439→        let mut violations = Vec::new();
   440→
   441→        match &node.operation {
   442→            OperationKind::Acquire { resource, kind } => {
   443→                // If this resource was already acquired and not yet released,
   444→                // that's not a violation per se (re-acquisition), but we
   445→                // record the new acquisition point.
   446→                self.live_resources
   447→                    .insert(*resource, (node.id, kind.clone()));
   448→                // If it was previously released, re-acquisition is fine
   449→                // (removes it from released set logically).
   450→            }
   451→            OperationKind::Release { resource, kind } => {
   452→                // Check for double-free
   453→                let count = self.release_count.entry(*resource).or_insert(0);
   454→                *count += 1;
   455→                if *count > 1 {
   456→                    violations.push(CleanupViolation {
   457→                        kind: ViolationKind::DoubleFree,
   458→                        resource: *resource,
   459→                        resource_kind: kind.clone(),
   460→                        path: self.path_labels.clone(),
   461→                        violation_node: node.id,
   462→                        description: format!(
   463→                            "{} released {} time(s) on this path",
   464→                            resource, count
   465→                        ),
   466→                    });
   467→                }
   468→                // Move from live to released
   469→                if let Some((_, rk)) = self.live_resources.remove(resource) {
   470→                    self.released_resources
   471→                        .insert(*resource, (node.id, rk));
   472→                }
   473→            }
   474→            OperationKind::Access { resource }
   475→                // Check for use-after-free: resource has been released
   476→                if self.released_resources.contains_key(resource) =>
   477→                {
   478→                    let kind = self
   479→                        .released_resources
   480→                        .get(resource)
   481→                        .map(|(_, k)| k.clone())
   482→                        .unwrap_or(ResourceKind::Memory);
   483→                    violations.push(CleanupViolation {
   484→                        kind: ViolationKind::UseAfterFree,
   485→                        resource: *resource,
   486→                        resource_kind: kind,
   487→                        path: self.path_labels.clone(),
   488→                        violation_node: node.id,
   489→                        description: format!(
   490→                            "{} accessed after being released on this path",
   491→                            resource
   492→                        ),
   493→                    });
   494→                }
   495→            _ => {}
   496→        }
   497→
   498→        self.path_labels.push(node.label.clone());
   499→        self.path_nodes.push(node.id);
   500→
   501→        violations
   502→    }
   503→
   504→    /// Check for leaks at a terminal node: any live resource that hasn't
   505→    /// been released is a leak.
   506→    fn check_leaks(&self, terminal_node: NodeId) -> Vec<CleanupViolation> {
   507→        let mut violations = Vec::new();
   508→        for (&resource, (_, kind)) in &self.live_resources {
   509→            violations.push(CleanupViolation {
   510→                kind: ViolationKind::Leak,
   511→                resource,
   512→                resource_kind: kind.clone(),
   513→                path: self.path_labels.clone(),
   514→                violation_node: terminal_node,
   515→                description: format!(
   516→                    "{} ({}) acquired but never released on this path",
   517→                    resource, kind
   518→                ),
   519→            });
   520→        }
   521→        violations
   522→    }
   523→}

   528→
   529→/// The cleanup invariant verifier.
   530→///
   531→/// Performs path-sensitive analysis on a [`CleanupGraph`] to detect:
   532→/// - Resource leaks (acquire without matching release on all paths)
   533→/// - Double-free (same resource released more than once on any path)
   534→/// - Use-after-free (access after release on any path)
   535→pub struct CleanupVerifier {
   536→    /// Maximum path length to explore (prevents infinite traversal on cycles).
   537→    max_path_length: usize,
   538→    /// Whether to emit detailed diagnostic logging.
   539→    verbose: bool,
   540→}
   541→
   542→/// The result of cleanup verification.
   543→#[derive(Debug, Clone)]
   544→pub struct CleanupReport {
   545→    /// All violations found.
   546→    pub violations: Vec<CleanupViolation>,
   547→    /// Whether the cleanup invariant holds (no violations).
   548→    pub clean: bool,
   549→    /// Number of paths explored.
   550→    pub paths_explored: usize,
   551→    /// Number of acquire nodes checked.
   552→    pub acquires_checked: usize,
   553→}
   554→
   555→impl CleanupReport {
   556→    /// Create a report from a list of violations.
   557→    pub fn from_violations(
   558→        violations: Vec<CleanupViolation>,
   559→        paths_explored: usize,
   560→        acquires_checked: usize,
   561→    ) -> Self {
   562→        let clean = violations.is_empty();
   563→        Self {
   564→            violations,
   565→            clean,
   566→            paths_explored,
   567→            acquires_checked,
   568→        }
   569→    }
   570→
   571→    /// Convert this report into a [`VerificationResult`] for integration
   572→    /// with the IVE verification engine.
   573→    pub fn to_verification_result(&self) -> VerificationResult {
   574→        if self.clean {
   575→            VerificationResult::new(
   576→                "cleanup",
   577→                VerificationStatus::Proven,
   578→                format!(
   579→                    "cleanup invariant verified: {} acquire(s) checked across {} path(s)",
   580→                    self.acquires_checked, self.paths_explored
   581→                ),
   582→            )
   583→        } else {
   584→            let first = &self.violations[0];
   585→            let path: Vec<String> = first.path.clone();
   586→            VerificationResult::new(
   587→                "cleanup",
   588→                VerificationStatus::Violated {
   589→                    counterexample: CounterExample::new(
   590→                        path,
   591→                        first.violation_node.to_string(),
   592→                        format!("{}", first),
   593→                    ),
   594→                },
   595→                format!(
   596→                    "cleanup invariant violated: {} violation(s) found",
   597→                    self.violations.len()
   598→                ),
   599→            )
   600→        }
   601→    }
   602→}
   603→
   604→impl CleanupVerifier {
   605→    /// Create a new verifier with default settings.
   606→    pub fn new() -> Self {
   607→        Self {
   608→            max_path_length: 256,
   609→            verbose: false,
   610→        }
   611→    }
   612→
   613→    /// Set the maximum path length for traversal.
   614→    pub fn with_max_path_length(mut self, len: usize) -> Self {
   615→        self.max_path_length = len;
   616→        self
   617→    }
   618→
   619→    /// Enable verbose diagnostic output.
   620→    pub fn with_verbose(mut self, verbose: bool) -> Self {
   621→        self.verbose = verbose;
   622→        self
   623→    }
   624→
   625→    /// Verify the cleanup invariant on the given graph.
   626→    ///
   627→    /// This performs a depth-first traversal of all execution paths from the
   628→    /// entry node (or from all nodes if no entry is set), checking for
   629→    /// leaks, double-frees, and use-after-free violations.
   630→    pub fn verify(&self, graph: &CleanupGraph) -> CleanupReport {
   631→        let mut all_violations: Vec<CleanupViolation> = Vec::new();
   632→        let mut paths_explored = 0usize;
   633→
   634→        // Determine starting nodes
   635→        let start_nodes: Vec<NodeId> = if let Some(entry) = graph.entry {
   636→            vec![entry]
   637→        } else {
   638→            // Start from all nodes that have no predecessors (entry points)
   639→            let entries: Vec<NodeId> = graph
   640→                .node_ids()
   641→                .filter(|&id| graph.predecessors_of(id).is_none_or(|p| p.is_empty()))
   642→                .collect();
   643→            if entries.is_empty() && graph.node_count() > 0 {
   644→                // Fallback: start from all nodes
   645→                graph.node_ids().collect()
   646→            } else {
   647→                entries
   648→            }
   649→        };
   650→
   651→        if start_nodes.is_empty() {
   652→            return CleanupReport::from_violations(vec![], 0, 0);
   653→        }
   654→
   655→        let acquires_checked = graph.acquire_nodes().len();
   656→
   657→        // DFS with path state tracking
   658→        for start in &start_nodes {
   659→            let initial_state = PathState::default();
   660→            self.dfs_verify(
   661→                graph,
   662→                *start,
   663→                initial_state,
   664→                &mut all_violations,
   665→                &mut paths_explored,
   666→                &mut BTreeSet::new(),
   667→            );
   668→        }
   669→
   670→        // Deduplicate violations (same kind + resource + violation_node)
   671→        let mut seen = BTreeSet::new();
   672→        let violations: Vec<CleanupViolation> = all_violations
   673→            .into_iter()
   674→            .filter(|v| {
   675→                let key = (v.kind.clone(), v.resource, v.violation_node);
   676→                seen.insert(key)
   677→            })
   678→            .collect();
   679→
   680→        CleanupReport::from_violations(violations, paths_explored, acquires_checked)
   681→    }
   682→
   683→    /// Recursive DFS that explores all paths from `current`, accumulating
   684→    /// violations and tracking resource state.
   685→    fn dfs_verify(
   686→        &self,
   687→        graph: &CleanupGraph,
   688→        current: NodeId,
   689→        mut state: PathState,
   690→        violations: &mut Vec<CleanupViolation>,
   691→        paths_explored: &mut usize,
   692→        visited_on_path: &mut BTreeSet<NodeId>,
   693→    ) {
   694→        // Cycle / path length guard
   695→        if state.path_nodes.len() >= self.max_path_length {
   696→            if self.verbose {
   697→                // Path length limit reached — log if verbose
   698→                let _ = current; // suppress unused warning
   699→            }
   700→            return;
   701→        }
   702→
   703→        // Simple cycle detection: if we've visited this node on the
   704→        // current path, skip (prevents infinite loops).
   705→        if visited_on_path.contains(&current) {
   706→            return;
   707→        }
   708→        visited_on_path.insert(current);
   709→
   710→        // Process the current node
   711→        if let Some(node) = graph.get_node(current) {
   712→            let node_violations = state.process_node(node);
   713→            violations.extend(node_violations);
   714→        } else {
   715→            visited_on_path.remove(&current);
   716→            return;
   717→        }
   718→
   719→        // Get successors
   720→        let succs: Vec<NodeId> = graph
   721→            .successors_of(current)
   722→            .map(|s| s.iter().copied().collect())
   723→            .unwrap_or_default();
   724→
   725→        if succs.is_empty() {
   726→            // Terminal node — check for leaks
   727→            *paths_explored += 1;
   728→            let leak_violations = state.check_leaks(current);
   729→            violations.extend(leak_violations);
   730→        } else {
   731→            // Explore each successor
   732→            for succ in succs {
   733→                self.dfs_verify(
   734→                    graph,
   735→                    succ,
   736→                    state.clone(),
   737→                    violations,
   738→                    paths_explored,
   739→                    visited_on_path,
   740→                );
   741→            }
   742→        }
   743→
   744→        visited_on_path.remove(&current);
   745→    }
   746→
   747→    /// Quick check: for each acquire node, does a matching release node
   748→    /// exist that is reachable from it? This is a fast O(V+E) per acquire
   749→    /// reachability check, but doesn't account for conditional paths.
   750→    ///
   751→    /// Returns a list of (acquire_node, resource) pairs where no release
   752→    /// is reachable.
   753→    pub fn quick_check_reachability(&self, graph: &CleanupGraph) -> Vec<(NodeId, ResourceId)> {
   754→        let mut unreachable: Vec<(NodeId, ResourceId)> = Vec::new();
   755→
   756→        for acquire_id in graph.acquire_nodes() {
   757→            if let Some(node) = graph.get_node(acquire_id) {
   758→                if let OperationKind::Acquire { resource, .. } = &node.operation {
   759→                    // Find all release nodes for this resource
   760→                    let release_ids = graph.release_nodes_for(*resource);
   761→                    let any_reachable = release_ids
   762→                        .iter()
   763→                        .any(|&rid| graph.has_path(acquire_id, rid));
   764→
   765→                    if !any_reachable {
   766→                        unreachable.push((acquire_id, *resource));
   767→                    }
   768→                }
   769→            }
   770→        }
   771→
   772→        unreachable
   773→    }
   774→}
   775→
   776→impl Default for CleanupVerifier {
   777→    fn default() -> Self {
   778→        Self::new()
   779→    }
   780→}
