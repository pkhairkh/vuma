     1→//! Recursive-descent parser for the VUMA language frontend.
     2→//!
     3→//! Consumes the token stream produced by [`crate::lexer::Lexer`] and builds
     4→//! an [`crate::ast::Program`]. The parser uses precedence climbing for
     5→//! expression sub-parsing and implements error recovery by skipping
     6→//! to the next statement boundary (`;` or `}`) or item boundary on failure.
     7→//!
     8→//! # Supported constructs
     9→//!
    10→//! **Items** (top-level declarations):
    11→//! - `fn`, `struct`, `enum`, `region`, `import`, `export`, `const`, `static`, `mod`
    12→//!
    13→//! **Statements**:
    14→//! - `let`, assignment, compound assignment (`+=`, `-=`, etc.), `if`, `while`, `for`,
    15→//!   `loop`, `match`, `return`, `break`, `continue`, `sync`, `free`, `allocate`,
    16→//!   `bd`/`repd`/`capd`/`reld` directives, expression statements
    17→//!
    18→//! **Expressions**:
    19→//! - Binary/unary ops, calls, field/index access, deref, addr-of, cast,
    20→//!   `sizeof`, `alignof`, `derive`, `async`, `spawn`, `allocate` (as expr),
    21→//!   literals (int, float, bool, string, address, null), struct init, namespace access
    22→
    23→use crate::ast::*;
    24→use crate::error::{suggest_keyword, ErrorRecovery, ParseError, ParseErrorKind, ParseResult, Span};
    25→use crate::lexer::{Lexer, Token, TokenKind};
    26→
    27→// ---------------------------------------------------------------------------
    28→// Parser
    29→// ---------------------------------------------------------------------------
    30→
    31→/// Default maximum recursion depth for expression parsing.
    32→const MAX_EXPR_DEPTH: u32 = 256;
    33→
    34→/// A recursive-descent parser that turns a token stream into an AST.
    35→pub struct Parser<'src> {
    36→    /// Underlying lexer.
    37→    lexer: Lexer<'src>,
    38→    /// The most recently consumed token.
    39→    current: Token,
    40→    /// Accumulated parse errors (for error recovery).
    41→    errors: Vec<ParseError>,
    42→    /// Pushback buffer for token rewinding (used in struct literal disambiguation).
    43→    pushback: std::collections::VecDeque<Token>,
    44→    /// Current recursion depth in expression parsing.
    45→    expr_depth: u32,
    46→    /// Maximum allowed recursion depth for expression parsing.
    47→    max_depth: u32,
    48→    /// When true, struct literal parsing is suppressed in `parse_postfix()`.
    49→    /// Used when parsing the end expression of a range (`..`) so that `{`
    50→    /// is not consumed as part of a struct literal and remains available
    51→    /// for the enclosing construct (e.g. a for-loop body).
    52→    no_struct_literal: bool,
    53→}
    54→
    55→/// Token kinds that begin a top-level item declaration.
    56→const ITEM_STARTERS: &[TokenKind] = &[
    57→    TokenKind::Fn,
    58→    TokenKind::Struct,
    59→    TokenKind::Enum,
    60→    TokenKind::Region,
    61→    TokenKind::Import,
    62→    TokenKind::Export,
    63→    TokenKind::Const,
    64→    TokenKind::Static,
    65→    TokenKind::Pub,
    66→    TokenKind::Hash,
    67→    // `mod` is TokenKind::Mod — handled directly
    68→];
    69→
    70→/// Compound-assignment token kinds mapped to their operators.
    71→///
    72→/// Returns `Err(ParseError)` for invalid tokens instead of panicking,
    73→/// so that callers can propagate the error cleanly with `?`.
    74→fn compound_op_from_token(kind: TokenKind, span: Span) -> Result<CompoundOp, ParseError> {
    75→    match kind {
    76→        TokenKind::PlusEq => Ok(CompoundOp::Add),
    77→        TokenKind::MinusEq => Ok(CompoundOp::Sub),
    78→        TokenKind::StarEq => Ok(CompoundOp::Mul),
    79→        TokenKind::SlashEq => Ok(CompoundOp::Div),
    80→        TokenKind::PercentEq => Ok(CompoundOp::Mod),
    81→        TokenKind::AmpEq => Ok(CompoundOp::BitAnd),
    82→        TokenKind::PipeEq => Ok(CompoundOp::BitOr),
    83→        TokenKind::CaretEq => Ok(CompoundOp::BitXor),
    84→        TokenKind::ShlEq => Ok(CompoundOp::Shl),
    85→        TokenKind::ShrEq => Ok(CompoundOp::Shr),
    86→        _ => Err(ParseError::invalid_compound_op(
    87→            format!("expected compound assignment operator, found {:?}", kind),
    88→            span,
    89→        )),
    90→    }
    91→}
    92→
    93→/// Check if a token kind is a compound assignment operator.
    94→fn is_compound_assign(kind: TokenKind) -> bool {
    95→    matches!(
    96→        kind,
    97→        TokenKind::PlusEq
    98→            | TokenKind::MinusEq
    99→            | TokenKind::StarEq
   100→            | TokenKind::SlashEq
   101→            | TokenKind::PercentEq
   102→            | TokenKind::AmpEq
   103→            | TokenKind::PipeEq
   104→            | TokenKind::CaretEq
   105→            | TokenKind::ShlEq
   106→            | TokenKind::ShrEq
   107→    )
   108→}
   109→
   110→impl<'src> Parser<'src> {
   111→    /// Create a new parser over the given source text.
   112→    pub fn new(source: &'src str) -> Self {
   113→        let mut lexer = Lexer::new(source);
   114→        let current = lexer.next_token();
   115→        Self {
   116→            lexer,
   117→            current,
   118→            errors: Vec::new(),
   119→            pushback: std::collections::VecDeque::new(),
   120→            expr_depth: 0,
   121→            max_depth: MAX_EXPR_DEPTH,
   122→            no_struct_literal: false,
   123→        }
   124→    }
   125→
   126→    /// Create a new parser with a custom maximum expression recursion depth.
   127→    pub fn with_max_depth(source: &'src str, max_depth: u32) -> Self {
   128→        let mut parser = Self::new(source);
   129→        parser.max_depth = max_depth;
   130→        parser
   131→    }
   132→
   133→    // -- public entry point --------------------------------------------------
   134→
   135→    /// Parse the full source into a [`Program`].
   136→    ///
   137→    /// If errors are encountered the parser attempts recovery and continues;
   138→    /// the returned [`ParseResult`] carries both the partial AST and any
   139→    /// accumulated non-fatal errors, enabling multi-error reporting in a
   140→    /// single pass.
   141→    pub fn parse_program(&mut self) -> ParseResult<Program> {
   142→        let start = self.current.span.start;
   143→        let mut items = Vec::new();
   144→
   145→        while !self.at(TokenKind::Eof) {
   146→            match self.parse_item() {
   147→                Ok(item) => items.push(item),
   148→                Err(err) => {
   149→                    self.errors.push(err);
   150→                    self.recover_to_item_boundary();
   151→                }
   152→            }
   153→        }
   154→
   155→        let end = self.current.span.end;
   156→        let program = Program {
   157→            items,
   158→            span: Span::new(start, end),
   159→        };
   160→
   161→        if self.errors.is_empty() {
   162→            ParseResult::ok(program)
   163→        } else {
   164→            ParseResult::ok_with_errors(program, std::mem::take(&mut self.errors))
   165→        }
   166→    }
   167→
   168→    /// Return all accumulated parse errors.
   169→    pub fn errors(&self) -> &[ParseError] {
   170→        &self.errors
   171→    }

   173→    // -- lookahead -----------------------------------------------------------
   174→
   175→    /// Peek at the token AFTER `self.current` without consuming anything.
   176→    /// Returns a clone of the peeked token.
   177→    fn peek_next(&mut self) -> Token {
   178→        self.lexer.peek().clone()
   179→    }
   180→
   181→    // -- item parsing --------------------------------------------------------
   182→
   183→    /// Parse a single top-level item or statement.
   184→    fn parse_item(&mut self) -> Result<Item, ParseError> {
   185→        // Parse outer attributes (#[...]) that precede the item.
   186→        let attrs = self.parse_outer_attributes()?;
   187→
   188→        // Parse visibility modifier.
   189→        let visibility = self.parse_visibility()?;
   190→
   191→        match self.current.kind {
   192→            TokenKind::Fn => self.parse_fn_def(false, visibility, attrs).map(Item::FnDef),
   193→            TokenKind::Async => {
   194→                // Could be `async fn` or `async { block }` as expression
   195→                let next = self.peek_next();
   196→                if next.kind == TokenKind::Fn {
   197→                    self.parse_fn_def(true, visibility, attrs).map(Item::FnDef)
   198→                } else {
   199→                    self.parse_stmt().map(Item::Stmt)
   200→                }
   201→            }
   202→            TokenKind::Struct => self
   203→                .parse_struct_def(visibility, attrs)
   204→                .map(Item::StructDef),
   205→            TokenKind::Enum => self.parse_enum_def(visibility, attrs).map(Item::EnumDef),
   206→            TokenKind::Region => {
   207→                // Distinguish: `region name = allocate(...)` vs `region` used as
   208→                // a variable name in an expression/assignment (e.g. `region = allocate(8);`)
   209→                let next = self.peek_next();
   210→                if next.kind == TokenKind::Ident {
   211→                    self.parse_region_def().map(Item::RegionDef)
   212→                } else {
   213→                    self.parse_stmt().map(Item::Stmt)
   214→                }
   215→            }
   216→            TokenKind::Import => self.parse_import().map(Item::Import),
   217→            TokenKind::Export => self.parse_export().map(Item::Export),
   218→            TokenKind::Const => self.parse_const_item(visibility, attrs).map(Item::Const),
   219→            TokenKind::Static => self.parse_static_item(visibility, attrs).map(Item::Static),
   220→            TokenKind::Mod => self.parse_module_def().map(Item::ModuleDef),
   221→            TokenKind::Trait => self.parse_trait_def(visibility, attrs).map(Item::TraitDef),
   222→            TokenKind::Impl => self.parse_impl_block(attrs).map(Item::ImplBlock),
   223→            TokenKind::Ident => {
   224→                let lexeme = self.current.lexeme.as_str();
   225→                match lexeme {
   226→                    "static" => self.parse_static_item(visibility, attrs).map(Item::Static),
   227→                    _ => self.parse_stmt().map(Item::Stmt),
   228→                }
   229→            }
   230→            _ => self.parse_stmt().map(Item::Stmt),
   231→        }
   232→    }
   233→
   234→    /// [`async`] `fn` <ident> [`<` type_params `>`] `(` <params>? `)` [`->` <type>] [`;` | `{` <block> `}`]
   235→    fn parse_fn_def(
   236→        &mut self,
   237→        is_async: bool,
   238→        _visibility: Visibility,
   239→        _attrs: Vec<Attribute>,
   240→    ) -> Result<FnDef, ParseError> {
   241→        let start = self.current.span.start;
   242→
   243→        // Consume 'async' if present
   244→        if self.at(TokenKind::Async) {
   245→            self.advance();
   246→        }
   247→
   248→        self.expect(TokenKind::Fn)?;
   249→
   250→        let name = self.expect_name()?;
   251→
   252→        // Optional generic type parameters: `<T, U>`
   253→        let type_params = if self.at(TokenKind::Lt) {
   254→            self.parse_generic_params()?
   255→        } else {
   256→            Vec::new()
   257→        };
   258→
   259→        self.expect(TokenKind::LParen)?;
   260→
   261→        let params = self.parse_params()?;
   262→
   263→        self.expect(TokenKind::RParen)?;
   264→
   265→        let return_type = if self.at(TokenKind::Arrow) {
   266→            self.advance();
   267→            Some(self.parse_type()?)
   268→        } else {
   269→            None
   270→        };
   271→
   272→        // Optional where clause
   273→        let where_clause = if self.at(TokenKind::Where) {
   274→            Some(self.parse_where_clause()?)
   275→        } else {
   276→            None
   277→        };
   278→
   279→        // For trait methods, we may have `;` instead of a body.
   280→        // We create a synthetic empty block for required method signatures.
   281→        let body = if self.at(TokenKind::LBrace) {
   282→            self.parse_block()?
   283→        } else if self.at(TokenKind::Semicolon) {
   284→            self.advance(); // consume ';'
   285→            Block {
   286→                statements: Vec::new(),
   287→                span: Span::new(self.current.span.start, self.current.span.end),
   288→            }
   289→        } else {
   290→            Block {
   291→                statements: Vec::new(),
   292→                span: Span::synthetic(),
   293→            }
   294→        };
   295→        let end = body.span.end;
   296→
   297→        Ok(FnDef {
   298→            visibility: _visibility,
   299→            attrs: _attrs,
   300→            name,
   301→            type_params,
   302→            params,
   303→            return_type,
   304→            body,
   305→            is_async,
   306→            where_clause,
   307→            span: Span::new(start, end),
   308→        })
   309→    }

   311→    /// `struct` <ident> [`<` type_params `>`] `{` <fields> `}`
   312→    fn parse_struct_def(
   313→        &mut self,
   314→        _visibility: Visibility,
   315→        _attrs: Vec<Attribute>,
   316→    ) -> Result<StructDef, ParseError> {
   317→        let start = self.current.span.start;
   318→        self.expect(TokenKind::Struct)?;
   319→
   320→        let name = self.expect_name()?;
   321→
   322→        // Optional generic type parameters with bounds
   323→        let type_params = if self.at(TokenKind::Lt) {
   324→            self.parse_type_params_with_bounds()?
   325→        } else {
   326→            Vec::new()
   327→        };
   328→
   329→        // Where clause comes before '{'
   330→        let where_clause = if self.at(TokenKind::Where) {
   331→            Some(self.parse_where_clause()?)
   332→        } else {
   333→            None
   334→        };
   335→
   336→        self.expect(TokenKind::LBrace)?;
   337→
   338→        let mut fields = Vec::new();
   339→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   340→            let f_start = self.current.span.start;
   341→            let fname = self.expect_name()?;
   342→            self.expect(TokenKind::Colon)?;
   343→            let ftype = self.parse_type()?;
   344→            let f_end = self.current.span.end;
   345→            fields.push(StructField {
   346→                name: fname,
   347→                ty: ftype,
   348→                span: Span::new(f_start, f_end),
   349→            });
   350→            if self.at(TokenKind::Comma) {
   351→                self.advance();
   352→            } else {
   353→                break;
   354→            }
   355→        }
   356→
   357→        self.expect(TokenKind::RBrace)?;
   358→
   359→        Ok(StructDef {
   360→            visibility: _visibility,
   361→            attrs: _attrs,
   362→            name,
   363→            type_params,
   364→            fields,
   365→            where_clause,
   366→            span: Span::new(start, self.current.span.end),
   367→        })
   368→    }

   370→    /// `enum` <ident> [`<` type_params `>`] `{` <variants> `}`
   371→    fn parse_enum_def(
   372→        &mut self,
   373→        _visibility: Visibility,
   374→        _attrs: Vec<Attribute>,
   375→    ) -> Result<EnumDef, ParseError> {
   376→        let start = self.current.span.start;
   377→        self.expect(TokenKind::Enum)?;
   378→
   379→        let name = self.expect_name()?;
   380→
   381→        let type_params = if self.at(TokenKind::Lt) {
   382→            self.parse_type_params_with_bounds()?
   383→        } else {
   384→            Vec::new()
   385→        };
   386→
   387→        // Where clause comes before '{'
   388→        let where_clause = if self.at(TokenKind::Where) {
   389→            Some(self.parse_where_clause()?)
   390→        } else {
   391→            None
   392→        };
   393→
   394→        self.expect(TokenKind::LBrace)?;
   395→
   396→        let mut variants = Vec::new();
   397→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   398→            let v_start = self.current.span.start;
   399→            let vname = self.expect_name()?;
   400→
   401→            let payload = if self.at(TokenKind::LParen) {
   402→                self.advance();
   403→                let ty = self.parse_type()?;
   404→                self.expect(TokenKind::RParen)?;
   405→                Some(ty)
   406→            } else {
   407→                None
   408→            };
   409→
   410→            let v_end = self.current.span.end;
   411→            variants.push(EnumVariant {
   412→                name: vname,
   413→                payload,
   414→                span: Span::new(v_start, v_end),
   415→            });
   416→            if self.at(TokenKind::Comma) {
   417→                self.advance();
   418→            } else {
   419→                break;
   420→            }
   421→        }
   422→
   423→        self.expect(TokenKind::RBrace)?;
   424→
   425→        Ok(EnumDef {
   426→            visibility: _visibility,
   427→            attrs: _attrs,
   428→            name,
   429→            type_params,
   430→            variants,
   431→            where_clause,
   432→            span: Span::new(start, self.current.span.end),
   433→        })
   434→    }

   436→    /// Parse comma-separated type parameter names inside `< … >`.
   437→    #[allow(dead_code)]
   438→    fn parse_type_params(&mut self) -> Result<Vec<String>, ParseError> {
   439→        self.expect(TokenKind::Lt)?;
   440→        let mut params = Vec::new();
   441→        while !self.at_closing_gt() && !self.at(TokenKind::Eof) {
   442→            params.push(self.expect_name()?);
   443→            if self.at(TokenKind::Comma) {
   444→                self.advance();
   445→            } else {
   446→                break;
   447→            }
   448→        }
   449→        self.expect_gt_closing_generic()?;
   450→        Ok(params)
   451→    }
   452→
   453→    /// Parse type parameters with optional bounds: `<T: Display + Clone, U>`
   454→    fn parse_type_params_with_bounds(&mut self) -> Result<Vec<TypeParam>, ParseError> {
   455→        self.expect(TokenKind::Lt)?;
   456→        let mut params = Vec::new();
   457→        while !self.at_closing_gt() && !self.at(TokenKind::Eof) {
   458→            let name = self.expect_name()?;
   459→            let bounds = if self.at(TokenKind::Colon) {
   460→                self.advance();
   461→                self.parse_trait_bounds()?
   462→            } else {
   463→                Vec::new()
   464→            };
   465→            params.push(TypeParam { name, bounds });
   466→            if self.at(TokenKind::Comma) {
   467→                self.advance();
   468→            } else {
   469→                break;
   470→            }
   471→        }
   472→        self.expect_gt_closing_generic()?;
   473→        Ok(params)
   474→    }
   475→
   476→    /// Parse trait bounds: `Trait + AnotherTrait + ...`
   477→    fn parse_trait_bounds(&mut self) -> Result<Vec<Type>, ParseError> {
   478→        let mut bounds = Vec::new();
   479→        bounds.push(self.parse_type()?);
   480→        while self.at(TokenKind::Plus) {
   481→            self.advance();
   482→            bounds.push(self.parse_type()?);
   483→        }
   484→        Ok(bounds)
   485→    }
   486→
   487→    /// Parse a where clause: `where T: Trait + AnotherTrait, U: Trait`
   488→    fn parse_where_clause(&mut self) -> Result<WhereClause, ParseError> {
   489→        self.expect(TokenKind::Where)?;
   490→        let mut predicates = Vec::new();
   491→        loop {
   492→            let type_name = self.expect_name()?;
   493→            self.expect(TokenKind::Colon)?;
   494→            let bounds = self.parse_trait_bounds()?;
   495→            predicates.push(WherePredicate { type_name, bounds });
   496→            if self.at(TokenKind::Comma) {
   497→                self.advance();
   498→            } else {
   499→                break;
   500→            }
   501→        }
   502→        Ok(WhereClause { predicates })
   503→    }
   504→
   505→    /// Parse generic parameter list `<T, U, ...>` with optional bounds.
   506→    /// Returns parsed type parameters for recording in the AST.
   507→    fn parse_generic_params(&mut self) -> Result<Vec<TypeParam>, ParseError> {
   508→        self.parse_type_params_with_bounds()
   509→    }
   510→
   511→    /// Parse a comma-separated parameter list (may be empty).
   512→    fn parse_params(&mut self) -> Result<Vec<Param>, ParseError> {
   513→        let mut params = Vec::new();
   514→        if self.at(TokenKind::RParen) {
   515→            return Ok(params);
   516→        }
   517→        loop {
   518→            let span = self.current.span;
   519→            let name = self.expect_name()?;
   520→            let ty = if self.at(TokenKind::Colon) {
   521→                self.advance();
   522→                Some(self.parse_type()?)
   523→            } else {
   524→                None
   525→            };
   526→            params.push(Param { name, ty, span });
   527→            if self.at(TokenKind::Comma) {
   528→                self.advance();
   529→            } else {
   530→                break;
   531→            }
   532→        }
   533→        Ok(params)
   534→    }

   536→    /// `region` <ident> `=` `allocate` `(` <expr> `)` `;`
   537→    fn parse_region_def(&mut self) -> Result<RegionDef, ParseError> {
   538→        let start = self.current.span.start;
   539→        self.expect(TokenKind::Region)?;
   540→
   541→        let name = self.expect_name()?;
   542→
   543→        self.expect(TokenKind::Assign)?;
   544→        self.expect(TokenKind::Allocate)?;
   545→        self.expect(TokenKind::LParen)?;
   546→
   547→        let size_expr = self.parse_expr()?;
   548→
   549→        self.expect(TokenKind::RParen)?;
   550→        self.expect(TokenKind::Semicolon)?;
   551→
   552→        Ok(RegionDef {
   553→            name,
   554→            size_expr,
   555→            span: Span::new(start, self.current.span.end),
   556→        })
   557→    }

   559→    /// `import` <string> [`{` <names> `}`] `;`
   560→    fn parse_import(&mut self) -> Result<Import, ParseError> {
   561→        let start = self.current.span.start;
   562→        self.expect(TokenKind::Import)?;
   563→
   564→        let path = self.expect_string()?;
   565→        let mut symbols = Vec::new();
   566→
   567→        if self.at(TokenKind::LBrace) {
   568→            self.advance();
   569→            while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   570→                symbols.push(self.expect_name()?);
   571→                if self.at(TokenKind::Comma) {
   572→                    self.advance();
   573→                } else {
   574→                    break;
   575→                }
   576→            }
   577→            self.expect(TokenKind::RBrace)?;
   578→        }
   579→
   580→        self.expect(TokenKind::Semicolon)?;
   581→
   582→        Ok(Import {
   583→            path,
   584→            symbols,
   585→            span: Span::new(start, self.current.span.end),
   586→        })
   587→    }

   592→        self.expect(TokenKind::Export)?;
   593→        let name = self.expect_name()?;

   595→        Ok(Export {
   596→            name,

   601→    /// `const` <name> [`:` <type>] `=` <expr> `;`

   625→        Ok(ConstDef {
   626→            visibility: _visibility,
   627→            attrs: _attrs,

   632→        })

   699→    /// `trait` <name> [`<` type_params `>`] `{` <members> `}`
   700→    fn parse_trait_def(
   701→        &mut self,
   702→        _visibility: Visibility,
   703→        _attrs: Vec<Attribute>,
   704→    ) -> Result<TraitDef, ParseError> {
   705→        let start = self.current.span.start;
   706→        self.expect(TokenKind::Trait)?;
   707→
   708→        let name = self.expect_name()?;
   709→
   710→        let type_params = if self.at(TokenKind::Lt) {
   711→            self.parse_type_params_with_bounds()?
   712→        } else {
   713→            Vec::new()
   714→        };
   715→
   716→        self.expect(TokenKind::LBrace)?;
   717→
   718→        let mut associated_types = Vec::new();
   719→        let mut associated_consts = Vec::new();
   720→        let mut required_methods = Vec::new();
   721→        let mut provided_methods = Vec::new();
   722→
   723→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   724→            // Associated type: `type Name;`
   725→            if self.at(TokenKind::Type) {
   726→                let next = self.peek_next();
   727→                if next.kind == TokenKind::Ident {
   728→                    self.advance(); // consume 'type'
   729→                    let ty_name = self.expect_name()?;
   730→                    self.expect(TokenKind::Semicolon)?;
   731→                    associated_types.push(ty_name);
   732→                    continue;
   733→                }
   734→            }
   735→            // Associated const: `const NAME: Type [= expr];`
   736→            if self.at(TokenKind::Const) {
   737→                let ac_start = self.current.span.start;
   738→                let next = self.peek_next();
   739→                if next.kind == TokenKind::Ident {
   740→                    self.advance(); // consume 'const'
   741→                    let ac_name = self.expect_name()?;
   742→                    self.expect(TokenKind::Colon)?;
   743→                    let ac_ty = self.parse_type()?;
   744→                    let ac_value = if self.at(TokenKind::Assign) {
   745→                        self.advance();
   746→                        Some(self.parse_expr()?)
   747→                    } else {
   748→                        None
   749→                    };
   750→                    self.expect(TokenKind::Semicolon)?;
   751→                    associated_consts.push(AssocConst {
   752→                        name: ac_name,
   753→                        ty: ac_ty,
   754→                        value: ac_value,
   755→                        span: Span::new(ac_start, self.current.span.end),
   756→                    });
   757→                    continue;
   758→                }
   759→            }
   760→            // Method: `fn name(...) -> T;` or `fn name(...) -> T { body }`
   761→            if self.at(TokenKind::Fn) {
   762→                let method = self.parse_fn_def(false, Visibility::default(), Vec::new())?;
   763→                if method.body.statements.is_empty()
   764→                    && !method.is_async
   765→                    && method.where_clause.is_none()
   766→                {
   767→                    // Empty body suggests a required method signature
   768→                    // But we can't easily tell — if the body is just {} it's provided
   769→                    // We'll check: if body span is just `{}` with nothing inside, treat as required
   770→                    // Actually, let's use a simpler heuristic: if the method has a non-empty
   771→                    // body (more than 0 statements), it's provided.
   772→                    required_methods.push(method);
   773→                } else {
   774→                    provided_methods.push(method);
   775→                }
   776→                continue;
   777→            }
   778→            // Skip unexpected tokens in trait body
   779→            self.advance();
   780→        }
   781→
   782→        self.expect(TokenKind::RBrace)?;
   783→
   784→        let where_clause = if self.at(TokenKind::Where) {
   785→            Some(self.parse_where_clause()?)
   786→        } else {
   787→            None
   788→        };
   789→
   790→        Ok(TraitDef {
   791→            visibility: _visibility,
   792→            attrs: _attrs,
   793→            name,
   794→            type_params,
   795→            associated_types,
   796→            associated_consts,
   797→            required_methods,
   798→            provided_methods,
   799→            where_clause,
   800→            span: Span::new(start, self.current.span.end),
   801→        })
   802→    }

   804→    /// `impl` [<trait_name> `for`] <type> `{` <methods> `}`
   805→    fn parse_impl_block(&mut self, _attrs: Vec<Attribute>) -> Result<ImplBlock, ParseError> {
   806→        let start = self.current.span.start;
   807→        self.expect(TokenKind::Impl)?;
   808→
   809→        // Parse optional generic params on the impl itself: `impl<T> ...`
   810→        // This must come before the trait/type name.
   811→        let type_params = if self.at(TokenKind::Lt) {
   812→            self.parse_generic_params()?
   813→        } else {
   814→            Vec::new()
   815→        };
   816→
   817→        // Parse the first name — could be a trait name or the target type
   818→        let first_name = self.expect_name()?;
   819→
   820→        // Check if `for` follows — if so, first_name is the trait name
   821→        let (trait_name, target_type) = if self.at(TokenKind::For) {
   822→            self.advance(); // consume 'for'
   823→            let target = self.parse_type()?;
   824→            (Some(first_name), target)
   825→        } else {
   826→            // No 'for' — first_name is the target type
   827→            // Check for generic args on the target type
   828→            let target = if self.at(TokenKind::Lt) {
   829→                // Generic type: `Name<Args>`
   830→                self.advance(); // consume '<'
   831→                let mut args = Vec::new();
   832→                if !self.at_closing_gt() {
   833→                    args.push(self.parse_type()?);
   834→                    while self.at(TokenKind::Comma) {
   835→                        self.advance();
   836→                        args.push(self.parse_type()?);
   837→                    }
   838→                }
   839→                self.expect_gt_closing_generic()?;
   840→                Type::Generic {
   841→                    name: first_name,
   842→                    args,
   843→                }
   844→            } else {
   845→                Type::BDBase(first_name)
   846→            };
   847→            (None, target)
   848→        };
   849→
   850→        self.expect(TokenKind::LBrace)?;
   851→
   852→        let mut methods = Vec::new();
   853→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   854→            if self.at(TokenKind::Fn) {
   855→                match self.parse_fn_def(false, Visibility::default(), Vec::new()) {
   856→                    Ok(method) => methods.push(method),
   857→                    Err(err) => {
   858→                        self.errors.push(err);
   859→                        self.recover_to_statement_boundary();
   860→                    }
   861→                }
   862→            } else {
   863→                self.advance(); // skip unexpected tokens
   864→            }
   865→        }
   866→
   867→        self.expect(TokenKind::RBrace)?;
   868→
   869→        let where_clause = if self.at(TokenKind::Where) {
   870→            Some(self.parse_where_clause()?)
   871→        } else {
   872→            None
   873→        };
   874→
   875→        Ok(ImplBlock {
   876→            attrs: _attrs,
   877→            type_params,
   878→            trait_name,
   879→            target_type,
   880→            methods,
   881→            where_clause,
   882→            span: Span::new(start, self.current.span.end),
   883→        })
   884→    }

   886→    // -- block & statements --------------------------------------------------
   887→
   888→    /// `{` <stmt>* `}`
   889→    fn parse_block(&mut self) -> Result<Block, ParseError> {
   890→        let start = self.current.span.start;
   891→        self.expect(TokenKind::LBrace)?;
   892→
   893→        let mut statements = Vec::new();
   894→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   895→            match self.parse_stmt() {
   896→                Ok(stmt) => statements.push(stmt),
   897→                Err(err) => {
   898→                    self.errors.push(err);
   899→                    self.recover_to_statement_boundary();
   900→                }
   901→            }
   902→        }
   903→
   904→        self.expect(TokenKind::RBrace)?;
   905→        let end = self.current.span.end;
   906→
   907→        Ok(Block {
   908→            statements,
   909→            span: Span::new(start, end),
   910→        })
   911→    }
   912→
   913→    /// Parse a single statement.
   914→    fn parse_stmt(&mut self) -> Result<Stmt, ParseError> {
   915→        match self.current.kind {
   916→            TokenKind::Let => self.parse_let_stmt(),
   917→            TokenKind::If => self.parse_if_stmt(),
   918→            TokenKind::While => self.parse_while_stmt(),
   919→            TokenKind::For => self.parse_for_stmt(),
   920→            TokenKind::Match => self.parse_match_stmt(),
   921→            TokenKind::Return => self.parse_return_stmt(),
   922→            TokenKind::Sync => self.parse_sync_stmt(),
   923→            TokenKind::Free => self.parse_free_stmt(),
   924→            TokenKind::Allocate => self.parse_allocate_stmt(),
   925→            // BD directives
   926→            TokenKind::Bd => self.parse_bd_directive(BdDirectiveKind::Bd),
   927→            TokenKind::Repd => self.parse_bd_directive(BdDirectiveKind::Repd),
   928→            TokenKind::Capd => self.parse_bd_directive(BdDirectiveKind::Capd),
   929→            TokenKind::Reld => self.parse_bd_directive(BdDirectiveKind::Reld),
   930→            // Handle `region` used as a variable name in assignments
   931→            TokenKind::Region => self.parse_assign_or_expr_stmt(),
   932→            // Break, Continue, and Loop are now proper keywords
   933→            TokenKind::Break => self.parse_break_stmt(),
   934→            TokenKind::Continue => self.parse_continue_stmt(),
   935→            TokenKind::Loop => self.parse_loop_stmt(),
   936→            // Unsafe block
   937→            TokenKind::Unsafe => self.parse_unsafe_block(),
   938→            // Handle Ident-based type-ascription declarations
   939→            TokenKind::Ident => {
   940→                // Check for type-ascription declaration: `name: type = expr;`
   941→                let next = self.peek_next();
   942→                if next.kind == TokenKind::Colon {
   943→                    self.parse_type_ascription_decl()
   944→                } else {
   945→                    self.parse_assign_or_expr_stmt()
   946→                }
   947→            }
   948→            _ => {
   949→                // Could be an assignment or an expression statement.
   950→                self.parse_assign_or_expr_stmt()
   951→            }
   952→        }
   953→    }
   954→
   955→    /// `let` <name> [`:` <type>] [`=` <expr>] `;`
   956→    fn parse_let_stmt(&mut self) -> Result<Stmt, ParseError> {
   957→        let start = self.current.span.start;
   958→        self.expect(TokenKind::Let)?;
   959→
   960→        let name = self.expect_name()?;
   961→
   962→        let ty = if self.at(TokenKind::Colon) {
   963→            self.advance();
   964→            Some(self.parse_type()?)
   965→        } else {
   966→            None
   967→        };
   968→
   969→        let value = if self.at(TokenKind::Assign) {
   970→            self.advance();
   971→            self.parse_expr()?
   972→        } else {
   973→            // `let x;` without initializer — properly represented as Uninitialized
   974→            Expr::Uninitialized {
   975→                span: Span::new(self.current.span.start, self.current.span.start),
   976→            }
   977→        };
   978→
   979→        self.expect(TokenKind::Semicolon)?;
   980→
   981→        Ok(Stmt::Let(LetStmt {
   982→            name,
   983→            ty,
   984→            value,
   985→            span: Span::new(start, self.current.span.end),
   986→        }))
   987→    }

   989→    /// <name> `:` <type> `=` <expr> `;`  (type-ascription declaration without `let`)
   990→    fn parse_type_ascription_decl(&mut self) -> Result<Stmt, ParseError> {
   991→        let start = self.current.span.start;
   992→
   993→        let name = self.expect_name()?;
   994→        self.expect(TokenKind::Colon)?;
   995→        let ty = self.parse_type()?;
   996→        self.expect(TokenKind::Assign)?;
   997→        let value = self.parse_expr()?;
   998→        self.expect(TokenKind::Semicolon)?;
   999→
  1000→        Ok(Stmt::Let(LetStmt {
  1001→            name,
  1002→            ty: Some(ty),
  1003→            value,
  1004→            span: Span::new(start, self.current.span.end),
  1005→        }))
  1006→    }

  1008→    /// Try to parse an assignment, compound assignment, or expression statement.
  1009→    ///
  1010→    /// Handles: `x = expr;`, `*x = expr;`, `(*x).field = expr;`, `x[i] = expr;`,
  1011→    /// `x += expr;`, `x -= expr;`, etc., or plain expression statement `expr;`.
  1012→    fn parse_assign_or_expr_stmt(&mut self) -> Result<Stmt, ParseError> {
  1013→        let start = self.current.span.start;
  1014→
  1015→        // Parse the full expression first (including prefix ops like `*`),
  1016→        // then check for `=` or compound assignment.
  1017→        let expr = self.parse_expr()?;
  1018→
  1019→        // Compound assignment: `expr += value;`, `expr -= value;`, etc.
  1020→        if is_compound_assign(self.current.kind) {
  1021→            let op = compound_op_from_token(self.current.kind, self.current.span)?;
  1022→            self.advance(); // consume compound op
  1023→            let value = self.parse_expr()?;
  1024→            self.expect(TokenKind::Semicolon)?;
  1025→
  1026→            let target = self.expr_to_assign_target(expr, start)?;
  1027→            return Ok(Stmt::CompoundAssign(CompoundAssignStmt {
  1028→                target,
  1029→                op,
  1030→                value,
  1031→                span: Span::new(start, self.current.span.end),
  1032→            }));
  1033→        }
  1034→
  1035→        if self.at(TokenKind::Assign) {
  1036→            self.advance(); // consume '='
  1037→            let value = self.parse_expr()?;
  1038→            self.expect(TokenKind::Semicolon)?;
  1039→
  1040→            let target = self.expr_to_assign_target(expr, start)?;
  1041→            return Ok(Stmt::Assign(AssignStmt {
  1042→                target,
  1043→                value,
  1044→                span: Span::new(start, self.current.span.end),
  1045→            }));
  1046→        }
  1047→
  1048→        // Plain expression statement.
  1049→        // Allow omitting semicolon before `}` (tail expression / block return value)
  1050→        // or before `)` (last argument in macro-like call) or at EOF.
  1051→        if !self.at(TokenKind::RBrace) && !self.at(TokenKind::RParen) && !self.at(TokenKind::Eof) {
  1052→            self.expect(TokenKind::Semicolon)?;
  1053→        }
  1054→        Ok(Stmt::Expr(ExprStmt {
  1055→            expr,
  1056→            span: Span::new(start, self.current.span.end),
  1057→        }))
  1058→    }
  1059→
  1060→    /// Convert an expression into an assignment target if possible.
  1061→    fn expr_to_assign_target(&self, expr: Expr, _start: usize) -> Result<AssignTarget, ParseError> {
  1062→        match expr {
  1063→            Expr::Var { name, span } => Ok(AssignTarget::Var { name, span }),
  1064→            Expr::Deref { expr, span } => Ok(AssignTarget::Deref { expr, span }),
  1065→            Expr::FieldAccess { expr, field, span } => {
  1066→                Ok(AssignTarget::DerefField { expr, field, span })
  1067→            }
  1068→            Expr::Index { expr, index, span } => Ok(AssignTarget::Index { expr, index, span }),
  1069→            _ => Err(ParseError::new(
  1070→                "invalid assignment target",
  1071→                expr.span(),
  1072→                ParseErrorKind::UnexpectedToken,
  1073→            )),
  1074→        }
  1075→    }

  1105→    /// `if` <expr> `{` <block> `}` [`else` `{` <block> `}` | `else if` …]
  1106→    fn parse_if_stmt(&mut self) -> Result<Stmt, ParseError> {
  1107→        let start = self.current.span.start;
  1108→        self.expect(TokenKind::If)?;
  1109→        let condition = self.parse_expr()?;
  1110→        let then_block = self.parse_block()?;
  1111→        let else_block = if self.at(TokenKind::Else) {
  1112→            self.advance();
  1113→            if self.at(TokenKind::If) {
  1114→                // `else if` — parse as a block containing an if statement
  1115→                let if_stmt = self.parse_if_stmt()?;
  1116→                let span = if_stmt.span();
  1117→                Some(Block {
  1118→                    statements: vec![if_stmt],
  1119→                    span,
  1120→                })
  1121→            } else {
  1122→                Some(self.parse_block()?)
  1123→            }
  1124→        } else {
  1125→            None
  1126→        };
  1127→        Ok(Stmt::If(IfStmt {
  1128→            condition,
  1129→            then_block,
  1130→            else_block,
  1131→            span: Span::new(start, self.current.span.end),
  1132→        }))
  1133→    }

  1148→    /// `for` <name> `in` <expr> `{` <block> `}`
  1149→    fn parse_for_stmt(&mut self) -> Result<Stmt, ParseError> {
  1150→        let start = self.current.span.start;
  1151→        self.expect(TokenKind::For)?;
  1152→        let name = self.expect_name()?;
  1153→        // "in" is tokenized as Ident
  1154→        if self.current.kind == TokenKind::Ident && self.current.lexeme == "in" {
  1155→            self.advance();
  1156→        } else {
  1157→            return Err(ParseError::unexpected(
  1158→                format!("expected 'in', found {}", self.current.kind),
  1159→                self.current.span,
  1160→            ));
  1161→        }
  1162→        let iter = self.parse_expr()?;
  1163→        let body = self.parse_block()?;
  1164→        Ok(Stmt::For(ForStmt {

  1194→    /// `match` <expr> `{` <arms> `}`
  1195→    fn parse_match_stmt(&mut self) -> Result<Stmt, ParseError> {
  1196→        let start = self.current.span.start;
  1197→        self.expect(TokenKind::Match)?;
  1198→        let subject = self.parse_expr()?;
  1199→        self.expect(TokenKind::LBrace)?;
  1200→
  1201→        let mut arms = Vec::new();
  1202→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
  1203→            let arm_start = self.current.span.start;
  1204→            let pattern = self.parse_match_pattern()?;
  1205→
  1206→            // Optional guard: `if expr`
  1207→            let guard = if self.at(TokenKind::If) {
  1208→                self.advance(); // consume 'if'
  1209→                Some(self.parse_expr()?)
  1210→            } else {
  1211→                None
  1212→            };
  1213→
  1214→            self.expect(TokenKind::FatArrow)?;
  1215→            let body = self.parse_expr()?;
  1216→            let arm_end = body.span().end;
  1217→            arms.push(MatchArm {
  1218→                pattern,
  1219→                guard,
  1220→                body,
  1221→                span: Span::new(arm_start, arm_end),
  1222→            });
  1223→            if self.at(TokenKind::Comma) {
  1224→                self.advance();
  1225→            }
  1226→        }
  1227→
  1228→        self.expect(TokenKind::RBrace)?;
  1229→        Ok(Stmt::Match(MatchStmt {
  1230→            subject,
  1231→            arms,
  1232→            span: Span::new(start, self.current.span.end),
  1233→        }))
  1234→    }

  1236→    /// Parse a match pattern.
  1237→    fn parse_match_pattern(&mut self) -> Result<MatchPattern, ParseError> {
  1238→        let span = self.current.span;
  1239→        match self.current.kind {
  1240→            // `_` wildcard — tokenized as either Ident("_") or TokenKind::Underscore
  1241→            TokenKind::Ident if self.current.lexeme == "_" => {
  1242→                self.advance();
  1243→                Ok(MatchPattern::Wildcard(span))
  1244→            }
  1245→            TokenKind::Underscore => {
  1246→                self.advance();
  1247→                Ok(MatchPattern::Wildcard(span))
  1248→            }
  1249→            TokenKind::Number => {
  1250→                let lexeme = self.current.lexeme.clone();
  1251→                self.advance();
  1252→                let value: i64 = lexeme.parse().map_err(|_| {
  1253→                    ParseError::new(
  1254→                        format!("invalid integer literal: {}", lexeme),
  1255→                        span,
  1256→                        ParseErrorKind::UnexpectedToken,
  1257→                    )
  1258→                })?;
  1259→                Ok(MatchPattern::Lit {
  1260→                    value: Lit::Int(value),
  1261→                    span,
  1262→                })
  1263→            }
  1264→            TokenKind::Float => {
  1265→                let lexeme = self.current.lexeme.clone();
  1266→                self.advance();
  1267→                let value: f64 = lexeme.parse().map_err(|_| {
  1268→                    ParseError::new(
  1269→                        format!("invalid float literal: {}", lexeme),
  1270→                        span,
  1271→                        ParseErrorKind::UnexpectedToken,
  1272→                    )
  1273→                })?;
  1274→                Ok(MatchPattern::Lit {
  1275→                    value: Lit::Float(value),
  1276→                    span,
  1277→                })
  1278→            }
  1279→            TokenKind::String => {
  1280→                let value = self.current.lexeme.clone();
  1281→                self.advance();
  1282→                Ok(MatchPattern::Lit {
  1283→                    value: Lit::String(value),
  1284→                    span,
  1285→                })
  1286→            }
  1287→            TokenKind::Address => {
  1288→                let lexeme = self.current.lexeme.clone();
  1289→                self.advance();
  1290→                let hex_str = lexeme.trim_start_matches("0x").trim_start_matches("0X");
  1291→                let value: u64 = u64::from_str_radix(hex_str, 16).map_err(|_| {
  1292→                    ParseError::invalid_address(format!("invalid hex address: {}", lexeme), span)
  1293→                })?;
  1294→                Ok(MatchPattern::Lit {
  1295→                    value: Lit::Address(value),
  1296→                    span,
  1297→                })
  1298→            }
  1299→            TokenKind::True => {
  1300→                self.advance();
  1301→                Ok(MatchPattern::Lit {
  1302→                    value: Lit::Bool(true),
  1303→                    span,
  1304→                })
  1305→            }
  1306→            TokenKind::False => {
  1307→                self.advance();
  1308→                Ok(MatchPattern::Lit {
  1309→                    value: Lit::Bool(false),
  1310→                    span,
  1311→                })
  1312→            }
  1313→            TokenKind::Ident => {
  1314→                let name = self.expect_name()?;
  1315→                // Check for struct pattern: `Name { field, ... }`
  1316→                if self.at(TokenKind::LBrace) {
  1317→                    self.advance();
  1318→                    let mut fields = Vec::new();
  1319→                    while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
  1320→                        fields.push(self.expect_name()?);
  1321→                        if self.at(TokenKind::Comma) {
  1322→                            self.advance();
  1323→                        } else {
  1324→                            break;
  1325→                        }
  1326→                    }
  1327→                    self.expect(TokenKind::RBrace)?;
  1328→                    Ok(MatchPattern::Struct { name, fields, span })
  1329→                } else if self.at(TokenKind::LParen) {
  1330→                    // Enum variant pattern: `Some(v)` or `None()`
  1331→                    self.advance();
  1332→                    let binding = if self.at(TokenKind::RParen) {
  1333→                        None
  1334→                    } else {
  1335→                        let b = self.expect_name()?;
  1336→                        Some(b)
  1337→                    };
  1338→                    self.expect(TokenKind::RParen)?;
  1339→                    Ok(MatchPattern::Enum {
  1340→                        name,
  1341→                        binding,
  1342→                        span,
  1343→                    })
  1344→                } else {
  1345→                    Ok(MatchPattern::Ident { name, span })
  1346→                }
  1347→            }
  1348→            // Handle Option/Result keywords as enum variant patterns
  1349→            TokenKind::SomeKw | TokenKind::NoneKw | TokenKind::OkKw | TokenKind::ErrKw => {
  1350→                let name = self.current.lexeme.clone();
  1351→                self.advance();
  1352→                if self.at(TokenKind::LParen) {
  1353→                    self.advance();
  1354→                    let binding = if self.at(TokenKind::RParen) {
  1355→                        None
  1356→                    } else {
  1357→                        let b = self.expect_name()?;
  1358→                        Some(b)
  1359→                    };
  1360→                    self.expect(TokenKind::RParen)?;
  1361→                    Ok(MatchPattern::Enum {
  1362→                        name,
  1363→                        binding,
  1364→                        span,
  1365→                    })
  1366→                } else {
  1367→                    Ok(MatchPattern::Ident { name, span })
  1368→                }
  1369→            }
  1370→            _ => Err(ParseError::unexpected(
  1371→                format!("expected match pattern, found {}", self.current.kind),
  1372→                self.current.span,
  1373→            )),
  1374→        }
  1375→    }

  1452→    // -- expression parsing (precedence climbing) ----------------------------
  1453→
  1454→    /// Parse an expression with full operator precedence.
  1455→    pub fn parse_expr(&mut self) -> Result<Expr, ParseError> {
  1456→        self.parse_expr_with_precedence(0)
  1457→    }
  1458→
  1459→    /// Precedence-climbing expression parser.
  1460→    ///
  1461→    /// Precedence levels (higher = binds tighter):
  1462→    ///   0  =>  ||
  1463→    ///   1  =>  &&
  1464→    ///   2  =>  ==  !=
  1465→    ///   3  =>  <  <=  >  >=
  1466→    ///   4  =>  |  (bitwise OR)
  1467→    ///   5  =>  ^  (bitwise XOR)
  1468→    ///   6  =>  &  (bitwise AND)
  1469→    ///   7  =>  <<  >>
  1470→    ///   8  =>  +  -
  1471→    ///   9  =>  *  /  %
  1472→    fn parse_expr_with_precedence(&mut self, min_prec: u8) -> Result<Expr, ParseError> {
  1473→        self.expr_depth += 1;
  1474→        if self.expr_depth > self.max_depth {
  1475→            self.expr_depth -= 1;
  1476→            return Err(ParseError::new(
  1477→                format!(
  1478→                    "expression nesting depth exceeds maximum ({})",
  1479→                    self.max_depth
  1480→                ),
  1481→                self.current.span,
  1482→                ParseErrorKind::InvalidSyntax,
  1483→            ));
  1484→        }
  1485→        let result = self.parse_expr_with_precedence_inner(min_prec);
  1486→        self.expr_depth -= 1;
  1487→        result
  1488→    }
  1489→
  1490→    /// Inner implementation of precedence climbing (called after depth check).
  1491→    fn parse_expr_with_precedence_inner(&mut self, min_prec: u8) -> Result<Expr, ParseError> {
  1492→        let mut left = self.parse_unary()?;
  1493→
  1494→        loop {
  1495→            // Range expression: `start..end` — handled as a very low-precedence
  1496→            // binary-like construct that produces Expr::Range instead of BinOp.
  1497→            if self.current.kind == TokenKind::DotDot && min_prec == 0 {
  1498→                let start = left.span().start;
  1499→                self.advance(); // consume '..'
  1500→                // Suppress struct literal parsing for the range end so that
  1501→                // `0..n { … }` does not interpret `n {` as a struct literal.
  1502→                let prev = self.no_struct_literal;
  1503→                self.no_struct_literal = true;
  1504→                let end_expr = self.parse_expr_with_precedence(1)?;
  1505→                self.no_struct_literal = prev;
  1506→                let end = end_expr.span().end;
  1507→                left = Expr::Range {
  1508→                    start: Box::new(left),
  1509→                    end: Box::new(end_expr),
  1510→                    span: Span::new(start, end),
  1511→                };
  1512→                continue;
  1513→            }
  1514→
  1515→            let (op, prec) = match self.current.kind {
  1516→                // Logical
  1517→                TokenKind::OrOr => (BinOp::Or, 0),
  1518→                TokenKind::AndAnd => (BinOp::And, 1),
  1519→                // Comparison
  1520→                TokenKind::EqEq => (BinOp::Eq, 2),
  1521→                TokenKind::Ne => (BinOp::Ne, 2),
  1522→                TokenKind::Lt => (BinOp::Lt, 3),
  1523→                TokenKind::Le => (BinOp::Le, 3),
  1524→                TokenKind::Gt => (BinOp::Gt, 3),
  1525→                TokenKind::Ge => (BinOp::Ge, 3),
  1526→                // Bitwise
  1527→                TokenKind::Pipe => (BinOp::BitOr, 4),
  1528→                TokenKind::Caret => (BinOp::BitXor, 5),
  1529→                TokenKind::Ampersand => (BinOp::BitAnd, 6),
  1530→                TokenKind::Shl => (BinOp::Shl, 7),
  1531→                TokenKind::Shr => (BinOp::Shr, 7),
  1532→                // Additive
  1533→                TokenKind::Plus => (BinOp::Add, 8),
  1534→                TokenKind::Minus => (BinOp::Sub, 8),
  1535→                // Multiplicative
  1536→                TokenKind::Star => (BinOp::Mul, 9),
  1537→                TokenKind::Slash => (BinOp::Div, 9),
  1538→                TokenKind::Percent => (BinOp::Mod, 9),
  1539→                _ => break,
  1540→            };
  1541→
  1542→            if prec < min_prec {
  1543→                break;
  1544→            }
  1545→
  1546→            let start = left.span().start;
  1547→            self.advance(); // consume the operator
  1548→            let right = self.parse_expr_with_precedence(prec + 1)?;
  1549→            let end = right.span().end;
  1550→
  1551→            left = Expr::BinOp {
  1552→                op,
  1553→                lhs: Box::new(left),
  1554→                rhs: Box::new(right),
  1555→                span: Span::new(start, end),
  1556→            };
  1557→        }
  1558→
  1559→        Ok(left)
  1560→    }
  1561→
  1562→    /// Parse a unary expression: prefix `-`, `!`, `*`, `@`, `~`, or primary.
  1563→    fn parse_unary(&mut self) -> Result<Expr, ParseError> {
  1564→        let start = self.current.span.start;
  1565→        match self.current.kind {
  1566→            TokenKind::Minus => {
  1567→                self.advance();
  1568→                let expr = self.parse_unary()?;
  1569→                let end = expr.span().end;
  1570→                Ok(Expr::UnOp {
  1571→                    op: UnOp::Neg,
  1572→                    expr: Box::new(expr),
  1573→                    span: Span::new(start, end),
  1574→                })
  1575→            }
  1576→            TokenKind::Bang => {
  1577→                self.advance();
  1578→                let expr = self.parse_unary()?;
  1579→                let end = expr.span().end;
  1580→                Ok(Expr::UnOp {
  1581→                    op: UnOp::Not,
  1582→                    expr: Box::new(expr),
  1583→                    span: Span::new(start, end),
  1584→                })
  1585→            }
  1586→            TokenKind::Tilde => {
  1587→                self.advance();
  1588→                let expr = self.parse_unary()?;
  1589→                let end = expr.span().end;
  1590→                Ok(Expr::UnOp {
  1591→                    op: UnOp::BitNot,
  1592→                    expr: Box::new(expr),
  1593→                    span: Span::new(start, end),
  1594→                })
  1595→            }
  1596→            TokenKind::Star => {
  1597→                self.advance();
  1598→                let expr = self.parse_unary()?;
  1599→                let end = expr.span().end;
  1600→                Ok(Expr::Deref {
  1601→                    expr: Box::new(expr),
  1602→                    span: Span::new(start, end),
  1603→                })
  1604→            }
  1605→            TokenKind::Ampersand => {
  1606→                // Borrow / address-of: `&expr`
  1607→                self.advance();
  1608→                let expr = self.parse_unary()?;
  1609→                let end = expr.span().end;
  1610→                Ok(Expr::AddressOf {
  1611→                    expr: Box::new(expr),
  1612→                    span: Span::new(start, end),
  1613→                })
  1614→            }
  1615→            TokenKind::Ampersat => {
  1616→                // Address-of: `@expr` (VUMA-specific)
  1617→                self.advance();
  1618→                let expr = self.parse_unary()?;
  1619→                let end = expr.span().end;
  1620→                Ok(Expr::AddressOf {
  1621→                    expr: Box::new(expr),
  1622→                    span: Span::new(start, end),
  1623→                })
  1624→            }
  1625→            _ => self.parse_postfix(),
  1626→        }
  1627→    }
  1628→
  1629→    /// Parse postfix operators: calls, field access, indexing, `as` casts,
  1630→    /// namespace access `::`.
  1631→    fn parse_postfix(&mut self) -> Result<Expr, ParseError> {
  1632→        let mut expr = self.parse_primary()?;
  1633→
  1634→        loop {
  1635→            match self.current.kind {
  1636→                TokenKind::LParen => {
  1637→                    // Function call.
  1638→                    let start = expr.span().start;
  1639→                    self.advance(); // consume '('
  1640→                    let mut args = Vec::new();
  1641→                    if !self.at(TokenKind::RParen) {
  1642→                        args.push(self.parse_expr()?);
  1643→                        while self.at(TokenKind::Comma) {
  1644→                            self.advance();
  1645→                            args.push(self.parse_expr()?);
  1646→                        }
  1647→                    }
  1648→                    self.expect(TokenKind::RParen)?;
  1649→                    let end = self.current.span.end;
  1650→                    expr = Expr::Call {
  1651→                        callee: Box::new(expr),
  1652→                        args,
  1653→                        span: Span::new(start, end),
  1654→                    };
  1655→                }
  1656→                TokenKind::Dot => {
  1657→                    // Field access or .await
  1658→                    let start = expr.span().start;
  1659→                    self.advance(); // consume '.'
  1660→
  1661→                    // Check for .await
  1662→                    if self.current.kind == TokenKind::Await {
  1663→                        self.advance(); // consume 'await'
  1664→                        let end = self.current.span.end;
  1665→                        expr = Expr::Await {
  1666→                            expr: Box::new(expr),
  1667→                            span: Span::new(start, end),
  1668→                        };
  1669→                        continue;
  1670→                    }
  1671→
  1672→                    let field = self.expect_name()?;
  1673→                    let end = self.current.span.end;
  1674→                    expr = Expr::FieldAccess {
  1675→                        expr: Box::new(expr),
  1676→                        field,
  1677→                        span: Span::new(start, end),
  1678→                    };
  1679→                }
  1680→                TokenKind::LBracket => {
  1681→                    // Index access: expr[index]
  1682→                    let start = expr.span().start;
  1683→                    self.advance(); // consume '['
  1684→                    let index = self.parse_expr()?;
  1685→                    self.expect(TokenKind::RBracket)?;
  1686→                    let end = self.current.span.end;
  1687→                    expr = Expr::Index {
  1688→                        expr: Box::new(expr),
  1689→                        index: Box::new(index),
  1690→                        span: Span::new(start, end),
  1691→                    };
  1692→                }
  1693→                TokenKind::LBrace => {
  1694→                    // Struct literal (only if `expr` is an identifier-like name
  1695→                    // and the brace is followed by `ident :` — otherwise the `{`
  1696→                    // belongs to a block statement like `if cond { … }`).
  1697→                    //
  1698→                    // When `no_struct_literal` is set (e.g. parsing the end of a
  1699→                    // range expression like `0..n`), skip this branch entirely so
  1700→                    // that the `{` remains for the enclosing construct.
  1701→                    if self.no_struct_literal {
  1702→                        break;
  1703→                    }
  1704→                    let start = expr.span().start;
  1705→                    if let Expr::Var { name, .. } = &expr {
  1706→                        let name = name.clone();
  1707→                        let saved_lbrace = self.current.clone();
  1708→                        self.advance(); // consume '{'
  1709→
  1710→                        // Disambiguation: if the first token inside the braces
  1711→                        // is NOT `ident :`, this is not a struct literal.
  1712→                        // Rewind and let the caller (if/while/for/match) handle the block.
  1713→                        let is_struct_literal = if self.at(TokenKind::RBrace) {
  1714→                            // Empty braces: `Foo {}` is a valid struct literal
  1715→                            true
