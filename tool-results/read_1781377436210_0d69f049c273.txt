     1тЖТ     1тЖТ//! # LoongArch 64-bit Backend
     2тЖТ     2тЖТ//!
     3тЖТ     3тЖТ//! Implements the `Backend` trait for the LoongArch 64-bit target (LP64 ABI).
     4тЖТ     4тЖТ//! This module provides:
     5тЖТ     5тЖТ//!
     6тЖТ     6тЖТ//! - `Gpr` тАФ General-purpose register enum (r0тАУr31)
     7тЖТ     7тЖТ//! - `Fpr` тАФ Floating-point register enum (f0тАУf31)
     8тЖТ     8тЖТ//! - `Instruction` тАФ LoongArch64 instruction enum with correct encoding
     9тЖТ     9тЖТ//! - Encoding helpers for all 9 instruction formats (2R, 3R, 4R, 2RI8, 2RI12,
    10тЖТ    10тЖТ//!   2RI14, 2RI16, 1RI21, I26)
    11тЖТ    11тЖТ//! - `LoongArch64Backend` тАФ `Backend` implementation that lowers IR to LoongArch64
    12тЖТ    12тЖТ//!   machine code and emits ELF64 binaries
    13тЖТ    13тЖТ//!
    14тЖТ    14тЖТ//! ## LoongArch64 Register Convention (LP64 ABI)
    15тЖТ    15тЖТ//!
    16тЖТ    16тЖТ//! | Register(s) | ABI Name | Role                              |
    17тЖТ    17тЖТ//! |-------------|----------|-----------------------------------|
    18тЖТ    18тЖТ//! | r0          | zero     | Hardwired zero                    |
    19тЖТ    19тЖТ//! | r1          | ra       | Return address                    |
    20тЖТ    20тЖТ//! | r2          | tp       | Thread pointer                    |
    21тЖТ    21тЖТ//! | r3          | sp       | Stack pointer                     |
    22тЖТ    22тЖТ//! | r4тАУr11      | a0тАУa7    | Argument / return registers       |
    23тЖТ    23тЖТ//! | r12тАУr20     | t0тАУt8    | Caller-saved temporaries          |
    24тЖТ    24тЖТ//! | r21         | тАФ        | Reserved                          |
    25тЖТ    25тЖТ//! | r22         | fp       | Frame pointer (callee-saved)      |
    26тЖТ    26тЖТ//! | r23тАУr31     | s0тАУs8    | Callee-saved                      |
    27тЖТ    27тЖТ//!
    28тЖТ    28тЖТ//! ## LoongArch64 FP Register Convention (LP64 ABI)
    29тЖТ    29тЖТ//!
    30тЖТ    30тЖТ//! | Register(s) | ABI Name | Role                              |
    31тЖТ    31тЖТ//! |-------------|----------|-----------------------------------|
    32тЖТ    32тЖТ//! | f0тАУf7       | fa0тАУfa7  | FP argument / return registers    |
    33тЖТ    33тЖТ//! | f8тАУf23      | ft0тАУft15 | Caller-saved temporaries          |
    34тЖТ    34тЖТ//! | f24тАУf31     | fs0тАУfs7  | Callee-saved                      |
    35тЖТ    35тЖТ//!
    36тЖТ    36тЖТ//! ## Instruction Formats
    37тЖТ    37тЖТ//!
    38тЖТ    38тЖТ//! All instructions are 32 bits, little-endian, with no branch delay slots.
    39тЖТ    39тЖТ//! Nine formats: 2R, 3R, 4R, 2RI8, 2RI12, 2RI14, 2RI16, 1RI21, I26.
    40тЖТ    40тЖТ//!

    42тЖТ    42тЖТ//!

   327тЖТ   327тЖТ/// Encode a 2R format instruction.
   328тЖТ   328тЖТ///
   329тЖТ   329тЖТ/// Format: `opcode[31:10] | rj[9:5] | rd[4:0]`
   330тЖТ   330тЖТfn encode_2r(opcode: u32, rj: u32, rd: u32) -> [u8; 4] {
   331тЖТ   331тЖТ    let word = ((opcode & 0x3FF_FFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   332тЖТ   332тЖТ    word.to_le_bytes()
   333тЖТ   333тЖТ}
   334тЖТ   334тЖТ
   335тЖТ   335тЖТ/// Encode a 3R format instruction.
   336тЖТ   336тЖТ///
   337тЖТ   337тЖТ/// Format: `opcode[31:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   338тЖТ   338тЖТfn encode_3r(opcode: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   339тЖТ   339тЖТ    let word = ((opcode & 0x1FFFF) << 15) | ((rk & 0x1F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   340тЖТ   340тЖТ    word.to_le_bytes()
   341тЖТ   341тЖТ}

   345тЖТ   345тЖТ/// Format: `opcode[31:20] | ra[19:15] | rk[14:10] | rj[9:5] | rd[4:0]`

   352тЖТ   352тЖТ        | (rd & 0x1F);
   353тЖТ   353тЖТ    word.to_le_bytes()

  1528тЖТ  1528тЖТ/// Build a minimal ELF64 binary for LoongArch64 from raw code bytes.
  1529тЖТ  1529тЖТ///
  1530тЖТ  1530тЖТ/// Produces a static executable with two LOAD segments:
  1531тЖТ  1531тЖТ///   1. PF_R | PF_X тАФ .text (code)
  1532тЖТ  1532тЖТ///   2. PF_R | PF_W тАФ .data / stack (writable memory)
  1533тЖТ  1533тЖТfn build_loongarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1534тЖТ  1534тЖТ    const PAGE_SIZE: u64 = 0x10000; // 64 KB (LoongArch64 typical page size)
  1535тЖТ  1535тЖТ
  1536тЖТ  1536тЖТ    let elf_header_size: u64 = 64;
  1537тЖТ  1537тЖТ    let phdr_size: u64 = 56;
  1538тЖТ  1538тЖТ    let num_phdrs: u64 = 2;
  1539тЖТ  1539тЖТ    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1540тЖТ  1540тЖТ    // Page-align the text segment start in the file.
  1541тЖТ  1541тЖТ    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1542тЖТ  1542тЖТ    let text_size = code.len() as u64;
  1543тЖТ  1543тЖТ
  1544тЖТ  1544тЖТ    // The data segment starts on the next page after the text.
  1545тЖТ  1545тЖТ    let text_file_end = text_offset + text_size;
  1546тЖТ  1546тЖТ    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1547тЖТ  1547тЖТ    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1548тЖТ  1548тЖТ    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1549тЖТ  1549тЖТ    let entry_point = base_addr + text_offset;
  1550тЖТ  1550тЖТ
  1551тЖТ  1551тЖТ    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1552тЖТ  1552тЖТ
  1553тЖТ  1553тЖТ    // --- e_ident ---
  1554тЖТ  1554тЖТ    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1555тЖТ  1555тЖТ    elf.push(2); // ELFCLASS64
  1556тЖТ  1556тЖТ    elf.push(1); // ELFDATA2LSB
  1557тЖТ  1557тЖТ    elf.push(1); // EV_CURRENT
  1558тЖТ  1558тЖТ    elf.push(3); // ELFOSABI_LINUX
  1559тЖТ  1559тЖТ    elf.push(0); // padding
  1560тЖТ  1560тЖТ    elf.extend_from_slice(&[0u8; 7]); // padding
  1561тЖТ  1561тЖТ
  1562тЖТ  1562тЖТ    // --- ELF header fields ---
  1563тЖТ  1563тЖТ    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1564тЖТ  1564тЖТ    elf.extend_from_slice(&258u16.to_le_bytes()); // e_machine = EM_LOONGARCH
  1565тЖТ  1565тЖТ    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1566тЖТ  1566тЖТ    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1567тЖТ  1567тЖТ    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1568тЖТ  1568тЖТ    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1569тЖТ  1569тЖТ    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1570тЖТ  1570тЖТ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1571тЖТ  1571тЖТ    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1572тЖТ  1572тЖТ    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1573тЖТ  1573тЖТ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1574тЖТ  1574тЖТ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1575тЖТ  1575тЖТ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1576тЖТ  1576тЖТ
  1577тЖТ  1577тЖТ    // --- Program Header 1: LOAD (PF_R | PF_X) тАФ .text ---
  1578тЖТ  1578тЖТ    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1579тЖТ  1579тЖТ    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1580тЖТ  1580тЖТ    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1581тЖТ  1581тЖТ    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1582тЖТ  1582тЖТ    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1583тЖТ  1583тЖТ    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1584тЖТ  1584тЖТ    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1585тЖТ  1585тЖТ    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1586тЖТ  1586тЖТ
  1587тЖТ  1587тЖТ    // --- Program Header 2: LOAD (PF_R | PF_W) тАФ .data / stack ---
  1588тЖТ  1588тЖТ    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1589тЖТ  1589тЖТ    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1590тЖТ  1590тЖТ    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1591тЖТ  1591тЖТ    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1592тЖТ  1592тЖТ    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1593тЖТ  1593тЖТ    elf.extend_from_slice(&0u64.to_le_bytes()); // p_filesz (no initialized data)
  1594тЖТ  1594тЖТ    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1595тЖТ  1595тЖТ    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1596тЖТ  1596тЖТ
  1597тЖТ  1597тЖТ    // --- .text section ---
  1598тЖТ  1598тЖТ    // Pad to page-aligned text_offset
  1599тЖТ  1599тЖТ    while (elf.len() as u64) < text_offset {
  1600тЖТ  1600тЖТ        elf.push(0);
  1601тЖТ  1601тЖТ    }
  1602тЖТ  1602тЖТ    elf.extend_from_slice(code);
  1603тЖТ  1603тЖТ
  1604тЖТ  1604тЖТ    // --- Pad to data segment offset (if needed) ---
  1605тЖТ  1605тЖТ    while (elf.len() as u64) < data_offset {
  1606тЖТ  1606тЖТ        elf.push(0);
  1607тЖТ  1607тЖТ    }
  1608тЖТ  1608тЖТ
  1609тЖТ  1609тЖТ    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1610тЖТ  1610тЖТ
  1611тЖТ  1611тЖТ    elf
  1612тЖТ  1612тЖТ}

  1618тЖТ  1618тЖТ/// Decode a single LoongArch64 32-bit instruction word into a mnemonic string.
  1619тЖТ  1619тЖТfn decode_loongarch64_instruction(word: u32) -> String {
  1620тЖТ  1620тЖТ    let opcode = word & 0x7f;
  1621тЖТ  1621тЖТ    match opcode {
  1622тЖТ  1622тЖТ        0x00 => {
  1623тЖТ  1623тЖТ            let rd = (word >> 7) & 0x1f;
  1624тЖТ  1624тЖТ            let rj = (word >> 15) & 0x1f;
  1625тЖТ  1625тЖТ            let rk = (word >> 10) & 0x1f;
  1626тЖТ  1626тЖТ            format!("add.w $r{}, $r{}, $r{}", rd, rj, rk)
  1627тЖТ  1627тЖТ        }
  1628тЖТ  1628тЖТ        0x01 => {
  1629тЖТ  1629тЖТ            let rd = (word >> 7) & 0x1f;
  1630тЖТ  1630тЖТ            let rj = (word >> 15) & 0x1f;
  1631тЖТ  1631тЖТ            let rk = (word >> 10) & 0x1f;
  1632тЖТ  1632тЖТ            format!("sub.w $r{}, $r{}, $r{}", rd, rj, rk)
  1633тЖТ  1633тЖТ        }
  1634тЖТ  1634тЖТ        0x02 => {
  1635тЖТ  1635тЖТ            let rd = (word >> 7) & 0x1f;
  1636тЖТ  1636тЖТ            let rj = (word >> 15) & 0x1f;
  1637тЖТ  1637тЖТ            let rk = (word >> 10) & 0x1f;
  1638тЖТ  1638тЖТ            format!("mul.w $r{}, $r{}, $r{}", rd, rj, rk)
  1639тЖТ  1639тЖТ        }
  1640тЖТ  1640тЖТ        0x05 => {
  1641тЖТ  1641тЖТ            let rd = (word >> 7) & 0x1f;
  1642тЖТ  1642тЖТ            let _rj = (word >> 15) & 0x1f; // lu12i.w does not use rj in the disassembly
  1643тЖТ  1643тЖТ            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1644тЖТ  1644тЖТ            format!("lu12i.w $r{}, {}({})", rd, si12, si12)
  1645тЖТ  1645тЖТ        }
  1646тЖТ  1646тЖТ        0x08 => {
  1647тЖТ  1647тЖТ            let rd = (word >> 7) & 0x1f;
  1648тЖТ  1648тЖТ            let rj = (word >> 15) & 0x1f;
  1649тЖТ  1649тЖТ            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1650тЖТ  1650тЖТ            format!("ld.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1651тЖТ  1651тЖТ        }
  1652тЖТ  1652тЖТ        0x0a => {
  1653тЖТ  1653тЖТ            let rd = (word >> 7) & 0x1f;
  1654тЖТ  1654тЖТ            let rj = (word >> 15) & 0x1f;
  1655тЖТ  1655тЖТ            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1656тЖТ  1656тЖТ            format!("st.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1657тЖТ  1657тЖТ        }
  1658тЖТ  1658тЖТ        0x0c => {
  1659тЖТ  1659тЖТ            let rd = (word >> 7) & 0x1f;
  1660тЖТ  1660тЖТ            let rj = (word >> 15) & 0x1f;
  1661тЖТ  1661тЖТ            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1662тЖТ  1662тЖТ            format!("ld.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1663тЖТ  1663тЖТ        }
  1664тЖТ  1664тЖТ        0x0e => {
  1665тЖТ  1665тЖТ            let rd = (word >> 7) & 0x1f;
  1666тЖТ  1666тЖТ            let rj = (word >> 15) & 0x1f;
  1667тЖТ  1667тЖТ            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1668тЖТ  1668тЖТ            format!("st.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1669тЖТ  1669тЖТ        }
  1670тЖТ  1670тЖТ        0x10 => {
  1671тЖТ  1671тЖТ            let rj = (word >> 15) & 0x1f;
  1672тЖТ  1672тЖТ            let offs = ((word >> 10) as i32) << 12 >> 10;
  1673тЖТ  1673тЖТ            format!("beq $r{}, {}({})", rj, offs, offs)
  1674тЖТ  1674тЖТ        }
  1675тЖТ  1675тЖТ        0x11 => {
  1676тЖТ  1676тЖТ            let rj = (word >> 15) & 0x1f;
  1677тЖТ  1677тЖТ            let offs = ((word >> 10) as i32) << 12 >> 10;
  1678тЖТ  1678тЖТ            format!("bne $r{}, {}({})", rj, offs, offs)
  1679тЖТ  1679тЖТ        }
  1680тЖТ  1680тЖТ        0x14 => format!("bl {}", ((word >> 10) as i32) << 12 >> 10),
  1681тЖТ  1681тЖТ        0x15 => {
  1682тЖТ  1682тЖТ            let rd = (word >> 7) & 0x1f;
  1683тЖТ  1683тЖТ            format!("jirl $r{}, {}", rd, ((word >> 10) as i32) << 12 >> 10)
  1684тЖТ  1684тЖТ        }
  1685тЖТ  1685тЖТ        _ => format!(".word {:08x}", word),
  1686тЖТ  1686тЖТ    }
  1687тЖТ  1687тЖТ}

  1709тЖТ  1709тЖТ/// Compute the stack frame size for an IR function on LoongArch64.
  1710тЖТ  1710тЖТ///
  1711тЖТ  1711тЖТ/// Sums `Alloc` instruction sizes, adds 16 bytes for the ra/fp save pair,
  1712тЖТ  1712тЖТ/// and rounds up to 16-byte alignment.
  1713тЖТ  1713тЖТfn loongarch64_compute_frame_size(func: &IRFunction) -> usize {
  1714тЖТ  1714тЖТ    let mut total: u32 = 16; // ra + fp save pair
  1715тЖТ  1715тЖТ    for block in &func.blocks {
  1716тЖТ  1716тЖТ        for instr in &block.instructions {
  1717тЖТ  1717тЖТ            if let IRInstr::Alloc { size, .. } = instr {
  1718тЖТ  1718тЖТ                let aligned = (*size).div_ceil(16) * 16;
  1719тЖТ  1719тЖТ                total += aligned;
  1720тЖТ  1720тЖТ            }
  1721тЖТ  1721тЖТ        }
  1722тЖТ  1722тЖТ    }
  1723тЖТ  1723тЖТ    // Round up to 16-byte alignment
  1724тЖТ  1724тЖТ    total = (total + 15) & !15;
  1725тЖТ  1725тЖТ    total as usize
  1726тЖТ  1726тЖТ}

  1765тЖТ  1765тЖТ/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1766тЖТ  1766тЖТfn map_vreg_to_gpr(
  1767тЖТ  1767тЖТ    vreg_id: u32,
  1768тЖТ  1768тЖТ    arg_index: Option<usize>,
  1769тЖТ  1769тЖТ    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1770тЖТ  1770тЖТ) -> Gpr {
  1771тЖТ  1771тЖТ    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1772тЖТ  1772тЖТ        return *gpr;
  1773тЖТ  1773тЖТ    }
  1774тЖТ  1774тЖТ
  1775тЖТ  1775тЖТ    // If this is an argument, map it to the corresponding argument register.
