     1в†’//! # MIPS 64-bit Backend
     2в†’//!
     3в†’//! Implements the `Backend` trait for the MIPS 64-bit target (N64 ABI,
     4в†’//! big-endian).  This module provides:
     5в†’//!
     6в†’//! - `Gpr` вЂ” General-purpose register enum ($0вЂ“$31)
     7в†’//! - `Fpr` вЂ” Floating-point register enum ($f0вЂ“$f31)
     8в†’//! - `Instruction` вЂ” MIPS64 instruction enum with correct encoding for
     9в†’//!   R-type, I-type, and J-type formats
    10в†’//! - `Mips64Backend` вЂ” `Backend` implementation that lowers IR to MIPS64
    11в†’//!   machine code and emits ELF64 binaries
    12в†’//!
    13в†’//! ## MIPS64 Register Convention (N64 ABI)
    14в†’//!
    15в†’//! | Register(s) | ABI Name | Role                              |
    16в†’//! |-------------|----------|-----------------------------------|
    17в†’//! | $0          | $zero    | Hardwired zero                    |
    18в†’//! | $1          | $at      | Assembler temporary               |
    19в†’//! | $2вЂ“$3       | $v0вЂ“$v1  | Return values                     |
    20в†’//! | $4вЂ“$7       | $a0вЂ“$a3  | Argument registers                |
    21в†’//! | $8вЂ“$15      | $t0вЂ“$t7  | Caller-saved temporaries          |
    22в†’//! | $16вЂ“$23     | $s0вЂ“$s7  | Callee-saved                      |
    23в†’//! | $24вЂ“$25     | $t8вЂ“$t9  | Caller-saved temporaries          |
    24в†’//! | $26вЂ“$27     | $k0вЂ“$k1  | Kernel registers                  |
    25в†’//! | $28         | $gp      | Global pointer                    |
    26в†’//! | $29         | $sp      | Stack pointer                     |
    27в†’//! | $30         | $fp      | Frame pointer (callee-saved)      |
    28в†’//! | $31         | $ra      | Return address                    |
    29в†’//!
    30в†’//! ## Branch Delay Slots
    31в†’//!
    32в†’//! MIPS has branch delay slots: the instruction immediately following a branch
    33в†’//! or jump is always executed before the branch takes effect.  This backend
    34в†’//! inserts a NOP (0x00000000) in every delay slot for correctness.
    35в†’//!
    36в†’//! ## Instruction Encoding
    37в†’//!
    38в†’//! All instructions are 32 bits, **big-endian**, with three formats:
    39в†’//!
    40в†’//! - **R-type**: `opcode[31:26] | rs[25:21] | rt[20:16] | rd[15:11] | sa[10:6] | funct[5:0]`
    41в†’//! - **I-type**: `opcode[31:26] | rs[25:21] | rt[20:16] | imm[15:0]`
    42в†’//! - **J-type**: `opcode[31:26] | target[25:0]`
    43в†’//!
    44в†’//! ## References
    45в†’//!
    46в†’//! - MIPS64 Architecture for Programmers, Volume II: Instruction Set
    47в†’//! - <https://www.mips.com/products/architectures/mips64/>

   461в†’/// Encode an R-type instruction (big-endian).
   462в†’///
   463в†’/// Format: `opcode[31:26] | rs[25:21] | rt[20:16] | rd[15:11] | sa[10:6] | funct[5:0]`

  1493в†’/// Build a proper ELF64 binary for MIPS64 (big-endian) with 2 LOAD segments.
  1494в†’///
  1495в†’/// Produces a static executable with:
  1496в†’/// - Segment 1: LOAD RX вЂ” `.text` section (code)
  1497в†’/// - Segment 2: LOAD RW вЂ” `.data` / BSS (writable memory for stack/data)
  1498в†’///
  1499в†’/// All header fields are written in big-endian byte order.
  1500в†’/// e_flags is set to EF_MIPS_ARCH_64 (0x60000000) for the MIPS64 N64 ABI.
  1501в†’fn build_mips64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1502в†’    const PAGE_SIZE: u64 = 0x10000; // 64 KB (MIPS typical page size)
  1503в†’
  1504в†’    let elf_header_size: u64 = 64;
  1505в†’    let phdr_size: u64 = 56;
  1506в†’    let num_phdrs: u64 = 2;
  1507в†’    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1508в†’    // Page-align the text segment start in the file.
  1509в†’    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1510в†’    let text_size = code.len() as u64;
  1511в†’
  1512в†’    // The data segment starts on the next page after the text.
  1513в†’    let text_file_end = text_offset + text_size;
  1514в†’    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1515в†’    let data_offset = data_vaddr - base_addr;
  1516в†’    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1517в†’    let entry_point = base_addr + text_offset;
  1518в†’
  1519в†’    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1520в†’
  1521в†’    // --- e_ident ---
  1522в†’    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1523в†’    elf.push(2); // ELFCLASS64
  1524в†’    elf.push(2); // ELFDATA2MSB (big-endian)
  1525в†’    elf.push(1); // EV_CURRENT
  1526в†’    elf.push(3); // ELFOSABI_LINUX
  1527в†’    elf.push(0); // padding
  1528в†’    elf.extend_from_slice(&[0u8; 7]); // padding
  1529в†’
  1530в†’    // --- ELF header fields (big-endian) ---
  1531в†’    elf.extend_from_slice(&2u16.to_be_bytes()); // e_type = ET_EXEC
  1532в†’    elf.extend_from_slice(&8u16.to_be_bytes()); // e_machine = EM_MIPS
  1533в†’    elf.extend_from_slice(&1u32.to_be_bytes()); // e_version
  1534в†’    elf.extend_from_slice(&entry_point.to_be_bytes()); // e_entry
  1535в†’    elf.extend_from_slice(&elf_header_size.to_be_bytes()); // e_phoff
  1536в†’    elf.extend_from_slice(&0u64.to_be_bytes()); // e_shoff (no section headers)
  1537в†’    // e_flags: EF_MIPS_ARCH_64 = 0x60000000 (MIPS64 ISA, N64 ABI implied by ELFCLASS64)
  1538в†’    elf.extend_from_slice(&0x60000000u32.to_be_bytes()); // e_flags
  1539в†’    elf.extend_from_slice(&64u16.to_be_bytes()); // e_ehsize
  1540в†’    elf.extend_from_slice(&56u16.to_be_bytes()); // e_phentsize
  1541в†’    elf.extend_from_slice(&2u16.to_be_bytes()); // e_phnum = 2
  1542в†’    elf.extend_from_slice(&64u16.to_be_bytes()); // e_shentsize
  1543в†’    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shnum
  1544в†’    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shstrndx
  1545в†’
  1546в†’    // --- Program Header 1: LOAD (PF_R | PF_X) вЂ” .text ---
  1547в†’    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1548в†’    elf.extend_from_slice(&5u32.to_be_bytes()); // p_flags = PF_R | PF_X
  1549в†’    elf.extend_from_slice(&text_offset.to_be_bytes()); // p_offset
  1550в†’    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_vaddr
  1551в†’    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_paddr
  1552в†’    elf.extend_from_slice(&text_size.to_be_bytes()); // p_filesz
  1553в†’    elf.extend_from_slice(&text_size.to_be_bytes()); // p_memsz
  1554в†’    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1555в†’
  1556в†’    // --- Program Header 2: LOAD (PF_R | PF_W) вЂ” .data / stack ---
  1557в†’    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1558в†’    elf.extend_from_slice(&6u32.to_be_bytes()); // p_flags = PF_R | PF_W
  1559в†’    elf.extend_from_slice(&data_offset.to_be_bytes()); // p_offset
  1560в†’    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_vaddr
  1561в†’    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_paddr
  1562в†’    elf.extend_from_slice(&0u64.to_be_bytes()); // p_filesz (no initialized data)
  1563в†’    elf.extend_from_slice(&data_size.to_be_bytes()); // p_memsz (writable pages)
  1564в†’    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1565в†’
  1566в†’    // --- .text section ---
  1567в†’    // Pad to page-aligned text_offset
  1568в†’    while (elf.len() as u64) < text_offset {
  1569в†’        elf.push(0);
  1570в†’    }
  1571в†’    elf.extend_from_slice(code);
  1572в†’
  1573в†’    // --- Pad to data segment offset (if needed) ---
  1574в†’    while (elf.len() as u64) < data_offset {
  1575в†’        elf.push(0);
  1576в†’    }
  1577в†’
  1578в†’    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1579в†’
  1580в†’    elf
  1581в†’}

  1611в†’/// Compute the stack frame size for an IR function on MIPS64.
  1612в†’///
  1613в†’/// Sums `Alloc` instruction sizes, adds 8 bytes for the $ra save slot,
  1614в†’/// and rounds up to 16-byte alignment.
  1615в†’fn mips64_compute_frame_size(func: &IRFunction) -> usize {
  1616в†’    let mut total: u32 = 8; // $ra save slot
  1617в†’    for block in &func.blocks {
  1618в†’        for instr in &block.instructions {
  1619в†’            if let IRInstr::Alloc { size, .. } = instr {
  1620в†’                let aligned = (*size).div_ceil(16) * 16;
  1621в†’                total += aligned;
  1622в†’            }
  1623в†’        }
  1624в†’    }
  1625в†’    // Round up to 16-byte alignment
  1626в†’    total = (total + 15) & !15;
  1627в†’    total as usize
  1628в†’}
