     1→//! ARM64 Codegen End-to-End Tests
     2→//!
     3→//! Tests exercising the full ARM64 code-generation pipeline:
     4→//! SCG → IR → register allocation → ARM64 instruction selection → ELF emission.
     5→//!
     6→//! Each test constructs a codegen-level SCG, lowers it to IR, emits ARM64
     7→//! machine code, and validates structural properties of the output (correct
     8→//! instruction presence, proper prologue/epilogue, valid ELF headers, etc.).
     9→//!
    10→//! # Test Matrix
    11→//!
    12→//! | # | Test                                    | What it validates                              |
    13→//! |---|-----------------------------------------|------------------------------------------------|
    14→//! | 1 | Simple add function                     | Add → IR → ARM64 ADD instruction               |
    15→//! | 2 | Function with stack allocation          | Alloc → stack frame layout                     |
    16→//! | 3 | Load/store round-trip                   | Load/Store → LDR/STR encoding                  |
    17→//! | 4 | If/else control flow                    | CondBranch → CBNZ + branch fixups              |
    18→//! | 5 | Loop with back-edge                     | Loop header phi + back-edge branch             |
    19→//! | 6 | Function call                           | Call → BL relocation                           |
    20→//! | 7 | Multiple functions in one program       | Multi-function ELF with symbol table           |
    21→//! | 8 | Type system / calling convention        | IRType sizes, arg classification, stack layout  |
    22→//! | 9 | Bare-metal raw binary emission          | Raw binary output (no ELF headers)             |
    23→//! | 10| ARM64 instruction encoding correctness  | Encode/decode round-trip for key instructions   |
    24→
    25→use vuma_codegen::{
    26→    arm64::{BarrierOption, Condition, Instruction, Operand, Register},
    27→    emit::{emit_elf, emit_raw, EmitConfig, Emitter},
    28→    ir::{
    29→        alignment_of, classify_arg, compute_calling_conv, compute_stack_layout, size_of, ArgClass,
    30→        BinOpKind, IRInstr, IRProgram, IRTerminator, IRType, RegisterClass,
    31→    },
    32→    scg_to_ir::{
    33→        AccessNode, AllocationNode, CallNode, ComputationNode, ControlNode, IRBuilder, Scg,
    34→        ScgExpr, ScgFunction, ScgNode, ScgParam, ScgStatement, ScgType,
    35→    },
    36→};

    38→// ---------------------------------------------------------------------------
    39→// Helpers
    40→// ---------------------------------------------------------------------------
    41→
    42→/// Compile a codegen `Scg` through the full IR → ARM64 emission pipeline.
    43→fn compile_scg(scg: &Scg) -> (IRProgram, Vec<u32>) {
    44→    let mut builder = IRBuilder::new();
    45→    let ir_program = builder.build(scg).expect("IRBuilder should succeed");
    46→    let mut emitter = Emitter::new();
    47→    let code_words = emitter
    48→        .emit_function(&ir_program.functions[0])
    49→        .expect("Emission should succeed");
    50→    (ir_program, code_words)
    51→}
    52→
    53→/// Compile a full program (potentially multiple functions) and emit an ELF.
    54→fn compile_to_elf(scg: &Scg, config: &EmitConfig) -> Vec<u8> {
    55→    let mut builder = IRBuilder::new();
    56→    let ir_program = builder.build(scg).expect("IRBuilder should succeed");
    57→    emit_elf(&ir_program.functions, &ir_program.data_sections, config)
    58→        .expect("ELF emission should succeed")
    59→}

    64→
    65→/// Test: Lower a simple `fn add(a, b) -> i64 { a + b }` to ARM64.
    66→///
    67→/// Validates:
    68→/// - IR function has two params and an Add instruction
    69→/// - ARM64 code contains a valid ADD instruction encoding
    70→/// - The function returns the result in X0 (AAPCS64)
    71→#[test]
    72→fn test_codegen_simple_add() {
    73→    let scg = Scg {
    74→        nodes: vec![ScgNode::Function(ScgFunction {
    75→            name: "add".to_string(),
    76→            params: vec![
    77→                ScgParam {
    78→                    name: "a".to_string(),
    79→                    ty: ScgType::I64,
    80→                },
    81→                ScgParam {
    82→                    name: "b".to_string(),
    83→                    ty: ScgType::I64,
    84→                },
    85→            ],
    86→            results: vec![ScgType::I64],
    87→            body: vec![
    88→                ScgStatement::Computation(ComputationNode {
    89→                    dst: "result".to_string(),
    90→                    op: BinOpKind::Add,
    91→                    lhs: ScgExpr::Var("a".to_string()),
    92→                    rhs: ScgExpr::Var("b".to_string()),
    93→                    tail_call: false,
    94→                }),
    95→                ScgStatement::Return(vec![ScgExpr::Var("result".to_string())]),
    96→            ],
    97→        })],
    98→    };
    99→
   100→    let (ir_program, code_words) = compile_scg(&scg);
   101→
   102→    // Validate IR structure
   103→    let func = &ir_program.functions[0];
   104→    assert_eq!(func.name, "add");
   105→    assert_eq!(func.params.len(), 2);
   106→    assert_eq!(func.result_types.len(), 1);
   107→
   108→    // Check that an Add instruction exists
   109→    let has_add = func.blocks.iter().any(|b| {
   110→        b.instructions
   111→            .iter()
   112→            .any(|i| matches!(i, IRInstr::Add { .. }))
   113→    });
   114→    assert!(has_add, "IR should contain an Add instruction");
   115→
   116→    // Check ARM64 code was emitted
   117→    assert!(!code_words.is_empty(), "ARM64 code should be non-empty");
   118→
   119→    // First instruction should be STP (prologue)
   120→    let first = code_words[0];
   121→    assert_ne!(first, 0, "Prologue STP should not be zero-encoded");
   122→}

   471→// Test 8: Type system and calling convention
   472→// ---------------------------------------------------------------------------
   473→
   474→/// Test: Validate IR type system, argument classification, and calling convention.
   475→///
   476→/// Tests:
   477→/// - IRType size/alignment calculations are correct for ARM64 LP64
   478→/// - Argument classification follows AAPCS64 (integer → Integer, FP → FP, etc.)
   479→/// - Calling convention assigns correct registers (X0–X7 for integer args)
   480→#[test]
   481→fn test_codegen_type_system_calling_conv() {
   482→    // --- Size/alignment ---
   483→    assert_eq!(size_of(&IRType::I8), 1);
   484→    assert_eq!(size_of(&IRType::I64), 8);
   485→    assert_eq!(size_of(&IRType::U32), 4);
   486→    assert_eq!(size_of(&IRType::F64), 8);
   487→    assert_eq!(size_of(&IRType::Ptr), 8);
   488→    assert_eq!(size_of(&IRType::Void), 0);
   489→
   490→    assert_eq!(alignment_of(&IRType::I8), 1);
   491→    assert_eq!(alignment_of(&IRType::I64), 8);
   492→    assert_eq!(alignment_of(&IRType::F32), 4);
   493→
   494→    // --- Argument classification ---
   495→    assert_eq!(classify_arg(&IRType::I64), ArgClass::Integer);
   496→    assert_eq!(classify_arg(&IRType::F64), ArgClass::FP);
   497→    assert_eq!(classify_arg(&IRType::Ptr), ArgClass::Integer);
   498→    assert_eq!(classify_arg(&IRType::Void), ArgClass::Integer);
   499→
   500→    // --- Calling convention: 2 integer args → X0, X1 ---
   501→    let cc = compute_calling_conv(&[IRType::I64, IRType::I64], &IRType::I64);
   502→    assert_eq!(cc.arg_locations.len(), 2);
   503→    assert_eq!(cc.arg_locations[0].register, Some((RegisterClass::X, 0)));
   504→    assert_eq!(cc.arg_locations[1].register, Some((RegisterClass::X, 1)));
   505→    assert_eq!(cc.stack_args_size, 0, "No stack args for 2 register args");
   506→
   507→    // --- Calling convention: 9 integer args → 8 in regs + 1 on stack ---
   508→    let args_9: Vec<IRType> = (0..9).map(|_| IRType::I64).collect();
   509→    let cc9 = compute_calling_conv(&args_9, &IRType::I64);
   510→    assert_eq!(cc9.arg_locations.len(), 9);
   511→    let stack_args: Vec<_> = cc9
   512→        .arg_locations
   513→        .iter()
   514→        .filter(|a| a.register.is_none())
   515→        .collect();
   516→    assert_eq!(stack_args.len(), 1, "9th arg should be on the stack");
   517→    assert!(cc9.stack_args_size > 0, "Should need stack argument space");
   518→
   519→    // --- Return value location ---
   520→    assert_eq!(cc.ret_location.registers, vec![(RegisterClass::X, 0)]);
   521→}

   580→// Test 10: ARM64 instruction encoding correctness
   581→// ---------------------------------------------------------------------------
   582→
   583→/// Test: Validate ARM64 instruction encoding for key instructions.
   584→///
   585→/// Tests that common ARM64 instructions encode to valid 32-bit machine code
   586→/// words and that the encoding produces distinct values for different
   587→/// operations.
   588→#[test]
   589→fn test_arm64_instruction_encoding() {
   590→    // --- ADD X0, X1, X2 ---
   591→    let add = Instruction::ADD {
   592→        rd: Register::X0,
   593→        rn: Register::X1,
   594→        rm: Operand::Reg {
   595→            reg: Register::X2,
   596→            shift: None,
   597→        },
   598→    };
   599→    let add_encoded = add.encode().expect("ADD should encode");
   600→    assert_ne!(add_encoded, 0, "ADD encoding should be non-zero");
   601→
   602→    // --- SUB X0, X1, X2 ---
   603→    let sub = Instruction::SUB {
   604→        rd: Register::X0,
   605→        rn: Register::X1,
   606→        rm: Operand::Reg {
   607→            reg: Register::X2,
   608→            shift: None,
   609→        },
   610→    };
   611→    let sub_encoded = sub.encode().expect("SUB should encode");
   612→    assert_ne!(sub_encoded, 0, "SUB encoding should be non-zero");
   613→    assert_ne!(
   614→        add_encoded, sub_encoded,
   615→        "ADD and SUB should have different encodings"
   616→    );
   617→
   618→    // --- ADD with immediate: ADD X0, X1, #42 ---
   619→    let add_imm = Instruction::ADD {
   620→        rd: Register::X0,
   621→        rn: Register::X1,
   622→        rm: Operand::Imm12(42),
   623→    };
   624→    let add_imm_encoded = add_imm.encode().expect("ADD imm should encode");
   625→    assert_ne!(
   626→        add_imm_encoded, add_encoded,
   627→        "ADD imm should differ from ADD reg"
   628→    );
   629→
   630→    // --- LDR X0, [X1, #0] ---
   631→    let ldr = Instruction::LDR {
   632→        rt: Register::X0,
   633→        rn: Register::X1,
   634→        offset: 0,
   635→    };
   636→    let ldr_encoded = ldr.encode().expect("LDR should encode");
   637→    assert_ne!(ldr_encoded, 0, "LDR encoding should be non-zero");
   638→
   639→    // --- STR X0, [X1, #0] ---
   640→    let str_ = Instruction::STR {
   641→        rt: Register::X0,
   642→        rn: Register::X1,
   643→        offset: 0,
   644→    };
   645→    let str_encoded = str_.encode().expect("STR should encode");
   646→    assert_ne!(
   647→        ldr_encoded, str_encoded,
   648→        "LDR and STR should have different encodings"
   649→    );
   650→
   651→    // --- MOV X0, X1 ---
   652→    let mov = Instruction::MOV {
   653→        rd: Register::X0,
   654→        rm: Register::X1,
   655→    };
   656→    let mov_encoded = mov.encode().expect("MOV should encode");
   657→    assert_ne!(mov_encoded, 0);
   658→
   659→    // --- RET ---
   660→    let ret = Instruction::RET { rn: None };
   661→    let ret_encoded = ret.encode().expect("RET should encode");
   662→    assert_ne!(ret_encoded, 0);
   663→
   664→    // --- NOP ---
   665→    let nop = Instruction::NOP;
   666→    let nop_encoded = nop.encode().expect("NOP should encode");
   667→    assert_ne!(nop_encoded, 0);
   668→
   669→    // --- MOVZ X0, #42 ---
   670→    let movz = Instruction::MOVZ {
   671→        rd: Register::X0,
   672→        imm16: 42,
   673→        shift: 0,
   674→    };
   675→    let movz_encoded = movz.encode().expect("MOVZ should encode");
   676→    assert_ne!(movz_encoded, 0);
   677→
   678→    // --- Barrier instructions ---
   679→    let dmb = Instruction::DMB {
   680→        option: BarrierOption::ISH,
   681→    };
   682→    let dmb_encoded = dmb.encode().expect("DMB should encode");
   683→    assert_ne!(dmb_encoded, 0);
   684→
   685→    let isb = Instruction::ISB;
   686→    let isb_encoded = isb.encode().expect("ISB should encode");
   687→    assert_ne!(isb_encoded, 0);
   688→    assert_ne!(dmb_encoded, isb_encoded);
   689→
   690→    // --- Condition codes ---
   691→    assert_eq!(Condition::EQ.encoding(), 0b0000);
   692→    assert_eq!(Condition::NE.encoding(), 0b0001);
   693→    assert_eq!(Condition::GE.encoding(), 0b1010);
   694→    assert_eq!(Condition::LT.encoding(), 0b1011);
   695→    assert_eq!(Condition::EQ.invert(), Condition::NE);
   696→    assert_eq!(Condition::GE.invert(), Condition::LT);
   697→
   698→    // --- Register encoding ---
   699→    assert_eq!(Register::X0.encoding(), 0);
   700→    assert_eq!(Register::X7.encoding(), 7);
   701→    assert_eq!(Register::SP.encoding(), 31);
   702→    assert_eq!(Register::XZR.encoding(), 31);
   703→    assert!(Register::X19.is_callee_saved());
   704→    assert!(!Register::X9.is_callee_saved());
   705→    assert!(Register::X9.is_caller_saved());
   706→
   707→    // --- Argument register lookup ---
   708→    assert_eq!(Register::arg_register(0), Some(Register::X0));
   709→    assert_eq!(Register::arg_register(7), Some(Register::X7));
   710→    assert_eq!(Register::arg_register(8), None);
   711→    assert_eq!(Register::X3.arg_index(), Some(3));
   712→    assert_eq!(Register::X19.arg_index(), None);
   713→}
