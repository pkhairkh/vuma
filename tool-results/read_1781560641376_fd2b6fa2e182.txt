     1→//! # LoongArch 64-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the LoongArch 64-bit target (LP64 ABI).
     4→//! This module provides:
     5→//!
     6→//! - `Gpr` — General-purpose register enum (r0–r31)
     7→//! - `Fpr` — Floating-point register enum (f0–f31)
     8→//! - `Instruction` — LoongArch64 instruction enum with correct encoding
     9→//! - Encoding helpers for all 9 instruction formats (2R, 3R, 4R, 2RI8, 2RI12,
    10→//!   2RI14, 2RI16, 1RI21, I26)
    11→//! - `LoongArch64Backend` — `Backend` implementation that lowers IR to LoongArch64
    12→//!   machine code and emits ELF64 binaries
    13→//!
    14→//! ## LoongArch64 Register Convention (LP64 ABI)
    15→//!
    16→//! | Register(s) | ABI Name | Role                              |
    17→//! |-------------|----------|-----------------------------------|
    18→//! | r0          | zero     | Hardwired zero                    |
    19→//! | r1          | ra       | Return address                    |
    20→//! | r2          | tp       | Thread pointer                    |
    21→//! | r3          | sp       | Stack pointer                     |
    22→//! | r4–r11      | a0–a7    | Argument / return registers       |
    23→//! | r12–r20     | t0–t8    | Caller-saved temporaries          |
    24→//! | r21         | —        | Reserved                          |
    25→//! | r22         | fp       | Frame pointer (callee-saved)      |
    26→//! | r23–r31     | s0–s8    | Callee-saved                      |
    27→//!
    28→//! ## LoongArch64 FP Register Convention (LP64 ABI)
    29→//!
    30→//! | Register(s) | ABI Name | Role                              |
    31→//! |-------------|----------|-----------------------------------|
    32→//! | f0–f7       | fa0–fa7  | FP argument / return registers    |
    33→//! | f8–f23      | ft0–ft15 | Caller-saved temporaries          |
    34→//! | f24–f31     | fs0–fs7  | Callee-saved                      |
    35→//!
    36→//! ## Instruction Formats
    37→//!
    38→//! All instructions are 32 bits, little-endian, with no branch delay slots.
    39→//! Nine formats: 2R, 3R, 4R, 2RI8, 2RI12, 2RI14, 2RI16, 1RI21, I26.
    40→//!
    41→//! ## References
    42→//!
    43→//! - LoongArch Reference Manual, Volume 1: Basic Architecture
    44→//! - <https://loongson.github.io/LoongArch-Documentation/>
    45→

    58→/// LoongArch64 general-purpose registers (r0–r31).
    59→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]

    61→    R0 = 0,
    62→    Ra = 1,
    63→    Tp = 2,
    64→    Sp = 3,
    65→    A0 = 4,
    66→    A1 = 5,
    67→    A2 = 6,
    68→    A3 = 7,
    69→    A4 = 8,
    70→    A5 = 9,

    75→    T2 = 14,
    76→    T3 = 15,
    77→    T4 = 16,
    78→    T5 = 17,
    79→    T6 = 18,
    80→    T7 = 19,
    81→    T8 = 20,
    82→    R21 = 21,
    83→    Fp = 22,
    84→    S0 = 23,
    85→    S1 = 24,
    86→    S2 = 25,
    87→    S3 = 26,
    88→    S4 = 27,

    95→impl Gpr {
    96→    /// Returns the 5-bit encoding index for this register.
    97→    pub fn encoding(&self) -> u32 {
    98→        *self as u32
    99→    }
   100→
   101→    /// Convert a 5-bit encoding index back to a Gpr variant.
   102→    pub fn from_encoding(enc: u32) -> Gpr {
   103→        match enc {
   104→            0 => Gpr::R0,
   105→            1 => Gpr::Ra,
   106→            2 => Gpr::Tp,
   107→            3 => Gpr::Sp,
   108→            4 => Gpr::A0,
   109→            5 => Gpr::A1,
   110→            6 => Gpr::A2,
   111→            7 => Gpr::A3,
   112→            8 => Gpr::A4,
   113→            9 => Gpr::A5,
   114→            10 => Gpr::A6,
   115→            11 => Gpr::A7,
   116→            12 => Gpr::T0,
   117→            13 => Gpr::T1,
   118→            14 => Gpr::T2,
   119→            15 => Gpr::T3,
   120→            16 => Gpr::T4,
   121→            17 => Gpr::T5,
   122→            18 => Gpr::T6,
   123→            19 => Gpr::T7,
   124→            20 => Gpr::T8,
   125→            21 => Gpr::R21,
   126→            22 => Gpr::Fp,
   127→            23 => Gpr::S0,
   128→            24 => Gpr::S1,
   129→            25 => Gpr::S2,
   130→            26 => Gpr::S3,
   131→            27 => Gpr::S4,
   132→            28 => Gpr::S5,
   133→            29 => Gpr::S6,
   134→            30 => Gpr::S7,
   135→            31 => Gpr::S8,
   136→            _ => Gpr::R0, // fallback: zero register

   138→    }

   140→    /// Returns `true` if this register is available for register allocation.
   141→    ///
   142→    /// R0 (zero), Ra, Tp, and Sp are reserved.
   143→    pub fn is_allocatable(&self) -> bool {
   144→        !matches!(self, Gpr::R0 | Gpr::Ra | Gpr::Tp | Gpr::Sp)
   145→    }
   146→
   147→    /// Returns `true` if this register is callee-saved (fp, s0–s8).
   148→    pub fn is_callee_saved(&self) -> bool {
   149→        matches!(
   150→            self,
   151→            Gpr::Fp
   152→                | Gpr::S0
   153→                | Gpr::S1
   154→                | Gpr::S2
   155→                | Gpr::S3
   156→                | Gpr::S4
   157→                | Gpr::S5
   158→                | Gpr::S6
   159→                | Gpr::S7
   160→                | Gpr::S8
   161→        )
   162→    }
   163→
   164→    /// Returns `true` if this register is an argument register (a0–a7).
   165→    pub fn is_arg_reg(&self) -> bool {
   166→        matches!(
   167→            self,
   168→            Gpr::A0 | Gpr::A1 | Gpr::A2 | Gpr::A3 | Gpr::A4 | Gpr::A5 | Gpr::A6 | Gpr::A7
   169→        )
   170→    }

   182→            Gpr::A3 => "$a3",
   183→            Gpr::A4 => "$a4",
   184→            Gpr::A5 => "$a5",
   185→            Gpr::A6 => "$a6",
   186→            Gpr::A7 => "$a7",
   187→            Gpr::T0 => "$t0",
   188→            Gpr::T1 => "$t1",
   189→            Gpr::T2 => "$t2",
   190→            Gpr::T3 => "$t3",
   191→            Gpr::T4 => "$t4",
   192→            Gpr::T5 => "$t5",
   193→            Gpr::T6 => "$t6",
   194→            Gpr::T7 => "$t7",
   195→            Gpr::T8 => "$t8",
   196→            Gpr::R21 => "$r21",
   197→            Gpr::Fp => "$fp",
   198→            Gpr::S0 => "$s0",
   199→            Gpr::S1 => "$s1",
   200→            Gpr::S2 => "$s2",
   201→            Gpr::S3 => "$s3",
   202→            Gpr::S4 => "$s4",
   203→            Gpr::S5 => "$s5",
   204→            Gpr::S6 => "$s6",
   205→            Gpr::S7 => "$s7",

   212→    pub fn arg_register(index: usize) -> Option<Gpr> {
   213→        match index {
   214→            0 => Some(Gpr::A0),
   215→            1 => Some(Gpr::A1),
   216→            2 => Some(Gpr::A2),
   217→            3 => Some(Gpr::A3),

   362→// ===========================================================================
   363→// Instruction Encoding Helpers
   364→// ===========================================================================
   365→
   366→/// Encode a 2R format instruction.
   367→///
   368→/// Format: `opcode[31:10] | rj[9:5] | rd[4:0]`
   369→fn encode_2r(opcode: u32, rj: u32, rd: u32) -> [u8; 4] {
   370→    let word = ((opcode & 0x3FF_FFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   371→    word.to_le_bytes()
   372→}
   373→
   374→/// Encode a 3R format instruction.
   375→///
   376→/// Format: `opcode[31:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   377→fn encode_3r(opcode: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   378→    let word = ((opcode & 0x1FFFF) << 15) | ((rk & 0x1F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   379→    word.to_le_bytes()
   380→}
   381→
   382→/// Encode a 4R format instruction.
   383→///
   384→/// Format: `opcode[31:20] | ra[19:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   385→#[allow(dead_code)]
   386→fn encode_4r(opcode: u32, ra: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   387→    let word = ((opcode & 0xFFF) << 20)
   388→        | ((ra & 0x1F) << 15)
   389→        | ((rk & 0x1F) << 10)
   390→        | ((rj & 0x1F) << 5)
   391→        | (rd & 0x1F);
   392→    word.to_le_bytes()
   393→}
   394→
   395→/// Encode a 2RI8 format instruction.
   396→///
   397→/// Format: `opcode[31:22] | I8[21:14] | rj[9:5] | rd[4:0]`
   398→#[allow(dead_code)]
   399→fn encode_2ri8(opcode: u32, imm8: u32, rj: u32, rd: u32) -> [u8; 4] {
   400→    let word = ((opcode & 0x3FF) << 22) | ((imm8 & 0xFF) << 14) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   401→    word.to_le_bytes()
   402→}
   403→
   404→/// Encode a reg2i5 format instruction (17-bit opcode).
   405→///
   406→/// Format: `opcode[31:15] | I5[14:10] | rj[9:5] | rd[4:0]`
   407→fn encode_reg2i5(opcode: u32, imm5: u32, rj: u32, rd: u32) -> [u8; 4] {
   408→    let word = ((opcode & 0x1FFFF) << 15) | ((imm5 & 0x1F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   409→    word.to_le_bytes()
   410→}
   411→
   412→/// Encode a reg2i6 format instruction (16-bit opcode).
   413→///
   414→/// Format: `opcode[31:16] | I6[15:10] | rj[9:5] | rd[4:0]`
   415→fn encode_reg2i6(opcode: u32, imm6: u32, rj: u32, rd: u32) -> [u8; 4] {
   416→    let word = ((opcode & 0xFFFF) << 16) | ((imm6 & 0x3F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   417→    word.to_le_bytes()
   418→}
   419→
   420→/// Encode a 2RI12 format instruction.
   421→///
   422→/// Format: `opcode[31:22] | I12[21:10] | rj[9:5] | rd[4:0]`
   423→fn encode_2ri12(opcode: u32, imm12: u32, rj: u32, rd: u32) -> [u8; 4] {
   424→    let word =
   425→        ((opcode & 0x3FF) << 22) | ((imm12 & 0xFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   426→    word.to_le_bytes()
   427→}

   429→/// Encode a 2RI14 format instruction.
   430→///
   431→/// Format: `opcode[31:24] | I14[23:10] | rj[9:5] | rd[4:0]`
   432→fn encode_2ri14(opcode: u32, imm14: u32, rj: u32, rd: u32) -> [u8; 4] {
   433→    let word =
   434→        ((opcode & 0xFF) << 24) | ((imm14 & 0x3FFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   435→    word.to_le_bytes()
   436→}
   437→
   438→/// Encode a 2RI16 format instruction.
   439→///
   440→/// Format: `opcode[31:26] | I16[25:10] | rj[9:5] | rd[4:0]`
   441→fn encode_2ri16(opcode: u32, imm16: u32, rj: u32, rd: u32) -> [u8; 4] {
   442→    let word =
   443→        ((opcode & 0x3F) << 26) | ((imm16 & 0xFFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   444→    word.to_le_bytes()
   445→}
   446→
   447→/// Encode a 1RI21 format instruction (used for BEQZ, BNEZ).
   448→///
   449→/// Format: `opcode[31:26] | offs21[15:0] in bits[25:10] | rj[9:5] | offs21[20:16] in bits[4:0]`
   450→///
   451→/// Note: the offset bits are split non-linearly — the lower 16 bits go in the
   452→/// higher position (bits[25:10]), and the upper 5 bits go in the lower position
   453→/// (bits[4:0]). The register field `rj` sits between them at bits[9:5].
   454→fn encode_1ri21(opcode: u32, imm21: u32, rj: u32) -> [u8; 4] {
   455→    // 1RI21 format (BEQZ/BNEZ): opcode[31:26] | offs[15:0] at [25:10] | rj[9:5] | offs[20:16] at [4:0]
   456→    let word = ((opcode & 0x3F) << 26)
   457→        | ((imm21 & 0xFFFF) << 10)       // offs[15:0] at bits 25:10
   458→        | ((rj & 0x1F) << 5)             // rj at bits 9:5
   459→        | ((imm21 >> 16) & 0x1F);        // offs[20:16] at bits 4:0
   460→    word.to_le_bytes()
   461→}
   462→
   463→/// Encode an I26 format instruction (used for B, BL).
   464→///
   465→/// Format: `opcode[31:26] | offs26[15:0] in bits[25:10] | offs26[25:16] in bits[9:0]`
   466→///
   467→/// Note: the offset bits are SWAPPED compared to a linear layout.
   468→/// The lower 16 bits of the offset go in the higher position (bits[25:10]),
   469→/// and the upper 10 bits go in the lower position (bits[9:0]).
   470→fn encode_i26(opcode: u32, imm26: u32) -> [u8; 4] {
   471→    let word = ((opcode & 0x3F) << 26)
   472→        | ((imm26 & 0xFFFF) << 10)
   473→        | ((imm26 >> 16) & 0x3FF);
   474→    word.to_le_bytes()
   475→}

  1347→                encode_3r(OPC_FDIV_S, fk.encoding(), fj.encoding(), fd.encoding())
  1348→            }
  1349→            Instruction::FdivD { fd, fj, fk } => {
  1350→                encode_3r(OPC_FDIV_D, fk.encoding(), fj.encoding(), fd.encoding())
  1351→            }
  1352→
  1353→            // ── FP Move (2R) ──────────────────────────────────────
  1354→            Instruction::FmovS { fd, fj } => encode_2r(OPC_FMOV_S, fj.encoding(), fd.encoding()),
  1355→            Instruction::FmovD { fd, fj } => encode_2r(OPC_FMOV_D, fj.encoding(), fd.encoding()),
  1356→
  1357→            // ── FP Compare (4R-like) ──────────────────────────────
  1358→            Instruction::FCmpS { cond, fj, fk, cd } => encode_4r(
  1359→                OPC_FCMP_S,
  1360→                (*cond & 0x1F) as u32,
  1361→                fk.encoding(),
  1362→                fj.encoding(),
  1363→                (*cd & 0x1F) as u32,
  1364→            ),
  1365→            Instruction::FCmpD { cond, fj, fk, cd } => encode_4r(
  1366→                OPC_FCMP_D,
  1367→                (*cond & 0x1F) as u32,
  1368→                fk.encoding(),
  1369→                fj.encoding(),
  1370→                (*cd & 0x1F) as u32,
  1371→            ),
  1372→
  1373→            // ── No-op / Break ─────────────────────────────────────
  1374→            Instruction::Nop => {
  1375→                // NOP pseudo: and $r0, $r0, $r0
  1376→                encode_3r(
  1377→                    OPC_AND,
  1378→                    Gpr::R0.encoding(),
  1379→                    Gpr::R0.encoding(),
  1380→                    Gpr::R0.encoding(),
  1381→                )
  1382→            }
  1383→            Instruction::Syscall => {
  1384→                // SYSCALL = 0x002B0000 (with code=0)
  1385→                0x002B0000u32.to_le_bytes()
  1386→            }
  1387→            Instruction::Break => {
  1388→                // BREAK = 0x002A0000 (with code=0)
  1389→                0x002A0000u32.to_le_bytes()
  1390→            }
  1391→        }
  1392→    }

  1650→/// Build a minimal ELF64 binary for LoongArch64 from raw code bytes.
  1651→///
  1652→/// Produces a static executable with two LOAD segments:
  1653→///   1. PF_R | PF_X — .text (code)
  1654→///   2. PF_R | PF_W — .data / stack (writable memory)
  1655→fn build_loongarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1656→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (LoongArch64 typical page size)
  1657→
  1658→    let elf_header_size: u64 = 64;
  1659→    let phdr_size: u64 = 56;
  1660→    let num_phdrs: u64 = 2;
  1661→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1662→    // Page-align the text segment start in the file.
  1663→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1664→    let text_size = code.len() as u64;
  1665→
  1666→    // The data segment starts on the next page after the text.

  1733→    elf.extend_from_slice(&vec![0u8; data_size as usize]);

  1738→/// Patch a 4-instruction load-immediate sequence in `code` starting at `offset`
  1739→/// with the 64-bit value `val`.
  1740→///
  1741→/// The sequence is:
  1742→///   lu12i.w rd, bits[31:12]   — sets bits[31:12] and sign-extends to 64 bits
  1743→///   ori     rd, rd, bits[11:0] — sets bits[11:0]
  1744→///   lu32i.d rd, bits[51:32]   — sets bits[51:32]
  1745→///   lu52i.d rd, rd, bits[63:52] — sets bits[63:52]
  1746→///
  1747→/// The `rd` register is extracted from the existing first instruction.
  1748→fn patch_load_imm_64(code: &mut [u8], offset: usize, val: u64) {
  1749→    // Extract rd from the first instruction (lu12i.w): rd is at bits[4:0]
  1750→    let word0 = u32::from_le_bytes([code[offset], code[offset + 1], code[offset + 2], code[offset + 3]]);
  1751→    let rd_enc = (word0 & 0x1F) as u32;
  1752→    let rd = Gpr::from_encoding(rd_enc);
  1753→
  1754→    // Re-encode all 4 instructions with the new value
  1755→
  1756→    // Step 1: lu12i.w rd, bits[31:12]
  1757→    let hi20 = ((val >> 12) & 0xFFFFF) as i32;
  1758→    let new_word0 = u32::from_le_bytes(Instruction::Lu12iW { rd, imm20: hi20 }.encode());
  1759→    code[offset..offset + 4].copy_from_slice(&new_word0.to_le_bytes());
  1760→
  1761→    // Step 2: ori rd, rd, bits[11:0]
  1762→    let lo12 = (val & 0xFFF) as u32;
  1763→    let new_word1 = u32::from_le_bytes(Instruction::Ori { rd, rj: rd, imm12: lo12 }.encode());
  1764→    code[offset + 4..offset + 8].copy_from_slice(&new_word1.to_le_bytes());
  1765→
  1766→    // Step 3: lu32i.d rd, bits[51:32]
  1767→    let hi32 = ((val >> 32) & 0xFFFFF) as i32;
  1768→    let new_word2 = u32::from_le_bytes(Instruction::Lu32iD { rd, imm20: hi32 }.encode());
  1769→    code[offset + 8..offset + 12].copy_from_slice(&new_word2.to_le_bytes());
  1770→
  1771→    // Step 4: lu52i.d rd, rd, bits[63:52]
  1772→    let hi52 = ((val >> 52) & 0xFFF) as i32;
  1773→    let new_word3 = u32::from_le_bytes(Instruction::Lu52iD { rd, rj: rd, imm12: hi52 }.encode());
  1774→    code[offset + 12..offset + 16].copy_from_slice(&new_word3.to_le_bytes());
  1775→}

  1781→/// Decode a single LoongArch64 32-bit instruction word into a mnemonic string.
  1782→fn decode_loongarch64_instruction(word: u32) -> String {

  1790→        }

  1792→            let rd = (word >> 7) & 0x1f;
  1793→            let rj = (word >> 15) & 0x1f;

  1796→        }

  1872→/// Compute the stack frame size for an IR function on LoongArch64.

  1874→/// Sums `Alloc` instruction sizes, adds 16 bytes for the ra/fp save pair,
  1875→/// and rounds up to 16-byte alignment.
  1876→fn loongarch64_compute_frame_size(func: &IRFunction) -> usize {

  1878→    for block in &func.blocks {

  1940→        if let Some(reg) = Gpr::arg_register(idx) {
