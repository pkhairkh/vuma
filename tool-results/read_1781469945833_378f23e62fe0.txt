  2000→            });
  2001→            return result;
  2002→        }
  2003→        BinOpKind::Ne => {
  2004→            // xor rd, lhs, rhs ; sltu rd, $zero, rd (rd = (0 < xor_result))
  2005→            result.push(AllocatedInstruction {
  2006→                opcode: "xor".to_string(),
  2007→                reads: vec![
  2008→                    PhysicalReg::new(RegClass::Gpr, lhs_reg.encoding()),
  2009→                    PhysicalReg::new(RegClass::Gpr, rhs_reg.encoding()),
  2010→                ],
  2011→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())],
  2012→                encoded: Instruction::Xor {
  2013→                    rd: dst_reg,
  2014→                    rs: lhs_reg,
  2015→                    rt: rhs_reg,
  2016→                }
  2017→                .encode()
  2018→                .to_vec(),
  2019→            });
  2020→            result.push(AllocatedInstruction {
  2021→                opcode: "sltu".to_string(),
  2022→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())],
  2023→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())],
  2024→                encoded: Instruction::Sltu {
  2025→                    rd: dst_reg,
  2026→                    rs: Gpr::Zero,
  2027→                    rt: dst_reg,
  2028→                }
  2029→                .encode()
  2030→                .to_vec(),
  2031→            });
  2032→            return result;
  2033→        }
  2034→    };
  2035→
  2036→    let encoded = mips_inst.encode();
  2037→    let reads = vec![
  2038→        PhysicalReg::new(RegClass::Gpr, lhs_reg.encoding()),
  2039→        PhysicalReg::new(RegClass::Gpr, rhs_reg.encoding()),
  2040→    ];
  2041→    let writes = vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())];
  2042→
  2043→    // mul/div write to HI/LO, then we need mflo (and possibly mfhi for rem)
  2044→    if matches!(
  2045→        op,
  2046→        BinOpKind::Mul | BinOpKind::SDiv | BinOpKind::UDiv | BinOpKind::SRem | BinOpKind::URem
  2047→    ) {
  2048→        result.push(AllocatedInstruction {
  2049→            opcode: mips_inst.mnemonic().to_string(),
  2050→            reads,
  2051→            writes: vec![
  2052→                PhysicalReg::new(RegClass::Special, 0), // HI
  2053→                PhysicalReg::new(RegClass::Special, 1), // LO
  2054→            ],
  2055→            encoded: encoded.to_vec(),
  2056→        });
  2057→        if matches!(op, BinOpKind::SRem | BinOpKind::URem) {
  2058→            // mfhi to get remainder
  2059→            let mfhi = Instruction::Mfhi { rd: dst_reg };
  2060→            result.push(AllocatedInstruction {
  2061→                opcode: "mfhi".to_string(),
  2062→                reads: vec![PhysicalReg::new(RegClass::Special, 0)],
  2063→                writes: writes.clone(),
  2064→                encoded: mfhi.encode().to_vec(),
  2065→            });
  2066→        } else {
  2067→            // mflo to get quotient/product
  2068→            let mflo = Instruction::Mflo { rd: dst_reg };
  2069→            result.push(AllocatedInstruction {
  2070→                opcode: "mflo".to_string(),
  2071→                reads: vec![PhysicalReg::new(RegClass::Special, 1)],
  2072→                writes: writes.clone(),
  2073→                encoded: mflo.encode().to_vec(),
  2074→            });
  2075→        }
  2076→        return result;
  2077→    }
  2078→
  2079→    result.push(AllocatedInstruction {
  2080→        opcode: mips_inst.mnemonic().to_string(),
  2081→        reads,
  2082→        writes,
  2083→        encoded: encoded.to_vec(),
  2084→    });
  2085→    result
  2086→}

  2088→/// Lower an IR instruction to a sequence of MIPS64 `AllocatedInstruction`s,
  2089→/// handling branch delay slots by inserting NOPs after branches/jumps.
  2090→///
  2091→/// `frame_size` is the stack frame size (0 if not yet known), used by the
  2092→/// `Ret` handler to generate a proper epilogue that restores $ra and
  2093→/// deallocates the frame.
  2094→// ===========================================================================
  2095→// Stack-slot based register allocation for MIPS64
  2096→// ===========================================================================
  2097→
  2098→/// Stack-slot based allocate_registers for MIPS64.
  2099→/// Every vreg gets a stack slot; operations use scratch registers $t0-$t7.
  2100→fn mips64_allocate_registers_ss(func: &IRFunction) -> Result<AllocatedFunction, BackendError> {
  2101→    let func_name = func.name.clone();
  2102→
  2103→    // ── Phase 1: Collect all vreg IDs and compute stack layout ──
  2104→    let mut all_vreg_ids: std::collections::HashSet<u32> = std::collections::HashSet::new();
  2105→    for &id in func.vregs.keys() { all_vreg_ids.insert(id); }
  2106→    for param in &func.params {
  2107→        if let Some(id) = param.as_register() { all_vreg_ids.insert(id); }
  2108→    }
  2109→    for block in &func.blocks {
  2110→        for instr in &block.instructions {
  2111→            for id in instr.defined_regs() { all_vreg_ids.insert(id); }
  2112→            for id in instr.used_regs() { all_vreg_ids.insert(id); }
  2113→        }
  2114→        match &block.terminator {
  2115→            crate::ir::IRTerminator::Branch { cond, .. } => {
  2116→                if let Some(id) = cond.as_register() { all_vreg_ids.insert(id); }
  2117→            }
  2118→            crate::ir::IRTerminator::Return(vals) => {
  2119→                for val in vals { if let Some(id) = val.as_register() { all_vreg_ids.insert(id); } }
  2120→            }
  2121→            _ => {}
  2122→        }
  2123→    }
  2124→    for val in &func.results {
  2125→        if let Some(id) = val.as_register() { all_vreg_ids.insert(id); }
  2126→    }
  2127→
  2128→    // Identify Alloc vregs and their sizes
  2129→    let mut stack_alloc_vregs: std::collections::HashSet<u32> = std::collections::HashSet::new();
  2130→    let mut alloc_sizes: HashMap<u32, i32> = HashMap::new();
  2131→    for block in &func.blocks {
  2132→        for instr in &block.instructions {
  2133→            if let IRInstr::Alloc { dst, size } = instr {
  2134→                if let Some(id) = dst.as_register() {
  2135→                    stack_alloc_vregs.insert(id);
  2136→                    let aligned_size = ((*size as i32 + 15) & !15) as i32;
  2137→                    alloc_sizes.insert(id, aligned_size);
  2138→                }
  2139→            }
  2140→        }
  2141→    }
  2142→
  2143→    // ── Stack Layout ──
  2144→    // [high address]  = $fp = $sp + frame_size (after prologue)
  2145→    //   saved $ra     ← $fp - 8
  2146→    //   saved $fp     ← $fp - 16
  2147→    //   Alloc data N  ← $fp - (24 + vreg_total + alloc_N)
  2148→    //   ...
  2149→    //   Alloc data 1
  2150→    //   vreg slot M   ← $fp - (24 + 8*(M-1))
  2151→    //   ...
  2152→    //   vreg slot 1   ← $fp - 24
  2153→    // [low address]   = $sp
  2154→
  2155→    let mut alloc_offsets: HashMap<u32, i32> = HashMap::new();
  2156→    let mut current_offset: i32 = 24; // skip $ra(8) + $fp(8) + first_slot_gap(8)
  2157→    // Actually: $ra at fp-8, $fp_old at fp-16, then vreg slots start at fp-24
  2158→    // But alloc regions need to be after vreg slots.
  2159→    // Let's do: vreg slots first (at smaller negative offsets), then alloc regions (at larger offsets)
  2160→
  2161→    let mut vreg_stack_slots: HashMap<u32, i32> = HashMap::new();
  2162→    let mut all_vreg_ids_sorted: Vec<u32> = all_vreg_ids.iter().copied().collect();
  2163→    all_vreg_ids_sorted.sort();
  2164→    for &id in &all_vreg_ids_sorted {
  2165→        vreg_stack_slots.insert(id, current_offset);
  2166→        current_offset += 8;
  2167→    }
  2168→
  2169→    // Alloc regions after vreg slots
  2170→    let mut alloc_vreg_ids: Vec<u32> = stack_alloc_vregs.iter().copied().collect();
  2171→    alloc_vreg_ids.sort();
  2172→    for &id in &alloc_vreg_ids {
  2173→        let size = alloc_sizes[&id];
  2174→        current_offset += size;
  2175→        alloc_offsets.insert(id, current_offset);
  2176→    }
  2177→
  2178→    let frame_size = ((current_offset + 15) & !15) as usize;
  2179→
  2180→    // ── Phase 2: Helper functions ──
  2181→
  2182→    // Emit code to load a 64-bit immediate into a register
  2183→    // CRITICAL: In MIPS64, LUI sign-extends the result! So LUI with 0xFFFF
  2184→    // produces 0xFFFFFFFFFFFF0000, NOT 0x00000000FFFF0000. We must handle
  2185→    // this correctly for 32-bit values where the sign bit is set.
  2186→    fn ss_load_imm(dst: Gpr, val: i64) -> Vec<u8> {
  2187→        let mut code = Vec::new();
  2188→        if (-32768..=32767).contains(&val) {
  2189→            // Fits in 16-bit signed immediate
  2190→            code.extend_from_slice(&Instruction::Daddiu { rt: dst, rs: Gpr::Zero, imm: val as i32 }.encode());
  2191→        } else if val >= 0 && val <= 0xFFFF {
  2192→            // 16-bit unsigned: use ORI (zero-extends)
  2193→            code.extend_from_slice(&Instruction::Ori { rt: dst, rs: Gpr::Zero, imm: val as u32 }.encode());
  2194→        } else if val >= 0 && val <= 0xFFFFFFFF {
  2195→            // 32-bit unsigned value: must ensure zero-extension in upper 32 bits.
  2196→            // LUI sign-extends in MIPS64, so we must use LUI+ORI and then
  2197→            // DSLL32+DSRL32 to clear the upper bits if the sign bit is set.
  2198→            let upper = ((val as u32) >> 16) & 0xFFFF;
  2199→            let lower = (val as u32) & 0xFFFF;
  2200→            code.extend_from_slice(&Instruction::Lui { rt: dst, imm: upper }.encode());
  2201→            if lower != 0 {
  2202→                code.extend_from_slice(&Instruction::Ori { rt: dst, rs: dst, imm: lower }.encode());
  2203→            }
  2204→            // If the 32-bit value has bit 31 set, LUI sign-extended it.
  2205→            // Zero-extend by shifting left 32 then right 32.
  2206→            if (val as u32) >= 0x80000000 {
  2207→                code.extend_from_slice(&Instruction::Dsll { rd: dst, rt: dst, sa: 32 }.encode());
  2208→                code.extend_from_slice(&Instruction::Dsrl { rd: dst, rt: dst, sa: 32 }.encode());
  2209→            }
  2210→        } else {
  2211→            // Full 64-bit: lui + ori + dsll + ori + dsll + ori
  2212→            let bits63_48 = ((val as u64) >> 48) as u32 & 0xFFFF;
  2213→            let bits47_32 = ((val as u64) >> 32) as u32 & 0xFFFF;
  2214→            let bits31_16 = ((val as u64) >> 16) as u32 & 0xFFFF;
  2215→            let bits15_0 = (val as u64) as u32 & 0xFFFF;
  2216→            code.extend_from_slice(&Instruction::Lui { rt: dst, imm: bits63_48 }.encode());
  2217→            code.extend_from_slice(&Instruction::Ori { rt: dst, rs: dst, imm: bits47_32 }.encode());
  2218→            code.extend_from_slice(&Instruction::Dsll { rd: dst, rt: dst, sa: 16 }.encode());
  2219→            code.extend_from_slice(&Instruction::Ori { rt: dst, rs: dst, imm: bits31_16 }.encode());
  2220→            code.extend_from_slice(&Instruction::Dsll { rd: dst, rt: dst, sa: 16 }.encode());
  2221→            code.extend_from_slice(&Instruction::Ori { rt: dst, rs: dst, imm: bits15_0 }.encode());
  2222→        }
  2223→        code
  2224→    }
  2225→
  2226→    // Load from stack slot at [$fp - offset] into dst register
  2227→    fn ss_ld(dst: Gpr, offset: i32) -> Vec<u8> {
  2228→        let mut code = Vec::new();
  2229→        if (-32768..=32767).contains(&(-offset)) {
  2230→            code.extend_from_slice(&Instruction::Ld { rt: dst, base: Gpr::Fp, offset: -offset }.encode());
  2231→            code.extend_from_slice(&encode_nop());
  2232→        } else {
  2233→            // Large offset: compute address into $t3
  2234→            code.extend(ss_load_imm(Gpr::T3, offset as i64));
  2235→            code.extend_from_slice(&Instruction::Dsubu { rd: Gpr::T3, rs: Gpr::Fp, rt: Gpr::T3 }.encode());
  2236→            code.extend_from_slice(&encode_nop());
  2237→            code.extend_from_slice(&Instruction::Ld { rt: dst, base: Gpr::T3, offset: 0 }.encode());
  2238→            code.extend_from_slice(&encode_nop());
  2239→        }
  2240→        code
  2241→    }
  2242→
  2243→    // Store to stack slot at [$fp - offset] from src register
  2244→    fn ss_sd(src: Gpr, offset: i32) -> Vec<u8> {
  2245→        let mut code = Vec::new();
  2246→        if (-32768..=32767).contains(&(-offset)) {
  2247→            code.extend_from_slice(&Instruction::Sd { rt: src, base: Gpr::Fp, offset: -offset }.encode());
  2248→            code.extend_from_slice(&encode_nop());
  2249→        } else {
  2250→            code.extend(ss_load_imm(Gpr::T3, offset as i64));
  2251→            code.extend_from_slice(&Instruction::Dsubu { rd: Gpr::T3, rs: Gpr::Fp, rt: Gpr::T3 }.encode());
  2252→            code.extend_from_slice(&encode_nop());
  2253→            code.extend_from_slice(&Instruction::Sd { rt: src, base: Gpr::T3, offset: 0 }.encode());
  2254→            code.extend_from_slice(&encode_nop());
  2255→        }
  2256→        code
  2257→    }
  2258→
  2259→    // Load a typed value from stack slot
  2260→    fn ss_ld_typed(dst: Gpr, offset: i32, ty: &IRType) -> Vec<u8> {
  2261→        let mut code = Vec::new();
  2262→        let neg_off = -offset;
  2263→        if (-32768..=32767).contains(&neg_off) {
  2264→            match ty {
  2265→                IRType::I8 => { code.extend_from_slice(&Instruction::Lb { rt: dst, base: Gpr::Fp, offset: neg_off }.encode()); code.extend_from_slice(&encode_nop()); }
  2266→                IRType::U8 => { code.extend_from_slice(&Instruction::Lbu { rt: dst, base: Gpr::Fp, offset: neg_off }.encode()); code.extend_from_slice(&encode_nop()); }
  2267→                IRType::I32 | IRType::U32 => { code.extend_from_slice(&Instruction::Lwu { rt: dst, base: Gpr::Fp, offset: neg_off }.encode()); code.extend_from_slice(&encode_nop()); }
  2268→                _ => { code.extend_from_slice(&Instruction::Ld { rt: dst, base: Gpr::Fp, offset: neg_off }.encode()); code.extend_from_slice(&encode_nop()); }
  2269→            }
  2270→        } else {
  2271→            code.extend(ss_load_imm(Gpr::T3, offset as i64));
  2272→            code.extend_from_slice(&Instruction::Dsubu { rd: Gpr::T3, rs: Gpr::Fp, rt: Gpr::T3 }.encode());
  2273→            code.extend_from_slice(&encode_nop());
  2274→            match ty {
  2275→                IRType::I8 => { code.extend_from_slice(&Instruction::Lb { rt: dst, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2276→                IRType::U8 => { code.extend_from_slice(&Instruction::Lbu { rt: dst, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2277→                IRType::I32 | IRType::U32 => { code.extend_from_slice(&Instruction::Lwu { rt: dst, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2278→                _ => { code.extend_from_slice(&Instruction::Ld { rt: dst, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2279→            }
  2280→        }
  2281→        code
  2282→    }
  2283→
  2284→    // Store a typed value to stack slot
  2285→    fn ss_sd_typed(src: Gpr, offset: i32, ty: &IRType) -> Vec<u8> {
  2286→        let mut code = Vec::new();
  2287→        let neg_off = -offset;
  2288→        if (-32768..=32767).contains(&neg_off) {
  2289→            match ty {
  2290→                IRType::I8 | IRType::U8 => { code.extend_from_slice(&Instruction::Sb { rt: src, base: Gpr::Fp, offset: neg_off }.encode()); code.extend_from_slice(&encode_nop()); }
  2291→                IRType::I32 | IRType::U32 => { code.extend_from_slice(&Instruction::Sw { rt: src, base: Gpr::Fp, offset: neg_off }.encode()); code.extend_from_slice(&encode_nop()); }
  2292→                _ => { code.extend_from_slice(&Instruction::Sd { rt: src, base: Gpr::Fp, offset: neg_off }.encode()); code.extend_from_slice(&encode_nop()); }
  2293→            }
  2294→        } else {
  2295→            code.extend(ss_load_imm(Gpr::T3, offset as i64));
  2296→            code.extend_from_slice(&Instruction::Dsubu { rd: Gpr::T3, rs: Gpr::Fp, rt: Gpr::T3 }.encode());
  2297→            code.extend_from_slice(&encode_nop());
  2298→            match ty {
  2299→                IRType::I8 | IRType::U8 => { code.extend_from_slice(&Instruction::Sb { rt: src, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2300→                IRType::I32 | IRType::U32 => { code.extend_from_slice(&Instruction::Sw { rt: src, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2301→                _ => { code.extend_from_slice(&Instruction::Sd { rt: src, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2302→            }
  2303→        }
  2304→        code
  2305→    }
  2306→
  2307→    // Load an IRValue into a scratch register
  2308→    fn ss_load_value(val: &IRValue, slots: &HashMap<u32, i32>, scratch: Gpr) -> Vec<u8> {
  2309→        match val {
  2310→            IRValue::Register(id) => {
  2311→                let offset = slots.get(id).copied().unwrap_or(0);
  2312→                ss_ld(scratch, offset)
  2313→            }
  2314→            IRValue::Immediate(v) => ss_load_imm(scratch, *v),
  2315→            IRValue::Address(a) => ss_load_imm(scratch, *a as i64),
  2316→            IRValue::Label(_) => ss_load_imm(scratch, 0),
  2317→        }
  2318→    }
  2319→
  2320→    // ── Phase 3: Emit prologue ──
  2321→    let mut code: Vec<u8> = Vec::new();
  2322→
  2323→    // DADDIU $sp, $sp, -frame_size
  2324→    code.extend_from_slice(&Instruction::Daddiu { rt: Gpr::Sp, rs: Gpr::Sp, imm: -(frame_size as i32) }.encode());
  2325→    code.extend_from_slice(&encode_nop()); // delay slot (safe: SP update is committed)
  2326→    // SD $ra, frame_size-8($sp)
  2327→    code.extend_from_slice(&Instruction::Sd { rt: Gpr::Ra, base: Gpr::Sp, offset: (frame_size - 8) as i32 }.encode());
  2328→    code.extend_from_slice(&encode_nop());
  2329→    // SD $fp, frame_size-16($sp)
  2330→    code.extend_from_slice(&Instruction::Sd { rt: Gpr::Fp, base: Gpr::Sp, offset: (frame_size - 16) as i32 }.encode());
  2331→    code.extend_from_slice(&encode_nop());
  2332→    // DADDIU $fp, $sp, frame_size  (frame pointer = top of frame)
  2333→    code.extend_from_slice(&Instruction::Daddiu { rt: Gpr::Fp, rs: Gpr::Sp, imm: frame_size as i32 }.encode());
  2334→    code.extend_from_slice(&encode_nop());
  2335→
  2336→    // Store function parameters from $a0-$a3 to their stack slots
  2337→    let arg_regs = [Gpr::A0, Gpr::A1, Gpr::A2, Gpr::A3];
  2338→    for (i, param) in func.params.iter().enumerate() {
  2339→        if let Some(id) = param.as_register() {
  2340→            if i < 4 {
  2341→                let offset = vreg_stack_slots.get(&id).copied().unwrap_or(0);
  2342→                code.extend(ss_sd(arg_regs[i], offset));
  2343→            }
  2344→        }
  2345→    }
  2346→
  2347→    // ── Phase 4: Emit each basic block ──
  2348→    let mut label_offsets: HashMap<String, usize> = HashMap::new();
  2349→    let mut branch_fixups: Vec<(usize, String, i32)> = Vec::new(); // (code_offset, label, adjustment)
  2350→    let mut relocations: Vec<RelocationEntry> = Vec::new();
  2351→
  2352→    for block in &func.blocks {
  2353→        label_offsets.insert(block.label.clone(), code.len());
  2354→
  2355→        for instr in &block.instructions {
  2356→            match instr {
  2357→                // ── BinOp / Add / Sub / Mul / Div ──
  2358→                IRInstr::BinOp { op, dst, lhs, rhs, ty } => {
  2359→                    let actual_op = *op;
  2360→                    let dst_id = dst.as_register().unwrap_or(0);
  2361→                    let dst_off = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
  2362→                    code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::T0));
  2363→                    code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::T1));
  2364→
  2365→                    let is_32bit = ty.as_ref().map_or(false, |t| matches!(t, IRType::I32 | IRType::U32));
  2366→
  2367→                    match actual_op {
  2368→                        BinOpKind::Add => {
  2369→                            code.extend_from_slice(&Instruction::Daddu { rd: Gpr::T0, rs: Gpr::T0, rt: Gpr::T1 }.encode());
  2370→                            code.extend_from_slice(&Instruction::Dsll { rd: Gpr::T0, rt: Gpr::T0, sa: 32 }.encode());
  2371→                            code.extend_from_slice(&Instruction::Dsrl { rd: Gpr::T0, rt: Gpr::T0, sa: 32 }.encode());
  2372→                        }
  2373→                        BinOpKind::Sub => {
  2374→                            code.extend_from_slice(&Instruction::Dsubu { rd: Gpr::T0, rs: Gpr::T0, rt: Gpr::T1 }.encode());
  2375→                            code.extend_from_slice(&Instruction::Dsll { rd: Gpr::T0, rt: Gpr::T0, sa: 32 }.encode());
  2376→                            code.extend_from_slice(&Instruction::Dsrl { rd: Gpr::T0, rt: Gpr::T0, sa: 32 }.encode());
  2377→                        }
  2378→                        BinOpKind::Mul | BinOpKind::SDiv | BinOpKind::UDiv |
  2379→                        BinOpKind::SRem | BinOpKind::URem => {
  2380→                            if is_32bit {
