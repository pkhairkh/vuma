     1→     1→//! # PowerPC64 Backend
     2→     2→//!
     3→     3→//! Implements the `Backend` trait for the PowerPC 64-bit target (ELFv2 ABI,
     4→     4→//! little-endian default). This module provides:
     5→     5→//!
     6→     6→//! - `Gpr` — General-purpose register enum (R0–R31)
     7→     7→//! - `Fpr` — Floating-point register enum (F0–F31)
     8→     8→//! - `CrField` — Condition register field enum (CR0–CR7)
     9→     9→//! - `Instruction` — PPC64 instruction enum with correct 32-bit encoding
    10→    10→//! - `PPC64Backend` — `Backend` implementation that lowers IR to PPC64
    11→    11→//!   machine code and emits ELF64 binaries
    12→    12→//!
    13→    13→//! ## PowerPC64 Register Convention (ELFv2 ABI)
    14→    14→//!
    15→    15→//! | Register(s) | Role                                  |
    16→    16→//! |-------------|---------------------------------------|
    17→    17→//! | R0          | Volatile (not hardwired zero!)        |
    18→    18→//! | R1          | Stack pointer (SP)                    |
    19→    19→//! | R2          | TOC pointer                           |
    20→    20→//! | R3–R10      | Argument / return registers           |
    21→    21→//! | R11–R12     | Volatile scratch                      |
    22→    22→//! | R13         | Thread pointer                        |
    23→    23→//! | R14–R31     | Callee-saved                          |
    24→    24→//! | F0–F13      | FP argument / return, volatile        |
    25→    25→//! | F14–F31     | FP callee-saved                       |
    26→    26→//! | CR0–CR7     | Condition register fields (4 bits ea) |
    27→    27→//! | LR          | Link register (SPR)                   |
    28→    28→//! | CTR         | Count register (SPR)                  |
    29→    29→//!
    30→    30→//! ## Instruction Encoding
    31→    31→//!
    32→    32→//! All instructions are 32 bits, fixed-width. PPC64 is bi-endian; the default
    33→    33→//! for ppc64le is little-endian byte order for both data and instructions.
    34→    34→//! The primary opcode occupies bits \[0:5\] (MSB-first bit numbering).
    35→    35→//!
    36→    36→//! ## References
    37→    37→//!
    38→    38→//! - Power ISA Version 3.1
    39→    39→//! - <https://openpowerfoundation.org/specifications/isa/>
    40→    40→

   419→   419→// Instruction Encoding Helpers
   420→   420→// ===========================================================================
   421→   421→
   422→   422→/// Encode a PPC64 32-bit instruction word and return as little-endian bytes.
   423→   423→fn encode_word(word: u32) -> [u8; 4] {
   424→   424→    word.to_le_bytes()
   425→   425→}
   426→   426→
   427→   427→/// Build a D-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | d[16:31]
   428→   428→fn encode_d_form(opcode: u32, rt: u32, ra: u32, d: i32) -> [u8; 4] {
   429→   429→    let word =
   430→   430→        ((opcode & 0x3F) << 26) | ((rt & 0x1F) << 21) | ((ra & 0x1F) << 16) | ((d as u32) & 0xFFFF);
   431→   431→    encode_word(word)
   432→   432→}
   433→   433→
   434→   434→/// Build a DS-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | ds[16:29] | xo[30:31]
   435→   435→fn encode_ds_form(opcode: u32, rt: u32, ra: u32, ds: i32, xo: u32) -> [u8; 4] {
   436→   436→    let word = ((opcode & 0x3F) << 26)
   437→   437→        | ((rt & 0x1F) << 21)
   438→   438→        | ((ra & 0x1F) << 16)
   439→   439→        | (((ds as u32) & 0x3FFC) << 2)
   440→   440→        | (xo & 0x3);
   441→   441→    encode_word(word)
   442→   442→}
   443→   443→
   444→   444→/// Build an X-form instruction: opcode[0:5] | rS[6:10] | rA[11:15] | rB[16:20] | xo[21:30] | Rc[31]
   445→   445→fn encode_x_form(opcode: u32, rs: u32, ra: u32, rb: u32, xo: u32, rc: u32) -> [u8; 4] {
   446→   446→    let word = ((opcode & 0x3F) << 26)
   447→   447→        | ((rs & 0x1F) << 21)
   448→   448→        | ((ra & 0x1F) << 16)
   449→   449→        | ((rb & 0x1F) << 11)
   450→   450→        | ((xo & 0x3FF) << 1)
   451→   451→        | (rc & 1);
   452→   452→    encode_word(word)
   453→   453→}
   454→   454→
   455→   455→/// Build an XO-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | rB[16:20] | OE[21] | xo[22:30] | Rc[31]
   456→   456→fn encode_xo_form(opcode: u32, rt: u32, ra: u32, rb: u32, oe: u32, xo: u32, rc: u32) -> [u8; 4] {
   457→   457→    let word = ((opcode & 0x3F) << 26)
   458→   458→        | ((rt & 0x1F) << 21)
   459→   459→        | ((ra & 0x1F) << 16)
   460→   460→        | ((rb & 0x1F) << 11)
   461→   461→        | ((oe & 1) << 10)
   462→   462→        | ((xo & 0x1FF) << 1)
   463→   463→        | (rc & 1);
   464→   464→    encode_word(word)
   465→   465→}
   466→   466→
   467→   467→/// Build an M-form instruction: opcode[0:5] | rS[6:10] | rA[11:15] | SH[16:20] | MB[21:25] | ME[26:30] | Rc[31]
   468→   468→fn encode_m_form(opcode: u32, rs: u32, ra: u32, sh: u32, mb: u32, me: u32, rc: u32) -> [u8; 4] {
   469→   469→    let word = ((opcode & 0x3F) << 26)
   470→   470→        | ((rs & 0x1F) << 21)
   471→   471→        | ((ra & 0x1F) << 16)
   472→   472→        | ((sh & 0x1F) << 11)
   473→   473→        | ((mb & 0x1F) << 6)
   474→   474→        | ((me & 0x1F) << 1)
   475→   475→        | (rc & 1);
   476→   476→    encode_word(word)
   477→   477→}
   478→   478→
   479→   479→/// Build an I-form instruction: opcode[0:5] | LI[6:29] | AA[30] | LK[31]
   480→   480→fn encode_i_form(opcode: u32, li: i32, aa: u32, lk: u32) -> [u8; 4] {
   481→   481→    let word =
   482→   482→        ((opcode & 0x3F) << 26) | (((li as u32) & 0x03FF_FFFF) << 2) | ((aa & 1) << 1) | (lk & 1);
   483→   483→    encode_word(word)
   484→   484→}
   485→   485→
   486→   486→/// Build a B-form instruction: opcode[0:5] | BO[6:10] | BI[11:15] | BD[16:29] | AA[30] | LK[31]
   487→   487→fn encode_b_form(opcode: u32, bo: u32, bi: u32, bd: i32, aa: u32, lk: u32) -> [u8; 4] {
   488→   488→    let word = ((opcode & 0x3F) << 26)
   489→   489→        | ((bo & 0x1F) << 21)
   490→   490→        | ((bi & 0x1F) << 16)
   491→   491→        | (((bd as u32) & 0x3FFF) << 2)
   492→   492→        | ((aa & 1) << 1)
   493→   493→        | (lk & 1);
   494→   494→    encode_word(word)
   495→   495→}
   496→   496→
   497→   497→/// Build an XL-form instruction: opcode[0:5] | BO[6:10] | BI[11:15] | 0[16:18] | BH[19:21] | xo[21:30] | LK[31]
   498→   498→fn encode_xl_form(opcode: u32, bo: u32, bi: u32, bh: u32, xo: u32, lk: u32) -> [u8; 4] {
   499→   499→    let word = ((opcode & 0x3F) << 26)
   500→   500→        | ((bo & 0x1F) << 21)
   501→   501→        | ((bi & 0x1F) << 16)
   502→   502→        | ((bh & 0x7) << 11)
   503→   503→        | ((xo & 0x3FF) << 1)
   504→   504→        | (lk & 1);
   505→   505→    encode_word(word)
   506→   506→}

   707→   707→impl Instruction {
   708→   708→    /// Encode this instruction into a 4-byte little-endian machine code word.
   709→   709→    ///
   710→   710→    /// Encoding follows the Power ISA Version 3.1.
   711→   711→    pub fn encode(&self) -> [u8; 4] {
   712→   712→        match self {
   713→   713→            // ── Arithmetic ──────────────────────────────────────
   714→   714→            Instruction::Add { rt, ra, rb } => {
   715→   715→                // ADD rT, rA, rB: primary=31, OE=0, xo=266, Rc=0
   716→   716→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 266, 0)
   717→   717→            }
   718→   718→            Instruction::Addi { rt, ra, simm } => {
   719→   719→                // ADDI rT, rA, simm16: primary=14
   720→   720→                encode_d_form(14, rt.encoding(), ra.encoding(), *simm)
   721→   721→            }
   722→   722→            Instruction::Addis { rt, ra, simm } => {
   723→   723→                // ADDIS rT, rA, simm16: primary=15
   724→   724→                encode_d_form(15, rt.encoding(), ra.encoding(), *simm)
   725→   725→            }
   726→   726→            Instruction::Subf { rt, ra, rb } => {
   727→   727→                // SUBF rT, rA, rB: primary=31, OE=0, xo=40, Rc=0
   728→   728→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 40, 0)
   729→   729→            }
   730→   730→            Instruction::Mullw { rt, ra, rb } => {
   731→   731→                // MULLW rT, rA, rB: primary=31, OE=0, xo=235, Rc=0
   732→   732→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 235, 0)
   733→   733→            }
   734→   734→            Instruction::Mulld { rt, ra, rb } => {
   735→   735→                // MULLD rT, rA, rB: primary=31, OE=0, xo=233, Rc=0
   736→   736→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 233, 0)
   737→   737→            }
   738→   738→            Instruction::Mulhw { rt, ra, rb } => {
   739→   739→                // MULHW rT, rA, rB: primary=31, xo=75, Rc=0
   740→   740→                encode_x_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 75, 0)
   741→   741→            }
   742→   742→            Instruction::Mulhd { rt, ra, rb } => {
   743→   743→                // MULHD rT, rA, rB: primary=31, xo=73, Rc=0
   744→   744→                encode_x_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 73, 0)
   745→   745→            }
   746→   746→            Instruction::Divw { rt, ra, rb } => {
   747→   747→                // DIVW rT, rA, rB: primary=31, OE=0, xo=491, Rc=0
   748→   748→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 491, 0)
   749→   749→            }
   750→   750→            Instruction::Divd { rt, ra, rb } => {
   751→   751→                // DIVD rT, rA, rB: primary=31, OE=0, xo=459, Rc=0
   752→   752→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 459, 0)
   753→   753→            }
   754→   754→            Instruction::Neg { rt, ra } => {
   755→   755→                // NEG rT, rA: primary=31, OE=0, xo=104, Rc=0, rB=0
   756→   756→                encode_xo_form(31, rt.encoding(), ra.encoding(), 0, 0, 104, 0)
   757→   757→            }
   758→   758→            Instruction::Cntlzd { ra, rs } => {
   759→   759→                // CNTLZD rA, rS: primary=31, xo=58, Rc=0, rB=0
   760→   760→                encode_x_form(31, rs.encoding(), ra.encoding(), 0, 58, 0)
   761→   761→            }
   762→   762→            Instruction::Popcntd { ra, rs } => {
   763→   763→                // POPCNTD rA, rS: primary=31, xo=506, Rc=0, rB=0
   764→   764→                encode_x_form(31, rs.encoding(), ra.encoding(), 0, 506, 0)
   765→   765→            }
   766→   766→            Instruction::Extsw { ra, rs } => {
   767→   767→                // EXTSW rA, rS: primary=31, xo=986, Rc=0, rB=0
   768→   768→                encode_x_form(31, rs.encoding(), ra.encoding(), 0, 986, 0)
   769→   769→            }
   770→   770→
   771→   771→            // ── Logical ────────────────────────────────────────
   772→   772→            Instruction::And { ra, rs, rb } => {
   773→   773→                // AND rA, rS, rB: primary=31, xo=28, Rc=0
   774→   774→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 28, 0)
   775→   775→            }
   776→   776→            Instruction::Andi { ra, rs, uimm } => {
   777→   777→                // ANDI. rA, rS, uimm16: primary=28, Rc=1 (always)
   778→   778→                let word = ((28u32 & 0x3F) << 26)
   779→   779→                    | ((rs.encoding() & 0x1F) << 21)
   780→   780→                    | ((ra.encoding() & 0x1F) << 16)
   781→   781→                    | (uimm & 0xFFFF);
   782→   782→                encode_word(word)

  1201→  1201→/// Build a proper ELF64 binary for PPC64LE with 2 LOAD segments.
  1202→  1202→///
  1203→  1203→/// Produces a static executable with:
  1204→  1204→/// - Segment 1: LOAD (PF_R | PF_X) — .text
  1205→  1205→/// - Segment 2: LOAD (PF_R | PF_W) — .data / BSS (writable)
  1206→  1206→///
  1207→  1207→/// Uses little-endian byte order for ppc64le.  The text segment is placed
  1208→  1208→/// at a page-aligned offset so that the kernel's ELF loader can mmap() it.
  1209→  1209→fn build_ppc64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1210→  1210→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (PPC64 typical page size)
  1211→  1211→
  1212→  1212→    let elf_header_size: u64 = 64;
  1213→  1213→    let phdr_size: u64 = 56;
  1214→  1214→    let num_phdrs: u64 = 2;
  1215→  1215→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1216→  1216→    // Page-align the text segment start in the file.
  1217→  1217→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1218→  1218→    let text_size = code.len() as u64;
  1219→  1219→
  1220→  1220→    // The data segment starts on the next page after the text.
  1221→  1221→    let text_file_end = text_offset + text_size;
  1222→  1222→    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1223→  1223→    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1224→  1224→    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1225→  1225→    let entry_point = base_addr + text_offset;
  1226→  1226→
  1227→  1227→    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1228→  1228→
  1229→  1229→    // --- e_ident ---
  1230→  1230→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1231→  1231→    elf.push(2); // ELFCLASS64
  1232→  1232→    elf.push(1); // ELFDATA2LSB (ppc64le)
  1233→  1233→    elf.push(1); // EV_CURRENT
  1234→  1234→    elf.push(3); // ELFOSABI_LINUX
  1235→  1235→    elf.push(0); // padding
  1236→  1236→    elf.extend_from_slice(&[0u8; 7]); // padding
  1237→  1237→
  1238→  1238→    // --- ELF header fields ---
  1239→  1239→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1240→  1240→    elf.extend_from_slice(&21u16.to_le_bytes()); // e_machine = EM_PPC64
  1241→  1241→    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1242→  1242→    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1243→  1243→    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1244→  1244→    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1245→  1245→    // e_flags: EF_PPC64_ABI_V2 = 0x2 (required for PPC64LE ELFv2 ABI)
  1246→  1246→    elf.extend_from_slice(&2u32.to_le_bytes()); // e_flags
  1247→  1247→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1248→  1248→    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1249→  1249→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1250→  1250→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1251→  1251→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1252→  1252→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1253→  1253→
  1254→  1254→    // --- Program Header 1: LOAD (PF_R | PF_X) — .text ---
  1255→  1255→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1256→  1256→    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1257→  1257→    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1258→  1258→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1259→  1259→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1260→  1260→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1261→  1261→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1262→  1262→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1263→  1263→
  1264→  1264→    // --- Program Header 2: LOAD (PF_R | PF_W) — .data / stack ---
  1265→  1265→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1266→  1266→    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1267→  1267→    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1268→  1268→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1269→  1269→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1270→  1270→    elf.extend_from_slice(&0u64.to_le_bytes()); // p_filesz (no initialized data)
  1271→  1271→    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1272→  1272→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1273→  1273→
  1274→  1274→    // --- .text section ---
  1275→  1275→    // Pad to page-aligned text_offset
  1276→  1276→    while (elf.len() as u64) < text_offset {
  1277→  1277→        elf.push(0);
  1278→  1278→    }
  1279→  1279→    elf.extend_from_slice(code);
  1280→  1280→
  1281→  1281→    // --- Pad to data segment offset (if needed) ---
  1282→  1282→    while (elf.len() as u64) < data_offset {
  1283→  1283→        elf.push(0);
  1284→  1284→    }
  1285→  1285→
  1286→  1286→    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1287→  1287→
  1288→  1288→    elf
  1289→  1289→}

  1315→  1315→/// Compute the stack frame size for an IR function on PPC64.
  1316→  1316→///
  1317→  1317→/// Sums `Alloc` instruction sizes, adds 32 bytes for the LR/CR save area
  1318→  1318→/// (per ELFv2 ABI), and rounds up to 16-byte alignment.
  1319→  1319→fn ppc64_compute_frame_size(func: &IRFunction) -> usize {
  1320→  1320→    let mut total: u32 = 32; // LR save (8) + CR save (8) + back chain (8) + TOC save (8)
  1321→  1321→    for block in &func.blocks {
  1322→  1322→        for instr in &block.instructions {
  1323→  1323→            if let IRInstr::Alloc { size, .. } = instr {
  1324→  1324→                let aligned = (*size).div_ceil(16) * 16;
  1325→  1325→                total += aligned;
  1326→  1326→            }
  1327→  1327→        }
  1328→  1328→    }
  1329→  1329→    // Round up to 16-byte alignment
  1330→  1330→    total = (total + 15) & !15;
  1331→  1331→    total as usize
  1332→  1332→}

  1373→  1373→/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1374→  1374→fn map_vreg_to_gpr(
  1375→  1375→    vreg_id: u32,
  1376→  1376→    arg_index: Option<usize>,
  1377→  1377→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1378→  1378→) -> Gpr {
  1379→  1379→    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1380→  1380→        return *gpr;
  1381→  1381→    }
  1382→  1382→
  1383→  1383→    // If this is an argument, map it to the corresponding argument register.
  1384→  1384→    if let Some(idx) = arg_index {
  1385→  1385→        if let Some(reg) = Gpr::arg_register(idx) {
  1386→  1386→            vreg_map.insert(vreg_id, reg);
  1387→  1387→            return reg;
  1388→  1388→        }
  1389→  1389→    }
  1390→  1390→
  1391→  1391→    // Otherwise, pick the first available allocatable register.
  1392→  1392→    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1393→  1393→    for &reg in ALLOCATABLE_GPRS {
  1394→  1394→        if !used.contains(&reg) {
  1395→  1395→            vreg_map.insert(vreg_id, reg);
  1396→  1396→            return reg;
  1397→  1397→        }
  1398→  1398→    }
  1399→  1399→
  1400→  1400→    // Fallback: use R11 as a scratch register.
  1401→  1401→    Gpr::R11
  1402→  1402→}

  1426→  1426→/// Load a 64-bit immediate value into a GPR, emitting the minimal number of
  1427→  1427→/// PPC64 instructions.
  1428→  1428→///
  1429→  1429→/// Strategy:
  1430→  1430→/// - **`[-32768, 32767]`**: `li rd, val` (single `addi rd, r0, val`)
  1431→  1431→/// - **`[0, 65535]`**: `ori rd, r0, val`
  1432→  1432→/// - **`0xXXXX0000` (low 16 bits zero, fits 32-bit)**: `lis rd, upper16`
  1433→  1433→/// - **Other 32-bit**: `lis rd, upper16` + `ori rd, rd, lower16`
  1434→  1434→/// - **Full 64-bit**: `lis` + `ori` + `sldi 32` + `oris` + `ori`
  1435→  1435→fn load_immediate_ppc64(rd: Gpr, val: i64, out: &mut Vec<AllocatedInstruction>) {
  1436→  1436→    let rd_phys = PhysicalReg::new(RegClass::Gpr, rd.encoding());
  1437→  1437→    let uval = val as u64;
  1438→  1438→
  1439→  1439→    if (-32768..=32767).contains(&val) {
  1440→  1440→        // li rd, val (addi rd, r0, val)
  1441→  1441→        out.push(emit_alloc_instr(
  1442→  1442→            Instruction::Li {
  1443→  1443→                rt: rd,
  1444→  1444→                simm: val as i32,
  1445→  1445→            },
  1446→  1446→            vec![],
  1447→  1447→            vec![rd_phys],
  1448→  1448→        ));
  1449→  1449→    } else if uval <= 0xFFFF {
  1450→  1450→        // ori rd, r0, val (unsigned 16-bit)
  1451→  1451→        out.push(emit_alloc_instr(
  1452→  1452→            Instruction::Ori {
  1453→  1453→                ra: rd,
  1454→  1454→                rs: Gpr::R0,
  1455→  1455→                uimm: uval as u32 & 0xFFFF,
  1456→  1456→            },
  1457→  1457→            vec![],
  1458→  1458→            vec![rd_phys],
  1459→  1459→        ));
  1460→  1460→    } else if uval & 0xFFFF == 0 && (uval >> 32) == 0 {
  1461→  1461→        // lis rd, upper16 (0xXXXX0000)
  1462→  1462→        out.push(emit_alloc_instr(
  1463→  1463→            Instruction::Lis {
  1464→  1464→                rt: rd,
  1465→  1465→                simm: ((uval >> 16) & 0xFFFF) as i16 as i32,
  1466→  1466→            },
  1467→  1467→            vec![],
  1468→  1468→            vec![rd_phys],
  1469→  1469→        ));
  1470→  1470→    } else if uval >> 32 == 0 {
  1471→  1471→        // 32-bit value: lis + ori
  1472→  1472→        let hi16 = ((uval >> 16) & 0xFFFF) as u32;
  1473→  1473→        let lo16 = (uval & 0xFFFF) as u32;
  1474→  1474→        out.push(emit_alloc_instr(
  1475→  1475→            Instruction::Lis {
  1476→  1476→                rt: rd,
  1477→  1477→                simm: hi16 as i16 as i32,
  1478→  1478→            },
  1479→  1479→            vec![],
  1480→  1480→            vec![rd_phys],
  1481→  1481→        ));
  1482→  1482→        if lo16 != 0 {
  1483→  1483→            out.push(emit_alloc_instr(
  1484→  1484→                Instruction::Ori {
  1485→  1485→                    ra: rd,
  1486→  1486→                    rs: rd,
  1487→  1487→                    uimm: lo16,
  1488→  1488→                },
  1489→  1489→                vec![rd_phys],
  1490→  1490→                vec![rd_phys],
  1491→  1491→            ));
  1492→  1492→        }
  1493→  1493→    } else {
  1494→  1494→        // Full 64-bit: lis + ori + sldi 32 + oris + ori
  1495→  1495→        let hi32 = (uval >> 32) as u32;
  1496→  1496→        let lo32 = (uval & 0xFFFF_FFFF) as u32;
  1497→  1497→
  1498→  1498→        // lis rd, upper16(hi32)
  1499→  1499→        out.push(emit_alloc_instr(
  1500→  1500→            Instruction::Lis {
  1501→  1501→                rt: rd,
  1502→  1502→                simm: ((hi32 >> 16) & 0xFFFF) as i16 as i32,
  1503→  1503→            },
  1504→  1504→            vec![],
  1505→  1505→            vec![rd_phys],
  1506→  1506→        ));
  1507→  1507→
  1508→  1508→        // ori rd, rd, lower16(hi32)
  1509→  1509→        if hi32 & 0xFFFF != 0 {
  1510→  1510→            out.push(emit_alloc_instr(
  1511→  1511→                Instruction::Ori {
  1512→  1512→                    ra: rd,
  1513→  1513→                    rs: rd,
  1514→  1514→                    uimm: hi32 & 0xFFFF,
  1515→  1515→                },
  1516→  1516→                vec![rd_phys],
  1517→  1517→                vec![rd_phys],
  1518→  1518→            ));
  1519→  1519→        }
  1520→  1520→
  1521→  1521→        // sldi rd, rd, 32 = rldicr rd, rd, 32, 31
  1522→  1522→        // MD-form: [0:5]=30, [6:10]=rS, [11:15]=rA, [16:20]=SH[0:4],
  1523→  1523→        // [21:25]=ME[0:4], [26]=SH[5], [27]=ME[5], [28:30]=xo(=2), [31]=Rc(=0)
  1524→  1524→        let sh: u32 = 32;
  1525→  1525→        let me: u32 = 63 - sh;
  1526→  1526→        let sldi_word: u32 = (30u32 << 26)
  1527→  1527→            | (rd.encoding() << 21)
  1528→  1528→            | (rd.encoding() << 16)
  1529→  1529→            | ((sh & 0x1F) << 11)
  1530→  1530→            | ((me & 0x1F) << 6)
  1531→  1531→            | (((sh >> 5) & 1) << 5)
  1532→  1532→            | (((me >> 5) & 1) << 4)
  1533→  1533→            | (2u32 << 1);
  1534→  1534→        out.push(AllocatedInstruction {
  1535→  1535→            opcode: "rldicr".to_string(),
  1536→  1536→            reads: vec![rd_phys],
  1537→  1537→            writes: vec![rd_phys],
  1538→  1538→            encoded: encode_word(sldi_word).to_vec(),
  1539→  1539→        });
  1540→  1540→
  1541→  1541→        // oris rd, rd, upper16(lo32) — primary opcode 25, D-form
  1542→  1542→        if (lo32 >> 16) & 0xFFFF != 0 {
  1543→  1543→            let oris_word: u32 = (25u32 << 26)
  1544→  1544→                | (rd.encoding() << 21)
  1545→  1545→                | (rd.encoding() << 16)
  1546→  1546→                | ((lo32 >> 16) & 0xFFFF);
  1547→  1547→            out.push(AllocatedInstruction {
  1548→  1548→                opcode: "oris".to_string(),
  1549→  1549→                reads: vec![rd_phys],
  1550→  1550→                writes: vec![rd_phys],
  1551→  1551→                encoded: encode_word(oris_word).to_vec(),
  1552→  1552→            });
  1553→  1553→        }
  1554→  1554→
  1555→  1555→        // ori rd, rd, lower16(lo32)
  1556→  1556→        if lo32 & 0xFFFF != 0 {
  1557→  1557→            out.push(emit_alloc_instr(
  1558→  1558→                Instruction::Ori {
  1559→  1559→                    ra: rd,
  1560→  1560→                    rs: rd,
  1561→  1561→                    uimm: lo32 & 0xFFFF,
  1562→  1562→                },
  1563→  1563→                vec![rd_phys],
  1564→  1564→                vec![rd_phys],
  1565→  1565→            ));
  1566→  1566→        }
  1567→  1567→    }
  1568→  1568→}

  1570→  1570→/// Resolve an `IRValue` to a physical `Gpr`, emitting immediate-load
  1571→  1571→/// instructions as needed.
  1572→  1572→///
  1573→  1573→/// - `IRValue::Register(id)` → looks up `id` in `reg_map`
  1574→  1574→/// - `IRValue::Immediate(val)` → loads `val` into `scratch` via
  1575→  1575→///   `load_immediate_ppc64` and returns `scratch`
  1576→  1576→/// - `IRValue::Address(addr)` → loads `addr` into `scratch`
  1577→  1577→/// - `IRValue::Label(_)` → loads 0 into `scratch` (placeholder for relocation)
  1578→  1578→///
  1579→  1579→/// The caller must supply a scratch register that is not live at this point
  1580→  1580→/// (e.g., `R11` for the first operand, `R12` for the second).
  1581→  1581→fn resolve_gpr_ppc64(
  1582→  1582→    val: &IRValue,
  1583→  1583→    reg_map: &mut std::collections::HashMap<u32, Gpr>,
  1584→  1584→    scratch: Gpr,
  1585→  1585→    out: &mut Vec<AllocatedInstruction>,
  1586→  1586→) -> Gpr {
  1587→  1587→    match val {
  1588→  1588→        IRValue::Register(id) => map_vreg_to_gpr(*id, None, reg_map),
  1589→  1589→        IRValue::Immediate(imm) => {
  1590→  1590→            load_immediate_ppc64(scratch, *imm, out);
  1591→  1591→            scratch
  1592→  1592→        }
  1593→  1593→        IRValue::Address(addr) => {
  1594→  1594→            load_immediate_ppc64(scratch, *addr as i64, out);
  1595→  1595→            scratch
  1596→  1596→        }
  1597→  1597→        IRValue::Label(_) => {
  1598→  1598→            // Labels need linker relocation; emit li scratch, 0 as placeholder
  1599→  1599→            load_immediate_ppc64(scratch, 0, out);
  1600→  1600→            scratch
  1601→  1601→        }
  1602→  1602→    }
  1603→  1603→}

  1605→  1605→/// Lower a comparison kind to PPC64 instructions, producing 0 or 1 in `dst`.

  1650→  1650→                rb: rhs,

  1652→  1652→            1,
  1653→  1653→        ),
  1654→  1654→        CmpKind::SGe => (
  1655→  1655→            Instruction::Cmp {
  1656→  1656→                bf: CrField::CR0,
  1657→  1657→                l: 1,
  1658→  1658→                ra: lhs,
  1659→  1659→                rb: rhs,
  1660→  1660→            },
  1661→  1661→            1,
  1662→  1662→        ),
  1663→  1663→        CmpKind::ULt => (
  1664→  1664→            Instruction::Cmpl {
  1665→  1665→                bf: CrField::CR0,
  1666→  1666→                l: 1,
  1667→  1667→                ra: lhs,
  1668→  1668→                rb: rhs,
  1669→  1669→            },

  1678→  1678→            },

  1724→  1724→    };
  1725→  1725→    let _ = cond_bit;
  1726→  1726→    result.push(emit_alloc_instr(
  1727→  1727→        Instruction::Li { rt: dst, simm: 1 },
  1728→  1728→        vec![],
  1729→  1729→        vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1730→  1730→    ));
  1731→  1731→    // Patch: use a simplified approach — li dst,1 is always emitted but should be conditional.
  1732→  1732→    // For simplicity, use select-like pattern:
  1733→  1733→    // cmp; li dst, 0; bne cr0, +8; li dst, 1
  1734→  1734→    // Actually let's redo this more correctly:
  1735→  1735→    result.clear();
  1736→  1736→    result.push(emit_alloc_instr(
  1737→  1737→        cmp_inst,
  1738→  1738→        vec![
  1739→  1739→            PhysicalReg::new(RegClass::Gpr, lhs.encoding()),
  1740→  1740→            PhysicalReg::new(RegClass::Gpr, rhs.encoding()),
  1741→  1741→        ],
  1742→  1742→        vec![],
  1743→  1743→    ));
  1744→  1744→    result.push(emit_alloc_instr(
  1745→  1745→        Instruction::Li { rt: dst, simm: 0 },
  1746→  1746→        vec![],
  1747→  1747→        vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1748→  1748→    ));
  1749→  1749→    result.push(emit_alloc_instr(
  1750→  1750→        Instruction::Bc { bo, bi, bd: 2 },
  1751→  1751→        vec![],
  1752→  1752→        vec![],
  1753→  1753→    ));
  1754→  1754→    result.push(emit_alloc_instr(
  1755→  1755→        Instruction::Li { rt: dst, simm: 1 },
  1756→  1756→        vec![],
  1757→  1757→        vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1758→  1758→    ));
  1759→  1759→    result
  1760→  1760→}
