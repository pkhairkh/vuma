     1→//! # WebAssembly 32-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the Wasm32 target.  WebAssembly is
     4→//! fundamentally different from register-based ISAs: it is a stack machine
     5→//! with structured control flow and a binary format.  This module provides:
     6→//!
     7→//! - `WasmType` — value types (i32, i64, f32, f64)
     8→//! - `WasmInstr` — exhaustive instruction enum covering all ops needed for VUMA IR
     9→//! - LEB128 encoding/decoding (unsigned and signed)
    10→//! - Wasm binary-format encoder that emits complete `.wasm` modules
    11→//! - `Wasm32Backend` — `Backend` implementation that lowers IR to Wasm bytecode
    12→

    24→// ===========================================================================
    25→// Wasm value types
    26→// ===========================================================================
    27→
    28→/// WebAssembly value types.
    29→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
    30→pub enum WasmType {
    31→    I32,
    32→    I64,
    33→    F32,
    34→    F64,
    35→}
    36→
    37→impl WasmType {
    38→    /// Binary encoding of this type in the Wasm format.
    39→    pub fn to_byte(&self) -> u8 {
    40→        match self {
    41→            WasmType::I32 => 0x7F,
    42→            WasmType::I64 => 0x7E,
    43→            WasmType::F32 => 0x7D,
    44→            WasmType::F64 => 0x7C,
    45→        }
    46→    }
    47→
    48→    /// Size in bytes of this Wasm type on the stack.
    49→    pub fn byte_size(&self) -> usize {
    50→        match self {
    51→            WasmType::I32 | WasmType::F32 => 4,
    52→            WasmType::I64 | WasmType::F64 => 8,
    53→        }
    54→    }
    55→
    56→    /// Whether this is an integer type.
    57→    pub fn is_integer(&self) -> bool {
    58→        matches!(self, WasmType::I32 | WasmType::I64)
    59→    }
    60→
    61→    /// Whether this is a floating-point type.
    62→    pub fn is_float(&self) -> bool {
    63→        matches!(self, WasmType::F32 | WasmType::F64)
    64→    }
    65→
    66→    /// Map an IRType to the corresponding WasmType.
    67→    pub fn from_ir_type(ty: &IRType) -> Option<WasmType> {
    68→        match ty {
    69→            IRType::I8
    70→            | IRType::I16
    71→            | IRType::I32
    72→            | IRType::U8
    73→            | IRType::U16
    74→            | IRType::U32
    75→            | IRType::Ptr
    76→            | IRType::Func => Some(WasmType::I32),
    77→            IRType::I64 | IRType::U64 => Some(WasmType::I64),
    78→            IRType::F32 => Some(WasmType::F32),
    79→            IRType::F64 => Some(WasmType::F64),
    80→            IRType::Void | IRType::Struct { .. } | IRType::Array { .. } => None,
    81→        }
    82→    }
    83→
    84→    /// Decode a WasmType from its binary encoding byte.
    85→    pub fn from_byte(byte: u8) -> Option<WasmType> {
    86→        match byte {
    87→            0x7F => Some(WasmType::I32),
    88→            0x7E => Some(WasmType::I64),
    89→            0x7D => Some(WasmType::F32),
    90→            0x7C => Some(WasmType::F64),
    91→            _ => None,
    92→        }
    93→    }
    94→}
    95→
    96→impl std::fmt::Display for WasmType {
    97→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    98→        match self {
    99→            WasmType::I32 => write!(f, "i32"),
   100→            WasmType::I64 => write!(f, "i64"),
   101→            WasmType::F32 => write!(f, "f32"),
   102→            WasmType::F64 => write!(f, "f64"),
   103→        }
   104→    }
   105→}

   392→impl WasmInstr {
   393→    /// Encode this instruction into Wasm bytecode, appending bytes to `out`.
   394→    pub fn encode(&self, out: &mut Vec<u8>) {
   395→        match self {
   396→            // ── Control ──────────────────────────────────────────────
   397→            WasmInstr::Block(ty) => {
   398→                out.push(0x02);
   399→                match ty {
   400→                    Some(t) => out.push(t.to_byte()),
   401→                    None => out.push(0x40), // void block type
   402→                }
   403→            }
   404→            WasmInstr::Loop(ty) => {
   405→                out.push(0x03);
   406→                match ty {
   407→                    Some(t) => out.push(t.to_byte()),
   408→                    None => out.push(0x40), // void loop type
   409→                }
   410→            }
   411→            WasmInstr::If(ty) => {
   412→                out.push(0x04);
   413→                match ty {
   414→                    Some(t) => out.push(t.to_byte()),
   415→                    None => out.push(0x40), // void if type
   416→                }
   417→            }
   418→            WasmInstr::Else => out.push(0x05),
   419→            WasmInstr::End => out.push(0x0B),
   420→            WasmInstr::Br(idx) => {
   421→                out.push(0x0C);
   422→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   423→            }
   424→            WasmInstr::BrIf(idx) => {
   425→                out.push(0x0D);
   426→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   427→            }
   428→            WasmInstr::BrTable { labels, default } => {
   429→                out.push(0x0E);
   430→                out.extend_from_slice(&encode_unsigned_leb128(labels.len() as u64));
   431→                for &lbl in labels {
   432→                    out.extend_from_slice(&encode_unsigned_leb128(lbl as u64));
   433→                }
   434→                out.extend_from_slice(&encode_unsigned_leb128(*default as u64));
   435→            }
   436→            WasmInstr::Return => out.push(0x0F),
   437→            WasmInstr::Call(idx) => {
   438→                out.push(0x10);
   439→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   440→            }
   441→            WasmInstr::CallIndirect {
   442→                type_idx,
   443→                table_idx,
   444→            } => {
   445→                out.push(0x11);
   446→                out.extend_from_slice(&encode_unsigned_leb128(*type_idx as u64));
   447→                out.extend_from_slice(&encode_unsigned_leb128(*table_idx as u64));
   448→            }
   449→
   450→            // ── Parametric ──────────────────────────────────────────
   451→            WasmInstr::Select => out.push(0x1B),
   452→            WasmInstr::Drop => out.push(0x1A),
   453→
   454→            // ── Variable ────────────────────────────────────────────
   455→            WasmInstr::LocalGet(idx) => {
   456→                out.push(0x20);
   457→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   458→            }
   459→            WasmInstr::LocalSet(idx) => {
   460→                out.push(0x21);
   461→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   462→            }
   463→            WasmInstr::LocalTee(idx) => {
   464→                out.push(0x22);
   465→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   466→            }
   467→            WasmInstr::GlobalGet(idx) => {
   468→                out.push(0x23);
   469→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   470→            }
   471→            WasmInstr::GlobalSet(idx) => {
   472→                out.push(0x24);
   473→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   474→            }
   475→
   476→            // ── Memory ──────────────────────────────────────────────
   477→            WasmInstr::I32Load { align, offset } => {
   478→                out.push(0x28);
   479→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   480→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   481→            }
   482→            WasmInstr::I64Load { align, offset } => {
   483→                out.push(0x29);
   484→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   485→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   486→            }
   487→            WasmInstr::F32Load { align, offset } => {
   488→                out.push(0x2A);
   489→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   490→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   491→            }
   492→            WasmInstr::F64Load { align, offset } => {
   493→                out.push(0x2B);
   494→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   495→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   496→            }
   497→            WasmInstr::I32Store { align, offset } => {
   498→                out.push(0x36);

   501→            }

   503→                out.push(0x37);

   516→            }

   521→            }

   531→            }

   541→            }

   788→/// Encode an unsigned 64-bit value as unsigned LEB128.
   789→pub fn encode_unsigned_leb128(value: u64) -> Vec<u8> {
   790→    let mut result = Vec::new();
   791→    let mut val = value;
   792→    loop {
   793→        let mut byte = (val & 0x7F) as u8;
   794→        val >>= 7;
   795→        if val != 0 {
   796→            byte |= 0x80;
   797→        }
   798→        result.push(byte);
   799→        if val == 0 {
   800→            break;
   801→        }
   802→    }
   803→    result
   804→}
   805→
   806→/// Encode a signed 64-bit value as signed LEB128.
   807→pub fn encode_signed_leb128(value: i64) -> Vec<u8> {
   808→    let mut result = Vec::new();
   809→    let mut val = value;
   810→    loop {
   811→        let mut byte = (val & 0x7F) as u8;
   812→        val >>= 7;
   813→        // Sign bit of byte is bit 6
   814→        let sign_bit = (byte & 0x40) != 0;
   815→        // If val is 0 and sign bit is clear, or val is -1 and sign bit is set, we're done
   816→        if (val == 0 && !sign_bit) || (val == -1 && sign_bit) {
   817→            result.push(byte);
   818→            break;
   819→        }
   820→        byte |= 0x80;
   821→        result.push(byte);
   822→    }
   823→    result
   824→}
   825→
   826→/// Decode an unsigned LEB128 value from a byte slice.
   827→/// Returns (decoded_value, number_of_bytes_consumed).
   828→pub fn decode_unsigned_leb128(bytes: &[u8]) -> (u64, usize) {
   829→    let mut result: u64 = 0;
   830→    let mut shift: u32 = 0;
   831→    let mut count = 0;
   832→    for &byte in bytes {
   833→        count += 1;
   834→        result |= ((byte & 0x7F) as u64) << shift;
   835→        if (byte & 0x80) == 0 {
   836→            break;
   837→        }
   838→        shift += 7;
   839→    }
   840→    (result, count)
   841→}
   842→
   843→/// Decode a signed LEB128 value from a byte slice.
   844→/// Returns (decoded_value, number_of_bytes_consumed).
   845→pub fn decode_signed_leb128(bytes: &[u8]) -> (i64, usize) {
   846→    let mut result: i64 = 0;
   847→    let mut shift: u32 = 0;
   848→    let mut count = 0;
   849→    let mut byte: u8;
   850→    loop {
   851→        byte = bytes[count];
   852→        count += 1;
   853→        result |= ((byte & 0x7F) as i64) << shift;
   854→        shift += 7;
   855→        if (byte & 0x80) == 0 {
   856→            break;
   857→        }
   858→    }
   859→    // Sign-extend if the sign bit is set
   860→    if shift < 64 && (byte & 0x40) != 0 {
   861→        result |= !0i64 << shift;
   862→    }
   863→    (result, count)
   864→}

   887→/// A Wasm function type signature: (params) -> (results).
   888→///
   889→/// Supports multi-value returns (Wasm 2.0 feature): the `results` vector
   890→/// may contain more than one type.
   891→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]

   893→    pub params: Vec<WasmType>,
   894→    pub results: Vec<WasmType>,
   895→}

   897→impl WasmFuncType {
   898→    /// Encode this function type in Wasm binary format.
   899→    pub fn encode(&self) -> Vec<u8> {
   900→        let mut out = Vec::new();
   901→        out.push(0x60); // func type tag
   902→        out.extend_from_slice(&encode_unsigned_leb128(self.params.len() as u64));
   903→        for p in &self.params {
   904→            out.push(p.to_byte());
   905→        }
   906→        out.extend_from_slice(&encode_unsigned_leb128(self.results.len() as u64));
   907→        for r in &self.results {
   908→            out.push(r.to_byte());
   909→        }
   910→        out
   911→    }
   912→}

  1117→    /// Encode the complete module into bytes.
  1118→    pub fn encode(self) -> Vec<u8> {
  1119→        let mut module = Vec::new();
  1120→
  1121→        // Magic + version
  1122→        module.extend_from_slice(&WASM_MAGIC);
  1123→        module.extend_from_slice(&WASM_VERSION);
  1124→
  1125→        // Type section
  1126→        if !self.types.is_empty() {
  1127→            let mut section = Vec::new();
  1128→            section.extend_from_slice(&encode_unsigned_leb128(self.types.len() as u64));
  1129→            for ft in &self.types {
  1130→                section.extend_from_slice(&ft.encode());
  1131→            }
  1132→            emit_section(&mut module, SECTION_TYPE, &section);
  1133→        }
  1134→
  1135→        // Import section
  1136→        if !self.imports.is_empty() {
  1137→            let mut section = Vec::new();
  1138→            section.extend_from_slice(&encode_unsigned_leb128(self.imports.len() as u64));
  1139→            for imp in &self.imports {
  1140→                section.extend_from_slice(&encode_unsigned_leb128(imp.module.len() as u64));
  1141→                section.extend_from_slice(imp.module.as_bytes());
  1142→                section.extend_from_slice(&encode_unsigned_leb128(imp.name.len() as u64));
  1143→                section.extend_from_slice(imp.name.as_bytes());
  1144→                match &imp.kind {
  1145→                    WasmImportKind::Function { type_idx } => {
  1146→                        section.push(0x00);
  1147→                        section.extend_from_slice(&encode_unsigned_leb128(*type_idx as u64));
  1148→                    }
  1149→                    WasmImportKind::Table { elem_type, limits } => {
  1150→                        section.push(0x01);
  1151→                        section.push(*elem_type);
  1152→                        section.extend_from_slice(&limits.encode());
  1153→                    }
  1154→                    WasmImportKind::Memory { limits } => {
  1155→                        section.push(0x02);
  1156→                        section.extend_from_slice(&limits.encode());
  1157→                    }
  1158→                    WasmImportKind::Global { val_type, mutable } => {
  1159→                        section.push(0x03);
  1160→                        section.push(val_type.to_byte());
  1161→                        section.push(if *mutable { 0x01 } else { 0x00 });
  1162→                    }
  1163→                }
  1164→            }
  1165→            emit_section(&mut module, SECTION_IMPORT, &section);
  1166→        }
  1167→
  1168→        // Function section
  1169→        if !self.functions.is_empty() {
  1170→            let mut section = Vec::new();
  1171→            section.extend_from_slice(&encode_unsigned_leb128(self.functions.len() as u64));
  1172→            for &type_idx in &self.functions {
  1173→                section.extend_from_slice(&encode_unsigned_leb128(type_idx as u64));
  1174→            }
  1175→            emit_section(&mut module, SECTION_FUNCTION, &section);
  1176→        }
  1177→
  1178→        // Table section
  1179→        if !self.tables.is_empty() {
  1180→            let mut section = Vec::new();
  1181→            section.extend_from_slice(&encode_unsigned_leb128(self.tables.len() as u64));
  1182→            for (elem_type, limits) in &self.tables {
  1183→                section.push(*elem_type);
  1184→                section.extend_from_slice(&limits.encode());
  1185→            }
  1186→            emit_section(&mut module, SECTION_TABLE, &section);
  1187→        }
  1188→
  1189→        // Memory section
  1190→        if !self.memories.is_empty() {
  1191→            let mut section = Vec::new();
  1192→            section.extend_from_slice(&encode_unsigned_leb128(self.memories.len() as u64));
  1193→            for limits in &self.memories {
  1194→                section.extend_from_slice(&limits.encode());
  1195→            }
  1196→            emit_section(&mut module, SECTION_MEMORY, &section);
  1197→        }
  1198→
  1199→        // Global section
  1200→        if !self.globals.is_empty() {
  1201→            let mut section = Vec::new();
  1202→            section.extend_from_slice(&encode_unsigned_leb128(self.globals.len() as u64));
  1203→            for g in &self.globals {
  1204→                section.push(g.val_type.to_byte());
  1205→                section.push(if g.mutable { 0x01 } else { 0x00 });
  1206→                section.extend_from_slice(&g.init_expr);
  1207→            }
  1208→            emit_section(&mut module, SECTION_GLOBAL, &section);
  1209→        }
  1210→
  1211→        // Export section
  1212→        if !self.exports.is_empty() {
  1213→            let mut section = Vec::new();
  1214→            section.extend_from_slice(&encode_unsigned_leb128(self.exports.len() as u64));
  1215→            for exp in &self.exports {
  1216→                section.extend_from_slice(&encode_unsigned_leb128(exp.name.len() as u64));
  1217→                section.extend_from_slice(exp.name.as_bytes());
  1218→                section.push(exp.kind as u8);
  1219→                section.extend_from_slice(&encode_unsigned_leb128(exp.index as u64));
  1220→            }
  1221→            emit_section(&mut module, SECTION_EXPORT, &section);
  1222→        }
  1223→
  1224→        // Start section
  1225→        if let Some(func_idx) = self.start_func {
  1226→            let section = encode_unsigned_leb128(func_idx as u64);
  1227→            emit_section(&mut module, SECTION_START, &section);
  1228→        }
  1229→
  1230→        // Element section
  1231→        if !self.elements.is_empty() {
  1232→            let mut section = Vec::new();
  1233→            section.extend_from_slice(&encode_unsigned_leb128(self.elements.len() as u64));
  1234→            for elem in &self.elements {
  1235→                section.extend_from_slice(&encode_unsigned_leb128(elem.table_index as u64));
  1236→                section.extend_from_slice(&elem.offset_expr);
  1237→                section.extend_from_slice(&encode_unsigned_leb128(elem.func_indices.len() as u64));
  1238→                for &idx in &elem.func_indices {
  1239→                    section.extend_from_slice(&encode_unsigned_leb128(idx as u64));
  1240→                }
  1241→            }
  1242→            emit_section(&mut module, SECTION_ELEMENT, &section);
  1243→        }
  1244→
  1245→        // Code section
  1246→        if !self.code.is_empty() {
  1247→            let mut section = Vec::new();
  1248→            section.extend_from_slice(&encode_unsigned_leb128(self.code.len() as u64));
  1249→            for func_body in &self.code {
  1250→                // Encode the function body: locals + code + end byte
  1251→                let mut body_bytes = Vec::new();
  1252→                body_bytes
  1253→                    .extend_from_slice(&encode_unsigned_leb128(func_body.locals.len() as u64));
  1254→                for (count, ty) in &func_body.locals {
  1255→                    body_bytes.extend_from_slice(&encode_unsigned_leb128(*count as u64));
  1256→                    body_bytes.push(ty.to_byte());
  1257→                }
  1258→                body_bytes.extend_from_slice(&func_body.body);
  1259→                // Function body is size-prefixed
  1260→                let body_size = body_bytes.len() as u64;
  1261→                section.extend_from_slice(&encode_unsigned_leb128(body_size));
  1262→                section.extend_from_slice(&body_bytes);
  1263→            }
  1264→            emit_section(&mut module, SECTION_CODE, &section);
  1265→        }
  1266→
  1267→        // Data section
  1268→        if !self.data.is_empty() {
  1269→            let mut section = Vec::new();
  1270→            section.extend_from_slice(&encode_unsigned_leb128(self.data.len() as u64));
  1271→            for seg in &self.data {
  1272→                section.extend_from_slice(&encode_unsigned_leb128(seg.memory_index as u64));
  1273→                section.extend_from_slice(&seg.offset_expr);
  1274→                section.extend_from_slice(&encode_unsigned_leb128(seg.data.len() as u64));
  1275→                section.extend_from_slice(&seg.data);
  1276→            }
  1277→            emit_section(&mut module, SECTION_DATA, &section);
  1278→        }
  1279→
  1280→        module
  1281→    }
  1282→}
  1283→
  1284→/// Emit a Wasm section: section ID + LEB128 size + content.
  1285→fn emit_section(module: &mut Vec<u8>, section_id: u8, content: &[u8]) {
  1286→    module.push(section_id);
  1287→    module.extend_from_slice(&encode_unsigned_leb128(content.len() as u64));
  1288→    module.extend_from_slice(content);
  1289→}

  1291→/// Skip `count` unsigned LEB128 values in the byte slice, advancing `offset`.
  1292→fn skip_leb128(bytes: &[u8], offset: &mut usize, count: usize) {
  1293→    for _ in 0..count {
  1294→        while *offset < bytes.len() && (bytes[*offset] & 0x80) != 0 {
  1295→            *offset += 1;
  1296→        }
  1297→        if *offset < bytes.len() {
  1298→            *offset += 1;
  1299→        }
  1300→    }
  1301→}
  1302→
  1303→/// Skip `count` signed LEB128 values in the byte slice, advancing `offset`.
  1304→fn skip_signed_leb128(bytes: &[u8], offset: &mut usize, count: usize) {
  1305→    // Signed and unsigned LEB128 have the same encoding structure for skipping
  1306→    skip_leb128(bytes, offset, count);
  1307→}

  1309→/// Convert a WasmType (from this module) to the serializable WasmValueType

  1312→    match ty {
  1313→        WasmType::I32 => crate::backend::WasmValueType::I32,
  1314→        WasmType::I64 => crate::backend::WasmValueType::I64,
  1315→        WasmType::F32 => crate::backend::WasmValueType::F32,
  1316→        WasmType::F64 => crate::backend::WasmValueType::F64,
  1317→    }
  1318→}

  1320→/// Convert a serializable WasmValueType (from the backend module) back to

  1322→fn backend_to_wasm_type(ty: &crate::backend::WasmValueType) -> WasmType {
  1323→    match ty {
  1324→        crate::backend::WasmValueType::I32 => WasmType::I32,
  1325→        crate::backend::WasmValueType::I64 => WasmType::I64,
  1326→        crate::backend::WasmValueType::F32 => WasmType::F32,
  1327→        crate::backend::WasmValueType::F64 => WasmType::F64,
  1328→    }
  1329→}

  1331→/// Skip a single Wasm instruction starting at `offset`, advancing past it.
  1332→///
  1333→/// This is used by `allocate_registers` to split the body bytecode into
  1334→/// per-instruction `AllocatedInstruction` entries with real opcode names.
  1335→fn skip_one_instruction(bytes: &[u8], offset: &mut usize) {
  1336→    if *offset >= bytes.len() {
  1337→        return;
  1338→    }
  1339→    let byte = bytes[*offset];
  1340→    *offset += 1;
  1341→
  1342→    match byte {
  1343→        // ── Multi-byte opcodes ────────────────────────────────────
  1344→        0xFC => {
  1345→            // Bulk memory / saturated arithmetic prefix
  1346→            skip_leb128(bytes, offset, 1); // sub-opcode
  1347→            let (subop, _) = decode_unsigned_leb128(&bytes[*offset - 1..]);
  1348→            match subop {
  1349→                0x08 => skip_leb128(bytes, offset, 2), // memory.init
  1350→                0x0A => skip_leb128(bytes, offset, 2), // memory.copy
  1351→                0x0B => skip_leb128(bytes, offset, 1), // memory.fill
  1352→                _ => {}                                // other sub-ops may have 0 or more operands
  1353→            }
  1354→        }
  1355→        0xFD => {
  1356→            // SIMD prefix
  1357→            skip_leb128(bytes, offset, 1); // sub-opcode
  1358→            let (subop, _) = decode_unsigned_leb128(&bytes[*offset - 1..]);
  1359→            if subop == 0x0C {
  1360→                // v128.const: skip 16 bytes of payload
  1361→                *offset += 16;
  1362→            }
  1363→        }
  1364→
  1365→        // ── Control flow ─────────────────────────────────────────
  1366→        0x00 => {} // unreachable
  1367→        0x01 => {} // nop
  1368→        0x02..=0x04 => {
  1369→            *offset += 1;
  1370→        } // block/loop/if + blocktype
  1371→        0x05 | 0x0B => {} // else, end
  1372→        0x0C | 0x0D => {
  1373→            skip_leb128(bytes, offset, 1);
  1374→        } // br, br_if
  1375→        0x0E => {
  1376→            // br_table: count + labels + default
  1377→            let (count, size) = decode_unsigned_leb128(&bytes[*offset..]);
  1378→            *offset += size;
  1379→            skip_leb128(bytes, offset, count as usize + 1);
  1380→        }
  1381→        0x0F => {} // return
  1382→        0x10 => {
  1383→            skip_leb128(bytes, offset, 1);
  1384→        } // call
  1385→        0x11 => {
  1386→            skip_leb128(bytes, offset, 2);
  1387→        } // call_indirect
  1388→
  1389→        // ── Parametric ───────────────────────────────────────────
  1390→        0x1A | 0x1B => {} // drop, select
  1391→
  1392→        // ── Variable ─────────────────────────────────────────────
  1393→        0x20..=0x24 => {
  1394→            skip_leb128(bytes, offset, 1);
  1395→        } // local.get/set/tee, global.get/set
  1396→
  1397→        // ── Memory ───────────────────────────────────────────────
  1398→        0x28..=0x3E => {
  1399→            skip_leb128(bytes, offset, 2);
  1400→        } // loads/stores: align + offset
  1401→        0x3F | 0x40 => {
  1402→            skip_leb128(bytes, offset, 1);
  1403→        } // memory.size, memory.grow
  1404→
  1405→        // ── Numeric ──────────────────────────────────────────────
  1406→        0x41 => {
  1407→            skip_signed_leb128(bytes, offset, 1);
  1408→        } // i32.const
  1409→        0x42 => {
  1410→            skip_signed_leb128(bytes, offset, 1);
  1411→        } // i64.const
  1412→        0x43 => {
  1413→            *offset += 4;
  1414→        } // f32.const
  1415→        0x44 => {
  1416→            *offset += 8;
  1417→        } // f64.const
  1418→
  1419→        // All other single-byte opcodes (comparisons, arithmetic, conversions)
  1420→        0x45..=0x4F
  1421→        | 0x50..=0x5A
  1422→        | 0x5B..=0x66
  1423→        | 0x67..=0x78
  1424→        | 0x79..=0x8A
  1425→        | 0x8C
  1426→        | 0x91..=0x95
  1427→        | 0x9A
  1428→        | 0x9F..=0xA3
  1429→        | 0xA7..=0xBF => {}
  1430→
  1431→        _ => {} // Unknown opcode; nothing more to skip
  1432→    }
  1433→}

  1439→/// Context for lowering an IR function to Wasm bytecode.
  1440→struct LoweringContext {
  1441→    /// Map from virtual register ID to local index.
  1442→    vreg_to_local: HashMap<u32, u32>,
  1443→    /// Map from virtual register ID to its Wasm type.
  1444→    vreg_types: HashMap<u32, WasmType>,
  1445→    /// Number of locals (including parameters).
  1446→    num_locals: u32,
  1447→    /// Local declarations (for the function body).
  1448→    locals: Vec<(u32, WasmType)>,
  1449→    /// Map from block label to its label depth index for branch targets.
  1450→    block_labels: HashMap<String, u32>,
  1451→    /// Accumulated Wasm instructions.
  1452→    instrs: Vec<WasmInstr>,
  1453→    /// Expected result types for the function (used for return type coercion).
  1454→    result_types: Vec<WasmType>,
  1455→}
  1456→
  1457→impl LoweringContext {
  1458→    fn new(result_types: Vec<WasmType>) -> Self {
  1459→        Self {
  1460→            vreg_to_local: HashMap::new(),
  1461→            vreg_types: HashMap::new(),
  1462→            num_locals: 0,
  1463→            locals: Vec::new(),
  1464→            block_labels: HashMap::new(),
  1465→            instrs: Vec::new(),
  1466→            result_types,
  1467→        }
  1468→    }
  1469→
  1470→    /// Allocate a local for a virtual register, returning the local index.
  1471→    fn alloc_local(&mut self, vreg_id: u32, ty: WasmType) -> u32 {
  1472→        let idx = self.num_locals;
  1473→        self.vreg_to_local.insert(vreg_id, idx);
  1474→        self.vreg_types.insert(vreg_id, ty);
  1475→        self.num_locals += 1;
  1476→        // Try to merge with an existing (count, type) entry
  1477→        if let Some(last) = self.locals.last_mut() {
  1478→            if last.1 == ty {
  1479→                last.0 += 1;
  1480→                return idx;
  1481→            }
  1482→        }
  1483→        self.locals.push((1, ty));
  1484→        idx
  1485→    }
  1486→
  1487→    /// Get the local index for a virtual register.
  1488→    fn get_local(&self, vreg_id: u32) -> Option<u32> {
  1489→        self.vreg_to_local.get(&vreg_id).copied()
  1490→    }
  1491→
  1492→    /// Emit an instruction.
  1493→    fn emit(&mut self, instr: WasmInstr) {
  1494→        self.instrs.push(instr);
  1495→    }
  1496→
  1497→    /// Push a value onto the Wasm stack from an IRValue.
  1498→    /// On Wasm32, all integer values are pushed as i32.
  1499→    fn push_value(&mut self, val: &IRValue, type_hint: Option<&WasmType>) {
  1500→        match val {
  1501→            IRValue::Immediate(v) => {
  1502→                // On Wasm32, always use i32 for integer constants
  1503→                let ty = type_hint.copied().unwrap_or(WasmType::I32);
  1504→                match ty {
  1505→                    WasmType::F32 => self.emit(WasmInstr::F32Const(*v as f32)),
  1506→                    WasmType::F64 => self.emit(WasmInstr::F64Const(*v as f64)),
  1507→                    _ => self.emit(WasmInstr::I32Const(*v as i32)),
  1508→                }
  1509→            }
  1510→            IRValue::Register(id) => {
  1511→                // If the register hasn't been allocated yet, allocate it as i32.
  1512→                // On Wasm32, all integer locals are i32.
  1513→                if self.get_local(*id).is_none() {
  1514→                    self.alloc_local(*id, WasmType::I32);
  1515→                }
  1516→                if let Some(local_idx) = self.get_local(*id) {
  1517→                    self.emit(WasmInstr::LocalGet(local_idx));
  1518→                }
  1519→            }
  1520→            IRValue::Address(addr) => {
  1521→                self.emit(WasmInstr::I32Const(*addr as i32)); // wasm32: pointers are i32
  1522→            }
  1523→            IRValue::Label(_) => {
  1524→                // Labels are handled via block structure; not pushed as values
  1525→            }
  1526→        }
  1527→    }
  1528→
  1529→    /// Store the top of the Wasm stack into a virtual register's local.
  1530→    fn pop_to_vreg(&mut self, vreg_id: u32, ty: WasmType) {
  1531→        if !self.vreg_to_local.contains_key(&vreg_id) {
  1532→            self.alloc_local(vreg_id, ty);
  1533→        }
  1534→        if let Some(local_idx) = self.get_local(vreg_id) {
  1535→            self.emit(WasmInstr::LocalSet(local_idx));
  1536→        }
  1537→    }
  1538→}
  1539→
  1540→/// Determine the Wasm type for dedicated arithmetic IR instructions (Add, Sub,
  1541→/// Mul, Div, Cmp).  These instructions carry an optional `ty` field that
  1542→/// indicates the operand width.  On Wasm32, integer types narrower than I64
  1543→/// map to I32; I64/U64 are preserved for 64-bit arithmetic.
  1544→fn wasm_type_for_dedicated_arith(ir_ty: Option<&IRType>) -> WasmType {
  1545→    match ir_ty {
  1546→        Some(IRType::I64) | Some(IRType::U64) => WasmType::I64,
  1547→        Some(IRType::F32) => WasmType::F32,
  1548→        Some(IRType::F64) => WasmType::F64,
  1549→        _ => WasmType::I32, // default: I32 for all other integer types
  1550→    }
  1551→}
  1552→
  1553→/// Determine the Wasm type for an IR BinOp based on the operand types and
  1554→/// the optional IR type annotation.  On the Wasm32 target, all integer
  1555→/// operations use i32 since the address space is 32 bits; only float types
  1556→/// retain their original width.
  1557→fn wasm_type_for_binop(
  1558→    _op: &BinOpKind,
  1559→    _lhs: &IRValue,
  1560→    _rhs: &IRValue,
  1561→    ir_ty: Option<&IRType>,
  1562→    _vreg_types: &HashMap<u32, WasmType>,
  1563→) -> WasmType {
  1564→    // If the IR provides a type, use it (but map I64/U64 to I32 for Wasm32).

  1591→fn lower_function(func: &IRFunction) -> Result<(WasmFuncBody, WasmFuncType), BackendError> {

  1596→        .iter()

  1656→    // In Wasm32, all integer params/results are i32.

  1672→        results: result_types,
