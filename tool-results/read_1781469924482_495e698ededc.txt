     1→//! # Intermediate Representation (IR)
     2→//!
     3→//! Defines the IR types used as the central representation between the SCG
     4→//! (Semantic Computation Graph) front-end and the ARM64 code emitter.
     5→//!
     6→//! ## Hierarchy
     7→//!
     8→//! ```text
     9→//! IRProgram
    10→//!  ├── Vec<IRFunction>
    11→//!  │    ├── name, params, results, param_types, result_types
    12→//!  │    └── Vec<IRBlock>
    13→//!  │         ├── label
    14→//!  │         ├── Vec<IRInstr>
    15→//!  │         └── IRTerminator
    16→//!  └── Vec<DataSection>
    17→//! ```
    18→//!
    19→//! ## Type System
    20→//!
    21→//! The IR includes a type system (`IRType`) that models the ARM64 LP64 data
    22→//! model, with functions for computing sizes, alignments, AAPCS64 argument
    23→//! classification, calling-convention layout, and stack-frame layout.
    24→//!
    25→//! The IR is intentionally low-level but target-independent.  It uses virtual
    26→//! registers (`IRValue::Register(id)`) that are later mapped to physical
    27→//! ARM64 registers by the register allocator.
    28→
    29→use std::collections::{HashMap, HashSet};
    30→use std::fmt;
    31→
    32→// ---------------------------------------------------------------------------
    33→// IRType
    34→// ---------------------------------------------------------------------------
    35→
    36→/// A type in the IR type system.
    37→///
    38→/// Target-independent type representation.  The size of `Ptr` and `Func`
    39→/// depends on the target's pointer width (4 bytes on 32-bit targets like
    40→/// Wasm32/ARM32, 8 bytes on 64-bit targets like ARM64/x86_64).  All other
    41→/// types have fixed sizes regardless of target.
    42→#[derive(Debug, Clone, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
    43→pub enum IRType {
    44→    /// Signed 8-bit integer.
    45→    I8,
    46→    /// Signed 16-bit integer.
    47→    I16,
    48→    /// Signed 32-bit integer.
    49→    I32,
    50→    /// Signed 64-bit integer.
    51→    I64,
    52→    /// Unsigned 8-bit integer.
    53→    U8,
    54→    /// Unsigned 16-bit integer.
    55→    U16,
    56→    /// Unsigned 32-bit integer.
    57→    U32,
    58→    /// Unsigned 64-bit integer.
    59→    U64,
    60→    /// 32-bit IEEE 754 floating-point.
    61→    F32,
    62→    /// 64-bit IEEE 754 floating-point.
    63→    F64,
    64→    /// Opaque pointer.  Size is target-dependent: 4 bytes on 32-bit targets
    65→    /// (Wasm32, ARM32), 8 bytes on 64-bit targets (ARM64, x86_64, etc.).
    66→    Ptr,
    67→    /// Void type (size 0, only valid as a function return type).
    68→    Void,
    69→    /// Function pointer (pointer-sized).
    70→    Func,
    71→    /// Struct type with a name and ordered fields.
    72→    Struct {
    73→        /// Struct name (may be empty for anonymous structs).
    74→        name: String,
    75→        /// Fields in declaration order.
    76→        fields: Vec<IRType>,
    77→    },
    78→    /// Array type with an element type and element count.
    79→    Array {
    80→        /// Element type.
    81→        element: Box<IRType>,
    82→        /// Number of elements.
    83→        count: usize,
    84→    },
    85→}
    86→
    87→impl IRType {
    88→    /// Returns `true` if this is an integer type (signed or unsigned).
    89→    pub fn is_integer(&self) -> bool {
    90→        matches!(
    91→            self,
    92→            IRType::I8
    93→                | IRType::I16
    94→                | IRType::I32
    95→                | IRType::I64
    96→                | IRType::U8
    97→                | IRType::U16
    98→                | IRType::U32
    99→                | IRType::U64
   100→        )
   101→    }
   102→
   103→    /// Returns `true` if this is a floating-point type.
   104→    pub fn is_float(&self) -> bool {
   105→        matches!(self, IRType::F32 | IRType::F64)
   106→    }
   107→
   108→    /// Returns `true` if this type is passed by value in registers under
   109→    /// AAPCS64 (integer or FP primitive, pointer, func, or small struct/array).
   110→    pub fn is_register_passable(&self) -> bool {
   111→        match self {
   112→            IRType::I8
   113→            | IRType::I16
   114→            | IRType::I32
   115→            | IRType::I64
   116→            | IRType::U8
   117→            | IRType::U16
   118→            | IRType::U32
   119→            | IRType::U64
   120→            | IRType::F32
   121→            | IRType::F64
   122→            | IRType::Ptr
   123→            | IRType::Func => true,
   124→            IRType::Void => false,
   125→            IRType::Struct { fields, .. } => size_of(self) <= 16 && !fields.is_empty(),
   126→            IRType::Array { .. } => size_of(self) <= 16,
   127→        }
   128→    }
   129→
   130→    /// Returns `true` if this is a Homogeneous Floating-point Aggregate (HFA)
   131→    /// — a struct or array of 1–4 identical floating-point members.
   132→    pub fn is_hfa(&self) -> bool {
   133→        match self {
   134→            IRType::Struct { fields, .. } => {
   135→                if fields.is_empty() || fields.len() > 4 {
   136→                    return false;
   137→                }
   138→                let first = &fields[0];
   139→                if !first.is_float() {
   140→                    return false;
   141→                }
   142→                fields.iter().all(|f| f == first)
   143→            }
   144→            IRType::Array { element, count } => {
   145→                if *count == 0 || *count > 4 {
   146→                    return false;
   147→                }
   148→                element.is_float()
   149→            }
   150→            _ => false,
   151→        }
   152→    }
   153→
   154→    /// If this is an HFA, returns the element type and count.
   155→    pub fn hfa_info(&self) -> Option<(&IRType, usize)> {
   156→        if !self.is_hfa() {
   157→            return None;
   158→        }
   159→        match self {
   160→            IRType::Struct { fields, .. } => Some((&fields[0], fields.len())),
   161→            IRType::Array { element, count } => Some((element, *count)),
   162→            _ => None,
   163→        }
   164→    }
   165→}
   166→
   167→impl fmt::Display for IRType {
   168→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   169→        match self {
   170→            IRType::I8 => write!(f, "i8"),
   171→            IRType::I16 => write!(f, "i16"),
   172→            IRType::I32 => write!(f, "i32"),
   173→            IRType::I64 => write!(f, "i64"),
   174→            IRType::U8 => write!(f, "u8"),
   175→            IRType::U16 => write!(f, "u16"),
   176→            IRType::U32 => write!(f, "u32"),
   177→            IRType::U64 => write!(f, "u64"),
   178→            IRType::F32 => write!(f, "f32"),
   179→            IRType::F64 => write!(f, "f64"),
   180→            IRType::Ptr => write!(f, "ptr"),
   181→            IRType::Void => write!(f, "void"),
   182→            IRType::Func => write!(f, "func"),
   183→            IRType::Struct { name, fields } => {
   184→                let fields_str = fields
   185→                    .iter()
   186→                    .map(|t| format!("{}", t))
   187→                    .collect::<Vec<_>>()
   188→                    .join(", ");
   189→                write!(f, "struct {} {{ {} }}", name, fields_str)
   190→            }
   191→            IRType::Array { element, count } => {
   192→                write!(f, "[{}; {}]", element, count)
   193→            }
   194→        }
   195→    }
   196→}

   199→// size_of / alignment_of

   201→
   202→/// Returns the byte size of `t` for a target with the given pointer width.
   203→///
   204→/// - Integers: their bit-width / 8
   205→/// - Floats: 4 (f32) or 8 (f64)
   206→/// - Pointers / function pointers: `ptr_width` (4 or 8)
   207→/// - Void: 0
   208→/// - Structs: sum of field sizes with inter-field padding for alignment,
   209→///   rounded up to the struct's own alignment
   210→/// - Arrays: `size_of(element) * count`
   211→pub fn size_of_with_ptr_width(t: &IRType, ptr_width: usize) -> usize {
   212→    match t {
   213→        IRType::I8 | IRType::U8 => 1,
   214→        IRType::I16 | IRType::U16 => 2,
   215→        IRType::I32 | IRType::U32 => 4,
   216→        IRType::I64 | IRType::U64 => 8,
   217→        IRType::Ptr | IRType::Func => ptr_width,
   218→        IRType::F32 => 4,
   219→        IRType::F64 => 8,
   220→        IRType::Void => 0,
   221→        IRType::Struct { fields, .. } => {
   222→            let mut offset = 0usize;
   223→            for field in fields {
   224→                let field_align = alignment_of_with_ptr_width(field, ptr_width);
   225→                // Align the current offset to the field alignment.
   226→                offset = (offset + field_align - 1) & !(field_align - 1);
   227→                offset += size_of_with_ptr_width(field, ptr_width);
   228→            }
   229→            // Round up to struct alignment.
   230→            let struct_align = alignment_of_with_ptr_width(t, ptr_width);
   231→            if struct_align > 0 {
   232→                (offset + struct_align - 1) & !(struct_align - 1)
   233→            } else {
   234→                0
   235→            }
   236→        }
   237→        IRType::Array { element, count } => size_of_with_ptr_width(element, ptr_width) * count,
   238→    }
   239→}
   240→
   241→/// Returns the natural alignment of `t` for a target with the given pointer
   242→/// width.
   243→///
   244→/// - Primitives: their size
   245→/// - Pointers / function pointers: `ptr_width`
   246→/// - Void: 1 (by convention, so that `size_of` math works for empty structs)
   247→/// - Structs: maximum alignment of any field
   248→/// - Arrays: alignment of the element type
   249→pub fn alignment_of_with_ptr_width(t: &IRType, ptr_width: usize) -> usize {
   250→    match t {
   251→        IRType::I8 | IRType::U8 => 1,
   252→        IRType::I16 | IRType::U16 => 2,
   253→        IRType::I32 | IRType::U32 => 4,
   254→        IRType::I64 | IRType::U64 => 8,
   255→        IRType::Ptr | IRType::Func => ptr_width,
   256→        IRType::F32 => 4,
   257→        IRType::F64 => 8,
   258→        IRType::Void => 1,
   259→        IRType::Struct { fields, .. } => {
   260→            if fields.is_empty() {
   261→                return 1;
   262→            }
   263→            fields
   264→                .iter()
   265→                .map(|f| alignment_of_with_ptr_width(f, ptr_width))
   266→                .max()
   267→                .unwrap_or(1)
   268→        }
   269→        IRType::Array { element, .. } => alignment_of_with_ptr_width(element, ptr_width),
   270→    }
   271→}
   272→
   273→/// Returns the byte size of `t` assuming a 64-bit target (ARM64 LP64).
   274→///
   275→/// This is a convenience wrapper around [`size_of_with_ptr_width`] that
   276→/// assumes 8-byte pointers.  For target-correct sizes, use
   277→/// `TargetInfo::size_of` or [`size_of_with_ptr_width`] with the appropriate
   278→/// pointer width.
   279→pub fn size_of(t: &IRType) -> usize {
   280→    size_of_with_ptr_width(t, 8)
   281→}
   282→
   283→/// Returns the natural alignment of `t` assuming a 64-bit target (ARM64 LP64).
   284→///
   285→/// This is a convenience wrapper around [`alignment_of_with_ptr_width`] that
   286→/// assumes 8-byte pointers.  For target-correct alignments, use
   287→/// `TargetInfo::alignment_of` or [`alignment_of_with_ptr_width`] with the
   288→/// appropriate pointer width.
   289→pub fn alignment_of(t: &IRType) -> usize {
   290→    alignment_of_with_ptr_width(t, 8)
   291→}

   296→
   297→/// Classification of how an argument or return value is passed under AAPCS64.
   298→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
   299→pub enum ArgClass {
   300→    /// Passed in a general-purpose (X) register.
   301→    Integer,
   302→    /// Passed in a SIMD/FP (V) register.
   303→    FP,
   304→    /// Passed on the stack.
   305→    Stack,
   306→    /// Passed indirectly — the caller passes a pointer to the value.
   307→    /// For returns, the caller provides the address in X8.
   308→    Indirect,
   309→}
   310→
   311→impl fmt::Display for ArgClass {
   312→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   313→        f.write_str(match self {
   314→            ArgClass::Integer => "integer",
   315→            ArgClass::FP => "fp",
   316→            ArgClass::Stack => "stack",
   317→            ArgClass::Indirect => "indirect",
   318→        })
   319→    }
   320→}
   321→
   322→/// Classifies a single argument or return-value type according to AAPCS64.
   323→///
   324→/// ## Rules (simplified)
   325→///
   326→/// - Integer types, pointers, and function pointers → `Integer`
   327→/// - Floating-point types → `FP`
   328→/// - Void → `Integer` (only valid for return; caller should check)
   329→/// - Homogeneous FP aggregates (1–4 same-type FP members) → `FP`
   330→/// - Structs / arrays ≤ 16 bytes (not HFA) → `Integer`
   331→/// - Structs / arrays > 16 bytes → `Indirect`
   332→pub fn classify_arg(t: &IRType) -> ArgClass {
   333→    match t {
   334→        // Integer types → Integer class
   335→        IRType::I8
   336→        | IRType::I16
   337→        | IRType::I32
   338→        | IRType::I64
   339→        | IRType::U8
   340→        | IRType::U16
   341→        | IRType::U32
   342→        | IRType::U64
   343→        | IRType::Ptr
   344→        | IRType::Func => ArgClass::Integer,
   345→
   346→        // Floating-point → FP class
   347→        IRType::F32 | IRType::F64 => ArgClass::FP,
   348→
   349→        // Void — only meaningful as a return type; classify as Integer
   350→        // (the caller should treat a void return as "no return value").
   351→        IRType::Void => ArgClass::Integer,
   352→
   353→        // Struct / Array: check HFA first, then size.
   354→        IRType::Struct { .. } | IRType::Array { .. } => {
   355→            if t.is_hfa() {
   356→                ArgClass::FP
   357→            } else {
   358→                let sz = size_of(t);
   359→                if sz <= 16 {
   360→                    ArgClass::Integer
   361→                } else {
   362→                    ArgClass::Indirect
   363→                }
   364→            }
   365→        }
   366→    }
   367→}

   373→/// Which kind of register an argument or return value occupies.
   374→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
   375→pub enum RegisterClass {
   376→    /// General-purpose X register (X0–X30, SP, XZR).
   377→    X,
   378→    /// SIMD/FP V register (V0–V31).
   379→    V,
   380→}
   381→
   382→impl fmt::Display for RegisterClass {
   383→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   384→        f.write_str(match self {
   385→            RegisterClass::X => "x",
   386→            RegisterClass::V => "v",
   387→        })
   388→    }
   389→}

   391→/// Describes where a single argument or return value is placed.
   392→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   393→pub struct ArgLocation {
   394→    /// Index of the argument (0-based) in the argument list.
   395→    pub index: usize,
   396→    /// AAPCS64 classification of this argument.
   397→    pub class: ArgClass,
   398→    /// Register class and index, if passed in a register.
   399→    /// For `ArgClass::Integer` → `(RegisterClass::X, n)` where n is 0–7.
   400→    /// For `ArgClass::FP` → `(RegisterClass::V, n)` where n is 0–7.
   401→    /// `None` for stack or indirect arguments.
   402→    pub register: Option<(RegisterClass, u32)>,
   403→    /// Byte offset from SP (for stack arguments) or from the indirect
   404→    /// pointer, if applicable.  `None` for register arguments.
   405→    pub stack_offset: Option<i32>,
   406→}
   407→
   408→/// Describes the location of the return value.
   409→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   410→pub struct RetLocation {
   411→    /// AAPCS64 classification of the return.
   412→    pub class: ArgClass,
   413→    /// Register(s) used for the return value.
   414→    /// - Void: empty
   415→    /// - Integer ≤ 8 bytes: `[(X, 0)]`
   416→    /// - Integer 9–16 bytes (struct): `[(X, 0), (X, 1)]`
   417→    /// - FP (HFA 1 member): `[(V, 0)]`, etc.
   418→    /// - Indirect: `[(X, 8)]` — X8 holds the caller-allocated address
   419→    pub registers: Vec<(RegisterClass, u32)>,
   420→}
   421→
   422→/// Complete calling-convention information for a function signature.
   423→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   424→pub struct CallingConvInfo {
   425→    /// Location for each argument.
   426→    pub arg_locations: Vec<ArgLocation>,
   427→    /// Location for the return value.
   428→    pub ret_location: RetLocation,
   429→    /// Total bytes of stack argument space that the *caller* must reserve
   430→    /// for arguments that don't fit in registers.
   431→    pub stack_args_size: usize,
   432→}
   433→
   434→/// Computes calling-convention information for a function signature.
   435→///
   436→/// Walks the argument types, classifies each, and assigns registers (X0–X7
   437→/// for integer/pointer, V0–V7 for FP) or stack slots.  Also classifies the
   438→/// return value.
   439→///
   440→/// ## Return-value handling
   441→///
   442→/// - `void` → no registers
   443→/// - Integer/pointer ≤ 8 bytes → X0
   444→/// - Small struct (≤16 bytes, integer-like) → X0, X1
   445→/// - FP primitive → V0
   446→/// - HFA with N members → V0..V(N-1)
   447→/// - Large type (>16 bytes, not HFA) → indirect via X8
   448→pub fn compute_calling_conv(args: &[IRType], ret: &IRType) -> CallingConvInfo {
   449→    let mut x_reg_idx: u32 = 0;
   450→    let mut v_reg_idx: u32 = 0;
   451→    let mut stack_offset: i32 = 0;
   452→    let mut arg_locations = Vec::with_capacity(args.len());
   453→
   454→    // If the return value is indirect, X8 is used for the return-area
   455→    // pointer.  Per AAPCS64, this is passed as a hidden first argument in X8.
   456→    // We model it by setting x_reg_idx = 1 so that the first explicit
   457→    // argument starts at X0 (if not indirect) or X1 (since X8 is reserved).
   458→    // Actually, AAPCS64 says X8 is *separate* — it does not consume an
   459→    // argument register slot.  The integer argument registers X0–X7 are
   460→    // independent of X8.  So we don't need to adjust x_reg_idx.
   461→    let indirect_ret = classify_arg(ret) == ArgClass::Indirect;
   462→
   463→    for (i, arg_type) in args.iter().enumerate() {
   464→        let class = classify_arg(arg_type);
   465→        match class {
   466→            ArgClass::Integer => {
   467→                if x_reg_idx < 8 {
   468→                    arg_locations.push(ArgLocation {
   469→                        index: i,
   470→                        class,
   471→                        register: Some((RegisterClass::X, x_reg_idx)),
   472→                        stack_offset: None,
   473→                    });
   474→                    x_reg_idx += 1;
   475→                } else {
   476→                    // Spill to stack — each stack arg is 8-byte aligned.
   477→                    arg_locations.push(ArgLocation {
   478→                        index: i,
   479→                        class: ArgClass::Stack,
   480→                        register: None,
   481→                        stack_offset: Some(stack_offset),
   482→                    });
   483→                    stack_offset += 8;
   484→                }
   485→            }
   486→            ArgClass::FP => {
   487→                // HFA: may consume multiple V registers.
   488→                if let Some((elem_ty, count)) = arg_type.hfa_info() {
   489→                    let elem_size = size_of(elem_ty);
   490→                    if v_reg_idx + count as u32 <= 8 {
   491→                        // HFA fits in V registers — store the first register;
   492→                        // the remaining are consecutive.
   493→                        arg_locations.push(ArgLocation {
   494→                            index: i,
   495→                            class,
   496→                            register: Some((RegisterClass::V, v_reg_idx)),
   497→                            stack_offset: None,
   498→                        });
   499→                        v_reg_idx += count as u32;
   500→                    } else {
   501→                        // HFA spills to stack — each member is naturally aligned.
   502→                        let mut off = stack_offset;
   503→                        for _ in 0..count {
   504→                            let align = alignment_of(elem_ty) as i32;
   505→                            off = (off + align - 1) & !(align - 1);
   506→                            off += elem_size as i32;
   507→                        }
   508→                        arg_locations.push(ArgLocation {
   509→                            index: i,
   510→                            class: ArgClass::Stack,
   511→                            register: None,
   512→                            stack_offset: Some(stack_offset),
   513→                        });
   514→                        stack_offset = off;
   515→                    }
   516→                } else {
   517→                    // Single FP value.
   518→                    if v_reg_idx < 8 {
   519→                        arg_locations.push(ArgLocation {
   520→                            index: i,
   521→                            class,
   522→                            register: Some((RegisterClass::V, v_reg_idx)),
   523→                            stack_offset: None,
   524→                        });
   525→                        v_reg_idx += 1;
   526→                    } else {
   527→                        let align = alignment_of(arg_type) as i32;
   528→                        stack_offset = (stack_offset + align - 1) & !(align - 1);
   529→                        arg_locations.push(ArgLocation {
   530→                            index: i,
   531→                            class: ArgClass::Stack,
   532→                            register: None,
   533→                            stack_offset: Some(stack_offset),
   534→                        });
   535→                        stack_offset += size_of(arg_type) as i32;
   536→                    }
   537→                }
   538→            }
   539→            ArgClass::Indirect => {
   540→                // Pass a pointer to the value in an X register (or on stack).
   541→                if x_reg_idx < 8 {
   542→                    arg_locations.push(ArgLocation {
   543→                        index: i,
   544→                        class,
   545→                        register: Some((RegisterClass::X, x_reg_idx)),
   546→                        stack_offset: None,
   547→                    });
   548→                    x_reg_idx += 1;
   549→                } else {
   550→                    arg_locations.push(ArgLocation {
   551→                        index: i,
   552→                        class: ArgClass::Stack,
   553→                        register: None,
   554→                        stack_offset: Some(stack_offset),
   555→                    });
   556→                    stack_offset += 8;
   557→                }
   558→            }
   559→            ArgClass::Stack => {
   560→                let align = alignment_of(arg_type) as i32;
   561→                stack_offset = (stack_offset + align - 1) & !(align - 1);
   562→                arg_locations.push(ArgLocation {
   563→                    index: i,
   564→                    class,
   565→                    register: None,
   566→                    stack_offset: Some(stack_offset),
   567→                });
   568→                stack_offset += size_of(arg_type) as i32;
   569→            }
   570→        }
   571→    }
   572→
   573→    // Align stack args size to 16 bytes (AAPCS64 requirement).
   574→    let stack_args_size = ((stack_offset as usize) + 15) & !15;
   575→
   576→    // Compute return-value location.
   577→    let ret_class = classify_arg(ret);
   578→    let ret_location = match ret_class {
   579→        ArgClass::Integer => {
   580→            if *ret == IRType::Void {
   581→                RetLocation {
   582→                    class: ret_class,
   583→                    registers: vec![],
   584→                }
   585→            } else {
   586→                let sz = size_of(ret);
   587→                if sz <= 8 {
   588→                    RetLocation {
   589→                        class: ret_class,
   590→                        registers: vec![(RegisterClass::X, 0)],
   591→                    }
   592→                } else {
   593→                    // Struct ≤ 16 bytes returned in X0 + X1.
   594→                    RetLocation {
   595→                        class: ret_class,
   596→                        registers: vec![(RegisterClass::X, 0), (RegisterClass::X, 1)],
   597→                    }
   598→                }
   599→            }
   600→        }
   601→        ArgClass::FP => {
   602→            if let Some((_, count)) = ret.hfa_info() {
   603→                let regs: Vec<(RegisterClass, u32)> =
   604→                    (0..count as u32).map(|i| (RegisterClass::V, i)).collect();
   605→                RetLocation {
   606→                    class: ret_class,
   607→                    registers: regs,
   608→                }
   609→            } else {
   610→                RetLocation {
   611→                    class: ret_class,
   612→                    registers: vec![(RegisterClass::V, 0)],
   613→                }
   614→            }
   615→        }
   616→        ArgClass::Indirect => RetLocation {
   617→            class: ret_class,
   618→            registers: vec![(RegisterClass::X, 8)],
   619→        },
   620→        ArgClass::Stack => {
   621→            // Stack return is unusual; model as indirect for now.
   622→            RetLocation {
   623→                class: ArgClass::Indirect,
   624→                registers: vec![(RegisterClass::X, 8)],
   625→            }
   626→        }
   627→    };
   628→
   629→    let _ = indirect_ret; // X8 reservation is implicit via ret_location.
   630→
   631→    CallingConvInfo {
   632→        arg_locations,
   633→        ret_location,
   634→        stack_args_size,
   635→    }
   636→}

   642→/// A named slot in the stack frame.
   643→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   644→pub struct StackSlot {
   645→    /// Human-readable name for this slot.
   646→    pub name: String,
   647→    /// Offset from the frame pointer (FP / X29).  Negative values are in the
   648→    /// callee's frame (locals, saved regs); positive values point into the
   649→    /// caller's stack-argument area.
   650→    pub offset: i32,
   651→    /// Size of this slot in bytes.
   652→    pub size: usize,
   653→    /// Alignment of this slot in bytes.
   654→    pub alignment: usize,
   655→}
   656→
   657→/// Complete stack-frame layout for a function.
   658→///
   659→/// The stack frame layout follows the AAPCS64 convention:
   660→///
   661→/// ```text
   662→/// Higher addresses
   663→///   ┌─────────────────────┐
   664→///   │ Incoming stack args  │  FP+16, FP+24, …
   665→///   ├─────────────────────┤
   666→///   │ Saved FP (X29)       │  FP+0
   667→///   │ Saved LR (X30)       │  FP+8
   668→///   ├─────────────────────┤
   669→///   │ Callee-saved regs    │  FP-8, FP-16, …
   670→///   │ (X19..X28)           │
   671→///   ├─────────────────────┤
   672→///   │ Local variables      │  from Alloc instructions
   673→///   │ (aligned)            │
   674→///   ├─────────────────────┤
   675→///   │ Outgoing stack args  │  bottom of frame (for calls)
   676→///   │ (for nested calls)   │
   677→///   └─────────────────────┘  ← SP
   678→/// Lower addresses
   679→/// ```
   680→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   681→pub struct StackLayout {
   682→    /// Total stack frame size in bytes (always 16-byte aligned).
   683→    pub total_size: usize,
   684→    /// Slots for callee-saved registers.
   685→    pub callee_save_slots: Vec<StackSlot>,
   686→    /// Slots for local variables (from `Alloc` instructions).
   687→    pub local_slots: Vec<StackSlot>,
   688→    /// Slot for outgoing stack arguments to callees (single contiguous area).
   689→    pub outgoing_args_slot: Option<StackSlot>,
   690→    /// Slot for the saved FP (X29) — always at FP+0.
   691→    pub fp_slot: StackSlot,
   692→    /// Slot for the saved LR (X30) — always at FP+8.
   693→    pub lr_slot: StackSlot,
   694→    /// Number of callee-saved registers that are spilled.
   695→    pub callee_saves_count: usize,
   696→}
   697→
   698→/// Computes the stack-frame layout for an IR function.
   699→///
   700→/// This scans the function body for `Alloc` instructions (local variables) and
   701→/// `Call` instructions (outgoing stack arguments), and lays out the frame
   702→/// according to AAPCS64 conventions.
   703→///
   704→/// ## Parameters
   705→///
   706→/// - `func`: The IR function to compute the layout for.
   707→/// - `callee_saves_count`: How many callee-saved registers (X19–X28) the
   708→///   function uses.  Set by the register allocator.  Each consumes 8 bytes.
   709→/// - `call_arg_types`: For each `Call` instruction in the function, the
   710→///   argument types.  This is used to compute the outgoing stack-argument
   711→///   area size.  If empty or if calls have no stack args, no outgoing area
   712→///   is needed.
   713→pub fn compute_stack_layout(func: &IRFunction) -> StackLayout {
   714→    compute_stack_layout_with_info(func, 0, &[])
   715→}
   716→
   717→/// Computes the stack-frame layout with additional information from the
   718→/// register allocator and call-site type information.
   719→///
   720→/// See [`compute_stack_layout`] for the simplified version.
   721→pub fn compute_stack_layout_with_info(
   722→    func: &IRFunction,
   723→    callee_saves_count: usize,
   724→    call_arg_types: &[Vec<IRType>],
   725→) -> StackLayout {
   726→    // Fixed slots: saved FP at FP+0, saved LR at FP+8.
   727→    let fp_slot = StackSlot {
   728→        name: "saved_fp".to_string(),
   729→        offset: 0,
   730→        size: 8,
   731→        alignment: 8,
   732→    };
   733→    let lr_slot = StackSlot {
   734→        name: "saved_lr".to_string(),
   735→        offset: 8,
   736→        size: 8,
   737→        alignment: 8,
   738→    };
   739→
   740→    // Current negative offset from FP, growing downward.
   741→    let mut offset: i32 = 0;
   742→
   743→    // --- Callee-saved registers ---
   744→    let mut callee_save_slots = Vec::new();
   745→    for i in 0..callee_saves_count {
   746→        offset -= 8;
   747→        callee_save_slots.push(StackSlot {
   748→            name: format!("callee_save_{}", i),
   749→            offset,
   750→            size: 8,
   751→            alignment: 8,
   752→        });
   753→    }
   754→
   755→    // --- Local variables (from Alloc instructions) ---
   756→    let mut local_slots = Vec::new();
   757→    let mut alloc_index = 0;
   758→    for block in &func.blocks {
   759→        for instr in &block.instructions {
   760→            if let IRInstr::Alloc { dst, size } = instr {
   761→                let size = *size as usize;
   762→                let align = if size >= 8 {
   763→                    8usize
   764→                } else {
   765→                    size.next_power_of_two()
   766→                };
   767→                // Align offset.
   768→                offset = (offset - align as i32) & !(align as i32 - 1);
   769→                offset -= size as i32;
   770→                // Re-align offset to the start of the slot.
   771→                offset &= !(align as i32 - 1);
   772→                let name = match dst {
   773→                    IRValue::Register(id) => format!("local_%v{}", id),
   774→                    _ => format!("local_{}", alloc_index),
   775→                };
   776→                local_slots.push(StackSlot {
   777→                    name,
   778→                    offset,
   779→                    size,
   780→                    alignment: align,
   781→                });
   782→                alloc_index += 1;
   783→            }
   784→        }
   785→    }
   786→
   787→    // --- Outgoing stack arguments ---
   788→    // Compute the maximum stack-args size across all call sites.
   789→    let mut max_outgoing_stack = 0usize;
   790→    for arg_types in call_arg_types {
   791→        let cc = compute_calling_conv(arg_types, &IRType::Void);
   792→        max_outgoing_stack = max_outgoing_stack.max(cc.stack_args_size);
   793→    }
   794→
   795→    let outgoing_args_slot = if max_outgoing_stack > 0 {
   796→        // Align to 16 bytes.
   797→        let size = (max_outgoing_stack + 15) & !15;
   798→        offset -= size as i32;
   799→        Some(StackSlot {
   800→            name: "outgoing_args".to_string(),
   801→            offset,
   802→            size,
   803→            alignment: 16,
   804→        })
   805→    } else {
   806→        None
   807→    };
   808→
   809→    // Total frame size: absolute value of the lowest offset, plus 16 for
   810→    // FP/LR pair.  The frame pointer is at the saved-FP position; everything
   811→    // below is at negative offsets from FP.
   812→    let total_raw = ((-offset) as usize) + 16; // 16 = FP + LR
   813→    let total_size = (total_raw + 15) & !15; // 16-byte align.
   814→
   815→    StackLayout {
   816→        total_size,
   817→        callee_save_slots,
   818→        local_slots,
   819→        outgoing_args_slot,
   820→        fp_slot,
   821→        lr_slot,
   822→        callee_saves_count,
   823→    }
   824→}

   830→/// A value that can appear as an operand in an IR instruction.
   831→#[derive(Debug, Clone, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
   832→pub enum IRValue {
   833→    /// A virtual register identified by a numeric ID.
   834→    Register(u32),
   835→    /// An immediate constant.
   836→    Immediate(i64),
   837→    /// A memory address (absolute).  Stored as `u64` for 64-bit targets;
   838→    /// 32-bit backends should use [`IRValue::as_address_32bit`] to safely
   839→    /// truncate and validate the value.
   840→    Address(u64),
   841→    /// A named label (for branch targets).
   842→    Label(String),
   843→}
   844→
   845→impl IRValue {
   846→    /// Returns `true` if this is a virtual register.
   847→    pub fn is_register(&self) -> bool {
   848→        matches!(self, IRValue::Register(_))
   849→    }
   850→
   851→    /// Returns `true` if this is an immediate constant.
   852→    pub fn is_immediate(&self) -> bool {
   853→        matches!(self, IRValue::Immediate(_))
   854→    }
   855→
   856→    /// Extract the register ID, if this is a register value.
   857→    pub fn as_register(&self) -> Option<u32> {
   858→        match self {
   859→            IRValue::Register(id) => Some(*id),
   860→            _ => None,
   861→        }
   862→    }
   863→
   864→    /// Extract the immediate value, if this is an immediate.
   865→    pub fn as_immediate(&self) -> Option<i64> {
   866→        match self {
   867→            IRValue::Immediate(v) => Some(*v),
   868→            _ => None,
   869→        }
   870→    }
   871→
   872→    /// Extract the address as a 32-bit value, if this is an Address.
   873→    ///
   874→    /// Returns `None` if the address exceeds the 32-bit range, which would
   875→    /// be invalid on 32-bit targets like Wasm32 or ARM32.
   876→    pub fn as_address_32bit(&self) -> Option<u32> {
   877→        match self {
   878→            IRValue::Address(a) => {
   879→                if *a > u32::MAX as u64 {
   880→                    None
   881→                } else {
   882→                    Some(*a as u32)
   883→                }
   884→            }
   885→            _ => None,
   886→        }
   887→    }
   888→}
   889→
   890→impl fmt::Display for IRValue {
   891→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   892→        match self {
   893→            IRValue::Register(id) => write!(f, "%v{}", id),
   894→            IRValue::Immediate(v) => write!(f, "{}", v),
   895→            IRValue::Address(a) => write!(f, "0x{:016x}", a),
   896→            IRValue::Label(name) => write!(f, "@{}", name),
   897→        }
   898→    }
   899→}

  1370→impl IRInstr {
  1371→    /// Returns the set of virtual-register IDs that this instruction defines
  1372→    /// (writes to).
  1373→    pub fn defined_regs(&self) -> Vec<u32> {
  1374→        match self {
  1375→            IRInstr::Load { dst, .. }
  1376→            | IRInstr::BinOp { dst, .. }
  1377→            | IRInstr::UnaryOp { dst, .. }
  1378→            | IRInstr::Alloc { dst, .. }
  1379→            | IRInstr::Cast { dst, .. }
  1380→            | IRInstr::Phi { dst, .. }
  1381→            | IRInstr::GetAddress { dst, .. }
  1382→            | IRInstr::Offset { dst, .. }
  1383→            | IRInstr::Select { dst, .. } => dst.as_register().into_iter().collect(),
  1384→            IRInstr::Call { dst, .. } => dst
  1385→                .as_ref()
  1386→                .and_then(|v| v.as_register())
  1387→                .into_iter()
  1388→                .collect(),
  1389→            IRInstr::Add { dst, .. }
  1390→            | IRInstr::Sub { dst, .. }
  1391→            | IRInstr::Mul { dst, .. }
  1392→            | IRInstr::Div { dst, .. }
  1393→            | IRInstr::Cmp { dst, .. } => dst.as_register().into_iter().collect(),
  1394→            IRInstr::Store { .. }
  1395→            | IRInstr::Free { .. }
  1396→            | IRInstr::Ret { .. }
  1397→            | IRInstr::Branch { .. }
  1398→            | IRInstr::CondBranch { .. } => vec![],
  1399→        }
  1400→    }
  1401→
  1402→    /// Returns the set of virtual-register IDs that this instruction uses
  1403→    /// (reads from).
  1404→    pub fn used_regs(&self) -> Vec<u32> {
  1405→        match self {
  1406→            IRInstr::Load { addr, .. } => addr.as_register().into_iter().collect(),
  1407→            IRInstr::Store { value, addr, .. } => {
  1408→                let mut r = value.as_register().into_iter().collect::<Vec<_>>();
  1409→                r.extend(addr.as_register());
  1410→                r
  1411→            }
  1412→            IRInstr::BinOp { lhs, rhs, .. }
  1413→            | IRInstr::Add { lhs, rhs, .. }
  1414→            | IRInstr::Sub { lhs, rhs, .. }
  1415→            | IRInstr::Mul { lhs, rhs, .. }
  1416→            | IRInstr::Div { lhs, rhs, .. }
  1417→            | IRInstr::Cmp { lhs, rhs, .. } => {
  1418→                let mut r = lhs.as_register().into_iter().collect::<Vec<_>>();
  1419→                r.extend(rhs.as_register());
  1420→                r
  1421→            }
  1422→            IRInstr::UnaryOp { operand, .. } => operand.as_register().into_iter().collect(),
  1423→            IRInstr::Call { args, .. } => args.iter().filter_map(|v| v.as_register()).collect(),
  1424→            IRInstr::Alloc { .. } | IRInstr::GetAddress { .. } => vec![],
  1425→            IRInstr::Free { ptr } => ptr.as_register().into_iter().collect(),
  1426→            IRInstr::Cast { src, .. } => src.as_register().into_iter().collect(),
  1427→            IRInstr::Phi { incoming, .. } => incoming
  1428→                .iter()
  1429→                .filter_map(|(v, _)| v.as_register())
  1430→                .collect(),
  1431→            IRInstr::Offset { base, offset, .. } => {
  1432→                let mut r = base.as_register().into_iter().collect::<Vec<_>>();
  1433→                r.extend(offset.as_register());
  1434→                r
  1435→            }
  1436→            IRInstr::Select {
  1437→                cond,
  1438→                true_val,
  1439→                false_val, ty: _,
  1440→                ..
  1441→            } => {
  1442→                let mut r = cond.as_register().into_iter().collect::<Vec<_>>();
  1443→                r.extend(true_val.as_register());
  1444→                r.extend(false_val.as_register());
  1445→                r
  1446→            }
  1447→            IRInstr::Ret { values } => values.iter().filter_map(|v| v.as_register()).collect(),
  1448→            IRInstr::Branch { .. } => vec![],
  1449→            IRInstr::CondBranch { cond, .. } => cond.as_register().into_iter().collect(),
  1450→        }
  1451→    }
  1452→}

  1454→impl fmt::Display for IRInstr {
  1455→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
  1456→        match self {
  1457→            IRInstr::Load { dst, addr, offset, ty } => {
  1458→                if *offset != 0 {
  1459→                    write!(f, "{} = load [{} + {}] ({})", dst, addr, offset, ty)
  1460→                } else {
  1461→                    write!(f, "{} = load {} ({})", dst, addr, ty)
  1462→                }
  1463→            }
  1464→            IRInstr::Store { value, addr, offset, ty } => {
  1465→                if *offset != 0 {
  1466→                    write!(f, "store {}, [{} + {}] ({})", value, addr, offset, ty)
  1467→                } else {
  1468→                    write!(f, "store {}, {} ({})", value, addr, ty)
  1469→                }
  1470→            }
  1471→            IRInstr::BinOp { op, dst, lhs, rhs, ty: _ } => {
  1472→                write!(f, "{} = {} {}, {}", dst, op, lhs, rhs)
  1473→            }
  1474→            IRInstr::UnaryOp { op, dst, operand, ty: _ } => {
  1475→                write!(f, "{} = {} {}", dst, op, operand)
  1476→            }
  1477→            IRInstr::Call { dst, func, args } => {
  1478→                let args_str = args
  1479→                    .iter()
  1480→                    .map(|a| format!("{}", a))
  1481→                    .collect::<Vec<_>>()
  1482→                    .join(", ");
  1483→                match dst {
  1484→                    Some(d) => write!(f, "{} = call @{}({})", d, func, args_str),
  1485→                    None => write!(f, "call @{}({})", func, args_str),
  1486→                }
  1487→            }
  1488→            IRInstr::Alloc { dst, size } => write!(f, "{} = alloc {}", dst, size),
  1489→            IRInstr::Free { ptr } => write!(f, "free {}", ptr),
  1490→            IRInstr::Cast { kind, dst, src } => {
  1491→                write!(f, "{} = {} {}", dst, kind, src)
  1492→            }
  1493→            IRInstr::Phi { dst, incoming } => {
  1494→                let pairs = incoming
  1495→                    .iter()
  1496→                    .map(|(v, b)| format!("[{}, @{}]", v, b))
  1497→                    .collect::<Vec<_>>()
  1498→                    .join(", ");
  1499→                write!(f, "{} = phi {}", dst, pairs)
  1500→            }
  1501→            IRInstr::GetAddress { dst, name } => {
  1502→                write!(f, "{} = getaddress @{}", dst, name)
  1503→            }
  1504→            IRInstr::Offset { dst, base, offset } => {
  1505→                write!(f, "{} = offset {}, {}", dst, base, offset)
  1506→            }
  1507→            IRInstr::Select {
  1508→                dst,
  1509→                cond,
  1510→                true_val,
  1511→                false_val, ty: _,
  1512→            } => {
  1513→                write!(f, "{} = select {}, {}, {}", dst, cond, true_val, false_val)
  1514→            }
  1515→            IRInstr::Add { dst, lhs, rhs, ty: _ } => write!(f, "{} = add {}, {}", dst, lhs, rhs),
  1516→            IRInstr::Sub { dst, lhs, rhs, ty: _ } => write!(f, "{} = sub {}, {}", dst, lhs, rhs),
  1517→            IRInstr::Mul { dst, lhs, rhs, ty: _ } => write!(f, "{} = mul {}, {}", dst, lhs, rhs),
  1518→            IRInstr::Div { dst, lhs, rhs, ty: _ } => write!(f, "{} = div {}, {}", dst, lhs, rhs),
  1519→            IRInstr::Cmp {
  1520→                kind,
  1521→                dst,
  1522→                lhs,
  1523→                rhs, ty: _,
  1524→            } => {
  1525→                write!(f, "{} = {} {}, {}", dst, kind, lhs, rhs)
  1526→            }
  1527→            IRInstr::Ret { values } => {
  1528→                let vals_str = values
  1529→                    .iter()
  1530→                    .map(|v| format!("{}", v))
  1531→                    .collect::<Vec<_>>()
  1532→                    .join(", ");
  1533→                write!(f, "ret {}", vals_str)
  1534→            }
  1535→            IRInstr::Branch { target } => write!(f, "br @{}", target),
  1536→            IRInstr::CondBranch {
  1537→                cond,
  1538→                true_target,
  1539→                false_target,
  1540→            } => {
  1541→                write!(f, "br {}, @{}, @{}", cond, true_target, false_target)
  1542→            }
  1543→        }
  1544→    }
  1545→}

  1551→/// A block terminator — the last "instruction" in an `IRBlock` that transfers
  1552→/// control flow.
  1553→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
  1554→pub enum IRTerminator {
  1555→    /// Unconditional jump to a label.
  1556→    Jump(String),
  1557→    /// Conditional branch: if `cond` is non-zero, go to `true_block`;
  1558→    /// otherwise go to `false_block`.
  1559→    Branch {
  1560→        /// Condition value; branches to `true_block` if non-zero.
  1561→        cond: IRValue,
  1562→        /// Target block label when the condition is true.
  1563→        true_block: String,
  1564→        /// Target block label when the condition is false.
  1565→        false_block: String,
  1566→    },
  1567→    /// Return from the current function with optional values.
  1568→    Return(Vec<IRValue>),
  1569→    /// Unreachable code marker (e.g. after a diverging call).
  1570→    Unreachable,
  1571→    /// Switch dispatch: branch to one of several targets based on the
  1572→    /// discriminator value, or fall through to `default`.
  1573→    Switch {
  1574→        /// Discriminator value.
  1575→        discr: IRValue,
  1576→        /// (value, target_label) pairs.
  1577→        targets: Vec<(i64, String)>,
  1578→        /// Default target if no value matches.
  1579→        default: String,
  1580→    },
  1581→    /// Invoke: call a function that may throw, with separate normal and
  1582→    /// unwind continuations.
  1583→    Invoke {
  1584→        /// Destination register for the return value.
  1585→        dst: Option<IRValue>,
  1586→        /// Function name.
  1587→        func: String,
  1588→        /// Arguments.
  1589→        args: Vec<IRValue>,
  1590→        /// Normal continuation label.
  1591→        normal: String,
  1592→        /// Unwind (exception) continuation label.
  1593→        unwind: String,
  1594→    },
  1595→    /// Tail call: jump to the callee, reusing the current stack frame.
  1596→    TailCall {
  1597→        /// Function name.
  1598→        func: String,
  1599→        /// Arguments.
  1600→        args: Vec<IRValue>,
  1601→    },
  1602→    /// Resume unwinding with the given exception value.
  1603→    Resume {
  1604→        /// Exception value to resume with.
  1605→        value: IRValue,
  1606→    },
  1607→}
  1608→
  1609→impl IRTerminator {
  1610→    /// Returns the labels of all successor blocks referenced by this terminator.
  1611→    pub fn successor_labels(&self) -> Vec<&str> {
  1612→        match self {
  1613→            IRTerminator::Jump(target) => vec![target],
  1614→            IRTerminator::Branch {
  1615→                true_block,
  1616→                false_block,
  1617→                ..
  1618→            } => vec![true_block, false_block],
  1619→            IRTerminator::Return(_) | IRTerminator::Unreachable => vec![],
  1620→            IRTerminator::Switch {
  1621→                targets, default, ..
  1622→            } => {
  1623→                let mut labels: Vec<&str> = targets.iter().map(|(_, l)| l.as_str()).collect();
  1624→                labels.push(default);
  1625→                labels
  1626→            }
  1627→            IRTerminator::Invoke { normal, unwind, .. } => vec![normal, unwind],
  1628→            IRTerminator::TailCall { .. } | IRTerminator::Resume { .. } => vec![],
  1629→        }
  1630→    }
  1631→}
  1632→
  1633→impl fmt::Display for IRTerminator {
  1634→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
  1635→        match self {
  1636→            IRTerminator::Jump(target) => write!(f, "jump @{}", target),
  1637→            IRTerminator::Branch {
  1638→                cond,
  1639→                true_block,
  1640→                false_block,
  1641→            } => {
  1642→                write!(f, "br {}, @{}, @{}", cond, true_block, false_block)
  1643→            }
  1644→            IRTerminator::Return(vals) => {
  1645→                let vals_str = vals
  1646→                    .iter()
  1647→                    .map(|v| format!("{}", v))
  1648→                    .collect::<Vec<_>>()
  1649→                    .join(", ");
  1650→                write!(f, "ret {}", vals_str)
  1651→            }
  1652→            IRTerminator::Unreachable => write!(f, "unreachable"),
  1653→            IRTerminator::Switch {
  1654→                discr,
  1655→                targets,
  1656→                default,
  1657→            } => {
  1658→                let pairs = targets
  1659→                    .iter()
  1660→                    .map(|(v, l)| format!("{}: @{}", v, l))
  1661→                    .collect::<Vec<_>>()
  1662→                    .join(", ");
  1663→                write!(f, "switch {}, [{}] default @{}", discr, pairs, default)
  1664→            }
  1665→            IRTerminator::Invoke {
  1666→                dst,
  1667→                func,
  1668→                args,
  1669→                normal,
  1670→                unwind,
  1671→            } => {
  1672→                let args_str = args
  1673→                    .iter()
  1674→                    .map(|a| format!("{}", a))
  1675→                    .collect::<Vec<_>>()
  1676→                    .join(", ");
  1677→                match dst {
  1678→                    Some(d) => write!(
  1679→                        f,
  1680→                        "invoke {} = @{}({}) normal @{} unwind @{}",
  1681→                        d, func, args_str, normal, unwind
  1682→                    ),
  1683→                    None => write!(
  1684→                        f,
  1685→                        "invoke @{}({}) normal @{} unwind @{}",
  1686→                        func, args_str, normal, unwind
  1687→                    ),
  1688→                }
  1689→            }
  1690→            IRTerminator::TailCall { func, args } => {
  1691→                let args_str = args
  1692→                    .iter()
  1693→                    .map(|a| format!("{}", a))
  1694→                    .collect::<Vec<_>>()
  1695→                    .join(", ");
  1696→                write!(f, "tailcall @{}({})", func, args_str)
  1697→            }
  1698→            IRTerminator::Resume { value } => write!(f, "resume {}", value),
  1699→        }
  1700→    }
  1701→}

  1707→/// A basic block within an IR function.
  1708→///
  1709→/// Execution enters at the top and falls through each instruction.  The block
  1710→/// always ends with exactly one terminator.
  1711→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
  1712→pub struct IRBlock {
  1713→    /// Block label (used as a branch target).
  1714→    pub label: String,
  1715→    /// Ordered instructions in this block.
  1716→    pub instructions: Vec<IRInstr>,
  1717→    /// The terminating control-flow instruction.
  1718→    pub terminator: IRTerminator,
  1719→    /// Labels of predecessor blocks (populated after CFG construction).
  1720→    pub predecessors: HashSet<String>,
  1721→    /// Labels of successor blocks (populated after CFG construction).
  1722→    pub successors: HashSet<String>,
  1723→}
  1724→
  1725→impl IRBlock {
  1726→    /// Create a new empty block with the given label and an `Unreachable`
  1727→    /// terminator placeholder (callers should replace it).
  1728→    pub fn new(label: impl Into<String>) -> Self {
  1729→        Self {
  1730→            label: label.into(),

  1838→    /// Rebuild predecessor/successor sets for all blocks from terminators.
  1839→    pub fn rebuild_cfg(&mut self) {
  1840→        let label_to_idx: HashMap<String, usize> = self
  1841→            .blocks
  1842→            .iter()
  1843→            .enumerate()
  1844→            .map(|(i, b)| (b.label.clone(), i))
  1845→            .collect();
  1846→
  1847→        for block in &mut self.blocks {
  1848→            block.predecessors.clear();
  1849→            block.successors.clear();
  1850→        }
  1851→
  1852→        // Collect edge data first to avoid borrow conflicts.
  1853→        let mut edges: Vec<(usize, usize)> = Vec::new();
  1854→        for i in 0..self.blocks.len() {
  1855→            self.blocks[i].update_successors_from_terminator();
  1856→            let succ_labels: Vec<String> = self.blocks[i].successors.iter().cloned().collect();
  1857→            for succ_label in succ_labels {
  1858→                if let Some(&succ_idx) = label_to_idx.get(&succ_label) {
  1859→                    edges.push((i, succ_idx));
  1860→                }
  1861→            }
  1862→        }
  1863→
  1864→        // Now apply predecessor edges.
  1865→        let src_labels: Vec<(usize, String)> = self
  1866→            .blocks
  1867→            .iter()
  1868→            .enumerate()
  1869→            .map(|(i, b)| (i, b.label.clone()))
  1870→            .collect();
  1871→        for (src_idx, tgt_idx) in edges {
  1872→            self.blocks[tgt_idx]
  1873→                .predecessors
  1874→                .insert(src_labels[src_idx].1.clone());
  1875→        }
  1876→    }
