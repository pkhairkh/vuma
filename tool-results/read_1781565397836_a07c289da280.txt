     1в†’//! Structured JSON diagnostics output for LLM integration.
     2в†’//!
     3в†’//! This module provides machine-readable diagnostic messages that can be
     4в†’//! serialized to JSON, enabling LLM-based tools and IDE integrations to
     5в†’//! programmatically consume compiler errors, warnings, and hints.
     6в†’//!
     7в†’//! # Architecture
     8в†’//!
     9в†’//! Each diagnostic carries a structured error code (`"E001"`, `"W001"`, etc.),
    10в†’//! a severity level, a human-readable message, a source identifier (which
    11в†’//! compiler stage produced it), a precise source location, optional related
    12в†’//! information, quick-fix suggestions with edit ranges, and an optional
    13в†’//! causal chain linking to parent diagnostics.
    14в†’//!
    15в†’//! # Error Catalog
    16в†’//!
    17в†’//! The diagnostic codes are organized into the following ranges:
    18в†’//!
    19в†’//! | Range    | Category       | Description                              |
    20в†’//! |----------|----------------|------------------------------------------|
    21в†’//! | E001вЂ“E030| Compilation    | Syntax, type, name resolution errors     |
    22в†’//! | E031вЂ“E040| Codegen        | Register allocation, encoding, relocation|
    23в†’//! | E041вЂ“E050| Verification   | Invariant violations, proof failures     |
    24в†’//! | W001вЂ“W010| Warnings       | Unused vars, performance hints            |
    25в†’//! | I001вЂ“I005| Informational  | General compiler information messages     |
    26в†’//!
    27в†’//! # Example JSON output
    28в†’//!
    29в†’//! ```json
    30в†’//! [
    31в†’//!   {
    32в†’//!     "code": "E002",
    33в†’//!     "severity": "error",
    34в†’//!     "message": "undefined variable `foo`",
    35в†’//!     "source": "parser",
    36в†’//!     "location": {
    37в†’//!       "file": "main.vu",
    38в†’//!       "start_line": 10,
    39в†’//!       "start_col": 4,
    40в†’//!       "end_line": 10,
    41в†’//!       "end_col": 7
    42в†’//!     },
    43в†’//!     "related": [],
    44в†’//!     "suggestions": ["did you mean `for`?"],
    45в†’//!     "chain": []
    46в†’//!   }
    47в†’//! ]
    48в†’//! ```
    49в†’//!
    50в†’//! # Integration
    51в†’//!
    52в†’//! Use [`from_parse_error`] to convert parser errors, [`from_vuma_error`] for
    53в†’//! pipeline-level errors, and [`from_codegen_error`] for code-generation
    54в†’//! failures.

    69в†’/// Severity level for a machine-readable diagnostic.

   542в†’/// A machine-readable diagnostic with structured fields for LLM consumption.
   543в†’///
   544в†’/// This is the primary output type for JSON diagnostics. Every field is
   545в†’/// serializable and carries enough information for an LLM or IDE to
   546в†’/// understand, categorize, and potentially auto-fix the issue.
   547в†’///
   548в†’/// # Error Chaining
   549в†’///
   550в†’/// Diagnostics support causal chains via the [`chain`](VumaDiagnostic::chain)
   551в†’/// method. A chain represents the cause of a diagnostic: for example,
   552в†’/// "E003: Type mismatch" may be caused by "E023: Unknown type 'int'".
   553в†’///
   554в†’/// # Suggestions
   555в†’///
   556в†’/// Each diagnostic can carry zero or more [`Suggestion`] values, which may
   557в†’/// include precise edit ranges for automatic code fixes.
   558в†’#[derive(Debug, Clone, Serialize, Deserialize)]
   559в†’pub struct VumaDiagnostic {
   560в†’    /// Structured diagnostic code, e.g. `"E001"`, `"W001"`.
   561в†’    pub code: String,
   562в†’    /// Severity level.
   563в†’    pub severity: DiagnosticSeverity,
   564в†’    /// Human-readable message (suitable for display).
   565в†’    pub message: String,
   566в†’    /// Compiler stage that produced this diagnostic
   567в†’    /// (`"parser"`, `"scg"`, `"ir"`, `"codegen"`, etc.).
   568в†’    pub source: String,
   569в†’    /// Precise source location.
   570в†’    pub location: DiagnosticSourceLocation,
   571в†’    /// Related source locations with explanatory messages.
   572в†’    pub related: Vec<RelatedInfo>,
   573в†’    /// Quick-fix suggestions (structured with optional edit ranges).
   574в†’    #[serde(default)]
   575в†’    pub suggestions: Vec<Suggestion>,
   576в†’    /// Causal chain: list of diagnostics that caused this one.
   577в†’    /// The first element is the immediate cause, the last is the root cause.
   578в†’    #[serde(default)]
   579в†’    pub chain: Vec<Box<VumaDiagnostic>>,
   580в†’    /// Backward-compatible plain-text suggestions (stored alongside
   581в†’    /// structured suggestions for JSON serialization compatibility).
   582в†’    #[serde(default, skip_serializing_if = "Vec::is_empty")]
   583в†’    pub legacy_suggestions: Vec<String>,
   584в†’}

   782в†’                "  \x1b[32mhelp\x1b[0m: {}",

   813в†’                out.push_str(&format!(
   814в†’                    "{}  \x1b[2m--> {}:{}:{}\x1b[0m\n",

   826в†’    /// - `range`: 0-based Position range
   827в†’    /// - `severity`: LSP numeric severity (1=Error, 2=Warning, 3=Info, 4=Hint)
   828в†’    /// - `code`: the diagnostic code string
   829в†’    /// - `source`: the compiler stage

   832в†’    pub fn to_lsp(&self) -> serde_json::Value {
   833в†’        use serde_json::json;
   834в†’
   835в†’        let start_line = if self.location.start_line > 0 {
   836в†’            self.location.start_line - 1
   837в†’        } else {
   838в†’            0
   839в†’        };
   840в†’        let start_col = if self.location.start_col > 0 {
   841в†’            self.location.start_col - 1
   842в†’        } else {
   843в†’            0
   844в†’        };
   845в†’        let end_line = if self.location.end_line > 0 {
   846в†’            self.location.end_line - 1
   847в†’        } else {
   848в†’            0
   849в†’        };
   850в†’        let end_col = if self.location.end_col > 0 {
   851в†’            self.location.end_col - 1
   852в†’        } else {
   853в†’            0
   854в†’        };
   855в†’
   856в†’        let mut diagnostic = json!({
   857в†’            "range": {
   858в†’                "start": { "line": start_line, "character": start_col },
   859в†’                "end": { "line": end_line, "character": end_col }
   860в†’            },
   861в†’            "severity": self.severity.to_lsp_severity(),
   862в†’            "code": self.code,
   863в†’            "source": self.source,
   864в†’            "message": self.message,
   865в†’        });
   866в†’

   868в†’        if !self.related.is_empty() {
   869в†’            let related: Vec<serde_json::Value> = self.related.iter().map(|r| {
   870в†’                let r_start_line = if r.location.start_line > 0 { r.location.start_line - 1 } else { 0 };
   871в†’                let r_start_col = if r.location.start_col > 0 { r.location.start_col - 1 } else { 0 };
   872в†’                let r_end_line = if r.location.end_line > 0 { r.location.end_line - 1 } else { 0 };
   873в†’                let r_end_col = if r.location.end_col > 0 { r.location.end_col - 1 } else { 0 };
   874в†’                json!({
   875в†’                    "location": {
   876в†’                        "uri": format!("file://{}", r.location.file),
   877в†’                        "range": {
   878в†’                            "start": { "line": r_start_line, "character": r_start_col },
   879в†’                            "end": { "line": r_end_line, "character": r_end_col }
   880в†’                        }
   881в†’                    },
   882в†’                    "message": r.message
   883в†’                })
   884в†’            }).collect();
   885в†’            diagnostic["relatedInformation"] = json!(related);
   886в†’        }
   887в†’
   888в†’        // Add LSP CodeDescription with href if it's a known code
   889в†’        let desc = code_description(&self.code);
   890в†’        if desc != "Unknown diagnostic code" {
   891в†’            diagnostic["codeDescription"] = json!({
   892в†’                "href": format!("https://vuma.dev/docs/diagnostics/{}", self.code)
   893в†’            });
   894в†’        }
   895в†’
   896в†’        // Add tags for warnings (Unnecessary = 1, Deprecated = 2)
   897в†’        let mut tags: Vec<u32> = Vec::new();
   898в†’        match self.code.as_str() {
   899в†’            "W001" | "W004" | "W009" => tags.push(1), // Unnecessary
   900в†’            "W008" => tags.push(2),                     // Deprecated
   901в†’            _ => {}
   902в†’        }
   903в†’        if !tags.is_empty() {
   904в†’            diagnostic["tags"] = json!(tags);
   905в†’        }
   906в†’
   907в†’        diagnostic
   908в†’    }
   909в†’}

  1075в†’/// Convert a [`ParseError`] to a [`VumaDiagnostic`].
  1076в†’///
  1077в†’/// `source` is the full source text (used to compute line/column from byte
  1078в†’/// offsets). `file` is the optional file path.
  1079в†’pub fn from_parse_error(
  1080в†’    err: &ParseError,
  1081в†’    source: &str,
  1082в†’    file: Option<&str>,
  1083в†’) -> VumaDiagnostic {
  1084в†’    let code = code_for_parse_error_kind(&err.kind).to_string();
  1085в†’    let loc = offset_to_location(source, err.span.start, file);
  1086в†’
  1087в†’    // Compute end position from span end.
  1088в†’    let end_loc = if err.span.end > err.span.start && err.span.end <= source.len() {
  1089в†’        offset_to_location(source, err.span.end, file)
  1090в†’    } else {
  1091в†’        loc.clone()
  1092в†’    };
  1093в†’
  1094в†’    let diag_loc = DiagnosticSourceLocation {
  1095в†’        file: loc.file.clone().unwrap_or_default(),
  1096в†’        start_line: loc.line as u32 + 1, // convert 0-based to 1-based
  1097в†’        start_col: loc.column as u32 + 1,
  1098в†’        end_line: end_loc.line as u32 + 1,
  1099в†’        end_col: end_loc.column as u32 + 1,
  1100в†’    };
  1101в†’
  1102в†’    let mut diag = VumaDiagnostic::new(
  1103в†’        code,
  1104в†’        DiagnosticSeverity::Error,
  1105в†’        &err.message,
  1106в†’        "parser",
  1107в†’        diag_loc,
  1108в†’    );
  1109в†’
  1110в†’    if let Some(ref suggestion) = err.suggestion {
  1111в†’        diag = diag.with_suggestion(format!("did you mean '{}'?", suggestion));
  1112в†’    }
  1113в†’
  1114в†’    // Store the line text as a hint if available.
  1115в†’    if let Some(ref line_text) = loc.line_text {
  1116в†’        diag = diag.with_suggestion(format!("source: {}", line_text.trim()));
  1117в†’    }
  1118в†’
  1119в†’    diag
  1120в†’}
