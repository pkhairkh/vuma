     1‚Üí     1‚Üí//! Abstract Syntax Tree types for the VUMA language frontend.
     2‚Üí     2‚Üí//!
     3‚Üí     3‚Üí//! The AST is the central data structure produced by the parser. Every node
     4‚Üí     4‚Üí//! carries an optional [`Span`] so that downstream passes (type checking,
     5‚Üí     5‚Üí//! SCG generation, error reporting) can trace back to the original source.
     6‚Üí     6‚Üí//!
     7‚Üí     7‚Üí//! Design principles:
     8‚Üí     8‚Üí//! - **Minimal**: only the constructs needed for a memory-oriented,
     9‚Üí     9‚Üí//!   AI-consumable language.
    10‚Üí    10‚Üí//! - **Explicit**: allocation, free, cast, and region operations are
    11‚Üí    11‚Üí//!   first-class statements rather than library calls.
    12‚Üí    12‚Üí//! - **Spanned**: every node can point back to its source location.
    13‚Üí    13‚Üí
    14‚Üí    14‚Üíuse crate::error::Span;
    15‚Üí    15‚Üíuse serde::{Deserialize, Serialize};

    21‚Üí    21‚Üí/// Visibility modifier for items.
    22‚Üí    22‚Üí#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, Default)]
    23‚Üí    23‚Üípub enum Visibility {
    24‚Üí    24‚Üí    /// Default (private) visibility ‚Äî no modifier.
    25‚Üí    25‚Üí    #[default]
    26‚Üí    26‚Üí    Private,
    27‚Üí    27‚Üí    /// `pub` ‚Äî public visibility.
    28‚Üí    28‚Üí    Public,
    29‚Üí    29‚Üí    /// `pub(crate)` ‚Äî visible within the current crate.
    30‚Üí    30‚Üí    PublicCrate,
    31‚Üí    31‚Üí    /// `pub(super)` ‚Äî visible in the parent module.
    32‚Üí    32‚Üí    PublicSuper,
    33‚Üí    33‚Üí    /// `pub(in path)` ‚Äî visible in the specified path.
    34‚Üí    34‚Üí    PublicIn(String),
    35‚Üí    35‚Üí}
    36‚Üí    36‚Üí
    37‚Üí    37‚Üíimpl std::fmt::Display for Visibility {
    38‚Üí    38‚Üí    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    39‚Üí    39‚Üí        match self {
    40‚Üí    40‚Üí            Visibility::Private => write!(f, ""),
    41‚Üí    41‚Üí            Visibility::Public => write!(f, "pub"),
    42‚Üí    42‚Üí            Visibility::PublicCrate => write!(f, "pub(crate)"),
    43‚Üí    43‚Üí            Visibility::PublicSuper => write!(f, "pub(super)"),
    44‚Üí    44‚Üí            Visibility::PublicIn(path) => write!(f, "pub(in {})", path),
    45‚Üí    45‚Üí        }
    46‚Üí    46‚Üí    }
    47‚Üí    47‚Üí}

  1184‚Üí  1184‚Üíimpl std::fmt::Display for Type {
  1185‚Üí  1185‚Üí    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
  1186‚Üí  1186‚Üí        match self {
  1187‚Üí  1187‚Üí            Type::BDBase(name) => write!(f, "{}", name),
  1188‚Üí  1188‚Üí            Type::Ptr(inner) => write!(f, "*{}", inner),
  1189‚Üí  1189‚Üí            Type::RegionPtr { inner, region } => write!(f, "*{} @ {}", inner, region),
  1190‚Üí  1190‚Üí            Type::Array { element, size } => write!(f, "[{}; {}]", element, size),
  1191‚Üí  1191‚Üí            Type::Struct { name, .. } => write!(f, "{}", name),
  1192‚Üí  1192‚Üí            Type::Generic { name, args } => {
  1193‚Üí  1193‚Üí                let a: Vec<String> = args.iter().map(|t| t.to_string()).collect();
  1194‚Üí  1194‚Üí                write!(f, "{}<{}>", name, a.join(", "))
  1195‚Üí  1195‚Üí            }
  1196‚Üí  1196‚Üí            Type::Func {
  1197‚Üí  1197‚Üí                params,
  1198‚Üí  1198‚Üí                return_type,
  1199‚Üí  1199‚Üí            } => {
  1200‚Üí  1200‚Üí                let p: Vec<String> = params.iter().map(|t| t.to_string()).collect();
  1201‚Üí  1201‚Üí                write!(f, "({})", p.join(", "))?;
  1202‚Üí  1202‚Üí                if let Some(rt) = return_type {
  1203‚Üí  1203‚Üí                    write!(f, " -> {}", rt)?;
  1204‚Üí  1204‚Üí                }
  1205‚Üí  1205‚Üí                Ok(())
  1206‚Üí  1206‚Üí            }
  1207‚Üí  1207‚Üí            Type::BdAnnot { name } => write!(f, "#bd({})", name),
  1208‚Üí  1208‚Üí        }
  1209‚Üí  1209‚Üí    }
  1210‚Üí  1210‚Üí}
