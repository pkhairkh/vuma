     1→     1→//! # LoongArch 64-bit Backend
     2→     2→//!
     3→     3→//! Implements the `Backend` trait for the LoongArch 64-bit target (LP64 ABI).
     4→     4→//! This module provides:
     5→     5→//!
     6→     6→//! - `Gpr` — General-purpose register enum (r0–r31)
     7→     7→//! - `Fpr` — Floating-point register enum (f0–f31)
     8→     8→//! - `Instruction` — LoongArch64 instruction enum with correct encoding
     9→     9→//! - Encoding helpers for all 9 instruction formats (2R, 3R, 4R, 2RI8, 2RI12,
    10→    10→//!   2RI14, 2RI16, 1RI21, I26)
    11→    11→//! - `LoongArch64Backend` — `Backend` implementation that lowers IR to LoongArch64
    12→    12→//!   machine code and emits ELF64 binaries
    13→    13→//!
    14→    14→//! ## LoongArch64 Register Convention (LP64 ABI)
    15→    15→//!
    16→    16→//! | Register(s) | ABI Name | Role                              |
    17→    17→//! |-------------|----------|-----------------------------------|
    18→    18→//! | r0          | zero     | Hardwired zero                    |
    19→    19→//! | r1          | ra       | Return address                    |
    20→    20→//! | r2          | tp       | Thread pointer                    |
    21→    21→//! | r3          | sp       | Stack pointer                     |
    22→    22→//! | r4–r11      | a0–a7    | Argument / return registers       |
    23→    23→//! | r12–r20     | t0–t8    | Caller-saved temporaries          |
    24→    24→//! | r21         | —        | Reserved                          |
    25→    25→//! | r22         | fp       | Frame pointer (callee-saved)      |
    26→    26→//! | r23–r31     | s0–s8    | Callee-saved                      |
    27→    27→//!
    28→    28→//! ## LoongArch64 FP Register Convention (LP64 ABI)
    29→    29→//!
    30→    30→//! | Register(s) | ABI Name | Role                              |
    31→    31→//! |-------------|----------|-----------------------------------|
    32→    32→//! | f0–f7       | fa0–fa7  | FP argument / return registers    |
    33→    33→//! | f8–f23      | ft0–ft15 | Caller-saved temporaries          |
    34→    34→//! | f24–f31     | fs0–fs7  | Callee-saved                      |
    35→    35→//!
    36→    36→//! ## Instruction Formats
    37→    37→//!
    38→    38→//! All instructions are 32 bits, little-endian, with no branch delay slots.
    39→    39→//! Nine formats: 2R, 3R, 4R, 2RI8, 2RI12, 2RI14, 2RI16, 1RI21, I26.
    40→    40→//!
    41→    41→//! ## References
    42→    42→//!
    43→    43→//! - LoongArch Reference Manual, Volume 1: Basic Architecture
    44→    44→//! - <https://loongson.github.io/LoongArch-Documentation/>

   324→   324→// Instruction Encoding Helpers
   325→   325→// ===========================================================================
   326→   326→
   327→   327→/// Encode a 2R format instruction.
   328→   328→///
   329→   329→/// Format: `opcode[31:10] | rj[9:5] | rd[4:0]`
   330→   330→fn encode_2r(opcode: u32, rj: u32, rd: u32) -> [u8; 4] {
   331→   331→    let word = ((opcode & 0x3FF_FFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   332→   332→    word.to_le_bytes()
   333→   333→}
   334→   334→
   335→   335→/// Encode a 3R format instruction.
   336→   336→///
   337→   337→/// Format: `opcode[31:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   338→   338→fn encode_3r(opcode: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   339→   339→    let word = ((opcode & 0x1FFFF) << 15) | ((rk & 0x1F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   340→   340→    word.to_le_bytes()
   341→   341→}
   342→   342→
   343→   343→/// Encode a 4R format instruction.
   344→   344→///
   345→   345→/// Format: `opcode[31:20] | ra[19:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   346→   346→#[allow(dead_code)]
   347→   347→fn encode_4r(opcode: u32, ra: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   348→   348→    let word = ((opcode & 0xFFF) << 20)
   349→   349→        | ((ra & 0x1F) << 15)
   350→   350→        | ((rk & 0x1F) << 10)
   351→   351→        | ((rj & 0x1F) << 5)
   352→   352→        | (rd & 0x1F);
   353→   353→    word.to_le_bytes()
   354→   354→}
   355→   355→
   356→   356→/// Encode a 2RI8 format instruction.
   357→   357→///
   358→   358→/// Format: `opcode[31:22] | I8[21:14] | rj[9:5] | rd[4:0]`
   359→   359→fn encode_2ri8(opcode: u32, imm8: u32, rj: u32, rd: u32) -> [u8; 4] {
   360→   360→    let word = ((opcode & 0x3FF) << 22) | ((imm8 & 0xFF) << 14) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   361→   361→    word.to_le_bytes()
   362→   362→}
   363→   363→
   364→   364→/// Encode a 2RI12 format instruction.
   365→   365→///
   366→   366→/// Format: `opcode[31:22] | I12[21:10] | rj[9:5] | rd[4:0]`
   367→   367→fn encode_2ri12(opcode: u32, imm12: u32, rj: u32, rd: u32) -> [u8; 4] {
   368→   368→    let word =
   369→   369→        ((opcode & 0x3FF) << 22) | ((imm12 & 0xFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   370→   370→    word.to_le_bytes()
   371→   371→}
   372→   372→
   373→   373→/// Encode a 2RI14 format instruction.
   374→   374→///
   375→   375→/// Format: `opcode[31:24] | I14[23:10] | rj[9:5] | rd[4:0]`
   376→   376→fn encode_2ri14(opcode: u32, imm14: u32, rj: u32, rd: u32) -> [u8; 4] {
   377→   377→    let word =
   378→   378→        ((opcode & 0xFF) << 24) | ((imm14 & 0x3FFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   379→   379→    word.to_le_bytes()
   380→   380→}
   381→   381→
   382→   382→/// Encode a 2RI16 format instruction.
   383→   383→///
   384→   384→/// Format: `opcode[31:26] | I16[25:10] | rj[9:5] | rd[4:0]`
   385→   385→fn encode_2ri16(opcode: u32, imm16: u32, rj: u32, rd: u32) -> [u8; 4] {
   386→   386→    let word =
   387→   387→        ((opcode & 0x3F) << 26) | ((imm16 & 0xFFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   388→   388→    word.to_le_bytes()
   389→   389→}
   390→   390→
   391→   391→/// Encode a 1RI21 format instruction (used for BEQZ, BNEZ).
   392→   392→///
   393→   393→/// Format: `opcode[31:26] | offs21[15:0] in bits[25:10] | offs21[20:16] in bits[9:5] | rj[4:0]`
   394→   394→///
   395→   395→/// Note: the offset bits are SWAPPED compared to a linear layout.
   396→   396→/// The lower 16 bits of the offset go in the higher position (bits[25:10]),
   397→   397→/// and the upper 5 bits go in the lower position (bits[9:5]).
   398→   398→fn encode_1ri21(opcode: u32, imm21: u32, rj: u32) -> [u8; 4] {
   399→   399→    let word = ((opcode & 0x3F) << 26)
   400→   400→        | ((imm21 & 0xFFFF) << 10)
   401→   401→        | ((imm21 >> 16) & 0x1F) << 5
   402→   402→        | (rj & 0x1F);
   403→   403→    word.to_le_bytes()
   404→   404→}
   405→   405→
   406→   406→/// Encode an I26 format instruction (used for B, BL).
   407→   407→///
   408→   408→/// Format: `opcode[31:26] | offs26[15:0] in bits[25:10] | offs26[25:16] in bits[9:0]`
   409→   409→///
   410→   410→/// Note: the offset bits are SWAPPED compared to a linear layout.
   411→   411→/// The lower 16 bits of the offset go in the higher position (bits[25:10]),
   412→   412→/// and the upper 10 bits go in the lower position (bits[9:0]).
   413→   413→fn encode_i26(opcode: u32, imm26: u32) -> [u8; 4] {
   414→   414→    let word = ((opcode & 0x3F) << 26)
   415→   415→        | ((imm26 & 0xFFFF) << 10)
   416→   416→        | ((imm26 >> 16) & 0x3FF);
   417→   417→    word.to_le_bytes()
   418→   418→}

  1538→  1538→/// Build a minimal ELF64 binary for LoongArch64 from raw code bytes.
  1539→  1539→///
  1540→  1540→/// Produces a static executable with two LOAD segments:
  1541→  1541→///   1. PF_R | PF_X — .text (code)
  1542→  1542→///   2. PF_R | PF_W — .data / stack (writable memory)
  1543→  1543→fn build_loongarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1544→  1544→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (LoongArch64 typical page size)
  1545→  1545→
  1546→  1546→    let elf_header_size: u64 = 64;
  1547→  1547→    let phdr_size: u64 = 56;
  1548→  1548→    let num_phdrs: u64 = 2;
  1549→  1549→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1550→  1550→    // Page-align the text segment start in the file.
  1551→  1551→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1552→  1552→    let text_size = code.len() as u64;
  1553→  1553→
  1554→  1554→    // The data segment starts on the next page after the text.
  1555→  1555→    let text_file_end = text_offset + text_size;
  1556→  1556→    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1557→  1557→    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1558→  1558→    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1559→  1559→    let entry_point = base_addr + text_offset;
  1560→  1560→
  1561→  1561→    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1562→  1562→
  1563→  1563→    // --- e_ident ---
  1564→  1564→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1565→  1565→    elf.push(2); // ELFCLASS64
  1566→  1566→    elf.push(1); // ELFDATA2LSB
  1567→  1567→    elf.push(1); // EV_CURRENT
  1568→  1568→    elf.push(3); // ELFOSABI_LINUX
  1569→  1569→    elf.push(0); // padding
  1570→  1570→    elf.extend_from_slice(&[0u8; 7]); // padding
  1571→  1571→
  1572→  1572→    // --- ELF header fields ---
  1573→  1573→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1574→  1574→    elf.extend_from_slice(&258u16.to_le_bytes()); // e_machine = EM_LOONGARCH
  1575→  1575→    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1576→  1576→    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1577→  1577→    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1578→  1578→    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1579→  1579→    elf.extend_from_slice(&0x43u32.to_le_bytes()); // e_flags = 0x43 (LP64D ABI double-float)
  1580→  1580→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1581→  1581→    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1582→  1582→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1583→  1583→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1584→  1584→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1585→  1585→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1586→  1586→
  1587→  1587→    // --- Program Header 1: LOAD (PF_R | PF_X) — .text ---
  1588→  1588→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1589→  1589→    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1590→  1590→    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1591→  1591→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1592→  1592→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1593→  1593→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1594→  1594→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1595→  1595→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1596→  1596→
  1597→  1597→    // --- Program Header 2: LOAD (PF_R | PF_W) — .data / stack ---
  1598→  1598→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1599→  1599→    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1600→  1600→    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1601→  1601→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1602→  1602→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1603→  1603→    elf.extend_from_slice(&0u64.to_le_bytes()); // p_filesz (no initialized data)
  1604→  1604→    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1605→  1605→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1606→  1606→
  1607→  1607→    // --- .text section ---
  1608→  1608→    // Pad to page-aligned text_offset
  1609→  1609→    while (elf.len() as u64) < text_offset {
  1610→  1610→        elf.push(0);
  1611→  1611→    }
  1612→  1612→    elf.extend_from_slice(code);
  1613→  1613→
  1614→  1614→    // --- Pad to data segment offset (if needed) ---
  1615→  1615→    while (elf.len() as u64) < data_offset {
  1616→  1616→        elf.push(0);
  1617→  1617→    }
  1618→  1618→
  1619→  1619→    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1620→  1620→
  1621→  1621→    elf
  1622→  1622→}

  1628→  1628→/// Decode a single LoongArch64 32-bit instruction word into a mnemonic string.
  1629→  1629→fn decode_loongarch64_instruction(word: u32) -> String {
  1630→  1630→    let opcode = word & 0x7f;
  1631→  1631→    match opcode {
  1632→  1632→        0x00 => {
  1633→  1633→            let rd = (word >> 7) & 0x1f;
  1634→  1634→            let rj = (word >> 15) & 0x1f;
  1635→  1635→            let rk = (word >> 10) & 0x1f;
  1636→  1636→            format!("add.w $r{}, $r{}, $r{}", rd, rj, rk)
  1637→  1637→        }
  1638→  1638→        0x01 => {
  1639→  1639→            let rd = (word >> 7) & 0x1f;
  1640→  1640→            let rj = (word >> 15) & 0x1f;
  1641→  1641→            let rk = (word >> 10) & 0x1f;
  1642→  1642→            format!("sub.w $r{}, $r{}, $r{}", rd, rj, rk)
  1643→  1643→        }
  1644→  1644→        0x02 => {
  1645→  1645→            let rd = (word >> 7) & 0x1f;
  1646→  1646→            let rj = (word >> 15) & 0x1f;
  1647→  1647→            let rk = (word >> 10) & 0x1f;
  1648→  1648→            format!("mul.w $r{}, $r{}, $r{}", rd, rj, rk)
  1649→  1649→        }
  1650→  1650→        0x05 => {
  1651→  1651→            let rd = (word >> 7) & 0x1f;
  1652→  1652→            let _rj = (word >> 15) & 0x1f; // lu12i.w does not use rj in the disassembly
  1653→  1653→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1654→  1654→            format!("lu12i.w $r{}, {}({})", rd, si12, si12)
  1655→  1655→        }
  1656→  1656→        0x08 => {
  1657→  1657→            let rd = (word >> 7) & 0x1f;
  1658→  1658→            let rj = (word >> 15) & 0x1f;
  1659→  1659→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1660→  1660→            format!("ld.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1661→  1661→        }
  1662→  1662→        0x0a => {
  1663→  1663→            let rd = (word >> 7) & 0x1f;
  1664→  1664→            let rj = (word >> 15) & 0x1f;
  1665→  1665→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1666→  1666→            format!("st.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1667→  1667→        }
  1668→  1668→        0x0c => {
  1669→  1669→            let rd = (word >> 7) & 0x1f;
  1670→  1670→            let rj = (word >> 15) & 0x1f;
  1671→  1671→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1672→  1672→            format!("ld.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1673→  1673→        }
  1674→  1674→        0x0e => {
  1675→  1675→            let rd = (word >> 7) & 0x1f;
  1676→  1676→            let rj = (word >> 15) & 0x1f;
  1677→  1677→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1678→  1678→            format!("st.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1679→  1679→        }
  1680→  1680→        0x10 => {
  1681→  1681→            let rj = (word >> 15) & 0x1f;
  1682→  1682→            let offs = ((word >> 10) as i32) << 12 >> 10;
  1683→  1683→            format!("beq $r{}, {}({})", rj, offs, offs)
  1684→  1684→        }
  1685→  1685→        0x11 => {
  1686→  1686→            let rj = (word >> 15) & 0x1f;
  1687→  1687→            let offs = ((word >> 10) as i32) << 12 >> 10;
  1688→  1688→            format!("bne $r{}, {}({})", rj, offs, offs)
  1689→  1689→        }
  1690→  1690→        0x14 => format!("bl {}", ((word >> 10) as i32) << 12 >> 10),
  1691→  1691→        0x15 => {
  1692→  1692→            let rd = (word >> 7) & 0x1f;
  1693→  1693→            format!("jirl $r{}, {}", rd, ((word >> 10) as i32) << 12 >> 10)
  1694→  1694→        }
  1695→  1695→        _ => format!(".word {:08x}", word),
  1696→  1696→    }
  1697→  1697→}

  1719→  1719→/// Compute the stack frame size for an IR function on LoongArch64.
  1720→  1720→///
  1721→  1721→/// Sums `Alloc` instruction sizes, adds 16 bytes for the ra/fp save pair,
  1722→  1722→/// and rounds up to 16-byte alignment.
  1723→  1723→fn loongarch64_compute_frame_size(func: &IRFunction) -> usize {
  1724→  1724→    let mut total: u32 = 16; // ra + fp save pair
  1725→  1725→    for block in &func.blocks {
  1726→  1726→        for instr in &block.instructions {
  1727→  1727→            if let IRInstr::Alloc { size, .. } = instr {
  1728→  1728→                let aligned = (*size).div_ceil(16) * 16;
  1729→  1729→                total += aligned;
  1730→  1730→            }
  1731→  1731→        }
  1732→  1732→    }
  1733→  1733→    // Round up to 16-byte alignment
  1734→  1734→    total = (total + 15) & !15;
  1735→  1735→    total as usize
  1736→  1736→}
