     1→     1→//! # ARM64 Code Emission
     2→     2→//!
     3→     3→//! Lowers IR to ARM64 machine code and produces ELF binaries or raw binaries
     4→     4→//! suitable for the AArch64 (Cortex-A76, ARMv8.2-A).
     5→     5→//!
     6→     6→//! ## Pipeline
     7→     7→//!
     8→     8→//! 1. **IR → ARM64 Instructions**: Each IR instruction is pattern-matched and
     9→     9→//!    lowered to one or more [`Instruction`]
    10→    10→//!    values, with virtual registers replaced by physical registers from the
    11→    11→//!    register allocator.
    12→    12→//! 2. **Emit function**: Each IR function is emitted as a sequence of 32-bit
    13→    13→//!    ARM64 code words.
    14→    14→//! 3. **Emit program**: All functions are collected, data sections are laid
    15→    15→//!    out, and a minimal ELF64 binary is produced.
    16→    16→//!
    17→    17→//! ## Output Formats
    18→    18→//!
    19→    19→//! - **ELF**: Full ELF64 binary with program headers, section headers, symbol
    20→    20→//!   table, and string table.  Suitable for Linux on AArch64.
    21→    21→//! - **Raw**: Flat binary image for bare-metal AArch64 (loaded at 0x80000).
    22→    22→//! - **Obj**: Relocatable ELF object file (`ET_REL`) for linking.
    23→    23→//!
    24→    24→//! ## ELF Layout (executable)
    25→    25→//!
    26→    26→//! ```text
    27→    27→//! ┌─────────────────────┐
    28→    28→//! │ ELF Header           │  64 bytes
    29→    29→//! ├─────────────────────┤
    30→    30→//! │ Program Headers      │  2 × 56 bytes (LOAD segments)
    31→    31→//! ├─────────────────────┤
    32→    32→//! │ .text                │  emitted code
    33→    33→//! ├─────────────────────┤
    34→    34→//! │ .rodata              │  read-only data
    35→    35→//! ├─────────────────────┤
    36→    36→//! │ .data                │  initialized data
    37→    37→//! ├─────────────────────┤
    38→    38→//! │ .symtab              │  symbol table entries
    39→    39→//! ├─────────────────────┤
    40→    40→//! │ .strtab              │  symbol string table
    41→    41→//! ├─────────────────────┤
    42→    42→//! │ .shstrtab            │  section header string table
    43→    43→//! ├─────────────────────┤
    44→    44→//! │ Section Headers      │  N × 64 bytes
    45→    45→//! └─────────────────────┘
    46→    46→//! .bss is virtual-only (memsz > filesz in the data LOAD segment).
    47→    47→//! ```
    48→    48→//!
    49→    49→//! ## Relocation Support
    50→    50→//!
    51→    51→//! Inter-function calls (`BL`) are resolved in a fixup pass after all
    52→    52→//! functions have been emitted.  The emitter records the word offset of each
    53→    53→//! `BL` instruction and the target function name; once function addresses
    54→    54→//! are known, the branch offsets are patched into the encoded instructions.
    55→    55→

   470→   470→            debug_info: false,
   471→   471→        }
   472→   472→    }

   508→   508→// ---------------------------------------------------------------------------
   509→   509→// Inter-function call relocation record
   510→   510→// ---------------------------------------------------------------------------
   511→   511→
   512→   512→/// A record for a `BL` instruction that needs to be patched with the address
   513→   513→/// of a named function.  After all functions are emitted, these records are
   514→   514→/// resolved to compute the correct branch offset.
   515→   515→#[derive(Debug, Clone)]
   516→   516→struct CallRelocation {
   517→   517→    /// Byte offset within the text section where the BL instruction lives.
   518→   518→    text_byte_offset: u64,
   519→   519→    /// Name of the target function.
   520→   520→    target_func: String,
   521→   521→}
   522→   522→
   523→   523→// ---------------------------------------------------------------------------
   524→   524→// ELF RelaEntry
   525→   525→// ---------------------------------------------------------------------------
   526→   526→
   527→   527→/// An ELF64 relocation entry with an explicit addend (SHT_RELA).
   528→   528→///
   529→   529→/// Each entry specifies where in the section a relocation must be applied,
   530→   530→/// which symbol the relocation references, the type of relocation, and an
   531→   531→/// addend value used in the relocation computation.
   532→   532→#[derive(Debug, Clone)]
   533→   533→pub struct RelaEntry {
   534→   534→    /// Byte offset within the section where the relocation applies.
   535→   535→    pub offset: u64,
   536→   536→    /// Symbol index (upper 32 bits) and relocation type (lower 32 bits),
   537→   537→    /// packed as: `(sym_idx << 32) | r_type`.
   538→   538→    pub info: u64,
   539→   539→    /// Addend value used in the relocation computation.
   540→   540→    pub addend: i64,
   541→   541→}
   542→   542→
   543→   543→impl RelaEntry {
   544→   544→    /// Create a new relocation entry.
   545→   545→    pub fn new(offset: u64, sym_idx: u32, r_type: u32, addend: i64) -> Self {
   546→   546→        Self {
   547→   547→            offset,
   548→   548→            info: ((sym_idx as u64) << 32) | (r_type as u64),
   549→   549→            addend,
   550→   550→        }
   551→   551→    }
   552→   552→
   553→   553→    /// Extract the symbol index from `info`.
   554→   554→    pub fn sym_idx(&self) -> u32 {
   555→   555→        (self.info >> 32) as u32
   556→   556→    }
   557→   557→
   558→   558→    /// Extract the relocation type from `info`.
   559→   559→    pub fn r_type(&self) -> u32 {
   560→   560→        (self.info & 0xFFFFFFFF) as u32
   561→   561→    }
   562→   562→
   563→   563→    /// Serialize the entry to 24 bytes (little-endian).
   564→   564→    pub fn to_bytes(&self) -> [u8; 24] {
   565→   565→        let mut buf = [0u8; 24];
   566→   566→        buf[0..8].copy_from_slice(&self.offset.to_le_bytes());
   567→   567→        buf[8..16].copy_from_slice(&self.info.to_le_bytes());
   568→   568→        buf[16..24].copy_from_slice(&self.addend.to_le_bytes());
   569→   569→        buf
   570→   570→    }
   571→   571→}

   574→   574→// Emitter
   575→   575→// ---------------------------------------------------------------------------
   576→   576→
   577→   577→/// The ARM64 code emitter.
   578→   578→///
   579→   579→/// Holds state for the current emission context: the register allocator,
   580→   580→/// accumulated code, and fixup records for branch targets.
   581→   581→pub struct Emitter {
   582→   582→    /// Register allocator used during emission.
   583→   583→    reg_alloc: RegAllocator,
   584→   584→    /// Accumulated machine code for the current function (32-bit words).
   585→   585→    code: Vec<u32>,
   586→   586→    /// Fixup records for intra-function branches: (word index, target label name, format).
   587→   587→    fixups: Vec<(usize, String, BranchFormat)>,
   588→   588→    /// Map from label name to code offset (in words) within the current function.
   589→   589→    label_offsets: HashMap<String, usize>,
   590→   590→    /// Inter-function call relocations for the current function.
   591→   591→    call_relocs: Vec<CallRelocation>,
   592→   592→    /// Relocation entries using the `RelocationEntry` infrastructure.
   593→   593→    ///
   594→   594→    /// Each entry records a byte offset within the function's encoded output
   595→   595→    /// where a symbolic reference must be resolved, the target symbol name,
   596→   596→    /// and the ISA-specific relocation type (e.g., `"R_AARCH64_CALL26"`).
   597→   597→    relocations: Vec<RelocationEntry>,
   598→   598→    /// Name of the function currently being emitted.
   599→   599→    current_func_name: String,
   600→   600→    /// Byte offset of the current function within the text section (set externally).
   601→   601→    func_text_offset: u64,
   602→   602→    /// Computed stack frame size (in bytes) for the current function.
   603→   603→    frame_size: u32,
   604→   604→    /// Registers pinned for the current instruction (auto-unpinned after each instruction).
   605→   605→    instr_pinned_regs: Vec<Register>,
   606→   606→}
   607→   607→
   608→   608→impl Emitter {
   609→   609→    /// Create a new emitter with a fresh register allocator.
   610→   610→    pub fn new() -> Self {
   611→   611→        Self {
   612→   612→            reg_alloc: RegAllocator::new(),
   613→   613→            code: Vec::new(),
   614→   614→            fixups: Vec::new(),
   615→   615→            label_offsets: HashMap::new(),
   616→   616→            call_relocs: Vec::new(),
   617→   617→            relocations: Vec::new(),
   618→   618→            current_func_name: String::new(),
   619→   619→            func_text_offset: 0,
   620→   620→            frame_size: 0,
   621→   621→            instr_pinned_regs: Vec::new(),
   622→   622→        }
   623→   623→    }
   624→   624→
   625→   625→    // -----------------------------------------------------------------------
   626→   626→    // Function emission
   627→   627→    // -----------------------------------------------------------------------
   628→   628→
   629→   629→    /// Emit a single IR function to ARM64 machine code.
   630→   630→    ///
   631→   631→    /// For functions with more than `STACK_SLOT_VREG_THRESHOLD` virtual
   632→   632→    /// registers, the stack-slot emitter is used (every vreg gets a stack
   633→   633→    /// slot, scratch registers only).  For simpler functions, the greedy
   634→   634→    /// register allocator is used.
   635→   635→    ///
   636→   636→    /// Returns a vector of 32-bit ARM64 instruction words.
   637→   637→    pub fn emit_function(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   638→   638→        let vreg_count = count_vregs(func);
   639→   639→        if vreg_count > STACK_SLOT_VREG_THRESHOLD {
   640→   640→            self.emit_function_stack_slot(func)
   641→   641→        } else {
   642→   642→            self.emit_function_greedy(func)
   643→   643→        }
   644→   644→    }
   645→   645→
   646→   646→    /// Emit a single IR function using the greedy register allocator.
   647→   647→    ///
   648→   648→    /// This is the original emission strategy — suitable for functions with
   649→   649→    /// a small number of virtual registers.
   650→   650→    fn emit_function_greedy(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   651→   651→        self.code.clear();
   652→   652→        self.fixups.clear();
   653→   653→        self.label_offsets.clear();
   654→   654→        self.call_relocs.clear();
   655→   655→        self.relocations.clear();
   656→   656→        self.current_func_name = func.name.clone();
   657→   657→        self.reg_alloc.reset();
   658→   658→
   659→   659→        // Pre-allocate registers for parameters (AAPCS64: X0–X7).
   660→   660→        // We must explicitly assign parameter virtual registers to the
   661→   661→        // correct argument registers so the calling convention is respected.
   662→   662→        let arg_regs = [
   663→   663→            Register::X0, Register::X1, Register::X2, Register::X3,
   664→   664→            Register::X4, Register::X5, Register::X6, Register::X7,
   665→   665→        ];
   666→   666→        for (i, param) in func.params.iter().enumerate() {
   667→   667→            if let IRValue::Register(vreg_id) = param {
   668→   668→                if i < 8 {
   669→   669→                    self.reg_alloc.preassign(*vreg_id, arg_regs[i]);
   670→   670→                }
   671→   671→            }
   672→   672→        }
   673→   673→
   674→   674→        // Emit prologue:
   675→   675→        // 1. SUB SP, SP, #16          (make room for FP/LR save pair)
   676→   676→        // 2. STP X29, X30, [SP]       (save FP and LR at new SP)
   677→   677→        // 3. ADD X29, SP, #0           (set frame pointer to new SP)
   678→   678→        // 4. SUB SP, SP, #frame_size   (make room for local variables)
   679→   679→        //
   680→   680→        // NOTE: We cannot use pre-indexed STP [SP, #-16]! because the
   681→   681→        // Instruction::STP encoding only supports signed-offset mode.
   682→   682→        // Instead, we explicitly decrement SP with SUB before the STP.
   683→   683→        self.emit_instruction(Instruction::SUB {
   684→   684→            rd: Register::SP,
   685→   685→            rn: Register::SP,
   686→   686→            rm: Operand::Imm12(16),
   687→   687→        })?;
   688→   688→        self.emit_instruction(Instruction::STP {
   689→   689→            rt1: Register::X29,
   690→   690→            rt2: Register::X30,
   691→   691→            rn: Register::SP,
   692→   692→            offset: 0,
   693→   693→        })?;
   694→   694→
   695→   695→        // MOV X29, SP (set frame pointer)
   696→   696→        // IMPORTANT: Cannot use MOV (ORR Xd, XZR, SP) because ORR treats
   697→   697→        // Rm=31 as XZR, yielding zero. Use ADD X29, SP, #0 instead.
   698→   698→        self.emit_instruction(Instruction::ADD {
   699→   699→            rd: Register::X29,
   700→   700→            rn: Register::SP,
   701→   701→            rm: Operand::Imm12(0),
   702→   702→        })?;
   703→   703→
   704→   704→        // Reserve space for spill slots only. Each Alloc instruction handles
   705→   705→        // its own SUB SP, SP, #aligned_size. The epilogue will restore only
   706→   706→        // the spill area reservation. The Alloc areas are NOT restored by the
   707→   707→        // epilogue — they're effectively "leaked" stack space. This is fine
   708→   708→        // because the function's stack frame is restored by the LDP + ADD SP, #16.
   709→   709→        //
   710→   710→        // For spill slots, we need to estimate the maximum number of spills
   711→   711→        // that could occur during emission. Since we can't know the exact
   712→   712→        // count beforehand, we use the compute_frame_size function which
   713→   713→        // estimates based on the vreg count.
   714→   714→        let aligned_stack = compute_frame_size(func);
   715→   715→        // frame_size = only the spill area (NOT the Alloc sizes, since each
   716→   716→        // Alloc instruction does its own SUB SP)
   717→   717→        // Compute the spill-only portion:
   718→   718→        let mut alloc_total: u32 = 0;
   719→   719→        for block in &func.blocks {
   720→   720→            for instr in &block.instructions {
   721→   721→                if let IRInstr::Alloc { size, .. } = instr {
   722→   722→                    let aligned = (*size).div_ceil(16) * 16;
   723→   723→                    alloc_total += aligned;
   724→   724→                }
   725→   725→            }
   726→   726→        }
   727→   727→        // Spill area = total frame - alloc area
   728→   728→        let spill_area = if aligned_stack > alloc_total {
   729→   729→            aligned_stack - alloc_total
   730→   730→        } else {
   731→   731→            // Minimal 16-byte reservation for safety
   732→   732→            16
   733→   733→        };
   734→   734→        let spill_area_aligned = (spill_area + 15) & !15;
   735→   735→        self.frame_size = spill_area_aligned;
   736→   736→        if spill_area_aligned > 0 {
   737→   737→            if spill_area_aligned <= 4095 {
   738→   738→                self.emit_instruction(Instruction::SUB {
   739→   739→                    rd: Register::SP,
   740→   740→                    rn: Register::SP,
   741→   741→                    rm: Operand::Imm12(spill_area_aligned as u16),
   742→   742→                })?;
   743→   743→            } else {
   744→   744→                self.emit_load_immediate(Register::X9, spill_area_aligned as i64)?;
   745→   745→                self.emit_instruction(Instruction::SUB {
   746→   746→                    rd: Register::SP,
   747→   747→                    rn: Register::SP,
   748→   748→                    rm: Operand::Reg {
   749→   749→                        reg: Register::X9,
   750→   750→                        shift: None,
   751→   751→                    },
   752→   752→                })?;
   753→   753→            }
   754→   754→        }
   755→   755→
   756→   756→        // Emit each basic block.
   757→   757→        for block in &func.blocks {
   758→   758→            self.label_offsets
   759→   759→                .insert(block.label.clone(), self.code.len());
   760→   760→            self.emit_block(block)?;
   761→   761→        }
   762→   762→
   763→   763→        // Apply fixups — resolve intra-function branch targets.
   764→   764→        self.apply_fixups()?;
   765→   765→
   766→   766→        Ok(self.code.clone())
   767→   767→    }
   768→   768→
   769→   769→    /// Returns the relocation entries recorded during emission of the current
   770→   770→    /// function.
   771→   771→    ///
   772→   772→    /// Each `RelocationEntry` records a byte offset within the function's
   773→   773→    /// encoded output where a symbolic reference must be resolved, the target
   774→   774→    /// symbol name, and the ISA-specific relocation type (e.g.,
   775→   775→    /// `"R_AARCH64_CALL26"`).
   776→   776→    pub fn relocations(&self) -> &[RelocationEntry] {
   777→   777→        &self.relocations
   778→   778→    }
   779→   779→
   780→   780→    /// Emit instructions for a single IR basic block.
   781→   781→    fn emit_block(&mut self, block: &IRBlock) -> Result<()> {
   782→   782→        for instr in &block.instructions {
   783→   783→            self.emit_ir_instr(instr)?;
   784→   784→            // Auto-unpin all registers that were pinned during this instruction.
   785→   785→            for reg in self.instr_pinned_regs.drain(..) {
   786→   786→                self.reg_alloc.unpin(reg);
   787→   787→            }
   788→   788→        }
   789→   789→        self.emit_terminator(&block.terminator)?;
   790→   790→        Ok(())
   791→   791→    }
   792→   792→
   793→   793→    // -----------------------------------------------------------------------
   794→   794→    // IR → ARM64 instruction lowering
   795→   795→    // -----------------------------------------------------------------------
   796→   796→
   797→   797→    /// Lower a single IR instruction to ARM64 instructions.
   798→   798→    fn emit_ir_instr(&mut self, instr: &IRInstr) -> Result<()> {
   799→   799→        match instr {
   800→   800→            IRInstr::Load { dst, addr, offset, ty } => {
   801→   801→                let rn = self.resolve_reg(addr)?;
   802→   802→                let rt = self.resolve_reg(dst)?;
   803→   803→                self.emit_load(rt, rn, *offset, ty)?;
   804→   804→            }
   805→   805→
   806→   806→            IRInstr::Store { value, addr, offset, ty } => {
   807→   807→                let rt = self.resolve_reg(value)?;
   808→   808→                let rn = self.resolve_reg(addr)?;
   809→   809→                if rt == rn {
   810→   810→                    // Both resolved to the same register — use scratch for value
   811→   811→                    self.emit_instruction(Instruction::MOV {
   812→   812→                        rd: Register::X16,
   813→   813→                        rm: rt,
   814→   814→                    })?;
   815→   815→                    self.emit_store(Register::X16, rn, *offset, ty)?;
   816→   816→                } else {
   817→   817→                    self.emit_store(rt, rn, *offset, ty)?;
   818→   818→                }
   819→   819→            }
   820→   820→
   821→   821→            IRInstr::BinOp { op, dst, lhs, rhs, ty } => {
   822→   822→                self.emit_binop(*op, dst, lhs, rhs, ty.as_ref())?;
   823→   823→            }
   824→   824→
   825→   825→            IRInstr::UnaryOp { op, dst, operand, ty: _ } => {
   826→   826→                let rd = self.resolve_reg(dst)?;
   827→   827→                let rn = self.resolve_reg(operand)?;
   828→   828→                match op {
   829→   829→                    UnaryOpKind::Neg => {
   830→   830→                        self.emit_instruction(Instruction::SUB {
   831→   831→                            rd,
   832→   832→                            rn: Register::XZR,
   833→   833→                            rm: Operand::Reg {
   834→   834→                                reg: rn,
   835→   835→                                shift: None,
   836→   836→                            },
   837→   837→                        })?;
   838→   838→                    }
   839→   839→                    UnaryOpKind::Not => {
   840→   840→                        self.emit_load_immediate(Register::X9, -1)?;
   841→   841→                        self.emit_instruction(Instruction::EOR {
   842→   842→                            rd,
   843→   843→                            rn,
   844→   844→                            rm: Register::X9,
   845→   845→                        })?;
   846→   846→                    }
   847→   847→                    UnaryOpKind::Clz => {
   848→   848→                        // CLZ Xd, Xn — count leading zeros (native ARM64 instruction)
   849→   849→                        self.emit_instruction(Instruction::CLZ { rd, rn })?;
   850→   850→                    }
   851→   851→                    UnaryOpKind::Ctz => {
   852→   852→                        // CTZ = RBIT + CLZ: reverse bits then count leading zeros.
   853→   853→                        // Use X9 as scratch if rd == rn (need intermediate result).
   854→   854→                        if rd == rn {
   855→   855→                            self.emit_instruction(Instruction::RBIT {
   856→   856→                                rd: Register::X9,
   857→   857→                                rn,
   858→   858→                            })?;
   859→   859→                            self.emit_instruction(Instruction::CLZ {
   860→   860→                                rd,
   861→   861→                                rn: Register::X9,
   862→   862→                            })?;
   863→   863→                        } else {
   864→   864→                            self.emit_instruction(Instruction::RBIT { rd, rn })?;
   865→   865→                            self.emit_instruction(Instruction::CLZ { rd, rn: rd })?;
   866→   866→                        }
   867→   867→                    }
   868→   868→                    UnaryOpKind::Popcnt => {
   869→   869→                        // POPCNT via FMOV+CNT+ADDV+UMOV sequence:
   870→   870→                        // FMOV D8, Xn        — move GPR value to SIMD register (V8 is caller-saved)
   871→   871→                        // CNT V8.8B, V8.8B   — count bits per byte
   872→   872→                        // ADDV B8, V8.8B     — horizontal sum of byte counts
   873→   873→                        // UMOV Xd, V8.B[0]   — move result back to GPR (zero-extends to 64 bits)
   874→   874→                        const SIMD_SCRATCH: u8 = 8; // V8 is caller-saved in AAPCS64
   875→   875→                        self.emit_instruction(Instruction::FMOV_DX {
   876→   876→                            vd: SIMD_SCRATCH,
   877→   877→                            rn,
   878→   878→                        })?;
   879→   879→                        self.emit_instruction(Instruction::CNT {
   880→   880→                            vd: SIMD_SCRATCH,
   881→   881→                            vn: SIMD_SCRATCH,
   882→   882→                        })?;
   883→   883→                        self.emit_instruction(Instruction::ADDV {
   884→   884→                            vd: SIMD_SCRATCH,
   885→   885→                            vn: SIMD_SCRATCH,
   886→   886→                        })?;
   887→   887→                        self.emit_instruction(Instruction::UMOV {
   888→   888→                            rd,
   889→   889→                            vn: SIMD_SCRATCH,
   890→   890→                        })?;
   891→   891→                    }
   892→   892→                }
   893→   893→            }
   894→   894→
   895→   895→            IRInstr::Call {
   896→   896→                dst,
   897→   897→                func: target_name,
   898→   898→                args,
   899→   899→            } => {
   900→   900→                // Resolve all argument source registers FIRST, before moving
   901→   901→                // any of them. This prevents a later move from overwriting a
   902→   902→                // register that an earlier argument is in.
   903→   903→                //
   904→   904→                // IMPORTANT: For immediate arguments, resolve_reg always loads
   905→   905→                // into X9. When multiple immediates are present, we must use
   906→   906→                // different scratch registers to avoid overwriting. We load
   907→   907→                // immediates into X0-X7 directly (since they're the target
   908→   908→                // registers anyway) or use X11-X15 as temporary scratch.
   909→   909→                let arg_regs = [
   910→   910→                    Register::X0, Register::X1, Register::X2, Register::X3,
   911→   911→                    Register::X4, Register::X5, Register::X6, Register::X7,
   912→   912→                ];
   913→   913→                // Scratch registers for loading immediates when the target arg
   914→   914→                // register isn't available yet (because of conflicts).
   915→   915→                let scratch_regs = [
   916→   916→                    Register::X11, Register::X12, Register::X13, Register::X14,
   917→   917→                    Register::X15, Register::X3, Register::X4, Register::X5,
   918→   918→                ];
   919→   919→
   920→   920→                let mut src_regs: Vec<Register> = Vec::new();
   921→   921→                let mut immediate_scratch_used: Vec<usize> = Vec::new(); // which scratch reg each immediate uses
   922→   922→                let mut next_scratch = 0;
   923→   923→
   924→   924→                for (i, arg) in args.iter().enumerate() {
   925→   925→                    if i >= 8 {
   926→   926→                        break;
   927→   927→                    }
   928→   928→                    let is_immediate = matches!(arg, IRValue::Immediate(_) | IRValue::Address(_));
   929→   929→                    if is_immediate {
   930→   930→                        let v = match arg {
   931→   931→                            IRValue::Immediate(v) => *v,
   932→   932→                            IRValue::Address(a) => *a as i64,
   933→   933→                            _ => unreachable!(),
   934→   934→                        };
   935→   935→                        // Load the immediate directly into the target register
   936→   936→                        // if it's not needed by another argument.
   937→   937→                        // For simplicity, always load into a scratch register
   938→   938→                        // and record it.
   939→   939→                        if next_scratch < scratch_regs.len() {
   940→   940→                            let scratch = scratch_regs[next_scratch];
   941→   941→                            next_scratch += 1;
   942→   942→                            self.emit_load_immediate(scratch, v)?;
   943→   943→                            src_regs.push(scratch);
   944→   944→                        } else {
   945→   945→                            // Fallback to X9 (rare, only with 8+ immediates)
   946→   946→                            let scratch = Register::X9;
   947→   947→                            self.emit_load_immediate(scratch, v)?;
   948→   948→                            src_regs.push(scratch);
   949→   949→                        }
   950→   950→                    } else {
   951→   951→                        let src = self.resolve_reg(arg)?;
   952→   952→                        src_regs.push(src);
   953→   953→                    }
   954→   954→                }
   955→   955→
   956→   956→                // Now move arguments into X0–X7. We need to handle the case
   957→   957→                // where a source register is also a target register (cycle).
   958→   958→                // Strategy: first move arguments that don't conflict, then
   959→   959→                // handle conflicts using a scratch register.
   960→   960→                let n = src_regs.len();
   961→   961→                let mut moved = vec![false; n];
   962→   962→
   963→   963→                // Pass 1: move args where src != any dst that hasn't been moved yet
   964→   964→                // Simple approach: iterate and move non-conflicting ones first
   965→   965→                for pass in 0..2 {
   966→   966→                    for i in 0..n {
   967→   967→                        if moved[i] {
   968→   968→                            continue;
   969→   969→                        }
   970→   970→                        let src = src_regs[i];
   971→   971→                        let dst = arg_regs[i];
   972→   972→                        if src == dst {
   973→   973→                            // Already in the right register
   974→   974→                            moved[i] = true;
   975→   975→                            continue;
   976→   976→                        }
   977→   977→                        if pass == 0 {
   978→   978→                            // Check if src is needed as a source by a later unmoved arg
   979→   979→                            let mut conflict = false;
   980→   980→                            for j in 0..n {
   981→   981→                                if !moved[j] && j != i && src_regs[j] == dst {
   982→   982→                                    // Moving arg i to dst would overwrite arg j's source
   983→   983→                                    conflict = true;
   984→   984→                                    break;
   985→   985→                                }
   986→   986→                            }
   987→   987→                            if conflict {
   988→   988→                                continue; // Defer to pass 2
   989→   989→                            }
   990→   990→                        }
   991→   991→                        // Safe to move now
   992→   992→                        self.emit_instruction(Instruction::MOV {
   993→   993→                            rd: dst,
   994→   994→                            rm: src,
   995→   995→                        })?;
   996→   996→                        moved[i] = true;
   997→   997→                    }
   998→   998→                }
   999→   999→
  1000→  1000→                // Pass 2: handle any remaining conflicts with scratch register
  1001→  1001→                for i in 0..n {
  1002→  1002→                    if moved[i] {
  1003→  1003→                        continue;
  1004→  1004→                    }
  1005→  1005→                    let src = src_regs[i];
  1006→  1006→                    let dst = arg_regs[i];
  1007→  1007→                    if src == dst {
  1008→  1008→                        moved[i] = true;
  1009→  1009→                        continue;
  1010→  1010→                    }
  1011→  1011→                    // Use X16 as scratch to break the cycle
  1012→  1012→                    self.emit_instruction(Instruction::MOV {
  1013→  1013→                        rd: Register::X16,
  1014→  1014→                        rm: src,
  1015→  1015→                    })?;
  1016→  1016→                    self.emit_instruction(Instruction::MOV {
  1017→  1017→                        rd: dst,
  1018→  1018→                        rm: Register::X16,
  1019→  1019→                    })?;
  1020→  1020→                    moved[i] = true;
  1021→  1021→                }
  1022→  1022→
  1023→  1023→                // BL — record a relocation for later patching.
  1024→  1024→                let bl_word_idx = self.code.len();
  1025→  1025→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1026→  1026→                self.call_relocs.push(CallRelocation {
  1027→  1027→                    text_byte_offset: bl_byte_offset,
  1028→  1028→                    target_func: target_name.clone(),
  1029→  1029→                });
  1030→  1030→                // Register a RelocationEntry so the linker can patch this BL.
  1031→  1031→                self.relocations.push(RelocationEntry {
  1032→  1032→                    offset: bl_byte_offset,
  1033→  1033→                    symbol: target_name.clone(),
  1034→  1034→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1035→  1035→                });
  1036→  1036→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1037→  1037→
  1038→  1038→                if let Some(dst_val) = dst {
  1039→  1039→                    let rd = self.resolve_reg(dst_val)?;
  1040→  1040→                    if rd != Register::X0 {
  1041→  1041→                        self.emit_instruction(Instruction::MOV {
  1042→  1042→                            rd,
  1043→  1043→                            rm: Register::X0,
  1044→  1044→                        })?;
  1045→  1045→                    }
  1046→  1046→                }
  1047→  1047→            }

  1178→  1178→                                reg: temp,
  1179→  1179→                                shift: None,

  1241→  1241→                            shift: None,
  1242→  1242→                        },

  1338→  1338→
  1339→  1339→    /// Emit a binary operation (shared by `BinOp` and dedicated `Add`/`Sub`/…).
  1340→  1340→    ///
  1341→  1341→    /// Some ARM64 instructions (ADD, SUB, LSL, LSR, ASR) accept a 12-bit
  1342→  1342→    /// unsigned immediate operand directly.  Most others (MUL, AND, ORR, EOR,
  1343→  1343→    /// SDIV, UDIV, remainder, comparisons) **require** a register operand.
  1344→  1344→    /// When the RHS is an immediate that the target instruction cannot accept
  1345→  1345→    /// directly, we "spill" it into a scratch register (X16 / IP0) first via
  1346→  1346→    /// MOVZ + MOVK and then use that register as the operand.
  1347→  1347→    ///
  1348→  1348→    /// X16 (IP0) is chosen as the scratch register because it is **not** in
  1349→  1349→    /// the register allocator's free pool, so it will never conflict with
  1350→  1350→    /// `rd` or `rn`.
  1351→  1351→    fn emit_binop(
  1352→  1352→        &mut self,
  1353→  1353→        op: BinOpKind,
  1354→  1354→        dst: &IRValue,
  1355→  1355→        lhs: &IRValue,
  1356→  1356→        rhs: &IRValue,
  1357→  1357→        ty: Option<&IRType>,
  1358→  1358→    ) -> Result<()> {
  1359→  1359→        let width = RegWidth::from_ir_type(ty);
  1360→  1360→        let rd = self.resolve_reg(dst)?;
  1361→  1361→        let rn = self.resolve_reg(lhs)?;
  1362→  1362→
  1363→  1363→        // Determine whether the operation can accept an immediate operand.
  1364→  1364→        // ADD, SUB, LSL, LSR, ASR all have immediate forms in ARM64.
  1365→  1365→        // All other operations require a register operand.
  1366→  1366→        let supports_imm = matches!(
  1367→  1367→            op,
  1368→  1368→            BinOpKind::Add | BinOpKind::Sub | BinOpKind::Shl | BinOpKind::ShrL | BinOpKind::ShrA
  1369→  1369→        );
  1370→  1370→
  1371→  1371→        let rm = match rhs {
  1372→  1372→            IRValue::Immediate(v) => {
  1373→  1373→                if supports_imm && *v >= 0 && *v <= 4095 {
  1374→  1374→                    // Small immediate that the instruction accepts directly.
  1375→  1375→                    Operand::Imm12(*v as u16)
  1376→  1376→                } else {
  1377→  1377→                    // Load the immediate into scratch register X16 (IP0).
  1378→  1378→                    // X16 is not in the register allocator's pool, so it cannot
  1379→  1379→                    // conflict with `rd` or `rn`.
  1380→  1380→                    let scratch = Register::X16;
  1381→  1381→                    self.emit_load_immediate_with_width(scratch, *v, width)?;
  1382→  1382→                    Operand::Reg {
  1383→  1383→                        reg: scratch,
  1384→  1384→                        shift: None,
  1385→  1385→                    }
  1386→  1386→                }
  1387→  1387→            }
  1388→  1388→            _ => Operand::Reg {
  1389→  1389→                reg: self.resolve_reg(rhs)?,
  1390→  1390→                shift: None,
  1391→  1391→            },
  1392→  1392→        };
  1393→  1393→
  1394→  1394→        match op {
  1395→  1395→            BinOpKind::Add => {
  1396→  1396→                self.emit_instruction_with_width(Instruction::ADD { rd, rn, rm }, width)?;
  1397→  1397→            }
  1398→  1398→            BinOpKind::Sub => {
  1399→  1399→                self.emit_instruction_with_width(Instruction::SUB { rd, rn, rm }, width)?;
  1400→  1400→            }
  1401→  1401→            BinOpKind::Mul => {
  1402→  1402→                let rm_reg = self.operand_to_reg(&rm)?;
  1403→  1403→                self.emit_instruction_with_width(Instruction::MUL { rd, rn, rm: rm_reg }, width)?;
  1404→  1404→            }
  1405→  1405→            BinOpKind::SDiv => {
  1406→  1406→                let rm_reg = self.operand_to_reg(&rm)?;
  1407→  1407→                self.emit_instruction_with_width(Instruction::SDIV { rd, rn, rm: rm_reg }, width)?;
  1408→  1408→            }
  1409→  1409→            BinOpKind::UDiv => {
  1410→  1410→                let rm_reg = self.operand_to_reg(&rm)?;
  1411→  1411→                self.emit_instruction_with_width(Instruction::UDIV { rd, rn, rm: rm_reg }, width)?;
  1412→  1412→            }
  1413→  1413→            BinOpKind::And => {
  1414→  1414→                let rm_reg = self.operand_to_reg(&rm)?;
  1415→  1415→                self.emit_instruction_with_width(Instruction::AND { rd, rn, rm: rm_reg }, width)?;
  1416→  1416→            }
  1417→  1417→            BinOpKind::Or => {
  1418→  1418→                let rm_reg = self.operand_to_reg(&rm)?;
  1419→  1419→                self.emit_instruction_with_width(Instruction::ORR { rd, rn, rm: rm_reg }, width)?;
  1420→  1420→            }
  1421→  1421→            BinOpKind::Xor => {
  1422→  1422→                let rm_reg = self.operand_to_reg(&rm)?;
  1423→  1423→                self.emit_instruction_with_width(Instruction::EOR { rd, rn, rm: rm_reg }, width)?;
  1424→  1424→            }
  1425→  1425→            BinOpKind::Shl => {
  1426→  1426→                self.emit_instruction_with_width(Instruction::LSL { rd, rn, rm }, width)?;
  1427→  1427→            }
  1428→  1428→            BinOpKind::ShrL => {
  1429→  1429→                self.emit_instruction_with_width(Instruction::LSR { rd, rn, rm }, width)?;
  1430→  1430→            }
  1431→  1431→            BinOpKind::ShrA => {
  1432→  1432→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm }, width)?;
  1433→  1433→            }
  1434→  1434→            BinOpKind::Ror | BinOpKind::Rol => {
  1435→  1435→                let rm_reg = self.operand_to_reg(&rm)?;
  1436→  1436→                // ARM64 rotation: use ASR as placeholder
  1437→  1437→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm: Operand::Reg { reg: rm_reg, shift: None } }, width)?;
  1438→  1438→            }
  1439→  1439→            BinOpKind::SRem | BinOpKind::URem => {
  1440→  1440→                let rm_reg = self.operand_to_reg(&rm)?;
  1441→  1441→                let div_instr = if op == BinOpKind::SRem {
  1442→  1442→                    Instruction::SDIV { rd, rn, rm: rm_reg }
  1443→  1443→                } else {
  1444→  1444→                    Instruction::UDIV { rd, rn, rm: rm_reg }
  1445→  1445→                };
  1446→  1446→                self.emit_instruction_with_width(div_instr, width)?;
  1447→  1447→                // MSUB rd, rd, rm, rn  =>  rd = rn - rd * rm  =  dividend - quotient * divisor
  1448→  1448→                self.emit_instruction_with_width(Instruction::MSUB {
  1449→  1449→                    rd,
  1450→  1450→                    rn: rd,     // quotient (result of DIV)
  1451→  1451→                    rm: rm_reg, // divisor
  1452→  1452→                    ra: rn,     // dividend
  1453→  1453→                }, width)?;
  1454→  1454→            }
  1455→  1455→            BinOpKind::SLt
  1456→  1456→            | BinOpKind::SLe
  1457→  1457→            | BinOpKind::SGt
  1458→  1458→            | BinOpKind::SGe
  1459→  1459→            | BinOpKind::ULt
  1460→  1460→            | BinOpKind::ULe
  1461→  1461→            | BinOpKind::UGt
  1462→  1462→            | BinOpKind::UGe
  1463→  1463→            | BinOpKind::Eq
  1464→  1464→            | BinOpKind::Ne => {
  1465→  1465→                let rm_reg = self.operand_to_reg(&rm)?;
  1466→  1466→                self.emit_instruction_with_width(Instruction::CMP {
  1467→  1467→                    rn,
  1468→  1468→                    rm: Operand::Reg {
  1469→  1469→                        reg: rm_reg,
  1470→  1470→                        shift: None,
  1471→  1471→                    },
  1472→  1472→                }, width)?;
  1473→  1473→                let cond = binop_kind_to_condition(&op);
  1474→  1474→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1475→  1475→            }
  1476→  1476→        }
  1477→  1477→        Ok(())
  1478→  1478→    }
  1479→  1479→
  1480→  1480→    /// Emit the block terminator.
  1481→  1481→    fn emit_terminator(&mut self, term: &IRTerminator) -> Result<()> {
  1482→  1482→        match term {
  1483→  1483→            IRTerminator::Jump(target) => {
  1484→  1484→                let fixup_idx = self.code.len();
  1485→  1485→                self.fixups.push((fixup_idx, target.clone(), BranchFormat::B26));
  1486→  1486→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1487→  1487→            }
  1488→  1488→            IRTerminator::Branch {
  1489→  1489→                cond,
  1490→  1490→                true_block,
  1491→  1491→                false_block,
  1492→  1492→            } => {
  1493→  1493→                let rt = self.resolve_reg(cond)?;
  1494→  1494→                let fixup_cbnz = self.code.len();
  1495→  1495→                self.fixups.push((fixup_cbnz, true_block.clone(), BranchFormat::Cond19));
  1496→  1496→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1497→  1497→                let fixup_b = self.code.len();
  1498→  1498→                self.fixups.push((fixup_b, false_block.clone(), BranchFormat::B26));
  1499→  1499→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1500→  1500→            }
  1501→  1501→            IRTerminator::Return(vals) => {
  1502→  1502→                for (i, val) in vals.iter().enumerate() {
  1503→  1503→                    if i >= 8 {
  1504→  1504→                        break;
  1505→  1505→                    }
  1506→  1506→                    let src = self.resolve_reg(val)?;
  1507→  1507→                    let dst_reg = match i {
  1508→  1508→                        0 => Register::X0,
  1509→  1509→                        1 => Register::X1,
  1510→  1510→                        2 => Register::X2,
  1511→  1511→                        3 => Register::X3,
  1512→  1512→                        4 => Register::X4,
  1513→  1513→                        5 => Register::X5,
  1514→  1514→                        6 => Register::X6,
  1515→  1515→                        7 => Register::X7,
  1516→  1516→                        _ => unreachable!(),
  1517→  1517→                    };
  1518→  1518→                    if src != dst_reg {
  1519→  1519→                        self.emit_instruction(Instruction::MOV {
  1520→  1520→                            rd: dst_reg,
  1521→  1521→                            rm: src,
  1522→  1522→                        })?;
  1523→  1523→                    }
  1524→  1524→                }
  1525→  1525→                // Restore SP to point to the saved FP/LR pair.
  1526→  1526→                // X29 was set to SP right after the prologue's SUB SP, SP, #16
  1527→  1527→                // and STP X29, X30, [SP]. So X29 points to the FP/LR save area.
  1528→  1528→                // Using MOV SP, X29 (via ADD SP, X29, #0) is the most robust
  1529→  1529→                // way to restore SP regardless of how many Alloc instructions
  1530→  1530→                // modified it during the function body.
  1531→  1531→                self.emit_instruction(Instruction::ADD {
  1532→  1532→                    rd: Register::SP,
  1533→  1533→                    rn: Register::X29,
  1534→  1534→                    rm: Operand::Imm12(0),
  1535→  1535→                })?;
  1536→  1536→                // FP/LR were stored at [SP] after the prologue's SUB SP, SP, #16.
  1537→  1537→                // After the ADD above, SP points back to the FP/LR save area.
  1538→  1538→                // Load from [SP, #0], then restore SP by adding 16.
  1539→  1539→                self.emit_instruction(Instruction::LDP {
  1540→  1540→                    rt1: Register::X29,
  1541→  1541→                    rt2: Register::X30,
  1542→  1542→                    rn: Register::SP,
  1543→  1543→                    offset: 0,
  1544→  1544→                })?;
  1545→  1545→                self.emit_instruction(Instruction::ADD {
  1546→  1546→                    rd: Register::SP,
  1547→  1547→                    rn: Register::SP,
  1548→  1548→                    rm: Operand::Imm12(16),
  1549→  1549→                })?;
  1550→  1550→                self.emit_instruction(Instruction::RET { rn: None })?;
  1551→  1551→            }
  1552→  1552→            IRTerminator::Unreachable => {
  1553→  1553→                self.emit_instruction(Instruction::MOV {
  1554→  1554→                    rd: Register::XZR,
  1555→  1555→                    rm: Register::XZR,
  1556→  1556→                })?;
  1557→  1557→            }
  1558→  1558→            // Switch, Invoke, TailCall, and Resume are lowered by the
  1559→  1559→            // control_flow module before reaching the emitter. If they
  1560→  1560→            // appear here, it means the lowering pass was not run.
  1561→  1561→            IRTerminator::Switch { .. } => {
  1562→  1562→                return Err(CodegenError::InvalidInstruction(
  1563→  1563→                    "Switch terminator must be lowered before emission".to_string(),
  1564→  1564→                ));
  1565→  1565→            }
  1566→  1566→            IRTerminator::Invoke { .. } => {
  1567→  1567→                return Err(CodegenError::InvalidInstruction(
  1568→  1568→                    "Invoke terminator must be lowered before emission".to_string(),
  1569→  1569→                ));
  1570→  1570→            }
  1571→  1571→            IRTerminator::TailCall { .. } => {
  1572→  1572→                return Err(CodegenError::InvalidInstruction(
  1573→  1573→                    "TailCall terminator must be lowered before emission".to_string(),
  1574→  1574→                ));
  1575→  1575→            }
  1576→  1576→            IRTerminator::Resume { .. } => {
  1577→  1577→                return Err(CodegenError::InvalidInstruction(
  1578→  1578→                    "Resume terminator must be lowered before emission".to_string(),
  1579→  1579→                ));
  1580→  1580→            }
  1581→  1581→        }
  1582→  1582→        Ok(())
  1583→  1583→    }

  1589→  1589→    fn emit_instruction(&mut self, instr: Instruction) -> Result<()> {
  1590→  1590→        let word = instr.encode()?;
  1591→  1591→        self.code.push(word);
  1592→  1592→        Ok(())
  1593→  1593→    }

  1595→  1595→    /// Emit an instruction with a specific register width (32-bit W or 64-bit X).
  1596→  1596→    ///
  1597→  1597→    /// This is the primary emission method for arithmetic and logical instructions
  1598→  1598→    /// where the operand width affects the encoding. Using `RegWidth::W32` produces
  1599→  1599→    /// instructions that operate on W sub-registers, giving automatic 32-bit
  1600→  1600→    /// wrapping arithmetic.
  1601→  1601→    fn emit_instruction_with_width(&mut self, instr: Instruction, width: RegWidth) -> Result<()> {
  1602→  1602→        let word = instr.encode_with_width(width)?;
  1603→  1603→        self.code.push(word);
  1604→  1604→        Ok(())
  1605→  1605→    }
  1606→  1606→
  1607→  1607→    /// Emit a load instruction with the given offset and IR type.
  1608→  1608→    ///
  1609→  1609→    /// Selects the correct ARM64 load variant based on the IR type:
  1610→  1610→    /// - I8/U8 → LDRB (byte, zero-extended)
  1611→  1611→    /// - I16/U16 → LDRH (halfword, zero-extended)
  1612→  1612→    /// - I32/U32/Ptr/Func → LDR (word or doubleword)
  1613→  1613→    /// - I64/U64 → LDR (doubleword)
  1614→  1614→    ///
  1615→  1615→    /// If the offset fits the ARM64 unsigned-offset immediate encoding for the
  1616→  1616→    /// selected instruction variant, it is encoded directly. Otherwise, the
  1617→  1617→    /// effective address is computed in a scratch register (X9) first, and
  1618→  1618→    /// the load is performed at [X9 + 0].
  1619→  1619→    fn emit_load(&mut self, rt: Register, rn: Register, offset: i32, ty: &IRType) -> Result<()> {
  1620→  1620→        // Determine the ARM64 load instruction and its immediate-offset constraints.
  1621→  1621→        // For each variant: (scale, max_imm12) where the encoded offset = imm12 << scale.
  1622→  1622→        // LDRB:  scale=0, imm12 0..4095 → offset 0..4095
  1623→  1623→        // LDRH:  scale=1, imm12 0..4095 → offset 0..8190, even
  1624→  1624→        // LDRSW: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1625→  1625→        // LDR W: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1626→  1626→        // LDR X: scale=3, imm12 0..4095 → offset 0..32760, multiple of 8
  1627→  1627→        match ty {
  1628→  1628→            IRType::I8 | IRType::U8 => {
  1629→  1629→                if offset >= 0 && offset <= 4095 {
  1630→  1630→                    self.emit_instruction(Instruction::LDRB { rt, rn, offset })?;
  1631→  1631→                } else {
  1632→  1632→                    self.emit_address_with_offset(rn, offset)?;
  1633→  1633→                    self.emit_instruction(Instruction::LDRB {
  1634→  1634→                        rt,
  1635→  1635→                        rn: Register::X9,
  1636→  1636→                        offset: 0,
  1637→  1637→                    })?;
  1638→  1638→                }
  1639→  1639→                // Sign-extend for signed byte loads (SXTB = SBFM Xd, Xn, #0, #7)
  1640→  1640→                if *ty == IRType::I8 {
  1641→  1641→                    self.emit_instruction(Instruction::SBFM {
  1642→  1642→                        rd: rt,
  1643→  1643→                        rn: rt,
  1644→  1644→                        immr: 0,
  1645→  1645→                        imms: 7,
  1646→  1646→                    })?;
  1647→  1647→                }
  1648→  1648→            }
  1649→  1649→            IRType::I16 | IRType::U16 => {
  1650→  1650→                if offset >= 0 && offset <= 8190 && offset % 2 == 0 {
  1651→  1651→                    self.emit_instruction(Instruction::LDRH { rt, rn, offset })?;
  1652→  1652→                } else {
  1653→  1653→                    self.emit_address_with_offset(rn, offset)?;
  1654→  1654→                    self.emit_instruction(Instruction::LDRH {
  1655→  1655→                        rt,
  1656→  1656→                        rn: Register::X9,
  1657→  1657→                        offset: 0,
  1658→  1658→                    })?;
  1659→  1659→                }
  1660→  1660→                // Sign-extend for signed halfword loads (SXTH = SBFM Xd, Xn, #0, #15)
  1661→  1661→                if *ty == IRType::I16 {
  1662→  1662→                    self.emit_instruction(Instruction::SBFM {
  1663→  1663→                        rd: rt,
  1664→  1664→                        rn: rt,
  1665→  1665→                        immr: 0,
  1666→  1666→                        imms: 15,
  1667→  1667→                    })?;
  1668→  1668→                }
  1669→  1669→            }
  1670→  1670→            IRType::I32 | IRType::U32 => {
  1671→  1671→                // 32-bit load uses LDR Wt encoding (scale=2, offset must be multiple of 4)
  1672→  1672→                if offset >= 0 && offset <= 16380 && offset % 4 == 0 {
  1673→  1673→                    self.emit_instruction(Instruction::LDR_W { rt, rn, offset })?;
  1674→  1674→                } else {
  1675→  1675→                    self.emit_address_with_offset(rn, offset)?;
  1676→  1676→                    self.emit_instruction(Instruction::LDR_W {
  1677→  1677→                        rt,
  1678→  1678→                        rn: Register::X9,
  1679→  1679→                        offset: 0,
  1680→  1680→                    })?;
  1681→  1681→                }
  1682→  1682→                // Sign-extend for signed word loads (SXTW = SBFM Xd, Xn, #0, #31)
  1683→  1683→                if *ty == IRType::I32 {
  1684→  1684→                    self.emit_instruction(Instruction::SBFM {
  1685→  1685→                        rd: rt,
  1686→  1686→                        rn: rt,
  1687→  1687→                        immr: 0,
  1688→  1688→                        imms: 31,
  1689→  1689→                    })?;
  1690→  1690→                }
  1691→  1691→            }
  1692→  1692→            IRType::I64 | IRType::U64 | IRType::Ptr | IRType::Func => {
  1693→  1693→                // 64-bit load uses LDR Xt encoding (scale=3, offset must be multiple of 8)
  1694→  1694→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1695→  1695→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1696→  1696→                } else {
  1697→  1697→                    self.emit_address_with_offset(rn, offset)?;
  1698→  1698→                    self.emit_instruction(Instruction::LDR {
  1699→  1699→                        rt,
  1700→  1700→                        rn: Register::X9,
  1701→  1701→                        offset: 0,
  1702→  1702→                    })?;
  1703→  1703→                }
  1704→  1704→            }
  1705→  1705→            _ => {
  1706→  1706→                // Default: treat as 64-bit load
  1707→  1707→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1708→  1708→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1709→  1709→                } else {
  1710→  1710→                    self.emit_address_with_offset(rn, offset)?;
  1711→  1711→                    self.emit_instruction(Instruction::LDR {
  1712→  1712→                        rt,
  1713→  1713→                        rn: Register::X9,
  1714→  1714→                        offset: 0,
  1715→  1715→                    })?;
  1716→  1716→                }
  1717→  1717→            }
  1718→  1718→        }
  1719→  1719→        Ok(())
  1720→  1720→    }
  1721→  1721→
  1722→  1722→    /// Emit a store instruction with the given offset and IR type.
  1723→  1723→    ///
  1724→  1724→    /// Selects the correct ARM64 store variant based on the IR type:
  1725→  1725→    /// - I8/U8 → STRB (byte)
  1726→  1726→    /// - I16/U16 → STRH (halfword)
  1727→  1727→    /// - I32/U32 → STR_W (32-bit word)
  1728→  1728→    /// - I64/U64/Ptr/Func → STR (64-bit doubleword)
  1729→  1729→    ///
  1730→  1730→    /// If the offset fits the ARM64 unsigned-offset immediate encoding for the
  1731→  1731→    /// selected instruction variant, it is encoded directly. Otherwise, the
  1732→  1732→    /// effective address is computed in a scratch register (X9) first, and
  1733→  1733→    /// the store is performed at [X9 + 0].
  1734→  1734→    fn emit_store(&mut self, rt: Register, rn: Register, offset: i32, ty: &IRType) -> Result<()> {
  1735→  1735→        match ty {
  1736→  1736→            IRType::I8 | IRType::U8 => {
  1737→  1737→                if offset >= 0 && offset <= 4095 {
  1738→  1738→                    self.emit_instruction(Instruction::STRB { rt, rn, offset })?;
  1739→  1739→                } else {
  1740→  1740→                    self.emit_address_with_offset(rn, offset)?;
  1741→  1741→                    self.emit_instruction(Instruction::STRB {
  1742→  1742→                        rt,
  1743→  1743→                        rn: Register::X9,
  1744→  1744→                        offset: 0,
  1745→  1745→                    })?;
  1746→  1746→                }
  1747→  1747→            }
  1748→  1748→            IRType::I16 | IRType::U16 => {
  1749→  1749→                if offset >= 0 && offset <= 8190 && offset % 2 == 0 {
