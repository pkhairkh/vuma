     1→//! Abstract Syntax Tree types for the VUMA language frontend.
     2→//!
     3→//! The AST is the central data structure produced by the parser. Every node
     4→//! carries an optional [`Span`] so that downstream passes (type checking,
     5→//! SCG generation, error reporting) can trace back to the original source.
     6→//!
     7→//! Design principles:
     8→//! - **Minimal**: only the constructs needed for a memory-oriented,
     9→//!   AI-consumable language.
    10→//! - **Explicit**: allocation, free, cast, and region operations are
    11→//!   first-class statements rather than library calls.
    12→//! - **Spanned**: every node can point back to its source location.
    13→
    14→use crate::error::Span;
    15→use serde::{Deserialize, Serialize};
    16→
    17→// ---------------------------------------------------------------------------
    18→// Visibility
    19→// ---------------------------------------------------------------------------
    20→
    21→/// Visibility modifier for items.
    22→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, Default)]
    23→pub enum Visibility {
    24→    /// Default (private) visibility — no modifier.
    25→    #[default]
    26→    Private,
    27→    /// `pub` — public visibility.
    28→    Public,
    29→    /// `pub(crate)` — visible within the current crate.
    30→    PublicCrate,
    31→    /// `pub(super)` — visible in the parent module.
    32→    PublicSuper,
    33→    /// `pub(in path)` — visible in the specified path.
    34→    PublicIn(String),
    35→}
    36→
    37→impl std::fmt::Display for Visibility {
    38→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    39→        match self {
    40→            Visibility::Private => write!(f, ""),
    41→            Visibility::Public => write!(f, "pub"),
    42→            Visibility::PublicCrate => write!(f, "pub(crate)"),
    43→            Visibility::PublicSuper => write!(f, "pub(super)"),
    44→            Visibility::PublicIn(path) => write!(f, "pub(in {})", path),
    45→        }
    46→    }
    47→}

    53→/// An outer attribute: `#[attr]` or `#[attr = value]` or `#[attr(key = value)]`
    54→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
    55→pub struct Attribute {
    56→    /// Whether this is an inner attribute (`#![...]`) or outer (`#[...]`).
    57→    pub is_inner: bool,
    58→    /// The attribute path/name (e.g. `inline`, `derive`, `cfg`).
    59→    pub name: String,
    60→    /// Optional single value: `#[inline(always)]` or `#[cfg(test)]`.
    61→    /// Stored as a string; structured parsing is left to later passes.
    62→    pub value: Option<AttrValue>,
    63→    /// Source span.
    64→    pub span: Span,
    65→}
    66→
    67→/// The value part of an attribute.
    68→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
    69→pub enum AttrValue {
    70→    /// A single identifier or literal: `#[inline(always)]` → `always`
    71→    Single(String),
    72→    /// Key-value pairs: `#[derive(Debug)]`, `#[cfg(test)]`, `#[allow(dead_code)]`
    73→    List(Vec<String>),
    74→    /// Key = value: `#[repr(C)]` → the `C` part
    75→    KeyValue {
    76→        /// Attribute key.
    77→        key: String,
    78→        /// Attribute value.
    79→        value: String,
    80→    },
    81→}

   100→/// A top-level declaration within a [`Program`].
   101→#[derive(Debug, Clone, Serialize, Deserialize)]
   102→pub enum Item {
   103→    /// Function definition: `fn name(params) -> T { … }`
   104→    FnDef(FnDef),
   105→    /// Struct definition: `struct Name { field: Type, … }`
   106→    StructDef(StructDef),
   107→    /// Enum definition: `enum Name { Variant, … }`
   108→    EnumDef(EnumDef),
   109→    /// Region declaration: `region name = allocate(size);`
   110→    RegionDef(RegionDef),
   111→    /// Import declaration: `import "path";`
   112→    Import(Import),
   113→    /// Export declaration: `export name;`
   114→    Export(Export),
   115→    /// Constant definition: `const name: T = value;`
   116→    Const(ConstDef),
   117→    /// Static definition: `static name: T = value;`
   118→    Static(StaticDef),
   119→    /// Module declaration: `module name { items }`
   120→    ModuleDef(ModuleDef),
   121→    /// Trait definition: `trait Name<T> { … }`
   122→    TraitDef(TraitDef),
   123→    /// Impl block: `impl TraitName for Type { … }` or `impl Type { … }`
   124→    ImplBlock(ImplBlock),
   125→    /// Extern block: `extern "C" { fn ...; fn ...; }`
   126→    ExternBlock(ExternBlockDef),
   127→    /// Top-level statement (assignment, expression, free, etc.) that
   128→    /// appears outside of any function body.
   129→    Stmt(Stmt),
   130→}

   443→/// A single statement within a [`Block`].
   444→#[derive(Debug, Clone, Serialize, Deserialize)]
   445→pub enum Stmt {
   446→    /// Variable binding: `let name [: T] = expr;`
   447→    Let(LetStmt),
   448→    /// Assignment: `name = expr;`  or  `*name = expr;`
   449→    Assign(AssignStmt),
   450→    /// Compound assignment: `name += expr;`, `name -= expr;`, etc.
   451→    CompoundAssign(CompoundAssignStmt),
   452→    /// Memory allocation: `allocate(expr)` — typically inside a `region` decl.
   453→    Allocate(AllocateStmt),
   454→    /// Memory deallocation: `free(expr);`
   455→    Free(FreeStmt),
   456→    /// Memory access / dereference: `*expr`  or  `(*expr).field`
   457→    Access(AccessStmt),
   458→    /// Type cast: `expr as Type`
   459→    Cast(CastStmt),
   460→    /// Conditional: `if expr { … } else { … }`
   461→    If(IfStmt),
   462→    /// Loop: `while expr { … }`
   463→    While(WhileStmt),
   464→    /// For loop: `for name in expr { … }`
   465→    For(ForStmt),
   466→    /// Infinite loop: `loop { … }`
   467→    Loop(LoopStmt),
   468→    /// An unsafe block (`unsafe { ... }`)
   469→    UnsafeBlock {
   470→        /// Block body.
   471→        body: Block,
   472→        /// Source span.
   473→        span: Span,
   474→    },
   475→    /// Pattern match: `match expr { arms }`
   476→    Match(MatchStmt),
   477→    /// Sync block: `sync { … }`
   478→    Sync(SyncBlock),
   479→    /// Return: `return [expr];`
   480→    Return(ReturnStmt),
   481→    /// Break: `break;` or `break expr;`
   482→    Break(BreakStmt),
   483→    /// Continue: `continue;`
   484→    Continue(ContinueStmt),
   485→    /// BD directive: `bd(name, expr)`, `repd(name, expr)`, `capd(name, expr)`, `reld(name, expr)`
   486→    BdDirective(BdDirectiveStmt),
   487→    /// Expression statement: `expr;`
   488→    Expr(ExprStmt),
   489→}

  1304→impl std::fmt::Display for Type {
  1305→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
  1306→        match self {
  1307→            Type::BDBase(name) => write!(f, "{}", name),
  1308→            Type::Ptr(inner) => write!(f, "*{}", inner),
  1309→            Type::RegionPtr { inner, region } => write!(f, "*{} @ {}", inner, region),
  1310→            Type::Array { element, size } => write!(f, "[{}; {}]", element, size),
  1311→            Type::Struct { name, .. } => write!(f, "{}", name),
  1312→            Type::Generic { name, args } => {
  1313→                let a: Vec<String> = args.iter().map(|t| t.to_string()).collect();
  1314→                write!(f, "{}<{}>", name, a.join(", "))
  1315→            }
  1316→            Type::Func {
  1317→                params,
