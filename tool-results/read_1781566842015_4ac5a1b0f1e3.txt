     1→//! # String Formatting and Output
     2→//!
     3→//! This module provides VUMA-verified string formatting primitives for
     4→//! printf-style output, integer/float conversion, padding, joining, and
     5→//! buffered writing. These are the foundational formatting operations that
     6→//! LLMs need when generating VUMA programs that produce human-readable
     7→//! output or serialize data to text streams.
     8→//!
     9→//! ## Relationship to VUMA Programs
    10→//!
    11→//! In `.vuma` source, these functions serve as the standard formatting
    12→//! library — the equivalent of C's `printf`/`sprintf` family or Rust's
    13→//! `std::fmt` module. They allow VUMA programs to convert numeric values
    14→//! to text representations in various bases (decimal, hexadecimal, octal,
    15→//! binary), format floating-point numbers with configurable precision, pad
    16→//! strings to fixed widths, join multiple strings, and write formatted
    17→//! output into byte buffers.
    18→//!
    19→//! ## BD Annotations
    20→//!
    21→//! All functions in this module are annotated with Behavioral Descriptions.
    22→//! Pure formatting functions (those that return `VumaString`) declare only
    23→//! { Read, Compare } capabilities. Buffer-writing functions additionally
    24→//! declare { Write } since they mutate the provided buffer.
    25→
    26→use crate::collections::VumaString;
    27→use crate::primitives::{CapD, CapFlag};

    29→// ---------------------------------------------------------------------------
    30→// Integer Formatting
    31→// ---------------------------------------------------------------------------
    32→
    33→/// Format a signed 64-bit integer as a string in the given base with
    34→/// minimum width padding.
    35→///
    36→/// Converts `value` to its textual representation in `base` (2–36). If the
    37→/// textual representation is shorter than `width`, it is left-padded with
    38→/// `'0'` characters. A negative value is prefixed with `'-'`; the sign
    39→/// occupies one character position and is not counted as a digit.
    40→///
    41→/// ## Supported Bases
    42→///
    43→/// | Base | Name       | Digits                    |
    44→/// |------|------------|---------------------------|
    45→/// | 2    | Binary     | 0–1                       |
    46→/// | 8    | Octal      | 0–7                       |
    47→/// | 10   | Decimal    | 0–9                       |
    48→/// | 16   | Hexadecimal| 0–9, a–f                  |
    49→/// | 36   | Base-36    | 0–9, a–z                  |
    50→///
    51→/// Bases outside the range 2–36 are treated as base 10.
    52→///
    53→/// ## VUMA Program Equivalent
    54→///
    55→/// ```vuma
    56→/// fn format_int(value: i64, base: u32, width: u32) -> VumaString {
    57→///     negative: bool = value < 0;
    58→///     abs_val: u64 = if negative { -value as u64 } else { value as u64 };
    59→///     digits: VumaString = format_uint(abs_val, base, width);
    60→///     if negative {
    61→///         return "-" + digits;
    62→///     }
    63→///     return digits;
    64→/// }
    65→/// ```
    66→///
    67→/// ## BD Annotations
    68→///
    69→/// - CapD: { Read, Compare } — pure conversion, no side effects
    70→// VUMA-VERIFIED: integer formatting is correct for all i64 values in bases 2–36
    71→pub fn format_int(value: i64, base: u32, width: u32) -> VumaString {
    72→    let effective_base = if base >= 2 && base <= 36 { base } else { 10 };
    73→
    74→    if value < 0 {
    75→        let abs_val = (value as i128).unsigned_abs() as u64;
    76→        let digits = format_uint_inner(abs_val, effective_base, if width > 1 { width - 1 } else { 0 });
    77→        let mut result = VumaString::new();
    78→        result.push('-');
    79→        result.push_str(digits.as_str());
    80→        result
    81→    } else {
    82→        format_uint_inner(value as u64, effective_base, width)
    83→    }
    84→}
    85→
    86→/// Format an unsigned 64-bit integer as a string in the given base with
    87→/// minimum width padding.
    88→///
    89→/// Converts `value` to its textual representation in `base` (2–36). If the
    90→/// textual representation is shorter than `width`, it is left-padded with
    91→/// `'0'` characters.
    92→///
    93→/// ## Supported Bases
    94→///
    95→/// Same as [`format_int`]. Bases outside the range 2–36 are treated as
    96→/// base 10.
    97→///
    98→/// ## VUMA Program Equivalent
    99→///
   100→/// ```vuma
   101→/// fn format_uint(value: u64, base: u32, width: u32) -> VumaString {
   102→///     if value == 0 {
   103→///         return pad_left("0", width, '0');
   104→///     }
   105→///     digits: [u8; 64];
   106→///     pos: u32 = 0;
   107→///     v: u64 = value;
   108→///     while v > 0 {
   109→///         digit: u64 = v % (base as u64);
   110→///         if digit < 10 {
   111→///             digits[pos] = '0' + (digit as u8);
   112→///         } else {
   113→///             digits[pos] = 'a' + ((digit - 10) as u8);
   114→///         }
   115→///         pos = pos + 1;
   116→///         v = v / (base as u64);
   117→///     }
   118→///     result: VumaString = "";
   119→///     for i in (0..pos).rev() {
   120→///         result.push(digits[i] as char);
   121→///     }
   122→///     return pad_left(result, width, '0');
   123→/// }
   124→/// ```
   125→///
   126→/// ## BD Annotations
   127→///
   128→/// - CapD: { Read, Compare } — pure conversion, no side effects
   129→// VUMA-VERIFIED: unsigned integer formatting is correct for all u64 values in bases 2–36
   130→pub fn format_uint(value: u64, base: u32, width: u32) -> VumaString {
   131→    let effective_base = if base >= 2 && base <= 36 { base } else { 10 };
   132→    format_uint_inner(value, effective_base, width)
   133→}
   134→
   135→/// Internal helper: format a u64 value in the given base with zero-padding.
   136→fn format_uint_inner(value: u64, base: u32, width: u32) -> VumaString {
   137→    const DIGITS: &[u8; 36] = b"0123456789abcdefghijklmnopqrstuvwxyz";
   138→
   139→    if value == 0 {
   140→        let zero_count = width.saturating_sub(1) as usize;
   141→        let mut result = VumaString::new();
   142→        for _ in 0..zero_count {
   143→            result.push('0');
   144→        }
   145→        result.push('0');
   146→        return result;
   147→    }
   148→
   149→    // Maximum digits for u64 in base 2 is 64
   150→    let mut buf = [0u8; 64];
   151→    let mut pos = 0usize;
   152→    let mut v = value;
   153→
   154→    while v > 0 {
   155→        let digit = (v % base as u64) as usize;
   156→        buf[pos] = DIGITS[digit];
   157→        pos += 1;
   158→        v /= base as u64;
   159→    }
   160→
   161→    // Apply zero-padding
   162→    let padding = (width as usize).saturating_sub(pos);
   163→
   164→    let mut result = VumaString::with_capacity(padding + pos);
   165→    for _ in 0..padding {
   166→        result.push('0');
   167→    }
   168→    // Write digits in reverse order
   169→    for i in (0..pos).rev() {
   170→        result.push(buf[i] as char);
   171→    }
   172→
   173→    result
   174→}

   179→
   180→/// Format a 64-bit floating-point number as a string with the given
   181→/// decimal precision.
   182→///
   183→/// Produces a decimal representation of `value` with exactly `precision`
   184→/// digits after the decimal point. If `precision` is 0, no decimal point
   185→/// is emitted.
   186→///
   187→/// ## Special Values
   188→///
   189→/// - `NaN` is formatted as `"nan"`.
   190→/// - Positive infinity is formatted as `"inf"`.
   191→/// - Negative infinity is formatted as `"-inf"`.
   192→/// - Zero is formatted as `"0"` (precision 0) or `"0.00..."` (precision > 0).
   193→///
   194→/// ## VUMA Program Equivalent
   195→///
   196→/// ```vuma
   197→/// fn format_float(value: f64, precision: u32) -> VumaString {
   198→///     // Delegates to the host's formatted float conversion
   199→///     return host_float_to_str(value, precision);
   200→/// }
   201→/// ```
   202→///
   203→/// ## BD Annotations
   204→///
   205→/// - CapD: { Read, Compare } — pure conversion, no side effects
   206→// VUMA-VERIFIED: float formatting produces correct decimal representation
   207→pub fn format_float(value: f64, precision: u32) -> VumaString {
   208→    if value.is_nan() {
   209→        return VumaString::from("nan");
   210→    }
   211→    if value.is_infinite() {
   212→        return if value.is_sign_negative() {
   213→            VumaString::from("-inf")
   214→        } else {
   215→            VumaString::from("inf")
   216→        };
   217→    }
   218→
   219→    let formatted = if precision == 0 {
   220→        format!("{:.0}", value)
   221→    } else {
   222→        format!("{:.1$}", value, precision as usize)
   223→    };
   224→
   225→    VumaString::from(formatted.as_str())
   226→}

   334→/// Left-pad a string to `width` characters using `fill` as the padding
   335→/// character.
   336→///
   337→/// If `s` already has length >= `width`, returns `s` unchanged (no
   338→/// truncation is performed). Otherwise, prepends `width - len(s)`
   339→/// copies of `fill` to the left of `s`.
   340→///
   341→/// ## VUMA Program Equivalent
   342→///
   343→/// ```vuma
   344→/// fn pad_left(s: &str, width: u32, fill: char) -> VumaString {
   345→///     len: u32 = s.length() as u32;
   346→///     if len >= width {
   347→///         return VumaString::from(s);
   348→///     }
   349→///     pad_count: u32 = width - len;
   350→///     result: VumaString = "";
   351→///     for i in 0..pad_count {
   352→///         result.push(fill);
   353→///     }
   354→///     result.push_str(s);
   355→///     return result;
   356→/// }
   357→/// ```
   358→///
   359→/// ## BD Annotations
   360→///
   361→/// - CapD: { Read, Compare } — pure function, no side effects
   362→// VUMA-VERIFIED: left padding is correct for all widths and fill characters
   363→pub fn pad_left(s: &str, width: u32, fill: char) -> VumaString {
   364→    let len = s.chars().count() as u32;
   365→    if len >= width {
   366→        return VumaString::from(s);
   367→    }
   368→
   369→    let pad_count = (width - len) as usize;
   370→    let mut result = VumaString::with_capacity(pad_count + s.len());
   371→    for _ in 0..pad_count {
   372→        result.push(fill);
   373→    }
   374→    result.push_str(s);
   375→    result
   376→}

   378→/// Right-pad a string to `width` characters using `fill` as the padding
   379→/// character.
   380→///
   381→/// If `s` already has length >= `width`, returns `s` unchanged (no
   382→/// truncation is performed). Otherwise, appends `width - len(s)` copies
   383→/// of `fill` to the right of `s`.
   384→///
   385→/// ## VUMA Program Equivalent
   386→///
   387→/// ```vuma
   388→/// fn pad_right(s: &str, width: u32, fill: char) -> VumaString {
   389→///     len: u32 = s.length() as u32;
   390→///     if len >= width {
   391→///         return VumaString::from(s);
   392→///     }
   393→///     pad_count: u32 = width - len;
   394→///     result: VumaString = VumaString::from(s);
   395→///     for i in 0..pad_count {
   396→///         result.push(fill);
   397→///     }
   398→///     return result;
   399→/// }
   400→/// ```
   401→///
   402→/// ## BD Annotations
   403→///
   404→/// - CapD: { Read, Compare } — pure function, no side effects
   405→// VUMA-VERIFIED: right padding is correct for all widths and fill characters
   406→pub fn pad_right(s: &str, width: u32, fill: char) -> VumaString {
   407→    let len = s.chars().count() as u32;
   408→    if len >= width {
   409→        return VumaString::from(s);
   410→    }
   411→
   412→    let pad_count = (width - len) as usize;
   413→    let mut result = VumaString::with_capacity(s.len() + pad_count);
   414→    result.push_str(s);
   415→    for _ in 0..pad_count {
   416→        result.push(fill);
   417→    }
   418→    result
   419→}

   425→/// Join a slice of string slices with a separator, producing a
   426→/// [`VumaString`].
   427→///
   428→/// Concatenates the elements of `parts`, placing `separator` between each
   429→/// adjacent pair. If `parts` is empty, returns an empty string. If
   430→/// `parts` has one element, returns that element without any separator.
   431→///
   432→/// ## VUMA Program Equivalent
   433→///
   434→/// ```vuma
   435→/// fn join(parts: &[&str], separator: &str) -> VumaString {
   436→///     if parts.length() == 0 {
   437→///         return VumaString::new();
   438→///     }
   439→///     result: VumaString = VumaString::from(parts[0]);
   440→///     for i in 1..parts.length() {
   441→///         result.push_str(separator);
   442→///         result.push_str(parts[i]);
   443→///     }
   444→///     return result;
   445→/// }
   446→/// ```
   447→///
   448→/// ## BD Annotations
   449→///
   450→/// - CapD: { Read, Compare } — pure function, no side effects
   451→// VUMA-VERIFIED: join produces correct concatenation with separator
   452→pub fn join(parts: &[&str], separator: &str) -> VumaString {
   453→    if parts.is_empty() {
   454→        return VumaString::new();
   455→    }
   456→
   457→    // Estimate capacity
   458→    let sep_len = separator.len();
   459→    let total_len: usize = parts.iter().map(|p| p.len()).sum::<usize>() + sep_len * (parts.len() - 1);
   460→    let mut result = VumaString::with_capacity(total_len);
   461→
   462→    result.push_str(parts[0]);
   463→    for part in &parts[1..] {
   464→        result.push_str(separator);
   465→        result.push_str(part);
   466→    }
   467→    result
   468→}

   474→/// Write a string slice into a byte buffer, returning the number of bytes
   475→/// written.
   476→///
   477→/// Copies the UTF-8 bytes of `s` into `buf`, up to `buf.len()` bytes. If
   478→/// `s` is longer than the buffer, only the prefix that fits is written —
   479→/// the output is **not** guaranteed to be valid UTF-8 if truncation occurs
   480→/// in the middle of a multi-byte sequence.
   481→///
   482→/// ## Return Value
   483→///
   484→/// The number of bytes actually written (which is `min(s.len(), buf.len())`).
   485→///
   486→/// ## VUMA Program Equivalent
   487→///
   488→/// ```vuma
   489→/// fn write_str(buf: &mut [u8], s: &str) -> u32 {
   490→///     n: u32 = min(s.length(), buf.length()) as u32;
   491→///     for i in 0..n {
   492→///         buf[i] = s[i];
   493→///     }
   494→///     return n;
   495→/// }
   496→/// ```
   497→///
   498→/// ## BD Annotations
   499→///
   500→/// - CapD: { Read, Write } — reads from string, writes to buffer
   501→/// - SyncEdge: produces a WriteEdge covering the written bytes
   502→// VUMA-VERIFIED: write_str is bounded by buffer length
   503→pub fn write_str(buf: &mut [u8], s: &str) -> u32 {
   504→    let s_bytes = s.as_bytes();
   505→    let n = buf.len().min(s_bytes.len());
   506→    buf[..n].copy_from_slice(&s_bytes[..n]);
   507→    n as u32
   508→}
