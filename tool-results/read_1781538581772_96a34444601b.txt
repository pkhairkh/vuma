     1→//! VUMA CLI — Command-line interface for the VUMA compiler.
     2→//!
     3→//! Subcommands:
     4→//! - `vuma build <file>` — Parse + compile to ARM64 ELF (default), save to output file
     5→//! - `vuma run <file>`   — Build + execute (via QEMU aarch64 or native)
     6→//! - `vuma check <file>` — Parse + SCG + BD inference + IVE verification only
     7→//! - `vuma emit <isa> <file>` — Compile to specific ISA
     8→//! - `vuma disasm <file>` — Read binary and disassemble
     9→//! - `vuma verify <file>` — Run IVE 5-invariant verification
    10→//! - `vuma repl` — Interactive REPL (parse expr, print AST)
    11→//! - `vuma lsp`  — Start Language Server (LSP) for IDE/LLM integration
    12→

   270→/// Read source file content with a human-readable error on failure.
   271→fn read_source(path: &PathBuf) -> Result<String, String> {

   290→/// Print compilation errors to stderr in a human-readable format.

   292→    for err in errors {
   293→        eprintln!("error[{}]: {}", err.stage(), err);

   297→/// Determine the default output path for a given input file.
   298→fn default_output_path(input: &Path) -> PathBuf {
   299→    let stem = input.file_stem().unwrap_or_default().to_string_lossy();
   300→    let dir = input.parent().unwrap_or(std::path::Path::new("."));
   301→    dir.join(format!("{}.o", stem))
   302→}

   308→/// `vuma build <file>` — Parse + compile to ELF, save to output file.
   309→fn cmd_build(
   310→    cli: &Cli,
   311→    file: &PathBuf,
   312→    output: &Option<PathBuf>,
   313→    target: &TargetArg,
   314→) -> Result<(), String> {
   315→    let source = read_source(file)?;
   316→    let config = make_config(cli, CompileTarget::from(*target));
   317→    let result = compile_with_path(&source, Some(file), &config).map_err(|errors| {
   318→        print_errors(&errors);
   319→        format!("compilation failed with {} error(s)", errors.len())
   320→    })?;
   321→
   322→    let out_path = output
   323→        .as_ref()
   324→        .cloned()
   325→        .unwrap_or_else(|| default_output_path(file));
   326→    fs::write(&out_path, &result.binary).map_err(|e| {
   327→        format!(
   328→            "error: cannot write output file '{}': {}",
   329→            out_path.display(),
   330→            e
   331→        )
   332→    })?;
   333→    // Make the output file executable on Unix.
   334→    #[cfg(unix)]
   335→    {
   336→        use std::os::unix::fs::PermissionsExt;
   337→        let mut perms = fs::metadata(&out_path)
   338→            .map_err(|e| format!("error: cannot stat '{}': {}", out_path.display(), e))?
   339→            .permissions();
   340→        perms.set_mode(0o755);
   341→        fs::set_permissions(&out_path, perms)
   342→            .map_err(|e| format!("error: cannot chmod '{}': {}", out_path.display(), e))?;
   343→    }
   344→
   345→    println!(
   346→        "Compiled {} -> {} ({} bytes, {} SCG nodes, {} IR instructions)",
   347→        file.display(),
   348→        out_path.display(),
   349→        result.binary.len(),
   350→        result.scg.node_count(),
   351→        result.ir_instruction_count,
   352→    );
   353→
   354→    // Print stage timings.
   355→    for (stage, ms) in &result.stage_timings {
   356→        println!("  {:20} {}ms", stage, ms);
   357→    }
   358→
   359→    Ok(())
   360→}

   362→/// `vuma run <file>` — Build + execute.
   363→fn cmd_run(cli: &Cli, file: &PathBuf, args: &[String]) -> Result<(), String> {
   364→    let source = read_source(file)?;
   365→    let config = make_config(cli, CompileTarget::Linux);
   366→    let result = compile_with_path(&source, Some(file), &config).map_err(|errors| {
   367→        print_errors(&errors);
   368→        format!("compilation failed with {} error(s)", errors.len())
   369→    })?;
   370→
   371→    // Write binary to a temp file.
   372→    let tmp_dir = std::env::temp_dir();
   373→    let exe_path = tmp_dir.join(format!("vuma_run_{}", std::process::id()));
   374→    fs::write(&exe_path, &result.binary).map_err(|e| {
   375→        format!(
   376→            "error: cannot write temporary executable '{}': {}",
   377→            exe_path.display(),
   378→            e
   379→        )
   380→    })?;
   381→
   382→    // Make the file executable on Unix.
   383→    #[cfg(unix)]
   384→    {
   385→        use std::os::unix::fs::PermissionsExt;
   386→        let mut perms = fs::metadata(&exe_path)
   387→            .map_err(|e| format!("error: cannot stat '{}': {}", exe_path.display(), e))?
   388→            .permissions();
   389→        perms.set_mode(0o755);
   390→        fs::set_permissions(&exe_path, perms)
   391→            .map_err(|e| format!("error: cannot chmod '{}': {}", exe_path.display(), e))?;
   392→    }
   393→
   394→    // Try to execute: first natively (if on aarch64), then via qemu-aarch64.
   395→    let native_output = Command::new(&exe_path).args(args).output();
   396→
   397→    let output = match native_output {
   398→        Ok(out) => out,
   399→        Err(_) => {
   400→            // Native execution failed; try qemu-aarch64.
   401→            let qemu_output = Command::new("qemu-aarch64")
   402→                .arg(&exe_path)
   403→                .args(args)
   404→                .output()
   405→                .map_err(|e| {
   406→                    format!(
   407→                        "error: failed to execute binary (neither native nor qemu-aarch64): {}",
   408→                        e
   409→                    )
   410→                })?;
   411→            qemu_output
   412→        }
   413→    };
   414→
   415→    io::stdout()
   416→        .write_all(&output.stdout)
   417→        .map_err(|e| format!("error: failed to write program output: {}", e))?;
   418→    io::stderr()
   419→        .write_all(&output.stderr)
   420→        .map_err(|e| format!("error: failed to write program stderr: {}", e))?;
   421→
   422→    // Clean up temp file.
   423→    let _ = fs::remove_file(&exe_path);
   424→
   425→    let code = output.status.code().unwrap_or(1);
   426→    if code != 0 {
   427→        return Err(format!("program exited with code {}", code));
   428→    }
   429→
   430→    Ok(())
   431→}

   433→/// `vuma check <file>` — Parse + SCG + BD inference + IVE verification only.
   434→fn cmd_check(cli: &Cli, file: &PathBuf) -> Result<(), String> {
   435→    let source = read_source(file)?;
   436→    let mut config = make_config(cli, CompileTarget::Linux);
   437→    // check mode: always run verification, don't skip
   438→    if config.verification_level == VerificationLevel::None {
   439→        config.verification_level = VerificationLevel::Normal;
   440→    }
   441→
   442→    // Run the full compile — the check command doesn't save the binary,
   443→    // it just verifies the program compiles and passes verification.
   444→    let result = compile_with_path(&source, Some(file), &config).map_err(|errors| {
   445→        print_errors(&errors);
   446→        format!("check failed with {} error(s)", errors.len())
   447→    })?;
   448→
   449→    println!("Check passed for {}", file.display());
   450→    println!("  SCG nodes:      {}", result.scg.node_count());
   451→    println!("  IR functions:   {}", result.ir_function_count);
   452→    println!("  IR instructions: {}", result.ir_instruction_count);
   453→
   454→    if let Some(ref verification) = result.verification {
   455→        println!("  Verification:   {}", verification.overall);
   456→    }
   457→
   458→    // Print stage timings.
   459→    for (stage, ms) in &result.stage_timings {
   460→        println!("  {:20} {}ms", stage, ms);
   461→    }
   462→
   463→    Ok(())
   464→}

   466→/// `vuma emit <isa> <file>` — Compile to a specific ISA target.
   467→///
   468→/// Uses the direct AST → codegen SCG → IR path for better code quality
   469→/// than the main pipeline (which goes through the semantic SCG and loses
   470→/// most program semantics in the SCG→IR bridge).
   471→fn cmd_emit(
   472→    cli: &Cli,
   473→    isa: &IsaArg,
   474→    file: &PathBuf,
   475→    output: &Option<PathBuf>,
   476→) -> Result<(), String> {
   477→    let source = read_source(file)?;
   478→    let backend_kind = BackendKind::from(*isa);
   479→
   480→    // Step 1: Parse source → AST.
   481→    let mut parser = vuma_parser::Parser::new(&source);
   482→    let parse_result = parser.parse_program();
   483→    if parse_result.is_err() {
   484→        return Err(format!("parse error: {:?}", parse_result.errors));
   485→    }
   486→    if !parse_result.errors.is_empty() {
   487→        eprintln!("[emit] WARNING: {} non-fatal parse errors:", parse_result.errors.len());
   488→        for err in &parse_result.errors {
   489→            eprintln!("[emit]   {:?}", err);
   490→        }
   491→    }
   492→    let program = parse_result.value.unwrap();
   493→
   494→    // Step 2: Bridge parser AST → codegen SCG.
   495→    let codegen_scg = bridge_ast_to_codegen_scg(&program);
   496→
   497→    // Step 3: Lower codegen SCG → IR.
   498→    let mut ir_builder = ScgToIr::new();
   499→    let ir_program = ir_builder.convert(&codegen_scg).map_err(|e| {
   500→        format!("IR conversion error: {}", e)
   501→    })?;
   502→
   503→    eprintln!("[emit] IR program has {} functions", ir_program.functions.len());
   504→    for func in &ir_program.functions {
   505→        eprintln!("[emit] Function: {} ({} params, {} vregs)", func.name, func.params.len(), func.vregs.len());
   506→        for block in &func.blocks {
   507→            eprintln!("[emit]   Block: {}", block.label);
   508→            for instr in &block.instructions {
   509→                eprintln!("[emit]     {:?}", instr);
   510→            }
   511→        }
   512→    }
   513→
   514→    // Step 4: Create backend and allocate registers.
   515→    let backend = create_backend(backend_kind).map_err(|e| {
   516→        format!("error: cannot create {} backend: {}", backend_kind.isa_name(), e)
   517→    })?;
   518→
   519→    let mut allocated_functions = Vec::new();
   520→    for func in &ir_program.functions {
   521→        match backend.allocate_registers(func) {
   522→            Ok(allocated) => allocated_functions.push(allocated),
   523→            Err(e) => {
   524→                eprintln!("warning: register allocation failed for '{}': {}", func.name, e);
   525→            }
   526→        }
   527→    }
   528→
   529→    let out_path = output
   530→        .as_ref()
   531→        .cloned()
   532→        .unwrap_or_else(|| default_output_path(file));
   533→
   534→    // Step 5: Encode and write output.
   535→    if !allocated_functions.is_empty() {
   536→        let allocated_program = vuma_codegen::backend::AllocatedProgram {
   537→            functions: allocated_functions,
   538→            total_code_size: 0,
   539→            total_data_size: 0,
   540→        };
   541→        match backend.encode_program(&allocated_program) {
   542→            Ok(bytes) => {
   543→                fs::write(&out_path, &bytes).map_err(|e| {
   544→                    format!("error: cannot write output file '{}': {}", out_path.display(), e)
   545→                })?;
   546→                // Make the output file executable on Unix.
   547→                #[cfg(unix)]
   548→                {
   549→                    use std::os::unix::fs::PermissionsExt;
   550→                    let mut perms = fs::metadata(&out_path)
   551→                        .map_err(|e| format!("error: cannot stat '{}': {}", out_path.display(), e))?
   552→                        .permissions();
   553→                    perms.set_mode(0o755);
   554→                    fs::set_permissions(&out_path, perms)
   555→                        .map_err(|e| format!("error: cannot chmod '{}': {}", out_path.display(), e))?;
   556→                }
   557→                println!(
   558→                    "Emitted {} -> {} ({} bytes, ISA: {})",
   559→                    file.display(),
   560→                    out_path.display(),
   561→                    bytes.len(),
   562→                    backend.name(),
   563→                );
   564→            }
   565→            Err(e) => {
   566→                return Err(format!("{} encoding failed: {}", backend.name(), e));
   567→            }
   568→        }
   569→    } else {
   570→        return Err("no functions were successfully allocated".to_string());
   571→    }
   572→
   573→    Ok(())
   574→}
   575→
   576→/// Bridge parser AST → codegen SCG.
   577→///
   578→/// This converts the parser's AST into the codegen's SCG representation,
   579→/// which can then be lowered to IR. This bypasses the main pipeline's
   580→/// semantic SCG which loses most program semantics.
   581→///
   582→/// Key design: expressions are flattened into three-address code (TAC) by
   583→/// introducing temporary variables for sub-expressions. This preserves the
   584→/// full semantics of nested binary operations, function calls, dereferences,
   585→/// and casts.
   586→fn bridge_ast_to_codegen_scg(program: &vuma_parser::ast::Program) -> Scg {
   587→    use vuma_parser::ast::Item;
   588→
   589→    let mut nodes: Vec<ScgNode> = Vec::new();
   590→
   591→    for item in &program.items {
   592→        if let Item::FnDef(fn_def) = item {
   593→            let params: Vec<ScgParam> = fn_def
   594→                .params
   595→                .iter()
   596→                .map(|p| ScgParam {
   597→                    name: p.name.clone(),
   598→                    ty: bridge_type_to_codegen_scg(&p.ty),
   599→                })
   600→                .collect();
   601→
   602→            let results = if let Some(ref ret_ty) = fn_def.return_type {
   603→                vec![bridge_type_to_codegen_scg(&Some(ret_ty.clone()))]
   604→            } else {
   605→                vec![]
   606→            };
   607→
   608→            let mut ctx = BridgeCtx::new();
   609→            let mut body = bridge_block_to_scg_stmts(&fn_def.body, &mut ctx);
   610→
   611→            // Ensure every function ends with a Return statement.
   612→            // If the body doesn't end with a Return, add an implicit one.
   613→            // When the function has a return type and the last statement was an
   614→            // expression, use ctx.last_expr_result as the return value.
   615→            let has_return = body.last().map_or(false, |s| matches!(s, ScgStatement::Return(_)));
   616→            if !has_return {
   617→                let ret_val = if !results.is_empty() {
   618→                    // First check if the last expression was tracked.
   619→                    if let Some(ref expr) = ctx.last_expr_result {
   620→                        Some(expr.clone())
   621→                    } else {
   622→                        // Otherwise, look for the last computation/call result.
   623→                        body.iter().rev().find_map(|s| match s {
   624→                            ScgStatement::Computation(comp) => Some(ScgExpr::Var(comp.dst.clone())),
   625→                            ScgStatement::Call(call) => call.dst.as_ref().map(|d| ScgExpr::Var(d.clone())),
   626→                            _ => None,
   627→                        })
   628→                    }
   629→                } else {
   630→                    None
   631→                };
   632→                body.push(ScgStatement::Return(ret_val.into_iter().collect()));
   633→            }
   634→
   635→            nodes.push(ScgNode::Function(ScgFunction {
   636→                name: fn_def.name.clone(),
   637→                params,
   638→                results,
   639→                body,
   640→            }));
   641→        }
   642→    }
   643→
   644→    Scg { nodes }
   645→}
   646→
   647→/// Context for the AST → codegen SCG bridge, tracking a monotonic temp counter
   648→/// and the last expression result for implicit returns.
   649→struct BridgeCtx {
   650→    temp_counter: u32,
   651→    /// The result of the last expression statement, if any.
   652→    /// Used for implicit return when a function body ends with an expression.
   653→    last_expr_result: Option<ScgExpr>,
   654→}
   655→
   656→impl BridgeCtx {
   657→    fn new() -> Self {
   658→        Self { temp_counter: 0, last_expr_result: None }
   659→    }
   660→
   661→    /// Allocate a unique temporary variable name.
   662→    fn alloc_temp(&mut self) -> String {
   663→        let name = format!("__t{}", self.temp_counter);
   664→        self.temp_counter += 1;
   665→        name
   666→    }
   667→}
   668→
   669→/// Convert a parser type annotation to a codegen SCG type.
   670→fn bridge_type_to_codegen_scg(ty: &Option<vuma_parser::ast::Type>) -> ScgType {
   671→    match ty {
   672→        Some(vuma_parser::ast::Type::BDBase(name)) => match name.as_str() {
   673→            "i8" => ScgType::I8,
   674→            "i16" => ScgType::I16,
   675→            "i32" => ScgType::I32,
   676→            "i64" => ScgType::I64,
   677→            "u8" => ScgType::U8,
   678→            "u16" => ScgType::U16,
   679→            "u32" => ScgType::U32,
   680→            "u64" => ScgType::U64,
   681→            _ => ScgType::I64,
   682→        },
   683→        Some(vuma_parser::ast::Type::Ptr(_)) => ScgType::Ptr,
   684→        Some(vuma_parser::ast::Type::RegionPtr { .. }) => ScgType::Ptr,
   685→        _ => ScgType::Void,
   686→    }
   687→}

   690→/// Convert a parser block into codegen SCG statements, flattening expressions
   691→/// into three-address code with temporaries.
   692→fn bridge_block_to_scg_stmts(block: &vuma_parser::ast::Block, ctx: &mut BridgeCtx) -> Vec<ScgStatement> {
   693→    block
   694→        .statements
   695→        .iter()
   696→        .flat_map(|s| bridge_stmt_to_scg(s, ctx))
   697→        .collect()
   698→}
   699→
   700→/// Flatten an expression into three-address code. Returns the ScgExpr that
   701→/// holds the result, and appends any intermediate computation statements
   702→/// to `stmts`.
   703→///
   704→/// This is the core of the bridge: it recursively decomposes nested
   705→/// expressions into a sequence of simple computation nodes, each operating
   706→/// on at most two operands and producing one result. This preserves the
   707→/// full semantics of the original expression tree.
   708→fn flatten_expr(
   709→    expr: &vuma_parser::ast::Expr,
   710→    stmts: &mut Vec<ScgStatement>,
   711→    ctx: &mut BridgeCtx,
   712→) -> ScgExpr {
   713→    use vuma_parser::ast::{Expr, Lit, UnOp};
   714→
   715→    match expr {
   716→        // ── Leaf expressions: return directly, no flattening needed ──
   717→        Expr::Var { name, .. } => ScgExpr::Var(name.clone()),
   718→        Expr::Lit { value, .. } => match value {
   719→            Lit::Int(n) => ScgExpr::Int(*n),
   720→            Lit::Float(f) => ScgExpr::Float(*f),
   721→            Lit::Bool(b) => ScgExpr::Int(if *b { 1 } else { 0 }),
   722→            Lit::Address(a) => ScgExpr::Int(*a as i64),
   723→            Lit::String(_) => ScgExpr::Int(0),
   724→        },
   725→
   726→        // ── Binary operations: flatten lhs and rhs, then emit one Computation ──
   727→        Expr::BinOp { op, lhs, rhs, .. } => {
   728→            let lhs_expr = flatten_expr(lhs, stmts, ctx);
   729→            let rhs_expr = flatten_expr(rhs, stmts, ctx);
   730→            let dst = ctx.alloc_temp();
   731→            let binop_kind = map_ast_binop(op);
   732→            stmts.push(ScgStatement::Computation(ComputationNode {
   733→                dst: dst.clone(),
   734→                op: binop_kind,
   735→                lhs: lhs_expr,
   736→                rhs: rhs_expr,
   737→                tail_call: false,
   738→            }));
   739→            ScgExpr::Var(dst)
   740→        }
   741→
   742→        // ── Unary operations: flatten operand, then emit one Computation ──
   743→        Expr::UnOp { op, expr: operand, .. } => {
   744→            let operand_expr = flatten_expr(operand, stmts, ctx);
   745→            let dst = ctx.alloc_temp();
   746→            match op {
   747→                UnOp::BitNot => {
   748→                    stmts.push(ScgStatement::Computation(ComputationNode {
   749→                        dst: dst.clone(),
   750→                        op: BinOpKind::Xor,
   751→                        lhs: operand_expr,
   752→                        rhs: ScgExpr::Int(-1),
   753→                        tail_call: false,
   754→                    }));
   755→                }
   756→                UnOp::Neg => {
   757→                    stmts.push(ScgStatement::Computation(ComputationNode {
   758→                        dst: dst.clone(),
   759→                        op: BinOpKind::Sub,
   760→                        lhs: ScgExpr::Int(0),
   761→                        rhs: operand_expr,
   762→                        tail_call: false,
   763→                    }));
   764→                }
   765→                UnOp::Not => {
   766→                    stmts.push(ScgStatement::Computation(ComputationNode {
   767→                        dst: dst.clone(),
   768→                        op: BinOpKind::Eq,
   769→                        lhs: operand_expr,
   770→                        rhs: ScgExpr::Int(0),
   771→                        tail_call: false,
   772→                    }));
   773→                }
   774→                UnOp::Deref => {
   775→                    stmts.push(ScgStatement::Access(AccessNode::Load {
   776→                        dst: dst.clone(),
   777→                        ptr: operand_expr,
   778→                        offset: None,
   779→                    }));
   780→                }
   781→            }
   782→            ScgExpr::Var(dst)
   783→        }
   784→
   785→        // ── Function call: flatten args, emit CallNode ──
   786→        Expr::Call { callee, args, .. } => {
   787→            let func_name = match callee.as_ref() {
   788→                Expr::Var { name, .. } => name.clone(),
   789→                _ => "_unknown".into(),
   790→            };
   791→            let flat_args: Vec<ScgExpr> = args.iter()
   792→                .map(|a| flatten_expr(a, stmts, ctx))
   793→                .collect();
   794→            let dst = ctx.alloc_temp();
   795→            stmts.push(ScgStatement::Call(CallNode {
   796→                dst: Some(dst.clone()),
   797→                func: func_name,
   798→                args: flat_args,
   799→            }));
   800→            ScgExpr::Var(dst)
   801→        }
   802→
   803→        // ── Dereference: flatten the address, emit Load ──
   804→        Expr::Deref { expr, .. } => {
   805→            let addr = flatten_expr(expr, stmts, ctx);
   806→            let dst = ctx.alloc_temp();
   807→            stmts.push(ScgStatement::Access(AccessNode::Load {
   808→                dst: dst.clone(),
   809→                ptr: addr,
   810→                offset: None,
   811→            }));
   812→            ScgExpr::Var(dst)
   813→        }
   814→
   815→        // ── Offset (pointer arithmetic): flatten base and offset, emit Add ──
   816→        Expr::Offset { base, offset, .. } => {
   817→            let base_expr = flatten_expr(base, stmts, ctx);
   818→            let off_expr = flatten_expr(offset, stmts, ctx);
   819→            let dst = ctx.alloc_temp();
   820→            stmts.push(ScgStatement::Computation(ComputationNode {
   821→                dst: dst.clone(),
   822→                op: BinOpKind::Add,
   823→                lhs: base_expr,
   824→                rhs: off_expr,
   825→                tail_call: false,
   826→            }));
   827→            ScgExpr::Var(dst)
   828→        }
   829→
   830→        // ── Cast: flatten operand, pass through ──
   831→        Expr::Cast { expr, .. } => flatten_expr(expr, stmts, ctx),
   832→
   833→        // ── TypeAscription: flatten inner expression ──
   834→        Expr::TypeAscription { expr, .. } => flatten_expr(expr, stmts, ctx),
   835→
   836→        // ── Index: flatten base and index, compute addr, emit Load ──
   837→        Expr::Index { expr, index, .. } => {
   838→            let base_expr = flatten_expr(expr, stmts, ctx);
   839→            let idx_expr = flatten_expr(index, stmts, ctx);
   840→            let addr = ctx.alloc_temp();
   841→            stmts.push(ScgStatement::Computation(ComputationNode {
   842→                dst: addr.clone(),
   843→                op: BinOpKind::Add,
   844→                lhs: base_expr,
   845→                rhs: idx_expr,
   846→                tail_call: false,
   847→            }));
   848→            let dst = ctx.alloc_temp();
   849→            stmts.push(ScgStatement::Access(AccessNode::Load {
   850→                dst: dst.clone(),
   851→                ptr: ScgExpr::Var(addr),
   852→                offset: None,
   853→            }));
   854→            ScgExpr::Var(dst)
   855→        }
   856→
   857→        // ── Range: just flatten start (range is handled by For) ──
   858→        Expr::Range { start, .. } => flatten_expr(start, stmts, ctx),
   859→
   860→        // ── Allocate: emit as a heap allocation call ──
   861→        Expr::Allocate { size, .. } => {
   862→            let size_expr = flatten_expr(size, stmts, ctx);
   863→            let dst = ctx.alloc_temp();
   864→            stmts.push(ScgStatement::Call(CallNode {
   865→                dst: Some(dst.clone()),
   866→                func: "__vuma_alloc".to_string(),
   867→                args: vec![size_expr],
   868→            }));
   869→            ScgExpr::Var(dst)
   870→        }
   871→
   872→        // ── Null → 0 ──
   873→        Expr::Null { .. } => ScgExpr::Int(0),
   874→
   875→        // ── Uninitialized → 0 ──
   876→        Expr::Uninitialized { .. } => ScgExpr::Int(0),
   877→
   878→        // ── Fallback for unsupported expression types ──
   879→        _ => ScgExpr::Int(0),
   880→    }
   881→}
   882→
   883→/// Map a VUMA AST BinOp to a codegen BinOpKind.
   884→fn map_ast_binop(op: &vuma_parser::ast::BinOp) -> BinOpKind {
   885→    use vuma_parser::ast::BinOp;
   886→    match op {
   887→        BinOp::Add => BinOpKind::Add,
   888→        BinOp::Sub => BinOpKind::Sub,
   889→        BinOp::Mul => BinOpKind::Mul,
   890→        BinOp::Div => BinOpKind::SDiv,
   891→        BinOp::Mod => BinOpKind::SRem,
   892→        BinOp::And => BinOpKind::And,
   893→        BinOp::Or => BinOpKind::Or,
   894→        BinOp::BitAnd => BinOpKind::And,
   895→        BinOp::BitOr => BinOpKind::Or,
   896→        BinOp::BitXor => BinOpKind::Xor,
   897→        BinOp::Shl => BinOpKind::Shl,
   898→        BinOp::Shr => BinOpKind::ShrL,
   899→        BinOp::Eq => BinOpKind::Eq,
   900→        BinOp::Ne => BinOpKind::Ne,
   901→        BinOp::Lt => BinOpKind::SLt,
   902→        BinOp::Le => BinOpKind::SLe,
   903→        BinOp::Gt => BinOpKind::SGt,
   904→        BinOp::Ge => BinOpKind::SGe,
   905→    }
   906→}

   908→/// Convert a single parser statement into zero or more codegen SCG statements.
   909→/// Uses `flatten_expr` to decompose nested expressions into three-address code.
   910→fn bridge_stmt_to_scg(stmt: &vuma_parser::ast::Stmt, ctx: &mut BridgeCtx) -> Vec<ScgStatement> {
   911→    use vuma_parser::ast::Stmt as PStmt;
   912→
   913→    match stmt {
   914→        // ── let x [: T] = expr ──
   915→        PStmt::Let(let_stmt) => {
   916→            let mut stmts = Vec::new();
   917→
   918→            // Check if the RHS is an allocate() call → AllocationNode::Stack
   919→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &let_stmt.value {
   920→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
   921→                    if name == "allocate" {
   922→                        let size: u32 = args.first()
   923→                            .and_then(|a| {
   924→                                if let vuma_parser::ast::Expr::Lit { value, .. } = a {
   925→                                    if let vuma_parser::ast::Lit::Int(n) = value {
   926→                                        return Some(*n as u32);
   927→                                    }
   928→                                }
   929→                                None
   930→                            })
   931→                            .unwrap_or(8);
   932→                        return vec![ScgStatement::Allocation(AllocationNode::Stack {
   933→                            name: let_stmt.name.clone(),
   934→                            size,
   935→                            ty: ScgType::Ptr,
   936→                        })];
   937→                    }
   938→                    // Other function calls → CallNode (flatten args)
   939→                    let flat_args: Vec<ScgExpr> = args.iter()
   940→                        .map(|a| flatten_expr(a, &mut stmts, ctx))
   941→                        .collect();
   942→                    stmts.push(ScgStatement::Call(CallNode {
   943→                        dst: Some(let_stmt.name.clone()),
   944→                        func: name.clone(),
   945→                        args: flat_args,
   946→                    }));
   947→                    return stmts;
   948→                }
   949→            }
   950→
   951→            // Check if the RHS is an Allocate expression → AllocationNode::Stack
   952→            if let vuma_parser::ast::Expr::Allocate { size, .. } = &let_stmt.value {
   953→                let size_val: u32 = match size.as_ref() {
   954→                    vuma_parser::ast::Expr::Lit { value, .. } => {
   955→                        if let vuma_parser::ast::Lit::Int(n) = value {
   956→                            *n as u32
   957→                        } else {
   958→                            8
   959→                        }
   960→                    }
   961→                    _ => 8,
   962→                };
   963→                return vec![ScgStatement::Allocation(AllocationNode::Stack {
   964→                    name: let_stmt.name.clone(),
   965→                    size: size_val,
   966→                    ty: ScgType::Ptr,
   967→                })];
   968→            }
   969→
   970→            // General case: flatten the expression and assign to dst
   971→            let result = flatten_expr(&let_stmt.value, &mut stmts, ctx);
   972→            match &result {
   973→                ScgExpr::Var(name) if name == &let_stmt.name => {}
   974→                _ => {
   975→                    stmts.push(ScgStatement::Computation(ComputationNode {
   976→                        dst: let_stmt.name.clone(),
   977→                        op: BinOpKind::Add,
   978→                        lhs: result,
   979→                        rhs: ScgExpr::Int(0),
   980→                        tail_call: false,
   981→                    }));
   982→                }
   983→            }
   984→            stmts
   985→        }
   986→
   987→        // ── target = value ──
   988→        PStmt::Assign(assign_stmt) => {
   989→            let mut stmts = Vec::new();
   990→
   991→            // Detect dereference writes: `*expr = val` → Access::Store
   992→            if let vuma_parser::ast::AssignTarget::Deref { expr, .. } = &assign_stmt.target {
   993→                let ptr = flatten_expr(expr, &mut stmts, ctx);
   994→                let value = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
   995→                stmts.push(ScgStatement::Access(AccessNode::Store {
   996→                    ptr,
   997→                    offset: None,
   998→                    value,
   999→                }));
  1000→                return stmts;
  1001→            }
  1002→
  1003→            // Handle Index target: ptr[index] = value
  1004→            if let vuma_parser::ast::AssignTarget::Index { expr, index, .. } = &assign_stmt.target {
  1005→                let base = flatten_expr(expr, &mut stmts, ctx);
  1006→                let idx = flatten_expr(index, &mut stmts, ctx);
  1007→                let addr = ctx.alloc_temp();
  1008→                stmts.push(ScgStatement::Computation(ComputationNode {
  1009→                    dst: addr.clone(),
  1010→                    op: BinOpKind::Add,
  1011→                    lhs: base,
  1012→                    rhs: idx,
  1013→                    tail_call: false,
  1014→                }));
  1015→                let value = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
  1016→                stmts.push(ScgStatement::Access(AccessNode::Store {
  1017→                    ptr: ScgExpr::Var(addr),
  1018→                    offset: None,
  1019→                    value,
  1020→                }));
  1021→                return stmts;
  1022→            }
  1023→
  1024→            let dst = match &assign_stmt.target {
  1025→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
  1026→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
  1027→                vuma_parser::ast::AssignTarget::Deref { .. } => "_deref".into(),
  1028→                vuma_parser::ast::AssignTarget::Index { .. } => "_index".into(),
  1029→            };
  1030→
  1031→            // Detect allocate() expression → AllocationNode::Stack
  1032→            if let vuma_parser::ast::Expr::Allocate { size, .. } = &assign_stmt.value {
  1033→                let size_val: u32 = match size.as_ref() {
  1034→                    vuma_parser::ast::Expr::Lit { value, .. } => {
  1035→                        if let vuma_parser::ast::Lit::Int(n) = value {
  1036→                            *n as u32
  1037→                        } else {
  1038→                            8
  1039→                        }
  1040→                    }
  1041→                    _ => 8,
  1042→                };
  1043→                return vec![ScgStatement::Allocation(AllocationNode::Stack {
  1044→                    name: dst,
  1045→                    size: size_val,
  1046→                    ty: ScgType::Ptr,
  1047→                })];
  1048→            }
  1049→
  1050→            // Detect function calls in assign: `x = foo(args)` → CallNode (flatten args)
  1051→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &assign_stmt.value {
  1052→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
  1053→                    let flat_args: Vec<ScgExpr> = args.iter()
  1054→                        .map(|a| flatten_expr(a, &mut stmts, ctx))
  1055→                        .collect();
  1056→                    stmts.push(ScgStatement::Call(CallNode {
  1057→                        dst: Some(dst),
  1058→                        func: name.clone(),
  1059→                        args: flat_args,
  1060→                    }));
  1061→                    return stmts;
  1062→                }
  1063→            }
  1064→
  1065→            // General case: flatten the value expression and assign to dst
  1066→            let result = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
  1067→            match &result {
  1068→                ScgExpr::Var(name) if name == &dst => {}
  1069→                _ => {
  1070→                    stmts.push(ScgStatement::Computation(ComputationNode {
  1071→                        dst,
  1072→                        op: BinOpKind::Add,
  1073→                        lhs: result,
  1074→                        rhs: ScgExpr::Int(0),
  1075→                        tail_call: false,
  1076→                    }));
  1077→                }
  1078→            }
  1079→            stmts
  1080→        }
  1081→
  1082→        // ── target op= value ──
  1083→        PStmt::CompoundAssign(ca_stmt) => {
  1084→            let mut stmts = Vec::new();
  1085→            let dst = match &ca_stmt.target {
  1086→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
  1087→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
  1088→                _ => "_".into(),
  1089→            };
  1090→            let binop = match ca_stmt.op {
  1091→                vuma_parser::ast::CompoundOp::Add => BinOpKind::Add,
  1092→                vuma_parser::ast::CompoundOp::Sub => BinOpKind::Sub,
  1093→                vuma_parser::ast::CompoundOp::Mul => BinOpKind::Mul,
  1094→                vuma_parser::ast::CompoundOp::Div => BinOpKind::SDiv,
  1095→                vuma_parser::ast::CompoundOp::Mod => BinOpKind::SRem,
  1096→                vuma_parser::ast::CompoundOp::BitAnd => BinOpKind::And,
  1097→                vuma_parser::ast::CompoundOp::BitOr => BinOpKind::Or,
  1098→                vuma_parser::ast::CompoundOp::BitXor => BinOpKind::Xor,
  1099→                vuma_parser::ast::CompoundOp::Shl => BinOpKind::Shl,
  1100→                vuma_parser::ast::CompoundOp::Shr => BinOpKind::ShrL,
  1101→            };
  1102→            let rhs = flatten_expr(&ca_stmt.value, &mut stmts, ctx);
  1103→            stmts.push(ScgStatement::Computation(ComputationNode {
  1104→                dst: dst.clone(),
  1105→                op: binop,
  1106→                lhs: ScgExpr::Var(dst),
  1107→                rhs,
  1108→                tail_call: false,
  1109→            }));
  1110→            stmts
  1111→        }
  1112→
  1113→        PStmt::Allocate(_) => vec![],
  1114→        PStmt::Free(_) => vec![],
  1115→
  1116→        PStmt::Cast(cast_stmt) => {
  1117→            let mut stmts = Vec::new();
  1118→            let _ = flatten_expr(&cast_stmt.expr, &mut stmts, ctx);
  1119→            stmts
  1120→        }
  1121→
  1122→        PStmt::Return(ret_stmt) => {
  1123→            let mut stmts = Vec::new();
  1124→            let values = match &ret_stmt.value {
  1125→                Some(expr) => vec![flatten_expr(expr, &mut stmts, ctx)],
  1126→                None => vec![],
  1127→            };
  1128→            stmts.push(ScgStatement::Return(values));
  1129→            stmts
  1130→        }
  1131→
  1132→        PStmt::Expr(expr_stmt) => {
  1133→            let mut stmts = Vec::new();
  1134→            let result = flatten_expr(&expr_stmt.expr, &mut stmts, ctx);
  1135→            ctx.last_expr_result = Some(result);
  1136→            stmts
  1137→        }
  1138→
  1139→        PStmt::If(if_stmt) => {
  1140→            let mut pre_stmts = Vec::new();
  1141→            let cond = flatten_expr(&if_stmt.condition, &mut pre_stmts, ctx);
  1142→            let then_body = bridge_block_to_scg_stmts(&if_stmt.then_block, ctx);
  1143→            let else_body = if_stmt
  1144→                .else_block
  1145→                .as_ref()
  1146→                .map(|b| bridge_block_to_scg_stmts(b, ctx));
  1147→            let mut result = pre_stmts;
  1148→            result.push(ScgStatement::Control(ControlNode::If {
  1149→                cond,
  1150→                then_body,
  1151→                else_body,
  1152→            }));
  1153→            result
  1154→        }
  1155→
  1156→        // while condition { body }
  1157→        // Lowered as: Loop { <compute cond>; if cond { body } else { break } }
  1158→        PStmt::While(while_stmt) => {
  1159→            let then_body = bridge_block_to_scg_stmts(&while_stmt.body, ctx);
  1160→            let mut loop_body = Vec::new();
  1161→            let cond = flatten_expr(&while_stmt.condition, &mut loop_body, ctx);
  1162→            loop_body.push(ScgStatement::Control(ControlNode::If {
  1163→                cond,
  1164→                then_body,
  1165→                else_body: Some(vec![ScgStatement::Control(ControlNode::Break)]),
  1166→            }));
  1167→            vec![ScgStatement::Control(ControlNode::Loop { body: loop_body })]
  1168→        }
  1169→
  1170→        // for name in start..end { body }
  1171→        // Lowered as:
  1172→        //   name = start
  1173→        //   loop { _cond = name < end; if _cond { body; name = name + 1 } else { break } }
  1174→        PStmt::For(for_stmt) => {
  1175→            let mut pre_stmts = Vec::new();
  1176→            let (start_expr, end_expr) = match &for_stmt.iter {
  1177→                vuma_parser::ast::Expr::Range { start, end, .. } => {
  1178→                    let s = flatten_expr(start, &mut pre_stmts, ctx);
  1179→                    let e = flatten_expr(end, &mut pre_stmts, ctx);
  1180→                    (s, e)
  1181→                }
  1182→                other => {
  1183→                    (ScgExpr::Int(0), ScgExpr::Int(0))
  1184→                }
  1185→            };
  1186→
  1187→            let init_stmt = ScgStatement::Computation(ComputationNode {
  1188→                dst: for_stmt.name.clone(),
  1189→                op: BinOpKind::Add,
  1190→                lhs: start_expr,
  1191→                rhs: ScgExpr::Int(0),
  1192→                tail_call: false,
  1193→            });
  1194→
  1195→            let mut loop_body = Vec::new();
  1196→            let cond_temp = format!("{}_cond", for_stmt.name);
  1197→            loop_body.push(ScgStatement::Computation(ComputationNode {
  1198→                dst: cond_temp.clone(),
  1199→                op: BinOpKind::SLt,
  1200→                lhs: ScgExpr::Var(for_stmt.name.clone()),
  1201→                rhs: end_expr.clone(),
  1202→                tail_call: false,
  1203→            }));
  1204→
  1205→            let inner_body = bridge_block_to_scg_stmts(&for_stmt.body, ctx);
  1206→            let mut full_then = inner_body;
  1207→            full_then.push(ScgStatement::Computation(ComputationNode {
  1208→                dst: for_stmt.name.clone(),
  1209→                op: BinOpKind::Add,
  1210→                lhs: ScgExpr::Var(for_stmt.name.clone()),
  1211→                rhs: ScgExpr::Int(1),
  1212→                tail_call: false,
  1213→            }));
  1214→
  1215→            loop_body.push(ScgStatement::Control(ControlNode::If {
  1216→                cond: ScgExpr::Var(cond_temp),
  1217→                then_body: full_then,
  1218→                else_body: Some(vec![ScgStatement::Control(ControlNode::Break)]),
  1219→            }));
  1220→
  1221→            let mut result = pre_stmts;
  1222→            result.push(init_stmt);
  1223→            result.push(ScgStatement::Control(ControlNode::Loop { body: loop_body }));
  1224→            result
  1225→        }
  1226→
  1227→        PStmt::Loop(loop_stmt) => {
  1228→            let body = bridge_block_to_scg_stmts(&loop_stmt.body, ctx);
  1229→            vec![ScgStatement::Control(ControlNode::Loop { body })]
  1230→        }
  1231→
  1232→        PStmt::Break(_) => vec![ScgStatement::Control(ControlNode::Break)],
  1233→        PStmt::Continue(_) => vec![ScgStatement::Control(ControlNode::Continue)],
  1234→        PStmt::BdDirective(_)
  1235→        | PStmt::Sync(_)
  1236→        | PStmt::UnsafeBlock { .. }
  1237→        | PStmt::Match(_)
  1238→        | PStmt::Access(_) => vec![],
  1239→    }
  1240→}
  1241→fn cmd_disasm(file: &PathBuf, isa: &IsaArg, base_addr_str: &str) -> Result<(), String> {
  1242→    let bytes = fs::read(file)
  1243→        .map_err(|e| format!("error: cannot read binary file '{}': {}", file.display(), e))?;
  1244→
  1245→    let base_addr = u64::from_str_radix(base_addr_str.trim_start_matches("0x"), 16)
  1246→        .map_err(|e| format!("error: invalid base address '{}': {}", base_addr_str, e))?;
  1247→
  1248→    let backend_kind = BackendKind::from(*isa);
  1249→    let backend = create_backend(backend_kind).map_err(|e| {
  1250→        format!(
  1251→            "error: cannot create {} backend: {}",
  1252→            backend_kind.isa_name(),
  1253→            e
  1254→        )
  1255→    })?;
  1256→
  1257→    let instructions = backend.disassemble(&bytes, base_addr);
  1258→
  1259→    println!(
  1260→        "Disassembly of {} ({} bytes, ISA: {}):",
  1261→        file.display(),
  1262→        bytes.len(),
  1263→        backend.name()
  1264→    );
  1265→    for line in &instructions {
  1266→        println!("{}", line);
  1267→    }
  1268→
  1269→    Ok(())
  1270→}

  1272→/// `vuma verify <file>` — Run IVE 5-invariant verification.
  1273→fn cmd_verify(cli: &Cli, file: &PathBuf) -> Result<(), String> {
  1274→    let source = read_source(file)?;
  1275→    let mut config = make_config(cli, CompileTarget::Linux);
  1276→    // Force exhaustive verification for the verify subcommand.

  1283→            errors.len()

  1287→    match result.verification {

  1292→                "  Summary: {} invariant(s) checked",
  1293→                verification.summary.total_checked
  1294→            );
  1295→            println!("           {} passed", verification.summary.passed);
  1296→            println!("           {} failed", verification.summary.failed);
  1297→            println!("           {} unverified", verification.summary.unverified);
  1298→
  1299→            // Print individual invariant results.
  1300→            for per_inv in &verification.per_invariant {
  1301→                let status_str = match &per_inv.result.status {
  1302→                    vuma_ive::VerificationStatus::Proven => "PROVEN",
  1303→                    vuma_ive::VerificationStatus::ProbablySafe { .. } => "PROBABLY_SAFE",
  1304→                    vuma_ive::VerificationStatus::Unverified { .. } => "UNVERIFIED",
  1305→                    vuma_ive::VerificationStatus::Violated { .. } => "VIOLATED",
  1306→                };
  1307→                println!(
  1308→                    "  [{}] {:?} — {}",
  1309→                    status_str, per_inv.kind, per_inv.result.message,
  1310→                );
  1311→                if let Some(ref evidence) = per_inv.result.evidence {
  1312→                    println!("    evidence: {}", evidence);
  1313→                }
  1314→            }
  1315→        }
  1316→        None => {
  1317→            println!("Verification was skipped (verification level set to None).");
  1318→        }
  1319→    }
  1320→
  1321→    Ok(())
  1322→}

  1324→/// `vuma repl` — Interactive REPL: parse expressions and print AST.
  1325→fn cmd_repl() -> Result<(), String> {
  1326→    println!("VUMA REPL v0.1.0");
  1327→    println!("Type VUMA expressions or statements. Enter ':quit' to exit, ':help' for help.");
  1328→
  1329→    let stdin = io::stdin();
  1330→    let mut input = String::new();
  1331→
  1332→    loop {
  1333→        print!("vuma> ");
  1334→        io::stdout()
  1335→            .flush()
  1336→            .map_err(|e| format!("flush error: {}", e))?;
  1337→
  1338→        input.clear();
  1339→        match stdin.read_line(&mut input) {
  1340→            Ok(0) => break, // EOF
  1341→            Ok(_) => {}
  1342→            Err(e) => return Err(format!("error reading input: {}", e)),
  1343→        }
  1344→
  1345→        let trimmed = input.trim();
  1346→        if trimmed.is_empty() {
  1347→            continue;
  1348→        }
  1349→        if trimmed == ":quit" || trimmed == ":q" {
  1350→            break;
  1351→        }
  1352→        if trimmed == ":help" || trimmed == ":h" {
  1353→            println!("  :quit, :q   — Exit the REPL");
  1354→            println!("  :help, :h   — Show this help");
  1355→            println!("  <expr>      — Parse and display the AST");
  1356→            println!("  <stmt>      — Parse and display the AST");
  1357→            println!("  <program>   — Parse and display the full program AST");
  1358→            continue;
  1359→        }
  1360→
  1361→        // Try to parse as a full program first.
  1362→        let mut parser = vuma_parser::Parser::new(trimmed);
  1363→        let result = parser.parse_program();
  1364→        if !result.has_errors() {
  1365→            // Successfully parsed; display the AST.
  1366→            let program = result.unwrap();
  1367→            println!("{:#?}", program);
  1368→            continue;
  1369→        }
  1370→        // Program parse had errors; try to parse as an expression via
  1371→        // wrapping it in a dummy program context.
  1372→        let program_errors = result.errors.clone();
  1373→        // Since parse_expr is private, we re-parse as a minimal program.
  1374→        let wrapped = format!("fn _repl_expr() {{ {} }}", trimmed);
  1375→        let mut expr_parser = vuma_parser::Parser::new(&wrapped);
  1376→        let expr_result = expr_parser.parse_program();
  1377→        if !expr_result.has_errors() {
  1378→            // Print the function body AST.
  1379→            let program = expr_result.unwrap();
  1380→            if let Some(item) = program.items.first() {
  1381→                println!("{:#?}", item);
  1382→            } else {
  1383→                println!("{:#?}", program);
  1384→            }
  1385→        } else {
  1386→            // Show the original parse errors.
  1387→            for err in &program_errors {
  1388→                eprintln!("parse error: {}", err);
  1389→            }
  1390→        }
  1391→    }
  1392→
  1393→    Ok(())
  1394→}

  1634→// Main entry point
  1635→// ═══════════════════════════════════════════════════════════════════════════

  1637→fn main() {
  1638→    // Initialize logger.
  1639→    env_logger::init();
  1640→
  1641→    let cli = Cli::parse();
  1642→
  1643→    // Handle --bench flag: run the benchmark suite.
  1644→    if cli.bench {
  1645→        cmd_bench(&cli);
  1646→        return;
  1647→    }
  1648→
  1649→    // Handle --repl flag: launch the full VumaRepl instead of subcommand.
  1650→    if cli.repl {
  1651→        let mut repl = vuma_core::repl::VumaRepl::new();
  1652→        if let Err(e) = repl.run() {
  1653→            eprintln!("REPL error: {e}");
  1654→            std::process::exit(1);
  1655→        }
  1656→        return;
  1657→    }
  1658→
  1659→    let command = match cli.command {
  1660→        Some(ref cmd) => cmd.clone(),
  1661→        None => {
  1662→            // No subcommand and no --repl: print help.
  1663→            eprintln!("No subcommand specified. Use `vuma --help` or `vuma --repl`.");
  1664→            std::process::exit(1);
  1665→        }
  1666→    };
  1667→
  1668→    let result = match command {
  1669→        Commands::Build {
  1670→            ref file,
  1671→            ref output,
  1672→            ref target,
  1673→        } => cmd_build(&cli, file, output, target),
  1674→        Commands::Run { ref file, ref args } => cmd_run(&cli, file, args),
  1675→        Commands::Check { ref file } => cmd_check(&cli, file),
  1676→        Commands::Emit {
  1677→            ref isa,
  1678→            ref file,
  1679→            ref output,
  1680→        } => cmd_emit(&cli, isa, file, output),
  1681→        Commands::Disasm {
  1682→            ref file,
  1683→            ref isa,
  1684→            ref base_addr,
  1685→        } => cmd_disasm(file, isa, base_addr),
  1686→        Commands::Verify { ref file } => cmd_verify(&cli, file),
  1687→        Commands::Repl => cmd_repl(),
  1688→        Commands::Lsp => cmd_lsp(),
  1689→        Commands::Pkg { ref cmd } => cmd_pkg(cmd),
  1690→    };
  1691→
  1692→    if let Err(err) = result {
  1693→        eprintln!("{}", err);
  1694→        std::process::exit(1);
  1695→    }
  1696→}

  1698→/// `vuma pkg init/build/add` — Package manager subcommands.
  1699→fn cmd_pkg(cmd: &PkgCommand) -> Result<(), String> {
  1700→    let dir = std::env::current_dir().map_err(|e| format!("cannot get current directory: {}", e))?;
  1701→    match cmd {
  1702→        PkgCommand::Init { name } => {
  1703→            vuma::init_package(&dir, name).map_err(|e| format!("pkg init failed: {}", e))?;
  1704→            println!("Initialized VUMA package '{}' in {}", name, dir.display());
  1705→            Ok(())
  1706→        }
  1707→        PkgCommand::Build => {
  1708→            vuma::build_package(&dir).map_err(|e| format!("pkg build failed: {}", e))?;
  1709→            println!("Package built successfully");
  1710→            Ok(())
  1711→        }
  1712→        PkgCommand::Add { dep, version } => {
  1713→            vuma::add_dependency(&dir, dep, version).map_err(|e| format!("pkg add failed: {}", e))?;
  1714→            println!("Added dependency {} @ {}", dep, version);
  1715→            Ok(())
  1716→        }
  1717→    }
  1718→}
