   274‚Üí
   275‚Üí/// Target-specific information needed during code generation.
   276‚Üí///
   277‚Üí/// This trait provides a data-driven interface for query target properties.
   278‚Üí/// It must be implementable for ALL supported ISAs, including fundamentally
   279‚Üí/// different architectures like Wasm (stack machine, no registers) and
   280‚Üí/// MIPS (branch delay slots).
   281‚Üí///
   282‚Üí/// # Object Safety
   283‚Üí///
   284‚Üí/// This trait is object-safe: all methods take `&self` and return only
   285‚Üí/// owned types or references with `'static` lifetime.
   286‚Üípub trait TargetInfo: Send + Sync + 'static {
   287‚Üí    // === Identity ===
   288‚Üí
   289‚Üí    /// ISA name in lowercase (e.g., "aarch64", "riscv64", "wasm32").
   290‚Üí    fn isa_name(&self) -> &'static str;
   291‚Üí
   292‚Üí    /// LLVM-style target triple (e.g., "aarch64-unknown-linux-gnu").
   293‚Üí    fn target_triple(&self) -> &'static str;
   294‚Üí
   295‚Üí    /// ELF `e_machine` value.  Returns 0 for non-ELF targets (Wasm).
   296‚Üí    fn elf_machine_type(&self) -> u16;
   297‚Üí
   298‚Üí    /// Default base address for the .text section.
   299‚Üí    fn default_base_address(&self) -> u64;
   300‚Üí
   301‚Üí    // === Data model ===
   302‚Üí
   303‚Üí    /// Pointer width in bytes (4 for 32-bit, 8 for 64-bit).
   304‚Üí    fn pointer_width(&self) -> usize;
   305‚Üí
   306‚Üí    /// Size in bytes of `ty` on this target.
   307‚Üí    fn size_of(&self, ty: &IRType) -> usize;
   308‚Üí
   309‚Üí    /// Natural alignment in bytes of `ty` on this target.
   310‚Üí    fn alignment_of(&self, ty: &IRType) -> usize;
   311‚Üí
   312‚Üí    /// Byte order of this target.
   313‚Üí    fn endianness(&self) -> Endianness;
   314‚Üí
   315‚Üí    // === Register architecture ===
   316‚Üí
   317‚Üí    /// Whether this target has registers at all.  `false` for Wasm (stack machine).
   318‚Üí    fn has_registers(&self) -> bool;
   319‚Üí
   320‚Üí    /// Number of general-purpose registers.  0 for Wasm.
   321‚Üí    fn num_gp_regs(&self) -> usize;
   322‚Üí
   323‚Üí    /// Number of SIMD/FP registers.  0 for Wasm.
   324‚Üí    fn num_simd_fp_regs(&self) -> usize;
   325‚Üí
   326‚Üí    /// Whether the ISA has a hardwired-zero register (RISC-V x0, LoongArch r0).
   327‚Üí    fn has_hardwired_zero(&self) -> bool;
   328‚Üí
   329‚Üí    /// Whether the ISA uses a link register (ARM, RISC-V, MIPS, PPC) rather than
   330‚Üí    /// pushing the return address on the stack (x86_64).
   331‚Üí    fn has_link_register(&self) -> bool;
   332‚Üí
   333‚Üí    /// Whether branches have delay slots (MIPS only).
   334‚Üí    fn has_branch_delay_slots(&self) -> bool;
   335‚Üí
   336‚Üí    /// Whether this ISA uses a TOC (Table of Contents) pointer (PPC64 r2).
   337‚Üí    fn has_toc_pointer(&self) -> bool;
   338‚Üí
   339‚Üí    /// Whether this ISA has dedicated condition register fields (PPC64 CR0-CR7).
   340‚Üí    fn has_condition_registers(&self) -> bool;
   341‚Üí
   342‚Üí    // === Calling convention ===
   343‚Üí
   344‚Üí    /// Name of the calling convention (e.g., "aapcs64", "lp64d", "systemv").
   345‚Üí    fn calling_convention_name(&self) -> &'static str;
   346‚Üí
   347‚Üí    /// Number of integer argument registers.
   348‚Üí    fn num_int_arg_regs(&self) -> usize;
   349‚Üí
   350‚Üí    /// Number of FP/SIMD argument registers.
   351‚Üí    fn num_fp_arg_regs(&self) -> usize;
   352‚Üí
   353‚Üí    /// Required stack alignment in bytes.
   354‚Üí    fn stack_alignment(&self) -> usize;
   355‚Üí
   356‚Üí    // === Instruction encoding ===
   357‚Üí
   358‚Üí    /// Alignment requirement for instructions in bytes (4 for fixed-width RISCs,
   359‚Üí    /// 1 for x86_64 and Wasm).
   360‚Üí    fn instruction_alignment(&self) -> usize;
   361‚Üí
   362‚Üí    /// Minimum and maximum instruction width in bytes.
   363‚Üí    /// - Fixed-width 32-bit ISAs: (4, 4)
   364‚Üí    /// - x86_64: (1, 15)
   365‚Üí    /// - RISC-V with RVC: (2, 4)
   366‚Üí    /// - Wasm: (1, ~) but typically (1, 15)
   367‚Üí    fn instruction_width_range(&self) -> (usize, usize);
   368‚Üí
   369‚Üí    // === Output format ===
   370‚Üí
   371‚Üí    /// Binary format produced by this backend.
   372‚Üí    fn output_format(&self) -> OutputFormat;
   373‚Üí}

   379‚Üí/// A code generation backend for a specific target architecture.
   380‚Üí///
   381‚Üí/// Each supported ISA implements this trait, providing register allocation,
   382‚Üí/// instruction encoding, program emission, and disassembly.
   383‚Üí///
   384‚Üí/// # Object Safety
   385‚Üí///
   386‚Üí/// This trait is object-safe.
   387‚Üípub trait Backend: Send + Sync + 'static {
   388‚Üí    /// Returns a reference to this backend's target info.
   389‚Üí    fn target_info(&self) -> &dyn TargetInfo;
   390‚Üí
   391‚Üí    /// Allocate physical registers for an IR function.
   392‚Üí    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError>;
   393‚Üí
   394‚Üí    /// Encode a single allocated function into machine code bytes.
   395‚Üí    fn encode_function(&self, func: &AllocatedFunction) -> Result<Vec<u8>, BackendError>;
   396‚Üí
   397‚Üí    /// Encode an entire allocated program into its final binary form
   398‚Üí    /// (ELF, .wasm, raw binary, etc.).
   399‚Üí    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError>;
   400‚Üí
   401‚Üí    /// Returns the bytes for a minimal return stub (e.g., `RET` on ARM64,
   402‚Üí    /// `mov eax, 0; ret` on x86_64, `end` on Wasm).
   403‚Üí    fn return_stub(&self) -> Vec<u8>;
   404‚Üí
   405‚Üí    /// Returns a trampoline that jumps to `entry_addr`.
   406‚Üí    fn trampoline(&self, entry_addr: u64) -> Vec<u8>;
   407‚Üí
   408‚Üí    /// Disassemble `bytes` starting at virtual address `addr`.
   409‚Üí    fn disassemble(&self, bytes: &[u8], addr: u64) -> Vec<String>;
   410‚Üí
   411‚Üí    /// Human-readable name of this backend.
   412‚Üí    fn name(&self) -> &'static str;
   413‚Üí}

  1118‚Üí/// Decode a 32-bit AArch64 instruction word into a human-readable mnemonic.
  1119‚Üí///
  1120‚Üí/// Covers the most common AArch64 instructions: ADD, SUB, MOV, LDR, STR, B,
  1121‚Üí/// BL, RET, CMP, B.cond, STP, LDP, NOP, MUL, SDIV, UDIV, AND, ORR, EOR,
  1122‚Üí/// plus several additional frequently-encountered encodings.
  1123‚Üífn decode_aarch64(word: u32) -> String {
  1124‚Üí    let rd = word & 0x1F;
  1125‚Üí    let rn = (word >> 5) & 0x1F;
  1126‚Üí    let rt = rd; // alias for load/store destination
  1127‚Üí    let rm = (word >> 16) & 0x1F;
  1128‚Üí    let imm12 = (word >> 10) & 0xFFF;
  1129‚Üí    let cond = word & 0xF;
  1130‚Üí
  1131‚Üí    // NOP: d503201f
  1132‚Üí    if word == 0xD503201F {
  1133‚Üí        return "nop".to_string();
  1134‚Üí    }
  1135‚Üí
  1136‚Üí    // RET: d65f03c0
  1137‚Üí    if word == 0xD65F03C0 {
  1138‚Üí        return "ret".to_string();
  1139‚Üí    }
  1140‚Üí
  1141‚Üí    let _top8 = word >> 24;
  1142‚Üí    let _top10 = word >> 22;
  1143‚Üí
  1144‚Üí    // --- ADD/SUB (immediate): 100100xx ...
  1145‚Üí    if (word >> 23) & 0x1FF == 0b1_0010_0010 {
  1146‚Üí        // ADD Xd, Xn, #imm12
  1147‚Üí        return format!("add x{}, x{}, #{}", rd, rn, imm12);
  1148‚Üí    }
  1149‚Üí    if (word >> 23) & 0x1FF == 0b1_1010_0010 {
  1150‚Üí        // SUB Xd, Xn, #imm12
  1151‚Üí        return format!("sub x{}, x{}, #{}", rd, rn, imm12);
  1152‚Üí    }
  1153‚Üí
  1154‚Üí    // --- ADD (shifted register): 1_00_0101_1_xxx ...
  1155‚Üí    if (word >> 24) & 0xFF == 0b1000_1011 {
  1156‚Üí        // ADD Xd, Xn, Xm
  1157‚Üí        return format!("add x{}, x{}, x{}", rd, rn, rm);
  1158‚Üí    }
  1159‚Üí
  1160‚Üí    // --- SUB (shifted register): 1_00_0101_1_xxx ... with S=1 at bit30
  1161‚Üí    if (word >> 24) & 0xFF == 0b1101_0110 {
  1162‚Üí        // SUB Xd, Xn, Xm (bit 30 set = sub)
  1163‚Üí        return format!("sub x{}, x{}, x{}", rd, rn, rm);
  1164‚Üí    }
  1165‚Üí
  1166‚Üí    // --- AND (shifted register): 1_00_0101_0_00_xxx
  1167‚Üí    if (word >> 24) & 0xFE == 0b1000_1010 {
  1168‚Üí        // Check bit 21-15: opcode[31:21] = 10001010_000
  1169‚Üí        if (word >> 21) & 0x7FF == 0b10001010000 {
  1170‚Üí            return format!("and x{}, x{}, x{}", rd, rn, rm);
  1171‚Üí        }
  1172‚Üí    }
  1173‚Üí
  1174‚Üí    // --- ORR (shifted register): 1_01_0101_0_00_xxx
  1175‚Üí    if (word >> 21) & 0x7FF == 0b10101010000 {
  1176‚Üí        return format!("orr x{}, x{}, x{}", rd, rn, rm);
  1177‚Üí    }
  1178‚Üí
  1179‚Üí    // --- EOR (shifted register): 1_10_0101_0_00_xxx
  1180‚Üí    if (word >> 21) & 0x7FF == 0b11001010000 {
  1181‚Üí        return format!("eor x{}, x{}, x{}", rd, rn, rm);
  1182‚Üí    }
  1183‚Üí
  1184‚Üí    // --- MOV (register): alias for ORR Xd, XZR, Xm
  1185‚Üí    // ORR Xd, XZR, Xm: 10101010_000_rm_000000_xzr_rd
  1186‚Üí    // More general: ORR with Rn=XZR(31)
  1187‚Üí    if (word >> 21) & 0x7FF == 0b10101010000 && rn == 31 {
  1188‚Üí        return format!("mov x{}, x{}", rd, rm);
  1189‚Üí    }
  1190‚Üí
  1191‚Üí    // --- MUL: MADD Xd, Xn, Xm, XZR
  1192‚Üí    // Encoding: 1_00_1101_1_000_Rm_0_01111_Rn_Rd
  1193‚Üí    if (word >> 21) & 0x7FF == 0b10011011000 && ((word >> 10) & 0x1F) == 0b01111 {
  1194‚Üí        return format!("mul x{}, x{}, x{}", rd, rn, rm);
  1195‚Üí    }
  1196‚Üí
  1197‚Üí    // --- SDIV: 1_00_1101_1_0100_00000_00001_Rn_Rd  (actually 1_00_1101_0100_Rm_00001_Rn_Rd)
  1198‚Üí    if (word >> 21) & 0x7FF == 0b10011010100 && (word >> 10) & 0x1F == 0b00001 {
  1199‚Üí        return format!("sdiv x{}, x{}, x{}", rd, rn, rm);
  1200‚Üí    }
  1201‚Üí
  1202‚Üí    // --- UDIV: 1_00_1101_0000_Rm_00001_Rn_Rd
  1203‚Üí    if (word >> 21) & 0x7FF == 0b10011010000 && (word >> 10) & 0x1F == 0b00001 {
  1204‚Üí        return format!("udiv x{}, x{}, x{}", rd, rn, rm);
  1205‚Üí    }
  1206‚Üí
  1207‚Üí    // --- CMP (immediate): SUBS XZR, Xn, #imm12
  1208‚Üí    // 11100001_00_xxx_xxx_xxx_xxx_xxx_11111_xxx
  1209‚Üí    if (word >> 23) & 0x1FF == 0b1_1110_0010 && rd == 31 {
  1210‚Üí        return format!("cmp x{}, #{}", rn, imm12);
  1211‚Üí    }
  1212‚Üí
  1213‚Üí    // --- CMP (register): SUBS XZR, Xn, Xm
  1214‚Üí    if (word >> 21) & 0x7FF == 0b11101011000 && rd == 31 {
  1215‚Üí        return format!("cmp x{}, x{}", rn, rm);
  1216‚Üí    }
  1217‚Üí
  1218‚Üí    // --- B.cond: 0101010x xxxxxxxxxx xxxxxx cond
  1219‚Üí    if (word >> 24) & 0xFF == 0x54 {
  1220‚Üí        let cond_name = match cond {
  1221‚Üí            0 => "eq",
  1222‚Üí            1 => "ne",
  1223‚Üí            2 => "cs",
  1224‚Üí            3 => "cc",
  1225‚Üí            4 => "mi",
  1226‚Üí            5 => "pl",
  1227‚Üí            6 => "vs",
  1228‚Üí            7 => "vc",
  1229‚Üí            8 => "hi",
  1230‚Üí            9 => "ls",
  1231‚Üí            10 => "ge",
  1232‚Üí            11 => "lt",
  1233‚Üí            12 => "gt",
  1234‚Üí            13 => "le",
  1235‚Üí            14 => "al",
  1236‚Üí            _ => "??",
  1237‚Üí        };
  1238‚Üí        let imm19 = (word >> 5) & 0x7FFFF;
  1239‚Üí        let offset = ((imm19 as i32) << 13) >> 11; // sign-extend and *4
  1240‚Üí        return format!("b.{} {:+}", cond_name, offset);
  1241‚Üí    }
  1242‚Üí
  1243‚Üí    // --- B (unconditional): 000101xx xxxxxxxxxxxxxxxxxxxx
  1244‚Üí    if (word >> 26) & 0x3F == 0b000101 {
  1245‚Üí        let imm26 = word & 0x3FFFFFF;
  1246‚Üí        let offset = ((imm26 as i32) << 6) >> 4; // sign-extend and *4
  1247‚Üí        return format!("b {:+}", offset);
  1248‚Üí    }
  1249‚Üí
  1250‚Üí    // --- BL: 100101xx xxxxxxxxxxxxxxxxxxxx
  1251‚Üí    if (word >> 26) & 0x3F == 0b100101 {
  1252‚Üí        let imm26 = word & 0x3FFFFFF;
  1253‚Üí        let offset = ((imm26 as i32) << 6) >> 4;
  1254‚Üí        return format!("bl {:+}", offset);
  1255‚Üí    }
  1256‚Üí
  1257‚Üí    // --- LDR (unsigned offset): 11111001_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1258‚Üí    if (word >> 22) & 0x3FF == 0b1111100101 {
  1259‚Üí        let imm12_raw = (word >> 10) & 0xFFF;
  1260‚Üí        let offset = imm12_raw * 8; // scale by 8 for 64-bit
  1261‚Üí        return format!("ldr x{}, [x{}, #{}]", rt, rn, offset);
  1262‚Üí    }
  1263‚Üí
  1264‚Üí    // --- STR (unsigned offset): 11111000_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1265‚Üí    if (word >> 22) & 0x3FF == 0b1111100001 {
  1266‚Üí        let imm12_raw = (word >> 10) & 0xFFF;
  1267‚Üí        let offset = imm12_raw * 8;
  1268‚Üí        return format!("str x{}, [x{}, #{}]", rt, rn, offset);
  1269‚Üí    }
  1270‚Üí
  1271‚Üí    // --- LDP (signed offset, 64-bit): 101_0100_110_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1272‚Üí    if (word >> 22) & 0x3FF == 0b1010100110 {
  1273‚Üí        let rt2 = (word >> 10) & 0x1F;
  1274‚Üí        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1275‚Üí        let offset = imm7 * 8;
  1276‚Üí        return format!("ldp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1277‚Üí    }
  1278‚Üí
  1279‚Üí    // --- STP (signed offset, 64-bit): 101_0100_010_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1280‚Üí    if (word >> 22) & 0x3FF == 0b1010100010 {
  1281‚Üí        let rt2 = (word >> 10) & 0x1F;
  1282‚Üí        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1283‚Üí        let offset = imm7 * 8;
  1284‚Üí        return format!("stp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1285‚Üí    }
  1286‚Üí
  1287‚Üí    // --- MOVZ: 110100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1288‚Üí    if (word >> 23) & 0x1FF == 0b110100101 {
  1289‚Üí        let hw = (word >> 21) & 0x3;
  1290‚Üí        let imm16 = (word >> 5) & 0xFFFF;
  1291‚Üí        return format!("movz x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1292‚Üí    }
  1293‚Üí
  1294‚Üí    // --- MOVK: 111100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1295‚Üí    if (word >> 23) & 0x1FF == 0b111100101 {
  1296‚Üí        let hw = (word >> 21) & 0x3;
  1297‚Üí        let imm16 = (word >> 5) & 0xFFFF;
  1298‚Üí        return format!("movk x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1299‚Üí    }
  1300‚Üí
  1301‚Üí    format!(".word {:08x}", word)
  1302‚Üí}

  1331‚Üí/// Compute the stack frame size for an IR function.
  1332‚Üí///
  1333‚Üí/// Replicates the private `compute_frame_size` function in `emit.rs`:
  1334‚Üí/// sums `Alloc` instruction sizes and rounds up to 16-byte alignment.
  1335‚Üí/// NOTE: Does NOT include 16 bytes for FP/LR pair (handled by prologue separately).
  1336‚Üífn aarch64_compute_frame_size(func: &IRFunction) -> usize {
  1337‚Üí    let mut total: u32 = 0; // Alloc sizes only; FP/LR handled separately
  1338‚Üí    for block in &func.blocks {
  1339‚Üí        for instr in &block.instructions {
  1340‚Üí            if let IRInstr::Alloc { size, .. } = instr {
  1341‚Üí                let aligned = (*size).div_ceil(16) * 16;
  1342‚Üí                total += aligned;
  1343‚Üí            }
  1344‚Üí        }
  1345‚Üí    }
  1346‚Üí    // Round up to 16-byte alignment
  1347‚Üí    total = (total + 15) & !15;
  1348‚Üí    total as usize
  1349‚Üí}
  1350‚Üí
  1351‚Üí/// Build an ELF64 binary for AArch64 Linux with 2 LOAD segments.
  1352‚Üí///
  1353‚Üí/// Segment 1: PF_R | PF_X ‚Äî contains .text (code)
  1354‚Üí/// Segment 2: PF_R | PF_W ‚Äî contains .data + stack space (writable)
  1355‚Üí///
  1356‚Üí/// The two segments are separated by page alignment (4KB) to ensure
  1357‚Üí/// the kernel maps them with different permissions. Without this,
  1358‚Üí/// a single PF_R|PF_W|PF_X segment is insecure and may cause
  1359‚Üí/// QEMU/Linux to reject the executable.
  1360‚Üífn build_aarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1361‚Üí    const PAGE_SIZE: u64 = 0x1000; // 4 KB
  1362‚Üí
  1363‚Üí    let elf_header_size: u64 = 64;
  1364‚Üí    let phdr_size: u64 = 56;
  1365‚Üí    let num_phdrs: u64 = 2;
  1366‚Üí    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1367‚Üí    // Page-align the text segment start in the file.  The kernel's ELF
  1368‚Üí    // loader mmap()s each LOAD segment, and the file offset must be
  1369‚Üí    // congruent with the virtual address modulo the page size.  The
  1370‚Üí    // simplest way to guarantee this is to place the text at a
  1371‚Üí    // page-aligned file offset, with vaddr = base_addr.
  1372‚Üí    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1373‚Üí    let text_size = code.len() as u64;
  1374‚Üí

  1376‚Üí    let text_file_end = text_offset + text_size;
  1377‚Üí    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1378‚Üí    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1379‚Üí    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1380‚Üí    let entry_point = base_addr + text_offset;
  1381‚Üí
  1382‚Üí    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1383‚Üí
  1384‚Üí    // --- e_ident ---
  1385‚Üí    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1386‚Üí    elf.push(2); // ELFCLASS64
  1387‚Üí    elf.push(1); // ELFDATA2LSB
  1388‚Üí    elf.push(1); // EV_CURRENT
  1389‚Üí    elf.push(3); // ELFOSABI_LINUX
  1390‚Üí    elf.push(0); // padding
  1391‚Üí    elf.extend_from_slice(&[0u8; 7]); // padding
  1392‚Üí
  1393‚Üí    // --- ELF header fields ---
  1394‚Üí    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1395‚Üí    elf.extend_from_slice(&183u16.to_le_bytes()); // e_machine = EM_AARCH64
  1396‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1397‚Üí    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1398‚Üí    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1399‚Üí    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1400‚Üí    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1401‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1402‚Üí    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1403‚Üí    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1404‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1405‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1406‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1407‚Üí
  1408‚Üí    // --- Program Header 1: LOAD (PF_R | PF_X) ‚Äî .text ---
  1409‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1410‚Üí    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1411‚Üí    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1412‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1413‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1414‚Üí    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1415‚Üí    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1416‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1417‚Üí
  1418‚Üí    // --- Program Header 2: LOAD (PF_R | PF_W) ‚Äî .data / stack ---
  1419‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1420‚Üí    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1421‚Üí    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1422‚Üí    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1423‚Üí    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1424‚Üí    elf.extend_from_slice(&0u64.to_le_bytes()); // p_filesz (no initialized data)
  1425‚Üí    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1426‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1427‚Üí
  1428‚Üí    // --- .text section ---
  1429‚Üí    // Pad to page-aligned text_offset
  1430‚Üí    while (elf.len() as u64) < text_offset {
  1431‚Üí        elf.push(0);
  1432‚Üí    }
  1433‚Üí    elf.extend_from_slice(code);
  1434‚Üí
  1435‚Üí    // --- Pad to data segment offset (if needed) ---
  1436‚Üí    while (elf.len() as u64) < data_offset {
  1437‚Üí        elf.push(0);
  1438‚Üí    }
  1439‚Üí
  1440‚Üí    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1441‚Üí
  1442‚Üí    elf
  1443‚Üí}

  1445‚Üí/// Build ARM64 runtime I/O functions using Linux SVC syscalls.
  1446‚Üí///
  1447‚Üí/// Provides:
  1448‚Üí/// - `__vuma_print_hex`: Print X0 as 8 hex digits to stdout (FD=1)
  1449‚Üí///   Uses sys_write (X8=64) via SVC #0.
  1450‚Üí///   Saves/restores X1,X2,X3,X8,X9,X10 on stack.
  1451‚Üí///
  1452‚Üí/// - `__vuma_print_int`: Print X0 as a decimal integer to stdout (FD=1)
  1453‚Üí///   Converts digit-by-digit into a stack buffer, then sys_write.
  1454‚Üí///
  1455‚Üí/// - `__vuma_print_newline`: Print a newline character to stdout.
  1456‚Üí///
  1457‚Üí/// All functions follow AAPCS64: X0 is the argument, X1-X7 are caller-saved,
  1458‚Üí/// X8 is the indirect result register / syscall number, X19-X28 are callee-saved.
  1459‚Üífn build_aarch64_runtime() -> Vec<u8> {
  1460‚Üí    let mut code = Vec::new();
  1461‚Üí
  1462‚Üí    // ‚îÄ‚îÄ __vuma_print_hex ‚îÄ‚îÄ
  1463‚Üí    // Input: X0 = 64-bit value to print as 8 hex digits
  1464‚Üí    // Clobbers: X1, X2, X3, X8, X9, X10
  1465‚Üí    // Stack frame: 48 bytes (8 for pair save + 8 for hex chars + 32 padding)
  1466‚Üí
  1467‚Üí    // Prologue: save callee-saved and link register
  1468‚Üí    // STP X29, X30, [SP, #-48]!  (pre-indexed, 48-byte frame)
  1469‚Üí    code.extend_from_slice(&0xA9BF7BFDu32.to_le_bytes());
  1470‚Üí    // ADD X29, SP, #0
  1471‚Üí    code.extend_from_slice(&0x910003FDu32.to_le_bytes());
  1472‚Üí
  1473‚Üí    // Save X9, X10 on stack at [SP, #16]
  1474‚Üí    // STP X9, X10, [SP, #16]
  1475‚Üí    code.extend_from_slice(&0xA90127E9u32.to_le_bytes());
  1476‚Üí
  1477‚Üí    // Save X1, X2 on stack at [SP, #32]
  1478‚Üí    // STP X1, X2, [SP, #32]
  1479‚Üí    code.extend_from_slice(&0xA9020441u32.to_le_bytes());
  1480‚Üí
  1481‚Üí    // Save X3, X8 on stack at [SP, #40]  (offset 40 = 5*8)
  1482‚Üí    // Wait, 40 = 5*8 but STP needs offset/8 for signed-offset. 40/8 = 5.
  1483‚Üí    // STP X3, X8, [SP, #40]
  1484‚Üí    code.extend_from_slice(&0xA9052C63u32.to_le_bytes());
  1485‚Üí
  1486‚Üí    // Hex lookup table: we'll use ADD + LDRB pattern.
  1487‚Üí    // Instead, compute hex digit directly:
  1488‚Üí    // For each nibble: digit = nibble; if digit < 10 then char = digit + '0' else char = digit - 10 + 'a'
  1489‚Üí
  1490‚Üí    // We'll write 8 hex chars starting at SP+16 (reuse save area after we load saves)
  1491‚Üí    // Actually, let's use SP+16 as the char buffer (8 bytes).
  1492‚Üí    // First, save the important regs, then overwrite the buffer area.
  1493‚Üí
  1494‚Üí    // Process 8 nibbles from most significant to least significant
  1495‚Üí    // X0 = value, iterate 8 times shifting right by 28, 24, 20, ..., 4, 0
  1496‚Üí
  1497‚Üí    // X9 = char buffer pointer = SP
  1498‚Üí    // X10 = loop counter (0..8)
  1499‚Üí
  1500‚Üí    // ADD X9, SP, #16  (buffer starts at SP+16)
  1501‚Üí    code.extend_from_slice(&0x910043E9u32.to_le_bytes());
  1502‚Üí    // MOV X10, #0
  1503‚Üí    code.extend_from_slice(&0xD280001Au32.to_le_bytes());
  1504‚Üí    // MOV X3, #0  (bit shift amount, starts at 28)
  1505‚Üí    code.extend_from_slice(&0xD280003Du32.to_le_bytes()); // MOVZ X3, #28>>4... wait
  1506‚Üí    // MOV X3, #28
  1507‚Üí    code.extend_from_slice(&0xD2800383u32.to_le_bytes()); // MOVZ X3, #28
  1508‚Üí
  1509‚Üí    // Loop label:
  1510‚Üí    let loop_offset = code.len();
  1511‚Üí
  1512‚Üí    // Extract nibble: X2 = (X0 >> X3) & 0xF
  1513‚Üí    // LSR X2, X0, X3
  1514‚Üí    code.extend_from_slice(&0x9AC02482u32.to_le_bytes()); // LSR X2, X0, X3
  1515‚Üí    // AND X2, X2, #0xF
  1516‚Üí    code.extend_from_slice(&0x92400C42u32.to_le_bytes()); // AND X2, X2, #15
  1517‚Üí
  1518‚Üí    // Convert nibble to hex char:
  1519‚Üí    // if X2 < 10: char = X2 + 48 ('0')
  1520‚Üí    // else: char = X2 - 10 + 97 ('a')
  1521‚Üí    // CMP X2, #10
  1522‚Üí    code.extend_from_slice(&0xF100141Fu32.to_le_bytes()); // CMP X2, #10? No...
  1523‚Üí    // CMP X2, #10: use ADD X1, X2, #0 then CMP... Actually:
  1524‚Üí    // SUB X1, X2, #10 ‚Üí if < 10, result is negative (borrow set)
  1525‚Üí    // Let's use: CMP X2, #9; B.GT hex_alpha
  1526‚Üí    // CMP X2, #9
  1527‚Üí    code.extend_from_slice(&0xF100129Fu32.to_le_bytes()); // CMP X2, #9
  1528‚Üí
  1529‚Üí    // ADD X1, X2, #48  (default: '0' + digit)
  1530‚Üí    code.extend_from_slice(&0x91000C41u32.to_le_bytes()); // ADD X1, X2, #48
  1531‚Üí
  1532‚Üí    // B.GT hex_alpha (placeholder, will be +1 instruction)
  1533‚Üí    code.extend_from_slice(&0x54000069u32.to_le_bytes()); // B.GT +4*2 = +8 bytes
  1534‚Üí
  1535‚Üí    // B store_char
  1536‚Üí    code.extend_from_slice(&0x14000002u32.to_le_bytes()); // B +2 instructions
  1537‚Üí
  1538‚Üí    // hex_alpha: ADD X1, X2, #87  (87 = 97-10, so digit-10+'a')
  1539‚Üí    code.extend_from_slice(&0x91002AC1u32.to_le_bytes()); // ADD X1, X2, #87
  1540‚Üí
  1541‚Üí    // store_char: STRB W1, [X9, X10]
  1542‚Üí    // STRB W1, [X9, X10] ‚Äî register offset
  1543‚Üí    code.extend_from_slice(&0x382A6821u32.to_le_bytes()); // STRB W1, [X9, X10]
  1544‚Üí
  1545‚Üí    // Increment: SUB X3, X3, #4; ADD X10, X10, #1; CMP X10, #8; B.LT loop
  1546‚Üí    code.extend_from_slice(&0xD1000863u32.to_le_bytes()); // SUB X3, X3, #4
  1547‚Üí    code.extend_from_slice(&0x9100054Au32.to_le_bytes()); // ADD X10, X10, #1
  1548‚Üí    code.extend_from_slice(&0xF100151Fu32.to_le_bytes()); // CMP X10, #8
  1549‚Üí
  1550‚Üí    // Calculate branch back to loop start
  1551‚Üí    let current_pos = code.len() as i64;
  1552‚Üí    let loop_start = loop_offset as i64;
  1553‚Üí    let back_offset = (loop_start - current_pos) / 4 - 1; // B.LT is relative to PC+4
  1554‚Üí    let bcond = 0x5400000Bu32 | ((back_offset as u32) & 0x7FFFF); // B.LT
  1555‚Üí    code.extend_from_slice(&bcond.to_le_bytes());
  1556‚Üí
  1557‚Üí    // ‚îÄ‚îÄ sys_write(1, buf, 8) ‚îÄ‚îÄ
  1558‚Üí    // MOV X0, #1  (stdout fd)
  1559‚Üí    code.extend_from_slice(&0xD2800020u32.to_le_bytes()); // MOVZ X0, #1
  1560‚Üí    // ADD X1, SP, #16  (buffer pointer)
  1561‚Üí    code.extend_from_slice(&0x910043E1u32.to_le_bytes());
  1562‚Üí    // MOV X2, #8  (length)
  1563‚Üí    code.extend_from_slice(&0xD2800102u32.to_le_bytes()); // MOVZ X2, #8
  1564‚Üí    // MOV X8, #64  (sys_write)
  1565‚Üí    code.extend_from_slice(&0xD2800808u32.to_le_bytes()); // MOVZ X8, #64
  1566‚Üí    // SVC #0
  1567‚Üí    code.extend_from_slice(&0xD4000001u32.to_le_bytes());
  1568‚Üí
  1569‚Üí    // Restore registers
  1570‚Üí    // LDP X9, X10, [SP, #16]
  1571‚Üí    code.extend_from_slice(&0xA94127E9u32.to_le_bytes());
  1572‚Üí    // LDP X1, X2, [SP, #32]
  1573‚Üí    code.extend_from_slice(&0xA9420441u32.to_le_bytes());
  1574‚Üí    // LDP X3, X8, [SP, #40]
  1575‚Üí    code.extend_from_slice(&0xA9452C63u32.to_le_bytes());
  1576‚Üí    // LDP X29, X30, [SP], #48  (post-indexed load)
  1577‚Üí    code.extend_from_slice(&0xA8C37BFDu32.to_le_bytes());

  1579‚Üí    // RET
  1580‚Üí    code.extend_from_slice(&0xD65F03C0u32.to_le_bytes());

  1582‚Üí    // ‚îÄ‚îÄ __vuma_print_int ‚îÄ‚îÄ

  1585‚Üí    // X9 = digit count, X10 = buffer pointer (SP-based)
  1586‚Üí    // Stack frame: 64 bytes (16 for save pair + 48 for digit buffer)

  1591‚Üí    code.extend_from_slice(&0x910003FDu32.to_le_bytes());

  1599‚Üí    // Handle negative numbers
  1600‚Üí    // CMP X0, #0
  1601‚Üí    code.extend_from_slice(&0xB100001Fu32.to_le_bytes());

  1609‚Üí    code.extend_from_slice(&0x390013E1u32.to_le_bytes());

  1611‚Üí    code.extend_from_slice(&0xD2800020u32.to_le_bytes()); // MOV X0, #1
  1612‚Üí    code.extend_from_slice(&0x910043E1u32.to_le_bytes()); // ADD X1, SP, #16

  1616‚Üí

  1618‚Üí    // NEG X0, X0 = SUB X0, XZR, X0

  1623‚Üí    code.extend_from_slice(&0xD2800019u32.to_le_bytes()); // MOVZ X9, #0
  1624‚Üí    // X10 = buffer pointer = SP + 17 (leave room for potential '-')
  1625‚Üí    code.extend_from_slice(&0x910047EAu32.to_le_bytes()); // ADD X10, SP, #17
  1626‚Üí

  1628‚Üí    let div_loop_start = code.len();

  1637‚Üí    code.extend_from_slice(&0x9AC10802u32.to_le_bytes()); // UDIV X2, X0, X1

  1639‚Üí    code.extend_from_slice(&0x9B017C43u32.to_le_bytes()); // MSUB X3, X2, X1, X0

  1641‚Üí    code.extend_from_slice(&0x91001863u32.to_le_bytes()); // ADD X3, X3, #48

  1643‚Üí    code.extend_from_slice(&0x382B6863u32.to_le_bytes()); // STRB W3, [X10, X9]

  1645‚Üí    code.extend_from_slice(&0x91000529u32.to_le_bytes()); // ADD X9, X9, #1
  1646‚Üí    // MOV X0, X2  (quotient becomes new value)
  1647‚Üí    code.extend_from_slice(&0xAA0203E0u32.to_le_bytes()); // MOV X0, X2

  1649‚Üí    let cur = code.len() as i64;
  1650‚Üí    let back = (div_loop_start as i64 - cur as i64) / 4 - 1;
  1651‚Üí    code.extend_from_slice(&(0x14000000u32 | ((back as u32) & 0x3FFFFFF)).to_le_bytes());
  1652‚Üí
  1653‚Üí    // done_digits: digits are in reverse order in buffer.
  1654‚Üí    // Reverse them in-place using X1 and X3 as temp pointers.
  1655‚Üí    // CBZ X9, write_digits  (if count == 0, print "0")
  1656‚Üí    code.extend_from_slice(&0xB4000039u32.to_le_bytes()); // CBZ X9, write_digits (+6 instr = 24 bytes)
  1657‚Üí
  1658‚Üí    // If count == 0, print "0"
  1659‚Üí    // MOV X1, #48  ('0')
  1660‚Üí    code.extend_from_slice(&0xD2800601u32.to_le_bytes()); // MOVZ X1, #48
  1661‚Üí    // STRB W1, [X10]
  1662‚Üí    code.extend_from_slice(&0x39001501u32.to_le_bytes());
  1663‚Üí    // MOV X9, #1
