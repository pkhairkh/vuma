     1→//! ARM64 Codegen End-to-End Tests
     2→//!
     3→//! Tests exercising the full ARM64 code-generation pipeline:
     4→//! SCG → IR → register allocation → ARM64 instruction selection → ELF emission.
     5→//!
     6→//! Each test constructs a codegen-level SCG, lowers it to IR, emits ARM64
     7→//! machine code, and validates structural properties of the output (correct
     8→//! instruction presence, proper prologue/epilogue, valid ELF headers, etc.).
     9→//!
    10→//! # Test Matrix
    11→//!
    12→//! | # | Test                                    | What it validates                              |
    13→//! |---|-----------------------------------------|------------------------------------------------|
    14→//! | 1 | Simple add function                     | Add → IR → ARM64 ADD instruction               |
    15→//! | 2 | Function with stack allocation          | Alloc → stack frame layout                     |
    16→//! | 3 | Load/store round-trip                   | Load/Store → LDR/STR encoding                  |
    17→//! | 4 | If/else control flow                    | CondBranch → CBNZ + branch fixups              |
    18→//! | 5 | Loop with back-edge                     | Loop header phi + back-edge branch             |
    19→//! | 6 | Function call                           | Call → BL relocation                           |
    20→//! | 7 | Multiple functions in one program       | Multi-function ELF with symbol table           |
    21→//! | 8 | Type system / calling convention        | IRType sizes, arg classification, stack layout  |
    22→//! | 9 | Bare-metal raw binary emission          | Raw binary output (no ELF headers)             |
    23→//! | 10| ARM64 instruction encoding correctness  | Encode/decode round-trip for key instructions   |
    24→
    25→use vuma_codegen::{
    26→    arm64::{BarrierOption, Condition, Instruction, Operand, Register},
    27→    emit::{emit_elf, emit_raw, EmitConfig, Emitter},
    28→    ir::{
    29→        alignment_of, classify_arg, compute_calling_conv, compute_stack_layout, size_of, ArgClass,
    30→        BinOpKind, IRInstr, IRProgram, IRTerminator, IRType, RegisterClass,
    31→    },
    32→    scg_to_ir::{

    38→// ---------------------------------------------------------------------------
    39→// Helpers

    41→
    42→/// Compile a codegen `Scg` through the full IR → ARM64 emission pipeline.
    43→fn compile_scg(scg: &Scg) -> (IRProgram, Vec<u32>) {
    44→    let mut builder = IRBuilder::new();
    45→    let ir_program = builder.build(scg).expect("IRBuilder should succeed");
    46→    let mut emitter = Emitter::new();
    47→    let code_words = emitter
    48→        .emit_function(&ir_program.functions[0])
    49→        .expect("Emission should succeed");
    50→    (ir_program, code_words)
    51→}
    52→
    53→/// Compile a full program (potentially multiple functions) and emit an ELF.
    54→fn compile_to_elf(scg: &Scg, config: &EmitConfig) -> Vec<u8> {
    55→    let mut builder = IRBuilder::new();
    56→    let ir_program = builder.build(scg).expect("IRBuilder should succeed");
    57→    emit_elf(&ir_program.functions, &ir_program.data_sections, config)
    58→        .expect("ELF emission should succeed")
    59→}

    65→/// Test: Lower a simple `fn add(a, b) -> i64 { a + b }` to ARM64.
    66→///
    67→/// Validates:
    68→/// - IR function has two params and an Add instruction
    69→/// - ARM64 code contains a valid ADD instruction encoding
    70→/// - The function returns the result in X0 (AAPCS64)
    71→#[test]
    72→fn test_codegen_simple_add() {
    73→    let scg = Scg {
    74→        nodes: vec![ScgNode::Function(ScgFunction {
    75→            name: "add".to_string(),
    76→            params: vec![
    77→                ScgParam {
    78→                    name: "a".to_string(),
    79→                    ty: ScgType::I64,
    80→                },
    81→                ScgParam {
    82→                    name: "b".to_string(),
    83→                    ty: ScgType::I64,
    84→                },
    85→            ],
    86→            results: vec![ScgType::I64],
    87→            body: vec![
    88→                ScgStatement::Computation(ComputationNode {
    89→                    dst: "result".to_string(),
    90→                    op: BinOpKind::Add,
    91→                    lhs: ScgExpr::Var("a".to_string()),
    92→                    rhs: ScgExpr::Var("b".to_string()),
    93→                    tail_call: false,
    94→                }),
    95→                ScgStatement::Return(vec![ScgExpr::Var("result".to_string())]),
    96→            ],
    97→        })],
    98→    };
    99→
   100→    let (ir_program, code_words) = compile_scg(&scg);
   101→
   102→    // Validate IR structure
   103→    let func = &ir_program.functions[0];
   104→    assert_eq!(func.name, "add");
   105→    assert_eq!(func.params.len(), 2);
   106→    assert_eq!(func.result_types.len(), 1);
   107→
   108→    // Check that an Add instruction exists
   109→    let has_add = func.blocks.iter().any(|b| {
   110→        b.instructions
   111→            .iter()
   112→            .any(|i| matches!(i, IRInstr::Add { .. }))
   113→    });
   114→    assert!(has_add, "IR should contain an Add instruction");
   115→
   116→    // Check ARM64 code was emitted
   117→    assert!(!code_words.is_empty(), "ARM64 code should be non-empty");
   118→
   119→    // First instruction should be STP (prologue)
   120→    let first = code_words[0];
   121→    assert_ne!(first, 0, "Prologue STP should not be zero-encoded");
   122→}

   473→// Test 8: Type system and calling convention

   475→
   476→/// Test: Validate IR type system, argument classification, and calling convention.
   477→///
   478→/// Tests:
   479→/// - IRType size/alignment calculations are correct for ARM64 LP64
   480→/// - Argument classification follows AAPCS64 (integer → Integer, FP → FP, etc.)
   481→/// - Calling convention assigns correct registers (X0–X7 for integer args)
   482→#[test]
   483→fn test_codegen_type_system_calling_conv() {
   484→    // --- Size/alignment ---
   485→    assert_eq!(size_of(&IRType::I8), 1);
   486→    assert_eq!(size_of(&IRType::I64), 8);
   487→    assert_eq!(size_of(&IRType::U32), 4);
   488→    assert_eq!(size_of(&IRType::F64), 8);
   489→    assert_eq!(size_of(&IRType::Ptr), 8);
   490→    assert_eq!(size_of(&IRType::Void), 0);
   491→
   492→    assert_eq!(alignment_of(&IRType::I8), 1);
   493→    assert_eq!(alignment_of(&IRType::I64), 8);
   494→    assert_eq!(alignment_of(&IRType::F32), 4);
   495→
   496→    // --- Argument classification ---
   497→    assert_eq!(classify_arg(&IRType::I64), ArgClass::Integer);
   498→    assert_eq!(classify_arg(&IRType::F64), ArgClass::FP);
   499→    assert_eq!(classify_arg(&IRType::Ptr), ArgClass::Integer);
   500→    assert_eq!(classify_arg(&IRType::Void), ArgClass::Integer);
   501→
   502→    // --- Calling convention: 2 integer args → X0, X1 ---
   503→    let cc = compute_calling_conv(&[IRType::I64, IRType::I64], &IRType::I64);
   504→    assert_eq!(cc.arg_locations.len(), 2);
   505→    assert_eq!(cc.arg_locations[0].register, Some((RegisterClass::X, 0)));
   506→    assert_eq!(cc.arg_locations[1].register, Some((RegisterClass::X, 1)));
   507→    assert_eq!(cc.stack_args_size, 0, "No stack args for 2 register args");
   508→
   509→    // --- Calling convention: 9 integer args → 8 in regs + 1 on stack ---
   510→    let args_9: Vec<IRType> = (0..9).map(|_| IRType::I64).collect();
   511→    let cc9 = compute_calling_conv(&args_9, &IRType::I64);
   512→    assert_eq!(cc9.arg_locations.len(), 9);
   513→    let stack_args: Vec<_> = cc9
   514→        .arg_locations
   515→        .iter()
   516→        .filter(|a| a.register.is_none())
   517→        .collect();
   518→    assert_eq!(stack_args.len(), 1, "9th arg should be on the stack");
   519→    assert!(cc9.stack_args_size > 0, "Should need stack argument space");
   520→
   521→    // --- Return value location ---
   522→    assert_eq!(cc.ret_location.registers, vec![(RegisterClass::X, 0)]);
   523→}

   529→/// Test: Emit a bare-metal raw binary (no ELF headers).
   530→///
   531→/// Validates:
   532→/// - Raw binary is produced without ELF magic
   533→/// - Binary consists of concatenated ARM64 code words
   534→/// - Each word is 4 bytes and non-zero
   535→#[test]
   536→fn test_codegen_bare_metal_raw() {
   537→    let scg = Scg {
   538→        nodes: vec![ScgNode::Function(ScgFunction {
   539→            name: "_start".to_string(),
   540→            params: vec![],
   541→            results: vec![ScgType::Void],
   542→            body: vec![
   543→                ScgStatement::Computation(ComputationNode {
   544→                    dst: "x".to_string(),
   545→                    op: BinOpKind::Add,
   546→                    lhs: ScgExpr::Int(1),
   547→                    rhs: ScgExpr::Int(2),
   548→                    tail_call: false,
   549→                }),
   550→                ScgStatement::Return(vec![]),
   551→            ],
   552→        })],
   553→    };
   554→
   555→    let mut builder = IRBuilder::new();
   556→    let ir_program = builder.build(&scg).expect("IRBuilder should succeed");
   557→
   558→    let config = EmitConfig::bare_metal_raw();
   559→    let raw_bytes = emit_raw(&ir_program.functions, &[], &config).expect("Raw emission should succeed");
   560→
   561→    // Raw binary should NOT start with ELF magic
   562→    if raw_bytes.len() >= 4 {
   563→        assert_ne!(
   564→            &raw_bytes[0..4],
   565→            &[0x7f, b'E', b'L', b'F'],
   566→            "Raw binary should not have ELF magic"
   567→        );
   568→    }
   569→
   570→    // Should be a multiple of 4 bytes (ARM64 instructions)
   571→    assert_eq!(
   572→        raw_bytes.len() % 4,
   573→        0,
   574→        "Raw binary should be 4-byte aligned"
   575→    );
   576→
   577→    // Should have at least the prologue (STP + MOV + SUB = 3 instr = 12 bytes)

   582→// Test 10: ARM64 instruction encoding correctness
   583→// ---------------------------------------------------------------------------

   585→/// Test: Validate ARM64 instruction encoding for key instructions.
   586→///
   587→/// Tests that common ARM64 instructions encode to valid 32-bit machine code
   588→/// words and that the encoding produces distinct values for different
   589→/// operations.
   590→#[test]
   591→fn test_arm64_instruction_encoding() {
   592→    // --- ADD X0, X1, X2 ---
   593→    let add = Instruction::ADD {
   594→        rd: Register::X0,
   595→        rn: Register::X1,
   596→        rm: Operand::Reg {
   597→            reg: Register::X2,
   598→            shift: None,
   599→        },
   600→    };
   601→    let add_encoded = add.encode().expect("ADD should encode");
   602→    assert_ne!(add_encoded, 0, "ADD encoding should be non-zero");
   603→
   604→    // --- SUB X0, X1, X2 ---
   605→    let sub = Instruction::SUB {
   606→        rd: Register::X0,
   607→        rn: Register::X1,
   608→        rm: Operand::Reg {
   609→            reg: Register::X2,
   610→            shift: None,
   611→        },
   612→    };
   613→    let sub_encoded = sub.encode().expect("SUB should encode");
   614→    assert_ne!(sub_encoded, 0, "SUB encoding should be non-zero");
   615→    assert_ne!(
   616→        add_encoded, sub_encoded,
   617→        "ADD and SUB should have different encodings"
   618→    );
   619→
   620→    // --- ADD with immediate: ADD X0, X1, #42 ---
   621→    let add_imm = Instruction::ADD {
   622→        rd: Register::X0,
   623→        rn: Register::X1,
   624→        rm: Operand::Imm12(42),
   625→    };
   626→    let add_imm_encoded = add_imm.encode().expect("ADD imm should encode");
   627→    assert_ne!(
   628→        add_imm_encoded, add_encoded,
   629→        "ADD imm should differ from ADD reg"
   630→    );
   631→
   632→    // --- LDR X0, [X1, #0] ---
   633→    let ldr = Instruction::LDR {
   634→        rt: Register::X0,
   635→        rn: Register::X1,
   636→        offset: 0,
   637→    };
   638→    let ldr_encoded = ldr.encode().expect("LDR should encode");
   639→    assert_ne!(ldr_encoded, 0, "LDR encoding should be non-zero");
   640→
   641→    // --- STR X0, [X1, #0] ---
   642→    let str_ = Instruction::STR {
   643→        rt: Register::X0,
   644→        rn: Register::X1,
   645→        offset: 0,
   646→    };
   647→    let str_encoded = str_.encode().expect("STR should encode");
   648→    assert_ne!(
   649→        ldr_encoded, str_encoded,
   650→        "LDR and STR should have different encodings"
   651→    );
   652→
   653→    // --- MOV X0, X1 ---
   654→    let mov = Instruction::MOV {
   655→        rd: Register::X0,
   656→        rm: Register::X1,
   657→    };
   658→    let mov_encoded = mov.encode().expect("MOV should encode");
   659→    assert_ne!(mov_encoded, 0);
   660→
   661→    // --- RET ---
   662→    let ret = Instruction::RET { rn: None };
   663→    let ret_encoded = ret.encode().expect("RET should encode");
   664→    assert_ne!(ret_encoded, 0);
   665→
   666→    // --- NOP ---
   667→    let nop = Instruction::NOP;
   668→    let nop_encoded = nop.encode().expect("NOP should encode");
   669→    assert_ne!(nop_encoded, 0);
   670→
   671→    // --- MOVZ X0, #42 ---
   672→    let movz = Instruction::MOVZ {
   673→        rd: Register::X0,
   674→        imm16: 42,
   675→        shift: 0,
   676→    };
   677→    let movz_encoded = movz.encode().expect("MOVZ should encode");
   678→    assert_ne!(movz_encoded, 0);
   679→
   680→    // --- Barrier instructions ---
   681→    let dmb = Instruction::DMB {
   682→        option: BarrierOption::ISH,
   683→    };
   684→    let dmb_encoded = dmb.encode().expect("DMB should encode");
   685→    assert_ne!(dmb_encoded, 0);
   686→
   687→    let isb = Instruction::ISB;
   688→    let isb_encoded = isb.encode().expect("ISB should encode");
   689→    assert_ne!(isb_encoded, 0);
   690→    assert_ne!(dmb_encoded, isb_encoded);
   691→
   692→    // --- Condition codes ---
   693→    assert_eq!(Condition::EQ.encoding(), 0b0000);
   694→    assert_eq!(Condition::NE.encoding(), 0b0001);
   695→    assert_eq!(Condition::GE.encoding(), 0b1010);
   696→    assert_eq!(Condition::LT.encoding(), 0b1011);
   697→    assert_eq!(Condition::EQ.invert(), Condition::NE);
   698→    assert_eq!(Condition::GE.invert(), Condition::LT);
   699→
   700→    // --- Register encoding ---
   701→    assert_eq!(Register::X0.encoding(), 0);
   702→    assert_eq!(Register::X7.encoding(), 7);
   703→    assert_eq!(Register::SP.encoding(), 31);
   704→    assert_eq!(Register::XZR.encoding(), 31);
   705→    assert!(Register::X19.is_callee_saved());
   706→    assert!(!Register::X9.is_callee_saved());
   707→    assert!(Register::X9.is_caller_saved());
   708→
   709→    // --- Argument register lookup ---
   710→    assert_eq!(Register::arg_register(0), Some(Register::X0));
   711→    assert_eq!(Register::arg_register(7), Some(Register::X7));
   712→    assert_eq!(Register::arg_register(8), None);
   713→    assert_eq!(Register::X3.arg_index(), Some(3));
   714→    assert_eq!(Register::X19.arg_index(), None);
   715→}
