     1→//! Abstract Syntax Tree types for the VUMA language frontend.
     2→//!
     3→//! The AST is the central data structure produced by the parser. Every node
     4→//! carries an optional [`Span`] so that downstream passes (type checking,
     5→//! SCG generation, error reporting) can trace back to the original source.
     6→//!
     7→//! Design principles:
     8→//! - **Minimal**: only the constructs needed for a memory-oriented,
     9→//!   AI-consumable language.
    10→//! - **Explicit**: allocation, free, cast, and region operations are
    11→//!   first-class statements rather than library calls.
    12→//! - **Spanned**: every node can point back to its source location.
    13→
    14→use crate::error::Span;
    15→use serde::{Deserialize, Serialize};
    16→
    17→// ---------------------------------------------------------------------------
    18→// Visibility
    19→// ---------------------------------------------------------------------------
    20→
    21→/// Visibility modifier for items.
    22→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, Default)]
    23→pub enum Visibility {
    24→    /// Default (private) visibility — no modifier.
    25→    #[default]
    26→    Private,
    27→    /// `pub` — public visibility.
    28→    Public,
    29→    /// `pub(crate)` — visible within the current crate.
    30→    PublicCrate,
    31→    /// `pub(super)` — visible in the parent module.
    32→    PublicSuper,
    33→    /// `pub(in path)` — visible in the specified path.
    34→    PublicIn(String),
    35→}
    36→
    37→impl std::fmt::Display for Visibility {
    38→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    39→        match self {
    40→            Visibility::Private => write!(f, ""),
    41→            Visibility::Public => write!(f, "pub"),
    42→            Visibility::PublicCrate => write!(f, "pub(crate)"),
    43→            Visibility::PublicSuper => write!(f, "pub(super)"),
    44→            Visibility::PublicIn(path) => write!(f, "pub(in {})", path),
    45→        }
    46→    }
    47→}

    53→/// An outer attribute: `#[attr]` or `#[attr = value]` or `#[attr(key = value)]`
    54→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
    55→pub struct Attribute {
    56→    /// Whether this is an inner attribute (`#![...]`) or outer (`#[...]`).
    57→    pub is_inner: bool,
    58→    /// The attribute path/name (e.g. `inline`, `derive`, `cfg`).
    59→    pub name: String,
    60→    /// Optional single value: `#[inline(always)]` or `#[cfg(test)]`.
    61→    /// Stored as a string; structured parsing is left to later passes.
    62→    pub value: Option<AttrValue>,
    63→    /// Source span.
    64→    pub span: Span,
    65→}
    66→
    67→/// The value part of an attribute.
    68→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
    69→pub enum AttrValue {
    70→    /// A single identifier or literal: `#[inline(always)]` → `always`
    71→    Single(String),
    72→    /// Key-value pairs: `#[derive(Debug)]`, `#[cfg(test)]`, `#[allow(dead_code)]`
    73→    List(Vec<String>),
    74→    /// Key = value: `#[repr(C)]` → the `C` part
    75→    KeyValue {
    76→        /// Attribute key.
    77→        key: String,
    78→        /// Attribute value.
    79→        value: String,
    80→    },
    81→}

   100→/// A top-level declaration within a [`Program`].
   101→#[derive(Debug, Clone, Serialize, Deserialize)]
   102→pub enum Item {
   103→    /// Function definition: `fn name(params) -> T { … }`
   104→    FnDef(FnDef),
   105→    /// Struct definition: `struct Name { field: Type, … }`
   106→    StructDef(StructDef),
   107→    /// Enum definition: `enum Name { Variant, … }`
   108→    EnumDef(EnumDef),
   109→    /// Region declaration: `region name = allocate(size);`
   110→    RegionDef(RegionDef),
   111→    /// Import declaration: `import "path";`
   112→    Import(Import),
   113→    /// Export declaration: `export name;`
   114→    Export(Export),
   115→    /// Constant definition: `const name: T = value;`
   116→    Const(ConstDef),
   117→    /// Static definition: `static name: T = value;`
   118→    Static(StaticDef),
   119→    /// Module declaration: `module name { items }`
   120→    ModuleDef(ModuleDef),
   121→    /// Trait definition: `trait Name<T> { … }`
   122→    TraitDef(TraitDef),
   123→    /// Impl block: `impl TraitName for Type { … }` or `impl Type { … }`
   124→    ImplBlock(ImplBlock),
   125→    /// Top-level statement (assignment, expression, free, etc.) that
   126→    /// appears outside of any function body.
   127→    Stmt(Stmt),
   128→}

   409→/// A single statement within a [`Block`].
   410→#[derive(Debug, Clone, Serialize, Deserialize)]
   411→pub enum Stmt {
   412→    /// Variable binding: `let name [: T] = expr;`
   413→    Let(LetStmt),
   414→    /// Assignment: `name = expr;`  or  `*name = expr;`
   415→    Assign(AssignStmt),
   416→    /// Compound assignment: `name += expr;`, `name -= expr;`, etc.
   417→    CompoundAssign(CompoundAssignStmt),
   418→    /// Memory allocation: `allocate(expr)` — typically inside a `region` decl.
   419→    Allocate(AllocateStmt),
   420→    /// Memory deallocation: `free(expr);`
   421→    Free(FreeStmt),
   422→    /// Memory access / dereference: `*expr`  or  `(*expr).field`
   423→    Access(AccessStmt),
   424→    /// Type cast: `expr as Type`
   425→    Cast(CastStmt),
   426→    /// Conditional: `if expr { … } else { … }`
   427→    If(IfStmt),
   428→    /// Loop: `while expr { … }`
   429→    While(WhileStmt),
   430→    /// For loop: `for name in expr { … }`
   431→    For(ForStmt),
   432→    /// Infinite loop: `loop { … }`
   433→    Loop(LoopStmt),
   434→    /// An unsafe block (`unsafe { ... }`)
   435→    UnsafeBlock {
   436→        /// Block body.
   437→        body: Block,
   438→        /// Source span.
   439→        span: Span,
   440→    },
   441→    /// Pattern match: `match expr { arms }`
   442→    Match(MatchStmt),
   443→    /// Sync block: `sync { … }`
   444→    Sync(SyncBlock),
   445→    /// Return: `return [expr];`
   446→    Return(ReturnStmt),
   447→    /// Break: `break;` or `break expr;`
   448→    Break(BreakStmt),
   449→    /// Continue: `continue;`
   450→    Continue(ContinueStmt),
   451→    /// BD directive: `bd(name, expr)`, `repd(name, expr)`, `capd(name, expr)`, `reld(name, expr)`
   452→    BdDirective(BdDirectiveStmt),
   453→    /// Expression statement: `expr;`
   454→    Expr(ExprStmt),
   455→}

   629→/// A match pattern (simplified for VUMA).
   630→#[derive(Debug, Clone, Serialize, Deserialize)]
   631→pub enum MatchPattern {
   632→    /// Wildcard: `_`
   633→    Wildcard(Span),
   634→    /// Literal pattern.
   635→    Lit {
   636→        /// The literal value to match.
   637→        value: Lit,
   638→        /// Source span.
   639→        span: Span,
   640→    },
   641→    /// Identifier / variant name pattern.
   642→    Ident {
   643→        /// The identifier or variant name.
   644→        name: String,
   645→        /// Source span.
   646→        span: Span,
   647→    },
   648→    /// Struct-like pattern: `Name { field, … }`
   649→    Struct {
   650→        /// Struct name.
   651→        name: String,
   652→        /// Field names to match.
   653→        fields: Vec<String>,
   654→        /// Source span.
   655→        span: Span,
   656→    },
   657→    /// Enum variant pattern: `Some(v)` or `None`
   658→    Enum {
   659→        /// Variant name.
   660→        name: String,
   661→        /// Optional binding for the variant payload.

  1131→/// Type annotation (optional in many positions).
  1132→///
  1133→/// The type system covers primitive types, pointer types, region-annotated
  1134→/// types, struct types, generic type applications, and BD annotations.
  1135→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
  1136→pub enum Type {
  1137→    /// Base domain type (from the BD subsystem).
  1138→    /// Examples: `u8`, `u32`, `i64`, `bool`, `void`.
  1139→    BDBase(String),
  1140→    /// Pointer type: `*T`
  1141→    Ptr(Box<Type>),
  1142→    /// Region-annotated pointer: `*T @ region`
  1143→    RegionPtr {
  1144→        /// The pointed-to type.
  1145→        inner: Box<Type>,
  1146→        /// Region name annotation.
  1147→        region: String,
  1148→    },
  1149→    /// Fixed-size array: `[T; N]`
  1150→    Array {
  1151→        /// Element type.
  1152→        element: Box<Type>,
  1153→        /// Number of elements.
  1154→        size: usize,
  1155→    },
  1156→    /// Struct type: named collection of typed fields.
  1157→    Struct {
  1158→        /// Struct name.
  1159→        name: String,
  1160→        /// Ordered field (name, type) pairs.
  1161→        fields: Vec<(String, Type)>,
  1162→    },
  1163→    /// Generic type application: `Name<T1, T2, …>`
  1164→    Generic {
  1165→        /// Type name.
  1166→        name: String,
  1167→        /// Generic arguments.
  1168→        args: Vec<Type>,
  1169→    },
  1170→    /// Function type: `(params) -> return_type`
  1171→    Func {
  1172→        /// Parameter types.
  1173→        params: Vec<Type>,
  1174→        /// Optional return type.
  1175→        return_type: Option<Box<Type>>,
  1176→    },
  1177→    /// BD annotation type: `#bd(Name)`
  1178→    BdAnnot {
  1179→        /// BD annotation name.
  1180→        name: String,
  1181→    },
  1182→}
  1183→
  1184→impl std::fmt::Display for Type {
  1185→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
  1186→        match self {
  1187→            Type::BDBase(name) => write!(f, "{}", name),
  1188→            Type::Ptr(inner) => write!(f, "*{}", inner),
  1189→            Type::RegionPtr { inner, region } => write!(f, "*{} @ {}", inner, region),
  1190→            Type::Array { element, size } => write!(f, "[{}; {}]", element, size),
  1191→            Type::Struct { name, .. } => write!(f, "{}", name),
  1192→            Type::Generic { name, args } => {
  1193→                let a: Vec<String> = args.iter().map(|t| t.to_string()).collect();
  1194→                write!(f, "{}<{}>", name, a.join(", "))
  1195→            }
  1196→            Type::Func {
  1197→                params,
  1198→                return_type,
  1199→            } => {
  1200→                let p: Vec<String> = params.iter().map(|t| t.to_string()).collect();
  1201→                write!(f, "({})", p.join(", "))?;
  1202→                if let Some(rt) = return_type {
  1203→                    write!(f, " -> {}", rt)?;
  1204→                }
  1205→                Ok(())
  1206→            }
  1207→            Type::BdAnnot { name } => write!(f, "#bd({})", name),
  1208→        }
  1209→    }
  1210→}
