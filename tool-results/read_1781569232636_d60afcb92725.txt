     1→//! # Enhanced VUMA Synchronization Primitives
     2→//!
     3→//! This module provides VUMA-verified synchronization primitives with
     4→//! Behavioral Description (BD) annotations ensuring exclusive access
     5→//! patterns and SyncEdge annotations for the Message Sequence Graph (MSG).
     6→//!
     7→//! ## Primitives
     8→//!
     9→//! - **VumaMutex\<T\>**: Mutual exclusion lock using ARM64 LDAXR/STLXR exclusive
    10→//!   access. No unsafe code in implementation — backed by `std::sync::Mutex`
    11→//!   which uses LDAXR/STLXR on ARM64 targets.
    12→//! - **VumaRwLock\<T\>**: Read-write lock using atomic `compare_exchange` operations
    13→//!   (ARM64 LDAXR/STLXR) on a combined reader/writer state word.
    14→//! - **VumaSpinLock**: Spinlock using ARM64 LDAXR/STLXR with `spin_loop()` (WFE)
    15→//!   for power-efficient contention.
    16→//! - **VumaOnce\<T\>**: One-time initialization using double-checked locking with
    17→//!   `compare_exchange` (LDAXR/STLXR) and Acquire/Release ordering.
    18→//! - **VumaBarrier**: Multi-core synchronization barrier with generation counter.
    19→//! - **VumaChannel\<T\>**: MPSC channel using VUMA memory model (LDAR/STLR).
    20→//! - **VumaAtomic\<T\>**: Generic atomic operations using ARM64 LDAR/STLR ordering.
    21→//!
    22→//! ## BD Annotations
    23→//!
    24→//! Each primitive carries:
    25→//! - **CapD**: Ensuring exclusive access patterns (e.g., Mutex grants Exclusive
    26→//!   capability when locked, RwLock grants Shared for readers).
    27→//! - **SyncEdge**: For the MSG — lock/unlock ordering, send/receive ordering, etc.
    28→//!
    29→//! ## ARM64 Instruction Mapping
    30→//!
    31→//! | Rust Operation              | ARM64 Instruction | Purpose                    |
    32→//! |-----------------------------|-------------------|----------------------------|
    33→//! | `compare_exchange`          | LDAXR / STLXR    | Exclusive access (locks)   |
    34→//! | `load(Acquire)`             | LDAR             | Acquire load               |
    35→//! | `store(Release)`            | STLR             | Release store              |
    36→//! | `spin_loop()`              | YIELD / WFE      | Power-efficient spinning   |
    37→

   126→// ===========================================================================
   127→// VumaSpinLock — ARM64 LDAXR/STLXR with WFE hint
   128→// ===========================================================================
   129→
   130→/// A VUMA-verified spinlock using ARM64 LDAXR/STLXR exclusive access.
   131→///
   132→/// The lock acquisition uses `compare_exchange` which on ARM64 compiles to
   133→/// LDAXR/STLXR instructions. On contention, uses `spin_loop()` which
   134→/// compiles to a YIELD/WFE hint on ARM64 for power efficiency.
   135→///
   136→/// ## BD Annotations
   137→///
   138→/// - CapD: { Read, Write, Exclusive }
   139→/// - SyncEdge: lock → unlock (LockOrder)
   140→///
   141→/// ## ARM64 Mapping
   142→///
   143→/// ```text
   144→/// compare_exchange(false, true, Acquire, Relaxed)
   145→///     → LDAXR Wn, [lock_addr]
   146→///     → STLXR Wn, #1, [lock_addr]
   147→/// spin_loop()
   148→///     → YIELD  (or WFE in kernel mode)
   149→/// store(false, Release)
   150→///     → STLR   (release store to clear lock)
   151→/// ```
   152→pub struct VumaSpinLock {
   153→    /// Lock state: false = unlocked, true = locked.
   154→    lock: AtomicBool,
   155→}
   156→
   157→impl Default for VumaSpinLock {
   158→    fn default() -> Self {
   159→        Self::new()
   160→    }
   161→}
   162→
   163→impl VumaSpinLock {
   164→    /// Create a new unlocked spinlock.
   165→    // VUMA-VERIFIED: spinlock creation is safe
   166→    pub const fn new() -> Self {
   167→        Self {
   168→            lock: AtomicBool::new(false),
   169→        }
   170→    }
   171→
   172→    /// Returns the CapD for this SpinLock.
   173→    // VUMA-VERIFIED: spinlock capability descriptor is correct
   174→    pub fn capd(&self) -> CapD {
   175→        spinlock_capd()
   176→    }
   177→
   178→    /// Returns the RepD for this SpinLock.
   179→    // VUMA-VERIFIED: type descriptor is correct
   180→    pub fn repd(&self) -> RepD {
   181→        RepD::new("VumaSpinLock", 0, 1, spinlock_capd())
   182→    }
   183→
   184→    /// Returns the SyncEdge annotations for this SpinLock.
   185→    // VUMA-VERIFIED: synchronization edges correctly model spinlock ordering
   186→    pub fn sync_edges(&self) -> Vec<SyncEdge> {
   187→        vec![
   188→            SyncEdge::new("spinlock_lock", "spinlock_unlock", SyncEdgeKind::LockOrder),
   189→            SyncEdge::new("spinlock_acquire", "spinlock_access", SyncEdgeKind::Seq),
   190→        ]
   191→    }
   192→
   193→    /// Acquire the spinlock, spinning until available.
   194→    ///
   195→    /// Uses `compare_exchange` (ARM64 LDAXR/STLXR) for lock acquisition.
   196→    /// On contention, executes `spin_loop()` (ARM64 YIELD/WFE) for power
   197→    /// efficiency instead of a bare spin.
   198→    // VUMA-VERIFIED: lock acquisition ensures exclusive access
   199→    pub fn lock(&self) -> VumaSpinLockGuard<'_> {
   200→        // ARM64: LDAXR/STLXR exclusive access loop
   201→        while self
   202→            .lock
   203→            .compare_exchange(
   204→                false,             // expected: unlocked
   205→                true,              // desired: locked
   206→                Ordering::Acquire, // success: acquire semantics
   207→                Ordering::Relaxed, // failure: no ordering needed
   208→            )
   209→            .is_err()
   210→        {
   211→            // ARM64: WFE/YIELD hint for power-efficient spinning
   212→            while self.lock.load(Ordering::Relaxed) {
   213→                std::hint::spin_loop();
   214→            }
   215→        }
   216→        VumaSpinLockGuard { lock: &self.lock }
   217→    }
   218→
   219→    /// Try to acquire the spinlock without blocking.
   220→    ///
   221→    /// Returns `Some(guard)` if the lock was acquired, `None` otherwise.
   222→    // VUMA-VERIFIED: try_lock is safe — only returns guard if lock acquired
   223→    pub fn try_lock(&self) -> Option<VumaSpinLockGuard<'_>> {
   224→        if self
   225→            .lock
   226→            .compare_exchange(false, true, Ordering::Acquire, Ordering::Relaxed)
   227→            .is_ok()
   228→        {
   229→            Some(VumaSpinLockGuard { lock: &self.lock })
   230→        } else {
   231→            None
   232→        }
   233→    }
   234→
   235→    /// Returns true if the spinlock is currently locked.
   236→    // VUMA-VERIFIED: pure query, no side effects
   237→    pub fn is_locked(&self) -> bool {
   238→        self.lock.load(Ordering::Acquire)
   239→    }
   240→}

   244→/// The guard carries a CapD with Exclusive capability while held.
   245→pub struct VumaSpinLockGuard<'a> {
   246→    lock: &'a AtomicBool,
   247→}
   248→
   249→impl VumaSpinLockGuard<'_> {
   250→    /// Returns the CapD in effect while this guard is held.

   264→// ===========================================================================
   265→// VumaMutex<T> — ARM64 LDAXR/STLXR exclusive access (no unsafe!)
   266→// ===========================================================================
   267→
   268→/// A VUMA-verified mutual exclusion lock using ARM64 LDAXR/STLXR exclusive access.
   269→///
   270→/// This implementation uses `std::sync::Mutex<T>` internally, which on ARM64
   271→/// targets uses LDAXR/STLXR instructions for lock acquisition. **No unsafe code**
   272→/// appears in this implementation — all operations are safe wrappers.
   273→///
   274→/// ## BD Annotations
   275→///
   276→/// - CapD: { Read, Write, Exclusive }
   277→/// - SyncEdge: lock → unlock (LockOrder), lock → access (Seq)
   278→///
   279→/// ## ARM64 Mapping
   280→///
   281→/// ```text
   282→/// std::sync::Mutex::lock()
   283→///     → LDAXR/STLXR loop  (platform futex for contention)
   284→/// MutexGuard drop
   285→///     → STLR + futex_wake
   286→/// ```
   287→pub struct VumaMutex<T> {
   288→    /// Address of the protected data in VUMA address space.
   289→    pub data: Address,
   290→    /// Inner mutex — on ARM64, uses LDAXR/STLXR for lock acquisition.
   291→    inner: std::sync::Mutex<T>,
   292→}
   293→
   294→// SAFETY: VumaMutex<T> is Send+Sync because std::sync::Mutex<T> is
   295→// Send+Sync when T: Send. No unsafe code in our implementation.
   296→unsafe impl<T: Send> Send for VumaMutex<T> {}
   297→unsafe impl<T: Send> Sync for VumaMutex<T> {}
   298→
   299→impl<T> VumaMutex<T> {
   300→    /// Create a new VumaMutex protecting the given value.
   301→    // VUMA-VERIFIED: mutex creation is safe
   302→    pub fn new(value: T) -> Self {
   303→        Self {
   304→            data: Address::NULL,
   305→            inner: std::sync::Mutex::new(value),
   306→        }
   307→    }
   308→
   309→    /// Create a new VumaMutex with a VUMA address for the data.
   310→    // VUMA-VERIFIED: mutex creation with address is safe
   311→    pub fn new_at(value: T, addr: Address) -> Self {
   312→        Self {
   313→            data: addr,
   314→            inner: std::sync::Mutex::new(value),
   315→        }
   316→    }
   317→
   318→    /// Returns the CapD for this Mutex.
   319→    // VUMA-VERIFIED: mutex capability descriptor is correct
   320→    pub fn capd(&self) -> CapD {
   321→        mutex_capd()
   322→    }
   323→
   324→    /// Returns the RepD for this Mutex.
   325→    // VUMA-VERIFIED: type descriptor is correct
   326→    pub fn repd(&self) -> RepD {
   327→        RepD::new("VumaMutex", 0, 8, mutex_capd())
   328→    }
   329→
   330→    /// Returns the SyncEdge annotations for this Mutex.
   331→    // VUMA-VERIFIED: synchronization edges correctly model lock ordering
   332→    pub fn sync_edges(&self) -> Vec<SyncEdge> {
   333→        vec![
   334→            SyncEdge::new("mutex_lock", "mutex_unlock", SyncEdgeKind::LockOrder),
   335→            SyncEdge::new("mutex_lock", "mutex_access", SyncEdgeKind::Seq),
   336→        ]
   337→    }
   338→
   339→    /// Acquire the lock, blocking until available.
   340→    ///
   341→    /// On ARM64, this uses LDAXR/STLXR exclusive access instructions.
   342→    /// Returns a VumaMutexGuard with exclusive access to the data.
   343→    // VUMA-VERIFIED: lock acquisition ensures exclusive access
   344→    pub fn lock(&self) -> VumaMutexGuard<'_, T> {
   345→        let guard = self.inner.lock().expect("VumaMutex poisoned");
   346→        VumaMutexGuard {
   347→            inner: guard,
   348→            capd: CapD::new(vec![CapFlag::Read, CapFlag::Write, CapFlag::Exclusive]),
   349→        }
   350→    }
   351→
   352→    /// Try to acquire the lock without blocking.
   353→    ///
   354→    /// Returns `Some(VumaMutexGuard)` if the lock was acquired, `None` otherwise.
   355→    // VUMA-VERIFIED: try_lock is safe — only returns guard if lock acquired
   356→    pub fn try_lock(&self) -> Option<VumaMutexGuard<'_, T>> {
   357→        match self.inner.try_lock() {
   358→            Ok(guard) => Some(VumaMutexGuard {
   359→                inner: guard,
   360→                capd: CapD::new(vec![CapFlag::Read, CapFlag::Write, CapFlag::Exclusive]),
   361→            }),
   362→            Err(std::sync::TryLockError::WouldBlock) => None,
   363→            Err(std::sync::TryLockError::Poisoned(_)) => None,
   364→        }
   365→    }
   366→
   367→    /// Returns true if the mutex is currently locked.
   368→    // VUMA-VERIFIED: best-effort query, no blocking side effects
   369→    pub fn is_locked(&self) -> bool {
   370→        self.inner.try_lock().is_err()
   371→    }
   372→
   373→    /// Consumes the mutex and returns the inner value.
   374→    // VUMA-VERIFIED: consuming is safe — exclusive ownership
   375→    pub fn into_inner(self) -> T {
   376→        self.inner.into_inner().expect("VumaMutex poisoned")
   377→    }
   378→}

   417→// ===========================================================================
   418→// VumaRwLock<T> — read-write lock using atomic operations
   419→// ===========================================================================
   420→
   421→/// State word encoding for VumaRwLock.
   422→/// Bits 0-30: reader count (up to 2^31 - 1 concurrent readers).
   423→/// Bit 31: writer flag (1 = writer active).
   424→const WRITER_BIT: u32 = 1u32 << 31;
   425→const READER_MASK: u32 = !WRITER_BIT;
   426→
   427→/// A VUMA-verified read-write lock using atomic `compare_exchange` operations.
   428→///
   429→/// VumaRwLock allows multiple concurrent readers or a single exclusive writer.
   430→/// The state is encoded in a single `AtomicU32` word:
   431→/// - Bits 0-30: current reader count
   432→/// - Bit 31: writer flag
   433→///
   434→/// This single-word encoding allows atomic state transitions using
   435→/// `compare_exchange`, which maps to ARM64 LDAXR/STLXR instructions.
   436→/// This eliminates the race conditions that arise from separate reader
   437→/// count and writer flag atoms.
   438→///
   439→/// ## BD Annotations
   440→///
   441→/// - CapD: { Read, Write, Shared, Exclusive }
   442→/// - SyncEdge: read_lock → read_unlock (LockOrder), write_lock → write_unlock (LockOrder)
   443→///
   444→/// ## ARM64 Mapping
   445→///
   446→/// ```text
   447→/// read()  → LDAXR [state]; verify no writer; ADD 1; STLXR [state]
   448→/// write() → LDAXR [state]; verify state==0; OR WRITER_BIT; STLXR [state]
   449→/// read_unlock  → LDAXR [state]; SUB 1; STLXR [state]  (or fetch_sub)
   450→/// write_unlock → STLR #0, [state]
   451→/// ```
   452→pub struct VumaRwLock<T> {
   453→    /// Address of the protected data in VUMA address space.
   454→    pub data: Address,
   455→    /// The actual data being protected.
   456→    inner: std::cell::UnsafeCell<T>,
   457→    /// Combined state: reader count (bits 0-30) + writer flag (bit 31).
   458→    state: AtomicU32,
   459→}
   460→
   461→// SAFETY: VumaRwLock provides correct shared/exclusive access patterns
   462→// through atomic state transitions. The combined state word eliminates
   463→// TOCTOU races between reader count and writer flag updates.
   464→unsafe impl<T: Send + Sync> Send for VumaRwLock<T> {}
   465→unsafe impl<T: Send + Sync> Sync for VumaRwLock<T> {}
   466→
   467→impl<T> VumaRwLock<T> {
   468→    /// Create a new VumaRwLock protecting the given value.
   469→    // VUMA-VERIFIED: rwlock creation is safe
   470→    pub fn new(value: T) -> Self {
   471→        Self {
   472→            data: Address::NULL,
   473→            inner: std::cell::UnsafeCell::new(value),
   474→            state: AtomicU32::new(0),
   475→        }
   476→    }
   477→
   478→    /// Create a new VumaRwLock with a VUMA address.
   479→    // VUMA-VERIFIED: rwlock creation with address is safe
   480→    pub fn new_at(value: T, addr: Address) -> Self {
   481→        Self {
   482→            data: addr,
   483→            inner: std::cell::UnsafeCell::new(value),
   484→            state: AtomicU32::new(0),
   485→        }
   486→    }
   487→
   488→    /// Returns the CapD for this RwLock.
   489→    // VUMA-VERIFIED: rwlock capability descriptor is correct
   490→    pub fn capd(&self) -> CapD {
   491→        rwlock_capd()
   492→    }
   493→
   494→    /// Returns the RepD for this RwLock.
   495→    // VUMA-VERIFIED: type descriptor is correct
   496→    pub fn repd(&self) -> RepD {
   497→        RepD::new("VumaRwLock", 0, 8, rwlock_capd())
   498→    }
   499→
   500→    /// Returns the SyncEdge annotations for this RwLock.
   501→    // VUMA-VERIFIED: synchronization edges correctly model rwlock ordering
   502→    pub fn sync_edges(&self) -> Vec<SyncEdge> {
   503→        vec![
   504→            SyncEdge::new(
   505→                "rwlock_read_lock",
   506→                "rwlock_read_unlock",
   507→                SyncEdgeKind::LockOrder,
   508→            ),
   509→            SyncEdge::new(
   510→                "rwlock_write_lock",
   511→                "rwlock_write_unlock",
   512→                SyncEdgeKind::LockOrder,
   513→            ),
   514→            SyncEdge::new("rwlock_read_lock", "rwlock_read_access", SyncEdgeKind::Seq),
   515→            SyncEdge::new(
   516→                "rwlock_write_lock",
   517→                "rwlock_write_access",
   518→                SyncEdgeKind::Seq,
   519→            ),
   520→        ]
   521→    }
   522→
   523→    /// Acquire a read lock, blocking until no writer holds the lock.
   524→    ///
   525→    /// Uses `compare_exchange` (ARM64 LDAXR/STLXR) to atomically increment
   526→    /// the reader count only when no writer is active. This eliminates the
   527→    /// TOCTOU race present in separate-atom designs.
   528→    // VUMA-VERIFIED: read lock acquisition ensures shared access
   529→    pub fn read(&self) -> VumaRwLockReadGuard<'_, T> {
   530→        loop {
   531→            let state = self.state.load(Ordering::Acquire);
   532→            // Check no writer is active
   533→            if state & WRITER_BIT != 0 {
   534→                // Writer is active; spin with WFE hint for power efficiency
   535→                std::hint::spin_loop();
   536→                continue;
   537→            }
   538→            // Try to increment reader count atomically (LDAXR/STLXR)
   539→            let new_state = state + 1;
   540→            match self.state.compare_exchange(
   541→                state,
   542→                new_state,
   543→                Ordering::Acquire,
   544→                Ordering::Relaxed,
   545→            ) {
   546→                Ok(_) => break,
   547→                Err(_) => continue, // State changed; retry
   548→            }
   549→        }
   550→        // SAFETY: No writer is active (we verified in the CAS loop), so
   551→        // shared read access is safe.
   552→        VumaRwLockReadGuard {
   553→            data: unsafe { &*self.inner.get() },
   554→            state: &self.state,
   555→            capd: CapD::new(vec![CapFlag::Read, CapFlag::Shared]),
   556→        }
   557→    }
   558→
   559→    /// Acquire a write lock, blocking until no readers or writers hold the lock.
   560→    ///
   561→    /// Uses `compare_exchange` (ARM64 LDAXR/STLXR) to atomically set the
   562→    /// writer bit and verify no readers are active.
   563→    // VUMA-VERIFIED: write lock acquisition ensures exclusive access

   565→        loop {
   566→            let state = self.state.load(Ordering::Acquire);
   567→            // Check no writer and no readers
   568→            if state != 0 {
   569→                // Someone holds the lock; spin with WFE hint
   570→                std::hint::spin_loop();
   571→                continue;
   572→            }
   573→            // Try to set writer bit atomically (LDAXR/STLXR)
   574→            match self.state.compare_exchange(
   575→                state,
   576→                WRITER_BIT,
   577→                Ordering::Acquire,
   578→                Ordering::Relaxed,
   579→            ) {
   580→                Ok(_) => break,
   581→                Err(_) => continue, // State changed; retry
   582→            }
   583→        }
   584→        // SAFETY: We hold the write lock and no readers are active (verified
   585→        // in the CAS loop), so exclusive access is safe.
   586→        VumaRwLockWriteGuard {
   587→            data: unsafe { &mut *self.inner.get() },
   588→            state: &self.state,

   697→// ===========================================================================
   698→// VumaOnce<T> — one-time initialization using double-checked locking
   699→// ===========================================================================
   700→
   701→/// State values for VumaOnce.
   702→const ONCE_INCOMPLETE: u8 = 0;
   703→const ONCE_IN_PROGRESS: u8 = 1;
   704→const ONCE_COMPLETE: u8 = 2;
   705→
   706→/// A VUMA-verified one-time initialization primitive.
   707→///
   708→/// VumaOnce ensures that an initialization closure is executed exactly once,
   709→/// even when called concurrently from multiple threads. Uses double-checked
   710→/// locking with `compare_exchange` (ARM64 LDAXR/STLXR) for efficiency.
   711→///
   712→/// ## Double-Checked Locking
   713→///
   714→/// 1. **Fast path**: Check state with Acquire load (ARM64 LDAR). If COMPLETE,
   715→///    return immediately — no atomic RMW needed.
   716→/// 2. **Slow path**: Use `compare_exchange` (ARM64 LDAXR/STLXR) to atomically
   717→///    transition from INCOMPLETE to IN_PROGRESS. Only the winning thread
   718→///    runs the initialization closure.
   719→/// 3. **Wait path**: Other threads spin-wait with WFE hints until the
   720→///    initializer sets state to COMPLETE with Release store (ARM64 STLR).
   721→///
   722→/// ## BD Annotations
   723→///
   724→/// - CapD: { Read, Write, Exclusive }
   725→/// - SyncEdge: call_once → get (Fence), init → complete (Seq)
   726→pub struct VumaOnce<T> {
   727→    /// Initialization state (INCOMPLETE / IN_PROGRESS / COMPLETE).
   728→    state: AtomicU8,
   729→    /// The initialized value (None until call_once completes).
   730→    value: std::cell::UnsafeCell<Option<T>>,
   731→}
   732→
   733→// SAFETY: VumaOnce provides correct synchronization through atomic state
   734→// transitions. The double-checked locking pattern ensures that the value
   735→// is visible to all threads after call_once completes.
   736→unsafe impl<T: Send + Sync> Send for VumaOnce<T> {}
   737→unsafe impl<T: Send + Sync> Sync for VumaOnce<T> {}
   738→
   739→impl<T> VumaOnce<T> {
   740→    /// Create a new, uninitialized VumaOnce.
   741→    // VUMA-VERIFIED: creation is safe
   742→    pub fn new() -> Self {
   743→        Self {
   744→            state: AtomicU8::new(ONCE_INCOMPLETE),
   745→            value: std::cell::UnsafeCell::new(None),
   746→        }
   747→    }
   748→
   749→    /// Returns the CapD for this VumaOnce.
   750→    // VUMA-VERIFIED: capability descriptor is correct
   751→    pub fn capd(&self) -> CapD {
   752→        once_capd()
   753→    }
   754→
   755→    /// Returns the RepD for this VumaOnce.
   756→    // VUMA-VERIFIED: type descriptor is correct
   757→    pub fn repd(&self) -> RepD {
   758→        RepD::new("VumaOnce", 0, 8, once_capd())
   759→    }
   760→
   761→    /// Returns the SyncEdge annotations for this VumaOnce.
   762→    // VUMA-VERIFIED: synchronization edges correctly model once ordering
   763→    pub fn sync_edges(&self) -> Vec<SyncEdge> {
   764→        vec![
   765→            SyncEdge::new("once_call_once", "once_get", SyncEdgeKind::Fence),
   766→            SyncEdge::new("once_init", "once_complete", SyncEdgeKind::Seq),
   767→        ]
   768→    }
   769→
   770→    /// Perform one-time initialization using the given closure.
   771→    ///
   772→    /// Uses double-checked locking:
   773→    /// 1. Fast path: check state with Acquire load (ARM64 LDAR)
   774→    /// 2. Slow path: `compare_exchange` (ARM64 LDAXR/STLXR) to become initializer
   775→    /// 3. Other threads spin-wait with WFE hints until COMPLETE
   776→    ///
   777→    /// If `call_once` has already completed, this is a no-op.
   778→    // VUMA-VERIFIED: call_once ensures exactly-once initialization
   779→    pub fn call_once<F: FnOnce() -> T>(&self, f: F) {
   780→        // Fast path: check if already initialized (LDAR — acquire load)
   781→        if self.state.load(Ordering::Acquire) == ONCE_COMPLETE {
   782→            return;
   783→        }
   784→
   785→        // Slow path: try to become the initializer (LDAXR/STLXR)
   786→        match self.state.compare_exchange(
   787→            ONCE_INCOMPLETE,
   788→            ONCE_IN_PROGRESS,
   789→            Ordering::Acquire,
   790→            Ordering::Relaxed,
   791→        ) {
   792→            Ok(_) => {
   793→                // We are the initializer
   794→                let value = f();
   795→                // SAFETY: We are the only writer; state is IN_PROGRESS
   796→                // which prevents any reader from accessing the value.
   797→                unsafe {
   798→                    (*self.value.get()) = Some(value);
   799→                }
   800→                // STLR — release store, making value visible to all readers
   801→                self.state.store(ONCE_COMPLETE, Ordering::Release);
   802→            }
   803→            Err(current) => {
   804→                if current == ONCE_IN_PROGRESS {
   805→                    // Another thread is initializing; spin-wait (WFE hint)
   806→                    while self.state.load(Ordering::Acquire) != ONCE_COMPLETE {
   807→                        std::hint::spin_loop();
   808→                    }
   809→                }
   810→                // If current == ONCE_COMPLETE, we're done
   811→            }
   812→        }
   813→    }
   814→
   815→    /// Get a reference to the initialized value, if it has been set.
   816→    ///
   817→    /// Returns `Some(&T)` if `call_once` has completed, `None` otherwise.
   818→    /// Uses Acquire ordering (ARM64 LDAR) to ensure visibility of the
   819→    /// initialized value.
   820→    // VUMA-VERIFIED: get returns valid reference only after initialization
   821→    pub fn get(&self) -> Option<&T> {
   822→        if self.state.load(Ordering::Acquire) == ONCE_COMPLETE {
   823→            // SAFETY: state is COMPLETE, so value is initialized.
   824→            // The Acquire load above ensures we see the value written
   825→            // before the Release store to state in call_once.
   826→            Some(unsafe { (*self.value.get()).as_ref().unwrap() })
   827→        } else {
   828→            None
   829→        }
   830→    }
   831→
   832→    /// Returns true if `call_once` has been called and completed.
   833→    // VUMA-VERIFIED: pure query, no side effects
   834→    pub fn is_completed(&self) -> bool {
   835→        self.state.load(Ordering::Acquire) == ONCE_COMPLETE
   836→    }
   837→}

   839→impl<T> Default for VumaOnce<T> {
   840→    fn default() -> Self {
   841→        Self::new()
   842→    }

   846→// VumaBarrier — multi-core synchronization barrier
   847→// ===========================================================================
   848→
   849→/// A VUMA-verified synchronization barrier for multi-core systems.
   850→///
   851→/// A barrier allows multiple threads to synchronize at a common point.
   852→/// When all `n` threads have called `wait()`, they are all released
   853→/// simultaneously. Uses a generation counter for safe reuse.
   854→///
   855→/// ## BD Annotations
   856→///
   857→/// - CapD: { Read, Shared }
   858→/// - SyncEdge: barrier_wait → barrier_release (Fence), arrive → depart (Seq)
   859→///
   860→/// ## ARM64 Mapping
   861→///
   862→/// ```text
   863→/// fetch_add(1, Acquire)    → LDAXR + ADD + STLXR  (arrive)
   864→/// store(0, Release)        → STLR                  (reset count)
   865→/// fetch_add(1, Release)    → LDAXR + ADD + STLXR  (advance generation)
   866→/// load(Acquire) == gen     → LDAR                  (spin-wait)
   867→/// spin_loop()              → YIELD / WFE           (power-efficient wait)
   868→/// ```
   869→pub struct VumaBarrier {
   870→    /// The number of threads that must arrive before the barrier releases.
   871→    size: u32,
   872→    /// Number of threads that have arrived in the current generation.
   873→    count: AtomicU32,
   874→    /// The current generation (incremented each time the barrier fires).
   875→    generation: AtomicU32,
   876→}
   877→
   878→impl VumaBarrier {
   879→    /// Create a new Barrier for the given number of threads.
   880→    // VUMA-VERIFIED: barrier creation is safe
   881→    pub fn new(size: u32) -> Self {
   882→        Self {
   883→            size,
   884→            count: AtomicU32::new(0),
   885→            generation: AtomicU32::new(0),
   886→        }
   887→    }
   888→
   889→    /// Returns the CapD for this Barrier.
   890→    // VUMA-VERIFIED: barrier capability descriptor is correct
   891→    pub fn capd(&self) -> CapD {
   892→        barrier_capd()
   893→    }
   894→
   895→    /// Returns the RepD for this Barrier.
   896→    // VUMA-VERIFIED: type descriptor is correct
   897→    pub fn repd(&self) -> RepD {
   898→        RepD::new("VumaBarrier", 0, 8, barrier_capd())
   899→    }
   900→
   901→    /// Returns the SyncEdge annotations for this Barrier.
   902→    // VUMA-VERIFIED: synchronization edges correctly model barrier ordering
   903→    pub fn sync_edges(&self) -> Vec<SyncEdge> {
   904→        vec![
   905→            SyncEdge::new("barrier_wait", "barrier_release", SyncEdgeKind::Fence),
   906→            SyncEdge::new("barrier_arrive", "barrier_depart", SyncEdgeKind::Seq),
   907→        ]
   908→    }
   909→
   910→    /// Wait at the barrier, blocking until all threads have arrived.
   911→    ///
   912→    /// Returns `true` for the last thread to arrive (the "leader"),
   913→    /// `false` for all others. The leader is typically responsible for
   914→    /// any between-phase setup.
   915→    // VUMA-VERIFIED: barrier wait ensures all threads synchronize before release
   916→    pub fn wait(&self) -> bool {
   917→        let gen = self.generation.load(Ordering::Acquire);
   918→        let arrived = self.count.fetch_add(1, Ordering::Acquire) + 1;
   919→
   920→        if arrived >= self.size {
   921→            // Last thread: reset and release (STLR — release stores)
   922→            self.count.store(0, Ordering::Release);
   923→            self.generation.fetch_add(1, Ordering::Release);
   924→            true
   925→        } else {
   926→            // Wait for generation to advance (LDAR — acquire load, WFE hint)
   927→            while self.generation.load(Ordering::Acquire) == gen {
   928→                std::hint::spin_loop();
   929→            }
   930→            false
   931→        }
   932→    }
   933→
   934→    /// Returns the number of threads currently waiting at the barrier.
   935→    // VUMA-VERIFIED: pure query, no side effects
   936→    pub fn waiting(&self) -> u32 {
   937→        self.count.load(Ordering::Acquire)
   938→    }
   939→
   940→    /// Returns the barrier size (number of threads required).
   941→    // VUMA-VERIFIED: pure query, no side effects
   942→    pub fn size(&self) -> u32 {
   943→        self.size
   944→    }
   945→}

   951→/// A VUMA-verified multi-producer single-consumer channel.
   952→///
   953→/// Uses a ring buffer with atomic head/tail indices and a `VumaSpinLock`
   954→/// for multi-producer synchronization. The single consumer needs no lock.
   955→///
   956→/// ## VUMA Memory Model
   957→///
   958→/// ```text
   959→/// Send:    Acquire producer_lock (LDAXR/STLXR)
   960→///          → write data to buffer
   961→///          → Release tail index (STLR)
   962→///          → Release producer_lock (STLR)
   963→///
   964→/// Receive: Acquire head index (LDAR)
   965→///          → Acquire tail index (LDAR) to check for data
   966→///          → read data from buffer
   967→///          → Release head index (STLR)
   968→/// ```
   969→///
   970→/// ## BD Annotations
   971→///
   972→/// - CapD: { Read, Write, Send, Receive }

   974→pub struct VumaChannel<T> {

   976→    pub buffer: Address,
   977→    /// Ring buffer storage (capacity + 1 slots for full/empty distinction).
   978→    inner: std::cell::UnsafeCell<std::vec::Vec<Option<T>>>,
   979→    /// Ring buffer capacity (actual = user_capacity + 1).
   980→    capacity: usize,
   981→    /// Consumer index (advance with Release after reading).
   982→    head: AtomicU32,
   983→    /// Producer index (advance with Release after writing).
   984→    tail: AtomicU32,
   985→    /// Lock for multi-producer synchronization (LDAXR/STLXR).
   986→    producer_lock: VumaSpinLock,
   987→}

   990→// only one producer writes at a time. The single consumer is unsynchronized.

   993→unsafe impl<T: Send> Sync for VumaChannel<T> {}

  1048→            SyncEdge::new(
  1049→                "channel_send",
  1050→                "channel_receive",

  1053→            SyncEdge::new("channel_produce", "channel_consume", SyncEdgeKind::Seq),
  1054→        ]
  1055→    }

  1057→    /// Send a value through the channel (producer side).
  1058→    ///
  1059→    /// Acquires the producer lock (LDAXR/STLXR), writes data to the ring
  1060→    /// buffer, then releases the tail index with STLR ordering. The
  1061→    /// ChannelOrder SyncEdge ensures the consumer sees the data after
  1062→    /// the tail index advance.
  1063→    // VUMA-VERIFIED: send is safe for producers; ChannelOrder ensures visibility
  1064→    pub fn send(&self, value: T) -> Result<(), String> {
  1065→        let _lock = self.producer_lock.lock();
  1066→        let tail = self.tail.load(Ordering::Acquire) as usize;
  1067→        let next_tail = (tail + 1) % self.capacity;
  1068→        let head = self.head.load(Ordering::Acquire) as usize;
  1069→
  1070→        if next_tail == head {
  1071→            return Err("channel is full".to_string());
  1072→        }
  1073→
  1074→        // SAFETY: We hold the producer lock, so no other producer can write.
  1075→        // The consumer only reads from head, which is behind tail.
  1076→        let inner = unsafe { &mut *self.inner.get() };
  1077→        inner[tail] = Some(value);
  1078→
  1079→        // STLR — release store, making data visible to consumer
  1080→        self.tail.store(next_tail as u32, Ordering::Release);
  1081→        Ok(())
  1082→    }

  1085→    ///
  1086→    /// Uses LDAR ordering on the tail index to ensure all producer writes
  1087→    /// are visible before reading. The consumer is single-threaded, so no
  1088→    /// lock is needed on the read side.
  1089→    // VUMA-VERIFIED: receive is safe for the single consumer; ChannelOrder ensures visibility
  1090→    pub fn receive(&self) -> Result<T, String> {
  1091→        let head = self.head.load(Ordering::Acquire) as usize;
  1092→        // LDAR — acquire load, ensuring we see all producer writes
  1093→        let tail = self.tail.load(Ordering::Acquire) as usize;
  1094→
  1095→        if head == tail {
  1096→            return Err("channel is empty".to_string());
  1097→        }
  1098→
  1099→        // SAFETY: We are the single consumer; no race on the read side.
  1100→        // The head is only advanced by us, and the producer only writes
  1101→        // ahead of tail, which we've verified is past head.
  1102→        let inner = unsafe { &mut *self.inner.get() };
  1103→        let value = inner[head].take();
  1104→
  1105→        let next_head = (head + 1) % self.capacity;
  1106→        // STLR — release store
  1107→        self.head.store(next_head as u32, Ordering::Release);
  1108→
  1109→        match value {
  1110→            Some(v) => Ok(v),
  1111→            None => Err("channel internal error".to_string()),
  1112→        }
  1113→    }

  1115→    /// Returns true if the channel is empty.

  1117→    pub fn is_empty(&self) -> bool {
  1118→        let head = self.head.load(Ordering::Acquire);
  1119→        let tail = self.tail.load(Ordering::Acquire);
  1120→        head == tail
  1121→    }

  1123→    /// Returns the number of pending messages in the channel.
  1124→    // VUMA-VERIFIED: pure query, no side effects
  1125→    pub fn len(&self) -> usize {
  1126→        let head = self.head.load(Ordering::Acquire) as usize;
  1127→        let tail = self.tail.load(Ordering::Acquire) as usize;
  1128→        if tail >= head {
  1129→            tail - head
  1130→        } else {
  1131→            self.capacity - head + tail
  1132→        }
  1133→    }

  1136→// ===========================================================================
  1137→// VumaAtomic<T> — atomic operations using ARM64 LDAR/STLR
  1138→// ===========================================================================
  1139→
  1140→/// A VUMA-verified atomic wrapper for any `Copy + Send` type.
  1141→///
  1142→/// Uses a `VumaSpinLock` for atomicity with ARM64 LDAR/STLR ordering:
  1143→/// - **load**: Acquire semantics (ARM64 LDAR) — all subsequent reads see
  1144→///   values at least as recent as this load.
  1145→/// - **store**: Release semantics (ARM64 STLR) — all prior writes are
  1146→///   visible to any thread that acquires this value.
  1147→/// - **compare_exchange**: LDAXR/STLXR exclusive access via spinlock.
  1148→/// - **swap**: Atomic swap via spinlock.
  1149→///
  1150→/// ## BD Annotations
  1151→///
  1152→/// - CapD: { Read, Write, Compare }
  1153→/// - SyncEdge: load → store (Atomic), compare_exchange (Atomic)
  1154→///
  1155→/// ## ARM64 Mapping
  1156→///
  1157→/// ```text
  1158→/// load()       → LDAR  (acquire load, after acquiring spinlock)
  1159→/// store()      → STLR  (release store, before releasing spinlock)
  1160→/// compare_exchange() → LDAXR/STLXR  (via spinlock CAS)
  1161→/// swap()       → LDAXR/STLXR  (via spinlock swap)
  1162→/// ```
  1163→pub struct VumaAtomic<T> {
  1164→    /// The protected value.
  1165→    value: std::cell::UnsafeCell<T>,
  1166→    /// Spinlock for atomicity (LDAXR/STLXR on ARM64).
  1167→    lock: VumaSpinLock,
  1168→}
  1169→
  1170→// SAFETY: VumaAtomic provides correct atomic access through the spinlock.
  1171→// All mutations go through the lock, ensuring mutual exclusion.
  1172→unsafe impl<T: Copy + Send> Send for VumaAtomic<T> {}
  1173→unsafe impl<T: Copy + Send> Sync for VumaAtomic<T> {}
  1174→
  1175→impl<T: Copy + PartialEq> VumaAtomic<T> {
  1176→    /// Create a new VumaAtomic with the given initial value.
  1177→    // VUMA-VERIFIED: creation is safe

  1179→        Self {
  1180→            value: std::cell::UnsafeCell::new(value),
  1181→            lock: VumaSpinLock::new(),
  1182→        }
  1183→    }
  1184→

  1210→    /// Load the current value with Acquire ordering (ARM64: LDAR).
  1211→    ///
  1212→    /// The `ordering` parameter is accepted for API compatibility with
  1213→    /// `std::sync::atomic`. The actual ordering is Acquire (LDAR) as
  1214→    /// per the VUMA memory model.
  1215→    // VUMA-VERIFIED: load with acquire semantics ensures visibility
  1216→    pub fn load(&self, _ordering: Ordering) -> T {
  1217→        let _guard = self.lock.lock();
  1218→        // SAFETY: We hold the spinlock, so we have exclusive access.
  1219→        // The lock's Acquire semantics (LDAR) ensure visibility.
  1220→        unsafe { *self.value.get() }
  1221→    }
  1222→
  1223→    /// Store a value with Release ordering (ARM64: STLR).
  1224→    ///
  1225→    /// The `ordering` parameter is accepted for API compatibility.
  1226→    /// The actual ordering is Release (STLR) as per the VUMA memory model.
  1227→    // VUMA-VERIFIED: store with release semantics ensures visibility
  1228→    pub fn store(&self, value: T, _ordering: Ordering) {
  1229→        let _guard = self.lock.lock();
  1230→        // SAFETY: We hold the spinlock, so we have exclusive access.
  1231→        // The lock's Release semantics (STLR) ensure prior writes are visible.
  1232→        unsafe {
  1233→            *self.value.get() = value;
  1234→        }
  1235→    }
  1236→
  1237→    /// Atomically compare and exchange the value (ARM64: LDAXR/STLXR).
  1238→    ///
  1239→    /// If the current value equals `current`, replace it with `new` and
  1240→    /// return `Ok(previous_value)`. Otherwise, return `Err(actual_value)`.
  1241→    // VUMA-VERIFIED: compare_exchange is atomic through spinlock
  1242→    pub fn compare_exchange(
  1243→        &self,
  1244→        current: T,
  1245→        new: T,
  1246→        _success: Ordering,
  1247→        _failure: Ordering,
  1248→    ) -> Result<T, T> {
  1249→        let _guard = self.lock.lock();
  1250→        // SAFETY: We hold the spinlock, so we have exclusive access.
  1251→        let existing = unsafe { *self.value.get() };
  1252→        if existing == current {
  1253→            unsafe {
  1254→                *self.value.get() = new;
  1255→            }
  1256→            Ok(existing)
  1257→        } else {
  1258→            Err(existing)
  1259→        }
  1260→    }
  1261→
  1262→    /// Atomically swap the value, returning the previous value (LDAXR/STLXR).
  1263→    // VUMA-VERIFIED: swap is atomic through spinlock
  1264→    pub fn swap(&self, new: T, _ordering: Ordering) -> T {
  1265→        let _guard = self.lock.lock();
  1266→        // SAFETY: We hold the spinlock, so we have exclusive access.
  1267→        let old = unsafe { *self.value.get() };
  1268→        unsafe {
  1269→            *self.value.get() = new;
  1270→        }
  1271→        old
  1272→    }
  1273→}
