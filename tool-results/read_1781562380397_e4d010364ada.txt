     1→//! # WebAssembly 32-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the Wasm32 target.  WebAssembly is
     4→//! fundamentally different from register-based ISAs: it is a stack machine
     5→//! with structured control flow and a binary format.  This module provides:
     6→//!
     7→//! - `WasmType` — value types (i32, i64, f32, f64)
     8→//! - `WasmInstr` — exhaustive instruction enum covering all ops needed for VUMA IR
     9→//! - LEB128 encoding/decoding (unsigned and signed)
    10→//! - Wasm binary-format encoder that emits complete `.wasm` modules
    11→//! - `Wasm32Backend` — `Backend` implementation that lowers IR to Wasm bytecode
    12→

    24→// ===========================================================================
    25→// Wasm value types
    26→// ===========================================================================
    27→
    28→/// WebAssembly value types.
    29→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
    30→pub enum WasmType {
    31→    I32,
    32→    I64,
    33→    F32,
    34→    F64,
    35→}
    36→
    37→impl WasmType {
    38→    /// Binary encoding of this type in the Wasm format.
    39→    pub fn to_byte(&self) -> u8 {
    40→        match self {
    41→            WasmType::I32 => 0x7F,
    42→            WasmType::I64 => 0x7E,
    43→            WasmType::F32 => 0x7D,
    44→            WasmType::F64 => 0x7C,
    45→        }
    46→    }
    47→
    48→    /// Size in bytes of this Wasm type on the stack.
    49→    pub fn byte_size(&self) -> usize {
    50→        match self {
    51→            WasmType::I32 | WasmType::F32 => 4,
    52→            WasmType::I64 | WasmType::F64 => 8,
    53→        }
    54→    }
    55→
    56→    /// Whether this is an integer type.
    57→    pub fn is_integer(&self) -> bool {
    58→        matches!(self, WasmType::I32 | WasmType::I64)
    59→    }
    60→
    61→    /// Whether this is a floating-point type.
    62→    pub fn is_float(&self) -> bool {
    63→        matches!(self, WasmType::F32 | WasmType::F64)
    64→    }
    65→
    66→    /// Map an IRType to the corresponding WasmType.
    67→    pub fn from_ir_type(ty: &IRType) -> Option<WasmType> {
    68→        match ty {
    69→            IRType::I8
    70→            | IRType::I16
    71→            | IRType::I32
    72→            | IRType::U8
    73→            | IRType::U16
    74→            | IRType::U32
    75→            | IRType::Ptr
    76→            | IRType::Func => Some(WasmType::I32),
    77→            IRType::I64 | IRType::U64 => Some(WasmType::I64),
    78→            IRType::F32 => Some(WasmType::F32),
    79→            IRType::F64 => Some(WasmType::F64),
    80→            IRType::Void | IRType::Struct { .. } | IRType::Array { .. } | IRType::TaggedUnion { .. } => None,
    81→        }
    82→    }
    83→
    84→    /// Decode a WasmType from its binary encoding byte.
    85→    pub fn from_byte(byte: u8) -> Option<WasmType> {
    86→        match byte {
    87→            0x7F => Some(WasmType::I32),
    88→            0x7E => Some(WasmType::I64),
    89→            0x7D => Some(WasmType::F32),
    90→            0x7C => Some(WasmType::F64),
    91→            _ => None,
    92→        }
    93→    }
    94→}
    95→
    96→impl std::fmt::Display for WasmType {
    97→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    98→        match self {
    99→            WasmType::I32 => write!(f, "i32"),
   100→            WasmType::I64 => write!(f, "i64"),
   101→            WasmType::F32 => write!(f, "f32"),
   102→            WasmType::F64 => write!(f, "f64"),
   103→        }
   104→    }
   105→}

   456→impl WasmInstr {
   457→    /// Encode this instruction into Wasm bytecode, appending bytes to `out`.
   458→    pub fn encode(&self, out: &mut Vec<u8>) {
   459→        match self {
   460→            // ── Control ──────────────────────────────────────────────
   461→            WasmInstr::Block(ty) => {
   462→                out.push(0x02);
   463→                match ty {
   464→                    Some(t) => out.push(t.to_byte()),
   465→                    None => out.push(0x40), // void block type
   466→                }
   467→            }
   468→            WasmInstr::Loop(ty) => {
   469→                out.push(0x03);
   470→                match ty {
   471→                    Some(t) => out.push(t.to_byte()),
   472→                    None => out.push(0x40), // void loop type
   473→                }
   474→            }
   475→            WasmInstr::If(ty) => {
   476→                out.push(0x04);
   477→                match ty {
   478→                    Some(t) => out.push(t.to_byte()),
   479→                    None => out.push(0x40), // void if type
   480→                }
   481→            }
   482→            WasmInstr::Else => out.push(0x05),
   483→            WasmInstr::End => out.push(0x0B),
   484→            WasmInstr::Br(idx) => {
   485→                out.push(0x0C);
   486→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   487→            }
   488→            WasmInstr::BrIf(idx) => {
   489→                out.push(0x0D);
   490→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   491→            }
   492→            WasmInstr::BrTable { labels, default } => {
   493→                out.push(0x0E);
   494→                out.extend_from_slice(&encode_unsigned_leb128(labels.len() as u64));
   495→                for &lbl in labels {
   496→                    out.extend_from_slice(&encode_unsigned_leb128(lbl as u64));
   497→                }
   498→                out.extend_from_slice(&encode_unsigned_leb128(*default as u64));
   499→            }
   500→            WasmInstr::Return => out.push(0x0F),
   501→            WasmInstr::Call(idx) => {
   502→                out.push(0x10);
   503→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   504→            }
   505→            WasmInstr::CallIndirect {
   506→                type_idx,
   507→                table_idx,
   508→            } => {
   509→                out.push(0x11);
   510→                out.extend_from_slice(&encode_unsigned_leb128(*type_idx as u64));
   511→                out.extend_from_slice(&encode_unsigned_leb128(*table_idx as u64));
   512→            }
   513→
   514→            // ── Parametric ──────────────────────────────────────────
   515→            WasmInstr::Select => out.push(0x1B),
   516→            WasmInstr::Drop => out.push(0x1A),
   517→
   518→            // ── Variable ────────────────────────────────────────────
   519→            WasmInstr::LocalGet(idx) => {
   520→                out.push(0x20);
   521→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   522→            }
   523→            WasmInstr::LocalSet(idx) => {
   524→                out.push(0x21);
   525→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   526→            }
   527→            WasmInstr::LocalTee(idx) => {
   528→                out.push(0x22);
   529→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   530→            }
   531→            WasmInstr::GlobalGet(idx) => {
   532→                out.push(0x23);
   533→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   534→            }
   535→            WasmInstr::GlobalSet(idx) => {
   536→                out.push(0x24);
   537→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   538→            }
   539→
   540→            // ── Memory ──────────────────────────────────────────────
   541→            WasmInstr::I32Load { align, offset } => {
   542→                out.push(0x28);
   543→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   544→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   545→            }
   546→            WasmInstr::I64Load { align, offset } => {
   547→                out.push(0x29);
   548→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   549→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   550→            }
   551→            WasmInstr::F32Load { align, offset } => {
   552→                out.push(0x2A);
   553→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   554→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   555→            }
   556→            WasmInstr::F64Load { align, offset } => {
   557→                out.push(0x2B);
   558→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   559→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   560→            }

   947→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   948→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   949→            }
   950→            WasmInstr::I64AtomicRmw16CmpxchgU { align, offset } => {
   951→                out.push(0xFE);
   952→                out.extend_from_slice(&encode_unsigned_leb128(0x4D)); // i64.atomic.rmw16.cmpxchg_u
   953→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   954→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   955→            }
   956→            WasmInstr::I64AtomicRmw32CmpxchgU { align, offset } => {
   957→                out.push(0xFE);
   958→                out.extend_from_slice(&encode_unsigned_leb128(0x4E)); // i64.atomic.rmw32.cmpxchg_u
   959→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   960→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   961→            }
   962→
   963→            // ── Bulk Memory Operations ────────────────────────────
   964→            WasmInstr::MemoryCopy { src_mem, dst_mem } => {
   965→                out.push(0xFC); // multi-byte op prefix
   966→                out.extend_from_slice(&encode_unsigned_leb128(0x0A)); // memory.copy
   967→                out.extend_from_slice(&encode_unsigned_leb128(*src_mem as u64));
   968→                out.extend_from_slice(&encode_unsigned_leb128(*dst_mem as u64));
   969→            }
   970→            WasmInstr::MemoryFill { mem } => {
   971→                out.push(0xFC);
   972→                out.extend_from_slice(&encode_unsigned_leb128(0x0B)); // memory.fill
   973→                out.extend_from_slice(&encode_unsigned_leb128(*mem as u64));
   974→            }
   975→            WasmInstr::MemoryInit { data_idx, mem } => {
   976→                out.push(0xFC);
   977→                out.extend_from_slice(&encode_unsigned_leb128(0x08)); // memory.init
   978→                out.extend_from_slice(&encode_unsigned_leb128(*data_idx as u64));
   979→                out.extend_from_slice(&encode_unsigned_leb128(*mem as u64));
   980→            }
   981→        }
   982→    }
   983→
   984→    /// Encode this instruction and return the bytes as a new Vec.
   985→    pub fn to_bytes(&self) -> Vec<u8> {
   986→        let mut out = Vec::new();
   987→        self.encode(&mut out);
   988→        out
   989→    }
   990→}

   996→/// Encode an unsigned 64-bit value as unsigned LEB128.
   997→pub fn encode_unsigned_leb128(value: u64) -> Vec<u8> {
   998→    let mut result = Vec::new();
   999→    let mut val = value;
  1000→    loop {
  1001→        let mut byte = (val & 0x7F) as u8;
  1002→        val >>= 7;
  1003→        if val != 0 {
  1004→            byte |= 0x80;
  1005→        }
  1006→        result.push(byte);
  1007→        if val == 0 {
  1008→            break;
  1009→        }
  1010→    }
  1011→    result
  1012→}
  1013→
  1014→/// Encode a signed 64-bit value as signed LEB128.
  1015→pub fn encode_signed_leb128(value: i64) -> Vec<u8> {
  1016→    let mut result = Vec::new();
  1017→    let mut val = value;
  1018→    loop {
  1019→        let mut byte = (val & 0x7F) as u8;
  1020→        val >>= 7;
  1021→        // Sign bit of byte is bit 6
  1022→        let sign_bit = (byte & 0x40) != 0;
  1023→        // If val is 0 and sign bit is clear, or val is -1 and sign bit is set, we're done
  1024→        if (val == 0 && !sign_bit) || (val == -1 && sign_bit) {
  1025→            result.push(byte);
  1026→            break;
  1027→        }
  1028→        byte |= 0x80;
  1029→        result.push(byte);
  1030→    }
  1031→    result
  1032→}
  1033→
  1034→/// Decode an unsigned LEB128 value from a byte slice.
  1035→/// Returns (decoded_value, number_of_bytes_consumed).
  1036→pub fn decode_unsigned_leb128(bytes: &[u8]) -> (u64, usize) {
  1037→    let mut result: u64 = 0;
  1038→    let mut shift: u32 = 0;
  1039→    let mut count = 0;
  1040→    for &byte in bytes {
  1041→        count += 1;
  1042→        result |= ((byte & 0x7F) as u64) << shift;
  1043→        if (byte & 0x80) == 0 {
  1044→            break;
  1045→        }
  1046→        shift += 7;
  1047→    }
  1048→    (result, count)
  1049→}
  1050→
  1051→/// Decode a signed LEB128 value from a byte slice.
  1052→/// Returns (decoded_value, number_of_bytes_consumed).
  1053→pub fn decode_signed_leb128(bytes: &[u8]) -> (i64, usize) {
  1054→    let mut result: i64 = 0;
  1055→    let mut shift: u32 = 0;
  1056→    let mut count = 0;
  1057→    let mut byte: u8;
  1058→    loop {
  1059→        byte = bytes[count];
  1060→        count += 1;
  1061→        result |= ((byte & 0x7F) as i64) << shift;
  1062→        shift += 7;
  1063→        if (byte & 0x80) == 0 {
  1064→            break;
  1065→        }
  1066→    }
  1067→    // Sign-extend if the sign bit is set
  1068→    if shift < 64 && (byte & 0x40) != 0 {
  1069→        result |= !0i64 << shift;
  1070→    }
  1071→    (result, count)
  1072→}

  1174→    fn encode(&self) -> Vec<u8> {
  1175→        let mut out = Vec::new();

  1177→        // Flag bits: bit 0 = has_max, bit 1 = shared

  1183→        }
  1184→        out
  1185→    }

  1366→    /// Encode the complete module into bytes.
  1367→    pub fn encode(self) -> Vec<u8> {
  1368→        let mut module = Vec::new();
  1369→
  1370→        // Magic + version
  1371→        module.extend_from_slice(&WASM_MAGIC);
  1372→        module.extend_from_slice(&WASM_VERSION);
  1373→
  1374→        // Type section
  1375→        if !self.types.is_empty() {
  1376→            let mut section = Vec::new();
  1377→            section.extend_from_slice(&encode_unsigned_leb128(self.types.len() as u64));
  1378→            for ft in &self.types {
  1379→                section.extend_from_slice(&ft.encode());
  1380→            }
  1381→            emit_section(&mut module, SECTION_TYPE, &section);
  1382→        }
  1383→
  1384→        // Import section
  1385→        if !self.imports.is_empty() {
  1386→            let mut section = Vec::new();
  1387→            section.extend_from_slice(&encode_unsigned_leb128(self.imports.len() as u64));
  1388→            for imp in &self.imports {
  1389→                section.extend_from_slice(&encode_unsigned_leb128(imp.module.len() as u64));
  1390→                section.extend_from_slice(imp.module.as_bytes());
  1391→                section.extend_from_slice(&encode_unsigned_leb128(imp.name.len() as u64));
  1392→                section.extend_from_slice(imp.name.as_bytes());
  1393→                match &imp.kind {
  1394→                    WasmImportKind::Function { type_idx } => {
  1395→                        section.push(0x00);
  1396→                        section.extend_from_slice(&encode_unsigned_leb128(*type_idx as u64));
  1397→                    }
  1398→                    WasmImportKind::Table { elem_type, limits } => {
  1399→                        section.push(0x01);
  1400→                        section.push(*elem_type);
  1401→                        section.extend_from_slice(&limits.encode());
  1402→                    }
  1403→                    WasmImportKind::Memory { limits } => {
  1404→                        section.push(0x02);
  1405→                        section.extend_from_slice(&limits.encode());
  1406→                    }
  1407→                    WasmImportKind::Global { val_type, mutable } => {
  1408→                        section.push(0x03);
  1409→                        section.push(val_type.to_byte());
  1410→                        section.push(if *mutable { 0x01 } else { 0x00 });
  1411→                    }
  1412→                }
  1413→            }
  1414→            emit_section(&mut module, SECTION_IMPORT, &section);
  1415→        }
  1416→
  1417→        // Function section
  1418→        if !self.functions.is_empty() {
  1419→            let mut section = Vec::new();
  1420→            section.extend_from_slice(&encode_unsigned_leb128(self.functions.len() as u64));
  1421→            for &type_idx in &self.functions {
  1422→                section.extend_from_slice(&encode_unsigned_leb128(type_idx as u64));
  1423→            }
  1424→            emit_section(&mut module, SECTION_FUNCTION, &section);
  1425→        }
  1426→
  1427→        // Table section
  1428→        if !self.tables.is_empty() {
  1429→            let mut section = Vec::new();
  1430→            section.extend_from_slice(&encode_unsigned_leb128(self.tables.len() as u64));
  1431→            for (elem_type, limits) in &self.tables {
  1432→                section.push(*elem_type);
  1433→                section.extend_from_slice(&limits.encode());
  1434→            }
  1435→            emit_section(&mut module, SECTION_TABLE, &section);
  1436→        }
  1437→
  1438→        // Memory section
  1439→        if !self.memories.is_empty() {
  1440→            let mut section = Vec::new();
  1441→            section.extend_from_slice(&encode_unsigned_leb128(self.memories.len() as u64));
  1442→            for limits in &self.memories {
  1443→                section.extend_from_slice(&limits.encode());
  1444→            }
  1445→            emit_section(&mut module, SECTION_MEMORY, &section);
  1446→        }
  1447→
  1448→        // Global section
  1449→        if !self.globals.is_empty() {
  1450→            let mut section = Vec::new();
  1451→            section.extend_from_slice(&encode_unsigned_leb128(self.globals.len() as u64));
  1452→            for g in &self.globals {
  1453→                section.push(g.ty.to_byte());
  1454→                section.push(if g.mutable { 0x01 } else { 0x00 });
  1455→                // Emit init expr: <const opcode> <LEB128 value> end
  1456→                match g.ty {
  1457→                    WasmType::I32 => {
  1458→                        section.push(0x41); // i32.const
  1459→                        section.extend_from_slice(&encode_signed_leb128(g.init_value));
  1460→                    }
  1461→                    WasmType::I64 => {
  1462→                        section.push(0x42); // i64.const
  1463→                        section.extend_from_slice(&encode_signed_leb128(g.init_value));
  1464→                    }
  1465→                    WasmType::F32 => {
  1466→                        section.push(0x43); // f32.const
  1467→                        section.extend_from_slice(&(g.init_value as f32).to_le_bytes());
  1468→                    }
  1469→                    WasmType::F64 => {
  1470→                        section.push(0x44); // f64.const
  1471→                        section.extend_from_slice(&(g.init_value as f64).to_le_bytes());
  1472→                    }
  1473→                }
  1474→                section.push(0x0B); // end
  1475→            }
  1476→            emit_section(&mut module, SECTION_GLOBAL, &section);
  1477→        }
  1478→
  1479→        // Export section
  1480→        if !self.exports.is_empty() {
  1481→            let mut section = Vec::new();
  1482→            section.extend_from_slice(&encode_unsigned_leb128(self.exports.len() as u64));
  1483→            for exp in &self.exports {
  1484→                section.extend_from_slice(&encode_unsigned_leb128(exp.name.len() as u64));
  1485→                section.extend_from_slice(exp.name.as_bytes());
  1486→                section.push(exp.kind as u8);
  1487→                section.extend_from_slice(&encode_unsigned_leb128(exp.index as u64));
  1488→            }
  1489→            emit_section(&mut module, SECTION_EXPORT, &section);
  1490→        }
  1491→
  1492→        // Start section
  1493→        if let Some(func_idx) = self.start_func {
  1494→            let section = encode_unsigned_leb128(func_idx as u64);
  1495→            emit_section(&mut module, SECTION_START, &section);
  1496→        }
  1497→
  1498→        // Element section
  1499→        if !self.elements.is_empty() {
  1500→            let mut section = Vec::new();
  1501→            section.extend_from_slice(&encode_unsigned_leb128(self.elements.len() as u64));
  1502→            for elem in &self.elements {
  1503→                section.extend_from_slice(&encode_unsigned_leb128(elem.table_index as u64));
  1504→                section.extend_from_slice(&elem.offset_expr);
  1505→                section.extend_from_slice(&encode_unsigned_leb128(elem.func_indices.len() as u64));
  1506→                for &idx in &elem.func_indices {
  1507→                    section.extend_from_slice(&encode_unsigned_leb128(idx as u64));
  1508→                }
  1509→            }
  1510→            emit_section(&mut module, SECTION_ELEMENT, &section);
  1511→        }
  1512→
  1513→        // Code section
  1514→        if !self.code.is_empty() {
  1515→            let mut section = Vec::new();
  1516→            section.extend_from_slice(&encode_unsigned_leb128(self.code.len() as u64));
  1517→            for func_body in &self.code {
  1518→                // Encode the function body: locals + code + end byte
  1519→                let mut body_bytes = Vec::new();
  1520→                body_bytes
  1521→                    .extend_from_slice(&encode_unsigned_leb128(func_body.locals.len() as u64));
  1522→                for (count, ty) in &func_body.locals {
  1523→                    body_bytes.extend_from_slice(&encode_unsigned_leb128(*count as u64));
  1524→                    body_bytes.push(ty.to_byte());
  1525→                }
  1526→                body_bytes.extend_from_slice(&func_body.body);
  1527→                // Function body is size-prefixed
  1528→                let body_size = body_bytes.len() as u64;
  1529→                section.extend_from_slice(&encode_unsigned_leb128(body_size));
  1530→                section.extend_from_slice(&body_bytes);
  1531→            }
  1532→            emit_section(&mut module, SECTION_CODE, &section);
  1533→        }
  1534→
  1535→        // Data section
  1536→        if !self.data.is_empty() {
  1537→            let mut section = Vec::new();
  1538→            section.extend_from_slice(&encode_unsigned_leb128(self.data.len() as u64));
  1539→            for seg in &self.data {
  1540→                section.extend_from_slice(&encode_unsigned_leb128(seg.memory_index as u64));
  1541→                section.extend_from_slice(&seg.offset_expr);
  1542→                section.extend_from_slice(&encode_unsigned_leb128(seg.data.len() as u64));
  1543→                section.extend_from_slice(&seg.data);
  1544→            }
  1545→            emit_section(&mut module, SECTION_DATA, &section);
  1546→        }
  1547→
  1548→        module
  1549→    }
  1550→}
  1551→
  1552→/// Emit a Wasm section: section ID + LEB128 size + content.
  1553→fn emit_section(module: &mut Vec<u8>, section_id: u8, content: &[u8]) {
  1554→    module.push(section_id);
  1555→    module.extend_from_slice(&encode_unsigned_leb128(content.len() as u64));
  1556→    module.extend_from_slice(content);
  1557→}
  1558→
  1559→/// Skip `count` unsigned LEB128 values in the byte slice, advancing `offset`.
  1560→fn skip_leb128(bytes: &[u8], offset: &mut usize, count: usize) {
  1561→    for _ in 0..count {
  1562→        while *offset < bytes.len() && (bytes[*offset] & 0x80) != 0 {
  1563→            *offset += 1;
  1564→        }
  1565→        if *offset < bytes.len() {
  1566→            *offset += 1;
  1567→        }
  1568→    }
  1569→}
  1570→
  1571→/// Skip `count` signed LEB128 values in the byte slice, advancing `offset`.
  1572→fn skip_signed_leb128(bytes: &[u8], offset: &mut usize, count: usize) {
  1573→    // Signed and unsigned LEB128 have the same encoding structure for skipping
  1574→    skip_leb128(bytes, offset, count);
  1575→}

  1577→/// Convert a WasmType (from this module) to the serializable WasmValueType

  1582→        WasmType::I64 => crate::backend::WasmValueType::I64,

  1584→        WasmType::F64 => crate::backend::WasmValueType::F64,

  1590→fn backend_to_wasm_type(ty: &crate::backend::WasmValueType) -> WasmType {
  1591→    match ty {
  1592→        crate::backend::WasmValueType::I32 => WasmType::I32,
  1593→        crate::backend::WasmValueType::I64 => WasmType::I64,
  1594→        crate::backend::WasmValueType::F32 => WasmType::F32,
  1595→        crate::backend::WasmValueType::F64 => WasmType::F64,
  1596→    }
  1597→}

  1599→/// Skip a single Wasm instruction starting at `offset`, advancing past it.
  1600→///
  1601→/// This is used by `allocate_registers` to split the body bytecode into
  1602→/// per-instruction `AllocatedInstruction` entries with real opcode names.
  1603→fn skip_one_instruction(bytes: &[u8], offset: &mut usize) {
  1604→    if *offset >= bytes.len() {
  1605→        return;
  1606→    }
  1607→    let byte = bytes[*offset];
  1608→    *offset += 1;
  1609→
  1610→    match byte {
  1611→        // ── Multi-byte opcodes ────────────────────────────────────
  1612→        0xFC => {
  1613→            // Bulk memory / saturated arithmetic prefix
  1614→            skip_leb128(bytes, offset, 1); // sub-opcode
  1615→            let (subop, _) = decode_unsigned_leb128(&bytes[*offset - 1..]);
  1616→            match subop {
  1617→                0x08 => skip_leb128(bytes, offset, 2), // memory.init
  1618→                0x0A => skip_leb128(bytes, offset, 2), // memory.copy
  1619→                0x0B => skip_leb128(bytes, offset, 1), // memory.fill
  1620→                _ => {}                                // other sub-ops may have 0 or more operands
  1621→            }
  1622→        }
  1623→        0xFD => {
  1624→            // SIMD prefix
  1625→            skip_leb128(bytes, offset, 1); // sub-opcode
  1626→            let (subop, _) = decode_unsigned_leb128(&bytes[*offset - 1..]);
  1627→            if subop == 0x0C {
  1628→                // v128.const: skip 16 bytes of payload
  1629→                *offset += 16;
  1630→            }
  1631→        }
  1632→
  1633→        // ── Control flow ─────────────────────────────────────────
  1634→        0x00 => {} // unreachable
  1635→        0x01 => {} // nop
  1636→        0x02..=0x04 => {
  1637→            *offset += 1;
  1638→        } // block/loop/if + blocktype
  1639→        0x05 | 0x0B => {} // else, end
  1640→        0x0C | 0x0D => {
  1641→            skip_leb128(bytes, offset, 1);
  1642→        } // br, br_if
  1643→        0x0E => {
  1644→            // br_table: count + labels + default
  1645→            let (count, size) = decode_unsigned_leb128(&bytes[*offset..]);
  1646→            *offset += size;
  1647→            skip_leb128(bytes, offset, count as usize + 1);
  1648→        }
  1649→        0x0F => {} // return
  1650→        0x10 => {
  1651→            skip_leb128(bytes, offset, 1);
  1652→        } // call
  1653→        0x11 => {
  1654→            skip_leb128(bytes, offset, 2);
  1655→        } // call_indirect
  1656→
  1657→        // ── Parametric ───────────────────────────────────────────
  1658→        0x1A | 0x1B => {} // drop, select
  1659→
  1660→        // ── Variable ─────────────────────────────────────────────
  1661→        0x20..=0x24 => {
  1662→            skip_leb128(bytes, offset, 1);
  1663→        } // local.get/set/tee, global.get/set
  1664→
  1665→        // ── Memory ───────────────────────────────────────────────
  1666→        0x28..=0x3E => {
  1667→            skip_leb128(bytes, offset, 2);
  1668→        } // loads/stores: align + offset
  1669→        0x3F | 0x40 => {
  1670→            skip_leb128(bytes, offset, 1);
  1671→        } // memory.size, memory.grow
  1672→
  1673→        // ── Numeric ──────────────────────────────────────────────
  1674→        0x41 => {
  1675→            skip_signed_leb128(bytes, offset, 1);
  1676→        } // i32.const
  1677→        0x42 => {
  1678→            skip_signed_leb128(bytes, offset, 1);
  1679→        } // i64.const
  1680→        0x43 => {
  1681→            *offset += 4;
  1682→        } // f32.const
  1683→        0x44 => {
  1684→            *offset += 8;
  1685→        } // f64.const
  1686→
  1687→        // All other single-byte opcodes (comparisons, arithmetic, conversions)
  1688→        0x45..=0x4F
  1689→        | 0x50..=0x5A
  1690→        | 0x5B..=0x66
  1691→        | 0x67..=0x78
  1692→        | 0x79..=0x8A
  1693→        | 0x8C
  1694→        | 0x91..=0x95
  1695→        | 0x9A
  1696→        | 0x9F..=0xA3
  1697→        | 0xA7..=0xBF => {}
  1698→
  1699→        _ => {} // Unknown opcode; nothing more to skip
  1700→    }
  1701→}

  1707→/// Context for lowering an IR function to Wasm bytecode.
  1708→struct LoweringContext {
  1709→    /// Map from virtual register ID to local index.
  1710→    vreg_to_local: HashMap<u32, u32>,
  1711→    /// Map from virtual register ID to its Wasm type.
  1712→    vreg_types: HashMap<u32, WasmType>,
  1713→    /// Number of locals (including parameters).
  1714→    num_locals: u32,
  1715→    /// Local declarations (for the function body).
  1716→    locals: Vec<(u32, WasmType)>,
  1717→    /// Map from block label to its label depth index for branch targets.
  1718→    block_labels: HashMap<String, u32>,
  1719→    /// Accumulated Wasm instructions.
  1720→    instrs: Vec<WasmInstr>,
  1721→    /// Expected result types for the function (used for return type coercion).
  1722→    result_types: Vec<WasmType>,
  1723→    /// For each Call instruction (indexed by position in `instrs`),
  1724→    /// record the target function name for later resolution during
  1725→    /// `encode_program`.  The placeholder `UNRESOLVED_CALL_IDX` is emitted
  1726→    /// as the call index and must be patched once the module's function
  1727→    /// index space is known.
  1728→    call_targets: Vec<(usize, String)>,
  1729→}
  1730→
  1731→impl LoweringContext {
  1732→    fn new(result_types: Vec<WasmType>) -> Self {
  1733→        Self {
  1734→            vreg_to_local: HashMap::new(),
  1735→            vreg_types: HashMap::new(),
  1736→            num_locals: 0,
  1737→            locals: Vec::new(),
  1738→            block_labels: HashMap::new(),
  1739→            instrs: Vec::new(),
  1740→            result_types,
  1741→            call_targets: Vec::new(),
  1742→        }
  1743→    }
  1744→
  1745→    /// Allocate a local for a virtual register, returning the local index.
  1746→    fn alloc_local(&mut self, vreg_id: u32, ty: WasmType) -> u32 {
  1747→        let idx = self.num_locals;
  1748→        self.vreg_to_local.insert(vreg_id, idx);
  1749→        self.vreg_types.insert(vreg_id, ty);
  1750→        self.num_locals += 1;
  1751→        // Try to merge with an existing (count, type) entry
  1752→        if let Some(last) = self.locals.last_mut() {
  1753→            if last.1 == ty {
  1754→                last.0 += 1;
  1755→                return idx;
  1756→            }
  1757→        }
  1758→        self.locals.push((1, ty));
  1759→        idx
  1760→    }
  1761→
  1762→    /// Get the local index for a virtual register.
  1763→    fn get_local(&self, vreg_id: u32) -> Option<u32> {
  1764→        self.vreg_to_local.get(&vreg_id).copied()
  1765→    }
  1766→
  1767→    /// Emit an instruction.
  1768→    fn emit(&mut self, instr: WasmInstr) {
  1769→        self.instrs.push(instr);
  1770→    }
  1771→
  1772→    /// Push a value onto the Wasm stack from an IRValue.
  1773→    /// On Wasm32, pointers and most integer values are pushed as i32.
  1774→    /// I64 is used only when the type hint explicitly requests it (e.g., for
  1775→    /// I64Load/I64Store of 64-bit values).  Addresses are always truncated
  1776→    /// to i32 since the Wasm32 address space is 32 bits.
  1777→    fn push_value(&mut self, val: &IRValue, type_hint: Option<&WasmType>) {
  1778→        match val {
  1779→            IRValue::Immediate(v) => {
  1780→                let ty = type_hint.copied().unwrap_or(WasmType::I32);
  1781→                match ty {
  1782→                    WasmType::I64 => self.emit(WasmInstr::I64Const(*v)),
  1783→                    WasmType::F32 => self.emit(WasmInstr::F32Const(*v as f32)),
  1784→                    WasmType::F64 => self.emit(WasmInstr::F64Const(*v as f64)),
  1785→                    _ => self.emit(WasmInstr::I32Const(*v as i32)),
  1786→                }
  1787→            }
  1788→            IRValue::Register(id) => {
  1789→                // If the register hasn't been allocated yet, allocate it as i32.
  1790→                // On Wasm32, all integer locals are i32.
  1791→                if self.get_local(*id).is_none() {
  1792→                    self.alloc_local(*id, WasmType::I32);
  1793→                }
  1794→                if let Some(local_idx) = self.get_local(*id) {
  1795→                    self.emit(WasmInstr::LocalGet(local_idx));
  1796→                }
  1797→            }
  1798→            IRValue::Address(addr) => {
  1799→                self.emit(WasmInstr::I32Const(*addr as i32)); // wasm32: pointers are i32
  1800→            }
  1801→            IRValue::Label(_) => {
  1802→                // Labels are handled via block structure; not pushed as values
  1803→            }
  1804→        }
  1805→    }
  1806→
  1807→    /// Store the top of the Wasm stack into a virtual register's local.
  1808→    fn pop_to_vreg(&mut self, vreg_id: u32, ty: WasmType) {
  1809→        if !self.vreg_to_local.contains_key(&vreg_id) {
  1810→            self.alloc_local(vreg_id, ty);
  1811→        }
  1812→        if let Some(local_idx) = self.get_local(vreg_id) {
  1813→            self.emit(WasmInstr::LocalSet(local_idx));
  1814→        }
  1815→    }
  1816→}
  1817→
  1818→/// Determine the Wasm type for dedicated arithmetic IR instructions (Add, Sub,
  1819→/// Mul, Div, Cmp).  These instructions carry an optional `ty` field that
  1820→/// indicates the operand width.  On the Wasm32 target, all integer types map
  1821→/// to I32 since pointers are 32 bits and the address space is 32 bits;
  1822→/// only float types retain their original width.  This is consistent with
  1823→/// `wasm_type_for_binop` and ensures pointer arithmetic always uses i32 ops.
  1824→fn wasm_type_for_dedicated_arith(ir_ty: Option<&IRType>) -> WasmType {
  1825→    match ir_ty {
  1826→        Some(IRType::F32) => WasmType::F32,
  1827→        Some(IRType::F64) => WasmType::F64,
  1828→        _ => WasmType::I32, // all integer types → i32 on wasm32 (pointers are i32)
  1829→    }
  1830→}
  1831→
  1832→/// Determine the Wasm type for an IR BinOp based on the operand types and
  1833→/// the optional IR type annotation.  On the Wasm32 target, all integer
  1834→/// operations use i32 since the address space is 32 bits; only float types
  1835→/// retain their original width.
  1836→fn wasm_type_for_binop(
  1837→    _op: &BinOpKind,
  1838→    _lhs: &IRValue,
  1839→    _rhs: &IRValue,
  1840→    ir_ty: Option<&IRType>,
  1841→    _vreg_types: &HashMap<u32, WasmType>,
  1842→) -> WasmType {
  1843→    // If the IR provides a type, use it (but map I64/U64 to I32 for Wasm32).
  1844→    if let Some(ty) = ir_ty {
  1845→        return match ty {
  1846→            IRType::F32 => WasmType::F32,
  1847→            IRType::F64 => WasmType::F64,
  1848→            _ => WasmType::I32, // all integer types → i32 on wasm32
  1849→        };
  1850→    }
  1851→    // Default to i32 for all integer ops on wasm32
  1852→    WasmType::I32
  1853→}
  1854→
  1855→/// Infer the Wasm type of an IR value based on its representation.
  1856→///
  1857→/// On the Wasm32 target, all integer values are i32.  Only float immediates
  1858→/// use the wider type.
  1859→fn infer_wasm_type(val: &IRValue) -> WasmType {
  1860→    match val {
  1861→        IRValue::Immediate(_)
  1862→        | IRValue::Register(_)
  1863→        | IRValue::Address(_)
  1864→        | IRValue::Label(_) => WasmType::I32,
  1865→    }
  1866→}
  1867→
  1868→/// Lower an IR function to Wasm bytecode, returning the function body,
  1869→/// type, and a list of call relocations that must be patched during
  1870→/// `encode_program`.
  1871→///
  1872→/// Each relocation is `(byte_offset, func_name)` where `byte_offset` is
  1873→/// the position *within the encoded body bytes* where the LEB128 function
  1874→/// index starts (i.e., one byte past the `0x10` Call opcode).
  1875→fn lower_function(
  1876→    func: &IRFunction,
  1877→) -> Result<(WasmFuncBody, WasmFuncType, Vec<(usize, String)>), BackendError> {
  1878→    // Compute result types.
  1879→    // In Wasm32, all integer results are i32 since it's a 32-bit target.
  1880→    let result_types: Vec<WasmType> = func
  1881→        .result_types
  1882→        .iter()
  1883→        .filter_map(WasmType::from_ir_type)
  1884→        .map(|t| if t.is_integer() { WasmType::I32 } else { t })
  1885→        .collect();
  1886→    let mut ctx = LoweringContext::new(result_types);
  1887→
  1888→    // Assign locals for parameters.
  1889→    // In Wasm32, all integer parameters are i32 regardless of IR type, since
  1890→    // pointers and all integer values fit in 32 bits on a 32-bit target.
  1891→    // Only float types retain their original width.
  1892→    for (i, param) in func.params.iter().enumerate() {
  1893→        let ty = func
  1894→            .param_types
  1895→            .get(i)
  1896→            .and_then(WasmType::from_ir_type)
  1897→            .map(|t| if t.is_integer() { WasmType::I32 } else { t })
  1898→            .unwrap_or(WasmType::I32);
  1899→        if let IRValue::Register(id) = param {
  1900→            let idx = ctx.num_locals;
  1901→            ctx.vreg_to_local.insert(*id, idx);
  1902→            ctx.vreg_types.insert(*id, ty);
  1903→            ctx.num_locals += 1;
  1904→            if let Some(last) = ctx.locals.last_mut() {
  1905→                if last.1 == ty {
  1906→                    last.0 += 1;
  1907→                } else {
  1908→                    ctx.locals.push((1, ty));
  1909→                }
  1910→            } else {
  1911→                ctx.locals.push((1, ty));
  1912→            }
  1913→        }
  1914→    }
  1915→
  1916→    // Lower each block
  1917→    for (block_idx, block) in func.blocks.iter().enumerate() {
  1918→        // Emit a block for structured control flow (except the entry block)
  1919→        if block_idx > 0 {
  1920→            ctx.emit(WasmInstr::Block(None)); // void block (no result value)
  1921→            ctx.block_labels
  1922→                .insert(block.label.clone(), block_idx as u32);
  1923→        } else {
  1924→            ctx.block_labels.insert(block.label.clone(), 0);
  1925→        }
  1926→
  1927→        // Lower instructions
  1928→        for instr in &block.instructions {
  1929→            lower_instruction(instr, &mut ctx)?;
  1930→        }
  1931→
  1932→        // Lower terminator
  1933→        lower_terminator(&block.terminator, &mut ctx, block_idx)?;
  1934→
  1935→        // End the block
  1936→        if block_idx > 0 {
  1937→            ctx.emit(WasmInstr::End);
  1938→        }
  1939→    }
  1940→
  1941→    // Build the function type.
  1942→    // In Wasm32, all integer params/results are i32.
  1943→    let param_types: Vec<WasmType> = func
  1944→        .param_types
  1945→        .iter()
  1946→        .filter_map(WasmType::from_ir_type)
  1947→        .map(|t| if t.is_integer() { WasmType::I32 } else { t })
  1948→        .collect();
  1949→    let result_types: Vec<WasmType> = func
  1950→        .result_types
  1951→        .iter()
  1952→        .filter_map(WasmType::from_ir_type)
