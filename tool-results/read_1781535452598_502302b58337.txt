     1→//! # Memory Safety Verification Module
     2→//!
     3→//! Compile-time and optional runtime memory safety checks for the VUMA compiler.
     4→//!
     5→//! ## Checks Provided
     6→//!
     7→//! | #  | Check              | Code  | Stage      | Description                                          |
     8→//! |----|--------------------|-------|------------|------------------------------------------------------|
     9→//! | 1  | Use-after-free     | E041  | Compile    | Value live after deallocation via SCG liveness       |
    10→//! | 2  | Double-free        | E042  | Compile    | Same allocation freed more than once                  |
    11→//! | 3  | Memory leak        | E043  | Compile    | Heap allocation with no matching free on exit paths   |
    12→//! | 4  | Bounds check       | E044  | Runtime    | Array index out-of-bounds (enabled by `--safe`)      |
    13→//! | 5  | Null deref         | E045  | Compile    | Dereference of pointer that may be null               |
    14→//! | 6  | Dangling pointer   | E046  | Compile    | Pointer to stack allocation that escapes its scope    |
    15→//! | 7  | Uninitialized read | E047  | Compile    | Read of allocation with no reaching write             |
    16→//! | 8  | Buffer overflow    | E048  | Runtime    | Write past allocation boundary (enabled by `--safe`)  |
    17→//! | 9  | Use-after-scope    | E049  | Compile    | Access to stack variable after scope exit             |
    18→//! | 10 | Invalid free       | E050  | Compile    | Free of non-heap pointer or already-freed pointer     |
    19→//!
    20→//! ## Integration
    21→//!
    22→//! The module integrates with the SCG liveness analysis from `vuma-scg` and
    23→//! with the diagnostics system (error codes E041–E050).  The `--safe` CLI
    24→//! flag enables runtime bounds-checking instrumentation.
    25→
    26→use std::collections::{HashMap, HashSet};
    27→use std::fmt;
    28→

    31→// dead-allocation detection. We depend on it through the codegen SCG bridge.
    32→
    33→use crate::scg_to_ir::{
    34→    AccessNode, AllocationNode, CallNode, ControlNode, Scg, ScgExpr, ScgFunction, ScgNode,
    35→    ScgStatement, ScgType,
    36→};

    38→// ═══════════════════════════════════════════════════════════════════════════
    39→// Error codes for memory safety diagnostics
    40→// ═══════════════════════════════════════════════════════════════════════════
    41→
    42→/// Memory safety violation kind, mapped to E041–E050.
    43→#[derive(Debug, Clone, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
    44→pub enum MemorySafetyViolation {
    45→    /// E041 — Use-after-free: value still live after deallocation.
    46→    UseAfterFree {
    47→        /// Name of the freed allocation.
    48→        allocation_name: String,
    49→        /// Line of the deallocation.
    50→        dealloc_line: Option<u32>,
    51→        /// Number of uses after free.
    52→        violation_count: usize,
    53→    },
    54→    /// E042 — Double-free: same pointer freed twice.
    55→    DoubleFree {
    56→        /// Name of the allocation freed twice.
    57→        allocation_name: String,
    58→        /// Line of the first free.
    59→        first_free_line: Option<u32>,
    60→        /// Line of the second free.
    61→        second_free_line: Option<u32>,
    62→    },
    63→    /// E043 — Memory leak: heap allocation never freed.
    64→    MemoryLeak {
    65→        /// Name of the leaked allocation.
    66→        allocation_name: String,
    67→        /// Line of the allocation.
    68→        alloc_line: Option<u32>,
    69→        /// Size of the leaked allocation in bytes.
    70→        alloc_size: Option<u32>,
    71→    },
    72→    /// E044 — Bounds check failure (runtime): array index out of bounds.
    73→    BoundsCheckFailure {
    74→        /// Name of the array being accessed.
    75→        array_name: String,
    76→        /// Index used.
    77→        index: i64,
    78→        /// Array length.
    79→        length: u64,
    80→    },
    81→    /// E045 — Null pointer dereference.
    82→    NullDereference {
    83→        /// Name of the pointer variable.
    84→        pointer_name: String,
    85→    },
    86→    /// E046 — Dangling pointer: stack address escapes its scope.
    87→    DanglingPointer {
    88→        /// Name of the escaped pointer.
    89→        pointer_name: String,
    90→        /// Scope where the allocation was made.
    91→        scope_name: String,
    92→    },
    93→    /// E047 — Uninitialized read: reading from allocation with no reaching write.
    94→    UninitializedRead {
    95→        /// Name of the variable being read.
    96→        variable_name: String,
    97→    },
    98→    /// E048 — Buffer overflow (runtime): write past allocation boundary.
    99→    BufferOverflow {
   100→        /// Name of the target buffer.
   101→        buffer_name: String,
   102→        /// Offset of the write.
   103→        offset: u64,
   104→        /// Size of the buffer.
   105→        buffer_size: u64,
   106→    },
   107→    /// E049 — Use after scope: access to stack variable after scope exit.
   108→    UseAfterScope {
   109→        /// Name of the variable.
   110→        variable_name: String,
   111→        /// Scope where the variable was defined.
   112→        scope_name: String,
   113→    },
   114→    /// E050 — Invalid free: freeing a non-heap pointer or already-freed pointer.
   115→    InvalidFree {
   116→        /// Name of the pointer being freed.
   117→        pointer_name: String,
   118→        /// Reason for the invalid free.
   119→        reason: String,
   120→    },
   121→}
   122→
   123→impl MemorySafetyViolation {
   124→    /// Returns the diagnostic code string for this violation.
   125→    pub fn code(&self) -> &'static str {
   126→        match self {
   127→            MemorySafetyViolation::UseAfterFree { .. } => "E041",
   128→            MemorySafetyViolation::DoubleFree { .. } => "E042",
   129→            MemorySafetyViolation::MemoryLeak { .. } => "E043",
   130→            MemorySafetyViolation::BoundsCheckFailure { .. } => "E044",
   131→            MemorySafetyViolation::NullDereference { .. } => "E045",
   132→            MemorySafetyViolation::DanglingPointer { .. } => "E046",
   133→            MemorySafetyViolation::UninitializedRead { .. } => "E047",
   134→            MemorySafetyViolation::BufferOverflow { .. } => "E048",
   135→            MemorySafetyViolation::UseAfterScope { .. } => "E049",
   136→            MemorySafetyViolation::InvalidFree { .. } => "E050",
   137→        }
   138→    }
   139→
   140→    /// Returns a human-readable description of this violation.
   141→    pub fn description(&self) -> String {
   142→        match self {
   143→            MemorySafetyViolation::UseAfterFree {
   144→                allocation_name,
   145→                dealloc_line,
   146→                violation_count,
   147→            } => format!(
   148→                "use-after-free: '{}' still used after free at line {} ({} violating use(s))",
   149→                allocation_name,
   150→                dealloc_line.unwrap_or(0),
   151→                violation_count
   152→            ),
   153→            MemorySafetyViolation::DoubleFree {
   154→                allocation_name,
   155→                first_free_line,
   156→                second_free_line,
   157→            } => format!(
   158→                "double-free: '{}' freed at line {} and again at line {}",
   159→                allocation_name,
   160→                first_free_line.unwrap_or(0),
   161→                second_free_line.unwrap_or(0)
   162→            ),
   163→            MemorySafetyViolation::MemoryLeak {
   164→                allocation_name,
   165→                alloc_line,
   166→                alloc_size,
   167→            } => format!(
   168→                "memory leak: '{}' (allocated at line {}, size {} bytes) never freed",
   169→                allocation_name,
   170→                alloc_line.unwrap_or(0),
   171→                alloc_size.unwrap_or(0)
   172→            ),
   173→            MemorySafetyViolation::BoundsCheckFailure {
   174→                array_name,
   175→                index,
   176→                length,
   177→            } => format!(
   178→                "bounds check failed: index {} out of bounds for array '{}' (length {})",
   179→                index, array_name, length
   180→            ),
   181→            MemorySafetyViolation::NullDereference { pointer_name } => {
   182→                format!("null pointer dereference: '{}'", pointer_name)
   183→            }
   184→            MemorySafetyViolation::DanglingPointer {
   185→                pointer_name,
   186→                scope_name,
   187→            } => format!(
   188→                "dangling pointer: '{}' escapes scope '{}'",
   189→                pointer_name, scope_name
   190→            ),
   191→            MemorySafetyViolation::UninitializedRead { variable_name } => {
   192→                format!("uninitialized read: variable '{}' has no reaching write", variable_name)
   193→            }
   194→            MemorySafetyViolation::BufferOverflow {
   195→                buffer_name,
   196→                offset,
   197→                buffer_size,
   198→            } => format!(
   199→                "buffer overflow: write at offset {} past buffer '{}' (size {})",
   200→                offset, buffer_name, buffer_size
   201→            ),
   202→            MemorySafetyViolation::UseAfterScope {
   203→                variable_name,
   204→                scope_name,
   205→            } => format!(
   206→                "use after scope: variable '{}' used after scope '{}' exits",
   207→                variable_name, scope_name
   208→            ),
   209→            MemorySafetyViolation::InvalidFree {
   210→                pointer_name,
   211→                reason,
   212→            } => format!(
   213→                "invalid free: pointer '{}' — {}",
   214→                pointer_name, reason
   215→            ),
   216→        }
   217→    }
   218→}
   219→
   220→impl fmt::Display for MemorySafetyViolation {
   221→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   222→        write!(f, "[{}] {}", self.code(), self.description())
   223→    }
   224→}

   227→// MemorySafetyConfig
   228→// ═══════════════════════════════════════════════════════════════════════════
   229→
   230→/// Configuration for memory safety checks.
   231→#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
   232→pub struct MemorySafetyConfig {
   233→    /// Enable runtime bounds checking for array accesses.
   234→    /// When enabled, the codegen inserts bounds-check instructions before
   235→    /// every array load/store.  This corresponds to the `--safe` CLI flag.
   236→    pub runtime_bounds_checks: bool,
   237→
   238→    /// Enable use-after-free detection at compile time.
   239→    pub check_use_after_free: bool,
   240→
   241→    /// Enable double-free detection at compile time.
   242→    pub check_double_free: bool,
   243→
   244→    /// Enable memory leak detection at compile time.
   245→    pub check_memory_leaks: bool,
   246→
   247→    /// Enable uninitialized read detection at compile time.
   248→    pub check_uninitialized_reads: bool,
   249→
   250→    /// Enable dangling pointer / scope escape detection.
   251→    pub check_dangling_pointers: bool,
   252→
   253→    /// Treat memory safety violations as errors (true) or warnings (false).
   254→    pub errors_are_fatal: bool,
   255→}
   256→
   257→impl Default for MemorySafetyConfig {
   258→    fn default() -> Self {
   259→        Self {
   260→            runtime_bounds_checks: false,
   261→            check_use_after_free: true,
   262→            check_double_free: true,
   263→            check_memory_leaks: true,
   264→            check_uninitialized_reads: true,
   265→            check_dangling_pointers: true,
   266→            errors_are_fatal: true,
   267→        }
   268→    }
   269→}
   270→
   271→impl MemorySafetyConfig {
   272→    /// Configuration enabled by the `--safe` CLI flag.
   273→    /// Enables runtime bounds checks in addition to all compile-time checks.
   274→    pub fn safe_mode() -> Self {
   275→        Self {
   276→            runtime_bounds_checks: true,
   277→            check_use_after_free: true,
   278→            check_double_free: true,
   279→            check_memory_leaks: true,
   280→            check_uninitialized_reads: true,
   281→            check_dangling_pointers: true,
   282→            errors_are_fatal: true,
   283→        }
   284→    }
   285→
   286→    /// Only compile-time checks, no runtime instrumentation.
   287→    pub fn compile_time_only() -> Self {
   288→        Self {
   289→            runtime_bounds_checks: false,
   290→            ..Self::default()
   291→        }
   292→    }
   293→}

   298→
   299→/// The result of running memory safety analysis on a program.
   300→#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
   301→pub struct MemorySafetyReport {
   302→    /// All violations found during analysis.
   303→    pub violations: Vec<MemorySafetyViolation>,
   304→
   305→    /// Number of heap allocations analyzed.
   306→    pub heap_allocations_analyzed: usize,
   307→
   308→    /// Number of stack allocations analyzed.
   309→    pub stack_allocations_analyzed: usize,
   310→
   311→    /// Number of deallocations analyzed.
   312→    pub deallocations_analyzed: usize,
   313→
   314→    /// Number of access sites analyzed.
   315→    pub access_sites_analyzed: usize,
   316→
   317→    /// Whether runtime bounds checks were instrumented.
   318→    pub runtime_bounds_instrumented: bool,
   319→
   320→    /// Total analysis time in microseconds.
   321→    pub analysis_time_us: u64,
   322→}
   323→
   324→impl MemorySafetyReport {
   325→    /// Create an empty report.
   326→    pub fn empty() -> Self {
   327→        Self {
   328→            violations: Vec::new(),
   329→            heap_allocations_analyzed: 0,
   330→            stack_allocations_analyzed: 0,
   331→            deallocations_analyzed: 0,
   332→            access_sites_analyzed: 0,
   333→            runtime_bounds_instrumented: false,
   334→            analysis_time_us: 0,
   335→        }
   336→    }
   337→
   338→    /// Returns `true` if no violations were found.
   339→    pub fn is_clean(&self) -> bool {
   340→        self.violations.is_empty()
   341→    }
   342→
   343→    /// Returns the number of errors (as opposed to warnings).
   344→    pub fn error_count(&self) -> usize {
   345→        self.violations.len()
   346→    }
   347→
   348→    /// Returns violations of a specific kind, identified by code.
   349→    pub fn violations_by_code(&self, code: &str) -> Vec<&MemorySafetyViolation> {
   350→        self.violations
   351→            .iter()
   352→            .filter(|v| v.code() == code)
   353→            .collect()
   354→    }
   355→}
   356→
   357→impl fmt::Display for MemorySafetyReport {
   358→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   359→        if self.violations.is_empty() {
   360→            writeln!(
   361→                f,
   362→                "Memory safety: CLEAN ({} heap allocs, {} stack allocs analyzed)",
   363→                self.heap_allocations_analyzed, self.stack_allocations_analyzed
   364→            )
   365→        } else {
   366→            writeln!(
   367→                f,
   368→                "Memory safety: {} violation(s) found",
   369→                self.violations.len()
   370→            )?;
   371→            for v in &self.violations {
   372→                writeln!(f, "  {}", v)?;
   373→            }
   374→            Ok(())
   375→        }
   376→    }
   377→}

   380→// Allocation tracking (for the codegen SCG representation)
   381→// ═══════════════════════════════════════════════════════════════════════════

   383→/// Information about a tracked allocation within the codegen SCG.
   384→#[derive(Debug, Clone)]
   385→struct AllocationInfo {
   386→    /// Variable name of the allocation.
   387→    name: String,
   388→    /// Whether this is a heap or stack allocation.
   389→    is_heap: bool,
   390→    /// Size in bytes (if known at compile time).
   391→    size: Option<u32>,
   392→    /// Source line number (best-effort).
   393→    line: Option<u32>,
   394→    /// Set of free/deallocation operations on this allocation.
   395→    frees: Vec<FreeInfo>,
   396→    /// Set of access operations on this allocation.
   397→    accesses: Vec<AccessInfo>,
   398→    /// Whether this allocation is returned from the function (escapes).
   399→    is_returned: bool,
   400→}
   401→
   402→/// Information about a free/deallocation operation.
   403→#[derive(Debug, Clone)]
   404→struct FreeInfo {
   405→    /// Variable name being freed.
   406→    name: String,
   407→    /// Source line number.
   408→    line: Option<u32>,
   409→    /// Whether this is an explicit free or implicit (end of scope for stack).
   410→    is_explicit: bool,
   411→}
   412→
   413→/// Information about an access (load/store) to an allocation.
   414→#[derive(Debug, Clone)]
   415→struct AccessInfo {
   416→    /// Variable name being accessed.
   417→    name: String,
   418→    /// Whether this is a read or write.
   419→    is_read: bool,
   420→    /// Source line number.
   421→    line: Option<u32>,
   422→    /// Optional offset expression (for array indexing).
   423→    offset_expr: Option<String>,
   424→}

   426→// ═══════════════════════════════════════════════════════════════════════════
   427→// Memory Safety Analysis Engine
   428→// ═══════════════════════════════════════════════════════════════════════════
   429→
   430→/// The memory safety analysis engine.
   431→///
   432→/// Walks the codegen SCG representation to track allocations, frees, and
   433→/// accesses, then runs compile-time checks for use-after-free, double-free,
   434→/// memory leaks, and uninitialized reads.
   435→///
   436→/// When `runtime_bounds_checks` is enabled, it also marks array access sites
   437→/// for instrumentation with bounds-check code during codegen.
   438→pub struct MemorySafetyAnalyzer {
   439→    config: MemorySafetyConfig,
   440→}
   441→
   442→impl MemorySafetyAnalyzer {
   443→    /// Create a new analyzer with the given configuration.
   444→    pub fn new(config: MemorySafetyConfig) -> Self {
   445→        Self { config }
   446→    }
   447→
   448→    /// Create an analyzer with the default configuration.
   449→    pub fn with_defaults() -> Self {
   450→        Self::new(MemorySafetyConfig::default())
   451→    }
   452→
   453→    /// Run memory safety analysis on a codegen SCG program.
   454→    ///
   455→    /// This is the primary entry point. It walks all functions in the SCG,
   456→    /// tracks allocations and frees, then runs each enabled check.
   457→    pub fn analyze(&self, scg: &Scg) -> MemorySafetyReport {
   458→        let start = std::time::Instant::now();
   459→
   460→        let mut report = MemorySafetyReport::empty();
   461→        let mut all_allocations: HashMap<String, AllocationInfo> = HashMap::new();
   462→
   463→        for node in &scg.nodes {
   464→            match node {
   465→                ScgNode::Function(func) => {
   466→                    let func_allocs = self.analyze_function(func);
   467→                    // Check for violations within the function
   468→                    self.check_function(func, &func_allocs, &mut report);
   469→                    // Merge into global tracking
   470→                    for (name, info) in func_allocs {
   471→                        all_allocations.insert(name, info);
   472→                    }
   473→                }
   474→                ScgNode::Data(_) => {
   475→                    // Data declarations don't have memory safety issues
   476→                }
   477→            }
   478→        }
   479→
   480→        // Global leak check: heap allocations with no frees
   481→        if self.config.check_memory_leaks {
   482→            self.check_memory_leaks(&all_allocations, &mut report);
   483→        }
   484→
   485→        report.runtime_bounds_instrumented = self.config.runtime_bounds_checks;
   486→        report.analysis_time_us = start.elapsed().as_micros() as u64;
   487→        report
   488→    }
   489→
   490→    /// Analyze a single function to track allocations, frees, and accesses.
   491→    fn analyze_function(&self, func: &ScgFunction) -> HashMap<String, AllocationInfo> {
   492→        let mut allocations: HashMap<String, AllocationInfo> = HashMap::new();
   493→        self.walk_statements(&func.body, &mut allocations, None);
   494→        allocations
   495→    }
   496→
   497→    /// Recursively walk SCG statements to collect allocation/access/free info.
   498→    fn walk_statements(
   499→        &self,
   500→        stmts: &[ScgStatement],
   501→        allocations: &mut HashMap<String, AllocationInfo>,
   502→        scope_name: Option<&str>,
   503→    ) {
   504→        for stmt in stmts {
   505→            match stmt {
   506→                ScgStatement::Allocation(alloc) => {
   507→                    match alloc {
   508→                        AllocationNode::Stack { name, size, .. } => {
   509→                            allocations.insert(
   510→                                name.clone(),
   511→                                AllocationInfo {
   512→                                    name: name.clone(),
   513→                                    is_heap: false,
   514→                                    size: Some(*size),
   515→                                    line: None,
   516→                                    frees: Vec::new(),
   517→                                    accesses: Vec::new(),
   518→                                    is_returned: false,
   519→                                },
   520→                            );
   521→                        }
   522→                        AllocationNode::Heap { name, .. } => {
   523→                            allocations.insert(
   524→                                name.clone(),
   525→                                AllocationInfo {
   526→                                    name: name.clone(),
   527→                                    is_heap: true,
   528→                                    size: None, // Dynamic size
   529→                                    line: None,
   530→                                    frees: Vec::new(),
   531→                                    accesses: Vec::new(),
   532→                                    is_returned: false,
   533→                                },
   534→                            );
   535→                        }
   536→                    }
   537→                }
   538→                ScgStatement::Access(access) => {
   539→                    match access {
   540→                        AccessNode::Load { dst, ptr, offset } => {
   541→                            let ptr_name = expr_to_name(ptr);
   542→                            if let Some(info) = allocations.get_mut(&ptr_name) {
   543→                                info.accesses.push(AccessInfo {
   544→                                    name: dst.clone(),
   545→                                    is_read: true,
   546→                                    line: None,
   547→                                    offset_expr: offset.as_ref().map(|e| format!("{:?}", e)),
   548→                                });
   549→                            }
   550→                        }
   551→                        AccessNode::Store { ptr, offset, value } => {
   552→                            let ptr_name = expr_to_name(ptr);
   553→                            if let Some(info) = allocations.get_mut(&ptr_name) {
   554→                                info.accesses.push(AccessInfo {
   555→                                    name: expr_to_name(value),
   556→                                    is_read: false,
   557→                                    line: None,
   558→                                    offset_expr: offset.as_ref().map(|e| format!("{:?}", e)),
   559→                                });
   560→                            }
   561→                        }
   562→                    }
   563→                }
   564→                ScgStatement::Call(call) => {
   565→                    // Check for deallocation calls (free, __vuma_free, etc.)
   566→                    let func_name = &call.func_name;
   567→                    if is_deallocation_call(func_name) {
   568→                        for arg in &call.args {
   569→                            let arg_name = expr_to_name(arg);
   570→                            if let Some(info) = allocations.get_mut(&arg_name) {
   571→                                info.frees.push(FreeInfo {
   572→                                    name: arg_name.clone(),
   573→                                    line: None,
   574→                                    is_explicit: true,
   575→                                });
   576→                            }
   577→                        }
   578→                    }
   579→                }
   580→                ScgStatement::Control(ctrl) => {
   581→                    match ctrl {
   582→                        ControlNode::If {
   583→                            then_body,
   584→                            else_body,
   585→                            ..
   586→                        } => {
   587→                            self.walk_statements(then_body, allocations, scope_name);
   588→                            if let Some(else_body) = else_body {
   589→                                self.walk_statements(else_body, allocations, scope_name);
   590→                            }
   591→                        }
   592→                        ControlNode::Loop { body, .. } => {
   593→                            self.walk_statements(body, allocations, scope_name);
   594→                        }
   595→                        ControlNode::Switch {
   596→                            arms, default_body, ..
   597→                        } => {
   598→                            for arm in arms {
   599→                                self.walk_statements(&arm.body, allocations, scope_name);
   600→                            }
   601→                            self.walk_statements(default_body, allocations, scope_name);
   602→                        }
   603→                        ControlNode::Break | ControlNode::Continue => {}
   604→                    }
   605→                }
   606→                ScgStatement::Return(values) => {
   607→                    // Mark any returned allocations as escaping
   608→                    for val in values {
   609→                        let name = expr_to_name(val);
   610→                        if let Some(info) = allocations.get_mut(&name) {
   611→                            info.is_returned = true;
   612→                        }
   613→                    }
   614→                }
   615→                ScgStatement::Computation(_) => {}
   616→                ScgStatement::UnaryComputation(_) => {}
   617→                ScgStatement::Cast(_) => {}
   618→            }
   619→        }
   620→    }
   621→
   622→    /// Check a function for memory safety violations.
   623→    fn check_function(
   624→        &self,
   625→        func: &ScgFunction,
   626→        allocations: &HashMap<String, AllocationInfo>,
   627→        report: &mut MemorySafetyReport,
   628→    ) {
   629→        for (name, info) in allocations {
   630→            // Count stats
   631→            if info.is_heap {
   632→                report.heap_allocations_analyzed += 1;
   633→            } else {
   634→                report.stack_allocations_analyzed += 1;
   635→            }
   636→            report.deallocations_analyzed += info.frees.len();
   637→            report.access_sites_analyzed += info.accesses.len();
   638→
   639→            // ── Double-free detection ──
   640→            if self.config.check_double_free && info.frees.len() > 1 {
   641→                // Multiple frees on the same allocation
   642→                let frees = &info.frees;
   643→                for i in 1..frees.len() {
   644→                    report.violations.push(MemorySafetyViolation::DoubleFree {
   645→                        allocation_name: name.clone(),
   646→                        first_free_line: frees[0].line,
   647→                        second_free_line: frees[i].line,
   648→                    });
   649→                }
   650→            }
   651→
   652→            // ── Use-after-free detection ──
   653→            // If there are frees, check if any access occurs after a free.
   654→            // In the codegen SCG (which is statement-order-based), we do a
   655→            // simplified check: if an allocation has both frees and accesses
   656→            // that come after the free (in statement order), it's a UAF.
   657→            if self.config.check_use_after_free && !info.frees.is_empty() {
   658→                // Walk the function body to find accesses after frees
   659→                let uaf_count = self.count_accesses_after_free(func, name);
   660→                if uaf_count > 0 {
   661→                    report.violations.push(MemorySafetyViolation::UseAfterFree {
   662→                        allocation_name: name.clone(),
   663→                        dealloc_line: info.frees.first().and_then(|f| f.line),
   664→                        violation_count: uaf_count,
   665→                    });
   666→                }
   667→            }
   668→
   669→            // ── Uninitialized read detection ──
   670→            if self.config.check_uninitialized_reads {
   671→                let has_write = info.accesses.iter().any(|a| !a.is_read);
   672→                let has_read = info.accesses.iter().any(|a| a.is_read);
   673→                if has_read && !has_write && !info.is_returned {
   674→                    // Reads without any writes (and not a parameter)
   675→                    // This is a simplified check; a full reaching-definitions
   676→                    // analysis would be more precise.
   677→                    report
   678→                        .violations
   679→                        .push(MemorySafetyViolation::UninitializedRead {
   680→                            variable_name: name.clone(),
   681→                        });
   682→                }
   683→            }
   684→        }
   685→    }
   686→
   687→    /// Count accesses to an allocation that appear after a free in statement order.
   688→    ///
   689→    /// This is a simplified analysis that works on the linear statement order
   690→    /// of the codegen SCG (which represents the program's control flow in
   691→    /// statement order). For more complex control flow (if/else, loops),
   692→    /// the full SCG liveness analysis from `vuma-scg` should be used.
   693→    fn count_accesses_after_free(&self, func: &ScgFunction, alloc_name: &str) -> usize {
   694→        let mut freed = false;
   695→        let mut count = 0;
   696→        self.count_accesses_after_free_stmts(&func.body, alloc_name, &mut freed, &mut count);
   697→        count
   698→    }
   699→
   700→    fn count_accesses_after_free_stmts(
   701→        &self,
   702→        stmts: &[ScgStatement],
   703→        alloc_name: &str,
   704→        freed: &mut bool,
   705→        count: &mut usize,
   706→    ) {
   707→        for stmt in stmts {
   708→            match stmt {
   709→                ScgStatement::Call(call) => {
   710→                    if is_deallocation_call(&call.func_name) {
   711→                        for arg in &call.args {
   712→                            if expr_to_name(arg) == alloc_name {
   713→                                *freed = true;
   714→                            }
   715→                        }
   716→                    }
   717→                }
   718→                ScgStatement::Access(access) => {
   719→                    if *freed {
   720→                        let ptr_name = match access {
   721→                            AccessNode::Load { ptr, .. } => expr_to_name(ptr),
   722→                            AccessNode::Store { ptr, .. } => expr_to_name(ptr),
   723→                        };
   724→                        if ptr_name == alloc_name {
   725→                            *count += 1;
   726→                        }
   727→                    }
   728→                }
   729→                ScgStatement::Control(ctrl) => {
   730→                    match ctrl {
   731→                        ControlNode::If {
   732→                            then_body,
   733→                            else_body,
   734→                            ..
   735→                        } => {
   736→                            // Check both branches conservatively
   737→                            let mut then_freed = *freed;
   738→                            let mut else_freed = *freed;
   739→                            let mut then_count = 0usize;
   740→                            let mut else_count = 0usize;
   741→                            self.count_accesses_after_free_stmts(
   742→                                then_body,
   743→                                alloc_name,
   744→                                &mut then_freed,
   745→                                &mut then_count,
   746→                            );
   747→                            if let Some(else_body) = else_body {
   748→                                self.count_accesses_after_free_stmts(
   749→                                    else_body,
   750→                                    alloc_name,
   751→                                    &mut else_freed,
   752→                                    &mut else_count,
   753→                                );
   754→                            }
   755→                            // If freed in either branch, conservatively mark as freed
   756→                            *freed = *freed || then_freed || else_freed;
   757→                            *count += then_count + else_count;
   758→                        }
   759→                        ControlNode::Loop { body, .. } => {
   760→                            // In a loop, a free in the body may free on every iteration
   761→                            self.count_accesses_after_free_stmts(
   762→                                body,
   763→                                alloc_name,
   764→                                freed,
   765→                                count,
   766→                            );
   767→                        }
   768→                        ControlNode::Switch {
   769→                            arms, default_body, ..
   770→                        } => {
   771→                            for arm in arms {
   772→                                self.count_accesses_after_free_stmts(
   773→                                    &arm.body,
   774→                                    alloc_name,
   775→                                    freed,
   776→                                    count,
   777→                                );
   778→                            }
   779→                            self.count_accesses_after_free_stmts(
   780→                                default_body,
   781→                                alloc_name,
   782→                                freed,
   783→                                count,
   784→                            );
   785→                        }
   786→                        ControlNode::Break | ControlNode::Continue => {}
   787→                    }
   788→                }
   789→                _ => {}
   790→            }
   791→        }
   792→    }
   793→
   794→    /// Check for memory leaks: heap allocations with no matching free.
   795→    fn check_memory_leaks(
   796→        &self,
   797→        allocations: &HashMap<String, AllocationInfo>,
   798→        report: &mut MemorySafetyReport,
   799→    ) {
   800→        for (name, info) in allocations {
   801→            if info.is_heap && info.frees.is_empty() && !info.is_returned {
   802→                report.violations.push(MemorySafetyViolation::MemoryLeak {
   803→                    allocation_name: name.clone(),
   804→                    alloc_line: info.line,
   805→                    alloc_size: info.size,
   806→                });
   807→            }
   808→        }
   809→    }
   810→}

   813→// Runtime bounds-check instrumentation
   814→// ═══════════════════════════════════════════════════════════════════════════
   815→
   816→/// Represents a site where a runtime bounds check should be inserted.
   817→#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
   818→pub struct BoundsCheckSite {
   819→    /// The function containing the access.
   820→    pub function_name: String,
   821→    /// The array/pointer being accessed.
   822→    pub array_name: String,
   823→    /// The index expression being used.
   824→    pub index_expr: String,
   825→    /// The length/bounds expression (if known).
   826→    pub length_expr: Option<String>,
   827→    /// Source line (best-effort).
   828→    pub line: Option<u32>,
   829→}
   830→
   831→/// Scan a codegen SCG for array access sites that need bounds checking.
   832→///
   833→/// When `--safe` is enabled, every array load/store gets a bounds check
   834→/// that traps if the index is out of range.
   835→pub fn find_bounds_check_sites(scg: &Scg) -> Vec<BoundsCheckSite> {
   836→    let mut sites = Vec::new();
   837→
   838→    for node in &scg.nodes {
   839→        if let ScgNode::Function(func) = node {
   840→            find_bounds_check_sites_in_stmts(&func.name, &func.body, &mut sites);
   841→        }
   842→    }
   843→
   844→    sites
   845→}
   846→
   847→fn find_bounds_check_sites_in_stmts(
   848→    func_name: &str,
   849→    stmts: &[ScgStatement],
   850→    sites: &mut Vec<BoundsCheckSite>,
   851→) {
   852→    for stmt in stmts {
   853→        match stmt {
   854→            ScgStatement::Access(access) => {
   855→                match access {
   856→                    AccessNode::Load { ptr, offset, .. } => {
   857→                        if offset.is_some() {
   858→                            sites.push(BoundsCheckSite {
   859→                                function_name: func_name.to_string(),
   860→                                array_name: expr_to_name(ptr),
   861→                                index_expr: offset
   862→                                    .as_ref()
   863→                                    .map(|e| format!("{:?}", e))
   864→                                    .unwrap_or_default(),
   865→                                length_expr: None,
   866→                                line: None,
   867→                            });
   868→                        }
   869→                    }
   870→                    AccessNode::Store { ptr, offset, .. } => {
   871→                        if offset.is_some() {
   872→                            sites.push(BoundsCheckSite {
   873→                                function_name: func_name.to_string(),
   874→                                array_name: expr_to_name(ptr),
   875→                                index_expr: offset
   876→                                    .as_ref()
   877→                                    .map(|e| format!("{:?}", e))
   878→                                    .unwrap_or_default(),
   879→                                length_expr: None,
   880→                                line: None,
   881→                            });
   882→                        }
   883→                    }
   884→                }
   885→            }
   886→            ScgStatement::Control(ctrl) => match ctrl {
   887→                ControlNode::If {
   888→                    then_body,
   889→                    else_body,
   890→                    ..
   891→                } => {
   892→                    find_bounds_check_sites_in_stmts(func_name, then_body, sites);
   893→                    if let Some(else_body) = else_body {
   894→                        find_bounds_check_sites_in_stmts(func_name, else_body, sites);
   895→                    }
   896→                }
   897→                ControlNode::Loop { body, .. } => {
   898→                    find_bounds_check_sites_in_stmts(func_name, body, sites);
   899→                }
   900→                ControlNode::Switch {
   901→                    arms, default_body, ..
   902→                } => {
   903→                    for arm in arms {
   904→                        find_bounds_check_sites_in_stmts(func_name, &arm.body, sites);
   905→                    }
   906→                    find_bounds_check_sites_in_stmts(func_name, default_body, sites);
   907→                }
   908→                ControlNode::Break | ControlNode::Continue => {}
   909→            },
   910→            _ => {}
   911→        }
   912→    }
   913→}

   916→// Helper functions
   917→// ═══════════════════════════════════════════════════════════════════════════
   918→
   919→/// Check if a function name is a deallocation call.
   920→fn is_deallocation_call(name: &str) -> bool {
   921→    matches!(
   922→        name,
   923→        "free"
   924→            | "__vuma_free"
   925→            | "dealloc"
   926→            | "deallocate"
   927→            | "drop"
   928→            | "__builtin_free"
   929→    )
   930→}
   931→
   932→/// Extract a variable name from an SCG expression (best-effort).
   933→fn expr_to_name(expr: &ScgExpr) -> String {
   934→    match expr {
   935→        ScgExpr::Var(name) => name.clone(),
   936→        ScgExpr::Const(_) => "<const>".to_string(),
   937→        ScgExpr::BinOp { op, lhs, rhs } => {
   938→            format!("({:?} {} {:?})", op, expr_to_name(lhs), expr_to_name(rhs))
   939→        }
   940→        ScgExpr::UnaryOp { op, operand } => {
   941→            format!("({:?} {:?})", op, expr_to_name(operand))
   942→        }
   943→        ScgExpr::Call { func, .. } => format!("{}(...)", func),
   944→        ScgExpr::Cast { expr, .. } => format!("cast({:?})", expr_to_name(expr)),
   945→        ScgExpr::Index { array, index } => {
   946→            format!("{}[{:?}]", expr_to_name(array), expr_to_name(index))
   947→        }
   948→        ScgExpr::Field { object, field } => {
   949→            format!("{}.{}", expr_to_name(object), field)
   950→        }
   951→        ScgExpr::Deref { operand } => format!("*{:?}", expr_to_name(operand)),
   952→        ScgExpr::AddrOf { operand } => format!("&{:?}", expr_to_name(operand)),
   953→        ScgExpr::SizeOf { .. } => "<sizeof>".to_string(),
   954→    }
   955→}

   957→// ═══════════════════════════════════════════════════════════════════════════
   958→// Integration with full SCG liveness analysis (vuma-scg)
   959→// ═══════════════════════════════════════════════════════════════════════════
   960→
   961→/// Run memory safety analysis using the full SCG liveness analysis from
   962→/// `vuma-scg`. This provides more precise use-after-free and dead-allocation
   963→/// detection by leveraging the graph-based SCG rather than the simplified
   964→/// codegen SCG.
   965→///
   966→/// This function is called from the main pipeline when the full SCG is
   967→/// available (after AST → SCG conversion). The results supplement the
   968→/// codegen-level analysis.
   969→pub fn analyze_with_scg_liveness(
   970→    scg_liveness: &vuma_scg::liveness::LivenessAnalysis,
   971→    scg: &vuma_scg::graph::SCG,
   972→    config: &MemorySafetyConfig,
   973→) -> Vec<MemorySafetyViolation> {
   974→    let mut violations = Vec::new();
   975→
   976→    // Use-after-free detection via liveness analysis
   977→    if config.check_use_after_free {
   978→        let uaf_violations = vuma_scg::liveness::find_use_after_free(scg, &scg_liveness.liveness);
   979→        for uaf in &uaf_violations {
   980→            violations.push(MemorySafetyViolation::UseAfterFree {
   981→                allocation_name: format!("node_{}", uaf.allocation),
   982→                dealloc_line: None, // Could be resolved with source mapping
   983→                violation_count: uaf.violating_uses.len(),
   984→            });
   985→        }
   986→    }
   987→
   988→    // Uninitialized read detection
   989→    if config.check_uninitialized_reads {
   990→        let uninit_reads = vuma_scg::liveness::find_uninitialized_reads(scg, &scg_liveness.liveness);
   991→        for node_id in &uninit_reads {
   992→            violations.push(MemorySafetyViolation::UninitializedRead {
   993→                variable_name: format!("node_{}", node_id),
   994→            });
   995→        }
   996→    }
   997→
   998→    // Dead allocation detection (potential memory leaks)
   999→    if config.check_memory_leaks {
  1000→        let dead_allocs = vuma_scg::liveness::find_dead_allocations(scg, &scg_liveness.liveness);
  1001→        for node_id in &dead_allocs {
  1002→            violations.push(MemorySafetyViolation::MemoryLeak {
  1003→                allocation_name: format!("node_{}", node_id),
  1004→                alloc_line: None,
  1005→                alloc_size: None,
  1006→            });
  1007→        }
  1008→    }
  1009→
  1010→    violations
  1011→}
