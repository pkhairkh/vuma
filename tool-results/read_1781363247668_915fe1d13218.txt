   273тЖТ   273тЖТ
   274тЖТ   274тЖТ/// Target-specific information needed during code generation.
   275тЖТ   275тЖТ///
   276тЖТ   276тЖТ/// This trait provides a data-driven interface for query target properties.
   277тЖТ   277тЖТ/// It must be implementable for ALL supported ISAs, including fundamentally
   278тЖТ   278тЖТ/// different architectures like Wasm (stack machine, no registers) and
   279тЖТ   279тЖТ/// MIPS (branch delay slots).
   280тЖТ   280тЖТ///
   281тЖТ   281тЖТ/// # Object Safety
   282тЖТ   282тЖТ///
   283тЖТ   283тЖТ/// This trait is object-safe: all methods take `&self` and return only
   284тЖТ   284тЖТ/// owned types or references with `'static` lifetime.
   285тЖТ   285тЖТpub trait TargetInfo: Send + Sync + 'static {
   286тЖТ   286тЖТ    // === Identity ===
   287тЖТ   287тЖТ
   288тЖТ   288тЖТ    /// ISA name in lowercase (e.g., "aarch64", "riscv64", "wasm32").
   289тЖТ   289тЖТ    fn isa_name(&self) -> &'static str;
   290тЖТ   290тЖТ
   291тЖТ   291тЖТ    /// LLVM-style target triple (e.g., "aarch64-unknown-linux-gnu").
   292тЖТ   292тЖТ    fn target_triple(&self) -> &'static str;
   293тЖТ   293тЖТ
   294тЖТ   294тЖТ    /// ELF `e_machine` value.  Returns 0 for non-ELF targets (Wasm).
   295тЖТ   295тЖТ    fn elf_machine_type(&self) -> u16;
   296тЖТ   296тЖТ
   297тЖТ   297тЖТ    /// Default base address for the .text section.
   298тЖТ   298тЖТ    fn default_base_address(&self) -> u64;
   299тЖТ   299тЖТ
   300тЖТ   300тЖТ    // === Data model ===
   301тЖТ   301тЖТ
   302тЖТ   302тЖТ    /// Pointer width in bytes (4 for 32-bit, 8 for 64-bit).
   303тЖТ   303тЖТ    fn pointer_width(&self) -> usize;
   304тЖТ   304тЖТ
   305тЖТ   305тЖТ    /// Size in bytes of `ty` on this target.
   306тЖТ   306тЖТ    fn size_of(&self, ty: &IRType) -> usize;
   307тЖТ   307тЖТ
   308тЖТ   308тЖТ    /// Natural alignment in bytes of `ty` on this target.
   309тЖТ   309тЖТ    fn alignment_of(&self, ty: &IRType) -> usize;
   310тЖТ   310тЖТ
   311тЖТ   311тЖТ    /// Byte order of this target.
   312тЖТ   312тЖТ    fn endianness(&self) -> Endianness;
   313тЖТ   313тЖТ
   314тЖТ   314тЖТ    // === Register architecture ===
   315тЖТ   315тЖТ
   316тЖТ   316тЖТ    /// Whether this target has registers at all.  `false` for Wasm (stack machine).
   317тЖТ   317тЖТ    fn has_registers(&self) -> bool;
   318тЖТ   318тЖТ
   319тЖТ   319тЖТ    /// Number of general-purpose registers.  0 for Wasm.
   320тЖТ   320тЖТ    fn num_gp_regs(&self) -> usize;
   321тЖТ   321тЖТ
   322тЖТ   322тЖТ    /// Number of SIMD/FP registers.  0 for Wasm.
   323тЖТ   323тЖТ    fn num_simd_fp_regs(&self) -> usize;
   324тЖТ   324тЖТ
   325тЖТ   325тЖТ    /// Whether the ISA has a hardwired-zero register (RISC-V x0, LoongArch r0).
   326тЖТ   326тЖТ    fn has_hardwired_zero(&self) -> bool;
   327тЖТ   327тЖТ
   328тЖТ   328тЖТ    /// Whether the ISA uses a link register (ARM, RISC-V, MIPS, PPC) rather than
   329тЖТ   329тЖТ    /// pushing the return address on the stack (x86_64).
   330тЖТ   330тЖТ    fn has_link_register(&self) -> bool;
   331тЖТ   331тЖТ
   332тЖТ   332тЖТ    /// Whether branches have delay slots (MIPS only).
   333тЖТ   333тЖТ    fn has_branch_delay_slots(&self) -> bool;
   334тЖТ   334тЖТ
   335тЖТ   335тЖТ    /// Whether this ISA uses a TOC (Table of Contents) pointer (PPC64 r2).
   336тЖТ   336тЖТ    fn has_toc_pointer(&self) -> bool;
   337тЖТ   337тЖТ
   338тЖТ   338тЖТ    /// Whether this ISA has dedicated condition register fields (PPC64 CR0-CR7).
   339тЖТ   339тЖТ    fn has_condition_registers(&self) -> bool;
   340тЖТ   340тЖТ
   341тЖТ   341тЖТ    // === Calling convention ===
   342тЖТ   342тЖТ
   343тЖТ   343тЖТ    /// Name of the calling convention (e.g., "aapcs64", "lp64d", "systemv").
   344тЖТ   344тЖТ    fn calling_convention_name(&self) -> &'static str;
   345тЖТ   345тЖТ
   346тЖТ   346тЖТ    /// Number of integer argument registers.
   347тЖТ   347тЖТ    fn num_int_arg_regs(&self) -> usize;
   348тЖТ   348тЖТ
   349тЖТ   349тЖТ    /// Number of FP/SIMD argument registers.
   350тЖТ   350тЖТ    fn num_fp_arg_regs(&self) -> usize;
   351тЖТ   351тЖТ
   352тЖТ   352тЖТ    /// Required stack alignment in bytes.
   353тЖТ   353тЖТ    fn stack_alignment(&self) -> usize;
   354тЖТ   354тЖТ
   355тЖТ   355тЖТ    // === Instruction encoding ===
   356тЖТ   356тЖТ
   357тЖТ   357тЖТ    /// Alignment requirement for instructions in bytes (4 for fixed-width RISCs,
   358тЖТ   358тЖТ    /// 1 for x86_64 and Wasm).
   359тЖТ   359тЖТ    fn instruction_alignment(&self) -> usize;
   360тЖТ   360тЖТ
   361тЖТ   361тЖТ    /// Minimum and maximum instruction width in bytes.
   362тЖТ   362тЖТ    /// - Fixed-width 32-bit ISAs: (4, 4)
   363тЖТ   363тЖТ    /// - x86_64: (1, 15)
   364тЖТ   364тЖТ    /// - RISC-V with RVC: (2, 4)
   365тЖТ   365тЖТ    /// - Wasm: (1, ~) but typically (1, 15)

  1115тЖТ  1115тЖТ// ---------------------------------------------------------------------------
  1116тЖТ  1116тЖТ
  1117тЖТ  1117тЖТ/// Decode a 32-bit AArch64 instruction word into a human-readable mnemonic.
  1118тЖТ  1118тЖТ///
  1119тЖТ  1119тЖТ/// Covers the most common AArch64 instructions: ADD, SUB, MOV, LDR, STR, B,
  1120тЖТ  1120тЖТ/// BL, RET, CMP, B.cond, STP, LDP, NOP, MUL, SDIV, UDIV, AND, ORR, EOR,
  1121тЖТ  1121тЖТ/// plus several additional frequently-encountered encodings.
  1122тЖТ  1122тЖТfn decode_aarch64(word: u32) -> String {
  1123тЖТ  1123тЖТ    let rd = word & 0x1F;
  1124тЖТ  1124тЖТ    let rn = (word >> 5) & 0x1F;
  1125тЖТ  1125тЖТ    let rt = rd; // alias for load/store destination
  1126тЖТ  1126тЖТ    let rm = (word >> 16) & 0x1F;
  1127тЖТ  1127тЖТ    let imm12 = (word >> 10) & 0xFFF;
  1128тЖТ  1128тЖТ    let cond = word & 0xF;
  1129тЖТ  1129тЖТ
  1130тЖТ  1130тЖТ    // NOP: d503201f
  1131тЖТ  1131тЖТ    if word == 0xD503201F {
  1132тЖТ  1132тЖТ        return "nop".to_string();
  1133тЖТ  1133тЖТ    }
  1134тЖТ  1134тЖТ
  1135тЖТ  1135тЖТ    // RET: d65f03c0
  1136тЖТ  1136тЖТ    if word == 0xD65F03C0 {
  1137тЖТ  1137тЖТ        return "ret".to_string();
  1138тЖТ  1138тЖТ    }
  1139тЖТ  1139тЖТ
  1140тЖТ  1140тЖТ    let _top8 = word >> 24;
  1141тЖТ  1141тЖТ    let _top10 = word >> 22;
  1142тЖТ  1142тЖТ
  1143тЖТ  1143тЖТ    // --- ADD/SUB (immediate): 100100xx ...
  1144тЖТ  1144тЖТ    if (word >> 23) & 0x1FF == 0b1_0010_0010 {
  1145тЖТ  1145тЖТ        // ADD Xd, Xn, #imm12
  1146тЖТ  1146тЖТ        return format!("add x{}, x{}, #{}", rd, rn, imm12);
  1147тЖТ  1147тЖТ    }
  1148тЖТ  1148тЖТ    if (word >> 23) & 0x1FF == 0b1_1010_0010 {
  1149тЖТ  1149тЖТ        // SUB Xd, Xn, #imm12
  1150тЖТ  1150тЖТ        return format!("sub x{}, x{}, #{}", rd, rn, imm12);
  1151тЖТ  1151тЖТ    }
  1152тЖТ  1152тЖТ
  1153тЖТ  1153тЖТ    // --- ADD (shifted register): 1_00_0101_1_xxx ...
  1154тЖТ  1154тЖТ    if (word >> 24) & 0xFF == 0b1000_1011 {
  1155тЖТ  1155тЖТ        // ADD Xd, Xn, Xm
  1156тЖТ  1156тЖТ        return format!("add x{}, x{}, x{}", rd, rn, rm);
  1157тЖТ  1157тЖТ    }
  1158тЖТ  1158тЖТ
  1159тЖТ  1159тЖТ    // --- SUB (shifted register): 1_00_0101_1_xxx ... with S=1 at bit30
  1160тЖТ  1160тЖТ    if (word >> 24) & 0xFF == 0b1101_0110 {
  1161тЖТ  1161тЖТ        // SUB Xd, Xn, Xm (bit 30 set = sub)
  1162тЖТ  1162тЖТ        return format!("sub x{}, x{}, x{}", rd, rn, rm);
  1163тЖТ  1163тЖТ    }
  1164тЖТ  1164тЖТ
  1165тЖТ  1165тЖТ    // --- AND (shifted register): 1_00_0101_0_00_xxx
  1166тЖТ  1166тЖТ    if (word >> 24) & 0xFE == 0b1000_1010 {
  1167тЖТ  1167тЖТ        // Check bit 21-15: opcode[31:21] = 10001010_000
  1168тЖТ  1168тЖТ        if (word >> 21) & 0x7FF == 0b10001010000 {
  1169тЖТ  1169тЖТ            return format!("and x{}, x{}, x{}", rd, rn, rm);
  1170тЖТ  1170тЖТ        }
  1171тЖТ  1171тЖТ    }
  1172тЖТ  1172тЖТ
  1173тЖТ  1173тЖТ    // --- ORR (shifted register): 1_01_0101_0_00_xxx
  1174тЖТ  1174тЖТ    if (word >> 21) & 0x7FF == 0b10101010000 {
  1175тЖТ  1175тЖТ        return format!("orr x{}, x{}, x{}", rd, rn, rm);
  1176тЖТ  1176тЖТ    }
  1177тЖТ  1177тЖТ
  1178тЖТ  1178тЖТ    // --- EOR (shifted register): 1_10_0101_0_00_xxx
  1179тЖТ  1179тЖТ    if (word >> 21) & 0x7FF == 0b11001010000 {
  1180тЖТ  1180тЖТ        return format!("eor x{}, x{}, x{}", rd, rn, rm);
  1181тЖТ  1181тЖТ    }
  1182тЖТ  1182тЖТ
  1183тЖТ  1183тЖТ    // --- MOV (register): alias for ORR Xd, XZR, Xm
  1184тЖТ  1184тЖТ    // ORR Xd, XZR, Xm: 10101010_000_rm_000000_xzr_rd
  1185тЖТ  1185тЖТ    // More general: ORR with Rn=XZR(31)
  1186тЖТ  1186тЖТ    if (word >> 21) & 0x7FF == 0b10101010000 && rn == 31 {
  1187тЖТ  1187тЖТ        return format!("mov x{}, x{}", rd, rm);
  1188тЖТ  1188тЖТ    }
  1189тЖТ  1189тЖТ
  1190тЖТ  1190тЖТ    // --- MUL: MADD Xd, Xn, Xm, XZR
  1191тЖТ  1191тЖТ    // Encoding: 1_00_1101_1_000_Rm_0_01111_Rn_Rd
  1192тЖТ  1192тЖТ    if (word >> 21) & 0x7FF == 0b10011011000 && ((word >> 10) & 0x1F) == 0b01111 {
  1193тЖТ  1193тЖТ        return format!("mul x{}, x{}, x{}", rd, rn, rm);
  1194тЖТ  1194тЖТ    }
  1195тЖТ  1195тЖТ
  1196тЖТ  1196тЖТ    // --- SDIV: 1_00_1101_1_0100_00000_00001_Rn_Rd  (actually 1_00_1101_0100_Rm_00001_Rn_Rd)
  1197тЖТ  1197тЖТ    if (word >> 21) & 0x7FF == 0b10011010100 && (word >> 10) & 0x1F == 0b00001 {
  1198тЖТ  1198тЖТ        return format!("sdiv x{}, x{}, x{}", rd, rn, rm);
  1199тЖТ  1199тЖТ    }
  1200тЖТ  1200тЖТ
  1201тЖТ  1201тЖТ    // --- UDIV: 1_00_1101_0000_Rm_00001_Rn_Rd
  1202тЖТ  1202тЖТ    if (word >> 21) & 0x7FF == 0b10011010000 && (word >> 10) & 0x1F == 0b00001 {
  1203тЖТ  1203тЖТ        return format!("udiv x{}, x{}, x{}", rd, rn, rm);
  1204тЖТ  1204тЖТ    }
  1205тЖТ  1205тЖТ
  1206тЖТ  1206тЖТ    // --- CMP (immediate): SUBS XZR, Xn, #imm12
  1207тЖТ  1207тЖТ    // 11100001_00_xxx_xxx_xxx_xxx_xxx_11111_xxx
  1208тЖТ  1208тЖТ    if (word >> 23) & 0x1FF == 0b1_1110_0010 && rd == 31 {
  1209тЖТ  1209тЖТ        return format!("cmp x{}, #{}", rn, imm12);
  1210тЖТ  1210тЖТ    }
  1211тЖТ  1211тЖТ
  1212тЖТ  1212тЖТ    // --- CMP (register): SUBS XZR, Xn, Xm
  1213тЖТ  1213тЖТ    if (word >> 21) & 0x7FF == 0b11101011000 && rd == 31 {
  1214тЖТ  1214тЖТ        return format!("cmp x{}, x{}", rn, rm);
  1215тЖТ  1215тЖТ    }
  1216тЖТ  1216тЖТ
  1217тЖТ  1217тЖТ    // --- B.cond: 0101010x xxxxxxxxxx xxxxxx cond
  1218тЖТ  1218тЖТ    if (word >> 24) & 0xFF == 0x54 {
  1219тЖТ  1219тЖТ        let cond_name = match cond {
  1220тЖТ  1220тЖТ            0 => "eq",
  1221тЖТ  1221тЖТ            1 => "ne",
  1222тЖТ  1222тЖТ            2 => "cs",
  1223тЖТ  1223тЖТ            3 => "cc",
  1224тЖТ  1224тЖТ            4 => "mi",
  1225тЖТ  1225тЖТ            5 => "pl",
  1226тЖТ  1226тЖТ            6 => "vs",
  1227тЖТ  1227тЖТ            7 => "vc",
  1228тЖТ  1228тЖТ            8 => "hi",
  1229тЖТ  1229тЖТ            9 => "ls",
  1230тЖТ  1230тЖТ            10 => "ge",
  1231тЖТ  1231тЖТ            11 => "lt",
  1232тЖТ  1232тЖТ            12 => "gt",
  1233тЖТ  1233тЖТ            13 => "le",
  1234тЖТ  1234тЖТ            14 => "al",
  1235тЖТ  1235тЖТ            _ => "??",
  1236тЖТ  1236тЖТ        };
  1237тЖТ  1237тЖТ        let imm19 = (word >> 5) & 0x7FFFF;
  1238тЖТ  1238тЖТ        let offset = ((imm19 as i32) << 13) >> 11; // sign-extend and *4
  1239тЖТ  1239тЖТ        return format!("b.{} {:+}", cond_name, offset);
  1240тЖТ  1240тЖТ    }
  1241тЖТ  1241тЖТ
  1242тЖТ  1242тЖТ    // --- B (unconditional): 000101xx xxxxxxxxxxxxxxxxxxxx
  1243тЖТ  1243тЖТ    if (word >> 26) & 0x3F == 0b000101 {
  1244тЖТ  1244тЖТ        let imm26 = word & 0x3FFFFFF;
  1245тЖТ  1245тЖТ        let offset = ((imm26 as i32) << 6) >> 4; // sign-extend and *4
  1246тЖТ  1246тЖТ        return format!("b {:+}", offset);
  1247тЖТ  1247тЖТ    }
  1248тЖТ  1248тЖТ
  1249тЖТ  1249тЖТ    // --- BL: 100101xx xxxxxxxxxxxxxxxxxxxx
  1250тЖТ  1250тЖТ    if (word >> 26) & 0x3F == 0b100101 {
  1251тЖТ  1251тЖТ        let imm26 = word & 0x3FFFFFF;
  1252тЖТ  1252тЖТ        let offset = ((imm26 as i32) << 6) >> 4;
  1253тЖТ  1253тЖТ        return format!("bl {:+}", offset);
  1254тЖТ  1254тЖТ    }
  1255тЖТ  1255тЖТ
  1256тЖТ  1256тЖТ    // --- LDR (unsigned offset): 11111001_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1257тЖТ  1257тЖТ    if (word >> 22) & 0x3FF == 0b1111100101 {
  1258тЖТ  1258тЖТ        let imm12_raw = (word >> 10) & 0xFFF;
  1259тЖТ  1259тЖТ        let offset = imm12_raw * 8; // scale by 8 for 64-bit
  1260тЖТ  1260тЖТ        return format!("ldr x{}, [x{}, #{}]", rt, rn, offset);
  1261тЖТ  1261тЖТ    }
  1262тЖТ  1262тЖТ
  1263тЖТ  1263тЖТ    // --- STR (unsigned offset): 11111000_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1264тЖТ  1264тЖТ    if (word >> 22) & 0x3FF == 0b1111100001 {
  1265тЖТ  1265тЖТ        let imm12_raw = (word >> 10) & 0xFFF;
  1266тЖТ  1266тЖТ        let offset = imm12_raw * 8;
  1267тЖТ  1267тЖТ        return format!("str x{}, [x{}, #{}]", rt, rn, offset);
  1268тЖТ  1268тЖТ    }
  1269тЖТ  1269тЖТ
  1270тЖТ  1270тЖТ    // --- LDP (signed offset, 64-bit): 101_0100_110_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1271тЖТ  1271тЖТ    if (word >> 22) & 0x3FF == 0b1010100110 {
  1272тЖТ  1272тЖТ        let rt2 = (word >> 10) & 0x1F;
  1273тЖТ  1273тЖТ        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1274тЖТ  1274тЖТ        let offset = imm7 * 8;
  1275тЖТ  1275тЖТ        return format!("ldp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1276тЖТ  1276тЖТ    }
  1277тЖТ  1277тЖТ
  1278тЖТ  1278тЖТ    // --- STP (signed offset, 64-bit): 101_0100_010_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1279тЖТ  1279тЖТ    if (word >> 22) & 0x3FF == 0b1010100010 {
  1280тЖТ  1280тЖТ        let rt2 = (word >> 10) & 0x1F;
  1281тЖТ  1281тЖТ        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1282тЖТ  1282тЖТ        let offset = imm7 * 8;
  1283тЖТ  1283тЖТ        return format!("stp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1284тЖТ  1284тЖТ    }
  1285тЖТ  1285тЖТ
  1286тЖТ  1286тЖТ    // --- MOVZ: 110100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1287тЖТ  1287тЖТ    if (word >> 23) & 0x1FF == 0b110100101 {
  1288тЖТ  1288тЖТ        let hw = (word >> 21) & 0x3;
  1289тЖТ  1289тЖТ        let imm16 = (word >> 5) & 0xFFFF;
  1290тЖТ  1290тЖТ        return format!("movz x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1291тЖТ  1291тЖТ    }
  1292тЖТ  1292тЖТ
  1293тЖТ  1293тЖТ    // --- MOVK: 111100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1294тЖТ  1294тЖТ    if (word >> 23) & 0x1FF == 0b111100101 {
  1295тЖТ  1295тЖТ        let hw = (word >> 21) & 0x3;
  1296тЖТ  1296тЖТ        let imm16 = (word >> 5) & 0xFFFF;
  1297тЖТ  1297тЖТ        return format!("movk x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1298тЖТ  1298тЖТ    }
  1299тЖТ  1299тЖТ
  1300тЖТ  1300тЖТ    format!(".word {:08x}", word)
  1301тЖТ  1301тЖТ}

  1330тЖТ  1330тЖТ/// Compute the stack frame size for an IR function.
  1331тЖТ  1331тЖТ///
  1332тЖТ  1332тЖТ/// Replicates the private `compute_frame_size` function in `emit.rs`:
  1333тЖТ  1333тЖТ/// sums `Alloc` instruction sizes and rounds up to 16-byte alignment.
  1334тЖТ  1334тЖТ/// NOTE: Does NOT include 16 bytes for FP/LR pair (handled by prologue separately).
  1335тЖТ  1335тЖТfn aarch64_compute_frame_size(func: &IRFunction) -> usize {
  1336тЖТ  1336тЖТ    let mut total: u32 = 0; // Alloc sizes only; FP/LR handled separately
  1337тЖТ  1337тЖТ    for block in &func.blocks {
  1338тЖТ  1338тЖТ        for instr in &block.instructions {
  1339тЖТ  1339тЖТ            if let IRInstr::Alloc { size, .. } = instr {
  1340тЖТ  1340тЖТ                let aligned = (*size).div_ceil(16) * 16;
  1341тЖТ  1341тЖТ                total += aligned;
  1342тЖТ  1342тЖТ            }
  1343тЖТ  1343тЖТ        }
  1344тЖТ  1344тЖТ    }
  1345тЖТ  1345тЖТ    // Round up to 16-byte alignment
  1346тЖТ  1346тЖТ    total = (total + 15) & !15;
  1347тЖТ  1347тЖТ    total as usize
  1348тЖТ  1348тЖТ}

  1350тЖТ  1350тЖТ/// Build an ELF64 binary for AArch64 Linux with 2 LOAD segments.
  1351тЖТ  1351тЖТ///
  1352тЖТ  1352тЖТ/// Segment 1: PF_R | PF_X тАФ contains .text (code)
  1353тЖТ  1353тЖТ/// Segment 2: PF_R | PF_W тАФ contains .data + stack space (writable)
  1354тЖТ  1354тЖТ///
  1355тЖТ  1355тЖТ/// The two segments are separated by page alignment (4KB) to ensure
  1356тЖТ  1356тЖТ/// the kernel maps them with different permissions. Without this,
  1357тЖТ  1357тЖТ/// a single PF_R|PF_W|PF_X segment is insecure and may cause
  1358тЖТ  1358тЖТ/// QEMU/Linux to reject the executable.
  1359тЖТ  1359тЖТfn build_aarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1360тЖТ  1360тЖТ    const PAGE_SIZE: u64 = 0x1000; // 4 KB
  1361тЖТ  1361тЖТ
  1362тЖТ  1362тЖТ    let elf_header_size: u64 = 64;
  1363тЖТ  1363тЖТ    let phdr_size: u64 = 56;
  1364тЖТ  1364тЖТ    let num_phdrs: u64 = 2;
  1365тЖТ  1365тЖТ    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1366тЖТ  1366тЖТ    // Page-align the text segment start in the file.  The kernel's ELF
  1367тЖТ  1367тЖТ    // loader mmap()s each LOAD segment, and the file offset must be
  1368тЖТ  1368тЖТ    // congruent with the virtual address modulo the page size.  The
  1369тЖТ  1369тЖТ    // simplest way to guarantee this is to place the text at a
  1370тЖТ  1370тЖТ    // page-aligned file offset, with vaddr = base_addr.
  1371тЖТ  1371тЖТ    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1372тЖТ  1372тЖТ    let text_size = code.len() as u64;
  1373тЖТ  1373тЖТ
  1374тЖТ  1374тЖТ    // The data segment starts on the next page after the text.
  1375тЖТ  1375тЖТ    let text_file_end = text_offset + text_size;
  1376тЖТ  1376тЖТ    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1377тЖТ  1377тЖТ    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1378тЖТ  1378тЖТ    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1379тЖТ  1379тЖТ    let entry_point = base_addr + text_offset;
  1380тЖТ  1380тЖТ
  1381тЖТ  1381тЖТ    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1382тЖТ  1382тЖТ
  1383тЖТ  1383тЖТ    // --- e_ident ---
  1384тЖТ  1384тЖТ    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1385тЖТ  1385тЖТ    elf.push(2); // ELFCLASS64
  1386тЖТ  1386тЖТ    elf.push(1); // ELFDATA2LSB
  1387тЖТ  1387тЖТ    elf.push(1); // EV_CURRENT
  1388тЖТ  1388тЖТ    elf.push(3); // ELFOSABI_LINUX
  1389тЖТ  1389тЖТ    elf.push(0); // padding
  1390тЖТ  1390тЖТ    elf.extend_from_slice(&[0u8; 7]); // padding
  1391тЖТ  1391тЖТ
  1392тЖТ  1392тЖТ    // --- ELF header fields ---
  1393тЖТ  1393тЖТ    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1394тЖТ  1394тЖТ    elf.extend_from_slice(&183u16.to_le_bytes()); // e_machine = EM_AARCH64
  1395тЖТ  1395тЖТ    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1396тЖТ  1396тЖТ    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1397тЖТ  1397тЖТ    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff

  1399тЖТ  1399тЖТ    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1400тЖТ  1400тЖТ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1401тЖТ  1401тЖТ    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1402тЖТ  1402тЖТ    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1403тЖТ  1403тЖТ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1404тЖТ  1404тЖТ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1405тЖТ  1405тЖТ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1406тЖТ  1406тЖТ
  1407тЖТ  1407тЖТ    // --- Program Header 1: LOAD (PF_R | PF_X) тАФ .text ---
  1408тЖТ  1408тЖТ    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1409тЖТ  1409тЖТ    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X

  1412тЖТ  1412тЖТ    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr

  1414тЖТ  1414тЖТ    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz

  1416тЖТ  1416тЖТ
  1417тЖТ  1417тЖТ    // --- Program Header 2: LOAD (PF_R | PF_W) тАФ .data / stack ---
  1418тЖТ  1418тЖТ    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1419тЖТ  1419тЖТ    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1420тЖТ  1420тЖТ    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1421тЖТ  1421тЖТ    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1422тЖТ  1422тЖТ    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1423тЖТ  1423тЖТ    elf.extend_from_slice(&0u64.to_le_bytes()); // p_filesz (no initialized data)
  1424тЖТ  1424тЖТ    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1425тЖТ  1425тЖТ    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1426тЖТ  1426тЖТ
  1427тЖТ  1427тЖТ    // --- .text section ---
  1428тЖТ  1428тЖТ    // Pad to page-aligned text_offset
  1429тЖТ  1429тЖТ    while (elf.len() as u64) < text_offset {
  1430тЖТ  1430тЖТ        elf.push(0);
  1431тЖТ  1431тЖТ    }
  1432тЖТ  1432тЖТ    elf.extend_from_slice(code);
  1433тЖТ  1433тЖТ
  1434тЖТ  1434тЖТ    // --- Pad to data segment offset (if needed) ---
  1435тЖТ  1435тЖТ    while (elf.len() as u64) < data_offset {
  1436тЖТ  1436тЖТ        elf.push(0);
  1437тЖТ  1437тЖТ    }
  1438тЖТ  1438тЖТ
  1439тЖТ  1439тЖТ    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1440тЖТ  1440тЖТ
  1441тЖТ  1441тЖТ    elf
  1442тЖТ  1442тЖТ}

  1444тЖТ  1444тЖТ/// Build ARM64 runtime I/O functions using Linux SVC syscalls.
  1445тЖТ  1445тЖТ///
  1446тЖТ  1446тЖТ/// Provides:
  1447тЖТ  1447тЖТ/// - `__vuma_print_hex`: Print X0 as 8 hex digits to stdout (FD=1)
  1448тЖТ  1448тЖТ///   Uses sys_write (X8=64) via SVC #0.
  1449тЖТ  1449тЖТ///   Saves/restores X1,X2,X3,X8,X9,X10 on stack.
  1450тЖТ  1450тЖТ///
  1451тЖТ  1451тЖТ/// - `__vuma_print_int`: Print X0 as a decimal integer to stdout (FD=1)
  1452тЖТ  1452тЖТ///   Converts digit-by-digit into a stack buffer, then sys_write.
  1453тЖТ  1453тЖТ///
  1454тЖТ  1454тЖТ/// - `__vuma_print_newline`: Print a newline character to stdout.
  1455тЖТ  1455тЖТ///
  1456тЖТ  1456тЖТ/// All functions follow AAPCS64: X0 is the argument, X1-X7 are caller-saved,
  1457тЖТ  1457тЖТ/// X8 is the indirect result register / syscall number, X19-X28 are callee-saved.
  1458тЖТ  1458тЖТfn build_aarch64_runtime() -> Vec<u8> {
  1459тЖТ  1459тЖТ    let mut code = Vec::new();
  1460тЖТ  1460тЖТ
  1461тЖТ  1461тЖТ    // тФАтФА __vuma_print_hex тФАтФА
  1462тЖТ  1462тЖТ    // Input: X0 = 64-bit value to print as 8 hex digits
  1463тЖТ  1463тЖТ    // Clobbers: X1, X2, X3, X8, X9, X10
  1464тЖТ  1464тЖТ    // Stack frame: 48 bytes (8 for pair save + 8 for hex chars + 32 padding)
  1465тЖТ  1465тЖТ
  1466тЖТ  1466тЖТ    // Prologue: save callee-saved and link register
  1467тЖТ  1467тЖТ    // STP X29, X30, [SP, #-48]!  (pre-indexed, 48-byte frame)
  1468тЖТ  1468тЖТ    code.extend_from_slice(&0xA9BF7BFDu32.to_le_bytes());
  1469тЖТ  1469тЖТ    // ADD X29, SP, #0
  1470тЖТ  1470тЖТ    code.extend_from_slice(&0x910003FDu32.to_le_bytes());
  1471тЖТ  1471тЖТ
  1472тЖТ  1472тЖТ    // Save X9, X10 on stack at [SP, #16]
  1473тЖТ  1473тЖТ    // STP X9, X10, [SP, #16]
  1474тЖТ  1474тЖТ    code.extend_from_slice(&0xA90127E9u32.to_le_bytes());
  1475тЖТ  1475тЖТ
  1476тЖТ  1476тЖТ    // Save X1, X2 on stack at [SP, #32]
  1477тЖТ  1477тЖТ    // STP X1, X2, [SP, #32]
  1478тЖТ  1478тЖТ    code.extend_from_slice(&0xA9020441u32.to_le_bytes());
  1479тЖТ  1479тЖТ
  1480тЖТ  1480тЖТ    // Save X3, X8 on stack at [SP, #40]  (offset 40 = 5*8)
  1481тЖТ  1481тЖТ    // Wait, 40 = 5*8 but STP needs offset/8 for signed-offset. 40/8 = 5.
  1482тЖТ  1482тЖТ    // STP X3, X8, [SP, #40]
  1483тЖТ  1483тЖТ    code.extend_from_slice(&0xA9052C63u32.to_le_bytes());
  1484тЖТ  1484тЖТ
  1485тЖТ  1485тЖТ    // Hex lookup table: we'll use ADD + LDRB pattern.
  1486тЖТ  1486тЖТ    // Instead, compute hex digit directly:
  1487тЖТ  1487тЖТ    // For each nibble: digit = nibble; if digit < 10 then char = digit + '0' else char = digit - 10 + 'a'
  1488тЖТ  1488тЖТ
  1489тЖТ  1489тЖТ    // We'll write 8 hex chars starting at SP+16 (reuse save area after we load saves)
  1490тЖТ  1490тЖТ    // Actually, let's use SP+16 as the char buffer (8 bytes).
  1491тЖТ  1491тЖТ    // First, save the important regs, then overwrite the buffer area.
  1492тЖТ  1492тЖТ
  1493тЖТ  1493тЖТ    // Process 8 nibbles from most significant to least significant
  1494тЖТ  1494тЖТ    // X0 = value, iterate 8 times shifting right by 28, 24, 20, ..., 4, 0
  1495тЖТ  1495тЖТ
  1496тЖТ  1496тЖТ    // X9 = char buffer pointer = SP
  1497тЖТ  1497тЖТ    // X10 = loop counter (0..8)
  1498тЖТ  1498тЖТ
  1499тЖТ  1499тЖТ    // ADD X9, SP, #16  (buffer starts at SP+16)
  1500тЖТ  1500тЖТ    code.extend_from_slice(&0x910043E9u32.to_le_bytes());
  1501тЖТ  1501тЖТ    // MOV X10, #0
  1502тЖТ  1502тЖТ    code.extend_from_slice(&0xD280001Au32.to_le_bytes());
  1503тЖТ  1503тЖТ    // MOV X3, #0  (bit shift amount, starts at 28)
  1504тЖТ  1504тЖТ    code.extend_from_slice(&0xD280003Du32.to_le_bytes()); // MOVZ X3, #28>>4... wait
  1505тЖТ  1505тЖТ    // MOV X3, #28
  1506тЖТ  1506тЖТ    code.extend_from_slice(&0xD2800383u32.to_le_bytes()); // MOVZ X3, #28
  1507тЖТ  1507тЖТ
  1508тЖТ  1508тЖТ    // Loop label:
  1509тЖТ  1509тЖТ    let loop_offset = code.len();
  1510тЖТ  1510тЖТ
  1511тЖТ  1511тЖТ    // Extract nibble: X2 = (X0 >> X3) & 0xF
  1512тЖТ  1512тЖТ    // LSR X2, X0, X3
  1513тЖТ  1513тЖТ    code.extend_from_slice(&0x9AC02482u32.to_le_bytes()); // LSR X2, X0, X3
  1514тЖТ  1514тЖТ    // AND X2, X2, #0xF
  1515тЖТ  1515тЖТ    code.extend_from_slice(&0x92400C42u32.to_le_bytes()); // AND X2, X2, #15
  1516тЖТ  1516тЖТ
  1517тЖТ  1517тЖТ    // Convert nibble to hex char:
  1518тЖТ  1518тЖТ    // if X2 < 10: char = X2 + 48 ('0')
  1519тЖТ  1519тЖТ    // else: char = X2 - 10 + 97 ('a')
  1520тЖТ  1520тЖТ    // CMP X2, #10
  1521тЖТ  1521тЖТ    code.extend_from_slice(&0xF100141Fu32.to_le_bytes()); // CMP X2, #10? No...
  1522тЖТ  1522тЖТ    // CMP X2, #10: use ADD X1, X2, #0 then CMP... Actually:
  1523тЖТ  1523тЖТ    // SUB X1, X2, #10 тЖТ if < 10, result is negative (borrow set)
  1524тЖТ  1524тЖТ    // Let's use: CMP X2, #9; B.GT hex_alpha
  1525тЖТ  1525тЖТ    // CMP X2, #9
  1526тЖТ  1526тЖТ    code.extend_from_slice(&0xF100129Fu32.to_le_bytes()); // CMP X2, #9
  1527тЖТ  1527тЖТ
  1528тЖТ  1528тЖТ    // ADD X1, X2, #48  (default: '0' + digit)
  1529тЖТ  1529тЖТ    code.extend_from_slice(&0x91000C41u32.to_le_bytes()); // ADD X1, X2, #48
  1530тЖТ  1530тЖТ
  1531тЖТ  1531тЖТ    // B.GT hex_alpha (placeholder, will be +1 instruction)
  1532тЖТ  1532тЖТ    code.extend_from_slice(&0x54000069u32.to_le_bytes()); // B.GT +4*2 = +8 bytes
  1533тЖТ  1533тЖТ
  1534тЖТ  1534тЖТ    // B store_char
  1535тЖТ  1535тЖТ    code.extend_from_slice(&0x14000002u32.to_le_bytes()); // B +2 instructions
  1536тЖТ  1536тЖТ
  1537тЖТ  1537тЖТ    // hex_alpha: ADD X1, X2, #87  (87 = 97-10, so digit-10+'a')
  1538тЖТ  1538тЖТ    code.extend_from_slice(&0x91002AC1u32.to_le_bytes()); // ADD X1, X2, #87
  1539тЖТ  1539тЖТ
  1540тЖТ  1540тЖТ    // store_char: STRB W1, [X9, X10]
  1541тЖТ  1541тЖТ    // STRB W1, [X9, X10] тАФ register offset
  1542тЖТ  1542тЖТ    code.extend_from_slice(&0x382A6821u32.to_le_bytes()); // STRB W1, [X9, X10]
  1543тЖТ  1543тЖТ
  1544тЖТ  1544тЖТ    // Increment: SUB X3, X3, #4; ADD X10, X10, #1; CMP X10, #8; B.LT loop
  1545тЖТ  1545тЖТ    code.extend_from_slice(&0xD1000863u32.to_le_bytes()); // SUB X3, X3, #4
  1546тЖТ  1546тЖТ    code.extend_from_slice(&0x9100054Au32.to_le_bytes()); // ADD X10, X10, #1
  1547тЖТ  1547тЖТ    code.extend_from_slice(&0xF100151Fu32.to_le_bytes()); // CMP X10, #8
  1548тЖТ  1548тЖТ
  1549тЖТ  1549тЖТ    // Calculate branch back to loop start
  1550тЖТ  1550тЖТ    let current_pos = code.len() as i64;
  1551тЖТ  1551тЖТ    let loop_start = loop_offset as i64;
  1552тЖТ  1552тЖТ    let back_offset = (loop_start - current_pos) / 4 - 1; // B.LT is relative to PC+4
  1553тЖТ  1553тЖТ    let bcond = 0x5400000Bu32 | ((back_offset as u32) & 0x7FFFF); // B.LT
  1554тЖТ  1554тЖТ    code.extend_from_slice(&bcond.to_le_bytes());
  1555тЖТ  1555тЖТ
  1556тЖТ  1556тЖТ    // тФАтФА sys_write(1, buf, 8) тФАтФА
  1557тЖТ  1557тЖТ    // MOV X0, #1  (stdout fd)
  1558тЖТ  1558тЖТ    code.extend_from_slice(&0xD2800020u32.to_le_bytes()); // MOVZ X0, #1
  1559тЖТ  1559тЖТ    // ADD X1, SP, #16  (buffer pointer)
  1560тЖТ  1560тЖТ    code.extend_from_slice(&0x910043E1u32.to_le_bytes());
  1561тЖТ  1561тЖТ    // MOV X2, #8  (length)
  1562тЖТ  1562тЖТ    code.extend_from_slice(&0xD2800102u32.to_le_bytes()); // MOVZ X2, #8
  1563тЖТ  1563тЖТ    // MOV X8, #64  (sys_write)
  1564тЖТ  1564тЖТ    code.extend_from_slice(&0xD2800808u32.to_le_bytes()); // MOVZ X8, #64
  1565тЖТ  1565тЖТ    // SVC #0
  1566тЖТ  1566тЖТ    code.extend_from_slice(&0xD4000001u32.to_le_bytes());
  1567тЖТ  1567тЖТ
  1568тЖТ  1568тЖТ    // Restore registers
  1569тЖТ  1569тЖТ    // LDP X9, X10, [SP, #16]
  1570тЖТ  1570тЖТ    code.extend_from_slice(&0xA94127E9u32.to_le_bytes());
  1571тЖТ  1571тЖТ    // LDP X1, X2, [SP, #32]
  1572тЖТ  1572тЖТ    code.extend_from_slice(&0xA9420441u32.to_le_bytes());
  1573тЖТ  1573тЖТ    // LDP X3, X8, [SP, #40]
  1574тЖТ  1574тЖТ    code.extend_from_slice(&0xA9452C63u32.to_le_bytes());
  1575тЖТ  1575тЖТ    // LDP X29, X30, [SP], #48  (post-indexed load)
  1576тЖТ  1576тЖТ    code.extend_from_slice(&0xA8C37BFDu32.to_le_bytes());
  1577тЖТ  1577тЖТ
  1578тЖТ  1578тЖТ    // RET
  1579тЖТ  1579тЖТ    code.extend_from_slice(&0xD65F03C0u32.to_le_bytes());
  1580тЖТ  1580тЖТ
  1581тЖТ  1581тЖТ    // тФАтФА __vuma_print_int тФАтФА
  1582тЖТ  1582тЖТ    // Input: X0 = 64-bit signed integer to print as decimal
  1583тЖТ  1583тЖТ    // Strategy: Divide by 10 repeatedly, push digits onto stack, then write.
  1584тЖТ  1584тЖТ    // X9 = digit count, X10 = buffer pointer (SP-based)
  1585тЖТ  1585тЖТ    // Stack frame: 64 bytes (16 for save pair + 48 for digit buffer)
  1586тЖТ  1586тЖТ
  1587тЖТ  1587тЖТ    // STP X29, X30, [SP, #-64]!
  1588тЖТ  1588тЖТ    code.extend_from_slice(&0xA9C27BFDu32.to_le_bytes());
  1589тЖТ  1589тЖТ    // ADD X29, SP, #0
  1590тЖТ  1590тЖТ    code.extend_from_slice(&0x910003FDu32.to_le_bytes());
  1591тЖТ  1591тЖТ    // STP X9, X10, [SP, #16]
  1592тЖТ  1592тЖТ    code.extend_from_slice(&0xA90127E9u32.to_le_bytes());
  1593тЖТ  1593тЖТ    // STP X1, X2, [SP, #32]
  1594тЖТ  1594тЖТ    code.extend_from_slice(&0xA9020441u32.to_le_bytes());
  1595тЖТ  1595тЖТ    // STP X3, X8, [SP, #48]
  1596тЖТ  1596тЖТ    code.extend_from_slice(&0xA9030463u32.to_le_bytes());
  1597тЖТ  1597тЖТ
  1598тЖТ  1598тЖТ    // Handle negative numbers
  1599тЖТ  1599тЖТ    // CMP X0, #0
  1600тЖТ  1600тЖТ    code.extend_from_slice(&0xB100001Fu32.to_le_bytes());
  1601тЖТ  1601тЖТ    // B.GE positive
  1602тЖТ  1602тЖТ    code.extend_from_slice(&0x5400002Au32.to_le_bytes()); // B.GE +4
  1603тЖТ  1603тЖТ
  1604тЖТ  1604тЖТ    // Print '-' for negative
  1605тЖТ  1605тЖТ    // MOV X1, #45  ('-')
  1606тЖТ  1606тЖТ    code.extend_from_slice(&0xD28005A1u32.to_le_bytes()); // MOVZ X1, #45
  1607тЖТ  1607тЖТ    // STRB W1, [SP, #16]
  1608тЖТ  1608тЖТ    code.extend_from_slice(&0x390013E1u32.to_le_bytes());
  1609тЖТ  1609тЖТ    // sys_write(1, SP+16, 1)
  1610тЖТ  1610тЖТ    code.extend_from_slice(&0xD2800020u32.to_le_bytes()); // MOV X0, #1
  1611тЖТ  1611тЖТ    code.extend_from_slice(&0x910043E1u32.to_le_bytes()); // ADD X1, SP, #16
  1612тЖТ  1612тЖТ    code.extend_from_slice(&0xD2800022u32.to_le_bytes()); // MOV X2, #1
  1613тЖТ  1613тЖТ    code.extend_from_slice(&0xD2800808u32.to_le_bytes()); // MOV X8, #64
  1614тЖТ  1614тЖТ    code.extend_from_slice(&0xD4000001u32.to_le_bytes()); // SVC #0
  1615тЖТ  1615тЖТ
  1616тЖТ  1616тЖТ    // Negate X0
  1617тЖТ  1617тЖТ    // NEG X0, X0 = SUB X0, XZR, X0
  1618тЖТ  1618тЖТ    code.extend_from_slice(&0xCB0003E0u32.to_le_bytes()); // SUB X0, XZR, X0
  1619тЖТ  1619тЖТ
  1620тЖТ  1620тЖТ    // positive: convert digits
  1621тЖТ  1621тЖТ    // X9 = digit count = 0
  1622тЖТ  1622тЖТ    code.extend_from_slice(&0xD2800019u32.to_le_bytes()); // MOVZ X9, #0
  1623тЖТ  1623тЖТ    // X10 = buffer pointer = SP + 17 (leave room for potential '-')
  1624тЖТ  1624тЖТ    code.extend_from_slice(&0x910047EAu32.to_le_bytes()); // ADD X10, SP, #17
  1625тЖТ  1625тЖТ
  1626тЖТ  1626тЖТ    // div_loop: UDIV X2, X0, #10; MSUB X1, X2, #10, X0; add '0'; push; X0 = X2
  1627тЖТ  1627тЖТ    let div_loop_start = code.len();
  1628тЖТ  1628тЖТ
  1629тЖТ  1629тЖТ    // CBZ X0, done_digits
  1630тЖТ  1630тЖТ    code.extend_from_slice(&0xB4000080u32.to_le_bytes()); // CBZ X0, done_digits (+16 bytes = 4 instr)
  1631тЖТ  1631тЖТ
  1632тЖТ  1632тЖТ    // UDIV X2, X0, #10  тЖТ but UDIV takes 2 regs, no immediate.
  1633тЖТ  1633тЖТ    // MOV X1, #10
  1634тЖТ  1634тЖТ    code.extend_from_slice(&0xD2800141u32.to_le_bytes()); // MOVZ X1, #10
  1635тЖТ  1635тЖТ    // UDIV X2, X0, X1
