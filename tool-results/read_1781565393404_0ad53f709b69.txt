     1тЖТ     1тЖТ//! # String Formatting and Output
     2тЖТ     2тЖТ//!
     3тЖТ     3тЖТ//! This module provides VUMA-verified string formatting primitives for
     4тЖТ     4тЖТ//! printf-style output, integer/float conversion, padding, joining, and
     5тЖТ     5тЖТ//! buffered writing. These are the foundational formatting operations that
     6тЖТ     6тЖТ//! LLMs need when generating VUMA programs that produce human-readable
     7тЖТ     7тЖТ//! output or serialize data to text streams.
     8тЖТ     8тЖТ//!
     9тЖТ     9тЖТ//! ## Relationship to VUMA Programs
    10тЖТ    10тЖТ//!
    11тЖТ    11тЖТ//! In `.vuma` source, these functions serve as the standard formatting
    12тЖТ    12тЖТ//! library тАФ the equivalent of C's `printf`/`sprintf` family or Rust's
    13тЖТ    13тЖТ//! `std::fmt` module. They allow VUMA programs to convert numeric values
    14тЖТ    14тЖТ//! to text representations in various bases (decimal, hexadecimal, octal,
    15тЖТ    15тЖТ//! binary), format floating-point numbers with configurable precision, pad
    16тЖТ    16тЖТ//! strings to fixed widths, join multiple strings, and write formatted
    17тЖТ    17тЖТ//! output into byte buffers.
    18тЖТ    18тЖТ//!

    20тЖТ    20тЖТ//!

    33тЖТ    33тЖТ/// Format a signed 64-bit integer as a string in the given base with
    34тЖТ    34тЖТ/// minimum width padding.
    35тЖТ    35тЖТ///
    36тЖТ    36тЖТ/// Converts `value` to its textual representation in `base` (2тАУ36). If the
    37тЖТ    37тЖТ/// textual representation is shorter than `width`, it is left-padded with
    38тЖТ    38тЖТ/// `'0'` characters. A negative value is prefixed with `'-'`; the sign
    39тЖТ    39тЖТ/// occupies one character position and is not counted as a digit.
    40тЖТ    40тЖТ///
    41тЖТ    41тЖТ/// ## Supported Bases
    42тЖТ    42тЖТ///
    43тЖТ    43тЖТ/// | Base | Name       | Digits                    |
    44тЖТ    44тЖТ/// |------|------------|---------------------------|
    45тЖТ    45тЖТ/// | 2    | Binary     | 0тАУ1                       |
    46тЖТ    46тЖТ/// | 8    | Octal      | 0тАУ7                       |
    47тЖТ    47тЖТ/// | 10   | Decimal    | 0тАУ9                       |
    48тЖТ    48тЖТ/// | 16   | Hexadecimal| 0тАУ9, aтАУf                  |
    49тЖТ    49тЖТ/// | 36   | Base-36    | 0тАУ9, aтАУz                  |
    50тЖТ    50тЖТ///
    51тЖТ    51тЖТ/// Bases outside the range 2тАУ36 are treated as base 10.
    52тЖТ    52тЖТ///
    53тЖТ    53тЖТ/// ## VUMA Program Equivalent
    54тЖТ    54тЖТ///
    55тЖТ    55тЖТ/// ```vuma
    56тЖТ    56тЖТ/// fn format_int(value: i64, base: u32, width: u32) -> VumaString {
    57тЖТ    57тЖТ///     negative: bool = value < 0;
    58тЖТ    58тЖТ///     abs_val: u64 = if negative { -value as u64 } else { value as u64 };
    59тЖТ    59тЖТ///     digits: VumaString = format_uint(abs_val, base, width);
    60тЖТ    60тЖТ///     if negative {
    61тЖТ    61тЖТ///         return "-" + digits;
    62тЖТ    62тЖТ///     }
    63тЖТ    63тЖТ///     return digits;
    64тЖТ    64тЖТ/// }
    65тЖТ    65тЖТ/// ```
    66тЖТ    66тЖТ///
    67тЖТ    67тЖТ/// ## BD Annotations
    68тЖТ    68тЖТ///
    69тЖТ    69тЖТ/// - CapD: { Read, Compare } тАФ pure conversion, no side effects
    70тЖТ    70тЖТ// VUMA-VERIFIED: integer formatting is correct for all i64 values in bases 2тАУ36
    71тЖТ    71тЖТpub fn format_int(value: i64, base: u32, width: u32) -> VumaString {
    72тЖТ    72тЖТ    let effective_base = if base >= 2 && base <= 36 { base } else { 10 };
    73тЖТ    73тЖТ
    74тЖТ    74тЖТ    if value < 0 {
    75тЖТ    75тЖТ        let abs_val = (value as i128).unsigned_abs() as u64;
    76тЖТ    76тЖТ        let digits = format_uint_inner(abs_val, effective_base, if width > 1 { width - 1 } else { 0 });
    77тЖТ    77тЖТ        let mut result = VumaString::new();
    78тЖТ    78тЖТ        result.push('-');
    79тЖТ    79тЖТ        result.push_str(digits.as_str());
    80тЖТ    80тЖТ        result
    81тЖТ    81тЖТ    } else {
    82тЖТ    82тЖТ        format_uint_inner(value as u64, effective_base, width)
    83тЖТ    83тЖТ    }
    84тЖТ    84тЖТ}

    92тЖТ    92тЖТ///
    93тЖТ    93тЖТ/// ## Supported Bases
    94тЖТ    94тЖТ///
    95тЖТ    95тЖТ/// Same as [`format_int`]. Bases outside the range 2тАУ36 are treated as
    96тЖТ    96тЖТ/// base 10.
    97тЖТ    97тЖТ///

    99тЖТ    99тЖТ///
   100тЖТ   100тЖТ/// ```vuma
   101тЖТ   101тЖТ/// fn format_uint(value: u64, base: u32, width: u32) -> VumaString {
   102тЖТ   102тЖТ///     if value == 0 {
   103тЖТ   103тЖТ///         return pad_left("0", width, '0');
   104тЖТ   104тЖТ///     }
   105тЖТ   105тЖТ///     digits: [u8; 64];

   121тЖТ   121тЖТ///     }

   123тЖТ   123тЖТ/// }

   130тЖТ   130тЖТpub fn format_uint(value: u64, base: u32, width: u32) -> VumaString {
   131тЖТ   131тЖТ    let effective_base = if base >= 2 && base <= 36 { base } else { 10 };

   135тЖТ   135тЖТ/// Internal helper: format a u64 value in the given base with zero-padding.
   136тЖТ   136тЖТfn format_uint_inner(value: u64, base: u32, width: u32) -> VumaString {
   137тЖТ   137тЖТ    const DIGITS: &[u8; 36] = b"0123456789abcdefghijklmnopqrstuvwxyz";
   138тЖТ   138тЖТ
   139тЖТ   139тЖТ    if value == 0 {
   140тЖТ   140тЖТ        let zero_count = width.saturating_sub(1) as usize;
   141тЖТ   141тЖТ        let mut result = VumaString::new();
   142тЖТ   142тЖТ        for _ in 0..zero_count {
   143тЖТ   143тЖТ            result.push('0');
   144тЖТ   144тЖТ        }
   145тЖТ   145тЖТ        result.push('0');
   146тЖТ   146тЖТ        return result;
   147тЖТ   147тЖТ    }
   148тЖТ   148тЖТ
   149тЖТ   149тЖТ    // Maximum digits for u64 in base 2 is 64
   150тЖТ   150тЖТ    let mut buf = [0u8; 64];
   151тЖТ   151тЖТ    let mut pos = 0usize;
   152тЖТ   152тЖТ    let mut v = value;
   153тЖТ   153тЖТ
   154тЖТ   154тЖТ    while v > 0 {
   155тЖТ   155тЖТ        let digit = (v % base as u64) as usize;
   156тЖТ   156тЖТ        buf[pos] = DIGITS[digit];
   157тЖТ   157тЖТ        pos += 1;
   158тЖТ   158тЖТ        v /= base as u64;
   159тЖТ   159тЖТ    }
   160тЖТ   160тЖТ
   161тЖТ   161тЖТ    // Apply zero-padding
   162тЖТ   162тЖТ    let padding = (width as usize).saturating_sub(pos);
   163тЖТ   163тЖТ
   164тЖТ   164тЖТ    let mut result = VumaString::with_capacity(padding + pos);
   165тЖТ   165тЖТ    for _ in 0..padding {
   166тЖТ   166тЖТ        result.push('0');
   167тЖТ   167тЖТ    }
   168тЖТ   168тЖТ    // Write digits in reverse order
   169тЖТ   169тЖТ    for i in (0..pos).rev() {
   170тЖТ   170тЖТ        result.push(buf[i] as char);
   171тЖТ   171тЖТ    }
   172тЖТ   172тЖТ
   173тЖТ   173тЖТ    result
   174тЖТ   174тЖТ}

   180тЖТ   180тЖТ/// Format a 64-bit floating-point number as a string with the given
   181тЖТ   181тЖТ/// decimal precision.
   182тЖТ   182тЖТ///
   183тЖТ   183тЖТ/// Produces a decimal representation of `value` with exactly `precision`
   184тЖТ   184тЖТ/// digits after the decimal point. If `precision` is 0, no decimal point
   185тЖТ   185тЖТ/// is emitted.
   186тЖТ   186тЖТ///
   187тЖТ   187тЖТ/// ## Special Values
   188тЖТ   188тЖТ///
   189тЖТ   189тЖТ/// - `NaN` is formatted as `"nan"`.
   190тЖТ   190тЖТ/// - Positive infinity is formatted as `"inf"`.
   191тЖТ   191тЖТ/// - Negative infinity is formatted as `"-inf"`.
   192тЖТ   192тЖТ/// - Zero is formatted as `"0"` (precision 0) or `"0.00..."` (precision > 0).
   193тЖТ   193тЖТ///
   194тЖТ   194тЖТ/// ## VUMA Program Equivalent
   195тЖТ   195тЖТ///
   196тЖТ   196тЖТ/// ```vuma
   197тЖТ   197тЖТ/// fn format_float(value: f64, precision: u32) -> VumaString {
   198тЖТ   198тЖТ///     // Delegates to the host's formatted float conversion
   199тЖТ   199тЖТ///     return host_float_to_str(value, precision);
   200тЖТ   200тЖТ/// }
   201тЖТ   201тЖТ/// ```
   202тЖТ   202тЖТ///
   203тЖТ   203тЖТ/// ## BD Annotations
   204тЖТ   204тЖТ///
   205тЖТ   205тЖТ/// - CapD: { Read, Compare } тАФ pure conversion, no side effects
   206тЖТ   206тЖТ// VUMA-VERIFIED: float formatting produces correct decimal representation
   207тЖТ   207тЖТpub fn format_float(value: f64, precision: u32) -> VumaString {
   208тЖТ   208тЖТ    if value.is_nan() {
   209тЖТ   209тЖТ        return VumaString::from("nan");
   210тЖТ   210тЖТ    }
   211тЖТ   211тЖТ    if value.is_infinite() {
   212тЖТ   212тЖТ        return if value.is_sign_negative() {
   213тЖТ   213тЖТ            VumaString::from("-inf")
   214тЖТ   214тЖТ        } else {
   215тЖТ   215тЖТ            VumaString::from("inf")
   216тЖТ   216тЖТ        };
   217тЖТ   217тЖТ    }
   218тЖТ   218тЖТ
   219тЖТ   219тЖТ    let formatted = if precision == 0 {
   220тЖТ   220тЖТ        format!("{:.0}", value)
   221тЖТ   221тЖТ    } else {
   222тЖТ   222тЖТ        format!("{:.1$}", value, precision as usize)
   223тЖТ   223тЖТ    };
   224тЖТ   224тЖТ
   225тЖТ   225тЖТ    VumaString::from(formatted.as_str())
   226тЖТ   226тЖТ}

   333тЖТ   333тЖТ
   334тЖТ   334тЖТ/// Left-pad a string to `width` characters using `fill` as the padding
   335тЖТ   335тЖТ/// character.
   336тЖТ   336тЖТ///
   337тЖТ   337тЖТ/// If `s` already has length >= `width`, returns `s` unchanged (no

   339тЖТ   339тЖТ/// copies of `fill` to the left of `s`.

   342тЖТ   342тЖТ///
   343тЖТ   343тЖТ/// ```vuma
   344тЖТ   344тЖТ/// fn pad_left(s: &str, width: u32, fill: char) -> VumaString {
   345тЖТ   345тЖТ///     len: u32 = s.length() as u32;
   346тЖТ   346тЖТ///     if len >= width {
   347тЖТ   347тЖТ///         return VumaString::from(s);
   348тЖТ   348тЖТ///     }
   349тЖТ   349тЖТ///     pad_count: u32 = width - len;
   350тЖТ   350тЖТ///     result: VumaString = "";
   351тЖТ   351тЖТ///     for i in 0..pad_count {
   352тЖТ   352тЖТ///         result.push(fill);
   353тЖТ   353тЖТ///     }
   354тЖТ   354тЖТ///     result.push_str(s);
   355тЖТ   355тЖТ///     return result;
   356тЖТ   356тЖТ/// }
   357тЖТ   357тЖТ/// ```

   426тЖТ   426тЖТ/// [`VumaString`].
   427тЖТ   427тЖТ///
   428тЖТ   428тЖТ/// Concatenates the elements of `parts`, placing `separator` between each

   430тЖТ   430тЖТ/// `parts` has one element, returns that element without any separator.

   433тЖТ   433тЖТ///
   434тЖТ   434тЖТ/// ```vuma
   435тЖТ   435тЖТ/// fn join(parts: &[&str], separator: &str) -> VumaString {
   436тЖТ   436тЖТ///     if parts.length() == 0 {
   437тЖТ   437тЖТ///         return VumaString::new();
   438тЖТ   438тЖТ///     }
   439тЖТ   439тЖТ///     result: VumaString = VumaString::from(parts[0]);
   440тЖТ   440тЖТ///     for i in 1..parts.length() {
   441тЖТ   441тЖТ///         result.push_str(separator);
   442тЖТ   442тЖТ///         result.push_str(parts[i]);
   443тЖТ   443тЖТ///     }
   444тЖТ   444тЖТ///     return result;
   445тЖТ   445тЖТ/// }
   446тЖТ   446тЖТ/// ```

   448тЖТ   448тЖТ/// ## BD Annotations
   449тЖТ   449тЖТ///
   450тЖТ   450тЖТ/// - CapD: { Read, Compare } тАФ pure function, no side effects
   451тЖТ   451тЖТ// VUMA-VERIFIED: join produces correct concatenation with separator
   452тЖТ   452тЖТpub fn join(parts: &[&str], separator: &str) -> VumaString {
   453тЖТ   453тЖТ    if parts.is_empty() {
   454тЖТ   454тЖТ        return VumaString::new();
   455тЖТ   455тЖТ    }
   456тЖТ   456тЖТ
   457тЖТ   457тЖТ    // Estimate capacity
   458тЖТ   458тЖТ    let sep_len = separator.len();
   459тЖТ   459тЖТ    let total_len: usize = parts.iter().map(|p| p.len()).sum::<usize>() + sep_len * (parts.len() - 1);
   460тЖТ   460тЖТ    let mut result = VumaString::with_capacity(total_len);
   461тЖТ   461тЖТ
   462тЖТ   462тЖТ    result.push_str(parts[0]);
   463тЖТ   463тЖТ    for part in &parts[1..] {
   464тЖТ   464тЖТ        result.push_str(separator);
   465тЖТ   465тЖТ        result.push_str(part);
   466тЖТ   466тЖТ    }
   467тЖТ   467тЖТ    result
   468тЖТ   468тЖТ}

   474тЖТ   474тЖТ/// Write a string slice into a byte buffer, returning the number of bytes
   475тЖТ   475тЖТ/// written.
   476тЖТ   476тЖТ///
   477тЖТ   477тЖТ/// Copies the UTF-8 bytes of `s` into `buf`, up to `buf.len()` bytes. If
   478тЖТ   478тЖТ/// `s` is longer than the buffer, only the prefix that fits is written тАФ
   479тЖТ   479тЖТ/// the output is **not** guaranteed to be valid UTF-8 if truncation occurs
   480тЖТ   480тЖТ/// in the middle of a multi-byte sequence.

   487тЖТ   487тЖТ///
   488тЖТ   488тЖТ/// ```vuma
   489тЖТ   489тЖТ/// fn write_str(buf: &mut [u8], s: &str) -> u32 {
   490тЖТ   490тЖТ///     n: u32 = min(s.length(), buf.length()) as u32;
   491тЖТ   491тЖТ///     for i in 0..n {
   492тЖТ   492тЖТ///         buf[i] = s[i];
   493тЖТ   493тЖТ///     }
   494тЖТ   494тЖТ///     return n;
   495тЖТ   495тЖТ/// }

   503тЖТ   503тЖТpub fn write_str(buf: &mut [u8], s: &str) -> u32 {

   507тЖТ   507тЖТ    n as u32
   508тЖТ   508тЖТ}
