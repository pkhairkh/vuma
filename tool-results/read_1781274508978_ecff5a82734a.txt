     1→//! # LoongArch 64-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the LoongArch 64-bit target (LP64 ABI).
     4→//! This module provides:
     5→//!
     6→//! - `Gpr` — General-purpose register enum (r0–r31)
     7→//! - `Fpr` — Floating-point register enum (f0–f31)
     8→//! - `Instruction` — LoongArch64 instruction enum with correct encoding
     9→//! - Encoding helpers for all 9 instruction formats (2R, 3R, 4R, 2RI8, 2RI12,
    10→//!   2RI14, 2RI16, 1RI21, I26)
    11→//! - `LoongArch64Backend` — `Backend` implementation that lowers IR to LoongArch64
    12→//!   machine code and emits ELF64 binaries
    13→//!
    14→//! ## LoongArch64 Register Convention (LP64 ABI)
    15→//!
    16→//! | Register(s) | ABI Name | Role                              |
    17→//! |-------------|----------|-----------------------------------|
    18→//! | r0          | zero     | Hardwired zero                    |
    19→//! | r1          | ra       | Return address                    |
    20→//! | r2          | tp       | Thread pointer                    |
    21→//! | r3          | sp       | Stack pointer                     |
    22→//! | r4–r11      | a0–a7    | Argument / return registers       |
    23→//! | r12–r20     | t0–t8    | Caller-saved temporaries          |
    24→//! | r21         | —        | Reserved                          |
    25→//! | r22         | fp       | Frame pointer (callee-saved)      |
    26→//! | r23–r31     | s0–s8    | Callee-saved                      |
    27→//!
    28→//! ## LoongArch64 FP Register Convention (LP64 ABI)
    29→//!
    30→//! | Register(s) | ABI Name | Role                              |
    31→//! |-------------|----------|-----------------------------------|
    32→//! | f0–f7       | fa0–fa7  | FP argument / return registers    |
    33→//! | f8–f23      | ft0–ft15 | Caller-saved temporaries          |
    34→//! | f24–f31     | fs0–fs7  | Callee-saved                      |
    35→//!
    36→//! ## Instruction Formats
    37→//!
    38→//! All instructions are 32 bits, little-endian, with no branch delay slots.
    39→//! Nine formats: 2R, 3R, 4R, 2RI8, 2RI12, 2RI14, 2RI16, 1RI21, I26.
    40→//!
    41→//! ## References
    42→//!
    43→//! - LoongArch Reference Manual, Volume 1: Basic Architecture
    44→//! - <https://loongson.github.io/LoongArch-Documentation/>
    45→

    60→    R0 = 0,

    63→    Sp = 3,
    64→    A0 = 4,
    65→    A1 = 5,

    94→impl Gpr {
    95→    /// Returns the 5-bit encoding index for this register.
    96→    pub fn encoding(&self) -> u32 {
    97→        *self as u32
    98→    }
    99→
   100→    /// Returns `true` if this register is available for register allocation.
   101→    ///
   102→    /// R0 (zero), Ra, Tp, Sp, and Fp are reserved.
   103→    pub fn is_allocatable(&self) -> bool {
   104→        !matches!(self, Gpr::R0 | Gpr::Ra | Gpr::Tp | Gpr::Sp | Gpr::Fp)
   105→    }
   106→
   107→    /// Returns `true` if this register is callee-saved (fp, s0–s8).
   108→    pub fn is_callee_saved(&self) -> bool {
   109→        matches!(
   110→            self,
   111→            Gpr::Fp
   112→                | Gpr::S0
   113→                | Gpr::S1
   114→                | Gpr::S2
   115→                | Gpr::S3
   116→                | Gpr::S4
   117→                | Gpr::S5
   118→                | Gpr::S6
   119→                | Gpr::S7
   120→                | Gpr::S8
   121→        )
   122→    }
   123→
   124→    /// Returns `true` if this register is an argument register (a0–a7).
   125→    pub fn is_arg_reg(&self) -> bool {
   126→        matches!(
   127→            self,
   128→            Gpr::A0 | Gpr::A1 | Gpr::A2 | Gpr::A3 | Gpr::A4 | Gpr::A5 | Gpr::A6 | Gpr::A7
   129→        )
   130→    }

   142→            Gpr::A3 => "$a3",
   143→            Gpr::A4 => "$a4",
   144→            Gpr::A5 => "$a5",
   145→            Gpr::A6 => "$a6",
   146→            Gpr::A7 => "$a7",
   147→            Gpr::T0 => "$t0",
   148→            Gpr::T1 => "$t1",
   149→            Gpr::T2 => "$t2",
   150→            Gpr::T3 => "$t3",
   151→            Gpr::T4 => "$t4",
   152→            Gpr::T5 => "$t5",
   153→            Gpr::T6 => "$t6",
   154→            Gpr::T7 => "$t7",
   155→            Gpr::T8 => "$t8",
   156→            Gpr::R21 => "$r21",
   157→            Gpr::Fp => "$fp",
   158→            Gpr::S0 => "$s0",
   159→            Gpr::S1 => "$s1",
   160→            Gpr::S2 => "$s2",
   161→            Gpr::S3 => "$s3",
   162→            Gpr::S4 => "$s4",
   163→            Gpr::S5 => "$s5",

   165→            Gpr::S7 => "$s7",
   166→            Gpr::S8 => "$s8",

   170→    /// Returns the Gpr for a given argument index (0–7). Returns `None` for
   171→    /// indices >= 8.
   172→    pub fn arg_register(index: usize) -> Option<Gpr> {
   173→        match index {
   174→            0 => Some(Gpr::A0),
   175→            1 => Some(Gpr::A1),
   176→            2 => Some(Gpr::A2),
   177→            3 => Some(Gpr::A3),

   181→            7 => Some(Gpr::A7),
   182→            _ => None,
   183→        }
   184→    }
   185→}

   322→// ===========================================================================
   323→// Instruction Encoding Helpers
   324→// ===========================================================================
   325→
   326→/// Encode a 2R format instruction.
   327→///
   328→/// Format: `opcode[31:10] | rj[9:5] | rd[4:0]`
   329→fn encode_2r(opcode: u32, rj: u32, rd: u32) -> [u8; 4] {
   330→    let word = ((opcode & 0x3FF_FFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   331→    word.to_le_bytes()
   332→}
   333→
   334→/// Encode a 3R format instruction.
   335→///
   336→/// Format: `opcode[31:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   337→fn encode_3r(opcode: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   338→    let word = ((opcode & 0x1FFFF) << 15) | ((rk & 0x1F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   339→    word.to_le_bytes()
   340→}
   341→
   342→/// Encode a 4R format instruction.
   343→///
   344→/// Format: `opcode[31:20] | ra[19:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   345→#[allow(dead_code)]
   346→fn encode_4r(opcode: u32, ra: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   347→    let word = ((opcode & 0xFFF) << 20)
   348→        | ((ra & 0x1F) << 15)
   349→        | ((rk & 0x1F) << 10)
   350→        | ((rj & 0x1F) << 5)
   351→        | (rd & 0x1F);
   352→    word.to_le_bytes()
   353→}
   354→
   355→/// Encode a 2RI8 format instruction.
   356→///
   357→/// Format: `opcode[31:22] | I8[21:14] | rj[9:5] | rd[4:0]`
   358→fn encode_2ri8(opcode: u32, imm8: u32, rj: u32, rd: u32) -> [u8; 4] {
   359→    let word = ((opcode & 0x3FF) << 22) | ((imm8 & 0xFF) << 14) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   360→    word.to_le_bytes()
   361→}
   362→
   363→/// Encode a 2RI12 format instruction.
   364→///
   365→/// Format: `opcode[31:22] | I12[21:10] | rj[9:5] | rd[4:0]`
   366→fn encode_2ri12(opcode: u32, imm12: u32, rj: u32, rd: u32) -> [u8; 4] {
   367→    let word =
   368→        ((opcode & 0x3FF) << 22) | ((imm12 & 0xFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   369→    word.to_le_bytes()
   370→}
   371→
   372→/// Encode a 2RI14 format instruction.
   373→///
   374→/// Format: `opcode[31:24] | I14[23:10] | rj[9:5] | rd[4:0]`
   375→fn encode_2ri14(opcode: u32, imm14: u32, rj: u32, rd: u32) -> [u8; 4] {
   376→    let word =
   377→        ((opcode & 0xFF) << 24) | ((imm14 & 0x3FFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   378→    word.to_le_bytes()
   379→}
   380→
   381→/// Encode a 2RI16 format instruction.
   382→///
   383→/// Format: `opcode[31:26] | I16[25:10] | rj[9:5] | rd[4:0]`
   384→fn encode_2ri16(opcode: u32, imm16: u32, rj: u32, rd: u32) -> [u8; 4] {
   385→    let word =
   386→        ((opcode & 0x3F) << 26) | ((imm16 & 0xFFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   387→    word.to_le_bytes()
   388→}
   389→
   390→/// Encode a 1RI21 format instruction.
   391→///
   392→/// Format: `opcode[31:26] | I21[25:5] | rj[9:5]`
   393→///
   394→/// The 21-bit immediate is split: bits[20:16] in bits[25:21], bits[15:0] in
   395→/// bits[20:5].
   396→fn encode_1ri21(opcode: u32, imm21: u32, rj: u32) -> [u8; 4] {
   397→    let word = ((opcode & 0x3F) << 26)
   398→        | ((imm21 & 0x1F) << 21)
   399→        | ((imm21 >> 5) & 0xFFFF) << 5
   400→        | (rj & 0x1F);
   401→    word.to_le_bytes()
   402→}
   403→
   404→/// Encode an I26 format instruction.
   405→///
   406→/// Format: `opcode[31:26] | I26[25:0]`
   407→///
   408→/// The 26-bit immediate is split: bits[25:16] in bits[25:16], bits[15:0] in
   409→/// bits[9:0] with bits[15:10] in bits[25:20] and bits[9:0] in bits[9:0].
   410→fn encode_i26(opcode: u32, imm26: u32) -> [u8; 4] {
   411→    let hi10 = (imm26 >> 16) & 0x3FF;
   412→    let lo16 = imm26 & 0xFFFF;
   413→    let word = ((opcode & 0x3F) << 26) | (hi10 << 16) | lo16;
   414→    word.to_le_bytes()
   415→}

   847→impl Instruction {
   848→    /// Encode this instruction into a 4-byte little-endian machine code word.
   849→    ///
   850→    /// Encoding follows the LoongArch ISA Specification.
   851→    pub fn encode(&self) -> [u8; 4] {
   852→        match self {
   853→            // ── Arithmetic (3R) ────────────────────────────────────
   854→            Instruction::AddW { rd, rj, rk } => {
   855→                encode_3r(OPC_ADD_W, rk.encoding(), rj.encoding(), rd.encoding())
   856→            }
   857→            Instruction::AddD { rd, rj, rk } => {
   858→                encode_3r(OPC_ADD_D, rk.encoding(), rj.encoding(), rd.encoding())
   859→            }
   860→            Instruction::SubW { rd, rj, rk } => {
   861→                encode_3r(OPC_SUB_W, rk.encoding(), rj.encoding(), rd.encoding())
   862→            }
   863→            Instruction::SubD { rd, rj, rk } => {
   864→                encode_3r(OPC_SUB_D, rk.encoding(), rj.encoding(), rd.encoding())
   865→            }
   866→            Instruction::Slt { rd, rj, rk } => {
   867→                encode_3r(OPC_SLT, rk.encoding(), rj.encoding(), rd.encoding())
   868→            }
   869→            Instruction::Sltu { rd, rj, rk } => {
   870→                encode_3r(OPC_SLTU, rk.encoding(), rj.encoding(), rd.encoding())
   871→            }
   872→            Instruction::MulW { rd, rj, rk } => {
   873→                encode_3r(OPC_MUL_W, rk.encoding(), rj.encoding(), rd.encoding())
   874→            }
   875→            Instruction::MulD { rd, rj, rk } => {
   876→                encode_3r(OPC_MUL_D, rk.encoding(), rj.encoding(), rd.encoding())
   877→            }
   878→            Instruction::DivW { rd, rj, rk } => {
   879→                encode_3r(OPC_DIV_W, rk.encoding(), rj.encoding(), rd.encoding())
   880→            }
   881→            Instruction::ModW { rd, rj, rk } => {
   882→                encode_3r(OPC_MOD_W, rk.encoding(), rj.encoding(), rd.encoding())
   883→            }
   884→            Instruction::DivD { rd, rj, rk } => {
   885→                encode_3r(OPC_DIV_D, rk.encoding(), rj.encoding(), rd.encoding())
   886→            }
   887→            Instruction::ModD { rd, rj, rk } => {
   888→                encode_3r(OPC_MOD_D, rk.encoding(), rj.encoding(), rd.encoding())
   889→            }
   890→
   891→            // ── Logical (3R) ──────────────────────────────────────
   892→            Instruction::And { rd, rj, rk } => {
   893→                encode_3r(OPC_AND, rk.encoding(), rj.encoding(), rd.encoding())
   894→            }
   895→            Instruction::Or { rd, rj, rk } => {
   896→                encode_3r(OPC_OR, rk.encoding(), rj.encoding(), rd.encoding())
   897→            }
   898→            Instruction::Xor { rd, rj, rk } => {
   899→                encode_3r(OPC_XOR, rk.encoding(), rj.encoding(), rd.encoding())
   900→            }
   901→            Instruction::Nor { rd, rj, rk } => {
   902→                encode_3r(OPC_NOR, rk.encoding(), rj.encoding(), rd.encoding())
   903→            }
   904→            Instruction::Andn { rd, rj, rk } => {
   905→                encode_3r(OPC_ANDN, rk.encoding(), rj.encoding(), rd.encoding())
   906→            }
   907→            Instruction::Orn { rd, rj, rk } => {
   908→                encode_3r(OPC_ORN, rk.encoding(), rj.encoding(), rd.encoding())
   909→            }
   910→

  1528→/// Build a minimal ELF64 binary for LoongArch64 from raw code bytes.
  1529→///
  1530→/// Produces a static executable with a single LOAD segment containing the
  1531→/// `.text` section. Entry point is at `base_addr` + header offset.
  1532→fn build_minimal_loongarch64_elf(code: &[u8], base_addr: u64) -> Vec<u8> {
  1533→    let elf_header_size: u64 = 64;
  1534→    let phdr_size: u64 = 56;
  1535→    let text_offset = elf_header_size + phdr_size;
  1536→    let text_size = code.len() as u64;
  1537→    let entry_point = base_addr + text_offset;
  1538→
  1539→    let mut elf = Vec::with_capacity(text_offset as usize + code.len());
  1540→
  1541→    // --- e_ident ---
  1542→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1543→    elf.push(2); // ELFCLASS64
  1544→    elf.push(1); // ELFDATA2LSB
  1545→    elf.push(1); // EV_CURRENT
  1546→    elf.push(3); // ELFOSABI_LINUX
  1547→    elf.push(0); // padding
  1548→    elf.extend_from_slice(&[0u8; 7]); // padding
  1549→
  1550→    // --- ELF header fields ---
  1551→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1552→    elf.extend_from_slice(&258u16.to_le_bytes()); // e_machine = EM_LOONGARCH
  1553→    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1554→    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1555→    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1556→    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1557→    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1558→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1559→    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1560→    elf.extend_from_slice(&1u16.to_le_bytes()); // e_phnum
  1561→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1562→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1563→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1564→
  1565→    // --- Program Header (single LOAD segment: PF_R | PF_X) ---
  1566→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1567→    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1568→    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1569→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1570→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1571→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1572→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1573→    elf.extend_from_slice(&16u64.to_le_bytes()); // p_align
  1574→
  1575→    // --- Code section ---
  1576→    elf.extend_from_slice(code);
  1577→
  1578→    elf
  1579→}

  1585→/// Decode a single LoongArch64 32-bit instruction word into a mnemonic string.
  1586→fn decode_loongarch64_instruction(word: u32) -> String {
  1587→    let opcode = word & 0x7f;
  1588→    match opcode {
  1589→        0x00 => {
  1590→            let rd = (word >> 7) & 0x1f;
  1591→            let rj = (word >> 15) & 0x1f;
  1592→            let rk = (word >> 10) & 0x1f;
  1593→            format!("add.w $r{}, $r{}, $r{}", rd, rj, rk)
  1594→        }
  1595→        0x01 => {
  1596→            let rd = (word >> 7) & 0x1f;
  1597→            let rj = (word >> 15) & 0x1f;
  1598→            let rk = (word >> 10) & 0x1f;
  1599→            format!("sub.w $r{}, $r{}, $r{}", rd, rj, rk)
  1600→        }
  1601→        0x02 => {
  1602→            let rd = (word >> 7) & 0x1f;
  1603→            let rj = (word >> 15) & 0x1f;
  1604→            let rk = (word >> 10) & 0x1f;
  1605→            format!("mul.w $r{}, $r{}, $r{}", rd, rj, rk)
  1606→        }
  1607→        0x05 => {
  1608→            let rd = (word >> 7) & 0x1f;
  1609→            let _rj = (word >> 15) & 0x1f; // lu12i.w does not use rj in the disassembly
  1610→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1611→            format!("lu12i.w $r{}, {}({})", rd, si12, si12)
  1612→        }
  1613→        0x08 => {
  1614→            let rd = (word >> 7) & 0x1f;
  1615→            let rj = (word >> 15) & 0x1f;
  1616→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1617→            format!("ld.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1618→        }
  1619→        0x0a => {
  1620→            let rd = (word >> 7) & 0x1f;
  1621→            let rj = (word >> 15) & 0x1f;
  1622→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1623→            format!("st.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1624→        }
  1625→        0x0c => {
  1626→            let rd = (word >> 7) & 0x1f;
  1627→            let rj = (word >> 15) & 0x1f;
  1628→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1629→            format!("ld.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1630→        }
  1631→        0x0e => {
  1632→            let rd = (word >> 7) & 0x1f;
  1633→            let rj = (word >> 15) & 0x1f;
  1634→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1635→            format!("st.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1636→        }
  1637→        0x10 => {
  1638→            let rj = (word >> 15) & 0x1f;
  1639→            let offs = ((word >> 10) as i32) << 12 >> 10;
  1640→            format!("beq $r{}, {}({})", rj, offs, offs)
  1641→        }
  1642→        0x11 => {
  1643→            let rj = (word >> 15) & 0x1f;
  1644→            let offs = ((word >> 10) as i32) << 12 >> 10;
  1645→            format!("bne $r{}, {}({})", rj, offs, offs)
  1646→        }
  1647→        0x14 => format!("bl {}", ((word >> 10) as i32) << 12 >> 10),
  1648→        0x15 => {
  1649→            let rd = (word >> 7) & 0x1f;
  1650→            format!("jirl $r{}, {}", rd, ((word >> 10) as i32) << 12 >> 10)
  1651→        }
  1652→        _ => format!(".word {:08x}", word),
  1653→    }
  1654→}

  1676→/// Compute the stack frame size for an IR function on LoongArch64.
  1677→///
  1678→/// Sums `Alloc` instruction sizes, adds 16 bytes for the ra/fp save pair,
  1679→/// and rounds up to 16-byte alignment.
  1680→fn loongarch64_compute_frame_size(func: &IRFunction) -> usize {
  1681→    let mut total: u32 = 16; // ra + fp save pair
  1682→    for block in &func.blocks {
  1683→        for instr in &block.instructions {
  1684→            if let IRInstr::Alloc { size, .. } = instr {
  1685→                let aligned = (*size).div_ceil(16) * 16;
  1686→                total += aligned;
  1687→            }
  1688→        }
  1689→    }
  1690→    // Round up to 16-byte alignment
  1691→    total = (total + 15) & !15;
  1692→    total as usize
  1693→}

  1730→/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1731→fn map_vreg_to_gpr(
  1732→    vreg_id: u32,
  1733→    arg_index: Option<usize>,
  1734→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1735→) -> Gpr {
  1736→    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1737→        return *gpr;
  1738→    }
  1739→
  1740→    // If this is an argument, map it to the corresponding argument register.
  1741→    if let Some(idx) = arg_index {
  1742→        if let Some(reg) = Gpr::arg_register(idx) {
  1743→            vreg_map.insert(vreg_id, reg);
  1744→            return reg;
  1745→        }
  1746→    }
  1747→
  1748→    // Otherwise, pick the first available allocatable register.
  1749→    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1750→    for &reg in ALLOCATABLE_GPRS {
  1751→        if !used.contains(&reg) {
  1752→            vreg_map.insert(vreg_id, reg);
  1753→            return reg;
  1754→        }
  1755→    }
  1756→
  1757→    // Fallback: use t8 as a scratch register.
  1758→    Gpr::T8
  1759→}

  1769→/// Load a 64-bit immediate into a register using the lu12i.w/ori/lu32i.d/lu52i.d
  1770→/// instruction sequence.
  1771→///
  1772→/// Returns the list of AllocatedInstructions emitted to perform the load.
  1773→fn load_imm_la64(rd: Gpr, imm: i64) -> Vec<AllocatedInstruction> {
  1774→    let mut result = Vec::new();
  1775→    let val = imm as u64;
  1776→
  1777→    // lu12i.w rd, bits[31:12]

  1799→    result.push(emit_alloc_instr(
  1800→        Instruction::Lu32iD { rd, imm20: hi32 },

  1805→    // lu52i.d rd, rd, bits[63:52]

  1820→/// Resolve an IRValue to a physical GPR.
  1821→///
  1822→/// - For `IRValue::Register`: looks up in `vreg_map`.
  1823→/// - For `IRValue::Immediate`: loads the value into `scratch` and returns
  1824→///   `(scratch, instructions)`.
  1825→/// - For `IRValue::Address`: loads the address into `scratch`.
  1826→/// - For `IRValue::Label`: loads a placeholder into `scratch`.
  1827→fn resolve_gpr_la64(
  1828→    val: &IRValue,
  1829→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1830→    scratch: Gpr,
  1831→) -> (Gpr, Vec<AllocatedInstruction>) {
  1832→    match val {
  1833→        IRValue::Register(id) => {
  1834→            let reg = map_vreg_to_gpr(*id, None, vreg_map);
  1835→            (reg, Vec::new())
  1836→        }
  1837→        IRValue::Immediate(imm) => {
  1838→            let code = load_imm_la64(scratch, *imm);
  1839→            (scratch, code)
  1840→        }
  1841→        IRValue::Address(addr) => {
  1842→            let code = load_imm_la64(scratch, *addr as i64);
  1843→            (scratch, code)
  1844→        }
  1845→        IRValue::Label(_) => {
  1846→            // Labels need relocation; emit a placeholder load
  1847→            let code = load_imm_la64(scratch, 0);
  1848→            (scratch, code)
  1849→        }
  1850→    }
  1851→}
