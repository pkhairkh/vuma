     1→//! # Target Description System
     2→//!
     3→//! Machine-readable ISA specifications that make adding new ISAs a data-driven
     4→//! process. Each `TargetDesc` contains the complete register file, calling
     5→//! convention details, and instruction category metadata for an ISA.
     6→
     7→use crate::backend::{Endianness, OutputFormat, RegClass};
     8→
     9→/// A complete machine-readable description of a target ISA.
    10→#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
    11→pub struct TargetDesc {
    12→    /// ISA name in lowercase (e.g., `"aarch64"`, `"riscv64"`).
    13→    pub name: &'static str,
    14→    /// LLVM-style target triple (e.g., `"aarch64-unknown-linux-gnu"`).
    15→    pub triple: &'static str,
    16→    /// ELF `e_machine` value (0 for non-ELF targets like Wasm).
    17→    pub elf_machine: u16,
    18→    /// Default base address for the `.text` section.
    19→    pub base_addr: u64,
    20→    /// Pointer width in bytes (4 for 32-bit, 8 for 64-bit).
    21→    pub pointer_width: usize,
    22→    /// Byte order of the target.
    23→    pub endianness: Endianness,
    24→    /// Binary output format produced by the backend.
    25→    pub output_format: OutputFormat,
    26→    /// Register file description.
    27→    pub registers: Vec<RegDesc>,
    28→    /// Calling convention details.
    29→    pub calling_convention: CallingConventionDesc,
    30→    /// Instruction category metadata.
    31→    pub instruction_categories: Vec<InstCategoryDesc>,
    32→}
    33→
    34→/// Description of a single register.
    35→#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
    36→pub struct RegDesc {
    37→    /// Register name (e.g., `"X0"`, `"RAX"`, `"x10"`).
    38→    pub name: &'static str,
    39→    /// Register class (GPR, SIMD/FP, Condition, Special).
    40→    pub class: RegClass,
    41→    /// Register index within its class (0-based).
    42→    pub index: usize,
    43→    /// Whether this register is available for the register allocator.
    44→    pub is_allocatable: bool,
    45→    /// Whether this register always reads as zero (e.g., RISC-V `x0`).
    46→    pub is_hardwired_zero: bool,
    47→    /// Whether this register serves as the stack pointer.
    48→    pub is_stack_pointer: bool,
    49→    /// Whether this register serves as the frame pointer.
    50→    pub is_frame_pointer: bool,
    51→    /// Whether this register holds the return address (link register).
    52→    pub is_link_register: bool,
    53→    /// Whether this register holds the TOC (table of contents) pointer.
    54→    pub is_toc_pointer: bool,
    55→    /// Whether this register must be preserved across function calls.
    56→    pub is_callee_saved: bool,
    57→    /// Whether this register is used for passing arguments.
    58→    pub is_arg_reg: bool,
    59→    /// For argument registers, the position in the argument list (0-based).
    60→    pub arg_position: Option<usize>,
    61→    /// Whether this register is used for returning values.
    62→    pub is_return_reg: bool,
    63→}
    64→
    65→impl RegDesc {
    66→    /// Create a new GPR descriptor (allocatable by default).
    67→    fn gpr(name: &'static str, index: usize) -> Self {
    68→        Self {
    69→            name,
    70→            class: RegClass::Gpr,
    71→            index,
    72→            is_allocatable: true,
    73→            is_hardwired_zero: false,
    74→            is_stack_pointer: false,
    75→            is_frame_pointer: false,
    76→            is_link_register: false,
    77→            is_toc_pointer: false,
    78→            is_callee_saved: false,
    79→            is_arg_reg: false,
    80→            arg_position: None,
    81→            is_return_reg: false,
    82→        }
    83→    }
    84→
    85→    /// Create a new SIMD/FP register descriptor (allocatable by default).
    86→    fn fpr(name: &'static str, index: usize) -> Self {
    87→        Self {
    88→            name,
    89→            class: RegClass::SimdFp,
    90→            index,
    91→            is_allocatable: true,
    92→            is_hardwired_zero: false,
    93→            is_stack_pointer: false,
    94→            is_frame_pointer: false,
    95→            is_link_register: false,
    96→            is_toc_pointer: false,
    97→            is_callee_saved: false,
    98→            is_arg_reg: false,
    99→            arg_position: None,
   100→            is_return_reg: false,
   101→        }
   102→    }
   103→
   104→    /// Create a new special-purpose register descriptor (not allocatable).
   105→    fn special_reg(name: &'static str, index: usize) -> Self {
   106→        Self {
   107→            name,
   108→            class: RegClass::Special,
   109→            index,
   110→            is_allocatable: false,
   111→            is_hardwired_zero: false,
   112→            is_stack_pointer: false,
   113→            is_frame_pointer: false,
   114→            is_link_register: false,
   115→            is_toc_pointer: false,
   116→            is_callee_saved: false,
   117→            is_arg_reg: false,
   118→            arg_position: None,
   119→            is_return_reg: false,
   120→        }
   121→    }
   122→
   123→    /// Create a new condition register descriptor (not allocatable).
   124→    fn cond_reg(name: &'static str, index: usize) -> Self {
   125→        Self {
   126→            name,
   127→            class: RegClass::Condition,
   128→            index,
   129→            is_allocatable: false,
   130→            is_hardwired_zero: false,
   131→            is_stack_pointer: false,
   132→            is_frame_pointer: false,
   133→            is_link_register: false,
   134→            is_toc_pointer: false,
   135→            is_callee_saved: false,
   136→            is_arg_reg: false,
   137→            arg_position: None,
   138→            is_return_reg: false,
   139→        }
   140→    }
   141→
   142→    // Builder-style modifiers
   143→
   144→    fn hardwired_zero(mut self) -> Self {
   145→        self.is_hardwired_zero = true;
   146→        self.is_allocatable = false;
   147→        self
   148→    }
   149→
   150→    fn stack_pointer(mut self) -> Self {
   151→        self.is_stack_pointer = true;
   152→        self.is_allocatable = false;
   153→        self
   154→    }
   155→
   156→    fn frame_pointer(mut self) -> Self {
   157→        self.is_frame_pointer = true;
   158→        self
   159→    }
   160→
   161→    fn link_register(mut self) -> Self {
   162→        self.is_link_register = true;
   163→        self.is_allocatable = false;
   164→        self
   165→    }
   166→
   167→    fn toc_pointer(mut self) -> Self {
   168→        self.is_toc_pointer = true;
   169→        self.is_allocatable = false;
   170→        self
   171→    }
   172→
   173→    fn callee_saved(mut self) -> Self {
   174→        self.is_callee_saved = true;
   175→        self
   176→    }
   177→
   178→    fn arg(mut self, pos: usize) -> Self {
   179→        self.is_arg_reg = true;
   180→        self.arg_position = Some(pos);
   181→        self
   182→    }
   183→
   184→    fn return_reg(mut self) -> Self {
   185→        self.is_return_reg = true;
   186→        self
   187→    }
   188→
   189→    fn not_allocatable(mut self) -> Self {
   190→        self.is_allocatable = false;
   191→        self
   192→    }
   193→}

   195→/// Description of a calling convention.
   196→#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
   197→pub struct CallingConventionDesc {
   198→    /// Calling convention name (e.g., `"aapcs64"`, `"lp64d"`, `"systemv"`).
   199→    pub name: &'static str,
   200→    /// GPR indices used for integer arguments, in order.
   201→    pub int_arg_regs: Vec<usize>,
   202→    /// FPR indices used for floating-point arguments, in order.
   203→    pub fp_arg_regs: Vec<usize>,
   204→    /// GPR indices used for integer return values, in order.
   205→    pub int_return_regs: Vec<usize>,
   206→    /// FPR indices used for floating-point return values, in order.
   207→    pub fp_return_regs: Vec<usize>,
   208→    /// GPR indices that are callee-saved.
   209→    pub callee_saved_gprs: Vec<usize>,
   210→    /// FPR indices that are callee-saved.
   211→    pub callee_saved_fps: Vec<usize>,
   212→    /// Required stack alignment in bytes.
   213→    pub stack_alignment: usize,
   214→    /// Whether the ISA uses a link register (vs pushing return address on stack).
   215→    pub has_link_register: bool,
   216→    /// Whether branches have delay slots (MIPS only).
   217→    pub has_branch_delay_slots: bool,
   218→    /// Whether the ISA uses a TOC (table of contents) pointer (PPC64).
   219→    pub has_toc_pointer: bool,
   220→}

   222→/// Description of an instruction category.
   223→#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
   224→pub struct InstCategoryDesc {
   225→    /// Category name (e.g., `"arithmetic"`, `"branch"`, `"load_store"`).
   226→    pub name: &'static str,
   227→    /// Instruction mnemonics belonging to this category.
   228→    pub insts: Vec<&'static str>,
   229→}

   231→/// Registry of all target descriptions.
   232→pub struct TargetDescRegistry {
   233→    descs: std::collections::HashMap<&'static str, TargetDesc>,
   234→}
   235→
   236→impl TargetDescRegistry {
   237→    /// Creates a new registry pre-populated with all supported target descriptions.
   238→    pub fn new() -> Self {
   239→        let mut descs = std::collections::HashMap::new();
   240→        descs.insert("aarch64", aarch64_target_desc());
   241→        descs.insert("riscv64", riscv64_target_desc());
   242→        descs.insert("wasm32", wasm32_target_desc());
   243→        descs.insert("loongarch64", loongarch64_target_desc());
   244→        descs.insert("x86_64", x86_64_target_desc());
   245→        descs.insert("arm32", arm32_target_desc());
   246→        descs.insert("mips64", mips64_target_desc());
   247→        descs.insert("ppc64", ppc64_target_desc());
   248→        Self { descs }
   249→    }
   250→
   251→    /// Looks up a target description by ISA name (e.g., `"aarch64"`).
   252→    pub fn get(&self, name: &str) -> Option<&TargetDesc> {
   253→        self.descs.get(name)
   254→    }
   255→
   256→    /// Returns the list of all registered ISA names.
   257→    pub fn isa_names(&self) -> Vec<&'static str> {
   258→        self.descs.keys().copied().collect()
   259→    }
   260→}
   261→
   262→impl Default for TargetDescRegistry {
   263→    fn default() -> Self {
   264→        Self::new()
   265→    }
   266→}
