  1400Ôćĺ    elf.extend_from_slice(code);
  1401Ôćĺ
  1402Ôćĺ    elf
  1403Ôćĺ}

  1417Ôćĺ                isa: "aarch64",
  1418Ôćĺ                reason: e.to_string(),
  1419Ôćĺ            })?;

  1440Ôćĺ            name: func_name,
  1441Ôćĺ            blocks: vec![AllocatedBlock {
  1442Ôćĺ                label: "entry".to_string(),

  1447Ôćĺ            callee_saved: vec![],
  1448Ôćĺ            spill_slots: 0,

  1450Ôćĺ            relocations: Vec::new(),
  1451Ôćĺ        })

  1454Ôćĺ    fn encode_function(&self, func: &AllocatedFunction) -> Result<Vec<u8>, BackendError> {
  1455Ôćĺ        let mut bytes = Vec::new();
  1456Ôćĺ        for block in &func.blocks {
  1457Ôćĺ            for instr in &block.instructions {

  1464Ôćĺ    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError> {
  1465Ôćĺ        // ARM64 Linux executable layout:
  1466Ôćĺ        // [_start stub (16 bytes)] [user functions] [runtime I/O functions]
  1467Ôćĺ        let start_stub_size: usize = 16; // 4 ARM64 instructions
  1468Ôćĺ
  1469Ôćĺ        // Compute function offsets (after _start stub)
  1470Ôćĺ        let mut func_offsets: HashMap<String, usize> = HashMap::new();
  1471Ôćĺ        let mut current_offset: usize = start_stub_size;
  1472Ôćĺ
  1473Ôćĺ        for func in &program.functions {
  1474Ôćĺ            func_offsets.insert(func.name.clone(), current_offset);
  1475Ôćĺ            let func_size: usize = func.blocks.iter()
  1476Ôćĺ                .flat_map(|b| b.instructions.iter())
  1477Ôćĺ                .map(|i| i.encoded.len())
  1478Ôćĺ                .sum();
  1479Ôćĺ            current_offset += func_size;
  1480Ôćĺ        }
  1481Ôćĺ
  1482Ôćĺ        // Runtime I/O functions: __vuma_write_stdout
  1483Ôćĺ        let write_stdout_offset = current_offset;
  1484Ôćĺ        for name in &["print_buf", "write_stdout", "__vuma_write_stdout", "print_hex"] {
  1485Ôćĺ            func_offsets.insert(name.to_string(), write_stdout_offset);
  1486Ôćĺ        }
  1487Ôćĺ        // ARM64 write_stdout: mov x2,x1; mov x1,x0; mov x0,#1; mov x8,#64; svc#0; ret
  1488Ôćĺ        let runtime_write_stdout: Vec<u8> = vec![
  1489Ôćĺ            0xE2, 0x03, 0x01, 0xAA, // mov x2, x1
  1490Ôćĺ            0xE1, 0x03, 0x00, 0xAA, // mov x1, x0
  1491Ôćĺ            0x20, 0x00, 0x80, 0xD2, // mov x0, #1
  1492Ôćĺ            0x08, 0x08, 0x80, 0xD2, // mov x8, #64
  1493Ôćĺ            0x01, 0x00, 0x00, 0xD4, // svc #0
  1494Ôćĺ            0xC0, 0x03, 0x5F, 0xD6, // ret
  1495Ôćĺ        ];
  1496Ôćĺ        current_offset += runtime_write_stdout.len();
  1497Ôćĺ
  1498Ôćĺ        // print_int: stub that returns 0
  1499Ôćĺ        let print_int_offset = current_offset;
  1500Ôćĺ        for name in &["print_int", "print", "__vuma_print_int"] {
  1501Ôćĺ            func_offsets.insert(name.to_string(), print_int_offset);
  1502Ôćĺ        }
  1503Ôćĺ        let runtime_print_int: Vec<u8> = vec![
  1504Ôćĺ            0x00, 0x00, 0x80, 0xD2, // mov x0, #0
  1505Ôćĺ            0xC0, 0x03, 0x5F, 0xD6, // ret
  1506Ôćĺ        ];
  1507Ôćĺ        current_offset += runtime_print_int.len();
  1508Ôćĺ
  1509Ôćĺ        // Build _start stub: bl main; mov x8, #93; svc #0; ret
  1510Ôćĺ        let mut start_stub = vec![0u8; start_stub_size];
  1511Ôćĺ        if let Some(&main_offset) = func_offsets.get("main") {
  1512Ôćĺ            let bl_offset = main_offset as i64; // from offset 0
  1513Ôćĺ            let imm26 = (bl_offset >> 2) as u32 & 0x03FFFFFF;
  1514Ôćĺ            let bl_instr = 0x94000000u32 | imm26;
  1515Ôćĺ            start_stub[0..4].copy_from_slice(&bl_instr.to_le_bytes());
  1516Ôćĺ        }
  1517Ôćĺ        // mov x8, #93 (sys_exit)
  1518Ôćĺ        start_stub[4..8].copy_from_slice(&0xD2800BA8u32.to_le_bytes());
  1519Ôćĺ        // svc #0
  1520Ôćĺ        start_stub[8..12].copy_from_slice(&0xD4000001u32.to_le_bytes());
  1521Ôćĺ        // ret
  1522Ôćĺ        start_stub[12..16].copy_from_slice(&0xD65F03C0u32.to_le_bytes());
  1523Ôćĺ
  1524Ôćĺ        // Concatenate all code
  1525Ôćĺ        let mut all_code = start_stub;
  1526Ôćĺ        for func in &program.functions {
  1527Ôćĺ            for block in &func.blocks {
  1528Ôćĺ                for instr in &block.instructions {
  1529Ôćĺ                    all_code.extend_from_slice(&instr.encoded);
  1530Ôćĺ                }
  1531Ôćĺ            }
  1532Ôćĺ        }
  1533Ôćĺ        all_code.extend_from_slice(&runtime_write_stdout);
  1534Ôćĺ        all_code.extend_from_slice(&runtime_print_int);
  1535Ôćĺ
  1536Ôćĺ        // Patch relocations for user functions
  1537Ôćĺ        let mut func_code_offset: usize = start_stub_size;
  1538Ôćĺ        for func in &program.functions {
  1539Ôćĺ            for reloc in &func.relocations {
  1540Ôćĺ                let abs_offset = func_code_offset + reloc.offset as usize;
  1541Ôćĺ                if abs_offset + 4 > all_code.len() {
  1542Ôćĺ                    continue;
  1543Ôćĺ                }
  1544Ôćĺ                // ARM64 BL relocation: 26-bit signed offset in bits[25:0]
  1545Ôćĺ                if reloc.reloc_type == "R_AARCH64_CALL26" {
  1546Ôćĺ                    if let Some(&target_offset) = func_offsets.get(&reloc.symbol) {
  1547Ôćĺ                        let current_bits = u32::from_le_bytes([
  1548Ôćĺ                            all_code[abs_offset],
  1549Ôćĺ                            all_code[abs_offset + 1],
  1550Ôćĺ                            all_code[abs_offset + 2],
  1551Ôćĺ                            all_code[abs_offset + 3],
  1552Ôćĺ                        ]);
  1553Ôćĺ                        // BL offset = (target - source) >> 2
  1554Ôćĺ                        let bl_offset = (target_offset as i64 - abs_offset as i64) >> 2;
  1555Ôćĺ                        let imm26 = (bl_offset as u32) & 0x03FFFFFF;
  1556Ôćĺ                        let patched = (current_bits & 0xFC000000) | imm26;
  1557Ôćĺ                        all_code[abs_offset..abs_offset + 4].copy_from_slice(&patched.to_le_bytes());
  1558Ôćĺ                    }
  1559Ôćĺ                }
  1560Ôćĺ            }
  1561Ôćĺ            let func_size: usize = func.blocks.iter()
  1562Ôćĺ                .flat_map(|b| b.instructions.iter())
  1563Ôćĺ                .map(|i| i.encoded.len())
  1564Ôćĺ                .sum();
  1565Ôćĺ            func_code_offset += func_size;
  1566Ôćĺ        }
  1567Ôćĺ
  1568Ôćĺ        Ok(build_minimal_aarch64_elf(&all_code, 0x400000))
  1569Ôćĺ    }
