     1‚Üí     1‚Üí//! # LoongArch 64-bit Backend
     2‚Üí     2‚Üí//!
     3‚Üí     3‚Üí//! Implements the `Backend` trait for the LoongArch 64-bit target (LP64 ABI).
     4‚Üí     4‚Üí//! This module provides:
     5‚Üí     5‚Üí//!
     6‚Üí     6‚Üí//! - `Gpr` ‚Äî General-purpose register enum (r0‚Äìr31)
     7‚Üí     7‚Üí//! - `Fpr` ‚Äî Floating-point register enum (f0‚Äìf31)
     8‚Üí     8‚Üí//! - `Instruction` ‚Äî LoongArch64 instruction enum with correct encoding
     9‚Üí     9‚Üí//! - Encoding helpers for all 9 instruction formats (2R, 3R, 4R, 2RI8, 2RI12,
    10‚Üí    10‚Üí//!   2RI14, 2RI16, 1RI21, I26)
    11‚Üí    11‚Üí//! - `LoongArch64Backend` ‚Äî `Backend` implementation that lowers IR to LoongArch64
    12‚Üí    12‚Üí//!   machine code and emits ELF64 binaries
    13‚Üí    13‚Üí//!
    14‚Üí    14‚Üí//! ## LoongArch64 Register Convention (LP64 ABI)
    15‚Üí    15‚Üí//!
    16‚Üí    16‚Üí//! | Register(s) | ABI Name | Role                              |
    17‚Üí    17‚Üí//! |-------------|----------|-----------------------------------|
    18‚Üí    18‚Üí//! | r0          | zero     | Hardwired zero                    |
    19‚Üí    19‚Üí//! | r1          | ra       | Return address                    |
    20‚Üí    20‚Üí//! | r2          | tp       | Thread pointer                    |
    21‚Üí    21‚Üí//! | r3          | sp       | Stack pointer                     |
    22‚Üí    22‚Üí//! | r4‚Äìr11      | a0‚Äìa7    | Argument / return registers       |
    23‚Üí    23‚Üí//! | r12‚Äìr20     | t0‚Äìt8    | Caller-saved temporaries          |
    24‚Üí    24‚Üí//! | r21         | ‚Äî        | Reserved                          |
    25‚Üí    25‚Üí//! | r22         | fp       | Frame pointer (callee-saved)      |
    26‚Üí    26‚Üí//! | r23‚Äìr31     | s0‚Äìs8    | Callee-saved                      |
    27‚Üí    27‚Üí//!
    28‚Üí    28‚Üí//! ## LoongArch64 FP Register Convention (LP64 ABI)
    29‚Üí    29‚Üí//!
    30‚Üí    30‚Üí//! | Register(s) | ABI Name | Role                              |
    31‚Üí    31‚Üí//! |-------------|----------|-----------------------------------|
    32‚Üí    32‚Üí//! | f0‚Äìf7       | fa0‚Äìfa7  | FP argument / return registers    |
    33‚Üí    33‚Üí//! | f8‚Äìf23      | ft0‚Äìft15 | Caller-saved temporaries          |
    34‚Üí    34‚Üí//! | f24‚Äìf31     | fs0‚Äìfs7  | Callee-saved                      |
    35‚Üí    35‚Üí//!
    36‚Üí    36‚Üí//! ## Instruction Formats
    37‚Üí    37‚Üí//!
    38‚Üí    38‚Üí//! All instructions are 32 bits, little-endian, with no branch delay slots.
    39‚Üí    39‚Üí//! Nine formats: 2R, 3R, 4R, 2RI8, 2RI12, 2RI14, 2RI16, 1RI21, I26.
    40‚Üí    40‚Üí//!
    41‚Üí    41‚Üí//! ## References
    42‚Üí    42‚Üí//!
    43‚Üí    43‚Üí//! - LoongArch Reference Manual, Volume 1: Basic Architecture

   327‚Üí   327‚Üí/// Encode a 2R format instruction.
   328‚Üí   328‚Üí///
   329‚Üí   329‚Üí/// Format: `opcode[31:10] | rj[9:5] | rd[4:0]`
   330‚Üí   330‚Üífn encode_2r(opcode: u32, rj: u32, rd: u32) -> [u8; 4] {
   331‚Üí   331‚Üí    let word = ((opcode & 0x3FF_FFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   332‚Üí   332‚Üí    word.to_le_bytes()
   333‚Üí   333‚Üí}

   391‚Üí   391‚Üí/// Encode a 1RI21 format instruction (used for BEQZ, BNEZ).
   392‚Üí   392‚Üí///
   393‚Üí   393‚Üí/// Format: `opcode[31:26] | offs21[15:0] in bits[25:10] | offs21[20:16] in bits[9:5] | rj[4:0]`
   394‚Üí   394‚Üí///
   395‚Üí   395‚Üí/// Note: the offset bits are SWAPPED compared to a linear layout.
   396‚Üí   396‚Üí/// The lower 16 bits of the offset go in the higher position (bits[25:10]),
   397‚Üí   397‚Üí/// and the upper 5 bits go in the lower position (bits[9:5]).
   398‚Üí   398‚Üífn encode_1ri21(opcode: u32, imm21: u32, rj: u32) -> [u8; 4] {
   399‚Üí   399‚Üí    let word = ((opcode & 0x3F) << 26)
   400‚Üí   400‚Üí        | ((imm21 & 0xFFFF) << 10)
   401‚Üí   401‚Üí        | ((imm21 >> 16) & 0x1F) << 5
   402‚Üí   402‚Üí        | (rj & 0x1F);
   403‚Üí   403‚Üí    word.to_le_bytes()
   404‚Üí   404‚Üí}

   410‚Üí   410‚Üí/// Note: the offset bits are SWAPPED compared to a linear layout.
   411‚Üí   411‚Üí/// The lower 16 bits of the offset go in the higher position (bits[25:10]),
   412‚Üí   412‚Üí/// and the upper 10 bits go in the lower position (bits[9:0]).

   417‚Üí   417‚Üí    word.to_le_bytes()
   418‚Üí   418‚Üí}

  1538‚Üí  1538‚Üí/// Build a minimal ELF64 binary for LoongArch64 from raw code bytes.
  1539‚Üí  1539‚Üí///
  1540‚Üí  1540‚Üí/// Produces a static executable with two LOAD segments:
  1541‚Üí  1541‚Üí///   1. PF_R | PF_X ‚Äî .text (code)
  1542‚Üí  1542‚Üí///   2. PF_R | PF_W ‚Äî .data / stack (writable memory)
  1543‚Üí  1543‚Üífn build_loongarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1544‚Üí  1544‚Üí    const PAGE_SIZE: u64 = 0x10000; // 64 KB (LoongArch64 typical page size)
  1545‚Üí  1545‚Üí
  1546‚Üí  1546‚Üí    let elf_header_size: u64 = 64;
  1547‚Üí  1547‚Üí    let phdr_size: u64 = 56;
  1548‚Üí  1548‚Üí    let num_phdrs: u64 = 2;
  1549‚Üí  1549‚Üí    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1550‚Üí  1550‚Üí    // Page-align the text segment start in the file.
  1551‚Üí  1551‚Üí    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1552‚Üí  1552‚Üí    let text_size = code.len() as u64;
  1553‚Üí  1553‚Üí
  1554‚Üí  1554‚Üí    // The data segment starts on the next page after the text.
  1555‚Üí  1555‚Üí    let text_file_end = text_offset + text_size;
  1556‚Üí  1556‚Üí    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1557‚Üí  1557‚Üí    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1558‚Üí  1558‚Üí    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1559‚Üí  1559‚Üí    let entry_point = base_addr + text_offset;
  1560‚Üí  1560‚Üí
  1561‚Üí  1561‚Üí    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1562‚Üí  1562‚Üí
  1563‚Üí  1563‚Üí    // --- e_ident ---
  1564‚Üí  1564‚Üí    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1565‚Üí  1565‚Üí    elf.push(2); // ELFCLASS64
  1566‚Üí  1566‚Üí    elf.push(1); // ELFDATA2LSB
  1567‚Üí  1567‚Üí    elf.push(1); // EV_CURRENT
  1568‚Üí  1568‚Üí    elf.push(3); // ELFOSABI_LINUX
  1569‚Üí  1569‚Üí    elf.push(0); // padding
  1570‚Üí  1570‚Üí    elf.extend_from_slice(&[0u8; 7]); // padding
  1571‚Üí  1571‚Üí
  1572‚Üí  1572‚Üí    // --- ELF header fields ---
  1573‚Üí  1573‚Üí    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1574‚Üí  1574‚Üí    elf.extend_from_slice(&258u16.to_le_bytes()); // e_machine = EM_LOONGARCH
  1575‚Üí  1575‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1576‚Üí  1576‚Üí    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1577‚Üí  1577‚Üí    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1578‚Üí  1578‚Üí    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1579‚Üí  1579‚Üí    elf.extend_from_slice(&0x43u32.to_le_bytes()); // e_flags = 0x43 (LP64D ABI double-float)
  1580‚Üí  1580‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1581‚Üí  1581‚Üí    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1582‚Üí  1582‚Üí    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1583‚Üí  1583‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1584‚Üí  1584‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1585‚Üí  1585‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1586‚Üí  1586‚Üí
  1587‚Üí  1587‚Üí    // --- Program Header 1: LOAD (PF_R | PF_X) ‚Äî .text ---
  1588‚Üí  1588‚Üí    // p_filesz = actual code size, p_memsz = page-aligned (QEMU needs this for mapping)
  1589‚Üí  1589‚Üí    let text_memsz = ((text_size + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1590‚Üí  1590‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1591‚Üí  1591‚Üí    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1592‚Üí  1592‚Üí    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1593‚Üí  1593‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1594‚Üí  1594‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1595‚Üí  1595‚Üí    elf.extend_from_slice(&text_memsz.to_le_bytes()); // p_filesz (cover full page for QEMU)
  1596‚Üí  1596‚Üí    elf.extend_from_slice(&text_memsz.to_le_bytes()); // p_memsz
  1597‚Üí  1597‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1598‚Üí  1598‚Üí
  1599‚Üí  1599‚Üí    // --- Program Header 2: LOAD (PF_R | PF_W) ‚Äî .data / stack ---
  1600‚Üí  1600‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1601‚Üí  1601‚Üí    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1602‚Üí  1602‚Üí    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1603‚Üí  1603‚Üí    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1604‚Üí  1604‚Üí    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1605‚Üí  1605‚Üí    elf.extend_from_slice(&data_size.to_le_bytes()); // p_filesz (write zeros so QEMU can mmap)
  1606‚Üí  1606‚Üí    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1607‚Üí  1607‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1608‚Üí  1608‚Üí
  1609‚Üí  1609‚Üí    // --- .text section ---
  1610‚Üí  1610‚Üí    // Pad to page-aligned text_offset
  1611‚Üí  1611‚Üí    while (elf.len() as u64) < text_offset {
  1612‚Üí  1612‚Üí        elf.push(0);
  1613‚Üí  1613‚Üí    }
  1614‚Üí  1614‚Üí    elf.extend_from_slice(code);
  1615‚Üí  1615‚Üí
  1616‚Üí  1616‚Üí    // --- Pad to data segment offset and write data segment ---
  1617‚Üí  1617‚Üí    while (elf.len() as u64) < data_offset {
  1618‚Üí  1618‚Üí        elf.push(0);
  1619‚Üí  1619‚Üí    }
  1620‚Üí  1620‚Üí    // Write data segment (all zeros, but must exist in file for QEMU mmap)
  1621‚Üí  1621‚Üí    elf.extend_from_slice(&vec![0u8; data_size as usize]);
  1622‚Üí  1622‚Üí
  1623‚Üí  1623‚Üí    elf
  1624‚Üí  1624‚Üí}

  1630‚Üí  1630‚Üí/// Decode a single LoongArch64 32-bit instruction word into a mnemonic string.
  1631‚Üí  1631‚Üífn decode_loongarch64_instruction(word: u32) -> String {
  1632‚Üí  1632‚Üí    let opcode = word & 0x7f;
  1633‚Üí  1633‚Üí    match opcode {
  1634‚Üí  1634‚Üí        0x00 => {
  1635‚Üí  1635‚Üí            let rd = (word >> 7) & 0x1f;
  1636‚Üí  1636‚Üí            let rj = (word >> 15) & 0x1f;
  1637‚Üí  1637‚Üí            let rk = (word >> 10) & 0x1f;
  1638‚Üí  1638‚Üí            format!("add.w $r{}, $r{}, $r{}", rd, rj, rk)
  1639‚Üí  1639‚Üí        }
  1640‚Üí  1640‚Üí        0x01 => {
  1641‚Üí  1641‚Üí            let rd = (word >> 7) & 0x1f;
  1642‚Üí  1642‚Üí            let rj = (word >> 15) & 0x1f;
  1643‚Üí  1643‚Üí            let rk = (word >> 10) & 0x1f;
  1644‚Üí  1644‚Üí            format!("sub.w $r{}, $r{}, $r{}", rd, rj, rk)
  1645‚Üí  1645‚Üí        }
  1646‚Üí  1646‚Üí        0x02 => {
  1647‚Üí  1647‚Üí            let rd = (word >> 7) & 0x1f;
  1648‚Üí  1648‚Üí            let rj = (word >> 15) & 0x1f;
  1649‚Üí  1649‚Üí            let rk = (word >> 10) & 0x1f;
  1650‚Üí  1650‚Üí            format!("mul.w $r{}, $r{}, $r{}", rd, rj, rk)
  1651‚Üí  1651‚Üí        }
  1652‚Üí  1652‚Üí        0x05 => {
  1653‚Üí  1653‚Üí            let rd = (word >> 7) & 0x1f;
  1654‚Üí  1654‚Üí            let _rj = (word >> 15) & 0x1f; // lu12i.w does not use rj in the disassembly
  1655‚Üí  1655‚Üí            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1656‚Üí  1656‚Üí            format!("lu12i.w $r{}, {}({})", rd, si12, si12)
  1657‚Üí  1657‚Üí        }
  1658‚Üí  1658‚Üí        0x08 => {
  1659‚Üí  1659‚Üí            let rd = (word >> 7) & 0x1f;
  1660‚Üí  1660‚Üí            let rj = (word >> 15) & 0x1f;
  1661‚Üí  1661‚Üí            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1662‚Üí  1662‚Üí            format!("ld.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1663‚Üí  1663‚Üí        }
  1664‚Üí  1664‚Üí        0x0a => {
  1665‚Üí  1665‚Üí            let rd = (word >> 7) & 0x1f;
  1666‚Üí  1666‚Üí            let rj = (word >> 15) & 0x1f;
  1667‚Üí  1667‚Üí            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1668‚Üí  1668‚Üí            format!("st.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1669‚Üí  1669‚Üí        }
  1670‚Üí  1670‚Üí        0x0c => {
  1671‚Üí  1671‚Üí            let rd = (word >> 7) & 0x1f;
  1672‚Üí  1672‚Üí            let rj = (word >> 15) & 0x1f;
  1673‚Üí  1673‚Üí            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1674‚Üí  1674‚Üí            format!("ld.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1675‚Üí  1675‚Üí        }
  1676‚Üí  1676‚Üí        0x0e => {
  1677‚Üí  1677‚Üí            let rd = (word >> 7) & 0x1f;
  1678‚Üí  1678‚Üí            let rj = (word >> 15) & 0x1f;
  1679‚Üí  1679‚Üí            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1680‚Üí  1680‚Üí            format!("st.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1681‚Üí  1681‚Üí        }
  1682‚Üí  1682‚Üí        0x10 => {
  1683‚Üí  1683‚Üí            let rj = (word >> 15) & 0x1f;
  1684‚Üí  1684‚Üí            let offs = ((word >> 10) as i32) << 12 >> 10;
  1685‚Üí  1685‚Üí            format!("beq $r{}, {}({})", rj, offs, offs)
  1686‚Üí  1686‚Üí        }
  1687‚Üí  1687‚Üí        0x11 => {
  1688‚Üí  1688‚Üí            let rj = (word >> 15) & 0x1f;
  1689‚Üí  1689‚Üí            let offs = ((word >> 10) as i32) << 12 >> 10;
  1690‚Üí  1690‚Üí            format!("bne $r{}, {}({})", rj, offs, offs)
  1691‚Üí  1691‚Üí        }
  1692‚Üí  1692‚Üí        0x14 => format!("bl {}", ((word >> 10) as i32) << 12 >> 10),
  1693‚Üí  1693‚Üí        0x15 => {
  1694‚Üí  1694‚Üí            let rd = (word >> 7) & 0x1f;
  1695‚Üí  1695‚Üí            format!("jirl $r{}, {}", rd, ((word >> 10) as i32) << 12 >> 10)
  1696‚Üí  1696‚Üí        }
  1697‚Üí  1697‚Üí        _ => format!(".word {:08x}", word),
  1698‚Üí  1698‚Üí    }
  1699‚Üí  1699‚Üí}

  1721‚Üí  1721‚Üí/// Compute the stack frame size for an IR function on LoongArch64.
  1722‚Üí  1722‚Üí///
  1723‚Üí  1723‚Üí/// Sums `Alloc` instruction sizes, adds 16 bytes for the ra/fp save pair,
  1724‚Üí  1724‚Üí/// and rounds up to 16-byte alignment.
  1725‚Üí  1725‚Üífn loongarch64_compute_frame_size(func: &IRFunction) -> usize {
  1726‚Üí  1726‚Üí    let mut total: u32 = 16; // ra + fp save pair
  1727‚Üí  1727‚Üí    for block in &func.blocks {
  1728‚Üí  1728‚Üí        for instr in &block.instructions {
  1729‚Üí  1729‚Üí            if let IRInstr::Alloc { size, .. } = instr {
  1730‚Üí  1730‚Üí                let aligned = (*size).div_ceil(16) * 16;
  1731‚Üí  1731‚Üí                total += aligned;
  1732‚Üí  1732‚Üí            }
  1733‚Üí  1733‚Üí        }
  1734‚Üí  1734‚Üí    }
  1735‚Üí  1735‚Üí    // Round up to 16-byte alignment
  1736‚Üí  1736‚Üí    total = (total + 15) & !15;
  1737‚Üí  1737‚Üí    total as usize
  1738‚Üí  1738‚Üí}

  1777‚Üí  1777‚Üí/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1778‚Üí  1778‚Üífn map_vreg_to_gpr(
  1779‚Üí  1779‚Üí    vreg_id: u32,
  1780‚Üí  1780‚Üí    arg_index: Option<usize>,
  1781‚Üí  1781‚Üí    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1782‚Üí  1782‚Üí) -> Gpr {
  1783‚Üí  1783‚Üí    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1784‚Üí  1784‚Üí        return *gpr;
  1785‚Üí  1785‚Üí    }
  1786‚Üí  1786‚Üí
  1787‚Üí  1787‚Üí    // If this is an argument, map it to the corresponding argument register.
  1788‚Üí  1788‚Üí    if let Some(idx) = arg_index {
  1789‚Üí  1789‚Üí        if let Some(reg) = Gpr::arg_register(idx) {
  1790‚Üí  1790‚Üí            vreg_map.insert(vreg_id, reg);
  1791‚Üí  1791‚Üí            return reg;
  1792‚Üí  1792‚Üí        }
  1793‚Üí  1793‚Üí    }
  1794‚Üí  1794‚Üí
  1795‚Üí  1795‚Üí    // Otherwise, pick the first available allocatable register.
  1796‚Üí  1796‚Üí    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1797‚Üí  1797‚Üí    for &reg in ALLOCATABLE_GPRS {
  1798‚Üí  1798‚Üí        if !used.contains(&reg) {
  1799‚Üí  1799‚Üí            vreg_map.insert(vreg_id, reg);
  1800‚Üí  1800‚Üí            return reg;
  1801‚Üí  1801‚Üí        }
  1802‚Üí  1802‚Üí    }
  1803‚Üí  1803‚Üí
  1804‚Üí  1804‚Üí    // Fallback: use t8 as a scratch register.
  1805‚Üí  1805‚Üí    Gpr::T8
  1806‚Üí  1806‚Üí}

  1816‚Üí  1816‚Üí/// Load a 64-bit immediate into a register using the lu12i.w/ori/lu32i.d/lu52i.d
  1817‚Üí  1817‚Üí/// instruction sequence.
  1818‚Üí  1818‚Üí///
  1819‚Üí  1819‚Üí/// Returns the list of AllocatedInstructions emitted to perform the load.
  1820‚Üí  1820‚Üífn load_imm_la64(rd: Gpr, imm: i64) -> Vec<AllocatedInstruction> {
  1821‚Üí  1821‚Üí    let mut result = Vec::new();
  1822‚Üí  1822‚Üí    let val = imm as u64;
  1823‚Üí  1823‚Üí
  1824‚Üí  1824‚Üí    // lu12i.w rd, bits[31:12]
  1825‚Üí  1825‚Üí    let hi20 = ((val >> 12) & 0xFFFFF) as i32;
  1826‚Üí  1826‚Üí    result.push(emit_alloc_instr(
  1827‚Üí  1827‚Üí        Instruction::Lu12iW { rd, imm20: hi20 },
  1828‚Üí  1828‚Üí        vec![],
  1829‚Üí  1829‚Üí        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1830‚Üí  1830‚Üí    ));
  1831‚Üí  1831‚Üí
  1832‚Üí  1832‚Üí    // ori rd, rd, bits[11:0]
  1833‚Üí  1833‚Üí    let lo12 = (val & 0xFFF) as u32;
  1834‚Üí  1834‚Üí    result.push(emit_alloc_instr(
  1835‚Üí  1835‚Üí        Instruction::Ori {
  1836‚Üí  1836‚Üí            rd,
  1837‚Üí  1837‚Üí            rj: rd,
  1838‚Üí  1838‚Üí            imm12: lo12,
  1839‚Üí  1839‚Üí        },
  1840‚Üí  1840‚Üí        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1841‚Üí  1841‚Üí        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1842‚Üí  1842‚Üí    ));
  1843‚Üí  1843‚Üí
  1844‚Üí  1844‚Üí    // lu32i.d is actually pcaddi in QEMU ‚Äî skip it
  1845‚Üí  1845‚Üí    // Instead, handle upper bits differently
  1846‚Üí  1846‚Üí
  1847‚Üí  1847‚Üí    let lower32 = val & 0xFFFFFFFF;
  1848‚Üí  1848‚Üí    let sign_ext = if lower32 & 0x80000000 != 0 {
  1849‚Üí  1849‚Üí        0xFFFFFFFF00000000u64
  1850‚Üí  1850‚Üí    } else {
  1851‚Üí  1851‚Üí        0u64
  1852‚Üí  1852‚Üí    };
  1853‚Üí  1853‚Üí
  1854‚Üí  1854‚Üí    if val >> 32 != sign_ext >> 32 {
  1855‚Üí  1855‚Üí        // Upper bits don't match sign extension
  1856‚Üí  1856‚Üí        if val >> 32 == 0 && lower32 & 0x80000000 != 0 {
  1857‚Üí  1857‚Üí            // Zero-extend: slli.d + srli.d
  1858‚Üí  1858‚Üí            result.push(emit_alloc_instr(
  1859‚Üí  1859‚Üí                Instruction::SlliD { rd, rj: rd, imm8: 32 },
  1860‚Üí  1860‚Üí                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1861‚Üí  1861‚Üí                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1862‚Üí  1862‚Üí            ));
  1863‚Üí  1863‚Üí            result.push(emit_alloc_instr(
  1864‚Üí  1864‚Üí                Instruction::SrliD { rd, rj: rd, imm8: 32 },
  1865‚Üí  1865‚Üí                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1866‚Üí  1866‚Üí                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1867‚Üí  1867‚Üí            ));
  1868‚Üí  1868‚Üí        } else {
  1869‚Üí  1869‚Üí            // Full 64-bit: use lu52i.d for bits[63:52] then slli+srli+or for bits[51:32]
  1870‚Üí  1870‚Üí            let hi52 = ((val >> 52) & 0xFFF) as i32;
  1871‚Üí  1871‚Üí            result.push(emit_alloc_instr(
  1872‚Üí  1872‚Üí                Instruction::Lu52iD {
  1873‚Üí  1873‚Üí                    rd,
  1874‚Üí  1874‚Üí                    rj: rd,
  1875‚Üí  1875‚Üí                    imm12: hi52,
  1876‚Üí  1876‚Üí                },
  1877‚Üí  1877‚Üí                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1878‚Üí  1878‚Üí                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1879‚Üí  1879‚Üí            ));
  1880‚Üí  1880‚Üí            let bits_51_32 = ((val >> 32) & 0xFFFFF) as u32;
  1881‚Üí  1881‚Üí            if bits_51_32 != 0 {
  1882‚Üí  1882‚Üí                let temp = Gpr::T0;
  1883‚Üí  1883‚Üí                let hi_51_32 = ((val >> 32) & 0xFFFFF) as i32;
  1884‚Üí  1884‚Üí                result.push(emit_alloc_instr(
  1885‚Üí  1885‚Üí                    Instruction::Lu12iW { rd: temp, imm20: hi_51_32 },
  1886‚Üí  1886‚Üí                    vec![],
  1887‚Üí  1887‚Üí                    vec![PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1888‚Üí  1888‚Üí                ));
  1889‚Üí  1889‚Üí                result.push(emit_alloc_instr(
  1890‚Üí  1890‚Üí                    Instruction::SlliD { rd: temp, rj: temp, imm8: 32 },
  1891‚Üí  1891‚Üí                    vec![PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1892‚Üí  1892‚Üí                    vec![PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1893‚Üí  1893‚Üí                ));
  1894‚Üí  1894‚Üí                result.push(emit_alloc_instr(
  1895‚Üí  1895‚Üí                    Instruction::SlliD { rd, rj: rd, imm8: 32 },
  1896‚Üí  1896‚Üí                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1897‚Üí  1897‚Üí                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1898‚Üí  1898‚Üí                ));
  1899‚Üí  1899‚Üí                result.push(emit_alloc_instr(
  1900‚Üí  1900‚Üí                    Instruction::SrliD { rd, rj: rd, imm8: 32 },
  1901‚Üí  1901‚Üí                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1902‚Üí  1902‚Üí                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1903‚Üí  1903‚Üí                ));
  1904‚Üí  1904‚Üí                result.push(emit_alloc_instr(
  1905‚Üí  1905‚Üí                    Instruction::Or { rd, rj: rd, rk: temp },
  1906‚Üí  1906‚Üí                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding()), PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1907‚Üí  1907‚Üí                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1908‚Üí  1908‚Üí                ));
  1909‚Üí  1909‚Üí            }
  1910‚Üí  1910‚Üí        }
  1911‚Üí  1911‚Üí    }
  1912‚Üí  1912‚Üí
  1913‚Üí  1913‚Üí    result
  1914‚Üí  1914‚Üí}

  1916‚Üí  1916‚Üí/// Resolve an IRValue to a physical GPR.
  1917‚Üí  1917‚Üí///
  1918‚Üí  1918‚Üí/// - For `IRValue::Register`: looks up in `vreg_map`.
  1919‚Üí  1919‚Üí/// - For `IRValue::Immediate`: loads the value into `scratch` and returns
  1920‚Üí  1920‚Üí///   `(scratch, instructions)`.
  1921‚Üí  1921‚Üí/// - For `IRValue::Address`: loads the address into `scratch`.
  1922‚Üí  1922‚Üí/// - For `IRValue::Label`: loads a placeholder into `scratch`.
  1923‚Üí  1923‚Üífn resolve_gpr_la64(
  1924‚Üí  1924‚Üí    val: &IRValue,
  1925‚Üí  1925‚Üí    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1926‚Üí  1926‚Üí    scratch: Gpr,
  1927‚Üí  1927‚Üí) -> (Gpr, Vec<AllocatedInstruction>) {
