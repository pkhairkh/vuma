     1→//! # VUMA Compiler API for LLM Consumption
     2→//!
     3→//! This module provides a clean, programmatic interface designed for LLM agents
     4→//! and automated tools that need to compile VUMA source code and receive
     5→//! structured, machine-readable results.
     6→//!
     7→//! ## Design Principles
     8→//!
     9→//! - **Always succeeds**: Every method returns a structured result; errors are
    10→//!   captured as diagnostics rather than panicking or returning `Err`.
    11→//! - **Serializable**: All result types derive `Serialize` / `Deserialize` so
    12→//!   they can be sent over JSON-based IPC channels.
    13→//! - **Summaries, not internals**: The API returns enough information for an
    14→//!   LLM to reason about the program (function signatures, call graph, node
    15→//!   counts) without exposing the full graph representation.
    16→//! - **Target-aware**: The API can compile for any supported backend
    17→//!   (x86_64, AArch64, RISC-V, etc.) and returns target-specific outputs.
    18→//!
    19→//! ## Quick Start
    20→//!
    21→//! ```rust
    22→//! use vuma::api::VumaCompiler;
    23→//!
    24→//! let compiler = VumaCompiler::new();
    25→//! let source = r#"
    26→//!     fn main() {
    27→//!         x = 1 + 2;
    28→//!     }
    29→//! "#;
    30→//!
    31→//! let result = compiler.compile(source);
    32→//! if result.success {
    33→//!     println!("Compiled {} functions", result.scg.unwrap().function_count);
    34→//! } else {
    35→//!     for diag in &result.diagnostics {
    36→//!         println!("[{}] {}", diag.severity, diag.message);
    37→//!     }
    38→//! }
    39→//! ```

    56→/// The primary compiler API for programmatic (LLM) consumption.
    57→///
    58→/// `VumaCompiler` is the main entry point for LLMs and automated tools.
    59→/// It wraps the full VUMA compilation pipeline and returns structured results
    60→/// that are easy to parse and reason about programmatically.
    61→#[derive(Debug, Clone)]
    62→pub struct VumaCompiler {
    63→    config: CompileConfig,
    64→}
    65→
    66→impl VumaCompiler {
    67→    /// Create a new compiler with default configuration.
    68→    pub fn new() -> Self {
    69→        Self {
    70→            config: CompileConfig::default(),
    71→        }
    72→    }
    73→
    74→    /// Create a compiler with a specific configuration.
    75→    pub fn with_config(config: CompileConfig) -> Self {
    76→        Self { config }
    77→    }

    79→    /// Compile from source string. Returns a structured result.
    80→    ///
    81→    /// This runs the full compilation pipeline:
    82→    /// Parse → AST → SCG → BD Inference → MSG → IVE Verification
    83→    /// → SCG Transforms → IR Lowering → Register Allocation → Code Emission
    84→    ///
    85→    /// The result always contains a value — check `result.success` to
    86→    /// determine if compilation succeeded, and inspect `result.diagnostics`
    87→    /// for any warnings or errors.
    88→    pub fn compile(&self, source: &str) -> CompileResult {
    89→        let start = Instant::now();
    90→        let source_lines = source.lines().count();
    91→        let source_bytes = source.len();
    92→
    93→        match pipeline::compile(source, &self.config) {
    94→            Ok(output) => {
    95→                let scg_summary = Some(build_scg_summary(&output.scg));
    96→
    97→                // Get disassembly from the binary using the default backend
    98→                let disasm = disassemble_default(&output.binary);
    99→
   100→                let target_output = Some(TargetOutput {
   101→                    backend: "aarch64".to_string(),
   102→                    binary: output.binary.clone(),
   103→                    binary_size: output.binary.len(),
   104→                    disassembly: disasm,
   105→                });
   106→
   107→                // Collect any non-fatal diagnostics
   108→                let diagnostics = Vec::new();
   109→
   110→                CompileResult {
   111→                    success: true,
   112→                    diagnostics,
   113→                    scg: scg_summary,
   114→                    target: target_output,
   115→                    metadata: CompileMetadata {
   116→                        compile_time_ms: start.elapsed().as_millis() as u64,
   117→                        source_lines,
   118→                        source_bytes,
   119→                    },
   120→                }
   121→            }
   122→            Err(errors) => {
   123→                let diagnostics = errors
   124→                    .iter()
   125→                    .flat_map(|e| diagnostics::from_vuma_error(e))
   126→                    .collect();
   127→
   128→                CompileResult {
   129→                    success: false,
   130→                    diagnostics,
   131→                    scg: None,
   132→                    target: None,
   133→                    metadata: CompileMetadata {
   134→                        compile_time_ms: start.elapsed().as_millis() as u64,
   135→                        source_lines,
   136→                        source_bytes,
   137→                    },
   138→                }
   139→            }
   140→        }
   141→    }
   142→
   143→    /// Compile for a specific target backend.
   144→    ///
   145→    /// Valid target strings: `"x86_64"`, `"aarch64"`, `"riscv64"`,
   146→    /// `"wasm32"`, `"loongarch64"`, `"arm32"`, `"mips64"`, `"ppc64"`.
   147→    ///
   148→    /// If the target string is not recognised, the result will contain
   149→    /// error diagnostics.
   150→    pub fn compile_for_target(&self, source: &str, target: &str) -> CompileResult {
   151→        let start = Instant::now();
   152→        let source_lines = source.lines().count();
   153→        let source_bytes = source.len();
   154→
   155→        // Parse the target string into a BackendKind
   156→        let backend_kind = match parse_target(target) {
   157→            Some(kind) => kind,
   158→            None => {
   159→                return CompileResult {
   160→                    success: false,
   161→                    diagnostics: vec![VumaDiagnostic::new(
   162→                        "E021",
   163→                        DiagnosticSeverity::Error,
   164→                        format!(
   165→                            "Unknown target '{}'. Available: x86_64, aarch64, riscv64, \
   166→                             wasm32, loongarch64, arm32, mips64, ppc64",
   167→                            target
   168→                        ),
   169→                        "target-selection",
   170→                        DiagnosticSourceLocation::unknown(),
   171→                    )],
   172→                    scg: None,
   173→                    target: None,
   174→                    metadata: CompileMetadata {
   175→                        compile_time_ms: start.elapsed().as_millis() as u64,
   176→                        source_lines,
   177→                        source_bytes,
   178→                    },
   179→                };
   180→            }
   181→        };
   182→
   183→        // Run the front-end pipeline (parse through SCG transforms)
   184→        let front_result = run_frontend(source, &self.config);
   185→
   186→        let (scg, mut diagnostics) = match front_result {
   187→            FrontendResult::Ok { scg } => (scg, Vec::new()),
   188→            FrontendResult::Err { diagnostics } => {
   189→                return CompileResult {
   190→                    success: false,
   191→                    diagnostics,
   192→                    scg: None,
   193→                    target: None,
   194→                    metadata: CompileMetadata {
   195→                        compile_time_ms: start.elapsed().as_millis() as u64,
   196→                        source_lines,
   197→                        source_bytes,
   198→                    },
   199→                };
   200→            }
   201→        };
   202→
   203→        // Build SCG summary from the validated SCG
   204→        let scg_summary = Some(build_scg_summary(&scg));
   205→
   206→        // Run target-specific codegen
   207→        let target_output = match run_backend_codegen(&scg, backend_kind) {
   208→            Ok(output) => Some(output),
   209→            Err(diags) => {
   210→                diagnostics.extend(diags);
   211→                return CompileResult {
   212→                    success: false,
   213→                    diagnostics,
   214→                    scg: scg_summary,
   215→                    target: None,
   216→                    metadata: CompileMetadata {
   217→                        compile_time_ms: start.elapsed().as_millis() as u64,
   218→                        source_lines,
   219→                        source_bytes,
   220→                    },
   221→                };
   222→            }
   223→        };
   224→
   225→        CompileResult {
   226→            success: true,
   227→            diagnostics,
   228→            scg: scg_summary,
   229→            target: target_output,
   230→            metadata: CompileMetadata {
   231→                compile_time_ms: start.elapsed().as_millis() as u64,
   232→                source_lines,
   233→                source_bytes,
   234→            },
   235→        }
   236→    }

   238→    /// Just parse — return AST/SCG without codegen.
   239→    ///
   240→    /// Useful for LLMs that want to understand the program structure
   241→    /// without incurring the cost of full code generation.
   242→    pub fn parse(&self, source: &str) -> ParseResult {
   243→        let start = Instant::now();
   244→
   245→        use vuma_parser::{AstToScg, Parser};
   246→
   247→        // Parse source to AST
   248→        let mut parser = Parser::new(source);
   249→        let parse_output = parser.parse_program();
   250→
   251→        if parse_output.has_errors() {
   252→            let diagnostics = diagnostics::from_parse_errors(&parse_output.errors, source, None);
   253→
   254→            return ParseResult {
   255→                success: false,
   256→                diagnostics,
   257→                ast_summary: None,
   258→                scg: None,
   259→                metadata: CompileMetadata {
   260→                    compile_time_ms: start.elapsed().as_millis() as u64,
   261→                    source_lines: source.lines().count(),
   262→                    source_bytes: source.len(),
   263→                },
   264→            };
   265→        }
   266→
   267→        let ast = parse_output.unwrap();
   268→        let ast_summary = Some(build_ast_summary(&ast));
   269→
   270→        // Convert AST to SCG
   271→        let mut converter = AstToScg::new();
   272→        match converter.convert(&ast) {
   273→            Ok(scg) => {
   274→                let scg_summary = Some(build_scg_summary(&scg));
   275→                ParseResult {
   276→                    success: true,
   277→                    diagnostics: Vec::new(),
   278→                    ast_summary,
   279→                    scg: scg_summary,
   280→                    metadata: CompileMetadata {
   281→                        compile_time_ms: start.elapsed().as_millis() as u64,
   282→                        source_lines: source.lines().count(),
   283→                        source_bytes: source.len(),
   284→                    },
   285→                }
   286→            }
   287→            Err(e) => {
   288→                let diagnostics = vec![VumaDiagnostic::new(
   289→                    "E019",
   290→                    DiagnosticSeverity::Error,
   291→                    format!("{}", e),
   292→                    "ast-to-scg",
   293→                    DiagnosticSourceLocation::unknown(),
   294→                )];
   295→                ParseResult {
   296→                    success: false,
   297→                    diagnostics,
   298→                    ast_summary,
   299→                    scg: None,
   300→                    metadata: CompileMetadata {
   301→                        compile_time_ms: start.elapsed().as_millis() as u64,
   302→                        source_lines: source.lines().count(),
   303→                        source_bytes: source.len(),
   304→                    },
   305→                }
   306→            }
   307→        }
   308→    }
   309→
   310→    /// Get SCG summary for a source string.
   311→    ///
   312→    /// This runs the front-end pipeline (parse + SCG construction +
   313→    /// validation + BD inference + SCG transforms) but skips codegen.
   314→    /// It is faster than `compile()` and is ideal for program analysis.
   315→    pub fn analyze(&self, source: &str) -> ScgSummary {
   316→        let front_result = run_frontend(source, &self.config);
   317→        match front_result {
   318→            FrontendResult::Ok { scg, .. } => build_scg_summary(&scg),
   319→            FrontendResult::Err { .. } => ScgSummary {
   320→                function_count: 0,
   321→                functions: Vec::new(),
   322→                total_nodes: 0,
   323→                total_edges: 0,
   324→            },
   325→        }
   326→    }

   328→    /// List available targets.
   329→    ///
   330→    /// Returns information about every backend the compiler supports.
   331→    pub fn available_targets(&self) -> Vec<ApiTargetInfo> {
   332→        use vuma_codegen::backend::{create_backend, BackendKind};
   333→
   334→        let all_kinds = [
   335→            BackendKind::AArch64,
   336→            BackendKind::X86_64,
   337→            BackendKind::RiscV64,
   338→            BackendKind::Wasm32,
   339→            BackendKind::LoongArch64,
   340→            BackendKind::Arm32,
   341→            BackendKind::Mips64,
   342→            BackendKind::PowerPC64,
   343→        ];
   344→
   345→        all_kinds
   346→            .iter()
   347→            .filter_map(|&kind| {
   348→                create_backend(kind).ok().map(|backend| {
   349→                    let info = backend.target_info();
   350→                    ApiTargetInfo {
   351→                        name: kind.isa_name().to_string(),
   352→                        triple: info.target_triple().to_string(),
   353→                        pointer_width: info.pointer_width() * 8, // bytes → bits
   354→                        endianness: match info.endianness() {
   355→                            vuma_codegen::backend::Endianness::Little => "little".to_string(),
   356→                            vuma_codegen::backend::Endianness::Big => "big".to_string(),
   357→                            vuma_codegen::backend::Endianness::Bi => "bi".to_string(),
   358→                        },
   359→                        output_format: match info.output_format() {
   360→                            vuma_codegen::backend::OutputFormat::Elf64 => "elf64".to_string(),
   361→                            vuma_codegen::backend::OutputFormat::Elf32 => "elf32".to_string(),
   362→                            vuma_codegen::backend::OutputFormat::WasmBinary => "wasm".to_string(),
   363→                            vuma_codegen::backend::OutputFormat::RawBinary => "raw".to_string(),
   364→                        },
   365→                    }
   366→                })
   367→            })
   368→            .collect()
   369→    }

   371→    /// Validate source without full compilation.
   372→    ///
   373→    /// Runs parsing and SCG validation, returning a list of diagnostics.
   374→    /// This is the fastest way to check if a program is well-formed.
   375→    pub fn validate(&self, source: &str) -> Vec<VumaDiagnostic> {
   376→        use vuma_parser::{AstToScg, Parser};
   377→
   378→        let mut all_diagnostics = Vec::new();
   379→
   380→        // Parse
   381→        let mut parser = Parser::new(source);
   382→        let parse_output = parser.parse_program();
   383→
   384→        if parse_output.has_errors() {
   385→            all_diagnostics.extend(diagnostics::from_parse_errors(
   386→                &parse_output.errors,
   387→                source,
   388→                None,
   389→            ));
   390→            return all_diagnostics;
   391→        }
   392→
   393→        let ast = parse_output.unwrap();
   394→
   395→        // AST → SCG
   396→        let mut converter = AstToScg::new();
   397→        let scg = match converter.convert(&ast) {
   398→            Ok(scg) => scg,
   399→            Err(e) => {
   400→                all_diagnostics.push(VumaDiagnostic::new(
   401→                    "E019",
   402→                    DiagnosticSeverity::Error,
   403→                    format!("{}", e),
   404→                    "ast-to-scg",
   405→                    DiagnosticSourceLocation::unknown(),
   406→                ));
   407→                return all_diagnostics;
   408→            }
   409→        };
   410→
   411→        // Validate SCG
   412→        let validation = scg.validate();
   413→        if !validation.is_valid {
   414→            for err in &validation.errors {
   415→                all_diagnostics.push(VumaDiagnostic::new(
   416→                    "E022",
   417→                    DiagnosticSeverity::Error,
   418→                    err.clone(),
   419→                    "scg-validation",
   420→                    DiagnosticSourceLocation::unknown(),
   421→                ));
   422→            }
   423→        }
   424→
   425→        all_diagnostics
   426→    }
   427→}

   439→/// Result of compiling a VUMA program.
   440→///
   441→/// This is the primary return type for `VumaCompiler::compile()` and
   442→/// `VumaCompiler::compile_for_target()`. It always contains a value —
   443→/// check `success` to determine the outcome.
   444→#[derive(Debug, Clone, Serialize, Deserialize)]
   445→pub struct CompileResult {
   446→    /// Whether compilation succeeded (produced a binary).
   447→    pub success: bool,
   448→    /// Any diagnostics (errors, warnings, notes) produced during compilation.
   449→    pub diagnostics: Vec<VumaDiagnostic>,
   450→    /// Summary of the SCG (Semantic Computation Graph).
   451→    ///
   452→    /// Present when parsing succeeds, even if codegen fails.
   453→    pub scg: Option<ScgSummary>,
   454→    /// Compiled output for the target.
   455→    ///
   456→    /// Present only when compilation succeeds.
   457→    pub target: Option<TargetOutput>,
   458→    /// Metadata about the compilation process.
   459→    pub metadata: CompileMetadata,
   460→}
   461→
   462→/// Result of parsing (without codegen).
   463→#[derive(Debug, Clone, Serialize, Deserialize)]

   465→    /// Whether parsing succeeded.
   466→    pub success: bool,
   467→    /// Any diagnostics produced during parsing.
   468→    pub diagnostics: Vec<VumaDiagnostic>,
   469→    /// Summary of the AST structure.
   470→    pub ast_summary: Option<AstSummary>,
   471→    /// Summary of the SCG (if AST → SCG conversion succeeded).
   472→    pub scg: Option<ScgSummary>,
   473→    /// Metadata about the parse process.
   474→    pub metadata: CompileMetadata,
   475→}

   589→/// Serialize `Vec<u8>` as a hex string for compact JSON representation.
   590→fn serialize_binary_hex<S: serde::Serializer>(data: &Vec<u8>, s: S) -> Result<S::Ok, S::Error> {
   591→    let hex: String = data.iter().map(|b| format!("{:02x}", b)).collect();
   592→    s.serialize_str(&hex)
   593→}
   594→
   595→/// Deserialize `Vec<u8>` from a hex string.
   596→fn deserialize_binary_hex<'de, D: serde::Deserializer<'de>>(d: D) -> Result<Vec<u8>, D::Error> {
   597→    use serde::de::Error;
   598→
   599→    struct HexVisitor;
   600→
   601→    impl<'de> serde::de::Visitor<'de> for HexVisitor {
   602→        type Value = Vec<u8>;
   603→
   604→        fn expecting(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   605→            write!(f, "a hex-encoded string")
   606→        }
   607→
   608→        fn visit_str<E: serde::de::Error>(self, v: &str) -> Result<Self::Value, E> {
   609→            if v.len() % 2 != 0 {
   610→                return Err(E::custom("hex string has odd length"));
   611→            }
   612→            (0..v.len())
   613→                .step_by(2)
   614→                .map(|i| {
   615→                    u8::from_str_radix(&v[i..i + 2], 16)
   616→                        .map_err(|e| E::custom(format!("hex decode error: {}", e)))
   617→                })
   618→                .collect()
   619→        }
   620→    }
   621→
   622→    d.deserialize_str(HexVisitor)
   623→}

   629→/// Front-end pipeline result (everything up to and including SCG transforms).
   630→enum FrontendResult {
   631→    Ok {
   632→        scg: vuma_scg::SCG,
   633→    },
   634→    Err {
   635→        diagnostics: Vec<VumaDiagnostic>,
   636→    },
   637→}

   639→/// Run the front-end pipeline: Parse → AST → SCG → Validate → BD Inference
   640→/// → IVE Verification → SCG Transforms.
   641→fn run_frontend(source: &str, config: &CompileConfig) -> FrontendResult {
   642→    use vuma_ive::{InvariantAggregator, VerificationLevel as IveVerificationLevel};
   643→    use vuma_parser::{AstToScg, Parser};
   644→
   645→    // Stage 1: Parse
   646→    let mut parser = Parser::new(source);
   647→    let parse_output = parser.parse_program();
   648→    if parse_output.has_errors() {
   649→        return FrontendResult::Err {
   650→            diagnostics: diagnostics::from_parse_errors(&parse_output.errors, source, None),
   651→        };
   652→    }
   653→    let ast = parse_output.unwrap();
   654→
   655→    // Stage 2: AST → SCG
   656→    let mut converter = AstToScg::new();
   657→    let mut scg = match converter.convert(&ast) {
   658→        Ok(scg) => scg,
   659→        Err(e) => {
   660→            return FrontendResult::Err {
   661→                diagnostics: vec![VumaDiagnostic::new(
   662→                    "E019",
   663→                    DiagnosticSeverity::Error,
   664→                    format!("{}", e),
   665→                    "ast-to-scg",
   666→                    DiagnosticSourceLocation::unknown(),
   667→                )],
   668→            };
   669→        }
   670→    };
   671→
   672→    // Stage 3: SCG Validation (non-fatal — warn but continue)
   673→    let _validation = scg.validate();
   674→
   675→    // Stage 4: BD Inference (refine types)
   676→    let inference_engine = vuma_ive::InferenceEngine::new();
   677→    let bd_results = inference_engine.infer_types(&scg);
   678→    pipeline::refine_scg_types_with_bd(&mut scg, &bd_results);
   679→
   680→    // Stage 5: IVE Verification (non-fatal)
   681→    if config.verification_level != VerificationLevel::None {
   682→        let ive_level = match config.verification_level {
   683→            VerificationLevel::Quick => IveVerificationLevel::Quick,
   684→            VerificationLevel::Normal => IveVerificationLevel::Normal,
   685→            VerificationLevel::Exhaustive => IveVerificationLevel::Exhaustive,
   686→            VerificationLevel::None => unreachable!(),
   687→        };
   688→        let aggregator = InvariantAggregator::new().with_level(ive_level);
   689→        let input = vuma_ive::verification::VerificationInput::from_scg(scg.clone());
   690→        let _ = aggregator.verify_all(&input);
   691→    }
   692→
   693→    // Stage 6: SCG Transforms
   694→    pipeline::run_scg_transforms(&mut scg, config);
   695→
   696→    FrontendResult::Ok { scg }
   697→}

   699→/// Run target-specific codegen using the Backend trait.
   700→fn run_backend_codegen(
   701→    scg: &vuma_scg::SCG,
   702→    backend_kind: vuma_codegen::backend::BackendKind,
   703→) -> Result<TargetOutput, Vec<VumaDiagnostic>> {
   704→    use vuma_codegen::backend::{create_backend, AllocatedProgram};
   705→    use vuma_codegen::regalloc::RegAllocator;
   706→    use vuma_codegen::scg_to_ir::IRBuilder;
   707→
   708→    // Bridge SCG to codegen SCG
   709→    let codegen_scg = pipeline::bridge_scg_to_codegen(scg);
   710→
   711→    // IR Lowering
   712→    let mut ir_builder = IRBuilder::new();
   713→    let ir_program = match ir_builder.build(&codegen_scg) {
   714→        Ok(ir) => ir,
   715→        Err(e) => {
   716→            return Err(vec![diagnostics::from_codegen_error(&e)]);
   717→        }
   718→    };
   719→
   720→    // Create backend
   721→    let backend = match create_backend(backend_kind) {
   722→        Ok(b) => b,
   723→        Err(e) => {
   724→            return Err(vec![VumaDiagnostic::new(
   725→                "E023",
   726→                DiagnosticSeverity::Error,
   727→                format!("{}", e),
   728→                "backend-creation",
   729→                DiagnosticSourceLocation::unknown(),
   730→            )]);
   731→        }
   732→    };
   733→
   734→    // Register allocation — delegate to the backend
   735→    let mut allocated_functions = Vec::new();
   736→    for func in &ir_program.functions {
   737→        match backend.allocate_registers(func) {
   738→            Ok(allocd) => allocated_functions.push(allocd),
   739→            Err(e) => {
   740→                return Err(vec![VumaDiagnostic::new(
   741→                    "E017",
   742→                    DiagnosticSeverity::Error,
   743→                    format!("{}: {}", func.name, e),
   744→                    "register-alloc",
   745→                    DiagnosticSourceLocation::unknown(),
   746→                )]);
   747→            }
   748→        }
   749→    }
   750→
   751→    let allocated_program = AllocatedProgram {
   752→        functions: allocated_functions,
   753→        total_code_size: 0,
   754→        total_data_size: 0,
   755→    };
   756→
   757→    // Encode the program
   758→    let binary = match backend.encode_program(&allocated_program) {
   759→        Ok(binary) => binary,
   760→        Err(e) => {
   761→            return Err(vec![VumaDiagnostic::new(
   762→                "E020",
   763→                DiagnosticSeverity::Error,
   764→                format!("{}", e),
   765→                "code-emission",
   766→                DiagnosticSourceLocation::unknown(),
   767→            )]);
   768→        }
   769→    };
   770→
   771→    // Disassemble
   772→    let base_addr = backend.target_info().default_base_address();
   773→    let disasm_lines = backend.disassemble(&binary, base_addr);
   774→    let disassembly = disasm_lines.join("\n");
   775→
   776→    Ok(TargetOutput {
   777→        backend: backend_kind.isa_name().to_string(),
   778→        binary_size: binary.len(),
   779→        binary,
   780→        disassembly,
   781→    })
   782→}

   784→/// Build an edge index from the SCG for efficient traversal.
   785→struct ScgEdgeIndex {
   786→    outgoing: HashMap<vuma_scg::NodeId, Vec<vuma_scg::EdgeData>>,
   787→    incoming: HashMap<vuma_scg::NodeId, Vec<vuma_scg::EdgeData>>,
   788→}
   789→
   790→impl ScgEdgeIndex {
   791→    fn build(scg: &vuma_scg::SCG) -> Self {
   792→        let mut outgoing: HashMap<vuma_scg::NodeId, Vec<vuma_scg::EdgeData>> = HashMap::new();
   793→        let mut incoming: HashMap<vuma_scg::NodeId, Vec<vuma_scg::EdgeData>> = HashMap::new();
   794→        for edge in scg.edges() {
   795→            outgoing.entry(edge.source).or_default().push(edge.clone());
   796→            incoming.entry(edge.target).or_default().push(edge.clone());
   797→        }
   798→        Self { outgoing, incoming }
   799→    }
   800→
   801→    fn outgoing(&self, id: vuma_scg::NodeId) -> &[vuma_scg::EdgeData] {
   802→        self.outgoing.get(&id).map(|v| v.as_slice()).unwrap_or(&[])
   803→    }
   804→
   805→    fn incoming(&self, id: vuma_scg::NodeId) -> &[vuma_scg::EdgeData] {
   806→        self.incoming.get(&id).map(|v| v.as_slice()).unwrap_or(&[])
   807→    }
   808→}

   810→/// Build an SCG summary from the full SCG graph.
   811→fn build_scg_summary(scg: &vuma_scg::SCG) -> ScgSummary {
   812→    use vuma_scg::{ControlKind, EdgeKind, NodePayload};
   813→
   814→    let total_nodes = scg.node_count();
   815→    let total_edges = scg.edge_count();
   816→    let edge_idx = ScgEdgeIndex::build(scg);
   817→
   818→    // Identify function entry nodes and build per-function summaries
   819→    let mut functions = Vec::new();
   820→
   821→    // Collect function entries
   822→    let func_entries: Vec<(vuma_scg::NodeId, String)> = scg
   823→        .nodes()
   824→        .filter_map(|n| {
   825→            if let NodePayload::Control(c) = &n.payload {
   826→                if c.kind == ControlKind::FunctionEntry {
   827→                    let name = c.label.clone().unwrap_or_else(|| "unknown".to_string());
   828→                    return Some((n.id, name));
   829→                }
   830→            }
   831→            None
   832→        })
   833→        .collect();
   834→
   835→    // For each function entry, count nodes reachable via ControlFlow
   836→    for (entry_id, func_name) in &func_entries {
   837→        let mut reachable = std::collections::HashSet::new();
   838→        let mut queue = std::collections::VecDeque::new();
   839→        queue.push_back(*entry_id);
   840→        reachable.insert(*entry_id);
   841→
   842→        while let Some(current) = queue.pop_front() {
   843→            for edge in edge_idx.outgoing(current) {
   844→                if edge.kind == EdgeKind::ControlFlow && !reachable.contains(&edge.target) {
   845→                    reachable.insert(edge.target);
   846→                    queue.push_back(edge.target);
   847→                }
   848→            }
   849→        }
   850→
   851→        let node_count = reachable.len();
   852→
   853→        // Extract parameters from outgoing DataFlow edges from the entry
   854→        let mut params = Vec::new();
   855→        for edge in edge_idx.outgoing(*entry_id) {
   856→            if edge.kind == EdgeKind::DataFlow {
   857→                if let Some(target_node) = scg.get_node(edge.target) {
   858→                    let name = match &target_node.payload {
   859→                        NodePayload::Allocation(a) => {
   860→                            a.type_name.clone().unwrap_or_else(|| "param".to_string())
   861→                        }
   862→                        NodePayload::Computation(c) => {
   863→                            c.result_type.clone().unwrap_or_else(|| "param".to_string())
   864→                        }
   865→                        _ => "param".to_string(),
   866→                    };
   867→                    let ty = match &target_node.payload {
   868→                        NodePayload::Allocation(a) => {
   869→                            a.type_name.as_deref().unwrap_or("i64").to_string()
   870→                        }
   871→                        NodePayload::Computation(c) => {
   872→                            c.result_type.as_deref().unwrap_or("i64").to_string()
   873→                        }
   874→                        NodePayload::Cast(c) => c.to_type.clone(),
   875→                        _ => "i64".to_string(),
   876→                    };
   877→                    params.push((name, ty));
   878→                }
   879→            }
   880→        }
   881→
   882→        // Find calls within this function
   883→        let mut calls = Vec::new();
   884→        for node_id in &reachable {
   885→            if let Some(node) = scg.get_node(*node_id) {
   886→                if let NodePayload::Computation(comp) = &node.payload {
   887→                    if !is_known_binop(&comp.operation) && !comp.operation.starts_with('_') {
   888→                        if !calls.contains(&comp.operation) {
   889→                            calls.push(comp.operation.clone());
   890→                        }
   891→                    }
   892→                }
   893→            }
   894→        }
   895→
   896→        // Determine return type
   897→        let return_type = find_return_type(scg, *entry_id, &edge_idx);
   898→
   899→        functions.push(FunctionSummary {
   900→            name: func_name.clone(),
   901→            params,
   902→            return_type,
   903→            node_count,
   904→            calls,
   905→        });
   906→    }
   907→
   908→    // If no function entries were found, create a single "main" summary
   909→    if functions.is_empty() {
   910→        let mut calls = Vec::new();
   911→        for node in scg.nodes() {
   912→            if let NodePayload::Computation(comp) = &node.payload {
   913→                if !is_known_binop(&comp.operation) && !comp.operation.starts_with('_') {
   914→                    if !calls.contains(&comp.operation) {
   915→                        calls.push(comp.operation.clone());
   916→                    }
   917→                }
   918→            }
   919→        }
   920→
   921→        functions.push(FunctionSummary {
   922→            name: "main".to_string(),
   923→            params: Vec::new(),
   924→            return_type: "void".to_string(),
   925→            node_count: total_nodes,
   926→            calls,
   927→        });
   928→    }
   929→
   930→    ScgSummary {
   931→        function_count: functions.len(),
   932→        functions,
   933→        total_nodes,
   934→        total_edges,
   935→    }
   936→}
   937→
   938→/// Check if an operation string is a known binary operation.
   939→fn is_known_binop(op: &str) -> bool {
   940→    matches!(
   941→        op,
   942→        "add" | "sub" | "mul" | "sdiv" | "udiv" | "srem" | "urem" | "and" | "or" | "xor"
   943→            | "shl" | "shr.l" | "shr.a" | "slt" | "sle" | "sgt" | "sge" | "ult" | "ule"
   944→            | "ugt" | "uge" | "eq" | "ne" | "+" | "-" | "*" | "/" | "%" | "&" | "|"
   945→            | "^" | "<<" | ">>" | "<" | "<=" | ">" | ">=" | "=="
   946→    )
   947→}

   949→/// Find the return type of a function by tracing to its FunctionReturn node.
   950→fn find_return_type(
   951→    scg: &vuma_scg::SCG,
   952→    entry_id: vuma_scg::NodeId,
   953→    edge_idx: &ScgEdgeIndex,
   954→) -> String {
   955→    use vuma_scg::{ControlKind, EdgeKind, NodePayload};
   956→
   957→    let mut visited = std::collections::HashSet::new();
   958→    let mut queue = std::collections::VecDeque::new();
   959→    queue.push_back(entry_id);
   960→    visited.insert(entry_id);
   961→
   962→    while let Some(current) = queue.pop_front() {
   963→        for edge in edge_idx.outgoing(current) {
   964→            if edge.kind == EdgeKind::ControlFlow && !visited.contains(&edge.target) {
   965→                visited.insert(edge.target);
   966→                if let Some(node) = scg.get_node(edge.target) {
   967→                    if let NodePayload::Control(c) = &node.payload {
   968→                        if c.kind == ControlKind::FunctionReturn {
   969→                            // Check incoming DataFlow edges for return value type
   970→                            for ret_edge in edge_idx.incoming(edge.target) {
   971→                                if ret_edge.kind == EdgeKind::DataFlow {
   972→                                    if let Some(src) = scg.get_node(ret_edge.source) {
   973→                                        if let NodePayload::Computation(comp) = &src.payload {
   974→                                            if let Some(rt) = &comp.result_type {
   975→                                                return rt.clone();
   976→                                            }
   977→                                        }
   978→                                    }
   979→                                }
   980→                            }
   981→                            return "void".to_string();
   982→                        }
   983→                    }
   984→                }
   985→                queue.push_back(edge.target);
   986→            }
   987→        }
   988→    }
   989→
   990→    "void".to_string()
   991→}

   993→/// Build an AST summary from the parsed program.
   994→fn build_ast_summary(ast: &vuma_parser::Program) -> AstSummary {
   995→    use vuma_parser::Item;
   996→
   997→    let mut function_names = Vec::new();
   998→    let mut region_names = Vec::new();
   999→    let mut import_count = 0;
  1000→
  1001→    for item in &ast.items {
  1002→        match item {
  1003→            Item::FnDef(fn_decl) => {
  1004→                function_names.push(fn_decl.name.clone());
  1005→            }
  1006→            Item::RegionDef(region_decl) => {
  1007→                region_names.push(region_decl.name.clone());
  1008→            }
  1009→            Item::Import(_) => {
  1010→                import_count += 1;
  1011→            }
  1012→            _ => {}
  1013→        }
  1014→    }
  1015→
  1016→    AstSummary {
  1017→        item_count: ast.items.len(),
  1018→        function_names,
  1019→        region_names,
  1020→        import_count,
  1021→    }
  1022→}

  1024→/// Disassemble the default (AArch64) binary output.

  1028→    match create_backend(BackendKind::AArch64) {

  1033→        Err(_) => format!("({} bytes of binary output)", binary.len()),
  1034→    }
