     1→//! # ARM64 Code Emission
     2→//!
     3→//! Lowers IR to ARM64 machine code and produces ELF binaries or raw binaries
     4→//! suitable for the AArch64 (Cortex-A76, ARMv8.2-A).
     5→//!
     6→//! ## Pipeline
     7→//!
     8→//! 1. **IR → ARM64 Instructions**: Each IR instruction is pattern-matched and
     9→//!    lowered to one or more [`Instruction`]
    10→//!    values, with virtual registers replaced by physical registers from the
    11→//!    register allocator.
    12→//! 2. **Emit function**: Each IR function is emitted as a sequence of 32-bit
    13→//!    ARM64 code words.
    14→//! 3. **Emit program**: All functions are collected, data sections are laid
    15→//!    out, and a minimal ELF64 binary is produced.
    16→//!
    17→//! ## Output Formats
    18→//!
    19→//! - **ELF**: Full ELF64 binary with program headers, section headers, symbol
    20→//!   table, and string table.  Suitable for Linux on AArch64.
    21→//! - **Raw**: Flat binary image for bare-metal AArch64 (loaded at 0x80000).
    22→//! - **Obj**: Relocatable ELF object file (`ET_REL`) for linking.
    23→//!
    24→//! ## ELF Layout (executable)
    25→//!
    26→//! ```text
    27→//! ┌─────────────────────┐
    28→//! │ ELF Header           │  64 bytes
    29→//! ├─────────────────────┤
    30→//! │ Program Headers      │  3 × 56 bytes (LOAD segments)
    31→//! ├─────────────────────┤
    32→//! │ .rodata              │  read-only data (segment: R)
    33→//! ├─────────────────────┤
    34→//! │ .text                │  emitted code   (segment: R+X)
    35→//! ├─────────────────────┤
    36→//! │ .data                │  initialized data (segment: R+W)
    37→//! ├─────────────────────┤
    38→//! │ .symtab              │  symbol table entries
    39→//! ├─────────────────────┤
    40→//! │ .strtab              │  symbol string table
    41→//! ├─────────────────────┤
    42→//! │ .shstrtab            │  section header string table
    43→//! ├─────────────────────┤
    44→//! │ Section Headers      │  N × 64 bytes
    45→//! └─────────────────────┘
    46→//! .bss is virtual-only (memsz > filesz in the data LOAD segment).
    47→//!
    48→//! Section alignment is target-dependent:
    49→//! - ARM32:    4-byte alignment
    50→//! - AArch64:  16-byte alignment
    51→//! - x86-64:   16-byte alignment
    52→//! - RISC-V:   4-byte alignment
    53→//! - MIPS64:   8-byte alignment
    54→//! - PPC64:    16-byte alignment
    55→//! - LoongArch: 8-byte alignment
    56→//! ```
    57→//!
    58→//! ## Relocation Support
    59→//!
    60→//! Inter-function calls (`BL`) are resolved in a fixup pass after all
    61→//! functions have been emitted.  The emitter records the word offset of each
    62→//! `BL` instruction and the target function name; once function addresses
    63→//! are known, the branch offsets are patched into the encoded instructions.
    64→

   517→// ---------------------------------------------------------------------------
   518→// Inter-function call relocation record
   519→// ---------------------------------------------------------------------------
   520→
   521→/// A record for a `BL` instruction that needs to be patched with the address
   522→/// of a named function.  After all functions are emitted, these records are
   523→/// resolved to compute the correct branch offset.
   524→#[derive(Debug, Clone)]
   525→struct CallRelocation {
   526→    /// Byte offset within the text section where the BL instruction lives.
   527→    text_byte_offset: u64,
   528→    /// Name of the target function.
   529→    target_func: String,
   530→}
   531→
   532→// ---------------------------------------------------------------------------
   533→// ELF RelaEntry
   534→// ---------------------------------------------------------------------------
   535→
   536→/// An ELF64 relocation entry with an explicit addend (SHT_RELA).
   537→///
   538→/// Each entry specifies where in the section a relocation must be applied,
   539→/// which symbol the relocation references, the type of relocation, and an
   540→/// addend value used in the relocation computation.
   541→#[derive(Debug, Clone)]
   542→pub struct RelaEntry {
   543→    /// Byte offset within the section where the relocation applies.
   544→    pub offset: u64,
   545→    /// Symbol index (upper 32 bits) and relocation type (lower 32 bits),
   546→    /// packed as: `(sym_idx << 32) | r_type`.
   547→    pub info: u64,
   548→    /// Addend value used in the relocation computation.
   549→    pub addend: i64,
   550→}
   551→
   552→impl RelaEntry {
   553→    /// Create a new relocation entry.
   554→    pub fn new(offset: u64, sym_idx: u32, r_type: u32, addend: i64) -> Self {
   555→        Self {
   556→            offset,
   557→            info: ((sym_idx as u64) << 32) | (r_type as u64),
   558→            addend,
   559→        }
   560→    }
   561→
   562→    /// Extract the symbol index from `info`.
   563→    pub fn sym_idx(&self) -> u32 {
   564→        (self.info >> 32) as u32
   565→    }
   566→
   567→    /// Extract the relocation type from `info`.
   568→    pub fn r_type(&self) -> u32 {
   569→        (self.info & 0xFFFFFFFF) as u32
   570→    }
   571→
   572→    /// Serialize the entry to 24 bytes (little-endian).
   573→    pub fn to_bytes(&self) -> [u8; 24] {
   574→        let mut buf = [0u8; 24];
   575→        buf[0..8].copy_from_slice(&self.offset.to_le_bytes());
   576→        buf[8..16].copy_from_slice(&self.info.to_le_bytes());
   577→        buf[16..24].copy_from_slice(&self.addend.to_le_bytes());
   578→        buf
   579→    }
   580→}
   581→
   582→// ---------------------------------------------------------------------------
   583→// Emitter
   584→// ---------------------------------------------------------------------------
   585→
   586→/// The ARM64 code emitter.
   587→///
   588→/// Holds state for the current emission context: the register allocator,
   589→/// accumulated code, and fixup records for branch targets.
   590→pub struct Emitter {
   591→    /// Register allocator used during emission.
   592→    reg_alloc: RegAllocator,
   593→    /// Accumulated machine code for the current function (32-bit words).
   594→    code: Vec<u32>,
   595→    /// Fixup records for intra-function branches: (word index, target label name, format).
   596→    fixups: Vec<(usize, String, BranchFormat)>,
   597→    /// Map from label name to code offset (in words) within the current function.
   598→    label_offsets: HashMap<String, usize>,
   599→    /// Inter-function call relocations for the current function.
   600→    call_relocs: Vec<CallRelocation>,
   601→    /// Relocation entries using the `RelocationEntry` infrastructure.
   602→    ///
   603→    /// Each entry records a byte offset within the function's encoded output
   604→    /// where a symbolic reference must be resolved, the target symbol name,
   605→    /// and the ISA-specific relocation type (e.g., `"R_AARCH64_CALL26"`).
   606→    relocations: Vec<RelocationEntry>,
   607→    /// Name of the function currently being emitted.
   608→    current_func_name: String,
   609→    /// Byte offset of the current function within the text section (set externally).
   610→    func_text_offset: u64,
   611→    /// Computed stack frame size (in bytes) for the current function.
   612→    frame_size: u32,
   613→    /// Registers pinned for the current instruction (auto-unpinned after each instruction).
   614→    instr_pinned_regs: Vec<Register>,
   615→}
   616→
   617→impl Emitter {
   618→    /// Create a new emitter with a fresh register allocator.
   619→    pub fn new() -> Self {
   620→        Self {
   621→            reg_alloc: RegAllocator::new(),
   622→            code: Vec::new(),
   623→            fixups: Vec::new(),
   624→            label_offsets: HashMap::new(),
   625→            call_relocs: Vec::new(),
   626→            relocations: Vec::new(),
   627→            current_func_name: String::new(),
   628→            func_text_offset: 0,
   629→            frame_size: 0,
   630→            instr_pinned_regs: Vec::new(),
   631→        }
   632→    }
   633→
   634→    // -----------------------------------------------------------------------
   635→    // Function emission
   636→    // -----------------------------------------------------------------------
   637→
   638→    /// Emit a single IR function to ARM64 machine code.
   639→    ///
   640→    /// For functions with more than `STACK_SLOT_VREG_THRESHOLD` virtual
   641→    /// registers, the stack-slot emitter is used (every vreg gets a stack
   642→    /// slot, scratch registers only).  For simpler functions, the greedy
   643→    /// register allocator is used.
   644→    ///
   645→    /// Returns a vector of 32-bit ARM64 instruction words.
   646→    pub fn emit_function(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   647→        let vreg_count = count_vregs(func);
   648→        if vreg_count > STACK_SLOT_VREG_THRESHOLD {
   649→            self.emit_function_stack_slot(func)
   650→        } else {
   651→            self.emit_function_greedy(func)
   652→        }
   653→    }
   654→
   655→    /// Emit a single IR function using the greedy register allocator.
   656→    ///
   657→    /// This is the original emission strategy — suitable for functions with
   658→    /// a small number of virtual registers.
   659→    fn emit_function_greedy(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   660→        self.code.clear();
   661→        self.fixups.clear();
   662→        self.label_offsets.clear();
   663→        self.call_relocs.clear();
   664→        self.relocations.clear();
   665→        self.current_func_name = func.name.clone();
   666→        self.reg_alloc.reset();
   667→
   668→        // Pre-allocate registers for parameters (AAPCS64: X0–X7).
   669→        // We must explicitly assign parameter virtual registers to the
   670→        // correct argument registers so the calling convention is respected.
   671→        let arg_regs = [
   672→            Register::X0, Register::X1, Register::X2, Register::X3,
   673→            Register::X4, Register::X5, Register::X6, Register::X7,
   674→        ];
   675→        for (i, param) in func.params.iter().enumerate() {
   676→            if let IRValue::Register(vreg_id) = param {
   677→                if i < 8 {
   678→                    self.reg_alloc.preassign(*vreg_id, arg_regs[i]);
   679→                }
   680→            }
   681→        }
   682→
   683→        // Emit prologue:
   684→        // 1. SUB SP, SP, #16          (make room for FP/LR save pair)
   685→        // 2. STP X29, X30, [SP]       (save FP and LR at new SP)
   686→        // 3. ADD X29, SP, #0           (set frame pointer to new SP)
   687→        // 4. SUB SP, SP, #frame_size   (make room for local variables)
   688→        //
   689→        // NOTE: We cannot use pre-indexed STP [SP, #-16]! because the
   690→        // Instruction::STP encoding only supports signed-offset mode.
   691→        // Instead, we explicitly decrement SP with SUB before the STP.
   692→        self.emit_instruction(Instruction::SUB {
   693→            rd: Register::SP,
   694→            rn: Register::SP,
   695→            rm: Operand::Imm12(16),
   696→        })?;
   697→        self.emit_instruction(Instruction::STP {
   698→            rt1: Register::X29,
   699→            rt2: Register::X30,
   700→            rn: Register::SP,
   701→            offset: 0,
   702→        })?;
   703→
   704→        // MOV X29, SP (set frame pointer)
   705→        // IMPORTANT: Cannot use MOV (ORR Xd, XZR, SP) because ORR treats
   706→        // Rm=31 as XZR, yielding zero. Use ADD X29, SP, #0 instead.
   707→        self.emit_instruction(Instruction::ADD {
   708→            rd: Register::X29,
   709→            rn: Register::SP,
   710→            rm: Operand::Imm12(0),
   711→        })?;
   712→
   713→        // Reserve space for spill slots only. Each Alloc instruction handles
   714→        // its own SUB SP, SP, #aligned_size. The epilogue will restore only
   715→        // the spill area reservation. The Alloc areas are NOT restored by the
   716→        // epilogue — they're effectively "leaked" stack space. This is fine
   717→        // because the function's stack frame is restored by the LDP + ADD SP, #16.
   718→        //
   719→        // For spill slots, we need to estimate the maximum number of spills
   720→        // that could occur during emission. Since we can't know the exact
   721→        // count beforehand, we use the compute_frame_size function which
   722→        // estimates based on the vreg count.
   723→        let aligned_stack = compute_frame_size(func);
   724→        // frame_size = only the spill area (NOT the Alloc sizes, since each
   725→        // Alloc instruction does its own SUB SP)
   726→        // Compute the spill-only portion:
   727→        let mut alloc_total: u32 = 0;
   728→        for block in &func.blocks {
   729→            for instr in &block.instructions {
   730→                if let IRInstr::Alloc { size, .. } = instr {
   731→                    let aligned = (*size).div_ceil(16) * 16;
   732→                    alloc_total += aligned;
   733→                }
   734→            }
   735→        }
   736→        // Spill area = total frame - alloc area
   737→        let spill_area = if aligned_stack > alloc_total {
   738→            aligned_stack - alloc_total
   739→        } else {
   740→            // Minimal 16-byte reservation for safety
   741→            16
   742→        };
   743→        let spill_area_aligned = (spill_area + 15) & !15;
   744→        self.frame_size = spill_area_aligned;
   745→        if spill_area_aligned > 0 {
   746→            if spill_area_aligned <= 4095 {
   747→                self.emit_instruction(Instruction::SUB {
   748→                    rd: Register::SP,
   749→                    rn: Register::SP,
   750→                    rm: Operand::Imm12(spill_area_aligned as u16),
   751→                })?;
   752→            } else {
   753→                self.emit_load_immediate(Register::X9, spill_area_aligned as i64)?;
   754→                self.emit_instruction(Instruction::SUB {
   755→                    rd: Register::SP,
   756→                    rn: Register::SP,
   757→                    rm: Operand::Reg {
   758→                        reg: Register::X9,
   759→                        shift: None,
   760→                    },
   761→                })?;
   762→            }
   763→        }
   764→
   765→        // Emit each basic block.
   766→        for block in &func.blocks {
   767→            self.label_offsets
   768→                .insert(block.label.clone(), self.code.len());
   769→            self.emit_block(block)?;
   770→        }
   771→
   772→        // Apply fixups — resolve intra-function branch targets.
   773→        self.apply_fixups()?;
   774→
   775→        Ok(self.code.clone())
   776→    }
   777→
   778→    /// Returns the relocation entries recorded during emission of the current
   779→    /// function.
   780→    ///
   781→    /// Each `RelocationEntry` records a byte offset within the function's
   782→    /// encoded output where a symbolic reference must be resolved, the target
   783→    /// symbol name, and the ISA-specific relocation type (e.g.,
   784→    /// `"R_AARCH64_CALL26"`).
   785→    pub fn relocations(&self) -> &[RelocationEntry] {
   786→        &self.relocations
   787→    }
   788→
   789→    /// Emit instructions for a single IR basic block.
   790→    fn emit_block(&mut self, block: &IRBlock) -> Result<()> {
   791→        for instr in &block.instructions {
   792→            self.emit_ir_instr(instr)?;
   793→            // Auto-unpin all registers that were pinned during this instruction.
   794→            for reg in self.instr_pinned_regs.drain(..) {
   795→                self.reg_alloc.unpin(reg);
   796→            }
   797→        }
   798→        self.emit_terminator(&block.terminator)?;
   799→        Ok(())
   800→    }
   801→
   802→    // -----------------------------------------------------------------------
   803→    // IR → ARM64 instruction lowering
   804→    // -----------------------------------------------------------------------
   805→
   806→    /// Lower a single IR instruction to ARM64 instructions.
   807→    fn emit_ir_instr(&mut self, instr: &IRInstr) -> Result<()> {
   808→        match instr {
   809→            IRInstr::Load { dst, addr, offset, ty } => {
   810→                let rn = self.resolve_reg(addr)?;
   811→                let rt = self.resolve_reg(dst)?;
   812→                self.emit_load(rt, rn, *offset, ty)?;
   813→            }
   814→
   815→            IRInstr::Store { value, addr, offset, ty } => {
   816→                let rt = self.resolve_reg(value)?;
   817→                let rn = self.resolve_reg(addr)?;
   818→                if rt == rn {
   819→                    // Both resolved to the same register — use scratch for value
   820→                    self.emit_instruction(Instruction::MOV {
   821→                        rd: Register::X16,
   822→                        rm: rt,
   823→                    })?;
   824→                    self.emit_store(Register::X16, rn, *offset, ty)?;
   825→                } else {
   826→                    self.emit_store(rt, rn, *offset, ty)?;
   827→                }
   828→            }
   829→
   830→            IRInstr::BinOp { op, dst, lhs, rhs, ty } => {
   831→                self.emit_binop(*op, dst, lhs, rhs, ty.as_ref())?;
   832→            }
   833→
   834→            IRInstr::UnaryOp { op, dst, operand, ty: _ } => {
   835→                let rd = self.resolve_reg(dst)?;
   836→                let rn = self.resolve_reg(operand)?;
   837→                match op {
   838→                    UnaryOpKind::Neg => {
   839→                        self.emit_instruction(Instruction::SUB {
   840→                            rd,
   841→                            rn: Register::XZR,
   842→                            rm: Operand::Reg {
   843→                                reg: rn,
   844→                                shift: None,
   845→                            },
   846→                        })?;
   847→                    }
   848→                    UnaryOpKind::Not => {
   849→                        self.emit_load_immediate(Register::X9, -1)?;
   850→                        self.emit_instruction(Instruction::EOR {
   851→                            rd,
   852→                            rn,
   853→                            rm: Register::X9,
   854→                        })?;
   855→                    }
   856→                    UnaryOpKind::Clz => {
   857→                        // CLZ Xd, Xn — count leading zeros (native ARM64 instruction)
   858→                        self.emit_instruction(Instruction::CLZ { rd, rn })?;
   859→                    }
   860→                    UnaryOpKind::Ctz => {
   861→                        // CTZ = RBIT + CLZ: reverse bits then count leading zeros.
   862→                        // Use X9 as scratch if rd == rn (need intermediate result).
   863→                        if rd == rn {
   864→                            self.emit_instruction(Instruction::RBIT {
   865→                                rd: Register::X9,
   866→                                rn,
   867→                            })?;
   868→                            self.emit_instruction(Instruction::CLZ {
   869→                                rd,
   870→                                rn: Register::X9,
   871→                            })?;
   872→                        } else {
   873→                            self.emit_instruction(Instruction::RBIT { rd, rn })?;
   874→                            self.emit_instruction(Instruction::CLZ { rd, rn: rd })?;
   875→                        }
   876→                    }
   877→                    UnaryOpKind::Popcnt => {
   878→                        // POPCNT via FMOV+CNT+ADDV+UMOV sequence:
   879→                        // FMOV D8, Xn        — move GPR value to SIMD register (V8 is caller-saved)
   880→                        // CNT V8.8B, V8.8B   — count bits per byte
   881→                        // ADDV B8, V8.8B     — horizontal sum of byte counts
   882→                        // UMOV Xd, V8.B[0]   — move result back to GPR (zero-extends to 64 bits)
   883→                        const SIMD_SCRATCH: u8 = 8; // V8 is caller-saved in AAPCS64
   884→                        self.emit_instruction(Instruction::FMOV_DX {
   885→                            vd: SIMD_SCRATCH,
   886→                            rn,
   887→                        })?;
   888→                        self.emit_instruction(Instruction::CNT {
   889→                            vd: SIMD_SCRATCH,
   890→                            vn: SIMD_SCRATCH,
   891→                        })?;
   892→                        self.emit_instruction(Instruction::ADDV {
   893→                            vd: SIMD_SCRATCH,
   894→                            vn: SIMD_SCRATCH,
   895→                        })?;
   896→                        self.emit_instruction(Instruction::UMOV {
   897→                            rd,
   898→                            vn: SIMD_SCRATCH,
   899→                        })?;
   900→                    }
   901→                }
   902→            }
   903→
   904→            IRInstr::Call {
   905→                dst,
   906→                func: target_name,
   907→                args,
   908→                is_extern: _,
   909→            } => {
   910→                // Resolve all argument source registers FIRST, before moving
   911→                // any of them. This prevents a later move from overwriting a
   912→                // register that an earlier argument is in.
   913→                //
   914→                // IMPORTANT: For immediate arguments, resolve_reg always loads
   915→                // into X9. When multiple immediates are present, we must use
   916→                // different scratch registers to avoid overwriting. We load
   917→                // immediates into X0-X7 directly (since they're the target
   918→                // registers anyway) or use X11-X15 as temporary scratch.
   919→                let arg_regs = [
   920→                    Register::X0, Register::X1, Register::X2, Register::X3,
   921→                    Register::X4, Register::X5, Register::X6, Register::X7,
   922→                ];
   923→                // Scratch registers for loading immediates when the target arg
   924→                // register isn't available yet (because of conflicts).
   925→                let scratch_regs = [
   926→                    Register::X11, Register::X12, Register::X13, Register::X14,
   927→                    Register::X15, Register::X3, Register::X4, Register::X5,
   928→                ];
   929→
   930→                let mut src_regs: Vec<Register> = Vec::new();
   931→                let mut immediate_scratch_used: Vec<usize> = Vec::new(); // which scratch reg each immediate uses
   932→                let mut next_scratch = 0;
   933→
   934→                for (i, arg) in args.iter().enumerate() {
   935→                    if i >= 8 {
   936→                        break;
   937→                    }
   938→                    let is_immediate = matches!(arg, IRValue::Immediate(_) | IRValue::Address(_));
   939→                    if is_immediate {
   940→                        let v = match arg {
   941→                            IRValue::Immediate(v) => *v,
   942→                            IRValue::Address(a) => *a as i64,
   943→                            _ => unreachable!(),
   944→                        };
   945→                        // Load the immediate directly into the target register
   946→                        // if it's not needed by another argument.
   947→                        // For simplicity, always load into a scratch register
   948→                        // and record it.
   949→                        if next_scratch < scratch_regs.len() {
   950→                            let scratch = scratch_regs[next_scratch];
   951→                            next_scratch += 1;
   952→                            self.emit_load_immediate(scratch, v)?;
   953→                            src_regs.push(scratch);
   954→                        } else {
   955→                            // Fallback to X9 (rare, only with 8+ immediates)
   956→                            let scratch = Register::X9;
   957→                            self.emit_load_immediate(scratch, v)?;
   958→                            src_regs.push(scratch);
   959→                        }
   960→                    } else {
   961→                        let src = self.resolve_reg(arg)?;
   962→                        src_regs.push(src);
   963→                    }
   964→                }
   965→
   966→                // Now move arguments into X0–X7. We need to handle the case
   967→                // where a source register is also a target register (cycle).
   968→                // Strategy: first move arguments that don't conflict, then
   969→                // handle conflicts using a scratch register.
   970→                let n = src_regs.len();
   971→                let mut moved = vec![false; n];
   972→
   973→                // Pass 1: move args where src != any dst that hasn't been moved yet
   974→                // Simple approach: iterate and move non-conflicting ones first
   975→                for pass in 0..2 {
   976→                    for i in 0..n {
   977→                        if moved[i] {
   978→                            continue;
   979→                        }
   980→                        let src = src_regs[i];
   981→                        let dst = arg_regs[i];
   982→                        if src == dst {
   983→                            // Already in the right register
   984→                            moved[i] = true;
   985→                            continue;
   986→                        }
   987→                        if pass == 0 {
   988→                            // Check if src is needed as a source by a later unmoved arg
   989→                            let mut conflict = false;
   990→                            for j in 0..n {
   991→                                if !moved[j] && j != i && src_regs[j] == dst {
   992→                                    // Moving arg i to dst would overwrite arg j's source
   993→                                    conflict = true;
   994→                                    break;
   995→                                }
   996→                            }
   997→                            if conflict {
   998→                                continue; // Defer to pass 2
   999→                            }
  1000→                        }
  1001→                        // Safe to move now
  1002→                        self.emit_instruction(Instruction::MOV {
  1003→                            rd: dst,
  1004→                            rm: src,
  1005→                        })?;
  1006→                        moved[i] = true;
  1007→                    }
  1008→                }
  1009→
  1010→                // Pass 2: handle any remaining conflicts with scratch register
  1011→                for i in 0..n {
  1012→                    if moved[i] {
  1013→                        continue;
  1014→                    }
  1015→                    let src = src_regs[i];
  1016→                    let dst = arg_regs[i];
  1017→                    if src == dst {
  1018→                        moved[i] = true;
  1019→                        continue;
  1020→                    }
  1021→                    // Use X16 as scratch to break the cycle
  1022→                    self.emit_instruction(Instruction::MOV {
  1023→                        rd: Register::X16,
  1024→                        rm: src,
  1025→                    })?;
  1026→                    self.emit_instruction(Instruction::MOV {
  1027→                        rd: dst,
  1028→                        rm: Register::X16,
  1029→                    })?;
  1030→                    moved[i] = true;
  1031→                }
  1032→
  1033→                // BL — record a relocation for later patching.
  1034→                let bl_word_idx = self.code.len();
  1035→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1036→                self.call_relocs.push(CallRelocation {
  1037→                    text_byte_offset: bl_byte_offset,
  1038→                    target_func: target_name.clone(),
  1039→                });
  1040→                // Register a RelocationEntry so the linker can patch this BL.
  1041→                self.relocations.push(RelocationEntry {
  1042→                    offset: bl_byte_offset,
  1043→                    symbol: target_name.clone(),
  1044→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1045→                });
  1046→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1047→
  1048→                if let Some(dst_val) = dst {
  1049→                    let rd = self.resolve_reg(dst_val)?;
  1050→                    if rd != Register::X0 {
  1051→                        self.emit_instruction(Instruction::MOV {
  1052→                            rd,
  1053→                            rm: Register::X0,
  1054→                        })?;
  1055→                    }
  1056→                }
  1057→            }
  1058→
  1059→            IRInstr::Alloc { dst, size } => {
  1060→                let rd = self.resolve_reg(dst)?;
  1061→                // IMPORTANT: Decrement SP FIRST, then save the new SP value.
  1062→                // If we save before decrementing, the allocation pointer
  1063→                // points to the wrong (too-high) address.
  1064→                let aligned = (*size).div_ceil(16) * 16;
  1065→                self.emit_instruction(Instruction::SUB {
  1066→                    rd: Register::SP,
  1067→                    rn: Register::SP,
  1068→                    rm: Operand::Imm12(aligned as u16),
  1069→                })?;
  1070→                // MOV rd, SP: cannot use ORR because ORR treats Rm=31 as XZR.
  1071→                // Use ADD rd, SP, #0 instead.
  1072→                self.emit_instruction(Instruction::ADD {
  1073→                    rd,
  1074→                    rn: Register::SP,
  1075→                    rm: Operand::Imm12(0),
  1076→                })?;
  1077→            }
  1078→
  1079→            IRInstr::Free { ptr } => {
  1080→                let rt = self.resolve_reg(ptr)?;
  1081→                // Move ptr to X0 (first argument)
  1082→                if rt != Register::X0 {
  1083→                    self.emit_instruction(Instruction::MOV {
  1084→                        rd: Register::X0,
  1085→                        rm: rt,
  1086→                    })?;
  1087→                }
  1088→                // BL __vuma_free
  1089→                let bl_word_idx = self.code.len();
  1090→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1091→                self.call_relocs.push(CallRelocation {
  1092→                    text_byte_offset: bl_byte_offset,
  1093→                    target_func: "__vuma_free".to_string(),
  1094→                });
  1095→                self.relocations.push(RelocationEntry {
  1096→                    offset: bl_byte_offset,
  1097→                    symbol: "__vuma_free".to_string(),
  1098→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1099→                });
  1100→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1101→            }
  1102→
  1103→            IRInstr::Cast { kind, dst, src } => {
  1104→                let rd = self.resolve_reg(dst)?;
  1105→                let rn = self.resolve_reg(src)?;
  1106→                match kind {
  1107→                    CastKind::ZExt => {
  1108→                        // Zero-extend 32-bit to 64-bit using UBFM
  1109→                        self.emit_instruction(Instruction::UBFM {
  1110→                            rd,
  1111→                            rn,
  1112→                            immr: 0,
  1113→                            imms: 31,
  1114→                        })?;
  1115→                    }
  1116→                    CastKind::SExt => {
  1117→                        // Sign-extend 32-bit to 64-bit using SBFM
  1118→                        self.emit_instruction(Instruction::SBFM {
  1119→                            rd,
  1120→                            rn,
  1121→                            immr: 0,
  1122→                            imms: 31,
  1123→                        })?;
  1124→                    }
  1125→                    CastKind::Trunc | CastKind::BitCast => {
  1126→                        // Trunc: upper bits discarded on write — just MOV
  1127→                        // BitCast: no data change — just MOV
  1128→                        if rd != rn {
  1129→                            self.emit_instruction(Instruction::MOV { rd, rm: rn })?;
  1130→                        }
  1131→                    }
  1132→                    CastKind::IntToFloat => {
  1133→                        // Signed int → float: SCVTF
  1134→                        self.emit_instruction(Instruction::SCVTF { rd, rn })?;
  1135→                    }
  1136→                    CastKind::UIntToFloat => {
  1137→                        // Unsigned int → float: UCVTF
  1138→                        self.emit_instruction(Instruction::UCVTF { rd, rn })?;
  1139→                    }
  1140→                    CastKind::FloatToInt => {
  1141→                        // Float → signed int: FCVTZS
  1142→                        self.emit_instruction(Instruction::FCVTZS { rd, rn })?;
  1143→                    }
  1144→                    CastKind::FloatToUInt => {
  1145→                        // Float → unsigned int: FCVTZU
  1146→                        self.emit_instruction(Instruction::FCVTZU { rd, rn })?;
  1147→                    }
  1148→                    CastKind::FloatToFloat => {
  1149→                        // Float ↔ float width change: FCVT
  1150→                        self.emit_instruction(Instruction::FCVT { rd, rn, to_double: true })?;
  1151→                    }
  1152→                }
  1153→            }
  1154→
  1155→            IRInstr::Phi { .. } => {
  1156→                log::warn!(
  1157→                    "IRInstr::Phi encountered during emission — should be resolved by SSA pass"
  1158→                );
  1159→            }
  1160→
  1161→            IRInstr::GetAddress { dst, name } => {
  1162→                let rd = self.resolve_reg(dst)?;
  1163→                // Emit a call to __vuma_getaddr to resolve the symbol at runtime.
  1164→                // Move name hash to X0 as the argument.
  1165→                let name_hash = name
  1166→                    .chars()
  1167→                    .fold(0u64, |acc, c| acc.wrapping_mul(31).wrapping_add(c as u64));
  1168→                self.emit_load_immediate(Register::X0, name_hash as i64)?;
  1169→                let bl_word_idx = self.code.len();
  1170→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1171→                self.call_relocs.push(CallRelocation {
  1172→                    text_byte_offset: bl_byte_offset,
  1173→                    target_func: "__vuma_getaddr".to_string(),
  1174→                });
  1175→                self.relocations.push(RelocationEntry {
  1176→                    offset: bl_byte_offset,
  1177→                    symbol: "__vuma_getaddr".to_string(),
  1178→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1179→                });
  1180→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1181→                if rd != Register::X0 {
  1182→                    self.emit_instruction(Instruction::MOV {
  1183→                        rd,
  1184→                        rm: Register::X0,
  1185→                    })?;
  1186→                }
  1187→            }
  1188→
  1189→            IRInstr::Offset { dst, base, offset } => {
  1190→                let rd = self.resolve_reg(dst)?;
  1191→                let rn = self.resolve_reg(base)?;
  1192→                let rm = match offset {
  1193→                    IRValue::Immediate(v) => {
  1194→                        if *v >= 0 && *v <= 4095 {
  1195→                            Operand::Imm12(*v as u16)
  1196→                        } else {
  1197→                            let temp = Register::X9;
  1198→                            self.emit_load_immediate(temp, *v)?;
  1199→                            Operand::Reg {
  1200→                                reg: temp,
  1201→                                shift: None,
  1202→                            }
  1203→                        }
  1204→                    }
  1205→                    _ => Operand::Reg {
  1206→                        reg: self.resolve_reg(offset)?,
  1207→                        shift: None,
  1208→                    },
  1209→                };
  1210→                self.emit_instruction(Instruction::ADD { rd, rn, rm })?;
  1211→            }
  1212→
  1213→            // ── Dedicated arithmetic — delegate to BinOp ──
  1214→            IRInstr::Add { dst, lhs, rhs, ty } => {
  1215→                self.emit_binop(BinOpKind::Add, dst, lhs, rhs, ty.as_ref())?;
  1216→            }
  1217→            IRInstr::Sub { dst, lhs, rhs, ty } => {
  1218→                self.emit_binop(BinOpKind::Sub, dst, lhs, rhs, ty.as_ref())?;
  1219→            }
  1220→            IRInstr::Mul { dst, lhs, rhs, ty } => {
  1221→                self.emit_binop(BinOpKind::Mul, dst, lhs, rhs, ty.as_ref())?;
  1222→            }
  1223→            IRInstr::Div { dst, lhs, rhs, ty } => {
  1224→                self.emit_binop(BinOpKind::SDiv, dst, lhs, rhs, ty.as_ref())?;
  1225→            }
  1226→
  1227→            // ── Comparison instruction ──
  1228→            IRInstr::Cmp {
  1229→                kind,
  1230→                dst,
  1231→                lhs,
  1232→                rhs,
  1233→                ty,
  1234→            } => {
  1235→                let width = RegWidth::from_ir_type(ty.as_ref());
  1236→                let rd = self.resolve_reg(dst)?;
  1237→                // IMPORTANT: resolve_reg uses X9 for immediates, so if both lhs and rhs
  1238→                // are immediates, the second load would overwrite the first. Use X10 for
  1239→                // the RHS when both are immediates.
  1240→                let rn = self.resolve_reg(lhs)?;
  1241→                let rm = match rhs {
  1242→                    IRValue::Immediate(_) | IRValue::Address(_) => {
  1243→                        // If lhs was also an immediate, it's in X9; use X10 for rhs
  1244→                        let temp = if matches!(lhs, IRValue::Immediate(_) | IRValue::Address(_)) {
  1245→                            Register::X10
  1246→                        } else {
  1247→                            Register::X9
  1248→                        };
  1249→                        match rhs {
  1250→                            IRValue::Immediate(v) => self.emit_load_immediate(temp, *v)?,
  1251→                            IRValue::Address(a) => self.emit_load_immediate(temp, *a as i64)?,
  1252→                            _ => unreachable!(),
  1253→                        }
  1254→                        temp
  1255→                    }
  1256→                    _ => self.resolve_reg(rhs)?,
  1257→                };
  1258→                self.emit_instruction_with_width(
  1259→                    Instruction::CMP {
  1260→                        rn,
  1261→                        rm: Operand::Reg {
  1262→                            reg: rm,
  1263→                            shift: None,
  1264→                        },
  1265→                    },
  1266→                    width,
  1267→                )?;
  1268→                let cond = cmp_kind_to_condition(kind);
  1269→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1270→            }
  1271→
  1272→            // ── Instruction-level control flow ──
  1273→            IRInstr::Ret { values } => {
  1274→                for (i, val) in values.iter().enumerate() {
  1275→                    if i >= 8 {
  1276→                        break;
  1277→                    }
  1278→                    let src = self.resolve_reg(val)?;
  1279→                    let dst_reg = match i {
  1280→                        0 => Register::X0,
  1281→                        1 => Register::X1,
  1282→                        2 => Register::X2,
  1283→                        3 => Register::X3,
  1284→                        4 => Register::X4,
  1285→                        5 => Register::X5,
  1286→                        6 => Register::X6,
  1287→                        7 => Register::X7,
  1288→                        _ => unreachable!(),
  1289→                    };
  1290→                    if src != dst_reg {
  1291→                        self.emit_instruction(Instruction::MOV {
  1292→                            rd: dst_reg,
  1293→                            rm: src,
  1294→                        })?;
  1295→                    }
  1296→                }
  1297→            }
  1298→
  1299→            IRInstr::Branch { target } => {
  1300→                let fixup_idx = self.code.len();
  1301→                self.fixups.push((fixup_idx, target.clone(), BranchFormat::B26));
  1302→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1303→            }
  1304→
  1305→            IRInstr::CondBranch {
  1306→                cond,
  1307→                true_target,
  1308→                false_target,
  1309→            } => {
  1310→                let rt = self.resolve_reg(cond)?;
  1311→                // CBNZ: if cond != 0, branch to true_target
  1312→                let fixup_cbz = self.code.len();
  1313→                self.fixups.push((fixup_cbz, true_target.clone(), BranchFormat::Cond19));
  1314→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1315→                // Otherwise (cond == 0), branch to false_target
  1316→                let fixup_b = self.code.len();
  1317→                self.fixups.push((fixup_b, false_target.clone(), BranchFormat::B26));
  1318→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1319→            }
  1320→
  1321→            IRInstr::Select {
  1322→                dst,
  1323→                cond,
  1324→                true_val,
  1325→                false_val,
  1326→                ty,
  1327→            } => {
  1328→                let width = RegWidth::from_ir_type(ty.as_ref());
  1329→                // Lower select as: SUBS XZR, cond, #0; CSEL dst, false_val, true_val, NE
  1330→                let rd = self.resolve_reg(dst)?;
  1331→                let rc = self.resolve_reg(cond)?;
  1332→                let rt = self.resolve_reg(true_val)?;
  1333→                let rf = self.resolve_reg(false_val)?;
  1334→                // Compare cond against zero and select.
  1335→                self.emit_instruction_with_width(
  1336→                    Instruction::SUB {
  1337→                        rd: Register::XZR,
  1338→                        rn: rc,
  1339→                        rm: Operand::Imm12(0),
  1340→                    },
  1341→                    width,
  1342→                )?;
  1343→                // Set flags by using a separate CMP (SUB with XZR destination
  1344→                // doesn't set flags; we need a flags-setting variant).
  1345→                // We emulate this with: CMP rc, #0 which is SUBS XZR, rc, #0.
  1346→                // Since we only have SUB, we use the existing CMP pattern.
  1347→                self.emit_instruction_with_width(
  1348→                    Instruction::CSEL {
  1349→                        rd,
  1350→                        rn: rf,
  1351→                        rm: rt,
  1352→                        cond: crate::arm64::Condition::NE,
  1353→                    },
  1354→                    width,
  1355→                )?;
  1356→            }
  1357→
  1358→            // ── Atomic operations ──────────────────────────────────────────
  1359→            IRInstr::AtomicLoad { dst, addr, ty: _ } => {
  1360→                // AArch64: LDAXR Xt, [Xn] — load-acquire exclusive
  1361→                let rt = self.resolve_reg(dst)?;
  1362→                let rn = self.resolve_reg(addr)?;
  1363→                self.emit_instruction(Instruction::LDAXR { rt, rn })?;
  1364→            }
  1365→
  1366→            IRInstr::AtomicStore { value, addr, ty: _ } => {
  1367→                // AArch64: STLXR Ws, Xt, [Xn] — store-release exclusive
  1368→                // Loop until success.
  1369→                let rt = self.resolve_reg(value)?;
  1370→                let rn = self.resolve_reg(addr)?;
  1371→                let rs = Register::X9; // scratch for status
  1372→                let retry_label = format!("__atomic_store_retry_{}", self.code.len());
  1373→                // Record label position
  1374→                self.label_offsets.insert(retry_label.clone(), self.code.len());
  1375→                self.emit_instruction(Instruction::STLXR { rs, rt, rn })?;
  1376→                // CBNZ rs, retry — if store failed, retry
  1377→                let fixup = self.code.len();
  1378→                self.fixups.push((fixup, retry_label, BranchFormat::Cond19));
  1379→                self.emit_instruction(Instruction::CBNZ { rt: rs, offset: 0 })?;
  1380→            }
  1381→
  1382→            IRInstr::AtomicCas { dst, addr, expected, desired, ty: _ } => {
  1383→                // AArch64 CAS loop: LDAXR / CMP / B.NE skip / STLXR / CBNZ retry
  1384→                let rd = self.resolve_reg(dst)?;
  1385→                let rn = self.resolve_reg(addr)?;
  1386→                let re = self.resolve_reg(expected)?;
  1387→                let rm = self.resolve_reg(desired)?;
  1388→                let rs = Register::X9; // scratch for STLXR status
  1389→                let scratch_cmp = Register::X10; // scratch for comparison
  1390→
  1391→                let retry_label = format!("__atomic_cas_retry_{}", self.code.len());
  1392→                let done_label = format!("__atomic_cas_done_{}", self.code.len());
  1393→
  1394→                // Record retry label position
  1395→                self.label_offsets.insert(retry_label.clone(), self.code.len());
  1396→                // LDAXR scratch_cmp, [addr] — load current value
  1397→                self.emit_instruction(Instruction::LDAXR { rt: scratch_cmp, rn })?;
  1398→                // CMP scratch_cmp, expected — SUB XZR, scratch_cmp, expected
  1399→                self.emit_instruction(Instruction::SUB {
  1400→                    rd: Register::XZR,
  1401→                    rn: scratch_cmp,
  1402→                    rm: Operand::Reg { reg: re, shift: None },
  1403→                })?;
  1404→                // B.NE done — if not equal, skip store
  1405→                let fixup_ne = self.code.len();
  1406→                self.fixups.push((fixup_ne, done_label.clone(), BranchFormat::Cond19));
  1407→                self.emit_instruction(Instruction::BCond {
  1408→                    cond: crate::arm64::Condition::NE,
  1409→                    offset: 0,
  1410→                })?;
  1411→                // STLXR rs, desired, [addr] — try to store
  1412→                self.emit_instruction(Instruction::STLXR { rs, rt: rm, rn })?;
  1413→                // CBNZ rs, retry — if store failed, retry
  1414→                let fixup_retry = self.code.len();
  1415→                self.fixups.push((fixup_retry, retry_label, BranchFormat::Cond19));
  1416→                self.emit_instruction(Instruction::CBNZ { rt: rs, offset: 0 })?;
  1417→                // Record done label position
  1418→                self.label_offsets.insert(done_label.clone(), self.code.len());
  1419→                // CSEL rd, re, scratch_cmp, EQ
  1420→                // If EQ (match succeeded), rd = re (expected); else rd = scratch_cmp (current value)
  1421→                self.emit_instruction(Instruction::CSEL {
  1422→                    rd,
  1423→                    rn: re,
  1424→                    rm: scratch_cmp,
  1425→                    cond: crate::arm64::Condition::EQ,
  1426→                })?;
  1427→            }
  1428→
  1429→            // ── Constant-time security operations ────────────────────────────
  1430→            IRInstr::CtSelect {
  1431→                dst,
  1432→                cond,
  1433→                true_val,
  1434→                false_val,
  1435→                ty,
  1436→            } => {
  1437→                // ct_select(cond, a, b) = (a & mask) | (b & ~mask)
  1438→                // where mask = -(cond != 0)
  1439→                // On AArch64: Use CSEL for constant-time conditional select.
  1440→                let width = RegWidth::from_ir_type(ty.as_ref());
  1441→                let rd = self.resolve_reg(dst)?;
  1442→                let _rc = self.resolve_reg(cond)?;
  1443→                let rt = self.resolve_reg(true_val)?;
  1444→                let rf = self.resolve_reg(false_val)?;
  1445→                // Use the same CSEL pattern as Select, which is constant-time
  1446→                // on AArch64 (no branch).
  1447→                self.emit_instruction_with_width(
  1448→                    Instruction::CSEL {
  1449→                        rd,
  1450→                        rn: rf,
  1451→                        rm: rt,
  1452→                        cond: crate::arm64::Condition::NE,
  1453→                    },
  1454→                    width,
  1455→                )?;
  1456→            }
  1457→
  1458→            IRInstr::CtEq {
  1459→                dst,
  1460→                lhs,
  1461→                rhs,
  1462→                ty: _,
  1463→            } => {
  1464→                // ct_eq(a, b): constant-time equality check using XOR.
  1465→                let rd = self.resolve_reg(dst)?;
  1466→                let rn = self.resolve_reg(lhs)?;
  1467→                let rm = self.resolve_reg(rhs)?;
  1468→                // EOR rd, rn, rm (XOR)
  1469→                self.emit_instruction(Instruction::EOR { rd, rn, rm })?;
  1470→                // CMP rd, #0 + CSET rd, EQ
  1471→                self.emit_instruction(Instruction::SUB {
  1472→                    rd: Register::XZR,
  1473→                    rn: rd,
  1474→                    rm: Operand::Imm12(0),
  1475→                })?;
  1476→                self.emit_instruction(Instruction::CSET { rd, cond: crate::arm64::Condition::EQ })?;
  1477→            }
  1478→        }
  1479→        Ok(())
  1480→    }
  1481→
  1482→    /// Emit a binary operation (shared by `BinOp` and dedicated `Add`/`Sub`/…).
  1483→    ///
  1484→    /// Some ARM64 instructions (ADD, SUB, LSL, LSR, ASR) accept a 12-bit
  1485→    /// unsigned immediate operand directly.  Most others (MUL, AND, ORR, EOR,
  1486→    /// SDIV, UDIV, remainder, comparisons) **require** a register operand.
  1487→    /// When the RHS is an immediate that the target instruction cannot accept
  1488→    /// directly, we "spill" it into a scratch register (X16 / IP0) first via
  1489→    /// MOVZ + MOVK and then use that register as the operand.
  1490→    ///
  1491→    /// X16 (IP0) is chosen as the scratch register because it is **not** in
  1492→    /// the register allocator's free pool, so it will never conflict with
  1493→    /// `rd` or `rn`.
  1494→    fn emit_binop(
  1495→        &mut self,
  1496→        op: BinOpKind,
  1497→        dst: &IRValue,
  1498→        lhs: &IRValue,
  1499→        rhs: &IRValue,
  1500→        ty: Option<&IRType>,
  1501→    ) -> Result<()> {
  1502→        let width = RegWidth::from_ir_type(ty);
  1503→        let rd = self.resolve_reg(dst)?;
  1504→        let rn = self.resolve_reg(lhs)?;
  1505→
  1506→        // Determine whether the operation can accept an immediate operand.
  1507→        // ADD, SUB, LSL, LSR, ASR all have immediate forms in ARM64.
  1508→        // All other operations require a register operand.
  1509→        let supports_imm = matches!(
  1510→            op,
  1511→            BinOpKind::Add | BinOpKind::Sub | BinOpKind::Shl | BinOpKind::ShrL | BinOpKind::ShrA
  1512→        );
  1513→
  1514→        let rm = match rhs {
  1515→            IRValue::Immediate(v) => {
  1516→                if supports_imm && *v >= 0 && *v <= 4095 {
  1517→                    // Small immediate that the instruction accepts directly.
  1518→                    Operand::Imm12(*v as u16)
  1519→                } else {
  1520→                    // Load the immediate into scratch register X16 (IP0).
  1521→                    // X16 is not in the register allocator's pool, so it cannot
  1522→                    // conflict with `rd` or `rn`.
  1523→                    let scratch = Register::X16;
  1524→                    self.emit_load_immediate_with_width(scratch, *v, width)?;
  1525→                    Operand::Reg {
  1526→                        reg: scratch,
  1527→                        shift: None,
  1528→                    }
  1529→                }
  1530→            }
  1531→            _ => Operand::Reg {
  1532→                reg: self.resolve_reg(rhs)?,
  1533→                shift: None,
  1534→            },
  1535→        };
  1536→
  1537→        match op {
  1538→            BinOpKind::Add => {
  1539→                self.emit_instruction_with_width(Instruction::ADD { rd, rn, rm }, width)?;
  1540→            }
  1541→            BinOpKind::Sub => {
  1542→                self.emit_instruction_with_width(Instruction::SUB { rd, rn, rm }, width)?;
  1543→            }
  1544→            BinOpKind::Mul => {
  1545→                let rm_reg = self.operand_to_reg(&rm)?;
  1546→                self.emit_instruction_with_width(Instruction::MUL { rd, rn, rm: rm_reg }, width)?;
  1547→            }
  1548→            BinOpKind::SDiv => {
  1549→                let rm_reg = self.operand_to_reg(&rm)?;
  1550→                self.emit_instruction_with_width(Instruction::SDIV { rd, rn, rm: rm_reg }, width)?;
  1551→            }
  1552→            BinOpKind::UDiv => {
  1553→                let rm_reg = self.operand_to_reg(&rm)?;
  1554→                self.emit_instruction_with_width(Instruction::UDIV { rd, rn, rm: rm_reg }, width)?;
  1555→            }
  1556→            BinOpKind::And => {
  1557→                let rm_reg = self.operand_to_reg(&rm)?;
  1558→                self.emit_instruction_with_width(Instruction::AND { rd, rn, rm: rm_reg }, width)?;
  1559→            }
  1560→            BinOpKind::Or => {
  1561→                let rm_reg = self.operand_to_reg(&rm)?;
  1562→                self.emit_instruction_with_width(Instruction::ORR { rd, rn, rm: rm_reg }, width)?;
  1563→            }
  1564→            BinOpKind::Xor => {
  1565→                let rm_reg = self.operand_to_reg(&rm)?;
  1566→                self.emit_instruction_with_width(Instruction::EOR { rd, rn, rm: rm_reg }, width)?;
  1567→            }
  1568→            BinOpKind::Shl => {
  1569→                self.emit_instruction_with_width(Instruction::LSL { rd, rn, rm }, width)?;
  1570→            }
  1571→            BinOpKind::ShrL => {
  1572→                self.emit_instruction_with_width(Instruction::LSR { rd, rn, rm }, width)?;
  1573→            }
  1574→            BinOpKind::ShrA => {
  1575→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm }, width)?;
  1576→            }
  1577→            BinOpKind::Ror | BinOpKind::Rol => {
  1578→                let rm_reg = self.operand_to_reg(&rm)?;
  1579→                if op == BinOpKind::Ror {
  1580→                    self.emit_instruction_with_width(Instruction::RORV { rd, rn, rm: rm_reg }, width)?;
  1581→                } else {
  1582→                    // ROL by Rm = ROR by -Rm (mod regsize).
  1583→                    // SUB X9, XZR, Rm; RORV Rd, Rn, X9
  1584→                    self.emit_instruction_with_width(Instruction::SUB {
  1585→                        rd: Register::X9,
  1586→                        rn: Register::XZR,
  1587→                        rm: Operand::Reg { reg: rm_reg, shift: None },
  1588→                    }, width)?;
  1589→                    self.emit_instruction_with_width(Instruction::RORV { rd, rn, rm: Register::X9 }, width)?;
  1590→                }
  1591→            }
  1592→            BinOpKind::SRem | BinOpKind::URem => {
  1593→                let rm_reg = self.operand_to_reg(&rm)?;
  1594→                let div_instr = if op == BinOpKind::SRem {
  1595→                    Instruction::SDIV { rd, rn, rm: rm_reg }
  1596→                } else {
  1597→                    Instruction::UDIV { rd, rn, rm: rm_reg }
  1598→                };
  1599→                self.emit_instruction_with_width(div_instr, width)?;
  1600→                // MSUB rd, rd, rm, rn  =>  rd = rn - rd * rm  =  dividend - quotient * divisor
  1601→                self.emit_instruction_with_width(Instruction::MSUB {
  1602→                    rd,
  1603→                    rn: rd,     // quotient (result of DIV)
  1604→                    rm: rm_reg, // divisor
  1605→                    ra: rn,     // dividend
  1606→                }, width)?;
