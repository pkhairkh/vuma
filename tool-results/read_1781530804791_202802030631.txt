     1→  2601→        IRInstr::Free { ptr: _ } => {
     2→  2602→            // Free is not directly implementable as a single instruction;
     3→  2603→            // emit a trap to catch any accidental execution at runtime.
     4→  2604→            result.push(emit_alloc_instr(Instruction::Trap, vec![], vec![]));
     5→  2605→        }
     6→  2606→
     7→  2607→        IRInstr::Phi { .. } => {
     8→  2608→            // Phi nodes are eliminated by SSA deconstruction; emit NOP.
     9→  2609→            result.push(emit_alloc_instr(Instruction::Nop, vec![], vec![]));
    10→  2610→        }
    11→  2611→    }
    12→  2612→
    13→  2613→    (result, relocations)
    14→  2614→}

    20→  2620→/// Load a 64-bit immediate into a GPR, returning encoded instruction bytes.
    21→  2621→fn ss_load_imm(rd: Gpr, val: i64) -> Vec<u8> {
    22→  2622→    let mut code = Vec::new();
    23→  2623→    let uval = val as u64;
    24→  2624→    if (-32768..=32767).contains(&val) {
    25→  2625→        code.extend_from_slice(&Instruction::Li { rt: rd, simm: val as i32 }.encode());
    26→  2626→    } else if uval <= 0xFFFF {
    27→  2627→        code.extend_from_slice(&Instruction::Ori { ra: rd, rs: Gpr::R0, uimm: uval as u32 & 0xFFFF }.encode());
    28→  2628→    } else if uval & 0xFFFF == 0 && (uval >> 32) == 0 {
    29→  2629→        code.extend_from_slice(&Instruction::Lis { rt: rd, simm: ((uval >> 16) & 0xFFFF) as i16 as i32 }.encode());
    30→  2630→        // If the high 16 bits have bit 15 set, LIS sign-extends to 64-bit all-1s.
    31→  2631→        // Clear upper 32 bits with rlwinm rd,rd,0,0,31 (equivalent to clrldi rd,rd,32)
    32→  2632→        if ((uval >> 16) & 0xFFFF) >= 0x8000 {
    33→  2633→            code.extend_from_slice(&Instruction::Rlwinm { ra: rd, rs: rd, sh: 0, mb: 0, me: 31 }.encode());
    34→  2634→        }
    35→  2635→    } else if uval >> 32 == 0 {
    36→  2636→        let hi16 = ((uval >> 16) & 0xFFFF) as u32;
    37→  2637→        let lo16 = (uval & 0xFFFF) as u32;
    38→  2638→        code.extend_from_slice(&Instruction::Lis { rt: rd, simm: hi16 as i16 as i32 }.encode());
    39→  2639→        if lo16 != 0 { code.extend_from_slice(&Instruction::Ori { ra: rd, rs: rd, uimm: lo16 }.encode()); }
    40→  2640→        // LIS sign-extends: if bit 31 of the 32-bit value is set, the upper 32 bits
    41→  2641→        // of the 64-bit register are all 1s. Clear them with rlwinm rd,rd,0,0,31
    42→  2642→        // (equivalent to clrldi rd,rd,32) which zero-extends to 64 bits.
    43→  2643→        if hi16 >= 0x8000 {
    44→  2644→            code.extend_from_slice(&Instruction::Rlwinm { ra: rd, rs: rd, sh: 0, mb: 0, me: 31 }.encode());
    45→  2645→        }
    46→  2646→    } else {
    47→  2647→        let hi32 = (uval >> 32) as u32;
    48→  2648→        let lo32 = (uval & 0xFFFF_FFFF) as u32;
    49→  2649→        code.extend_from_slice(&Instruction::Lis { rt: rd, simm: ((hi32 >> 16) & 0xFFFF) as i16 as i32 }.encode());
    50→  2650→        if hi32 & 0xFFFF != 0 { code.extend_from_slice(&Instruction::Ori { ra: rd, rs: rd, uimm: hi32 & 0xFFFF }.encode()); }
    51→  2651→        // SLDI rd, rd, 32 = RLDICR rd, rd, 32, 31 (shift left by 32 bits)
    52→  2652→        // Use rlwinm-based approach: SLDI is correct for upper-32 ops,
    53→  2653→        // but the MD-form encoding was buggy. Use SLW + clear instead:
    54→  2654→        // Load 32 into R12, then sld rd, rd, r12
    55→  2655→        code.extend_from_slice(&Instruction::Li { rt: Gpr::R12, simm: 32 }.encode());
    56→  2656→        code.extend_from_slice(&Instruction::Sld { ra: rd, rs: rd, rb: Gpr::R12 }.encode());
    57→  2657→        if (lo32 >> 16) & 0xFFFF != 0 {
    58→  2658→            let oris_word: u32 = (25u32<<26)|(rd.encoding()<<21)|(rd.encoding()<<16)|((lo32>>16)&0xFFFF);
    59→  2659→            code.extend_from_slice(&encode_word(oris_word));
    60→  2660→        }
    61→  2661→        if lo32 & 0xFFFF != 0 { code.extend_from_slice(&Instruction::Ori { ra: rd, rs: rd, uimm: lo32 & 0xFFFF }.encode()); }
    62→  2662→    }
    63→  2663→    code
    64→  2664→}

    66→  2666→/// Load from stack slot [R31 - offset_from_r31] into dst_reg.
    67→  2667→fn ss_load_from_slot(dst_reg: Gpr, offset_from_r31: i32) -> Vec<u8> {
    68→  2668→    let neg_off = -offset_from_r31;
    69→  2669→    if neg_off >= -32764 && neg_off <= 32764 && (neg_off & 3) == 0 {
    70→  2670→        Instruction::Ld { rt: dst_reg, ra: Gpr::R31, ds: neg_off }.encode().to_vec()
    71→  2671→    } else if neg_off >= -32768 && neg_off <= 32767 {
    72→  2672→        let mut code = Vec::new();
    73→  2673→        code.extend_from_slice(&Instruction::Addi { rt: Gpr::R12, ra: Gpr::R31, simm: neg_off }.encode());
    74→  2674→        code.extend_from_slice(&Instruction::Ld { rt: dst_reg, ra: Gpr::R12, ds: 0 }.encode());
    75→  2675→        code
    76→  2676→    } else {
    77→  2677→        let mut code = Vec::new();
    78→  2678→        code.extend(ss_load_imm(Gpr::R12, offset_from_r31 as i64));
    79→  2679→        code.extend_from_slice(&Instruction::Subf { rt: Gpr::R12, ra: Gpr::R12, rb: Gpr::R31 }.encode());
    80→  2680→        code.extend_from_slice(&Instruction::Ld { rt: dst_reg, ra: Gpr::R12, ds: 0 }.encode());
    81→  2681→        code
    82→  2682→    }
    83→  2683→}

    85→  2685→/// Store src_reg into stack slot [R31 - offset_from_r31].
    86→  2686→fn ss_store_to_slot(src_reg: Gpr, offset_from_r31: i32) -> Vec<u8> {
    87→  2687→    let neg_off = -offset_from_r31;
    88→  2688→    if neg_off >= -32764 && neg_off <= 32764 && (neg_off & 3) == 0 {
    89→  2689→        Instruction::Std { rs: src_reg, ra: Gpr::R31, ds: neg_off }.encode().to_vec()
    90→  2690→    } else if neg_off >= -32768 && neg_off <= 32767 {
    91→  2691→        let mut code = Vec::new();
    92→  2692→        code.extend_from_slice(&Instruction::Addi { rt: Gpr::R12, ra: Gpr::R31, simm: neg_off }.encode());
    93→  2693→        code.extend_from_slice(&Instruction::Std { rs: src_reg, ra: Gpr::R12, ds: 0 }.encode());
    94→  2694→        code
    95→  2695→    } else {
    96→  2696→        let mut code = Vec::new();
    97→  2697→        code.extend(ss_load_imm(Gpr::R12, offset_from_r31 as i64));
    98→  2698→        code.extend_from_slice(&Instruction::Subf { rt: Gpr::R12, ra: Gpr::R12, rb: Gpr::R31 }.encode());
    99→  2699→        code.extend_from_slice(&Instruction::Std { rs: src_reg, ra: Gpr::R12, ds: 0 }.encode());
   100→  2700→        code
   101→  2701→    }
   102→  2702→}

   104→  2704→/// Load an IRValue into a scratch register.
   105→  2705→fn ss_load_value(val: &IRValue, slots: &std::collections::HashMap<u32, i32>, scratch: Gpr) -> Vec<u8> {
   106→  2706→    match val {
   107→  2707→        IRValue::Register(id) => { let offset = slots.get(id).copied().unwrap_or(0); ss_load_from_slot(scratch, offset) }
   108→  2708→        IRValue::Immediate(v) => ss_load_imm(scratch, *v),
   109→  2709→        IRValue::Address(a) => ss_load_imm(scratch, *a as i64),
   110→  2710→        IRValue::Label(_) => Instruction::Li { rt: scratch, simm: 0 }.encode().to_vec(),
   111→  2711→    }
   112→  2712→}

   114→  2714→/// Emit comparison code producing 0 or 1 in dst.
   115→  2715→fn ss_emit_cmp(kind: &CmpKind, dst: Gpr, lhs: Gpr, rhs: Gpr) -> Vec<u8> {
   116→  2716→    let mut code = Vec::new();
   117→  2717→    let cmp_signed = !matches!(kind, CmpKind::ULt|CmpKind::ULe|CmpKind::UGt|CmpKind::UGe);
   118→  2718→    if cmp_signed {
   119→  2719→        code.extend_from_slice(&Instruction::Cmp { bf: CrField::CR0, l: 1, ra: lhs, rb: rhs }.encode());
   120→  2720→    } else {
   121→  2721→        code.extend_from_slice(&Instruction::Cmpl { bf: CrField::CR0, l: 1, ra: lhs, rb: rhs }.encode());
   122→  2722→    }
   123→  2723→    code.extend_from_slice(&Instruction::Li { rt: dst, simm: 0 }.encode());
   124→  2724→    let (bo, bi) = match kind {
   125→  2725→        CmpKind::Eq => (4, 2u32), CmpKind::Ne => (12, 2u32),
   126→  2726→        // SLt/ULt: branch if LT=0 (not less than) → skip setting 1; only set 1 when LT=1
   127→  2727→        CmpKind::SLt|CmpKind::ULt => (4, 0u32),
   128→  2728→        // SLe/ULe: branch if GT=1 (strictly greater) → skip setting 1; set 1 when LT=1 or EQ=1
   129→  2729→        CmpKind::SLe|CmpKind::ULe => (12, 1u32),
   130→  2730→        // SGt/UGt: branch if GT=0 (not greater) → skip setting 1; only set 1 when GT=1
   131→  2731→        CmpKind::SGt|CmpKind::UGt => (4, 1u32),
   132→  2732→        // SGe/UGe: branch if LT=1 (strictly less) → skip setting 1; set 1 when GT=1 or EQ=1
   133→  2733→        CmpKind::SGe|CmpKind::UGe => (12, 0u32),
   134→  2734→    };
   135→  2735→    code.extend_from_slice(&Instruction::Bc { bo, bi, bd: 2 }.encode());
   136→  2736→    code.extend_from_slice(&Instruction::Li { rt: dst, simm: 1 }.encode());
   137→  2737→    code
   138→  2738→}

   239→  2839→    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError> {
   240→  2840→        let func_name = func.name.clone();
   241→  2841→
   242→  2842→        // ── Phase 1: Collect all vreg IDs and compute stack layout ──
   243→  2843→        let mut all_vreg_ids: std::collections::HashSet<u32> = std::collections::HashSet::new();
   244→  2844→        for &id in func.vregs.keys() { all_vreg_ids.insert(id); }
   245→  2845→        for param in &func.params {
   246→  2846→            if let Some(id) = param.as_register() { all_vreg_ids.insert(id); }
   247→  2847→        }
   248→  2848→        for block in &func.blocks {
   249→  2849→            for instr in &block.instructions {
   250→  2850→                for id in instr.defined_regs() { all_vreg_ids.insert(id); }
   251→  2851→                for id in instr.used_regs() { all_vreg_ids.insert(id); }
   252→  2852→            }
   253→  2853→        }
   254→  2854→        for val in &func.results {
   255→  2855→            if let Some(id) = val.as_register() { all_vreg_ids.insert(id); }
   256→  2856→        }
   257→  2857→
   258→  2858→        let mut stack_alloc_vregs: std::collections::HashSet<u32> = std::collections::HashSet::new();
   259→  2859→        let mut alloc_sizes: std::collections::HashMap<u32, i32> = std::collections::HashMap::new();
   260→  2860→        for block in &func.blocks {
   261→  2861→            for instr in &block.instructions {
   262→  2862→                if let IRInstr::Alloc { dst, size } = instr {
   263→  2863→                    if let Some(id) = dst.as_register() {
   264→  2864→                        stack_alloc_vregs.insert(id);
   265→  2865→                        let aligned_size = ((*size as i32 + 15) & !15) as i32;
   266→  2866→                        alloc_sizes.insert(id, aligned_size);
   267→  2867→                    }
   268→  2868→                }
   269→  2869→            }
   270→  2870→        }
   271→  2871→
   272→  2872→        // Stack Layout (relative to R31, which = R1 + frame_size):
   273→  2873→        // [R31 + 8]  = saved LR
   274→  2874→        // [R31]      = back chain
   275→  2875→        // [R31 - 16] = saved R31
   276→  2876→        // Then alloc regions, then vreg slots at [R31 - offset]
   277→  2877→
   278→  2878→        let mut alloc_offsets: std::collections::HashMap<u32, i32> = std::collections::HashMap::new();
   279→  2879→        let mut current_offset: i32 = 24; // after saved R31 at -16
   280→  2880→
   281→  2881→        let mut alloc_vreg_ids: Vec<u32> = stack_alloc_vregs.iter().copied().collect();
   282→  2882→        alloc_vreg_ids.sort();
   283→  2883→        for &id in &alloc_vreg_ids {
   284→  2884→            let size = alloc_sizes[&id];
   285→  2885→            current_offset += size;
   286→  2886→            alloc_offsets.insert(id, current_offset);
   287→  2887→        }
   288→  2888→
   289→  2889→        let mut vreg_stack_slots: std::collections::HashMap<u32, i32> = std::collections::HashMap::new();
   290→  2890→        let mut all_vreg_ids_sorted: Vec<u32> = all_vreg_ids.iter().copied().collect();
   291→  2891→        all_vreg_ids_sorted.sort();
   292→  2892→        for &id in &all_vreg_ids_sorted {
   293→  2893→            current_offset += 8;
   294→  2894→            vreg_stack_slots.insert(id, current_offset);
   295→  2895→        }
   296→  2896→
   297→  2897→        let frame_size = ((current_offset + 15) & !15) as usize;
   298→  2898→        let fs = frame_size as i32;
   299→  2899→
   300→  2900→        // ── Phase 2: Emit prologue ──
   301→  2901→        let mut instructions: Vec<AllocatedInstruction> = Vec::new();
   302→  2902→        let mut relocations: Vec<crate::backend::RelocationEntry> = Vec::new();
   303→  2903→
   304→  2904→        // STDU R1, -frame_size(R1)
   305→  2905→        instructions.push(AllocatedInstruction {
   306→  2906→            opcode: "stdu".into(),
   307→  2907→            reads: vec![PhysicalReg::new(RegClass::Gpr, 1)],
   308→  2908→            writes: vec![PhysicalReg::new(RegClass::Gpr, 1)],
   309→  2909→            encoded: Instruction::Stdu { rs: Gpr::R1, ra: Gpr::R1, ds: -fs }.encode().to_vec(),
   310→  2910→        });
   311→  2911→        // MFLR R0
   312→  2912→        let mflr_word: u32 = (31u32 << 26) | (0u32 << 21) | (8u32 << 16) | (339 << 1);
   313→  2913→        instructions.push(AllocatedInstruction {
   314→  2914→            opcode: "mflr".into(), reads: vec![], writes: vec![PhysicalReg::new(RegClass::Gpr, 0)],
   315→  2915→            encoded: encode_word(mflr_word).to_vec(),
   316→  2916→        });
