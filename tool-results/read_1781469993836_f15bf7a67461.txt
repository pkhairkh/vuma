  2943→                                    // rlwinm operates on the low 32 bits and zero-extends the result

  2952→                                    // Use rlwnm (register-based rotate left word then AND mask)
  2953→                                    // rlwnm r3, r4, r5, 0, 31 — rotates low 32 bits of r4 left by r5[0:4],
  2954→                                    // then masks bits 0-31, clearing upper 32 bits.
  2955→                                    // Encoding: M-form, primary opcode 23
  2956→                                    // rlwnm RA, RS, RB, MB, ME: opcode=23, RS[6:10], RA[11:15], RB[16:20], MB[21:25], ME[26:30], Rc[31]
  2957→                                    let rlwnm_word: u32 = (23u32 << 26)
  2958→                                        | (Gpr::R4.encoding() << 21)
  2959→                                        | (Gpr::R3.encoding() << 16)
  2960→                                        | (Gpr::R5.encoding() << 11)

  2962→                                        | (31u32 << 1)   // ME = 31
