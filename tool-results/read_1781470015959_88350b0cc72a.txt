  3357→    fn encode_function(&self, func: &AllocatedFunction) -> Result<Vec<u8>, BackendError> {
  3358→        let mut bytes = Vec::new();
  3359→        for block in &func.blocks {
  3360→            for instr in &block.instructions {

  3366→
  3367→    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError> {
  3368→        // ── PPC64LE Linux static executable ──
  3369→        //
  3370→        // Layout:
  3371→        //   _start:  BL main           ; call main (result in R3)
  3372→        //            LI R0, 1          ; sys_exit = 1
  3373→        //            SC                ; syscall: exit(R3)
  3374→        //   <functions...>
  3375→        //
  3376→        // The _start stub is 3 instructions = 12 bytes.
  3377→        // After that come all user functions.
  3378→
  3379→        const R_PPC64_REL24: &str = "R_PPC64_REL24";
  3380→
  3381→        // ── _start stub ──
  3382→        // BL <main>      — offset 0, needs relocation
  3383→        // LI R0, 1       — offset 4 (sys_exit = 1 on PPC64 Linux)
  3384→        // SC             — offset 8
  3385→
  3386→        let start_stub_size: usize = 12; // 3 × 4-byte instructions
  3387→
  3388→        // ── Compute function offsets ──
  3389→        // _start stub comes first, then user functions.
  3390→        let mut func_offsets: std::collections::HashMap<String, usize> = std::collections::HashMap::new();
  3391→        let mut current_offset: usize = start_stub_size; // after _start
  3392→
  3393→        for func in &program.functions {
  3394→            func_offsets.insert(func.name.clone(), current_offset);
  3395→            let func_size: usize = func.blocks.iter()
  3396→                .flat_map(|b| b.instructions.iter())
  3397→                .map(|i| i.encoded.len())
  3398→                .sum();
  3399→            current_offset += func_size;
  3400→        }
  3401→
  3402→        // ── Build _start stub bytes ──
  3403→        let mut start_stub = Vec::with_capacity(start_stub_size);
  3404→
  3405→        // BL <main> — placeholder, will be patched
  3406→        // BL encoding: I-form, primary=18, LI=0, AA=0, LK=1
  3407→        start_stub.extend_from_slice(&Instruction::Bl { li: 0 }.encode());
  3408→
  3409→        // LI R0, 1 = ADDI R0, 0, 1 (sys_exit = 1)
  3410→        start_stub.extend_from_slice(&Instruction::Li { rt: Gpr::R0, simm: 1 }.encode());
  3411→
  3412→        // SC (syscall)
  3413→        start_stub.extend_from_slice(&Instruction::Sc.encode());
  3414→
  3415→        // ── Patch _start BL to main ──
  3416→        let main_key = func_offsets.keys()
  3417→            .find(|k| *k == "main" || k.starts_with("fn_main"))
  3418→            .cloned();
  3419→        if let Some(ref key) = main_key {
  3420→            let main_offset = func_offsets[key];
  3421→            // BL target = CIA + LI*4, where CIA = address of BL instruction.
  3422→            // BL is at offset 0 within all_code. main is at main_offset.
  3423→            // LI = (main_offset - 0) / 4
  3424→            let li_val = (main_offset as i64) / 4;
  3425→            // Mask to 24 bits (signed)
  3426→            let imm24 = (li_val as u32) & 0x00FF_FFFF;
  3427→            // BL word: primary=18 (bits 26-31), LI (bits 2-25), AA=0 (bit 1), LK=1 (bit 0)
  3428→            let bl_word: u32 = (18u32 << 26) | (imm24 << 2) | 1;
  3429→            start_stub[0..4].copy_from_slice(&bl_word.to_be_bytes());
  3430→        }
  3431→
  3432→        // ── Concatenate all code ──
  3433→        let mut all_code = start_stub;
  3434→        for func in &program.functions {
  3435→            for block in &func.blocks {
  3436→                for instr in &block.instructions {
  3437→                    all_code.extend_from_slice(&instr.encoded);
  3438→                }
  3439→            }
  3440→        }
  3441→
  3442→        // ── Patch BL relocations ──
  3443→        // Each function's relocations are relative to the start of that function's code.
  3444→        // We need to adjust them by the _start stub size + preceding functions' sizes.
  3445→        let mut func_code_offset: usize = start_stub_size;
  3446→        for func in &program.functions {
  3447→            for reloc in &func.relocations {
  3448→                let abs_offset = func_code_offset + reloc.offset as usize;
  3449→                if abs_offset + 4 > all_code.len() {
  3450→                    continue; // skip invalid relocations
  3451→                }
  3452→
  3453→                if reloc.reloc_type == R_PPC64_REL24 {
  3454→                    // R_PPC64_REL24: patch BL instruction's LI field (24 bits).
  3455→                    // BL target = CIA + LI*4, where CIA = address of BL instruction.
  3456→                    // So: LI = (target_addr - bl_addr) / 4
  3457→                    if let Some(&target_offset) = func_offsets.get(&reloc.symbol) {
  3458→                        let bl_addr = abs_offset as i64;
  3459→                        let target_addr = target_offset as i64;
  3460→                        let offset_words = (target_addr - bl_addr) / 4;
