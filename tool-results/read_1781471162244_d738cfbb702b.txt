     1→//! Lexer (tokeniser) for the VUMA language frontend.
     2→//!
     3→//! Transforms raw source text into a flat stream of [`Token`] values, each
     4→//! annotated with a [`Span`] and line/column position for error reporting.
     5→//! The lexer supports **full error recovery**: it never stops on the first
     6→//! error. Instead, it produces [`TokenKind::Error`] tokens and collects
     7→//! [`ParseError`] values that can be retrieved later via [`Lexer::errors`].
     8→//!
     9→//! # Example
    10→//!
    11→//! ```
    12→//! use vuma_parser::lexer::Lexer;
    13→//!
    14→//! let source = "region pool = allocate(1024);";
    15→//! let mut lexer = Lexer::new(source);
    16→//! let tokens = lexer.collect_tokens();
    17→//! for tok in &tokens {
    18→//!     println!("{:?} at {}:{} — '{}'", tok.kind, tok.line + 1, tok.column + 1, tok.lexeme);
    19→//! }
    20→//! ```
    21→
    22→use crate::error::{ParseError, ParseErrorKind, Span};
    23→use serde::{Deserialize, Serialize};
    24→
    25→// ---------------------------------------------------------------------------
    26→// Position
    27→// ---------------------------------------------------------------------------
    28→
    29→/// Source position: byte offset, line (0-based), and column (0-based).
    30→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
    31→pub struct Position {
    32→    /// Byte offset into the source text.
    33→    pub offset: usize,
    34→    /// Line number (0-based).
    35→    pub line: usize,
    36→    /// Column number (0-based, measured in Unicode code points).
    37→    pub column: usize,
    38→}
    39→
    40→impl Position {
    41→    /// Create a new position.
    42→    pub fn new(offset: usize, line: usize, column: usize) -> Self {
    43→        Self {
    44→            offset,
    45→            line,
    46→            column,
    47→        }
    48→    }
    49→
    50→    /// The initial position (offset 0, line 0, column 0).
    51→    pub fn zero() -> Self {
    52→        Self::new(0, 0, 0)
    53→    }
    54→}
    55→
    56→impl std::fmt::Display for Position {
    57→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    58→        write!(f, "{}:{}", self.line + 1, self.column + 1)
    59→    }
    60→}

    63→// Token
    64→// ---------------------------------------------------------------------------

    66→/// A single lexical token produced by the [`Lexer`].
    67→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    68→pub struct Token {
    69→    /// The classified kind of this token.
    70→    pub kind: TokenKind,
    71→    /// The exact source text that was consumed (for strings this is the
    72→    /// decoded content; for everything else it is the raw source slice).
    73→    pub lexeme: String,
    74→    /// Byte-offset span within the original source.
    75→    pub span: Span,
    76→    /// Line number where this token starts (0-based).
    77→    pub line: usize,
    78→    /// Column number where this token starts (0-based).
    79→    pub column: usize,
    80→}
    81→
    82→impl Token {
    83→    /// Convenience constructor.
    84→    pub fn new(
    85→        kind: TokenKind,
    86→        lexeme: impl Into<String>,
    87→        span: Span,
    88→        line: usize,
    89→        column: usize,
    90→    ) -> Self {
    91→        Self {
    92→            kind,
    93→            lexeme: lexeme.into(),
    94→            span,
    95→            line,
    96→            column,
    97→        }
    98→    }
    99→
   100→    /// True when this token is the end-of-file sentinel.
   101→    pub fn is_eof(&self) -> bool {
   102→        self.kind == TokenKind::Eof
   103→    }
   104→}

   624→        "sizeof" => Some(TokenKind::Sizeof),
   625→        "alignof" => Some(TokenKind::Alignof),
   626→
   627→        // Option/Result sugar
   628→        "Option" => Some(TokenKind::OptionKw),
   629→        "Some" => Some(TokenKind::SomeKw),
   630→        "None" => Some(TokenKind::NoneKw),
   631→        "Result" => Some(TokenKind::ResultKw),
   632→        "Ok" => Some(TokenKind::OkKw),
   633→        "Err" => Some(TokenKind::ErrKw),
   634→
   635→        _ => None,
   636→    }
   637→}

   639→// ---------------------------------------------------------------------------
   640→// Lexer
   641→// ---------------------------------------------------------------------------
   642→
   643→/// A streaming lexer that converts source text into [`Token`] values.
   644→///
   645→/// The lexer is *pull*-based: the caller invokes [`Lexer::next_token`]
   646→/// repeatedly until a token with [`TokenKind::Eof`] is returned.
   647→///
   648→/// **Error recovery**: the lexer never panics or returns `Err`. When it
   649→/// encounters invalid input it produces [`TokenKind::Error`] tokens and
   650→/// records the error internally. Call [`Lexer::errors`] or
   651→/// [`Lexer::take_errors`] to retrieve accumulated errors.
   652→///
   653→/// Whitespace and non-doc comments are silently skipped. Doc comments
   654→/// (`///` and `//!`) are emitted as tokens for downstream tooling.
   655→pub struct Lexer<'src> {
   656→    /// Full source text.
   657→    source: &'src str,
   658→    /// Characters yet to be consumed.
   659→    chars: std::iter::Peekable<std::str::Chars<'src>>,
   660→    /// Current byte offset into `source`.
   661→    offset: usize,
   662→    /// Current line number (0-based).
   663→    line: usize,
   664→    /// Current column number (0-based).
   665→    column: usize,
   666→    /// Lookahead token already consumed but not yet handed to the caller.
   667→    peeked: Option<Token>,
   668→    /// Cached EOF token used as a safe fallback in [`peek`] when the
   669→    /// peeked buffer is unexpectedly empty (defensive guard against panic).
   670→    eof_token: Token,
   671→    /// Accumulated lexical errors (for error recovery).
   672→    errors: Vec<ParseError>,
   673→}
   674→
   675→impl<'src> Lexer<'src> {
   676→    /// Create a new lexer over the given source text.
   677→    pub fn new(source: &'src str) -> Self {
   678→        Self {
   679→            source,
   680→            chars: source.chars().peekable(),
   681→            offset: 0,
   682→            line: 0,
   683→            column: 0,
   684→            peeked: None,
   685→            eof_token: Token::new(TokenKind::Eof, "", Span::new(0, 0), 0, 0),
   686→            errors: Vec::new(),
   687→        }
   688→    }
   689→
   690→    // -- public API ----------------------------------------------------------
   691→
   692→    /// Consume and return the next token from the source.
   693→    ///
   694→    /// On end of input a token with kind [`TokenKind::Eof`] is returned.
   695→    /// Lexical errors are reported via [`TokenKind::Error`] tokens and
   696→    /// collected internally (see [`Lexer::errors`]).
   697→    pub fn next_token(&mut self) -> Token {
   698→        if let Some(tok) = self.peeked.take() {
   699→            return tok;
   700→        }
   701→        self.advance()
   702→    }
   703→
   704→    /// Peek at the next token without consuming it.
   705→    ///
   706→    /// Successive calls to `peek` return the same token until
   707→    /// [`Lexer::next_token`] is called.
   708→    pub fn peek(&mut self) -> &Token {
   709→        if self.peeked.is_none() {
   710→            self.peeked = Some(self.advance());
   711→        }
   712→        // Defensive: if `peeked` is still `None` after the branch above
   713→        // (should never happen), fall back to the cached EOF token instead
   714→        // of panicking.
   715→        self.peeked.as_ref().unwrap_or(&self.eof_token)
   716→    }
   717→
   718→    /// Consume all remaining tokens and return them (including the final Eof).
   719→    pub fn collect_tokens(&mut self) -> Vec<Token> {
   720→        let mut tokens = Vec::new();
   721→        loop {
   722→            let tok = self.next_token();
   723→            let is_eof = tok.is_eof();
   724→            tokens.push(tok);
   725→            if is_eof {
   726→                break;
   727→            }
   728→        }
   729→        tokens
   730→    }
   731→
   732→    /// Return the accumulated lexical errors.
   733→    pub fn errors(&self) -> &[ParseError] {
   734→        &self.errors
   735→    }
   736→
   737→    /// Take all accumulated errors, leaving the internal buffer empty.
   738→    pub fn take_errors(&mut self) -> Vec<ParseError> {
   739→        std::mem::take(&mut self.errors)
   740→    }
   741→
   742→    /// Return the source text being lexed.
   743→    pub fn source(&self) -> &'src str {
   744→        self.source
   745→    }
   746→
   747→    // -- core advance --------------------------------------------------------
   748→
   749→    /// Core token production (bypasses peeked buffer).
   750→    fn advance(&mut self) -> Token {
   751→        // Loop: skip whitespace and non-doc comments until we find a real
   752→        // token or a doc-comment token.
   753→        loop {
   754→            self.skip_whitespace();
   755→
   756→            let start = self.offset;
   757→            let line = self.line;
   758→            let column = self.column;
   759→
   760→            let ch = match self.chars.peek() {
   761→                Some(&c) => c,
   762→                None => return self.eof_token(start, line, column),
   763→            };
   764→
   765→            // ---- Comment handling ----
   766→            if ch == '/' {
   767→                let second = self.peek_next(1);
   768→                if second == Some('/') {
   769→                    // `//` — line comment or doc comment
   770→                    self.bump(); // consume first /
   771→                    self.bump(); // consume second /
   772→                    let third = self.chars.peek().copied();
   773→                    if third == Some('/') {
   774→                        // `///` outer doc comment — emit as token
   775→                        self.bump(); // consume third /
   776→                        self.consume_to_eol();
   777→                        return self.make_token(TokenKind::DocComment, start, line, column);
   778→                    } else if third == Some('!') {
   779→                        // `//!` module doc comment — emit as token
   780→                        self.bump(); // consume !
   781→                        self.consume_to_eol();
   782→                        return self.make_token(TokenKind::ModuleDoc, start, line, column);
   783→                    } else {
   784→                        // Regular line comment — skip
   785→                        self.consume_to_eol();
   786→                        continue; // loop back for next token
   787→                    }
   788→                } else if second == Some('*') {
   789→                    // `/* */` block comment — skip
   790→                    self.bump(); // consume /
   791→                    self.bump(); // consume *
   792→                    self.skip_block_comment(start);
   793→                    continue; // loop back
   794→                } else {
   795→                    // Just `/` — the division operator; fall through to main match
   796→                    break;
   797→                }
   798→            }
   799→
   800→            // Not whitespace, not a comment — lex the token.
   801→            break;
   802→        }
   803→
   804→        self.lex_token()
   805→    }
   806→
   807→    /// Lex a single token starting at the current position.
   808→    fn lex_token(&mut self) -> Token {
   809→        let start = self.offset;
   810→        let line = self.line;
   811→        let column = self.column;
   812→
   813→        let ch = match self.chars.peek() {
   814→            Some(&c) => c,
   815→            None => return self.eof_token(start, line, column),
   816→        };
   817→
   818→        let kind = match ch {
   819→            // ---- Delimiters ----
   820→            '(' => {
   821→                self.bump();
   822→                TokenKind::LParen
   823→            }
   824→            ')' => {
   825→                self.bump();
   826→                TokenKind::RParen
   827→            }
   828→            '{' => {
   829→                self.bump();
   830→                TokenKind::LBrace
   831→            }
   832→            '}' => {
   833→                self.bump();
   834→                TokenKind::RBrace
   835→            }
   836→            '[' => {
   837→                self.bump();
   838→                TokenKind::LBracket
   839→            }
   840→            ']' => {
   841→                self.bump();
   842→                TokenKind::RBracket
   843→            }
   844→
   845→            // ---- Simple operators ----
   846→            '+' => return self.lex_plus(start, line, column),
   847→            '*' => return self.lex_star(start, line, column),
   848→            '/' => return self.lex_slash(start, line, column),
   849→            '%' => return self.lex_percent(start, line, column),
   850→            '^' => return self.lex_caret(start, line, column),
   851→            '~' => {
   852→                self.bump();
   853→                TokenKind::Tilde
   854→            }
   855→            '@' => {
   856→                self.bump();
   857→                TokenKind::Ampersat
   858→            }
   859→            '#' => {
   860→                self.bump();
   861→                TokenKind::Hash
   862→            }
   863→            '$' => {
   864→                self.bump();
   865→                TokenKind::Dollar
   866→            }
   867→            '?' => {
   868→                self.bump();
   869→                TokenKind::Question
   870→            }
   871→            ';' => {
   872→                self.bump();
   873→                TokenKind::Semicolon
   874→            }
   875→            ',' => {
   876→                self.bump();
   877→                TokenKind::Comma
   878→            }
   879→
   880→            // ---- Multi-char operators ----
   881→            '-' => return self.lex_minus(start, line, column),
   882→            '!' => return self.lex_bang(start, line, column),
   883→            '=' => return self.lex_eq(start, line, column),
   884→            '<' => return self.lex_lt(start, line, column),
   885→            '>' => return self.lex_gt(start, line, column),
   886→            '&' => return self.lex_ampersand(start, line, column),
   887→            '|' => return self.lex_pipe(start, line, column),
   888→            ':' => return self.lex_colon(start, line, column),
   889→            '.' => return self.lex_dot(start, line, column),
   890→
   891→            // ---- Literals ----
   892→            '0'..='9' => return self.lex_number(start, line, column),
   893→            '"' => return self.lex_string(start, line, column),
   894→            '\'' => return self.lex_char(start, line, column),
   895→
   896→            // Byte-string prefix `b"…"`
   897→            'b' => {
   898→                if self.peek_next(1) == Some('"') {
   899→                    self.bump(); // consume 'b'
   900→                    return self.lex_byte_string(start, line, column);
   901→                }
   902→                return self.lex_ident(start, line, column);
   903→            }
   904→
   905→            // Raw-string prefix `r"…"` or `r#"…"#`
   906→            'r' => {
   907→                let second = self.peek_next(1);
   908→                if second == Some('"') || second == Some('#') {
   909→                    self.bump(); // consume 'r'
   910→                    return self.lex_raw_string(start, line, column);
   911→                }
   912→                return self.lex_ident(start, line, column);
   913→            }
   914→
   915→            // Format-string prefix `f"…"`
   916→            'f' => {
   917→                if self.peek_next(1) == Some('"') {
   918→                    self.bump(); // consume 'f'
   919→                    return self.lex_format_string(start, line, column);
   920→                }
   921→                return self.lex_ident(start, line, column);
   922→            }
   923→
   924→            // ---- Identifier, keyword, or underscore ----
   925→            '_' => {
   926→                return self.lex_ident(start, line, column);
   927→            }
   928→            c if c.is_ascii_alphabetic() => {
   929→                return self.lex_ident(start, line, column);
   930→            }
   931→
   932→            // ---- Unknown character ----
   933→            _ => {
   934→                self.bump();
   935→                self.push_error(
   936→                    format!("unexpected character '{}'", ch),
   937→                    Span::new(start, self.offset),
   938→                );
   939→                TokenKind::Error
   940→            }
   941→        };
   942→
   943→        self.make_token(kind, start, line, column)
   944→    }
   945→

  1148→    /// Lex a numeric literal (integer or float).
  1149→    ///
  1150→    /// Handles: decimal, `0x` hex, `0b` binary, `0o` octal, float with
  1151→    /// `.` and/or exponent `e`/`E`.  Underscore separators are allowed.
  1152→    fn lex_number(&mut self, start: usize, line: usize, column: usize) -> Token {
  1153→        let first = self.chars.peek().copied();
  1154→        debug_assert!(first.is_some_and(|c| c.is_ascii_digit()));
  1155→
  1156→        if first == Some('0') {
  1157→            let second = self.peek_next(1);
  1158→            match second {
  1159→                Some('x') | Some('X') => {
  1160→                    // Hex prefix — only if followed by at least one hex digit
  1161→                    let third = self.peek_next(2);
  1162→                    if third.is_some_and(|c| c.is_ascii_hexdigit() || c == '_') {
  1163→                        self.bump(); // consume '0'
  1164→                        self.bump(); // consume 'x'/'X'
  1165→                        return self.lex_hex_digits(start, line, column);
  1166→                    }
  1167→                    // `0x` without digits -> treat `0` as integer
  1168→                    self.bump(); // consume '0'
  1169→                    return self.make_token(TokenKind::Number, start, line, column);
  1170→                }
  1171→                Some('b') | Some('B') => {
  1172→                    let third = self.peek_next(2);
  1173→                    if third.is_some_and(|c| c == '0' || c == '1' || c == '_') {
  1174→                        self.bump();
  1175→                        self.bump(); // consume '0b'
  1176→                        return self.lex_binary_digits(start, line, column);
  1177→                    }
  1178→                    self.bump(); // consume '0'
  1179→                    self.consume_decimal_digits();
  1180→                    return self.check_float_or_int(start, line, column);
  1181→                }
  1182→                Some('o') | Some('O') => {
  1183→                    let third = self.peek_next(2);
  1184→                    if third.is_some_and(|c| ('0'..='7').contains(&c) || c == '_') {
  1185→                        self.bump();
  1186→                        self.bump(); // consume '0o'
  1187→                        return self.lex_octal_digits(start, line, column);
  1188→                    }
  1189→                    self.bump(); // consume '0'
  1190→                    self.consume_decimal_digits();
  1191→                    return self.check_float_or_int(start, line, column);
  1192→                }
  1193→                Some('e') | Some('E') => {
  1194→                    // 0e10 — float with exponent
  1195→                    self.bump(); // consume '0'
  1196→                    self.lex_exponent();
  1197→                    return self.make_token(TokenKind::Float, start, line, column);
  1198→                }
  1199→                Some('.') => {
  1200→                    // 0.xxx — float
  1201→                    let third = self.peek_next(2);
  1202→                    if third.is_some_and(|c| c.is_ascii_digit()) {
  1203→                        self.bump(); // consume '0'
  1204→                        self.bump(); // consume '.'
  1205→                        self.consume_decimal_digits();
  1206→                        self.lex_exponent();
  1207→                        return self.make_token(TokenKind::Float, start, line, column);
  1208→                    }
  1209→                    // 0. -> integer 0 then dot
  1210→                    self.bump(); // consume '0'
  1211→                    return self.make_token(TokenKind::Number, start, line, column);
  1212→                }
  1213→                _ => {
  1214→                    // Plain 0, or 007 etc.
  1215→                    self.bump(); // consume '0'
  1216→                    self.consume_decimal_digits();
  1217→                    return self.check_float_or_int(start, line, column);
  1218→                }
  1219→            }
  1220→        }
  1221→
  1222→        // Non-zero decimal start
  1223→        self.consume_decimal_digits();
  1224→        self.check_float_or_int(start, line, column)
  1225→    }
  1226→
  1227→    /// After consuming the integer part, check for a float suffix (`.` or `e`).
  1228→    fn check_float_or_int(&mut self, start: usize, line: usize, column: usize) -> Token {
  1229→        if self.chars.peek() == Some(&'.') {
  1230→            let next = self.peek_next(1);
  1231→            if next.is_some_and(|c| c.is_ascii_digit()) {
  1232→                self.bump(); // consume '.'
  1233→                self.consume_decimal_digits();
  1234→                self.lex_exponent();
  1235→                return self.make_token(TokenKind::Float, start, line, column);
  1236→            }
  1237→            // `.` not followed by a digit — integer followed by Dot/DotDot
  1238→        }
  1239→        if self.chars.peek() == Some(&'e') || self.chars.peek() == Some(&'E') {
  1240→            self.lex_exponent();
  1241→            return self.make_token(TokenKind::Float, start, line, column);
  1242→        }
  1243→        self.make_token(TokenKind::Number, start, line, column)
  1244→    }
  1245→
  1246→    fn consume_decimal_digits(&mut self) {
  1247→        while let Some(&c) = self.chars.peek() {
  1248→            if c.is_ascii_digit() || c == '_' {
  1249→                self.bump();
  1250→            } else {
  1251→                break;
  1252→            }
  1253→        }
  1254→    }

  1256→    fn lex_hex_digits(&mut self, start: usize, line: usize, column: usize) -> Token {
  1257→        let digits_start = self.offset;
  1258→        while let Some(&c) = self.chars.peek() {
  1259→            if c.is_ascii_hexdigit() || c == '_' {
  1260→                self.bump();
  1261→            } else {
  1262→                break;
  1263→            }
  1264→        }
  1265→        if self.offset == digits_start {
  1266→            self.push_error("empty hex digits after '0x'", Span::new(start, self.offset));
  1267→            return self.make_token(TokenKind::Error, start, line, column);
  1268→        }
  1269→        self.make_token(TokenKind::Address, start, line, column)
  1270→    }

  1272→    fn lex_binary_digits(&mut self, start: usize, line: usize, column: usize) -> Token {
  1273→        let digits_start = self.offset;
  1274→        while let Some(&c) = self.chars.peek() {
  1275→            if c == '0' || c == '1' || c == '_' {
  1276→                self.bump();
  1277→            } else {
  1278→                break;
  1279→            }
  1280→        }
  1281→        if self.offset == digits_start {
  1282→            self.push_error(
  1283→                "empty binary digits after '0b'",
  1284→                Span::new(start, self.offset),
  1285→            );
  1286→            return self.make_token(TokenKind::Error, start, line, column);
  1287→        }
  1288→        self.make_token(TokenKind::Number, start, line, column)
  1289→    }

  1291→    fn lex_octal_digits(&mut self, start: usize, line: usize, column: usize) -> Token {
  1292→        let digits_start = self.offset;
  1293→        while let Some(&c) = self.chars.peek() {
  1294→            if ('0'..='7').contains(&c) || c == '_' {
  1295→                self.bump();
  1296→            } else {
  1297→                break;
  1298→            }
  1299→        }
  1300→        if self.offset == digits_start {
  1301→            self.push_error(
  1302→                "empty octal digits after '0o'",
  1303→                Span::new(start, self.offset),
  1304→            );
  1305→            return self.make_token(TokenKind::Error, start, line, column);
  1306→        }
  1307→        self.make_token(TokenKind::Number, start, line, column)
  1308→    }
  1309→
  1310→    /// Consume an optional exponent part: `[eE][+-]?[0-9_]+`.
  1311→    fn lex_exponent(&mut self) {
  1312→        if self.chars.peek() == Some(&'e') || self.chars.peek() == Some(&'E') {
  1313→            self.bump(); // consume 'e'/'E'
  1314→            if self.chars.peek() == Some(&'+') || self.chars.peek() == Some(&'-') {
  1315→                self.bump(); // consume sign
  1316→            }
  1317→            while let Some(&c) = self.chars.peek() {
  1318→                if c.is_ascii_digit() || c == '_' {
  1319→                    self.bump();
  1320→                } else {
  1321→                    break;
  1322→                }
  1323→            }
  1324→        }
  1325→    }

  1327→    /// Lex a double-quoted string literal with escape processing.
  1328→    fn lex_string(&mut self, start: usize, line: usize, column: usize) -> Token {
  1329→        self.bump(); // consume opening "
  1330→        let mut decoded = String::new();
  1331→
  1332→        loop {
  1333→            match self.chars.peek() {
  1334→                Some(&'"') => {
  1335→                    self.bump(); // consume closing "
  1336→                    break;
  1337→                }
  1338→                Some(&'\\') => {
  1339→                    self.bump(); // consume backslash
  1340→                    match self.chars.peek() {
  1341→                        Some(&'n') => {
  1342→                            self.bump();
  1343→                            decoded.push('\n');
  1344→                        }
  1345→                        Some(&'t') => {
  1346→                            self.bump();
  1347→                            decoded.push('\t');
  1348→                        }
  1349→                        Some(&'r') => {
  1350→                            self.bump();
  1351→                            decoded.push('\r');
  1352→                        }
  1353→                        Some(&'\\') => {
  1354→                            self.bump();
  1355→                            decoded.push('\\');
  1356→                        }
  1357→                        Some(&'"') => {
  1358→                            self.bump();
  1359→                            decoded.push('"');
  1360→                        }
  1361→                        Some(&'0') => {
  1362→                            self.bump();
  1363→                            decoded.push('\0');
  1364→                        }
  1365→                        Some(&'x') => {
  1366→                            self.bump(); // consume 'x'
  1367→                            let h1 = self.chars.peek().and_then(|c| c.to_digit(16));
  1368→                            if let Some(h1) = h1 {
  1369→                                self.bump();
  1370→                                let h2 = self.chars.peek().and_then(|c| c.to_digit(16));
  1371→                                if let Some(h2) = h2 {
  1372→                                    self.bump();
  1373→                                    decoded.push(char::from((h1 * 16 + h2) as u8));
  1374→                                } else {
  1375→                                    self.push_error(
  1376→                                        "incomplete hex escape in string",
  1377→                                        Span::new(self.offset.saturating_sub(2), self.offset),
  1378→                                    );
  1379→                                }
  1380→                            } else {
  1381→                                self.push_error(
  1382→                                    "incomplete hex escape in string",
  1383→                                    Span::new(self.offset.saturating_sub(1), self.offset),
  1384→                                );
  1385→                            }
  1386→                        }
  1387→                        Some(&'u') => {
  1388→                            // Unicode escape: \u{XXXX}
  1389→                            self.bump(); // consume 'u'
  1390→                            if self.chars.peek() == Some(&'{') {
  1391→                                self.bump(); // consume '{'
  1392→                                let mut hex_str = String::new();
  1393→                                while let Some(&c) = self.chars.peek() {
  1394→                                    if c.is_ascii_hexdigit() {
  1395→                                        self.bump();
  1396→                                        hex_str.push(c);
  1397→                                    } else {
  1398→                                        break;
  1399→                                    }
  1400→                                }
  1401→                                if self.chars.peek() == Some(&'}') {
  1402→                                    self.bump(); // consume '}'
  1403→                                    if let Ok(code_point) = u32::from_str_radix(&hex_str, 16) {
  1404→                                        if let Some(ch) = char::from_u32(code_point) {
  1405→                                            decoded.push(ch);
  1406→                                        } else {
  1407→                                            self.push_error(
  1408→                                                "invalid unicode code point in string",
  1409→                                                Span::new(start, self.offset),
  1410→                                            );
  1411→                                        }
  1412→                                    } else {
  1413→                                        self.push_error(
  1414→                                            "invalid unicode escape in string",
  1415→                                            Span::new(start, self.offset),
  1416→                                        );
  1417→                                    }
  1418→                                } else {
  1419→                                    self.push_error(
  1420→                                        "expected '}' after unicode escape",
  1421→                                        Span::new(start, self.offset),
  1422→                                    );
  1423→                                }
  1424→                            } else {
  1425→                                self.push_error(
  1426→                                    "expected '{' after \\u escape",
  1427→                                    Span::new(self.offset.saturating_sub(1), self.offset),
  1428→                                );
  1429→                            }
  1430→                        }
  1431→                        Some(&c) => {
  1432→                            self.bump();
  1433→                            decoded.push(c);
  1434→                        }
  1435→                        None => {
  1436→                            self.push_error(
  1437→                                "unterminated escape in string literal",
  1438→                                Span::new(start, self.offset),
  1439→                            );
  1440→                            return self.make_token_with_lexeme(
  1441→                                TokenKind::Error,
  1442→                                decoded,
  1443→                                start,
  1444→                                line,
  1445→                                column,
  1446→                            );
  1447→                        }
  1448→                    }
  1449→                }
  1450→                Some(&'\n') => {
  1451→                    self.push_error(
  1452→                        "unterminated string literal (newline)",
  1453→                        Span::new(start, self.offset),
  1454→                    );
  1455→                    return self.make_token_with_lexeme(
  1456→                        TokenKind::Error,
  1457→                        decoded,
  1458→                        start,
  1459→                        line,
  1460→                        column,
  1461→                    );
  1462→                }
  1463→                Some(&c) => {
  1464→                    self.bump();
  1465→                    decoded.push(c);
  1466→                }
  1467→                None => {
  1468→                    self.push_error("unterminated string literal", Span::new(start, self.offset));
  1469→                    return self.make_token_with_lexeme(
  1470→                        TokenKind::Error,
  1471→                        decoded,
  1472→                        start,
  1473→                        line,
  1474→                        column,
  1475→                    );
  1476→                }
  1477→            }
  1478→        }
  1479→
  1480→        self.make_token_with_lexeme(TokenKind::String, decoded, start, line, column)
  1481→    }

  1483→    /// Lex a character literal: `'c'` or `'\n'` etc.
  1484→    fn lex_char(&mut self, start: usize, line: usize, column: usize) -> Token {
  1485→        self.bump(); // consume opening '
  1486→
  1487→        let decoded = if self.chars.peek() == Some(&'\\') {
  1488→            self.bump(); // consume backslash
  1489→            match self.chars.peek() {
  1490→                Some(&'n') => {
  1491→                    self.bump();
  1492→                    Some('\n')
  1493→                }
  1494→                Some(&'t') => {
  1495→                    self.bump();
  1496→                    Some('\t')
  1497→                }
  1498→                Some(&'r') => {
  1499→                    self.bump();
  1500→                    Some('\r')
  1501→                }
  1502→                Some(&'\\') => {
  1503→                    self.bump();
  1504→                    Some('\\')
  1505→                }
  1506→                Some(&'\'') => {
  1507→                    self.bump();
  1508→                    Some('\'')
  1509→                }
  1510→                Some(&'0') => {
  1511→                    self.bump();
  1512→                    Some('\0')
  1513→                }
  1514→                Some(&'x') => {
  1515→                    self.bump(); // consume 'x'
  1516→                    let h1 = self.chars.peek().and_then(|c| c.to_digit(16));
  1517→                    if let Some(h1) = h1 {
  1518→                        self.bump();
  1519→                        let h2 = self.chars.peek().and_then(|c| c.to_digit(16));
  1520→                        if let Some(h2) = h2 {
  1521→                            self.bump();
  1522→                            Some(char::from((h1 * 16 + h2) as u8))
  1523→                        } else {
  1524→                            self.push_error(
  1525→                                "incomplete hex escape in char literal",
  1526→                                Span::new(self.offset.saturating_sub(2), self.offset),
  1527→                            );
  1528→                            None
  1529→                        }
  1530→                    } else {
  1531→                        self.push_error(
  1532→                            "incomplete hex escape in char literal",
  1533→                            Span::new(self.offset.saturating_sub(1), self.offset),
  1534→                        );
  1535→                        None
  1536→                    }
  1537→                }
  1538→                Some(&c) => {
  1539→                    self.bump();
  1540→                    Some(c)
  1541→                }
  1542→                None => {
  1543→                    self.push_error(
  1544→                        "unterminated escape in char literal",
  1545→                        Span::new(start, self.offset),
  1546→                    );
  1547→                    None
  1548→                }
  1549→            }
  1550→        } else if let Some(&c) = self.chars.peek() {
  1551→            self.bump();
  1552→            Some(c)
  1553→        } else {
  1554→            self.push_error("unterminated char literal", Span::new(start, self.offset));
  1555→            None
  1556→        };
  1557→
  1558→        // Expect closing '
  1559→        if self.chars.peek() == Some(&'\'') {
  1560→            self.bump();
  1561→        } else {
  1562→            self.push_error(
  1563→                "expected closing '\\'' in char literal",
  1564→                Span::new(start, self.offset),
  1565→            );
  1566→        }
  1567→
  1568→        let lexeme = decoded.map(|c| c.to_string()).unwrap_or_default();
  1569→        self.make_token_with_lexeme(TokenKind::Char, lexeme, start, line, column)
  1570→    }

  1572→    /// Lex a byte string literal: `b"…"`.
  1573→    fn lex_byte_string(&mut self, start: usize, line: usize, column: usize) -> Token {
  1574→        // 'b' already consumed; expect opening "
  1575→        if self.chars.peek() != Some(&'"') {
  1576→            self.push_error(
  1577→                "expected '\"' after 'b' in byte string",
  1578→                Span::new(start, self.offset),
  1579→            );
  1580→            return self.make_token(TokenKind::Error, start, line, column);
  1581→        }
  1582→        self.bump(); // consume opening "
  1583→
  1584→        while let Some(&c) = self.chars.peek() {
  1585→            match c {
  1586→                '"' => {
  1587→                    self.bump();
  1588→                    break;
  1589→                }
  1590→                '\\' => {
  1591→                    self.bump(); // consume backslash
  1592→                    if self.chars.peek().is_some() {
  1593→                        self.bump();
  1594→                    }
  1595→                }
  1596→                '\n' => {
  1597→                    self.push_error(
  1598→                        "unterminated byte string literal (newline)",
  1599→                        Span::new(start, self.offset),
  1600→                    );
  1601→                    break;
  1602→                }
  1603→                _ => {
  1604→                    self.bump();
  1605→                }
  1606→            }
  1607→        }
  1608→
  1609→        self.make_token(TokenKind::ByteStr, start, line, column)
  1610→    }

  1612→    /// Lex a raw string literal: `r"…"`, `r#"…"#`, `r##"…"##`, etc.
  1613→    fn lex_raw_string(&mut self, start: usize, line: usize, column: usize) -> Token {
  1614→        // 'r' already consumed; count '#' delimiters
  1615→        let mut num_hashes: usize = 0;
  1616→        while self.chars.peek() == Some(&'#') {
  1617→            self.bump();
  1618→            num_hashes += 1;
  1619→        }
  1620→
  1621→        // Expect opening "
  1622→        if self.chars.peek() != Some(&'"') {
  1623→            self.push_error(
  1624→                "expected '\"' after raw string prefix",
  1625→                Span::new(start, self.offset),
  1626→            );
  1627→            return self.make_token(TokenKind::Error, start, line, column);
  1628→        }
  1629→        self.bump(); // consume opening "
  1630→
  1631→        // Read until closing `"` followed by exactly `num_hashes` `#` chars
  1632→        loop {
  1633→            match self.chars.peek() {
  1634→                Some(&'"') => {
  1635→                    self.bump(); // consume '"'
  1636→                    let mut closing_hashes: usize = 0;
  1637→                    while closing_hashes < num_hashes && self.chars.peek() == Some(&'#') {
  1638→                        self.bump();
  1639→                        closing_hashes += 1;
  1640→                    }
  1641→                    if closing_hashes == num_hashes {
  1642→                        break; // found the end
  1643→                    }
  1644→                    // Not enough closing hashes — keep reading
  1645→                }
  1646→                Some(_) => {
  1647→                    self.bump();
  1648→                }
  1649→                None => {
  1650→                    self.push_error(
  1651→                        "unterminated raw string literal",
  1652→                        Span::new(start, self.offset),
  1653→                    );
  1654→                    break;
  1655→                }
  1656→            }
  1657→        }
  1658→
  1659→        self.make_token(TokenKind::RawStr, start, line, column)
  1660→    }

  1662→    /// Lex a format string: `f"…{expr}…"`
  1663→    ///
  1664→    /// The 'f' prefix has already been consumed. The content (between the
  1665→    /// quotes) is stored in the lexeme, including `{` and `}` markers.
  1666→    /// The parser is responsible for splitting into parts.
  1667→    fn lex_format_string(&mut self, start: usize, line: usize, column: usize) -> Token {
  1668→        // 'f' already consumed; expect opening "
  1669→        if self.chars.peek() != Some(&'"') {
  1670→            self.push_error(
  1671→                "expected '\"' after format string prefix 'f'",
  1672→                Span::new(start, self.offset),
  1673→            );
  1674→            return self.make_token(TokenKind::Error, start, line, column);
  1675→        }
  1676→        self.bump(); // consume opening "
  1677→
  1678→        // Read until closing ", handling escape sequences and { } braces
  1679→        loop {
  1680→            match self.chars.peek() {
  1681→                Some(&'"') => {
  1682→                    self.bump(); // consume closing "
  1683→                    break;
  1684→                }
  1685→                Some('\\') => {
  1686→                    self.bump(); // consume backslash
  1687→                    if self.chars.peek().is_some() {
  1688→                        self.bump(); // consume escaped character
  1689→                    }
  1690→                }
  1691→                Some(&'{') => {
  1692→                    self.bump(); // consume {
  1693→                                 // If {{, it's an escaped brace, not an interpolation
  1694→                    if self.chars.peek() == Some(&'}') {
  1695→                        // Empty {} — just consume the } and continue
  1696→                        self.bump();
  1697→                    }
  1698→                    // Otherwise the } will be consumed as part of the expression
  1699→                }
  1700→                Some(&'}') => {
  1701→                    self.bump(); // consume }
  1702→                }
  1703→                Some(_) => {
  1704→                    self.bump();
  1705→                }
  1706→                None => {
  1707→                    self.push_error(
  1708→                        "unterminated format string literal",
  1709→                        Span::new(start, self.offset),
  1710→                    );
  1711→                    break;
  1712→                }
  1713→            }
  1714→        }
  1715→
  1716→        self.make_token(TokenKind::FormatStr, start, line, column)
  1717→    }

  1719→    /// Lex an identifier or keyword.
  1720→    fn lex_ident(&mut self, start: usize, line: usize, column: usize) -> Token {
  1721→        self.bump(); // consume first character
  1722→        while let Some(&c) = self.chars.peek() {
  1723→            if c.is_ascii_alphanumeric() || c == '_' {
  1724→                self.bump();
  1725→            } else {
  1726→                break;
  1727→            }
  1728→        }
  1729→        let text = &self.source[start..self.offset];
  1730→        // Special case: a standalone `_` is the Underscore token.
  1731→        if text == "_" {
  1732→            return self.make_token(TokenKind::Underscore, start, line, column);
  1733→        }
  1734→        let kind = keyword_kind(text).unwrap_or(TokenKind::Ident);
  1735→        self.make_token(kind, start, line, column)
  1736→    }

  1751→    /// Skip characters until end of line (or EOF).
  1752→    fn consume_to_eol(&mut self) {
  1753→        while let Some(&c) = self.chars.peek() {
