     1‚Üí//! # PowerPC64 Backend
     2‚Üí//!
     3‚Üí//! Implements the `Backend` trait for the PowerPC 64-bit target (ELFv2 ABI,
     4‚Üí//! little-endian default). This module provides:
     5‚Üí//!
     6‚Üí//! - `Gpr` ‚Äî General-purpose register enum (R0‚ÄìR31)
     7‚Üí//! - `Fpr` ‚Äî Floating-point register enum (F0‚ÄìF31)
     8‚Üí//! - `CrField` ‚Äî Condition register field enum (CR0‚ÄìCR7)
     9‚Üí//! - `Instruction` ‚Äî PPC64 instruction enum with correct 32-bit encoding
    10‚Üí//! - `PPC64Backend` ‚Äî `Backend` implementation that lowers IR to PPC64
    11‚Üí//!   machine code and emits ELF64 binaries
    12‚Üí//!
    13‚Üí//! ## PowerPC64 Register Convention (ELFv2 ABI)
    14‚Üí//!
    15‚Üí//! | Register(s) | Role                                  |
    16‚Üí//! |-------------|---------------------------------------|
    17‚Üí//! | R0          | Volatile (not hardwired zero!)        |
    18‚Üí//! | R1          | Stack pointer (SP)                    |
    19‚Üí//! | R2          | TOC pointer                           |
    20‚Üí//! | R3‚ÄìR10      | Argument / return registers           |
    21‚Üí//! | R11‚ÄìR12     | Volatile scratch                      |
    22‚Üí//! | R13         | Thread pointer                        |
    23‚Üí//! | R14‚ÄìR31     | Callee-saved                          |
    24‚Üí//! | F0‚ÄìF13      | FP argument / return, volatile        |
    25‚Üí//! | F14‚ÄìF31     | FP callee-saved                       |
    26‚Üí//! | CR0‚ÄìCR7     | Condition register fields (4 bits ea) |
    27‚Üí//! | LR          | Link register (SPR)                   |
    28‚Üí//! | CTR         | Count register (SPR)                  |
    29‚Üí//!
    30‚Üí//! ## Instruction Encoding
    31‚Üí//!
    32‚Üí//! All instructions are 32 bits, fixed-width. PPC64 is bi-endian; the default
    33‚Üí//! for ppc64le is little-endian byte order for both data and instructions.
    34‚Üí//! The primary opcode occupies bits \[0:5\] (MSB-first bit numbering).
    35‚Üí//!
    36‚Üí//! ## References
    37‚Üí//!
    38‚Üí//! - Power ISA Version 3.1
    39‚Üí//! - <https://openpowerfoundation.org/specifications/isa/>

   421‚Üí
   422‚Üí/// Encode a PPC64 32-bit instruction word and return as big-endian bytes.
   423‚Üífn encode_word(word: u32) -> [u8; 4] {
   424‚Üí    word.to_be_bytes()
   425‚Üí}
   426‚Üí
   427‚Üí/// Build a D-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | d[16:31]
   428‚Üífn encode_d_form(opcode: u32, rt: u32, ra: u32, d: i32) -> [u8; 4] {
   429‚Üí    let word =
   430‚Üí        ((opcode & 0x3F) << 26) | ((rt & 0x1F) << 21) | ((ra & 0x1F) << 16) | ((d as u32) & 0xFFFF);
   431‚Üí    encode_word(word)
   432‚Üí}
   433‚Üí
   434‚Üí/// Build a DS-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | ds[16:29] | xo[30:31]
   435‚Üí/// The `ds` parameter is the byte offset; the DS field stores ds/4.
   436‚Üífn encode_ds_form(opcode: u32, rt: u32, ra: u32, ds: i32, xo: u32) -> [u8; 4] {
   437‚Üí    let word = ((opcode & 0x3F) << 26)
   438‚Üí        | ((rt & 0x1F) << 21)
   439‚Üí        | ((ra & 0x1F) << 16)
   440‚Üí        | (((ds >> 2) as u32) & 0x3FFF) << 2
   441‚Üí        | (xo & 0x3);
   442‚Üí    encode_word(word)
   443‚Üí}
   444‚Üí
   445‚Üí/// Build an X-form instruction: opcode[0:5] | rS[6:10] | rA[11:15] | rB[16:20] | xo[21:30] | Rc[31]
   446‚Üífn encode_x_form(opcode: u32, rs: u32, ra: u32, rb: u32, xo: u32, rc: u32) -> [u8; 4] {
   447‚Üí    let word = ((opcode & 0x3F) << 26)
   448‚Üí        | ((rs & 0x1F) << 21)
   449‚Üí        | ((ra & 0x1F) << 16)

   461‚Üí        | ((rb & 0x1F) << 11)

   463‚Üí        | ((xo & 0x1FF) << 1)

   466‚Üí}
   467‚Üí

   470‚Üí    let word = ((opcode & 0x3F) << 26)

   481‚Üífn encode_i_form(opcode: u32, li: i32, aa: u32, lk: u32) -> [u8; 4] {
   482‚Üí    let word =
   483‚Üí        ((opcode & 0x3F) << 26) | (((li as u32) & 0x03FF_FFFF) << 2) | ((aa & 1) << 1) | (lk & 1);

   489‚Üí    let word = ((opcode & 0x3F) << 26)
   490‚Üí        | ((bo & 0x1F) << 21)
   491‚Üí        | ((bi & 0x1F) << 16)
   492‚Üí        | (((bd as u32) & 0x3FFF) << 2)
   493‚Üí        | ((aa & 1) << 1)
   494‚Üí        | (lk & 1);
   495‚Üí    encode_word(word)
   496‚Üí}
   497‚Üí

   502‚Üí        | ((bi & 0x1F) << 16)
   503‚Üí        | ((bh & 0x7) << 11)
   504‚Üí        | ((xo & 0x3FF) << 1)
   505‚Üí        | (lk & 1);
   506‚Üí    encode_word(word)
   507‚Üí}

  1218‚Üí/// Build a proper ELF64 binary for PPC64LE with 2 LOAD segments.
  1219‚Üí///
  1220‚Üí/// Produces a static executable with:
  1221‚Üí/// - Segment 1: LOAD (PF_R | PF_X) ‚Äî .text
  1222‚Üí/// - Segment 2: LOAD (PF_R | PF_W) ‚Äî .data / BSS (writable)
  1223‚Üí///
  1224‚Üí/// Uses little-endian byte order for ppc64le.  The text segment is placed
  1225‚Üí/// at a page-aligned offset so that the kernel's ELF loader can mmap() it.
  1226‚Üífn build_ppc64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1227‚Üí    const PAGE_SIZE: u64 = 0x10000; // 64 KB (PPC64 typical page size)
  1228‚Üí
  1229‚Üí    let elf_header_size: u64 = 64;
  1230‚Üí    let phdr_size: u64 = 56;
  1231‚Üí    let num_phdrs: u64 = 2;
  1232‚Üí    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1233‚Üí    // Page-align the text segment start in the file.
  1234‚Üí    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1235‚Üí    let text_size = code.len() as u64;
  1236‚Üí
  1237‚Üí    // The data segment starts on the next page after the text.
  1238‚Üí    let text_file_end = text_offset + text_size;
  1239‚Üí    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1240‚Üí    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1241‚Üí    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1242‚Üí    let entry_point = base_addr + text_offset;
  1243‚Üí
  1244‚Üí    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1245‚Üí
  1246‚Üí    // --- e_ident ---
  1247‚Üí    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1248‚Üí    elf.push(2); // ELFCLASS64
  1249‚Üí    elf.push(2); // ELFDATA2MSB (big-endian PPC64)
  1250‚Üí    elf.push(1); // EV_CURRENT
  1251‚Üí    elf.push(3); // ELFOSABI_LINUX
  1252‚Üí    elf.push(0); // padding
  1253‚Üí    elf.extend_from_slice(&[0u8; 7]); // padding
  1254‚Üí
  1255‚Üí    // --- ELF header fields ---
  1256‚Üí    elf.extend_from_slice(&2u16.to_be_bytes()); // e_type = ET_EXEC
  1257‚Üí    elf.extend_from_slice(&21u16.to_be_bytes()); // e_machine = EM_PPC64
  1258‚Üí    elf.extend_from_slice(&1u32.to_be_bytes()); // e_version
  1259‚Üí    elf.extend_from_slice(&entry_point.to_be_bytes()); // e_entry

  1263‚Üí    elf.extend_from_slice(&2u32.to_be_bytes()); // e_flags
  1264‚Üí    elf.extend_from_slice(&64u16.to_be_bytes()); // e_ehsize
  1265‚Üí    elf.extend_from_slice(&56u16.to_be_bytes()); // e_phentsize

  1267‚Üí    elf.extend_from_slice(&64u16.to_be_bytes()); // e_shentsize
  1268‚Üí    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shnum
  1269‚Üí    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shstrndx
  1270‚Üí

  1272‚Üí    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1273‚Üí    elf.extend_from_slice(&5u32.to_be_bytes()); // p_flags = PF_R | PF_X
  1274‚Üí    elf.extend_from_slice(&text_offset.to_be_bytes()); // p_offset
  1275‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_vaddr
  1276‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_paddr
  1277‚Üí    elf.extend_from_slice(&text_size.to_be_bytes()); // p_filesz
  1278‚Üí    elf.extend_from_slice(&text_size.to_be_bytes()); // p_memsz
  1279‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1280‚Üí
  1281‚Üí    // --- Program Header 2: LOAD (PF_R | PF_W) ‚Äî .data / stack ---
  1282‚Üí    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1283‚Üí    elf.extend_from_slice(&6u32.to_be_bytes()); // p_flags = PF_R | PF_W
  1284‚Üí    elf.extend_from_slice(&data_offset.to_be_bytes()); // p_offset
  1285‚Üí    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_vaddr
  1286‚Üí    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_paddr
  1287‚Üí    elf.extend_from_slice(&0u64.to_be_bytes()); // p_filesz (no initialized data)
  1288‚Üí    elf.extend_from_slice(&data_size.to_be_bytes()); // p_memsz (writable pages)
  1289‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1290‚Üí
  1291‚Üí    // --- .text section ---
  1292‚Üí    // Pad to page-aligned text_offset
  1293‚Üí    while (elf.len() as u64) < text_offset {
  1294‚Üí        elf.push(0);
  1295‚Üí    }
  1296‚Üí    elf.extend_from_slice(code);
  1297‚Üí
  1298‚Üí    // --- Pad to data segment offset (if needed) ---
  1299‚Üí    while (elf.len() as u64) < data_offset {
  1300‚Üí        elf.push(0);
  1301‚Üí    }
  1302‚Üí
  1303‚Üí    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1304‚Üí
  1305‚Üí    elf
  1306‚Üí}

  1332‚Üí/// Compute the stack frame size for an IR function on PPC64.
  1333‚Üí///
  1334‚Üí/// Sums `Alloc` instruction sizes, adds 32 bytes for the LR/CR save area
  1335‚Üí/// (per ELFv2 ABI), and rounds up to 16-byte alignment.
  1336‚Üífn ppc64_compute_frame_size(func: &IRFunction) -> usize {
  1337‚Üí    let mut total: u32 = 32; // LR save (8) + CR save (8) + back chain (8) + TOC save (8)
  1338‚Üí    for block in &func.blocks {
  1339‚Üí        for instr in &block.instructions {
  1340‚Üí            if let IRInstr::Alloc { size, .. } = instr {
  1341‚Üí                let aligned = (*size).div_ceil(16) * 16;
  1342‚Üí                total += aligned;
  1343‚Üí            }
  1344‚Üí        }
  1345‚Üí    }
  1346‚Üí    // Round up to 16-byte alignment
  1347‚Üí    total = (total + 15) & !15;
  1348‚Üí    total as usize
  1349‚Üí}

  1351‚Üí/// Allocatable GPR registers for PPC64, in priority order.
  1352‚Üí///

  1354‚Üí/// R0 is reserved (not allocatable ‚Äî in some ISA contexts rA=0 means literal zero).

  1365‚Üí    Gpr::R9,
  1366‚Üí    Gpr::R10,
  1367‚Üí    // Volatile temporary
  1368‚Üí    Gpr::R12,
  1369‚Üí    // Callee-saved (require save/restore)
  1370‚Üí    Gpr::R14,
  1371‚Üí    Gpr::R15,
  1372‚Üí    Gpr::R16,
  1373‚Üí    Gpr::R17,
  1374‚Üí    Gpr::R18,
  1375‚Üí    Gpr::R19,
  1376‚Üí    Gpr::R20,
  1377‚Üí    Gpr::R21,
  1378‚Üí    Gpr::R22,
  1379‚Üí    Gpr::R23,
  1380‚Üí    Gpr::R24,
  1381‚Üí    Gpr::R25,
  1382‚Üí    Gpr::R26,
  1383‚Üí    Gpr::R27,
  1384‚Üí    Gpr::R28,

  1390‚Üí/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1391‚Üífn map_vreg_to_gpr(
  1392‚Üí    vreg_id: u32,
  1393‚Üí    arg_index: Option<usize>,
  1394‚Üí    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1395‚Üí) -> Gpr {
  1396‚Üí    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1397‚Üí        return *gpr;
  1398‚Üí    }
  1399‚Üí
  1400‚Üí    // If this is an argument, map it to the corresponding argument register.
  1401‚Üí    if let Some(idx) = arg_index {
  1402‚Üí        if let Some(reg) = Gpr::arg_register(idx) {
  1403‚Üí            vreg_map.insert(vreg_id, reg);
  1404‚Üí            return reg;
  1405‚Üí        }
  1406‚Üí    }
  1407‚Üí
  1408‚Üí    // Otherwise, pick the first available allocatable register.
  1409‚Üí    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1410‚Üí    for &reg in ALLOCATABLE_GPRS {
  1411‚Üí        if !used.contains(&reg) {
  1412‚Üí            vreg_map.insert(vreg_id, reg);
  1413‚Üí            return reg;
  1414‚Üí        }
  1415‚Üí    }
  1416‚Üí
  1417‚Üí    // Fallback: use R11 as a scratch register.
  1418‚Üí    Gpr::R11
  1419‚Üí}

  1443‚Üí/// Load a 64-bit immediate value into a GPR, emitting the minimal number of
  1444‚Üí/// PPC64 instructions.
  1445‚Üí///
  1446‚Üí/// Strategy:
  1447‚Üí/// - **`[-32768, 32767]`**: `li rd, val` (single `addi rd, r0, val`)
  1448‚Üí/// - **`[0, 65535]`**: `ori rd, r0, val`
  1449‚Üí/// - **`0xXXXX0000` (low 16 bits zero, fits 32-bit)**: `lis rd, upper16`
  1450‚Üí/// - **Other 32-bit**: `lis rd, upper16` + `ori rd, rd, lower16`
  1451‚Üí/// - **Full 64-bit**: `lis` + `ori` + `sldi 32` + `oris` + `ori`
  1452‚Üífn load_immediate_ppc64(rd: Gpr, val: i64, out: &mut Vec<AllocatedInstruction>) {
  1453‚Üí    let rd_phys = PhysicalReg::new(RegClass::Gpr, rd.encoding());
  1454‚Üí    let uval = val as u64;
  1455‚Üí
  1456‚Üí    if (-32768..=32767).contains(&val) {
  1457‚Üí        // li rd, val (addi rd, r0, val)
  1458‚Üí        out.push(emit_alloc_instr(
  1459‚Üí            Instruction::Li {
  1460‚Üí                rt: rd,
  1461‚Üí                simm: val as i32,
  1462‚Üí            },
  1463‚Üí            vec![],
  1464‚Üí            vec![rd_phys],
  1465‚Üí        ));
  1466‚Üí    } else if uval <= 0xFFFF {
  1467‚Üí        // ori rd, r0, val (unsigned 16-bit)
  1468‚Üí        out.push(emit_alloc_instr(
  1469‚Üí            Instruction::Ori {
  1470‚Üí                ra: rd,
  1471‚Üí                rs: Gpr::R0,
  1472‚Üí                uimm: uval as u32 & 0xFFFF,
  1473‚Üí            },
  1474‚Üí            vec![],
  1475‚Üí            vec![rd_phys],
  1476‚Üí        ));
  1477‚Üí    } else if uval & 0xFFFF == 0 && (uval >> 32) == 0 {
  1478‚Üí        // lis rd, upper16 (0xXXXX0000)
  1479‚Üí        out.push(emit_alloc_instr(
  1480‚Üí            Instruction::Lis {
  1481‚Üí                rt: rd,
  1482‚Üí                simm: ((uval >> 16) & 0xFFFF) as i16 as i32,
  1483‚Üí            },
  1484‚Üí            vec![],
  1485‚Üí            vec![rd_phys],
  1486‚Üí        ));
  1487‚Üí    } else if uval >> 32 == 0 {
  1488‚Üí        // 32-bit value: lis + ori
  1489‚Üí        let hi16 = ((uval >> 16) & 0xFFFF) as u32;
  1490‚Üí        let lo16 = (uval & 0xFFFF) as u32;
  1491‚Üí        out.push(emit_alloc_instr(
  1492‚Üí            Instruction::Lis {
  1493‚Üí                rt: rd,
  1494‚Üí                simm: hi16 as i16 as i32,
  1495‚Üí            },
  1496‚Üí            vec![],
  1497‚Üí            vec![rd_phys],
  1498‚Üí        ));
  1499‚Üí        if lo16 != 0 {
  1500‚Üí            out.push(emit_alloc_instr(
  1501‚Üí                Instruction::Ori {
  1502‚Üí                    ra: rd,
  1503‚Üí                    rs: rd,
  1504‚Üí                    uimm: lo16,
  1505‚Üí                },
  1506‚Üí                vec![rd_phys],
  1507‚Üí                vec![rd_phys],
  1508‚Üí            ));
  1509‚Üí        }
  1510‚Üí    } else {
  1511‚Üí        // Full 64-bit: lis + ori + sldi 32 + oris + ori
  1512‚Üí        let hi32 = (uval >> 32) as u32;
  1513‚Üí        let lo32 = (uval & 0xFFFF_FFFF) as u32;
  1514‚Üí
  1515‚Üí        // lis rd, upper16(hi32)
  1516‚Üí        out.push(emit_alloc_instr(
  1517‚Üí            Instruction::Lis {
  1518‚Üí                rt: rd,
  1519‚Üí                simm: ((hi32 >> 16) & 0xFFFF) as i16 as i32,
  1520‚Üí            },
  1521‚Üí            vec![],
  1522‚Üí            vec![rd_phys],
  1523‚Üí        ));
  1524‚Üí
  1525‚Üí        // ori rd, rd, lower16(hi32)
  1526‚Üí        if hi32 & 0xFFFF != 0 {
  1527‚Üí            out.push(emit_alloc_instr(
  1528‚Üí                Instruction::Ori {
  1529‚Üí                    ra: rd,
  1530‚Üí                    rs: rd,
  1531‚Üí                    uimm: hi32 & 0xFFFF,
  1532‚Üí                },
  1533‚Üí                vec![rd_phys],
  1534‚Üí                vec![rd_phys],
  1535‚Üí            ));
  1536‚Üí        }
  1537‚Üí
  1538‚Üí        // sldi rd, rd, 32 = rldicr rd, rd, 32, 31
  1539‚Üí        // MD-form: [0:5]=30, [6:10]=rS, [11:15]=rA, [16:20]=SH[0:4],
  1540‚Üí        // [21:25]=ME[0:4], [26]=SH[5], [27]=ME[5], [28:30]=xo(=2), [31]=Rc(=0)
  1541‚Üí        let sh: u32 = 32;
  1542‚Üí        let me: u32 = 63 - sh;
  1543‚Üí        let sldi_word: u32 = (30u32 << 26)
  1544‚Üí            | (rd.encoding() << 21)
  1545‚Üí            | (rd.encoding() << 16)
  1546‚Üí            | ((sh & 0x1F) << 11)
  1547‚Üí            | ((me & 0x1F) << 6)
  1548‚Üí            | (((sh >> 5) & 1) << 5)
  1549‚Üí            | (((me >> 5) & 1) << 4)
  1550‚Üí            | (2u32 << 1);
  1551‚Üí        out.push(AllocatedInstruction {
  1552‚Üí            opcode: "rldicr".to_string(),
  1553‚Üí            reads: vec![rd_phys],
  1554‚Üí            writes: vec![rd_phys],
  1555‚Üí            encoded: encode_word(sldi_word).to_vec(),
  1556‚Üí        });
  1557‚Üí
  1558‚Üí        // oris rd, rd, upper16(lo32) ‚Äî primary opcode 25, D-form
  1559‚Üí        if (lo32 >> 16) & 0xFFFF != 0 {
  1560‚Üí            let oris_word: u32 = (25u32 << 26)
  1561‚Üí                | (rd.encoding() << 21)
  1562‚Üí                | (rd.encoding() << 16)
  1563‚Üí                | ((lo32 >> 16) & 0xFFFF);
  1564‚Üí            out.push(AllocatedInstruction {
  1565‚Üí                opcode: "oris".to_string(),
  1566‚Üí                reads: vec![rd_phys],
  1567‚Üí                writes: vec![rd_phys],
  1568‚Üí                encoded: encode_word(oris_word).to_vec(),
  1569‚Üí            });
  1570‚Üí        }
  1571‚Üí
  1572‚Üí        // ori rd, rd, lower16(lo32)
  1573‚Üí        if lo32 & 0xFFFF != 0 {
  1574‚Üí            out.push(emit_alloc_instr(
  1575‚Üí                Instruction::Ori {
  1576‚Üí                    ra: rd,
  1577‚Üí                    rs: rd,
  1578‚Üí                    uimm: lo32 & 0xFFFF,
  1579‚Üí                },
  1580‚Üí                vec![rd_phys],
  1581‚Üí                vec![rd_phys],
  1582‚Üí            ));
  1583‚Üí        }
  1584‚Üí    }
  1585‚Üí}

  1587‚Üí/// Resolve an `IRValue` to a physical `Gpr`, emitting immediate-load
  1588‚Üí/// instructions as needed.
  1589‚Üí///
  1590‚Üí/// - `IRValue::Register(id)` ‚Üí looks up `id` in `reg_map`
  1591‚Üí/// - `IRValue::Immediate(val)` ‚Üí loads `val` into `scratch` via
  1592‚Üí///   `load_immediate_ppc64` and returns `scratch`
  1593‚Üí/// - `IRValue::Address(addr)` ‚Üí loads `addr` into `scratch`
  1594‚Üí/// - `IRValue::Label(_)` ‚Üí loads 0 into `scratch` (placeholder for relocation)
  1595‚Üí///
  1596‚Üí/// The caller must supply a scratch register that is not live at this point
  1597‚Üí/// (e.g., `R11` for the first operand, `R12` for the second).
  1598‚Üífn resolve_gpr_ppc64(
  1599‚Üí    val: &IRValue,
  1600‚Üí    reg_map: &mut std::collections::HashMap<u32, Gpr>,
  1601‚Üí    scratch: Gpr,
  1602‚Üí    out: &mut Vec<AllocatedInstruction>,
  1603‚Üí) -> Gpr {
  1604‚Üí    match val {
  1605‚Üí        IRValue::Register(id) => map_vreg_to_gpr(*id, None, reg_map),
  1606‚Üí        IRValue::Immediate(imm) => {
  1607‚Üí            load_immediate_ppc64(scratch, *imm, out);
  1608‚Üí            scratch
  1609‚Üí        }
  1610‚Üí        IRValue::Address(addr) => {
  1611‚Üí            load_immediate_ppc64(scratch, *addr as i64, out);
  1612‚Üí            scratch
  1613‚Üí        }
  1614‚Üí        IRValue::Label(_) => {
  1615‚Üí            // Labels need linker relocation; emit li scratch, 0 as placeholder
  1616‚Üí            load_immediate_ppc64(scratch, 0, out);
  1617‚Üí            scratch
  1618‚Üí        }
  1619‚Üí    }
  1620‚Üí}
