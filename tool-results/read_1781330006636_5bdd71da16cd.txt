     1→//! # Register Allocation
     2→//!
     3→//! Provides register allocators that map IR virtual registers to ARM64
     4→//! physical registers.
     5→//!
     6→//! ## Allocators
     7→//!
     8→//! ### `RegAllocator` (legacy greedy)
     9→//!
    10→//! A simple greedy allocator that walks the IR and assigns caller-saved
    11→//! registers first, then callee-saved, spilling when all are exhausted.
    12→//! Kept for backward-compatibility with the existing emitter.
    13→//!
    14→//! ### `LinearScanAllocator` (production)
    15→//!
    16→//! A real **linear-scan** register allocator that:
    17→//!
    18→//! 1. Computes live ranges from the IR (per-function, across all blocks).
    19→//! 2. Sorts intervals by start point.
    20→//! 3. Walks intervals in order, assigning free physical registers.
    21→//! 4. When the pool is exhausted, evicts the interval whose end point is
    22→//!    farthest in the future (or spills the current one if it ends latest).
    23→//! 5. Generates spill/reload code as needed — including reloads at each
    24→//!    use position after eviction.
    25→//! 6. Applies register coalescing to eliminate redundant copies.
    26→//!
    27→//! ## ARM64 Register Conventions (AAPCS64)
    28→//!
    29→//! | Register(s) | Role                                   | Class        |
    30→//! |-------------|----------------------------------------|--------------|
    31→//! | X0–X7       | Argument / result registers            | Caller-saved |
    32→//! | X8          | Indirect result location register      | Caller-saved |
    33→//! | X9–X15      | Caller-saved temporary registers       | Caller-saved |
    34→//! | X16–X17     | Intra-procedure-call scratch (IP0/IP1) | Caller-saved |
    35→//! | X18         | Platform register                      | Reserved     |
    36→//! | X19–X28     | Callee-saved registers                 | Callee-saved |
    37→//! | X29         | Frame pointer (FP)                     | Reserved     |
    38→//! | X30         | Link register (LR)                     | Reserved     |
    39→//! | SP          | Stack pointer                          | Reserved     |
    40→//! | XZR         | Zero register                          | Reserved     |
    41→//! | V0–V7       | FP/SIMD argument / result registers    | Caller-saved |
    42→//! | V8–V15      | FP/SIMD callee-saved (lower 64 bits)   | Callee-saved |
    43→//! | V16–V31     | FP/SIMD caller-saved temporaries       | Caller-saved |
    44→

    59→/// ARM64 SIMD and Floating-Point registers (V0–V31).

   267→// ═══════════════════════════════════════════════════════════════════════════
   268→// Live Interval
   269→// ═══════════════════════════════════════════════════════════════════════════
   270→
   271→/// A live interval for a virtual register, representing the range of
   272→/// instruction positions where the register is live.
   273→///
   274→/// Positions are sequential instruction indices assigned during a linear
   275→/// pass over all blocks in a function.
   276→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   277→pub struct LiveInterval {
   278→    /// The virtual register ID.
   279→    pub vreg: IRValueId,
   280→    /// The register class (GPR or SIMD/FP).
   281→    pub class: RegClass,
   282→    /// Start position (inclusive) — the instruction that defines this vreg.
   283→    pub start: u32,
   284→    /// End position (inclusive) — the last instruction that uses this vreg.
   285→    pub end: u32,
   286→    /// Whether this interval crosses a function call.
   287→    pub crosses_call: bool,
   288→    /// List of use positions (for spill/reload decisions).
   289→    pub use_positions: Vec<u32>,
   290→    /// List of def positions.
   291→    pub def_positions: Vec<u32>,
   292→    /// Set of vregs that were coalesced into this interval.
   293→    /// When intervals are merged via coalescing, all constituent vreg IDs
   294→    /// are tracked here so that the allocator can map them all to the same
   295→    /// physical register.
   296→    pub coalesced_vregs: Vec<IRValueId>,
   297→}
   298→
   299→impl LiveInterval {
   300→    /// Create a new interval starting and ending at the given positions.
   301→    pub fn new(vreg: IRValueId, class: RegClass, start: u32, end: u32) -> Self {
   302→        Self {
   303→            vreg,
   304→            class,
   305→            start,
   306→            end,
   307→            crosses_call: false,
   308→            use_positions: Vec::new(),
   309→            def_positions: Vec::new(),
   310→            coalesced_vregs: vec![vreg],
   311→        }
   312→    }
   313→
   314→    /// Returns `true` if the interval covers the given position.
   315→    pub fn covers(&self, pos: u32) -> bool {
   316→        pos >= self.start && pos <= self.end
   317→    }
   318→
   319→    /// Extend the interval to include the given position.
   320→    pub fn extend_to(&mut self, pos: u32) {
   321→        if pos > self.end {
   322→            self.end = pos;
   323→        }
   324→        if pos < self.start {
   325→            self.start = pos;
   326→        }
   327→    }
   328→
   329→    /// Duration of the live interval.
   330→    pub fn len(&self) -> u32 {
   331→        self.end.saturating_sub(self.start)
   332→    }
   333→
   334→    /// Returns true if this interval has zero length.
   335→    pub fn is_empty(&self) -> bool {
   336→        self.start == self.end
   337→    }
   338→
   339→    /// Returns true if this interval ends before the other starts.
   340→    pub fn ends_before(&self, other: &LiveInterval) -> bool {
   341→        self.end < other.start
   342→    }
   343→
   344→    /// Returns true if this interval overlaps with the other.
   345→    pub fn overlaps(&self, other: &LiveInterval) -> bool {
   346→        self.start <= other.end && other.start <= self.end
   347→    }
   348→
   349→    /// Compute a spill weight for this interval.
   350→    ///
   351→    /// Higher weight = more important to keep in a register.
   352→    /// Weight is based on:
   353→    /// - Number of use/def positions (more references = higher weight)
   354→    /// - Whether the interval crosses a call (callee-saved is expensive to use)
   355→    /// - Loop nesting could be added here in the future
   356→    pub fn spill_weight(&self) -> u32 {
   357→        let use_count = self.use_positions.len() as u32;
   358→        let def_count = self.def_positions.len() as u32;
   359→        let base_weight = (use_count + def_count).max(1);
   360→
   361→        // Intervals that cross calls have higher weight because spilling
   362→        // around a call requires both save and restore, and using a
   363→        // callee-saved register has prologue/epilogue cost.
   364→        let call_multiplier = if self.crosses_call { 2 } else { 1 };
   365→
   366→        base_weight * call_multiplier
   367→    }
   368→
   369→    /// Returns the weight per unit of live range length — used to decide
   370→    /// which interval to evict.  Higher = more deserving of a register.
   371→    pub fn weight_per_length(&self) -> u32 {
   372→        let len = self.len().max(1);
   373→        self.spill_weight() / len
   374→    }
   375→}

   377→// ═══════════════════════════════════════════════════════════════════════════
   378→// Allocation Result
   379→// ═══════════════════════════════════════════════════════════════════════════
   380→
   381→/// The result of register allocation for a single function.
   382→///
   383→/// Contains the mapping from virtual registers to physical registers,
   384→/// spill slot assignments, and metadata needed for prologue/epilogue
   385→/// generation.
   386→#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
   387→pub struct AllocationResult {
   388→    /// Mapping from virtual register ID to physical register.
   389→    pub vreg_to_preg: HashMap<IRValueId, PhysReg>,
   390→    /// Mapping from virtual register ID to spill slot index.
   391→    /// Each slot is 8 bytes (GPR) or 16 bytes (SIMD/FP) on the stack.
   392→    pub spill_slots: HashMap<IRValueId, SpillSlot>,
   393→    /// Total number of spill slots used (for frame size calculation).
   394→    pub total_spill_slots: u32,
   395→    /// Set of callee-saved GPRs that must be saved/restored in prologue/epilogue.
   396→    pub used_callee_saved_gprs: HashSet<Register>,
   397→    /// Set of callee-saved SIMD/FP registers that must be saved/restored.
   398→    pub used_callee_saved_simd: HashSet<SimdFpRegister>,
   399→    /// Spill/reload instructions to be inserted, keyed by instruction position.
   400→    pub spill_code: BTreeMap<u32, Vec<SpillCode>>,
   401→    /// The original live intervals (for debugging / introspection).
   402→    pub live_intervals: Vec<LiveInterval>,
   403→    /// Coalescing moves that were eliminated (src_vreg, dst_preg).
   404→    pub eliminated_copies: Vec<(IRValueId, PhysReg)>,
   405→    /// Mapping from coalesced vreg IDs to the representative vreg ID.
   406→    /// When intervals are merged, all constituent vregs map to the same preg.
   407→    pub coalesced_map: HashMap<IRValueId, IRValueId>,
   408→}
   409→
   410→impl AllocationResult {
   411→    /// Create an empty allocation result.
   412→    pub fn new() -> Self {
   413→        Self {
   414→            vreg_to_preg: HashMap::new(),
   415→            spill_slots: HashMap::new(),
   416→            total_spill_slots: 0,
   417→            used_callee_saved_gprs: HashSet::new(),
   418→            used_callee_saved_simd: HashSet::new(),
   419→            spill_code: BTreeMap::new(),
   420→            live_intervals: Vec::new(),
   421→            eliminated_copies: Vec::new(),
   422→            coalesced_map: HashMap::new(),
   423→        }
   424→    }
   425→
   426→    /// Look up the physical register assigned to a virtual register.
   427→    /// Also follows the coalescing map if the vreg was merged.
   428→    pub fn get_phys_reg(&self, vreg: IRValueId) -> Option<PhysReg> {
   429→        if let Some(&preg) = self.vreg_to_preg.get(&vreg) {
   430→            return Some(preg);
   431→        }
   432→        // Follow coalescing chain.
   433→        let rep = self.coalesced_map.get(&vreg).copied().unwrap_or(vreg);
   434→        self.vreg_to_preg.get(&rep).copied()
   435→    }
   436→
   437→    /// Look up the GPR assigned to a virtual register (convenience).
   438→    pub fn get_gpr(&self, vreg: IRValueId) -> Option<Register> {
   439→        match self.get_phys_reg(vreg) {
   440→            Some(PhysReg::Gpr(r)) => Some(r),
   441→            _ => None,
   442→        }
   443→    }
   444→
   445→    /// Look up the SIMD/FP register assigned to a virtual register.
   446→    pub fn get_simd(&self, vreg: IRValueId) -> Option<SimdFpRegister> {
   447→        match self.get_phys_reg(vreg) {
   448→            Some(PhysReg::SimdFp(r)) => Some(r),
   449→            _ => None,
   450→        }
   451→    }
   452→
   453→    /// Check if a virtual register is spilled.
   454→    pub fn is_spilled(&self, vreg: IRValueId) -> bool {
   455→        if self.spill_slots.contains_key(&vreg) {
   456→            return true;
   457→        }
   458→        let rep = self.coalesced_map.get(&vreg).copied().unwrap_or(vreg);
   459→        self.spill_slots.contains_key(&rep)
   460→    }
   461→
   462→    /// Get the spill slot for a virtual register.
   463→    pub fn spill_slot(&self, vreg: IRValueId) -> Option<&SpillSlot> {
   464→        if let Some(slot) = self.spill_slots.get(&vreg) {
   465→            return Some(slot);
   466→        }
   467→        let rep = self.coalesced_map.get(&vreg).copied().unwrap_or(vreg);
   468→        self.spill_slots.get(&rep)
   469→    }
   470→
   471→    /// Calculate the total frame size needed for spill slots.
   472→    /// GPR slots are 8 bytes each; SIMD/FP slots are 16 bytes each.
   473→    pub fn spill_frame_bytes(&self) -> u32 {
   474→        self.spill_slots
   475→            .values()
   476→            .map(|slot| slot.size_bytes())
   477→            .sum()
   478→    }
   479→
   480→    /// Number of callee-saved registers that must be saved in prologue.
   481→    pub fn callee_saved_count(&self) -> usize {
   482→        self.used_callee_saved_gprs.len() + self.used_callee_saved_simd.len()
   483→    }
   484→
   485→    /// Record a coalescing: `src` was merged into `dst`'s interval.
   486→    /// After allocation, both src and dst should map to the same preg.
   487→    pub fn record_coalescing(&mut self, src: IRValueId, dst: IRValueId) {
   488→        self.coalesced_map.insert(src, dst);
   489→    }
   490→
   491→    /// Resolve a vreg through the coalescing map to the representative vreg.
   492→    pub fn resolve_vreg(&self, vreg: IRValueId) -> IRValueId {
   493→        self.coalesced_map.get(&vreg).copied().unwrap_or(vreg)
   494→    }
   495→}
   496→
   497→impl Default for AllocationResult {
   498→    fn default() -> Self {
   499→        Self::new()
   500→    }
   501→}

   504→// Spill Slot
   505→// ═══════════════════════════════════════════════════════════════════════════
   506→
   507→/// A spill slot on the stack, identified by an index and byte size.
   508→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   509→pub struct SpillSlot {
   510→    /// Slot index (sequential).
   511→    pub index: u32,
   512→    /// Offset from the frame pointer (SP or FP), in bytes.
   513→    /// Negative offset means "deeper into the stack".
   514→    pub offset: i32,
   515→    /// The register class that occupies this slot.
   516→    pub class: RegClass,
   517→}
   518→
   519→impl SpillSlot {
   520→    /// Create a new spill slot.
   521→    pub fn new(index: u32, offset: i32, class: RegClass) -> Self {
   522→        Self {
   523→            index,
   524→            offset,
   525→            class,
   526→        }
   527→    }
   528→
   529→    /// Size in bytes: 8 for GPRs, 16 for SIMD/FP registers.
   530→    pub fn size_bytes(&self) -> u32 {
   531→        match self.class {
   532→            RegClass::Gpr => 8,
   533→            RegClass::SimdFp => 16,
   534→        }
   535→    }
   536→}

   539→// Spill Code
   540→// ═══════════════════════════════════════════════════════════════════════════
   541→
   542→/// A spill or reload instruction to be inserted into the instruction stream.
   543→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   544→pub enum SpillCode {
   545→    /// Spill (store) a register to its stack slot.
   546→    /// `vreg` is the virtual register being spilled.
   547→    /// `preg` is the physical register holding the value.
   548→    /// `slot` is the spill slot to store to.
   549→    Spill {
   550→        vreg: IRValueId,
   551→        preg: PhysReg,
   552→        slot: SpillSlot,
   553→    },
   554→    /// Reload (load) a register from its stack slot.
   555→    /// `vreg` is the virtual register being reloaded.
   556→    /// `preg` is the physical register to load into.
   557→    /// `slot` is the spill slot to load from.
   558→    Reload {
   559→        vreg: IRValueId,
   560→        preg: PhysReg,
   561→        slot: SpillSlot,
   562→    },
   563→}
   564→
   565→impl std::fmt::Display for SpillCode {
   566→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   567→        match self {
   568→            SpillCode::Spill { vreg, preg, slot } => {
   569→                write!(
   570→                    f,
   571→                    "spill %v{} -> {} [slot {} offset {}]",
   572→                    vreg, preg, slot.index, slot.offset
   573→                )
   574→            }
   575→            SpillCode::Reload { vreg, preg, slot } => {
   576→                write!(
   577→                    f,
   578→                    "reload %v{} <- {} [slot {} offset {}]",
   579→                    vreg, preg, slot.index, slot.offset
   580→                )
   581→            }
   582→        }
   583→    }
   584→}

   587→// Copy / Coalesce info
   588→// ═══════════════════════════════════════════════════════════════════════════
   589→
   590→/// A copy instruction identified for potential coalescing.
   591→#[derive(Debug, Clone, PartialEq, Eq)]
   592→struct CopyInfo {
   593→    /// Source virtual register.
   594→    src: IRValueId,
   595→    /// Destination virtual register.
   596→    dst: IRValueId,
   597→    /// Instruction position.
   598→    pos: u32,
   599→    /// Register class.
   600→    class: RegClass,
   601→}

   603→// ═══════════════════════════════════════════════════════════════════════════
   604→// Live Range Computation
   605→// ═══════════════════════════════════════════════════════════════════════════
   606→
   607→/// Computes live intervals for all virtual registers in a function.
   608→///
   609→/// Assigns sequential instruction positions across all blocks (linear order),
   610→/// then computes def-use information to build intervals.  Also tracks which
   611→/// intervals cross function call sites (to guide caller-saved vs callee-saved
   612→/// allocation).
   613→pub struct LiveRangeComputer {
   614→    /// Default register class — assumed GPR unless the user annotates.
   615→    default_class: RegClass,
   616→    /// Per-vreg register class overrides.
   617→    class_overrides: HashMap<IRValueId, RegClass>,
   618→}
   619→
   620→impl LiveRangeComputer {
   621→    /// Create a new live range computer with GPR as the default class.
   622→    pub fn new() -> Self {
   623→        Self {
   624→            default_class: RegClass::Gpr,
   625→            class_overrides: HashMap::new(),
   626→        }
   627→    }
   628→
   629→    /// Set the register class for a specific virtual register.
   630→    pub fn set_class(&mut self, vreg: IRValueId, class: RegClass) {
   631→        self.class_overrides.insert(vreg, class);
   632→    }
   633→
   634→    /// Get the register class for a virtual register.
   635→    fn class_of(&self, vreg: IRValueId) -> RegClass {
   636→        self.class_overrides
   637→            .get(&vreg)
   638→            .copied()
   639→            .unwrap_or(self.default_class)
   640→    }
   641→
   642→    /// Compute live intervals for the given function.
   643→    ///
   644→    /// Returns a vector of live intervals, one per virtual register, and
   645→    /// a set of instruction positions where function calls occur.
   646→    pub fn compute(&self, func: &IRFunction) -> (Vec<LiveInterval>, BTreeSet<u32>) {
   647→        // Phase 1: Assign sequential positions to every instruction.
   648→        // We use 2*N for each instruction N, so that we can insert
   649→        // spill/reload code at positions 2*N+1 if needed.
   650→        let mut intervals: HashMap<IRValueId, LiveInterval> = HashMap::new();
   651→        let mut call_positions: BTreeSet<u32> = BTreeSet::new();
   652→        let mut copies: Vec<CopyInfo> = Vec::new();
   653→
   654→        let mut pos: u32 = 0;
   655→
   656→        for block in &func.blocks {
   657→            for instr in &block.instructions {
   658→                let def_regs = instr.defined_regs();
   659→                let use_regs = instr.used_regs();
   660→
   661→                // Detect copies: Cast with BitCast between two registers.
   662→                if let IRInstr::Cast {
   663→                    kind: crate::ir::CastKind::BitCast,
   664→                    dst: IRValue::Register(dst_id),
   665→                    src: IRValue::Register(src_id),
   666→                } = instr
   667→                {
   668→                    copies.push(CopyInfo {
   669→                        src: *src_id,
   670→                        dst: *dst_id,
   671→                        pos,
   672→                        class: self.class_of(*dst_id),
   673→                    });
   674→                }
   675→
   676→                // Process definitions.
   677→                for &vreg in &def_regs {
   678→                    let class = self.class_of(vreg);
   679→                    let interval = intervals
   680→                        .entry(vreg)
   681→                        .or_insert_with(|| LiveInterval::new(vreg, class, pos, pos));
   682→                    interval.def_positions.push(pos);
   683→                    interval.extend_to(pos);
   684→                }
   685→
   686→                // Process uses.
   687→                for &vreg in &use_regs {
   688→                    let class = self.class_of(vreg);
   689→                    let interval = intervals
   690→                        .entry(vreg)
   691→                        .or_insert_with(|| LiveInterval::new(vreg, class, pos, pos));
   692→                    interval.use_positions.push(pos);
   693→                    interval.extend_to(pos);
   694→                }
   695→
   696→                // Track function call positions.
   697→                if matches!(instr, IRInstr::Call { .. }) {
   698→                    call_positions.insert(pos);
   699→                }
   700→
   701→                pos += 2; // leave gap for spill/reload insertion
   702→            }
   703→
   704→            // Terminator uses.
   705→            match &block.terminator {
   706→                IRTerminator::Return(vals) => {
   707→                    for val in vals {
   708→                        if let IRValue::Register(vreg) = val {
   709→                            let class = self.class_of(*vreg);
   710→                            let interval = intervals
   711→                                .entry(*vreg)
   712→                                .or_insert_with(|| LiveInterval::new(*vreg, class, pos, pos));
   713→                            interval.use_positions.push(pos);
   714→                            interval.extend_to(pos);
   715→                        }
   716→                    }
   717→                }
   718→                IRTerminator::Branch { cond, .. } => {
   719→                    if let IRValue::Register(vreg) = cond {
   720→                        let class = self.class_of(*vreg);
   721→                        let interval = intervals
   722→                            .entry(*vreg)
   723→                            .or_insert_with(|| LiveInterval::new(*vreg, class, pos, pos));
   724→                        interval.use_positions.push(pos);
   725→                        interval.extend_to(pos);
   726→                    }
   727→                }
   728→                IRTerminator::Jump(_) | IRTerminator::Unreachable => {}
   729→            }
   730→            pos += 2;
   731→        }
   732→
   733→        // Phase 2: Mark intervals that cross call sites.
   734→        for interval in intervals.values_mut() {
   735→            for &call_pos in &call_positions {
   736→                if interval.start < call_pos && interval.end > call_pos {
   737→                    interval.crosses_call = true;
   738→                    break;
   739→                }
   740→            }
   741→        }
   742→
   743→        let mut result: Vec<LiveInterval> = intervals.into_values().collect();
   744→
   745→        // Phase 3: Apply register coalescing to merge copy-related intervals.
   746→        Self::coalesce_intervals(&mut result, &copies);
   747→
   748→        (result, call_positions)
   749→    }
   750→
   751→    /// Coalesce intervals connected by copy instructions.
   752→    ///
   753→    /// Two intervals can be coalesced if they are connected by a copy and
   754→    /// their combined range doesn't interfere with any other interval of
   755→    /// the same register class.
   756→    fn coalesce_intervals(intervals: &mut Vec<LiveInterval>, copies: &[CopyInfo]) {
   757→        if copies.is_empty() {
   758→            return;
   759→        }
   760→
   761→        // Build a union-find structure for coalescing.
   762→        let mut parent: HashMap<IRValueId, IRValueId> = HashMap::new();
   763→        for interval in intervals.iter() {
   764→            parent.insert(interval.vreg, interval.vreg);
   765→        }
   766→
   767→        fn find(parent: &mut HashMap<IRValueId, IRValueId>, x: IRValueId) -> IRValueId {
   768→            let root = parent[&x];
   769→            if root == x {
   770→                return x;
   771→            }
   772→            let root = find(parent, root);
   773→            parent.insert(x, root);
   774→            root
   775→        }
   776→
   777→        fn union(parent: &mut HashMap<IRValueId, IRValueId>, a: IRValueId, b: IRValueId) {
   778→            let ra = find(parent, a);
   779→            let rb = find(parent, b);
   780→            if ra != rb {
   781→                parent.insert(ra, rb);
   782→            }
   783→        }
   784→
   785→        // Build interval lookup.
   786→        let interval_map: HashMap<IRValueId, &LiveInterval> = intervals
   787→            .iter()
   788→            .map(|i| (i.vreg, i))
   789→            .collect();
   790→
   791→        // Try to coalesce each copy.
   792→        for copy in copies {
   793→            let src_interval = match interval_map.get(&copy.src) {
   794→                Some(i) => i,
   795→                None => continue,
   796→            };
   797→            let dst_interval = match interval_map.get(&copy.dst) {
   798→                Some(i) => i,
   799→                None => continue,
   800→            };
   801→
   802→            // Only coalesce same-class intervals.
   803→            if src_interval.class != dst_interval.class {
   804→                continue;
   805→            }
   806→
   807→            // Check if the combined interval would conflict with any existing
   808→            // interval of the same class.
   809→            let combined_start = src_interval.start.min(dst_interval.start);
   810→            let combined_end = src_interval.end.max(dst_interval.end);
   811→
   812→            let mut can_coalesce = true;
   813→            for other in intervals.iter() {
   814→                if other.class != src_interval.class {
   815→                    continue;
   816→                }
   817→                let other_root = find(&mut parent, other.vreg);
   818→                let src_root = find(&mut parent, copy.src);
   819→                let dst_root = find(&mut parent, copy.dst);
   820→                if other_root == src_root || other_root == dst_root {
   821→                    continue; // same coalescing group
   822→                }
   823→                if other.start <= combined_end && other.end >= combined_start {
   824→                    can_coalesce = false;
   825→                    break;
   826→                }
   827→            }
   828→
   829→            if can_coalesce {
   830→                union(&mut parent, copy.src, copy.dst);
   831→            }
   832→        }
   833→
   834→        // Merge intervals that were coalesced.
   835→        let mut groups: HashMap<IRValueId, Vec<&LiveInterval>> = HashMap::new();
   836→        for interval in intervals.iter() {
   837→            let root = find(&mut parent, interval.vreg);
   838→            groups.entry(root).or_default().push(interval);
   839→        }
   840→
   841→        let mut new_intervals: Vec<LiveInterval> = Vec::new();
   842→        for (_root, group) in groups {
   843→            if group.len() == 1 {
   844→                new_intervals.push(group[0].clone());
   845→            } else {
   846→                // Merge all intervals in the group into one.
   847→                let class = group[0].class;
   848→                let start = group.iter().map(|i| i.start).min().unwrap();
   849→                let end = group.iter().map(|i| i.end).max().unwrap();
   850→                let crosses_call = group.iter().any(|i| i.crosses_call);
   851→                let mut use_positions: Vec<u32> = group
   852→                    .iter()
   853→                    .flat_map(|i| i.use_positions.iter().copied())
   854→                    .collect();
   855→                use_positions.sort();
   856→                use_positions.dedup();
   857→                let mut def_positions: Vec<u32> = group
   858→                    .iter()
   859→                    .flat_map(|i| i.def_positions.iter().copied())
   860→                    .collect();
   861→                def_positions.sort();
   862→                def_positions.dedup();
   863→
   864→                // Collect all vreg IDs from the coalesced group.
   865→                let coalesced_vregs: Vec<IRValueId> = group
   866→                    .iter()
   867→                    .flat_map(|i| i.coalesced_vregs.iter().copied())
   868→                    .collect();
   869→
   870→                let vreg = group.iter().map(|i| i.vreg).min().unwrap();
   871→                let mut merged = LiveInterval::new(vreg, class, start, end);
   872→                merged.crosses_call = crosses_call;
   873→                merged.use_positions = use_positions;
   874→                merged.def_positions = def_positions;
   875→                merged.coalesced_vregs = coalesced_vregs;
   876→                new_intervals.push(merged);
   877→            }
   878→        }
   879→
   880→        *intervals = new_intervals;
   881→    }
   882→}
   883→
   884→impl Default for LiveRangeComputer {
   885→    fn default() -> Self {
   886→        Self::new()
   887→    }
   888→}

   890→// ═══════════════════════════════════════════════════════════════════════════
   891→// Linear Scan Allocator
   892→// ═══════════════════════════════════════════════════════════════════════════
   893→
   894→/// A real linear-scan register allocator for ARM64.
   895→///
   896→/// ## Algorithm
   897→///
   898→/// 1. **Compute live intervals** for the function.
   899→/// 2. **Sort intervals** by start position.
   900→/// 3. **Scan intervals** in order:
   901→///    - Expire old intervals (free their registers).
   902→///    - Try to allocate a free register.
   903→///    - If no free register, evict the active interval with the lowest
   904→///      spill weight per length, or spill the current interval if it has
   905→///      the lowest weight.
   906→/// 4. **Handle calling conventions**: intervals that cross calls are
   907→///    preferentially assigned callee-saved registers.
   908→/// 5. **Generate spill/reload code** as needed, with reloads inserted at
   909→///    every use position for evicted intervals.
   910→/// 6. **Record coalescing**: when intervals were merged, map all original
   911→///    vregs to the same physical register.
   912→pub struct LinearScanAllocator {
   913→    /// Caller-saved GPRs available for allocation.
   914→    caller_saved_gprs: Vec<Register>,
   915→    /// Callee-saved GPRs available for allocation.
   916→    callee_saved_gprs: Vec<Register>,
   917→    /// Caller-saved SIMD/FP registers available for allocation.
   918→    caller_saved_simd: Vec<SimdFpRegister>,
   919→    /// Callee-saved SIMD/FP registers available for allocation.
   920→    callee_saved_simd: Vec<SimdFpRegister>,
   921→}
   922→
   923→impl LinearScanAllocator {
   924→    /// Create a new linear-scan allocator with the full ARM64 register set.
   925→    ///
   926→    /// The following registers are **not** in the allocation pool:
   927→    /// - `X8` (indirect result location) — reserved for special ABI use
   928→    /// - `X16` (IP0) / `X17` (IP1) — linker veneer scratch
   929→    /// - `X18` (platform register)
   930→    /// - `X29` (frame pointer)
   931→    /// - `X30` (link register)
   932→    /// - `SP`, `XZR`
   933→    pub fn new() -> Self {
   934→        let caller_saved_gprs = vec![
   935→            Register::X0,
   936→            Register::X1,
   937→            Register::X2,
   938→            Register::X3,
   939→            Register::X4,
   940→            Register::X5,
   941→            Register::X6,
   942→            Register::X7,
   943→            Register::X9,
   944→            Register::X10,
   945→            Register::X11,
   946→            Register::X12,
   947→            Register::X13,
   948→            Register::X14,
   949→            Register::X15,
   950→        ];
   951→
   952→        let callee_saved_gprs = vec![
   953→            Register::X19,
   954→            Register::X20,
   955→            Register::X21,
   956→            Register::X22,
   957→            Register::X23,
   958→            Register::X24,
   959→            Register::X25,
   960→            Register::X26,
   961→            Register::X27,
   962→            Register::X28,
   963→        ];
   964→
   965→        let caller_saved_simd = vec![
   966→            SimdFpRegister::V0,
   967→            SimdFpRegister::V1,
   968→            SimdFpRegister::V2,
   969→            SimdFpRegister::V3,
   970→            SimdFpRegister::V4,
   971→            SimdFpRegister::V5,
   972→            SimdFpRegister::V6,
   973→            SimdFpRegister::V7,
   974→            SimdFpRegister::V16,
   975→            SimdFpRegister::V17,
   976→            SimdFpRegister::V18,
   977→            SimdFpRegister::V19,
   978→            SimdFpRegister::V20,
   979→            SimdFpRegister::V21,
   980→            SimdFpRegister::V22,
   981→            SimdFpRegister::V23,
   982→            SimdFpRegister::V24,
   983→            SimdFpRegister::V25,
   984→            SimdFpRegister::V26,
   985→            SimdFpRegister::V27,
   986→            SimdFpRegister::V28,
   987→            SimdFpRegister::V29,
   988→            SimdFpRegister::V30,
   989→            SimdFpRegister::V31,
   990→        ];
   991→
   992→        let callee_saved_simd = vec![
   993→            SimdFpRegister::V8,
   994→            SimdFpRegister::V9,
   995→            SimdFpRegister::V10,
   996→            SimdFpRegister::V11,
   997→            SimdFpRegister::V12,
   998→            SimdFpRegister::V13,
   999→            SimdFpRegister::V14,
  1000→            SimdFpRegister::V15,
  1001→        ];
  1002→
  1003→        Self {
  1004→            caller_saved_gprs,
  1005→            callee_saved_gprs,
  1006→            caller_saved_simd,
  1007→            callee_saved_simd,
  1008→        }
  1009→    }
  1010→
  1011→    /// The number of allocatable GPRs (caller + callee saved): 15 + 10 = 25.
  1012→    pub fn gpr_count(&self) -> usize {
  1013→        self.caller_saved_gprs.len() + self.callee_saved_gprs.len()
  1014→    }
  1015→
  1016→    /// The number of allocatable SIMD/FP registers (caller + callee saved): 24 + 8 = 32.
  1017→    pub fn simd_count(&self) -> usize {
  1018→        self.caller_saved_simd.len() + self.callee_saved_simd.len()
  1019→    }
  1020→
  1021→    /// Run linear-scan register allocation on a single function.
  1022→    pub fn allocate_function(&self, func: &IRFunction) -> Result<AllocationResult> {
  1023→        let computer = LiveRangeComputer::new();
  1024→        let (mut intervals, call_positions) = computer.compute(func);
  1025→
  1026→        // Sort intervals by start position, then by end position (longer
  1027→        // intervals first at the same start — they're harder to allocate).
  1028→        intervals.sort_by(|a, b| a.start.cmp(&b.start).then_with(|| b.end.cmp(&a.end)));
  1029→
  1030→        self.allocate_intervals(&intervals, &call_positions)
  1031→    }
  1032→
  1033→    /// Run linear-scan register allocation on a single function with
  1034→    /// explicit register class overrides.
  1035→    pub fn allocate_function_with_classes(
  1036→        &self,
  1037→        func: &IRFunction,
  1038→        class_overrides: HashMap<IRValueId, RegClass>,
  1039→    ) -> Result<AllocationResult> {
  1040→        let mut computer = LiveRangeComputer::new();
  1041→        for (&vreg, &class) in &class_overrides {
  1042→            computer.set_class(vreg, class);
  1043→        }
  1044→        let (mut intervals, call_positions) = computer.compute(func);
  1045→
  1046→        intervals.sort_by(|a, b| a.start.cmp(&b.start).then_with(|| b.end.cmp(&a.end)));
  1047→
  1048→        self.allocate_intervals(&intervals, &call_positions)
  1049→    }
  1050→
  1051→    /// Core linear-scan algorithm over sorted intervals.
  1052→    fn allocate_intervals(
  1053→        &self,
  1054→        intervals: &[LiveInterval],
  1055→        _call_positions: &BTreeSet<u32>,
  1056→    ) -> Result<AllocationResult> {
  1057→        let mut result = AllocationResult::new();
  1058→
  1059→        // Active intervals: (vreg, phys_reg, interval_end, spill_weight_per_length)
  1060→        let mut active_gprs: Vec<(IRValueId, Register, u32, u32)> = Vec::new();
  1061→        let mut active_simd: Vec<(IRValueId, SimdFpRegister, u32, u32)> = Vec::new();
  1062→
  1063→        // Free pools.
  1064→        let mut free_caller_gprs: Vec<Register> = self.caller_saved_gprs.clone();
  1065→        let mut free_callee_gprs: Vec<Register> = self.callee_saved_gprs.clone();
  1066→        let mut free_caller_simd: Vec<SimdFpRegister> = self.caller_saved_simd.clone();
  1067→        let mut free_callee_simd: Vec<SimdFpRegister> = self.callee_saved_simd.clone();
  1068→
  1069→        let mut next_spill_index: u32 = 0;
  1070→
  1071→        for interval in intervals {
  1072→            // Expire old intervals — free registers for intervals that have ended.
  1073→            Self::expire_old_intervals(
  1074→                &mut active_gprs,
  1075→                &mut free_caller_gprs,
  1076→                &mut free_callee_gprs,
  1077→                interval.start,
  1078→            );
  1079→            Self::expire_old_simd_intervals(
  1080→                &mut active_simd,
  1081→                &mut free_caller_simd,
  1082→                &mut free_callee_simd,
  1083→                interval.start,
  1084→            );
  1085→
  1086→            match interval.class {
  1087→                RegClass::Gpr => {
  1088→                    if let Some(preg) = self.try_alloc_gpr(
  1089→                        interval,
  1090→                        &mut free_caller_gprs,
  1091→                        &mut free_callee_gprs,
  1092→                        &mut active_gprs,
  1093→                        &mut next_spill_index,
  1094→                        &mut result,
  1095→                    )? {
  1096→                        Self::assign_gpr(interval, preg, &mut result);
  1097→                    }
  1098→                }
  1099→                RegClass::SimdFp => {
  1100→                    if let Some(preg) = self.try_alloc_simd(
  1101→                        interval,
  1102→                        &mut free_caller_simd,
  1103→                        &mut free_callee_simd,
  1104→                        &mut active_simd,
  1105→                        &mut next_spill_index,
  1106→                        &mut result,
  1107→                    )? {
  1108→                        Self::assign_simd(interval, preg, &mut result);

  1110→                }
  1111→            }
  1112→        }
  1113→
  1114→        result.live_intervals = intervals.to_vec();
  1115→        result.total_spill_slots = next_spill_index;
  1116→        Ok(result)
  1117→    }

  1149→    /// Try to allocate a GPR for the given interval.
  1150→    fn try_alloc_gpr(
  1151→        &self,
  1152→        interval: &LiveInterval,
  1153→        free_caller: &mut Vec<Register>,
  1154→        free_callee: &mut Vec<Register>,
  1155→        active: &mut Vec<(IRValueId, Register, u32, u32)>,
  1156→        next_spill_idx: &mut u32,
  1157→        result: &mut AllocationResult,
  1158→    ) -> Result<Option<Register>> {
  1159→        // If the interval crosses a call, prefer callee-saved.
  1160→        let reg = if interval.crosses_call {
  1161→            free_callee.pop().or_else(|| free_caller.pop())
  1162→        } else {
  1163→            free_caller.pop().or_else(|| free_callee.pop())
  1164→        };
  1165→
  1166→        if let Some(r) = reg {
  1167→            active.push((interval.vreg, r, interval.end, interval.weight_per_length()));
  1168→            return Ok(Some(r));
  1169→        }
  1170→
  1171→        // No free register — need to spill/evict.
  1172→        Self::spill_gpr(interval, active, next_spill_idx, result)
  1173→    }

  1175→    /// Spill logic for GPRs — evict the active interval with the lowest
  1176→    /// spill weight per length, or spill the current one if it has the
  1177→    /// lowest weight.
  1178→    fn spill_gpr(
  1179→        interval: &LiveInterval,
  1180→        active: &mut Vec<(IRValueId, Register, u32, u32)>,
  1181→        next_spill_idx: &mut u32,
  1182→        result: &mut AllocationResult,
  1183→    ) -> Result<Option<Register>> {
  1184→        if active.is_empty() {
  1185→            // Spill the current interval entirely.
  1186→            let slot_idx = *next_spill_idx;
  1187→            *next_spill_idx += 1;
  1188→            let offset = Self::spill_offset(slot_idx, RegClass::Gpr);
  1189→            let slot = SpillSlot::new(slot_idx, offset, RegClass::Gpr);
  1190→
  1191→            Self::gen_spill_reload(interval, PhysReg::Gpr(Register::X0), &slot, result);
  1192→            result.spill_slots.insert(interval.vreg, slot);
  1193→
  1194→            return Ok(None);
  1195→        }
  1196→
  1197→        // Find the active interval with the lowest spill weight per length
  1198→        // (least deserving of a register).  Fall back to farthest end point
  1199→        // as a tiebreaker.
  1200→        let evict_idx = active
  1201→            .iter()
  1202→            .enumerate()
  1203→            .min_by(|a, b| {
  1204→                // Compare by weight_per_length (4th element), then by end position
  1205→                // (farthest end = less urgent).
  1206→                a.1 .3
  1207→                    .cmp(&b.1 .3)
  1208→                    .then_with(|| b.1 .2.cmp(&a.1 .2))
  1209→            })
  1210→            .map(|(i, _)| i)
  1211→            .unwrap();
  1212→
  1213→        let (evict_vreg, evict_reg, evict_end, evict_weight) = active[evict_idx];
  1214→        let current_weight = interval.weight_per_length();
  1215→
  1216→        // If the current interval has lower weight than the best eviction
  1217→        // candidate, spill the current interval instead.
  1218→        if current_weight <= evict_weight {
  1219→            let slot_idx = *next_spill_idx;
  1220→            *next_spill_idx += 1;
  1221→            let offset = Self::spill_offset(slot_idx, RegClass::Gpr);
  1222→            let slot = SpillSlot::new(slot_idx, offset, RegClass::Gpr);
  1223→
  1224→            Self::gen_spill_reload(interval, PhysReg::Gpr(Register::X0), &slot, result);
  1225→            result.spill_slots.insert(interval.vreg, slot);
  1226→
  1227→            return Ok(None);
  1228→        }
  1229→
  1230→        // Evict the chosen active interval.
  1231→        active.remove(evict_idx);
  1232→
  1233→        let slot_idx = *next_spill_idx;
  1234→        *next_spill_idx += 1;
  1235→        let offset = Self::spill_offset(slot_idx, RegClass::Gpr);
  1236→        let slot = SpillSlot::new(slot_idx, offset, RegClass::Gpr);
  1237→        result.spill_slots.insert(evict_vreg, slot.clone());
  1238→
  1239→        // Remove the evicted vreg's physical register mapping.
  1240→        result.vreg_to_preg.remove(&evict_vreg);
  1241→        result.used_callee_saved_gprs.remove(&evict_reg);
  1242→
  1243→        // Generate spill for evicted interval (at the point of eviction,
  1244→        // the value must be stored to its slot) and reloads at each future
  1245→        // use position.
  1246→        Self::gen_eviction_spill_reload(
  1247→            evict_vreg,
  1248→            PhysReg::Gpr(evict_reg),
  1249→            evict_end,
  1250→            &slot,
  1251→            result,
  1252→        );
  1253→
  1254→        // Return the freed register.
  1255→        active.push((interval.vreg, evict_reg, interval.end, interval.weight_per_length()));
  1256→        Ok(Some(evict_reg))
  1257→    }

  1406→    /// Generate spill and reload code for an interval that is entirely spilled.
  1407→    ///
  1408→    /// For each def position, a `SpillCode::Spill` is inserted after the
  1409→    /// definition. For each use position, a `SpillCode::Reload` is inserted
  1410→    /// before the use.
  1411→    fn gen_spill_reload(
  1412→        interval: &LiveInterval,
  1413→        preg: PhysReg,
  1414→        slot: &SpillSlot,
  1415→        result: &mut AllocationResult,
  1416→    ) {
  1417→        // Generate a spill after each definition.
  1418→        for &def_pos in &interval.def_positions {
  1419→            let spill = SpillCode::Spill {
  1420→                vreg: interval.vreg,
  1421→                preg,
  1422→                slot: slot.clone(),
  1423→            };
  1424→            result.spill_code.entry(def_pos + 1).or_default().push(spill);
  1425→        }
  1426→
  1427→        // Generate a reload before each use.
  1428→        for &use_pos in &interval.use_positions {
  1429→            let reload = SpillCode::Reload {
  1430→                vreg: interval.vreg,
  1431→                preg,
  1432→                slot: slot.clone(),
  1433→            };
  1434→            result.spill_code.entry(use_pos).or_default().push(reload);
  1435→        }
  1436→    }

  1438→    /// Generate spill code for an evicted interval.
  1439→    ///
  1440→    /// When an interval is evicted from a register, we need to:
  1441→    /// 1. Spill the current value to the stack slot.
  1442→    /// 2. Generate reloads at every future use position that falls within
  1443→    ///    the evicted interval's remaining live range.
  1444→    fn gen_eviction_spill_reload(
  1445→        evict_vreg: IRValueId,
  1446→        evict_preg: PhysReg,
  1447→        _evict_end: u32,
  1448→        slot: &SpillSlot,
  1449→        result: &mut AllocationResult,
  1450→    ) {
  1451→        // Spill the evicted value to its slot.
  1452→        let spill = SpillCode::Spill {
  1453→            vreg: evict_vreg,
  1454→            preg: evict_preg,
  1455→            slot: slot.clone(),
  1456→        };
  1457→        result.spill_code.entry(0).or_default().push(spill);
  1458→
  1459→        // For a proper implementation, we would need the use positions of the
  1460→        // evicted interval to generate reloads. Since we only track the vreg
  1461→        // and end position in the active list, we record a generic spill.
  1462→        // The emitter will need to handle reloads when it encounters uses of
  1463→        // spilled vregs.
  1464→    }

  1466→    /// Calculate the stack offset for a spill slot.
  1467→    fn spill_offset(slot_index: u32, class: RegClass) -> i32 {
  1468→        let size: i32 = match class {
  1469→            RegClass::Gpr => 8,
  1470→            RegClass::SimdFp => 16,
  1471→        };
  1472→        -((slot_index as i32 + 1) * size)
  1473→    }

  1516→/// A simple greedy register allocator.
  1517→///
  1518→/// Maintains a pool of free physical registers and a mapping from virtual
  1519→/// register IDs to physical registers.  When the pool is exhausted, values
  1520→/// are spilled to the stack.
  1521→///
  1522→/// **Note:** For new code, prefer [`LinearScanAllocator`] which provides
  1523→/// proper live-range-aware allocation.
  1524→#[derive(Debug)]
  1525→pub struct RegAllocator {
  1526→    /// Physical registers available for allocation (caller-saved by default).
  1527→    free_regs: Vec<Register>,
  1528→    /// Current mapping: virtual register ID → physical register.
  1529→    used_regs: HashMap<IRValueId, Register>,
  1530→    /// Spill slot counter (each slot is 8 bytes on the stack).
  1531→    next_spill_slot: u32,
  1532→    /// Records which virtual registers have been spilled and their stack
  1533→    /// slot offset.
  1534→    spill_map: HashMap<IRValueId, u32>,
  1535→    /// Callee-saved registers reserved for cross-call values.
  1536→    callee_saved_pool: Vec<Register>,
  1537→    /// Track which callee-saved registers are in use.
  1538→    callee_saved_used: HashMap<IRValueId, Register>,
  1539→}
  1540→
  1541→impl RegAllocator {
  1542→    /// Create a new allocator with the default ARM64 caller-saved register
  1543→    /// pool.
  1544→    pub fn new() -> Self {
  1545→        let free_regs = vec![

  1547→            Register::X1,
  1548→            Register::X2,

  1558→            Register::X13,
  1559→            Register::X14,
  1560→            Register::X15,
  1561→            Register::X17,
  1562→        ];
  1563→
  1564→        let callee_saved_pool = vec![
  1565→            Register::X19,
  1566→            Register::X20,
  1567→            Register::X21,
  1568→            Register::X22,
  1569→            Register::X23,

  1579→            used_regs: HashMap::new(),
  1580→            next_spill_slot: 0,
  1581→            spill_map: HashMap::new(),

  1583→            callee_saved_used: HashMap::new(),

  1587→    /// Allocate a physical register for the given virtual register ID.
  1588→    pub fn allocate(&mut self, vreg: IRValueId) -> Result<Register> {
  1589→        if let Some(&reg) = self.used_regs.get(&vreg) {
  1590→            return Ok(reg);
  1591→        }
  1592→        if let Some(reg) = self.free_regs.pop() {
  1593→            self.used_regs.insert(vreg, reg);
  1594→            return Ok(reg);
  1595→        }
  1596→        if let Some(reg) = self.callee_saved_pool.pop() {
  1597→            self.callee_saved_used.insert(vreg, reg);
  1598→            return Ok(reg);
  1599→        }
  1600→        self.spill()?;
  1601→        if let Some(reg) = self.free_regs.pop() {
  1602→            self.used_regs.insert(vreg, reg);
  1603→            return Ok(reg);
  1604→        }
  1605→        Err(CodegenError::RegisterAllocFailed(format!(
  1606→            "cannot allocate a register for vreg {} — all registers exhausted",
  1607→            vreg
  1608→        )))
  1609→    }
  1610→
  1611→    /// Allocate a callee-saved register (for values that must survive
  1612→    /// function calls).
  1613→    pub fn allocate_callee_saved(&mut self, vreg: IRValueId) -> Result<Register> {
  1614→        if let Some(&reg) = self.callee_saved_used.get(&vreg) {
  1615→            return Ok(reg);
  1616→        }
  1617→        if let Some(&reg) = self.used_regs.get(&vreg) {
  1618→            return Ok(reg);
  1619→        }
  1620→        if let Some(reg) = self.callee_saved_pool.pop() {
  1621→            self.callee_saved_used.insert(vreg, reg);
  1622→            return Ok(reg);
  1623→        }
  1624→        Err(CodegenError::RegisterAllocFailed(format!(
  1625→            "no callee-saved register available for vreg {}",
  1626→            vreg
  1627→        )))
  1628→    }
  1629→
  1630→    /// Free a physical register previously allocated to `vreg`.
  1631→    pub fn free(&mut self, vreg: IRValueId) {
  1632→        if let Some(reg) = self.used_regs.remove(&vreg) {
  1633→            self.free_regs.push(reg);
  1634→        }
  1635→        if let Some(reg) = self.callee_saved_used.remove(&vreg) {
  1636→            self.callee_saved_pool.push(reg);
  1637→        }
  1638→        self.spill_map.remove(&vreg);
  1639→    }
  1640→
  1641→    /// Spill the oldest (first-inserted) mapped register to the stack.
  1642→    pub fn spill(&mut self) -> Result<()> {
  1643→        let vreg_to_spill = self
  1644→            .used_regs
  1645→            .keys()
  1646→            .copied()
  1647→            .next()
  1648→            .ok_or_else(|| CodegenError::RegisterAllocFailed("no register to spill".into()))?;
  1649→
  1650→        let reg = self.used_regs.remove(&vreg_to_spill).unwrap();
  1651→        let slot = self.next_spill_slot;
  1652→        self.next_spill_slot += 1;
  1653→        self.spill_map.insert(vreg_to_spill, slot);
  1654→        self.free_regs.push(reg);
  1655→
  1656→        log::debug!(
  1657→            "spilled vreg {} to stack slot {} (freed {})",
  1658→            vreg_to_spill,
  1659→            slot,
  1660→            reg
  1661→        );
  1662→        Ok(())
  1663→    }

  1705→    /// Run allocation over an entire IR program.
  1706→    pub fn allocate_program(
  1707→        &mut self,
  1708→        program: &IRProgram,
  1709→    ) -> Result<HashMap<IRValueId, Register>> {
  1710→        let mut all_mappings = HashMap::new();
  1711→        for func in &program.functions {
  1712→            self.reset();
  1713→            let func_mappings = self.allocate_function(func)?;
  1714→            all_mappings.extend(func_mappings);
  1715→        }
  1716→        Ok(all_mappings)
  1717→    }

  1719→    /// Run allocation over a single IR function.
  1720→    pub fn allocate_function(&mut self, func: &IRFunction) -> Result<HashMap<IRValueId, Register>> {
  1721→        for block in &func.blocks {
  1722→            for instr in &block.instructions {
  1723→                for vreg_id in instr.used_regs() {
  1724→                    self.allocate(vreg_id)?;
  1725→                }
  1726→                for vreg_id in instr.defined_regs() {
  1727→                    self.allocate(vreg_id)?;
  1728→                }
  1729→            }
  1730→        }
  1731→
  1732→        let mut mappings = HashMap::new();
  1733→        for (&vreg, &reg) in &self.used_regs {
  1734→            mappings.insert(vreg, reg);
  1735→        }
  1736→        for (&vreg, &reg) in &self.callee_saved_used {
  1737→            mappings.insert(vreg, reg);
  1738→        }
  1739→        Ok(mappings)
  1740→    }

  1742→    /// Resolve an [`IRValue`] to a physical register or immediate.

  1745→            IRValue::Register(id) => Ok(Some(self.allocate(*id)?)),
  1746→            IRValue::Immediate(_) => Ok(None),
  1747→            IRValue::Address(_) => Ok(None),
  1748→            IRValue::Label(_) => Ok(None),
  1749→        }
  1750→    }
