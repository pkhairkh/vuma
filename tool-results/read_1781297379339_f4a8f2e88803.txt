  1941ŌåÆ  1941ŌåÆ/// Build a minimal ELF64 binary for RISC-V 64-bit from raw code bytes.
  1942ŌåÆ  1942ŌåÆ///
  1943ŌåÆ  1943ŌåÆ/// Produces a static executable with a single LOAD segment containing the
  1944ŌåÆ  1944ŌåÆ/// `.text` section.  Uses EM_RISCV (243) as the machine type.
  1945ŌåÆ  1945ŌåÆfn build_minimal_riscv64_elf(code: &[u8], base_addr: u64) -> Vec<u8> {
  1946ŌåÆ  1946ŌåÆ    // Layout: ELF header (64) | 1 program header (56) | code
  1947ŌåÆ  1947ŌåÆ    let elf_header_size: u64 = 64;
  1948ŌåÆ  1948ŌåÆ    let phdr_size: u64 = 56;
  1949ŌåÆ  1949ŌåÆ    let text_offset = elf_header_size + phdr_size;
  1950ŌåÆ  1950ŌåÆ    let text_size = code.len() as u64;
  1951ŌåÆ  1951ŌåÆ    let entry_point = base_addr + text_offset;
  1952ŌåÆ  1952ŌåÆ
  1953ŌåÆ  1953ŌåÆ    let mut elf = Vec::with_capacity(text_offset as usize + code.len());
  1954ŌåÆ  1954ŌåÆ
  1955ŌåÆ  1955ŌåÆ    // --- e_ident ---
  1956ŌåÆ  1956ŌåÆ    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1957ŌåÆ  1957ŌåÆ    elf.push(2); // ELFCLASS64
  1958ŌåÆ  1958ŌåÆ    elf.push(1); // ELFDATA2LSB
  1959ŌåÆ  1959ŌåÆ    elf.push(1); // EV_CURRENT
  1960ŌåÆ  1960ŌåÆ    elf.push(3); // ELFOSABI_LINUX
  1961ŌåÆ  1961ŌåÆ    elf.push(0); // padding
  1962ŌåÆ  1962ŌåÆ    elf.extend_from_slice(&[0u8; 7]); // padding
  1963ŌåÆ  1963ŌåÆ
  1964ŌåÆ  1964ŌåÆ    // --- ELF header fields ---
  1965ŌåÆ  1965ŌåÆ    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1966ŌåÆ  1966ŌåÆ    elf.extend_from_slice(&243u16.to_le_bytes()); // e_machine = EM_RISCV
  1967ŌåÆ  1967ŌåÆ    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1968ŌåÆ  1968ŌåÆ    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1969ŌåÆ  1969ŌåÆ    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1970ŌåÆ  1970ŌåÆ    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1971ŌåÆ  1971ŌåÆ                                                // e_flags: RISC-V-specific flags. Float ABI = double (0x5), RVC = 0.
  1972ŌåÆ  1972ŌåÆ                                                // EF_RISCV_FLOAT_ABI_DOUBLE = 0x5 << 5 = 0xA0
  1973ŌåÆ  1973ŌåÆ                                                // EF_RISCV_RVC = 0x0001 (we don't set it for now)
  1974ŌåÆ  1974ŌåÆ    elf.extend_from_slice(&0xA0u32.to_le_bytes()); // e_flags (LP64D ABI)
  1975ŌåÆ  1975ŌåÆ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1976ŌåÆ  1976ŌåÆ    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1977ŌåÆ  1977ŌåÆ    elf.extend_from_slice(&1u16.to_le_bytes()); // e_phnum
  1978ŌåÆ  1978ŌåÆ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1979ŌåÆ  1979ŌåÆ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1980ŌåÆ  1980ŌåÆ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1981ŌåÆ  1981ŌåÆ
  1982ŌåÆ  1982ŌåÆ    // --- Program Header (single LOAD segment: PF_R | PF_X) ---
  1983ŌåÆ  1983ŌåÆ    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1984ŌåÆ  1984ŌåÆ    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1985ŌåÆ  1985ŌåÆ    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1986ŌåÆ  1986ŌåÆ    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1987ŌåÆ  1987ŌåÆ    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1988ŌåÆ  1988ŌåÆ    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1989ŌåÆ  1989ŌåÆ    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1990ŌåÆ  1990ŌåÆ    elf.extend_from_slice(&16u64.to_le_bytes()); // p_align
  1991ŌåÆ  1991ŌåÆ
  1992ŌåÆ  1992ŌåÆ    // --- Code section ---
  1993ŌåÆ  1993ŌåÆ    elf.extend_from_slice(code);
  1994ŌåÆ  1994ŌåÆ
  1995ŌåÆ  1995ŌåÆ    elf
  1996ŌåÆ  1996ŌåÆ}
