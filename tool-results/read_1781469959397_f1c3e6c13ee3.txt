  2200→            code.extend_from_slice(&Instruction::Lui { rt: dst, imm: upper }.encode());
  2201→            if lower != 0 {
  2202→                code.extend_from_slice(&Instruction::Ori { rt: dst, rs: dst, imm: lower }.encode());
  2203→            }
  2204→            // If the 32-bit value has bit 31 set, LUI sign-extended it.
  2205→            // Zero-extend by shifting left 32 then right 32.
  2206→            if (val as u32) >= 0x80000000 {
  2207→                code.extend_from_slice(&Instruction::Dsll { rd: dst, rt: dst, sa: 32 }.encode());
  2208→                code.extend_from_slice(&Instruction::Dsrl { rd: dst, rt: dst, sa: 32 }.encode());
  2209→            }
  2210→        } else {
  2211→            // Full 64-bit: lui + ori + dsll + ori + dsll + ori
  2212→            let bits63_48 = ((val as u64) >> 48) as u32 & 0xFFFF;
  2213→            let bits47_32 = ((val as u64) >> 32) as u32 & 0xFFFF;
  2214→            let bits31_16 = ((val as u64) >> 16) as u32 & 0xFFFF;
  2215→            let bits15_0 = (val as u64) as u32 & 0xFFFF;
  2216→            code.extend_from_slice(&Instruction::Lui { rt: dst, imm: bits63_48 }.encode());
  2217→            code.extend_from_slice(&Instruction::Ori { rt: dst, rs: dst, imm: bits47_32 }.encode());
  2218→            code.extend_from_slice(&Instruction::Dsll { rd: dst, rt: dst, sa: 16 }.encode());
  2219→            code.extend_from_slice(&Instruction::Ori { rt: dst, rs: dst, imm: bits31_16 }.encode());
  2220→            code.extend_from_slice(&Instruction::Dsll { rd: dst, rt: dst, sa: 16 }.encode());
  2221→            code.extend_from_slice(&Instruction::Ori { rt: dst, rs: dst, imm: bits15_0 }.encode());
  2222→        }
  2223→        code
  2224→    }
  2225→
  2226→    // Load from stack slot at [$fp - offset] into dst register
  2227→    fn ss_ld(dst: Gpr, offset: i32) -> Vec<u8> {
  2228→        let mut code = Vec::new();
  2229→        if (-32768..=32767).contains(&(-offset)) {
  2230→            code.extend_from_slice(&Instruction::Ld { rt: dst, base: Gpr::Fp, offset: -offset }.encode());
  2231→            code.extend_from_slice(&encode_nop());
  2232→        } else {
  2233→            // Large offset: compute address into $t3
  2234→            code.extend(ss_load_imm(Gpr::T3, offset as i64));
  2235→            code.extend_from_slice(&Instruction::Dsubu { rd: Gpr::T3, rs: Gpr::Fp, rt: Gpr::T3 }.encode());
  2236→            code.extend_from_slice(&encode_nop());
  2237→            code.extend_from_slice(&Instruction::Ld { rt: dst, base: Gpr::T3, offset: 0 }.encode());
  2238→            code.extend_from_slice(&encode_nop());
  2239→        }
  2240→        code
  2241→    }

  2259→    // Load a typed value from stack slot
  2260→    fn ss_ld_typed(dst: Gpr, offset: i32, ty: &IRType) -> Vec<u8> {
  2261→        let mut code = Vec::new();
  2262→        let neg_off = -offset;
  2263→        if (-32768..=32767).contains(&neg_off) {
  2264→            match ty {

  2277→                IRType::I32 | IRType::U32 => { code.extend_from_slice(&Instruction::Lwu { rt: dst, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2278→                _ => { code.extend_from_slice(&Instruction::Ld { rt: dst, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2279→            }
  2280→        }
  2281→        code
  2282→    }

  2284→    // Store a typed value to stack slot
  2285→    fn ss_sd_typed(src: Gpr, offset: i32, ty: &IRType) -> Vec<u8> {
  2286→        let mut code = Vec::new();
  2287→        let neg_off = -offset;
  2288→        if (-32768..=32767).contains(&neg_off) {
  2289→            match ty {
  2290→                IRType::I8 | IRType::U8 => { code.extend_from_slice(&Instruction::Sb { rt: src, base: Gpr::Fp, offset: neg_off }.encode()); code.extend_from_slice(&encode_nop()); }
  2291→                IRType::I32 | IRType::U32 => { code.extend_from_slice(&Instruction::Sw { rt: src, base: Gpr::Fp, offset: neg_off }.encode()); code.extend_from_slice(&encode_nop()); }
  2292→                _ => { code.extend_from_slice(&Instruction::Sd { rt: src, base: Gpr::Fp, offset: neg_off }.encode()); code.extend_from_slice(&encode_nop()); }
  2293→            }
  2294→        } else {
  2295→            code.extend(ss_load_imm(Gpr::T3, offset as i64));
  2296→            code.extend_from_slice(&Instruction::Dsubu { rd: Gpr::T3, rs: Gpr::Fp, rt: Gpr::T3 }.encode());
  2297→            code.extend_from_slice(&encode_nop());
  2298→            match ty {
  2299→                IRType::I8 | IRType::U8 => { code.extend_from_slice(&Instruction::Sb { rt: src, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2300→                IRType::I32 | IRType::U32 => { code.extend_from_slice(&Instruction::Sw { rt: src, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2301→                _ => { code.extend_from_slice(&Instruction::Sd { rt: src, base: Gpr::T3, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }

  2303→        }

  2307→    // Load an IRValue into a scratch register
  2308→    fn ss_load_value(val: &IRValue, slots: &HashMap<u32, i32>, scratch: Gpr) -> Vec<u8> {
  2309→        match val {
  2310→            IRValue::Register(id) => {
  2311→                let offset = slots.get(id).copied().unwrap_or(0);
  2312→                ss_ld(scratch, offset)
  2313→            }
  2314→            IRValue::Immediate(v) => ss_load_imm(scratch, *v),
  2315→            IRValue::Address(a) => ss_load_imm(scratch, *a as i64),
  2316→            IRValue::Label(_) => ss_load_imm(scratch, 0),
  2317→        }
  2318→    }

  2320→    // ── Phase 3: Emit prologue ──
  2321→    let mut code: Vec<u8> = Vec::new();
  2322→
  2323→    // DADDIU $sp, $sp, -frame_size
  2324→    code.extend_from_slice(&Instruction::Daddiu { rt: Gpr::Sp, rs: Gpr::Sp, imm: -(frame_size as i32) }.encode());
  2325→    code.extend_from_slice(&encode_nop()); // delay slot (safe: SP update is committed)
  2326→    // SD $ra, frame_size-8($sp)
  2327→    code.extend_from_slice(&Instruction::Sd { rt: Gpr::Ra, base: Gpr::Sp, offset: (frame_size - 8) as i32 }.encode());
  2328→    code.extend_from_slice(&encode_nop());
  2329→    // SD $fp, frame_size-16($sp)
  2330→    code.extend_from_slice(&Instruction::Sd { rt: Gpr::Fp, base: Gpr::Sp, offset: (frame_size - 16) as i32 }.encode());
  2331→    code.extend_from_slice(&encode_nop());
  2332→    // DADDIU $fp, $sp, frame_size  (frame pointer = top of frame)
  2333→    code.extend_from_slice(&Instruction::Daddiu { rt: Gpr::Fp, rs: Gpr::Sp, imm: frame_size as i32 }.encode());
  2334→    code.extend_from_slice(&encode_nop());
  2335→
  2336→    // Store function parameters from $a0-$a3 to their stack slots
  2337→    let arg_regs = [Gpr::A0, Gpr::A1, Gpr::A2, Gpr::A3];
  2338→    for (i, param) in func.params.iter().enumerate() {
  2339→        if let Some(id) = param.as_register() {
  2340→            if i < 4 {
  2341→                let offset = vreg_stack_slots.get(&id).copied().unwrap_or(0);
  2342→                code.extend(ss_sd(arg_regs[i], offset));
  2343→            }
  2344→        }
  2345→    }
  2346→
  2347→    // ── Phase 4: Emit each basic block ──
  2348→    let mut label_offsets: HashMap<String, usize> = HashMap::new();

  2350→    let mut relocations: Vec<RelocationEntry> = Vec::new();
  2351→
  2352→    for block in &func.blocks {
  2353→        label_offsets.insert(block.label.clone(), code.len());
  2354→
  2355→        for instr in &block.instructions {
  2356→            match instr {
  2357→                // ── BinOp / Add / Sub / Mul / Div ──
  2358→                IRInstr::BinOp { op, dst, lhs, rhs, ty } => {
  2359→                    let actual_op = *op;
  2360→                    let dst_id = dst.as_register().unwrap_or(0);
  2361→                    let dst_off = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
  2362→                    code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::T0));
  2363→                    code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::T1));
  2364→

  2366→

  2375→                            code.extend_from_slice(&Instruction::Dsll { rd: Gpr::T0, rt: Gpr::T0, sa: 32 }.encode());
  2376→                            code.extend_from_slice(&Instruction::Dsrl { rd: Gpr::T0, rt: Gpr::T0, sa: 32 }.encode());
  2377→                        }
  2378→                        BinOpKind::Mul | BinOpKind::SDiv | BinOpKind::UDiv |
  2379→                        BinOpKind::SRem | BinOpKind::URem => {

  2385→                                    BinOpKind::SRem => { code.extend_from_slice(&Instruction::Div { rs: Gpr::T0, rt: Gpr::T1 }.encode()); code.extend_from_slice(&encode_nop()); code.extend_from_slice(&Instruction::Mfhi { rd: Gpr::T0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2386→                                    BinOpKind::URem => { code.extend_from_slice(&Instruction::Divu { rs: Gpr::T0, rt: Gpr::T1 }.encode()); code.extend_from_slice(&encode_nop()); code.extend_from_slice(&Instruction::Mfhi { rd: Gpr::T0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2387→                                    _ => {}
  2388→                                }
  2389→                            } else {

  2394→                                    BinOpKind::SRem => { code.extend_from_slice(&Instruction::Ddiv { rs: Gpr::T0, rt: Gpr::T1 }.encode()); code.extend_from_slice(&encode_nop()); code.extend_from_slice(&Instruction::Mfhi { rd: Gpr::T0 }.encode()); code.extend_from_slice(&encode_nop()); }

  2396→                                    _ => {}
  2397→                                }
  2398→                            }
  2399→                        }
  2400→                        BinOpKind::And => { code.extend_from_slice(&Instruction::And { rd: Gpr::T0, rs: Gpr::T0, rt: Gpr::T1 }.encode()); }
  2401→                        BinOpKind::Or => { code.extend_from_slice(&Instruction::Or { rd: Gpr::T0, rs: Gpr::T0, rt: Gpr::T1 }.encode()); }
