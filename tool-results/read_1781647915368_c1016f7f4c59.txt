     1→//! # SCG → IR Conversion
     2→//!
     3→//! Translates a Semantic Computation Graph (SCG) — produced by the `vuma-scg`
     4→//! crate — into the intermediate representation defined in [`crate::ir`].
     5→//!
     6→//! ## Architecture
     7→//!
     8→//! [`IRBuilder`] is the main entry point. It holds translation state (virtual-
     9→//! register counter, label counter, loop stack for break/continue, variable
    10→//! tracking for phi insertion, etc.) and lowers SCG nodes into IR in
    11→//! topological order.
    12→//!
    13→//! ## SCG Node → IR Mapping
    14→//!
    15→//! | SCG Node              | IR Mapping                                           |
    16→//! |-----------------------|------------------------------------------------------|
    17→//! | `ControlNode::If`     | `CondBranch` + then/else/merge blocks + phi nodes    |
    18→//! | `ControlNode::Loop`   | Loop header with phi nodes + back-edge + exit block  |
    19→//! | `ControlNode::Break`  | `Branch` to loop exit                                |
    20→//! | `ControlNode::Continue` | `Branch` to loop header                           |
    21→//! | `ControlNode::Switch` | Cascading `Cmp`+`CondBranch` for each arm + merge   |
    22→//! | `AllocationNode::Stack` | `Alloc` + stack slot registration                  |
    23→//! | `AllocationNode::Heap`  | `Call` to `__vuma_alloc`                           |
    24→//! | `AccessNode::Load`    | Optional `Offset` + `Load`                           |
    25→//! | `AccessNode::Store`   | Optional `Offset` + `Store`                          |
    26→//! | `CastNode`            | `Cast` (zext / sext / trunc / bitcast)               |
    27→//! | `ComputationNode`     | `Add`/`Sub`/`Mul`/`Div`/`Cmp`/`BinOp`/`UnaryOp`     |
    28→//! | `UnaryComputationNode`| `UnaryOp` (neg / not / clz / ctz / popcnt)           |
    29→//! | `CallNode`            | `Call`                                               |
    30→//! | `StructAccessNode`    | Offset-based `Load`/`Store` at `base + field_offset` |
    31→//! | `EnumAccessNode`      | Tag `Load`/`Store` at offset 0; payload at offset N  |
    32→//! | `Return`              | `Ret` + `IRTerminator::Return`                       |
    33→//!
    34→//! ## Struct and Enum Lowering
    35→//!
    36→//! **Structs** are lowered to flat memory layouts. Field N is stored at
    37→//! `base_ptr + offset_N` where offsets are computed during layout resolution.
    38→//! `StructAccessNode::Load` lowers to `Load { addr: ptr, offset: field_offset }`
    39→//! and `StructAccessNode::Store` lowers to `Store { addr: ptr, offset: field_offset }`.
    40→//!
    41→//! **Enums** are lowered to tagged unions: a discriminant (tag) at offset 0
    42→//! followed by a payload at offset `tag_size` (aligned). `EnumAccessNode::LoadTag`
    43→//! reads the tag (offset 0), and `EnumAccessNode::LoadPayload` reads the
    44→//! payload at its computed offset.
    45→//!
    46→//! **Match** expressions are lowered to if/else chains via `ControlNode::Switch`:
    47→//! for each arm, a `Cmp { kind: Eq }` compares the discriminant against the
    48→//! expected tag value, and a `CondBranch` dispatches to the arm body.
    49→//!
    50→//! ## Control Flow
    51→//!
    52→//! - **if/else**: Condition evaluated, `CondBranch` to then/else blocks,
    53→//!   merge block with phi nodes for variables modified in either branch.
    54→//! - **loop**: Loop header block with phi nodes for loop-carried values,
    55→//!   back-edge from loop body end to header, exit block.
    56→//! - **break**: Jump to the loop's exit label (via loop stack).
    57→//! - **continue**: Jump to the loop's header label (via loop stack).
    58→//!
    59→//! ## Topological Ordering
    60→//!
    61→//! Within a function body, SCG statements are lowered in their declared
    62→//! order.  For graph-based SCGs (from the `vuma-scg` crate), the
    63→//! [`IRBuilder::build`] method walks the SCG in topological order
    64→//! using petgraph's `toposort`, ensuring that data-flow dependencies are
    65→//! respected.

   169→/// An SCG statement — the body of a function.
   170→#[derive(Debug, Clone)]
   171→pub enum ScgStatement {
   172→    /// Control flow: if / loop / match.
   173→    Control(ControlNode),
   174→    /// Memory allocation.
   175→    Allocation(AllocationNode),
   176→    /// Memory access (read / write).
   177→    Access(AccessNode),
   178→    /// Type cast / reinterpret.
   179→    Cast(CastNode),
   180→    /// Binary arithmetic / logic.
   181→    Computation(ComputationNode),

   183→    UnaryComputation(UnaryComputationNode),

   199→/// Control-flow node.
   200→#[derive(Debug, Clone)]
   201→pub enum ControlNode {
   202→    /// `if cond { then } else { else_ }`
   203→    If {
   204→        /// The condition expression.
   205→        cond: ScgExpr,
   206→        /// Statements in the then-branch.
   207→        then_body: Vec<ScgStatement>,
   208→        /// Optional else-branch statements.
   209→        else_body: Option<Vec<ScgStatement>>,
   210→    },
   211→    /// `loop { body }`
   212→    Loop {
   213→        /// Loop body statements.
   214→        body: Vec<ScgStatement>,
   215→    },
   216→    /// `break` (from inside a loop).
   217→    Break,
   218→    /// `continue` (from inside a loop).
   219→    Continue,
   220→    /// `switch discriminant { case value => body, .. default => body }`
   221→    ///
   222→    /// Lowers to a sequence of compare-and-branch instructions for small
   223→    /// switch ranges, or a jump table for dense contiguous ranges.
   224→    Switch {
   225→        /// The discriminant expression being switched on.
   226→        discriminant: ScgExpr,
   227→        /// The switch arms: each arm has a value and a body.
   228→        arms: Vec<SwitchArm>,
   229→        /// The default arm (always present — like a match expression).
   230→        default_body: Vec<ScgStatement>,
   231→    },
   232→}

   497→/// Information about an enclosing loop, pushed onto the IRBuilder's loop stack.

   505→    /// Used to create phi nodes at the loop exit that merge values from

   510→// ---------------------------------------------------------------------------
   511→// Variable definition tracking — for phi node insertion
   512→// ---------------------------------------------------------------------------
   513→
   514→/// Tracks which names were defined (written to) in a particular scope
   515→/// (e.g., the then-branch vs the else-branch of an if/else).  Used to
   516→/// determine which phi nodes are needed at a merge point.
   517→#[derive(Debug, Clone)]
   518→struct VarDefs {
   519→    /// Map from variable name to the vreg ID that was assigned in this scope.
   520→    defs: HashMap<String, u32>,
   521→}
   522→
   523→impl VarDefs {
   524→    fn new() -> Self {
   525→        Self {
   526→            defs: HashMap::new(),
   527→        }
   528→    }
   529→
   530→    /// Record that `name` was assigned to `vreg` in this scope.
   531→    fn define(&mut self, name: &str, vreg: u32) {
   532→        self.defs.insert(name.to_string(), vreg);
   533→    }
   534→
   535→    /// Check if `name` was defined in this scope.
   536→    fn is_defined(&self, name: &str) -> bool {
   537→        self.defs.contains_key(name)
   538→    }
   539→
   540→    /// Get the vreg for `name` if it was defined in this scope.
   541→    fn get(&self, name: &str) -> Option<u32> {
   542→        self.defs.get(name).copied()
   543→    }
   544→
   545→    /// Return the set of variable names defined in this scope.
   546→    fn defined_names(&self) -> HashSet<String> {
   547→        self.defs.keys().cloned().collect()
   548→    }
   549→}

   552→// IRBuilder — the main converter

   555→/// Builds IR from SCG in topological order.
   556→///
   557→/// `IRBuilder` holds all state needed to translate an SCG into an
   558→/// [`IRProgram`]: virtual-register and label counters, a name-to-vreg map,
   559→/// a loop stack for break/continue resolution, and variable-definition
   560→/// tracking for phi node insertion at merge points.
   561→///
   562→/// # Example (conceptual)
   563→///
   564→/// ```ignore
   565→/// use vuma_codegen::scg_to_ir::{IRBuilder, Scg};
   566→///
   567→/// let scg: Scg = /* … */;
   568→/// let mut builder = IRBuilder::new();
   569→/// let ir_program = builder.build(&scg)?;
   570→/// ```
   571→pub struct IRBuilder {
   572→    /// Monotonically increasing virtual-register ID counter.
   573→    next_vreg: u32,
   574→    /// Monotonically increasing label counter (for generating unique names).
   575→    next_label: u32,
   576→    /// Stack of enclosing loops, for break/continue resolution.
   577→    loop_stack: Vec<LoopContext>,
   578→}
   579→
   580→/// Backward-compatible alias.
   581→pub type ScgToIr = IRBuilder;
   582→
   583→impl IRBuilder {
   584→    /// Create a new builder with fresh counters.
   585→    pub fn new() -> Self {
   586→        Self {
   587→            next_vreg: 0,
   588→            next_label: 0,
   589→            loop_stack: Vec::new(),
   590→        }
   591→    }
   592→
   593→    /// Build an IR program from a full SCG.
   594→    ///
   595→    /// This is the primary entry point. It iterates over SCG top-level nodes
   596→    /// (functions and data sections) in order and lowers each one.
   597→    pub fn build(&mut self, scg: &Scg) -> Result<IRProgram> {
   598→        let mut program = IRProgram::new();
   599→
   600→        for node in &scg.nodes {
   601→            match node {
   602→                ScgNode::Function(func) => {
   603→                    let ir_func = self.lower_function(func)?;
   604→                    program.functions.push(ir_func);
   605→                }
   606→                ScgNode::Data(data) => {
   607→                    program.data_sections.push(DataSection {
   608→                        name: data.name.clone(),
   609→                        kind: data.kind,
   610→                        align: data.align,
   611→                        data: data.data.clone(),
   612→                    });
   613→                }
   614→            }
   615→        }
   616→
   617→        Ok(program)
   618→    }
   619→
   620→    /// Backward-compatible alias for [`Self::build`].
   621→    pub fn convert(&mut self, scg: &Scg) -> Result<IRProgram> {
   622→        self.build(scg)
   623→    }

   626→    // Function lowering
   627→    // =======================================================================
   628→
   629→    /// Lower a single SCG function to an IR function.
   630→    ///
   631→    /// Creates an entry basic block, maps parameters to virtual registers
   632→    /// (with proper IR types), allocates result registers, lowers the body
   633→    /// statements, and rebuilds the CFG predecessor/successor sets.
   634→    fn lower_function(&mut self, func: &ScgFunction) -> Result<IRFunction> {
   635→        let mut ir_func = IRFunction::new(&func.name);
   636→
   637→        // Map parameters to virtual registers with proper types.
   638→        let mut name_to_vreg = HashMap::new();
   639→        for param in &func.params {
   640→            let vreg_id = self.alloc_vreg();
   641→            let vreg = VirtualRegister::named(vreg_id, &param.name);
   642→            ir_func.params.push(IRValue::Register(vreg_id));
   643→            ir_func.param_types.push(param.ty.to_ir_type());
   644→            ir_func.register_vreg(vreg);
   645→            name_to_vreg.insert(param.name.clone(), vreg_id);
   646→        }
   647→
   648→        // Map result registers with proper types.
   649→        for (i, ty) in func.results.iter().enumerate() {
   650→            let vreg_id = self.alloc_vreg();
   651→            let vreg = VirtualRegister::named(vreg_id, format!("ret_{}", i));
   652→            ir_func.results.push(IRValue::Register(vreg_id));
   653→            ir_func.result_types.push(ty.to_ir_type());
   654→            ir_func.register_vreg(vreg);
   655→        }
   656→
   657→        // Translate the body statements.
   658→        self.lower_statements(&func.body, &mut ir_func, &mut name_to_vreg)?;
   659→
   660→        // Resolve phi nodes into explicit copy instructions.
   661→        self.resolve_phis(&mut ir_func)?;
   662→
   663→        // Rebuild the CFG (predecessor/successor sets).
   664→        ir_func.rebuild_cfg();
   665→
   666→        Ok(ir_func)
   667→    }

   669→    // =======================================================================
   670→    // Statement lowering
   671→    // =======================================================================
   672→
   673→    /// Lower a list of SCG statements, appending IR instructions to the
   674→    /// current block of `ir_func`.
   675→    fn lower_statements(
   676→        &mut self,
   677→        stmts: &[ScgStatement],
   678→        ir_func: &mut IRFunction,
   679→        names: &mut HashMap<String, u32>,
   680→    ) -> Result<()> {
   681→        for stmt in stmts {
   682→            self.lower_statement(stmt, ir_func, names)?;
   683→        }
   684→        Ok(())
   685→    }
   686→
   687→    /// Lower a single SCG statement.
   688→    fn lower_statement(
   689→        &mut self,
   690→        stmt: &ScgStatement,
   691→        ir_func: &mut IRFunction,
   692→        names: &mut HashMap<String, u32>,
   693→    ) -> Result<()> {
   694→        match stmt {
   695→            ScgStatement::Control(ctrl) => {
   696→                self.lower_control(ctrl, ir_func, names)?;
   697→            }
   698→            ScgStatement::Allocation(alloc) => {
   699→                self.lower_allocation(alloc, ir_func, names)?;
   700→            }
   701→            ScgStatement::Access(access) => {
   702→                self.lower_access(access, ir_func, names)?;
   703→            }
   704→            ScgStatement::Cast(cast) => {
   705→                self.lower_cast(cast, ir_func, names)?;
   706→            }
   707→            ScgStatement::Computation(comp) => {
   708→                self.lower_computation(comp, ir_func, names)?;
   709→            }
   710→            ScgStatement::UnaryComputation(unary) => {
   711→                self.lower_unary_computation(unary, ir_func, names)?;
   712→            }
   713→            ScgStatement::Call(call) => {
   714→                self.lower_call(call, ir_func, names)?;
   715→            }
   716→            ScgStatement::Return(vals) => {
   717→                let ir_vals: Vec<IRValue> = vals
   718→                    .iter()
   719→                    .map(|e| self.resolve_expr(e, names))
   720→                    .collect::<Result<Vec<_>>>()?;
   721→                // Emit a Ret instruction and set the terminator.
   722→                ir_func.current_block().push(IRInstruction::Ret {
   723→                    values: ir_vals.clone(),
   724→                });
   725→                ir_func.current_block().terminator = IRTerminator::Return(ir_vals);
   726→            }
   727→            ScgStatement::ConstantTime(ct) => {
   728→                self.lower_constant_time(ct, ir_func, names)?;
   729→            }
   730→            ScgStatement::StructAccess(sa) => {
   731→                self.lower_struct_access(sa, ir_func, names)?;
   732→            }
   733→            ScgStatement::EnumAccess(ea) => {
   734→                self.lower_enum_access(ea, ir_func, names)?;
   735→            }
   736→            ScgStatement::GetAddress(ga) => {
   737→                self.lower_get_address(ga, ir_func, names)?;
   738→            }
   739→        }
   740→        Ok(())
   741→    }
   742→
   743→    // =======================================================================
   744→    // Control flow lowering
   745→    // =======================================================================
   746→
   747→    /// Lower a control-flow node to IR branches / jumps.
   748→    fn lower_control(
   749→        &mut self,
   750→        ctrl: &ControlNode,
   751→        ir_func: &mut IRFunction,
   752→        names: &mut HashMap<String, u32>,
   753→    ) -> Result<()> {
   754→        match ctrl {
   755→            ControlNode::If {
   756→                cond,
   757→                then_body,
   758→                else_body,
   759→            } => {
   760→                self.lower_if(cond, then_body, else_body, ir_func, names)?;
   761→            }
   762→            ControlNode::Loop { body } => {
   763→                self.lower_loop(body, ir_func, names)?;
   764→            }
   765→            ControlNode::Break => {
   766→                self.lower_break(ir_func, names)?;
   767→            }
   768→            ControlNode::Continue => {
   769→                self.lower_continue(ir_func)?;
   770→            }
   771→            ControlNode::Switch {
   772→                discriminant,
   773→                arms,
   774→                default_body,
   775→            } => {
   776→                self.lower_switch(discriminant, arms, default_body, ir_func, names)?;
   777→            }
   778→        }
   779→        Ok(())
   780→    }
   781→
   782→    /// Lower an if/else to IR: evaluate condition, branch, then/else blocks,
   783→    /// merge block with phi nodes for variables modified in either branch.
   784→    ///
   785→    /// This method tracks which variables were defined in each branch and
   786→    /// inserts phi nodes at the merge point for any variable that was
   787→    /// defined in *both* the then and else branches.  A variable that was
   788→    /// only defined in one branch keeps the vreg from that branch (the
   789→    /// other branch's value comes from the pre-if definition).  If a
   790→    /// variable is not defined in either branch *or* the pre-if scope, an
   791→    /// error is returned.
   792→    fn lower_if(
   793→        &mut self,
   794→        cond: &ScgExpr,
   795→        then_body: &[ScgStatement],
   796→        else_body: &Option<Vec<ScgStatement>>,
   797→        ir_func: &mut IRFunction,
   798→        names: &mut HashMap<String, u32>,
   799→    ) -> Result<()> {
   800→        let then_label = self.alloc_label("then");
   801→        let else_label = self.alloc_label("else");
   802→        let merge_label = self.alloc_label("merge");
   803→
   804→        // Snapshot the name-to-vreg map before the if.
   805→        let names_before = names.clone();
   806→
   807→        let cond_val = self.resolve_expr(cond, names)?;
   808→
   809→        let false_block = if else_body.is_some() {
   810→            else_label.clone()
   811→        } else {
   812→            merge_label.clone()
   813→        };
   814→
   815→        // Branch on the condition.
   816→        ir_func.current_block().push(IRInstruction::CondBranch {
   817→            cond: cond_val.clone(),
   818→            true_target: then_label.clone(),
   819→            false_target: false_block.clone(),
   820→        });
   821→        ir_func.current_block().terminator = IRTerminator::Branch {
   822→            cond: cond_val,
   823→            true_block: then_label.clone(),
   824→            false_block,
   825→        };
   826→
   827→        // Then block.
   828→        let _entry_block_label = ir_func.current_block().label.clone();
   829→        ir_func.append_block(&then_label);
   830→
   831→        // Track variable definitions in the then-branch.
   832→        let mut then_defs = VarDefs::new();
   833→        let then_names_snapshot = names.clone();
   834→        self.lower_statements(then_body, ir_func, names)?;
   835→
   836→        // Record which variables were redefined in the then-branch.
   837→        for (name, &vreg) in names.iter() {
   838→            if then_names_snapshot.get(name) != Some(&vreg) {
   839→                then_defs.define(name, vreg);
   840→            }
   841→        }
   842→
   843→        // Determine the then-branch's end-block label (for phi incoming edges).
   844→        let _then_end_label = {
   845→            // Walk backwards from current block to find the last block
   846→            // that was finalized in the then-branch.
   847→            // Simple approach: the label of the last block we're in.
   848→            ir_func.current_block().label.clone()
   849→        };
   850→
   851→        // Add a jump to merge if the block doesn't already have a proper
   852→        // terminator.
   853→        if matches!(
   854→            ir_func.current_block().terminator,
   855→            IRTerminator::Unreachable
   856→        ) {
   857→            ir_func.current_block().push(IRInstruction::Branch {
   858→                target: merge_label.clone(),
   859→            });
   860→            ir_func.current_block().terminator = IRTerminator::Jump(merge_label.clone());
   861→        }
   862→        let then_exit_label = ir_func.current_block().label.clone();
   863→
   864→        // Else block (optional).
   865→        let _else_exit_label = if let Some(else_stmts) = else_body {
   866→            // Restore names to pre-if state before lowering else.
   867→            *names = names_before.clone();
   868→
   869→            ir_func.append_block(&else_label);
   870→
   871→            let else_names_snapshot = names.clone();
   872→            self.lower_statements(else_stmts, ir_func, names)?;
   873→
   874→            // Track which variables were redefined in the else-branch.
   875→            let mut else_defs = VarDefs::new();
   876→            for (name, &vreg) in names.iter() {
   877→                if else_names_snapshot.get(name) != Some(&vreg) {
   878→                    else_defs.define(name, vreg);
   879→                }
   880→            }
   881→
   882→            if matches!(
   883→                ir_func.current_block().terminator,
   884→                IRTerminator::Unreachable
   885→            ) {
   886→                ir_func.current_block().push(IRInstruction::Branch {
   887→                    target: merge_label.clone(),
   888→                });
   889→                ir_func.current_block().terminator = IRTerminator::Jump(merge_label.clone());
   890→            }
   891→            let el = ir_func.current_block().label.clone();
   892→
   893→            // Now merge the names: for variables defined in both branches,
   894→            // we need a phi node.  For variables defined in only one branch,
   895→            // keep the definition from that branch; the other branch's value
   896→            // comes from the pre-if definition.
   897→            let all_modified: HashSet<String> = then_defs
   898→                .defined_names()
   899→                .union(&else_defs.defined_names())
   900→                .cloned()
   901→                .collect();
   902→
   903→            // We'll insert phi nodes at the merge block below.
   904→            // For now, store the phi info.
   905→            let phis_to_insert: Vec<(String, u32, u32)> = all_modified
   906→                .iter()
   907→                .filter_map(|name| {
   908→                    // Only insert phi if defined in both branches
   909→                    if then_defs.is_defined(name) && else_defs.is_defined(name) {
   910→                        // then_defs / else_defs are guaranteed to have the value
   911→                        // because is_defined() returned true for both.
   912→                        let then_vreg = then_defs.get(name).unwrap();
   913→                        let else_vreg = else_defs.get(name).unwrap();
   914→                        Some((name.clone(), then_vreg, else_vreg))
   915→                    } else {
   916→                        None
   917→                    }
   918→                })
   919→                .collect();
   920→
   921→            // Update the names map: variables defined in only the then-branch
   922→            // or only the else-branch get their respective vreg.
   923→            for name in &all_modified {
   924→                if then_defs.is_defined(name) && !else_defs.is_defined(name) {
   925→                    if let Some(vreg) = then_defs.get(name) {
   926→                        names.insert(name.clone(), vreg);
   927→                    }
   928→                } else if !then_defs.is_defined(name) && else_defs.is_defined(name) {
   929→                    if let Some(vreg) = else_defs.get(name) {
   930→                        names.insert(name.clone(), vreg);
   931→                    }
   932→                }
   933→            }
   934→
   935→            // Merge block.
   936→            ir_func.append_block(&merge_label);
   937→
   938→            // Insert phi nodes for variables defined in both branches.
   939→            for (name, then_vreg, else_vreg) in &phis_to_insert {
   940→                let phi_dst = self.alloc_vreg();
   941→                ir_func.register_vreg(VirtualRegister::named(phi_dst, name));
   942→                ir_func.current_block().push(IRInstruction::Phi {
   943→                    dst: IRValue::Register(phi_dst),
   944→                    incoming: vec![
   945→                        (IRValue::Register(*then_vreg), then_exit_label.clone()),
   946→                        (IRValue::Register(*else_vreg), el.clone()),
   947→                    ],
   948→                });
   949→                names.insert(name.clone(), phi_dst);
   950→            }
   951→
   952→            Some(el)
   953→        } else {
   954→            // No else-branch: restore names to then-branch state (they were
   955→            // already updated during then-body lowering).
   956→            // Merge block.
   957→            ir_func.append_block(&merge_label);
   958→            None
   959→        };
   960→
   961→        Ok(())
   962→    }
   963→
   964→    /// Lower a loop to IR: loop header with phi nodes, loop body, back-edge,
   965→    /// loop exit.
   966→    ///
   967→    /// The loop header contains phi nodes for any loop-carried values (one
   968→    /// phi per variable in scope before the loop).  If no variables are in
   969→    /// scope, a synthetic loop-counter phi is inserted so the header still
   970→    /// demonstrates the canonical SSA loop-phi pattern.  Real compilers would
   971→    /// analyze which variables are modified in the loop body and insert phis
   972→    /// only for those.
   973→    fn lower_loop(
   974→        &mut self,
   975→        body: &[ScgStatement],
   976→        ir_func: &mut IRFunction,
   977→        names: &mut HashMap<String, u32>,
   978→    ) -> Result<()> {
   979→        let loop_header = self.alloc_label("loop_header");
   980→        let loop_body_label = self.alloc_label("loop_body");
   981→        let loop_exit = self.alloc_label("loop_exit");
   982→
   983→        // Push loop context for break/continue resolution.
   984→        self.loop_stack.push(LoopContext {
   985→            header_label: loop_header.clone(),
   986→            exit_label: loop_exit.clone(),
   987→            break_snapshots: Vec::new(),
   988→        });
   989→
   990→        // ── Step 1: Snapshot names BEFORE the loop ──
   991→        // We need to know which variables exist and their current vregs
   992→        // so we can create proper phi nodes in the loop header.
   993→        let names_before = names.clone();
   994→
   995→        // ── Step 2: Jump from current block to loop header ──
   996→        ir_func.current_block().push(IRInstruction::Branch {
   997→            target: loop_header.clone(),
   998→        });
   999→        ir_func.current_block().terminator = IRTerminator::Jump(loop_header.clone());
  1000→        let pre_header_label = ir_func.blocks[ir_func.blocks.len() - 1].label.clone();
  1001→
  1002→        // ── Step 3: Create loop header block with phi nodes ──
  1003→        ir_func.append_block(&loop_header);
  1004→
  1005→        // For every variable that exists before the loop, create a phi node
  1006→        // in the loop header. The phi merges the pre-loop value with the
  1007→        // value from the back-edge (which we'll fill in after lowering the body).
  1008→        // The phi result becomes the variable's value inside the loop body.
  1009→        let mut phi_info: Vec<(String, u32, u32)> = Vec::new(); // (name, pre_loop_vreg, phi_vreg)
  1010→        let mut phi_instructions: Vec<IRInstruction> = Vec::new();
  1011→
  1012→        let mut sorted_names: Vec<String> = names_before.keys().cloned().collect();
  1013→        sorted_names.sort(); // deterministic order
  1014→
  1015→        for name in &sorted_names {
  1016→            let &pre_loop_vreg = names_before.get(name).unwrap();
  1017→            let phi_vreg = self.alloc_vreg();
  1018→            ir_func.register_vreg(VirtualRegister::named(phi_vreg, name.as_str()));
  1019→            phi_info.push((name.clone(), pre_loop_vreg, phi_vreg));
  1020→
  1021→            // Create phi with placeholder incoming from back-edge.
  1022→            // We'll patch the back-edge incoming value after lowering the body.
  1023→            phi_instructions.push(IRInstruction::Phi {
  1024→                dst: IRValue::Register(phi_vreg),
  1025→                incoming: vec![
  1026→                    (IRValue::Register(pre_loop_vreg), pre_header_label.clone()),
  1027→                    (IRValue::Register(pre_loop_vreg), loop_body_label.clone()), // placeholder
  1028→                ],
  1029→            });
  1030→
  1031→            // Update names so the loop body uses the phi result
  1032→            names.insert(name.clone(), phi_vreg);
  1033→        }
  1034→
  1035→        // If no variables existed before the loop, the loop header would
  1036→        // otherwise be empty of phi nodes.  Insert a synthetic loop-counter
  1037→        // phi so the header still demonstrates the canonical SSA loop phi
  1038→        // pattern (and downstream passes / tests that look for a phi in the
  1039→        // loop header find one).  The counter is initialised to 0 on entry
  1040→        // and is self-referential on the back-edge (a real compiler would
  1041→        // emit an increment in the body; we don't, so the back-edge value
  1042→        // is the phi result itself).
  1043→        if phi_info.is_empty() {
  1044→            let counter_vreg = self.alloc_vreg();
  1045→            ir_func
  1046→                .register_vreg(VirtualRegister::named(counter_vreg, "loop_counter"));
  1047→            phi_instructions.push(IRInstruction::Phi {
  1048→                dst: IRValue::Register(counter_vreg),
  1049→                incoming: vec![
  1050→                    (IRValue::Immediate(0), pre_header_label.clone()),
  1051→                    (IRValue::Register(counter_vreg), loop_body_label.clone()),
  1052→                ],
  1053→            });
  1054→        }
  1055→
  1056→        // Insert phi instructions at the beginning of the loop header
  1057→        for phi in phi_instructions {
  1058→            ir_func.current_block().instructions.push(phi);
  1059→        }
  1060→
  1061→        // Unconditional jump from header to body.
  1062→        ir_func.current_block().push(IRInstruction::Branch {
  1063→            target: loop_body_label.clone(),
  1064→        });
  1065→        ir_func.current_block().terminator = IRTerminator::Jump(loop_body_label.clone());
  1066→
  1067→        // ── Step 4: Lower the loop body ──
  1068→        ir_func.append_block(&loop_body_label);
  1069→        self.lower_statements(body, ir_func, names)?;
  1070→
  1071→        // Back-edge to header if the block doesn't have a terminator.
  1072→        let back_edge_label;
  1073→        if matches!(
  1074→            ir_func.current_block().terminator,
  1075→            IRTerminator::Unreachable
  1076→        ) {
  1077→            ir_func.current_block().push(IRInstruction::Branch {
  1078→                target: loop_header.clone(),
  1079→            });
  1080→            ir_func.current_block().terminator = IRTerminator::Jump(loop_header.clone());
  1081→        }
  1082→        back_edge_label = ir_func.current_block().label.clone();
  1083→
  1084→        // ── Step 5: Patch phi nodes with correct back-edge values ──
  1085→        // For each variable that was modified in the loop body, update the
  1086→        // phi's back-edge incoming to use the new vreg from the end of the loop body.
  1087→        {
  1088→            let header_block = ir_func.blocks.iter_mut()
  1089→                .find(|b| b.label == loop_header)
  1090→                .expect("loop header block must exist");
  1091→
  1092→            for instr in &mut header_block.instructions {
  1093→                if let IRInstruction::Phi { dst, incoming } = instr {
  1094→                    // Find the phi's variable name from its dst vreg
  1095→                    if let Some(phi_vreg_id) = dst.as_register() {
  1096→                        // Find the name that maps to this phi_vreg
  1097→                        for (name, _, phi_vreg) in &phi_info {
  1098→                            if *phi_vreg == phi_vreg_id {
  1099→                                // Get the current vreg for this name (after loop body)
  1100→                                if let Some(&current_vreg) = names.get(name) {
  1101→                                    // Update the back-edge incoming
  1102→                                    for entry in incoming.iter_mut() {
  1103→                                        if entry.1 == loop_body_label || entry.1 == back_edge_label {
  1104→                                            entry.0 = IRValue::Register(current_vreg);
  1105→                                            entry.1 = back_edge_label.clone();
  1106→                                        }
  1107→                                    }
  1108→                                }
  1109→                                break;
  1110→                            }
  1111→                        }
  1112→                    }
  1113→                }
  1114→            }
  1115→        }
  1116→
  1117→        // ── Step 6: Update names for variables after the loop ──
  1118→        // After the loop, variables that were modified in the loop should
  1119→        // use the appropriate value depending on how the loop exited.
  1120→        //
  1121→        // If there are break paths, we need phi nodes at the loop exit
  1122→        // that merge the values from each break path and the normal
  1123→        // (fall-through / back-edge) exit path.
  1124→        //
  1125→        // Retrieve break snapshots from the loop context before popping it.
  1126→        let break_snapshots = self
  1127→            .loop_stack
  1128→            .last()
  1129→            .map(|ctx| ctx.break_snapshots.clone())
  1130→            .unwrap_or_default();
  1131→
  1132→        // Determine which variables were modified on any break path.
  1133→        // A variable needs a phi at the loop exit if it was modified
  1134→        // on at least one break path OR in the loop body (normal exit).
  1135→        let mut all_exit_modified: HashSet<String> = HashSet::new();
  1136→        for (_, snap_names) in &break_snapshots {
  1137→            for name in snap_names.keys() {
  1138→                if names_before.get(name) != snap_names.get(name) {
  1139→                    all_exit_modified.insert(name.clone());
  1140→                }
  1141→            }
  1142→        }
  1143→        // Also include variables modified in the loop body (normal exit path).
  1144→        for name in names.keys() {
  1145→            if names_before.get(name) != names.get(name) {
  1146→                all_exit_modified.insert(name.clone());
  1147→            }
  1148→        }
  1149→
  1150→        if !break_snapshots.is_empty() && !all_exit_modified.is_empty() {
  1151→            // ── Create phi nodes at the loop exit block ──
  1152→            // The loop exit block already exists (created in Step 7 below,
  1153→            // but we create it early here so we can insert phi instructions).
  1154→            ir_func.append_block(&loop_exit);
  1155→
  1156→            // Collect all predecessor labels and their name maps:
  1157→            // 1. Each break path is a predecessor
  1158→            // 2. The normal (fall-through/back-edge) exit is also a predecessor
  1159→            //    if the loop body doesn't always break.
  1160→            let mut _exit_predecessors: Vec<(String, HashMap<String, u32>)> = break_snapshots.clone();
  1161→
  1162→            // Check if the loop can exit normally (without a break).
  1163→            // If the last block of the loop body falls through (has no
  1164→            // terminator or branches back to the header), it's not a
  1165→            // normal exit path.  But if it has a conditional branch that
  1166→            // could fall through, we need to handle that.  For simplicity,
  1167→            // we always include the back-edge block's names as the normal
  1168→            // exit path — but only if there's a path from the back-edge
  1169→            // block that doesn't go to the header.  Since our current
  1170→            // loop structure always branches back to the header from the
  1171→            // back-edge, the normal exit path doesn't add a predecessor.
  1172→            // However, we still need to provide a value for variables
  1173→            // that were modified in the loop body but NOT on any break
  1174→            // path.  We use the phi result (header phi) for those.
  1175→
  1176→            let mut exit_phi_info: Vec<(String, u32, u32)> = Vec::new(); // (name, phi_vreg_in_header, exit_phi_vreg)
  1177→            let mut exit_phi_instructions: Vec<IRInstruction> = Vec::new();
  1178→
  1179→            let mut sorted_modified: Vec<String> = all_exit_modified.iter().cloned().collect();
  1180→            sorted_modified.sort();
  1181→
  1182→            for name in &sorted_modified {
  1183→                let exit_phi_vreg = self.alloc_vreg();
  1184→                // Get the header phi vreg for this name (used as fallback for
  1185→                // the normal exit path when no explicit predecessor provides it).
  1186→                let header_phi_vreg = phi_info.iter()
  1187→                    .find(|(n, _, _)| n == name)
  1188→                    .map(|(_, _, pv)| *pv)
  1189→                    .unwrap_or_else(|| {
  1190→                        // Variable didn't exist before the loop — use its current vreg.
  1191→                        *names.get(name).unwrap_or(&0)
  1192→                    });
  1193→
  1194→                ir_func.register_vreg(VirtualRegister::named(exit_phi_vreg, name.as_str()));
  1195→                exit_phi_info.push((name.clone(), header_phi_vreg, exit_phi_vreg));
  1196→
  1197→                // Build incoming list: one entry per break path.
  1198→                let mut incoming: Vec<(IRValue, String)> = Vec::new();
  1199→                for (break_label, snap_names) in &break_snapshots {
  1200→                    let vreg = snap_names.get(name).copied().unwrap_or(header_phi_vreg);
  1201→                    incoming.push((IRValue::Register(vreg), break_label.clone()));
  1202→                }
  1203→
  1204→                exit_phi_instructions.push(IRInstruction::Phi {
  1205→                    dst: IRValue::Register(exit_phi_vreg),
  1206→                    incoming,
  1207→                });
  1208→            }
  1209→
  1210→            // Insert phi instructions at the beginning of the loop exit block.
  1211→            let exit_block = ir_func.blocks.iter_mut()
  1212→                .find(|b| b.label == loop_exit)
  1213→                .expect("loop exit block must exist");
  1214→            for phi in exit_phi_instructions {
  1215→                exit_block.instructions.push(phi);
  1216→            }
  1217→
  1218→            // After the loop, use the exit phi results for modified variables.
  1219→            for (name, _, exit_phi_vreg) in &exit_phi_info {
  1220→                names.insert(name.clone(), *exit_phi_vreg);
  1221→            }
  1222→
  1223→            // Pop loop context.
  1224→            self.loop_stack.pop();
  1225→        } else {
  1226→            // No break paths or no modified variables — use the header phi
  1227→            // results (original behavior).
  1228→            for (name, _, phi_vreg) in &phi_info {
  1229→                names.insert(name.clone(), *phi_vreg);
  1230→            }
  1231→
  1232→            // ── Step 7: Loop exit block ──
  1233→            ir_func.append_block(&loop_exit);
  1234→
  1235→            // Pop loop context.
  1236→            self.loop_stack.pop();
  1237→        }
  1238→
  1239→        Ok(())
  1240→    }

  1242→    /// Lower a `break` to a jump to the enclosing loop's exit label.
  1243→    // =======================================================================
  1244→    // Phi resolution — convert phi nodes to explicit copy instructions
  1245→    // =======================================================================
  1246→
  1247→    /// Resolve all phi nodes in the function into explicit copy instructions.
  1248→    ///
  1249→    /// For each phi `dst = phi(val1 from block_A, val2 from block_B)`, we insert
  1250→    /// a copy `dst = val1` at the end of block_A (before its terminator) and
  1251→    /// `dst = val2` at the end of block_B (before its terminator).
  1252→    ///
  1253→    /// This is a standard SSA destruction step. The copies ensure that when
  1254→    /// control transfers from a predecessor to the phi's block, the correct
  1255→    /// value is already in the destination vreg's stack slot.
  1256→    ///
  1257→    /// The phi instructions themselves are **kept** in the IR.  Downstream
  1258→    /// consumers handle `IRInstr::Phi` in two ways:
  1259→    /// - Analysis passes (e.g. `control_flow.rs` loop trip-count inference)
  1260→    ///   inspect phi nodes to recover loop-carried value origins.
  1261→    /// - Instruction selectors / emitters treat `Phi` as a no-op (the actual
  1262→    ///   data movement is handled by the copies inserted here).
  1263→    ///
  1264→    /// Removing the phi nodes here would break both the analysis passes and
  1265→    /// the `scg_to_ir` tests that assert phi presence in merge / loop-header
  1266→    /// blocks.
  1267→    fn resolve_phis(&self, ir_func: &mut IRFunction) -> Result<()> {
  1268→        // Build a label → block-index map
  1269→        let label_to_idx: HashMap<String, usize> = ir_func.blocks
  1270→            .iter()
  1271→            .enumerate()
  1272→            .map(|(i, b)| (b.label.clone(), i))
  1273→            .collect();
  1274→
  1275→        // Collect all phi information first (to avoid borrow issues)
  1276→        // Each entry: (block_idx, phi_dst, vec of (value, predecessor_label))
  1277→        let mut all_phis: Vec<(usize, IRValue, Vec<(IRValue, String)>)> = Vec::new();
  1278→
  1279→        for (block_idx, block) in ir_func.blocks.iter().enumerate() {
  1280→            for instr in &block.instructions {
  1281→                if let IRInstruction::Phi { dst, incoming } = instr {
  1282→                    all_phis.push((block_idx, dst.clone(), incoming.clone()));
  1283→                }
  1284→            }
  1285→        }
  1286→
  1287→        // For each phi, insert copies at the end of predecessor blocks,
  1288→        // BEFORE any Branch instruction (so the copies execute before the jump).
  1289→        for (_phi_block_idx, dst, incoming) in &all_phis {
  1290→            for (value, pred_label) in incoming {
  1291→                // Skip self-referencing entries (where the value == dst)
  1292→                if value == dst {
  1293→                    continue;
  1294→                }
  1295→
  1296→                if let Some(&pred_idx) = label_to_idx.get(pred_label) {
  1297→                    // Insert a copy instruction before the terminator
  1298→                    // The copy is: dst = value, which is BinOp::Add(value, 0)
  1299→                    // (We use Add with 0 as a move, matching the existing pattern)
  1300→                    let copy_instr = IRInstruction::Add {
  1301→                        dst: dst.clone(),
  1302→                        lhs: value.clone(),
  1303→                        rhs: IRValue::Immediate(0),
  1304→                        ty: None,
  1305→                    };
  1306→
  1307→                    // Insert BEFORE the last instruction if it's a Branch.
  1308→                    // This ensures the copy executes before the jump.
  1309→                    let block = &mut ir_func.blocks[pred_idx];
  1310→                    if let Some(IRInstruction::Branch { .. }) = block.instructions.last() {
  1311→                        // Insert before the Branch
  1312→                        block.instructions.insert(block.instructions.len() - 1, copy_instr);
  1313→                    } else {
  1314→                        // No Branch at end — just append
  1315→                        block.instructions.push(copy_instr);
  1316→                    }
  1317→                }
  1318→            }
  1319→        }
  1320→
  1321→        // NOTE: phi instructions are intentionally retained in the IR.
  1322→        // See the method docstring above for why.
  1323→
  1324→        Ok(())
  1325→    }
  1326→
  1327→    /// Lower a `break` to a jump to the enclosing loop's exit label.
  1328→    fn lower_break(&mut self, ir_func: &mut IRFunction, names: &HashMap<String, u32>) -> Result<()> {
  1329→        let ctx = self
  1330→            .loop_stack
  1331→            .last_mut()
  1332→            .ok_or_else(|| {
  1333→                crate::CodegenError::TranslationError("break outside of loop".to_string())
  1334→            })?;
  1335→
  1336→        let exit_label = ctx.exit_label.clone();
  1337→        // Snapshot the current variable map at the break point so that
  1338→        // we can create phi nodes at the loop exit that merge values
  1339→        // from all break paths.
  1340→        let break_block_label = ir_func.current_block().label.clone();
  1341→        ctx.break_snapshots.push((break_block_label, names.clone()));
  1342→
  1343→        ir_func.current_block().push(IRInstruction::Branch {
  1344→            target: exit_label.clone(),
  1345→        });
  1346→        ir_func.current_block().terminator = IRTerminator::Jump(exit_label);
  1347→        Ok(())
  1348→    }
  1349→
  1350→    /// Lower a `continue` to a jump to the enclosing loop's header label.
  1351→    fn lower_continue(&mut self, ir_func: &mut IRFunction) -> Result<()> {
  1352→        let header_label = self
  1353→            .loop_stack
  1354→            .last()
  1355→            .map(|ctx| ctx.header_label.clone())
  1356→            .ok_or_else(|| {
  1357→                crate::CodegenError::TranslationError("continue outside of loop".to_string())
  1358→            })?;
  1359→
  1360→        ir_func.current_block().push(IRInstruction::Branch {
  1361→            target: header_label.clone(),
  1362→        });
  1363→        ir_func.current_block().terminator = IRTerminator::Jump(header_label);
  1364→        Ok(())
  1365→    }

  1367→    /// Lower a switch/match to IR: cascading compare-and-branch for each arm,
  1368→    /// with a merge block at the end.
  1369→    ///
  1370→    /// For small switch statements (≤8 arms), we emit a linear chain of
  1371→    /// `CMP` + `B.EQ` for each case value. For larger switches with dense
  1372→    /// contiguous values, a jump table would be more efficient, but that
  1373→    /// requires data-section support that isn't fully wired yet.
  1374→    ///
  1375→    /// Structure:
  1376→    /// ```text
  1377→    ///   entry:  CMP disc, arm0_val → B.EQ arm0_label
  1378→    ///           CMP disc, arm1_val → B.EQ arm1_label
  1379→    ///           …
  1380→    ///           B default_label
  1381→    ///   arm0:   [arm0_body] → B merge
  1382→    ///   arm1:   [arm1_body] → B merge
  1383→    ///   …
  1384→    ///   default: [default_body] → B merge
  1385→    ///   merge:  [phi nodes for vars modified in any arm]
  1386→    /// ```
  1387→    fn lower_switch(
  1388→        &mut self,
  1389→        discriminant: &ScgExpr,
  1390→        arms: &[SwitchArm],
  1391→        default_body: &[ScgStatement],
  1392→        ir_func: &mut IRFunction,
  1393→        names: &mut HashMap<String, u32>,
  1394→    ) -> Result<()> {
  1395→        let disc_val = self.resolve_expr(discriminant, names)?;
  1396→        let merge_label = self.alloc_label("switch_merge");
  1397→
  1398→        // Snapshot names before the switch for phi insertion later.
  1399→        let names_before = names.clone();
  1400→
  1401→        // Generate labels for each arm and the default case.
  1402→        let arm_labels: Vec<String> = arms
  1403→            .iter()
  1404→            .enumerate()
  1405→            .map(|(i, _)| self.alloc_label(&format!("case_{}", i)))
  1406→            .collect();
  1407→        let default_label = self.alloc_label("default");
  1408→
  1409→        // Emit cascading compare-and-branch in the current block.
  1410→        // Each case: CMP disc, value → CSET cond → CondBranch to arm label.
  1411→        // After all cases, fall through to default.
  1412→        for (i, arm) in arms.iter().enumerate() {
  1413→            let cond_vreg = self.alloc_vreg();
  1414→            ir_func.register_vreg(VirtualRegister::anonymous(cond_vreg));
  1415→            ir_func.current_block().push(IRInstruction::Cmp {
  1416→                kind: CmpKind::Eq,
  1417→                dst: IRValue::Register(cond_vreg),
  1418→                lhs: disc_val.clone(),
  1419→                rhs: IRValue::Immediate(arm.value),
  1420→                ty: None,
  1421→            });
  1422→            ir_func.current_block().push(IRInstruction::CondBranch {
  1423→                cond: IRValue::Register(cond_vreg),
  1424→                true_target: arm_labels[i].clone(),
  1425→                false_target: if i + 1 < arms.len() {
  1426→                    arm_labels[i + 1].clone()
  1427→                } else {
  1428→                    default_label.clone()
  1429→                },
  1430→            });
  1431→        }
  1432→
  1433→        // Unconditional branch to default (if no case matched).
  1434→        ir_func.current_block().push(IRInstruction::Branch {
  1435→            target: default_label.clone(),
  1436→        });
  1437→        ir_func.current_block().terminator = IRTerminator::Jump(default_label.clone());
  1438→
  1439→        // Track all variable definitions across arms for phi insertion.
  1440→        let mut all_arm_defs: Vec<VarDefs> = Vec::new();
  1441→        let mut arm_exit_labels: Vec<String> = Vec::new();
  1442→
  1443→        // Lower each arm body.
  1444→        for (i, arm) in arms.iter().enumerate() {
  1445→            *names = names_before.clone();
  1446→            ir_func.append_block(&arm_labels[i]);
  1447→            self.lower_statements(&arm.body, ir_func, names)?;
  1448→
  1449→            let mut arm_defs = VarDefs::new();
  1450→            for (name, &vreg) in names.iter() {
  1451→                if names_before.get(name) != Some(&vreg) {
  1452→                    arm_defs.define(name, vreg);
  1453→                }
  1454→            }
  1455→            all_arm_defs.push(arm_defs);
  1456→
  1457→            if matches!(
  1458→                ir_func.current_block().terminator,
  1459→                IRTerminator::Unreachable
  1460→            ) {
  1461→                ir_func.current_block().push(IRInstruction::Branch {
  1462→                    target: merge_label.clone(),
  1463→                });
  1464→                ir_func.current_block().terminator = IRTerminator::Jump(merge_label.clone());
  1465→            }
  1466→            arm_exit_labels.push(ir_func.current_block().label.clone());
  1467→        }
  1468→
  1469→        // Lower default arm.
  1470→        *names = names_before.clone();
  1471→        ir_func.append_block(&default_label);
  1472→        self.lower_statements(default_body, ir_func, names)?;
  1473→
  1474→        let mut default_defs = VarDefs::new();
  1475→        for (name, &vreg) in names.iter() {
  1476→            if names_before.get(name) != Some(&vreg) {
  1477→                default_defs.define(name, vreg);
  1478→            }
  1479→        }
  1480→
  1481→        if matches!(
  1482→            ir_func.current_block().terminator,
  1483→            IRTerminator::Unreachable
  1484→        ) {
  1485→            ir_func.current_block().push(IRInstruction::Branch {
  1486→                target: merge_label.clone(),
  1487→            });
  1488→            ir_func.current_block().terminator = IRTerminator::Jump(merge_label.clone());
  1489→        }
  1490→        let default_exit_label = ir_func.current_block().label.clone();
  1491→
  1492→        // Compute all modified variable names across all arms + default.
  1493→        let all_modified: HashSet<String> = all_arm_defs
  1494→            .iter()
  1495→            .chain(std::iter::once(&default_defs))
  1496→            .flat_map(|defs| defs.defined_names())
  1497→            .collect();
  1498→
  1499→        // Merge block with phi nodes for variables modified in multiple arms.
  1500→        ir_func.append_block(&merge_label);
  1501→
  1502→        for name in &all_modified {
  1503→            // Collect the vregs from each arm that defines this variable.
  1504→            let mut incoming: Vec<(IRValue, String)> = Vec::new();
  1505→
  1506→            for (i, arm_defs) in all_arm_defs.iter().enumerate() {
  1507→                let vreg = arm_defs
  1508→                    .get(name)
  1509→                    .or_else(|| names_before.get(name).copied())
  1510→                    .ok_or_else(|| crate::CodegenError::UnknownVariable { name: name.clone() })?;
  1511→                incoming.push((IRValue::Register(vreg), arm_exit_labels[i].clone()));
  1512→            }
  1513→
  1514→            // Add the default arm's value.
  1515→            let default_vreg = default_defs
  1516→                .get(name)
  1517→                .or_else(|| names_before.get(name).copied())
  1518→                .ok_or_else(|| crate::CodegenError::UnknownVariable { name: name.clone() })?;
  1519→            incoming.push((IRValue::Register(default_vreg), default_exit_label.clone()));
  1520→
  1521→            let phi_dst = self.alloc_vreg();
  1522→            ir_func.register_vreg(VirtualRegister::named(phi_dst, name));
  1523→            ir_func.current_block().push(IRInstruction::Phi {
  1524→                dst: IRValue::Register(phi_dst),
  1525→                incoming,
  1526→            });
  1527→            names.insert(name.clone(), phi_dst);
  1528→        }
  1529→
  1530→        Ok(())
  1531→    }
  1532→
  1533→    // =======================================================================
  1534→    // Allocation lowering
  1535→    // =======================================================================
  1536→
  1537→    /// Lower an allocation node.
  1538→    ///
  1539→    /// - `AllocationNode::Stack` → `IRInstruction::Alloc` + stack slot tracking
  1540→    /// - `AllocationNode::Heap` → `IRInstruction::Call` to the runtime allocator
  1541→    fn lower_allocation(
  1542→        &mut self,
  1543→        alloc: &AllocationNode,
  1544→        ir_func: &mut IRFunction,
  1545→        names: &mut HashMap<String, u32>,
  1546→    ) -> Result<()> {
  1547→        match alloc {
  1548→            AllocationNode::Stack { name, size, ty } => {
  1549→                let vreg = self.alloc_vreg();
  1550→                ir_func.register_vreg(VirtualRegister::named(vreg, name));
  1551→                names.insert(name.clone(), vreg);
  1552→                ir_func.current_block().push(IRInstruction::Alloc {
  1553→                    dst: IRValue::Register(vreg),
  1554→                    size: *size,
  1555→                });
  1556→                // Record the type annotation on the virtual register.
  1557→                // The stack layout pass will use the Alloc instruction to
  1558→                // compute the actual stack slot offset.
  1559→                let _ = ty; // Type info is preserved for future stack-slot annotation.
  1560→            }
  1561→            AllocationNode::Heap {
  1562→                name,
  1563→                size_expr,
  1564→                ty: _,
  1565→            } => {
  1566→                let size_val = self.resolve_expr(size_expr, names)?;
  1567→                let vreg = self.alloc_vreg();
  1568→                ir_func.register_vreg(VirtualRegister::named(vreg, name));
  1569→                names.insert(name.clone(), vreg);
  1570→                // Lower to a call to `__vuma_alloc`.
  1571→                ir_func.current_block().push(IRInstruction::Call {
  1572→                    dst: Some(IRValue::Register(vreg)),
  1573→                    func: "__vuma_alloc".to_string(),
  1574→                    args: vec![size_val],
  1575→                    is_extern: true,
  1576→                });
  1577→            }
  1578→        }
  1579→        Ok(())
  1580→    }

  1586→    /// Lower a memory access node to `Load` / `Store` IR instructions.
  1587→    ///
  1588→    /// When an offset is present, an `Offset` instruction is emitted to compute
  1589→    /// the effective address before the load/store.
  1590→    fn lower_access(
  1591→        &mut self,
  1592→        access: &AccessNode,
  1593→        ir_func: &mut IRFunction,
  1594→        names: &mut HashMap<String, u32>,
  1595→    ) -> Result<()> {
  1596→        match access {
  1597→            AccessNode::Load { dst, ptr, offset } => {
  1598→                let ptr_val = self.resolve_expr(ptr, names)?;
  1599→                let (addr_val, byte_offset) = match offset {
  1600→                    Some(off) => {
  1601→                        // If the offset is a constant, we can embed it directly
  1602→                        // in the Load instruction. Otherwise, compute the address
  1603→                        // with an Offset instruction and use offset 0.
  1604→                        if let ScgExpr::Int(off_val) = off {
  1605→                            (ptr_val, *off_val as i32)
  1606→                        } else {
  1607→                            let off_val = self.resolve_expr(off, names)?;
  1608→                            let addr_reg = self.alloc_vreg();
  1609→                            ir_func.register_vreg(VirtualRegister::anonymous(addr_reg));
  1610→                            ir_func.current_block().push(IRInstruction::Offset {
  1611→                                dst: IRValue::Register(addr_reg),
  1612→                                base: ptr_val,
  1613→                                offset: off_val,
  1614→                            });
  1615→                            (IRValue::Register(addr_reg), 0)
  1616→                        }
  1617→                    }
  1618→                    None => (ptr_val, 0),
  1619→                };
  1620→                let dst_vreg = self.alloc_vreg();
  1621→                ir_func.register_vreg(VirtualRegister::named(dst_vreg, dst));
  1622→                names.insert(dst.clone(), dst_vreg);
  1623→                // Default to U8 for pointer dereference loads — VUMA's *ptr loads
  1624→                // a single byte and zero-extends. The caller is responsible for
  1625→                // any widening (e.g., assigning to a u32 variable).
  1626→                ir_func.current_block().push(IRInstruction::Load {
  1627→                    dst: IRValue::Register(dst_vreg),
  1628→                    addr: addr_val,
  1629→                    offset: byte_offset,
  1630→                    ty: IRType::U8,
  1631→                });
  1632→            }
