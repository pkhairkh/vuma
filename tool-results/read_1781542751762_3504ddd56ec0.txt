     1→//! Structured JSON diagnostics output for LLM integration.
     2→//!
     3→//! This module provides machine-readable diagnostic messages that can be
     4→//! serialized to JSON, enabling LLM-based tools and IDE integrations to
     5→//! programmatically consume compiler errors, warnings, and hints.
     6→//!
     7→//! # Architecture
     8→//!
     9→//! Each diagnostic carries a structured error code (`"E001"`, `"W001"`, etc.),
    10→//! a severity level, a human-readable message, a source identifier (which
    11→//! compiler stage produced it), a precise source location, optional related
    12→//! information, quick-fix suggestions with edit ranges, and an optional
    13→//! causal chain linking to parent diagnostics.
    14→//!
    15→//! # Error Catalog
    16→//!
    17→//! The diagnostic codes are organized into the following ranges:
    18→//!
    19→//! | Range    | Category       | Description                              |
    20→//! |----------|----------------|------------------------------------------|
    21→//! | E001–E030| Compilation    | Syntax, type, name resolution errors     |
    22→//! | E031–E040| Codegen        | Register allocation, encoding, relocation|
    23→//! | E041–E050| Verification   | Invariant violations, proof failures     |
    24→//! | W001–W010| Warnings       | Unused vars, performance hints            |
    25→//! | I001–I005| Informational  | General compiler information messages     |
    26→//!
    27→//! # Example JSON output
    28→//!
    29→//! ```json
    30→//! [
    31→//!   {
    32→//!     "code": "E002",
    33→//!     "severity": "error",
    34→//!     "message": "undefined variable `foo`",
    35→//!     "source": "parser",
    36→//!     "location": {
    37→//!       "file": "main.vu",
    38→//!       "start_line": 10,
    39→//!       "start_col": 4,
    40→//!       "end_line": 10,
    41→//!       "end_col": 7
    42→//!     },
    43→//!     "related": [],
    44→//!     "suggestions": ["did you mean `for`?"],
    45→//!     "chain": []
    46→//!   }
    47→//! ]
    48→//! ```
    49→//!
    50→//! # Integration
    51→//!
    52→//! Use [`from_parse_error`] to convert parser errors, [`from_vuma_error`] for
    53→//! pipeline-level errors, and [`from_codegen_error`] for code-generation
    54→//! failures.
    55→

    65→// ═══════════════════════════════════════════════════════════════════════════
    66→// DiagnosticSeverity
    67→// ═══════════════════════════════════════════════════════════════════════════
    68→
    69→/// Severity level for a machine-readable diagnostic.
    70→///
    71→/// Unlike the parser's `Severity` (which has `Note`), this uses `Hint` for
    72→/// LLM-consumable suggestions and `Info` for informational notes.
    73→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
    74→#[serde(rename_all = "lowercase")]
    75→pub enum DiagnosticSeverity {
    76→    /// A hard error — the program cannot be compiled.
    77→    Error,
    78→    /// A warning — suspicious code that is technically valid.
    79→    Warning,
    80→    /// An informational note, typically providing context.
    81→    Info,
    82→    /// A hint or quick-fix suggestion.
    83→    Hint,
    84→}
    85→
    86→impl fmt::Display for DiagnosticSeverity {
    87→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
    88→        match self {
    89→            DiagnosticSeverity::Error => write!(f, "error"),
    90→            DiagnosticSeverity::Warning => write!(f, "warning"),
    91→            DiagnosticSeverity::Info => write!(f, "info"),
    92→            DiagnosticSeverity::Hint => write!(f, "hint"),
    93→        }
    94→    }
    95→}
    96→
    97→impl From<ParserSeverity> for DiagnosticSeverity {
    98→    fn from(s: ParserSeverity) -> Self {
    99→        match s {
   100→            ParserSeverity::Error => DiagnosticSeverity::Error,
   101→            ParserSeverity::Warning => DiagnosticSeverity::Warning,
   102→            ParserSeverity::Note => DiagnosticSeverity::Info,
   103→        }
   104→    }
   105→}
   106→
   107→impl DiagnosticSeverity {
   108→    /// Returns the LSP numeric severity value.
   109→    pub fn to_lsp_severity(&self) -> u32 {
   110→        match self {
   111→            DiagnosticSeverity::Error => 1,
   112→            DiagnosticSeverity::Warning => 2,
   113→            DiagnosticSeverity::Info => 3,
   114→            DiagnosticSeverity::Hint => 4,
   115→        }
   116→    }
   117→}

   119→// ═══════════════════════════════════════════════════════════════════════════
   120→// DiagnosticSourceLocation
   121→// ═══════════════════════════════════════════════════════════════════════════
   122→
   123→/// Machine-readable source location with line/column ranges.
   124→///
   125→/// All positions are 1-based (as is conventional for compiler diagnostics)
   126→/// rather than the 0-based positions used internally by the parser.
   127→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   128→pub struct DiagnosticSourceLocation {
   129→    /// File path or module name.
   130→    pub file: String,
   131→    /// Start line (1-based).
   132→    pub start_line: u32,
   133→    /// Start column (1-based).
   134→    pub start_col: u32,
   135→    /// End line (1-based).
   136→    pub end_line: u32,
   137→    /// End column (1-based, exclusive).
   138→    pub end_col: u32,
   139→}
   140→
   141→impl DiagnosticSourceLocation {
   142→    /// Create a location spanning a single point (zero-width).
   143→    pub fn point(file: impl Into<String>, line: u32, col: u32) -> Self {
   144→        Self {
   145→            file: file.into(),
   146→            start_line: line,
   147→            start_col: col,
   148→            end_line: line,
   149→            end_col: col,
   150→        }
   151→    }
   152→
   153→    /// Create a location spanning a range within a single line.
   154→    pub fn range(
   155→        file: impl Into<String>,
   156→        line: u32,
   157→        start_col: u32,
   158→        end_col: u32,
   159→    ) -> Self {
   160→        Self {
   161→            file: file.into(),
   162→            start_line: line,
   163→            start_col,
   164→            end_line: line,
   165→            end_col,
   166→        }
   167→    }
   168→
   169→    /// Create a multi-line location.
   170→    pub fn multi_line(
   171→        file: impl Into<String>,
   172→        start_line: u32,
   173→        start_col: u32,
   174→        end_line: u32,
   175→        end_col: u32,
   176→    ) -> Self {
   177→        Self {
   178→            file: file.into(),
   179→            start_line,
   180→            start_col,
   181→            end_line,
   182→            end_col,
   183→        }
   184→    }
   185→
   186→    /// Create a synthetic "unknown" location for diagnostics that cannot be
   187→    /// tied to a specific source position.
   188→    pub fn unknown() -> Self {
   189→        Self {
   190→            file: String::new(),
   191→            start_line: 0,
   192→            start_col: 0,
   193→            end_line: 0,
   194→            end_col: 0,
   195→        }
   196→    }
   197→
   198→    /// Returns `true` if this location represents an unknown / synthetic
   199→    /// position (all fields zero, no file).
   200→    pub fn is_unknown(&self) -> bool {
   201→        self.file.is_empty() && self.start_line == 0 && self.start_col == 0
   202→    }
   203→}
   204→
   205→impl fmt::Display for DiagnosticSourceLocation {
   206→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   207→        if self.is_unknown() {
   208→            write!(f, "<unknown>")
   209→        } else if self.start_line == self.end_line && self.start_col == self.end_col {
   210→            write!(f, "{}:{}:{}", self.file, self.start_line, self.start_col)
   211→        } else {
   212→            write!(
   213→                f,
   214→                "{}:{}:{}-{}:{}",
   215→                self.file, self.start_line, self.start_col, self.end_line, self.end_col
   216→            )
   217→        }
   218→    }
   219→}

   238→        Self {
   239→            location,
   240→            message: message.into(),
   241→        }
   242→    }
   243→}

   245→// ═══════════════════════════════════════════════════════════════════════════
   246→// Suggestion (structured)
   247→// ═══════════════════════════════════════════════════════════════════════════
   248→
   249→/// A structured suggestion with an optional edit range for automatic fixing.
   250→///
   251→/// Suggestions can be simple text hints (no edit range) or precise code
   252→/// edits that an IDE or tool can apply automatically.
   253→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   254→pub struct Suggestion {
   255→    /// Human-readable description of the suggestion.
   256→    pub message: String,
   257→    /// Optional source location that this suggestion replaces.
   258→    #[serde(skip_serializing_if = "Option::is_none")]
   259→    pub edit_range: Option<DiagnosticSourceLocation>,
   260→    /// Optional replacement text for the edit range.
   261→    #[serde(skip_serializing_if = "Option::is_none")]
   262→    pub replacement: Option<String>,
   263→    /// The applicability of this suggestion.
   264→    #[serde(default = "default_applicability")]
   265→    pub applicability: SuggestionApplicability,
   266→}
   267→
   268→/// How likely a suggestion is to be the right fix.
   269→#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
   270→#[serde(rename_all = "snake_case")]
   271→pub enum SuggestionApplicability {
   272→    /// The suggestion is definitely what the user intended (machine-applicable).
   273→    MachineApplicable,
   274→    /// The suggestion has a very high likelihood of being correct.
   275→    MaybeIncorrect,
   276→    /// The suggestion may or may not be correct — requires human judgment.
   277→    HasPlaceholders,
   278→    /// The suggestion is a rough hint, not a precise fix.
   279→    Unspecified,
   280→}
   281→
   282→fn default_applicability() -> SuggestionApplicability {
   283→    SuggestionApplicability::Unspecified
   284→}
   285→
   286→impl Suggestion {
   287→    /// Create a simple text-only suggestion (no automatic edit).
   288→    pub fn text(message: impl Into<String>) -> Self {
   289→        Self {
   290→            message: message.into(),
   291→            edit_range: None,
   292→            replacement: None,
   293→            applicability: SuggestionApplicability::Unspecified,
   294→        }
   295→    }
   296→
   297→    /// Create a suggestion with an exact edit range and replacement text.
   298→    pub fn edit(
   299→        message: impl Into<String>,
   300→        range: DiagnosticSourceLocation,
   301→        replacement: impl Into<String>,
   302→    ) -> Self {
   303→        Self {
   304→            message: message.into(),
   305→            edit_range: Some(range),
   306→            replacement: Some(replacement.into()),
   307→            applicability: SuggestionApplicability::MachineApplicable,
   308→        }
   309→    }
   310→
   311→    /// Create a machine-applicable suggestion with an edit range.
   312→    pub fn machine_applicable(
   313→        message: impl Into<String>,
   314→        range: DiagnosticSourceLocation,
   315→        replacement: impl Into<String>,
   316→    ) -> Self {
   317→        Self {
   318→            message: message.into(),
   319→            edit_range: Some(range),
   320→            replacement: Some(replacement.into()),
   321→            applicability: SuggestionApplicability::MachineApplicable,
   322→        }
   323→    }
   324→
   325→    /// Create a suggestion with placeholders (needs human review).
   326→    pub fn with_placeholders(message: impl Into<String>) -> Self {
   327→        Self {
   328→            message: message.into(),
   329→            edit_range: None,
   330→            replacement: None,
   331→            applicability: SuggestionApplicability::HasPlaceholders,
   332→        }
   333→    }
   334→}

   492→/// Returns the category prefix for a diagnostic code.
   493→///
   494→/// - `"E"` for errors, `"W"` for warnings, `"I"` for informational.
   495→pub fn code_category(code: &str) -> &'static str {
   496→    if code.starts_with('E') {
   497→        "error"
   498→    } else if code.starts_with('W') {
   499→        "warning"
   500→    } else if code.starts_with('I') {
   501→        "info"
   502→    } else {
   503→        "unknown"
   504→    }
   505→}
   506→
   507→/// Returns the sub-category range for a diagnostic code.
   508→///
   509→/// - `"compilation"` for E001–E030
   510→/// - `"codegen"` for E031–E040
   511→/// - `"verification"` for E041–E050
   512→/// - `"warning"` for W001–W010
   513→/// - `"informational"` for I001–I005
   514→pub fn code_subcategory(code: &str) -> &'static str {
   515→    if let Ok(num) = code[1..].parse::<u32>() {
   516→        match code.chars().next() {
   517→            Some('E') => {
   518→                if num <= 30 {
   519→                    "compilation"
   520→                } else if num <= 40 {
   521→                    "codegen"
   522→                } else if num <= 50 {
   523→                    "verification"
   524→                } else {
   525→                    "unknown"
   526→                }
   527→            }
   528→            Some('W') => "warning",
   529→            Some('I') => "informational",
   530→            _ => "unknown",
   531→        }
   532→    } else {
   533→        "unknown"
   534→    }
   535→}

   538→// VumaDiagnostic
   539→// ═══════════════════════════════════════════════════════════════════════════
   540→
   541→/// A machine-readable diagnostic with structured fields for LLM consumption.
   542→///
   543→/// This is the primary output type for JSON diagnostics. Every field is
   544→/// serializable and carries enough information for an LLM or IDE to
   545→/// understand, categorize, and potentially auto-fix the issue.
   546→///
   547→/// # Error Chaining
   548→///
   549→/// Diagnostics support causal chains via the [`chain`](VumaDiagnostic::chain)
   550→/// method. A chain represents the cause of a diagnostic: for example,
   551→/// "E003: Type mismatch" may be caused by "E023: Unknown type 'int'".
   552→///
   553→/// # Suggestions
   554→///
   555→/// Each diagnostic can carry zero or more [`Suggestion`] values, which may
   556→/// include precise edit ranges for automatic code fixes.
   557→#[derive(Debug, Clone, Serialize, Deserialize)]
   558→pub struct VumaDiagnostic {
   559→    /// Structured diagnostic code, e.g. `"E001"`, `"W001"`.
   560→    pub code: String,
   561→    /// Severity level.
   562→    pub severity: DiagnosticSeverity,
   563→    /// Human-readable message (suitable for display).
   564→    pub message: String,
   565→    /// Compiler stage that produced this diagnostic
   566→    /// (`"parser"`, `"scg"`, `"ir"`, `"codegen"`, etc.).
   567→    pub source: String,
   568→    /// Precise source location.
   569→    pub location: DiagnosticSourceLocation,
   570→    /// Related source locations with explanatory messages.
   571→    pub related: Vec<RelatedInfo>,
   572→    /// Quick-fix suggestions (structured with optional edit ranges).
   573→    #[serde(default)]
   574→    pub suggestions: Vec<Suggestion>,
   575→    /// Causal chain: list of diagnostics that caused this one.
   576→    /// The first element is the immediate cause, the last is the root cause.
   577→    #[serde(default)]
   578→    pub chain: Vec<Box<VumaDiagnostic>>,
   579→    /// Backward-compatible plain-text suggestions (stored alongside
   580→    /// structured suggestions for JSON serialization compatibility).
   581→    #[serde(default, skip_serializing_if = "Vec::is_empty")]
   582→    pub legacy_suggestions: Vec<String>,
   583→}
   584→
   585→impl VumaDiagnostic {
   586→    /// Create a new diagnostic with all fields.
   587→    pub fn new(
   588→        code: impl Into<String>,
   589→        severity: DiagnosticSeverity,
   590→        message: impl Into<String>,
   591→        source: impl Into<String>,
   592→        location: DiagnosticSourceLocation,
   593→    ) -> Self {
   594→        Self {
   595→            code: code.into(),
   596→            severity,
   597→            message: message.into(),
   598→            source: source.into(),
   599→            location,
   600→            related: Vec::new(),
   601→            suggestions: Vec::new(),
   602→            chain: Vec::new(),
   603→            legacy_suggestions: Vec::new(),
   604→        }
   605→    }
   606→
   607→    /// Add a related-info entry.
   608→    pub fn with_related(mut self, info: RelatedInfo) -> Self {
   609→        self.related.push(info);
   610→        self
   611→    }
   612→
   613→    /// Add a quick-fix suggestion (plain text, backward-compatible).
   614→    pub fn with_suggestion(mut self, suggestion: impl Into<String>) -> Self {
   615→        self.legacy_suggestions.push(suggestion.into());
   616→        self
   617→    }
   618→
   619→    /// Add a structured suggestion with optional edit range.
   620→    pub fn with_structured_suggestion(mut self, suggestion: Suggestion) -> Self {
   621→        self.suggestions.push(suggestion);
   622→        self
   623→    }
   624→
   625→    /// Chain another diagnostic as the cause of this one.
   626→    ///
   627→    /// The chained diagnostic becomes the immediate cause. Subsequent
   628→    /// calls to `chain()` push further causes, building a causal chain
   629→    /// from immediate cause to root cause.
   630→    ///
   631→    /// # Example
   632→    ///
   633→    /// ```
   634→    /// use vuma::diagnostics::*;
   635→    ///
   636→    /// let root = VumaDiagnostic::new(
   637→    ///     "E023", DiagnosticSeverity::Error,
   638→    ///     "Unknown type 'int' — did you mean 'i32'?",
   639→    ///     "parser", DiagnosticSourceLocation::unknown()
   640→    /// );
   641→    /// let diag = VumaDiagnostic::new(
   642→    ///     "E003", DiagnosticSeverity::Error,
   643→    ///     "Type mismatch", "parser", DiagnosticSourceLocation::unknown()
   644→    /// ).chain(root);
   645→    ///
   646→    /// assert_eq!(diag.chain.len(), 1);
   647→    /// assert_eq!(diag.chain[0].code, "E023");
   648→    /// ```
   649→    pub fn chain(mut self, cause: VumaDiagnostic) -> Self {
   650→        self.chain.push(Box::new(cause));
   651→        self
   652→    }
   653→
   654→    /// Returns the root cause of this diagnostic (the last element in the
   655→    /// causal chain), or `None` if there is no chain.
   656→    pub fn root_cause(&self) -> Option<&VumaDiagnostic> {
   657→        self.chain.last().map(|b| b.as_ref())
   658→    }
   659→
   660→    /// Returns the immediate cause of this diagnostic (the first element
   661→    /// in the causal chain), or `None` if there is no chain.
   662→    pub fn immediate_cause(&self) -> Option<&VumaDiagnostic> {
   663→        self.chain.first().map(|b| b.as_ref())
   664→    }
   665→
   666→    /// Returns the full causal chain as a slice of diagnostics.
   667→    pub fn causal_chain(&self) -> &[Box<VumaDiagnostic>] {
   668→        &self.chain
   669→    }
   670→
   671→    /// Returns `true` if this diagnostic has a causal chain.
   672→    pub fn has_chain(&self) -> bool {
   673→        !self.chain.is_empty()
   674→    }
   675→
   676→    /// Returns all suggestions, combining both structured and legacy
   677→    /// plain-text suggestions.
   678→    pub fn all_suggestions(&self) -> Vec<&Suggestion> {
   679→        self.suggestions.iter().collect()
   680→    }
   681→
   682→    /// Returns `true` if this diagnostic has any machine-applicable
   683→    /// suggestions (i.e. suggestions with edit ranges).
   684→    pub fn has_machine_applicable_fixes(&self) -> bool {
   685→        self.suggestions.iter().any(|s| {
   686→            s.edit_range.is_some()
   687→                && s.replacement.is_some()
   688→                && s.applicability == SuggestionApplicability::MachineApplicable
   689→        })
   690→    }

   694→    /// Serialize this diagnostic as a JSON string.

   698→        })

   706→    /// Format this diagnostic as plain text (for logs, no ANSI colors).
   707→    ///
   708→    /// Format: `severity[code]: message (source at location)`
   709→    pub fn to_plain_text(&self) -> String {
   710→        let mut out = format!(
   711→            "{}[{}]: {} ({} at {})",
   712→            self.severity, self.code, self.message, self.source, self.location
   713→        );
   714→        for related in &self.related {
   715→            out.push_str(&format!("\n  note: {} at {}", related.message, related.location));
   716→        }
   717→        for suggestion in &self.suggestions {
   718→            out.push_str(&format!("\n  help: {}", suggestion.message));
   719→            if let Some(ref repl) = suggestion.replacement {
   720→                out.push_str(&format!(" → `{}`", repl));
   721→            }
   722→        }
   723→        for suggestion in &self.legacy_suggestions {
   724→            out.push_str(&format!("\n  help: {}", suggestion));
   725→        }
   726→        for cause in &self.chain {
   727→            out.push_str(&format!(
   728→                "\n  caused by: {}[{}]: {}",
   729→                cause.severity, cause.code, cause.message
   730→            ));
   731→        }
   732→        out
   733→    }
   734→
   735→    /// Format this diagnostic as ANSI-colored terminal output (rich text).
   736→    ///
   737→    /// Uses standard compiler diagnostic coloring:
   738→    /// - Errors in red, warnings in yellow, info in cyan, hints in blue.
   739→    /// - Code and source in bold.
   740→    /// - Suggestions in green.
   741→    /// - Causal chain indented with "caused by" prefix.
   742→    pub fn to_rich_text(&self) -> String {
   743→        let mut out = String::new();
   744→
   745→        // Severity prefix with color
   746→        let (sev_label, sev_code) = match self.severity {
   747→            DiagnosticSeverity::Error => ("\x1b[31merror\x1b[0m", "\x1b[31;1m"),
   748→            DiagnosticSeverity::Warning => ("\x1b[33mwarning\x1b[0m", "\x1b[33;1m"),
   749→            DiagnosticSeverity::Info => ("\x1b[36minfo\x1b[0m", "\x1b[36;1m"),
   750→            DiagnosticSeverity::Hint => ("\x1b[34mhint\x1b[0m", "\x1b[34;1m"),
   751→        };
   752→
   753→        // Main line: severity[code]: message
   754→        out.push_str(&format!(
   755→            "{}[\x1b[1m{}{}\x1b[0m]: {}\n",
   756→            sev_label, sev_code, self.code, self.message
   757→        ));
   758→
   759→        // Location line
   760→        if !self.location.is_unknown() {
   761→            out.push_str(&format!(
   762→                "  \x1b[2m--> {}:{}:{}\x1b[0m\n",
   763→                self.location.file, self.location.start_line, self.location.start_col
   764→            ));
   765→        }
   766→
   767→        // Source indicator
   768→        out.push_str(&format!("  \x1b[2m[{}]\x1b[0m\n", self.source));
   769→
   770→        // Related info
   771→        for related in &self.related {
   772→            out.push_str(&format!(
   773→                "  \x1b[36mnote\x1b[0m: {} \x1b[2m(at {})\x1b[0m\n",
   774→                related.message, related.location
   775→            ));
   776→        }
   777→
   778→        // Structured suggestions
   779→        for suggestion in &self.suggestions {
   780→            out.push_str(&format!(
   781→                "  \x1b[32mhelp\x1b[0m: {}",
   782→                suggestion.message
   783→            ));
   784→            if let Some(ref repl) = suggestion.replacement {
   785→                out.push_str(&format!(" → \x1b[32m`{}`\x1b[0m", repl));
   786→            }
   787→            if suggestion.applicability == SuggestionApplicability::MachineApplicable {
   788→                out.push_str(" \x1b[2m[machine-applicable]\x1b[0m");
   789→            }
   790→            out.push('\n');
   791→        }
   792→
   793→        // Legacy suggestions
   794→        for suggestion in &self.legacy_suggestions {
   795→            out.push_str(&format!("  \x1b[32mhelp\x1b[0m: {}\n", suggestion));
   796→        }
   797→
   798→        // Causal chain (recursively format)
   799→        for (i, cause) in self.chain.iter().enumerate() {
   800→            let indent = "  ".repeat(i + 1);
   801→            let (cause_sev, cause_code) = match cause.severity {
   802→                DiagnosticSeverity::Error => ("\x1b[31merror\x1b[0m", "\x1b[31;1m"),
   803→                DiagnosticSeverity::Warning => ("\x1b[33mwarning\x1b[0m", "\x1b[33;1m"),
   804→                DiagnosticSeverity::Info => ("\x1b[36minfo\x1b[0m", "\x1b[36;1m"),
   805→                DiagnosticSeverity::Hint => ("\x1b[34mhint\x1b[0m", "\x1b[34;1m"),
   806→            };
   807→            out.push_str(&format!(
   808→                "{}\x1b[2mcaused by:\x1b[0m {}[\x1b[1m{}{}\x1b[0m]: {}\n",
   809→                indent, cause_sev, cause_code, cause.code, cause.message
   810→            ));
   811→            if !cause.location.is_unknown() {
   812→                out.push_str(&format!(
   813→                    "{}  \x1b[2m--> {}:{}:{}\x1b[0m\n",
   814→                    indent, cause.location.file, cause.location.start_line, cause.location.start_col
   815→                ));
   816→            }
   817→        }
   818→
   819→        out
   820→    }
   821→
   822→    /// Format this diagnostic as an LSP Diagnostic.
   823→    ///
   824→    /// Returns a JSON `Value` conforming to the LSP Diagnostic structure:
   825→    /// - `range`: 0-based Position range
   826→    /// - `severity`: LSP numeric severity (1=Error, 2=Warning, 3=Info, 4=Hint)
   827→    /// - `code`: the diagnostic code string
   828→    /// - `source`: the compiler stage
   829→    /// - `message`: the human-readable message
   830→    /// - `relatedInformation`: optional related locations
   831→    pub fn to_lsp(&self) -> serde_json::Value {
   832→        use serde_json::json;
   833→
   834→        let start_line = if self.location.start_line > 0 {
   835→            self.location.start_line - 1
   836→        } else {
   837→            0
   838→        };
   839→        let start_col = if self.location.start_col > 0 {
   840→            self.location.start_col - 1
   841→        } else {
   842→            0
   843→        };
   844→        let end_line = if self.location.end_line > 0 {
   845→            self.location.end_line - 1
   846→        } else {
   847→            0
   848→        };
   849→        let end_col = if self.location.end_col > 0 {
   850→            self.location.end_col - 1
   851→        } else {
   852→            0
   853→        };
   854→
   855→        let mut diagnostic = json!({
   856→            "range": {
   857→                "start": { "line": start_line, "character": start_col },
   858→                "end": { "line": end_line, "character": end_col }
   859→            },
   860→            "severity": self.severity.to_lsp_severity(),
   861→            "code": self.code,
   862→            "source": self.source,
   863→            "message": self.message,
   864→        });
   865→
   866→        // Add related information if present
   867→        if !self.related.is_empty() {
   868→            let related: Vec<serde_json::Value> = self.related.iter().map(|r| {
   869→                let r_start_line = if r.location.start_line > 0 { r.location.start_line - 1 } else { 0 };
   870→                let r_start_col = if r.location.start_col > 0 { r.location.start_col - 1 } else { 0 };
   871→                let r_end_line = if r.location.end_line > 0 { r.location.end_line - 1 } else { 0 };
   872→                let r_end_col = if r.location.end_col > 0 { r.location.end_col - 1 } else { 0 };
   873→                json!({
   874→                    "location": {
   875→                        "uri": format!("file://{}", r.location.file),
   876→                        "range": {
   877→                            "start": { "line": r_start_line, "character": r_start_col },
   878→                            "end": { "line": r_end_line, "character": r_end_col }
   879→                        }
   880→                    },
   881→                    "message": r.message
   882→                })
   883→            }).collect();
   884→            diagnostic["relatedInformation"] = json!(related);
   885→        }
   886→
   887→        // Add LSP CodeDescription with href if it's a known code
   888→        let desc = code_description(&self.code);
   889→        if desc != "Unknown diagnostic code" {
   890→            diagnostic["codeDescription"] = json!({
   891→                "href": format!("https://vuma.dev/docs/diagnostics/{}", self.code)
   892→            });
   893→        }
   894→
   895→        // Add tags for warnings (Unnecessary = 1, Deprecated = 2)
   896→        let mut tags: Vec<u32> = Vec::new();
   897→        match self.code.as_str() {
   898→            "W001" | "W004" | "W009" => tags.push(1), // Unnecessary
   899→            "W008" => tags.push(2),                     // Deprecated
   900→            _ => {}
   901→        }
   902→        if !tags.is_empty() {
   903→            diagnostic["tags"] = json!(tags);
   904→        }
   905→
   906→        diagnostic
   907→    }
   908→}

   958→/// Summary statistics for a collection of diagnostics.
   959→///
   960→/// Counts diagnostics by severity and by individual code, providing
   961→/// a quick overview of the health of a compilation.
   962→#[derive(Debug, Clone, Default, Serialize, Deserialize)]
   963→pub struct DiagnosticSummary {
   964→    /// Total number of diagnostics.
   965→    pub total: usize,
   966→    /// Number of errors.
   967→    pub errors: usize,
   968→    /// Number of warnings.
   969→    pub warnings: usize,
   970→    /// Number of informational messages.
   971→    pub infos: usize,
   972→    /// Number of hints.
   973→    pub hints: usize,
   974→    /// Count per diagnostic code (e.g. `"E001"` → 3).
   975→    pub by_code: HashMap<String, usize>,
   976→    /// Count per source (e.g. `"parser"` → 5).
   977→    pub by_source: HashMap<String, usize>,
   978→    /// Count per subcategory (e.g. `"compilation"` → 3).
   979→    pub by_subcategory: HashMap<String, usize>,
   980→}
   981→
   982→impl DiagnosticSummary {
   983→    /// Create a new empty summary.
   984→    pub fn new() -> Self {
   985→        Self::default()
   986→    }
   987→
   988→    /// Build a summary from a slice of diagnostics.
   989→    pub fn from_diagnostics(diagnostics: &[VumaDiagnostic]) -> Self {
   990→        let mut summary = Self::new();
   991→        for diag in diagnostics {
   992→            summary.add(diag);
   993→        }
   994→        summary
   995→    }
   996→
   997→    /// Add a single diagnostic to the summary.
   998→    pub fn add(&mut self, diag: &VumaDiagnostic) {
   999→        self.total += 1;
  1000→        match diag.severity {
  1001→            DiagnosticSeverity::Error => self.errors += 1,
  1002→            DiagnosticSeverity::Warning => self.warnings += 1,
  1003→            DiagnosticSeverity::Info => self.infos += 1,
  1004→            DiagnosticSeverity::Hint => self.hints += 1,
  1005→        }
  1006→        *self.by_code.entry(diag.code.clone()).or_insert(0) += 1;
  1007→        *self.by_source.entry(diag.source.clone()).or_insert(0) += 1;
  1008→        let subcat = code_subcategory(&diag.code).to_string();
  1009→        *self.by_subcategory.entry(subcat).or_insert(0) += 1;
  1010→    }
  1011→
  1012→    /// Returns `true` if there are any errors.
  1013→    pub fn has_errors(&self) -> bool {
  1014→        self.errors > 0
  1015→    }
  1016→
  1017→    /// Returns `true` if there are any warnings.
  1018→    pub fn has_warnings(&self) -> bool {
  1019→        self.warnings > 0
  1020→    }
  1021→
  1022→    /// Returns the count for a specific diagnostic code.
  1023→    pub fn count_for_code(&self, code: &str) -> usize {
  1024→        self.by_code.get(code).copied().unwrap_or(0)
  1025→    }
  1026→
  1027→    /// Returns the count for a specific source.
  1028→    pub fn count_for_source(&self, source: &str) -> usize {
  1029→        self.by_source.get(source).copied().unwrap_or(0)
  1030→    }
  1031→
  1032→    /// Serialize this summary as JSON.
  1033→    pub fn to_json(&self) -> String {
  1034→        serde_json::to_string(self).unwrap_or_else(|_| "{}".to_string())
  1035→    }
  1036→
  1037→    /// Serialize this summary as pretty-printed JSON.
  1038→    pub fn to_json_pretty(&self) -> String {
  1039→        serde_json::to_string_pretty(self).unwrap_or_else(|_| "{}".to_string())
  1040→    }
  1041→}
  1042→
  1043→impl fmt::Display for DiagnosticSummary {
  1044→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
  1045→        write!(
  1046→            f,
  1047→            "diagnostic summary: {} total ({} errors, {} warnings, {} info, {} hints)",
  1048→            self.total, self.errors, self.warnings, self.infos, self.hints
  1049→        )?;
  1050→        if !self.by_code.is_empty() {
  1051→            write!(f, "\n  by code:")?;
  1052→            let mut codes: Vec<_> = self.by_code.iter().collect();
  1053→            codes.sort_by_key(|(k, _)| k.as_str());
  1054→            for (code, count) in codes {
  1055→                write!(f, "\n    {} × {}", code, count)?;
  1056→            }
  1057→        }
  1058→        if !self.by_source.is_empty() {
  1059→            write!(f, "\n  by source:")?;
  1060→            let mut sources: Vec<_> = self.by_source.iter().collect();
  1061→            sources.sort_by_key(|(k, _)| k.as_str());
  1062→            for (source, count) in sources {
  1063→                write!(f, "\n    {} × {}", source, count)?;
  1064→            }
  1065→        }
  1066→        Ok(())
  1067→    }
  1068→}

  1070→// ═══════════════════════════════════════════════════════════════════════════
  1071→// Integration: ParseError → VumaDiagnostic
  1072→// ═══════════════════════════════════════════════════════════════════════════
  1073→
  1074→/// Convert a [`ParseError`] to a [`VumaDiagnostic`].
  1075→///
  1076→/// `source` is the full source text (used to compute line/column from byte
  1077→/// offsets). `file` is the optional file path.
  1078→pub fn from_parse_error(
  1079→    err: &ParseError,
  1080→    source: &str,
  1081→    file: Option<&str>,
  1082→) -> VumaDiagnostic {
  1083→    let code = code_for_parse_error_kind(&err.kind).to_string();
  1084→    let loc = offset_to_location(source, err.span.start, file);
  1085→
  1086→    // Compute end position from span end.
  1087→    let end_loc = if err.span.end > err.span.start && err.span.end <= source.len() {
  1088→        offset_to_location(source, err.span.end, file)
  1089→    } else {
  1090→        loc.clone()
  1091→    };
  1092→
  1093→    let diag_loc = DiagnosticSourceLocation {
  1094→        file: loc.file.clone().unwrap_or_default(),
  1095→        start_line: loc.line as u32 + 1, // convert 0-based to 1-based
  1096→        start_col: loc.column as u32 + 1,
  1097→        end_line: end_loc.line as u32 + 1,
  1098→        end_col: end_loc.column as u32 + 1,
  1099→    };
  1100→
  1101→    let mut diag = VumaDiagnostic::new(
  1102→        code,
  1103→        DiagnosticSeverity::Error,
  1104→        &err.message,
  1105→        "parser",
  1106→        diag_loc,
  1107→    );
  1108→
  1109→    if let Some(ref suggestion) = err.suggestion {
  1110→        diag = diag.with_suggestion(format!("did you mean '{}'?", suggestion));
  1111→    }
  1112→
  1113→    // Store the line text as a hint if available.
  1114→    if let Some(ref line_text) = loc.line_text {
  1115→        diag = diag.with_suggestion(format!("source: {}", line_text.trim()));
  1116→    }
  1117→
  1118→    diag
  1119→}

  1121→/// Convert multiple [`ParseError`]s into [`VumaDiagnostic`]s.
  1122→pub fn from_parse_errors(
  1123→    errors: &[ParseError],
  1124→    source: &str,
  1125→    file: Option<&str>,
  1126→) -> Vec<VumaDiagnostic> {
  1127→    errors.iter().map(|e| from_parse_error(e, source, file)).collect()
  1128→}

  1131→// Integration: VumaError → Vec<VumaDiagnostic>

  1134→/// Convert a [`VumaError`] to one or more [`VumaDiagnostic`]s.
  1135→///
  1136→/// Since `VumaError` does not carry the original source text, the resulting
  1137→/// diagnostics will have [`DiagnosticSourceLocation::unknown()`] for variants
  1138→/// that don't carry location information. For parse errors, prefer
  1139→/// [`from_parse_error`] which can compute precise locations.
  1140→pub fn from_vuma_error(err: &VumaError) -> Vec<VumaDiagnostic> {
  1141→    match err {
  1142→        VumaError::Parse { errors } => {
  1143→            // Parse errors need source text for location computation;
  1144→            // without it, we produce diagnostics with unknown locations.
  1145→            errors
  1146→                .iter()
  1147→                .map(|pe| {
  1148→                    let code = code_for_parse_error_kind(&pe.kind).to_string();
  1149→                    let mut diag = VumaDiagnostic::new(
  1150→                        code,
  1151→                        DiagnosticSeverity::Error,
  1152→                        &pe.message,
  1153→                        "parser",
  1154→                        DiagnosticSourceLocation::unknown(),
  1155→                    );
  1156→                    if let Some(ref suggestion) = pe.suggestion {
  1157→                        diag = diag.with_suggestion(format!("did you mean '{}'?", suggestion));
  1158→                    }
  1159→                    diag
  1160→                })
  1161→                .collect()
  1162→        }
  1163→        VumaError::AstToScg { message } => {
  1164→            vec![VumaDiagnostic::new(
  1165→                "E024",
  1166→                DiagnosticSeverity::Error,
  1167→                message,
  1168→                "scg",
  1169→                DiagnosticSourceLocation::unknown(),
  1170→            )]
  1171→        }
  1172→        VumaError::ScgValidation { errors } => errors
  1173→            .iter()
  1174→            .map(|msg| {
  1175→                VumaDiagnostic::new(
  1176→                    "E041",
  1177→                    DiagnosticSeverity::Error,
  1178→                    msg,
  1179→                    "scg",
  1180→                    DiagnosticSourceLocation::unknown(),
  1181→                )
  1182→            })
  1183→            .collect(),
  1184→        VumaError::ScgToMsg { error } => {
  1185→            vec![VumaDiagnostic::new(
  1186→                "E034",
  1187→                DiagnosticSeverity::Error,
  1188→                error.to_string(),
  1189→                "msg",
  1190→                DiagnosticSourceLocation::unknown(),
  1191→            )]
  1192→        }
  1193→        VumaError::BdInference { node_id, message } => {
  1194→            let mut diag = VumaDiagnostic::new(
  1195→                "E048",
  1196→                DiagnosticSeverity::Error,
  1197→                message,
  1198→                "bd",
  1199→                DiagnosticSourceLocation::unknown(),
  1200→            );
  1201→            if let Some(id) = node_id {
  1202→                diag = diag.with_suggestion(format!("at SCG node {}", id));
  1203→            }
  1204→            vec![diag]
  1205→        }
  1206→        VumaError::Verification { result } => {
  1207→            vec![VumaDiagnostic::new(
  1208→                "E042",
  1209→                DiagnosticSeverity::Error,
  1210→                format!("verification failed: {}", result.overall),
  1211→                "ive",
  1212→                DiagnosticSourceLocation::unknown(),
  1213→            )]
  1214→        }
  1215→        VumaError::Transform { pass_name, errors } => errors
  1216→            .iter()
  1217→            .map(|msg| {
  1218→                VumaDiagnostic::new(
  1219→                    "E041",
  1220→                    DiagnosticSeverity::Error,
  1221→                    msg,
  1222→                    format!("scg-transform:{}", pass_name),
  1223→                    DiagnosticSourceLocation::unknown(),
  1224→                )
  1225→            })
  1226→            .collect(),
  1227→        VumaError::Codegen { error } => {
  1228→            vec![from_codegen_error(error)]
  1229→        }
  1230→        VumaError::RegisterAlloc { message } => {
  1231→            vec![VumaDiagnostic::new(
  1232→                "E032",
  1233→                DiagnosticSeverity::Error,
  1234→                message,
  1235→                "register-alloc",
  1236→                DiagnosticSourceLocation::unknown(),
  1237→            )]
  1238→        }
  1239→        VumaError::Emission { message } => {
  1240→            vec![VumaDiagnostic::new(
  1241→                "E035",
  1242→                DiagnosticSeverity::Error,
  1243→                message,
  1244→                "elf-emission",
  1245→                DiagnosticSourceLocation::unknown(),
  1246→            )]
  1247→        }
  1248→        VumaError::CorInit { message } => {
  1249→            vec![VumaDiagnostic::new(
  1250→                "E024",
  1251→                DiagnosticSeverity::Error,
  1252→                message,
  1253→                "cor-init",
  1254→                DiagnosticSourceLocation::unknown(),
  1255→            )]
  1256→        }
  1257→        VumaError::Multi { errors } => errors.iter().flat_map(from_vuma_error).collect(),
  1258→        VumaError::ModuleResolution { errors } => errors
  1259→            .iter()
  1260→            .flat_map(|e| {
  1261→                vec![VumaDiagnostic::new(
  1262→                    "E024",
  1263→                    DiagnosticSeverity::Error,
  1264→                    e.to_string(),
  1265→                    "module-resolution",
  1266→                    DiagnosticSourceLocation::unknown(),
  1267→                )]
  1268→            })
  1269→            .collect(),
  1270→        VumaError::BackendFallback { failed_backend, fallback_backend, error } => {
  1271→            let mut msg = format!("backend '{}' failed: {}", failed_backend, error);
  1272→            if let Some(fb) = fallback_backend {
  1273→                msg.push_str(&format!(", attempting fallback to '{}'", fb));
  1274→            }
  1275→            vec![VumaDiagnostic::new(
  1276→                "E031",
  1277→                DiagnosticSeverity::Warning,
  1278→                &msg,
  1279→                "backend-fallback",
  1280→                DiagnosticSourceLocation::unknown(),
  1281→            )]
  1282→        }
  1283→        VumaError::PanicCaught { stage, message } => {
  1284→            vec![VumaDiagnostic::new(
  1285→                "E050",
  1286→                DiagnosticSeverity::Error,
  1287→                &format!("internal panic in stage '{}': {}", stage, message),
  1288→                stage,
  1289→                DiagnosticSourceLocation::unknown(),
  1290→            )]
  1291→        }
  1292→    }
  1293→}

  1299→/// Convert a [`CodegenError`] to a [`VumaDiagnostic`].
  1300→pub fn from_codegen_error(err: &CodegenError) -> VumaDiagnostic {
  1301→    let code = code_for_codegen_error(err).to_string();
  1302→    let message = err.to_string();
  1303→
  1304→    let (source, suggestion) = match err {
  1305→        CodegenError::UnknownVariable { name } => (
  1306→            "codegen",
  1307→            Some(Suggestion::edit(
  1308→                format!("declare '{}' before use", name),
  1309→                DiagnosticSourceLocation::unknown(),
  1310→                format!("let {} = …;", name),
  1311→            )),
  1312→        ),
  1313→        CodegenError::RegisterAllocFailed(_) => (
  1314→            "register-alloc",
  1315→            Some(Suggestion::text(
  1316→                "reduce register pressure by splitting live ranges".to_string(),
  1317→            )),
  1318→        ),
  1319→        CodegenError::TranslationError(_) => (
  1320→            "ir",
  1321→            None,
  1322→        ),
  1323→        CodegenError::EncodingError(_) => (
  1324→            "codegen",
  1325→            None,
  1326→        ),
  1327→        CodegenError::ElfError(_) => (
  1328→            "emit",
  1329→            None,
  1330→        ),
  1331→        CodegenError::InvalidInstruction(_) => (
  1332→            "codegen",
  1333→            None,
  1334→        ),
  1335→        CodegenError::WasmSectionNotFound { section } => (
  1336→            "codegen",
  1337→            Some(Suggestion::text(
  1338→                format!("ensure '{}' section is generated", section),
  1339→            )),
  1340→        ),
  1341→    };
  1342→
  1343→    let mut diag = VumaDiagnostic::new(
  1344→        code,
  1345→        DiagnosticSeverity::Error,
  1346→        message,
  1347→        source,
  1348→        DiagnosticSourceLocation::unknown(),
  1349→    );
  1350→
  1351→    if let Some(s) = suggestion {
  1352→        diag = diag.with_structured_suggestion(s);
  1353→    }
  1354→
  1355→    diag
  1356→}

  1359→// Memory safety diagnostics (E041–E050)
  1360→// ═══════════════════════════════════════════════════════════════════════════
  1361→
  1362→/// Convert a [`MemorySafetyViolation`] into a [`VumaDiagnostic`].
  1363→///
  1364→/// Maps each violation to its E041–E050 code and creates a structured
  1365→/// diagnostic with source location and quick-fix suggestions where
  1366→/// applicable.
  1367→pub fn from_memory_safety_violation(violation: &vuma_codegen::MemorySafetyViolation) -> VumaDiagnostic {
  1368→    use vuma_codegen::MemorySafetyViolation;
  1369→
  1370→    let code = violation.code().to_string();
  1371→    let message = violation.description();
  1372→    let severity = DiagnosticSeverity::Error;
  1373→
  1374→    let (source, suggestion): (&str, Option<Suggestion>) = match violation {
  1375→        MemorySafetyViolation::UseAfterFree { allocation_name, .. } => (
  1376→            "memory-safety",
  1377→            Some(Suggestion::text(format!(
  1378→                "ensure '{}' is not accessed after it is freed", allocation_name
  1379→            ))),
  1380→        ),
  1381→        MemorySafetyViolation::DoubleFree { allocation_name, .. } => (
  1382→            "memory-safety",
  1383→            Some(Suggestion::text(format!(
  1384→                "'{}' is freed more than once — remove the duplicate free", allocation_name
  1385→            ))),
  1386→        ),
  1387→        MemorySafetyViolation::MemoryLeak { allocation_name, .. } => (
  1388→            "memory-safety",
  1389→            Some(Suggestion::text(format!(
  1390→                "add a 'free({})' before the function returns", allocation_name
  1391→            ))),
  1392→        ),
  1393→        MemorySafetyViolation::BoundsCheckFailure { array_name, .. } => (
  1394→            "runtime-safety",
  1395→            Some(Suggestion::text(format!(
  1396→                "check the index before accessing '{}'", array_name
  1397→            ))),
  1398→        ),
  1399→        MemorySafetyViolation::NullDereference { pointer_name } => (
  1400→            "memory-safety",
  1401→            Some(Suggestion::text(format!(
  1402→                "add a null check before dereferencing '{}'", pointer_name
  1403→            ))),
  1404→        ),
  1405→        MemorySafetyViolation::DanglingPointer { pointer_name, .. } => (
  1406→            "memory-safety",
  1407→            Some(Suggestion::text(format!(
  1408→                "'{}' references a stack allocation that escapes its scope — allocate on the heap instead",
  1409→                pointer_name
  1410→            ))),
  1411→        ),
  1412→        MemorySafetyViolation::UninitializedRead { variable_name } => (
  1413→            "memory-safety",
  1414→            Some(Suggestion::text(format!(
  1415→                "initialize '{}' before reading it", variable_name
  1416→            ))),
  1417→        ),
  1418→        MemorySafetyViolation::BufferOverflow { buffer_name, .. } => (
  1419→            "runtime-safety",
  1420→            Some(Suggestion::text(format!(
  1421→                "check the offset before writing to '{}'", buffer_name
  1422→            ))),
  1423→        ),
  1424→        MemorySafetyViolation::UseAfterScope { variable_name, .. } => (
  1425→            "memory-safety",
  1426→            Some(Suggestion::text(format!(
  1427→                "'{}' is used after its scope ends — extend the scope or copy the value",
  1428→                variable_name
  1429→            ))),
  1430→        ),
  1431→        MemorySafetyViolation::InvalidFree { pointer_name, reason } => (
  1432→            "memory-safety",
  1433→            Some(Suggestion::text(format!(
  1434→                "'{}' cannot be freed: {} — check the pointer origin", pointer_name, reason
  1435→            ))),
  1436→        ),
  1437→    };
  1438→
  1439→    let mut diag = VumaDiagnostic::new(
  1440→        code,
  1441→        severity,
  1442→        message,
  1443→        source,
  1444→        DiagnosticSourceLocation::unknown(),
  1445→    );
  1446→
  1447→    if let Some(s) = suggestion {
  1448→        diag = diag.with_structured_suggestion(s);
  1449→    }
  1450→
  1451→    diag
  1452→}

  1454→/// Convert all violations from a [`MemorySafetyReport`] into diagnostics.
  1455→pub fn from_memory_safety_report(report: &vuma_codegen::MemorySafetyReport) -> Vec<VumaDiagnostic> {
  1456→    report
  1457→        .violations
  1458→        .iter()
