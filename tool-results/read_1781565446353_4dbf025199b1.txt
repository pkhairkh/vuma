     1→//! # Cross-Backend Consistency Test Suite
     2→//!
     3→//! Compiles the same VUMA programs for all 8 backends and verifies they produce
     4→//! equivalent, structurally valid results.  Each test constructs IR functions
     5→//! directly (bypassing the SCG front-end), runs each backend's
     6→//! `allocate_registers` + `encode_function`, and validates:
     7→//!
     8→//! - Binary output exists and has reasonable size
     9→//! - For Wasm32: the module structure (magic, version, sections)
    10→//! - For ELF backends: the ELF header (magic, class, machine type)
    11→//!
    12→//! # Test Programs
    13→//!
    14→//! | # | Program        | Semantics                                        | Expected result |
    15→//! |---|----------------|--------------------------------------------------|-----------------|
    16→//! | 1 | Simple         | `fn main() -> i64 { return 42; }`               | 42              |
    17→//! | 2 | Arithmetic     | `fn main() -> i64 { return (10+20)*3 - 5; }`    | 85              |
    18→//! | 3 | Memory         | alloc 8B, store 0x42424242, load, return low byte| 66 (0x42)       |
    19→//! | 4 | Function call  | helper() returns 7; main returns helper()        | 7               |
    20→

    33→/// All 8 backend kinds, in a stable order for iteration.
    34→const ALL_BACKENDS: &[BackendKind] = &[
    35→    BackendKind::AArch64,
    36→    BackendKind::RiscV64,
    37→    BackendKind::Wasm32,

    45→/// Human-readable name for a BackendKind (for assertion messages).
    46→fn backend_name(kind: BackendKind) -> &'static str {
    47→    match kind {
    48→        BackendKind::AArch64 => "aarch64",
    49→        BackendKind::RiscV64 => "riscv64",
    50→        BackendKind::Wasm32 => "wasm32",
    51→        BackendKind::LoongArch64 => "loongarch64",
    52→        BackendKind::X86_64 => "x86_64",
    53→        BackendKind::Arm32 => "arm32",
    54→        BackendKind::Mips64 => "mips64",
    55→        BackendKind::PowerPC64 => "ppc64",
    56→    }
    57→}

    59→/// ELF machine type for a BackendKind (0 for non-ELF targets).
    60→fn elf_machine(kind: BackendKind) -> u16 {
    61→    match kind {
    62→        BackendKind::AArch64 => 183,   // EM_AARCH64
    63→        BackendKind::RiscV64 => 243,   // EM_RISCV
    64→        BackendKind::Wasm32 => 0,      // Not ELF
    65→        BackendKind::LoongArch64 => 258, // EM_LOONGARCH
    66→        BackendKind::X86_64 => 62,     // EM_X86_64
    67→        BackendKind::Arm32 => 40,      // EM_ARM
    68→        BackendKind::Mips64 => 8,      // EM_MIPS
    69→        BackendKind::PowerPC64 => 21,  // EM_PPC64
    70→    }
    71→}

    73→/// Expected output format for a BackendKind.
    74→fn expected_output_format(kind: BackendKind) -> OutputFormat {
    75→    match kind {
    76→        BackendKind::Arm32 => OutputFormat::Elf32,
    77→        BackendKind::Wasm32 => OutputFormat::WasmBinary,
    78→        _ => OutputFormat::Elf64,
    79→    }
    80→}

    82→/// Run the full `allocate_registers` + `encode_program` pipeline for a
    83→/// multi-function program and return the final binary.
    84→fn compile_program(
    85→    backend: &dyn Backend,
    86→    functions: &[IRFunction],
    87→    label: &str,
    88→) -> Vec<u8> {
    89→    let mut allocated_functions = Vec::new();
    90→    for func in functions {
    91→        let allocated = backend
    92→            .allocate_registers(func)
    93→            .unwrap_or_else(|e| {
    94→                panic!(
    95→                    "{}: allocate_registers failed for {} / {}: {}",
    96→                    backend.name(),
    97→                    label,
    98→                    func.name,
    99→                    e
   100→                )
   101→            });
   102→        allocated_functions.push(allocated);
   103→    }
   104→
   105→    let total_code_size: usize = allocated_functions
   106→        .iter()
   107→        .map(|f| f.code_size)
   108→        .sum();
   109→
   110→    let program = AllocatedProgram {
   111→        functions: allocated_functions,
   112→        total_code_size,
   113→        total_data_size: 0,
   114→    };
   115→
   116→    backend
   117→        .encode_program(&program)
   118→        .unwrap_or_else(|e| {
   119→            panic!(
   120→                "{}: encode_program failed for {}: {}",
   121→                backend.name(),
   122→                label,
   123→                e
   124→            )
   125→        })
   126→}

   128→/// Validate the ELF header of a compiled binary for the given backend.
   129→fn validate_elf_header(bytes: &[u8], kind: BackendKind) {
   130→    let name = backend_name(kind);
   131→
   132→    // ELF header must be at least 52 bytes (ELF32) or 64 bytes (ELF64).
   133→    let min_header = match expected_output_format(kind) {
   134→        OutputFormat::Elf32 => 52,
   135→        OutputFormat::Elf64 => 64,
   136→        _ => panic!("validate_elf_header called for non-ELF backend {}", name),
   137→    };
   138→    assert!(
   139→        bytes.len() >= min_header,
   140→        "{}: ELF binary too short ({} bytes, need at least {})",
   141→        name,
   142→        bytes.len(),
   143→        min_header
   144→    );
   145→
   146→    // Magic bytes: 0x7f 'E' 'L' 'F'
   147→    assert_eq!(
   148→        &bytes[0..4],
   149→        &[0x7f, b'E', b'L', b'F'],
   150→        "{}: ELF magic bytes incorrect",
   151→        name
   152→    );
   153→
   154→    // ELF class
   155→    let expected_class = match expected_output_format(kind) {
   156→        OutputFormat::Elf32 => 1u8, // ELFCLASS32
   157→        OutputFormat::Elf64 => 2u8, // ELFCLASS64
   158→        _ => unreachable!(),
   159→    };
   160→    assert_eq!(
   161→        bytes[4], expected_class,
   162→        "{}: ELF class should be {}",
   163→        name, expected_class
   164→    );
   165→
   166→    // ELF version must be EV_CURRENT (1)
   167→    assert_eq!(bytes[6], 1, "{}: ELF version should be EV_CURRENT (1)", name);
   168→
   169→    // Machine type at offset 18..20 — read using the ELF's declared byte order.
   170→    // Byte 5 (ei_data): 1 = little-endian, 2 = big-endian.
   171→    let e_machine = if bytes[5] == 2 {
   172→        u16::from_be_bytes([bytes[18], bytes[19]])
   173→    } else {
   174→        u16::from_le_bytes([bytes[18], bytes[19]])
   175→    };
   176→    assert_eq!(
   177→        e_machine,
   178→        elf_machine(kind),
   179→        "{}: ELF machine type should be {} (got {})",
   180→        name,
   181→        elf_machine(kind),
   182→        e_machine
   183→    );
   184→}

   186→/// Validate the Wasm module structure of a compiled binary.
   187→fn validate_wasm_module(bytes: &[u8]) {
   188→    // Must have at least 8 bytes (magic + version)
   189→    assert!(
   190→        bytes.len() >= 8,
   191→        "wasm32: binary too short ({} bytes, need at least 8)",
   192→        bytes.len()
   193→    );
   194→
   195→    // Magic: 0x00 0x61 0x73 0x6D ("\0asm")
   196→    assert_eq!(
   197→        &bytes[0..4],
   198→        &[0x00, 0x61, 0x73, 0x6D],
   199→        "wasm32: magic bytes should be \\0asm"
   200→    );
   201→
   202→    // Version: 0x01 0x00 0x00 0x00 (version 1)
   203→    assert_eq!(
   204→        &bytes[4..8],
   205→        &[0x01, 0x00, 0x00, 0x00],
   206→        "wasm32: version should be 1"
   207→    );
   208→
   209→    // Verify at least some sections exist after the header
   210→    assert!(
   211→        bytes.len() > 8,
   212→        "wasm32: module should have content after header"
   213→    );
   214→
   215→    // Walk sections and verify they appear in ascending ID order
   216→    let mut offset = 8usize;
   217→    let mut last_section_id: Option<u8> = None;
   218→    while offset < bytes.len() {
   219→        let section_id = bytes[offset];
   220→        offset += 1;
   221→
   222→        // Decode LEB128 size
   223→        let mut size: usize = 0;
   224→        let mut shift: usize = 0;
   225→        loop {
   226→            assert!(offset < bytes.len(), "wasm32: truncated section size");
   227→            let byte = bytes[offset];
   228→            offset += 1;
   229→            size |= ((byte & 0x7F) as usize) << shift;
   230→            shift += 7;
   231→            if byte & 0x80 == 0 {
   232→                break;
   233→            }
   234→        }
   235→
   236→        // Sections must appear in order of ascending ID (except custom sections, ID 0)
   237→        if section_id != 0 {
   238→            if let Some(prev) = last_section_id {
   239→                assert!(
   240→                    section_id > prev,
   241→                    "wasm32: sections out of order ({} after {})",
   242→                    section_id,
   243→                    prev
   244→                );
   245→            }
   246→            last_section_id = Some(section_id);
   247→        }
   248→
   249→        offset += size;
   250→    }
   251→}
   252→
   253→/// Validate a binary produced by any backend: check format-specific structure
   254→/// and that the output has a reasonable minimum size.
   255→fn validate_binary(bytes: &[u8], kind: BackendKind, min_size: usize) {
   256→    let name = backend_name(kind);
   257→
   258→    // Reasonable minimum size (at least a few instructions)
   259→    assert!(
   260→        bytes.len() >= min_size,
   261→        "{}: binary too small ({} bytes, expected at least {})",
   262→        name,
   263→        bytes.len(),
   264→        min_size
   265→    );
   266→
   267→    match expected_output_format(kind) {
   268→        OutputFormat::Elf32 | OutputFormat::Elf64 => validate_elf_header(bytes, kind),
   269→        OutputFormat::WasmBinary => validate_wasm_module(bytes),
   270→        OutputFormat::RawBinary => {
   271→            // No structural validation for raw binaries
   272→        }
   273→    }
   274→}

   280→/// Program 1: Simple — a function that returns 42.
   281→///
   282→/// ```text
   283→/// fn main() -> i64 { return 42; }
   284→/// ```
   285→fn make_simple_function() -> IRFunction {
   286→    let mut func = IRFunction::new("main");
   287→    func.result_types.push(IRType::I64);
   288→    func.results.push(IRValue::Register(0));
   289→    func.vregs.insert(0, VirtualRegister::new(0, Some("ret_val".to_string())));
   290→
   291→    func.current_block().terminator = IRTerminator::Return(vec![IRValue::Immediate(42)]);
   292→    func
   293→}

   295→/// Program 2: Arithmetic — computes (10 + 20) * 3 - 5 = 85.
   296→///
   297→/// ```text
   298→/// fn main() -> i64 {
   299→///     let a = 10 + 20;   // 30
   300→///     let b = a * 3;     // 90
   301→///     let c = b - 5;     // 85
   302→///     return c;
   303→/// }
   304→/// ```
   305→fn make_arithmetic_function() -> IRFunction {
   306→    let mut func = IRFunction::new("main");
   307→    func.result_types.push(IRType::I64);
   308→    func.results.push(IRValue::Register(2));
   309→    func.vregs.insert(0, VirtualRegister::new(0, Some("a".to_string())));
   310→    func.vregs.insert(1, VirtualRegister::new(1, Some("b".to_string())));
   311→    func.vregs.insert(2, VirtualRegister::new(2, Some("c".to_string())));
   312→
   313→    let block = func.current_block();
   314→
   315→    // a = 10 + 20
   316→    block.push(IRInstr::Add {
   317→        dst: IRValue::Register(0),
   318→        lhs: IRValue::Immediate(10),
   319→        rhs: IRValue::Immediate(20),
   320→        ty: Some(IRType::I64),
   321→    });
   322→
   323→    // b = a * 3
   324→    block.push(IRInstr::BinOp {
   325→        op: BinOpKind::Mul,
   326→        dst: IRValue::Register(1),
   327→        lhs: IRValue::Register(0),
   328→        rhs: IRValue::Immediate(3),
   329→        ty: Some(IRType::I64),
   330→    });
   331→
   332→    // c = b - 5
   333→    block.push(IRInstr::BinOp {
   334→        op: BinOpKind::Sub,
   335→        dst: IRValue::Register(2),
   336→        lhs: IRValue::Register(1),
   337→        rhs: IRValue::Immediate(5),
   338→        ty: Some(IRType::I64),
   339→    });
   340→
   341→    block.terminator = IRTerminator::Return(vec![IRValue::Register(2)]);
   342→    func
   343→}

   345→/// Program 3: Memory — allocates 8 bytes, writes 0x42424242, reads it back,
   346→/// returns the low byte (0x42 = 66).
   347→///
   348→/// ```text
   349→/// fn main() -> i64 {
   350→///     let ptr = alloc 8;
   351→///     store 0x42424242 at ptr;
   352→///     let val = load ptr as i64;
   353→///     let byte = val & 0xFF;
   354→///     return byte;          // 0x42 = 66
   355→/// }
   356→/// ```
   357→fn make_memory_function() -> IRFunction {
   358→    let mut func = IRFunction::new("main");
   359→    func.result_types.push(IRType::I64);
   360→    func.results.push(IRValue::Register(2));
   361→    func.vregs.insert(0, VirtualRegister::new(0, Some("ptr".to_string())));
   362→    func.vregs.insert(1, VirtualRegister::new(1, Some("val".to_string())));
   363→    func.vregs.insert(2, VirtualRegister::new(2, Some("byte".to_string())));
   364→
   365→    let block = func.current_block();
   366→
   367→    // ptr = alloc 8
   368→    block.push(IRInstr::Alloc {
   369→        dst: IRValue::Register(0),
   370→        size: 8,
   371→    });
   372→
   373→    // store 0x42424242 at ptr + 0
   374→    block.push(IRInstr::Store {
   375→        value: IRValue::Immediate(0x42424242),
   376→        addr: IRValue::Register(0),
   377→        offset: 0,
   378→        ty: IRType::I64,
   379→    });
   380→
   381→    // val = load ptr + 0 as i64
   382→    block.push(IRInstr::Load {
   383→        dst: IRValue::Register(1),
   384→        addr: IRValue::Register(0),
   385→        offset: 0,
   386→        ty: IRType::I64,
   387→    });
   388→
   389→    // byte = val & 0xFF
   390→    block.push(IRInstr::BinOp {
   391→        op: BinOpKind::And,
   392→        dst: IRValue::Register(2),
   393→        lhs: IRValue::Register(1),
   394→        rhs: IRValue::Immediate(0xFF),
   395→        ty: Some(IRType::I64),
   396→    });
   397→
   398→    block.terminator = IRTerminator::Return(vec![IRValue::Register(2)]);
   399→    func
   400→}

   402→/// Program 4: Function call — a helper that returns 7, main calls it
   403→/// and returns the result.
   404→///
   405→/// ```text
   406→/// fn helper() -> i64 { return 7; }
   407→/// fn main() -> i64 { return helper(); }
   408→/// ```
   409→fn make_function_call_program() -> Vec<IRFunction> {
   410→    // Helper function: returns 7
   411→    let mut helper = IRFunction::new("helper");
   412→    helper.result_types.push(IRType::I64);
   413→    helper.results.push(IRValue::Register(0));
   414→    helper
   415→        .vregs
   416→        .insert(0, VirtualRegister::new(0, Some("ret".to_string())));
   417→    helper.current_block().terminator = IRTerminator::Return(vec![IRValue::Immediate(7)]);
   418→
   419→    // Main function: calls helper and returns the result
   420→    let mut main_fn = IRFunction::new("main");
   421→    main_fn.result_types.push(IRType::I64);
   422→    main_fn.results.push(IRValue::Register(0));
   423→    main_fn
   424→        .vregs
   425→        .insert(0, VirtualRegister::new(0, Some("result".to_string())));
   426→
   427→    main_fn.current_block().push(IRInstr::Call {
   428→        dst: Some(IRValue::Register(0)),
   429→        func: "helper".to_string(),
   430→        args: vec![],
   431→        is_extern: false,
   432→    });
   433→    main_fn.current_block().terminator = IRTerminator::Return(vec![IRValue::Register(0)]);
   434→
   435→    vec![helper, main_fn]
   436→}

   442→/// Test 1: Simple program — `fn main() -> i64 { return 42; }`
   443→///
   444→/// Validates that all 8 backends can compile a trivial return-constant
   445→/// function and produce structurally valid output.
   446→#[test]
   447→fn test_cross_backend_simple_return() {
   448→    let func = make_simple_function();
   449→
   450→    for &kind in ALL_BACKENDS {
   451→        let name = backend_name(kind);
   452→        let backend = create_backend(kind).expect("backend creation should succeed");
   453→
   454→        // --- allocate_registers + encode_function ---
   455→        let allocated = backend
   456→            .allocate_registers(&func)
   457→            .unwrap_or_else(|e| panic!("{}: allocate_registers failed: {}", name, e));
   458→
   459→        // The allocated function should have at least one block
   460→        assert!(
   461→            !allocated.blocks.is_empty(),
   462→            "{}: allocated function should have at least one block",
   463→            name
   464→        );
   465→
   466→        // The block should contain at least one instruction
   467→        let total_instrs: usize = allocated
   468→            .blocks
   469→            .iter()
   470→            .map(|b| b.instructions.len())
   471→            .sum();
   472→        assert!(
   473→            total_instrs > 0,
   474→            "{}: allocated function should have instructions",
   475→            name
   476→        );
   477→
   478→        // Encode the function
   479→        let code = backend
   480→            .encode_function(&allocated)
   481→            .unwrap_or_else(|e| panic!("{}: encode_function failed: {}", name, e));
   482→
   483→        // Even a trivial function needs at least 4 bytes of machine code
   484→        assert!(
   485→            code.len() >= 4,
   486→            "{}: encoded function too small ({} bytes)",
   487→            name,
   488→            code.len()
   489→        );
   490→
   491→        // --- encode_program (full binary) ---
   492→        let program_bytes = compile_program(&*backend, &[func.clone()], "simple");
   493→        validate_binary(&program_bytes, kind, 16);
   494→    }
   495→}

   600→/// Test 4: Function call — helper returns 7, main returns helper()
   601→///
   602→/// Validates that all 8 backends can compile a multi-function program
   603→/// with an inter-function call and produce structurally valid output.
   604→#[test]
   605→fn test_cross_backend_function_call() {
   606→    let functions = make_function_call_program();
   607→
   608→    for &kind in ALL_BACKENDS {
   609→        let name = backend_name(kind);
   610→        let backend = create_backend(kind).expect("backend creation should succeed");
   611→
   612→        // Allocate registers for each function independently
   613→        let mut allocated_fns = Vec::new();
   614→        for func in &functions {
   615→            let allocated = backend
   616→                .allocate_registers(func)
   617→                .unwrap_or_else(|e| {
   618→                    panic!(
   619→                        "{}: allocate_registers failed for '{}': {}",
   620→                        name,
   621→                        func.name,
   622→                        e
   623→                    )
   624→                });
   625→            allocated_fns.push(allocated);
   626→        }
   627→
   628→        // We should have 2 allocated functions
   629→        assert_eq!(
   630→            allocated_fns.len(),
   631→            2,
   632→            "{}: should have 2 allocated functions",
   633→            name
   634→        );
   635→
   636→        // Encode each function individually
   637→        for alloc_fn in &allocated_fns {
   638→            let code = backend
   639→                .encode_function(alloc_fn)
   640→                .unwrap_or_else(|e| panic!("{}: encode_function failed: {}", name, e));
   641→            assert!(
   642→                code.len() >= 4,
   643→                "{}: encoded function '{}' too small ({} bytes)",
   644→                name,
   645→                alloc_fn.name,
   646→                code.len()
   647→            );
   648→        }
   649→
   650→        // Full program binary (links the two functions together)
   651→        let program_bytes = compile_program(&*backend, &functions, "func_call");
   652→        validate_binary(&program_bytes, kind, 16);
   653→
   654→        // The main function should have a relocation to the helper
   655→        let main_alloc = &allocated_fns[1]; // second function is "main"
   656→        let has_helper_reloc = main_alloc
   657→            .relocations
   658→            .iter()
   659→            .any(|r| r.symbol == "helper");
   660→        assert!(
   661→            has_helper_reloc,
   662→            "{}: main function should have a relocation to 'helper'",
   663→            name
   664→        );
   665→    }
   666→}

   848→/// Test 8: Wasm32-specific module structure validation
   849→///
   850→/// Compiles each program with the Wasm32 backend and performs detailed
   851→/// validation of the Wasm module structure: sections present, order,
   852→/// type section contents, and memory section.
   853→#[test]
   854→fn test_cross_backend_wasm32_module_structure() {
   855→    let simple = make_simple_function();
   856→    let arithmetic = make_arithmetic_function();
   857→    let memory = make_memory_function();
   858→    let func_call = make_function_call_program();
   859→
   860→    let programs: Vec<(&str, Vec<IRFunction>)> = vec![
   861→        ("simple", vec![simple]),
   862→        ("arithmetic", vec![arithmetic]),
   863→        ("memory", vec![memory]),
   864→        ("func_call", func_call),
   865→    ];
   866→
   867→    let backend = create_backend(BackendKind::Wasm32).expect("Wasm32 backend creation");
   868→
   869→    for (label, functions) in &programs {
   870→        let bytes = compile_program(&*backend, functions, label);
   871→
   872→        // Basic Wasm structure
   873→        assert!(
   874→            bytes.len() >= 8,
   875→            "wasm32/{}: module too short ({} bytes)",
   876→            label,
   877→            bytes.len()
   878→        );
   879→        assert_eq!(
   880→            &bytes[0..4],
   881→            &[0x00, 0x61, 0x73, 0x6D],
   882→            "wasm32/{}: magic bytes incorrect",
   883→            label
   884→        );
   885→        assert_eq!(
   886→            &bytes[4..8],
   887→            &[0x01, 0x00, 0x00, 0x00],
   888→            "wasm32/{}: version incorrect",
   889→            label
   890→        );
   891→
   892→        // Parse sections and verify presence of required sections
   893→        let mut found_type_section = false;
   894→        let mut found_function_section = false;
   895→        let mut found_memory_section = false;
   896→        let mut found_code_section = false;
   897→
   898→        let mut offset = 8usize;
   899→        while offset < bytes.len() {
   900→            let section_id = bytes[offset];
   901→            offset += 1;
   902→
   903→            // Decode LEB128 size
   904→            let mut size: usize = 0;
   905→            let mut shift: usize = 0;
   906→            loop {
   907→                assert!(offset < bytes.len(), "wasm32/{}: truncated section", label);
   908→                let byte = bytes[offset];
   909→                offset += 1;
   910→                size |= ((byte & 0x7F) as usize) << shift;
   911→                shift += 7;
   912→                if byte & 0x80 == 0 {
   913→                    break;
   914→                }
   915→            }
   916→
   917→            match section_id {
   918→                1 => found_type_section = true,
   919→                3 => found_function_section = true,
   920→                5 => found_memory_section = true,
   921→                10 => found_code_section = true,
   922→                _ => {}
   923→            }
   924→
   925→            offset += size;
   926→        }
   927→
   928→        assert!(
   929→            found_type_section,
   930→            "wasm32/{}: missing type section (ID 1)",
   931→            label
   932→        );
   933→        assert!(
   934→            found_function_section,
   935→            "wasm32/{}: missing function section (ID 3)",
   936→            label
   937→        );
   938→        assert!(
   939→            found_memory_section,
   940→            "wasm32/{}: missing memory section (ID 5)",
   941→            label
   942→        );
   943→        assert!(
   944→            found_code_section,
   945→            "wasm32/{}: missing code section (ID 10)",
   946→            label
   947→        );
   948→    }
   949→}

   951→/// Test 9: ELF-specific header validation for all ELF backends
   952→///
   953→/// Compiles each program with every ELF-producing backend and verifies
   954→/// the ELF header fields (magic, class, endianness, machine type, version).
   955→#[test]
   956→fn test_cross_backend_elf_header_validation() {
   957→    let simple = make_simple_function();
   958→    let arithmetic = make_arithmetic_function();
   959→    let memory = make_memory_function();
   960→    let func_call = make_function_call_program();
   961→
   962→    let programs: Vec<(&str, Vec<IRFunction>)> = vec![
   963→        ("simple", vec![simple]),
   964→        ("arithmetic", vec![arithmetic]),
   965→        ("memory", vec![memory]),
   966→        ("func_call", func_call),
   967→    ];
   968→
   969→    for &kind in ALL_BACKENDS {
   970→        let fmt = expected_output_format(kind);
   971→        if fmt != OutputFormat::Elf32 && fmt != OutputFormat::Elf64 {
   972→            continue; // Skip non-ELF backends
   973→        }
   974→
   975→        let name = backend_name(kind);
   976→        let backend = create_backend(kind).expect("backend creation should succeed");
   977→
   978→        for (label, functions) in &programs {
   979→            let bytes = compile_program(&*backend, functions, label);
   980→
   981→            // Minimum ELF header size
   982→            let min_hdr = if fmt == OutputFormat::Elf32 { 52 } else { 64 };
   983→            assert!(
   984→                bytes.len() >= min_hdr,
   985→                "{}/{}: ELF binary too short ({} bytes)",
   986→                name,
   987→                label,
   988→                bytes.len()
   989→            );
   990→
   991→            // Magic
   992→            assert_eq!(
   993→                &bytes[0..4],
   994→                &[0x7f, b'E', b'L', b'F'],
   995→                "{}/{}: bad ELF magic",
   996→                name,
   997→                label
   998→            );
   999→
  1000→            // Class
  1001→            let expected_class = if fmt == OutputFormat::Elf32 { 1u8 } else { 2u8 };
  1002→            assert_eq!(
  1003→                bytes[4], expected_class,
  1004→                "{}/{}: ELF class mismatch",
  1005→                name,
  1006→                label
  1007→            );
  1008→
  1009→            // Endianness
  1010→            let backend_obj = create_backend(kind).unwrap();
  1011→            let endian = backend_obj.target_info().endianness();
  1012→            let expected_data = match endian {
  1013→                vuma_codegen::backend::Endianness::Little => 1u8, // ELFDATA2LSB
  1014→                vuma_codegen::backend::Endianness::Big => 2u8,    // ELFDATA2MSB
  1015→                vuma_codegen::backend::Endianness::Bi => 2u8,     // Bi-endian defaults big
  1016→            };
  1017→            assert_eq!(
  1018→                bytes[5], expected_data,
  1019→                "{}/{}: ELF data encoding mismatch",
  1020→                name,
  1021→                label
  1022→            );
  1023→
  1024→            // Version
  1025→            assert_eq!(
  1026→                bytes[6], 1,
  1027→                "{}/{}: ELF version should be EV_CURRENT",
  1028→                name,
  1029→                label
  1030→            );
  1031→
  1032→            // Machine type
  1033→            let e_machine = u16::from_le_bytes([bytes[18], bytes[19]]);
  1034→            // For big-endian ELF files the header is still encoded in the
  1035→            // target's endianness, so we may need to read it as BE.
  1036→            let e_machine_val = if expected_data == 2u8 {
  1037→                u16::from_be_bytes([bytes[18], bytes[19]])
  1038→            } else {
  1039→                e_machine
  1040→            };
  1041→            assert_eq!(
  1042→                e_machine_val,
  1043→                elf_machine(kind),
  1044→                "{}/{}: machine type mismatch (expected {}, got {} / {} LE)",
  1045→                name,
  1046→                label,
  1047→                elf_machine(kind),
  1048→                e_machine,
  1049→                e_machine_val
  1050→            );
  1051→        }
  1052→    }
  1053→}
