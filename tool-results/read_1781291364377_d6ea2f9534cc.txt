     1→//! # ARM64 Code Emission
     2→//!
     3→//! Lowers IR to ARM64 machine code and produces ELF binaries or raw binaries
     4→//! suitable for the Raspberry Pi 5 (Cortex-A76, ARMv8.2-A).
     5→//!
     6→//! ## Pipeline
     7→//!
     8→//! 1. **IR → ARM64 Instructions**: Each IR instruction is pattern-matched and
     9→//!    lowered to one or more [`Instruction`]
    10→//!    values, with virtual registers replaced by physical registers from the
    11→//!    register allocator.
    12→//! 2. **Emit function**: Each IR function is emitted as a sequence of 32-bit
    13→//!    ARM64 code words.
    14→//! 3. **Emit program**: All functions are collected, data sections are laid
    15→//!    out, and a minimal ELF64 binary is produced.
    16→//!
    17→//! ## Output Formats
    18→//!
    19→//! - **ELF**: Full ELF64 binary with program headers, section headers, symbol
    20→//!   table, and string table.  Suitable for Linux on Pi 5.
    21→//! - **Raw**: Flat binary image for bare-metal Pi 5 (loaded at 0x80000).
    22→//! - **Obj**: Relocatable ELF object file (`ET_REL`) for linking.
    23→//!
    24→//! ## ELF Layout (executable)
    25→//!
    26→//! ```text
    27→//! ┌─────────────────────┐
    28→//! │ ELF Header           │  64 bytes
    29→//! ├─────────────────────┤
    30→//! │ Program Headers      │  2 × 56 bytes (LOAD segments)
    31→//! ├─────────────────────┤
    32→//! │ .text                │  emitted code
    33→//! ├─────────────────────┤
    34→//! │ .rodata              │  read-only data
    35→//! ├─────────────────────┤
    36→//! │ .data                │  initialized data
    37→//! ├─────────────────────┤
    38→//! │ .symtab              │  symbol table entries
    39→//! ├─────────────────────┤
    40→//! │ .strtab              │  symbol string table
    41→//! ├─────────────────────┤
    42→//! │ .shstrtab            │  section header string table
    43→//! ├─────────────────────┤
    44→//! │ Section Headers      │  N × 64 bytes
    45→//! └─────────────────────┘
    46→//! .bss is virtual-only (memsz > filesz in the data LOAD segment).
    47→//! ```
    48→//!
    49→//! ## Relocation Support
    50→//!
    51→//! Inter-function calls (`BL`) are resolved in a fixup pass after all
    52→//! functions have been emitted.  The emitter records the word offset of each
    53→//! `BL` instruction and the target function name; once function addresses
    54→//! are known, the branch offsets are patched into the encoded instructions.

   311→/// Output format for the code emitter.

   314→    /// Full ELF64 executable (Linux) or object file.

   316→    /// Flat raw binary image (bare-metal).
   317→    Raw,
   318→    /// Relocatable ELF object file (`.o`).

   324→        match self {
   325→            OutputFormat::ELF => write!(f, "elf"),
   326→            OutputFormat::Raw => write!(f, "raw"),
   327→            OutputFormat::Obj => write!(f, "obj"),

   468→/// A record for a `BL` instruction that needs to be patched with the address
   469→/// of a named function.  After all functions are emitted, these records are
   470→/// resolved to compute the correct branch offset.
   471→#[derive(Debug, Clone)]
   472→struct CallRelocation {
   473→    /// Byte offset within the text section where the BL instruction lives.
   474→    text_byte_offset: u64,
   475→    /// Name of the target function.
   476→    target_func: String,
   477→}
   478→
   479→// ---------------------------------------------------------------------------
   480→// ELF RelaEntry
   481→// ---------------------------------------------------------------------------
   482→
   483→/// An ELF64 relocation entry with an explicit addend (SHT_RELA).
   484→///
   485→/// Each entry specifies where in the section a relocation must be applied,
   486→/// which symbol the relocation references, the type of relocation, and an
   487→/// addend value used in the relocation computation.
   488→#[derive(Debug, Clone)]
   489→pub struct RelaEntry {
   490→    /// Byte offset within the section where the relocation applies.
   491→    pub offset: u64,
   492→    /// Symbol index (upper 32 bits) and relocation type (lower 32 bits),
   493→    /// packed as: `(sym_idx << 32) | r_type`.
   494→    pub info: u64,
   495→    /// Addend value used in the relocation computation.
   496→    pub addend: i64,
   497→}
   498→
   499→impl RelaEntry {
   500→    /// Create a new relocation entry.
   501→    pub fn new(offset: u64, sym_idx: u32, r_type: u32, addend: i64) -> Self {
   502→        Self {
   503→            offset,
   504→            info: ((sym_idx as u64) << 32) | (r_type as u64),
   505→            addend,
   506→        }
   507→    }
   508→
   509→    /// Extract the symbol index from `info`.
   510→    pub fn sym_idx(&self) -> u32 {
   511→        (self.info >> 32) as u32
   512→    }
   513→
   514→    /// Extract the relocation type from `info`.
   515→    pub fn r_type(&self) -> u32 {
   516→        (self.info & 0xFFFFFFFF) as u32
   517→    }
   518→
   519→    /// Serialize the entry to 24 bytes (little-endian).
   520→    pub fn to_bytes(&self) -> [u8; 24] {
   521→        let mut buf = [0u8; 24];
   522→        buf[0..8].copy_from_slice(&self.offset.to_le_bytes());
   523→        buf[8..16].copy_from_slice(&self.info.to_le_bytes());
   524→        buf[16..24].copy_from_slice(&self.addend.to_le_bytes());
   525→        buf
   526→    }
   527→}

   533→/// The ARM64 code emitter.
   534→///
   535→/// Holds state for the current emission context: the register allocator,
   536→/// accumulated code, and fixup records for branch targets.
   537→pub struct Emitter {
   538→    /// Register allocator used during emission.
   539→    reg_alloc: RegAllocator,
   540→    /// Accumulated machine code for the current function (32-bit words).
   541→    code: Vec<u32>,
   542→    /// Fixup records for intra-function branches: (word index, target label name).
   543→    fixups: Vec<(usize, String)>,
   544→    /// Map from label name to code offset (in words) within the current function.
   545→    label_offsets: HashMap<String, usize>,
   546→    /// Inter-function call relocations for the current function.
   547→    call_relocs: Vec<CallRelocation>,
   548→    /// Relocation entries using the `RelocationEntry` infrastructure.
   549→    ///
   550→    /// Each entry records a byte offset within the function's encoded output
   551→    /// where a symbolic reference must be resolved, the target symbol name,
   552→    /// and the ISA-specific relocation type (e.g., `"R_AARCH64_CALL26"`).
   553→    relocations: Vec<RelocationEntry>,
   554→    /// Name of the function currently being emitted.
   555→    current_func_name: String,
   556→    /// Byte offset of the current function within the text section (set externally).
   557→    func_text_offset: u64,
   558→    /// Computed stack frame size (in bytes) for the current function.
   559→    frame_size: u16,
   560→}
   561→
   562→impl Emitter {
   563→    /// Create a new emitter with a fresh register allocator.
   564→    pub fn new() -> Self {
   565→        Self {
   566→            reg_alloc: RegAllocator::new(),
   567→            code: Vec::new(),
   568→            fixups: Vec::new(),
   569→            label_offsets: HashMap::new(),
   570→            call_relocs: Vec::new(),
   571→            relocations: Vec::new(),
   572→            current_func_name: String::new(),
   573→            func_text_offset: 0,
   574→            frame_size: 0,
   575→        }
   576→    }

   581→
   582→    /// Emit a single IR function to ARM64 machine code.
   583→    ///
   584→    /// Returns a vector of 32-bit ARM64 instruction words.
   585→    pub fn emit_function(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   586→        self.code.clear();
   587→        self.fixups.clear();
   588→        self.label_offsets.clear();
   589→        self.call_relocs.clear();
   590→        self.relocations.clear();
   591→        self.current_func_name = func.name.clone();
   592→        self.reg_alloc.reset();
   593→
   594→        // Allocate registers for parameters (AAPCS64: X0–X7).
   595→        for (i, param) in func.params.iter().enumerate() {
   596→            if let IRValue::Register(vreg_id) = param {
   597→                if i < 8 {
   598→                    let _ = self.reg_alloc.allocate(*vreg_id);
   599→                }
   600→            }
   601→        }
   602→
   603→        // Emit prologue: SUB SP, SP, #frame_size+16 (space for saved FP/LR + locals)
   604→        // Then STP X29, X30, [SP, #frame_size] (save FP/LR at top of allocated frame)
   605→        let aligned_stack = compute_frame_size(func);
   606→        self.frame_size = aligned_stack;
   607→        let total_alloc = aligned_stack + 16; // 16 bytes for FP+LR pair
   608→        self.emit_instruction(Instruction::SUB {
   609→            rd: Register::SP,
   610→            rn: Register::SP,
   611→            rm: Operand::Imm12(total_alloc),
   612→        })?;
   613→        self.emit_instruction(Instruction::STP {
   614→            rt1: Register::X29,
   615→            rt2: Register::X30,
   616→            rn: Register::SP,
   617→            offset: aligned_stack as i32, // save FP/LR at top of frame
   618→        })?;
   619→
   620→        // Emit each basic block.
   621→        for block in &func.blocks {
   622→            self.label_offsets
   623→                .insert(block.label.clone(), self.code.len());
   624→            self.emit_block(block)?;
   625→        }
   626→
   627→        // Apply fixups — resolve intra-function branch targets.
   628→        self.apply_fixups()?;
   629→
   630→        Ok(self.code.clone())
   631→    }
   632→
   633→    /// Returns the relocation entries recorded during emission of the current
   634→    /// function.
   635→    ///
   636→    /// Each `RelocationEntry` records a byte offset within the function's
   637→    /// encoded output where a symbolic reference must be resolved, the target
   638→    /// symbol name, and the ISA-specific relocation type (e.g.,
   639→    /// `"R_AARCH64_CALL26"`).
   640→    pub fn relocations(&self) -> &[RelocationEntry] {
   641→        &self.relocations
   642→    }
   643→
   644→    /// Emit instructions for a single IR basic block.
   645→    fn emit_block(&mut self, block: &IRBlock) -> Result<()> {
   646→        for instr in &block.instructions {
   647→            self.emit_ir_instr(instr)?;
   648→        }
   649→        self.emit_terminator(&block.terminator)?;
   650→        Ok(())
   651→    }

   654→    // IR → ARM64 instruction lowering
   655→    // -----------------------------------------------------------------------
   656→
   657→    /// Lower a single IR instruction to ARM64 instructions.
   658→    fn emit_ir_instr(&mut self, instr: &IRInstr) -> Result<()> {
   659→        match instr {
   660→            IRInstr::Load { dst, addr, offset, ty } => {
   661→                let rt = self.resolve_reg(dst)?;
   662→                let rn = self.resolve_reg(addr)?;
   663→                self.emit_load(rt, rn, *offset, ty)?;
   664→            }
   665→
   666→            IRInstr::Store { value, addr, offset, ty } => {
   667→                let rt = self.resolve_reg(value)?;
   668→                let rn = self.resolve_reg(addr)?;
   669→                self.emit_store(rt, rn, *offset, ty)?;
   670→            }
   671→
   672→            IRInstr::BinOp { op, dst, lhs, rhs, ty } => {
   673→                self.emit_binop(*op, dst, lhs, rhs, ty.as_ref())?;
   674→            }
   675→
   676→            IRInstr::UnaryOp { op, dst, operand, ty: _ } => {
   677→                let rd = self.resolve_reg(dst)?;
   678→                let rn = self.resolve_reg(operand)?;
   679→                match op {
   680→                    UnaryOpKind::Neg => {
   681→                        self.emit_instruction(Instruction::SUB {
   682→                            rd,
   683→                            rn: Register::XZR,
   684→                            rm: Operand::Reg {
   685→                                reg: rn,
   686→                                shift: None,
   687→                            },
   688→                        })?;
   689→                    }
   690→                    UnaryOpKind::Not => {
   691→                        self.emit_load_immediate(Register::X9, -1)?;
   692→                        self.emit_instruction(Instruction::EOR {
   693→                            rd,
   694→                            rn,
   695→                            rm: Register::X9,
   696→                        })?;
   697→                    }
   698→                    UnaryOpKind::Clz => {
   699→                        // CLZ Xd, Xn — count leading zeros (native ARM64 instruction)
   700→                        self.emit_instruction(Instruction::CLZ { rd, rn })?;
   701→                    }
   702→                    UnaryOpKind::Ctz => {
   703→                        // CTZ = RBIT + CLZ: reverse bits then count leading zeros.
   704→                        // Use X9 as scratch if rd == rn (need intermediate result).
   705→                        if rd == rn {
   706→                            self.emit_instruction(Instruction::RBIT {
   707→                                rd: Register::X9,
   708→                                rn,
   709→                            })?;
   710→                            self.emit_instruction(Instruction::CLZ {
   711→                                rd,
   712→                                rn: Register::X9,
   713→                            })?;
   714→                        } else {
   715→                            self.emit_instruction(Instruction::RBIT { rd, rn })?;
   716→                            self.emit_instruction(Instruction::CLZ { rd, rn: rd })?;
   717→                        }
   718→                    }
   719→                    UnaryOpKind::Popcnt => {
   720→                        // POPCNT via FMOV+CNT+ADDV+UMOV sequence:
   721→                        // FMOV D8, Xn        — move GPR value to SIMD register (V8 is caller-saved)
   722→                        // CNT V8.8B, V8.8B   — count bits per byte
   723→                        // ADDV B8, V8.8B     — horizontal sum of byte counts
   724→                        // UMOV Xd, V8.B[0]   — move result back to GPR (zero-extends to 64 bits)
   725→                        const SIMD_SCRATCH: u8 = 8; // V8 is caller-saved in AAPCS64
   726→                        self.emit_instruction(Instruction::FMOV_DX {
   727→                            vd: SIMD_SCRATCH,
   728→                            rn,
   729→                        })?;
   730→                        self.emit_instruction(Instruction::CNT {
   731→                            vd: SIMD_SCRATCH,
   732→                            vn: SIMD_SCRATCH,
   733→                        })?;
   734→                        self.emit_instruction(Instruction::ADDV {
   735→                            vd: SIMD_SCRATCH,
   736→                            vn: SIMD_SCRATCH,
   737→                        })?;
   738→                        self.emit_instruction(Instruction::UMOV {
   739→                            rd,
   740→                            vn: SIMD_SCRATCH,
   741→                        })?;
   742→                    }
   743→                }
   744→            }
   745→
   746→            IRInstr::Call {
   747→                dst,
   748→                func: target_name,
   749→                args,
   750→            } => {
   751→                // Move arguments into X0–X7.
   752→                for (i, arg) in args.iter().enumerate() {
   753→                    if i >= 8 {
   754→                        break;
   755→                    }
   756→                    let src = self.resolve_reg(arg)?;
   757→                    let dst_reg = match i {
   758→                        0 => Register::X0,
   759→                        1 => Register::X1,
   760→                        2 => Register::X2,
   761→                        3 => Register::X3,
   762→                        4 => Register::X4,
   763→                        5 => Register::X5,
   764→                        6 => Register::X6,
   765→                        7 => Register::X7,
   766→                        _ => unreachable!(),
   767→                    };
   768→                    if src != dst_reg {
   769→                        self.emit_instruction(Instruction::MOV {
   770→                            rd: dst_reg,
   771→                            rm: src,
   772→                        })?;
   773→                    }
   774→                }
   775→
   776→                // BL — record a relocation for later patching.
   777→                let bl_word_idx = self.code.len();
   778→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
   779→                self.call_relocs.push(CallRelocation {
   780→                    text_byte_offset: bl_byte_offset,
   781→                    target_func: target_name.clone(),
   782→                });
   783→                // Register a RelocationEntry so the linker can patch this BL.
   784→                self.relocations.push(RelocationEntry {
   785→                    offset: bl_byte_offset,
   786→                    symbol: target_name.clone(),
   787→                    reloc_type: "R_AARCH64_CALL26".to_string(),
   788→                });
   789→                self.emit_instruction(Instruction::BL { offset: 0 })?;
   790→
   791→                if let Some(dst_val) = dst {
   792→                    let rd = self.resolve_reg(dst_val)?;
   793→                    if rd != Register::X0 {
   794→                        self.emit_instruction(Instruction::MOV {
   795→                            rd,
   796→                            rm: Register::X0,
   797→                        })?;
   798→                    }
   799→                }
   800→            }
   801→
   802→            IRInstr::Alloc { dst, size } => {
   803→                let rd = self.resolve_reg(dst)?;
   804→                self.emit_instruction(Instruction::MOV {
   805→                    rd,
   806→                    rm: Register::SP,
   807→                })?;
   808→                let aligned = (*size).div_ceil(16) * 16;
   809→                self.emit_instruction(Instruction::SUB {
   810→                    rd: Register::SP,
   811→                    rn: Register::SP,
   812→                    rm: Operand::Imm12(aligned as u16),
   813→                })?;
   814→            }
   815→
   816→            IRInstr::Free { ptr } => {
   817→                let rt = self.resolve_reg(ptr)?;
   818→                // Move ptr to X0 (first argument)
   819→                if rt != Register::X0 {
   820→                    self.emit_instruction(Instruction::MOV {
   821→                        rd: Register::X0,
   822→                        rm: rt,
   823→                    })?;
   824→                }
   825→                // BL __vuma_free
   826→                let bl_word_idx = self.code.len();
   827→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
   828→                self.call_relocs.push(CallRelocation {
   829→                    text_byte_offset: bl_byte_offset,
   830→                    target_func: "__vuma_free".to_string(),
   831→                });
   832→                self.relocations.push(RelocationEntry {
   833→                    offset: bl_byte_offset,
   834→                    symbol: "__vuma_free".to_string(),
   835→                    reloc_type: "R_AARCH64_CALL26".to_string(),
   836→                });
   837→                self.emit_instruction(Instruction::BL { offset: 0 })?;
   838→            }
   839→
   840→            IRInstr::Cast { kind, dst, src } => {
   841→                let rd = self.resolve_reg(dst)?;
   842→                let rn = self.resolve_reg(src)?;
   843→                match kind {
   844→                    CastKind::ZExt => {
   845→                        // Zero-extend 32-bit to 64-bit using UBFM
   846→                        self.emit_instruction(Instruction::UBFM {
   847→                            rd,
   848→                            rn,
   849→                            immr: 0,
   850→                            imms: 31,
   851→                        })?;
   852→                    }
   853→                    CastKind::SExt => {
   854→                        // Sign-extend 32-bit to 64-bit using SBFM
   855→                        self.emit_instruction(Instruction::SBFM {
   856→                            rd,
   857→                            rn,
   858→                            immr: 0,
   859→                            imms: 31,
   860→                        })?;
   861→                    }
   862→                    CastKind::Trunc | CastKind::BitCast => {
   863→                        // Trunc: upper bits discarded on write — just MOV
   864→                        // BitCast: no data change — just MOV
   865→                        if rd != rn {
   866→                            self.emit_instruction(Instruction::MOV { rd, rm: rn })?;
   867→                        }
   868→                    }
   869→                }
   870→            }
   871→
   872→            IRInstr::Phi { .. } => {
   873→                log::warn!(
   874→                    "IRInstr::Phi encountered during emission — should be resolved by SSA pass"
   875→                );
   876→            }
   877→
   878→            IRInstr::GetAddress { dst, name } => {
   879→                let rd = self.resolve_reg(dst)?;
   880→                // Emit a call to __vuma_getaddr to resolve the symbol at runtime.
   881→                // Move name hash to X0 as the argument.
   882→                let name_hash = name
   883→                    .chars()
   884→                    .fold(0u64, |acc, c| acc.wrapping_mul(31).wrapping_add(c as u64));
   885→                self.emit_load_immediate(Register::X0, name_hash as i64)?;
   886→                let bl_word_idx = self.code.len();
   887→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
   888→                self.call_relocs.push(CallRelocation {
   889→                    text_byte_offset: bl_byte_offset,
   890→                    target_func: "__vuma_getaddr".to_string(),
   891→                });
   892→                self.relocations.push(RelocationEntry {
   893→                    offset: bl_byte_offset,
   894→                    symbol: "__vuma_getaddr".to_string(),
   895→                    reloc_type: "R_AARCH64_CALL26".to_string(),
   896→                });
   897→                self.emit_instruction(Instruction::BL { offset: 0 })?;
   898→                if rd != Register::X0 {
   899→                    self.emit_instruction(Instruction::MOV {
   900→                        rd,
   901→                        rm: Register::X0,
   902→                    })?;
   903→                }
   904→            }
   905→
   906→            IRInstr::Offset { dst, base, offset } => {
   907→                let rd = self.resolve_reg(dst)?;
   908→                let rn = self.resolve_reg(base)?;
   909→                let rm = match offset {
   910→                    IRValue::Immediate(v) => {
   911→                        if *v >= 0 && *v <= 4095 {
   912→                            Operand::Imm12(*v as u16)
   913→                        } else {
   914→                            let temp = Register::X9;
   915→                            self.emit_load_immediate(temp, *v)?;
   916→                            Operand::Reg {
   917→                                reg: temp,
   918→                                shift: None,
   919→                            }
   920→                        }
   921→                    }
   922→                    _ => Operand::Reg {
   923→                        reg: self.resolve_reg(offset)?,
   924→                        shift: None,
   925→                    },
   926→                };
   927→                self.emit_instruction(Instruction::ADD { rd, rn, rm })?;
   928→            }
   929→
   930→            // ── Dedicated arithmetic — delegate to BinOp ──
   931→            IRInstr::Add { dst, lhs, rhs, ty } => {
   932→                self.emit_binop(BinOpKind::Add, dst, lhs, rhs, ty.as_ref())?;
   933→            }
   934→            IRInstr::Sub { dst, lhs, rhs, ty } => {
   935→                self.emit_binop(BinOpKind::Sub, dst, lhs, rhs, ty.as_ref())?;
   936→            }
   937→            IRInstr::Mul { dst, lhs, rhs, ty } => {
   938→                self.emit_binop(BinOpKind::Mul, dst, lhs, rhs, ty.as_ref())?;
   939→            }
   940→            IRInstr::Div { dst, lhs, rhs, ty } => {
   941→                self.emit_binop(BinOpKind::SDiv, dst, lhs, rhs, ty.as_ref())?;
   942→            }
   943→
   944→            // ── Comparison instruction ──
   945→            IRInstr::Cmp {
   946→                kind,
   947→                dst,
   948→                lhs,
   949→                rhs,
   950→                ty,
   951→            } => {
   952→                let width = RegWidth::from_ir_type(ty.as_ref());
   953→                let rd = self.resolve_reg(dst)?;
   954→                let rn = self.resolve_reg(lhs)?;
   955→                let rm = self.resolve_reg(rhs)?;
   956→                self.emit_instruction_with_width(
   957→                    Instruction::CMP {
   958→                        rn,
   959→                        rm: Operand::Reg {
   960→                            reg: rm,
   961→                            shift: None,
   962→                        },
   963→                    },
   964→                    width,
   965→                )?;
   966→                let cond = cmp_kind_to_condition(kind);
   967→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
   968→            }
   969→
   970→            // ── Operating-system interface ──
   971→            IRInstr::Syscall { dst, sysno, args } => {
   972→                // ARM64 Linux syscall convention:
   973→                //   x8 = syscall number, x0-x5 = arguments, svc #0, result in x0
   974→                // Move arguments into X0–X5.
   975→                for (i, arg) in args.iter().enumerate() {
   976→                    if i >= 6 {
   977→                        break;
   978→                    }
   979→                    let src = self.resolve_reg(arg)?;
   980→                    let dst_reg = match i {
   981→                        0 => Register::X0,
   982→                        1 => Register::X1,
   983→                        2 => Register::X2,
   984→                        3 => Register::X3,
   985→                        4 => Register::X4,
   986→                        5 => Register::X5,
   987→                        _ => unreachable!(),
   988→                    };
   989→                    if src != dst_reg {
   990→                        self.emit_instruction(Instruction::MOV {
   991→                            rd: dst_reg,
   992→                            rm: src,
   993→                        })?;
   994→                    }
   995→                }
   996→                // Move syscall number into X8
   997→                self.emit_instruction(Instruction::MOVZ {
   998→                    rd: Register::X8,
   999→                    imm16: *sysno as u16,
  1000→                    shift: 0,
  1001→                })?;
  1002→                // If sysno > 65535, use MOVK for upper bits
  1003→                if *sysno > 0xFFFF {
  1004→                    self.emit_instruction(Instruction::MOVK {
  1005→                        rd: Register::X8,
  1006→                        imm16: (*sysno >> 16) as u16,
  1007→                        shift: 16,
  1008→                    })?;
  1009→                }
  1010→                // svc #0
  1011→                self.emit_instruction(Instruction::SVC { imm16: 0 })?;
  1012→                // Move result from X0 to destination
  1013→                if let Some(dst_val) = dst {
  1014→                    let rd = self.resolve_reg(dst_val)?;
  1015→                    if rd != Register::X0 {
  1016→                        self.emit_instruction(Instruction::MOV {
  1017→                            rd,
  1018→                            rm: Register::X0,
  1019→                        })?;
  1020→                    }
  1021→                }
  1022→            }
  1023→
  1024→            // ── Instruction-level control flow ──
  1025→            IRInstr::Ret { values } => {
  1026→                for (i, val) in values.iter().enumerate() {
  1027→                    if i >= 8 {
  1028→                        break;
  1029→                    }
  1030→                    let src = self.resolve_reg(val)?;
  1031→                    let dst_reg = match i {
  1032→                        0 => Register::X0,
  1033→                        1 => Register::X1,
  1034→                        2 => Register::X2,
  1035→                        3 => Register::X3,
  1036→                        4 => Register::X4,
  1037→                        5 => Register::X5,
  1038→                        6 => Register::X6,
  1039→                        7 => Register::X7,
  1040→                        _ => unreachable!(),
  1041→                    };
  1042→                    if src != dst_reg {
  1043→                        self.emit_instruction(Instruction::MOV {
  1044→                            rd: dst_reg,
  1045→                            rm: src,
  1046→                        })?;
  1047→                    }
  1048→                }
  1049→            }
  1050→
  1051→            IRInstr::Branch { target } => {
  1052→                let fixup_idx = self.code.len();
  1053→                self.fixups.push((fixup_idx, target.clone()));
  1054→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1055→            }
  1056→
  1057→            IRInstr::CondBranch {
  1058→                cond,
  1059→                true_target,
  1060→                false_target,
  1061→            } => {
  1062→                let rt = self.resolve_reg(cond)?;
  1063→                let fixup_cbz = self.code.len();
  1064→                self.fixups.push((fixup_cbz, false_target.clone()));
  1065→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1066→                let fixup_b = self.code.len();
  1067→                self.fixups.push((fixup_b, true_target.clone()));
  1068→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1069→            }
  1070→
  1071→            IRInstr::Select {
  1072→                dst,
  1073→                cond,
  1074→                true_val,
  1075→                false_val,
  1076→                ty,
  1077→            } => {
  1078→                let width = RegWidth::from_ir_type(ty.as_ref());
  1079→                // Lower select as: SUBS XZR, cond, #0; CSEL dst, false_val, true_val, NE
  1080→                let rd = self.resolve_reg(dst)?;
  1081→                let rc = self.resolve_reg(cond)?;
  1082→                let rt = self.resolve_reg(true_val)?;
  1083→                let rf = self.resolve_reg(false_val)?;
  1084→                // Compare cond against zero and select.
  1085→                self.emit_instruction_with_width(
  1086→                    Instruction::SUB {
  1087→                        rd: Register::XZR,
  1088→                        rn: rc,
  1089→                        rm: Operand::Imm12(0),
  1090→                    },
  1091→                    width,
  1092→                )?;
  1093→                // Set flags by using a separate CMP (SUB with XZR destination
  1094→                // doesn't set flags; we need a flags-setting variant).
  1095→                // We emulate this with: CMP rc, #0 which is SUBS XZR, rc, #0.
  1096→                // Since we only have SUB, we use the existing CMP pattern.
  1097→                self.emit_instruction_with_width(
  1098→                    Instruction::CSEL {
  1099→                        rd,
  1100→                        rn: rf,
  1101→                        rm: rt,
  1102→                        cond: crate::arm64::Condition::NE,
  1103→                    },
  1104→                    width,
  1105→                )?;
  1106→            }
  1107→        }
  1108→        Ok(())
  1109→    }
  1110→
  1111→    /// Emit a binary operation (shared by `BinOp` and dedicated `Add`/`Sub`/…).
  1112→    ///
  1113→    /// Some ARM64 instructions (ADD, SUB, LSL, LSR, ASR) accept a 12-bit
  1114→    /// unsigned immediate operand directly.  Most others (MUL, AND, ORR, EOR,
  1115→    /// SDIV, UDIV, remainder, comparisons) **require** a register operand.
  1116→    /// When the RHS is an immediate that the target instruction cannot accept
  1117→    /// directly, we "spill" it into a scratch register (X16 / IP0) first via
  1118→    /// MOVZ + MOVK and then use that register as the operand.
  1119→    ///
  1120→    /// X16 (IP0) is chosen as the scratch register because it is **not** in
  1121→    /// the register allocator's free pool, so it will never conflict with
  1122→    /// `rd` or `rn`.
  1123→    fn emit_binop(
  1124→        &mut self,
  1125→        op: BinOpKind,
  1126→        dst: &IRValue,
  1127→        lhs: &IRValue,
  1128→        rhs: &IRValue,
  1129→        ty: Option<&IRType>,
  1130→    ) -> Result<()> {
  1131→        let width = RegWidth::from_ir_type(ty);
  1132→        let rd = self.resolve_reg(dst)?;
  1133→        let rn = self.resolve_reg(lhs)?;
  1134→
  1135→        // Determine whether the operation can accept an immediate operand.
  1136→        // ADD, SUB, LSL, LSR, ASR all have immediate forms in ARM64.
  1137→        // All other operations require a register operand.
  1138→        let supports_imm = matches!(
  1139→            op,
  1140→            BinOpKind::Add | BinOpKind::Sub | BinOpKind::Shl | BinOpKind::ShrL | BinOpKind::ShrA
  1141→        );
  1142→
  1143→        let rm = match rhs {
  1144→            IRValue::Immediate(v) => {
  1145→                if supports_imm && *v >= 0 && *v <= 4095 {
  1146→                    // Small immediate that the instruction accepts directly.
  1147→                    Operand::Imm12(*v as u16)
  1148→                } else {
  1149→                    // Load the immediate into scratch register X16 (IP0).
  1150→                    // X16 is not in the register allocator's pool, so it cannot
  1151→                    // conflict with `rd` or `rn`.
  1152→                    let scratch = Register::X16;
  1153→                    self.emit_load_immediate_with_width(scratch, *v, width)?;
  1154→                    Operand::Reg {
  1155→                        reg: scratch,
  1156→                        shift: None,
  1157→                    }
  1158→                }
  1159→            }
  1160→            _ => Operand::Reg {
  1161→                reg: self.resolve_reg(rhs)?,
  1162→                shift: None,
  1163→            },
  1164→        };
  1165→
  1166→        match op {
  1167→            BinOpKind::Add => {
  1168→                self.emit_instruction_with_width(Instruction::ADD { rd, rn, rm }, width)?;
  1169→            }
  1170→            BinOpKind::Sub => {
  1171→                self.emit_instruction_with_width(Instruction::SUB { rd, rn, rm }, width)?;
  1172→            }
  1173→            BinOpKind::Mul => {
  1174→                let rm_reg = self.operand_to_reg(&rm)?;
  1175→                self.emit_instruction_with_width(Instruction::MUL { rd, rn, rm: rm_reg }, width)?;
  1176→            }
  1177→            BinOpKind::SDiv => {
  1178→                let rm_reg = self.operand_to_reg(&rm)?;
  1179→                self.emit_instruction_with_width(Instruction::SDIV { rd, rn, rm: rm_reg }, width)?;
  1180→            }
  1181→            BinOpKind::UDiv => {
  1182→                let rm_reg = self.operand_to_reg(&rm)?;
  1183→                self.emit_instruction_with_width(Instruction::UDIV { rd, rn, rm: rm_reg }, width)?;
  1184→            }
  1185→            BinOpKind::And => {
  1186→                let rm_reg = self.operand_to_reg(&rm)?;
  1187→                self.emit_instruction_with_width(Instruction::AND { rd, rn, rm: rm_reg }, width)?;
  1188→            }
  1189→            BinOpKind::Or => {
  1190→                let rm_reg = self.operand_to_reg(&rm)?;
  1191→                self.emit_instruction_with_width(Instruction::ORR { rd, rn, rm: rm_reg }, width)?;
  1192→            }
  1193→            BinOpKind::Xor => {
  1194→                let rm_reg = self.operand_to_reg(&rm)?;
  1195→                self.emit_instruction_with_width(Instruction::EOR { rd, rn, rm: rm_reg }, width)?;
  1196→            }
  1197→            BinOpKind::Shl => {
  1198→                self.emit_instruction_with_width(Instruction::LSL { rd, rn, rm }, width)?;
  1199→            }
  1200→            BinOpKind::ShrL => {
  1201→                self.emit_instruction_with_width(Instruction::LSR { rd, rn, rm }, width)?;
  1202→            }
  1203→            BinOpKind::ShrA => {
  1204→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm }, width)?;
  1205→            }
  1206→            BinOpKind::Ror | BinOpKind::Rol => {
  1207→                let rm_reg = self.operand_to_reg(&rm)?;
  1208→                // ARM64 rotation: use ASR as placeholder
  1209→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm: Operand::Reg { reg: rm_reg, shift: None } }, width)?;
  1210→            }
  1211→            BinOpKind::SRem | BinOpKind::URem => {
  1212→                let rm_reg = self.operand_to_reg(&rm)?;
  1213→                let div_instr = if op == BinOpKind::SRem {
  1214→                    Instruction::SDIV { rd, rn, rm: rm_reg }
  1215→                } else {
  1216→                    Instruction::UDIV { rd, rn, rm: rm_reg }
  1217→                };
  1218→                self.emit_instruction_with_width(div_instr, width)?;
  1219→                // MSUB rd, rd, rm, rn  =>  rd = rn - rd * rm  =  dividend - quotient * divisor
  1220→                self.emit_instruction_with_width(Instruction::MSUB {
  1221→                    rd,
  1222→                    rn: rd,     // quotient (result of DIV)
  1223→                    rm: rm_reg, // divisor
  1224→                    ra: rn,     // dividend
  1225→                }, width)?;
  1226→            }
  1227→            BinOpKind::SLt
  1228→            | BinOpKind::SLe
  1229→            | BinOpKind::SGt
  1230→            | BinOpKind::SGe
  1231→            | BinOpKind::ULt
  1232→            | BinOpKind::ULe
  1233→            | BinOpKind::UGt
  1234→            | BinOpKind::UGe
  1235→            | BinOpKind::Eq
  1236→            | BinOpKind::Ne => {
  1237→                let rm_reg = self.operand_to_reg(&rm)?;
  1238→                self.emit_instruction_with_width(Instruction::CMP {
  1239→                    rn,
  1240→                    rm: Operand::Reg {
  1241→                        reg: rm_reg,
  1242→                        shift: None,
  1243→                    },
  1244→                }, width)?;
  1245→                let cond = binop_kind_to_condition(&op);
  1246→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1247→            }
  1248→        }
  1249→        Ok(())
  1250→    }
  1251→
  1252→    /// Emit the block terminator.
  1253→    fn emit_terminator(&mut self, term: &IRTerminator) -> Result<()> {
  1254→        match term {
  1255→            IRTerminator::Jump(target) => {
  1256→                let fixup_idx = self.code.len();
  1257→                self.fixups.push((fixup_idx, target.clone()));
  1258→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1259→            }
  1260→            IRTerminator::Branch {
  1261→                cond,
  1262→                true_block,
  1263→                false_block,
  1264→            } => {
  1265→                let rt = self.resolve_reg(cond)?;
  1266→                let fixup_cbnz = self.code.len();
  1267→                self.fixups.push((fixup_cbnz, true_block.clone()));
  1268→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1269→                let fixup_b = self.code.len();
  1270→                self.fixups.push((fixup_b, false_block.clone()));
  1271→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1272→            }
  1273→            IRTerminator::Return(vals) => {
  1274→                for (i, val) in vals.iter().enumerate() {
  1275→                    if i >= 8 {
  1276→                        break;
  1277→                    }
  1278→                    let src = self.resolve_reg(val)?;
  1279→                    let dst_reg = match i {
  1280→                        0 => Register::X0,
  1281→                        1 => Register::X1,
  1282→                        2 => Register::X2,
  1283→                        3 => Register::X3,
  1284→                        4 => Register::X4,
  1285→                        5 => Register::X5,
  1286→                        6 => Register::X6,
  1287→                        7 => Register::X7,
  1288→                        _ => unreachable!(),
  1289→                    };
  1290→                    if src != dst_reg {
  1291→                        self.emit_instruction(Instruction::MOV {
  1292→                            rd: dst_reg,
  1293→                            rm: src,
  1294→                        })?;
  1295→                    }
  1296→                }
  1297→                // Use the same frame size computed in the prologue
  1298→                let frame_size = self.frame_size;
  1299→                self.emit_instruction(Instruction::LDP {
  1300→                    rt1: Register::X29,
  1301→                    rt2: Register::X30,
  1302→                    rn: Register::SP,
  1303→                    offset: frame_size as i32, // load FP/LR from top of frame
  1304→                })?;
  1305→                self.emit_instruction(Instruction::ADD {
  1306→                    rd: Register::SP,
  1307→                    rn: Register::SP,
  1308→                    rm: Operand::Imm12(frame_size + 16), // restore SP
  1309→                })?;
  1310→                self.emit_instruction(Instruction::RET { rn: None })?;
  1311→            }
  1312→            IRTerminator::Unreachable => {
  1313→                self.emit_instruction(Instruction::MOV {
  1314→                    rd: Register::XZR,
  1315→                    rm: Register::XZR,
  1316→                })?;
  1317→            }
  1318→            // Switch, Invoke, TailCall, and Resume are lowered by the
  1319→            // control_flow module before reaching the emitter. If they
  1320→            // appear here, it means the lowering pass was not run.
  1321→            IRTerminator::Switch { .. } => {
  1322→                return Err(CodegenError::InvalidInstruction(
  1323→                    "Switch terminator must be lowered before emission".to_string(),
  1324→                ));
  1325→            }
  1326→            IRTerminator::Invoke { .. } => {
  1327→                return Err(CodegenError::InvalidInstruction(
  1328→                    "Invoke terminator must be lowered before emission".to_string(),
  1329→                ));
  1330→            }
  1331→            IRTerminator::TailCall { .. } => {
  1332→                return Err(CodegenError::InvalidInstruction(
  1333→                    "TailCall terminator must be lowered before emission".to_string(),
  1334→                ));
  1335→            }
  1336→            IRTerminator::Resume { .. } => {
  1337→                return Err(CodegenError::InvalidInstruction(
  1338→                    "Resume terminator must be lowered before emission".to_string(),
  1339→                ));
  1340→            }
  1341→        }
  1342→        Ok(())
  1343→    }

  1355→    /// Emit an instruction with a specific register width (32-bit W or 64-bit X).

  1357→    /// This is the primary emission method for arithmetic and logical instructions
  1358→    /// where the operand width affects the encoding. Using `RegWidth::W32` produces
  1359→    /// instructions that operate on W sub-registers, giving automatic 32-bit

  1367→    /// Emit a load instruction with the given offset and IR type.
  1368→    ///
  1369→    /// Selects the correct ARM64 load variant based on the IR type:
  1370→    /// - I8/U8 → LDRB (byte, zero-extended)
  1371→    /// - I16/U16 → LDRH (halfword, zero-extended)
  1372→    /// - I32/U32/Ptr/Func → LDR (word or doubleword)
  1373→    /// - I64/U64 → LDR (doubleword)
  1374→    ///
  1375→    /// If the offset fits the ARM64 unsigned-offset immediate encoding for the
  1376→    /// selected instruction variant, it is encoded directly. Otherwise, the
  1377→    /// effective address is computed in a scratch register (X9) first, and
  1378→    /// the load is performed at [X9 + 0].
  1379→    fn emit_load(&mut self, rt: Register, rn: Register, offset: i32, ty: &IRType) -> Result<()> {
  1380→        // Determine the ARM64 load instruction and its immediate-offset constraints.
  1381→        // For each variant: (scale, max_imm12) where the encoded offset = imm12 << scale.
  1382→        // LDRB:  scale=0, imm12 0..4095 → offset 0..4095
  1383→        // LDRH:  scale=1, imm12 0..4095 → offset 0..8190, even
  1384→        // LDRSW: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1385→        // LDR W: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1386→        // LDR X: scale=3, imm12 0..4095 → offset 0..32760, multiple of 8
  1387→        match ty {
  1388→            IRType::I8 | IRType::U8 => {
  1389→                if offset >= 0 && offset <= 4095 {
  1390→                    self.emit_instruction(Instruction::LDRB { rt, rn, offset })?;
  1391→                } else {
  1392→                    self.emit_address_with_offset(rn, offset)?;
  1393→                    self.emit_instruction(Instruction::LDRB {
  1394→                        rt,
  1395→                        rn: Register::X9,
  1396→                        offset: 0,
  1397→                    })?;
  1398→                }
  1399→                // Sign-extend for signed byte loads (SXTB = SBFM Xd, Xn, #0, #7)
  1400→                if *ty == IRType::I8 {
  1401→                    self.emit_instruction(Instruction::SBFM {
  1402→                        rd: rt,
  1403→                        rn: rt,
  1404→                        immr: 0,
  1405→                        imms: 7,
  1406→                    })?;
  1407→                }
  1408→            }
  1409→            IRType::I16 | IRType::U16 => {
  1410→                if offset >= 0 && offset <= 8190 && offset % 2 == 0 {
  1411→                    self.emit_instruction(Instruction::LDRH { rt, rn, offset })?;
  1412→                } else {
  1413→                    self.emit_address_with_offset(rn, offset)?;
  1414→                    self.emit_instruction(Instruction::LDRH {
  1415→                        rt,
  1416→                        rn: Register::X9,
  1417→                        offset: 0,
  1418→                    })?;
  1419→                }
  1420→                // Sign-extend for signed halfword loads (SXTH = SBFM Xd, Xn, #0, #15)
  1421→                if *ty == IRType::I16 {
  1422→                    self.emit_instruction(Instruction::SBFM {
  1423→                        rd: rt,
  1424→                        rn: rt,
  1425→                        immr: 0,
  1426→                        imms: 15,
  1427→                    })?;
  1428→                }
  1429→            }
  1430→            IRType::I32 | IRType::U32 => {
  1431→                // 32-bit load uses LDR Wt encoding (scale=2, offset must be multiple of 4)
  1432→                if offset >= 0 && offset <= 16380 && offset % 4 == 0 {
  1433→                    self.emit_instruction(Instruction::LDR_W { rt, rn, offset })?;
  1434→                } else {
  1435→                    self.emit_address_with_offset(rn, offset)?;
  1436→                    self.emit_instruction(Instruction::LDR_W {
  1437→                        rt,
  1438→                        rn: Register::X9,
  1439→                        offset: 0,
  1440→                    })?;
  1441→                }
  1442→                // Sign-extend for signed word loads (SXTW = SBFM Xd, Xn, #0, #31)
  1443→                if *ty == IRType::I32 {
  1444→                    self.emit_instruction(Instruction::SBFM {
  1445→                        rd: rt,
  1446→                        rn: rt,
  1447→                        immr: 0,
  1448→                        imms: 31,
  1449→                    })?;
  1450→                }
  1451→            }
  1452→            IRType::I64 | IRType::U64 | IRType::Ptr | IRType::Func => {
  1453→                // 64-bit load uses LDR Xt encoding (scale=3, offset must be multiple of 8)
  1454→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1455→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1456→                } else {
  1457→                    self.emit_address_with_offset(rn, offset)?;
  1458→                    self.emit_instruction(Instruction::LDR {
  1459→                        rt,
  1460→                        rn: Register::X9,
  1461→                        offset: 0,
  1462→                    })?;
  1463→                }
  1464→            }
  1465→            _ => {
  1466→                // Default: treat as 64-bit load
  1467→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1468→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1469→                } else {
  1470→                    self.emit_address_with_offset(rn, offset)?;
  1471→                    self.emit_instruction(Instruction::LDR {
  1472→                        rt,
  1473→                        rn: Register::X9,
  1474→                        offset: 0,
  1475→                    })?;
  1476→                }
  1477→            }
  1478→        }
  1479→        Ok(())
  1480→    }
  1481→
  1482→    /// Emit a store instruction with the given offset and IR type.
  1483→    ///
  1484→    /// Selects the correct ARM64 store variant based on the IR type:
  1485→    /// - I8/U8 → STRB (byte)
  1486→    /// - I16/U16 → STRH (halfword)
  1487→    /// - I32/U32 → STR_W (32-bit word)
  1488→    /// - I64/U64/Ptr/Func → STR (64-bit doubleword)
  1489→    ///
  1490→    /// If the offset fits the ARM64 unsigned-offset immediate encoding for the
  1491→    /// selected instruction variant, it is encoded directly. Otherwise, the
  1492→    /// effective address is computed in a scratch register (X9) first, and
  1493→    /// the store is performed at [X9 + 0].
  1494→    fn emit_store(&mut self, rt: Register, rn: Register, offset: i32, ty: &IRType) -> Result<()> {

  1496→            IRType::I8 | IRType::U8 => {

  1498→                    self.emit_instruction(Instruction::STRB { rt, rn, offset })?;

  1500→                    self.emit_address_with_offset(rn, offset)?;

  1503→                        rn: Register::X9,

  1520→            IRType::I32 | IRType::U32 => {

  1533→            IRType::I64 | IRType::U64 | IRType::Ptr | IRType::Func => {
  1534→                // 64-bit store uses STR Xt encoding (scale=3, offset must be multiple of 8)
  1535→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {

  1537→                } else {
  1538→                    self.emit_address_with_offset(rn, offset)?;

  1540→                        rt,
  1541→                        rn: Register::X9,

  1543→                    })?;
  1544→                }
  1545→            }
  1546→            _ => {
  1547→                // Default: treat as 64-bit store
  1548→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1549→                    self.emit_instruction(Instruction::STR { rt, rn, offset })?;
  1550→                } else {
  1551→                    self.emit_address_with_offset(rn, offset)?;
  1552→                    self.emit_instruction(Instruction::STR {
  1553→                        rt,
  1554→                        rn: Register::X9,
  1555→                        offset: 0,
  1556→                    })?;
  1557→                }
  1558→            }
  1559→        }
  1560→        Ok(())
  1561→    }
  1562→
  1563→    /// Compute `X9 = rn + offset` for load/store with an offset that does not
  1564→    /// fit the ARM64 unsigned-offset immediate encoding.
  1565→    ///
  1566→    /// Uses X9 as a scratch register. If the offset fits in a 12-bit unsigned
  1567→    /// immediate (0..4095), emits `ADD X9, rn, #offset`. Otherwise, loads the
  1568→    /// offset into X9 with MOVZ/MOVK and then emits `ADD X9, rn, X9`.
  1569→    fn emit_address_with_offset(
  1570→        &mut self,
  1571→        rn: Register,
  1572→        offset: i32,
  1573→    ) -> Result<()> {
  1574→        if offset >= 0 && offset <= 4095 {
  1575→            // Small positive offset: ADD X9, rn, #offset
  1576→            self.emit_instruction(Instruction::ADD {
  1577→                rd: Register::X9,
  1578→                rn,
  1579→                rm: Operand::Imm12(offset as u16),
  1580→            })?;
  1581→        } else {
  1582→            // Large or negative offset: load offset into X9, then ADD X9, rn, X9
  1583→            self.emit_load_immediate(Register::X9, offset as i64)?;
  1584→            self.emit_instruction(Instruction::ADD {
  1585→                rd: Register::X9,
  1586→                rn,
  1587→                rm: Operand::Reg {
  1588→                    reg: Register::X9,
  1589→                    shift: None,
  1590→                },
  1591→            })?;
  1592→        }
  1593→        Ok(())
  1594→    }
  1595→
  1596→    fn resolve_reg(&mut self, val: &IRValue) -> Result<Register> {
  1597→        match val {
  1598→            IRValue::Register(id) => self.reg_alloc.allocate(*id),
  1599→            IRValue::Immediate(v) => {
  1600→                let temp = Register::X9;
  1601→                self.emit_load_immediate(temp, *v)?;
  1602→                Ok(temp)
  1603→            }
  1604→            IRValue::Address(addr) => {
  1605→                let temp = Register::X10;
  1606→                self.emit_load_immediate(temp, *addr as i64)?;
  1607→                Ok(temp)
  1608→            }
  1609→            IRValue::Label(_) => Err(CodegenError::EncodingError(
  1610→                "label value cannot be resolved to a register".into(),
  1611→            )),
  1612→        }
  1613→    }
  1614→
  1615→    fn emit_load_immediate(&mut self, rd: Register, value: i64) -> Result<()> {
  1616→        if (0..=65535).contains(&value) {
  1617→            self.emit_instruction(Instruction::MOVZ {
  1618→                rd,
  1619→                imm16: value as u16,
  1620→                shift: 0,
  1621→            })?;
  1622→            return Ok(());
  1623→        }
  1624→        if (0..=0xFFFF_FFFF).contains(&value) {
  1625→            let lo = (value & 0xFFFF) as u16;
  1626→            let hi = ((value >> 16) & 0xFFFF) as u16;
  1627→            self.emit_instruction(Instruction::MOVZ {
  1628→                rd,
  1629→                imm16: lo,
  1630→                shift: 0,
  1631→            })?;
  1632→            self.emit_instruction(Instruction::MOVK {
  1633→                rd,
  1634→                imm16: hi,
  1635→                shift: 16,
  1636→            })?;
  1637→            return Ok(());
  1638→        }
  1639→        let w0 = (value & 0xFFFF) as u16;
  1640→        let w1 = ((value >> 16) & 0xFFFF) as u16;
  1641→        let w2 = ((value >> 32) & 0xFFFF) as u16;

  1695→                        RegWidth::W32,

  1701→                        Instruction::MOVZ {
  1702→                            rd,
  1703→                            imm16: lo,
  1704→                            shift: 0,

  1706→                        RegWidth::W32,

  1710→                            Instruction::MOVK {

  1712→                                imm16: hi,
  1713→                                shift: 16,
  1714→                            },
  1715→                            RegWidth::W32,
  1716→                        )?;
  1717→                    }
  1718→                }

  1742→    fn apply_fixups(&mut self) -> Result<()> {
  1743→        let fixups = std::mem::take(&mut self.fixups);
  1744→        for (word_idx, label) in &fixups {
  1745→            let target_offset = self.label_offsets.get(label).copied().unwrap_or(0);
  1746→            let offset = (target_offset as i32) - (*word_idx as i32);
  1747→            let old_word = self.code[*word_idx];
  1748→            let patched = (old_word & !0x03FFFFFF) | ((offset & 0x03FFFFFF) as u32);
  1749→            self.code[*word_idx] = patched;

  1751→        Ok(())

  1814→/// Compute the stack frame size for a function by summing its `Alloc`
  1815→/// instructions, adding 16 bytes for the FP/LR save pair, and rounding up
  1816→/// to 16-byte alignment.
  1817→fn compute_frame_size(func: &IRFunction) -> u16 {
  1818→    let mut total: u32 = 16; // FP/LR save pair
  1819→    for block in &func.blocks {
  1820→        for instr in &block.instructions {
  1821→            if let IRInstr::Alloc { size, .. } = instr {
  1822→                let aligned = (*size).div_ceil(16) * 16;
  1823→                total += aligned;
  1824→            }
  1825→        }
  1826→    }
  1827→    // Round up to 16-byte alignment (should already be, but be safe).
  1828→    total = (total + 15) & !15;
  1829→    total as u16
  1830→}
