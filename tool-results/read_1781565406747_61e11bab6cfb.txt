    83тЖТ    83тЖТ/// SHA-256 compression function: process one 512-bit (64-byte) block.
    84тЖТ    84тЖТfn sha256_transform(state: &mut [u32; 8], block: &[u8; 64]) {
    85тЖТ    85тЖТ    // Build message schedule W[0..63].
    86тЖТ    86тЖТ    let mut w = [0u32; 64];
    87тЖТ    87тЖТ    for i in 0..16 {
    88тЖТ    88тЖТ        w[i] = u32::from_be_bytes([
    89тЖТ    89тЖТ            block[i * 4],
    90тЖТ    90тЖТ            block[i * 4 + 1],
    91тЖТ    91тЖТ            block[i * 4 + 2],
    92тЖТ    92тЖТ            block[i * 4 + 3],
    93тЖТ    93тЖТ        ]);
    94тЖТ    94тЖТ    }
    95тЖТ    95тЖТ    for i in 16..64 {
    96тЖТ    96тЖТ        w[i] = small_sigma1(w[i - 2])
    97тЖТ    97тЖТ            .wrapping_add(w[i - 7])
    98тЖТ    98тЖТ            .wrapping_add(small_sigma0(w[i - 15]))
    99тЖТ    99тЖТ            .wrapping_add(w[i - 16]);
   100тЖТ   100тЖТ    }
   101тЖТ   101тЖТ
   102тЖТ   102тЖТ    // Initialize working variables.
   103тЖТ   103тЖТ    let [mut a, mut b, mut c, mut d, mut e, mut f, mut g, mut h] = *state;
   104тЖТ   104тЖТ
   105тЖТ   105тЖТ    // 64-round compression.
   106тЖТ   106тЖТ    for i in 0..64 {
   107тЖТ   107тЖТ        let t1 = h
   108тЖТ   108тЖТ            .wrapping_add(big_sigma1(e))
   109тЖТ   109тЖТ            .wrapping_add(ch(e, f, g))
   110тЖТ   110тЖТ            .wrapping_add(K[i])
   111тЖТ   111тЖТ            .wrapping_add(w[i]);
   112тЖТ   112тЖТ        let t2 = big_sigma0(a).wrapping_add(maj(a, b, c));
   113тЖТ   113тЖТ
   114тЖТ   114тЖТ        h = g;
   115тЖТ   115тЖТ        g = f;
   116тЖТ   116тЖТ        f = e;
   117тЖТ   117тЖТ        e = d.wrapping_add(t1);
   118тЖТ   118тЖТ        d = c;
   119тЖТ   119тЖТ        c = b;
   120тЖТ   120тЖТ        b = a;
   121тЖТ   121тЖТ        a = t1.wrapping_add(t2);
   122тЖТ   122тЖТ    }
   123тЖТ   123тЖТ
   124тЖТ   124тЖТ    // Add compressed chunk to hash state.
   125тЖТ   125тЖТ    state[0] = state[0].wrapping_add(a);
   126тЖТ   126тЖТ    state[1] = state[1].wrapping_add(b);
   127тЖТ   127тЖТ    state[2] = state[2].wrapping_add(c);
   128тЖТ   128тЖТ    state[3] = state[3].wrapping_add(d);
   129тЖТ   129тЖТ    state[4] = state[4].wrapping_add(e);
   130тЖТ   130тЖТ    state[5] = state[5].wrapping_add(f);
   131тЖТ   131тЖТ    state[6] = state[6].wrapping_add(g);
   132тЖТ   132тЖТ    state[7] = state[7].wrapping_add(h);
   133тЖТ   133тЖТ}

   135тЖТ   135тЖТ/// Compute SHA-256 of an arbitrary byte message, returning 32-byte digest.
   136тЖТ   136тЖТfn sha256(message: &[u8]) -> [u8; 32] {
   137тЖТ   137тЖТ    let mut state = H_INIT;
   138тЖТ   138тЖТ
   139тЖТ   139тЖТ    // Padding: append 0x80, then zeros, then 64-bit big-endian bit length.
   140тЖТ   140тЖТ    // Total padded length must be a multiple of 64 bytes.
   141тЖТ   141тЖТ    let msg_len = message.len();
   142тЖТ   142тЖТ    let bit_len = (msg_len as u64) * 8;
   143тЖТ   143тЖТ
   144тЖТ   144тЖТ    // Determine padded length: next multiple of 64 that can fit the padding.
   145тЖТ   145тЖТ    // We need at least 1 byte (0x80) + 8 bytes (length) = 9 bytes after message.
   146тЖТ   146тЖТ    let padded_len = if msg_len % 64 < 56 {
   147тЖТ   147тЖТ        (msg_len / 64 + 1) * 64
   148тЖТ   148тЖТ    } else {
   149тЖТ   149тЖТ        (msg_len / 64 + 2) * 64
   150тЖТ   150тЖТ    };
   151тЖТ   151тЖТ
   152тЖТ   152тЖТ    let mut padded = vec![0u8; padded_len];
   153тЖТ   153тЖТ    padded[..msg_len].copy_from_slice(message);
   154тЖТ   154тЖТ    padded[msg_len] = 0x80;
   155тЖТ   155тЖТ    let len_bytes = bit_len.to_be_bytes();
   156тЖТ   156тЖТ    padded[padded_len - 8..].copy_from_slice(&len_bytes);
   157тЖТ   157тЖТ
   158тЖТ   158тЖТ    // Process each 64-byte block.
   159тЖТ   159тЖТ    for chunk_start in (0..padded_len).step_by(64) {
   160тЖТ   160тЖТ        let mut block = [0u8; 64];
   161тЖТ   161тЖТ        block.copy_from_slice(&padded[chunk_start..chunk_start + 64]);
   162тЖТ   162тЖТ        sha256_transform(&mut state, &block);
   163тЖТ   163тЖТ    }
   164тЖТ   164тЖТ
   165тЖТ   165тЖТ    // Produce digest (8 x u32 in big-endian).
   166тЖТ   166тЖТ    let mut digest = [0u8; 32];
   167тЖТ   167тЖТ    for i in 0..8 {
   168тЖТ   168тЖТ        digest[i * 4..i * 4 + 4].copy_from_slice(&state[i].to_be_bytes());
   169тЖТ   169тЖТ    }
   170тЖТ   170тЖТ    digest
   171тЖТ   171тЖТ}

   173тЖТ   173тЖТ/// Compute SHA256d: SHA-256(SHA-256(message)).
   174тЖТ   174тЖТ/// The inner SHA-256 produces a 32-byte digest, which is then hashed
   175тЖТ   175тЖТ/// again by the outer SHA-256.
   176тЖТ   176тЖТfn sha256d(message: &[u8]) -> [u8; 32] {
   177тЖТ   177тЖТ    let inner = sha256(message);
   178тЖТ   178тЖТ    sha256(&inner)
   179тЖТ   179тЖТ}
