  1400→  1400→        ty: Option<IRType>,
  1401→  1401→    },
  1402→  1402→
  1403→  1403→    // ── Comparison ──────────────────────────────────────────────────
  1404→  1404→    /// Comparison: `dst = cmp kind lhs rhs` — produces 1 or 0.
  1405→  1405→    Cmp {
  1406→  1406→        /// Comparison kind.
  1407→  1407→        kind: CmpKind,
  1408→  1408→        /// Destination register (boolean result).
  1409→  1409→        dst: IRValue,
  1410→  1410→        /// Left-hand side operand.
  1411→  1411→        lhs: IRValue,
  1412→  1412→        /// Right-hand side operand.
  1413→  1413→        rhs: IRValue,
  1414→  1414→        /// Type of the operands (determines 32-bit vs 64-bit comparison).
  1415→  1415→        /// `None` defaults to 64-bit (X register) behavior.
  1416→  1416→        ty: Option<IRType>,
  1417→  1417→    },
  1418→  1418→
  1419→  1419→    // ── Atomic operations ──────────────────────────────────────────────
  1420→  1420→    /// Atomic load: `dst = atomic_load addr`
  1421→  1421→    ///
  1422→  1422→    /// Emits target-specific acquire-load instructions:
  1423→  1423→    /// - AArch64: `LDAXR`
  1424→  1424→    /// - x86_64: `LOCK` prefix or plain `MOV` (x86 is already atomic for aligned loads)
  1425→  1425→    /// - RISC-V: `LR.D`
  1426→  1426→    AtomicLoad {
  1427→  1427→        /// Destination register.
  1428→  1428→        dst: IRValue,
  1429→  1429→        /// Source address register.
  1430→  1430→        addr: IRValue,
  1431→  1431→        /// Type of the value being loaded (determines access size).
  1432→  1432→        ty: IRType,
  1433→  1433→    },
  1434→  1434→
  1435→  1435→    /// Atomic store: `atomic_store value, addr`
  1436→  1436→    ///
  1437→  1437→    /// Emits target-specific release-store instructions:
  1438→  1438→    /// - AArch64: `STLXR`
  1439→  1439→    /// - x86_64: `LOCK` prefix or plain `MOV`
  1440→  1440→    /// - RISC-V: `SC.D`
  1441→  1441→    AtomicStore {
  1442→  1442→        /// Value to store.
  1443→  1443→        value: IRValue,
  1444→  1444→        /// Target address register.
  1445→  1445→        addr: IRValue,
  1446→  1446→        /// Type of the value being stored.
  1447→  1447→        ty: IRType,
  1448→  1448→    },
  1449→  1449→
  1450→  1450→    /// Atomic compare-and-swap: `dst = atomic_cas addr, expected, desired`
  1451→  1451→    ///
  1452→  1452→    /// Returns the old value at `addr`. If the old value equals `expected`,
  1453→  1453→    /// the new value `desired` is written. The `dst` receives the old value
  1454→  1454→    /// so the caller can check whether the swap succeeded.
  1455→  1455→    ///
  1456→  1456→    /// Emits target-specific LL/SC or CMPXCHG sequences:
  1457→  1457→    /// - AArch64: `LDAXR` / `CMP` / `B.NE` / `STLXR` loop
  1458→  1458→    /// - x86_64: `LOCK CMPXCHG`
  1459→  1459→    /// - RISC-V: `LR.D` / `BNE` / `SC.D` loop
  1460→  1460→    AtomicCas {
  1461→  1461→        /// Destination register (receives the old value at addr).
  1462→  1462→        dst: IRValue,
  1463→  1463→        /// Target address register.
  1464→  1464→        addr: IRValue,
  1465→  1465→        /// Expected value.
  1466→  1466→        expected: IRValue,
  1467→  1467→        /// Desired new value.
  1468→  1468→        desired: IRValue,
  1469→  1469→        /// Type of the value (determines access size).
  1470→  1470→        ty: IRType,
  1471→  1471→    },
  1472→  1472→
  1473→  1473→    // ── Instruction-level control flow ───────────────────────────────
  1474→  1474→    /// Return from the current function with optional values.
  1475→  1475→    Ret {
  1476→  1476→        /// Return value registers.
  1477→  1477→        values: Vec<IRValue>,
  1478→  1478→    },
  1479→  1479→    /// Unconditional branch to a label.
  1480→  1480→    Branch {
  1481→  1481→        /// Target label.
  1482→  1482→        target: String,
  1483→  1483→    },
  1484→  1484→    /// Conditional branch: if `cond` is non-zero, go to `true_target`;
  1485→  1485→    /// otherwise go to `false_target`.
  1486→  1486→    CondBranch {
  1487→  1487→        /// Condition value.
  1488→  1488→        cond: IRValue,
  1489→  1489→        /// Label to branch to when condition is true.
  1490→  1490→        true_target: String,
  1491→  1491→        /// Label to branch to when condition is false.
  1492→  1492→        false_target: String,
  1493→  1493→    },
  1494→  1494→
  1495→  1495→    // ── Constant-time security operations ───────────────────────────
  1496→  1496→    /// Constant-time conditional select: `dst = ct_select(cond, a, b)`.
  1497→  1497→    ///
  1498→  1498→    /// Returns `a` if `cond != 0`, else `b`, using only bitwise operations
  1499→  1499→    /// (no branches) to prevent timing side-channel attacks.
  1500→  1500→    ///
  1501→  1501→    /// Lowered to: `(a & mask) | (b & ~mask)` where `mask = -(cond != 0)`.
  1502→  1502→    CtSelect {
  1503→  1503→        /// Destination register.
  1504→  1504→        dst: IRValue,
  1505→  1505→        /// Condition value.
  1506→  1506→        cond: IRValue,
  1507→  1507→        /// Value selected when condition is true.
  1508→  1508→        true_val: IRValue,
  1509→  1509→        /// Value selected when condition is false.
  1510→  1510→        false_val: IRValue,
  1511→  1511→        /// Type of the result (determines 32-bit vs 64-bit encoding).
  1512→  1512→        /// `None` defaults to 64-bit (X register) behavior.
  1513→  1513→        ty: Option<IRType>,
  1514→  1514→    },
  1515→  1515→
  1516→  1516→    /// Constant-time equality check: `dst = ct_eq(a, b)`.
  1517→  1517→    ///
  1518→  1518→    /// Returns 1 if `a == b`, else 0, using only bitwise operations
  1519→  1519→    /// (no branches) to prevent timing side-channel attacks.
  1520→  1520→    ///
  1521→  1521→    /// Lowered to: XOR-based comparison with constant-time result.
  1522→  1522→    CtEq {
  1523→  1523→        /// Destination register (boolean result: 1 or 0).
  1524→  1524→        dst: IRValue,
  1525→  1525→        /// Left-hand side operand.
  1526→  1526→        lhs: IRValue,
  1527→  1527→        /// Right-hand side operand.
  1528→  1528→        rhs: IRValue,
  1529→  1529→        /// Type of the operands (determines 32-bit vs 64-bit comparison).
  1530→  1530→        /// `None` defaults to 64-bit (X register) behavior.
  1531→  1531→        ty: Option<IRType>,
  1532→  1532→    },
  1533→  1533→}

  1535→  1535→impl IRInstr {
  1536→  1536→    /// Returns the set of virtual-register IDs that this instruction defines
  1537→  1537→    /// (writes to).
  1538→  1538→    pub fn defined_regs(&self) -> Vec<u32> {
  1539→  1539→        match self {
  1540→  1540→            IRInstr::Load { dst, .. }
  1541→  1541→            | IRInstr::BinOp { dst, .. }
  1542→  1542→            | IRInstr::UnaryOp { dst, .. }
  1543→  1543→            | IRInstr::Alloc { dst, .. }
  1544→  1544→            | IRInstr::Cast { dst, .. }
  1545→  1545→            | IRInstr::Phi { dst, .. }
  1546→  1546→            | IRInstr::GetAddress { dst, .. }
  1547→  1547→            | IRInstr::Offset { dst, .. }
  1548→  1548→            | IRInstr::Select { dst, .. } => dst.as_register().into_iter().collect(),
  1549→  1549→            IRInstr::Call { dst, .. } => dst
  1550→  1550→                .as_ref()
  1551→  1551→                .and_then(|v| v.as_register())
  1552→  1552→                .into_iter()
  1553→  1553→                .collect(),
  1554→  1554→            IRInstr::Add { dst, .. }
  1555→  1555→            | IRInstr::Sub { dst, .. }
  1556→  1556→            | IRInstr::Mul { dst, .. }
  1557→  1557→            | IRInstr::Div { dst, .. }
  1558→  1558→            | IRInstr::Cmp { dst, .. }
  1559→  1559→            | IRInstr::CtSelect { dst, .. }
  1560→  1560→            | IRInstr::CtEq { dst, .. }
  1561→  1561→            | IRInstr::AtomicLoad { dst, .. }
  1562→  1562→            | IRInstr::AtomicCas { dst, .. } => dst.as_register().into_iter().collect(),
  1563→  1563→            IRInstr::Store { .. }
  1564→  1564→            | IRInstr::Free { .. }
  1565→  1565→            | IRInstr::Ret { .. }
  1566→  1566→            | IRInstr::Branch { .. }
  1567→  1567→            | IRInstr::CondBranch { .. }
  1568→  1568→            | IRInstr::AtomicStore { .. } => vec![],
  1569→  1569→        }
  1570→  1570→    }
  1571→  1571→
  1572→  1572→    /// Returns the set of virtual-register IDs that this instruction uses
  1573→  1573→    /// (reads from).
  1574→  1574→    pub fn used_regs(&self) -> Vec<u32> {
  1575→  1575→        match self {
  1576→  1576→            IRInstr::Load { addr, .. } => addr.as_register().into_iter().collect(),
  1577→  1577→            IRInstr::Store { value, addr, .. } => {
  1578→  1578→                let mut r = value.as_register().into_iter().collect::<Vec<_>>();
  1579→  1579→                r.extend(addr.as_register());
  1580→  1580→                r
  1581→  1581→            }
  1582→  1582→            IRInstr::BinOp { lhs, rhs, .. }
  1583→  1583→            | IRInstr::Add { lhs, rhs, .. }
  1584→  1584→            | IRInstr::Sub { lhs, rhs, .. }
  1585→  1585→            | IRInstr::Mul { lhs, rhs, .. }
  1586→  1586→            | IRInstr::Div { lhs, rhs, .. }
  1587→  1587→            | IRInstr::Cmp { lhs, rhs, .. } => {
  1588→  1588→                let mut r = lhs.as_register().into_iter().collect::<Vec<_>>();
  1589→  1589→                r.extend(rhs.as_register());
  1590→  1590→                r
  1591→  1591→            }
  1592→  1592→            IRInstr::UnaryOp { operand, .. } => operand.as_register().into_iter().collect(),
  1593→  1593→            IRInstr::Call { args, .. } => args.iter().filter_map(|v| v.as_register()).collect(),
  1594→  1594→            IRInstr::Alloc { .. } | IRInstr::GetAddress { .. } => vec![],
  1595→  1595→            IRInstr::Free { ptr } => ptr.as_register().into_iter().collect(),
  1596→  1596→            IRInstr::Cast { src, .. } => src.as_register().into_iter().collect(),
  1597→  1597→            IRInstr::Phi { incoming, .. } => incoming
  1598→  1598→                .iter()
  1599→  1599→                .filter_map(|(v, _)| v.as_register())
  1600→  1600→                .collect(),
  1601→  1601→            IRInstr::Offset { base, offset, .. } => {
  1602→  1602→                let mut r = base.as_register().into_iter().collect::<Vec<_>>();
  1603→  1603→                r.extend(offset.as_register());
  1604→  1604→                r
  1605→  1605→            }
  1606→  1606→            IRInstr::Select {
  1607→  1607→                cond,
  1608→  1608→                true_val,
  1609→  1609→                false_val, ty: _,
  1610→  1610→                ..
  1611→  1611→            } => {
  1612→  1612→                let mut r = cond.as_register().into_iter().collect::<Vec<_>>();
  1613→  1613→                r.extend(true_val.as_register());
  1614→  1614→                r.extend(false_val.as_register());
  1615→  1615→                r
  1616→  1616→            }
  1617→  1617→            IRInstr::Ret { values } => values.iter().filter_map(|v| v.as_register()).collect(),
  1618→  1618→            IRInstr::Branch { .. } => vec![],
  1619→  1619→            IRInstr::CondBranch { cond, .. } => cond.as_register().into_iter().collect(),
  1620→  1620→            IRInstr::CtSelect {
  1621→  1621→                cond,
  1622→  1622→                true_val,
  1623→  1623→                false_val,
  1624→  1624→                ..
  1625→  1625→            } => {
  1626→  1626→                let mut r = cond.as_register().into_iter().collect::<Vec<_>>();
  1627→  1627→                r.extend(true_val.as_register());
  1628→  1628→                r.extend(false_val.as_register());
  1629→  1629→                r
  1630→  1630→            }
  1631→  1631→            IRInstr::CtEq { lhs, rhs, .. } => {
  1632→  1632→                let mut r = lhs.as_register().into_iter().collect::<Vec<_>>();
  1633→  1633→                r.extend(rhs.as_register());
  1634→  1634→                r
  1635→  1635→            }
  1636→  1636→            IRInstr::AtomicLoad { addr, .. } => addr.as_register().into_iter().collect(),
  1637→  1637→            IRInstr::AtomicStore { value, addr, .. } => {
  1638→  1638→                let mut r = value.as_register().into_iter().collect::<Vec<_>>();
  1639→  1639→                r.extend(addr.as_register());
  1640→  1640→                r
  1641→  1641→            }
  1642→  1642→            IRInstr::AtomicCas { addr, expected, desired, .. } => {
  1643→  1643→                let mut r = addr.as_register().into_iter().collect::<Vec<_>>();
  1644→  1644→                r.extend(expected.as_register());
  1645→  1645→                r.extend(desired.as_register());
  1646→  1646→                r
  1647→  1647→            }
  1648→  1648→        }
  1649→  1649→    }
  1650→  1650→}

  1783→  1783→/// A block terminator — the last "instruction" in an `IRBlock` that transfers

  1790→  1790→    /// otherwise go to `false_block`.

  1793→  1793→        cond: IRValue,

  1795→  1795→        true_block: String,

  1802→  1802→    Unreachable,
  1803→  1803→    /// Switch dispatch: branch to one of several targets based on the
  1804→  1804→    /// discriminator value, or fall through to `default`.
  1805→  1805→    Switch {
  1806→  1806→        /// Discriminator value.
  1807→  1807→        discr: IRValue,
  1808→  1808→        /// (value, target_label) pairs.
