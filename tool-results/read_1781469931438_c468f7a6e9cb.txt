    50тЖТ    50тЖТ/// ARM64 register width тАФ selects between 64-bit X registers and 32-bit W
    51тЖТ    51тЖТ/// sub-registers.
    52тЖТ    52тЖТ///
    53тЖТ    53тЖТ/// On AArch64, the 5-bit register encoding is the same for Xn and Wn; the
    54тЖТ    54тЖТ/// instruction's **sf** bit (bit 31) or **size** field (bits 31:30) selects
    55тЖТ    55тЖТ/// the operand width. Using W registers gives automatic 32-bit wrapping
    56тЖТ    56тЖТ/// arithmetic, which is required for algorithms like SHA-256 that operate on
    57тЖТ    57тЖТ/// `u32` values.
    58тЖТ    58тЖТ#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
    59тЖТ    59тЖТpub enum RegWidth {
    60тЖТ    60тЖТ    /// 32-bit W registers (W0тАУW30, WZR, WSP).
    61тЖТ    61тЖТ    W32,
    62тЖТ    62тЖТ    /// 64-bit X registers (X0тАУX30, XZR, SP) тАФ default.
    63тЖТ    63тЖТ    X64,
    64тЖТ    64тЖТ}
    65тЖТ    65тЖТ
    66тЖТ    66тЖТimpl RegWidth {
    67тЖТ    67тЖТ    /// Returns `true` if this is a 32-bit register width.
    68тЖТ    68тЖТ    pub fn is_32bit(&self) -> bool {
    69тЖТ    69тЖТ        matches!(self, RegWidth::W32)
    70тЖТ    70тЖТ    }
    71тЖТ    71тЖТ
    72тЖТ    72тЖТ    /// Returns the **sf** bit value for this width (bit 31 of the encoding).
    73тЖТ    73тЖТ    ///
    74тЖТ    74тЖТ    /// - `X64` тЖТ 1
    75тЖТ    75тЖТ    /// - `W32` тЖТ 0
    76тЖТ    76тЖТ    pub fn sf_bit(&self) -> u32 {
    77тЖТ    77тЖТ        match self {
    78тЖТ    78тЖТ            RegWidth::X64 => 1,
    79тЖТ    79тЖТ            RegWidth::W32 => 0,
    80тЖТ    80тЖТ        }
    81тЖТ    81тЖТ    }
    82тЖТ    82тЖТ
    83тЖТ    83тЖТ    /// Returns the register size in bits (64 or 32).
    84тЖТ    84тЖТ    pub fn bits(&self) -> u32 {
    85тЖТ    85тЖТ        match self {
    86тЖТ    86тЖТ            RegWidth::X64 => 64,
    87тЖТ    87тЖТ            RegWidth::W32 => 32,
    88тЖТ    88тЖТ        }
    89тЖТ    89тЖТ    }
    90тЖТ    90тЖТ
    91тЖТ    91тЖТ    /// Returns the register size in bytes (8 or 4).
    92тЖТ    92тЖТ    pub fn bytes(&self) -> u32 {
    93тЖТ    93тЖТ        self.bits() / 8
    94тЖТ    94тЖТ    }
    95тЖТ    95тЖТ
    96тЖТ    96тЖТ    /// Returns the log2 scale for unsigned-offset load/store encoding.
    97тЖТ    97тЖТ    ///
    98тЖТ    98тЖТ    /// - `X64` тЖТ 3 (offset is divided by 8 for 64-bit LDR/STR)
    99тЖТ    99тЖТ    /// - `W32` тЖТ 2 (offset is divided by 4 for 32-bit LDR/STR)
   100тЖТ   100тЖТ    pub fn scale(&self) -> u32 {
   101тЖТ   101тЖТ        match self {
   102тЖТ   102тЖТ            RegWidth::X64 => 3,
   103тЖТ   103тЖТ            RegWidth::W32 => 2,
   104тЖТ   104тЖТ        }
   105тЖТ   105тЖТ    }
   106тЖТ   106тЖТ
   107тЖТ   107тЖТ    /// Returns the mask value used in UBFM/SBFM for the `imms` field
   108тЖТ   108тЖТ    /// (register size - 1).
   109тЖТ   109тЖТ    pub fn size_minus_1(&self) -> u32 {
   110тЖТ   110тЖТ        match self {
   111тЖТ   111тЖТ            RegWidth::X64 => 63,
   112тЖТ   112тЖТ            RegWidth::W32 => 31,
   113тЖТ   113тЖТ        }
   114тЖТ   114тЖТ    }
   115тЖТ   115тЖТ
   116тЖТ   116тЖТ    /// Derive the register width from an optional IR type.
   117тЖТ   117тЖТ    ///
   118тЖТ   118тЖТ    /// Returns `W32` for `I32`/`U32` types, `X64` otherwise (including `None`).
   119тЖТ   119тЖТ    pub fn from_ir_type(ty: Option<&IRType>) -> RegWidth {

   122тЖТ   122тЖТ            _ => RegWidth::X64,
   123тЖТ   123тЖТ        }
   124тЖТ   124тЖТ    }
   125тЖТ   125тЖТ}

   701тЖТ   701тЖТ/// - **Base + offset**: `LDR Xt, [Xn, #offset]`

   712тЖТ   712тЖТ    PreIndex { base: Register, offset: i32 },

   717тЖТ   717тЖТ    /// the element size.

  1895тЖТ  1895тЖТ    /// Encode this instruction with a specific register width.
  1896тЖТ  1896тЖТ    ///
  1897тЖТ  1897тЖТ    /// For instructions that support 32-bit W-register mode (arithmetic,
  1898тЖТ  1898тЖТ    /// logical, shift, move, load/store, compare, conditional select, and
  1899тЖТ  1899тЖТ    /// bitfield operations), the `width` parameter controls whether the
  1900тЖТ  1900тЖТ    /// sf bit (bit 31) or size field (bits 31:30) is set for 32-bit or
  1901тЖТ  1901тЖТ    /// 64-bit operation.
  1902тЖТ  1902тЖТ    ///
  1903тЖТ  1903тЖТ    /// For instructions that do not have a width variant (branches, system,
  1904тЖТ  1904тЖТ    /// SIMD, etc.), the width parameter is ignored and the default 64-bit
  1905тЖТ  1905тЖТ    /// encoding is used.
  1906тЖТ  1906тЖТ    ///
  1907тЖТ  1907тЖТ    /// Using `RegWidth::W32` produces instructions that operate on W
  1908тЖТ  1908тЖТ    /// sub-registers, giving automatic 32-bit wrapping arithmetic тАФ essential
  1909тЖТ  1909тЖТ    /// for algorithms like SHA-256 that require `u32` modular arithmetic.
  1910тЖТ  1910тЖТ    pub fn encode_with_width(&self, width: RegWidth) -> Result<u32> {
  1911тЖТ  1911тЖТ        let sf = width.sf_bit();
  1912тЖТ  1912тЖТ        match self {
  1913тЖТ  1913тЖТ            // ---- ADD (shifted register): sf 0 0 0 1 0 1 1 shift 0 Rm imm6 Rn Rd ----
  1914тЖТ  1914тЖТ            // ADD (shifted register): sf op S 01011 shift 0 Rm imm6 Rn Rd
  1915тЖТ  1915тЖТ            // 64-bit base=0x8B000000, 32-bit base=0x0B000000
  1916тЖТ  1916тЖТ            Instruction::ADD { rd, rn, rm } => match rm {
  1917тЖТ  1917тЖТ                Operand::Reg { reg, shift } => {
  1918тЖТ  1918тЖТ                    let (hw, imm6) = shift.map(|(k, v)| (k.encoding(), v)).unwrap_or((0, 0));
  1919тЖТ  1919тЖТ                    Ok((sf << 31)
  1920тЖТ  1920тЖТ                        | 0x0B000000u32
  1921тЖТ  1921тЖТ                        | (hw << 22)
  1922тЖТ  1922тЖТ                        | (reg.encoding() << 16)
  1923тЖТ  1923тЖТ                        | (imm6 << 10)
  1924тЖТ  1924тЖТ                        | (rn.encoding() << 5)
  1925тЖТ  1925тЖТ                        | rd.encoding())
  1926тЖТ  1926тЖТ                }
  1927тЖТ  1927тЖТ                // ADD (immediate): sf op S 10001 shift imm12 Rn Rd
  1928тЖТ  1928тЖТ                // 64-bit base=0x91000000, 32-bit base=0x11000000
  1929тЖТ  1929тЖТ                Operand::Imm12(imm) => Ok((sf << 31)
  1930тЖТ  1930тЖТ                    | 0x11000000u32
  1931тЖТ  1931тЖТ                    | ((*imm as u32) << 10)
  1932тЖТ  1932тЖТ                    | (rn.encoding() << 5)
  1933тЖТ  1933тЖТ                    | rd.encoding()),
  1934тЖТ  1934тЖТ            },
  1935тЖТ  1935тЖТ
  1936тЖТ  1936тЖТ            // ---- SUB (shifted register): sf 1 0 0 1 0 1 1 shift 0 Rm imm6 Rn Rd ----
  1937тЖТ  1937тЖТ            // SUB (shifted register): sf op S 01011 shift 0 Rm imm6 Rn Rd
