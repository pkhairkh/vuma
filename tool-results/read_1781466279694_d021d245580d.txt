     1тЖТ     1тЖТ//! Adaptive deployment for the Continuous Optimization Runtime.
     2тЖТ     2тЖТ//!
     3тЖТ     3тЖТ//! The COR can distribute compiled regions across heterogeneous execution
     4тЖТ     4тЖТ//! targets: the local process, remote endpoints (e.g. a cloud compute
     5тЖТ     5тЖТ//! instance), or Raspberry Pi 5 boards running either bare-metal or Linux.
     6тЖТ     6тЖТ//!
     7тЖТ     7тЖТ//! # Core types
     8тЖТ     8тЖТ//!
     9тЖТ     9тЖТ//! - [`DeploymentTarget`] тАФ where to deploy (Local, Pi5Bare, Pi5Linux, Remote).
    10тЖТ    10тЖТ//! - [`DeploymentPackage`] тАФ compiled binary + metadata + debug info.
    11тЖТ    11тЖТ//! - [`DeploymentManager`] тАФ orchestrates deploys, hot-swaps, rollbacks, and
    12тЖТ    12тЖТ//!   delta transfers.
    13тЖТ    13тЖТ//!
    14тЖТ    14тЖТ//! # Hot-swap (Pi 5)
    15тЖТ    15тЖТ//!
    16тЖТ    16тЖТ//! On Raspberry Pi 5 targets, the deployment manager supports *hot-swapping*:
    17тЖТ    17тЖТ//! replacing a region's compiled code while execution continues. A shadow
    18тЖТ    18тЖТ//! buffer is prepared, then atomically swapped in via a fence. The previous
    19тЖТ    19тЖТ//! version is retained for potential rollback.
    20тЖТ    20тЖТ//!
    21тЖТ    21тЖТ//! # Version management
    22тЖТ    22тЖТ//!
    23тЖТ    23тЖТ//! Every successful deployment is recorded in a [`VersionLog`]. The log
    24тЖТ    24тЖТ//! enables rollback to any previously deployed version of a region.
    25тЖТ    25тЖТ//!
    26тЖТ    26тЖТ//! # Delta deployment
    27тЖТ    27тЖТ//!
    28тЖТ    28тЖТ//! When re-deploying a region whose previous version is already present on
    29тЖТ    29тЖТ//! the target, the manager computes a [`DeploymentDelta`] containing only the
    30тЖТ    30тЖТ//! changed bytes, reducing transfer size.
    31тЖТ    31тЖТ

   220тЖТ   220тЖТ/// A delta (diff) between two deployment packages, used for bandwidth-
   221тЖТ   221тЖТ/// efficient re-deployment.
   222тЖТ   222тЖТ#[derive(Debug, Clone, Serialize, Deserialize)]
   223тЖТ   223тЖТpub struct DeploymentDelta {
   224тЖТ   224тЖТ    /// The region this delta applies to.
   225тЖТ   225тЖТ    pub region_id: RegionId,
   226тЖТ   226тЖТ    /// Version we are diffing **from**.
   227тЖТ   227тЖТ    pub from_version: PackageVersion,
   228тЖТ   228тЖТ    /// Version we are diffing **to**.
   229тЖТ   229тЖТ    pub to_version: PackageVersion,
   230тЖТ   230тЖТ    /// Patches: (offset_in_code, replacement_bytes).
   231тЖТ   231тЖТ    /// Offsets that are not listed are unchanged.
   232тЖТ   232тЖТ    pub patches: Vec<(usize, Vec<u8>)>,
   233тЖТ   233тЖТ    /// Bytes that should be **deleted** starting at the given offset.
   234тЖТ   234тЖТ    pub deletions: Vec<(usize, usize)>,
   235тЖТ   235тЖТ    /// Bytes that should be **inserted** at the given offset.
   236тЖТ   236тЖТ    pub insertions: Vec<(usize, Vec<u8>)>,
   237тЖТ   237тЖТ}
   238тЖТ   238тЖТ
   239тЖТ   239тЖТimpl DeploymentDelta {
   240тЖТ   240тЖТ    /// Computes a delta between two code buffers.
   241тЖТ   241тЖТ    ///
   242тЖТ   242тЖТ    /// The algorithm performs a simple block-level diff:
   243тЖТ   243тЖТ    /// - Scans both buffers in [`DELTA_BLOCK_SIZE`] chunks.
   244тЖТ   244тЖТ    /// - Chunks that differ become patches.
   245тЖТ   245тЖТ    /// - Trailing bytes in the new buffer become insertions.
   246тЖТ   246тЖТ    /// - Trailing bytes in the old buffer become deletions.
   247тЖТ   247тЖТ    pub fn compute(
   248тЖТ   248тЖТ        region_id: RegionId,
   249тЖТ   249тЖТ        from_version: PackageVersion,
   250тЖТ   250тЖТ        to_version: PackageVersion,
   251тЖТ   251тЖТ        old_code: &[u8],
   252тЖТ   252тЖТ        new_code: &[u8],
   253тЖТ   253тЖТ    ) -> Self {
   254тЖТ   254тЖТ        const BLOCK: usize = 64;
   255тЖТ   255тЖТ        let min_len = old_code.len().min(new_code.len());
   256тЖТ   256тЖТ        let mut patches = Vec::new();
   257тЖТ   257тЖТ
   258тЖТ   258тЖТ        let mut offset = 0;
   259тЖТ   259тЖТ        while offset + BLOCK <= min_len {
   260тЖТ   260тЖТ            if old_code[offset..offset + BLOCK] != new_code[offset..offset + BLOCK] {
   261тЖТ   261тЖТ                patches.push((offset, new_code[offset..offset + BLOCK].to_vec()));
   262тЖТ   262тЖТ            }
   263тЖТ   263тЖТ            offset += BLOCK;
   264тЖТ   264тЖТ        }
   265тЖТ   265тЖТ        // Handle the tail (< BLOCK bytes).
   266тЖТ   266тЖТ        if offset < min_len && old_code[offset..min_len] != new_code[offset..min_len] {
   267тЖТ   267тЖТ            patches.push((offset, new_code[offset..min_len].to_vec()));
   268тЖТ   268тЖТ        }
   269тЖТ   269тЖТ
   270тЖТ   270тЖТ        let mut deletions = Vec::new();
   271тЖТ   271тЖТ        let mut insertions = Vec::new();
   272тЖТ   272тЖТ
   273тЖТ   273тЖТ        if new_code.len() > old_code.len() {
   274тЖТ   274тЖТ            insertions.push((old_code.len(), new_code[old_code.len()..].to_vec()));
   275тЖТ   275тЖТ        } else if old_code.len() > new_code.len() {
   276тЖТ   276тЖТ            deletions.push((new_code.len(), old_code.len() - new_code.len()));
   277тЖТ   277тЖТ        }
   278тЖТ   278тЖТ
   279тЖТ   279тЖТ        DeploymentDelta {
   280тЖТ   280тЖТ            region_id,
   281тЖТ   281тЖТ            from_version,
   282тЖТ   282тЖТ            to_version,
   283тЖТ   283тЖТ            patches,
   284тЖТ   284тЖТ            deletions,
   285тЖТ   285тЖТ            insertions,
   286тЖТ   286тЖТ        }
   287тЖТ   287тЖТ    }
   288тЖТ   288тЖТ
   289тЖТ   289тЖТ    /// Applies this delta to an old code buffer, producing the new code
   290тЖТ   290тЖТ    /// buffer.
   291тЖТ   291тЖТ    pub fn apply(&self, old_code: &[u8]) -> Result<Vec<u8>, DeploymentError> {
   292тЖТ   292тЖТ        let mut result = old_code.to_vec();
   293тЖТ   293тЖТ
   294тЖТ   294тЖТ        // Apply patches (same-length replacements).
   295тЖТ   295тЖТ        for (offset, data) in &self.patches {
   296тЖТ   296тЖТ            let end = offset.checked_add(data.len()).ok_or_else(|| {
   297тЖТ   297тЖТ                DeploymentError::DeltaApplyFailed(
   298тЖТ   298тЖТ                    self.region_id,
   299тЖТ   299тЖТ                    "patch offset + length overflow".to_owned(),
   300тЖТ   300тЖТ                )
   301тЖТ   301тЖТ            })?;
   302тЖТ   302тЖТ            if end > result.len() {
   303тЖТ   303тЖТ                return Err(DeploymentError::DeltaApplyFailed(
   304тЖТ   304тЖТ                    self.region_id,
   305тЖТ   305тЖТ                    format!(
   306тЖТ   306тЖТ                        "patch at {}..{} exceeds buffer len {}",
   307тЖТ   307тЖТ                        offset,
   308тЖТ   308тЖТ                        end,
   309тЖТ   309тЖТ                        result.len()
   310тЖТ   310тЖТ                    ),
   311тЖТ   311тЖТ                ));
   312тЖТ   312тЖТ            }
   313тЖТ   313тЖТ            result[*offset..end].copy_from_slice(data);
   314тЖТ   314тЖТ        }
   315тЖТ   315тЖТ
   316тЖТ   316тЖТ        // Apply deletions (truncate tail only тАФ ordered back-to-front).
   317тЖТ   317тЖТ        let mut sorted_deletions = self.deletions.clone();
   318тЖТ   318тЖТ        sorted_deletions.sort_by_key(|b| std::cmp::Reverse(b.0));
   319тЖТ   319тЖТ        for (offset, len) in &sorted_deletions {
   320тЖТ   320тЖТ            let end = *offset + len;
   321тЖТ   321тЖТ            if end > result.len() {
   322тЖТ   322тЖТ                return Err(DeploymentError::DeltaApplyFailed(
   323тЖТ   323тЖТ                    self.region_id,
   324тЖТ   324тЖТ                    format!("deletion at {}..{} exceeds buffer", offset, end),
   325тЖТ   325тЖТ                ));
   326тЖТ   326тЖТ            }
   327тЖТ   327тЖТ            result.drain(*offset..*offset + len);
   328тЖТ   328тЖТ        }
   329тЖТ   329тЖТ
   330тЖТ   330тЖТ        // Apply insertions (at specific offsets).
   331тЖТ   331тЖТ        // Sort by offset descending so later insertions don't shift earlier ones.
   332тЖТ   332тЖТ        let mut sorted_insertions = self.insertions.clone();
   333тЖТ   333тЖТ        sorted_insertions.sort_by_key(|b| std::cmp::Reverse(b.0));
   334тЖТ   334тЖТ        for (offset, data) in &sorted_insertions {
   335тЖТ   335тЖТ            if *offset > result.len() {
   336тЖТ   336тЖТ                return Err(DeploymentError::DeltaApplyFailed(
   337тЖТ   337тЖТ                    self.region_id,
   338тЖТ   338тЖТ                    format!(
   339тЖТ   339тЖТ                        "insertion at {} exceeds buffer len {}",
   340тЖТ   340тЖТ                        offset,
   341тЖТ   341тЖТ                        result.len()
   342тЖТ   342тЖТ                    ),
   343тЖТ   343тЖТ                ));
   344тЖТ   344тЖТ            }
   345тЖТ   345тЖТ            result.splice(*offset..*offset, data.iter().copied());
   346тЖТ   346тЖТ        }
   347тЖТ   347тЖТ
   348тЖТ   348тЖТ        Ok(result)
   349тЖТ   349тЖТ    }
   350тЖТ   350тЖТ
   351тЖТ   351тЖТ    /// Returns an estimate of the delta size in bytes (for bandwidth
   352тЖТ   352тЖТ    /// planning).
   353тЖТ   353тЖТ    pub fn estimated_size(&self) -> usize {
   354тЖТ   354тЖТ        self.patches.iter().map(|(_, d)| d.len()).sum::<usize>()
   355тЖТ   355тЖТ            + self.insertions.iter().map(|(_, d)| d.len()).sum::<usize>()
   356тЖТ   356тЖТ    }
   357тЖТ   357тЖТ
   358тЖТ   358тЖТ    /// Returns true if the delta contains no changes.
   359тЖТ   359тЖТ    pub fn is_empty(&self) -> bool {

   559тЖТ   559тЖТ/// Computes deployment plans and manages live migrations.
   560тЖТ   560тЖТ#[derive(Debug)]
   561тЖТ   561тЖТpub struct DeploymentPlanner {
   562тЖТ   562тЖТ    /// Current deployment plan.
   563тЖТ   563тЖТ    plan: DeploymentPlan,

   569тЖТ   569тЖТ    /// Creates a new deployment planner with the given configuration.
   570тЖТ   570тЖТ    pub fn new(config: Config) -> Self {
   571тЖТ   571тЖТ        DeploymentPlanner {
   572тЖТ   572тЖТ            plan: DeploymentPlan::new(),
   573тЖТ   573тЖТ            config,
   574тЖТ   574тЖТ        }
   575тЖТ   575тЖТ    }
   576тЖТ   576тЖТ
   577тЖТ   577тЖТ    /// Computes a new deployment plan based on profile data and the
   578тЖТ   578тЖТ    /// current configuration.
   579тЖТ   579тЖТ    ///
   580тЖТ   580тЖТ    /// # Strategy
   581тЖТ   581тЖТ    ///
   582тЖТ   582тЖТ    /// 1. **Hot regions** (high call count) are deployed locally for
   583тЖТ   583тЖТ    ///    minimal latency.
   584тЖТ   584тЖТ    /// 2. **Cold regions** are candidates for remote execution.
   585тЖТ   585тЖТ    /// 3. On Pi 5 targets, regions are round-robin-assigned to cores.
   586тЖТ   586тЖТ    pub fn compute_deployment_plan(
   587тЖТ   587тЖТ        &mut self,
   588тЖТ   588тЖТ        region_ids: &[RegionId],
   589тЖТ   589тЖТ        profile: &ProfileData,
   590тЖТ   590тЖТ    ) -> &DeploymentPlan {
   591тЖТ   591тЖТ        self.plan.regions.clear();
   592тЖТ   592тЖТ
   593тЖТ   593тЖТ        let hot_threshold = 100;
   594тЖТ   594тЖТ
   595тЖТ   595тЖТ        for &region_id in region_ids {
   596тЖТ   596тЖТ            let _is_hot = profile
   597тЖТ   597тЖТ                .call_counts
   598тЖТ   598тЖТ                .values()
   599тЖТ   599тЖТ                .any(|&count| count > hot_threshold);
   600тЖТ   600тЖТ
   601тЖТ   601тЖТ            let target = if self.config.is_pi5_target() {
   602тЖТ   602тЖТ                let core_id = (region_id % 4) as u32;
   603тЖТ   603тЖТ                if self.config.enable_speculative {
   604тЖТ   604тЖТ                    DeploymentTarget::Pi5Bare {
   605тЖТ   605тЖТ                        board_id: "default".to_owned(),
   606тЖТ   606тЖТ                        core_id,
   607тЖТ   607тЖТ                    }
   608тЖТ   608тЖТ                } else {
   609тЖТ   609тЖТ                    DeploymentTarget::Pi5Linux {
   610тЖТ   610тЖТ                        host: "localhost".to_owned(),
   611тЖТ   611тЖТ                        core_affinity: Some(core_id),
   612тЖТ   612тЖТ                    }
   613тЖТ   613тЖТ                }
   614тЖТ   614тЖТ            } else {
   615тЖТ   615тЖТ                DeploymentTarget::Local
   616тЖТ   616тЖТ            };
   617тЖТ   617тЖТ
   618тЖТ   618тЖТ            self.plan.regions.push((region_id, target));
   619тЖТ   619тЖТ        }
   620тЖТ   620тЖТ
   621тЖТ   621тЖТ        &self.plan
   622тЖТ   622тЖТ    }
   623тЖТ   623тЖТ
   624тЖТ   624тЖТ    /// Migrates a single region to a new target.
   625тЖТ   625тЖТ    pub fn migrate_region(
   626тЖТ   626тЖТ        &mut self,
   627тЖТ   627тЖТ        region_id: RegionId,
   628тЖТ   628тЖТ        new_target: DeploymentTarget,
   629тЖТ   629тЖТ    ) -> Result<(), DeploymentError> {
   630тЖТ   630тЖТ        let entry = self
   631тЖТ   631тЖТ            .plan
   632тЖТ   632тЖТ            .regions
   633тЖТ   633тЖТ            .iter_mut()
   634тЖТ   634тЖТ            .find(|(r, _)| *r == region_id)
   635тЖТ   635тЖТ            .ok_or(DeploymentError::RegionNotFound(region_id))?;
   636тЖТ   636тЖТ
   637тЖТ   637тЖТ        let old_target = entry.1.clone();
   638тЖТ   638тЖТ        entry.1 = new_target.clone();
   639тЖТ   639тЖТ
   640тЖТ   640тЖТ        log::info!(
   641тЖТ   641тЖТ            "Migrated region {} from {} to {}",
   642тЖТ   642тЖТ            region_id,
   643тЖТ   643тЖТ            old_target,
   644тЖТ   644тЖТ            new_target
   645тЖТ   645тЖТ        );
   646тЖТ   646тЖТ
   647тЖТ   647тЖТ        Ok(())
   648тЖТ   648тЖТ    }
   649тЖТ   649тЖТ
   650тЖТ   650тЖТ    /// Rebalances the deployment plan by re-running the planner logic.
   651тЖТ   651тЖТ    pub fn rebalance(&mut self, region_ids: &[RegionId], profile: &ProfileData) -> &DeploymentPlan {
   652тЖТ   652тЖТ        log::info!(
   653тЖТ   653тЖТ            "Rebalancing deployment plan for {} regions",
   654тЖТ   654тЖТ            region_ids.len()
   655тЖТ   655тЖТ        );
   656тЖТ   656тЖТ        self.compute_deployment_plan(region_ids, profile)
   657тЖТ   657тЖТ    }

   660тЖТ   660тЖТ    pub fn plan(&self) -> &DeploymentPlan {
   661тЖТ   661тЖТ        &self.plan
   662тЖТ   662тЖТ    }

   668тЖТ   668тЖТ
   669тЖТ   669тЖТ/// Manages the deployment lifecycle of compiled regions.
   670тЖТ   670тЖТ///
   671тЖТ   671тЖТ/// The `DeploymentManager` is the central orchestrator for:
   672тЖТ   672тЖТ/// - Deploying packages to targets ([`deploy`](DeploymentManager::deploy)).
   673тЖТ   673тЖТ/// - Hot-swapping running code on Pi 5 targets
   674тЖТ   674тЖТ///   ([`hot_swap`](DeploymentManager::hot_swap)).
   675тЖТ   675тЖТ/// - Version tracking and rollback
   676тЖТ   676тЖТ///   ([`rollback`](DeploymentManager::rollback)).
   677тЖТ   677тЖТ/// - Delta-based re-deployment
   678тЖТ   678тЖТ///   ([`deploy_delta`](DeploymentManager::deploy_delta)).
   679тЖТ   679тЖТ#[derive(Debug)]
   680тЖТ   680тЖТpub struct DeploymentManager {
   681тЖТ   681тЖТ    /// Runtime configuration.
   682тЖТ   682тЖТ    config: Config,
   683тЖТ   683тЖТ    /// Version history for all deployed regions.
   684тЖТ   684тЖТ    version_log: VersionLog,
   685тЖТ   685тЖТ    /// Hot-swap state per region.
   686тЖТ   686тЖТ    hot_swap_states: HashMap<RegionId, HotSwapState>,
   687тЖТ   687тЖТ    /// Deployment planner for computing placement.
   688тЖТ   688тЖТ    planner: DeploymentPlanner,
   689тЖТ   689тЖТ}
   690тЖТ   690тЖТ
   691тЖТ   691тЖТimpl DeploymentManager {
   692тЖТ   692тЖТ    /// Creates a new deployment manager with the given configuration.
   693тЖТ   693тЖТ    pub fn new(config: Config) -> Self {
   694тЖТ   694тЖТ        let planner = DeploymentPlanner::new(config.clone());
   695тЖТ   695тЖТ        DeploymentManager {
   696тЖТ   696тЖТ            config,
   697тЖТ   697тЖТ            version_log: VersionLog::new(),
   698тЖТ   698тЖТ            hot_swap_states: HashMap::new(),
   699тЖТ   699тЖТ            planner,
   700тЖТ   700тЖТ        }
   701тЖТ   701тЖТ    }
   702тЖТ   702тЖТ
   703тЖТ   703тЖТ    /// Deploys a package to the specified target.
   704тЖТ   704тЖТ    ///
   705тЖТ   705тЖТ    /// This is the primary deployment entry point. It:
   706тЖТ   706тЖТ    /// 1. Validates the package checksum.
   707тЖТ   707тЖТ    /// 2. If a previous version exists on the target and the target supports
   708тЖТ   708тЖТ    ///    hot-swap, performs a hot-swap.
   709тЖТ   709тЖТ    /// 3. Otherwise performs a full deployment.
   710тЖТ   710тЖТ    /// 4. Records the version in the log.
   711тЖТ   711тЖТ    ///
   712тЖТ   712тЖТ    /// # Errors
   713тЖТ   713тЖТ    ///
   714тЖТ   714тЖТ    /// Returns [`DeploymentError::ChecksumMismatch`] if the package checksum
   715тЖТ   715тЖТ    /// is invalid, or [`DeploymentError::TargetUnreachable`] if the target
   716тЖТ   716тЖТ    /// cannot be reached (simulated).
   717тЖТ   717тЖТ    pub fn deploy(
   718тЖТ   718тЖТ        &mut self,
   719тЖТ   719тЖТ        package: &DeploymentPackage,
   720тЖТ   720тЖТ        target: &DeploymentTarget,
   721тЖТ   721тЖТ    ) -> Result<DeploymentResult, DeploymentError> {
   722тЖТ   722тЖТ        // Step 1: Validate checksum.
   723тЖТ   723тЖТ        if !package.validate_checksum() {
   724тЖТ   724тЖТ            return Err(DeploymentError::ChecksumMismatch(
   725тЖТ   725тЖТ                package.metadata.region_id,
   726тЖТ   726тЖТ            ));
   727тЖТ   727тЖТ        }
   728тЖТ   728тЖТ
   729тЖТ   729тЖТ        let start = Instant::now();
   730тЖТ   730тЖТ
   731тЖТ   731тЖТ        // Step 2: Check if we can hot-swap.
   732тЖТ   732тЖТ        let region_id = package.metadata.region_id;
   733тЖТ   733тЖТ        let previous = self.version_log.latest(region_id).cloned();
   734тЖТ   734тЖТ        let mut hot_swapped = false;
   735тЖТ   735тЖТ
   736тЖТ   736тЖТ        if let Some(ref prev) = previous {
   737тЖТ   737тЖТ            if target.supports_hot_swap() && prev.target == *target {
   738тЖТ   738тЖТ                // Hot-swap path.
   739тЖТ   739тЖТ                self.hot_swap(package, target)?;
   740тЖТ   740тЖТ                hot_swapped = true;
   741тЖТ   741тЖТ            }
   742тЖТ   742тЖТ        }
   743тЖТ   743тЖТ
   744тЖТ   744тЖТ        // Step 3: Record version.
   745тЖТ   745тЖТ        self.version_log.record(package, target);
   746тЖТ   746тЖТ
   747тЖТ   747тЖТ        let duration = start.elapsed();
   748тЖТ   748тЖТ
   749тЖТ   749тЖТ        Ok(DeploymentResult {
   750тЖТ   750тЖТ            region_id,
   751тЖТ   751тЖТ            version: package.metadata.version.clone(),
   752тЖТ   752тЖТ            target: target.clone(),
   753тЖТ   753тЖТ            duration,
   754тЖТ   754тЖТ            bytes_transferred: package.metadata.code_size,
   755тЖТ   755тЖТ            hot_swapped,
   756тЖТ   756тЖТ            was_delta: false,
   757тЖТ   757тЖТ        })
   758тЖТ   758тЖТ    }
   759тЖТ   759тЖТ
   760тЖТ   760тЖТ    /// Performs a hot-swap of a running region on a Pi 5 target.
   761тЖТ   761тЖТ    ///
   762тЖТ   762тЖТ    /// The hot-swap proceeds through these phases:
   763тЖТ   763тЖТ    /// 1. **PreparingShadow** тАФ allocate and fill shadow buffer with new code.
   764тЖТ   764тЖТ    /// 2. **AwaitingSafePoint** тАФ wait until no threads are in the region.
   765тЖТ   765тЖТ    /// 3. **Swapping** тАФ atomically swap the code pointer.
   766тЖТ   766тЖТ    /// 4. **Completed** тАФ old code is retained in the version log for
   767тЖТ   767тЖТ    ///    rollback.
   768тЖТ   768тЖТ    ///
   769тЖТ   769тЖТ    /// In this simulation the phases advance immediately. A production
   770тЖТ   770тЖТ    /// implementation would coordinate with the runtime's safe-point
   771тЖТ   771тЖТ    /// mechanism.
   772тЖТ   772тЖТ    pub fn hot_swap(
   773тЖТ   773тЖТ        &mut self,
   774тЖТ   774тЖТ        package: &DeploymentPackage,
   775тЖТ   775тЖТ        target: &DeploymentTarget,
   776тЖТ   776тЖТ    ) -> Result<HotSwapPhase, DeploymentError> {
   777тЖТ   777тЖТ        if !target.supports_hot_swap() {
   778тЖТ   778тЖТ            return Err(DeploymentError::HotSwapNotSupported(
   779тЖТ   779тЖТ                target.kind_label().to_owned(),
   780тЖТ   780тЖТ            ));
   781тЖТ   781тЖТ        }
   782тЖТ   782тЖТ
   783тЖТ   783тЖТ        let region_id = package.metadata.region_id;
   784тЖТ   784тЖТ        let from_version = self
   785тЖТ   785тЖТ            .version_log
   786тЖТ   786тЖТ            .latest(region_id)
   787тЖТ   787тЖТ            .map(|r| r.version.clone())
   788тЖТ   788тЖТ            .unwrap_or(PackageVersion(0));
   789тЖТ   789тЖТ
   790тЖТ   790тЖТ        let state = self
   791тЖТ   791тЖТ            .hot_swap_states
   792тЖТ   792тЖТ            .entry(region_id)
   793тЖТ   793тЖТ            .or_insert_with(|| HotSwapState::new(region_id));
   794тЖТ   794тЖТ
   795тЖТ   795тЖТ        // Phase 1: Prepare shadow buffer.
   796тЖТ   796тЖТ        state.phase = HotSwapPhase::PreparingShadow;
   797тЖТ   797тЖТ        state.from_version = from_version.clone();
   798тЖТ   798тЖТ        state.to_version = package.metadata.version.clone();
   799тЖТ   799тЖТ
   800тЖТ   800тЖТ        // Phase 2: Await safe point (simulated тАФ immediate).
   801тЖТ   801тЖТ        state.phase = HotSwapPhase::AwaitingSafePoint;
   802тЖТ   802тЖТ
   803тЖТ   803тЖТ        // Phase 3: Atomic swap.
   804тЖТ   804тЖТ        state.phase = HotSwapPhase::Swapping;
   805тЖТ   805тЖТ
   806тЖТ   806тЖТ        // Phase 4: Completed.
   807тЖТ   807тЖТ        state.phase = HotSwapPhase::Completed;
   808тЖТ   808тЖТ
   809тЖТ   809тЖТ        log::info!(
   810тЖТ   810тЖТ            "Hot-swapped region {} from {} to {} on {}",
   811тЖТ   811тЖТ            region_id,
   812тЖТ   812тЖТ            from_version,
   813тЖТ   813тЖТ            package.metadata.version,
   814тЖТ   814тЖТ            target
   815тЖТ   815тЖТ        );
   816тЖТ   816тЖТ
   817тЖТ   817тЖТ        Ok(HotSwapPhase::Completed)
   818тЖТ   818тЖТ    }
   819тЖТ   819тЖТ
   820тЖТ   820тЖТ    /// Rolls back a region to the previous deployed version.
   821тЖТ   821тЖТ    ///
   822тЖТ   822тЖТ    /// The previous version's code is re-deployed to the same target. The
   823тЖТ   823тЖТ    /// rollback itself is recorded in the version log.
   824тЖТ   824тЖТ    pub fn rollback(&mut self, region_id: RegionId) -> Result<DeploymentResult, DeploymentError> {
   825тЖТ   825тЖТ        let prev_record = self
   826тЖТ   826тЖТ            .version_log
   827тЖТ   827тЖТ            .previous(region_id)
   828тЖТ   828тЖТ            .cloned()
   829тЖТ   829тЖТ            .ok_or(DeploymentError::NoRollbackTarget(region_id))?;
   830тЖТ   830тЖТ
   831тЖТ   831тЖТ        // Determine the new version number: must be strictly greater than
   832тЖТ   832тЖТ        // the current latest.
   833тЖТ   833тЖТ        let latest_version = self
   834тЖТ   834тЖТ            .version_log
   835тЖТ   835тЖТ            .latest(region_id)
   836тЖТ   836тЖТ            .map(|r| r.version.0)
   837тЖТ   837тЖТ            .unwrap_or(0);
   838тЖТ   838тЖТ
   839тЖТ   839тЖТ        // Build a package from the previous record.
   840тЖТ   840тЖТ        let rollback_pkg = DeploymentPackage::new(
   841тЖТ   841тЖТ            prev_record.code.clone(),
   842тЖТ   842тЖТ            prev_record.region_id,
   843тЖТ   843тЖТ            // Bump version: new version = latest + 1.
   844тЖТ   844тЖТ            PackageVersion(latest_version + 1),
   845тЖТ   845тЖТ            "rollback",
   846тЖТ   846тЖТ        );
   847тЖТ   847тЖТ
   848тЖТ   848тЖТ        let target = prev_record.target.clone();
   849тЖТ   849тЖТ        self.deploy(&rollback_pkg, &target)
   850тЖТ   850тЖТ    }
   851тЖТ   851тЖТ
   852тЖТ   852тЖТ    /// Deploys only the delta (diff) between the current deployed version
   853тЖТ   853тЖТ    /// and the new package.
   854тЖТ   854тЖТ    ///
   855тЖТ   855тЖТ    /// This is more bandwidth-efficient than a full deploy when the target
   856тЖТ   856тЖТ    /// already holds a previous version. The delta is computed, sent, and
   857тЖТ   857тЖТ    /// applied on the "remote" side (simulated locally).
   858тЖТ   858тЖТ    pub fn deploy_delta(
   859тЖТ   859тЖТ        &mut self,
   860тЖТ   860тЖТ        package: &DeploymentPackage,
   861тЖТ   861тЖТ        target: &DeploymentTarget,
   862тЖТ   862тЖТ    ) -> Result<DeploymentResult, DeploymentError> {
   863тЖТ   863тЖТ        let region_id = package.metadata.region_id;
   864тЖТ   864тЖТ
   865тЖТ   865тЖТ        let prev_code = self
   866тЖТ   866тЖТ            .version_log
   867тЖТ   867тЖТ            .latest(region_id)
   868тЖТ   868тЖТ            .map(|r| r.code.clone())
   869тЖТ   869тЖТ            .ok_or(DeploymentError::NoPreviousVersion(region_id))?;
   870тЖТ   870тЖТ
   871тЖТ   871тЖТ        let from_version = self
   872тЖТ   872тЖТ            .version_log
   873тЖТ   873тЖТ            .latest(region_id)
   874тЖТ   874тЖТ            .map(|r| r.version.clone())
   875тЖТ   875тЖТ            .unwrap_or(PackageVersion(0));
   876тЖТ   876тЖТ
   877тЖТ   877тЖТ        // Compute delta.
   878тЖТ   878тЖТ        let delta = DeploymentDelta::compute(
   879тЖТ   879тЖТ            region_id,
   880тЖТ   880тЖТ            from_version,
   881тЖТ   881тЖТ            package.metadata.version.clone(),
   882тЖТ   882тЖТ            &prev_code,
   883тЖТ   883тЖТ            &package.code,
   884тЖТ   884тЖТ        );
   885тЖТ   885тЖТ
   886тЖТ   886тЖТ        // Simulate applying the delta on the remote side.
   887тЖТ   887тЖТ        let reconstructed = delta
   888тЖТ   888тЖТ            .apply(&prev_code)
   889тЖТ   889тЖТ            .map_err(|e| DeploymentError::DeltaApplyFailed(region_id, format!("{:?}", e)))?;
   890тЖТ   890тЖТ
   891тЖТ   891тЖТ        // Verify the reconstructed code matches.
   892тЖТ   892тЖТ        if reconstructed != package.code {
   893тЖТ   893тЖТ            return Err(DeploymentError::DeltaApplyFailed(
   894тЖТ   894тЖТ                region_id,
   895тЖТ   895тЖТ                "reconstructed code does not match expected".to_owned(),
   896тЖТ   896тЖТ            ));
   897тЖТ   897тЖТ        }
   898тЖТ   898тЖТ
   899тЖТ   899тЖТ        let start = Instant::now();
   900тЖТ   900тЖТ        let delta_size = delta.estimated_size();
   901тЖТ   901тЖТ
   902тЖТ   902тЖТ        // If the delta is not empty, we also try hot-swap on supported
   903тЖТ   903тЖТ        // targets.
   904тЖТ   904тЖТ        let mut hot_swapped = false;
   905тЖТ   905тЖТ        if target.supports_hot_swap() && !delta.is_empty() {
   906тЖТ   906тЖТ            // For a delta hot-swap, we reconstruct on the target side and
   907тЖТ   907тЖТ            // then swap. Here we just flag it.
   908тЖТ   908тЖТ            hot_swapped = true;
   909тЖТ   909тЖТ        }
   910тЖТ   910тЖТ
   911тЖТ   911тЖТ        // Record the version (full code stored for future deltas).
   912тЖТ   912тЖТ        self.version_log.record(package, target);
   913тЖТ   913тЖТ
   914тЖТ   914тЖТ        let duration = start.elapsed();
   915тЖТ   915тЖТ
   916тЖТ   916тЖТ        Ok(DeploymentResult {
   917тЖТ   917тЖТ            region_id,
   918тЖТ   918тЖТ            version: package.metadata.version.clone(),
   919тЖТ   919тЖТ            target: target.clone(),
   920тЖТ   920тЖТ            duration,
   921тЖТ   921тЖТ            bytes_transferred: delta_size,
   922тЖТ   922тЖТ            hot_swapped,
   923тЖТ   923тЖТ            was_delta: true,
   924тЖТ   924тЖТ        })
   925тЖТ   925тЖТ    }
   926тЖТ   926тЖТ

   997тЖТ   997тЖТ/// Simple CRC32 computation (stub тАФ a real implementation would use the
   998тЖТ   998тЖТ/// `crc32fast` crate or hardware CRC). We use a basic table-driven CRC32
   999тЖТ   999тЖТ/// for determinism in tests.
  1000тЖТ  1000тЖТfn crc32(data: &[u8]) -> u32 {
  1001тЖТ  1001тЖТ    // CRC32 (IEEE 802.3 polynomial: 0xEDB88320 reflected).
  1002тЖТ  1002тЖТ    const POLY: u32 = 0xEDB88320;
  1003тЖТ  1003тЖТ    let mut table = [0u32; 256];
  1004тЖТ  1004тЖТ    for i in 0..256u32 {
  1005тЖТ  1005тЖТ        let mut crc = i;
  1006тЖТ  1006тЖТ        for _ in 0..8 {
  1007тЖТ  1007тЖТ            if crc & 1 != 0 {
  1008тЖТ  1008тЖТ                crc = (crc >> 1) ^ POLY;
  1009тЖТ  1009тЖТ            } else {
  1010тЖТ  1010тЖТ                crc >>= 1;
  1011тЖТ  1011тЖТ            }
  1012тЖТ  1012тЖТ        }
  1013тЖТ  1013тЖТ        table[i as usize] = crc;
  1014тЖТ  1014тЖТ    }
  1015тЖТ  1015тЖТ
  1016тЖТ  1016тЖТ    let mut crc: u32 = 0xFFFF_FFFF;
  1017тЖТ  1017тЖТ    for &byte in data {
  1018тЖТ  1018тЖТ        let index = ((crc ^ byte as u32) & 0xFF) as usize;
  1019тЖТ  1019тЖТ        crc = (crc >> 8) ^ table[index];
  1020тЖТ  1020тЖТ    }
  1021тЖТ  1021тЖТ    !crc
  1022тЖТ  1022тЖТ}

  1024тЖТ  1024тЖТ/// Returns the current time as seconds since the Unix epoch (best-effort).
  1025тЖТ  1025тЖТfn epoch_secs() -> u64 {
  1026тЖТ  1026тЖТ    std::time::SystemTime::now()

  1028тЖТ  1028тЖТ        .map(|d| d.as_secs())
  1029тЖТ  1029тЖТ        .unwrap_or(0)
  1030тЖТ  1030тЖТ}
