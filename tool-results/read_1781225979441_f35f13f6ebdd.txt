     1→//! VUMA CLI — Command-line interface for the VUMA compiler.
     2→//!
     3→//! Subcommands:
     4→//! - `vuma build <file>` — Parse + compile to ARM64 ELF (default), save to output file
     5→//! - `vuma run <file>`   — Build + execute (via QEMU aarch64 or native)
     6→//! - `vuma check <file>` — Parse + SCG + BD inference + IVE verification only
     7→//! - `vuma emit <isa> <file>` — Compile to specific ISA
     8→//! - `vuma disasm <file>` — Read binary and disassemble
     9→//! - `vuma verify <file>` — Run IVE 5-invariant verification
    10→//! - `vuma repl` — Interactive REPL (parse expr, print AST)
    11→

   226→/// Read source file content with a human-readable error on failure.
   227→fn read_source(path: &PathBuf) -> Result<String, String> {
   228→    fs::read_to_string(path)
   229→        .map_err(|e| format!("error: cannot read source file '{}': {}", path.display(), e))
   230→}

   232→/// Build a `CompileConfig` from the global CLI flags.
   233→fn make_config(cli: &Cli, target: CompileTarget) -> CompileConfig {
   234→    CompileConfig {
   235→        target,
   236→        opt_level: OptLevel::from(cli.opt_level),
   237→        verification_level: VerificationLevel::from(cli.verification),
   238→        debug_info: cli.debug,
   239→        ..CompileConfig::default()
   240→    }
   241→}

   243→/// Print compilation errors to stderr in a human-readable format.
   244→fn print_errors(errors: &[VumaError]) {
   245→    for err in errors {
   246→        eprintln!("error[{}]: {}", err.stage(), err);
   247→    }
   248→}

   250→/// Determine the default output path for a given input file.
   251→fn default_output_path(input: &Path) -> PathBuf {
   252→    let stem = input.file_stem().unwrap_or_default().to_string_lossy();
   253→    let dir = input.parent().unwrap_or(std::path::Path::new("."));
   254→    dir.join(format!("{}.o", stem))
   255→}

   261→/// `vuma build <file>` — Parse + compile to ELF, save to output file.
   262→fn cmd_build(
   263→    cli: &Cli,
   264→    file: &PathBuf,
   265→    output: &Option<PathBuf>,
   266→    target: TargetArg,
   267→) -> Result<(), String> {
   268→    let source = read_source(file)?;
   269→    let config = make_config(cli, CompileTarget::from(target));
   270→    let result = compile(&source, &config).map_err(|errors| {
   271→        print_errors(&errors);
   272→        format!("compilation failed with {} error(s)", errors.len())
   273→    })?;
   274→
   275→    let out_path = output
   276→        .as_ref()
   277→        .cloned()
   278→        .unwrap_or_else(|| default_output_path(file));
   279→    fs::write(&out_path, &result.binary).map_err(|e| {
   280→        format!(
   281→            "error: cannot write output file '{}': {}",
   282→            out_path.display(),
   283→            e
   284→        )
   285→    })?;
   286→
   287→    println!(
   288→        "Compiled {} -> {} ({} bytes, {} SCG nodes, {} IR instructions)",
   289→        file.display(),
   290→        out_path.display(),
   291→        result.binary.len(),
   292→        result.scg.node_count(),
   293→        result.ir_instruction_count,
   294→    );
   295→
   296→    // Print stage timings.
   297→    for (stage, ms) in &result.stage_timings {
   298→        println!("  {:20} {}ms", stage, ms);
   299→    }
   300→
   301→    Ok(())
   302→}

   304→/// `vuma run <file>` — Build + execute.
   305→fn cmd_run(cli: &Cli, file: &PathBuf, args: &[String]) -> Result<(), String> {
   306→    let source = read_source(file)?;
   307→    let config = make_config(cli, CompileTarget::Pi5Linux);
   308→    let result = compile(&source, &config).map_err(|errors| {
   309→        print_errors(&errors);
   310→        format!("compilation failed with {} error(s)", errors.len())
   311→    })?;
   312→
   313→    // Write binary to a temp file.
   314→    let tmp_dir = std::env::temp_dir();
   315→    let exe_path = tmp_dir.join(format!("vuma_run_{}", std::process::id()));
   316→    fs::write(&exe_path, &result.binary).map_err(|e| {
   317→        format!(
   318→            "error: cannot write temporary executable '{}': {}",
   319→            exe_path.display(),
   320→            e
   321→        )
   322→    })?;
   323→
   324→    // Make the file executable on Unix.
   325→    #[cfg(unix)]
   326→    {
   327→        use std::os::unix::fs::PermissionsExt;
   328→        let mut perms = fs::metadata(&exe_path)
   329→            .map_err(|e| format!("error: cannot stat '{}': {}", exe_path.display(), e))?
   330→            .permissions();
   331→        perms.set_mode(0o755);
   332→        fs::set_permissions(&exe_path, perms)
   333→            .map_err(|e| format!("error: cannot chmod '{}': {}", exe_path.display(), e))?;
   334→    }
   335→
   336→    // Try to execute: first natively (if on aarch64), then via qemu-aarch64.
   337→    let native_output = Command::new(&exe_path).args(args).output();
   338→
   339→    let output = match native_output {
   340→        Ok(out) => out,
   341→        Err(_) => {
   342→            // Native execution failed; try qemu-aarch64.
   343→            let qemu_output = Command::new("qemu-aarch64")
   344→                .arg(&exe_path)
   345→                .args(args)
   346→                .output()
   347→                .map_err(|e| {
   348→                    format!(
   349→                        "error: failed to execute binary (neither native nor qemu-aarch64): {}",
   350→                        e
   351→                    )
   352→                })?;
   353→            qemu_output
   354→        }
   355→    };
   356→
   357→    io::stdout()
   358→        .write_all(&output.stdout)
   359→        .map_err(|e| format!("error: failed to write program output: {}", e))?;
   360→    io::stderr()
   361→        .write_all(&output.stderr)
   362→        .map_err(|e| format!("error: failed to write program stderr: {}", e))?;
   363→
   364→    // Clean up temp file.
   365→    let _ = fs::remove_file(&exe_path);
   366→
   367→    let code = output.status.code().unwrap_or(1);
   368→    if code != 0 {
   369→        return Err(format!("program exited with code {}", code));
   370→    }
   371→
   372→    Ok(())
   373→}

   375→/// `vuma check <file>` — Parse + SCG + BD inference + IVE verification only.
   376→fn cmd_check(cli: &Cli, file: &PathBuf) -> Result<(), String> {
   377→    let source = read_source(file)?;
   378→    let mut config = make_config(cli, CompileTarget::Pi5Linux);
   379→    // check mode: always run verification, don't skip
   380→    if config.verification_level == VerificationLevel::None {
   381→        config.verification_level = VerificationLevel::Normal;
   382→    }
   383→
   384→    // Run the full compile — the check command doesn't save the binary,
   385→    // it just verifies the program compiles and passes verification.
   386→    let result = compile(&source, &config).map_err(|errors| {
   387→        print_errors(&errors);
   388→        format!("check failed with {} error(s)", errors.len())
   389→    })?;
   390→
   391→    println!("Check passed for {}", file.display());
   392→    println!("  SCG nodes:      {}", result.scg.node_count());
   393→    println!("  IR functions:   {}", result.ir_function_count);
   394→    println!("  IR instructions: {}", result.ir_instruction_count);
   395→
   396→    if let Some(ref verification) = result.verification {
   397→        println!("  Verification:   {}", verification.overall);
   398→    }
   399→
   400→    // Print stage timings.
   401→    for (stage, ms) in &result.stage_timings {
   402→        println!("  {:20} {}ms", stage, ms);
   403→    }
   404→
   405→    Ok(())
   406→}

   408→/// `vuma emit <isa> <file>` — Compile to a specific ISA target.
   409→fn cmd_emit(
   410→    cli: &Cli,
   411→    isa: IsaArg,
   412→    file: &PathBuf,
   413→    output: &Option<PathBuf>,
   414→) -> Result<(), String> {
   415→    let source = read_source(file)?;
   416→    let backend_kind = BackendKind::from(isa);
   417→    let config = make_config(cli, CompileTarget::Pi5Linux);
   418→    let result = compile(&source, &config).map_err(|errors| {
   419→        print_errors(&errors);
   420→        format!("compilation failed with {} error(s)", errors.len())
   421→    })?;
   422→
   423→    // Use the multi-arch backend to produce ISA-specific output.
   424→    let backend = create_backend(backend_kind).map_err(|e| {
   425→        format!(
   426→            "error: cannot create {} backend: {}",
   427→            backend_kind.isa_name(),
   428→            e
   429→        )
   430→    })?;
   431→
   432→    // Allocate registers and encode using the target backend.
   433→    let mut allocated_functions = Vec::new();
   434→    if let Some(ref debug_info) = result.debug_info {
   435→        if let Some(ref ir_program) = debug_info.ir_pre_regalloc {
   436→            for func in &ir_program.functions {
   437→                match backend.allocate_registers(func) {
   438→                    Ok(allocated) => allocated_functions.push(allocated),
   439→                    Err(e) => {
   440→                        eprintln!(
   441→                            "warning: register allocation failed for '{}': {}",
   442→                            func.name, e
   443→                        );
   444→                    }
   445→                }
   446→            }
   447→        }
   448→    }
   449→
   450→    let out_path = output
   451→        .as_ref()
   452→        .cloned()
   453→        .unwrap_or_else(|| default_output_path(file));
   454→
   455→    // If we have allocated functions, encode them; otherwise write the ARM64 binary.
   456→    if !allocated_functions.is_empty() {
   457→        let allocated_program = vuma_codegen::backend::AllocatedProgram {
   458→            functions: allocated_functions,
   459→            total_code_size: 0,
   460→            total_data_size: 0,
   461→        };
   462→        match backend.encode_program(&allocated_program) {
   463→            Ok(bytes) => {
   464→                fs::write(&out_path, &bytes).map_err(|e| {
   465→                    format!(
   466→                        "error: cannot write output file '{}': {}",
   467→                        out_path.display(),
   468→                        e
   469→                    )
   470→                })?;
   471→                println!(
   472→                    "Emitted {} -> {} ({} bytes, ISA: {})",
   473→                    file.display(),
   474→                    out_path.display(),
   475→                    bytes.len(),
   476→                    backend.name(),
   477→                );
   478→            }
   479→            Err(e) => {
   480→                // Backend encode failed; fall back to writing the ARM64 ELF.
   481→                eprintln!(
   482→                    "warning: {} encoding failed ({}), writing ARM64 ELF instead",
   483→                    backend.name(),
   484→                    e
   485→                );
   486→                fs::write(&out_path, &result.binary).map_err(|e2| {
   487→                    format!(
   488→                        "error: cannot write output file '{}': {}",
   489→                        out_path.display(),
   490→                        e2
   491→                    )
   492→                })?;
   493→                println!(
   494→                    "Emitted {} -> {} ({} bytes, ARM64 ELF fallback)",
   495→                    file.display(),
   496→                    out_path.display(),
   497→                    result.binary.len(),
   498→                );
   499→            }
   500→        }
   501→    } else {
   502→        // No IR program available (debug info not captured), write the ARM64 binary.
   503→        fs::write(&out_path, &result.binary).map_err(|e| {
   504→            format!(
   505→                "error: cannot write output file '{}': {}",
   506→                out_path.display(),
   507→                e
   508→            )
   509→        })?;
   510→        println!(
   511→            "Emitted {} -> {} ({} bytes, ARM64 ELF)",
   512→            file.display(),
   513→            out_path.display(),
   514→            result.binary.len(),
   515→        );
   516→    }
   517→
   518→    Ok(())
   519→}

   521→/// `vuma disasm <file>` — Read binary and disassemble.
   522→fn cmd_disasm(file: &PathBuf, isa: IsaArg, base_addr_str: &str) -> Result<(), String> {
   523→    let bytes = fs::read(file)
   524→        .map_err(|e| format!("error: cannot read binary file '{}': {}", file.display(), e))?;
   525→
   526→    let base_addr = u64::from_str_radix(base_addr_str.trim_start_matches("0x"), 16)
   527→        .map_err(|e| format!("error: invalid base address '{}': {}", base_addr_str, e))?;
   528→
   529→    let backend_kind = BackendKind::from(isa);
   530→    let backend = create_backend(backend_kind).map_err(|e| {
   531→        format!(
   532→            "error: cannot create {} backend: {}",
   533→            backend_kind.isa_name(),
   534→            e
   535→        )
   536→    })?;
   537→
   538→    let instructions = backend.disassemble(&bytes, base_addr);
   539→
   540→    println!(
   541→        "Disassembly of {} ({} bytes, ISA: {}):",
   542→        file.display(),
   543→        bytes.len(),
   544→        backend.name()
   545→    );
   546→    for line in &instructions {
   547→        println!("{}", line);
   548→    }
   549→
   550→    Ok(())
   551→}

   553→/// `vuma verify <file>` — Run IVE 5-invariant verification.
   554→fn cmd_verify(cli: &Cli, file: &PathBuf) -> Result<(), String> {
   555→    let source = read_source(file)?;
   556→    let mut config = make_config(cli, CompileTarget::Pi5Linux);
   557→    // Force exhaustive verification for the verify subcommand.
   558→    config.verification_level = VerificationLevel::Exhaustive;
   559→
   560→    let result = compile(&source, &config).map_err(|errors| {
   561→        print_errors(&errors);
   562→        format!(
   563→            "compilation/verification failed with {} error(s)",
   564→            errors.len()
   565→        )
   566→    })?;
   567→
   568→    match result.verification {
   569→        Some(ref verification) => {
   570→            println!("IVE Verification Results for {}", file.display());
   571→            println!("  Overall verdict: {}", verification.overall);
   572→            println!(
   573→                "  Summary: {} invariant(s) checked",
   574→                verification.summary.total_checked
   575→            );
   576→            println!("           {} passed", verification.summary.passed);
   577→            println!("           {} failed", verification.summary.failed);
   578→            println!("           {} unverified", verification.summary.unverified);
   579→
   580→            // Print individual invariant results.
   581→            for per_inv in &verification.per_invariant {
   582→                let status_str = match &per_inv.result.status {
   583→                    vuma_ive::VerificationStatus::Proven => "PROVEN",
   584→                    vuma_ive::VerificationStatus::ProbablySafe { .. } => "PROBABLY_SAFE",
   585→                    vuma_ive::VerificationStatus::Unverified { .. } => "UNVERIFIED",
   586→                    vuma_ive::VerificationStatus::Violated { .. } => "VIOLATED",
   587→                };
   588→                println!(
   589→                    "  [{}] {:?} — {}",
   590→                    status_str, per_inv.kind, per_inv.result.message,
   591→                );
   592→                if let Some(ref evidence) = per_inv.result.evidence {
   593→                    println!("    evidence: {}", evidence);
   594→                }
   595→            }
   596→        }
   597→        None => {
   598→            println!("Verification was skipped (verification level set to None).");
   599→        }
   600→    }
   601→
   602→    Ok(())
   603→}
   604→
   605→/// `vuma repl` — Interactive REPL: parse expressions and print AST.
   606→fn cmd_repl() -> Result<(), String> {
   607→    println!("VUMA REPL v0.1.0");
   608→    println!("Type VUMA expressions or statements. Enter ':quit' to exit, ':help' for help.");
   609→
   610→    let stdin = io::stdin();
   611→    let mut input = String::new();
   612→
   613→    loop {
   614→        print!("vuma> ");
   615→        io::stdout()
   616→            .flush()
   617→            .map_err(|e| format!("flush error: {}", e))?;
   618→
   619→        input.clear();
   620→        match stdin.read_line(&mut input) {
   621→            Ok(0) => break, // EOF
   622→            Ok(_) => {}
   623→            Err(e) => return Err(format!("error reading input: {}", e)),
   624→        }
   625→
   626→        let trimmed = input.trim();
   627→        if trimmed.is_empty() {
   628→            continue;
   629→        }
   630→        if trimmed == ":quit" || trimmed == ":q" {
   631→            break;
   632→        }
   633→        if trimmed == ":help" || trimmed == ":h" {
   634→            println!("  :quit, :q   — Exit the REPL");
   635→            println!("  :help, :h   — Show this help");
   636→            println!("  <expr>      — Parse and display the AST");
   637→            println!("  <stmt>      — Parse and display the AST");
   638→            println!("  <program>   — Parse and display the full program AST");
   639→            continue;
   640→        }
   641→
   642→        // Try to parse as a full program first.
   643→        let mut parser = vuma_parser::Parser::new(trimmed);
   644→        let result = parser.parse_program();
   645→        if !result.has_errors() {
   646→            // Successfully parsed; display the AST.
   647→            let program = result.unwrap();
   648→            println!("{:#?}", program);
   649→            continue;
   650→        }
   651→        // Program parse had errors; try to parse as an expression via
   652→        // wrapping it in a dummy program context.
   653→        let program_errors = result.errors.clone();
   654→        // Since parse_expr is private, we re-parse as a minimal program.
   655→        let wrapped = format!("fn _repl_expr() {{ {} }}", trimmed);
   656→        let mut expr_parser = vuma_parser::Parser::new(&wrapped);
   657→        let expr_result = expr_parser.parse_program();
   658→        if !expr_result.has_errors() {
   659→            // Print the function body AST.
   660→            let program = expr_result.unwrap();
   661→            if let Some(item) = program.items.first() {
   662→                println!("{:#?}", item);
   663→            } else {
   664→                println!("{:#?}", program);
   665→            }
   666→        } else {
   667→            // Show the original parse errors.
   668→            for err in &program_errors {
   669→                eprintln!("parse error: {}", err);
   670→            }
   671→        }
   672→    }
   673→
   674→    Ok(())
   675→}

   677→// ═══════════════════════════════════════════════════════════════════════════
   678→// Main entry point
   679→// ═══════════════════════════════════════════════════════════════════════════
   680→
   681→fn main() {
   682→    // Initialize logger.
   683→    env_logger::init();
   684→
   685→    let cli = Cli::parse();
   686→
   687→    let result = match cli.command {
   688→        Commands::Build {
   689→            ref file,
   690→            ref output,
   691→            target,
   692→        } => cmd_build(&cli, file, output, target),
   693→        Commands::Run { ref file, ref args } => cmd_run(&cli, file, args),
   694→        Commands::Check { ref file } => cmd_check(&cli, file),
   695→        Commands::Emit {
   696→            isa,
   697→            ref file,
   698→            ref output,
   699→        } => cmd_emit(&cli, isa, file, output),
   700→        Commands::Disasm {
   701→            ref file,
   702→            isa,
   703→            ref base_addr,
   704→        } => cmd_disasm(file, isa, base_addr),
   705→        Commands::Verify { ref file } => cmd_verify(&cli, file),
   706→        Commands::Repl => cmd_repl(),
   707→    };
   708→
   709→    if let Err(err) = result {
   710→        eprintln!("{}", err);
   711→        std::process::exit(1);
   712→    }
   713→}
