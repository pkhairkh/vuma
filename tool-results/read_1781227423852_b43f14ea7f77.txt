   242тЖТ   242тЖТ/// Generate a REX prefix byte.
   243тЖТ   243тЖТ///
   244тЖТ   244тЖТ/// - `w`: REX.W тАФ 64-bit operand size
   245тЖТ   245тЖТ/// - `r`: Extension of the ModR/M `reg` field (for R8тАУR15 / XMM8тАУXMM15)
   246тЖТ   246тЖТ/// - `x`: Extension of the SIB `index` field
   247тЖТ   247тЖТ/// - `b`: Extension of the ModR/M `rm` field or SIB `base` field
   248тЖТ   248тЖТ///
   249тЖТ   249тЖТ/// Returns `None` if no REX byte is needed (all bits are 0).
   250тЖТ   250тЖТfn rex_prefix(w: bool, r: bool, x: bool, b: bool) -> Option<u8> {
   251тЖТ   251тЖТ    let byte = 0x40 | (w as u8) << 3 | (r as u8) << 2 | (x as u8) << 1 | b as u8;
   252тЖТ   252тЖТ    if byte > 0x40 {
   253тЖТ   253тЖТ        Some(byte)
   254тЖТ   254тЖТ    } else {
   255тЖТ   255тЖТ        None
   256тЖТ   256тЖТ    }
   257тЖТ   257тЖТ}

   262тЖТ   262тЖТ
   263тЖТ   263тЖТ/// Encode a ModR/M byte.
   264тЖТ   264тЖТ///
   265тЖТ   265тЖТ/// - `mod_bits`: 2-bit mod field (0=mem, 1=mem+disp8, 2=mem+disp32, 3=reg)
   266тЖТ   266тЖТ/// - `reg`: 3-bit reg field (register or opcode extension)
   267тЖТ   267тЖТ/// - `rm`: 3-bit r/m field (register or memory operand)
   268тЖТ   268тЖТfn modrm(mod_bits: u8, reg: u8, rm: u8) -> u8 {
   269тЖТ   269тЖТ    (mod_bits & 3) << 6 | (reg & 7) << 3 | (rm & 7)
   270тЖТ   270тЖТ}
   271тЖТ   271тЖТ
   272тЖТ   272тЖТ/// Encode a SIB byte.
   273тЖТ   273тЖТ///
   274тЖТ   274тЖТ/// - `scale`: 2-bit scale factor (0=1, 1=2, 2=4, 3=8)
   275тЖТ   275тЖТ/// - `index`: 3-bit index register
   276тЖТ   276тЖТ/// - `base`: 3-bit base register
   277тЖТ   277тЖТfn sib(scale: u8, index: u8, base: u8) -> u8 {
   278тЖТ   278тЖТ    (scale & 3) << 6 | (index & 7) << 3 | (base & 7)
   279тЖТ   279тЖТ}

   285тЖТ   285тЖТ/// Emit a REX.W prefix plus opcode, then a ModR/M byte for reg-reg operations.
   286тЖТ   286тЖТ///
   287тЖТ   287тЖТ/// This is the common pattern for 64-bit ALU instructions: REX.W + opcode + ModR/M(mod=3, reg, rm).
   288тЖТ   288тЖТfn emit_rexw_reg_reg(code: &mut Vec<u8>, opcode: u8, reg: Gpr, rm: Gpr) {
   289тЖТ   289тЖТ    let r = reg.needs_rex();
   290тЖТ   290тЖТ    let b = rm.needs_rex();
   291тЖТ   291тЖТ    if let Some(rex) = rex_prefix(true, r, false, b) {
   292тЖТ   292тЖТ        code.push(rex);
   293тЖТ   293тЖТ    } else {
   294тЖТ   294тЖТ        // REX.W is still needed for 64-bit operations even if r and b are 0
   295тЖТ   295тЖТ        code.push(0x48);
   296тЖТ   296тЖТ    }
   297тЖТ   297тЖТ    code.push(opcode);
   298тЖТ   298тЖТ    code.push(modrm(3, reg.encoding() & 7, rm.encoding() & 7));
   299тЖТ   299тЖТ}
   300тЖТ   300тЖТ
   301тЖТ   301тЖТ/// Emit a REX.W prefix (always), then opcode, then ModR/M for reg-reg with
   302тЖТ   302тЖТ/// specific reg field (opcode extension) and rm register.
   303тЖТ   303тЖТfn emit_rexw_opext_reg(code: &mut Vec<u8>, opcode: u8, opext: u8, rm: Gpr) {
   304тЖТ   304тЖТ    let b = rm.needs_rex();
   305тЖТ   305тЖТ    // Always emit REX.W for 64-bit
   306тЖТ   306тЖТ    let rex = 0x48 | (b as u8);
   307тЖТ   307тЖТ    code.push(rex);
   308тЖТ   308тЖТ    code.push(opcode);
   309тЖТ   309тЖТ    code.push(modrm(3, opext & 7, rm.encoding() & 7));
   310тЖТ   310тЖТ}

   312тЖТ   312тЖТ/// Encode MOV r64, r64 (REX.W + 89 /r)
   313тЖТ   313тЖТpub fn encode_mov_reg_reg(dst: Gpr, src: Gpr) -> Vec<u8> {
   314тЖТ   314тЖТ    let mut code = Vec::with_capacity(3);
   315тЖТ   315тЖТ    emit_rexw_reg_reg(&mut code, 0x89, src, dst);
   316тЖТ   316тЖТ    code
   317тЖТ   317тЖТ}

   319тЖТ   319тЖТ/// Encode MOV r64, imm64 (REX.W + B8+rd + 8-byte imm)
   320тЖТ   320тЖТpub fn encode_mov_reg_imm64(dst: Gpr, imm: u64) -> Vec<u8> {
   321тЖТ   321тЖТ    let mut code = Vec::with_capacity(10);
   322тЖТ   322тЖТ    let b = dst.needs_rex();
   323тЖТ   323тЖТ    if let Some(rex) = rex_prefix(true, false, false, b) {
   324тЖТ   324тЖТ        code.push(rex);
   325тЖТ   325тЖТ    } else {
   326тЖТ   326тЖТ        code.push(0x48);
   327тЖТ   327тЖТ    }
   328тЖТ   328тЖТ    code.push(0xB8 + (dst.encoding() & 7));
   329тЖТ   329тЖТ    code.extend_from_slice(&imm.to_le_bytes());
   330тЖТ   330тЖТ    code
   331тЖТ   331тЖТ}

   333тЖТ   333тЖТ/// Encode MOV r64, imm32 (REX.W + C7 /0 + 4-byte imm, sign-extended)
   334тЖТ   334тЖТpub fn encode_mov_reg_imm32(dst: Gpr, imm: i32) -> Vec<u8> {
   335тЖТ   335тЖТ    let mut code = Vec::with_capacity(7);
   336тЖТ   336тЖТ    let b = dst.needs_rex();
   337тЖТ   337тЖТ    if let Some(rex) = rex_prefix(true, false, false, b) {
   338тЖТ   338тЖТ        code.push(rex);
   339тЖТ   339тЖТ    } else {
   340тЖТ   340тЖТ        code.push(0x48);
   341тЖТ   341тЖТ    }
   342тЖТ   342тЖТ    code.push(0xC7);

   348тЖТ   348тЖТ/// Encode MOV r64, [r64+offset] (REX.W + 8B /r + displacement)
   349тЖТ   349тЖТ///
   350тЖТ   350тЖТ/// Handles special cases:
   351тЖТ   351тЖТ/// - RSP/R12 as base: SIB byte required
   352тЖТ   352тЖТ/// - RBP/R13 as base: mod=1 with disp8=0 even for zero offset
   353тЖТ   353тЖТ/// - Offset fits in i8: disp8; otherwise: disp32
   354тЖТ   354тЖТpub fn encode_mov_reg_mem(dst: Gpr, base: Gpr, offset: i32) -> Vec<u8> {
   355тЖТ   355тЖТ    let mut code = Vec::with_capacity(8);
   356тЖТ   356тЖТ    let r = dst.needs_rex();
   357тЖТ   357тЖТ    let b = base.needs_rex();
   358тЖТ   358тЖТ
   359тЖТ   359тЖТ    if let Some(rex) = rex_prefix(true, r, false, b) {
   360тЖТ   360тЖТ        code.push(rex);
   361тЖТ   361тЖТ    } else {
   362тЖТ   362тЖТ        code.push(0x48);
   363тЖТ   363тЖТ    }
   364тЖТ   364тЖТ    code.push(0x8B);
   365тЖТ   365тЖТ
   366тЖТ   366тЖТ    let needs_sib = base == Gpr::Rsp || base == Gpr::R12;
   367тЖТ   367тЖТ    let needs_disp8_for_zero = base == Gpr::Rbp || base == Gpr::R13;
   368тЖТ   368тЖТ
   369тЖТ   369тЖТ    if offset == 0 && !needs_disp8_for_zero && !needs_sib {
   370тЖТ   370тЖТ        // mod=00, no displacement
   371тЖТ   371тЖТ        code.push(modrm(0, dst.encoding() & 7, base.encoding() & 7));
   372тЖТ   372тЖТ    } else if needs_sib {
   373тЖТ   373тЖТ        // SIB byte required: base = RSP(4), index = RSP(4) means "no index"
   374тЖТ   374тЖТ        if offset == 0 {
   375тЖТ   375тЖТ            code.push(modrm(0, dst.encoding() & 7, 4));
   376тЖТ   376тЖТ            code.push(sib(0, 4, base.encoding() & 7));
   377тЖТ   377тЖТ        } else if (-128..=127).contains(&offset) {
   378тЖТ   378тЖТ            code.push(modrm(1, dst.encoding() & 7, 4));
   379тЖТ   379тЖТ            code.push(sib(0, 4, base.encoding() & 7));
   380тЖТ   380тЖТ            code.push(offset as u8);
   381тЖТ   381тЖТ        } else {
   382тЖТ   382тЖТ            code.push(modrm(2, dst.encoding() & 7, 4));
   383тЖТ   383тЖТ            code.push(sib(0, 4, base.encoding() & 7));
   384тЖТ   384тЖТ            code.extend_from_slice(&offset.to_le_bytes());
   385тЖТ   385тЖТ        }
   386тЖТ   386тЖТ    } else if (-128..=127).contains(&offset) {
   387тЖТ   387тЖТ        // mod=01, disp8
   388тЖТ   388тЖТ        code.push(modrm(1, dst.encoding() & 7, base.encoding() & 7));
   389тЖТ   389тЖТ        code.push(offset as u8);
   390тЖТ   390тЖТ    } else {
   391тЖТ   391тЖТ        // mod=10, disp32
   392тЖТ   392тЖТ        code.push(modrm(2, dst.encoding() & 7, base.encoding() & 7));
   393тЖТ   393тЖТ        code.extend_from_slice(&offset.to_le_bytes());
   394тЖТ   394тЖТ    }
   395тЖТ   395тЖТ
   396тЖТ   396тЖТ    code
   397тЖТ   397тЖТ}

   848тЖТ   848тЖТ
   849тЖТ   849тЖТ/// Decode x86_64 bytes into mnemonic strings with (offset, mnemonic) pairs.
   850тЖТ   850тЖТ///
   851тЖТ   851тЖТ/// Handles the top 20+ most common x86_64 instructions including mov, add, sub,
   852тЖТ   852тЖТ/// push, pop, call, ret, jmp, cmp, test, lea, xor, and, or, shl, shr, nop,
   853тЖТ   853тЖТ/// mul, div, imul.
   854тЖТ   854тЖТfn disassemble_x86_64_mnemonic(bytes: &[u8], addr: u64) -> Vec<String> {
   855тЖТ   855тЖТ    let mut lines = Vec::new();
   856тЖТ   856тЖТ    let mut offset = 0usize;
   857тЖТ   857тЖТ    let mut pc = addr;
   858тЖТ   858тЖТ
   859тЖТ   859тЖТ    while offset < bytes.len() {
   860тЖТ   860тЖТ        let start = offset;
   861тЖТ   861тЖТ        let start_pc = pc;
   862тЖТ   862тЖТ        let mut pos = offset;
   863тЖТ   863тЖТ
   864тЖТ   864тЖТ        // Skip legacy prefixes
   865тЖТ   865тЖТ        while pos < bytes.len() && matches!(bytes[pos], 0x66 | 0x67 | 0xF2 | 0xF3) {
   866тЖТ   866тЖТ            pos += 1;
   867тЖТ   867тЖТ        }
   868тЖТ   868тЖТ
   869тЖТ   869тЖТ        // REX prefix
   870тЖТ   870тЖТ        let mut rex = 0u8;
   871тЖТ   871тЖТ        let mut _rex_w = false;
   872тЖТ   872тЖТ        let mut rex_r = false;
   873тЖТ   873тЖТ        let mut rex_b = false;
   874тЖТ   874тЖТ        if pos < bytes.len() && bytes[pos] >= 0x40 && bytes[pos] <= 0x4F {
   875тЖТ   875тЖТ            rex = bytes[pos];
   876тЖТ   876тЖТ            _rex_w = (rex & 0x08) != 0;
   877тЖТ   877тЖТ            rex_r = (rex & 0x04) != 0;
   878тЖТ   878тЖТ            rex_b = (rex & 0x01) != 0;
   879тЖТ   879тЖТ            pos += 1;
   880тЖТ   880тЖТ        }
   881тЖТ   881тЖТ
   882тЖТ   882тЖТ        if pos >= bytes.len() {
   883тЖТ   883тЖТ            let end = pos.min(bytes.len());
   884тЖТ   884тЖТ            let hex_bytes: Vec<String> = bytes[start..end]
   885тЖТ   885тЖТ                .iter()
   886тЖТ   886тЖТ                .map(|b| format!("{:02x}", b))
   887тЖТ   887тЖТ                .collect();
   888тЖТ   888тЖТ            lines.push(format!("{:#010x}:  {}", start_pc, hex_bytes.join(" ")));
   889тЖТ   889тЖТ            offset = end;
   890тЖТ   890тЖТ            pc = start_pc + (end - start) as u64;
   891тЖТ   891тЖТ            continue;
   892тЖТ   892тЖТ        }
   893тЖТ   893тЖТ
   894тЖТ   894тЖТ        let opcode = bytes[pos];
   895тЖТ   895тЖТ        pos += 1;
   896тЖТ   896тЖТ
   897тЖТ   897тЖТ        let mnemonic = match opcode {
   898тЖТ   898тЖТ            // NOP
   899тЖТ   899тЖТ            0x90 => "nop".to_string(),
   900тЖТ   900тЖТ
   901тЖТ   901тЖТ            // RET
   902тЖТ   902тЖТ            0xC3 => "ret".to_string(),
   903тЖТ   903тЖТ
   904тЖТ   904тЖТ            // INT3
   905тЖТ   905тЖТ            0xCC => "int3".to_string(),
   906тЖТ   906тЖТ
   907тЖТ   907тЖТ            // PUSH r64
   908тЖТ   908тЖТ            0x50..=0x57 => {
   909тЖТ   909тЖТ                let reg_idx = (opcode - 0x50) | (if rex_b { 8 } else { 0 });
   910тЖТ   910тЖТ                format!("push {}", gpr_name_64(reg_idx))
   911тЖТ   911тЖТ            }
   912тЖТ   912тЖТ
   913тЖТ   913тЖТ            // POP r64
   914тЖТ   914тЖТ            0x58..=0x5F => {
   915тЖТ   915тЖТ                let reg_idx = (opcode - 0x58) | (if rex_b { 8 } else { 0 });
   916тЖТ   916тЖТ                format!("pop {}", gpr_name_64(reg_idx))
   917тЖТ   917тЖТ            }
   918тЖТ   918тЖТ
   919тЖТ   919тЖТ            // MOV r64, imm64 (B8+rd)
   920тЖТ   920тЖТ            0xB8..=0xBF => {
   921тЖТ   921тЖТ                let reg_idx = (opcode - 0xB8) | (if rex_b { 8 } else { 0 });
   922тЖТ   922тЖТ                if pos + 8 <= bytes.len() {
   923тЖТ   923тЖТ                    let imm = u64::from_le_bytes(bytes[pos..pos + 8].try_into().unwrap_or([0; 8]));
   924тЖТ   924тЖТ                    pos += 8;
   925тЖТ   925тЖТ                    format!("mov {}, {:#x}", gpr_name_64(reg_idx), imm)
   926тЖТ   926тЖТ                } else {
   927тЖТ   927тЖТ                    pos = bytes.len();
   928тЖТ   928тЖТ                    format!("mov {}, ???", gpr_name_64(reg_idx))
   929тЖТ   929тЖТ                }
   930тЖТ   930тЖТ            }
   931тЖТ   931тЖТ
   932тЖТ   932тЖТ            // JMP rel32
   933тЖТ   933тЖТ            0xE9 => {
   934тЖТ   934тЖТ                if pos + 4 <= bytes.len() {
   935тЖТ   935тЖТ                    let rel = i32::from_le_bytes(bytes[pos..pos + 4].try_into().unwrap_or([0; 4]));
   936тЖТ   936тЖТ                    pos += 4;
   937тЖТ   937тЖТ                    format!(
   938тЖТ   938тЖТ                        "jmp {:#x}",
   939тЖТ   939тЖТ                        (start_pc + (pos - start) as u64).wrapping_add(rel as u64)
   940тЖТ   940тЖТ                    )
   941тЖТ   941тЖТ                } else {
   942тЖТ   942тЖТ                    pos = bytes.len();
   943тЖТ   943тЖТ                    "jmp ???".to_string()
   944тЖТ   944тЖТ                }
   945тЖТ   945тЖТ            }
   946тЖТ   946тЖТ
   947тЖТ   947тЖТ            // CALL rel32
   948тЖТ   948тЖТ            0xE8 => {
   949тЖТ   949тЖТ                if pos + 4 <= bytes.len() {
   950тЖТ   950тЖТ                    let rel = i32::from_le_bytes(bytes[pos..pos + 4].try_into().unwrap_or([0; 4]));
   951тЖТ   951тЖТ                    pos += 4;
   952тЖТ   952тЖТ                    format!(
   953тЖТ   953тЖТ                        "call {:#x}",
   954тЖТ   954тЖТ                        (start_pc + (pos - start) as u64).wrapping_add(rel as u64)
   955тЖТ   955тЖТ                    )
   956тЖТ   956тЖТ                } else {
   957тЖТ   957тЖТ                    pos = bytes.len();
   958тЖТ   958тЖТ                    "call ???".to_string()
   959тЖТ   959тЖТ                }
   960тЖТ   960тЖТ            }
   961тЖТ   961тЖТ
   962тЖТ   962тЖТ            // Two-byte opcode (0F xx)
   963тЖТ   963тЖТ            0x0F => {
   964тЖТ   964тЖТ                if pos >= bytes.len() {
   965тЖТ   965тЖТ                    "0f ???".to_string()
   966тЖТ   966тЖТ                } else {
   967тЖТ   967тЖТ                    let op2 = bytes[pos];
   968тЖТ   968тЖТ                    pos += 1;
   969тЖТ   969тЖТ                    match op2 {
   970тЖТ   970тЖТ                        // SYSCALL
   971тЖТ   971тЖТ                        0x05 => "syscall".to_string(),
   972тЖТ   972тЖТ                        // IMUL r64, r64
   973тЖТ   973тЖТ                        0xAF => {
   974тЖТ   974тЖТ                            let (r, rm, new_pos) = decode_modrm_reg_rm(bytes, pos, rex_r, rex_b);
   975тЖТ   975тЖТ                            pos = new_pos;
   976тЖТ   976тЖТ                            format!("imul {}, {}", gpr_name_64(r), gpr_name_64(rm))
   977тЖТ   977тЖТ                        }
   978тЖТ   978тЖТ                        // Jcc rel32
   979тЖТ   979тЖТ                        0x80..=0x8F => {
   980тЖТ   980тЖТ                            let cc_name = match op2 & 0xF {
   981тЖТ   981тЖТ                                0 => "jo",
   982тЖТ   982тЖТ                                1 => "jno",

  1380тЖТ  1380тЖТfn decode_modrm_reg_rm(bytes: &[u8], pos: usize, rex_r: bool, rex_b: bool) -> (u8, u8, usize) {

  1382тЖТ  1382тЖТ        return (0, 0, pos);
  1383тЖТ  1383тЖТ    }
  1384тЖТ  1384тЖТ    let modrm = bytes[pos];

  1387тЖТ  1387тЖТ    let reg = ((modrm >> 3) & 7) | (if rex_r { 8 } else { 0 });
  1388тЖТ  1388тЖТ    let rm = (modrm & 7) | (if rex_b { 8 } else { 0 });
  1389тЖТ  1389тЖТ
  1390тЖТ  1390тЖТ    if mod_bits == 3 {
  1391тЖТ  1391тЖТ        // Register-register
  1392тЖТ  1392тЖТ        (reg, rm, new_pos)
  1393тЖТ  1393тЖТ    } else {

  1395тЖТ  1395тЖТ        (reg, rm, new_pos)
  1396тЖТ  1396тЖТ    }
  1397тЖТ  1397тЖТ}

  1403тЖТ  1403тЖТ/// Build a minimal ELF64 binary for x86_64 from raw code bytes.
  1404тЖТ  1404тЖТ///
  1405тЖТ  1405тЖТ/// Produces a static executable with a single LOAD segment containing the
  1406тЖТ  1406тЖТ/// `.text` section. Entry point is at `base_addr` + header offset.
  1407тЖТ  1407тЖТfn build_minimal_x86_64_elf(code: &[u8], base_addr: u64) -> Vec<u8> {
  1408тЖТ  1408тЖТ    let elf_header_size: u64 = 64;
  1409тЖТ  1409тЖТ    let phdr_size: u64 = 56;
  1410тЖТ  1410тЖТ    let text_offset = elf_header_size + phdr_size;
  1411тЖТ  1411тЖТ    let text_size = code.len() as u64;
  1412тЖТ  1412тЖТ    let entry_point = base_addr + text_offset;
  1413тЖТ  1413тЖТ
  1414тЖТ  1414тЖТ    let mut elf = Vec::with_capacity(text_offset as usize + code.len());
  1415тЖТ  1415тЖТ
  1416тЖТ  1416тЖТ    // --- e_ident ---
  1417тЖТ  1417тЖТ    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1418тЖТ  1418тЖТ    elf.push(2); // ELFCLASS64
  1419тЖТ  1419тЖТ    elf.push(1); // ELFDATA2LSB
  1420тЖТ  1420тЖТ    elf.push(1); // EV_CURRENT
  1421тЖТ  1421тЖТ    elf.push(3); // ELFOSABI_LINUX
  1422тЖТ  1422тЖТ    elf.push(0); // padding
  1423тЖТ  1423тЖТ    elf.extend_from_slice(&[0u8; 7]); // padding
  1424тЖТ  1424тЖТ
  1425тЖТ  1425тЖТ    // --- ELF header fields ---
  1426тЖТ  1426тЖТ    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1427тЖТ  1427тЖТ    elf.extend_from_slice(&62u16.to_le_bytes()); // e_machine = EM_X86_64
  1428тЖТ  1428тЖТ    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1429тЖТ  1429тЖТ    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1430тЖТ  1430тЖТ    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1431тЖТ  1431тЖТ    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1432тЖТ  1432тЖТ    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1433тЖТ  1433тЖТ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1434тЖТ  1434тЖТ    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1435тЖТ  1435тЖТ    elf.extend_from_slice(&1u16.to_le_bytes()); // e_phnum
  1436тЖТ  1436тЖТ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1437тЖТ  1437тЖТ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1438тЖТ  1438тЖТ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1439тЖТ  1439тЖТ
  1440тЖТ  1440тЖТ    // --- Program Header (single LOAD segment: PF_R | PF_X) ---
  1441тЖТ  1441тЖТ    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1442тЖТ  1442тЖТ    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1443тЖТ  1443тЖТ    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1444тЖТ  1444тЖТ    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1445тЖТ  1445тЖТ    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1446тЖТ  1446тЖТ    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1447тЖТ  1447тЖТ    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1448тЖТ  1448тЖТ    elf.extend_from_slice(&16u64.to_le_bytes()); // p_align
  1449тЖТ  1449тЖТ
  1450тЖТ  1450тЖТ    // --- Code section ---
  1451тЖТ  1451тЖТ    elf.extend_from_slice(code);
  1452тЖТ  1452тЖТ
  1453тЖТ  1453тЖТ    elf
  1454тЖТ  1454тЖТ}

  1480тЖТ  1480тЖТ/// Compute the stack frame size for an IR function on x86_64.
  1481тЖТ  1481тЖТ///
  1482тЖТ  1482тЖТ/// Sums `Alloc` instruction sizes, adds 8 bytes for the RBP save,
  1483тЖТ  1483тЖТ/// and rounds up to 16-byte alignment.
  1484тЖТ  1484тЖТfn x86_64_compute_frame_size(func: &IRFunction) -> usize {
  1485тЖТ  1485тЖТ    let mut total: usize = 8; // Saved RBP
  1486тЖТ  1486тЖТ    for block in &func.blocks {
  1487тЖТ  1487тЖТ        for instr in &block.instructions {
  1488тЖТ  1488тЖТ            if let crate::ir::IRInstr::Alloc { size, .. } = instr {
  1489тЖТ  1489тЖТ                let aligned = (*size as usize).div_ceil(16) * 16;
  1490тЖТ  1490тЖТ                total += aligned;
  1491тЖТ  1491тЖТ            }
  1492тЖТ  1492тЖТ        }
  1493тЖТ  1493тЖТ    }
  1494тЖТ  1494тЖТ    // Round up to 16-byte alignment
  1495тЖТ  1495тЖТ    (total + 15) & !15
  1496тЖТ  1496тЖТ}
