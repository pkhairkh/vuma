   215тЖТ
   216тЖТ/// Compute the absolute value of a signed 64-bit integer.
   217тЖТ///
   218тЖТ/// Returns the magnitude of `x` without its sign. For non-negative inputs,
   219тЖТ/// the result is identical to the input. For negative inputs, the result
   220тЖТ/// is `-x`.
   221тЖТ///
   222тЖТ/// ## Edge Case: i64::MIN
   223тЖТ///
   224тЖТ/// `i64::MIN` (-9223372036854775808) has no positive counterpart in the
   225тЖТ/// `i64` range. Calling `abs(i64::MIN)` returns `i64::MIN` itself,
   226тЖТ/// matching the behavior of Rust's `i64::wrapping_abs()`. This is a
   227тЖТ/// known overflow that callers should handle explicitly if it may occur.
   228тЖТ///
   229тЖТ/// ## VUMA Program Equivalent
   230тЖТ///
   231тЖТ/// ```vuma
   232тЖТ/// fn abs(x: i64) -> i64 {
   233тЖТ///     if x < 0 {
   234тЖТ///         return -x;
   235тЖТ///     }
   236тЖТ///     return x;
   237тЖТ/// }
   238тЖТ/// ```
   239тЖТ///
   240тЖТ/// ## BD Annotations
   241тЖТ///
   242тЖТ/// - CapD: { Read, Compare } тАФ pure function, no side effects
   243тЖТ// VUMA-VERIFIED: absolute value is correct for all non-MIN inputs
   244тЖТpub fn abs(x: i64) -> i64 {
   245тЖТ    if x < 0 {
   246тЖТ        x.wrapping_neg()
   247тЖТ    } else {
   248тЖТ        x
   249тЖТ    }
   250тЖТ}

   314тЖТ// ---------------------------------------------------------------------------
   315тЖТ// Clamp
   316тЖТ// ---------------------------------------------------------------------------
   317тЖТ
   318тЖТ/// Constrain a value to lie within the inclusive range `[lo, hi]`.
   319тЖТ///
   320тЖТ/// Returns `lo` if `x < lo`, `hi` if `x > hi`, or `x` otherwise.
   321тЖТ/// This is equivalent to `max(lo, min(hi, x))` but expressed as a single
   322тЖТ/// operation for clarity and efficiency.
   323тЖТ///
   324тЖТ/// ## Panics
   325тЖТ///
   326тЖТ/// In debug builds, panics if `lo > hi`. In release builds, the behavior
   327тЖТ/// is `max(lo, min(hi, x))`, which effectively swaps the bounds.
   328тЖТ///
   329тЖТ/// ## VUMA Program Equivalent
   330тЖТ///
   331тЖТ/// ```vuma
   332тЖТ/// fn clamp(x: i64, lo: i64, hi: i64) -> i64 {
   333тЖТ///     if x < lo {
   334тЖТ///         return lo;
   335тЖТ///     }
   336тЖТ///     if x > hi {
   337тЖТ///         return hi;
   338тЖТ///     }
   339тЖТ///     return x;
   340тЖТ/// }
   341тЖТ/// ```
   342тЖТ///
   343тЖТ/// ## Common Use Cases
   344тЖТ///
   345тЖТ/// - Clamping array indices: `clamp(idx, 0, len - 1)`
   346тЖТ/// - Clamping color channels: `clamp(channel, 0, 255)`
   347тЖТ/// - Clamping buffer offsets: `clamp(offset, 0, buf_size - 1)`
   348тЖТ/// - Saturation arithmetic: `clamp(a + b, i64::MIN, i64::MAX)`
   349тЖТ///
   350тЖТ/// ## BD Annotations
   351тЖТ///
   352тЖТ/// - CapD: { Read, Compare } тАФ pure function, no side effects
   353тЖТ// VUMA-VERIFIED: clamp is correct when lo <= hi
   354тЖТpub fn clamp(x: i64, lo: i64, hi: i64) -> i64 {
   355тЖТ    debug_assert!(lo <= hi, "clamp: lo ({}) must be <= hi ({})", lo, hi);
   356тЖТ    if x < lo {
   357тЖТ        lo
   358тЖТ    } else if x > hi {
   359тЖТ        hi
   360тЖТ    } else {
   361тЖТ        x
   362тЖТ    }
   363тЖТ}
