     1в†’//! Structured JSON diagnostics output for LLM integration.
     2в†’//!
     3в†’//! This module provides machine-readable diagnostic messages that can be
     4в†’//! serialized to JSON, enabling LLM-based tools and IDE integrations to
     5в†’//! programmatically consume compiler errors, warnings, and hints.
     6в†’//!
     7в†’//! # Architecture
     8в†’//!
     9в†’//! Each diagnostic carries a structured error code (`"E001"`, `"W001"`, etc.),
    10в†’//! a severity level, a human-readable message, a source identifier (which
    11в†’//! compiler stage produced it), a precise source location, optional related
    12в†’//! information, quick-fix suggestions with edit ranges, and an optional
    13в†’//! causal chain linking to parent diagnostics.
    14в†’//!
    15в†’//! # Error Catalog
    16в†’//!
    17в†’//! The diagnostic codes are organized into the following ranges:
    18в†’//!
    19в†’//! | Range    | Category       | Description                              |
    20в†’//! |----------|----------------|------------------------------------------|
    21в†’//! | E001вЂ“E030| Compilation    | Syntax, type, name resolution errors     |
    22в†’//! | E031вЂ“E040| Codegen        | Register allocation, encoding, relocation|
    23в†’//! | E041вЂ“E050| Verification   | Invariant violations, proof failures     |
    24в†’//! | W001вЂ“W010| Warnings       | Unused vars, performance hints            |
    25в†’//! | I001вЂ“I005| Informational  | General compiler information messages     |
    26в†’//!
    27в†’//! # Example JSON output
    28в†’//!
    29в†’//! ```json
    30в†’//! [
    31в†’//!   {
    32в†’//!     "code": "E002",
    33в†’//!     "severity": "error",
    34в†’//!     "message": "undefined variable `foo`",
    35в†’//!     "source": "parser",
    36в†’//!     "location": {
    37в†’//!       "file": "main.vu",
    38в†’//!       "start_line": 10,
    39в†’//!       "start_col": 4,
    40в†’//!       "end_line": 10,
    41в†’//!       "end_col": 7
    42в†’//!     },
    43в†’//!     "related": [],
    44в†’//!     "suggestions": ["did you mean `for`?"],
    45в†’//!     "chain": []
    46в†’//!   }
    47в†’//! ]
    48в†’//! ```
    49в†’//!
    50в†’//! # Integration
    51в†’//!
    52в†’//! Use [`from_parse_error`] to convert parser errors, [`from_vuma_error`] for
    53в†’//! pipeline-level errors, and [`from_codegen_error`] for code-generation
    54в†’//! failures.

    69в†’/// Severity level for a machine-readable diagnostic.

   541в†’/// A machine-readable diagnostic with structured fields for LLM consumption.
   542в†’///
   543в†’/// This is the primary output type for JSON diagnostics. Every field is
   544в†’/// serializable and carries enough information for an LLM or IDE to
   545в†’/// understand, categorize, and potentially auto-fix the issue.
   546в†’///
   547в†’/// # Error Chaining
   548в†’///
   549в†’/// Diagnostics support causal chains via the [`chain`](VumaDiagnostic::chain)
   550в†’/// method. A chain represents the cause of a diagnostic: for example,
   551в†’/// "E003: Type mismatch" may be caused by "E023: Unknown type 'int'".
   552в†’///
   553в†’/// # Suggestions
   554в†’///
   555в†’/// Each diagnostic can carry zero or more [`Suggestion`] values, which may
   556в†’/// include precise edit ranges for automatic code fixes.
   557в†’#[derive(Debug, Clone, Serialize, Deserialize)]
   558в†’pub struct VumaDiagnostic {
   559в†’    /// Structured diagnostic code, e.g. `"E001"`, `"W001"`.
   560в†’    pub code: String,
   561в†’    /// Severity level.
   562в†’    pub severity: DiagnosticSeverity,
   563в†’    /// Human-readable message (suitable for display).
   564в†’    pub message: String,
   565в†’    /// Compiler stage that produced this diagnostic
   566в†’    /// (`"parser"`, `"scg"`, `"ir"`, `"codegen"`, etc.).
   567в†’    pub source: String,
   568в†’    /// Precise source location.
   569в†’    pub location: DiagnosticSourceLocation,
   570в†’    /// Related source locations with explanatory messages.
   571в†’    pub related: Vec<RelatedInfo>,
   572в†’    /// Quick-fix suggestions (structured with optional edit ranges).
   573в†’    #[serde(default)]
   574в†’    pub suggestions: Vec<Suggestion>,
   575в†’    /// Causal chain: list of diagnostics that caused this one.
   576в†’    /// The first element is the immediate cause, the last is the root cause.
   577в†’    #[serde(default)]
   578в†’    pub chain: Vec<Box<VumaDiagnostic>>,
   579в†’    /// Backward-compatible plain-text suggestions (stored alongside
   580в†’    /// structured suggestions for JSON serialization compatibility).
   581в†’    #[serde(default, skip_serializing_if = "Vec::is_empty")]
   582в†’    pub legacy_suggestions: Vec<String>,
   583в†’}

   781в†’                "  \x1b[32mhelp\x1b[0m: {}",

   823в†’    ///
   824в†’    /// Returns a JSON `Value` conforming to the LSP Diagnostic structure:
   825в†’    /// - `range`: 0-based Position range
   826в†’    /// - `severity`: LSP numeric severity (1=Error, 2=Warning, 3=Info, 4=Hint)
   827в†’    /// - `code`: the diagnostic code string
   828в†’    /// - `source`: the compiler stage

   831в†’    pub fn to_lsp(&self) -> serde_json::Value {
   832в†’        use serde_json::json;
   833в†’
   834в†’        let start_line = if self.location.start_line > 0 {
   835в†’            self.location.start_line - 1
   836в†’        } else {
   837в†’            0
   838в†’        };
   839в†’        let start_col = if self.location.start_col > 0 {
   840в†’            self.location.start_col - 1
   841в†’        } else {
   842в†’            0
   843в†’        };
   844в†’        let end_line = if self.location.end_line > 0 {
   845в†’            self.location.end_line - 1
   846в†’        } else {
   847в†’            0
   848в†’        };
   849в†’        let end_col = if self.location.end_col > 0 {
   850в†’            self.location.end_col - 1
   851в†’        } else {
   852в†’            0
   853в†’        };
   854в†’
   855в†’        let mut diagnostic = json!({
   856в†’            "range": {
   857в†’                "start": { "line": start_line, "character": start_col },
   858в†’                "end": { "line": end_line, "character": end_col }
   859в†’            },
   860в†’            "severity": self.severity.to_lsp_severity(),
   861в†’            "code": self.code,
   862в†’            "source": self.source,
   863в†’            "message": self.message,
   864в†’        });
   865в†’
   866в†’        // Add related information if present
   867в†’        if !self.related.is_empty() {
   868в†’            let related: Vec<serde_json::Value> = self.related.iter().map(|r| {
   869в†’                let r_start_line = if r.location.start_line > 0 { r.location.start_line - 1 } else { 0 };
   870в†’                let r_start_col = if r.location.start_col > 0 { r.location.start_col - 1 } else { 0 };
   871в†’                let r_end_line = if r.location.end_line > 0 { r.location.end_line - 1 } else { 0 };
   872в†’                let r_end_col = if r.location.end_col > 0 { r.location.end_col - 1 } else { 0 };
   873в†’                json!({
   874в†’                    "location": {
   875в†’                        "uri": format!("file://{}", r.location.file),
   876в†’                        "range": {
   877в†’                            "start": { "line": r_start_line, "character": r_start_col },
   878в†’                            "end": { "line": r_end_line, "character": r_end_col }
   879в†’                        }
   880в†’                    },
   881в†’                    "message": r.message
   882в†’                })
   883в†’            }).collect();
   884в†’            diagnostic["relatedInformation"] = json!(related);
   885в†’        }
   886в†’
   887в†’        // Add LSP CodeDescription with href if it's a known code
   888в†’        let desc = code_description(&self.code);
   889в†’        if desc != "Unknown diagnostic code" {
   890в†’            diagnostic["codeDescription"] = json!({
   891в†’                "href": format!("https://vuma.dev/docs/diagnostics/{}", self.code)
   892в†’            });
   893в†’        }
   894в†’
   895в†’        // Add tags for warnings (Unnecessary = 1, Deprecated = 2)
   896в†’        let mut tags: Vec<u32> = Vec::new();
   897в†’        match self.code.as_str() {
   898в†’            "W001" | "W004" | "W009" => tags.push(1), // Unnecessary
   899в†’            "W008" => tags.push(2),                     // Deprecated
   900в†’            _ => {}
   901в†’        }
   902в†’        if !tags.is_empty() {
   903в†’            diagnostic["tags"] = json!(tags);
   904в†’        }
   905в†’
   906в†’        diagnostic
   907в†’    }
   908в†’}

  1074в†’/// Convert a [`ParseError`] to a [`VumaDiagnostic`].

  1079в†’    err: &ParseError,
  1080в†’    source: &str,
  1081в†’    file: Option<&str>,
  1082в†’) -> VumaDiagnostic {
  1083в†’    let code = code_for_parse_error_kind(&err.kind).to_string();
  1084в†’    let loc = offset_to_location(source, err.span.start, file);
  1085в†’
  1086в†’    // Compute end position from span end.
  1087в†’    let end_loc = if err.span.end > err.span.start && err.span.end <= source.len() {
  1088в†’        offset_to_location(source, err.span.end, file)
  1089в†’    } else {
  1090в†’        loc.clone()
  1091в†’    };
  1092в†’
  1093в†’    let diag_loc = DiagnosticSourceLocation {
  1094в†’        file: loc.file.clone().unwrap_or_default(),
  1095в†’        start_line: loc.line as u32 + 1, // convert 0-based to 1-based
  1096в†’        start_col: loc.column as u32 + 1,
  1097в†’        end_line: end_loc.line as u32 + 1,
  1098в†’        end_col: end_loc.column as u32 + 1,
  1099в†’    };
  1100в†’
  1101в†’    let mut diag = VumaDiagnostic::new(
  1102в†’        code,
  1103в†’        DiagnosticSeverity::Error,
  1104в†’        &err.message,
  1105в†’        "parser",
  1106в†’        diag_loc,
  1107в†’    );
  1108в†’
  1109в†’    if let Some(ref suggestion) = err.suggestion {
  1110в†’        diag = diag.with_suggestion(format!("did you mean '{}'?", suggestion));

  1112в†’
  1113в†’    // Store the line text as a hint if available.
  1114в†’    if let Some(ref line_text) = loc.line_text {

  1116в†’    }
