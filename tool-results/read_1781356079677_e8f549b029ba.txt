   273‚Üí
   274‚Üí/// Target-specific information needed during code generation.
   275‚Üí///
   276‚Üí/// This trait provides a data-driven interface for query target properties.
   277‚Üí/// It must be implementable for ALL supported ISAs, including fundamentally
   278‚Üí/// different architectures like Wasm (stack machine, no registers) and
   279‚Üí/// MIPS (branch delay slots).
   280‚Üí///
   281‚Üí/// # Object Safety
   282‚Üí///
   283‚Üí/// This trait is object-safe: all methods take `&self` and return only
   284‚Üí/// owned types or references with `'static` lifetime.
   285‚Üípub trait TargetInfo: Send + Sync + 'static {
   286‚Üí    // === Identity ===
   287‚Üí
   288‚Üí    /// ISA name in lowercase (e.g., "aarch64", "riscv64", "wasm32").
   289‚Üí    fn isa_name(&self) -> &'static str;
   290‚Üí
   291‚Üí    /// LLVM-style target triple (e.g., "aarch64-unknown-linux-gnu").
   292‚Üí    fn target_triple(&self) -> &'static str;
   293‚Üí
   294‚Üí    /// ELF `e_machine` value.  Returns 0 for non-ELF targets (Wasm).
   295‚Üí    fn elf_machine_type(&self) -> u16;
   296‚Üí
   297‚Üí    /// Default base address for the .text section.
   298‚Üí    fn default_base_address(&self) -> u64;
   299‚Üí
   300‚Üí    // === Data model ===
   301‚Üí
   302‚Üí    /// Pointer width in bytes (4 for 32-bit, 8 for 64-bit).
   303‚Üí    fn pointer_width(&self) -> usize;
   304‚Üí
   305‚Üí    /// Size in bytes of `ty` on this target.
   306‚Üí    fn size_of(&self, ty: &IRType) -> usize;
   307‚Üí
   308‚Üí    /// Natural alignment in bytes of `ty` on this target.
   309‚Üí    fn alignment_of(&self, ty: &IRType) -> usize;
   310‚Üí
   311‚Üí    /// Byte order of this target.
   312‚Üí    fn endianness(&self) -> Endianness;
   313‚Üí
   314‚Üí    // === Register architecture ===
   315‚Üí
   316‚Üí    /// Whether this target has registers at all.  `false` for Wasm (stack machine).
   317‚Üí    fn has_registers(&self) -> bool;
   318‚Üí
   319‚Üí    /// Number of general-purpose registers.  0 for Wasm.
   320‚Üí    fn num_gp_regs(&self) -> usize;
   321‚Üí
   322‚Üí    /// Number of SIMD/FP registers.  0 for Wasm.
   323‚Üí    fn num_simd_fp_regs(&self) -> usize;
   324‚Üí
   325‚Üí    /// Whether the ISA has a hardwired-zero register (RISC-V x0, LoongArch r0).
   326‚Üí    fn has_hardwired_zero(&self) -> bool;
   327‚Üí
   328‚Üí    /// Whether the ISA uses a link register (ARM, RISC-V, MIPS, PPC) rather than
   329‚Üí    /// pushing the return address on the stack (x86_64).
   330‚Üí    fn has_link_register(&self) -> bool;
   331‚Üí
   332‚Üí    /// Whether branches have delay slots (MIPS only).
   333‚Üí    fn has_branch_delay_slots(&self) -> bool;
   334‚Üí
   335‚Üí    /// Whether this ISA uses a TOC (Table of Contents) pointer (PPC64 r2).
   336‚Üí    fn has_toc_pointer(&self) -> bool;
   337‚Üí
   338‚Üí    /// Whether this ISA has dedicated condition register fields (PPC64 CR0-CR7).
   339‚Üí    fn has_condition_registers(&self) -> bool;
   340‚Üí
   341‚Üí    // === Calling convention ===
   342‚Üí
   343‚Üí    /// Name of the calling convention (e.g., "aapcs64", "lp64d", "systemv").
   344‚Üí    fn calling_convention_name(&self) -> &'static str;
   345‚Üí
   346‚Üí    /// Number of integer argument registers.
   347‚Üí    fn num_int_arg_regs(&self) -> usize;
   348‚Üí
   349‚Üí    /// Number of FP/SIMD argument registers.
   350‚Üí    fn num_fp_arg_regs(&self) -> usize;
   351‚Üí
   352‚Üí    /// Required stack alignment in bytes.
   353‚Üí    fn stack_alignment(&self) -> usize;
   354‚Üí
   355‚Üí    // === Instruction encoding ===
   356‚Üí
   357‚Üí    /// Alignment requirement for instructions in bytes (4 for fixed-width RISCs,
   358‚Üí    /// 1 for x86_64 and Wasm).
   359‚Üí    fn instruction_alignment(&self) -> usize;
   360‚Üí
   361‚Üí    /// Minimum and maximum instruction width in bytes.
   362‚Üí    /// - Fixed-width 32-bit ISAs: (4, 4)
   363‚Üí    /// - x86_64: (1, 15)
   364‚Üí    /// - RISC-V with RVC: (2, 4)
   365‚Üí    /// - Wasm: (1, ~) but typically (1, 15)
   366‚Üí    fn instruction_width_range(&self) -> (usize, usize);
   367‚Üí
   368‚Üí    // === Output format ===
   369‚Üí
   370‚Üí    /// Binary format produced by this backend.
   371‚Üí    fn output_format(&self) -> OutputFormat;
   372‚Üí}

   378‚Üí/// A code generation backend for a specific target architecture.
   379‚Üí///
   380‚Üí/// Each supported ISA implements this trait, providing register allocation,
   381‚Üí/// instruction encoding, program emission, and disassembly.
   382‚Üí///
   383‚Üí/// # Object Safety
   384‚Üí///
   385‚Üí/// This trait is object-safe.
   386‚Üípub trait Backend: Send + Sync + 'static {
   387‚Üí    /// Returns a reference to this backend's target info.
   388‚Üí    fn target_info(&self) -> &dyn TargetInfo;
   389‚Üí
   390‚Üí    /// Allocate physical registers for an IR function.
   391‚Üí    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError>;
   392‚Üí
   393‚Üí    /// Encode a single allocated function into machine code bytes.
   394‚Üí    fn encode_function(&self, func: &AllocatedFunction) -> Result<Vec<u8>, BackendError>;
   395‚Üí
   396‚Üí    /// Encode an entire allocated program into its final binary form
   397‚Üí    /// (ELF, .wasm, raw binary, etc.).
   398‚Üí    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError>;
   399‚Üí
   400‚Üí    /// Returns the bytes for a minimal return stub (e.g., `RET` on ARM64,
   401‚Üí    /// `mov eax, 0; ret` on x86_64, `end` on Wasm).
   402‚Üí    fn return_stub(&self) -> Vec<u8>;
   403‚Üí
   404‚Üí    /// Returns a trampoline that jumps to `entry_addr`.
   405‚Üí    fn trampoline(&self, entry_addr: u64) -> Vec<u8>;
   406‚Üí
   407‚Üí    /// Disassemble `bytes` starting at virtual address `addr`.
   408‚Üí    fn disassemble(&self, bytes: &[u8], addr: u64) -> Vec<String>;
   409‚Üí
   410‚Üí    /// Human-readable name of this backend.
   411‚Üí    fn name(&self) -> &'static str;
   412‚Üí}

  1118‚Üí///
  1119‚Üí/// Covers the most common AArch64 instructions: ADD, SUB, MOV, LDR, STR, B,
  1120‚Üí/// BL, RET, CMP, B.cond, STP, LDP, NOP, MUL, SDIV, UDIV, AND, ORR, EOR,
  1121‚Üí/// plus several additional frequently-encountered encodings.
  1122‚Üífn decode_aarch64(word: u32) -> String {

  1124‚Üí    let rn = (word >> 5) & 0x1F;
  1125‚Üí    let rt = rd; // alias for load/store destination
  1126‚Üí    let rm = (word >> 16) & 0x1F;

  1128‚Üí    let cond = word & 0xF;

  1130‚Üí    // NOP: d503201f
  1131‚Üí    if word == 0xD503201F {
  1132‚Üí        return "nop".to_string();
  1133‚Üí    }
  1134‚Üí
  1135‚Üí    // RET: d65f03c0
  1136‚Üí    if word == 0xD65F03C0 {
  1137‚Üí        return "ret".to_string();
  1138‚Üí    }
  1139‚Üí
  1140‚Üí    let _top8 = word >> 24;
  1141‚Üí    let _top10 = word >> 22;
  1142‚Üí
  1143‚Üí    // --- ADD/SUB (immediate): 100100xx ...
  1144‚Üí    if (word >> 23) & 0x1FF == 0b1_0010_0010 {
  1145‚Üí        // ADD Xd, Xn, #imm12
  1146‚Üí        return format!("add x{}, x{}, #{}", rd, rn, imm12);
  1147‚Üí    }
  1148‚Üí    if (word >> 23) & 0x1FF == 0b1_1010_0010 {
  1149‚Üí        // SUB Xd, Xn, #imm12
  1150‚Üí        return format!("sub x{}, x{}, #{}", rd, rn, imm12);
  1151‚Üí    }
  1152‚Üí
  1153‚Üí    // --- ADD (shifted register): 1_00_0101_1_xxx ...
  1154‚Üí    if (word >> 24) & 0xFF == 0b1000_1011 {
  1155‚Üí        // ADD Xd, Xn, Xm
  1156‚Üí        return format!("add x{}, x{}, x{}", rd, rn, rm);
  1157‚Üí    }
  1158‚Üí
  1159‚Üí    // --- SUB (shifted register): 1_00_0101_1_xxx ... with S=1 at bit30
  1160‚Üí    if (word >> 24) & 0xFF == 0b1101_0110 {
  1161‚Üí        // SUB Xd, Xn, Xm (bit 30 set = sub)
  1162‚Üí        return format!("sub x{}, x{}, x{}", rd, rn, rm);
  1163‚Üí    }
  1164‚Üí
  1165‚Üí    // --- AND (shifted register): 1_00_0101_0_00_xxx
  1166‚Üí    if (word >> 24) & 0xFE == 0b1000_1010 {
  1167‚Üí        // Check bit 21-15: opcode[31:21] = 10001010_000
  1168‚Üí        if (word >> 21) & 0x7FF == 0b10001010000 {
  1169‚Üí            return format!("and x{}, x{}, x{}", rd, rn, rm);
  1170‚Üí        }
  1171‚Üí    }
  1172‚Üí
  1173‚Üí    // --- ORR (shifted register): 1_01_0101_0_00_xxx
  1174‚Üí    if (word >> 21) & 0x7FF == 0b10101010000 {
  1175‚Üí        return format!("orr x{}, x{}, x{}", rd, rn, rm);
  1176‚Üí    }
  1177‚Üí
  1178‚Üí    // --- EOR (shifted register): 1_10_0101_0_00_xxx
  1179‚Üí    if (word >> 21) & 0x7FF == 0b11001010000 {
  1180‚Üí        return format!("eor x{}, x{}, x{}", rd, rn, rm);
  1181‚Üí    }
  1182‚Üí
  1183‚Üí    // --- MOV (register): alias for ORR Xd, XZR, Xm
  1184‚Üí    // ORR Xd, XZR, Xm: 10101010_000_rm_000000_xzr_rd
  1185‚Üí    // More general: ORR with Rn=XZR(31)
  1186‚Üí    if (word >> 21) & 0x7FF == 0b10101010000 && rn == 31 {
  1187‚Üí        return format!("mov x{}, x{}", rd, rm);
  1188‚Üí    }
  1189‚Üí
  1190‚Üí    // --- MUL: MADD Xd, Xn, Xm, XZR
  1191‚Üí    // Encoding: 1_00_1101_1_000_Rm_0_01111_Rn_Rd
  1192‚Üí    if (word >> 21) & 0x7FF == 0b10011011000 && ((word >> 10) & 0x1F) == 0b01111 {
  1193‚Üí        return format!("mul x{}, x{}, x{}", rd, rn, rm);
  1194‚Üí    }
  1195‚Üí
  1196‚Üí    // --- SDIV: 1_00_1101_1_0100_00000_00001_Rn_Rd  (actually 1_00_1101_0100_Rm_00001_Rn_Rd)
  1197‚Üí    if (word >> 21) & 0x7FF == 0b10011010100 && (word >> 10) & 0x1F == 0b00001 {
  1198‚Üí        return format!("sdiv x{}, x{}, x{}", rd, rn, rm);
  1199‚Üí    }
  1200‚Üí
  1201‚Üí    // --- UDIV: 1_00_1101_0000_Rm_00001_Rn_Rd
  1202‚Üí    if (word >> 21) & 0x7FF == 0b10011010000 && (word >> 10) & 0x1F == 0b00001 {
  1203‚Üí        return format!("udiv x{}, x{}, x{}", rd, rn, rm);
  1204‚Üí    }
  1205‚Üí
  1206‚Üí    // --- CMP (immediate): SUBS XZR, Xn, #imm12
  1207‚Üí    // 11100001_00_xxx_xxx_xxx_xxx_xxx_11111_xxx
  1208‚Üí    if (word >> 23) & 0x1FF == 0b1_1110_0010 && rd == 31 {
  1209‚Üí        return format!("cmp x{}, #{}", rn, imm12);
  1210‚Üí    }
  1211‚Üí
  1212‚Üí    // --- CMP (register): SUBS XZR, Xn, Xm
  1213‚Üí    if (word >> 21) & 0x7FF == 0b11101011000 && rd == 31 {
  1214‚Üí        return format!("cmp x{}, x{}", rn, rm);
  1215‚Üí    }
  1216‚Üí
  1217‚Üí    // --- B.cond: 0101010x xxxxxxxxxx xxxxxx cond
  1218‚Üí    if (word >> 24) & 0xFF == 0x54 {
  1219‚Üí        let cond_name = match cond {
  1220‚Üí            0 => "eq",
  1221‚Üí            1 => "ne",
  1222‚Üí            2 => "cs",
  1223‚Üí            3 => "cc",
  1224‚Üí            4 => "mi",
  1225‚Üí            5 => "pl",
  1226‚Üí            6 => "vs",
  1227‚Üí            7 => "vc",
  1228‚Üí            8 => "hi",
  1229‚Üí            9 => "ls",
  1230‚Üí            10 => "ge",
  1231‚Üí            11 => "lt",
  1232‚Üí            12 => "gt",
  1233‚Üí            13 => "le",
  1234‚Üí            14 => "al",
  1235‚Üí            _ => "??",
  1236‚Üí        };
  1237‚Üí        let imm19 = (word >> 5) & 0x7FFFF;
  1238‚Üí        let offset = ((imm19 as i32) << 13) >> 11; // sign-extend and *4
  1239‚Üí        return format!("b.{} {:+}", cond_name, offset);
  1240‚Üí    }
  1241‚Üí
  1242‚Üí    // --- B (unconditional): 000101xx xxxxxxxxxxxxxxxxxxxx
  1243‚Üí    if (word >> 26) & 0x3F == 0b000101 {
  1244‚Üí        let imm26 = word & 0x3FFFFFF;
  1245‚Üí        let offset = ((imm26 as i32) << 6) >> 4; // sign-extend and *4
  1246‚Üí        return format!("b {:+}", offset);
  1247‚Üí    }
  1248‚Üí
  1249‚Üí    // --- BL: 100101xx xxxxxxxxxxxxxxxxxxxx
  1250‚Üí    if (word >> 26) & 0x3F == 0b100101 {
  1251‚Üí        let imm26 = word & 0x3FFFFFF;
  1252‚Üí        let offset = ((imm26 as i32) << 6) >> 4;
  1253‚Üí        return format!("bl {:+}", offset);
  1254‚Üí    }
  1255‚Üí
  1256‚Üí    // --- LDR (unsigned offset): 11111001_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1257‚Üí    if (word >> 22) & 0x3FF == 0b1111100101 {
  1258‚Üí        let imm12_raw = (word >> 10) & 0xFFF;
  1259‚Üí        let offset = imm12_raw * 8; // scale by 8 for 64-bit
  1260‚Üí        return format!("ldr x{}, [x{}, #{}]", rt, rn, offset);
  1261‚Üí    }
  1262‚Üí
  1263‚Üí    // --- STR (unsigned offset): 11111000_01_xxx_xxx_xxx_xxx_xxx_xn_rt

  1265‚Üí        let imm12_raw = (word >> 10) & 0xFFF;
  1266‚Üí        let offset = imm12_raw * 8;

  1330‚Üí/// Compute the stack frame size for an IR function.

  1349‚Üí
  1350‚Üí/// Build an ELF64 binary for AArch64 Linux with 2 LOAD segments.
  1351‚Üí///
  1352‚Üí/// Segment 1: PF_R | PF_X ‚Äî contains .text (code)
  1353‚Üí/// Segment 2: PF_R | PF_W ‚Äî contains .data + stack space (writable)
  1354‚Üí///
  1355‚Üí/// The two segments are separated by page alignment (4KB) to ensure
  1356‚Üí/// the kernel maps them with different permissions. Without this,
  1357‚Üí/// a single PF_R|PF_W|PF_X segment is insecure and may cause

  1360‚Üí    const PAGE_SIZE: u64 = 0x1000; // 4 KB
  1361‚Üí
  1362‚Üí    let elf_header_size: u64 = 64;
  1363‚Üí    let phdr_size: u64 = 56;
  1364‚Üí    let num_phdrs: u64 = 2;
  1365‚Üí    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1366‚Üí    // Page-align the text segment start in the file.  The kernel's ELF
  1367‚Üí    // loader mmap()s each LOAD segment, and the file offset must be
  1368‚Üí    // congruent with the virtual address modulo the page size.  The
  1369‚Üí    // simplest way to guarantee this is to place the text at a
  1370‚Üí    // page-aligned file offset, with vaddr = base_addr.
  1371‚Üí    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1372‚Üí    let text_size = code.len() as u64;
  1373‚Üí
  1374‚Üí    // The data segment starts on the next page after the text.
  1375‚Üí    let text_file_end = text_offset + text_size;
  1376‚Üí    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1377‚Üí    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1378‚Üí    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1379‚Üí    let entry_point = base_addr + text_offset;
  1380‚Üí
  1381‚Üí    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1382‚Üí
  1383‚Üí    // --- e_ident ---
  1384‚Üí    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1385‚Üí    elf.push(2); // ELFCLASS64
  1386‚Üí    elf.push(1); // ELFDATA2LSB
  1387‚Üí    elf.push(1); // EV_CURRENT
  1388‚Üí    elf.push(3); // ELFOSABI_LINUX
  1389‚Üí    elf.push(0); // padding
  1390‚Üí    elf.extend_from_slice(&[0u8; 7]); // padding
  1391‚Üí
  1392‚Üí    // --- ELF header fields ---
  1393‚Üí    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1394‚Üí    elf.extend_from_slice(&183u16.to_le_bytes()); // e_machine = EM_AARCH64
  1395‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1396‚Üí    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1397‚Üí    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1398‚Üí    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1399‚Üí    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1400‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1401‚Üí    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1402‚Üí    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1403‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1404‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1405‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1406‚Üí
  1407‚Üí    // --- Program Header 1: LOAD (PF_R | PF_X) ‚Äî .text ---
  1408‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1409‚Üí    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1410‚Üí    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1411‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1412‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1413‚Üí    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1414‚Üí    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1415‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1416‚Üí
  1417‚Üí    // --- Program Header 2: LOAD (PF_R | PF_W) ‚Äî .data / stack ---
  1418‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1419‚Üí    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1420‚Üí    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1421‚Üí    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1422‚Üí    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1423‚Üí    elf.extend_from_slice(&0u64.to_le_bytes()); // p_filesz (no initialized data)
  1424‚Üí    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1425‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1426‚Üí
  1427‚Üí    // --- .text section ---
  1428‚Üí    // Pad to page-aligned text_offset
  1429‚Üí    while (elf.len() as u64) < text_offset {
  1430‚Üí        elf.push(0);
  1431‚Üí    }
  1432‚Üí    elf.extend_from_slice(code);
  1433‚Üí
  1434‚Üí    // --- Pad to data segment offset (if needed) ---
  1435‚Üí    while (elf.len() as u64) < data_offset {
  1436‚Üí        elf.push(0);
  1437‚Üí    }
  1438‚Üí
  1439‚Üí    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1440‚Üí
  1441‚Üí    elf
  1442‚Üí}

  1444‚Üí/// Build ARM64 runtime I/O functions using Linux SVC syscalls.
  1445‚Üí///
  1446‚Üí/// Provides:
  1447‚Üí/// - `__vuma_print_hex`: Print X0 as 8 hex digits to stdout (FD=1)
  1448‚Üí///   Uses sys_write (X8=64) via SVC #0.
  1449‚Üí///   Saves/restores X1,X2,X3,X8,X9,X10 on stack.
  1450‚Üí///
  1451‚Üí/// - `__vuma_print_int`: Print X0 as a decimal integer to stdout (FD=1)
  1452‚Üí///   Converts digit-by-digit into a stack buffer, then sys_write.
  1453‚Üí///
  1454‚Üí/// - `__vuma_print_newline`: Print a newline character to stdout.
  1455‚Üí///
  1456‚Üí/// All functions follow AAPCS64: X0 is the argument, X1-X7 are caller-saved,
  1457‚Üí/// X8 is the indirect result register / syscall number, X19-X28 are callee-saved.
  1458‚Üífn build_aarch64_runtime() -> Vec<u8> {
  1459‚Üí    let mut code = Vec::new();
  1460‚Üí
  1461‚Üí    // ‚îÄ‚îÄ __vuma_print_hex ‚îÄ‚îÄ
  1462‚Üí    // Input: X0 = 64-bit value to print as 8 hex digits
  1463‚Üí    // Clobbers: X1, X2, X3, X8, X9, X10
  1464‚Üí    // Stack frame: 48 bytes (8 for pair save + 8 for hex chars + 32 padding)
  1465‚Üí
  1466‚Üí    // Prologue: save callee-saved and link register
  1467‚Üí    // STP X29, X30, [SP, #-48]!  (pre-indexed, 48-byte frame)
  1468‚Üí    code.extend_from_slice(&0xA9BF7BFDu32.to_le_bytes());
  1469‚Üí    // ADD X29, SP, #0
  1470‚Üí    code.extend_from_slice(&0x910003FDu32.to_le_bytes());
  1471‚Üí
  1472‚Üí    // Save X9, X10 on stack at [SP, #16]
  1473‚Üí    // STP X9, X10, [SP, #16]
  1474‚Üí    code.extend_from_slice(&0xA90127E9u32.to_le_bytes());
  1475‚Üí
  1476‚Üí    // Save X1, X2 on stack at [SP, #32]
  1477‚Üí    // STP X1, X2, [SP, #32]
  1478‚Üí    code.extend_from_slice(&0xA9020441u32.to_le_bytes());
  1479‚Üí
  1480‚Üí    // Save X3, X8 on stack at [SP, #40]  (offset 40 = 5*8)
  1481‚Üí    // Wait, 40 = 5*8 but STP needs offset/8 for signed-offset. 40/8 = 5.
  1482‚Üí    // STP X3, X8, [SP, #40]
  1483‚Üí    code.extend_from_slice(&0xA9052C63u32.to_le_bytes());
  1484‚Üí
  1485‚Üí    // Hex lookup table: we'll use ADD + LDRB pattern.
  1486‚Üí    // Instead, compute hex digit directly:
  1487‚Üí    // For each nibble: digit = nibble; if digit < 10 then char = digit + '0' else char = digit - 10 + 'a'
  1488‚Üí
  1489‚Üí    // We'll write 8 hex chars starting at SP+16 (reuse save area after we load saves)
  1490‚Üí    // Actually, let's use SP+16 as the char buffer (8 bytes).
  1491‚Üí    // First, save the important regs, then overwrite the buffer area.
  1492‚Üí
  1493‚Üí    // Process 8 nibbles from most significant to least significant
  1494‚Üí    // X0 = value, iterate 8 times shifting right by 28, 24, 20, ..., 4, 0
  1495‚Üí
  1496‚Üí    // X9 = char buffer pointer = SP
  1497‚Üí    // X10 = loop counter (0..8)
  1498‚Üí
  1499‚Üí    // ADD X9, SP, #16  (buffer starts at SP+16)
  1500‚Üí    code.extend_from_slice(&0x910043E9u32.to_le_bytes());
  1501‚Üí    // MOV X10, #0
  1502‚Üí    code.extend_from_slice(&0xD280001Au32.to_le_bytes());
  1503‚Üí    // MOV X3, #0  (bit shift amount, starts at 28)
  1504‚Üí    code.extend_from_slice(&0xD280003Du32.to_le_bytes()); // MOVZ X3, #28>>4... wait
  1505‚Üí    // MOV X3, #28
  1506‚Üí    code.extend_from_slice(&0xD2800383u32.to_le_bytes()); // MOVZ X3, #28
  1507‚Üí
  1508‚Üí    // Loop label:
  1509‚Üí    let loop_offset = code.len();
  1510‚Üí
  1511‚Üí    // Extract nibble: X2 = (X0 >> X3) & 0xF
  1512‚Üí    // LSR X2, X0, X3
  1513‚Üí    code.extend_from_slice(&0x9AC02482u32.to_le_bytes()); // LSR X2, X0, X3
  1514‚Üí    // AND X2, X2, #0xF
  1515‚Üí    code.extend_from_slice(&0x92400C42u32.to_le_bytes()); // AND X2, X2, #15
  1516‚Üí
  1517‚Üí    // Convert nibble to hex char:
  1518‚Üí    // if X2 < 10: char = X2 + 48 ('0')
  1519‚Üí    // else: char = X2 - 10 + 97 ('a')
  1520‚Üí    // CMP X2, #10
  1521‚Üí    code.extend_from_slice(&0xF100141Fu32.to_le_bytes()); // CMP X2, #10? No...
  1522‚Üí    // CMP X2, #10: use ADD X1, X2, #0 then CMP... Actually:
  1523‚Üí    // SUB X1, X2, #10 ‚Üí if < 10, result is negative (borrow set)
  1524‚Üí    // Let's use: CMP X2, #9; B.GT hex_alpha
  1525‚Üí    // CMP X2, #9
  1526‚Üí    code.extend_from_slice(&0xF100129Fu32.to_le_bytes()); // CMP X2, #9
  1527‚Üí
  1528‚Üí    // ADD X1, X2, #48  (default: '0' + digit)
  1529‚Üí    code.extend_from_slice(&0x91000C41u32.to_le_bytes()); // ADD X1, X2, #48
  1530‚Üí
  1531‚Üí    // B.GT hex_alpha (placeholder, will be +1 instruction)
  1532‚Üí    code.extend_from_slice(&0x54000069u32.to_le_bytes()); // B.GT +4*2 = +8 bytes
  1533‚Üí
  1534‚Üí    // B store_char
  1535‚Üí    code.extend_from_slice(&0x14000002u32.to_le_bytes()); // B +2 instructions
  1536‚Üí
  1537‚Üí    // hex_alpha: ADD X1, X2, #87  (87 = 97-10, so digit-10+'a')
  1538‚Üí    code.extend_from_slice(&0x91002AC1u32.to_le_bytes()); // ADD X1, X2, #87
  1539‚Üí
  1540‚Üí    // store_char: STRB W1, [X9, X10]
  1541‚Üí    // STRB W1, [X9, X10] ‚Äî register offset
  1542‚Üí    code.extend_from_slice(&0x382A6821u32.to_le_bytes()); // STRB W1, [X9, X10]
  1543‚Üí
  1544‚Üí    // Increment: SUB X3, X3, #4; ADD X10, X10, #1; CMP X10, #8; B.LT loop
  1545‚Üí    code.extend_from_slice(&0xD1000863u32.to_le_bytes()); // SUB X3, X3, #4
  1546‚Üí    code.extend_from_slice(&0x9100054Au32.to_le_bytes()); // ADD X10, X10, #1
  1547‚Üí    code.extend_from_slice(&0xF100151Fu32.to_le_bytes()); // CMP X10, #8
  1548‚Üí
  1549‚Üí    // Calculate branch back to loop start
  1550‚Üí    let current_pos = code.len() as i64;
  1551‚Üí    let loop_start = loop_offset as i64;
  1552‚Üí    let back_offset = (loop_start - current_pos) / 4 - 1; // B.LT is relative to PC+4
  1553‚Üí    let bcond = 0x5400000Bu32 | ((back_offset as u32) & 0x7FFFF); // B.LT
  1554‚Üí    code.extend_from_slice(&bcond.to_le_bytes());

  1556‚Üí    // ‚îÄ‚îÄ sys_write(1, buf, 8) ‚îÄ‚îÄ
  1557‚Üí    // MOV X0, #1  (stdout fd)
  1558‚Üí    code.extend_from_slice(&0xD2800020u32.to_le_bytes()); // MOVZ X0, #1
  1559‚Üí    // ADD X1, SP, #16  (buffer pointer)
  1560‚Üí    code.extend_from_slice(&0x910043E1u32.to_le_bytes());
  1561‚Üí    // MOV X2, #8  (length)
  1562‚Üí    code.extend_from_slice(&0xD2800102u32.to_le_bytes()); // MOVZ X2, #8
  1563‚Üí    // MOV X8, #64  (sys_write)
  1564‚Üí    code.extend_from_slice(&0xD2800808u32.to_le_bytes()); // MOVZ X8, #64

  1566‚Üí    code.extend_from_slice(&0xD4000001u32.to_le_bytes());
  1567‚Üí
  1568‚Üí    // Restore registers
  1569‚Üí    // LDP X9, X10, [SP, #16]
  1570‚Üí    code.extend_from_slice(&0xA94127E9u32.to_le_bytes());
  1571‚Üí    // LDP X1, X2, [SP, #32]
  1572‚Üí    code.extend_from_slice(&0xA9420441u32.to_le_bytes());
  1573‚Üí    // LDP X3, X8, [SP, #40]
  1574‚Üí    code.extend_from_slice(&0xA9452C63u32.to_le_bytes());

  1579‚Üí    code.extend_from_slice(&0xD65F03C0u32.to_le_bytes());

  1581‚Üí    // ‚îÄ‚îÄ __vuma_print_int ‚îÄ‚îÄ

  1598‚Üí    // Handle negative numbers
  1599‚Üí    // CMP X0, #0

  1627‚Üí    let div_loop_start = code.len();

  1638‚Üí    code.extend_from_slice(&0x9B017C43u32.to_le_bytes()); // MSUB X3, X2, X1, X0

  1642‚Üí    code.extend_from_slice(&0x382B6863u32.to_le_bytes()); // STRB W3, [X10, X9]

  1644‚Üí    code.extend_from_slice(&0x91000529u32.to_le_bytes()); // ADD X9, X9, #1
  1645‚Üí    // MOV X0, X2  (quotient becomes new value)
  1646‚Üí    code.extend_from_slice(&0xAA0203E0u32.to_le_bytes()); // MOV X0, X2

  1649‚Üí    let back = (div_loop_start as i64 - cur as i64) / 4 - 1;
  1650‚Üí    code.extend_from_slice(&(0x14000000u32 | ((back as u32) & 0x3FFFFFF)).to_le_bytes());
  1651‚Üí

  1653‚Üí    // Reverse them in-place using X1 and X3 as temp pointers.
  1654‚Üí    // CBZ X9, write_digits  (if count == 0, print "0")
  1655‚Üí    code.extend_from_slice(&0xB4000039u32.to_le_bytes()); // CBZ X9, write_digits (+6 instr = 24 bytes)
  1656‚Üí
  1657‚Üí    // If count == 0, print "0"
  1658‚Üí    // MOV X1, #48  ('0')
  1659‚Üí    code.extend_from_slice(&0xD2800601u32.to_le_bytes()); // MOVZ X1, #48
  1660‚Üí    // STRB W1, [X10]
  1661‚Üí    code.extend_from_slice(&0x39001501u32.to_le_bytes());
  1662‚Üí    // MOV X9, #1
