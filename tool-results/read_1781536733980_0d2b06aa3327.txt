     1→//! VUMA CLI — Command-line interface for the VUMA compiler.
     2→//!
     3→//! Subcommands:
     4→//! - `vuma build <file>` — Parse + compile to ARM64 ELF (default), save to output file
     5→//! - `vuma run <file>`   — Build + execute (via QEMU aarch64 or native)
     6→//! - `vuma check <file>` — Parse + SCG + BD inference + IVE verification only
     7→//! - `vuma emit <isa> <file>` — Compile to specific ISA
     8→//! - `vuma disasm <file>` — Read binary and disassemble
     9→//! - `vuma verify <file>` — Run IVE 5-invariant verification
    10→//! - `vuma repl` — Interactive REPL (parse expr, print AST)
    11→//! - `vuma lsp`  — Start Language Server (LSP) for IDE/LLM integration
    12→

   244→/// Read source file content with a human-readable error on failure.
   245→fn read_source(path: &PathBuf) -> Result<String, String> {
   246→    fs::read_to_string(path)
   247→        .map_err(|e| format!("error: cannot read source file '{}': {}", path.display(), e))
   248→}

   250→/// Build a `CompileConfig` from the global CLI flags.
   251→fn make_config(cli: &Cli, target: CompileTarget) -> CompileConfig {
   252→    CompileConfig {

   254→        opt_level: OptLevel::from(cli.opt_level),

   256→        debug_info: cli.debug,
   257→        section_headers: cli.sections,
   258→        runtime_bounds_checks: cli.safe,
   259→        memory_safety: true,
   260→        ..CompileConfig::default()
   261→    }

   264→/// Print compilation errors to stderr in a human-readable format.
   265→fn print_errors(errors: &[VumaError]) {
   266→    for err in errors {
   267→        eprintln!("error[{}]: {}", err.stage(), err);
   268→    }

   271→/// Determine the default output path for a given input file.
   272→fn default_output_path(input: &Path) -> PathBuf {
   273→    let stem = input.file_stem().unwrap_or_default().to_string_lossy();
   274→    let dir = input.parent().unwrap_or(std::path::Path::new("."));
   275→    dir.join(format!("{}.o", stem))
   276→}

   282→/// `vuma build <file>` — Parse + compile to ELF, save to output file.
   283→fn cmd_build(
   284→    cli: &Cli,
   285→    file: &PathBuf,
   286→    output: &Option<PathBuf>,
   287→    target: &TargetArg,
   288→) -> Result<(), String> {
   289→    let source = read_source(file)?;
   290→    let config = make_config(cli, CompileTarget::from(*target));
   291→    let result = compile_with_path(&source, Some(file), &config).map_err(|errors| {
   292→        print_errors(&errors);
   293→        format!("compilation failed with {} error(s)", errors.len())
   294→    })?;
   295→
   296→    let out_path = output
   297→        .as_ref()
   298→        .cloned()
   299→        .unwrap_or_else(|| default_output_path(file));
   300→    fs::write(&out_path, &result.binary).map_err(|e| {
   301→        format!(
   302→            "error: cannot write output file '{}': {}",
   303→            out_path.display(),
   304→            e
   305→        )
   306→    })?;
   307→    // Make the output file executable on Unix.
   308→    #[cfg(unix)]
   309→    {
   310→        use std::os::unix::fs::PermissionsExt;
   311→        let mut perms = fs::metadata(&out_path)
   312→            .map_err(|e| format!("error: cannot stat '{}': {}", out_path.display(), e))?
   313→            .permissions();
   314→        perms.set_mode(0o755);
   315→        fs::set_permissions(&out_path, perms)
   316→            .map_err(|e| format!("error: cannot chmod '{}': {}", out_path.display(), e))?;
   317→    }
   318→
   319→    println!(
   320→        "Compiled {} -> {} ({} bytes, {} SCG nodes, {} IR instructions)",
   321→        file.display(),
   322→        out_path.display(),
   323→        result.binary.len(),
   324→        result.scg.node_count(),
   325→        result.ir_instruction_count,
   326→    );
   327→
   328→    // Print stage timings.
   329→    for (stage, ms) in &result.stage_timings {
   330→        println!("  {:20} {}ms", stage, ms);
   331→    }
   332→
   333→    Ok(())
   334→}

   336→/// `vuma run <file>` — Build + execute.
   337→fn cmd_run(cli: &Cli, file: &PathBuf, args: &[String]) -> Result<(), String> {
   338→    let source = read_source(file)?;
   339→    let config = make_config(cli, CompileTarget::Linux);
   340→    let result = compile_with_path(&source, Some(file), &config).map_err(|errors| {
   341→        print_errors(&errors);
   342→        format!("compilation failed with {} error(s)", errors.len())
   343→    })?;
   344→
   345→    // Write binary to a temp file.
   346→    let tmp_dir = std::env::temp_dir();
   347→    let exe_path = tmp_dir.join(format!("vuma_run_{}", std::process::id()));
   348→    fs::write(&exe_path, &result.binary).map_err(|e| {
   349→        format!(
   350→            "error: cannot write temporary executable '{}': {}",
   351→            exe_path.display(),
   352→            e
   353→        )
   354→    })?;
   355→
   356→    // Make the file executable on Unix.
   357→    #[cfg(unix)]
   358→    {
   359→        use std::os::unix::fs::PermissionsExt;
   360→        let mut perms = fs::metadata(&exe_path)
   361→            .map_err(|e| format!("error: cannot stat '{}': {}", exe_path.display(), e))?
   362→            .permissions();
   363→        perms.set_mode(0o755);
   364→        fs::set_permissions(&exe_path, perms)
   365→            .map_err(|e| format!("error: cannot chmod '{}': {}", exe_path.display(), e))?;
   366→    }
   367→
   368→    // Try to execute: first natively (if on aarch64), then via qemu-aarch64.
   369→    let native_output = Command::new(&exe_path).args(args).output();
   370→
   371→    let output = match native_output {
   372→        Ok(out) => out,
   373→        Err(_) => {
   374→            // Native execution failed; try qemu-aarch64.
   375→            let qemu_output = Command::new("qemu-aarch64")
   376→                .arg(&exe_path)
   377→                .args(args)
   378→                .output()
   379→                .map_err(|e| {
   380→                    format!(
   381→                        "error: failed to execute binary (neither native nor qemu-aarch64): {}",
   382→                        e
   383→                    )
   384→                })?;
   385→            qemu_output
   386→        }
   387→    };
   388→
   389→    io::stdout()
   390→        .write_all(&output.stdout)
   391→        .map_err(|e| format!("error: failed to write program output: {}", e))?;
   392→    io::stderr()
   393→        .write_all(&output.stderr)
   394→        .map_err(|e| format!("error: failed to write program stderr: {}", e))?;
   395→
   396→    // Clean up temp file.
   397→    let _ = fs::remove_file(&exe_path);
   398→
   399→    let code = output.status.code().unwrap_or(1);
   400→    if code != 0 {
   401→        return Err(format!("program exited with code {}", code));
   402→    }
   403→
   404→    Ok(())
   405→}

   407→/// `vuma check <file>` — Parse + SCG + BD inference + IVE verification only.
   408→fn cmd_check(cli: &Cli, file: &PathBuf) -> Result<(), String> {
   409→    let source = read_source(file)?;
   410→    let mut config = make_config(cli, CompileTarget::Linux);
   411→    // check mode: always run verification, don't skip
   412→    if config.verification_level == VerificationLevel::None {
   413→        config.verification_level = VerificationLevel::Normal;
   414→    }
   415→
   416→    // Run the full compile — the check command doesn't save the binary,
   417→    // it just verifies the program compiles and passes verification.
   418→    let result = compile_with_path(&source, Some(file), &config).map_err(|errors| {
   419→        print_errors(&errors);
   420→        format!("check failed with {} error(s)", errors.len())
   421→    })?;
   422→
   423→    println!("Check passed for {}", file.display());
   424→    println!("  SCG nodes:      {}", result.scg.node_count());
   425→    println!("  IR functions:   {}", result.ir_function_count);
   426→    println!("  IR instructions: {}", result.ir_instruction_count);
   427→
   428→    if let Some(ref verification) = result.verification {
   429→        println!("  Verification:   {}", verification.overall);
   430→    }
   431→
   432→    // Print stage timings.
   433→    for (stage, ms) in &result.stage_timings {
   434→        println!("  {:20} {}ms", stage, ms);
   435→    }
   436→
   437→    Ok(())
   438→}

   440→/// `vuma emit <isa> <file>` — Compile to a specific ISA target.
   441→///
   442→/// Uses the direct AST → codegen SCG → IR path for better code quality
   443→/// than the main pipeline (which goes through the semantic SCG and loses
   444→/// most program semantics in the SCG→IR bridge).
   445→fn cmd_emit(
   446→    cli: &Cli,
   447→    isa: &IsaArg,
   448→    file: &PathBuf,
   449→    output: &Option<PathBuf>,
   450→) -> Result<(), String> {
   451→    let source = read_source(file)?;
   452→    let backend_kind = BackendKind::from(*isa);
   453→
   454→    // Step 1: Parse source → AST.
   455→    let mut parser = vuma_parser::Parser::new(&source);
   456→    let parse_result = parser.parse_program();
   457→    if parse_result.is_err() {
   458→        return Err(format!("parse error: {:?}", parse_result.errors));
   459→    }
   460→    if !parse_result.errors.is_empty() {
   461→        eprintln!("[emit] WARNING: {} non-fatal parse errors:", parse_result.errors.len());
   462→        for err in &parse_result.errors {
   463→            eprintln!("[emit]   {:?}", err);
   464→        }
   465→    }
   466→    let program = parse_result.value.unwrap();
   467→
   468→    // Step 2: Bridge parser AST → codegen SCG.
   469→    let codegen_scg = bridge_ast_to_codegen_scg(&program);
   470→
   471→    // Step 3: Lower codegen SCG → IR.
   472→    let mut ir_builder = ScgToIr::new();
   473→    let ir_program = ir_builder.convert(&codegen_scg).map_err(|e| {
   474→        format!("IR conversion error: {}", e)
   475→    })?;
   476→
   477→    eprintln!("[emit] IR program has {} functions", ir_program.functions.len());
   478→    for func in &ir_program.functions {
   479→        eprintln!("[emit] Function: {} ({} params, {} vregs)", func.name, func.params.len(), func.vregs.len());
   480→        for block in &func.blocks {
   481→            eprintln!("[emit]   Block: {}", block.label);
   482→            for instr in &block.instructions {
   483→                eprintln!("[emit]     {:?}", instr);
   484→            }
   485→        }
   486→    }
   487→
   488→    // Step 4: Create backend and allocate registers.
   489→    let backend = create_backend(backend_kind).map_err(|e| {
   490→        format!("error: cannot create {} backend: {}", backend_kind.isa_name(), e)
   491→    })?;
   492→
   493→    let mut allocated_functions = Vec::new();
   494→    for func in &ir_program.functions {
   495→        match backend.allocate_registers(func) {
   496→            Ok(allocated) => allocated_functions.push(allocated),
   497→            Err(e) => {
   498→                eprintln!("warning: register allocation failed for '{}': {}", func.name, e);
   499→            }
   500→        }
   501→    }
   502→
   503→    let out_path = output
   504→        .as_ref()
   505→        .cloned()
   506→        .unwrap_or_else(|| default_output_path(file));
   507→
   508→    // Step 5: Encode and write output.
   509→    if !allocated_functions.is_empty() {
   510→        let allocated_program = vuma_codegen::backend::AllocatedProgram {
   511→            functions: allocated_functions,
   512→            total_code_size: 0,
   513→            total_data_size: 0,
   514→        };
   515→        match backend.encode_program(&allocated_program) {
   516→            Ok(bytes) => {
   517→                fs::write(&out_path, &bytes).map_err(|e| {
   518→                    format!("error: cannot write output file '{}': {}", out_path.display(), e)
   519→                })?;
   520→                // Make the output file executable on Unix.
   521→                #[cfg(unix)]
   522→                {
   523→                    use std::os::unix::fs::PermissionsExt;
   524→                    let mut perms = fs::metadata(&out_path)
   525→                        .map_err(|e| format!("error: cannot stat '{}': {}", out_path.display(), e))?
   526→                        .permissions();
   527→                    perms.set_mode(0o755);
   528→                    fs::set_permissions(&out_path, perms)
   529→                        .map_err(|e| format!("error: cannot chmod '{}': {}", out_path.display(), e))?;
   530→                }
   531→                println!(
   532→                    "Emitted {} -> {} ({} bytes, ISA: {})",
   533→                    file.display(),
   534→                    out_path.display(),
   535→                    bytes.len(),
   536→                    backend.name(),
   537→                );
   538→            }
   539→            Err(e) => {
   540→                return Err(format!("{} encoding failed: {}", backend.name(), e));
   541→            }
   542→        }
   543→    } else {
   544→        return Err("no functions were successfully allocated".to_string());
   545→    }
   546→
   547→    Ok(())
   548→}

   550→/// Bridge parser AST → codegen SCG.
   551→///
   552→/// This converts the parser's AST into the codegen's SCG representation,
   553→/// which can then be lowered to IR. This bypasses the main pipeline's
   554→/// semantic SCG which loses most program semantics.
   555→///
   556→/// Key design: expressions are flattened into three-address code (TAC) by
   557→/// introducing temporary variables for sub-expressions. This preserves the
   558→/// full semantics of nested binary operations, function calls, dereferences,
   559→/// and casts.
   560→fn bridge_ast_to_codegen_scg(program: &vuma_parser::ast::Program) -> Scg {
   561→    use vuma_parser::ast::Item;
   562→
   563→    let mut nodes: Vec<ScgNode> = Vec::new();
   564→
   565→    for item in &program.items {
   566→        if let Item::FnDef(fn_def) = item {
   567→            let params: Vec<ScgParam> = fn_def
   568→                .params
   569→                .iter()
   570→                .map(|p| ScgParam {
   571→                    name: p.name.clone(),
   572→                    ty: bridge_type_to_codegen_scg(&p.ty),
   573→                })
   574→                .collect();
   575→
   576→            let results = if let Some(ref ret_ty) = fn_def.return_type {
   577→                vec![bridge_type_to_codegen_scg(&Some(ret_ty.clone()))]
   578→            } else {
   579→                vec![]
   580→            };
   581→
   582→            let mut ctx = BridgeCtx::new();
   583→            let mut body = bridge_block_to_scg_stmts(&fn_def.body, &mut ctx);
   584→
   585→            // Ensure every function ends with a Return statement.
   586→            // If the body doesn't end with a Return, add an implicit one.
   587→            // When the function has a return type and the last statement was an
   588→            // expression, use ctx.last_expr_result as the return value.
   589→            let has_return = body.last().map_or(false, |s| matches!(s, ScgStatement::Return(_)));
   590→            if !has_return {
   591→                let ret_val = if !results.is_empty() {
   592→                    // First check if the last expression was tracked.
   593→                    if let Some(ref expr) = ctx.last_expr_result {
   594→                        Some(expr.clone())
   595→                    } else {
   596→                        // Otherwise, look for the last computation/call result.
   597→                        body.iter().rev().find_map(|s| match s {
   598→                            ScgStatement::Computation(comp) => Some(ScgExpr::Var(comp.dst.clone())),
   599→                            ScgStatement::Call(call) => call.dst.as_ref().map(|d| ScgExpr::Var(d.clone())),
   600→                            _ => None,
   601→                        })
   602→                    }
   603→                } else {
   604→                    None
   605→                };
   606→                body.push(ScgStatement::Return(ret_val.into_iter().collect()));
   607→            }
   608→
   609→            nodes.push(ScgNode::Function(ScgFunction {
   610→                name: fn_def.name.clone(),
   611→                params,
   612→                results,
   613→                body,
   614→            }));
   615→        }
   616→    }
   617→
   618→    Scg { nodes }
   619→}
   620→
   621→/// Context for the AST → codegen SCG bridge, tracking a monotonic temp counter
   622→/// and the last expression result for implicit returns.
   623→struct BridgeCtx {
   624→    temp_counter: u32,
   625→    /// The result of the last expression statement, if any.
   626→    /// Used for implicit return when a function body ends with an expression.
   627→    last_expr_result: Option<ScgExpr>,
   628→}
   629→
   630→impl BridgeCtx {
   631→    fn new() -> Self {
   632→        Self { temp_counter: 0, last_expr_result: None }
   633→    }
   634→
   635→    /// Allocate a unique temporary variable name.
   636→    fn alloc_temp(&mut self) -> String {
   637→        let name = format!("__t{}", self.temp_counter);
   638→        self.temp_counter += 1;
   639→        name
   640→    }
   641→}

   643→/// Convert a parser type annotation to a codegen SCG type.
   644→fn bridge_type_to_codegen_scg(ty: &Option<vuma_parser::ast::Type>) -> ScgType {
   645→    match ty {
   646→        Some(vuma_parser::ast::Type::BDBase(name)) => match name.as_str() {
   647→            "i8" => ScgType::I8,
   648→            "i16" => ScgType::I16,
   649→            "i32" => ScgType::I32,
   650→            "i64" => ScgType::I64,
   651→            "u8" => ScgType::U8,
   652→            "u16" => ScgType::U16,
   653→            "u32" => ScgType::U32,
   654→            "u64" => ScgType::U64,
   655→            _ => ScgType::I64,
   656→        },
   657→        Some(vuma_parser::ast::Type::Ptr(_)) => ScgType::Ptr,
   658→        Some(vuma_parser::ast::Type::RegionPtr { .. }) => ScgType::Ptr,
   659→        _ => ScgType::Void,
   660→    }
   661→}

   664→/// Convert a parser block into codegen SCG statements, flattening expressions
   665→/// into three-address code with temporaries.
   666→fn bridge_block_to_scg_stmts(block: &vuma_parser::ast::Block, ctx: &mut BridgeCtx) -> Vec<ScgStatement> {
   667→    block
   668→        .statements
   669→        .iter()
   670→        .flat_map(|s| bridge_stmt_to_scg(s, ctx))
   671→        .collect()
   672→}
   673→
   674→/// Flatten an expression into three-address code. Returns the ScgExpr that
   675→/// holds the result, and appends any intermediate computation statements
   676→/// to `stmts`.
   677→///
   678→/// This is the core of the bridge: it recursively decomposes nested
   679→/// expressions into a sequence of simple computation nodes, each operating
   680→/// on at most two operands and producing one result. This preserves the
   681→/// full semantics of the original expression tree.
   682→fn flatten_expr(
   683→    expr: &vuma_parser::ast::Expr,
   684→    stmts: &mut Vec<ScgStatement>,
   685→    ctx: &mut BridgeCtx,
   686→) -> ScgExpr {
   687→    use vuma_parser::ast::{Expr, Lit, UnOp};
   688→
   689→    match expr {
   690→        // ── Leaf expressions: return directly, no flattening needed ──
   691→        Expr::Var { name, .. } => ScgExpr::Var(name.clone()),
   692→        Expr::Lit { value, .. } => match value {
   693→            Lit::Int(n) => ScgExpr::Int(*n),
   694→            Lit::Float(f) => ScgExpr::Float(*f),
   695→            Lit::Bool(b) => ScgExpr::Int(if *b { 1 } else { 0 }),
   696→            Lit::Address(a) => ScgExpr::Int(*a as i64),
   697→            Lit::String(_) => ScgExpr::Int(0),
   698→        },
   699→
   700→        // ── Binary operations: flatten lhs and rhs, then emit one Computation ──
   701→        Expr::BinOp { op, lhs, rhs, .. } => {
   702→            let lhs_expr = flatten_expr(lhs, stmts, ctx);
   703→            let rhs_expr = flatten_expr(rhs, stmts, ctx);
   704→            let dst = ctx.alloc_temp();
   705→            let binop_kind = map_ast_binop(op);
   706→            stmts.push(ScgStatement::Computation(ComputationNode {
   707→                dst: dst.clone(),
   708→                op: binop_kind,
   709→                lhs: lhs_expr,
   710→                rhs: rhs_expr,
   711→                tail_call: false,
   712→            }));
   713→            ScgExpr::Var(dst)
   714→        }
   715→
   716→        // ── Unary operations: flatten operand, then emit one Computation ──
   717→        Expr::UnOp { op, expr: operand, .. } => {
   718→            let operand_expr = flatten_expr(operand, stmts, ctx);
   719→            let dst = ctx.alloc_temp();
   720→            match op {
   721→                UnOp::BitNot => {
   722→                    stmts.push(ScgStatement::Computation(ComputationNode {
   723→                        dst: dst.clone(),
   724→                        op: BinOpKind::Xor,
   725→                        lhs: operand_expr,
   726→                        rhs: ScgExpr::Int(-1),
   727→                        tail_call: false,
   728→                    }));
   729→                }
   730→                UnOp::Neg => {
   731→                    stmts.push(ScgStatement::Computation(ComputationNode {
   732→                        dst: dst.clone(),
   733→                        op: BinOpKind::Sub,
   734→                        lhs: ScgExpr::Int(0),
   735→                        rhs: operand_expr,
   736→                        tail_call: false,
   737→                    }));
   738→                }
   739→                UnOp::Not => {
   740→                    stmts.push(ScgStatement::Computation(ComputationNode {
   741→                        dst: dst.clone(),
   742→                        op: BinOpKind::Eq,
   743→                        lhs: operand_expr,
   744→                        rhs: ScgExpr::Int(0),
   745→                        tail_call: false,
   746→                    }));
   747→                }
   748→                UnOp::Deref => {
   749→                    stmts.push(ScgStatement::Access(AccessNode::Load {
   750→                        dst: dst.clone(),
   751→                        ptr: operand_expr,
   752→                        offset: None,
   753→                    }));
   754→                }
   755→            }
   756→            ScgExpr::Var(dst)
   757→        }
   758→
   759→        // ── Function call: flatten args, emit CallNode ──
   760→        Expr::Call { callee, args, .. } => {
   761→            let func_name = match callee.as_ref() {
   762→                Expr::Var { name, .. } => name.clone(),
   763→                _ => "_unknown".into(),
   764→            };
   765→            let flat_args: Vec<ScgExpr> = args.iter()
   766→                .map(|a| flatten_expr(a, stmts, ctx))
   767→                .collect();
   768→            let dst = ctx.alloc_temp();
   769→            stmts.push(ScgStatement::Call(CallNode {
   770→                dst: Some(dst.clone()),
   771→                func: func_name,
   772→                args: flat_args,
   773→            }));
   774→            ScgExpr::Var(dst)
   775→        }
   776→
   777→        // ── Dereference: flatten the address, emit Load ──
   778→        Expr::Deref { expr, .. } => {
   779→            let addr = flatten_expr(expr, stmts, ctx);
   780→            let dst = ctx.alloc_temp();
   781→            stmts.push(ScgStatement::Access(AccessNode::Load {
   782→                dst: dst.clone(),
   783→                ptr: addr,
   784→                offset: None,
   785→            }));
   786→            ScgExpr::Var(dst)
   787→        }
   788→
   789→        // ── Offset (pointer arithmetic): flatten base and offset, emit Add ──
   790→        Expr::Offset { base, offset, .. } => {
   791→            let base_expr = flatten_expr(base, stmts, ctx);
   792→            let off_expr = flatten_expr(offset, stmts, ctx);
   793→            let dst = ctx.alloc_temp();
   794→            stmts.push(ScgStatement::Computation(ComputationNode {
   795→                dst: dst.clone(),
   796→                op: BinOpKind::Add,
   797→                lhs: base_expr,
   798→                rhs: off_expr,
   799→                tail_call: false,
   800→            }));
   801→            ScgExpr::Var(dst)
   802→        }
   803→
   804→        // ── Cast: flatten operand, pass through ──
   805→        Expr::Cast { expr, .. } => flatten_expr(expr, stmts, ctx),
   806→
   807→        // ── TypeAscription: flatten inner expression ──
   808→        Expr::TypeAscription { expr, .. } => flatten_expr(expr, stmts, ctx),
   809→
   810→        // ── Index: flatten base and index, compute addr, emit Load ──
   811→        Expr::Index { expr, index, .. } => {
   812→            let base_expr = flatten_expr(expr, stmts, ctx);
   813→            let idx_expr = flatten_expr(index, stmts, ctx);
   814→            let addr = ctx.alloc_temp();
   815→            stmts.push(ScgStatement::Computation(ComputationNode {
   816→                dst: addr.clone(),
   817→                op: BinOpKind::Add,
   818→                lhs: base_expr,
   819→                rhs: idx_expr,
   820→                tail_call: false,
   821→            }));
   822→            let dst = ctx.alloc_temp();
   823→            stmts.push(ScgStatement::Access(AccessNode::Load {
   824→                dst: dst.clone(),
   825→                ptr: ScgExpr::Var(addr),
   826→                offset: None,
   827→            }));
   828→            ScgExpr::Var(dst)
   829→        }
   830→
   831→        // ── Range: just flatten start (range is handled by For) ──
   832→        Expr::Range { start, .. } => flatten_expr(start, stmts, ctx),
   833→
   834→        // ── Allocate: emit as a heap allocation call ──
   835→        Expr::Allocate { size, .. } => {
   836→            let size_expr = flatten_expr(size, stmts, ctx);
   837→            let dst = ctx.alloc_temp();
   838→            stmts.push(ScgStatement::Call(CallNode {
   839→                dst: Some(dst.clone()),
   840→                func: "__vuma_alloc".to_string(),
   841→                args: vec![size_expr],
   842→            }));
   843→            ScgExpr::Var(dst)
   844→        }
   845→
   846→        // ── Null → 0 ──
   847→        Expr::Null { .. } => ScgExpr::Int(0),
   848→
   849→        // ── Uninitialized → 0 ──
   850→        Expr::Uninitialized { .. } => ScgExpr::Int(0),
   851→
   852→        // ── Fallback for unsupported expression types ──
   853→        _ => ScgExpr::Int(0),
   854→    }
   855→}
   856→
   857→/// Map a VUMA AST BinOp to a codegen BinOpKind.
   858→fn map_ast_binop(op: &vuma_parser::ast::BinOp) -> BinOpKind {
   859→    use vuma_parser::ast::BinOp;
   860→    match op {
   861→        BinOp::Add => BinOpKind::Add,
   862→        BinOp::Sub => BinOpKind::Sub,
   863→        BinOp::Mul => BinOpKind::Mul,
   864→        BinOp::Div => BinOpKind::SDiv,
   865→        BinOp::Mod => BinOpKind::SRem,
   866→        BinOp::And => BinOpKind::And,
   867→        BinOp::Or => BinOpKind::Or,
   868→        BinOp::BitAnd => BinOpKind::And,
   869→        BinOp::BitOr => BinOpKind::Or,
   870→        BinOp::BitXor => BinOpKind::Xor,
   871→        BinOp::Shl => BinOpKind::Shl,
   872→        BinOp::Shr => BinOpKind::ShrL,
   873→        BinOp::Eq => BinOpKind::Eq,
   874→        BinOp::Ne => BinOpKind::Ne,
   875→        BinOp::Lt => BinOpKind::SLt,
   876→        BinOp::Le => BinOpKind::SLe,
   877→        BinOp::Gt => BinOpKind::SGt,
   878→        BinOp::Ge => BinOpKind::SGe,
   879→    }
   880→}

   882→/// Convert a single parser statement into zero or more codegen SCG statements.
   883→/// Uses `flatten_expr` to decompose nested expressions into three-address code.
   884→fn bridge_stmt_to_scg(stmt: &vuma_parser::ast::Stmt, ctx: &mut BridgeCtx) -> Vec<ScgStatement> {
   885→    use vuma_parser::ast::Stmt as PStmt;
   886→
   887→    match stmt {
   888→        // ── let x [: T] = expr ──
   889→        PStmt::Let(let_stmt) => {
   890→            let mut stmts = Vec::new();
   891→
   892→            // Check if the RHS is an allocate() call → AllocationNode::Stack
   893→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &let_stmt.value {
   894→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
   895→                    if name == "allocate" {
   896→                        let size: u32 = args.first()
   897→                            .and_then(|a| {
   898→                                if let vuma_parser::ast::Expr::Lit { value, .. } = a {
   899→                                    if let vuma_parser::ast::Lit::Int(n) = value {
   900→                                        return Some(*n as u32);
   901→                                    }
   902→                                }
   903→                                None
   904→                            })
   905→                            .unwrap_or(8);
   906→                        return vec![ScgStatement::Allocation(AllocationNode::Stack {
   907→                            name: let_stmt.name.clone(),
   908→                            size,
   909→                            ty: ScgType::Ptr,
   910→                        })];
   911→                    }
   912→                    // Other function calls → CallNode (flatten args)
   913→                    let flat_args: Vec<ScgExpr> = args.iter()
   914→                        .map(|a| flatten_expr(a, &mut stmts, ctx))
   915→                        .collect();
   916→                    stmts.push(ScgStatement::Call(CallNode {
   917→                        dst: Some(let_stmt.name.clone()),
   918→                        func: name.clone(),
   919→                        args: flat_args,
   920→                    }));
   921→                    return stmts;
   922→                }
   923→            }
   924→
   925→            // Check if the RHS is an Allocate expression → AllocationNode::Stack
   926→            if let vuma_parser::ast::Expr::Allocate { size, .. } = &let_stmt.value {
   927→                let size_val: u32 = match size.as_ref() {
   928→                    vuma_parser::ast::Expr::Lit { value, .. } => {
   929→                        if let vuma_parser::ast::Lit::Int(n) = value {
   930→                            *n as u32
   931→                        } else {
   932→                            8
   933→                        }
   934→                    }
   935→                    _ => 8,
   936→                };
   937→                return vec![ScgStatement::Allocation(AllocationNode::Stack {
   938→                    name: let_stmt.name.clone(),
   939→                    size: size_val,
   940→                    ty: ScgType::Ptr,
   941→                })];
   942→            }
   943→
   944→            // General case: flatten the expression and assign to dst
   945→            let result = flatten_expr(&let_stmt.value, &mut stmts, ctx);
   946→            match &result {
   947→                ScgExpr::Var(name) if name == &let_stmt.name => {}
   948→                _ => {
   949→                    stmts.push(ScgStatement::Computation(ComputationNode {
   950→                        dst: let_stmt.name.clone(),
   951→                        op: BinOpKind::Add,
   952→                        lhs: result,
   953→                        rhs: ScgExpr::Int(0),
   954→                        tail_call: false,
   955→                    }));
   956→                }
   957→            }
   958→            stmts
   959→        }
   960→
   961→        // ── target = value ──
   962→        PStmt::Assign(assign_stmt) => {
   963→            let mut stmts = Vec::new();
   964→
   965→            // Detect dereference writes: `*expr = val` → Access::Store
   966→            if let vuma_parser::ast::AssignTarget::Deref { expr, .. } = &assign_stmt.target {
   967→                let ptr = flatten_expr(expr, &mut stmts, ctx);
   968→                let value = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
   969→                stmts.push(ScgStatement::Access(AccessNode::Store {
   970→                    ptr,
   971→                    offset: None,
   972→                    value,
   973→                }));
   974→                return stmts;
   975→            }
   976→
   977→            // Handle Index target: ptr[index] = value
   978→            if let vuma_parser::ast::AssignTarget::Index { expr, index, .. } = &assign_stmt.target {
   979→                let base = flatten_expr(expr, &mut stmts, ctx);
   980→                let idx = flatten_expr(index, &mut stmts, ctx);
   981→                let addr = ctx.alloc_temp();
   982→                stmts.push(ScgStatement::Computation(ComputationNode {
   983→                    dst: addr.clone(),
   984→                    op: BinOpKind::Add,
   985→                    lhs: base,
   986→                    rhs: idx,
   987→                    tail_call: false,
   988→                }));
   989→                let value = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
   990→                stmts.push(ScgStatement::Access(AccessNode::Store {
   991→                    ptr: ScgExpr::Var(addr),
   992→                    offset: None,
   993→                    value,
   994→                }));
   995→                return stmts;
   996→            }
   997→
   998→            let dst = match &assign_stmt.target {
   999→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
  1000→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
  1001→                vuma_parser::ast::AssignTarget::Deref { .. } => "_deref".into(),
  1002→                vuma_parser::ast::AssignTarget::Index { .. } => "_index".into(),
  1003→            };
  1004→
  1005→            // Detect allocate() expression → AllocationNode::Stack
  1006→            if let vuma_parser::ast::Expr::Allocate { size, .. } = &assign_stmt.value {
  1007→                let size_val: u32 = match size.as_ref() {
  1008→                    vuma_parser::ast::Expr::Lit { value, .. } => {
  1009→                        if let vuma_parser::ast::Lit::Int(n) = value {
  1010→                            *n as u32
  1011→                        } else {
  1012→                            8
  1013→                        }
  1014→                    }
  1015→                    _ => 8,
  1016→                };
  1017→                return vec![ScgStatement::Allocation(AllocationNode::Stack {
  1018→                    name: dst,
  1019→                    size: size_val,
  1020→                    ty: ScgType::Ptr,
  1021→                })];
  1022→            }
  1023→
  1024→            // Detect function calls in assign: `x = foo(args)` → CallNode (flatten args)
  1025→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &assign_stmt.value {
  1026→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
  1027→                    let flat_args: Vec<ScgExpr> = args.iter()
  1028→                        .map(|a| flatten_expr(a, &mut stmts, ctx))
  1029→                        .collect();
  1030→                    stmts.push(ScgStatement::Call(CallNode {
  1031→                        dst: Some(dst),
  1032→                        func: name.clone(),
  1033→                        args: flat_args,
  1034→                    }));
  1035→                    return stmts;
  1036→                }
  1037→            }
  1038→
  1039→            // General case: flatten the value expression and assign to dst
  1040→            let result = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
  1041→            match &result {
  1042→                ScgExpr::Var(name) if name == &dst => {}
  1043→                _ => {
  1044→                    stmts.push(ScgStatement::Computation(ComputationNode {
  1045→                        dst,
  1046→                        op: BinOpKind::Add,
  1047→                        lhs: result,
  1048→                        rhs: ScgExpr::Int(0),
  1049→                        tail_call: false,
  1050→                    }));
  1051→                }
  1052→            }
  1053→            stmts
  1054→        }
  1055→
  1056→        // ── target op= value ──
  1057→        PStmt::CompoundAssign(ca_stmt) => {
  1058→            let mut stmts = Vec::new();
  1059→            let dst = match &ca_stmt.target {
  1060→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
  1061→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
  1062→                _ => "_".into(),
  1063→            };
  1064→            let binop = match ca_stmt.op {
  1065→                vuma_parser::ast::CompoundOp::Add => BinOpKind::Add,
  1066→                vuma_parser::ast::CompoundOp::Sub => BinOpKind::Sub,
  1067→                vuma_parser::ast::CompoundOp::Mul => BinOpKind::Mul,
  1068→                vuma_parser::ast::CompoundOp::Div => BinOpKind::SDiv,
  1069→                vuma_parser::ast::CompoundOp::Mod => BinOpKind::SRem,
  1070→                vuma_parser::ast::CompoundOp::BitAnd => BinOpKind::And,
  1071→                vuma_parser::ast::CompoundOp::BitOr => BinOpKind::Or,
  1072→                vuma_parser::ast::CompoundOp::BitXor => BinOpKind::Xor,
  1073→                vuma_parser::ast::CompoundOp::Shl => BinOpKind::Shl,
  1074→                vuma_parser::ast::CompoundOp::Shr => BinOpKind::ShrL,
  1075→            };
  1076→            let rhs = flatten_expr(&ca_stmt.value, &mut stmts, ctx);
  1077→            stmts.push(ScgStatement::Computation(ComputationNode {
  1078→                dst: dst.clone(),
  1079→                op: binop,
  1080→                lhs: ScgExpr::Var(dst),
  1081→                rhs,
  1082→                tail_call: false,
  1083→            }));
  1084→            stmts
  1085→        }
  1086→
  1087→        PStmt::Allocate(_) => vec![],
  1088→        PStmt::Free(_) => vec![],
  1089→
  1090→        PStmt::Cast(cast_stmt) => {
  1091→            let mut stmts = Vec::new();
  1092→            let _ = flatten_expr(&cast_stmt.expr, &mut stmts, ctx);
  1093→            stmts
  1094→        }
  1095→
  1096→        PStmt::Return(ret_stmt) => {
  1097→            let mut stmts = Vec::new();
  1098→            let values = match &ret_stmt.value {
  1099→                Some(expr) => vec![flatten_expr(expr, &mut stmts, ctx)],
  1100→                None => vec![],
  1101→            };
  1102→            stmts.push(ScgStatement::Return(values));
  1103→            stmts
  1104→        }
  1105→
  1106→        PStmt::Expr(expr_stmt) => {
  1107→            let mut stmts = Vec::new();
  1108→            let result = flatten_expr(&expr_stmt.expr, &mut stmts, ctx);
  1109→            ctx.last_expr_result = Some(result);
  1110→            stmts
  1111→        }
  1112→
  1113→        PStmt::If(if_stmt) => {
  1114→            let mut pre_stmts = Vec::new();
  1115→            let cond = flatten_expr(&if_stmt.condition, &mut pre_stmts, ctx);
  1116→            let then_body = bridge_block_to_scg_stmts(&if_stmt.then_block, ctx);
  1117→            let else_body = if_stmt
  1118→                .else_block
  1119→                .as_ref()
  1120→                .map(|b| bridge_block_to_scg_stmts(b, ctx));
  1121→            let mut result = pre_stmts;
  1122→            result.push(ScgStatement::Control(ControlNode::If {
  1123→                cond,
  1124→                then_body,
  1125→                else_body,
  1126→            }));
  1127→            result
  1128→        }
  1129→
  1130→        // while condition { body }
  1131→        // Lowered as: Loop { <compute cond>; if cond { body } else { break } }
  1132→        PStmt::While(while_stmt) => {
  1133→            let then_body = bridge_block_to_scg_stmts(&while_stmt.body, ctx);
  1134→            let mut loop_body = Vec::new();
  1135→            let cond = flatten_expr(&while_stmt.condition, &mut loop_body, ctx);
  1136→            loop_body.push(ScgStatement::Control(ControlNode::If {
  1137→                cond,
  1138→                then_body,
  1139→                else_body: Some(vec![ScgStatement::Control(ControlNode::Break)]),
  1140→            }));
  1141→            vec![ScgStatement::Control(ControlNode::Loop { body: loop_body })]
  1142→        }
  1143→
  1144→        // for name in start..end { body }
  1145→        // Lowered as:
  1146→        //   name = start
  1147→        //   loop { _cond = name < end; if _cond { body; name = name + 1 } else { break } }
  1148→        PStmt::For(for_stmt) => {
  1149→            let mut pre_stmts = Vec::new();
  1150→            let (start_expr, end_expr) = match &for_stmt.iter {
  1151→                vuma_parser::ast::Expr::Range { start, end, .. } => {
  1152→                    let s = flatten_expr(start, &mut pre_stmts, ctx);
  1153→                    let e = flatten_expr(end, &mut pre_stmts, ctx);
  1154→                    (s, e)
  1155→                }
  1156→                other => {
  1157→                    (ScgExpr::Int(0), ScgExpr::Int(0))
  1158→                }
  1159→            };
  1160→
  1161→            let init_stmt = ScgStatement::Computation(ComputationNode {
  1162→                dst: for_stmt.name.clone(),
  1163→                op: BinOpKind::Add,
  1164→                lhs: start_expr,
  1165→                rhs: ScgExpr::Int(0),
  1166→                tail_call: false,
  1167→            });
  1168→
  1169→            let mut loop_body = Vec::new();
  1170→            let cond_temp = format!("{}_cond", for_stmt.name);
  1171→            loop_body.push(ScgStatement::Computation(ComputationNode {
  1172→                dst: cond_temp.clone(),
  1173→                op: BinOpKind::SLt,
  1174→                lhs: ScgExpr::Var(for_stmt.name.clone()),
  1175→                rhs: end_expr.clone(),
  1176→                tail_call: false,
  1177→            }));
  1178→
  1179→            let inner_body = bridge_block_to_scg_stmts(&for_stmt.body, ctx);
  1180→            let mut full_then = inner_body;
  1181→            full_then.push(ScgStatement::Computation(ComputationNode {
  1182→                dst: for_stmt.name.clone(),
  1183→                op: BinOpKind::Add,
  1184→                lhs: ScgExpr::Var(for_stmt.name.clone()),
  1185→                rhs: ScgExpr::Int(1),
  1186→                tail_call: false,
  1187→            }));
  1188→
  1189→            loop_body.push(ScgStatement::Control(ControlNode::If {
  1190→                cond: ScgExpr::Var(cond_temp),
  1191→                then_body: full_then,
  1192→                else_body: Some(vec![ScgStatement::Control(ControlNode::Break)]),
  1193→            }));
  1194→
  1195→            let mut result = pre_stmts;
  1196→            result.push(init_stmt);
  1197→            result.push(ScgStatement::Control(ControlNode::Loop { body: loop_body }));
  1198→            result
  1199→        }
  1200→
  1201→        PStmt::Loop(loop_stmt) => {
  1202→            let body = bridge_block_to_scg_stmts(&loop_stmt.body, ctx);
  1203→            vec![ScgStatement::Control(ControlNode::Loop { body })]
  1204→        }
  1205→
  1206→        PStmt::Break(_) => vec![ScgStatement::Control(ControlNode::Break)],
  1207→        PStmt::Continue(_) => vec![ScgStatement::Control(ControlNode::Continue)],
  1208→        PStmt::BdDirective(_)
  1209→        | PStmt::Sync(_)
  1210→        | PStmt::UnsafeBlock { .. }
  1211→        | PStmt::Match(_)
  1212→        | PStmt::Access(_) => vec![],
  1213→    }
  1214→}
  1215→fn cmd_disasm(file: &PathBuf, isa: &IsaArg, base_addr_str: &str) -> Result<(), String> {
  1216→    let bytes = fs::read(file)
  1217→        .map_err(|e| format!("error: cannot read binary file '{}': {}", file.display(), e))?;
  1218→
  1219→    let base_addr = u64::from_str_radix(base_addr_str.trim_start_matches("0x"), 16)
  1220→        .map_err(|e| format!("error: invalid base address '{}': {}", base_addr_str, e))?;
  1221→
  1222→    let backend_kind = BackendKind::from(*isa);
  1223→    let backend = create_backend(backend_kind).map_err(|e| {
  1224→        format!(
  1225→            "error: cannot create {} backend: {}",
  1226→            backend_kind.isa_name(),
  1227→            e
  1228→        )
  1229→    })?;
  1230→
  1231→    let instructions = backend.disassemble(&bytes, base_addr);
  1232→
  1233→    println!(
  1234→        "Disassembly of {} ({} bytes, ISA: {}):",
  1235→        file.display(),
  1236→        bytes.len(),
  1237→        backend.name()
  1238→    );
  1239→    for line in &instructions {
  1240→        println!("{}", line);
  1241→    }
  1242→
  1243→    Ok(())
  1244→}

  1246→/// `vuma verify <file>` — Run IVE 5-invariant verification.
  1247→fn cmd_verify(cli: &Cli, file: &PathBuf) -> Result<(), String> {
  1248→    let source = read_source(file)?;
  1249→    let mut config = make_config(cli, CompileTarget::Linux);
  1250→    // Force exhaustive verification for the verify subcommand.
  1251→    config.verification_level = VerificationLevel::Exhaustive;
  1252→
  1253→    let result = compile_with_path(&source, Some(file), &config).map_err(|errors| {
  1254→        print_errors(&errors);
  1255→        format!(
  1256→            "compilation/verification failed with {} error(s)",
  1257→            errors.len()
  1258→        )
  1259→    })?;
  1260→
  1261→    match result.verification {
  1262→        Some(ref verification) => {
  1263→            println!("IVE Verification Results for {}", file.display());
  1264→            println!("  Overall verdict: {}", verification.overall);
  1265→            println!(
  1266→                "  Summary: {} invariant(s) checked",
  1267→                verification.summary.total_checked
  1268→            );
  1269→            println!("           {} passed", verification.summary.passed);
  1270→            println!("           {} failed", verification.summary.failed);
  1271→            println!("           {} unverified", verification.summary.unverified);
  1272→
  1273→            // Print individual invariant results.
  1274→            for per_inv in &verification.per_invariant {
  1275→                let status_str = match &per_inv.result.status {
  1276→                    vuma_ive::VerificationStatus::Proven => "PROVEN",
  1277→                    vuma_ive::VerificationStatus::ProbablySafe { .. } => "PROBABLY_SAFE",
  1278→                    vuma_ive::VerificationStatus::Unverified { .. } => "UNVERIFIED",
  1279→                    vuma_ive::VerificationStatus::Violated { .. } => "VIOLATED",
  1280→                };
  1281→                println!(
  1282→                    "  [{}] {:?} — {}",
  1283→                    status_str, per_inv.kind, per_inv.result.message,
  1284→                );
  1285→                if let Some(ref evidence) = per_inv.result.evidence {
  1286→                    println!("    evidence: {}", evidence);
  1287→                }
  1288→            }
  1289→        }
  1290→        None => {
  1291→            println!("Verification was skipped (verification level set to None).");
  1292→        }
  1293→    }
  1294→
  1295→    Ok(())
  1296→}

  1298→/// `vuma repl` — Interactive REPL: parse expressions and print AST.
  1299→fn cmd_repl() -> Result<(), String> {
  1300→    println!("VUMA REPL v0.1.0");
  1301→    println!("Type VUMA expressions or statements. Enter ':quit' to exit, ':help' for help.");
  1302→
  1303→    let stdin = io::stdin();
  1304→    let mut input = String::new();
  1305→
  1306→    loop {
  1307→        print!("vuma> ");
  1308→        io::stdout()
  1309→            .flush()
  1310→            .map_err(|e| format!("flush error: {}", e))?;
  1311→
  1312→        input.clear();
  1313→        match stdin.read_line(&mut input) {
  1314→            Ok(0) => break, // EOF
  1315→            Ok(_) => {}
  1316→            Err(e) => return Err(format!("error reading input: {}", e)),
  1317→        }
  1318→
  1319→        let trimmed = input.trim();
  1320→        if trimmed.is_empty() {
  1321→            continue;
  1322→        }
  1323→        if trimmed == ":quit" || trimmed == ":q" {
  1324→            break;
  1325→        }
  1326→        if trimmed == ":help" || trimmed == ":h" {
  1327→            println!("  :quit, :q   — Exit the REPL");
  1328→            println!("  :help, :h   — Show this help");
  1329→            println!("  <expr>      — Parse and display the AST");
  1330→            println!("  <stmt>      — Parse and display the AST");
  1331→            println!("  <program>   — Parse and display the full program AST");
  1332→            continue;
  1333→        }
  1334→
  1335→        // Try to parse as a full program first.
  1336→        let mut parser = vuma_parser::Parser::new(trimmed);
  1337→        let result = parser.parse_program();
  1338→        if !result.has_errors() {
  1339→            // Successfully parsed; display the AST.
  1340→            let program = result.unwrap();
  1341→            println!("{:#?}", program);
  1342→            continue;
  1343→        }
  1344→        // Program parse had errors; try to parse as an expression via
  1345→        // wrapping it in a dummy program context.
  1346→        let program_errors = result.errors.clone();
  1347→        // Since parse_expr is private, we re-parse as a minimal program.
  1348→        let wrapped = format!("fn _repl_expr() {{ {} }}", trimmed);
  1349→        let mut expr_parser = vuma_parser::Parser::new(&wrapped);
  1350→        let expr_result = expr_parser.parse_program();
  1351→        if !expr_result.has_errors() {
  1352→            // Print the function body AST.
  1353→            let program = expr_result.unwrap();
  1354→            if let Some(item) = program.items.first() {
  1355→                println!("{:#?}", item);
  1356→            } else {
  1357→                println!("{:#?}", program);
  1358→            }
  1359→        } else {
  1360→            // Show the original parse errors.
  1361→            for err in &program_errors {
  1362→                eprintln!("parse error: {}", err);
  1363→            }
  1364→        }
  1365→    }
  1366→
  1367→    Ok(())
  1368→}

  1609→// ═══════════════════════════════════════════════════════════════════════════
  1610→
  1611→fn main() {
  1612→    // Initialize logger.
  1613→    env_logger::init();
  1614→
  1615→    let cli = Cli::parse();
  1616→
  1617→    // Handle --bench flag: run the benchmark suite.
  1618→    if cli.bench {
  1619→        cmd_bench(&cli);
  1620→        return;
  1621→    }
  1622→
  1623→    // Handle --repl flag: launch the full VumaRepl instead of subcommand.
  1624→    if cli.repl {
  1625→        let mut repl = vuma_core::repl::VumaRepl::new();
  1626→        if let Err(e) = repl.run() {
  1627→            eprintln!("REPL error: {e}");
  1628→            std::process::exit(1);
  1629→        }
  1630→        return;
  1631→    }
  1632→
  1633→    let command = match cli.command {
  1634→        Some(ref cmd) => cmd.clone(),
  1635→        None => {
  1636→            // No subcommand and no --repl: print help.
  1637→            eprintln!("No subcommand specified. Use `vuma --help` or `vuma --repl`.");
  1638→            std::process::exit(1);
  1639→        }
  1640→    };
  1641→
  1642→    let result = match command {
  1643→        Commands::Build {
  1644→            ref file,
  1645→            ref output,
  1646→            ref target,
  1647→        } => cmd_build(&cli, file, output, target),
  1648→        Commands::Run { ref file, ref args } => cmd_run(&cli, file, args),
  1649→        Commands::Check { ref file } => cmd_check(&cli, file),
  1650→        Commands::Emit {
  1651→            ref isa,
  1652→            ref file,
  1653→            ref output,
  1654→        } => cmd_emit(&cli, isa, file, output),
  1655→        Commands::Disasm {
  1656→            ref file,
  1657→            ref isa,
  1658→            ref base_addr,
  1659→        } => cmd_disasm(file, isa, base_addr),
  1660→        Commands::Verify { ref file } => cmd_verify(&cli, file),
  1661→        Commands::Repl => cmd_repl(),
  1662→        Commands::Lsp => cmd_lsp(),
  1663→    };
  1664→
  1665→    if let Err(err) = result {
  1666→        eprintln!("{}", err);
  1667→        std::process::exit(1);
  1668→    }
  1669→}
