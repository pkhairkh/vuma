     1→//! Structured JSON diagnostics output for LLM integration.
     2→//!
     3→//! This module provides machine-readable diagnostic messages that can be
     4→//! serialized to JSON, enabling LLM-based tools and IDE integrations to
     5→//! programmatically consume compiler errors, warnings, and hints.
     6→//!
     7→//! # Architecture
     8→//!
     9→//! Each diagnostic carries a structured error code (`"E001"`, `"W001"`, etc.),
    10→//! a severity level, a human-readable message, a source identifier (which
    11→//! compiler stage produced it), a precise source location, optional related
    12→//! information, and quick-fix suggestions.
    13→//!
    14→//! # Example JSON output
    15→//!
    16→//! ```json
    17→//! [
    18→//!   {
    19→//!     "code": "E002",
    20→//!     "severity": "error",
    21→//!     "message": "undefined variable `foo`",
    22→//!     "source": "parser",
    23→//!     "location": {
    24→//!       "file": "main.vu",
    25→//!       "start_line": 10,
    26→//!       "start_col": 4,
    27→//!       "end_line": 10,
    28→//!       "end_col": 7
    29→//!     },
    30→//!     "related": [],
    31→//!     "suggestions": ["did you mean `for`?"]
    32→//!   }
    33→//! ]
    34→//! ```
    35→//!
    36→//! # Integration
    37→//!
    38→//! Use [`from_parse_error`] to convert parser errors, [`from_vuma_error`] for
    39→//! pipeline-level errors, and [`from_codegen_error`] for code-generation
    40→//! failures.
    41→
    42→use serde::{Deserialize, Serialize};
    43→use std::fmt;
    44→
    45→use vuma_codegen::CodegenError;
    46→use vuma_parser::{offset_to_location, ParseError, ParseErrorKind, Severity as ParserSeverity};
    47→
    48→use crate::pipeline::VumaError;

    50→// ═══════════════════════════════════════════════════════════════════════════
    51→// DiagnosticSeverity
    52→// ═══════════════════════════════════════════════════════════════════════════
    53→
    54→/// Severity level for a machine-readable diagnostic.
    55→///
    56→/// Unlike the parser's `Severity` (which has `Note`), this uses `Hint` for
    57→/// LLM-consumable suggestions and `Info` for informational notes.
    58→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
    59→#[serde(rename_all = "lowercase")]
    60→pub enum DiagnosticSeverity {
    61→    /// A hard error — the program cannot be compiled.
    62→    Error,
    63→    /// A warning — suspicious code that is technically valid.
    64→    Warning,
    65→    /// An informational note, typically providing context.
    66→    Info,
    67→    /// A hint or quick-fix suggestion.
    68→    Hint,
    69→}
    70→
    71→impl fmt::Display for DiagnosticSeverity {
    72→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
    73→        match self {
    74→            DiagnosticSeverity::Error => write!(f, "error"),
    75→            DiagnosticSeverity::Warning => write!(f, "warning"),
    76→            DiagnosticSeverity::Info => write!(f, "info"),
    77→            DiagnosticSeverity::Hint => write!(f, "hint"),
    78→        }
    79→    }
    80→}
    81→
    82→impl From<ParserSeverity> for DiagnosticSeverity {
    83→    fn from(s: ParserSeverity) -> Self {
    84→        match s {
    85→            ParserSeverity::Error => DiagnosticSeverity::Error,
    86→            ParserSeverity::Warning => DiagnosticSeverity::Warning,
    87→            ParserSeverity::Note => DiagnosticSeverity::Info,
    88→        }
    89→    }
    90→}

    93→// DiagnosticSourceLocation
    94→// ═══════════════════════════════════════════════════════════════════════════
    95→
    96→/// Machine-readable source location with line/column ranges.
    97→///
    98→/// All positions are 1-based (as is conventional for compiler diagnostics)
    99→/// rather than the 0-based positions used internally by the parser.
   100→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   101→pub struct DiagnosticSourceLocation {
   102→    /// File path or module name.
   103→    pub file: String,
   104→    /// Start line (1-based).
   105→    pub start_line: u32,
   106→    /// Start column (1-based).
   107→    pub start_col: u32,
   108→    /// End line (1-based).
   109→    pub end_line: u32,
   110→    /// End column (1-based, exclusive).
   111→    pub end_col: u32,
   112→}
   113→
   114→impl DiagnosticSourceLocation {
   115→    /// Create a location spanning a single point (zero-width).
   116→    pub fn point(file: impl Into<String>, line: u32, col: u32) -> Self {
   117→        Self {
   118→            file: file.into(),
   119→            start_line: line,
   120→            start_col: col,
   121→            end_line: line,
   122→            end_col: col,
   123→        }
   124→    }
   125→
   126→    /// Create a location spanning a range within a single line.
   127→    pub fn range(
   128→        file: impl Into<String>,
   129→        line: u32,
   130→        start_col: u32,
   131→        end_col: u32,
   132→    ) -> Self {
   133→        Self {
   134→            file: file.into(),
   135→            start_line: line,
   136→            start_col,
   137→            end_line: line,
   138→            end_col,
   139→        }
   140→    }
   141→
   142→    /// Create a synthetic "unknown" location for diagnostics that cannot be
   143→    /// tied to a specific source position.
   144→    pub fn unknown() -> Self {
   145→        Self {
   146→            file: String::new(),
   147→            start_line: 0,
   148→            start_col: 0,
   149→            end_line: 0,
   150→            end_col: 0,
   151→        }
   152→    }
   153→
   154→    /// Returns `true` if this location represents an unknown / synthetic
   155→    /// position (all fields zero, no file).
   156→    pub fn is_unknown(&self) -> bool {
   157→        self.file.is_empty() && self.start_line == 0 && self.start_col == 0
   158→    }
   159→}
   160→
   161→impl fmt::Display for DiagnosticSourceLocation {
   162→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   163→        if self.is_unknown() {
   164→            write!(f, "<unknown>")
   165→        } else if self.start_line == self.end_line && self.start_col == self.end_col {
   166→            write!(f, "{}:{}:{}", self.file, self.start_line, self.start_col)
   167→        } else {
   168→            write!(
   169→                f,
   170→                "{}:{}:{}-{}:{}",
   171→                self.file, self.start_line, self.start_col, self.end_line, self.end_col
   172→            )
   173→        }
   174→    }
   175→}

   178→// RelatedInfo
   179→// ═══════════════════════════════════════════════════════════════════════════
   180→
   181→/// Additional context for a diagnostic: a related source location with a
   182→/// message explaining its relevance.
   183→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   184→pub struct RelatedInfo {
   185→    /// The related source location.
   186→    pub location: DiagnosticSourceLocation,
   187→    /// Why this location is relevant to the primary diagnostic.
   188→    pub message: String,
   189→}
   190→
   191→impl RelatedInfo {
   192→    /// Create a new related-info entry.
   193→    pub fn new(location: DiagnosticSourceLocation, message: impl Into<String>) -> Self {
   194→        Self {
   195→            location,
   196→            message: message.into(),
   197→        }
   198→    }
   199→}

   202→// Diagnostic codes
   203→// ═══════════════════════════════════════════════════════════════════════════
   204→
   205→/// Returns the diagnostic code string for a given [`ParseErrorKind`].
   206→///
   207→/// Mapping:
   208→///
   209→/// | Code  | Kind                    | Description                |
   210→/// |-------|-------------------------|----------------------------|
   211→/// | E001  | InvalidSyntax           | Syntax error               |
   212→/// | E002  | UndefinedReference / UndefinedVariable | Undefined variable |
   213→/// | E003  | TypeMismatch            | Type mismatch              |
   214→/// | E004  | DuplicateDefinition     | Duplicate definition       |
   215→/// | E005  | —                       | Invalid argument count     |
   216→/// | E006  | —                       | Invalid type               |
   217→/// | E007  | —                       | Missing return             |
   218→/// | E008  | —                       | Unreachable code           |
   219→/// | W001  | —                       | Unused variable            |
   220→/// | W002  | —                       | Implicit type conversion   |
   221→/// | W003  | —                       | Large constant (perf hint) |
   222→/// | E009  | UnexpectedToken         | Unexpected token           |
   223→/// | E010  | ExpectedToken           | Expected token             |
   224→/// | E011  | RegionError             | Region error               |
   225→/// | E012  | BDAnnotationError       | BD annotation error        |
   226→/// | E013  | InvalidCompoundOp       | Invalid compound operator  |
   227→/// | E014  | MissingSemicolon        | Missing semicolon          |
   228→/// | E015  | InvalidAddress          | Invalid address literal    |
   229→pub fn code_for_parse_error_kind(kind: &ParseErrorKind) -> &'static str {
   230→    match kind {
   231→        ParseErrorKind::InvalidSyntax => "E001",
   232→        ParseErrorKind::UndefinedReference | ParseErrorKind::UndefinedVariable => "E002",
   233→        ParseErrorKind::TypeMismatch => "E003",
   234→        ParseErrorKind::DuplicateDefinition => "E004",
   235→        ParseErrorKind::UnexpectedToken => "E009",
   236→        ParseErrorKind::ExpectedToken => "E010",
   237→        ParseErrorKind::RegionError => "E011",
   238→        ParseErrorKind::BDAnnotationError => "E012",
   239→        ParseErrorKind::InvalidCompoundOp => "E013",
   240→        ParseErrorKind::MissingSemicolon => "E014",
   241→        ParseErrorKind::InvalidAddress => "E015",
   242→    }
   243→}
   244→
   245→/// Returns the diagnostic code string for a given [`CodegenError`].
   246→pub fn code_for_codegen_error(err: &CodegenError) -> &'static str {
   247→    match err {
   248→        CodegenError::InvalidInstruction(_) => "E016",
   249→        CodegenError::RegisterAllocFailed(_) => "E017",
   250→        CodegenError::EncodingError(_) => "E018",
   251→        CodegenError::TranslationError(_) => "E019",
   252→        CodegenError::ElfError(_) => "E020",
   253→        CodegenError::UnknownVariable { .. } => "E002",
   254→        CodegenError::WasmSectionNotFound { .. } => "E016",
   255→    }
   256→}
   257→
   258→/// Human-readable description of a diagnostic code.
   259→pub fn code_description(code: &str) -> &'static str {
   260→    match code {
   261→        "E001" => "Syntax error",
   262→        "E002" => "Undefined variable",
   263→        "E003" => "Type mismatch",
   264→        "E004" => "Duplicate definition",
   265→        "E005" => "Invalid argument count",
   266→        "E006" => "Invalid type",
   267→        "E007" => "Missing return",
   268→        "E008" => "Unreachable code",
   269→        "E009" => "Unexpected token",
   270→        "E010" => "Expected token",
   271→        "E011" => "Region error",
   272→        "E012" => "BD annotation error",
   273→        "E013" => "Invalid compound assignment operator",
   274→        "E014" => "Missing semicolon",
   275→        "E015" => "Invalid address literal",
   276→        "E016" => "Invalid instruction",
   277→        "E017" => "Register allocation failed",
   278→        "E018" => "Encoding error",
   279→        "E019" => "IR translation error",
   280→        "E020" => "ELF emission error",
   281→        "W001" => "Unused variable",
   282→        "W002" => "Implicit type conversion",
   283→        "W003" => "Large constant (performance hint)",
   284→        _ => "Unknown diagnostic code",
   285→    }
   286→}

   289→// VumaDiagnostic
   290→// ═══════════════════════════════════════════════════════════════════════════
   291→
   292→/// A machine-readable diagnostic with structured fields for LLM consumption.
   293→///
   294→/// This is the primary output type for JSON diagnostics. Every field is
   295→/// serializable and carries enough information for an LLM or IDE to
   296→/// understand, categorize, and potentially auto-fix the issue.
   297→#[derive(Debug, Clone, Serialize, Deserialize)]
   298→pub struct VumaDiagnostic {
   299→    /// Structured diagnostic code, e.g. `"E001"`, `"W001"`.
   300→    pub code: String,
   301→    /// Severity level.
   302→    pub severity: DiagnosticSeverity,
   303→    /// Human-readable message (suitable for display).
   304→    pub message: String,
   305→    /// Compiler stage that produced this diagnostic
   306→    /// (`"parser"`, `"scg"`, `"ir"`, `"codegen"`, etc.).
   307→    pub source: String,
   308→    /// Precise source location.
   309→    pub location: DiagnosticSourceLocation,
   310→    /// Related source locations with explanatory messages.
   311→    pub related: Vec<RelatedInfo>,
   312→    /// Quick-fix suggestions (e.g. `"did you mean 'fn'?"`,
   313→    /// `"add semicolon"`).
   314→    pub suggestions: Vec<String>,
   315→}
   316→
   317→impl VumaDiagnostic {
   318→    /// Create a new diagnostic with all fields.
   319→    pub fn new(
   320→        code: impl Into<String>,
   321→        severity: DiagnosticSeverity,
   322→        message: impl Into<String>,
   323→        source: impl Into<String>,
   324→        location: DiagnosticSourceLocation,
   325→    ) -> Self {
   326→        Self {
   327→            code: code.into(),
   328→            severity,
   329→            message: message.into(),
   330→            source: source.into(),
   331→            location,
   332→            related: Vec::new(),
   333→            suggestions: Vec::new(),
   334→        }
   335→    }
   336→
   337→    /// Add a related-info entry.
   338→    pub fn with_related(mut self, info: RelatedInfo) -> Self {
   339→        self.related.push(info);
   340→        self
   341→    }
   342→
   343→    /// Add a quick-fix suggestion.
   344→    pub fn with_suggestion(mut self, suggestion: impl Into<String>) -> Self {
   345→        self.suggestions.push(suggestion.into());
   346→        self
   347→    }
   348→
   349→    /// Serialize this diagnostic as a JSON string.
   350→    pub fn to_json(&self) -> String {
   351→        serde_json::to_string(self).unwrap_or_else(|_| {
   352→            r#"{"code":"E000","severity":"error","message":"JSON serialization failed","source":"diagnostics","location":{"file":"","start_line":0,"start_col":0,"end_line":0,"end_col":0},"related":[],"suggestions":[]}"#.to_string()
   353→        })
   354→    }
   355→
   356→    /// Serialize this diagnostic as a pretty-printed JSON string.
   357→    pub fn to_json_pretty(&self) -> String {
   358→        serde_json::to_string_pretty(self).unwrap_or_else(|_| self.to_json())
   359→    }
   360→}
   361→
   362→impl fmt::Display for VumaDiagnostic {
   363→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   364→        write!(
   365→            f,
   366→            "{}[{}]: {} ({} at {})",
   367→            self.severity, self.code, self.message, self.source, self.location
   368→        )?;
   369→        for related in &self.related {
   370→            write!(f, "\n  note: {} at {}", related.message, related.location)?;
   371→        }
   372→        for suggestion in &self.suggestions {
   373→            write!(f, "\n  help: {}", suggestion)?;
   374→        }
   375→        Ok(())
   376→    }
   377→}

   383→/// Serialize a slice of diagnostics as a JSON array.

   388→/// Serialize a slice of diagnostics as a pretty-printed JSON array.

   393→// ═══════════════════════════════════════════════════════════════════════════
   394→// Integration: ParseError → VumaDiagnostic
   395→// ═══════════════════════════════════════════════════════════════════════════
   396→
   397→/// Convert a [`ParseError`] to a [`VumaDiagnostic`].
   398→///
   399→/// `source` is the full source text (used to compute line/column from byte
   400→/// offsets). `file` is the optional file path.
   401→pub fn from_parse_error(
   402→    err: &ParseError,
   403→    source: &str,
   404→    file: Option<&str>,
   405→) -> VumaDiagnostic {
   406→    let code = code_for_parse_error_kind(&err.kind).to_string();
   407→    let loc = offset_to_location(source, err.span.start, file);
   408→
   409→    // Compute end position from span end.
   410→    let end_loc = if err.span.end > err.span.start && err.span.end <= source.len() {
   411→        offset_to_location(source, err.span.end, file)
   412→    } else {
   413→        loc.clone()
   414→    };
   415→
   416→    let diag_loc = DiagnosticSourceLocation {
   417→        file: loc.file.clone().unwrap_or_default(),
   418→        start_line: loc.line as u32 + 1, // convert 0-based to 1-based
   419→        start_col: loc.column as u32 + 1,
   420→        end_line: end_loc.line as u32 + 1,
   421→        end_col: end_loc.column as u32 + 1,
   422→    };
   423→
   424→    let mut diag = VumaDiagnostic::new(
   425→        code,
   426→        DiagnosticSeverity::Error,
   427→        &err.message,
   428→        "parser",
   429→        diag_loc,
   430→    );
   431→
   432→    if let Some(ref suggestion) = err.suggestion {
   433→        diag = diag.with_suggestion(format!("did you mean '{}'?", suggestion));
   434→    }
   435→
   436→    // Store the line text as a hint if available.
   437→    if let Some(ref line_text) = loc.line_text {
   438→        diag = diag.with_suggestion(format!("source: {}", line_text.trim()));
   439→    }
   440→
   441→    diag
   442→}
   443→
   444→/// Convert multiple [`ParseError`]s into [`VumaDiagnostic`]s.
   445→pub fn from_parse_errors(
   446→    errors: &[ParseError],
   447→    source: &str,
   448→    file: Option<&str>,
   449→) -> Vec<VumaDiagnostic> {
   450→    errors.iter().map(|e| from_parse_error(e, source, file)).collect()
   451→}

   453→// ═══════════════════════════════════════════════════════════════════════════
   454→// Integration: VumaError → Vec<VumaDiagnostic>
   455→// ═══════════════════════════════════════════════════════════════════════════
   456→
   457→/// Convert a [`VumaError`] to one or more [`VumaDiagnostic`]s.
   458→///
   459→/// Since `VumaError` does not carry the original source text, the resulting
   460→/// diagnostics will have [`DiagnosticSourceLocation::unknown()`] for variants
   461→/// that don't carry location information. For parse errors, prefer
   462→/// [`from_parse_error`] which can compute precise locations.
   463→pub fn from_vuma_error(err: &VumaError) -> Vec<VumaDiagnostic> {
   464→    match err {
   465→        VumaError::Parse { errors } => {
   466→            // Parse errors need source text for location computation;
   467→            // without it, we produce diagnostics with unknown locations.
   468→            errors
   469→                .iter()
   470→                .map(|pe| {
   471→                    let code = code_for_parse_error_kind(&pe.kind).to_string();
   472→                    let mut diag = VumaDiagnostic::new(
   473→                        code,
   474→                        DiagnosticSeverity::Error,
   475→                        &pe.message,
   476→                        "parser",
   477→                        DiagnosticSourceLocation::unknown(),
   478→                    );
   479→                    if let Some(ref suggestion) = pe.suggestion {
   480→                        diag = diag.with_suggestion(format!("did you mean '{}'?", suggestion));
   481→                    }
   482→                    diag
   483→                })
   484→                .collect()
   485→        }
   486→        VumaError::AstToScg { message } => {
   487→            vec![VumaDiagnostic::new(
   488→                "E001",
   489→                DiagnosticSeverity::Error,
   490→                message,
   491→                "scg",
   492→                DiagnosticSourceLocation::unknown(),
   493→            )]
   494→        }
   495→        VumaError::ScgValidation { errors } => errors
   496→            .iter()
   497→            .map(|msg| {
   498→                VumaDiagnostic::new(
   499→                    "E001",
   500→                    DiagnosticSeverity::Error,
   501→                    msg,
   502→                    "scg",
   503→                    DiagnosticSourceLocation::unknown(),
   504→                )
   505→            })
   506→            .collect(),
   507→        VumaError::ScgToMsg { error } => {
   508→            vec![VumaDiagnostic::new(
   509→                "E001",
   510→                DiagnosticSeverity::Error,
   511→                error.to_string(),
   512→                "msg",
   513→                DiagnosticSourceLocation::unknown(),
   514→            )]
   515→        }
   516→        VumaError::BdInference { node_id, message } => {
   517→            let mut diag = VumaDiagnostic::new(
   518→                "E001",
   519→                DiagnosticSeverity::Error,
   520→                message,
   521→                "bd",
   522→                DiagnosticSourceLocation::unknown(),
   523→            );
   524→            if let Some(id) = node_id {
   525→                diag = diag.with_suggestion(format!("at SCG node {}", id));
   526→            }
   527→            vec![diag]
   528→        }
   529→        VumaError::Verification { result } => {
   530→            vec![VumaDiagnostic::new(
   531→                "E001",
   532→                DiagnosticSeverity::Error,
   533→                format!("verification failed: {}", result.overall),
   534→                "ive",
   535→                DiagnosticSourceLocation::unknown(),
   536→            )]
   537→        }
   538→        VumaError::Transform { pass_name, errors } => errors
   539→            .iter()
   540→            .map(|msg| {
   541→                VumaDiagnostic::new(
   542→                    "E001",
   543→                    DiagnosticSeverity::Error,
   544→                    msg,
   545→                    format!("scg-transform:{}", pass_name),
   546→                    DiagnosticSourceLocation::unknown(),
   547→                )
   548→            })
   549→            .collect(),
   550→        VumaError::Codegen { error } => {
   551→            vec![from_codegen_error(error)]
   552→        }
   553→        VumaError::RegisterAlloc { message } => {
   554→            vec![VumaDiagnostic::new(
   555→                "E017",
   556→                DiagnosticSeverity::Error,
   557→                message,
   558→                "register-alloc",
   559→                DiagnosticSourceLocation::unknown(),
   560→            )]
   561→        }
   562→        VumaError::Emission { message } => {
   563→            vec![VumaDiagnostic::new(
   564→                "E020",
   565→                DiagnosticSeverity::Error,
   566→                message,
   567→                "elf-emission",
   568→                DiagnosticSourceLocation::unknown(),
   569→            )]
   570→        }
   571→        VumaError::CorInit { message } => {
   572→            vec![VumaDiagnostic::new(
   573→                "E001",
   574→                DiagnosticSeverity::Error,
   575→                message,
   576→                "cor-init",
   577→                DiagnosticSourceLocation::unknown(),
   578→            )]
   579→        }
   580→        VumaError::Multi { errors } => errors.iter().flat_map(from_vuma_error).collect(),
   581→    }
   582→}

   588→/// Convert a [`CodegenError`] to a [`VumaDiagnostic`].
   589→pub fn from_codegen_error(err: &CodegenError) -> VumaDiagnostic {
   590→    let code = code_for_codegen_error(err).to_string();
   591→    let message = err.to_string();
   592→
   593→    let (source, suggestion) = match err {
   594→        CodegenError::UnknownVariable { name } => (
   595→            "codegen",
   596→            Some(format!("declare '{}' before use", name)),
   597→        ),
   598→        CodegenError::RegisterAllocFailed(_) => (
   599→            "register-alloc",
   600→            Some("reduce register pressure by splitting live ranges".to_string()),
   601→        ),
   602→        CodegenError::TranslationError(_) => (
   603→            "ir",
   604→            None,
   605→        ),
   606→        CodegenError::EncodingError(_) => (
   607→            "codegen",
   608→            None,
   609→        ),
   610→        CodegenError::ElfError(_) => (
   611→            "emit",
   612→            None,
   613→        ),
   614→        CodegenError::InvalidInstruction(_) => (
   615→            "codegen",
   616→            None,
   617→        ),
   618→        CodegenError::WasmSectionNotFound { section } => (
   619→            "codegen",
   620→            Some(format!("ensure '{}' section is generated", section)),
   621→        ),
   622→    };
   623→
   624→    let mut diag = VumaDiagnostic::new(
   625→        code,
   626→        DiagnosticSeverity::Error,
   627→        message,
   628→        source,
   629→        DiagnosticSourceLocation::unknown(),
   630→    );
   631→
   632→    if let Some(s) = suggestion {
   633→        diag = diag.with_suggestion(s);
   634→    }
   635→
   636→    diag
   637→}
