     1→//! Recursive-descent parser for the VUMA language frontend.
     2→//!
     3→//! Consumes the token stream produced by [`crate::lexer::Lexer`] and builds
     4→//! an [`crate::ast::Program`]. The parser uses precedence climbing for
     5→//! expression sub-parsing and implements error recovery by skipping
     6→//! to the next statement boundary (`;` or `}`) or item boundary on failure.
     7→//!
     8→//! # Supported constructs
     9→//!
    10→//! **Items** (top-level declarations):
    11→//! - `fn`, `struct`, `enum`, `region`, `import`, `export`, `const`, `static`, `mod`
    12→//!
    13→//! **Statements**:
    14→//! - `let`, assignment, compound assignment (`+=`, `-=`, etc.), `if`, `while`, `for`,
    15→//!   `loop`, `match`, `return`, `break`, `continue`, `sync`, `free`, `allocate`,
    16→//!   `bd`/`repd`/`capd`/`reld` directives, expression statements
    17→//!
    18→//! **Expressions**:
    19→//! - Binary/unary ops, calls, field/index access, deref, addr-of, cast,
    20→//!   `sizeof`, `alignof`, `derive`, `async`, `spawn`, `allocate` (as expr),
    21→//!   literals (int, float, bool, string, address, null), struct init, namespace access
    22→
    23→use crate::ast::*;
    24→use crate::error::{
    25→    check_llm_construct, suggest_keyword, suggest_vuma_type, ErrorRecovery, ParseError,
    26→    ParseErrorKind, ParseResult, Span,
    27→};
    28→use crate::lexer::{Lexer, Token, TokenKind};
    29→

    31→// Parser

    33→
    34→/// Default maximum recursion depth for expression parsing.
    35→const MAX_EXPR_DEPTH: u32 = 256;
    36→
    37→/// A recursive-descent parser that turns a token stream into an AST.
    38→pub struct Parser<'src> {
    39→    /// Underlying lexer.
    40→    lexer: Lexer<'src>,
    41→    /// The most recently consumed token.
    42→    current: Token,
    43→    /// Accumulated parse errors (for error recovery).
    44→    errors: Vec<ParseError>,
    45→    /// Pushback buffer for token rewinding (used in struct literal disambiguation).
    46→    pushback: std::collections::VecDeque<Token>,
    47→    /// Current recursion depth in expression parsing.
    48→    expr_depth: u32,
    49→    /// Maximum allowed recursion depth for expression parsing.
    50→    max_depth: u32,
    51→    /// When true, struct literal parsing is suppressed in `parse_postfix()`.
    52→    /// Used when parsing the end expression of a range (`..`) so that `{`
    53→    /// is not consumed as part of a struct literal and remains available
    54→    /// for the enclosing construct (e.g. a for-loop body).
    55→    no_struct_literal: bool,
    56→    /// Name of the function currently being parsed (for context-aware errors).
    57→    current_fn_name: Option<String>,
    58→}
    59→
    60→/// Token kinds that begin a top-level item declaration.
    61→const ITEM_STARTERS: &[TokenKind] = &[
    62→    TokenKind::Fn,
    63→    TokenKind::Struct,
    64→    TokenKind::Enum,
    65→    TokenKind::Region,
    66→    TokenKind::Import,
    67→    TokenKind::Export,
    68→    TokenKind::Const,
    69→    TokenKind::Static,
    70→    TokenKind::Pub,
    71→    TokenKind::Hash,
    72→    // `mod` is TokenKind::Mod — handled directly
    73→];
    74→
    75→/// Compound-assignment token kinds mapped to their operators.
    76→///
    77→/// Returns `Err(ParseError)` for invalid tokens instead of panicking,
    78→/// so that callers can propagate the error cleanly with `?`.
    79→fn compound_op_from_token(kind: TokenKind, span: Span) -> Result<CompoundOp, ParseError> {
    80→    match kind {
    81→        TokenKind::PlusEq => Ok(CompoundOp::Add),
    82→        TokenKind::MinusEq => Ok(CompoundOp::Sub),
    83→        TokenKind::StarEq => Ok(CompoundOp::Mul),
    84→        TokenKind::SlashEq => Ok(CompoundOp::Div),
    85→        TokenKind::PercentEq => Ok(CompoundOp::Mod),
    86→        TokenKind::AmpEq => Ok(CompoundOp::BitAnd),
    87→        TokenKind::PipeEq => Ok(CompoundOp::BitOr),
    88→        TokenKind::CaretEq => Ok(CompoundOp::BitXor),
    89→        TokenKind::ShlEq => Ok(CompoundOp::Shl),
    90→        TokenKind::ShrEq => Ok(CompoundOp::Shr),
    91→        _ => Err(ParseError::invalid_compound_op(
    92→            format!("expected compound assignment operator, found {:?}", kind),
    93→            span,
    94→        )),
    95→    }
    96→}
    97→
    98→/// Check if a token kind is a compound assignment operator.
    99→fn is_compound_assign(kind: TokenKind) -> bool {
   100→    matches!(
   101→        kind,
   102→        TokenKind::PlusEq
   103→            | TokenKind::MinusEq
   104→            | TokenKind::StarEq
   105→            | TokenKind::SlashEq
   106→            | TokenKind::PercentEq
   107→            | TokenKind::AmpEq
   108→            | TokenKind::PipeEq
   109→            | TokenKind::CaretEq
   110→            | TokenKind::ShlEq
   111→            | TokenKind::ShrEq
   112→    )
   113→}
   114→
   115→impl<'src> Parser<'src> {
   116→    /// Create a new parser over the given source text.
   117→    pub fn new(source: &'src str) -> Self {
   118→        let mut lexer = Lexer::new(source);
   119→        let current = lexer.next_token();
   120→        Self {
   121→            lexer,
   122→            current,
   123→            errors: Vec::new(),
   124→            pushback: std::collections::VecDeque::new(),
   125→            expr_depth: 0,
   126→            max_depth: MAX_EXPR_DEPTH,
   127→            no_struct_literal: false,
   128→            current_fn_name: None,
   129→        }
   130→    }
   131→
   132→    /// Create a new parser with a custom maximum expression recursion depth.
   133→    pub fn with_max_depth(source: &'src str, max_depth: u32) -> Self {
   134→        let mut parser = Self::new(source);
   135→        parser.max_depth = max_depth;
   136→        parser
   137→    }
   138→
   139→    // -- public entry point --------------------------------------------------
   140→
   141→    /// Parse the full source into a [`Program`].
   142→    ///
   143→    /// If errors are encountered the parser attempts recovery and continues;
   144→    /// the returned [`ParseResult`] carries both the partial AST and any
   145→    /// accumulated non-fatal errors, enabling multi-error reporting in a
   146→    /// single pass. All errors are collected — the parser never stops at
   147→    /// the first error.
   148→    pub fn parse_program(&mut self) -> ParseResult<Program> {
   149→        let start = self.current.span.start;
   150→        let mut items = Vec::new();
   151→
   152→        // Also collect any lexer-level errors
   153→        let lexer_errors = self.lexer.take_errors();
   154→        self.errors.extend(lexer_errors);
   155→
   156→        while !self.at(TokenKind::Eof) {
   157→            match self.parse_item() {
   158→                Ok(item) => items.push(item),
   159→                Err(err) => {
   160→                    self.errors.push(err);
   161→                    self.recover_to_item_boundary();
   162→                }
   163→            }
   164→        }
   165→
   166→        // Collect any remaining lexer errors that appeared during parsing
   167→        let lexer_errors = self.lexer.take_errors();
   168→        self.errors.extend(lexer_errors);
   169→
   170→        let end = self.current.span.end;
   171→        let program = Program {
   172→            items,
   173→            span: Span::new(start, end),
   174→        };
   175→
   176→        // Resolve line/column for all accumulated errors
   177→        let source = self.lexer.source();
   178→        for err in &mut self.errors {
   179→            err.resolve_location(source, None);
   180→        }
   181→
   182→        if self.errors.is_empty() {
   183→            ParseResult::ok(program)
   184→        } else {
   185→            ParseResult::ok_with_errors(program, std::mem::take(&mut self.errors))
   186→        }
   187→    }
   188→
   189→    /// Return all accumulated parse errors.
   190→    pub fn errors(&self) -> &[ParseError] {
   191→        &self.errors
   192→    }

   194→    // -- lookahead -----------------------------------------------------------
   195→
   196→    /// Peek at the token AFTER `self.current` without consuming anything.
   197→    /// Returns a clone of the peeked token.
   198→    fn peek_next(&mut self) -> Token {
   199→        self.lexer.peek().clone()
   200→    }
   201→
   202→    // -- item parsing --------------------------------------------------------
   203→
   204→    /// Parse a single top-level item or statement.
   205→    fn parse_item(&mut self) -> Result<Item, ParseError> {
   206→        // Parse outer attributes (#[...]) that precede the item.
   207→        let attrs = self.parse_outer_attributes()?;
   208→
   209→        // Parse visibility modifier.
   210→        let visibility = self.parse_visibility()?;
   211→
   212→        match self.current.kind {
   213→            TokenKind::Fn => self.parse_fn_def(false, visibility, attrs).map(Item::FnDef),
   214→            TokenKind::Async => {
   215→                // Could be `async fn` or `async { block }` as expression
   216→                let next = self.peek_next();
   217→                if next.kind == TokenKind::Fn {
   218→                    self.parse_fn_def(true, visibility, attrs).map(Item::FnDef)
   219→                } else {
   220→                    self.parse_stmt().map(Item::Stmt)
   221→                }
   222→            }
   223→            TokenKind::Struct => self
   224→                .parse_struct_def(visibility, attrs)
   225→                .map(Item::StructDef),
   226→            TokenKind::Enum => self.parse_enum_def(visibility, attrs).map(Item::EnumDef),
   227→            TokenKind::Region => {
   228→                // Distinguish: `region name = allocate(...)` vs `region` used as
   229→                // a variable name in an expression/assignment (e.g. `region = allocate(8);`)
   230→                let next = self.peek_next();
   231→                if next.kind == TokenKind::Ident {
   232→                    self.parse_region_def().map(Item::RegionDef)
   233→                } else {
   234→                    self.parse_stmt().map(Item::Stmt)
   235→                }
   236→            }
   237→            TokenKind::Import => self.parse_import().map(Item::Import),
   238→            TokenKind::Export => self.parse_export().map(Item::Export),
   239→            TokenKind::Const => self.parse_const_item(visibility, attrs).map(Item::Const),
   240→            TokenKind::Static => self.parse_static_item(visibility, attrs).map(Item::Static),
   241→            TokenKind::Mod => self.parse_module_def().map(Item::ModuleDef),
   242→            TokenKind::Trait => self.parse_trait_def(visibility, attrs).map(Item::TraitDef),
   243→            TokenKind::Impl => self.parse_impl_block(attrs).map(Item::ImplBlock),
   244→            TokenKind::Ident => {
   245→                let lexeme = self.current.lexeme.as_str();
   246→                match lexeme {
   247→                    "static" => self.parse_static_item(visibility, attrs).map(Item::Static),
   248→                    _ => self.parse_stmt().map(Item::Stmt),
   249→                }
   250→            }
   251→            _ => self.parse_stmt().map(Item::Stmt),
   252→        }
   253→    }
   254→
   255→    /// [`async`] `fn` <ident> [`<` type_params `>`] `(` <params>? `)` [`->` <type>] [`;` | `{` <block> `}`]
   256→    fn parse_fn_def(
   257→        &mut self,
   258→        is_async: bool,
   259→        _visibility: Visibility,
   260→        _attrs: Vec<Attribute>,
   261→    ) -> Result<FnDef, ParseError> {
   262→        let start = self.current.span.start;
   263→
   264→        // Consume 'async' if present
   265→        if self.at(TokenKind::Async) {
   266→            self.advance();
   267→        }
   268→
   269→        self.expect(TokenKind::Fn)?;
   270→
   271→        let name = self.expect_name()?;
   272→
   273→        // Track current function name for context-aware error messages
   274→        self.current_fn_name = Some(name.clone());
   275→
   276→        // Optional generic type parameters: `<T, U>`
   277→        let type_params = if self.at(TokenKind::Lt) {
   278→            self.parse_generic_params()?
   279→        } else {
   280→            Vec::new()
   281→        };
   282→
   283→        self.expect(TokenKind::LParen)?;
   284→
   285→        let params = self.parse_params()?;
   286→
   287→        self.expect(TokenKind::RParen)?;
   288→
   289→        let return_type = if self.at(TokenKind::Arrow) {
   290→            self.advance();
   291→            Some(self.parse_type()?)
   292→        } else {
   293→            None
   294→        };
   295→
   296→        // Optional where clause
   297→        let where_clause = if self.at(TokenKind::Where) {
   298→            Some(self.parse_where_clause()?)
   299→        } else {
   300→            None
   301→        };
   302→
   303→        // For trait methods, we may have `;` instead of a body.
   304→        // We create a synthetic empty block for required method signatures.
   305→        let body = if self.at(TokenKind::LBrace) {
   306→            self.parse_block()?
   307→        } else if self.at(TokenKind::Semicolon) {
   308→            self.advance(); // consume ';'
   309→            Block {
   310→                statements: Vec::new(),
   311→                span: Span::new(self.current.span.start, self.current.span.end),
   312→            }
   313→        } else {
   314→            Block {
   315→                statements: Vec::new(),
   316→                span: Span::synthetic(),
   317→            }
   318→        };
   319→        let end = body.span.end;
   320→
   321→        // Clear the current function name after we're done parsing it
   322→        self.current_fn_name = None;
   323→
   324→        Ok(FnDef {
   325→            visibility: _visibility,
   326→            attrs: _attrs,
   327→            name,
   328→            type_params,
   329→            params,
   330→            return_type,
   331→            body,
   332→            is_async,
   333→            where_clause,
   334→            span: Span::new(start, end),
   335→        })
   336→    }

   338→    /// `struct` <ident> [`<` type_params `>`] `{` <fields> `}`
   339→    fn parse_struct_def(
   340→        &mut self,
   341→        _visibility: Visibility,
   342→        _attrs: Vec<Attribute>,
   343→    ) -> Result<StructDef, ParseError> {
   344→        let start = self.current.span.start;
   345→        self.expect(TokenKind::Struct)?;
   346→
   347→        let name = self.expect_name()?;
   348→
   349→        // Optional generic type parameters with bounds
   350→        let type_params = if self.at(TokenKind::Lt) {
   351→            self.parse_type_params_with_bounds()?
   352→        } else {
   353→            Vec::new()
   354→        };
   355→
   356→        // Where clause comes before '{'
   357→        let where_clause = if self.at(TokenKind::Where) {
   358→            Some(self.parse_where_clause()?)
   359→        } else {
   360→            None
   361→        };
   362→
   363→        self.expect(TokenKind::LBrace)?;
   364→
   365→        let mut fields = Vec::new();
   366→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   367→            let f_start = self.current.span.start;
   368→            let fname = self.expect_name()?;
   369→            self.expect(TokenKind::Colon)?;
   370→            let ftype = self.parse_type()?;
   371→            let f_end = self.current.span.end;
   372→            fields.push(StructField {
   373→                name: fname,
   374→                ty: ftype,
   375→                span: Span::new(f_start, f_end),
   376→            });
   377→            if self.at(TokenKind::Comma) {
   378→                self.advance();
   379→            } else {
   380→                break;
   381→            }
   382→        }
   383→
   384→        self.expect(TokenKind::RBrace)?;
   385→
   386→        Ok(StructDef {
   387→            visibility: _visibility,
   388→            attrs: _attrs,
   389→            name,
   390→            type_params,
   391→            fields,
   392→            where_clause,
   393→            span: Span::new(start, self.current.span.end),
   394→        })
   395→    }

   397→    /// `enum` <ident> [`<` type_params `>`] `{` <variants> `}`
   398→    fn parse_enum_def(
   399→        &mut self,
   400→        _visibility: Visibility,
   401→        _attrs: Vec<Attribute>,
   402→    ) -> Result<EnumDef, ParseError> {
   403→        let start = self.current.span.start;
   404→        self.expect(TokenKind::Enum)?;
   405→
   406→        let name = self.expect_name()?;
   407→
   408→        let type_params = if self.at(TokenKind::Lt) {
   409→            self.parse_type_params_with_bounds()?
   410→        } else {
   411→            Vec::new()
   412→        };
   413→
   414→        // Where clause comes before '{'
   415→        let where_clause = if self.at(TokenKind::Where) {
   416→            Some(self.parse_where_clause()?)
   417→        } else {
   418→            None
   419→        };
   420→
   421→        self.expect(TokenKind::LBrace)?;
   422→
   423→        let mut variants = Vec::new();
   424→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   425→            let v_start = self.current.span.start;
   426→            let vname = self.expect_name()?;
   427→
   428→            let payload = if self.at(TokenKind::LParen) {
   429→                self.advance();
   430→                let ty = self.parse_type()?;
   431→                self.expect(TokenKind::RParen)?;
   432→                Some(ty)
   433→            } else {
   434→                None
   435→            };
   436→
   437→            let v_end = self.current.span.end;
   438→            variants.push(EnumVariant {
   439→                name: vname,
   440→                payload,
   441→                span: Span::new(v_start, v_end),
   442→            });
   443→            if self.at(TokenKind::Comma) {
   444→                self.advance();
   445→            } else {
   446→                break;
   447→            }
   448→        }
   449→
   450→        self.expect(TokenKind::RBrace)?;
   451→
   452→        Ok(EnumDef {
   453→            visibility: _visibility,
   454→            attrs: _attrs,
   455→            name,
   456→            type_params,
   457→            variants,
   458→            where_clause,
   459→            span: Span::new(start, self.current.span.end),
   460→        })
   461→    }

   463→    /// Parse comma-separated type parameter names inside `< … >`.
   464→    #[allow(dead_code)]
   465→    fn parse_type_params(&mut self) -> Result<Vec<String>, ParseError> {
   466→        self.expect(TokenKind::Lt)?;
   467→        let mut params = Vec::new();
   468→        while !self.at_closing_gt() && !self.at(TokenKind::Eof) {
   469→            params.push(self.expect_name()?);
   470→            if self.at(TokenKind::Comma) {
   471→                self.advance();
   472→            } else {
   473→                break;
   474→            }
   475→        }
   476→        self.expect_gt_closing_generic()?;
   477→        Ok(params)
   478→    }
   479→
   480→    /// Parse type parameters with optional bounds: `<T: Display + Clone, U>`
   481→    fn parse_type_params_with_bounds(&mut self) -> Result<Vec<TypeParam>, ParseError> {
   482→        self.expect(TokenKind::Lt)?;
   483→        let mut params = Vec::new();
   484→        while !self.at_closing_gt() && !self.at(TokenKind::Eof) {
   485→            let name = self.expect_name()?;
   486→            let bounds = if self.at(TokenKind::Colon) {
   487→                self.advance();
   488→                self.parse_trait_bounds()?
   489→            } else {
   490→                Vec::new()
   491→            };
   492→            params.push(TypeParam { name, bounds });
   493→            if self.at(TokenKind::Comma) {
   494→                self.advance();
   495→            } else {
   496→                break;
   497→            }
   498→        }
   499→        self.expect_gt_closing_generic()?;
   500→        Ok(params)
   501→    }
   502→
   503→    /// Parse trait bounds: `Trait + AnotherTrait + ...`
   504→    fn parse_trait_bounds(&mut self) -> Result<Vec<Type>, ParseError> {
   505→        let mut bounds = Vec::new();
   506→        bounds.push(self.parse_type()?);
   507→        while self.at(TokenKind::Plus) {
   508→            self.advance();
   509→            bounds.push(self.parse_type()?);
   510→        }
   511→        Ok(bounds)
   512→    }
   513→
   514→    /// Parse a where clause: `where T: Trait + AnotherTrait, U: Trait`
   515→    fn parse_where_clause(&mut self) -> Result<WhereClause, ParseError> {
   516→        self.expect(TokenKind::Where)?;
   517→        let mut predicates = Vec::new();
   518→        loop {
   519→            let type_name = self.expect_name()?;
   520→            self.expect(TokenKind::Colon)?;
   521→            let bounds = self.parse_trait_bounds()?;
   522→            predicates.push(WherePredicate { type_name, bounds });
   523→            if self.at(TokenKind::Comma) {
   524→                self.advance();
   525→            } else {
   526→                break;
   527→            }
   528→        }
   529→        Ok(WhereClause { predicates })
   530→    }
   531→
   532→    /// Parse generic parameter list `<T, U, ...>` with optional bounds.
   533→    /// Returns parsed type parameters for recording in the AST.
   534→    fn parse_generic_params(&mut self) -> Result<Vec<TypeParam>, ParseError> {
   535→        self.parse_type_params_with_bounds()
   536→    }
   537→
   538→    /// Parse a comma-separated parameter list (may be empty).
   539→    fn parse_params(&mut self) -> Result<Vec<Param>, ParseError> {
   540→        let mut params = Vec::new();
   541→        if self.at(TokenKind::RParen) {
   542→            return Ok(params);
   543→        }
   544→        loop {
   545→            let span = self.current.span;
   546→            let name = self.expect_name()?;
   547→            let ty = if self.at(TokenKind::Colon) {
   548→                self.advance();
   549→                Some(self.parse_type()?)
   550→            } else {
   551→                None
   552→            };
   553→            params.push(Param { name, ty, span });
   554→            if self.at(TokenKind::Comma) {
   555→                self.advance();
   556→            } else {
   557→                break;
   558→            }
   559→        }
   560→        Ok(params)
   561→    }

   563→    /// `region` <ident> `=` `allocate` `(` <expr> `)` `;`
   564→    fn parse_region_def(&mut self) -> Result<RegionDef, ParseError> {
   565→        let start = self.current.span.start;
   566→        self.expect(TokenKind::Region)?;
   567→
   568→        let name = self.expect_name()?;
   569→
   570→        self.expect(TokenKind::Assign)?;
   571→        self.expect(TokenKind::Allocate)?;
   572→        self.expect(TokenKind::LParen)?;
   573→
   574→        let size_expr = self.parse_expr()?;
   575→
   576→        self.expect(TokenKind::RParen)?;
   577→        self.expect(TokenKind::Semicolon)?;
   578→
   579→        Ok(RegionDef {
   580→            name,
   581→            size_expr,
   582→            span: Span::new(start, self.current.span.end),
   583→        })
   584→    }

   586→    /// `import` <string> [`::` `{` <names> `}`] `;`
   587→    ///
   588→    /// Supported forms:
   589→    /// - `import "crypto.vuma";` — import all functions from file
   590→    /// - `import "crypto.vuma"::{sha256, sha256d};` — import specific functions
   591→    /// - `import "crypto.vuma" {sha256, sha256d};` — legacy form (also accepted)
   592→    fn parse_import(&mut self) -> Result<Import, ParseError> {
   593→        let start = self.current.span.start;
   594→        self.expect(TokenKind::Import)?;
   595→
   596→        let path = self.expect_string()?;
   597→        let mut symbols = Vec::new();
   598→
   599→        // Accept optional `::` before `{` for the new syntax
   600→        if self.at(TokenKind::PathSep) {
   601→            self.advance();
   602→        }
   603→
   604→        if self.at(TokenKind::LBrace) {
   605→            self.advance();
   606→            while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   607→                symbols.push(self.expect_name()?);
   608→                if self.at(TokenKind::Comma) {
   609→                    self.advance();
   610→                } else {
   611→                    break;
   612→                }
   613→            }
   614→            self.expect(TokenKind::RBrace)?;
   615→        }
   616→
   617→        self.expect(TokenKind::Semicolon)?;
   618→
   619→        Ok(Import {
   620→            path,
   621→            symbols,
   622→            span: Span::new(start, self.current.span.end),
   623→        })
   624→    }

   707→    /// `mod` <name> `{` <items>* `}`
   708→    fn parse_module_def(&mut self) -> Result<ModuleDef, ParseError> {
   709→        let start = self.current.span.start;
   710→        self.expect(TokenKind::Mod)?;
   711→
   712→        let name = self.expect_name()?;
   713→
   714→        self.expect(TokenKind::LBrace)?;
   715→
   716→        let mut items = Vec::new();
   717→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   718→            match self.parse_item() {

   720→                Err(err) => {
   721→                    self.errors.push(err);
   722→                    self.recover_to_item_boundary();
   723→                }
   724→            }
   725→        }
   726→
   727→        self.expect(TokenKind::RBrace)?;
   728→

   736→    /// `trait` <name> [`<` type_params `>`] `{` <members> `}`
   737→    fn parse_trait_def(
   738→        &mut self,
   739→        _visibility: Visibility,
   740→        _attrs: Vec<Attribute>,
   741→    ) -> Result<TraitDef, ParseError> {
   742→        let start = self.current.span.start;
   743→        self.expect(TokenKind::Trait)?;
   744→
   745→        let name = self.expect_name()?;
   746→
   747→        let type_params = if self.at(TokenKind::Lt) {
   748→            self.parse_type_params_with_bounds()?
   749→        } else {
   750→            Vec::new()
   751→        };
   752→
   753→        self.expect(TokenKind::LBrace)?;
   754→
   755→        let mut associated_types = Vec::new();
   756→        let mut associated_consts = Vec::new();
   757→        let mut required_methods = Vec::new();
   758→        let mut provided_methods = Vec::new();
   759→
   760→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   761→            // Associated type: `type Name;`
   762→            if self.at(TokenKind::Type) {
   763→                let next = self.peek_next();
   764→                if next.kind == TokenKind::Ident {
   765→                    self.advance(); // consume 'type'
   766→                    let ty_name = self.expect_name()?;
   767→                    self.expect(TokenKind::Semicolon)?;
   768→                    associated_types.push(ty_name);
   769→                    continue;
   770→                }
   771→            }
   772→            // Associated const: `const NAME: Type [= expr];`
   773→            if self.at(TokenKind::Const) {
   774→                let ac_start = self.current.span.start;
   775→                let next = self.peek_next();
   776→                if next.kind == TokenKind::Ident {
   777→                    self.advance(); // consume 'const'
   778→                    let ac_name = self.expect_name()?;
   779→                    self.expect(TokenKind::Colon)?;
   780→                    let ac_ty = self.parse_type()?;
   781→                    let ac_value = if self.at(TokenKind::Assign) {
   782→                        self.advance();
   783→                        Some(self.parse_expr()?)
   784→                    } else {
   785→                        None
   786→                    };
   787→                    self.expect(TokenKind::Semicolon)?;
   788→                    associated_consts.push(AssocConst {
   789→                        name: ac_name,
   790→                        ty: ac_ty,
   791→                        value: ac_value,
   792→                        span: Span::new(ac_start, self.current.span.end),
   793→                    });
   794→                    continue;
   795→                }
   796→            }
   797→            // Method: `fn name(...) -> T;` or `fn name(...) -> T { body }`
   798→            if self.at(TokenKind::Fn) {
   799→                let method = self.parse_fn_def(false, Visibility::default(), Vec::new())?;
   800→                if method.body.statements.is_empty()
   801→                    && !method.is_async
   802→                    && method.where_clause.is_none()
   803→                {
   804→                    // Empty body suggests a required method signature
   805→                    // But we can't easily tell — if the body is just {} it's provided
   806→                    // We'll check: if body span is just `{}` with nothing inside, treat as required
   807→                    // Actually, let's use a simpler heuristic: if the method has a non-empty
   808→                    // body (more than 0 statements), it's provided.
   809→                    required_methods.push(method);
   810→                } else {
   811→                    provided_methods.push(method);
   812→                }
   813→                continue;
   814→            }
   815→            // Skip unexpected tokens in trait body
   816→            self.advance();
   817→        }
   818→
   819→        self.expect(TokenKind::RBrace)?;
   820→
   821→        let where_clause = if self.at(TokenKind::Where) {
   822→            Some(self.parse_where_clause()?)
   823→        } else {
   824→            None
   825→        };
   826→
   827→        Ok(TraitDef {
   828→            visibility: _visibility,
   829→            attrs: _attrs,
   830→            name,
   831→            type_params,
   832→            associated_types,
   833→            associated_consts,
   834→            required_methods,
   835→            provided_methods,
   836→            where_clause,
   837→            span: Span::new(start, self.current.span.end),
   838→        })
   839→    }

   841→    /// `impl` [<trait_name> `for`] <type> `{` <methods> `}`
   842→    fn parse_impl_block(&mut self, _attrs: Vec<Attribute>) -> Result<ImplBlock, ParseError> {
   843→        let start = self.current.span.start;
   844→        self.expect(TokenKind::Impl)?;
   845→
   846→        // Parse optional generic params on the impl itself: `impl<T> ...`
   847→        // This must come before the trait/type name.
   848→        let type_params = if self.at(TokenKind::Lt) {
   849→            self.parse_generic_params()?
   850→        } else {
   851→            Vec::new()
   852→        };
   853→
   854→        // Parse the first name — could be a trait name or the target type
   855→        let first_name = self.expect_name()?;
   856→
   857→        // Check if `for` follows — if so, first_name is the trait name
   858→        let (trait_name, target_type) = if self.at(TokenKind::For) {
   859→            self.advance(); // consume 'for'
   860→            let target = self.parse_type()?;
   861→            (Some(first_name), target)
   862→        } else {
   863→            // No 'for' — first_name is the target type
   864→            // Check for generic args on the target type
   865→            let target = if self.at(TokenKind::Lt) {
   866→                // Generic type: `Name<Args>`
   867→                self.advance(); // consume '<'
   868→                let mut args = Vec::new();
   869→                if !self.at_closing_gt() {
   870→                    args.push(self.parse_type()?);
   871→                    while self.at(TokenKind::Comma) {
   872→                        self.advance();
   873→                        args.push(self.parse_type()?);
   874→                    }
   875→                }
   876→                self.expect_gt_closing_generic()?;
   877→                Type::Generic {
   878→                    name: first_name,
   879→                    args,
   880→                }
   881→            } else {
   882→                Type::BDBase(first_name)
   883→            };
   884→            (None, target)
   885→        };
   886→
   887→        self.expect(TokenKind::LBrace)?;
   888→
   889→        let mut methods = Vec::new();
   890→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   891→            if self.at(TokenKind::Fn) {
   892→                match self.parse_fn_def(false, Visibility::default(), Vec::new()) {
   893→                    Ok(method) => methods.push(method),
   894→                    Err(err) => {
   895→                        self.errors.push(err);
   896→                        self.recover_to_statement_boundary();
   897→                    }
   898→                }
   899→            } else {
   900→                self.advance(); // skip unexpected tokens
   901→            }
   902→        }
   903→
   904→        self.expect(TokenKind::RBrace)?;
   905→
   906→        let where_clause = if self.at(TokenKind::Where) {
   907→            Some(self.parse_where_clause()?)
   908→        } else {
   909→            None
   910→        };
   911→
   912→        Ok(ImplBlock {
   913→            attrs: _attrs,
   914→            type_params,
   915→            trait_name,
   916→            target_type,
   917→            methods,
   918→            where_clause,
   919→            span: Span::new(start, self.current.span.end),
   920→        })
   921→    }
   922→
   923→    // -- block & statements --------------------------------------------------
   924→
   925→    /// `{` <stmt>* `}`
   926→    ///
   927→    /// On error within a statement, the parser pushes the error to
   928→    /// [`self.errors`] and recovers to the next statement boundary, allowing
   929→    /// the rest of the block to be parsed and all errors to be collected.
   930→    fn parse_block(&mut self) -> Result<Block, ParseError> {
   931→        let start = self.current.span.start;
   932→        self.expect(TokenKind::LBrace)?;
   933→
   934→        let mut statements = Vec::new();
   935→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   936→            match self.parse_stmt() {
   937→                Ok(stmt) => statements.push(stmt),
   938→                Err(err) => {
   939→                    self.errors.push(err);
   940→                    self.recover_to_statement_boundary();
   941→                }
   942→            }
   943→        }
   944→
   945→        // If we hit EOF without finding '}', push an error but don't fail
   946→        if self.at(TokenKind::Eof) {
   947→            self.errors.push(ParseError::expected(
   948→                "'}'",
   949→                "end of file",
   950→                self.current.span,
   951→            ));
   952→            return Ok(Block {
   953→                statements,
   954→                span: Span::new(start, self.current.span.end),
   955→            });
   956→        }
   957→
   958→        self.expect(TokenKind::RBrace)?;
   959→        let end = self.current.span.end;
   960→
   961→        Ok(Block {
   962→            statements,
   963→            span: Span::new(start, end),
   964→        })
   965→    }
   966→
   967→    /// Parse a single statement.
   968→    fn parse_stmt(&mut self) -> Result<Stmt, ParseError> {
   969→        match self.current.kind {
   970→            TokenKind::Let => self.parse_let_stmt(),
   971→            TokenKind::Mut => {
   972→                // LLM mistake: `mut` is not a VUMA keyword for declarations
   973→                let span = self.current.span;
   974→                self.errors.push(ParseError::llm_mistake(
   975→                    "`mut` is not used in VUMA — variables are mutable by default; just use assignment without `mut`",
   976→                    span,
   977→                    "remove `mut`",
   978→                ));
   979→                self.advance(); // consume `mut`
   980→                // Try to continue parsing what comes after `mut`
   981→                self.parse_stmt()
   982→            }
   983→            TokenKind::MacroIdent => {
   984→                // LLM mistake: Rust-style macro invocation (println!, etc.)
   985→                let span = self.current.span;
   986→                let lexeme = self.current.lexeme.clone();
   987→                let name_without_bang = lexeme.strip_suffix('!').unwrap_or(&lexeme);
   988→                let hint = check_llm_construct(name_without_bang)
   989→                    .unwrap_or("VUMA does not support macro syntax with `!`");
   990→                self.errors.push(ParseError::llm_mistake(
   991→                    format!("`{}` is not valid VUMA syntax — {}", lexeme, hint),
   992→                    span,
   993→                    "use `write` or format strings instead",
   994→                ));
   995→                self.advance(); // consume the macro identifier
   996→                // Try to skip the macro arguments and continue
   997→                if self.at(TokenKind::LParen) {
   998→                    self.skip_balanced_parens();
   999→                }
  1000→                if self.at(TokenKind::Semicolon) {
  1001→                    self.advance();
  1002→                }
  1003→                // Return a synthetic expression statement to allow recovery
  1004→                Ok(Stmt::Expr(ExprStmt {
  1005→                    expr: Expr::Uninitialized {
  1006→                        span: Span::new(span.start, span.end),
  1007→                    },
  1008→                    span,
  1009→                }))
  1010→            }
  1011→            TokenKind::If => self.parse_if_stmt(),
  1012→            TokenKind::While => self.parse_while_stmt(),
  1013→            TokenKind::For => self.parse_for_stmt(),
  1014→            TokenKind::Match => self.parse_match_stmt(),
  1015→            TokenKind::Return => self.parse_return_stmt(),
  1016→            TokenKind::Sync => self.parse_sync_stmt(),
  1017→            TokenKind::Free => self.parse_free_stmt(),
  1018→            TokenKind::Allocate => self.parse_allocate_stmt(),
  1019→            // BD directives
  1020→            TokenKind::Bd => self.parse_bd_directive(BdDirectiveKind::Bd),
  1021→            TokenKind::Repd => self.parse_bd_directive(BdDirectiveKind::Repd),
  1022→            TokenKind::Capd => self.parse_bd_directive(BdDirectiveKind::Capd),
  1023→            TokenKind::Reld => self.parse_bd_directive(BdDirectiveKind::Reld),
  1024→            // Handle `region` used as a variable name in assignments
  1025→            TokenKind::Region => self.parse_assign_or_expr_stmt(),
  1026→            // Break, Continue, and Loop are now proper keywords
  1027→            TokenKind::Break => self.parse_break_stmt(),
  1028→            TokenKind::Continue => self.parse_continue_stmt(),
  1029→            TokenKind::Loop => self.parse_loop_stmt(),
  1030→            // Unsafe block
  1031→            TokenKind::Unsafe => self.parse_unsafe_block(),
  1032→            // Handle Ident-based type-ascription declarations
  1033→            TokenKind::Ident => {
  1034→                // Check for type-ascription declaration: `name: type = expr;`
  1035→                let next = self.peek_next();
  1036→                if next.kind == TokenKind::Colon {
  1037→                    self.parse_type_ascription_decl()
  1038→                } else {
  1039→                    self.parse_assign_or_expr_stmt()
  1040→                }
  1041→            }
  1042→            _ => {
  1043→                // Could be an assignment or an expression statement.
  1044→                self.parse_assign_or_expr_stmt()
  1045→            }
  1046→        }
  1047→    }
  1048→
  1049→    /// `let` <name> [`:` <type>] [`=` <expr>] `;`
  1050→    fn parse_let_stmt(&mut self) -> Result<Stmt, ParseError> {
  1051→        let start = self.current.span.start;
  1052→        self.expect(TokenKind::Let)?;
  1053→
  1054→        let name = self.expect_name()?;
  1055→
  1056→        let ty = if self.at(TokenKind::Colon) {
  1057→            self.advance();
  1058→            Some(self.parse_type()?)
  1059→        } else {
  1060→            None
  1061→        };
  1062→
  1063→        let value = if self.at(TokenKind::Assign) {
  1064→            self.advance();
  1065→            self.parse_expr()?
  1066→        } else {
  1067→            // `let x;` without initializer — properly represented as Uninitialized
  1068→            Expr::Uninitialized {
  1069→                span: Span::new(self.current.span.start, self.current.span.start),
  1070→            }
  1071→        };
  1072→
  1073→        self.expect(TokenKind::Semicolon)?;
  1074→
  1075→        Ok(Stmt::Let(LetStmt {
  1076→            name,
  1077→            ty,
  1078→            value,
  1079→            span: Span::new(start, self.current.span.end),
  1080→        }))
  1081→    }
  1082→
  1083→    /// <name> `:` <type> `=` <expr> `;`  (type-ascription declaration without `let`)
  1084→    fn parse_type_ascription_decl(&mut self) -> Result<Stmt, ParseError> {
  1085→        let start = self.current.span.start;
  1086→
  1087→        let name = self.expect_name()?;
  1088→        self.expect(TokenKind::Colon)?;
  1089→        let ty = self.parse_type()?;
  1090→        self.expect(TokenKind::Assign)?;
  1091→        let value = self.parse_expr()?;
  1092→        self.expect(TokenKind::Semicolon)?;
  1093→
  1094→        Ok(Stmt::Let(LetStmt {
  1095→            name,
  1096→            ty: Some(ty),
  1097→            value,
  1098→            span: Span::new(start, self.current.span.end),
  1099→        }))
  1100→    }
  1101→
  1102→    /// Try to parse an assignment, compound assignment, or expression statement.
  1103→    ///
  1104→    /// Handles: `x = expr;`, `*x = expr;`, `(*x).field = expr;`, `x[i] = expr;`,
  1105→    /// `x += expr;`, `x -= expr;`, etc., or plain expression statement `expr;`.
  1106→    fn parse_assign_or_expr_stmt(&mut self) -> Result<Stmt, ParseError> {
  1107→        let start = self.current.span.start;
  1108→
  1109→        // Parse the full expression first (including prefix ops like `*`),
  1110→        // then check for `=` or compound assignment.
  1111→        let expr = self.parse_expr()?;
  1112→
  1113→        // Compound assignment: `expr += value;`, `expr -= value;`, etc.
  1114→        if is_compound_assign(self.current.kind) {
  1115→            let op = compound_op_from_token(self.current.kind, self.current.span)?;
  1116→            self.advance(); // consume compound op
  1117→            let value = self.parse_expr()?;
  1118→            self.expect(TokenKind::Semicolon)?;
  1119→
  1120→            let target = self.expr_to_assign_target(expr, start)?;
  1121→            return Ok(Stmt::CompoundAssign(CompoundAssignStmt {
  1122→                target,
  1123→                op,
  1124→                value,
  1125→                span: Span::new(start, self.current.span.end),
  1126→            }));
  1127→        }
  1128→
  1129→        if self.at(TokenKind::Assign) {
  1130→            self.advance(); // consume '='
  1131→            let value = self.parse_expr()?;
  1132→            self.expect(TokenKind::Semicolon)?;
  1133→
  1134→            let target = self.expr_to_assign_target(expr, start)?;
  1135→            return Ok(Stmt::Assign(AssignStmt {
  1136→                target,
  1137→                value,
  1138→                span: Span::new(start, self.current.span.end),
  1139→            }));
  1140→        }
  1141→
  1142→        // Plain expression statement.
  1143→        // Allow omitting semicolon before `}` (tail expression / block return value)
  1144→        // or before `)` (last argument in macro-like call) or at EOF.
  1145→        if !self.at(TokenKind::RBrace) && !self.at(TokenKind::RParen) && !self.at(TokenKind::Eof) {
  1146→            self.expect(TokenKind::Semicolon)?;
  1147→        }
  1148→        Ok(Stmt::Expr(ExprStmt {
  1149→            expr,
  1150→            span: Span::new(start, self.current.span.end),
  1151→        }))
  1152→    }
  1153→
  1154→    /// Convert an expression into an assignment target if possible.
  1155→    fn expr_to_assign_target(&self, expr: Expr, _start: usize) -> Result<AssignTarget, ParseError> {
  1156→        match expr {
  1157→            Expr::Var { name, span } => Ok(AssignTarget::Var { name, span }),
  1158→            Expr::Deref { expr, span } => Ok(AssignTarget::Deref { expr, span }),
  1159→            Expr::FieldAccess { expr, field, span } => {
  1160→                Ok(AssignTarget::DerefField { expr, field, span })
  1161→            }
  1162→            Expr::Index { expr, index, span } => Ok(AssignTarget::Index { expr, index, span }),
  1163→            _ => Err(ParseError::new(

  1312→    /// `match` <expr> `{` <arms> `}`
  1313→    fn parse_match_stmt(&mut self) -> Result<Stmt, ParseError> {
  1314→        let start = self.current.span.start;
  1315→        self.expect(TokenKind::Match)?;
  1316→        let subject = self.parse_expr()?;
  1317→        self.expect(TokenKind::LBrace)?;
  1318→
  1319→        let mut arms = Vec::new();
  1320→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
  1321→            let arm_start = self.current.span.start;
  1322→            let pattern = self.parse_match_pattern()?;
  1323→
  1324→            // Optional guard: `if expr`
  1325→            let guard = if self.at(TokenKind::If) {
  1326→                self.advance(); // consume 'if'
  1327→                Some(self.parse_expr()?)
  1328→            } else {
  1329→                None
  1330→            };
  1331→
  1332→            self.expect(TokenKind::FatArrow)?;
  1333→            let body = self.parse_expr()?;
  1334→            let arm_end = body.span().end;
  1335→            arms.push(MatchArm {
  1336→                pattern,
  1337→                guard,
  1338→                body,
  1339→                span: Span::new(arm_start, arm_end),
  1340→            });
  1341→            if self.at(TokenKind::Comma) {
  1342→                self.advance();
  1343→            }
  1344→        }
  1345→
  1346→        self.expect(TokenKind::RBrace)?;
  1347→        Ok(Stmt::Match(MatchStmt {
  1348→            subject,
  1349→            arms,
  1350→            span: Span::new(start, self.current.span.end),
  1351→        }))
  1352→    }

  1354→    /// Parse a match pattern.
  1355→    fn parse_match_pattern(&mut self) -> Result<MatchPattern, ParseError> {
  1356→        let span = self.current.span;
  1357→        match self.current.kind {
  1358→            // `_` wildcard — tokenized as either Ident("_") or TokenKind::Underscore
  1359→            TokenKind::Ident if self.current.lexeme == "_" => {
  1360→                self.advance();
  1361→                Ok(MatchPattern::Wildcard(span))
  1362→            }
  1363→            TokenKind::Underscore => {
  1364→                self.advance();
  1365→                Ok(MatchPattern::Wildcard(span))
  1366→            }
  1367→            TokenKind::Number => {
  1368→                let lexeme = self.current.lexeme.clone();
  1369→                self.advance();
  1370→                let value: i64 = lexeme.parse().map_err(|_| {
  1371→                    ParseError::new(
  1372→                        format!("invalid integer literal: {}", lexeme),
  1373→                        span,
  1374→                        ParseErrorKind::UnexpectedToken,
  1375→                    )
  1376→                })?;
  1377→                Ok(MatchPattern::Lit {
  1378→                    value: Lit::Int(value),
  1379→                    span,
  1380→                })
  1381→            }
  1382→            TokenKind::Float => {
  1383→                let lexeme = self.current.lexeme.clone();
  1384→                self.advance();
  1385→                let value: f64 = lexeme.parse().map_err(|_| {
  1386→                    ParseError::new(
  1387→                        format!("invalid float literal: {}", lexeme),
  1388→                        span,
  1389→                        ParseErrorKind::UnexpectedToken,
  1390→                    )
  1391→                })?;
  1392→                Ok(MatchPattern::Lit {
  1393→                    value: Lit::Float(value),
  1394→                    span,
  1395→                })
  1396→            }
  1397→            TokenKind::String => {
  1398→                let value = self.current.lexeme.clone();
  1399→                self.advance();
  1400→                Ok(MatchPattern::Lit {
  1401→                    value: Lit::String(value),
  1402→                    span,
  1403→                })
  1404→            }
  1405→            TokenKind::Address => {
  1406→                let lexeme = self.current.lexeme.clone();
  1407→                self.advance();
  1408→                let hex_str = lexeme.trim_start_matches("0x").trim_start_matches("0X");
  1409→                let value: u64 = u64::from_str_radix(hex_str, 16).map_err(|_| {
  1410→                    ParseError::invalid_address(format!("invalid hex address: {}", lexeme), span)
  1411→                })?;
  1412→                Ok(MatchPattern::Lit {
  1413→                    value: Lit::Address(value),
  1414→                    span,
  1415→                })
  1416→            }
  1417→            TokenKind::True => {
  1418→                self.advance();
  1419→                Ok(MatchPattern::Lit {
  1420→                    value: Lit::Bool(true),
  1421→                    span,
  1422→                })
  1423→            }
  1424→            TokenKind::False => {
  1425→                self.advance();
  1426→                Ok(MatchPattern::Lit {
  1427→                    value: Lit::Bool(false),
  1428→                    span,
  1429→                })
  1430→            }
  1431→            TokenKind::Ident => {
  1432→                let name = self.expect_name()?;
  1433→                // Check for struct pattern: `Name { field, ... }`
  1434→                if self.at(TokenKind::LBrace) {
  1435→                    self.advance();
  1436→                    let mut fields = Vec::new();
  1437→                    while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
  1438→                        fields.push(self.expect_name()?);
  1439→                        if self.at(TokenKind::Comma) {
  1440→                            self.advance();
  1441→                        } else {
  1442→                            break;
  1443→                        }
  1444→                    }
  1445→                    self.expect(TokenKind::RBrace)?;
  1446→                    Ok(MatchPattern::Struct { name, fields, span })
  1447→                } else if self.at(TokenKind::LParen) {
  1448→                    // Enum variant pattern: `Some(v)` or `None()`
  1449→                    self.advance();
  1450→                    let binding = if self.at(TokenKind::RParen) {
  1451→                        None
  1452→                    } else {
  1453→                        let b = self.expect_name()?;
  1454→                        Some(b)
  1455→                    };
  1456→                    self.expect(TokenKind::RParen)?;
  1457→                    Ok(MatchPattern::Enum {
  1458→                        name,
  1459→                        binding,
  1460→                        span,
  1461→                    })
  1462→                } else {
  1463→                    Ok(MatchPattern::Ident { name, span })
  1464→                }
  1465→            }
  1466→            // Handle Option/Result keywords as enum variant patterns
  1467→            TokenKind::SomeKw | TokenKind::NoneKw | TokenKind::OkKw | TokenKind::ErrKw => {
  1468→                let name = self.current.lexeme.clone();
  1469→                self.advance();
  1470→                if self.at(TokenKind::LParen) {
  1471→                    self.advance();
  1472→                    let binding = if self.at(TokenKind::RParen) {
  1473→                        None
  1474→                    } else {
  1475→                        let b = self.expect_name()?;
  1476→                        Some(b)
  1477→                    };
  1478→                    self.expect(TokenKind::RParen)?;
  1479→                    Ok(MatchPattern::Enum {
  1480→                        name,
  1481→                        binding,
  1482→                        span,
  1483→                    })
  1484→                } else {
  1485→                    Ok(MatchPattern::Ident { name, span })
  1486→                }
  1487→            }
  1488→            _ => Err(ParseError::unexpected(
  1489→                format!("expected match pattern, found {}", self.current.kind),
  1490→                self.current.span,
  1491→            )),
  1492→        }
  1493→    }

  1570→    // -- expression parsing (precedence climbing) ----------------------------
  1571→
  1572→    /// Parse an expression with full operator precedence.
  1573→    pub fn parse_expr(&mut self) -> Result<Expr, ParseError> {
  1574→        self.parse_expr_with_precedence(0)
  1575→    }
  1576→
  1577→    /// Precedence-climbing expression parser.
  1578→    ///
  1579→    /// Precedence levels (higher = binds tighter):
  1580→    ///   0  =>  ||
  1581→    ///   1  =>  &&
  1582→    ///   2  =>  ==  !=
  1583→    ///   3  =>  <  <=  >  >=
  1584→    ///   4  =>  |  (bitwise OR)
  1585→    ///   5  =>  ^  (bitwise XOR)
  1586→    ///   6  =>  &  (bitwise AND)
  1587→    ///   7  =>  <<  >>
  1588→    ///   8  =>  +  -
  1589→    ///   9  =>  *  /  %
  1590→    fn parse_expr_with_precedence(&mut self, min_prec: u8) -> Result<Expr, ParseError> {
  1591→        self.expr_depth += 1;
  1592→        if self.expr_depth > self.max_depth {
  1593→            self.expr_depth -= 1;
  1594→            return Err(ParseError::new(
  1595→                format!(
  1596→                    "expression nesting depth exceeds maximum ({})",
  1597→                    self.max_depth
  1598→                ),
  1599→                self.current.span,
  1600→                ParseErrorKind::InvalidSyntax,
  1601→            ));
  1602→        }
  1603→        let result = self.parse_expr_with_precedence_inner(min_prec);
  1604→        self.expr_depth -= 1;
  1605→        result
  1606→    }
  1607→
  1608→    /// Inner implementation of precedence climbing (called after depth check).
  1609→    fn parse_expr_with_precedence_inner(&mut self, min_prec: u8) -> Result<Expr, ParseError> {
  1610→        let mut left = self.parse_unary()?;
  1611→
  1612→        loop {
  1613→            // Range expression: `start..end` — handled as a very low-precedence
  1614→            // binary-like construct that produces Expr::Range instead of BinOp.
  1615→            if self.current.kind == TokenKind::DotDot && min_prec == 0 {
  1616→                let start = left.span().start;
  1617→                self.advance(); // consume '..'
  1618→                // Suppress struct literal parsing for the range end so that
  1619→                // `0..n { … }` does not interpret `n {` as a struct literal.
  1620→                let prev = self.no_struct_literal;
  1621→                self.no_struct_literal = true;
  1622→                let end_expr = self.parse_expr_with_precedence(1)?;
  1623→                self.no_struct_literal = prev;
  1624→                let end = end_expr.span().end;
  1625→                left = Expr::Range {
  1626→                    start: Box::new(left),
  1627→                    end: Box::new(end_expr),
  1628→                    span: Span::new(start, end),
  1629→                };
  1630→                continue;
  1631→            }
  1632→
  1633→            let (op, prec) = match self.current.kind {
  1634→                // Logical
  1635→                TokenKind::OrOr => (BinOp::Or, 0),
  1636→                TokenKind::AndAnd => (BinOp::And, 1),
  1637→                // Comparison
  1638→                TokenKind::EqEq => (BinOp::Eq, 2),
  1639→                TokenKind::Ne => (BinOp::Ne, 2),
  1640→                TokenKind::Lt => (BinOp::Lt, 3),
  1641→                TokenKind::Le => (BinOp::Le, 3),
  1642→                TokenKind::Gt => (BinOp::Gt, 3),
  1643→                TokenKind::Ge => (BinOp::Ge, 3),
  1644→                // Bitwise
  1645→                TokenKind::Pipe => (BinOp::BitOr, 4),
  1646→                TokenKind::Caret => (BinOp::BitXor, 5),
  1647→                TokenKind::Ampersand => (BinOp::BitAnd, 6),
  1648→                TokenKind::Shl => (BinOp::Shl, 7),
  1649→                TokenKind::Shr => (BinOp::Shr, 7),
  1650→                // Additive
  1651→                TokenKind::Plus => (BinOp::Add, 8),
  1652→                TokenKind::Minus => (BinOp::Sub, 8),
  1653→                // Multiplicative
  1654→                TokenKind::Star => (BinOp::Mul, 9),
  1655→                TokenKind::Slash => (BinOp::Div, 9),
  1656→                TokenKind::Percent => (BinOp::Mod, 9),
  1657→                _ => break,
  1658→            };
  1659→
  1660→            if prec < min_prec {
  1661→                break;
  1662→            }
  1663→
  1664→            let start = left.span().start;
  1665→            self.advance(); // consume the operator
  1666→            let right = self.parse_expr_with_precedence(prec + 1)?;
  1667→            let end = right.span().end;
  1668→
  1669→            left = Expr::BinOp {
  1670→                op,
  1671→                lhs: Box::new(left),
  1672→                rhs: Box::new(right),
  1673→                span: Span::new(start, end),
  1674→            };
  1675→        }
  1676→
  1677→        Ok(left)
  1678→    }
  1679→
  1680→    /// Parse a unary expression: prefix `-`, `!`, `*`, `@`, `~`, or primary.
  1681→    fn parse_unary(&mut self) -> Result<Expr, ParseError> {
  1682→        let start = self.current.span.start;
  1683→        match self.current.kind {
  1684→            TokenKind::Minus => {
  1685→                self.advance();
  1686→                let expr = self.parse_unary()?;
  1687→                let end = expr.span().end;
  1688→                Ok(Expr::UnOp {
  1689→                    op: UnOp::Neg,
  1690→                    expr: Box::new(expr),
  1691→                    span: Span::new(start, end),
  1692→                })
  1693→            }
  1694→            TokenKind::Bang => {
  1695→                self.advance();
  1696→                let expr = self.parse_unary()?;
  1697→                let end = expr.span().end;
  1698→                Ok(Expr::UnOp {
  1699→                    op: UnOp::Not,
  1700→                    expr: Box::new(expr),
  1701→                    span: Span::new(start, end),
  1702→                })
  1703→            }
  1704→            TokenKind::Tilde => {
  1705→                self.advance();
  1706→                let expr = self.parse_unary()?;

  1749→    fn parse_postfix(&mut self) -> Result<Expr, ParseError> {
  1750→        let mut expr = self.parse_primary()?;
  1751→

  1814→                    // belongs to a block statement like `if cond { … }`).

  1833→                            true

  1904→        Ok(expr)
  1905→    }
