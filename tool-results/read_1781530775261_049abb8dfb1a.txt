     1тЖТ     1тЖТ//! VUMA Integration Test Framework
     2тЖТ     2тЖТ//!
     3тЖТ     3тЖТ//! This module provides a comprehensive test framework for the VUMA project,
     4тЖТ     4тЖТ//! offering high-level helpers that compose the full compilation and
     5тЖТ     5тЖТ//! verification pipeline:
     6тЖТ     6тЖТ//!
     7тЖТ     7тЖТ//! 1. **Source parsing** тАФ `vuma-parser` lexes and parses VUMA source text into
     8тЖТ     8тЖТ//!    an AST, which is then converted to the parser's local SCG representation.
     9тЖТ     9тЖТ//! 2. **SCG bridging** тАФ The parser SCG is translated into `vuma-scg::SCG` for
    10тЖТ    10тЖТ//!    graph operations, validation, dominance/liveness analysis.
    11тЖТ    11тЖТ//! 3. **Invariant verification** тАФ The `vuma-ive` invariant aggregator runs all
    12тЖТ    12тЖТ//!    five VUMA invariant checks (liveness, exclusivity, interpretation, origin,
    13тЖТ    13тЖТ//!    cleanup) against the program.
    14тЖТ    14тЖТ//! 4. **Code generation** тАФ The `vuma-codegen` crate lowers the SCG through IR
    15тЖТ    15тЖТ//!    to ARM64 machine code, producing a minimal ELF binary.
    16тЖТ    16тЖТ//!
    17тЖТ    17тЖТ//! # Test Categories
    18тЖТ    18тЖТ//!
    19тЖТ    19тЖТ//! | Category       | Scope                                                        |
    20тЖТ    20тЖТ//! |----------------|--------------------------------------------------------------|
    21тЖТ    21тЖТ//! | [`Unit`]       | Individual crate functions, data structures, edge cases.     |
    22тЖТ    22тЖТ//! | [`Integration`]| Cross-crate pipelines (parse -> SCG -> verify, etc.).        |
    23тЖТ    23тЖТ//! | [`Verification`]| IVE invariant checks, proof generation, counterexamples.    |
    24тЖТ    24тЖТ//! | [`Codegen`]    | ARM64 code emission, register allocation, ELF generation.    |
    25тЖТ    25тЖТ//!
    26тЖТ    26тЖТ//! # Helper Macros
    27тЖТ    27тЖТ//!
    28тЖТ    28тЖТ//! The framework provides declarative macros that automatically annotate tests
    29тЖТ    29тЖТ//! with the correct [`TestCategory`] and integrate with the [`TestRegistry`]:
    30тЖТ    30тЖТ//!
    31тЖТ    31тЖТ//! - [`vuma_unit_test!`] тАФ unit test category
    32тЖТ    32тЖТ//! - [`vuma_integration_test!`] тАФ integration test category
    33тЖТ    33тЖТ//! - [`vuma_verification_test!`] тАФ verification test category
    34тЖТ    34тЖТ//! - [`vuma_codegen_test!`] тАФ codegen test category
    35тЖТ    35тЖТ//!
    36тЖТ    36тЖТ//! # Quick Start
    37тЖТ    37тЖТ//!
    38тЖТ    38тЖТ//! ```rust,ignore
    39тЖТ    39тЖТ//! use vuma_tests::framework::{build_scg_from_source, assert_verifies, assert_violation};
    40тЖТ    40тЖТ//! use vuma_ive::InvariantKind;
    41тЖТ    41тЖТ//!
    42тЖТ    42тЖТ//! // Build an SCG from source text.
    43тЖТ    43тЖТ//! let scg = build_scg_from_source("region pool = allocate(1024); free(pool);").unwrap();

    45тЖТ    45тЖТ//! // Assert that a program passes all invariant checks.

   439тЖТ   439тЖТ/// A registry for tracking test executions and reporting results.
   440тЖТ   440тЖТ///
   441тЖТ   441тЖТ/// The test registry accumulates [`TestRecord`]s as tests are run and
   442тЖТ   442тЖТ/// provides summary statistics and reporting. It is designed for use
   443тЖТ   443тЖТ/// with the [`run_test`] helper function.
   444тЖТ   444тЖТ///
   445тЖТ   445тЖТ/// # Thread Safety
   446тЖТ   446тЖТ///
   447тЖТ   447тЖТ/// The global registry uses atomic counters for pass/fail/ignore totals,
   448тЖТ   448тЖТ/// making it safe to use from multiple threads (e.g., with `cargo test`
   449тЖТ   449тЖТ/// parallelism). Detailed per-test records are not collected in the
   450тЖТ   450тЖТ/// global registry; use a local [`TestRegistry`] instance instead.
   451тЖТ   451тЖТpub struct TestRegistry {
   452тЖТ   452тЖТ    records: Vec<TestRecord>,
   453тЖТ   453тЖТ    pass_count: AtomicUsize,
   454тЖТ   454тЖТ    fail_count: AtomicUsize,
   455тЖТ   455тЖТ    ignore_count: AtomicUsize,
   456тЖТ   456тЖТ}
   457тЖТ   457тЖТ
   458тЖТ   458тЖТimpl TestRegistry {
   459тЖТ   459тЖТ    /// Create a new empty test registry.
   460тЖТ   460тЖТ    pub fn new() -> Self {
   461тЖТ   461тЖТ        Self {
   462тЖТ   462тЖТ            records: Vec::new(),
   463тЖТ   463тЖТ            pass_count: AtomicUsize::new(0),
   464тЖТ   464тЖТ            fail_count: AtomicUsize::new(0),
   465тЖТ   465тЖТ            ignore_count: AtomicUsize::new(0),
   466тЖТ   466тЖТ        }
   467тЖТ   467тЖТ    }
   468тЖТ   468тЖТ
   469тЖТ   469тЖТ    /// Record a test result.
   470тЖТ   470тЖТ    pub fn record(&mut self, record: TestRecord) {
   471тЖТ   471тЖТ        match record.outcome {
   472тЖТ   472тЖТ            TestOutcome::Pass => self.pass_count.fetch_add(1, Ordering::Relaxed),
   473тЖТ   473тЖТ            TestOutcome::Fail => self.fail_count.fetch_add(1, Ordering::Relaxed),
   474тЖТ   474тЖТ            TestOutcome::Ignore => self.ignore_count.fetch_add(1, Ordering::Relaxed),
   475тЖТ   475тЖТ        };
   476тЖТ   476тЖТ        self.records.push(record);
   477тЖТ   477тЖТ    }
   478тЖТ   478тЖТ
   479тЖТ   479тЖТ    /// Returns the total number of tests recorded.
   480тЖТ   480тЖТ    pub fn total(&self) -> usize {
   481тЖТ   481тЖТ        self.records.len()
   482тЖТ   482тЖТ    }
   483тЖТ   483тЖТ
   484тЖТ   484тЖТ    /// Returns the number of passing tests.
   485тЖТ   485тЖТ    pub fn pass_count(&self) -> usize {
   486тЖТ   486тЖТ        self.pass_count.load(Ordering::Relaxed)
   487тЖТ   487тЖТ    }
   488тЖТ   488тЖТ
   489тЖТ   489тЖТ    /// Returns the number of failing tests.
   490тЖТ   490тЖТ    pub fn fail_count(&self) -> usize {

   492тЖТ   492тЖТ    }
   493тЖТ   493тЖТ
   494тЖТ   494тЖТ    /// Returns the number of ignored tests.
   495тЖТ   495тЖТ    pub fn ignore_count(&self) -> usize {
   496тЖТ   496тЖТ        self.ignore_count.load(Ordering::Relaxed)
   497тЖТ   497тЖТ    }
   498тЖТ   498тЖТ
   499тЖТ   499тЖТ    /// Returns `true` if all recorded tests passed.
   500тЖТ   500тЖТ    pub fn all_passed(&self) -> bool {
   501тЖТ   501тЖТ        self.fail_count.load(Ordering::Relaxed) == 0 && self.pass_count.load(Ordering::Relaxed) > 0
   502тЖТ   502тЖТ    }
   503тЖТ   503тЖТ
   504тЖТ   504тЖТ    /// Returns records filtered by category.
   505тЖТ   505тЖТ    pub fn by_category(&self, category: TestCategory) -> Vec<&TestRecord> {
   506тЖТ   506тЖТ        self.records
   507тЖТ   507тЖТ            .iter()
   508тЖТ   508тЖТ            .filter(|r| r.category == category)
   509тЖТ   509тЖТ            .collect()
   510тЖТ   510тЖТ    }
   511тЖТ   511тЖТ
   512тЖТ   512тЖТ    /// Generate a summary report of all recorded tests.
   513тЖТ   513тЖТ    pub fn report(&self) -> TestReport<'_> {

   515тЖТ   515тЖТ        for record in &self.records {

   519тЖТ   519тЖТ        TestReport {
   520тЖТ   520тЖТ            total: self.total(),
   521тЖТ   521тЖТ            passed: self.pass_count(),

   596тЖТ   596тЖТ
   597тЖТ   597тЖТ/// Parse VUMA source text and convert it into a `vuma_scg::SCG`.
   598тЖТ   598тЖТ///
   599тЖТ   599тЖТ/// This is the primary entry point for the test framework. It:
   600тЖТ   600тЖТ/// 1. Lexes and parses the source using `vuma_parser`.
   601тЖТ   601тЖТ/// 2. Converts the AST to the parser's local SCG via `AstToScg`.
   602тЖТ   602тЖТ/// 3. Bridges the parser SCG into the full `vuma_scg::SCG` with regions,
   603тЖТ   603тЖТ///    typed node payloads, and proper edge kinds.
   604тЖТ   604тЖТ///
   605тЖТ   605тЖТ/// # Errors
   606тЖТ   606тЖТ///
   607тЖТ   607тЖТ/// Returns `Err(Vec<ParseError>)` if the source cannot be parsed or
   608тЖТ   608тЖТ/// converted to the parser SCG.
   609тЖТ   609тЖТ///
   610тЖТ   610тЖТ/// # Example
   611тЖТ   611тЖТ///
   612тЖТ   612тЖТ/// ```rust,ignore
   613тЖТ   613тЖТ/// let scg = build_scg_from_source("region pool = allocate(1024); free(pool);").unwrap();
   614тЖТ   614тЖТ/// assert!(scg.node_count() > 0);
   615тЖТ   615тЖТ/// ```
   616тЖТ   616тЖТpub fn build_scg_from_source(source: &str) -> Result<SCG, Vec<ParseError>> {
   617тЖТ   617тЖТ    // Step 1: Parse source -> AST.
   618тЖТ   618тЖТ    let mut parser = Parser::new(source);
   619тЖТ   619тЖТ    let result = parser.parse_program();
   620тЖТ   620тЖТ    if result.has_errors() {
   621тЖТ   621тЖТ        return Err(result.errors);
   622тЖТ   622тЖТ    }
   623тЖТ   623тЖТ    let program = result.unwrap();
   624тЖТ   624тЖТ
   625тЖТ   625тЖТ    // Step 2: Convert AST -> vuma_scg::SCG (AstToScg now produces SCG directly).
   626тЖТ   626тЖТ    let mut converter = AstToScg::new();
   627тЖТ   627тЖТ    let scg = converter.convert(&program).map_err(|e| vec![e])?;
   628тЖТ   628тЖТ
   629тЖТ   629тЖТ    Ok(scg)
   630тЖТ   630тЖТ}
   631тЖТ   631тЖТ
   632тЖТ   632тЖТ/// Parse VUMA source, build an SCG, and run all five IVE invariant checks.
   633тЖТ   633тЖТ///
   634тЖТ   634тЖТ/// Returns an [`AggregatedResult`] with per-invariant outcomes, an overall
   635тЖТ   635тЖТ/// verdict, and a summary of statistics.
   636тЖТ   636тЖТ///
   637тЖТ   637тЖТ/// # Example
   638тЖТ   638тЖТ///
   639тЖТ   639тЖТ/// ```rust,ignore
   640тЖТ   640тЖТ/// let result = verify_program("region buf = allocate(256); free(buf);");
   641тЖТ   641тЖТ/// println!("Verdict: {}", result.overall);
   642тЖТ   642тЖТ/// ```
   643тЖТ   643тЖТpub fn verify_program(source: &str) -> AggregatedResult {
   644тЖТ   644тЖТ    // Build the vuma_scg::SCG (silently treating parse errors as empty programs).
   645тЖТ   645тЖТ    let scg = build_scg_from_source(source).unwrap_or_default();
   646тЖТ   646тЖТ
   647тЖТ   647тЖТ    // Build verification input from the SCG.
   648тЖТ   648тЖТ    let input = VerificationInput::from_scg(scg);
   649тЖТ   649тЖТ
   650тЖТ   650тЖТ    // Run all five invariant checks at Normal level.
   651тЖТ   651тЖТ    let aggregator = InvariantAggregator::new();
   652тЖТ   652тЖТ    aggregator.verify_all(&input)
   653тЖТ   653тЖТ}
   654тЖТ   654тЖТ
   655тЖТ   655тЖТ/// Parse VUMA source, build an SCG, and run IVE checks at a specific
   656тЖТ   656тЖТ/// verification level.
   657тЖТ   657тЖТ///
   658тЖТ   658тЖТ/// This is like [`verify_program`] but allows control over the verification
   659тЖТ   659тЖТ/// level (Quick, Normal, or Exhaustive).
   660тЖТ   660тЖТ///
   661тЖТ   661тЖТ/// # Example
   662тЖТ   662тЖТ///
   663тЖТ   663тЖТ/// ```rust,ignore
   664тЖТ   664тЖТ/// use vuma_ive::VerificationLevel;
   665тЖТ   665тЖТ/// let result = verify_program_at_level("region buf = allocate(256);", VerificationLevel::Quick);
   666тЖТ   666тЖТ/// assert_eq!(result.per_invariant.len(), 2); // Quick only runs 2 checks
   667тЖТ   667тЖТ/// ```
   668тЖТ   668тЖТpub fn verify_program_at_level(source: &str, level: VerificationLevel) -> AggregatedResult {
   669тЖТ   669тЖТ    let scg = build_scg_from_source(source).unwrap_or_default();
   670тЖТ   670тЖТ    let input = VerificationInput::from_scg(scg);
   671тЖТ   671тЖТ
   672тЖТ   672тЖТ    let aggregator = InvariantAggregator::new().with_level(level);
   673тЖТ   673тЖТ    aggregator.verify_all(&input)
   674тЖТ   674тЖТ}
   675тЖТ   675тЖТ
   676тЖТ   676тЖТ/// Run the full pipeline with stage-by-stage tracking.
   677тЖТ   677тЖТ///
   678тЖТ   678тЖТ/// Returns a [`PipelineResult`] that records which stages succeeded or
   679тЖТ   679тЖТ/// failed, the constructed SCG (if any), and the verification result
   680тЖТ   680тЖТ/// (if the verification stage was reached).
   681тЖТ   681тЖТ///
   682тЖТ   682тЖТ/// This is the most detailed pipeline function, suitable for integration
   683тЖТ   683тЖТ/// tests that need to diagnose *where* in the pipeline a failure occurs.
   684тЖТ   684тЖТ///
   685тЖТ   685тЖТ/// # Example
   686тЖТ   686тЖТ///
   687тЖТ   687тЖТ/// ```rust,ignore
   688тЖТ   688тЖТ/// let result = verify_program_detailed("region buf = allocate(64); free(buf);");
   689тЖТ   689тЖТ/// assert!(result.all_passed());
   690тЖТ   690тЖТ/// assert!(result.scg.is_some());
   691тЖТ   691тЖТ/// assert!(result.verification.is_some());
   692тЖТ   692тЖТ/// ```
   693тЖТ   693тЖТpub fn verify_program_detailed(source: &str) -> PipelineResult {
   694тЖТ   694тЖТ    let start = Instant::now();
   695тЖТ   695тЖТ    let mut stages = Vec::new();
   696тЖТ   696тЖТ    let scg_result: Option<SCG>;
   697тЖТ   697тЖТ
   698тЖТ   698тЖТ    // Stage 1: Parse source -> AST.
   699тЖТ   699тЖТ    let mut parser = Parser::new(source);
   700тЖТ   700тЖТ    let parse_result = parser.parse_program();
   701тЖТ   701тЖТ    let program = if parse_result.has_errors() {
   702тЖТ   702тЖТ        stages.push((PipelineStage::Parse, StageOutcome::Failed));
   703тЖТ   703тЖТ        return PipelineResult {
   704тЖТ   704тЖТ            stages,
   705тЖТ   705тЖТ            scg: None,
   706тЖТ   706тЖТ            verification: None,
   707тЖТ   707тЖТ            elapsed_ms: start.elapsed().as_millis() as u64,
   708тЖТ   708тЖТ        };
   709тЖТ   709тЖТ    } else {
   710тЖТ   710тЖТ        stages.push((PipelineStage::Parse, StageOutcome::Passed));
   711тЖТ   711тЖТ        parse_result.unwrap()
   712тЖТ   712тЖТ    };
   713тЖТ   713тЖТ
   714тЖТ   714тЖТ    // Stage 2: AST -> vuma_scg::SCG (AstToScg now produces SCG directly).
   715тЖТ   715тЖТ    let mut converter = AstToScg::new();
   716тЖТ   716тЖТ    let scg = match converter.convert(&program) {
   717тЖТ   717тЖТ        Ok(s) => {
   718тЖТ   718тЖТ            stages.push((PipelineStage::AstToScg, StageOutcome::Passed));
   719тЖТ   719тЖТ            s
   720тЖТ   720тЖТ        }
   721тЖТ   721тЖТ        Err(_) => {
   722тЖТ   722тЖТ            stages.push((PipelineStage::AstToScg, StageOutcome::Failed));
   723тЖТ   723тЖТ            return PipelineResult {
   724тЖТ   724тЖТ                stages,
   725тЖТ   725тЖТ                scg: None,
   726тЖТ   726тЖТ                verification: None,
   727тЖТ   727тЖТ                elapsed_ms: start.elapsed().as_millis() as u64,
   728тЖТ   728тЖТ            };
   729тЖТ   729тЖТ        }
   730тЖТ   730тЖТ    };
   731тЖТ   731тЖТ
   732тЖТ   732тЖТ    // Stage 3: Bridge step no longer needed (AstToScg produces vuma_scg::SCG directly).
   733тЖТ   733тЖТ    stages.push((PipelineStage::ScgBridge, StageOutcome::Passed));
   734тЖТ   734тЖТ    scg_result = Some(scg.clone());
   735тЖТ   735тЖТ
   736тЖТ   736тЖТ    // Stage 4: Validate SCG.
   737тЖТ   737тЖТ    let validation = scg.validate();
   738тЖТ   738тЖТ    if validation.is_valid {
   739тЖТ   739тЖТ        stages.push((PipelineStage::ScgValidation, StageOutcome::Passed));
   740тЖТ   740тЖТ    } else {
   741тЖТ   741тЖТ        stages.push((PipelineStage::ScgValidation, StageOutcome::Failed));
   742тЖТ   742тЖТ    }
   743тЖТ   743тЖТ
   744тЖТ   744тЖТ    // Stage 5: IVE verification.
   745тЖТ   745тЖТ    let input = VerificationInput::from_scg(scg);
   746тЖТ   746тЖТ    let aggregator = InvariantAggregator::new();
   747тЖТ   747тЖТ    let aggregated = aggregator.verify_all(&input);
   748тЖТ   748тЖТ    stages.push((PipelineStage::IveVerification, StageOutcome::Passed));
   749тЖТ   749тЖТ    let verification_result: Option<AggregatedResult> = Some(aggregated);
   750тЖТ   750тЖТ
   751тЖТ   751тЖТ    // Stage 6: Codegen тАФ compile through SCG тЖТ IR тЖТ ARM64 ELF pipeline.
   752тЖТ   752тЖТ    let codegen_outcome = match compile_to_arm64(source) {
   753тЖТ   753тЖТ        Ok(_) => StageOutcome::Passed,
   754тЖТ   754тЖТ        Err(_) => StageOutcome::Failed,
   755тЖТ   755тЖТ    };
   756тЖТ   756тЖТ    stages.push((PipelineStage::Codegen, codegen_outcome));
   757тЖТ   757тЖТ
   758тЖТ   758тЖТ    PipelineResult {
   759тЖТ   759тЖТ        stages,
   760тЖТ   760тЖТ        scg: scg_result,
   761тЖТ   761тЖТ        verification: verification_result,
   762тЖТ   762тЖТ        elapsed_ms: start.elapsed().as_millis() as u64,
   763тЖТ   763тЖТ    }
   764тЖТ   764тЖТ}
   765тЖТ   765тЖТ
   766тЖТ   766тЖТ/// Run the full compilation pipeline: source -> SCG -> IR -> ARM64 ELF binary.
   767тЖТ   767тЖТ///
   768тЖТ   768тЖТ/// Returns the raw bytes of a minimal ELF binary for Linux/AArch64.
   769тЖТ   769тЖТ///
   770тЖТ   770тЖТ/// # Pipeline
   771тЖТ   771тЖТ///
   772тЖТ   772тЖТ/// 1. Parse source тЖТ AST.
   773тЖТ   773тЖТ/// 2. Convert AST тЖТ codegen SCG (bridge from parser types).
   774тЖТ   774тЖТ/// 3. Lower codegen SCG тЖТ IR via [`ScgToIr`].
   775тЖТ   775тЖТ/// 4. Emit ARM64 machine code тЖТ ELF binary via [`Emitter`].
   776тЖТ   776тЖТ///
   777тЖТ   777тЖТ/// # Errors
   778тЖТ   778тЖТ///
   779тЖТ   779тЖТ/// Returns [`CompileError`] if any stage fails:
   780тЖТ   780тЖТ/// - [`CompileError::Parse`] тАФ source parsing failed.
   781тЖТ   781тЖТ/// - [`CompileError::ScgConversion`] тАФ SCG bridge or validation failed.
   782тЖТ   782тЖТ/// - [`CompileError::Codegen`] тАФ IR translation or emission failed.
   783тЖТ   783тЖТpub fn compile_to_arm64(source: &str) -> Result<Vec<u8>, Vec<CompileError>> {
   784тЖТ   784тЖТ    // Step 1: Parse source -> AST.
   785тЖТ   785тЖТ    let mut parser = Parser::new(source);
   786тЖТ   786тЖТ    let result = parser.parse_program();
   787тЖТ   787тЖТ    if result.has_errors() {
   788тЖТ   788тЖТ        return Err(vec![CompileError::Parse(result.errors)]);
   789тЖТ   789тЖТ    }
   790тЖТ   790тЖТ    let program = result.unwrap();
   791тЖТ   791тЖТ
   792тЖТ   792тЖТ    // Step 2: Bridge parser AST тЖТ codegen SCG.
   793тЖТ   793тЖТ    let codegen_scg = bridge_ast_to_codegen_scg(&program);
   794тЖТ   794тЖТ
   795тЖТ   795тЖТ    // Step 3: Lower codegen SCG тЖТ IR.
   796тЖТ   796тЖТ    let mut ir_builder = ScgToIr::new();
   797тЖТ   797тЖТ    let ir_program = ir_builder
   798тЖТ   798тЖТ        .convert(&codegen_scg)
   799тЖТ   799тЖТ        .map_err(|e| vec![CompileError::Codegen(e.to_string())])?;
   800тЖТ   800тЖТ
   801тЖТ   801тЖТ    // Step 4: Emit ARM64 ELF binary.
   802тЖТ   802тЖТ    let mut emitter = Emitter::new();
   803тЖТ   803тЖТ    let elf_bytes = emitter
   804тЖТ   804тЖТ        .emit_program(&ir_program)
   805тЖТ   805тЖТ        .map_err(|e| vec![CompileError::Codegen(e.to_string())])?;
   806тЖТ   806тЖТ
   807тЖТ   807тЖТ    Ok(elf_bytes)
   808тЖТ   808тЖТ}

   810тЖТ   810тЖТ// ===========================================================================
   811тЖТ   811тЖТ// AST тЖТ Codegen SCG Bridge
   812тЖТ   812тЖТ// ===========================================================================
   813тЖТ   813тЖТ
   814тЖТ   814тЖТ/// Bridge a parsed VUMA AST into the codegen crate's SCG representation.
   815тЖТ   815тЖТ///
   816тЖТ   816тЖТ/// The codegen SCG is a simplified representation that only contains function
   817тЖТ   817тЖТ/// definitions and data declarations. This bridge walks the parser's `Program`
   818тЖТ   818тЖТ/// AST and converts each `FnDef` into a `CodegenScgFunction`. Other top-level
   819тЖТ   819тЖТ/// items (structs, enums, imports, etc.) are currently skipped since they do
   820тЖТ   820тЖТ/// not directly contribute to code emission.
   821тЖТ   821тЖТ///
   822тЖТ   822тЖТ/// Function bodies are translated on a best-effort basis: simple statements
   823тЖТ   823тЖТ/// (let bindings, assignments, returns, expressions) are converted, while
   824тЖТ   824тЖТ/// complex control flow (match, loops, etc.) falls back to a placeholder.
   825тЖТ   825тЖТfn bridge_ast_to_codegen_scg(program: &vuma_parser::ast::Program) -> CodegenScg {
   826тЖТ   826тЖТ    use vuma_parser::ast::Item;
   827тЖТ   827тЖТ
   828тЖТ   828тЖТ    let mut nodes: Vec<CodegenScgNode> = Vec::new();
   829тЖТ   829тЖТ
   830тЖТ   830тЖТ    for item in &program.items {
   831тЖТ   831тЖТ        if let Item::FnDef(fn_def) = item {
   832тЖТ   832тЖТ            let params: Vec<CodegenScgParam> = fn_def
   833тЖТ   833тЖТ                .params
   834тЖТ   834тЖТ                .iter()
   835тЖТ   835тЖТ                .map(|p| CodegenScgParam {
   836тЖТ   836тЖТ                    name: p.name.clone(),
   837тЖТ   837тЖТ                    ty: bridge_type_to_codegen(&p.ty),
   838тЖТ   838тЖТ                })
   839тЖТ   839тЖТ                .collect();
   840тЖТ   840тЖТ
   841тЖТ   841тЖТ            let results = if let Some(ref ret_ty) = fn_def.return_type {
   842тЖТ   842тЖТ                vec![bridge_type_to_codegen(&Some(ret_ty.clone()))]
   843тЖТ   843тЖТ            } else {
   844тЖТ   844тЖТ                vec![]
   845тЖТ   845тЖТ            };
   846тЖТ   846тЖТ
   847тЖТ   847тЖТ            let body = bridge_block_to_codegen_stmts(&fn_def.body);
   848тЖТ   848тЖТ
   849тЖТ   849тЖТ            nodes.push(CodegenScgNode::Function(CodegenScgFunction {
   850тЖТ   850тЖТ                name: fn_def.name.clone(),
   851тЖТ   851тЖТ                params,
   852тЖТ   852тЖТ                results,
   853тЖТ   853тЖТ                body,
   854тЖТ   854тЖТ            }));
   855тЖТ   855тЖТ        }
   856тЖТ   856тЖТ    }
   857тЖТ   857тЖТ
   858тЖТ   858тЖТ    CodegenScg { nodes }
   859тЖТ   859тЖТ}
   860тЖТ   860тЖТ
   861тЖТ   861тЖТ/// Convert a parser type annotation to a codegen SCG type.
   862тЖТ   862тЖТfn bridge_type_to_codegen(ty: &Option<vuma_parser::ast::Type>) -> CodegenScgType {
   863тЖТ   863тЖТ    match ty {
   864тЖТ   864тЖТ        Some(vuma_parser::ast::Type::BDBase(name)) => match name.as_str() {
   865тЖТ   865тЖТ            "i8" => CodegenScgType::I8,
   866тЖТ   866тЖТ            "i16" => CodegenScgType::I16,
   867тЖТ   867тЖТ            "i32" => CodegenScgType::I32,
   868тЖТ   868тЖТ            "i64" => CodegenScgType::I64,
   869тЖТ   869тЖТ            "u8" => CodegenScgType::U8,
   870тЖТ   870тЖТ            "u16" => CodegenScgType::U16,
   871тЖТ   871тЖТ            "u32" => CodegenScgType::U32,
   872тЖТ   872тЖТ            "u64" => CodegenScgType::U64,
   873тЖТ   873тЖТ            _ => CodegenScgType::I64, // default to pointer-sized int
   874тЖТ   874тЖТ        },
   875тЖТ   875тЖТ        Some(vuma_parser::ast::Type::Ptr(_)) => CodegenScgType::Ptr,
   876тЖТ   876тЖТ        Some(vuma_parser::ast::Type::RegionPtr { .. }) => CodegenScgType::Ptr,
   877тЖТ   877тЖТ        _ => CodegenScgType::Void,
   878тЖТ   878тЖТ    }
   879тЖТ   879тЖТ}
   880тЖТ   880тЖТ
   881тЖТ   881тЖТ/// Convert a parser block into codegen SCG statements.
   882тЖТ   882тЖТfn bridge_block_to_codegen_stmts(block: &vuma_parser::ast::Block) -> Vec<CodegenScgStatement> {
   883тЖТ   883тЖТ    block
   884тЖТ   884тЖТ        .statements
   885тЖТ   885тЖТ        .iter()
   886тЖТ   886тЖТ        .flat_map(bridge_stmt_to_codegen)
   887тЖТ   887тЖТ        .collect()
   888тЖТ   888тЖТ}
   889тЖТ   889тЖТ
   890тЖТ   890тЖТ/// Convert a single parser statement into zero or more codegen SCG statements.
   891тЖТ   891тЖТ///
   892тЖТ   892тЖТ/// Supports: Let, Assign, CompoundAssign, Allocate, Free, Cast, Return,
   893тЖТ   893тЖТ/// Expr, If, While (with condition), For, Loop, Break, Continue.
   894тЖТ   894тЖТ/// Falls back to empty for BdDirective, Sync, UnsafeBlock, Match, Access
   895тЖТ   895тЖТ/// (these are not directly representable in the codegen SCG and require
   896тЖТ   896тЖТ/// higher-level pipeline support).
   897тЖТ   897тЖТfn bridge_stmt_to_codegen(stmt: &vuma_parser::ast::Stmt) -> Vec<CodegenScgStatement> {
   898тЖТ   898тЖТ    use vuma_parser::ast::Stmt as PStmt;
   899тЖТ   899тЖТ
   900тЖТ   900тЖТ    match stmt {
   901тЖТ   901тЖТ        PStmt::Let(let_stmt) => {
   902тЖТ   902тЖТ            // Convert the right-hand side expression and emit a computation.
   903тЖТ   903тЖТ            let dst = let_stmt.name.clone();
   904тЖТ   904тЖТ            let (op, lhs, rhs) = bridge_expr_to_binop(&let_stmt.value, &dst);
   905тЖТ   905тЖТ            vec![CodegenScgStatement::Computation(CodegenComputationNode {
   906тЖТ   906тЖТ                dst,
   907тЖТ   907тЖТ                op,
   908тЖТ   908тЖТ                lhs,
   909тЖТ   909тЖТ                rhs,
   910тЖТ   910тЖТ                tail_call: false,
   911тЖТ   911тЖТ            })]
   912тЖТ   912тЖТ        }
   913тЖТ   913тЖТ        PStmt::Assign(assign_stmt) => {
   914тЖТ   914тЖТ            let dst = match &assign_stmt.target {
   915тЖТ   915тЖТ                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
   916тЖТ   916тЖТ                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
   917тЖТ   917тЖТ                vuma_parser::ast::AssignTarget::Deref { .. } => "_deref".into(),
   918тЖТ   918тЖТ                vuma_parser::ast::AssignTarget::Index { .. } => "_index".into(),
   919тЖТ   919тЖТ            };
   920тЖТ   920тЖТ            let (op, lhs, rhs) = bridge_expr_to_binop(&assign_stmt.value, &dst);
   921тЖТ   921тЖТ            vec![CodegenScgStatement::Computation(CodegenComputationNode {
   922тЖТ   922тЖТ                dst,
   923тЖТ   923тЖТ                op,
   924тЖТ   924тЖТ                lhs,
   925тЖТ   925тЖТ                rhs,
   926тЖТ   926тЖТ                tail_call: false,
   927тЖТ   927тЖТ            })]
   928тЖТ   928тЖТ        }
   929тЖТ   929тЖТ        PStmt::CompoundAssign(ca_stmt) => {
   930тЖТ   930тЖТ            // Compound assignment: `x += expr` тЖТ `x = x + expr`
   931тЖТ   931тЖТ            let dst = match &ca_stmt.target {
   932тЖТ   932тЖТ                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
   933тЖТ   933тЖТ                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
   934тЖТ   934тЖТ                _ => "_".into(),
   935тЖТ   935тЖТ            };
   936тЖТ   936тЖТ            let binop = match ca_stmt.op {
   937тЖТ   937тЖТ                vuma_parser::ast::CompoundOp::Add => BinOpKind::Add,
   938тЖТ   938тЖТ                vuma_parser::ast::CompoundOp::Sub => BinOpKind::Sub,
   939тЖТ   939тЖТ                vuma_parser::ast::CompoundOp::Mul => BinOpKind::Mul,
   940тЖТ   940тЖТ                vuma_parser::ast::CompoundOp::Div => BinOpKind::SDiv,
   941тЖТ   941тЖТ                vuma_parser::ast::CompoundOp::Mod => BinOpKind::SRem,
   942тЖТ   942тЖТ                vuma_parser::ast::CompoundOp::BitAnd => BinOpKind::And,
   943тЖТ   943тЖТ                vuma_parser::ast::CompoundOp::BitOr => BinOpKind::Or,
   944тЖТ   944тЖТ                vuma_parser::ast::CompoundOp::BitXor => BinOpKind::Xor,
   945тЖТ   945тЖТ                vuma_parser::ast::CompoundOp::Shl => BinOpKind::Shl,
   946тЖТ   946тЖТ                vuma_parser::ast::CompoundOp::Shr => BinOpKind::ShrL,
   947тЖТ   947тЖТ            };
   948тЖТ   948тЖТ            vec![CodegenScgStatement::Computation(CodegenComputationNode {
   949тЖТ   949тЖТ                dst: dst.clone(),
   950тЖТ   950тЖТ                op: binop,
   951тЖТ   951тЖТ                lhs: CodegenScgExpr::Var(dst),
   952тЖТ   952тЖТ                rhs: bridge_expr_to_scg_expr(&ca_stmt.value),
   953тЖТ   953тЖТ                tail_call: false,
   954тЖТ   954тЖТ            })]
   955тЖТ   955тЖТ        }
   956тЖТ   956тЖТ        PStmt::Allocate(alloc_stmt) => {
   957тЖТ   957тЖТ            // Allocation: emit as computation `dst = allocate(size)`.
   958тЖТ   958тЖТ            // Represented as Add(dst_var, 0) with a named destination so
   959тЖТ   959тЖТ            // the IR builder can track the allocation point.
   960тЖТ   960тЖТ            vec![CodegenScgStatement::Computation(CodegenComputationNode {
   961тЖТ   961тЖТ                dst: "_alloc".into(),
   962тЖТ   962тЖТ                op: BinOpKind::Add,
   963тЖТ   963тЖТ                lhs: CodegenScgExpr::Int(0),
   964тЖТ   964тЖТ                rhs: bridge_expr_to_scg_expr(&alloc_stmt.size),
   965тЖТ   965тЖТ                tail_call: false,
   966тЖТ   966тЖТ            })]
   967тЖТ   967тЖТ        }
   968тЖТ   968тЖТ        PStmt::Free(free_stmt) => {
   969тЖТ   969тЖТ            // Deallocation: emit as computation to preserve the free point
   970тЖТ   970тЖТ            // in the SCG. The actual free is a runtime call.
   971тЖТ   971тЖТ            vec![CodegenScgStatement::Computation(CodegenComputationNode {
   972тЖТ   972тЖТ                dst: "_free".into(),
   973тЖТ   973тЖТ                op: BinOpKind::Add,
   974тЖТ   974тЖТ                lhs: bridge_expr_to_scg_expr(&free_stmt.ptr),
   975тЖТ   975тЖТ                rhs: CodegenScgExpr::Int(0),
   976тЖТ   976тЖТ                tail_call: false,
   977тЖТ   977тЖТ            })]
   978тЖТ   978тЖТ        }
   979тЖТ   979тЖТ        PStmt::Cast(cast_stmt) => {
   980тЖТ   980тЖТ            // Cast: `expr as Type` тЖТ emit as computation preserving the value.
   981тЖТ   981тЖТ            vec![CodegenScgStatement::Computation(CodegenComputationNode {
   982тЖТ   982тЖТ                dst: "_cast".into(),
   983тЖТ   983тЖТ                op: BinOpKind::Add,
   984тЖТ   984тЖТ                lhs: bridge_expr_to_scg_expr(&cast_stmt.expr),
   985тЖТ   985тЖТ                rhs: CodegenScgExpr::Int(0),
   986тЖТ   986тЖТ                tail_call: false,
   987тЖТ   987тЖТ            })]
   988тЖТ   988тЖТ        }
   989тЖТ   989тЖТ        PStmt::Return(ret_stmt) => {
   990тЖТ   990тЖТ            let values = match &ret_stmt.value {
   991тЖТ   991тЖТ                Some(expr) => vec![bridge_expr_to_scg_expr(expr)],
   992тЖТ   992тЖТ                None => vec![],
   993тЖТ   993тЖТ            };
   994тЖТ   994тЖТ            vec![CodegenScgStatement::Return(values)]
   995тЖТ   995тЖТ        }
   996тЖТ   996тЖТ        PStmt::Expr(expr_stmt) => {
   997тЖТ   997тЖТ            // Expression statement: evaluate and discard result.
   998тЖТ   998тЖТ            let (op, lhs, rhs) = bridge_expr_to_binop(&expr_stmt.expr, "_");
   999тЖТ   999тЖТ            vec![CodegenScgStatement::Computation(CodegenComputationNode {
  1000тЖТ  1000тЖТ                dst: "_".into(),
  1001тЖТ  1001тЖТ                op,
  1002тЖТ  1002тЖТ                lhs,
  1003тЖТ  1003тЖТ                rhs,
  1004тЖТ  1004тЖТ                tail_call: false,
  1005тЖТ  1005тЖТ            })]
  1006тЖТ  1006тЖТ        }
  1007тЖТ  1007тЖТ        PStmt::If(if_stmt) => {
  1008тЖТ  1008тЖТ            let cond = bridge_expr_to_scg_expr(&if_stmt.condition);
  1009тЖТ  1009тЖТ            let then_body = bridge_block_to_codegen_stmts(&if_stmt.then_block);
  1010тЖТ  1010тЖТ            let else_body = if_stmt
  1011тЖТ  1011тЖТ                .else_block
  1012тЖТ  1012тЖТ                .as_ref()
  1013тЖТ  1013тЖТ                .map(bridge_block_to_codegen_stmts);
  1014тЖТ  1014тЖТ            vec![CodegenScgStatement::Control(CodegenControlNode::If {
  1015тЖТ  1015тЖТ                cond,
  1016тЖТ  1016тЖТ                then_body,
  1017тЖТ  1017тЖТ                else_body,
  1018тЖТ  1018тЖТ            })]
  1019тЖТ  1019тЖТ        }
  1020тЖТ  1020тЖТ        PStmt::While(while_stmt) => {
  1021тЖТ  1021тЖТ            let cond = bridge_expr_to_scg_expr(&while_stmt.condition);
  1022тЖТ  1022тЖТ            let body = bridge_block_to_codegen_stmts(&while_stmt.body);
  1023тЖТ  1023тЖТ            vec![CodegenScgStatement::Control(CodegenControlNode::If {
  1024тЖТ  1024тЖТ                cond,
  1025тЖТ  1025тЖТ                then_body: body,
  1026тЖТ  1026тЖТ                else_body: Some(vec![CodegenScgStatement::Control(
  1027тЖТ  1027тЖТ                    CodegenControlNode::Break,
  1028тЖТ  1028тЖТ                )]),
  1029тЖТ  1029тЖТ            })]
  1030тЖТ  1030тЖТ        }
  1031тЖТ  1031тЖТ        PStmt::For(for_stmt) => {
  1032тЖТ  1032тЖТ            // For loop: `for name in iter { body }`
  1033тЖТ  1033тЖТ            // Bridge as a loop with an iterator variable and break condition.
  1034тЖТ  1034тЖТ            // The iterator variable is tracked as a computation node.
  1035тЖТ  1035тЖТ            let iter_init = CodegenScgStatement::Computation(CodegenComputationNode {
  1036тЖТ  1036тЖТ                dst: for_stmt.name.clone(),
  1037тЖТ  1037тЖТ                op: BinOpKind::Add,
  1038тЖТ  1038тЖТ                lhs: CodegenScgExpr::Int(0),
  1039тЖТ  1039тЖТ                rhs: bridge_expr_to_scg_expr(&for_stmt.iter),
  1040тЖТ  1040тЖТ                tail_call: false,
  1041тЖТ  1041тЖТ            });
  1042тЖТ  1042тЖТ            let body = bridge_block_to_codegen_stmts(&for_stmt.body);
  1043тЖТ  1043тЖТ            let mut loop_stmts = vec![iter_init];
  1044тЖТ  1044тЖТ            loop_stmts.push(CodegenScgStatement::Control(CodegenControlNode::Loop {
  1045тЖТ  1045тЖТ                body,
  1046тЖТ  1046тЖТ            }));
  1047тЖТ  1047тЖТ            loop_stmts
  1048тЖТ  1048тЖТ        }
  1049тЖТ  1049тЖТ        PStmt::Loop(loop_stmt) => {
  1050тЖТ  1050тЖТ            let body = bridge_block_to_codegen_stmts(&loop_stmt.body);
  1051тЖТ  1051тЖТ            vec![CodegenScgStatement::Control(CodegenControlNode::Loop {
  1052тЖТ  1052тЖТ                body,
  1053тЖТ  1053тЖТ            })]
  1054тЖТ  1054тЖТ        }
  1055тЖТ  1055тЖТ        PStmt::Break(_) => vec![CodegenScgStatement::Control(CodegenControlNode::Break)],
  1056тЖТ  1056тЖТ        PStmt::Continue(_) => vec![CodegenScgStatement::Control(CodegenControlNode::Continue)],
  1057тЖТ  1057тЖТ        // BdDirective, Sync, UnsafeBlock, Match, Access are not directly
  1058тЖТ  1058тЖТ        // representable in the codegen SCG тАФ they require higher-level
  1059тЖТ  1059тЖТ        // pipeline support. Skip them gracefully.
  1060тЖТ  1060тЖТ        PStmt::BdDirective(_)
  1061тЖТ  1061тЖТ        | PStmt::Sync(_)
  1062тЖТ  1062тЖТ        | PStmt::UnsafeBlock { .. }
  1063тЖТ  1063тЖТ        | PStmt::Match(_)
  1064тЖТ  1064тЖТ        | PStmt::Access(_) => vec![],
  1065тЖТ  1065тЖТ    }
  1066тЖТ  1066тЖТ}

  1068тЖТ  1068тЖТ/// Convert a parser expression into a codegen SCG expression.
  1069тЖТ  1069тЖТ///
  1070тЖТ  1070тЖТ/// Handles: Var, Lit (Int/Float/Bool), BinOp, UnaryOp, Call, Deref,
  1071тЖТ  1071тЖТ/// Cast, Index, Range. Complex expressions that cannot be represented
  1072тЖТ  1072тЖТ/// directly in the codegen SCG are preserved as named variable references
  1073тЖТ  1073тЖТ/// or zero-valued placeholders.
  1074тЖТ  1074тЖТfn bridge_expr_to_scg_expr(expr: &vuma_parser::ast::Expr) -> CodegenScgExpr {
  1075тЖТ  1075тЖТ    use vuma_parser::ast::{Expr, Lit, UnOp};
  1076тЖТ  1076тЖТ
  1077тЖТ  1077тЖТ    match expr {
  1078тЖТ  1078тЖТ        Expr::Var { name, .. } => CodegenScgExpr::Var(name.clone()),
  1079тЖТ  1079тЖТ        Expr::Lit { value, .. } => match value {
  1080тЖТ  1080тЖТ            Lit::Int(n) => CodegenScgExpr::Int(*n),
  1081тЖТ  1081тЖТ            Lit::Float(f) => CodegenScgExpr::Float(*f),
  1082тЖТ  1082тЖТ            Lit::Bool(b) => CodegenScgExpr::Int(if *b { 1 } else { 0 }),
  1083тЖТ  1083тЖТ            _ => CodegenScgExpr::Int(0),
  1084тЖТ  1084тЖТ        },
  1085тЖТ  1085тЖТ        Expr::BinOp { op, lhs, rhs: _, .. } => {
  1086тЖТ  1086тЖТ            // Recursively bridge nested binary operations.
  1087тЖТ  1087тЖТ            let cg_op = match op {
  1088тЖТ  1088тЖТ                vuma_parser::ast::BinOp::Add => BinOpKind::Add,
  1089тЖТ  1089тЖТ                vuma_parser::ast::BinOp::Sub => BinOpKind::Sub,
  1090тЖТ  1090тЖТ                vuma_parser::ast::BinOp::Mul => BinOpKind::Mul,
  1091тЖТ  1091тЖТ                vuma_parser::ast::BinOp::Div => BinOpKind::SDiv,
  1092тЖТ  1092тЖТ                vuma_parser::ast::BinOp::Mod => BinOpKind::SRem,
  1093тЖТ  1093тЖТ                vuma_parser::ast::BinOp::And => BinOpKind::And,
  1094тЖТ  1094тЖТ                vuma_parser::ast::BinOp::Or => BinOpKind::Or,
  1095тЖТ  1095тЖТ                vuma_parser::ast::BinOp::BitAnd => BinOpKind::And,
  1096тЖТ  1096тЖТ                vuma_parser::ast::BinOp::BitOr => BinOpKind::Or,
  1097тЖТ  1097тЖТ                vuma_parser::ast::BinOp::BitXor => BinOpKind::Xor,
  1098тЖТ  1098тЖТ                vuma_parser::ast::BinOp::Shl => BinOpKind::Shl,
  1099тЖТ  1099тЖТ                vuma_parser::ast::BinOp::Shr => BinOpKind::ShrL,
  1100тЖТ  1100тЖТ                vuma_parser::ast::BinOp::Eq => BinOpKind::Eq,
  1101тЖТ  1101тЖТ                vuma_parser::ast::BinOp::Ne => BinOpKind::Ne,
  1102тЖТ  1102тЖТ                vuma_parser::ast::BinOp::Lt => BinOpKind::SLt,
  1103тЖТ  1103тЖТ                vuma_parser::ast::BinOp::Le => BinOpKind::SLe,
  1104тЖТ  1104тЖТ                vuma_parser::ast::BinOp::Gt => BinOpKind::SGt,
  1105тЖТ  1105тЖТ                vuma_parser::ast::BinOp::Ge => BinOpKind::SGe,
  1106тЖТ  1106тЖТ            };
  1107тЖТ  1107тЖТ            // For SCG expressions, represent as the LHS of a BinOp computation.
  1108тЖТ  1108тЖТ            // The full computation is captured at the ComputationNode level.
  1109тЖТ  1109тЖТ            let _ = cg_op;
  1110тЖТ  1110тЖТ            bridge_expr_to_scg_expr(lhs)
  1111тЖТ  1111тЖТ        }
  1112тЖТ  1112тЖТ        Expr::UnOp { op, expr: operand, .. } => {
  1113тЖТ  1113тЖТ            match op {
  1114тЖТ  1114тЖТ                UnOp::BitNot => {
  1115тЖТ  1115тЖТ                    // ~x тЖТ XOR(x, -1)
  1116тЖТ  1116тЖТ                    CodegenScgExpr::Int(-1)
  1117тЖТ  1117тЖТ                }
  1118тЖТ  1118тЖТ                UnOp::Neg => {
  1119тЖТ  1119тЖТ                    // -x тЖТ SUB(0, x)
  1120тЖТ  1120тЖТ                    CodegenScgExpr::Int(0)
  1121тЖТ  1121тЖТ                }
  1122тЖТ  1122тЖТ                _ => bridge_expr_to_scg_expr(operand),
  1123тЖТ  1123тЖТ            }
  1124тЖТ  1124тЖТ        }
  1125тЖТ  1125тЖТ        Expr::Call { callee, args, .. } => {
  1126тЖТ  1126тЖТ            // Function call: preserve the callee name as a variable
  1127тЖТ  1127тЖТ            // reference. The actual call semantics require IR-level support.
  1128тЖТ  1128тЖТ            let _ = args;
  1129тЖТ  1129тЖТ            match callee.as_ref() {
  1130тЖТ  1130тЖТ                Expr::Var { name, .. } => CodegenScgExpr::Var(format!("_call_{}", name)),
  1131тЖТ  1131тЖТ                _ => CodegenScgExpr::Int(0),
  1132тЖТ  1132тЖТ            }
  1133тЖТ  1133тЖТ        }
  1134тЖТ  1134тЖТ        Expr::Deref { expr, .. } => {
  1135тЖТ  1135тЖТ            // Dereference: track the inner expression.
  1136тЖТ  1136тЖТ            bridge_expr_to_scg_expr(expr)
  1137тЖТ  1137тЖТ        }
  1138тЖТ  1138тЖТ        Expr::Cast { expr, .. } => {
  1139тЖТ  1139тЖТ            // Cast: the value is preserved, only the type changes.
  1140тЖТ  1140тЖТ            bridge_expr_to_scg_expr(expr)
  1141тЖТ  1141тЖТ        }
  1142тЖТ  1142тЖТ        Expr::Index { expr, index, .. } => {
  1143тЖТ  1143тЖТ            // Index: track base + index for address computation.
  1144тЖТ  1144тЖТ            let _ = index;
  1145тЖТ  1145тЖТ            bridge_expr_to_scg_expr(expr)
  1146тЖТ  1146тЖТ        }
  1147тЖТ  1147тЖТ        Expr::Range { start, end, .. } => {
  1148тЖТ  1148тЖТ            // Range: track the start value as the primary expression.
  1149тЖТ  1149тЖТ            let _ = end;
  1150тЖТ  1150тЖТ            bridge_expr_to_scg_expr(start)
  1151тЖТ  1151тЖТ        }
  1152тЖТ  1152тЖТ        _ => CodegenScgExpr::Int(0), // fallback for Field, Group, etc.
  1153тЖТ  1153тЖТ    }
  1154тЖТ  1154тЖТ}
  1155тЖТ  1155тЖТ
  1156тЖТ  1156тЖТ/// Convert a parser expression into a (BinOpKind, lhs, rhs) triple for use
  1157тЖТ  1157тЖТ/// in a `ComputationNode`. Simple expressions like integer literals or variable
  1158тЖТ  1158тЖТ/// references are converted into `Add(x, 0)` to represent a "move" operation.
  1159тЖТ  1159тЖТ///
  1160тЖТ  1160тЖТ/// Handles: Lit, Var, BinOp (with recursive decomposition), UnaryOp,
  1161тЖТ  1161тЖТ/// Call (as a placeholder), Cast, Deref, Index, Range.
  1162тЖТ  1162тЖТfn bridge_expr_to_binop(
  1163тЖТ  1163тЖТ    expr: &vuma_parser::ast::Expr,
  1164тЖТ  1164тЖТ    _dst: &str,
  1165тЖТ  1165тЖТ) -> (BinOpKind, CodegenScgExpr, CodegenScgExpr) {
  1166тЖТ  1166тЖТ    use vuma_parser::ast::{BinOp, Expr, Lit, UnOp};
  1167тЖТ  1167тЖТ
  1168тЖТ  1168тЖТ    match expr {
  1169тЖТ  1169тЖТ        Expr::Lit {
  1170тЖТ  1170тЖТ            value: Lit::Int(n), ..
  1171тЖТ  1171тЖТ        } => (
  1172тЖТ  1172тЖТ            BinOpKind::Add,
  1173тЖТ  1173тЖТ            CodegenScgExpr::Int(0),
  1174тЖТ  1174тЖТ            CodegenScgExpr::Int(*n),
  1175тЖТ  1175тЖТ        ),
  1176тЖТ  1176тЖТ        Expr::Var { name, .. } => (
  1177тЖТ  1177тЖТ            BinOpKind::Add,
  1178тЖТ  1178тЖТ            CodegenScgExpr::Var(name.clone()),
  1179тЖТ  1179тЖТ            CodegenScgExpr::Int(0),
  1180тЖТ  1180тЖТ        ),
  1181тЖТ  1181тЖТ        Expr::BinOp { op, lhs, rhs, .. } => {
  1182тЖТ  1182тЖТ            let cg_op = match op {
  1183тЖТ  1183тЖТ                BinOp::Add => BinOpKind::Add,
  1184тЖТ  1184тЖТ                BinOp::Sub => BinOpKind::Sub,
  1185тЖТ  1185тЖТ                BinOp::Mul => BinOpKind::Mul,
  1186тЖТ  1186тЖТ                BinOp::Div => BinOpKind::SDiv,
  1187тЖТ  1187тЖТ                BinOp::Mod => BinOpKind::SRem,
  1188тЖТ  1188тЖТ                BinOp::And => BinOpKind::And,
  1189тЖТ  1189тЖТ                BinOp::Or => BinOpKind::Or,
  1190тЖТ  1190тЖТ                BinOp::BitAnd => BinOpKind::And,
  1191тЖТ  1191тЖТ                BinOp::BitOr => BinOpKind::Or,
  1192тЖТ  1192тЖТ                BinOp::BitXor => BinOpKind::Xor,
  1193тЖТ  1193тЖТ                BinOp::Shl => BinOpKind::Shl,
  1194тЖТ  1194тЖТ                BinOp::Shr => BinOpKind::ShrL,
  1195тЖТ  1195тЖТ                BinOp::Eq => BinOpKind::Eq,
  1196тЖТ  1196тЖТ                BinOp::Ne => BinOpKind::Ne,
  1197тЖТ  1197тЖТ                BinOp::Lt => BinOpKind::SLt,
  1198тЖТ  1198тЖТ                BinOp::Le => BinOpKind::SLe,
  1199тЖТ  1199тЖТ                BinOp::Gt => BinOpKind::SGt,
  1200тЖТ  1200тЖТ                BinOp::Ge => BinOpKind::SGe,
  1201тЖТ  1201тЖТ            };
  1202тЖТ  1202тЖТ            (
  1203тЖТ  1203тЖТ                cg_op,
  1204тЖТ  1204тЖТ                bridge_expr_to_scg_expr(lhs),
  1205тЖТ  1205тЖТ                bridge_expr_to_scg_expr(rhs),
  1206тЖТ  1206тЖТ            )
  1207тЖТ  1207тЖТ        }
  1208тЖТ  1208тЖТ        Expr::UnOp { op, expr: operand, .. } => {
  1209тЖТ  1209тЖТ            match op {
  1210тЖТ  1210тЖТ                UnOp::BitNot => (
  1211тЖТ  1211тЖТ                    BinOpKind::Xor,
  1212тЖТ  1212тЖТ                    bridge_expr_to_scg_expr(operand),
  1213тЖТ  1213тЖТ                    CodegenScgExpr::Int(-1),
  1214тЖТ  1214тЖТ                ),
  1215тЖТ  1215тЖТ                UnOp::Neg => (
  1216тЖТ  1216тЖТ                    BinOpKind::Sub,
  1217тЖТ  1217тЖТ                    CodegenScgExpr::Int(0),
  1218тЖТ  1218тЖТ                    bridge_expr_to_scg_expr(operand),
  1219тЖТ  1219тЖТ                ),
  1220тЖТ  1220тЖТ                _ => (
  1221тЖТ  1221тЖТ                    BinOpKind::Add,
  1222тЖТ  1222тЖТ                    bridge_expr_to_scg_expr(operand),
  1223тЖТ  1223тЖТ                    CodegenScgExpr::Int(0),
  1224тЖТ  1224тЖТ                ),
  1225тЖТ  1225тЖТ            }
  1226тЖТ  1226тЖТ        }
  1227тЖТ  1227тЖТ        Expr::Call { callee, args, .. } => {
  1228тЖТ  1228тЖТ            // Function call: preserve callee and arguments.
  1229тЖТ  1229тЖТ            let lhs = args.first()
  1230тЖТ  1230тЖТ                .map(|a| bridge_expr_to_scg_expr(a))
  1231тЖТ  1231тЖТ                .unwrap_or(CodegenScgExpr::Int(0));
  1232тЖТ  1232тЖТ            let _ = callee;
  1233тЖТ  1233тЖТ            (
  1234тЖТ  1234тЖТ                BinOpKind::Add,
  1235тЖТ  1235тЖТ                lhs,
  1236тЖТ  1236тЖТ                CodegenScgExpr::Int(0),
  1237тЖТ  1237тЖТ            )
  1238тЖТ  1238тЖТ        }
  1239тЖТ  1239тЖТ        Expr::Deref { expr, .. } => (
  1240тЖТ  1240тЖТ            BinOpKind::Add,
  1241тЖТ  1241тЖТ            bridge_expr_to_scg_expr(expr),
  1242тЖТ  1242тЖТ            CodegenScgExpr::Int(0),
  1243тЖТ  1243тЖТ        ),
  1244тЖТ  1244тЖТ        Expr::Cast { expr, .. } => (
  1245тЖТ  1245тЖТ            BinOpKind::Add,
  1246тЖТ  1246тЖТ            bridge_expr_to_scg_expr(expr),
  1247тЖТ  1247тЖТ            CodegenScgExpr::Int(0),
  1248тЖТ  1248тЖТ        ),
  1249тЖТ  1249тЖТ        Expr::Index { expr, index, .. } => {
  1250тЖТ  1250тЖТ            (
  1251тЖТ  1251тЖТ                BinOpKind::Add,
  1252тЖТ  1252тЖТ                bridge_expr_to_scg_expr(expr),
  1253тЖТ  1253тЖТ                bridge_expr_to_scg_expr(index),
  1254тЖТ  1254тЖТ            )
  1255тЖТ  1255тЖТ        }
  1256тЖТ  1256тЖТ        Expr::Range { start, end, .. } => {
  1257тЖТ  1257тЖТ            (
  1258тЖТ  1258тЖТ                BinOpKind::Add,
  1259тЖТ  1259тЖТ                bridge_expr_to_scg_expr(start),
  1260тЖТ  1260тЖТ                bridge_expr_to_scg_expr(end),
  1261тЖТ  1261тЖТ            )
  1262тЖТ  1262тЖТ        }
  1263тЖТ  1263тЖТ        _ => {
  1264тЖТ  1264тЖТ            (
  1265тЖТ  1265тЖТ                BinOpKind::Add,
  1266тЖТ  1266тЖТ                CodegenScgExpr::Int(0),
  1267тЖТ  1267тЖТ                CodegenScgExpr::Int(0),
  1268тЖТ  1268тЖТ            )
  1269тЖТ  1269тЖТ        }
  1270тЖТ  1270тЖТ    }
  1271тЖТ  1271тЖТ}

  1277тЖТ  1277тЖТ/// Assert that the given VUMA source program passes all invariant checks.
  1278тЖТ  1278тЖТ///
  1279тЖТ  1279тЖТ/// This panics if the aggregated verification result contains any
  1280тЖТ  1280тЖТ/// concrete invariant **violations**.
  1281тЖТ  1281тЖТ///
  1282тЖТ  1282тЖТ/// In the current implementation, all IVE checks return `Unverified`, so the
  1283тЖТ  1283тЖТ/// overall verdict is `Inconclusive`. This assertion accepts `Inconclusive`
  1284тЖТ  1284тЖТ/// as a passing result until the IVE checks are fully implemented. It only
  1285тЖТ  1285тЖТ/// fails when a concrete invariant *violation* is detected.
  1286тЖТ  1286тЖТ///
  1287тЖТ  1287тЖТ/// # Panics
  1288тЖТ  1288тЖТ///
  1289тЖТ  1289тЖТ/// Panics if any invariant check returns a `Violated` status.
  1290тЖТ  1290тЖТpub fn assert_verifies(source: &str) {
  1291тЖТ  1291тЖТ    let result = verify_program(source);
  1292тЖТ  1292тЖТ
  1293тЖТ  1293тЖТ    // Check for any violations.
  1294тЖТ  1294тЖТ    let violations: Vec<_> = result
  1295тЖТ  1295тЖТ        .per_invariant
  1296тЖТ  1296тЖТ        .iter()
  1297тЖТ  1297тЖТ        .filter(|pir| pir.is_fail())
  1298тЖТ  1298тЖТ        .collect();
  1299тЖТ  1299тЖТ
  1300тЖТ  1300тЖТ    if !violations.is_empty() {
  1301тЖТ  1301тЖТ        let violation_labels: Vec<String> = violations
  1302тЖТ  1302тЖТ            .iter()
  1303тЖТ  1303тЖТ            .map(|v| format!("{}: {}", v.kind, v.result.message))
  1304тЖТ  1304тЖТ            .collect();
  1305тЖТ  1305тЖТ        panic!(
  1306тЖТ  1306тЖТ            "Program failed invariant checks:\n  {}\n\nOverall verdict: {}\nSource:\n{}",
  1307тЖТ  1307тЖТ            violation_labels.join("\n  "),
  1308тЖТ  1308тЖТ            result.overall,
  1309тЖТ  1309тЖТ            source
  1310тЖТ  1310тЖТ        );
  1311тЖТ  1311тЖТ    }
  1312тЖТ  1312тЖТ}

  1314тЖТ  1314тЖТ/// Assert that the given VUMA source program violates a specific invariant.
  1315тЖТ  1315тЖТ///
  1316тЖТ  1316тЖТ/// This panics if the specified invariant is *not* violated (i.e., it is
  1317тЖТ  1317тЖТ/// proven, probably safe, or unverified).
  1318тЖТ  1318тЖТ///
  1319тЖТ  1319тЖТ/// **Note**: Since the IVE verification engine is currently in placeholder
  1320тЖТ  1320тЖТ/// mode (returning `Unverified` for all checks), this assertion will fail
  1321тЖТ  1321тЖТ/// for all programs until the actual verification logic is implemented.
  1322тЖТ  1322тЖТ/// Use this assertion in tests that are explicitly marked as
  1323тЖТ  1323тЖТ/// `#[ignore]` pending IVE implementation.
  1324тЖТ  1324тЖТ///
  1325тЖТ  1325тЖТ/// # Panics
  1326тЖТ  1326тЖТ///
  1327тЖТ  1327тЖТ/// Panics if the specified invariant is not violated.
  1328тЖТ  1328тЖТpub fn assert_violation(source: &str, invariant: InvariantKind) {
  1329тЖТ  1329тЖТ    let result = verify_program(source);
  1330тЖТ  1330тЖТ
  1331тЖТ  1331тЖТ    let pir = result
  1332тЖТ  1332тЖТ        .per_invariant
  1333тЖТ  1333тЖТ        .iter()
  1334тЖТ  1334тЖТ        .find(|pir| pir.kind == invariant);
  1335тЖТ  1335тЖТ
  1336тЖТ  1336тЖТ    match pir {
  1337тЖТ  1337тЖТ        Some(pir) if pir.is_fail() => {
  1338тЖТ  1338тЖТ            // Expected: the invariant is violated.
  1339тЖТ  1339тЖТ        }
  1340тЖТ  1340тЖТ        Some(pir) => {
  1341тЖТ  1341тЖТ            panic!(
  1342тЖТ  1342тЖТ                "Expected violation of {:?}, but got status: {:?}\n\
  1343тЖТ  1343тЖТ                 Message: {}\n\
  1344тЖТ  1344тЖТ                 Overall verdict: {}\n\
  1345тЖТ  1345тЖТ                 Source:\n{}",
  1346тЖТ  1346тЖТ                invariant, pir.result.status, pir.result.message, result.overall, source
  1347тЖТ  1347тЖТ            );
  1348тЖТ  1348тЖТ        }
  1349тЖТ  1349тЖТ        None => {
  1350тЖТ  1350тЖТ            panic!(

  1392тЖТ  1392тЖТ/// Run a test closure and record the result in a [`TestRegistry`].
  1393тЖТ  1393тЖТ///
  1394тЖТ  1394тЖТ/// This is the registry-aware version of [`run_test`]. It measures elapsed
  1395тЖТ  1395тЖТ/// time and records a [`TestRecord`] in the provided registry.
  1396тЖТ  1396тЖТpub fn run_registered_test(
  1397тЖТ  1397тЖТ    registry: &mut TestRegistry,
  1398тЖТ  1398тЖТ    category: TestCategory,
  1399тЖТ  1399тЖТ    name: &str,
  1400тЖТ  1400тЖТ    f: impl FnOnce() + panic::UnwindSafe,
  1401тЖТ  1401тЖТ) -> TestOutcome {
  1402тЖТ  1402тЖТ    let start = Instant::now();
  1403тЖТ  1403тЖТ    let result = panic::catch_unwind(f);
  1404тЖТ  1404тЖТ    let elapsed_us = start.elapsed().as_micros() as u64;
  1405тЖТ  1405тЖТ
  1406тЖТ  1406тЖТ    let (outcome, message) = match result {
  1407тЖТ  1407тЖТ        Ok(()) => (TestOutcome::Pass, None),
  1408тЖТ  1408тЖТ        Err(_) => (TestOutcome::Fail, Some("test panicked".to_string())),
  1409тЖТ  1409тЖТ    };
  1410тЖТ  1410тЖТ
  1411тЖТ  1411тЖТ    let icon = match outcome {
  1412тЖТ  1412тЖТ        TestOutcome::Pass => "PASS",
  1413тЖТ  1413тЖТ        TestOutcome::Fail => "FAIL",
  1414тЖТ  1414тЖТ        TestOutcome::Ignore => "SKIP",
  1415тЖТ  1415тЖТ    };
  1416тЖТ  1416тЖТ    println!("[{}] [{:?}] {} ({}us)", icon, category, name, elapsed_us);
  1417тЖТ  1417тЖТ
  1418тЖТ  1418тЖТ    registry.record(TestRecord {
  1419тЖТ  1419тЖТ        name: name.to_string(),
  1420тЖТ  1420тЖТ        category,
  1421тЖТ  1421тЖТ        outcome,
  1422тЖТ  1422тЖТ        elapsed_us,
  1423тЖТ  1423тЖТ        message,
  1424тЖТ  1424тЖТ    });
  1425тЖТ  1425тЖТ
  1426тЖТ  1426тЖТ    outcome
  1427тЖТ  1427тЖТ}
