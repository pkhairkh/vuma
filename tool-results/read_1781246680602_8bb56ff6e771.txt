     1→//! VUMA CLI — Command-line interface for the VUMA compiler.
     2→//!
     3→//! Subcommands:
     4→//! - `vuma build <file>` — Parse + compile to ARM64 ELF (default), save to output file
     5→//! - `vuma run <file>`   — Build + execute (via QEMU aarch64 or native)
     6→//! - `vuma check <file>` — Parse + SCG + BD inference + IVE verification only
     7→//! - `vuma emit <isa> <file>` — Compile to specific ISA
     8→//! - `vuma disasm <file>` — Read binary and disassemble
     9→//! - `vuma verify <file>` — Run IVE 5-invariant verification
    10→//! - `vuma repl` — Interactive REPL (parse expr, print AST)

   230→/// Read source file content with a human-readable error on failure.
   231→fn read_source(path: &PathBuf) -> Result<String, String> {
   232→    fs::read_to_string(path)
   233→        .map_err(|e| format!("error: cannot read source file '{}': {}", path.display(), e))
   234→}

   238→    CompileConfig {

   247→/// Print compilation errors to stderr in a human-readable format.
   248→fn print_errors(errors: &[VumaError]) {
   249→    for err in errors {
   250→        eprintln!("error[{}]: {}", err.stage(), err);
   251→    }

   254→/// Determine the default output path for a given input file.
   255→fn default_output_path(input: &Path) -> PathBuf {
   256→    let stem = input.file_stem().unwrap_or_default().to_string_lossy();
   257→    let dir = input.parent().unwrap_or(std::path::Path::new("."));
   258→    dir.join(format!("{}.o", stem))
   259→}

   265→/// `vuma build <file>` — Parse + compile to ELF, save to output file.
   266→fn cmd_build(
   267→    cli: &Cli,
   268→    file: &PathBuf,
   269→    output: &Option<PathBuf>,
   270→    target: TargetArg,
   271→) -> Result<(), String> {
   272→    let source = read_source(file)?;
   273→    let config = make_config(cli, CompileTarget::from(target));
   274→    let result = compile(&source, &config).map_err(|errors| {
   275→        print_errors(&errors);
   276→        format!("compilation failed with {} error(s)", errors.len())
   277→    })?;
   278→
   279→    let out_path = output
   280→        .as_ref()
   281→        .cloned()
   282→        .unwrap_or_else(|| default_output_path(file));
   283→    fs::write(&out_path, &result.binary).map_err(|e| {
   284→        format!(
   285→            "error: cannot write output file '{}': {}",
   286→            out_path.display(),
   287→            e
   288→        )
   289→    })?;
   290→
   291→    println!(
   292→        "Compiled {} -> {} ({} bytes, {} SCG nodes, {} IR instructions)",
   293→        file.display(),
   294→        out_path.display(),
   295→        result.binary.len(),
   296→        result.scg.node_count(),
   297→        result.ir_instruction_count,
   298→    );
   299→
   300→    // Print stage timings.
   301→    for (stage, ms) in &result.stage_timings {
   302→        println!("  {:20} {}ms", stage, ms);
   303→    }
   304→
   305→    Ok(())
   306→}
   307→
   308→/// `vuma run <file>` — Build + execute.
   309→fn cmd_run(cli: &Cli, file: &PathBuf, args: &[String]) -> Result<(), String> {
   310→    let source = read_source(file)?;
   311→    let config = make_config(cli, CompileTarget::Pi5Linux);
   312→    let result = compile(&source, &config).map_err(|errors| {
   313→        print_errors(&errors);
   314→        format!("compilation failed with {} error(s)", errors.len())
   315→    })?;
   316→
   317→    // Write binary to a temp file.
   318→    let tmp_dir = std::env::temp_dir();
   319→    let exe_path = tmp_dir.join(format!("vuma_run_{}", std::process::id()));
   320→    fs::write(&exe_path, &result.binary).map_err(|e| {
   321→        format!(
   322→            "error: cannot write temporary executable '{}': {}",
   323→            exe_path.display(),
   324→            e
   325→        )
   326→    })?;
   327→
   328→    // Make the file executable on Unix.
   329→    #[cfg(unix)]
   330→    {
   331→        use std::os::unix::fs::PermissionsExt;
   332→        let mut perms = fs::metadata(&exe_path)
   333→            .map_err(|e| format!("error: cannot stat '{}': {}", exe_path.display(), e))?
   334→            .permissions();
   335→        perms.set_mode(0o755);
   336→        fs::set_permissions(&exe_path, perms)
   337→            .map_err(|e| format!("error: cannot chmod '{}': {}", exe_path.display(), e))?;
   338→    }
   339→
   340→    // Try to execute: first natively (if on aarch64), then via qemu-aarch64.
   341→    let native_output = Command::new(&exe_path).args(args).output();
   342→
   343→    let output = match native_output {
   344→        Ok(out) => out,
   345→        Err(_) => {
   346→            // Native execution failed; try qemu-aarch64.
   347→            let qemu_output = Command::new("qemu-aarch64")
   348→                .arg(&exe_path)
   349→                .args(args)
   350→                .output()
   351→                .map_err(|e| {
   352→                    format!(
   353→                        "error: failed to execute binary (neither native nor qemu-aarch64): {}",
   354→                        e
   355→                    )
   356→                })?;
   357→            qemu_output
   358→        }
   359→    };
   360→
   361→    io::stdout()
   362→        .write_all(&output.stdout)
   363→        .map_err(|e| format!("error: failed to write program output: {}", e))?;
   364→    io::stderr()
   365→        .write_all(&output.stderr)
   366→        .map_err(|e| format!("error: failed to write program stderr: {}", e))?;
   367→
   368→    // Clean up temp file.
   369→    let _ = fs::remove_file(&exe_path);
   370→
   371→    let code = output.status.code().unwrap_or(1);
   372→    if code != 0 {
   373→        return Err(format!("program exited with code {}", code));
   374→    }
   375→
   376→    Ok(())
   377→}

   379→/// `vuma check <file>` — Parse + SCG + BD inference + IVE verification only.
   380→fn cmd_check(cli: &Cli, file: &PathBuf) -> Result<(), String> {
   381→    let source = read_source(file)?;
   382→    let mut config = make_config(cli, CompileTarget::Pi5Linux);
   383→    // check mode: always run verification, don't skip
   384→    if config.verification_level == VerificationLevel::None {
   385→        config.verification_level = VerificationLevel::Normal;
   386→    }
   387→
   388→    // Run the full compile — the check command doesn't save the binary,
   389→    // it just verifies the program compiles and passes verification.
   390→    let result = compile(&source, &config).map_err(|errors| {
   391→        print_errors(&errors);
   392→        format!("check failed with {} error(s)", errors.len())
   393→    })?;
   394→
   395→    println!("Check passed for {}", file.display());
   396→    println!("  SCG nodes:      {}", result.scg.node_count());
   397→    println!("  IR functions:   {}", result.ir_function_count);
   398→    println!("  IR instructions: {}", result.ir_instruction_count);
   399→
   400→    if let Some(ref verification) = result.verification {
   401→        println!("  Verification:   {}", verification.overall);
   402→    }
   403→
   404→    // Print stage timings.
   405→    for (stage, ms) in &result.stage_timings {
   406→        println!("  {:20} {}ms", stage, ms);
   407→    }
   408→
   409→    Ok(())
   410→}

   412→/// `vuma emit <isa> <file>` — Compile to a specific ISA target.
   413→///
   414→/// Uses the direct AST → codegen SCG → IR path for better code quality
   415→/// than the main pipeline (which goes through the semantic SCG and loses
   416→/// most program semantics in the SCG→IR bridge).
   417→fn cmd_emit(
   418→    cli: &Cli,
   419→    isa: IsaArg,
   420→    file: &PathBuf,
   421→    output: &Option<PathBuf>,
   422→) -> Result<(), String> {
   423→    let source = read_source(file)?;
   424→    let backend_kind = BackendKind::from(isa);
   425→
   426→    // Step 1: Parse source → AST.
   427→    let mut parser = vuma_parser::Parser::new(&source);
   428→    let parse_result = parser.parse_program();
   429→    if parse_result.is_err() {
   430→        return Err(format!("parse error: {:?}", parse_result.errors));
   431→    }
   432→    let program = parse_result.value.unwrap();
   433→
   434→    // Step 2: Bridge parser AST → codegen SCG.
   435→    let codegen_scg = bridge_ast_to_codegen_scg(&program);
   436→
   437→    // Step 3: Lower codegen SCG → IR.
   438→    let mut ir_builder = ScgToIr::new();
   439→    let ir_program = ir_builder.convert(&codegen_scg).map_err(|e| {
   440→        format!("IR conversion error: {}", e)
   441→    })?;
   442→
   443→    eprintln!("[emit] IR program has {} functions", ir_program.functions.len());
   444→
   445→    // Step 4: Create backend and allocate registers.
   446→    let backend = create_backend(backend_kind).map_err(|e| {
   447→        format!("error: cannot create {} backend: {}", backend_kind.isa_name(), e)
   448→    })?;
   449→
   450→    let mut allocated_functions = Vec::new();
   451→    for func in &ir_program.functions {
   452→        match backend.allocate_registers(func) {
   453→            Ok(allocated) => allocated_functions.push(allocated),
   454→            Err(e) => {
   455→                eprintln!("warning: register allocation failed for '{}': {}", func.name, e);
   456→            }
   457→        }
   458→    }
   459→
   460→    let out_path = output
   461→        .as_ref()
   462→        .cloned()
   463→        .unwrap_or_else(|| default_output_path(file));
   464→
   465→    // Step 5: Encode and write output.
   466→    if !allocated_functions.is_empty() {
   467→        let allocated_program = vuma_codegen::backend::AllocatedProgram {
   468→            functions: allocated_functions,
   469→            total_code_size: 0,
   470→            total_data_size: 0,
   471→        };
   472→        match backend.encode_program(&allocated_program) {
   473→            Ok(bytes) => {
   474→                fs::write(&out_path, &bytes).map_err(|e| {
   475→                    format!("error: cannot write output file '{}': {}", out_path.display(), e)
   476→                })?;
   477→                println!(
   478→                    "Emitted {} -> {} ({} bytes, ISA: {})",
   479→                    file.display(),
   480→                    out_path.display(),
   481→                    bytes.len(),
   482→                    backend.name(),
   483→                );
   484→            }
   485→            Err(e) => {
   486→                return Err(format!("{} encoding failed: {}", backend.name(), e));
   487→            }
   488→        }
   489→    } else {
   490→        return Err("no functions were successfully allocated".to_string());
   491→    }
   492→
   493→    Ok(())
   494→}
   495→
   496→/// Bridge parser AST → codegen SCG.
   497→///
   498→/// This converts the parser's AST into the codegen's SCG representation,
   499→/// which can then be lowered to IR. This bypasses the main pipeline's
   500→/// semantic SCG which loses most program semantics.
   501→fn bridge_ast_to_codegen_scg(program: &vuma_parser::ast::Program) -> Scg {
   502→    use vuma_parser::ast::Item;
   503→
   504→    let mut nodes: Vec<ScgNode> = Vec::new();
   505→
   506→    for item in &program.items {
   507→        if let Item::FnDef(fn_def) = item {
   508→            let params: Vec<ScgParam> = fn_def
   509→                .params
   510→                .iter()
   511→                .map(|p| ScgParam {
   512→                    name: p.name.clone(),
   513→                    ty: bridge_type_to_codegen_scg(&p.ty),
   514→                })
   515→                .collect();
   516→
   517→            let results = if let Some(ref ret_ty) = fn_def.return_type {
   518→                vec![bridge_type_to_codegen_scg(&Some(ret_ty.clone()))]
   519→            } else {
   520→                vec![]
   521→            };
   522→
   523→            let body = bridge_block_to_scg_stmts(&fn_def.body);
   524→
   525→            nodes.push(ScgNode::Function(ScgFunction {
   526→                name: fn_def.name.clone(),
   527→                params,
   528→                results,
   529→                body,
   530→            }));
   531→        }
   532→    }
   533→
   534→    Scg { nodes }
   535→}
   536→
   537→/// Convert a parser type annotation to a codegen SCG type.
   538→fn bridge_type_to_codegen_scg(ty: &Option<vuma_parser::ast::Type>) -> ScgType {
   539→    match ty {
   540→        Some(vuma_parser::ast::Type::BDBase(name)) => match name.as_str() {
   541→            "i8" => ScgType::I8,
   542→            "i16" => ScgType::I16,
   543→            "i32" => ScgType::I32,
   544→            "i64" => ScgType::I64,
   545→            "u8" => ScgType::U8,
   546→            "u16" => ScgType::U16,
   547→            "u32" => ScgType::U32,
   548→            "u64" => ScgType::U64,
   549→            _ => ScgType::I64,
   550→        },
   551→        Some(vuma_parser::ast::Type::Ptr(_)) => ScgType::Ptr,
   552→        Some(vuma_parser::ast::Type::RegionPtr { .. }) => ScgType::Ptr,
   553→        _ => ScgType::Void,
   554→    }
   555→}
   556→
   557→/// Convert a parser block into codegen SCG statements.
   558→fn bridge_block_to_scg_stmts(block: &vuma_parser::ast::Block) -> Vec<ScgStatement> {
   559→    block
   560→        .statements
   561→        .iter()
   562→        .flat_map(bridge_stmt_to_scg)
   563→        .collect()
   564→}
   565→
   566→/// Convert a single parser statement into zero or more codegen SCG statements.
   567→fn bridge_stmt_to_scg(stmt: &vuma_parser::ast::Stmt) -> Vec<ScgStatement> {
   568→    use vuma_parser::ast::Stmt as PStmt;
   569→
   570→    match stmt {
   571→        PStmt::Let(let_stmt) => {
   572→            // Check if the RHS is an allocate() call
   573→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &let_stmt.value {
   574→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
   575→                    if name == "allocate" {
   576→                        let size: u32 = args.first()
   577→                            .and_then(|a| {
   578→                                if let vuma_parser::ast::Expr::Lit { value, .. } = a {
   579→                                    if let vuma_parser::ast::Lit::Int(n) = value {
   580→                                        return Some(*n as u32);
   581→                                    }
   582→                                }
   583→                                None
   584→                            })
   585→                            .unwrap_or(8);
   586→                        return vec![ScgStatement::Allocation(AllocationNode::Stack {
   587→                            name: let_stmt.name.clone(),
   588→                            size,
   589→                            ty: ScgType::Ptr,
   590→                        })];
   591→                    }
   592→                    // Other function calls → CallNode
   593→                    let scg_args: Vec<ScgExpr> = args.iter()
   594→                        .map(|a| bridge_expr_to_scg_expr(a))
   595→                        .collect();
   596→                    return vec![ScgStatement::Call(CallNode {
   597→                        dst: Some(let_stmt.name.clone()),
   598→                        func: name.clone(),
   599→                        args: scg_args,
   600→                    })];
   601→                }
   602→            }
   603→            // Check if the RHS is a dereference (*ptr) → Access::Load
   604→            if let vuma_parser::ast::Expr::Deref { expr, .. } = &let_stmt.value {
   605→                let ptr = bridge_expr_to_scg_expr(expr);
   606→                return vec![ScgStatement::Access(AccessNode::Load {
   607→                    dst: let_stmt.name.clone(),
   608→                    ptr,
   609→                    offset: None,
   610→                })];
   611→            }
   612→            let dst = let_stmt.name.clone();
   613→            let (op, lhs, rhs) = bridge_expr_to_binop(&let_stmt.value, &dst);
   614→            vec![ScgStatement::Computation(ComputationNode {
   615→                dst,
   616→                op,
   617→                lhs,
   618→                rhs,
   619→                tail_call: false,
   620→            })]
   621→        }
   622→        PStmt::Assign(assign_stmt) => {
   623→            // Detect dereference writes: `*ptr = val` → Access::Store
   624→            if let vuma_parser::ast::AssignTarget::Deref { expr, .. } = &assign_stmt.target {
   625→                let ptr = bridge_expr_to_scg_expr(expr);
   626→                let value = bridge_expr_to_scg_expr(&assign_stmt.value);
   627→                return vec![ScgStatement::Access(AccessNode::Store {
   628→                    ptr,
   629→                    offset: None,
   630→                    value,
   631→                })];
   632→            }
   633→            let dst = match &assign_stmt.target {
   634→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
   635→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
   636→                vuma_parser::ast::AssignTarget::Deref { .. } => "_deref".into(),
   637→                vuma_parser::ast::AssignTarget::Index { .. } => "_index".into(),
   638→            };
   639→            // Detect allocate() expression: `region = allocate(n)` → AllocationNode::Stack
   640→            if let vuma_parser::ast::Expr::Allocate { size, .. } = &assign_stmt.value {
   641→                let size_val: u32 = match size.as_ref() {
   642→                    vuma_parser::ast::Expr::Lit { value, .. } => {
   643→                        if let vuma_parser::ast::Lit::Int(n) = value {
   644→                            *n as u32
   645→                        } else {
   646→                            8
   647→                        }
   648→                    }
   649→                    _ => 8,
   650→                };
   651→                return vec![ScgStatement::Allocation(AllocationNode::Stack {
   652→                    name: dst,
   653→                    size: size_val,
   654→                    ty: ScgType::Ptr,
   655→                })];
   656→            }
   657→            // Detect function calls in assign: `x = foo(args)` → CallNode
   658→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &assign_stmt.value {
   659→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
   660→                    let scg_args: Vec<ScgExpr> = args.iter()
   661→                        .map(|a| bridge_expr_to_scg_expr(a))
   662→                        .collect();
   663→                    return vec![ScgStatement::Call(CallNode {
   664→                        dst: Some(dst),
   665→                        func: name.clone(),
   666→                        args: scg_args,
   667→                    })];
   668→                }
   669→            }
   670→            let (op, lhs, rhs) = bridge_expr_to_binop(&assign_stmt.value, &dst);
   671→            vec![ScgStatement::Computation(ComputationNode {
   672→                dst,
   673→                op,
   674→                lhs,
   675→                rhs,
   676→                tail_call: false,
   677→            })]
   678→        }
   679→        PStmt::CompoundAssign(ca_stmt) => {
   680→            let dst = match &ca_stmt.target {
   681→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
   682→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
   683→                _ => "_".into(),
   684→            };
   685→            let binop = match ca_stmt.op {
   686→                vuma_parser::ast::CompoundOp::Add => BinOpKind::Add,
   687→                vuma_parser::ast::CompoundOp::Sub => BinOpKind::Sub,
   688→                vuma_parser::ast::CompoundOp::Mul => BinOpKind::Mul,
   689→                vuma_parser::ast::CompoundOp::Div => BinOpKind::SDiv,
   690→                vuma_parser::ast::CompoundOp::Mod => BinOpKind::SRem,
   691→                vuma_parser::ast::CompoundOp::BitAnd => BinOpKind::And,
   692→                vuma_parser::ast::CompoundOp::BitOr => BinOpKind::Or,
   693→                vuma_parser::ast::CompoundOp::BitXor => BinOpKind::Xor,
   694→                vuma_parser::ast::CompoundOp::Shl => BinOpKind::Shl,
   695→                vuma_parser::ast::CompoundOp::Shr => BinOpKind::ShrL,
   696→            };
   697→            vec![ScgStatement::Computation(ComputationNode {
   698→                dst: dst.clone(),
   699→                op: binop,
   700→                lhs: ScgExpr::Var(dst),
   701→                rhs: bridge_expr_to_scg_expr(&ca_stmt.value),
   702→                tail_call: false,
   703→            })]
   704→        }
   705→        PStmt::Allocate(alloc_stmt) => {
   706→            vec![ScgStatement::Allocation(AllocationNode::Stack {
   707→                name: "_alloc".into(),
   708→                size: 8, // default size
   709→                ty: ScgType::Ptr,
   710→            })]
   711→        }
   712→        PStmt::Free(_free_stmt) => {
   713→            // Stack allocations are freed automatically when the function returns.
   714→            // For now, skip emitting free calls since we don't have a heap allocator.
   715→            vec![]
   716→        }
   717→        PStmt::Cast(cast_stmt) => {
   718→            vec![ScgStatement::Computation(ComputationNode {
   719→                dst: "_cast".into(),
   720→                op: BinOpKind::Add,
   721→                lhs: bridge_expr_to_scg_expr(&cast_stmt.expr),
   722→                rhs: ScgExpr::Int(0),
   723→                tail_call: false,
   724→            })]
   725→        }
   726→        PStmt::Return(ret_stmt) => {
   727→            let values = match &ret_stmt.value {
   728→                Some(expr) => vec![bridge_expr_to_scg_expr(expr)],
   729→                None => vec![],
   730→            };
   731→            vec![ScgStatement::Return(values)]
   732→        }
   733→        PStmt::Expr(expr_stmt) => {
   734→            // Handle standalone function calls like free(ptr)
   735→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &expr_stmt.expr {
   736→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
   737→                    let scg_args: Vec<ScgExpr> = args.iter()
   738→                        .map(|a| bridge_expr_to_scg_expr(a))
   739→                        .collect();
   740→                    return vec![ScgStatement::Call(CallNode {
   741→                        dst: None,
   742→                        func: name.clone(),
   743→                        args: scg_args,
   744→                    })];
   745→                }
   746→            }
   747→            let (op, lhs, rhs) = bridge_expr_to_binop(&expr_stmt.expr, "_");
   748→            vec![ScgStatement::Computation(ComputationNode {
   749→                dst: "_".into(),
   750→                op,
   751→                lhs,
   752→                rhs,
   753→                tail_call: false,
   754→            })]
   755→        }
   756→        PStmt::If(if_stmt) => {
   757→            let cond = bridge_expr_to_scg_expr(&if_stmt.condition);
   758→            let then_body = bridge_block_to_scg_stmts(&if_stmt.then_block);
   759→            let else_body = if_stmt
   760→                .else_block
   761→                .as_ref()
   762→                .map(bridge_block_to_scg_stmts);
   763→            vec![ScgStatement::Control(ControlNode::If {
   764→                cond,
   765→                then_body,
   766→                else_body,
   767→            })]
   768→        }
   769→        PStmt::While(while_stmt) => {
   770→            let cond = bridge_expr_to_scg_expr(&while_stmt.condition);
   771→            let body = bridge_block_to_scg_stmts(&while_stmt.body);
   772→            vec![ScgStatement::Control(ControlNode::If {
   773→                cond,
   774→                then_body: body,
   775→                else_body: Some(vec![ScgStatement::Control(ControlNode::Break)]),
   776→            })]
   777→        }
   778→        PStmt::For(for_stmt) => {
   779→            let iter_init = ScgStatement::Computation(ComputationNode {
   780→                dst: for_stmt.name.clone(),
   781→                op: BinOpKind::Add,
   782→                lhs: ScgExpr::Int(0),
   783→                rhs: bridge_expr_to_scg_expr(&for_stmt.iter),
   784→                tail_call: false,
   785→            });
   786→            let body = bridge_block_to_scg_stmts(&for_stmt.body);
   787→            let mut loop_stmts = vec![iter_init];
   788→            loop_stmts.push(ScgStatement::Control(ControlNode::Loop { body }));
   789→            loop_stmts
   790→        }
   791→        PStmt::Loop(loop_stmt) => {
   792→            let body = bridge_block_to_scg_stmts(&loop_stmt.body);
   793→            vec![ScgStatement::Control(ControlNode::Loop { body })]
   794→        }
   795→        PStmt::Break(_) => vec![ScgStatement::Control(ControlNode::Break)],
   796→        PStmt::Continue(_) => vec![ScgStatement::Control(ControlNode::Continue)],
   797→        PStmt::BdDirective(_)
   798→        | PStmt::Sync(_)
   799→        | PStmt::UnsafeBlock { .. }
   800→        | PStmt::Match(_)
   801→        | PStmt::Access(_) => vec![],
   802→    }
   803→}

   805→/// Convert a parser expression into a codegen SCG expression.
   806→fn bridge_expr_to_scg_expr(expr: &vuma_parser::ast::Expr) -> ScgExpr {
   807→    use vuma_parser::ast::{Expr, Lit, UnOp};
   808→
   809→    match expr {
   810→        Expr::Var { name, .. } => ScgExpr::Var(name.clone()),
   811→        Expr::Lit { value, .. } => match value {
   812→            Lit::Int(n) => ScgExpr::Int(*n),
   813→            Lit::Float(f) => ScgExpr::Float(*f),
   814→            Lit::Bool(b) => ScgExpr::Int(if *b { 1 } else { 0 }),
   815→            _ => ScgExpr::Int(0),
   816→        },
   817→        Expr::BinOp { op, lhs, rhs: _, .. } => {
   818→            let _ = op;
   819→            bridge_expr_to_scg_expr(lhs)
   820→        }
   821→        Expr::UnOp { op, expr: operand, .. } => {
   822→            match op {
   823→                UnOp::BitNot => ScgExpr::Int(-1),
   824→                UnOp::Neg => ScgExpr::Int(0),
   825→                _ => bridge_expr_to_scg_expr(operand),
   826→            }
   827→        }
   828→        Expr::Call { callee, args, .. } => {
   829→            let _ = args;
   830→            match callee.as_ref() {
   831→                Expr::Var { name, .. } => ScgExpr::Var(format!("_call_{}", name)),
   832→                _ => ScgExpr::Int(0),
   833→            }
   834→        }
   835→        Expr::Deref { expr, .. } => bridge_expr_to_scg_expr(expr),
   836→        Expr::Cast { expr, .. } => bridge_expr_to_scg_expr(expr),
   837→        Expr::Index { expr, index, .. } => {
   838→            let _ = index;
   839→            bridge_expr_to_scg_expr(expr)
   840→        }
   841→        Expr::Range { start, end, .. } => {
   842→            let _ = end;
   843→            bridge_expr_to_scg_expr(start)
   844→        }
   845→        _ => ScgExpr::Int(0),
   846→    }
   847→}

   849→/// Convert a parser expression into a (BinOpKind, lhs, rhs) triple for use
   850→/// in a ComputationNode.
   851→fn bridge_expr_to_binop(
   852→    expr: &vuma_parser::ast::Expr,
   853→    _dst: &str,
   854→) -> (BinOpKind, ScgExpr, ScgExpr) {
   855→    use vuma_parser::ast::{BinOp, Expr, Lit, UnOp};
   856→
   857→    match expr {
   858→        Expr::Lit {
   859→            value: Lit::Int(n), ..
   860→        } => (
   861→            BinOpKind::Add,
   862→            ScgExpr::Int(0),
   863→            ScgExpr::Int(*n),
   864→        ),
   865→        Expr::Var { name, .. } => (
   866→            BinOpKind::Add,
   867→            ScgExpr::Var(name.clone()),
   868→            ScgExpr::Int(0),
   869→        ),
   870→        Expr::BinOp { op, lhs, rhs, .. } => {
   871→            let cg_op = match op {
   872→                BinOp::Add => BinOpKind::Add,
   873→                BinOp::Sub => BinOpKind::Sub,
   874→                BinOp::Mul => BinOpKind::Mul,
   875→                BinOp::Div => BinOpKind::SDiv,
   876→                BinOp::Mod => BinOpKind::SRem,
   877→                BinOp::And => BinOpKind::And,
   878→                BinOp::Or => BinOpKind::Or,
   879→                BinOp::BitAnd => BinOpKind::And,
   880→                BinOp::BitOr => BinOpKind::Or,
   881→                BinOp::BitXor => BinOpKind::Xor,
   882→                BinOp::Shl => BinOpKind::Shl,
   883→                BinOp::Shr => BinOpKind::ShrL,
   884→                BinOp::Eq => BinOpKind::Eq,
   885→                BinOp::Ne => BinOpKind::Ne,
   886→                BinOp::Lt => BinOpKind::SLt,
   887→                BinOp::Le => BinOpKind::SLe,
   888→                BinOp::Gt => BinOpKind::SGt,
   889→                BinOp::Ge => BinOpKind::SGe,
   890→            };
   891→            (
   892→                cg_op,
   893→                bridge_expr_to_scg_expr(lhs),
   894→                bridge_expr_to_scg_expr(rhs),
   895→            )
   896→        }
   897→        Expr::UnOp { op, expr: operand, .. } => {
   898→            match op {
   899→                UnOp::BitNot => (
   900→                    BinOpKind::Xor,
   901→                    bridge_expr_to_scg_expr(operand),
   902→                    ScgExpr::Int(-1),
   903→                ),
   904→                UnOp::Neg => (
   905→                    BinOpKind::Sub,
   906→                    ScgExpr::Int(0),
   907→                    bridge_expr_to_scg_expr(operand),
   908→                ),
   909→                _ => (
   910→                    BinOpKind::Add,
   911→                    bridge_expr_to_scg_expr(operand),
   912→                    ScgExpr::Int(0),
   913→                ),
   914→            }
   915→        }
   916→        Expr::Call { callee, args, .. } => {
   917→            // For function calls in expression position, emit a CallNode instead.
   918→            // But this function returns (BinOpKind, lhs, rhs), so we approximate.
   919→            let func_name = match callee.as_ref() {
   920→                Expr::Var { name, .. } => name.clone(),
   921→                _ => "_unknown".into(),
   922→            };
   923→            let lhs = args.first()
   924→                .map(|a| bridge_expr_to_scg_expr(a))
   925→                .unwrap_or(ScgExpr::Int(0));
   926→            (
   927→                BinOpKind::Add,
   928→                ScgExpr::Var(format!("_call_{}", func_name)),
   929→                lhs,
   930→            )
   931→        }
   932→        Expr::Deref { expr, .. } => (
   933→            BinOpKind::Add,
   934→            bridge_expr_to_scg_expr(expr),
   935→            ScgExpr::Int(0),
   936→        ),
   937→        Expr::Cast { expr, .. } => (
   938→            BinOpKind::Add,
   939→            bridge_expr_to_scg_expr(expr),
   940→            ScgExpr::Int(0),
   941→        ),
   942→        _ => {
   943→            (
   944→                BinOpKind::Add,
   945→                ScgExpr::Int(0),
   946→                ScgExpr::Int(0),
   947→            )
   948→        }
   949→    }
   950→}

   952→/// `vuma disasm <file>` — Read binary and disassemble.
   953→fn cmd_disasm(file: &PathBuf, isa: IsaArg, base_addr_str: &str) -> Result<(), String> {
   954→    let bytes = fs::read(file)
   955→        .map_err(|e| format!("error: cannot read binary file '{}': {}", file.display(), e))?;
   956→
   957→    let base_addr = u64::from_str_radix(base_addr_str.trim_start_matches("0x"), 16)
   958→        .map_err(|e| format!("error: invalid base address '{}': {}", base_addr_str, e))?;
   959→
   960→    let backend_kind = BackendKind::from(isa);
   961→    let backend = create_backend(backend_kind).map_err(|e| {
   962→        format!(
   963→            "error: cannot create {} backend: {}",
   964→            backend_kind.isa_name(),
   965→            e
   966→        )
   967→    })?;
   968→
   969→    let instructions = backend.disassemble(&bytes, base_addr);
   970→
   971→    println!(
   972→        "Disassembly of {} ({} bytes, ISA: {}):",
   973→        file.display(),
   974→        bytes.len(),
   975→        backend.name()
   976→    );
   977→    for line in &instructions {
   978→        println!("{}", line);
   979→    }
   980→
   981→    Ok(())
   982→}

   984→/// `vuma verify <file>` — Run IVE 5-invariant verification.
   985→fn cmd_verify(cli: &Cli, file: &PathBuf) -> Result<(), String> {
   986→    let source = read_source(file)?;
   987→    let mut config = make_config(cli, CompileTarget::Pi5Linux);
   988→    // Force exhaustive verification for the verify subcommand.
   989→    config.verification_level = VerificationLevel::Exhaustive;
   990→
   991→    let result = compile(&source, &config).map_err(|errors| {
   992→        print_errors(&errors);
   993→        format!(
   994→            "compilation/verification failed with {} error(s)",
   995→            errors.len()
   996→        )
   997→    })?;
   998→
   999→    match result.verification {
  1000→        Some(ref verification) => {
  1001→            println!("IVE Verification Results for {}", file.display());
  1002→            println!("  Overall verdict: {}", verification.overall);
  1003→            println!(
  1004→                "  Summary: {} invariant(s) checked",
  1005→                verification.summary.total_checked
  1006→            );
  1007→            println!("           {} passed", verification.summary.passed);
  1008→            println!("           {} failed", verification.summary.failed);
  1009→            println!("           {} unverified", verification.summary.unverified);
  1010→
  1011→            // Print individual invariant results.
  1012→            for per_inv in &verification.per_invariant {
  1013→                let status_str = match &per_inv.result.status {
  1014→                    vuma_ive::VerificationStatus::Proven => "PROVEN",
  1015→                    vuma_ive::VerificationStatus::ProbablySafe { .. } => "PROBABLY_SAFE",
  1016→                    vuma_ive::VerificationStatus::Unverified { .. } => "UNVERIFIED",
  1017→                    vuma_ive::VerificationStatus::Violated { .. } => "VIOLATED",
  1018→                };
  1019→                println!(
  1020→                    "  [{}] {:?} — {}",
  1021→                    status_str, per_inv.kind, per_inv.result.message,
  1022→                );
  1023→                if let Some(ref evidence) = per_inv.result.evidence {
  1024→                    println!("    evidence: {}", evidence);
  1025→                }
  1026→            }
  1027→        }
  1028→        None => {
  1029→            println!("Verification was skipped (verification level set to None).");
  1030→        }
  1031→    }
  1032→
  1033→    Ok(())
  1034→}

  1036→/// `vuma repl` — Interactive REPL: parse expressions and print AST.
  1037→fn cmd_repl() -> Result<(), String> {
  1038→    println!("VUMA REPL v0.1.0");
  1039→    println!("Type VUMA expressions or statements. Enter ':quit' to exit, ':help' for help.");
  1040→
  1041→    let stdin = io::stdin();
  1042→    let mut input = String::new();
  1043→
  1044→    loop {
  1045→        print!("vuma> ");
  1046→        io::stdout()
  1047→            .flush()
  1048→            .map_err(|e| format!("flush error: {}", e))?;
  1049→
  1050→        input.clear();
  1051→        match stdin.read_line(&mut input) {
  1052→            Ok(0) => break, // EOF
  1053→            Ok(_) => {}
  1054→            Err(e) => return Err(format!("error reading input: {}", e)),
  1055→        }
  1056→
  1057→        let trimmed = input.trim();
  1058→        if trimmed.is_empty() {
  1059→            continue;
  1060→        }
  1061→        if trimmed == ":quit" || trimmed == ":q" {
  1062→            break;
  1063→        }
  1064→        if trimmed == ":help" || trimmed == ":h" {
  1065→            println!("  :quit, :q   — Exit the REPL");
  1066→            println!("  :help, :h   — Show this help");
  1067→            println!("  <expr>      — Parse and display the AST");
  1068→            println!("  <stmt>      — Parse and display the AST");
  1069→            println!("  <program>   — Parse and display the full program AST");
  1070→            continue;
  1071→        }
  1072→
  1073→        // Try to parse as a full program first.
  1074→        let mut parser = vuma_parser::Parser::new(trimmed);
  1075→        let result = parser.parse_program();
  1076→        if !result.has_errors() {
  1077→            // Successfully parsed; display the AST.
  1078→            let program = result.unwrap();
  1079→            println!("{:#?}", program);
  1080→            continue;
  1081→        }
  1082→        // Program parse had errors; try to parse as an expression via
  1083→        // wrapping it in a dummy program context.
  1084→        let program_errors = result.errors.clone();
  1085→        // Since parse_expr is private, we re-parse as a minimal program.
  1086→        let wrapped = format!("fn _repl_expr() {{ {} }}", trimmed);
  1087→        let mut expr_parser = vuma_parser::Parser::new(&wrapped);
  1088→        let expr_result = expr_parser.parse_program();
  1089→        if !expr_result.has_errors() {
  1090→            // Print the function body AST.
  1091→            let program = expr_result.unwrap();
  1092→            if let Some(item) = program.items.first() {
  1093→                println!("{:#?}", item);
  1094→            } else {
  1095→                println!("{:#?}", program);
  1096→            }
  1097→        } else {
  1098→            // Show the original parse errors.
  1099→            for err in &program_errors {
  1100→                eprintln!("parse error: {}", err);
  1101→            }
  1102→        }
  1103→    }
  1104→
  1105→    Ok(())
  1106→}

  1109→// Main entry point
  1110→// ═══════════════════════════════════════════════════════════════════════════
  1111→
  1112→fn main() {
  1113→    // Initialize logger.
  1114→    env_logger::init();
  1115→
  1116→    let cli = Cli::parse();
  1117→
  1118→    let result = match cli.command {
  1119→        Commands::Build {
  1120→            ref file,
  1121→            ref output,
  1122→            target,
  1123→        } => cmd_build(&cli, file, output, target),
  1124→        Commands::Run { ref file, ref args } => cmd_run(&cli, file, args),
  1125→        Commands::Check { ref file } => cmd_check(&cli, file),
  1126→        Commands::Emit {
  1127→            isa,
  1128→            ref file,
  1129→            ref output,
  1130→        } => cmd_emit(&cli, isa, file, output),
  1131→        Commands::Disasm {
  1132→            ref file,
  1133→            isa,
  1134→            ref base_addr,
  1135→        } => cmd_disasm(file, isa, base_addr),
  1136→        Commands::Verify { ref file } => cmd_verify(&cli, file),
  1137→        Commands::Repl => cmd_repl(),
  1138→    };
  1139→
  1140→    if let Err(err) = result {
  1141→        eprintln!("{}", err);
  1142→        std::process::exit(1);
  1143→    }
  1144→}
