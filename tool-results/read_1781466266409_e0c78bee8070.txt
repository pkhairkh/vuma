     1→//! Raspberry Pi 5 Hardware Tests (Mock)
     2→//!
     3→//! End-to-end tests for Pi 5–targeted hardware abstractions: GPIO, UART,
     4→//! system timer, and SMP (multi-core).  Because the test suite runs on the
     5→//! host (x86_64 or aarch64), all hardware interactions are exercised through
     6→//! the VUMA compilation pipeline and the codegen emitter, producing ARM64
     7→//! machine code that would perform the described operations on actual Pi 5
     8→//! hardware.
     9→//!
    10→//! # Test Matrix
    11→//!
    12→//! | # | Test                              | Hardware Subsystem | Strategy                        |
    13→//! |---|-----------------------------------|--------------------|---------------------------------|
    14→//! | 1 | GPIO pin set output               | GPIO               | Parse → SCG → verify → emit     |
    15→//! | 2 | GPIO pin read input               | GPIO               | Parse → SCG → verify → emit     |
    16→//! | 3 | UART transmit byte                | UART               | Parse → SCG → verify → emit     |
    17→//! | 4 | UART receive byte with polling    | UART               | Parse → SCG → verify → emit     |
    18→//! | 5 | System timer delay                | Timer              | Parse → SCG → verify → emit     |
    19→//! | 6 | System timer read counter         | Timer              | Parse → SCG → verify → emit     |
    20→//! | 7 | SMP core bootstrap                | SMP                | Parse → SCG → verify → emit     |
    21→//! | 8 | SMP inter-core mailbox            | SMP                | Parse → SCG → verify → emit     |
    22→//! | 9 | GPIO+UART combined workflow       | GPIO + UART        | Parse → SCG → verify → emit     |
    23→//! | 10| MMIO write followed by barrier    | System             | Parse → SCG → codegen           |

   349→// Helper: compile a codegen Scg to ARM64 via the full IR → emit pipeline

   352→/// Compile a codegen `Scg` through the IR builder and ARM64 emitter,
   353→/// returning the IR program and the raw emitted code words.
   354→fn compile_scg_to_arm64(scg: &Scg) -> (IRProgram, Vec<u32>) {
   355→    let mut builder = IRBuilder::new();
   356→    let ir_program = builder.build(scg).expect("IRBuilder should succeed");
   357→    let mut emitter = Emitter::new();
   358→    let code_words = emitter
   359→        .emit_function(&ir_program.functions[0])
   360→        .expect("Emission should succeed");
   361→    (ir_program, code_words)
   362→}

   368→/// Test: GPIO pin set output through the full pipeline.
   369→///
   370→/// Parses a VUMA source program that models writing to a GPIO register,
   371→/// builds the SCG, verifies IVE invariants, then builds a codegen SCG
   372→/// for the GPIO function and emits ARM64 machine code.
   373→#[test]
   374→fn test_gpio_set_output_pipeline() {
   375→    // Phase 1: Parse → SCG → Verify
   376→    let scg = build_scg_from_source(gpio_set_output_source())
   377→        .expect("GPIO set output source should parse");
   378→    assert!(scg.node_count() > 0, "SCG should have nodes");
   379→    let validation = scg.validate();
   380→    assert!(
   381→        validation.is_valid,
   382→        "SCG should validate: {:?}",
   383→        validation.errors
   384→    );
   385→
   386→    // Phase 2: IVE verification (no violations expected)
   387→    let result = verify_program(gpio_set_output_source());
   388→    let violations: Vec<_> = result
   389→        .per_invariant
   390→        .iter()
   391→        .filter(|pir| pir.is_fail())
   392→        .collect();
   393→    assert!(
   394→        violations.is_empty(),
   395→        "GPIO set output should have no violations"
   396→    );
   397→
   398→    // Phase 3: Codegen — build GPIO function SCG → IR → ARM64
   399→    let cg_scg = build_gpio_set_output_scg();
   400→    let (ir_program, code_words) = compile_scg_to_arm64(&cg_scg);
   401→    assert!(
   402→        !ir_program.functions.is_empty(),
   403→        "IR should contain functions"
   404→    );
   405→    assert!(!code_words.is_empty(), "ARM64 code should be emitted");
   406→
   407→    // Verify the emitted code contains a prologue (STP X29, X30)
   408→    let first_word = code_words[0];
   409→    // STP encoding: top bits should indicate STP instruction
   410→    assert_ne!(first_word, 0, "First instruction should not be zero");
   411→}
