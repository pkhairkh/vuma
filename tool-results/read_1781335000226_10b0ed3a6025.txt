     1→//! # ARM64 Code Emission
     2→//!
     3→//! Lowers IR to ARM64 machine code and produces ELF binaries or raw binaries
     4→//! suitable for the Raspberry Pi 5 (Cortex-A76, ARMv8.2-A).
     5→//!
     6→//! ## Pipeline
     7→//!
     8→//! 1. **IR → ARM64 Instructions**: Each IR instruction is pattern-matched and
     9→//!    lowered to one or more [`Instruction`]
    10→//!    values, with virtual registers replaced by physical registers from the
    11→//!    register allocator.
    12→//! 2. **Emit function**: Each IR function is emitted as a sequence of 32-bit
    13→//!    ARM64 code words.
    14→//! 3. **Emit program**: All functions are collected, data sections are laid
    15→//!    out, and a minimal ELF64 binary is produced.
    16→//!
    17→//! ## Output Formats
    18→//!
    19→//! - **ELF**: Full ELF64 binary with program headers, section headers, symbol
    20→//!   table, and string table.  Suitable for Linux on Pi 5.
    21→//! - **Raw**: Flat binary image for bare-metal Pi 5 (loaded at 0x80000).
    22→//! - **Obj**: Relocatable ELF object file (`ET_REL`) for linking.
    23→//!
    24→//! ## ELF Layout (executable)
    25→//!
    26→//! ```text
    27→//! ┌─────────────────────┐
    28→//! │ ELF Header           │  64 bytes
    29→//! ├─────────────────────┤
    30→//! │ Program Headers      │  2 × 56 bytes (LOAD segments)
    31→//! ├─────────────────────┤
    32→//! │ .text                │  emitted code
    33→//! ├─────────────────────┤
    34→//! │ .rodata              │  read-only data
    35→//! ├─────────────────────┤
    36→//! │ .data                │  initialized data
    37→//! ├─────────────────────┤
    38→//! │ .symtab              │  symbol table entries
    39→//! ├─────────────────────┤
    40→//! │ .strtab              │  symbol string table
    41→//! ├─────────────────────┤
    42→//! │ .shstrtab            │  section header string table
    43→//! ├─────────────────────┤
    44→//! │ Section Headers      │  N × 64 bytes
    45→//! └─────────────────────┘
    46→//! .bss is virtual-only (memsz > filesz in the data LOAD segment).
    47→//! ```
    48→//!
    49→//! ## Relocation Support
    50→//!
    51→//! Inter-function calls (`BL`) are resolved in a fixup pass after all
    52→//! functions have been emitted.  The emitter records the word offset of each
    53→//! `BL` instruction and the target function name; once function addresses
    54→//! are known, the branch offsets are patched into the encoded instructions.
    55→

   487→// ---------------------------------------------------------------------------
   488→// Inter-function call relocation record
   489→// ---------------------------------------------------------------------------
   490→
   491→/// A record for a `BL` instruction that needs to be patched with the address
   492→/// of a named function.  After all functions are emitted, these records are
   493→/// resolved to compute the correct branch offset.
   494→#[derive(Debug, Clone)]
   495→struct CallRelocation {
   496→    /// Byte offset within the text section where the BL instruction lives.
   497→    text_byte_offset: u64,
   498→    /// Name of the target function.
   499→    target_func: String,
   500→}
   501→
   502→// ---------------------------------------------------------------------------
   503→// ELF RelaEntry
   504→// ---------------------------------------------------------------------------
   505→
   506→/// An ELF64 relocation entry with an explicit addend (SHT_RELA).
   507→///
   508→/// Each entry specifies where in the section a relocation must be applied,
   509→/// which symbol the relocation references, the type of relocation, and an
   510→/// addend value used in the relocation computation.
   511→#[derive(Debug, Clone)]
   512→pub struct RelaEntry {
   513→    /// Byte offset within the section where the relocation applies.
   514→    pub offset: u64,
   515→    /// Symbol index (upper 32 bits) and relocation type (lower 32 bits),
   516→    /// packed as: `(sym_idx << 32) | r_type`.
   517→    pub info: u64,
   518→    /// Addend value used in the relocation computation.
   519→    pub addend: i64,
   520→}
   521→
   522→impl RelaEntry {
   523→    /// Create a new relocation entry.
   524→    pub fn new(offset: u64, sym_idx: u32, r_type: u32, addend: i64) -> Self {
   525→        Self {
   526→            offset,
   527→            info: ((sym_idx as u64) << 32) | (r_type as u64),
   528→            addend,
   529→        }
   530→    }
   531→
   532→    /// Extract the symbol index from `info`.
   533→    pub fn sym_idx(&self) -> u32 {
   534→        (self.info >> 32) as u32
   535→    }
   536→
   537→    /// Extract the relocation type from `info`.
   538→    pub fn r_type(&self) -> u32 {
   539→        (self.info & 0xFFFFFFFF) as u32
   540→    }
   541→
   542→    /// Serialize the entry to 24 bytes (little-endian).
   543→    pub fn to_bytes(&self) -> [u8; 24] {
   544→        let mut buf = [0u8; 24];
   545→        buf[0..8].copy_from_slice(&self.offset.to_le_bytes());
   546→        buf[8..16].copy_from_slice(&self.info.to_le_bytes());
   547→        buf[16..24].copy_from_slice(&self.addend.to_le_bytes());
   548→        buf
   549→    }
   550→}

   556→/// The ARM64 code emitter.
   557→///
   558→/// Holds state for the current emission context: the register allocator,
   559→/// accumulated code, and fixup records for branch targets.
   560→pub struct Emitter {
   561→    /// Register allocator used during emission.
   562→    reg_alloc: RegAllocator,
   563→    /// Accumulated machine code for the current function (32-bit words).
   564→    code: Vec<u32>,
   565→    /// Fixup records for intra-function branches: (word index, target label name, format).
   566→    fixups: Vec<(usize, String, BranchFormat)>,
   567→    /// Map from label name to code offset (in words) within the current function.
   568→    label_offsets: HashMap<String, usize>,
   569→    /// Inter-function call relocations for the current function.
   570→    call_relocs: Vec<CallRelocation>,
   571→    /// Relocation entries using the `RelocationEntry` infrastructure.
   572→    ///
   573→    /// Each entry records a byte offset within the function's encoded output
   574→    /// where a symbolic reference must be resolved, the target symbol name,
   575→    /// and the ISA-specific relocation type (e.g., `"R_AARCH64_CALL26"`).
   576→    relocations: Vec<RelocationEntry>,
   577→    /// Name of the function currently being emitted.
   578→    current_func_name: String,
   579→    /// Byte offset of the current function within the text section (set externally).
   580→    func_text_offset: u64,
   581→    /// Computed stack frame size (in bytes) for the current function.
   582→    frame_size: u32,
   583→    /// Registers pinned for the current instruction (auto-unpinned after each instruction).
   584→    instr_pinned_regs: Vec<Register>,
   585→}
   586→
   587→impl Emitter {
   588→    /// Create a new emitter with a fresh register allocator.
   589→    pub fn new() -> Self {
   590→        Self {
   591→            reg_alloc: RegAllocator::new(),
   592→            code: Vec::new(),
   593→            fixups: Vec::new(),
   594→            label_offsets: HashMap::new(),
   595→            call_relocs: Vec::new(),
   596→            relocations: Vec::new(),
   597→            current_func_name: String::new(),
   598→            func_text_offset: 0,
   599→            frame_size: 0,
   600→            instr_pinned_regs: Vec::new(),
   601→        }
   602→    }
   603→
   604→    // -----------------------------------------------------------------------
   605→    // Function emission
   606→    // -----------------------------------------------------------------------
   607→
   608→    /// Emit a single IR function to ARM64 machine code.
   609→    ///
   610→    /// For functions with more than `STACK_SLOT_VREG_THRESHOLD` virtual
   611→    /// registers, the stack-slot emitter is used (every vreg gets a stack
   612→    /// slot, scratch registers only).  For simpler functions, the greedy
   613→    /// register allocator is used.
   614→    ///
   615→    /// Returns a vector of 32-bit ARM64 instruction words.
   616→    pub fn emit_function(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   617→        let vreg_count = count_vregs(func);
   618→        if vreg_count > STACK_SLOT_VREG_THRESHOLD {
   619→            self.emit_function_stack_slot(func)
   620→        } else {
   621→            self.emit_function_greedy(func)
   622→        }
   623→    }
   624→
   625→    /// Emit a single IR function using the greedy register allocator.
   626→    ///
   627→    /// This is the original emission strategy — suitable for functions with
   628→    /// a small number of virtual registers.
   629→    fn emit_function_greedy(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   630→        self.code.clear();
   631→        self.fixups.clear();
   632→        self.label_offsets.clear();
   633→        self.call_relocs.clear();
   634→        self.relocations.clear();
   635→        self.current_func_name = func.name.clone();
   636→        self.reg_alloc.reset();
   637→
   638→        // Pre-allocate registers for parameters (AAPCS64: X0–X7).
   639→        // We must explicitly assign parameter virtual registers to the
   640→        // correct argument registers so the calling convention is respected.
   641→        let arg_regs = [
   642→            Register::X0, Register::X1, Register::X2, Register::X3,
   643→            Register::X4, Register::X5, Register::X6, Register::X7,
   644→        ];
   645→        for (i, param) in func.params.iter().enumerate() {
   646→            if let IRValue::Register(vreg_id) = param {
   647→                if i < 8 {
   648→                    self.reg_alloc.preassign(*vreg_id, arg_regs[i]);
   649→                }
   650→            }
   651→        }
   652→
   653→        // Emit prologue: SUB SP, SP, #frame_size+16 (space for saved FP/LR + locals)
   654→        // Then STP X29, X30, [SP, #frame_size] (save FP/LR at top of allocated frame)
   655→        let aligned_stack = compute_frame_size(func);
   656→        self.frame_size = aligned_stack;
   657→        let total_alloc = aligned_stack + 16; // 16 bytes for FP+LR pair
   658→        self.emit_instruction(Instruction::SUB {
   659→            rd: Register::SP,
   660→            rn: Register::SP,
   661→            rm: Operand::Imm12(total_alloc),
   662→        })?;
   663→        self.emit_instruction(Instruction::STP {
   664→            rt1: Register::X29,
   665→            rt2: Register::X30,
   666→            rn: Register::SP,
   667→            offset: aligned_stack as i32, // save FP/LR at top of frame
   668→        })?;
   669→
   670→        // Emit each basic block.
   671→        for block in &func.blocks {
   672→            self.label_offsets
   673→                .insert(block.label.clone(), self.code.len());
   674→            self.emit_block(block)?;
   675→        }
   676→
   677→        // Apply fixups — resolve intra-function branch targets.
   678→        self.apply_fixups()?;
   679→
   680→        Ok(self.code.clone())
   681→    }
   682→
   683→    /// Returns the relocation entries recorded during emission of the current
   684→    /// function.
   685→    ///
   686→    /// Each `RelocationEntry` records a byte offset within the function's
   687→    /// encoded output where a symbolic reference must be resolved, the target
   688→    /// symbol name, and the ISA-specific relocation type (e.g.,
   689→    /// `"R_AARCH64_CALL26"`).
   690→    pub fn relocations(&self) -> &[RelocationEntry] {
   691→        &self.relocations
   692→    }
   693→
   694→    /// Emit instructions for a single IR basic block.
   695→    fn emit_block(&mut self, block: &IRBlock) -> Result<()> {
   696→        for instr in &block.instructions {
   697→            self.emit_ir_instr(instr)?;
   698→            // Auto-unpin all registers that were pinned during this instruction.
   699→            for reg in self.instr_pinned_regs.drain(..) {
   700→                self.reg_alloc.unpin(reg);
   701→            }
   702→        }
   703→        self.emit_terminator(&block.terminator)?;
   704→        Ok(())
   705→    }
   706→
   707→    // -----------------------------------------------------------------------
   708→    // IR → ARM64 instruction lowering
   709→    // -----------------------------------------------------------------------
   710→
   711→    /// Lower a single IR instruction to ARM64 instructions.
   712→    fn emit_ir_instr(&mut self, instr: &IRInstr) -> Result<()> {
   713→        match instr {
   714→            IRInstr::Load { dst, addr, offset, ty } => {
   715→                let rn = self.resolve_reg(addr)?;
   716→                let rt = self.resolve_reg(dst)?;
   717→                self.emit_load(rt, rn, *offset, ty)?;
   718→            }
   719→
   720→            IRInstr::Store { value, addr, offset, ty } => {
   721→                let rt = self.resolve_reg(value)?;
   722→                let rn = self.resolve_reg(addr)?;
   723→                if rt == rn {
   724→                    // Both resolved to the same register — use scratch for value
   725→                    self.emit_instruction(Instruction::MOV {
   726→                        rd: Register::X16,
   727→                        rm: rt,
   728→                    })?;
   729→                    self.emit_store(Register::X16, rn, *offset, ty)?;
   730→                } else {
   731→                    self.emit_store(rt, rn, *offset, ty)?;
   732→                }
   733→            }
   734→
   735→            IRInstr::BinOp { op, dst, lhs, rhs, ty } => {
   736→                self.emit_binop(*op, dst, lhs, rhs, ty.as_ref())?;
   737→            }
   738→
   739→            IRInstr::UnaryOp { op, dst, operand, ty: _ } => {
   740→                let rd = self.resolve_reg(dst)?;
   741→                let rn = self.resolve_reg(operand)?;
   742→                match op {
   743→                    UnaryOpKind::Neg => {
   744→                        self.emit_instruction(Instruction::SUB {
   745→                            rd,
   746→                            rn: Register::XZR,
   747→                            rm: Operand::Reg {
   748→                                reg: rn,
   749→                                shift: None,
   750→                            },
   751→                        })?;
   752→                    }
   753→                    UnaryOpKind::Not => {
   754→                        self.emit_load_immediate(Register::X9, -1)?;
   755→                        self.emit_instruction(Instruction::EOR {
   756→                            rd,
   757→                            rn,
   758→                            rm: Register::X9,
   759→                        })?;
   760→                    }
   761→                    UnaryOpKind::Clz => {
   762→                        // CLZ Xd, Xn — count leading zeros (native ARM64 instruction)
   763→                        self.emit_instruction(Instruction::CLZ { rd, rn })?;
   764→                    }
   765→                    UnaryOpKind::Ctz => {
   766→                        // CTZ = RBIT + CLZ: reverse bits then count leading zeros.
   767→                        // Use X9 as scratch if rd == rn (need intermediate result).
   768→                        if rd == rn {
   769→                            self.emit_instruction(Instruction::RBIT {
   770→                                rd: Register::X9,
   771→                                rn,
   772→                            })?;
   773→                            self.emit_instruction(Instruction::CLZ {
   774→                                rd,
   775→                                rn: Register::X9,
   776→                            })?;
   777→                        } else {
   778→                            self.emit_instruction(Instruction::RBIT { rd, rn })?;
   779→                            self.emit_instruction(Instruction::CLZ { rd, rn: rd })?;
   780→                        }
   781→                    }
   782→                    UnaryOpKind::Popcnt => {
   783→                        // POPCNT via FMOV+CNT+ADDV+UMOV sequence:
   784→                        // FMOV D8, Xn        — move GPR value to SIMD register (V8 is caller-saved)
   785→                        // CNT V8.8B, V8.8B   — count bits per byte
   786→                        // ADDV B8, V8.8B     — horizontal sum of byte counts
   787→                        // UMOV Xd, V8.B[0]   — move result back to GPR (zero-extends to 64 bits)
   788→                        const SIMD_SCRATCH: u8 = 8; // V8 is caller-saved in AAPCS64
   789→                        self.emit_instruction(Instruction::FMOV_DX {
   790→                            vd: SIMD_SCRATCH,
   791→                            rn,
   792→                        })?;
   793→                        self.emit_instruction(Instruction::CNT {
   794→                            vd: SIMD_SCRATCH,
   795→                            vn: SIMD_SCRATCH,
   796→                        })?;
   797→                        self.emit_instruction(Instruction::ADDV {
   798→                            vd: SIMD_SCRATCH,
   799→                            vn: SIMD_SCRATCH,
   800→                        })?;
   801→                        self.emit_instruction(Instruction::UMOV {
   802→                            rd,
   803→                            vn: SIMD_SCRATCH,
   804→                        })?;
   805→                    }
   806→                }
   807→            }
   808→
   809→            IRInstr::Call {
   810→                dst,
   811→                func: target_name,
   812→                args,
   813→            } => {
   814→                // Resolve all argument source registers FIRST, before moving
   815→                // any of them. This prevents a later move from overwriting a
   816→                // register that an earlier argument is in.
   817→                //
   818→                // IMPORTANT: For immediate arguments, resolve_reg always loads
   819→                // into X9. When multiple immediates are present, we must use
   820→                // different scratch registers to avoid overwriting. We load
   821→                // immediates into X0-X7 directly (since they're the target
   822→                // registers anyway) or use X11-X15 as temporary scratch.
   823→                let arg_regs = [
   824→                    Register::X0, Register::X1, Register::X2, Register::X3,
   825→                    Register::X4, Register::X5, Register::X6, Register::X7,
   826→                ];
   827→                // Scratch registers for loading immediates when the target arg
   828→                // register isn't available yet (because of conflicts).
   829→                let scratch_regs = [
   830→                    Register::X11, Register::X12, Register::X13, Register::X14,
   831→                    Register::X15, Register::X3, Register::X4, Register::X5,
   832→                ];
   833→
   834→                let mut src_regs: Vec<Register> = Vec::new();
   835→                let mut immediate_scratch_used: Vec<usize> = Vec::new(); // which scratch reg each immediate uses
   836→                let mut next_scratch = 0;
   837→
   838→                for (i, arg) in args.iter().enumerate() {
   839→                    if i >= 8 {
   840→                        break;
   841→                    }
   842→                    let is_immediate = matches!(arg, IRValue::Immediate(_) | IRValue::Address(_));
   843→                    if is_immediate {
   844→                        let v = match arg {
   845→                            IRValue::Immediate(v) => *v,
   846→                            IRValue::Address(a) => *a as i64,
   847→                            _ => unreachable!(),
   848→                        };
   849→                        // Load the immediate directly into the target register
   850→                        // if it's not needed by another argument.
   851→                        // For simplicity, always load into a scratch register
   852→                        // and record it.
   853→                        if next_scratch < scratch_regs.len() {
   854→                            let scratch = scratch_regs[next_scratch];
   855→                            next_scratch += 1;
   856→                            self.emit_load_immediate(scratch, v)?;
   857→                            src_regs.push(scratch);
   858→                        } else {
   859→                            // Fallback to X9 (rare, only with 8+ immediates)
   860→                            let scratch = Register::X9;
   861→                            self.emit_load_immediate(scratch, v)?;
   862→                            src_regs.push(scratch);
   863→                        }
   864→                    } else {
   865→                        let src = self.resolve_reg(arg)?;
   866→                        src_regs.push(src);
   867→                    }
   868→                }
   869→
   870→                // Now move arguments into X0–X7. We need to handle the case
   871→                // where a source register is also a target register (cycle).
   872→                // Strategy: first move arguments that don't conflict, then
   873→                // handle conflicts using a scratch register.
   874→                let n = src_regs.len();
   875→                let mut moved = vec![false; n];
   876→
   877→                // Pass 1: move args where src != any dst that hasn't been moved yet
   878→                // Simple approach: iterate and move non-conflicting ones first
   879→                for pass in 0..2 {
   880→                    for i in 0..n {
   881→                        if moved[i] {
   882→                            continue;
   883→                        }
   884→                        let src = src_regs[i];
   885→                        let dst = arg_regs[i];
   886→                        if src == dst {
   887→                            // Already in the right register
   888→                            moved[i] = true;
   889→                            continue;
   890→                        }
   891→                        if pass == 0 {
   892→                            // Check if src is needed as a source by a later unmoved arg
   893→                            let mut conflict = false;
   894→                            for j in 0..n {
   895→                                if !moved[j] && j != i && src_regs[j] == dst {
   896→                                    // Moving arg i to dst would overwrite arg j's source
   897→                                    conflict = true;
   898→                                    break;
   899→                                }
   900→                            }
   901→                            if conflict {
   902→                                continue; // Defer to pass 2
   903→                            }
   904→                        }
   905→                        // Safe to move now
   906→                        self.emit_instruction(Instruction::MOV {
   907→                            rd: dst,
   908→                            rm: src,
   909→                        })?;
   910→                        moved[i] = true;
   911→                    }
   912→                }
   913→
   914→                // Pass 2: handle any remaining conflicts with scratch register
   915→                for i in 0..n {
   916→                    if moved[i] {
   917→                        continue;
   918→                    }
   919→                    let src = src_regs[i];
   920→                    let dst = arg_regs[i];
   921→                    if src == dst {
   922→                        moved[i] = true;
   923→                        continue;
   924→                    }
   925→                    // Use X16 as scratch to break the cycle
   926→                    self.emit_instruction(Instruction::MOV {
   927→                        rd: Register::X16,
   928→                        rm: src,
   929→                    })?;
   930→                    self.emit_instruction(Instruction::MOV {
   931→                        rd: dst,
   932→                        rm: Register::X16,
   933→                    })?;
   934→                    moved[i] = true;
   935→                }
   936→
   937→                // BL — record a relocation for later patching.
   938→                let bl_word_idx = self.code.len();
   939→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
   940→                self.call_relocs.push(CallRelocation {
   941→                    text_byte_offset: bl_byte_offset,
   942→                    target_func: target_name.clone(),
   943→                });
   944→                // Register a RelocationEntry so the linker can patch this BL.
   945→                self.relocations.push(RelocationEntry {
   946→                    offset: bl_byte_offset,
   947→                    symbol: target_name.clone(),
   948→                    reloc_type: "R_AARCH64_CALL26".to_string(),
   949→                });
   950→                self.emit_instruction(Instruction::BL { offset: 0 })?;
   951→
   952→                if let Some(dst_val) = dst {
   953→                    let rd = self.resolve_reg(dst_val)?;
   954→                    if rd != Register::X0 {
   955→                        self.emit_instruction(Instruction::MOV {
   956→                            rd,
   957→                            rm: Register::X0,
   958→                        })?;
   959→                    }
   960→                }
   961→            }
   962→
   963→            IRInstr::Alloc { dst, size } => {
   964→                let rd = self.resolve_reg(dst)?;
   965→                // IMPORTANT: Decrement SP FIRST, then save the new SP value.
   966→                // If we save before decrementing, the allocation pointer
   967→                // points to the wrong (too-high) address.
   968→                let aligned = (*size).div_ceil(16) * 16;
   969→                self.emit_instruction(Instruction::SUB {
   970→                    rd: Register::SP,
   971→                    rn: Register::SP,
   972→                    rm: Operand::Imm12(aligned as u16),
   973→                })?;
   974→                // MOV rd, SP: cannot use ORR because ORR treats Rm=31 as XZR.
   975→                // Use ADD rd, SP, #0 instead.
   976→                self.emit_instruction(Instruction::ADD {
   977→                    rd,
   978→                    rn: Register::SP,
   979→                    rm: Operand::Imm12(0),
   980→                })?;
   981→            }
   982→
   983→            IRInstr::Free { ptr } => {
   984→                let rt = self.resolve_reg(ptr)?;
   985→                // Move ptr to X0 (first argument)
   986→                if rt != Register::X0 {
   987→                    self.emit_instruction(Instruction::MOV {
   988→                        rd: Register::X0,
   989→                        rm: rt,
   990→                    })?;
   991→                }
   992→                // BL __vuma_free
   993→                let bl_word_idx = self.code.len();
   994→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
   995→                self.call_relocs.push(CallRelocation {
   996→                    text_byte_offset: bl_byte_offset,
   997→                    target_func: "__vuma_free".to_string(),
   998→                });
   999→                self.relocations.push(RelocationEntry {
  1000→                    offset: bl_byte_offset,
  1001→                    symbol: "__vuma_free".to_string(),
  1002→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1003→                });
  1004→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1005→            }
  1006→
  1007→            IRInstr::Cast { kind, dst, src } => {
  1008→                let rd = self.resolve_reg(dst)?;
  1009→                let rn = self.resolve_reg(src)?;
  1010→                match kind {
  1011→                    CastKind::ZExt => {
  1012→                        // Zero-extend 32-bit to 64-bit using UBFM
  1013→                        self.emit_instruction(Instruction::UBFM {
  1014→                            rd,
  1015→                            rn,
  1016→                            immr: 0,
  1017→                            imms: 31,
  1018→                        })?;
  1019→                    }
  1020→                    CastKind::SExt => {
  1021→                        // Sign-extend 32-bit to 64-bit using SBFM
  1022→                        self.emit_instruction(Instruction::SBFM {
  1023→                            rd,
  1024→                            rn,
  1025→                            immr: 0,
  1026→                            imms: 31,
  1027→                        })?;
  1028→                    }
  1029→                    CastKind::Trunc | CastKind::BitCast => {
  1030→                        // Trunc: upper bits discarded on write — just MOV
  1031→                        // BitCast: no data change — just MOV
  1032→                        if rd != rn {
  1033→                            self.emit_instruction(Instruction::MOV { rd, rm: rn })?;
  1034→                        }
  1035→                    }
  1036→                }
  1037→            }
  1038→
  1039→            IRInstr::Phi { .. } => {
  1040→                log::warn!(
  1041→                    "IRInstr::Phi encountered during emission — should be resolved by SSA pass"
  1042→                );
  1043→            }
  1044→
  1045→            IRInstr::GetAddress { dst, name } => {
  1046→                let rd = self.resolve_reg(dst)?;
  1047→                // Emit a call to __vuma_getaddr to resolve the symbol at runtime.
  1048→                // Move name hash to X0 as the argument.
  1049→                let name_hash = name
  1050→                    .chars()
  1051→                    .fold(0u64, |acc, c| acc.wrapping_mul(31).wrapping_add(c as u64));
  1052→                self.emit_load_immediate(Register::X0, name_hash as i64)?;
  1053→                let bl_word_idx = self.code.len();
  1054→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1055→                self.call_relocs.push(CallRelocation {
  1056→                    text_byte_offset: bl_byte_offset,
  1057→                    target_func: "__vuma_getaddr".to_string(),
  1058→                });
  1059→                self.relocations.push(RelocationEntry {
  1060→                    offset: bl_byte_offset,
  1061→                    symbol: "__vuma_getaddr".to_string(),
  1062→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1063→                });
  1064→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1065→                if rd != Register::X0 {
  1066→                    self.emit_instruction(Instruction::MOV {
  1067→                        rd,
  1068→                        rm: Register::X0,
  1069→                    })?;
  1070→                }
  1071→            }
  1072→
  1073→            IRInstr::Offset { dst, base, offset } => {
  1074→                let rd = self.resolve_reg(dst)?;
  1075→                let rn = self.resolve_reg(base)?;
  1076→                let rm = match offset {
  1077→                    IRValue::Immediate(v) => {
  1078→                        if *v >= 0 && *v <= 4095 {
  1079→                            Operand::Imm12(*v as u16)
  1080→                        } else {
  1081→                            let temp = Register::X9;
  1082→                            self.emit_load_immediate(temp, *v)?;
  1083→                            Operand::Reg {
  1084→                                reg: temp,
  1085→                                shift: None,
  1086→                            }
  1087→                        }
  1088→                    }
  1089→                    _ => Operand::Reg {
  1090→                        reg: self.resolve_reg(offset)?,
  1091→                        shift: None,
  1092→                    },
  1093→                };
  1094→                self.emit_instruction(Instruction::ADD { rd, rn, rm })?;
  1095→            }
  1096→
  1097→            // ── Dedicated arithmetic — delegate to BinOp ──
  1098→            IRInstr::Add { dst, lhs, rhs, ty } => {
  1099→                self.emit_binop(BinOpKind::Add, dst, lhs, rhs, ty.as_ref())?;
  1100→            }
  1101→            IRInstr::Sub { dst, lhs, rhs, ty } => {
  1102→                self.emit_binop(BinOpKind::Sub, dst, lhs, rhs, ty.as_ref())?;
  1103→            }
  1104→            IRInstr::Mul { dst, lhs, rhs, ty } => {
  1105→                self.emit_binop(BinOpKind::Mul, dst, lhs, rhs, ty.as_ref())?;
  1106→            }
  1107→            IRInstr::Div { dst, lhs, rhs, ty } => {
  1108→                self.emit_binop(BinOpKind::SDiv, dst, lhs, rhs, ty.as_ref())?;
  1109→            }
  1110→
  1111→            // ── Comparison instruction ──
  1112→            IRInstr::Cmp {
  1113→                kind,
  1114→                dst,
  1115→                lhs,
  1116→                rhs,
  1117→                ty,
  1118→            } => {
  1119→                let width = RegWidth::from_ir_type(ty.as_ref());
  1120→                let rd = self.resolve_reg(dst)?;
  1121→                // IMPORTANT: resolve_reg uses X9 for immediates, so if both lhs and rhs
  1122→                // are immediates, the second load would overwrite the first. Use X10 for
  1123→                // the RHS when both are immediates.
  1124→                let rn = self.resolve_reg(lhs)?;
  1125→                let rm = match rhs {
  1126→                    IRValue::Immediate(_) | IRValue::Address(_) => {
  1127→                        // If lhs was also an immediate, it's in X9; use X10 for rhs
  1128→                        let temp = if matches!(lhs, IRValue::Immediate(_) | IRValue::Address(_)) {
  1129→                            Register::X10
  1130→                        } else {
  1131→                            Register::X9
  1132→                        };
  1133→                        match rhs {
  1134→                            IRValue::Immediate(v) => self.emit_load_immediate(temp, *v)?,
  1135→                            IRValue::Address(a) => self.emit_load_immediate(temp, *a as i64)?,
  1136→                            _ => unreachable!(),
  1137→                        }
  1138→                        temp
  1139→                    }
  1140→                    _ => self.resolve_reg(rhs)?,
  1141→                };
  1142→                self.emit_instruction_with_width(
  1143→                    Instruction::CMP {
  1144→                        rn,
  1145→                        rm: Operand::Reg {
  1146→                            reg: rm,
  1147→                            shift: None,
  1148→                        },
  1149→                    },
  1150→                    width,
  1151→                )?;
  1152→                let cond = cmp_kind_to_condition(kind);
  1153→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1154→            }
  1155→
  1156→            // ── Instruction-level control flow ──
  1157→            IRInstr::Ret { values } => {
  1158→                for (i, val) in values.iter().enumerate() {
  1159→                    if i >= 8 {
  1160→                        break;
  1161→                    }
  1162→                    let src = self.resolve_reg(val)?;
  1163→                    let dst_reg = match i {
  1164→                        0 => Register::X0,
  1165→                        1 => Register::X1,
  1166→                        2 => Register::X2,
  1167→                        3 => Register::X3,
  1168→                        4 => Register::X4,
  1169→                        5 => Register::X5,
  1170→                        6 => Register::X6,
  1171→                        7 => Register::X7,
  1172→                        _ => unreachable!(),
  1173→                    };
  1174→                    if src != dst_reg {
  1175→                        self.emit_instruction(Instruction::MOV {
  1176→                            rd: dst_reg,
  1177→                            rm: src,
  1178→                        })?;
  1179→                    }
  1180→                }
  1181→            }
  1182→
  1183→            IRInstr::Branch { target } => {
  1184→                let fixup_idx = self.code.len();
  1185→                self.fixups.push((fixup_idx, target.clone(), BranchFormat::B26));
  1186→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1187→            }
  1188→
  1189→            IRInstr::CondBranch {
  1190→                cond,
  1191→                true_target,
  1192→                false_target,
  1193→            } => {
  1194→                let rt = self.resolve_reg(cond)?;
  1195→                // CBNZ: if cond != 0, branch to true_target
  1196→                let fixup_cbz = self.code.len();
  1197→                self.fixups.push((fixup_cbz, true_target.clone(), BranchFormat::Cond19));
  1198→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1199→                // Otherwise (cond == 0), branch to false_target
  1200→                let fixup_b = self.code.len();
  1201→                self.fixups.push((fixup_b, false_target.clone(), BranchFormat::B26));
  1202→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1203→            }
  1204→
  1205→            IRInstr::Select {
  1206→                dst,
  1207→                cond,
  1208→                true_val,
  1209→                false_val,
  1210→                ty,
  1211→            } => {
  1212→                let width = RegWidth::from_ir_type(ty.as_ref());
  1213→                // Lower select as: SUBS XZR, cond, #0; CSEL dst, false_val, true_val, NE
  1214→                let rd = self.resolve_reg(dst)?;
  1215→                let rc = self.resolve_reg(cond)?;
  1216→                let rt = self.resolve_reg(true_val)?;
  1217→                let rf = self.resolve_reg(false_val)?;
  1218→                // Compare cond against zero and select.
  1219→                self.emit_instruction_with_width(
  1220→                    Instruction::SUB {
  1221→                        rd: Register::XZR,
  1222→                        rn: rc,
  1223→                        rm: Operand::Imm12(0),
  1224→                    },
  1225→                    width,
  1226→                )?;
  1227→                // Set flags by using a separate CMP (SUB with XZR destination
  1228→                // doesn't set flags; we need a flags-setting variant).
  1229→                // We emulate this with: CMP rc, #0 which is SUBS XZR, rc, #0.
  1230→                // Since we only have SUB, we use the existing CMP pattern.
  1231→                self.emit_instruction_with_width(
  1232→                    Instruction::CSEL {
  1233→                        rd,
  1234→                        rn: rf,
  1235→                        rm: rt,
  1236→                        cond: crate::arm64::Condition::NE,
  1237→                    },
  1238→                    width,
  1239→                )?;
  1240→            }
  1241→        }
  1242→        Ok(())
  1243→    }
  1244→
  1245→    /// Emit a binary operation (shared by `BinOp` and dedicated `Add`/`Sub`/…).
  1246→    ///
  1247→    /// Some ARM64 instructions (ADD, SUB, LSL, LSR, ASR) accept a 12-bit
  1248→    /// unsigned immediate operand directly.  Most others (MUL, AND, ORR, EOR,
  1249→    /// SDIV, UDIV, remainder, comparisons) **require** a register operand.
  1250→    /// When the RHS is an immediate that the target instruction cannot accept
  1251→    /// directly, we "spill" it into a scratch register (X16 / IP0) first via
  1252→    /// MOVZ + MOVK and then use that register as the operand.
  1253→    ///
  1254→    /// X16 (IP0) is chosen as the scratch register because it is **not** in
  1255→    /// the register allocator's free pool, so it will never conflict with
  1256→    /// `rd` or `rn`.
  1257→    fn emit_binop(
  1258→        &mut self,
  1259→        op: BinOpKind,
  1260→        dst: &IRValue,
  1261→        lhs: &IRValue,
  1262→        rhs: &IRValue,
  1263→        ty: Option<&IRType>,
  1264→    ) -> Result<()> {
  1265→        let width = RegWidth::from_ir_type(ty);
  1266→        let rd = self.resolve_reg(dst)?;
  1267→        let rn = self.resolve_reg(lhs)?;
  1268→
  1269→        // Determine whether the operation can accept an immediate operand.
  1270→        // ADD, SUB, LSL, LSR, ASR all have immediate forms in ARM64.
  1271→        // All other operations require a register operand.
  1272→        let supports_imm = matches!(
  1273→            op,
  1274→            BinOpKind::Add | BinOpKind::Sub | BinOpKind::Shl | BinOpKind::ShrL | BinOpKind::ShrA
  1275→        );
  1276→
  1277→        let rm = match rhs {
  1278→            IRValue::Immediate(v) => {
  1279→                if supports_imm && *v >= 0 && *v <= 4095 {
  1280→                    // Small immediate that the instruction accepts directly.
  1281→                    Operand::Imm12(*v as u16)
  1282→                } else {
  1283→                    // Load the immediate into scratch register X16 (IP0).
  1284→                    // X16 is not in the register allocator's pool, so it cannot
  1285→                    // conflict with `rd` or `rn`.
  1286→                    let scratch = Register::X16;
  1287→                    self.emit_load_immediate_with_width(scratch, *v, width)?;
  1288→                    Operand::Reg {
  1289→                        reg: scratch,
  1290→                        shift: None,
  1291→                    }
  1292→                }
  1293→            }
  1294→            _ => Operand::Reg {
  1295→                reg: self.resolve_reg(rhs)?,
  1296→                shift: None,
  1297→            },
  1298→        };
  1299→
  1300→        match op {
  1301→            BinOpKind::Add => {
  1302→                self.emit_instruction_with_width(Instruction::ADD { rd, rn, rm }, width)?;
  1303→            }
  1304→            BinOpKind::Sub => {
  1305→                self.emit_instruction_with_width(Instruction::SUB { rd, rn, rm }, width)?;
  1306→            }
  1307→            BinOpKind::Mul => {
  1308→                let rm_reg = self.operand_to_reg(&rm)?;
  1309→                self.emit_instruction_with_width(Instruction::MUL { rd, rn, rm: rm_reg }, width)?;
  1310→            }
  1311→            BinOpKind::SDiv => {
  1312→                let rm_reg = self.operand_to_reg(&rm)?;
  1313→                self.emit_instruction_with_width(Instruction::SDIV { rd, rn, rm: rm_reg }, width)?;
  1314→            }
  1315→            BinOpKind::UDiv => {
  1316→                let rm_reg = self.operand_to_reg(&rm)?;
  1317→                self.emit_instruction_with_width(Instruction::UDIV { rd, rn, rm: rm_reg }, width)?;
  1318→            }
  1319→            BinOpKind::And => {
  1320→                let rm_reg = self.operand_to_reg(&rm)?;
  1321→                self.emit_instruction_with_width(Instruction::AND { rd, rn, rm: rm_reg }, width)?;
  1322→            }
  1323→            BinOpKind::Or => {
  1324→                let rm_reg = self.operand_to_reg(&rm)?;
  1325→                self.emit_instruction_with_width(Instruction::ORR { rd, rn, rm: rm_reg }, width)?;
  1326→            }
  1327→            BinOpKind::Xor => {
  1328→                let rm_reg = self.operand_to_reg(&rm)?;
  1329→                self.emit_instruction_with_width(Instruction::EOR { rd, rn, rm: rm_reg }, width)?;
  1330→            }
  1331→            BinOpKind::Shl => {
  1332→                self.emit_instruction_with_width(Instruction::LSL { rd, rn, rm }, width)?;
  1333→            }
  1334→            BinOpKind::ShrL => {
  1335→                self.emit_instruction_with_width(Instruction::LSR { rd, rn, rm }, width)?;
  1336→            }
  1337→            BinOpKind::ShrA => {
  1338→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm }, width)?;
  1339→            }
  1340→            BinOpKind::Ror | BinOpKind::Rol => {
  1341→                let rm_reg = self.operand_to_reg(&rm)?;
  1342→                // ARM64 rotation: use ASR as placeholder
  1343→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm: Operand::Reg { reg: rm_reg, shift: None } }, width)?;
  1344→            }
  1345→            BinOpKind::SRem | BinOpKind::URem => {
  1346→                let rm_reg = self.operand_to_reg(&rm)?;
  1347→                let div_instr = if op == BinOpKind::SRem {
  1348→                    Instruction::SDIV { rd, rn, rm: rm_reg }
  1349→                } else {
  1350→                    Instruction::UDIV { rd, rn, rm: rm_reg }
  1351→                };
  1352→                self.emit_instruction_with_width(div_instr, width)?;
  1353→                // MSUB rd, rd, rm, rn  =>  rd = rn - rd * rm  =  dividend - quotient * divisor
  1354→                self.emit_instruction_with_width(Instruction::MSUB {
  1355→                    rd,
  1356→                    rn: rd,     // quotient (result of DIV)
  1357→                    rm: rm_reg, // divisor
  1358→                    ra: rn,     // dividend
  1359→                }, width)?;
  1360→            }
  1361→            BinOpKind::SLt
  1362→            | BinOpKind::SLe
  1363→            | BinOpKind::SGt
  1364→            | BinOpKind::SGe
  1365→            | BinOpKind::ULt
  1366→            | BinOpKind::ULe
  1367→            | BinOpKind::UGt
  1368→            | BinOpKind::UGe
  1369→            | BinOpKind::Eq
  1370→            | BinOpKind::Ne => {
  1371→                let rm_reg = self.operand_to_reg(&rm)?;
  1372→                self.emit_instruction_with_width(Instruction::CMP {
  1373→                    rn,
  1374→                    rm: Operand::Reg {
  1375→                        reg: rm_reg,
  1376→                        shift: None,
  1377→                    },
  1378→                }, width)?;
  1379→                let cond = binop_kind_to_condition(&op);
  1380→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1381→            }
  1382→        }
  1383→        Ok(())
  1384→    }
  1385→
  1386→    /// Emit the block terminator.
  1387→    fn emit_terminator(&mut self, term: &IRTerminator) -> Result<()> {
  1388→        match term {
  1389→            IRTerminator::Jump(target) => {
  1390→                let fixup_idx = self.code.len();
  1391→                self.fixups.push((fixup_idx, target.clone(), BranchFormat::B26));
  1392→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1393→            }
  1394→            IRTerminator::Branch {
  1395→                cond,
  1396→                true_block,
  1397→                false_block,
  1398→            } => {
  1399→                let rt = self.resolve_reg(cond)?;
  1400→                let fixup_cbnz = self.code.len();
  1401→                self.fixups.push((fixup_cbnz, true_block.clone(), BranchFormat::Cond19));
  1402→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1403→                let fixup_b = self.code.len();
  1404→                self.fixups.push((fixup_b, false_block.clone(), BranchFormat::B26));
  1405→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1406→            }
  1407→            IRTerminator::Return(vals) => {
  1408→                for (i, val) in vals.iter().enumerate() {
  1409→                    if i >= 8 {
  1410→                        break;
  1411→                    }
  1412→                    let src = self.resolve_reg(val)?;
  1413→                    let dst_reg = match i {
  1414→                        0 => Register::X0,
  1415→                        1 => Register::X1,
  1416→                        2 => Register::X2,
  1417→                        3 => Register::X3,
  1418→                        4 => Register::X4,
  1419→                        5 => Register::X5,
  1420→                        6 => Register::X6,
  1421→                        7 => Register::X7,
  1422→                        _ => unreachable!(),
  1423→                    };
  1424→                    if src != dst_reg {
  1425→                        self.emit_instruction(Instruction::MOV {
  1426→                            rd: dst_reg,
  1427→                            rm: src,
  1428→                        })?;
  1429→                    }
  1430→                }
  1431→                // Use the same frame size computed in the prologue
  1432→                let frame_size = self.frame_size;
  1433→                self.emit_instruction(Instruction::LDP {
  1434→                    rt1: Register::X29,
  1435→                    rt2: Register::X30,
  1436→                    rn: Register::SP,
  1437→                    offset: frame_size as i32, // load FP/LR from top of frame
  1438→                })?;
  1439→                self.emit_instruction(Instruction::ADD {
  1440→                    rd: Register::SP,
  1441→                    rn: Register::SP,
  1442→                    rm: Operand::Imm12(frame_size + 16), // restore SP
  1443→                })?;
  1444→                self.emit_instruction(Instruction::RET { rn: None })?;
  1445→            }
  1446→            IRTerminator::Unreachable => {
  1447→                self.emit_instruction(Instruction::MOV {
  1448→                    rd: Register::XZR,
  1449→                    rm: Register::XZR,
  1450→                })?;
  1451→            }
  1452→            // Switch, Invoke, TailCall, and Resume are lowered by the
  1453→            // control_flow module before reaching the emitter. If they
  1454→            // appear here, it means the lowering pass was not run.
  1455→            IRTerminator::Switch { .. } => {
  1456→                return Err(CodegenError::InvalidInstruction(
  1457→                    "Switch terminator must be lowered before emission".to_string(),
  1458→                ));
  1459→            }
  1460→            IRTerminator::Invoke { .. } => {
  1461→                return Err(CodegenError::InvalidInstruction(
  1462→                    "Invoke terminator must be lowered before emission".to_string(),
  1463→                ));
  1464→            }
  1465→            IRTerminator::TailCall { .. } => {
  1466→                return Err(CodegenError::InvalidInstruction(
  1467→                    "TailCall terminator must be lowered before emission".to_string(),
  1468→                ));
  1469→            }
  1470→            IRTerminator::Resume { .. } => {
  1471→                return Err(CodegenError::InvalidInstruction(
  1472→                    "Resume terminator must be lowered before emission".to_string(),
  1473→                ));
  1474→            }
  1475→        }
  1476→        Ok(())
  1477→    }

  1483→    fn emit_instruction(&mut self, instr: Instruction) -> Result<()> {

  1485→        self.code.push(word);
  1486→        Ok(())

  1489→    /// Emit an instruction with a specific register width (32-bit W or 64-bit X).
  1490→    ///
  1491→    /// This is the primary emission method for arithmetic and logical instructions
  1492→    /// where the operand width affects the encoding. Using `RegWidth::W32` produces
  1493→    /// instructions that operate on W sub-registers, giving automatic 32-bit
  1494→    /// wrapping arithmetic.
  1495→    fn emit_instruction_with_width(&mut self, instr: Instruction, width: RegWidth) -> Result<()> {
  1496→        let word = instr.encode_with_width(width)?;
  1497→        self.code.push(word);
  1498→        Ok(())
  1499→    }

  1501→    /// Emit a load instruction with the given offset and IR type.
  1502→    ///
  1503→    /// Selects the correct ARM64 load variant based on the IR type:
  1504→    /// - I8/U8 → LDRB (byte, zero-extended)
  1505→    /// - I16/U16 → LDRH (halfword, zero-extended)
  1506→    /// - I32/U32/Ptr/Func → LDR (word or doubleword)
  1507→    /// - I64/U64 → LDR (doubleword)
  1508→    ///
  1509→    /// If the offset fits the ARM64 unsigned-offset immediate encoding for the
  1510→    /// selected instruction variant, it is encoded directly. Otherwise, the
  1511→    /// effective address is computed in a scratch register (X9) first, and
  1512→    /// the load is performed at [X9 + 0].
  1513→    fn emit_load(&mut self, rt: Register, rn: Register, offset: i32, ty: &IRType) -> Result<()> {
  1514→        // Determine the ARM64 load instruction and its immediate-offset constraints.
  1515→        // For each variant: (scale, max_imm12) where the encoded offset = imm12 << scale.
  1516→        // LDRB:  scale=0, imm12 0..4095 → offset 0..4095
  1517→        // LDRH:  scale=1, imm12 0..4095 → offset 0..8190, even
  1518→        // LDRSW: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1519→        // LDR W: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1520→        // LDR X: scale=3, imm12 0..4095 → offset 0..32760, multiple of 8
  1521→        match ty {
  1522→            IRType::I8 | IRType::U8 => {
  1523→                if offset >= 0 && offset <= 4095 {
  1524→                    self.emit_instruction(Instruction::LDRB { rt, rn, offset })?;
  1525→                } else {
  1526→                    self.emit_address_with_offset(rn, offset)?;
  1527→                    self.emit_instruction(Instruction::LDRB {
  1528→                        rt,
  1529→                        rn: Register::X9,
  1530→                        offset: 0,
  1531→                    })?;
  1532→                }
  1533→                // Sign-extend for signed byte loads (SXTB = SBFM Xd, Xn, #0, #7)
  1534→                if *ty == IRType::I8 {
  1535→                    self.emit_instruction(Instruction::SBFM {
  1536→                        rd: rt,
  1537→                        rn: rt,
  1538→                        immr: 0,
  1539→                        imms: 7,
  1540→                    })?;
  1541→                }
  1542→            }
  1543→            IRType::I16 | IRType::U16 => {
  1544→                if offset >= 0 && offset <= 8190 && offset % 2 == 0 {
  1545→                    self.emit_instruction(Instruction::LDRH { rt, rn, offset })?;
  1546→                } else {
  1547→                    self.emit_address_with_offset(rn, offset)?;
  1548→                    self.emit_instruction(Instruction::LDRH {
  1549→                        rt,
  1550→                        rn: Register::X9,
  1551→                        offset: 0,
  1552→                    })?;
  1553→                }
  1554→                // Sign-extend for signed halfword loads (SXTH = SBFM Xd, Xn, #0, #15)
  1555→                if *ty == IRType::I16 {
  1556→                    self.emit_instruction(Instruction::SBFM {
  1557→                        rd: rt,
  1558→                        rn: rt,
  1559→                        immr: 0,
  1560→                        imms: 15,
  1561→                    })?;
  1562→                }
  1563→            }
  1564→            IRType::I32 | IRType::U32 => {
  1565→                // 32-bit load uses LDR Wt encoding (scale=2, offset must be multiple of 4)
  1566→                if offset >= 0 && offset <= 16380 && offset % 4 == 0 {
  1567→                    self.emit_instruction(Instruction::LDR_W { rt, rn, offset })?;
  1568→                } else {
  1569→                    self.emit_address_with_offset(rn, offset)?;
  1570→                    self.emit_instruction(Instruction::LDR_W {
  1571→                        rt,
  1572→                        rn: Register::X9,
  1573→                        offset: 0,
  1574→                    })?;
  1575→                }
  1576→                // Sign-extend for signed word loads (SXTW = SBFM Xd, Xn, #0, #31)
  1577→                if *ty == IRType::I32 {
  1578→                    self.emit_instruction(Instruction::SBFM {
  1579→                        rd: rt,
  1580→                        rn: rt,
  1581→                        immr: 0,
  1582→                        imms: 31,
  1583→                    })?;
  1584→                }
  1585→            }
  1586→            IRType::I64 | IRType::U64 | IRType::Ptr | IRType::Func => {
  1587→                // 64-bit load uses LDR Xt encoding (scale=3, offset must be multiple of 8)
  1588→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1589→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1590→                } else {
  1591→                    self.emit_address_with_offset(rn, offset)?;
  1592→                    self.emit_instruction(Instruction::LDR {
  1593→                        rt,
  1594→                        rn: Register::X9,
  1595→                        offset: 0,
  1596→                    })?;
  1597→                }
  1598→            }
  1599→            _ => {
  1600→                // Default: treat as 64-bit load
  1601→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1602→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1603→                } else {
  1604→                    self.emit_address_with_offset(rn, offset)?;
  1605→                    self.emit_instruction(Instruction::LDR {
  1606→                        rt,
  1607→                        rn: Register::X9,
  1608→                        offset: 0,
  1609→                    })?;
  1610→                }
  1611→            }
  1612→        }
  1613→        Ok(())
  1614→    }
  1615→
  1616→    /// Emit a store instruction with the given offset and IR type.
  1617→    ///
  1618→    /// Selects the correct ARM64 store variant based on the IR type:
  1619→    /// - I8/U8 → STRB (byte)
  1620→    /// - I16/U16 → STRH (halfword)
  1621→    /// - I32/U32 → STR_W (32-bit word)
  1622→    /// - I64/U64/Ptr/Func → STR (64-bit doubleword)
  1623→    ///

  1737→                // This prevents subsequent resolve_reg calls within the same

  1809→    fn emit_load_immediate(&mut self, rd: Register, value: i64) -> Result<()> {
  1810→        if (0..=65535).contains(&value) {

  1813→                imm16: value as u16,

  1828→                imm16: hi,

  1936→    fn apply_fixups(&mut self) -> Result<()> {
  1937→        let fixups = std::mem::take(&mut self.fixups);
  1938→        for (word_idx, label, format) in &fixups {
  1939→            let target_offset = self.label_offsets.get(label).copied().unwrap_or(0);
  1940→            let offset = (target_offset as i32) - (*word_idx as i32);
  1941→            let old_word = self.code[*word_idx];
  1942→            let patched = match format {
  1943→                BranchFormat::B26 => {
  1944→                    // B / BL: offset in bits[25:0], word-aligned
  1945→                    (old_word & !0x03FFFFFF) | ((offset & 0x03FFFFFF) as u32)
  1946→                }
  1947→                BranchFormat::Cond19 => {
  1948→                    // CBZ / CBNZ: imm19 in bits[23:5], word-aligned
  1949→                    // Clear bits[23:5], then set imm19 there
  1950→                    let imm19 = offset & 0x7FFFF;
  1951→                    (old_word & !(0x7FFFF << 5)) | ((imm19 as u32) << 5)
  1952→                }
  1953→                BranchFormat::BCond19 => {
  1954→                    // B.cond: imm19 in bits[23:5], word-aligned
  1955→                    let imm19 = offset & 0x7FFFF;
  1956→                    (old_word & !(0x7FFFF << 5)) | ((imm19 as u32) << 5)
  1957→                }
  1958→            };
  1959→            self.code[*word_idx] = patched;
  1960→        }
  1961→        Ok(())
