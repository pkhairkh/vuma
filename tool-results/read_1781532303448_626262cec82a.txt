     1→     1→//! # Register Allocation
     2→     2→//!
     3→     3→//! Provides register allocators that map IR virtual registers to physical
     4→     4→//! registers. The module contains two families of allocators:
     5→     5→//!
     6→     6→//! ## ARM64-specific allocators (legacy)
     7→     7→//!
     8→     8→//! ### `RegAllocator` (legacy greedy)
     9→     9→//!
    10→    10→//! A simple greedy allocator that walks the IR and assigns caller-saved
    11→    11→//! registers first, then callee-saved, spilling when all are exhausted.
    12→    12→//! Kept for backward-compatibility with the existing emitter.
    13→    13→//!
    14→    14→//! ### `LinearScanAllocator` (ARM64 production)
    15→    15→//!
    16→    16→//! A real **linear-scan** register allocator for ARM64 that:
    17→    17→//!
    18→    18→//! 1. Computes live ranges from the IR (per-function, across all blocks).
    19→    19→//! 2. Sorts intervals by start point.
    20→    20→//! 3. Walks intervals in order, assigning free physical registers.
    21→    21→//! 4. When the pool is exhausted, evicts the interval whose end point is
    22→    22→//!    farthest in the future (or spills the current one if it ends latest).
    23→    23→//! 5. Generates spill/reload code as needed — including reloads at each
    24→    24→//!    use position after eviction.
    25→    25→//! 6. Applies register coalescing to eliminate redundant copies.
    26→    26→//!
    27→    27→//! ## Target-agnostic allocator (new)
    28→    28→//!
    29→    29→//! ### `TargetAgnosticRegAlloc`
    30→    30→//!
    31→    31→//! A **target-agnostic** linear-scan register allocator driven by the
    32→    32→//! `TargetDesc` data from `target_desc.rs`. It derives the available
    33→    33→//! register pool (caller-saved, callee-saved, per class) from the target
    34→    34→//! description rather than hard-coding ARM64 registers. Any backend can
    35→    35→//! use this allocator by passing its `TargetDesc`.
    36→    36→//!
    37→    37→//! ## ARM64 Register Conventions (AAPCS64)
    38→    38→//!
    39→    39→//! | Register(s) | Role                                   | Class        |
    40→    40→//! |-------------|----------------------------------------|--------------|
    41→    41→//! | X0–X7       | Argument / result registers            | Caller-saved |
    42→    42→//! | X8          | Indirect result location register      | Caller-saved |
    43→    43→//! | X9–X15      | Caller-saved temporary registers       | Caller-saved |
    44→    44→//! | X16–X17     | Intra-procedure-call scratch (IP0/IP1) | Caller-saved |
    45→    45→//! | X18         | Platform register                      | Reserved     |
    46→    46→//! | X19–X28     | Callee-saved registers                 | Callee-saved |
    47→    47→//! | X29         | Frame pointer (FP)                     | Reserved     |
    48→    48→//! | X30         | Link register (LR)                     | Reserved     |
    49→    49→//! | SP          | Stack pointer                          | Reserved     |
    50→    50→//! | XZR         | Zero register                          | Reserved     |
    51→    51→//! | V0–V7       | FP/SIMD argument / result registers    | Caller-saved |
    52→    52→//! | V8–V15      | FP/SIMD callee-saved (lower 64 bits)   | Callee-saved |
    53→    53→//! | V16–V31     | FP/SIMD caller-saved temporaries       | Caller-saved |
    54→    54→

   305→   305→// Live Interval
   306→   306→// ═══════════════════════════════════════════════════════════════════════════
   307→   307→
   308→   308→/// A live interval for a virtual register, representing the range of
   309→   309→/// instruction positions where the register is live.
   310→   310→///
   311→   311→/// Positions are sequential instruction indices assigned during a linear
   312→   312→/// pass over all blocks in a function.
   313→   313→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   314→   314→pub struct LiveInterval {
   315→   315→    /// The virtual register ID.
   316→   316→    pub vreg: IRValueId,
   317→   317→    /// The register class (GPR or SIMD/FP).
   318→   318→    pub class: RegClass,
   319→   319→    /// Start position (inclusive) — the instruction that defines this vreg.
   320→   320→    pub start: u32,
   321→   321→    /// End position (inclusive) — the last instruction that uses this vreg.
   322→   322→    pub end: u32,
   323→   323→    /// Whether this interval crosses a function call.
   324→   324→    pub crosses_call: bool,
   325→   325→    /// List of use positions (for spill/reload decisions).
   326→   326→    pub use_positions: Vec<u32>,
   327→   327→    /// List of def positions.
   328→   328→    pub def_positions: Vec<u32>,
   329→   329→    /// Set of vregs that were coalesced into this interval.
   330→   330→    /// When intervals are merged via coalescing, all constituent vreg IDs
   331→   331→    /// are tracked here so that the allocator can map them all to the same
   332→   332→    /// physical register.
   333→   333→    pub coalesced_vregs: Vec<IRValueId>,
   334→   334→}
   335→   335→
   336→   336→impl LiveInterval {
   337→   337→    /// Create a new interval starting and ending at the given positions.
   338→   338→    pub fn new(vreg: IRValueId, class: RegClass, start: u32, end: u32) -> Self {
   339→   339→        Self {
   340→   340→            vreg,
   341→   341→            class,
   342→   342→            start,
   343→   343→            end,
   344→   344→            crosses_call: false,
   345→   345→            use_positions: Vec::new(),
   346→   346→            def_positions: Vec::new(),
   347→   347→            coalesced_vregs: vec![vreg],
   348→   348→        }
   349→   349→    }
   350→   350→
   351→   351→    /// Returns `true` if the interval covers the given position.
   352→   352→    pub fn covers(&self, pos: u32) -> bool {
   353→   353→        pos >= self.start && pos <= self.end
   354→   354→    }
   355→   355→
   356→   356→    /// Extend the interval to include the given position.
   357→   357→    pub fn extend_to(&mut self, pos: u32) {
   358→   358→        if pos > self.end {
   359→   359→            self.end = pos;
   360→   360→        }
   361→   361→        if pos < self.start {
   362→   362→            self.start = pos;
   363→   363→        }
   364→   364→    }
   365→   365→
   366→   366→    /// Duration of the live interval.
   367→   367→    pub fn len(&self) -> u32 {
   368→   368→        self.end.saturating_sub(self.start)
   369→   369→    }
   370→   370→
   371→   371→    /// Returns true if this interval has zero length.
   372→   372→    pub fn is_empty(&self) -> bool {
   373→   373→        self.start == self.end
   374→   374→    }
   375→   375→
   376→   376→    /// Returns true if this interval ends before the other starts.
   377→   377→    pub fn ends_before(&self, other: &LiveInterval) -> bool {
   378→   378→        self.end < other.start
   379→   379→    }
   380→   380→
   381→   381→    /// Returns true if this interval overlaps with the other.
   382→   382→    pub fn overlaps(&self, other: &LiveInterval) -> bool {
   383→   383→        self.start <= other.end && other.start <= self.end
   384→   384→    }
   385→   385→
   386→   386→    /// Compute a spill weight for this interval.
   387→   387→    ///
   388→   388→    /// Higher weight = more important to keep in a register.
   389→   389→    /// Weight is based on:
   390→   390→    /// - Number of use/def positions (more references = higher weight)
   391→   391→    /// - Whether the interval crosses a call (callee-saved is expensive to use)
   392→   392→    /// - Loop nesting could be added here in the future
   393→   393→    pub fn spill_weight(&self) -> u32 {
   394→   394→        let use_count = self.use_positions.len() as u32;
   395→   395→        let def_count = self.def_positions.len() as u32;
   396→   396→        let base_weight = (use_count + def_count).max(1);
   397→   397→
   398→   398→        // Intervals that cross calls have higher weight because spilling
   399→   399→        // around a call requires both save and restore, and using a
   400→   400→        // callee-saved register has prologue/epilogue cost.
   401→   401→        let call_multiplier = if self.crosses_call { 2 } else { 1 };
   402→   402→
   403→   403→        base_weight * call_multiplier
   404→   404→    }
   405→   405→
   406→   406→    /// Returns the weight per unit of live range length — used to decide
   407→   407→    /// which interval to evict.  Higher = more deserving of a register.
   408→   408→    pub fn weight_per_length(&self) -> u32 {
   409→   409→        let len = self.len().max(1);
   410→   410→        self.spill_weight() / len
   411→   411→    }
   412→   412→}

   436→   436→    /// Spill/reload instructions to be inserted, keyed by instruction position.

   438→   438→    /// The original live intervals (for debugging / introspection).

   441→   441→    pub eliminated_copies: Vec<(IRValueId, PhysReg)>,

   443→   443→    /// When intervals are merged, all constituent vregs map to the same preg.
   444→   444→    pub coalesced_map: HashMap<IRValueId, IRValueId>,
   445→   445→}

   452→   452→            spill_slots: HashMap::new(),
   453→   453→            total_spill_slots: 0,
   454→   454→            used_callee_saved_gprs: HashSet::new(),
   455→   455→            used_callee_saved_simd: HashSet::new(),
   456→   456→            spill_code: BTreeMap::new(),
   457→   457→            live_intervals: Vec::new(),
   458→   458→            eliminated_copies: Vec::new(),
   459→   459→            coalesced_map: HashMap::new(),
   460→   460→        }
   461→   461→    }
   462→   462→
   463→   463→    /// Look up the physical register assigned to a virtual register.
   464→   464→    /// Also follows the coalescing map if the vreg was merged.
   465→   465→    pub fn get_phys_reg(&self, vreg: IRValueId) -> Option<PhysReg> {
   466→   466→        if let Some(&preg) = self.vreg_to_preg.get(&vreg) {
   467→   467→            return Some(preg);
   468→   468→        }
   469→   469→        // Follow coalescing chain.
   470→   470→        let rep = self.coalesced_map.get(&vreg).copied().unwrap_or(vreg);
   471→   471→        self.vreg_to_preg.get(&rep).copied()
   472→   472→    }
   473→   473→
   474→   474→    /// Look up the GPR assigned to a virtual register (convenience).
   475→   475→    pub fn get_gpr(&self, vreg: IRValueId) -> Option<Register> {
   476→   476→        match self.get_phys_reg(vreg) {
   477→   477→            Some(PhysReg::Gpr(r)) => Some(r),
   478→   478→            _ => None,
   479→   479→        }
   480→   480→    }
   481→   481→
   482→   482→    /// Look up the SIMD/FP register assigned to a virtual register.
   483→   483→    pub fn get_simd(&self, vreg: IRValueId) -> Option<SimdFpRegister> {
   484→   484→        match self.get_phys_reg(vreg) {
   485→   485→            Some(PhysReg::SimdFp(r)) => Some(r),
   486→   486→            _ => None,
   487→   487→        }
   488→   488→    }
   489→   489→
   490→   490→    /// Check if a virtual register is spilled.
   491→   491→    pub fn is_spilled(&self, vreg: IRValueId) -> bool {
   492→   492→        if self.spill_slots.contains_key(&vreg) {
   493→   493→            return true;
   494→   494→        }
   495→   495→        let rep = self.coalesced_map.get(&vreg).copied().unwrap_or(vreg);
   496→   496→        self.spill_slots.contains_key(&rep)
   497→   497→    }
   498→   498→
   499→   499→    /// Get the spill slot for a virtual register.
   500→   500→    pub fn spill_slot(&self, vreg: IRValueId) -> Option<&SpillSlot> {
   501→   501→        if let Some(slot) = self.spill_slots.get(&vreg) {
   502→   502→            return Some(slot);
   503→   503→        }
   504→   504→        let rep = self.coalesced_map.get(&vreg).copied().unwrap_or(vreg);
   505→   505→        self.spill_slots.get(&rep)
   506→   506→    }
   507→   507→
   508→   508→    /// Calculate the total frame size needed for spill slots.
   509→   509→    /// GPR slots are 8 bytes each; SIMD/FP slots are 16 bytes each.
   510→   510→    pub fn spill_frame_bytes(&self) -> u32 {
   511→   511→        self.spill_slots
   512→   512→            .values()
   513→   513→            .map(|slot| slot.size_bytes())
   514→   514→            .sum()
   515→   515→    }
   516→   516→
   517→   517→    /// Number of callee-saved registers that must be saved in prologue.
   518→   518→    pub fn callee_saved_count(&self) -> usize {
   519→   519→        self.used_callee_saved_gprs.len() + self.used_callee_saved_simd.len()
   520→   520→    }
   521→   521→
   522→   522→    /// Record a coalescing: `src` was merged into `dst`'s interval.
   523→   523→    /// After allocation, both src and dst should map to the same preg.
   524→   524→    pub fn record_coalescing(&mut self, src: IRValueId, dst: IRValueId) {
   525→   525→        self.coalesced_map.insert(src, dst);
   526→   526→    }
   527→   527→
   528→   528→    /// Resolve a vreg through the coalescing map to the representative vreg.
   529→   529→    pub fn resolve_vreg(&self, vreg: IRValueId) -> IRValueId {
   530→   530→        self.coalesced_map.get(&vreg).copied().unwrap_or(vreg)
   531→   531→    }
   532→   532→}

   534→   534→impl Default for AllocationResult {

   537→   537→    }

   627→   627→/// A copy instruction identified for potential coalescing.

   629→   629→struct CopyInfo {
   630→   630→    /// Source virtual register.
   631→   631→    src: IRValueId,
   632→   632→    /// Destination virtual register.
   633→   633→    dst: IRValueId,

   640→   640→// ═══════════════════════════════════════════════════════════════════════════
   641→   641→// Live Range Computation
   642→   642→// ═══════════════════════════════════════════════════════════════════════════
   643→   643→
   644→   644→/// Computes live intervals for all virtual registers in a function.
   645→   645→///
   646→   646→/// Assigns sequential instruction positions across all blocks (linear order),
   647→   647→/// then computes def-use information to build intervals.  Also tracks which
   648→   648→/// intervals cross function call sites (to guide caller-saved vs callee-saved
   649→   649→/// allocation).
   650→   650→pub struct LiveRangeComputer {
   651→   651→    /// Default register class — assumed GPR unless the user annotates.
   652→   652→    default_class: RegClass,
   653→   653→    /// Per-vreg register class overrides.
   654→   654→    class_overrides: HashMap<IRValueId, RegClass>,
   655→   655→}
   656→   656→
   657→   657→impl LiveRangeComputer {
   658→   658→    /// Create a new live range computer with GPR as the default class.
   659→   659→    pub fn new() -> Self {
   660→   660→        Self {
   661→   661→            default_class: RegClass::Gpr,
   662→   662→            class_overrides: HashMap::new(),
   663→   663→        }
   664→   664→    }
   665→   665→
   666→   666→    /// Set the register class for a specific virtual register.
   667→   667→    pub fn set_class(&mut self, vreg: IRValueId, class: RegClass) {
   668→   668→        self.class_overrides.insert(vreg, class);
   669→   669→    }
   670→   670→
   671→   671→    /// Get the register class for a virtual register.
   672→   672→    fn class_of(&self, vreg: IRValueId) -> RegClass {
   673→   673→        self.class_overrides
   674→   674→            .get(&vreg)
   675→   675→            .copied()
   676→   676→            .unwrap_or(self.default_class)
   677→   677→    }
   678→   678→
   679→   679→    /// Compute live intervals for the given function.
   680→   680→    ///
   681→   681→    /// Returns a vector of live intervals, one per virtual register, and
   682→   682→    /// a set of instruction positions where function calls occur.
   683→   683→    pub fn compute(&self, func: &IRFunction) -> (Vec<LiveInterval>, BTreeSet<u32>) {
   684→   684→        // Phase 1: Assign sequential positions to every instruction.
   685→   685→        // We use 2*N for each instruction N, so that we can insert
   686→   686→        // spill/reload code at positions 2*N+1 if needed.
   687→   687→        let mut intervals: HashMap<IRValueId, LiveInterval> = HashMap::new();
   688→   688→        let mut call_positions: BTreeSet<u32> = BTreeSet::new();
   689→   689→        let mut copies: Vec<CopyInfo> = Vec::new();
   690→   690→
   691→   691→        let mut pos: u32 = 0;
   692→   692→
   693→   693→        for block in &func.blocks {
   694→   694→            for instr in &block.instructions {
   695→   695→                let def_regs = instr.defined_regs();
   696→   696→                let use_regs = instr.used_regs();
   697→   697→
   698→   698→                // Detect copies: Cast with BitCast between two registers.
   699→   699→                if let IRInstr::Cast {
   700→   700→                    kind: crate::ir::CastKind::BitCast,
   701→   701→                    dst: IRValue::Register(dst_id),
   702→   702→                    src: IRValue::Register(src_id),
   703→   703→                } = instr
   704→   704→                {
   705→   705→                    copies.push(CopyInfo {
   706→   706→                        src: *src_id,
   707→   707→                        dst: *dst_id,
   708→   708→                        pos,
   709→   709→                        class: self.class_of(*dst_id),
   710→   710→                    });
   711→   711→                }
   712→   712→
   713→   713→                // Process definitions.
   714→   714→                for &vreg in &def_regs {
   715→   715→                    let class = self.class_of(vreg);
   716→   716→                    let interval = intervals
   717→   717→                        .entry(vreg)
   718→   718→                        .or_insert_with(|| LiveInterval::new(vreg, class, pos, pos));
   719→   719→                    interval.def_positions.push(pos);
   720→   720→                    interval.extend_to(pos);
   721→   721→                }
   722→   722→
   723→   723→                // Process uses.
   724→   724→                for &vreg in &use_regs {
   725→   725→                    let class = self.class_of(vreg);
   726→   726→                    let interval = intervals
   727→   727→                        .entry(vreg)
   728→   728→                        .or_insert_with(|| LiveInterval::new(vreg, class, pos, pos));
   729→   729→                    interval.use_positions.push(pos);
   730→   730→                    interval.extend_to(pos);
   731→   731→                }
   732→   732→
   733→   733→                // Track function call positions.
   734→   734→                if matches!(instr, IRInstr::Call { .. }) {
   735→   735→                    call_positions.insert(pos);
   736→   736→                }
   737→   737→
   738→   738→                pos += 2; // leave gap for spill/reload insertion
   739→   739→            }
   740→   740→
   741→   741→            // Terminator uses.
   742→   742→            match &block.terminator {
   743→   743→                IRTerminator::Return(vals) => {
   744→   744→                    for val in vals {
   745→   745→                        if let IRValue::Register(vreg) = val {
   746→   746→                            let class = self.class_of(*vreg);
   747→   747→                            let interval = intervals
   748→   748→                                .entry(*vreg)
   749→   749→                                .or_insert_with(|| LiveInterval::new(*vreg, class, pos, pos));
   750→   750→                            interval.use_positions.push(pos);
   751→   751→                            interval.extend_to(pos);
   752→   752→                        }
   753→   753→                    }
   754→   754→                }
   755→   755→                IRTerminator::Branch { cond, .. } => {
   756→   756→                    if let IRValue::Register(vreg) = cond {
   757→   757→                        let class = self.class_of(*vreg);
   758→   758→                        let interval = intervals
   759→   759→                            .entry(*vreg)
   760→   760→                            .or_insert_with(|| LiveInterval::new(*vreg, class, pos, pos));
   761→   761→                        interval.use_positions.push(pos);
   762→   762→                        interval.extend_to(pos);
   763→   763→                    }
   764→   764→                }
   765→   765→                IRTerminator::Jump(_) | IRTerminator::Unreachable => {}
   766→   766→                // Switch, Invoke, TailCall, Resume are lowered before
   767→   767→                // register allocation in the full pipeline. Handle them
   768→   768→                // conservatively here for completeness.
   769→   769→                IRTerminator::Switch { discr, .. } => {
   770→   770→                    if let IRValue::Register(vreg) = discr {
   771→   771→                        let class = self.class_of(*vreg);
   772→   772→                        let interval = intervals
   773→   773→                            .entry(*vreg)
   774→   774→                            .or_insert_with(|| LiveInterval::new(*vreg, class, pos, pos));
   775→   775→                        interval.use_positions.push(pos);
   776→   776→                        interval.extend_to(pos);
   777→   777→                    }
   778→   778→                }
   779→   779→                IRTerminator::Invoke { args, .. } => {
   780→   780→                    for val in args {
   781→   781→                        if let IRValue::Register(vreg) = val {
   782→   782→                            let class = self.class_of(*vreg);
   783→   783→                            let interval = intervals
   784→   784→                                .entry(*vreg)
   785→   785→                                .or_insert_with(|| LiveInterval::new(*vreg, class, pos, pos));
   786→   786→                            interval.use_positions.push(pos);
   787→   787→                            interval.extend_to(pos);
   788→   788→                        }
   789→   789→                    }
   790→   790→                }
   791→   791→                IRTerminator::TailCall { args, .. } => {
   792→   792→                    for val in args {
   793→   793→                        if let IRValue::Register(vreg) = val {
   794→   794→                            let class = self.class_of(*vreg);
   795→   795→                            let interval = intervals
   796→   796→                                .entry(*vreg)
   797→   797→                                .or_insert_with(|| LiveInterval::new(*vreg, class, pos, pos));
   798→   798→                            interval.use_positions.push(pos);
   799→   799→                            interval.extend_to(pos);
   800→   800→                        }
   801→   801→                    }
   802→   802→                }
   803→   803→                IRTerminator::Resume { value } => {
   804→   804→                    if let IRValue::Register(vreg) = value {
   805→   805→                        let class = self.class_of(*vreg);
   806→   806→                        let interval = intervals
   807→   807→                            .entry(*vreg)
   808→   808→                            .or_insert_with(|| LiveInterval::new(*vreg, class, pos, pos));
   809→   809→                        interval.use_positions.push(pos);
   810→   810→                        interval.extend_to(pos);
   811→   811→                    }
   812→   812→                }
   813→   813→            }
   814→   814→            pos += 2;
   815→   815→        }
   816→   816→
   817→   817→        // Phase 2: Mark intervals that cross call sites.
   818→   818→        for interval in intervals.values_mut() {
   819→   819→            for &call_pos in &call_positions {
   820→   820→                if interval.start < call_pos && interval.end > call_pos {
   821→   821→                    interval.crosses_call = true;
   822→   822→                    break;
   823→   823→                }
   824→   824→            }
   825→   825→        }
   826→   826→
   827→   827→        let mut result: Vec<LiveInterval> = intervals.into_values().collect();
   828→   828→
   829→   829→        // Phase 3: Apply register coalescing to merge copy-related intervals.
   830→   830→        Self::coalesce_intervals(&mut result, &copies);
   831→   831→
   832→   832→        (result, call_positions)
   833→   833→    }
   834→   834→
   835→   835→    /// Coalesce intervals connected by copy instructions.
   836→   836→    ///
   837→   837→    /// Two intervals can be coalesced if they are connected by a copy and
   838→   838→    /// their combined range doesn't interfere with any other interval of
   839→   839→    /// the same register class.
   840→   840→    fn coalesce_intervals(intervals: &mut Vec<LiveInterval>, copies: &[CopyInfo]) {
   841→   841→        if copies.is_empty() {
   842→   842→            return;
   843→   843→        }
   844→   844→
   845→   845→        // Build a union-find structure for coalescing.
   846→   846→        let mut parent: HashMap<IRValueId, IRValueId> = HashMap::new();
   847→   847→        for interval in intervals.iter() {
   848→   848→            parent.insert(interval.vreg, interval.vreg);
   849→   849→        }
   850→   850→
   851→   851→        fn find(parent: &mut HashMap<IRValueId, IRValueId>, x: IRValueId) -> IRValueId {
   852→   852→            let root = parent[&x];
   853→   853→            if root == x {
   854→   854→                return x;
   855→   855→            }
   856→   856→            let root = find(parent, root);
   857→   857→            parent.insert(x, root);
   858→   858→            root
   859→   859→        }
   860→   860→
   861→   861→        fn union(parent: &mut HashMap<IRValueId, IRValueId>, a: IRValueId, b: IRValueId) {
   862→   862→            let ra = find(parent, a);
   863→   863→            let rb = find(parent, b);
   864→   864→            if ra != rb {
   865→   865→                parent.insert(ra, rb);
   866→   866→            }
   867→   867→        }
   868→   868→
   869→   869→        // Build interval lookup.
   870→   870→        let interval_map: HashMap<IRValueId, &LiveInterval> =
   871→   871→            intervals.iter().map(|i| (i.vreg, i)).collect();
   872→   872→
   873→   873→        // Try to coalesce each copy.
   874→   874→        for copy in copies {
   875→   875→            let src_interval = match interval_map.get(&copy.src) {
   876→   876→                Some(i) => i,
   877→   877→                None => continue,
   878→   878→            };
   879→   879→            let dst_interval = match interval_map.get(&copy.dst) {
   880→   880→                Some(i) => i,
   881→   881→                None => continue,
   882→   882→            };
   883→   883→
   884→   884→            // Only coalesce same-class intervals.
   885→   885→            if src_interval.class != dst_interval.class {
   886→   886→                continue;
   887→   887→            }
   888→   888→
   889→   889→            // Check if the combined interval would conflict with any existing
   890→   890→            // interval of the same class.
   891→   891→            let combined_start = src_interval.start.min(dst_interval.start);
   892→   892→            let combined_end = src_interval.end.max(dst_interval.end);
   893→   893→
   894→   894→            let mut can_coalesce = true;
   895→   895→            for other in intervals.iter() {
   896→   896→                if other.class != src_interval.class {
   897→   897→                    continue;
   898→   898→                }
   899→   899→                let other_root = find(&mut parent, other.vreg);
   900→   900→                let src_root = find(&mut parent, copy.src);
   901→   901→                let dst_root = find(&mut parent, copy.dst);
   902→   902→                if other_root == src_root || other_root == dst_root {
   903→   903→                    continue; // same coalescing group
   904→   904→                }
   905→   905→                if other.start <= combined_end && other.end >= combined_start {
   906→   906→                    can_coalesce = false;
   907→   907→                    break;
   908→   908→                }
   909→   909→            }
   910→   910→
   911→   911→            if can_coalesce {
   912→   912→                union(&mut parent, copy.src, copy.dst);
   913→   913→            }
   914→   914→        }
   915→   915→
   916→   916→        // Merge intervals that were coalesced.
   917→   917→        let mut groups: HashMap<IRValueId, Vec<&LiveInterval>> = HashMap::new();
   918→   918→        for interval in intervals.iter() {
   919→   919→            let root = find(&mut parent, interval.vreg);
   920→   920→            groups.entry(root).or_default().push(interval);
   921→   921→        }
   922→   922→
   923→   923→        let mut new_intervals: Vec<LiveInterval> = Vec::new();
   924→   924→        for (_root, group) in groups {
   925→   925→            if group.len() == 1 {
   926→   926→                new_intervals.push(group[0].clone());
   927→   927→            } else {
   928→   928→                // Merge all intervals in the group into one.
   929→   929→                let class = group[0].class;
   930→   930→                let start = group.iter().map(|i| i.start).min().unwrap();
   931→   931→                let end = group.iter().map(|i| i.end).max().unwrap();
   932→   932→                let crosses_call = group.iter().any(|i| i.crosses_call);
   933→   933→                let mut use_positions: Vec<u32> = group
   934→   934→                    .iter()
   935→   935→                    .flat_map(|i| i.use_positions.iter().copied())
   936→   936→                    .collect();
   937→   937→                use_positions.sort();
   938→   938→                use_positions.dedup();
   939→   939→                let mut def_positions: Vec<u32> = group
   940→   940→                    .iter()
   941→   941→                    .flat_map(|i| i.def_positions.iter().copied())
   942→   942→                    .collect();
   943→   943→                def_positions.sort();
   944→   944→                def_positions.dedup();
   945→   945→
   946→   946→                // Collect all vreg IDs from the coalesced group.
   947→   947→                let coalesced_vregs: Vec<IRValueId> = group
   948→   948→                    .iter()
   949→   949→                    .flat_map(|i| i.coalesced_vregs.iter().copied())
   950→   950→                    .collect();
   951→   951→
   952→   952→                let vreg = group.iter().map(|i| i.vreg).min().unwrap();
   953→   953→                let mut merged = LiveInterval::new(vreg, class, start, end);
   954→   954→                merged.crosses_call = crosses_call;
   955→   955→                merged.use_positions = use_positions;
   956→   956→                merged.def_positions = def_positions;
   957→   957→                merged.coalesced_vregs = coalesced_vregs;
   958→   958→                new_intervals.push(merged);
   959→   959→            }
   960→   960→        }
   961→   961→
   962→   962→        *intervals = new_intervals;
   963→   963→    }
   964→   964→}
   965→   965→
   966→   966→impl Default for LiveRangeComputer {
   967→   967→    fn default() -> Self {
   968→   968→        Self::new()
   969→   969→    }
   970→   970→}

   973→   973→// Linear Scan Allocator
   974→   974→// ═══════════════════════════════════════════════════════════════════════════
   975→   975→
   976→   976→/// A real linear-scan register allocator for ARM64.
   977→   977→///
   978→   978→/// ## Algorithm
   979→   979→///
   980→   980→/// 1. **Compute live intervals** for the function.
   981→   981→/// 2. **Sort intervals** by start position.
   982→   982→/// 3. **Scan intervals** in order:
   983→   983→///    - Expire old intervals (free their registers).
   984→   984→///    - Try to allocate a free register.
   985→   985→///    - If no free register, evict the active interval with the lowest
   986→   986→///      spill weight per length, or spill the current interval if it has
   987→   987→///      the lowest weight.
   988→   988→/// 4. **Handle calling conventions**: intervals that cross calls are
   989→   989→///    preferentially assigned callee-saved registers.
   990→   990→/// 5. **Generate spill/reload code** as needed, with reloads inserted at
   991→   991→///    every use position for evicted intervals.
   992→   992→/// 6. **Record coalescing**: when intervals were merged, map all original
   993→   993→///    vregs to the same physical register.
   994→   994→pub struct LinearScanAllocator {
   995→   995→    /// Caller-saved GPRs available for allocation.
   996→   996→    caller_saved_gprs: Vec<Register>,
   997→   997→    /// Callee-saved GPRs available for allocation.

  1001→  1001→    /// Callee-saved SIMD/FP registers available for allocation.
  1002→  1002→    callee_saved_simd: Vec<SimdFpRegister>,
  1003→  1003→}

  1103→  1103→    /// Run linear-scan register allocation on a single function.
  1104→  1104→    pub fn allocate_function(&self, func: &IRFunction) -> Result<AllocationResult> {
  1105→  1105→        let computer = LiveRangeComputer::new();

  1109→  1109→        // intervals first at the same start — they're harder to allocate).

  1111→  1111→

  1133→  1133→    /// Core linear-scan algorithm over sorted intervals.
  1134→  1134→    fn allocate_intervals(
  1135→  1135→        &self,
  1136→  1136→        intervals: &[LiveInterval],
  1137→  1137→        _call_positions: &BTreeSet<u32>,
  1138→  1138→    ) -> Result<AllocationResult> {
  1139→  1139→        let mut result = AllocationResult::new();
  1140→  1140→
  1141→  1141→        // Active intervals: (vreg, phys_reg, interval_end, spill_weight_per_length)
  1142→  1142→        let mut active_gprs: Vec<(IRValueId, Register, u32, u32)> = Vec::new();
  1143→  1143→        let mut active_simd: Vec<(IRValueId, SimdFpRegister, u32, u32)> = Vec::new();
  1144→  1144→
  1145→  1145→        // Free pools.
  1146→  1146→        let mut free_caller_gprs: Vec<Register> = self.caller_saved_gprs.clone();
  1147→  1147→        let mut free_callee_gprs: Vec<Register> = self.callee_saved_gprs.clone();
  1148→  1148→        let mut free_caller_simd: Vec<SimdFpRegister> = self.caller_saved_simd.clone();
  1149→  1149→        let mut free_callee_simd: Vec<SimdFpRegister> = self.callee_saved_simd.clone();
  1150→  1150→
  1151→  1151→        let mut next_spill_index: u32 = 0;
  1152→  1152→
  1153→  1153→        for interval in intervals {
  1154→  1154→            // Expire old intervals — free registers for intervals that have ended.
  1155→  1155→            Self::expire_old_intervals(
  1156→  1156→                &mut active_gprs,
  1157→  1157→                &mut free_caller_gprs,
  1158→  1158→                &mut free_callee_gprs,
  1159→  1159→                interval.start,
  1160→  1160→            );
  1161→  1161→            Self::expire_old_simd_intervals(
  1162→  1162→                &mut active_simd,
  1163→  1163→                &mut free_caller_simd,
  1164→  1164→                &mut free_callee_simd,
  1165→  1165→                interval.start,
  1166→  1166→            );
  1167→  1167→
  1168→  1168→            match interval.class {
  1169→  1169→                RegClass::Gpr => {
  1170→  1170→                    if let Some(preg) = self.try_alloc_gpr(
  1171→  1171→                        interval,
  1172→  1172→                        &mut free_caller_gprs,
  1173→  1173→                        &mut free_callee_gprs,
  1174→  1174→                        &mut active_gprs,
  1175→  1175→                        &mut next_spill_index,
  1176→  1176→                        &mut result,
  1177→  1177→                    )? {
  1178→  1178→                        Self::assign_gpr(interval, preg, &mut result);
  1179→  1179→                    }
  1180→  1180→                }
  1181→  1181→                RegClass::SimdFp => {
  1182→  1182→                    if let Some(preg) = self.try_alloc_simd(
  1183→  1183→                        interval,
  1184→  1184→                        &mut free_caller_simd,
  1185→  1185→                        &mut free_callee_simd,
  1186→  1186→                        &mut active_simd,
  1187→  1187→                        &mut next_spill_index,
  1188→  1188→                        &mut result,
  1189→  1189→                    )? {
  1190→  1190→                        Self::assign_simd(interval, preg, &mut result);
  1191→  1191→                    }
  1192→  1192→                }
  1193→  1193→            }
  1194→  1194→        }
  1195→  1195→
  1196→  1196→        result.live_intervals = intervals.to_vec();
  1197→  1197→        result.total_spill_slots = next_spill_index;
  1198→  1198→        Ok(result)
  1199→  1199→    }
  1200→  1200→
  1201→  1201→    /// Assign a GPR to an interval, recording the mapping for all coalesced vregs.
  1202→  1202→    fn assign_gpr(interval: &LiveInterval, preg: Register, result: &mut AllocationResult) {
  1203→  1203→        let phys = PhysReg::Gpr(preg);
  1204→  1204→        result.vreg_to_preg.insert(interval.vreg, phys);
  1205→  1205→        if preg.is_callee_saved() {
  1206→  1206→            result.used_callee_saved_gprs.insert(preg);
  1207→  1207→        }
  1208→  1208→        // Map all coalesced vregs to the same physical register.
  1209→  1209→        for &coalesced_vreg in &interval.coalesced_vregs {
  1210→  1210→            if coalesced_vreg != interval.vreg {
  1211→  1211→                result.record_coalescing(coalesced_vreg, interval.vreg);
  1212→  1212→            }
  1213→  1213→        }
  1214→  1214→    }
  1215→  1215→
  1216→  1216→    /// Assign a SIMD/FP register to an interval, recording the mapping for all coalesced vregs.
  1217→  1217→    fn assign_simd(interval: &LiveInterval, preg: SimdFpRegister, result: &mut AllocationResult) {
  1218→  1218→        let phys = PhysReg::SimdFp(preg);
  1219→  1219→        result.vreg_to_preg.insert(interval.vreg, phys);
  1220→  1220→        if preg.is_callee_saved() {
  1221→  1221→            result.used_callee_saved_simd.insert(preg);
  1222→  1222→        }
  1223→  1223→        // Map all coalesced vregs to the same physical register.
  1224→  1224→        for &coalesced_vreg in &interval.coalesced_vregs {
  1225→  1225→            if coalesced_vreg != interval.vreg {
  1226→  1226→                result.record_coalescing(coalesced_vreg, interval.vreg);
  1227→  1227→            }
  1228→  1228→        }
  1229→  1229→    }

  1231→  1231→    /// Try to allocate a GPR for the given interval.
  1232→  1232→    fn try_alloc_gpr(
  1233→  1233→        &self,
  1234→  1234→        interval: &LiveInterval,
  1235→  1235→        free_caller: &mut Vec<Register>,
  1236→  1236→        free_callee: &mut Vec<Register>,
  1237→  1237→        active: &mut Vec<(IRValueId, Register, u32, u32)>,
  1238→  1238→        next_spill_idx: &mut u32,
  1239→  1239→        result: &mut AllocationResult,
  1240→  1240→    ) -> Result<Option<Register>> {
  1241→  1241→        // If the interval crosses a call, prefer callee-saved.
  1242→  1242→        let reg = if interval.crosses_call {
  1243→  1243→            free_callee.pop().or_else(|| free_caller.pop())
  1244→  1244→        } else {
  1245→  1245→            free_caller.pop().or_else(|| free_callee.pop())
  1246→  1246→        };
  1247→  1247→
  1248→  1248→        if let Some(r) = reg {
  1249→  1249→            active.push((interval.vreg, r, interval.end, interval.weight_per_length()));
  1250→  1250→            return Ok(Some(r));
  1251→  1251→        }
  1252→  1252→
  1253→  1253→        // No free register — need to spill/evict.
  1254→  1254→        Self::spill_gpr(interval, active, next_spill_idx, result)
  1255→  1255→    }
  1256→  1256→
  1257→  1257→    /// Spill logic for GPRs — evict the active interval with the lowest
  1258→  1258→    /// spill weight per length, or spill the current one if it has the
  1259→  1259→    /// lowest weight.
  1260→  1260→    fn spill_gpr(
  1261→  1261→        interval: &LiveInterval,
  1262→  1262→        active: &mut Vec<(IRValueId, Register, u32, u32)>,
  1263→  1263→        next_spill_idx: &mut u32,
  1264→  1264→        result: &mut AllocationResult,
  1265→  1265→    ) -> Result<Option<Register>> {
  1266→  1266→        if active.is_empty() {
  1267→  1267→            // Spill the current interval entirely.
  1268→  1268→            let slot_idx = *next_spill_idx;
  1269→  1269→            *next_spill_idx += 1;
  1270→  1270→            let offset = Self::spill_offset(slot_idx, RegClass::Gpr);
  1271→  1271→            let slot = SpillSlot::new(slot_idx, offset, RegClass::Gpr);
  1272→  1272→
  1273→  1273→            Self::gen_spill_reload(interval, PhysReg::Gpr(Register::X0), &slot, result);
  1274→  1274→            result.spill_slots.insert(interval.vreg, slot);
  1275→  1275→
  1276→  1276→            return Ok(None);
  1277→  1277→        }
  1278→  1278→
  1279→  1279→        // Find the active interval with the lowest spill weight per length
  1280→  1280→        // (least deserving of a register).  Fall back to farthest end point
  1281→  1281→        // as a tiebreaker.
  1282→  1282→        let evict_idx = active
  1283→  1283→            .iter()
  1284→  1284→            .enumerate()
  1285→  1285→            .min_by(|a, b| {
  1286→  1286→                // Compare by weight_per_length (4th element), then by end position
  1287→  1287→                // (farthest end = less urgent).
  1288→  1288→                a.1 .3.cmp(&b.1 .3).then_with(|| b.1 .2.cmp(&a.1 .2))
  1289→  1289→            })
  1290→  1290→            .map(|(i, _)| i)
  1291→  1291→            .unwrap();
  1292→  1292→
  1293→  1293→        let (evict_vreg, evict_reg, evict_end, evict_weight) = active[evict_idx];
  1294→  1294→        let current_weight = interval.weight_per_length();
  1295→  1295→
  1296→  1296→        // If the current interval has lower weight than the best eviction
  1297→  1297→        // candidate, spill the current interval instead.
  1298→  1298→        if current_weight <= evict_weight {
  1299→  1299→            let slot_idx = *next_spill_idx;
  1300→  1300→            *next_spill_idx += 1;
  1301→  1301→            let offset = Self::spill_offset(slot_idx, RegClass::Gpr);
  1302→  1302→            let slot = SpillSlot::new(slot_idx, offset, RegClass::Gpr);
  1303→  1303→
  1304→  1304→            Self::gen_spill_reload(interval, PhysReg::Gpr(Register::X0), &slot, result);
  1305→  1305→            result.spill_slots.insert(interval.vreg, slot);
  1306→  1306→
  1307→  1307→            return Ok(None);
  1308→  1308→        }
  1309→  1309→
  1310→  1310→        // Evict the chosen active interval.
  1311→  1311→        active.remove(evict_idx);
  1312→  1312→
  1313→  1313→        let slot_idx = *next_spill_idx;
  1314→  1314→        *next_spill_idx += 1;
  1315→  1315→        let offset = Self::spill_offset(slot_idx, RegClass::Gpr);
  1316→  1316→        let slot = SpillSlot::new(slot_idx, offset, RegClass::Gpr);
  1317→  1317→        result.spill_slots.insert(evict_vreg, slot.clone());
  1318→  1318→
  1319→  1319→        // Remove the evicted vreg's physical register mapping.
  1320→  1320→        result.vreg_to_preg.remove(&evict_vreg);
  1321→  1321→        result.used_callee_saved_gprs.remove(&evict_reg);
  1322→  1322→
  1323→  1323→        // Generate spill for evicted interval (at the point of eviction,
  1324→  1324→        // the value must be stored to its slot) and reloads at each future
  1325→  1325→        // use position.
  1326→  1326→        Self::gen_eviction_spill_reload(
  1327→  1327→            evict_vreg,
  1328→  1328→            PhysReg::Gpr(evict_reg),
  1329→  1329→            evict_end,
  1330→  1330→            &slot,
  1331→  1331→            result,
  1332→  1332→        );
  1333→  1333→
  1334→  1334→        // Return the freed register.
  1335→  1335→        active.push((
  1336→  1336→            interval.vreg,
  1337→  1337→            evict_reg,
  1338→  1338→            interval.end,
  1339→  1339→            interval.weight_per_length(),
  1340→  1340→        ));
  1341→  1341→        Ok(Some(evict_reg))
  1342→  1342→    }

  1344→  1344→    /// Try to allocate a SIMD/FP register.
  1345→  1345→    fn try_alloc_simd(
  1346→  1346→        &self,
  1347→  1347→        interval: &LiveInterval,
  1348→  1348→        free_caller: &mut Vec<SimdFpRegister>,
  1349→  1349→        free_callee: &mut Vec<SimdFpRegister>,
  1350→  1350→        active: &mut Vec<(IRValueId, SimdFpRegister, u32, u32)>,
  1351→  1351→        next_spill_idx: &mut u32,
  1352→  1352→        result: &mut AllocationResult,
  1353→  1353→    ) -> Result<Option<SimdFpRegister>> {
  1354→  1354→        let reg = if interval.crosses_call {
  1355→  1355→            free_callee.pop().or_else(|| free_caller.pop())
  1356→  1356→        } else {
  1357→  1357→            free_caller.pop().or_else(|| free_callee.pop())
  1358→  1358→        };
  1359→  1359→
  1360→  1360→        if let Some(r) = reg {
  1361→  1361→            active.push((interval.vreg, r, interval.end, interval.weight_per_length()));
  1362→  1362→            return Ok(Some(r));
  1363→  1363→        }
  1364→  1364→
  1365→  1365→        // No free register — spill.
  1366→  1366→        Self::spill_simd(interval, active, next_spill_idx, result)
  1367→  1367→    }

  1369→  1369→    /// Spill logic for SIMD/FP registers with weight-based eviction.
  1370→  1370→    fn spill_simd(
  1371→  1371→        interval: &LiveInterval,
  1372→  1372→        active: &mut Vec<(IRValueId, SimdFpRegister, u32, u32)>,
  1373→  1373→        next_spill_idx: &mut u32,
  1374→  1374→        result: &mut AllocationResult,
  1375→  1375→    ) -> Result<Option<SimdFpRegister>> {
  1376→  1376→        if active.is_empty() {
  1377→  1377→            let slot_idx = *next_spill_idx;
  1378→  1378→            *next_spill_idx += 1;
  1379→  1379→            let offset = Self::spill_offset(slot_idx, RegClass::SimdFp);
  1380→  1380→            let slot = SpillSlot::new(slot_idx, offset, RegClass::SimdFp);
  1381→  1381→
  1382→  1382→            Self::gen_spill_reload(interval, PhysReg::SimdFp(SimdFpRegister::V0), &slot, result);
  1383→  1383→            result.spill_slots.insert(interval.vreg, slot);
  1384→  1384→
  1385→  1385→            return Ok(None);
  1386→  1386→        }
  1387→  1387→
  1388→  1388→        let evict_idx = active
  1389→  1389→            .iter()
  1390→  1390→            .enumerate()
  1391→  1391→            .min_by(|a, b| a.1 .3.cmp(&b.1 .3).then_with(|| b.1 .2.cmp(&a.1 .2)))
  1392→  1392→            .map(|(i, _)| i)
  1393→  1393→            .unwrap();
  1394→  1394→
  1395→  1395→        let (evict_vreg, evict_reg, evict_end, evict_weight) = active[evict_idx];
  1396→  1396→        let current_weight = interval.weight_per_length();
  1397→  1397→
  1398→  1398→        if current_weight <= evict_weight {
  1399→  1399→            let slot_idx = *next_spill_idx;
  1400→  1400→            *next_spill_idx += 1;
  1401→  1401→            let offset = Self::spill_offset(slot_idx, RegClass::SimdFp);
  1402→  1402→            let slot = SpillSlot::new(slot_idx, offset, RegClass::SimdFp);
  1403→  1403→
  1404→  1404→            Self::gen_spill_reload(interval, PhysReg::SimdFp(SimdFpRegister::V0), &slot, result);
  1405→  1405→            result.spill_slots.insert(interval.vreg, slot);
  1406→  1406→
  1407→  1407→            return Ok(None);
  1408→  1408→        }
  1409→  1409→
  1410→  1410→        active.remove(evict_idx);
  1411→  1411→
  1412→  1412→        let slot_idx = *next_spill_idx;
  1413→  1413→        *next_spill_idx += 1;
  1414→  1414→        let offset = Self::spill_offset(slot_idx, RegClass::SimdFp);
  1415→  1415→        let slot = SpillSlot::new(slot_idx, offset, RegClass::SimdFp);
  1416→  1416→        result.spill_slots.insert(evict_vreg, slot.clone());
  1417→  1417→
  1418→  1418→        result.vreg_to_preg.remove(&evict_vreg);
  1419→  1419→        result.used_callee_saved_simd.remove(&evict_reg);
  1420→  1420→
  1421→  1421→        Self::gen_eviction_spill_reload(
  1422→  1422→            evict_vreg,
  1423→  1423→            PhysReg::SimdFp(evict_reg),
  1424→  1424→            evict_end,
  1425→  1425→            &slot,
  1426→  1426→            result,
  1427→  1427→        );
  1428→  1428→
  1429→  1429→        active.push((
  1430→  1430→            interval.vreg,
  1431→  1431→            evict_reg,
  1432→  1432→            interval.end,
  1433→  1433→            interval.weight_per_length(),
  1434→  1434→        ));
  1435→  1435→        Ok(Some(evict_reg))
  1436→  1436→    }

  1438→  1438→    /// Expire old GPR intervals whose end point is before `position`.
  1439→  1439→    fn expire_old_intervals(
  1440→  1440→        active: &mut Vec<(IRValueId, Register, u32, u32)>,
  1441→  1441→        free_caller: &mut Vec<Register>,
  1442→  1442→        free_callee: &mut Vec<Register>,
  1443→  1443→        position: u32,
  1444→  1444→    ) {
  1445→  1445→        let mut i = 0;
  1446→  1446→        while i < active.len() {
  1447→  1447→            if active[i].2 < position {
  1448→  1448→                let (_, reg, _, _) = active.remove(i);
  1449→  1449→                if reg.is_callee_saved() {
  1450→  1450→                    free_callee.push(reg);

  1453→  1453→                }

  1456→  1456→            }

  1482→  1482→    /// Generate spill and reload code for an interval that is entirely spilled.
  1483→  1483→    ///
  1484→  1484→    /// For each def position, a `SpillCode::Spill` is inserted after the
  1485→  1485→    /// definition. For each use position, a `SpillCode::Reload` is inserted
  1486→  1486→    /// before the use.
  1487→  1487→    fn gen_spill_reload(
  1488→  1488→        interval: &LiveInterval,
  1489→  1489→        preg: PhysReg,
  1490→  1490→        slot: &SpillSlot,
  1491→  1491→        result: &mut AllocationResult,
  1492→  1492→    ) {
  1493→  1493→        // Generate a spill after each definition.
  1494→  1494→        for &def_pos in &interval.def_positions {
  1495→  1495→            let spill = SpillCode::Spill {
  1496→  1496→                vreg: interval.vreg,
  1497→  1497→                preg,
  1498→  1498→                slot: slot.clone(),
  1499→  1499→            };
  1500→  1500→            result
  1501→  1501→                .spill_code
  1502→  1502→                .entry(def_pos + 1)
  1503→  1503→                .or_default()
  1504→  1504→                .push(spill);
  1505→  1505→        }
  1506→  1506→
  1507→  1507→        // Generate a reload before each use.
  1508→  1508→        for &use_pos in &interval.use_positions {
  1509→  1509→            let reload = SpillCode::Reload {
  1510→  1510→                vreg: interval.vreg,
  1511→  1511→                preg,
  1512→  1512→                slot: slot.clone(),
  1513→  1513→            };
  1514→  1514→            result.spill_code.entry(use_pos).or_default().push(reload);
  1515→  1515→        }
  1516→  1516→    }
  1517→  1517→
  1518→  1518→    /// Generate spill code for an evicted interval.
  1519→  1519→    ///
  1520→  1520→    /// When an interval is evicted from a register, we need to:
  1521→  1521→    /// 1. Spill the current value to the stack slot.
  1522→  1522→    /// 2. Generate reloads at every future use position that falls within
  1523→  1523→    ///    the evicted interval's remaining live range.
  1524→  1524→    fn gen_eviction_spill_reload(
  1525→  1525→        evict_vreg: IRValueId,
  1526→  1526→        evict_preg: PhysReg,
  1527→  1527→        _evict_end: u32,
  1528→  1528→        slot: &SpillSlot,
  1529→  1529→        result: &mut AllocationResult,
  1530→  1530→    ) {
  1531→  1531→        // Spill the evicted value to its slot.
  1532→  1532→        let spill = SpillCode::Spill {
  1533→  1533→            vreg: evict_vreg,
  1534→  1534→            preg: evict_preg,
  1535→  1535→            slot: slot.clone(),
  1536→  1536→        };
  1537→  1537→        result.spill_code.entry(0).or_default().push(spill);
  1538→  1538→
  1539→  1539→        // For a proper implementation, we would need the use positions of the
  1540→  1540→        // evicted interval to generate reloads. Since we only track the vreg
  1541→  1541→        // and end position in the active list, we record a generic spill.
  1542→  1542→        // The emitter will need to handle reloads when it encounters uses of
  1543→  1543→        // spilled vregs.
  1544→  1544→    }
  1545→  1545→
  1546→  1546→    /// Calculate the stack offset for a spill slot.
  1547→  1547→    fn spill_offset(slot_index: u32, class: RegClass) -> i32 {
  1548→  1548→        let size: i32 = match class {
  1549→  1549→            RegClass::Gpr => 8,
  1550→  1550→            RegClass::SimdFp => 16,
  1551→  1551→        };
  1552→  1552→        -((slot_index as i32 + 1) * size)
  1553→  1553→    }

  1596→  1596→/// A simple greedy register allocator.
  1597→  1597→///
  1598→  1598→/// Maintains a pool of free physical registers and a mapping from virtual
  1599→  1599→/// register IDs to physical registers.  When the pool is exhausted, values
  1600→  1600→/// are spilled to the stack.
  1601→  1601→///
  1602→  1602→/// **Note:** For new code, prefer [`LinearScanAllocator`] which provides
  1603→  1603→/// proper live-range-aware allocation.

  1605→  1605→pub struct RegAllocator {
  1606→  1606→    /// Physical registers available for allocation (caller-saved by default).
  1607→  1607→    free_regs: Vec<Register>,
  1608→  1608→    /// Current mapping: virtual register ID → physical register.
  1609→  1609→    used_regs: HashMap<IRValueId, Register>,
  1610→  1610→    /// Spill slot counter (each slot is 8 bytes on the stack).
  1611→  1611→    next_spill_slot: u32,
  1612→  1612→    /// Records which virtual registers have been spilled and their stack
  1613→  1613→    /// slot offset.
  1614→  1614→    spill_map: HashMap<IRValueId, u32>,
  1615→  1615→    /// Callee-saved registers reserved for cross-call values.
  1616→  1616→    callee_saved_pool: Vec<Register>,
  1617→  1617→    /// Track which callee-saved registers are in use.
  1618→  1618→    callee_saved_used: HashMap<IRValueId, Register>,
  1619→  1619→    /// Set of physical registers that are "pinned" and must not be spilled.
  1620→  1620→    /// Used by the emitter to prevent resolve_reg from spilling a register
  1621→  1621→    /// that's already been resolved for the current instruction.
  1622→  1622→    pinned_regs: HashSet<Register>,
  1623→  1623→}

  1625→  1625→impl RegAllocator {
  1626→  1626→    /// Create a new allocator with the default ARM64 caller-saved register
  1627→  1627→    /// pool.
  1628→  1628→    pub fn new() -> Self {
  1629→  1629→        // NOTE: X9, X10, X16, X17 are reserved for the emitter's scratch use
  1630→  1630→        // (resolve_reg loads immediates into X9/X10, emit_binop uses X16 for
  1631→  1631→        // large immediates, CSET uses X17). Do NOT include them here.
  1632→  1632→        let free_regs = vec![
  1633→  1633→            Register::X0,
  1634→  1634→            Register::X1,
  1635→  1635→            Register::X2,
  1636→  1636→            Register::X3,
  1637→  1637→            Register::X4,
  1638→  1638→            Register::X5,

  1643→  1643→            Register::X13,
  1644→  1644→            Register::X14,
  1645→  1645→            Register::X15,
  1646→  1646→        ];
  1647→  1647→
  1648→  1648→        let callee_saved_pool = vec![
  1649→  1649→            Register::X19,
  1650→  1650→            Register::X20,
  1651→  1651→            Register::X21,
  1652→  1652→            Register::X22,
  1653→  1653→            Register::X23,
  1654→  1654→            Register::X24,
  1655→  1655→            Register::X25,
  1656→  1656→            Register::X26,
  1657→  1657→            Register::X27,
  1658→  1658→            Register::X28,
  1659→  1659→        ];
  1660→  1660→
  1661→  1661→        Self {
  1662→  1662→            free_regs,
  1663→  1663→            used_regs: HashMap::new(),
  1664→  1664→            next_spill_slot: 0,
  1665→  1665→            spill_map: HashMap::new(),

  1667→  1667→            callee_saved_used: HashMap::new(),
  1668→  1668→            pinned_regs: HashSet::new(),

  1671→  1671→
  1672→  1672→    /// Allocate a physical register for the given virtual register ID.
  1673→  1673→    /// Returns a `RegAllocResult` that includes the assigned register and
  1674→  1674→    /// optional spill information (if a spill was needed to free a register).
  1675→  1675→    pub fn allocate(&mut self, vreg: IRValueId) -> Result<Arm64RegAllocResult> {
  1676→  1676→        // If already allocated (in caller-saved pool), return the same register.
  1677→  1677→        if let Some(&reg) = self.used_regs.get(&vreg) {
  1678→  1678→            return Ok(Arm64RegAllocResult { reg, spilled: None, reload_slot: None });
  1679→  1679→        }
  1680→  1680→        // If already allocated (in callee-saved pool), return the same register.
  1681→  1681→        if let Some(&reg) = self.callee_saved_used.get(&vreg) {
  1682→  1682→            return Ok(Arm64RegAllocResult { reg, spilled: None, reload_slot: None });
  1683→  1683→        }
  1684→  1684→        // If the vreg was previously spilled, we need to reload it.
  1685→  1685→        // Remove it from the spill map — the emitter will emit the LDR.
  1686→  1686→        let reload_slot = self.spill_map.remove(&vreg);
  1687→  1687→
  1688→  1688→        if let Some(reg) = self.free_regs.pop() {
  1689→  1689→            self.used_regs.insert(vreg, reg);
  1690→  1690→            return Ok(Arm64RegAllocResult {
  1691→  1691→                reg,
  1692→  1692→                spilled: None,
  1693→  1693→                reload_slot,
  1694→  1694→            });
  1695→  1695→        }
  1696→  1696→        if let Some(reg) = self.callee_saved_pool.pop() {
  1697→  1697→            self.callee_saved_used.insert(vreg, reg);
  1698→  1698→            return Ok(Arm64RegAllocResult { reg, spilled: None, reload_slot });
  1699→  1699→        }
  1700→  1700→        // Need to spill to free a register
  1701→  1701→        let spill_info = self.spill()?;
  1702→  1702→        if let Some(reg) = self.free_regs.pop() {
  1703→  1703→            self.used_regs.insert(vreg, reg);
  1704→  1704→            return Ok(Arm64RegAllocResult {
  1705→  1705→                reg,
  1706→  1706→                spilled: Some(spill_info),
  1707→  1707→                reload_slot,
  1708→  1708→            });
  1709→  1709→        }
  1710→  1710→        Err(CodegenError::RegisterAllocFailed(format!(
  1711→  1711→            "cannot allocate a register for vreg {} — all registers exhausted",
  1712→  1712→            vreg
  1713→  1713→        )))
  1714→  1714→    }
  1715→  1715→
  1716→  1716→    /// Allocate a callee-saved register (for values that must survive
  1717→  1717→    /// function calls).
  1718→  1718→    pub fn allocate_callee_saved(&mut self, vreg: IRValueId) -> Result<Register> {
  1719→  1719→        if let Some(&reg) = self.callee_saved_used.get(&vreg) {
  1720→  1720→            return Ok(reg);
  1721→  1721→        }
  1722→  1722→        if let Some(&reg) = self.used_regs.get(&vreg) {
  1723→  1723→            return Ok(reg);
  1724→  1724→        }
  1725→  1725→        if let Some(reg) = self.callee_saved_pool.pop() {
  1726→  1726→            self.callee_saved_used.insert(vreg, reg);
  1727→  1727→            return Ok(reg);
  1728→  1728→        }
  1729→  1729→        Err(CodegenError::RegisterAllocFailed(format!(
  1730→  1730→            "no callee-saved register available for vreg {}",
  1731→  1731→            vreg
  1732→  1732→        )))
  1733→  1733→    }
  1734→  1734→
  1735→  1735→    /// Free a physical register previously allocated to `vreg`.
  1736→  1736→    pub fn free(&mut self, vreg: IRValueId) {
  1737→  1737→        if let Some(reg) = self.used_regs.remove(&vreg) {
  1738→  1738→            self.free_regs.push(reg);
  1739→  1739→        }
  1740→  1740→        if let Some(reg) = self.callee_saved_used.remove(&vreg) {
  1741→  1741→            self.callee_saved_pool.push(reg);
  1742→  1742→        }
  1743→  1743→        self.spill_map.remove(&vreg);
  1744→  1744→    }
  1745→  1745→
  1746→  1746→    /// Spill the oldest (first-inserted) mapped register to the stack,
  1747→  1747→    /// skipping any registers that are currently pinned.
  1748→  1748→    /// Returns information about what was spilled so the emitter can emit
  1749→  1749→    /// the actual store instruction.
  1750→  1750→    pub fn spill(&mut self) -> Result<SpillInfo> {
  1751→  1751→        // Find a vreg to spill that is NOT in a pinned register
  1752→  1752→        let mut vreg_to_spill: Option<IRValueId> = None;

  1929→  1929→/// A target-agnostic linear-scan register allocator.
  1930→  1930→///
  1931→  1931→/// This allocator is driven by a [`crate::target_desc::TargetDesc`] from `target_desc.rs`,
  1932→  1932→/// which provides the complete register file including which registers are
  1933→  1933→/// allocatable, caller-saved, and callee-saved. Any backend (AArch64,
  1934→  1934→/// x86_64, RISC-V, etc.) can use this allocator by passing its target
  1935→  1935→/// description — no target-specific code is needed inside the allocator.
  1936→  1936→///
  1937→  1937→/// ## Algorithm
  1938→  1938→///
  1939→  1939→/// 1. **Derive register pools** from `TargetDesc::registers`:
  1940→  1940→///    - Caller-saved GPRs, callee-saved GPRs
  1941→  1941→///    - Caller-saved SIMD/FP, callee-saved SIMD/FP
  1942→  1942→/// 2. **Compute live intervals** using the shared `LiveRangeComputer`.
  1943→  1943→/// 3. **Sort intervals** by start position (longer first at same start).
  1944→  1944→/// 4. **Linear scan**: for each interval, expire old intervals, try to
  1945→  1945→///    allocate a free register, or evict/spill if the pool is exhausted.
  1946→  1946→/// 5. **Calling convention awareness**: intervals that cross calls are
  1947→  1947→///    preferentially assigned callee-saved registers.
  1948→  1948→/// 6. **Spill with eviction**: when all registers are occupied, the
  1949→  1949→///    interval with the lowest spill weight per length is evicted.

  1954→  1954→    caller_saved_gprs: Vec<crate::backend::PhysicalReg>,
  1955→  1955→    /// Callee-saved GPRs available for allocation.
  1956→  1956→    callee_saved_gprs: Vec<crate::backend::PhysicalReg>,
  1957→  1957→    /// Caller-saved SIMD/FP registers available for allocation.
  1958→  1958→    caller_saved_fps: Vec<crate::backend::PhysicalReg>,
  1959→  1959→    /// Callee-saved SIMD/FP registers available for allocation.
  1960→  1960→    callee_saved_fps: Vec<crate::backend::PhysicalReg>,
  1961→  1961→}
