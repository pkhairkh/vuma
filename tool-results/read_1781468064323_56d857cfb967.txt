     1→//! # LoongArch 64-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the LoongArch 64-bit target (LP64 ABI).
     4→//! This module provides:
     5→//!
     6→//! - `Gpr` — General-purpose register enum (r0–r31)
     7→//! - `Fpr` — Floating-point register enum (f0–f31)
     8→//! - `Instruction` — LoongArch64 instruction enum with correct encoding
     9→//! - Encoding helpers for all 9 instruction formats (2R, 3R, 4R, 2RI8, 2RI12,
    10→//!   2RI14, 2RI16, 1RI21, I26)
    11→//! - `LoongArch64Backend` — `Backend` implementation that lowers IR to LoongArch64
    12→//!   machine code and emits ELF64 binaries
    13→//!
    14→//! ## LoongArch64 Register Convention (LP64 ABI)
    15→//!
    16→//! | Register(s) | ABI Name | Role                              |
    17→//! |-------------|----------|-----------------------------------|
    18→//! | r0          | zero     | Hardwired zero                    |
    19→//! | r1          | ra       | Return address                    |
    20→//! | r2          | tp       | Thread pointer                    |
    21→//! | r3          | sp       | Stack pointer                     |
    22→//! | r4–r11      | a0–a7    | Argument / return registers       |
    23→//! | r12–r20     | t0–t8    | Caller-saved temporaries          |
    24→//! | r21         | —        | Reserved                          |
    25→//! | r22         | fp       | Frame pointer (callee-saved)      |
    26→//! | r23–r31     | s0–s8    | Callee-saved                      |
    27→//!
    28→//! ## LoongArch64 FP Register Convention (LP64 ABI)
    29→//!
    30→//! | Register(s) | ABI Name | Role                              |
    31→//! |-------------|----------|-----------------------------------|
    32→//! | f0–f7       | fa0–fa7  | FP argument / return registers    |
    33→//! | f8–f23      | ft0–ft15 | Caller-saved temporaries          |
    34→//! | f24–f31     | fs0–fs7  | Callee-saved                      |
    35→//!
    36→//! ## Instruction Formats
    37→//!
    38→//! All instructions are 32 bits, little-endian, with no branch delay slots.
    39→//! Nine formats: 2R, 3R, 4R, 2RI8, 2RI12, 2RI14, 2RI16, 1RI21, I26.
    40→//!
    41→//! ## References
    42→//!
    43→//! - LoongArch Reference Manual, Volume 1: Basic Architecture
    44→//! - <https://loongson.github.io/LoongArch-Documentation/>
    45→

    95→impl Gpr {
    96→    /// Returns the 5-bit encoding index for this register.
    97→    pub fn encoding(&self) -> u32 {
    98→        *self as u32
    99→    }
   100→
   101→    /// Returns `true` if this register is available for register allocation.
   102→    ///
   103→    /// R0 (zero), Ra, Tp, and Sp are reserved.
   104→    pub fn is_allocatable(&self) -> bool {
   105→        !matches!(self, Gpr::R0 | Gpr::Ra | Gpr::Tp | Gpr::Sp)
   106→    }
   107→
   108→    /// Returns `true` if this register is callee-saved (fp, s0–s8).
   109→    pub fn is_callee_saved(&self) -> bool {
   110→        matches!(
   111→            self,
   112→            Gpr::Fp
   113→                | Gpr::S0
   114→                | Gpr::S1
   115→                | Gpr::S2
   116→                | Gpr::S3
   117→                | Gpr::S4
   118→                | Gpr::S5
   119→                | Gpr::S6
   120→                | Gpr::S7
   121→                | Gpr::S8
   122→        )
   123→    }
   124→
   125→    /// Returns `true` if this register is an argument register (a0–a7).

   127→        matches!(
   128→            self,
   129→            Gpr::A0 | Gpr::A1 | Gpr::A2 | Gpr::A3 | Gpr::A4 | Gpr::A5 | Gpr::A6 | Gpr::A7
   130→        )
   131→    }

   323→// ===========================================================================
   324→// Instruction Encoding Helpers
   325→// ===========================================================================
   326→
   327→/// Encode a 2R format instruction.
   328→///
   329→/// Format: `opcode[31:10] | rj[9:5] | rd[4:0]`
   330→fn encode_2r(opcode: u32, rj: u32, rd: u32) -> [u8; 4] {
   331→    let word = ((opcode & 0x3FF_FFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   332→    word.to_le_bytes()
   333→}
   334→
   335→/// Encode a 3R format instruction.
   336→///
   337→/// Format: `opcode[31:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   338→fn encode_3r(opcode: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   339→    let word = ((opcode & 0x1FFFF) << 15) | ((rk & 0x1F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   340→    word.to_le_bytes()
   341→}
   342→
   343→/// Encode a 4R format instruction.
   344→///
   345→/// Format: `opcode[31:20] | ra[19:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   346→#[allow(dead_code)]
   347→fn encode_4r(opcode: u32, ra: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   348→    let word = ((opcode & 0xFFF) << 20)
   349→        | ((ra & 0x1F) << 15)
   350→        | ((rk & 0x1F) << 10)
   351→        | ((rj & 0x1F) << 5)
   352→        | (rd & 0x1F);
   353→    word.to_le_bytes()
   354→}
   355→
   356→/// Encode a 2RI8 format instruction.
   357→///
   358→/// Format: `opcode[31:22] | I8[21:14] | rj[9:5] | rd[4:0]`
   359→#[allow(dead_code)]
   360→fn encode_2ri8(opcode: u32, imm8: u32, rj: u32, rd: u32) -> [u8; 4] {
   361→    let word = ((opcode & 0x3FF) << 22) | ((imm8 & 0xFF) << 14) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   362→    word.to_le_bytes()
   363→}
   364→
   365→/// Encode a reg2i5 format instruction (17-bit opcode).
   366→///
   367→/// Format: `opcode[31:15] | I5[14:10] | rj[9:5] | rd[4:0]`
   368→fn encode_reg2i5(opcode: u32, imm5: u32, rj: u32, rd: u32) -> [u8; 4] {
   369→    let word = ((opcode & 0x1FFFF) << 15) | ((imm5 & 0x1F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   370→    word.to_le_bytes()
   371→}
   372→
   373→/// Encode a reg2i6 format instruction (16-bit opcode).
   374→///
   375→/// Format: `opcode[31:16] | I6[15:10] | rj[9:5] | rd[4:0]`
   376→fn encode_reg2i6(opcode: u32, imm6: u32, rj: u32, rd: u32) -> [u8; 4] {
   377→    let word = ((opcode & 0xFFFF) << 16) | ((imm6 & 0x3F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   378→    word.to_le_bytes()
   379→}
   380→
   381→/// Encode a 2RI12 format instruction.
   382→///
   383→/// Format: `opcode[31:22] | I12[21:10] | rj[9:5] | rd[4:0]`
   384→fn encode_2ri12(opcode: u32, imm12: u32, rj: u32, rd: u32) -> [u8; 4] {
   385→    let word =
   386→        ((opcode & 0x3FF) << 22) | ((imm12 & 0xFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   387→    word.to_le_bytes()
   388→}
   389→
   390→/// Encode a 2RI14 format instruction.
   391→///
   392→/// Format: `opcode[31:24] | I14[23:10] | rj[9:5] | rd[4:0]`
   393→fn encode_2ri14(opcode: u32, imm14: u32, rj: u32, rd: u32) -> [u8; 4] {
   394→    let word =
   395→        ((opcode & 0xFF) << 24) | ((imm14 & 0x3FFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   396→    word.to_le_bytes()
   397→}
   398→
   399→/// Encode a 2RI16 format instruction.
   400→///
   401→/// Format: `opcode[31:26] | I16[25:10] | rj[9:5] | rd[4:0]`
   402→fn encode_2ri16(opcode: u32, imm16: u32, rj: u32, rd: u32) -> [u8; 4] {
   403→    let word =
   404→        ((opcode & 0x3F) << 26) | ((imm16 & 0xFFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   405→    word.to_le_bytes()
   406→}
   407→
   408→/// Encode a 1RI21 format instruction (used for BEQZ, BNEZ).
   409→///
   410→/// Format: `opcode[31:26] | offs21[15:0] in bits[25:10] | rj[9:5] | offs21[20:16] in bits[4:0]`
   411→///
   412→/// Note: the offset bits are split non-linearly — the lower 16 bits go in the
   413→/// higher position (bits[25:10]), and the upper 5 bits go in the lower position
   414→/// (bits[4:0]). The register field `rj` sits between them at bits[9:5].
   415→fn encode_1ri21(opcode: u32, imm21: u32, rj: u32) -> [u8; 4] {
   416→    // 1RI21 format (BEQZ/BNEZ): opcode[31:26] | offs[15:0] at [25:10] | rj[9:5] | offs[20:16] at [4:0]
   417→    let word = ((opcode & 0x3F) << 26)
   418→        | ((imm21 & 0xFFFF) << 10)       // offs[15:0] at bits 25:10
   419→        | ((rj & 0x1F) << 5)             // rj at bits 9:5
   420→        | ((imm21 >> 16) & 0x1F);        // offs[20:16] at bits 4:0
   421→    word.to_le_bytes()
   422→}
   423→
   424→/// Encode an I26 format instruction (used for B, BL).
   425→///
   426→/// Format: `opcode[31:26] | offs26[15:0] in bits[25:10] | offs26[25:16] in bits[9:0]`
   427→///
   428→/// Note: the offset bits are SWAPPED compared to a linear layout.
   429→/// The lower 16 bits of the offset go in the higher position (bits[25:10]),
   430→/// and the upper 10 bits go in the lower position (bits[9:0]).
   431→fn encode_i26(opcode: u32, imm26: u32) -> [u8; 4] {
   432→    let word = ((opcode & 0x3F) << 26)
   433→        | ((imm26 & 0xFFFF) << 10)
   434→        | ((imm26 >> 16) & 0x3FF);
   435→    word.to_le_bytes()
   436→}

   885→impl Instruction {
   886→    /// Encode this instruction into a 4-byte little-endian machine code word.
   887→    ///
   888→    /// Encoding follows the LoongArch ISA Specification.
   889→    pub fn encode(&self) -> [u8; 4] {
   890→        match self {
   891→            // ── Arithmetic (3R) ────────────────────────────────────
   892→            Instruction::AddW { rd, rj, rk } => {
   893→                encode_3r(OPC_ADD_W, rk.encoding(), rj.encoding(), rd.encoding())
   894→            }
   895→            Instruction::AddD { rd, rj, rk } => {
   896→                encode_3r(OPC_ADD_D, rk.encoding(), rj.encoding(), rd.encoding())
   897→            }
   898→            Instruction::SubW { rd, rj, rk } => {
   899→                encode_3r(OPC_SUB_W, rk.encoding(), rj.encoding(), rd.encoding())

  1339→    /// Returns the mnemonic name of this instruction.
  1340→    pub fn mnemonic(&self) -> &'static str {
  1341→        match self {
  1342→            Instruction::AddW { .. } => "add.w",
  1343→            Instruction::AddD { .. } => "add.d",
  1344→            Instruction::SubW { .. } => "sub.w",
  1345→            Instruction::SubD { .. } => "sub.d",

  1591→/// Build a minimal ELF64 binary for LoongArch64 from raw code bytes.
  1592→///
  1593→/// Produces a static executable with two LOAD segments:
  1594→///   1. PF_R | PF_X — .text (code)
  1595→///   2. PF_R | PF_W — .data / stack (writable memory)
  1596→fn build_loongarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1597→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (LoongArch64 typical page size)
  1598→
  1599→    let elf_header_size: u64 = 64;
  1600→    let phdr_size: u64 = 56;
  1601→    let num_phdrs: u64 = 2;
  1602→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1603→    // Page-align the text segment start in the file.
  1604→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1605→    let text_size = code.len() as u64;
  1606→
  1607→    // The data segment starts on the next page after the text.
  1608→    let text_file_end = text_offset + text_size;
  1609→    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1610→    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1611→    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1612→    let entry_point = base_addr + text_offset;
  1613→
  1614→    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1615→
  1616→    // --- e_ident ---
  1617→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1618→    elf.push(2); // ELFCLASS64
  1619→    elf.push(1); // ELFDATA2LSB
  1620→    elf.push(1); // EV_CURRENT
  1621→    elf.push(3); // ELFOSABI_LINUX
  1622→    elf.push(0); // padding
  1623→    elf.extend_from_slice(&[0u8; 7]); // padding
  1624→
  1625→    // --- ELF header fields ---
  1626→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1627→    elf.extend_from_slice(&258u16.to_le_bytes()); // e_machine = EM_LOONGARCH
  1628→    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1629→    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1630→    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1631→    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1632→    elf.extend_from_slice(&0x43u32.to_le_bytes()); // e_flags = 0x43 (LP64D ABI double-float)
  1633→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1634→    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1635→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1636→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1637→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1638→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1639→
  1640→    // --- Program Header 1: LOAD (PF_R | PF_X) — .text ---
  1641→    // p_filesz = actual code size, p_memsz = page-aligned (QEMU needs this for mapping)
  1642→    let text_memsz = ((text_size + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1643→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1644→    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1645→    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1646→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1647→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1648→    elf.extend_from_slice(&text_memsz.to_le_bytes()); // p_filesz (cover full page for QEMU)
  1649→    elf.extend_from_slice(&text_memsz.to_le_bytes()); // p_memsz
  1650→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1651→
  1652→    // --- Program Header 2: LOAD (PF_R | PF_W) — .data / stack ---
  1653→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1654→    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1655→    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1656→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1657→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1658→    elf.extend_from_slice(&data_size.to_le_bytes()); // p_filesz (write zeros so QEMU can mmap)
  1659→    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1660→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1661→
  1662→    // --- .text section ---
  1663→    // Pad to page-aligned text_offset
  1664→    while (elf.len() as u64) < text_offset {
  1665→        elf.push(0);
  1666→    }
  1667→    elf.extend_from_slice(code);
  1668→
  1669→    // --- Pad to data segment offset and write data segment ---
  1670→    while (elf.len() as u64) < data_offset {
  1671→        elf.push(0);
  1672→    }
  1673→    // Write data segment (all zeros, but must exist in file for QEMU mmap)
  1674→    elf.extend_from_slice(&vec![0u8; data_size as usize]);
  1675→
  1676→    elf
  1677→}

  1683→/// Decode a single LoongArch64 32-bit instruction word into a mnemonic string.
  1684→fn decode_loongarch64_instruction(word: u32) -> String {
  1685→    let opcode = word & 0x7f;
  1686→    match opcode {
  1687→        0x00 => {
  1688→            let rd = (word >> 7) & 0x1f;
  1689→            let rj = (word >> 15) & 0x1f;
  1690→            let rk = (word >> 10) & 0x1f;
  1691→            format!("add.w $r{}, $r{}, $r{}", rd, rj, rk)
  1692→        }
  1693→        0x01 => {
  1694→            let rd = (word >> 7) & 0x1f;
  1695→            let rj = (word >> 15) & 0x1f;
  1696→            let rk = (word >> 10) & 0x1f;
  1697→            format!("sub.w $r{}, $r{}, $r{}", rd, rj, rk)
  1698→        }
  1699→        0x02 => {
  1700→            let rd = (word >> 7) & 0x1f;
  1701→            let rj = (word >> 15) & 0x1f;
  1702→            let rk = (word >> 10) & 0x1f;
  1703→            format!("mul.w $r{}, $r{}, $r{}", rd, rj, rk)
  1704→        }
  1705→        0x05 => {
  1706→            let rd = (word >> 7) & 0x1f;
  1707→            let _rj = (word >> 15) & 0x1f; // lu12i.w does not use rj in the disassembly
  1708→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1709→            format!("lu12i.w $r{}, {}({})", rd, si12, si12)
  1710→        }
  1711→        0x08 => {
  1712→            let rd = (word >> 7) & 0x1f;
  1713→            let rj = (word >> 15) & 0x1f;
  1714→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1715→            format!("ld.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1716→        }
  1717→        0x0a => {
  1718→            let rd = (word >> 7) & 0x1f;
  1719→            let rj = (word >> 15) & 0x1f;
  1720→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1721→            format!("st.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1722→        }
  1723→        0x0c => {
  1724→            let rd = (word >> 7) & 0x1f;
  1725→            let rj = (word >> 15) & 0x1f;
  1726→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1727→            format!("ld.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1728→        }
  1729→        0x0e => {
  1730→            let rd = (word >> 7) & 0x1f;
  1731→            let rj = (word >> 15) & 0x1f;
  1732→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1733→            format!("st.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1734→        }
  1735→        0x10 => {
  1736→            let rj = (word >> 15) & 0x1f;
  1737→            let offs = ((word >> 10) as i32) << 12 >> 10;
  1738→            format!("beq $r{}, {}({})", rj, offs, offs)
  1739→        }
  1740→        0x11 => {
  1741→            let rj = (word >> 15) & 0x1f;
  1742→            let offs = ((word >> 10) as i32) << 12 >> 10;
  1743→            format!("bne $r{}, {}({})", rj, offs, offs)
  1744→        }
  1745→        0x14 => format!("bl {}", ((word >> 10) as i32) << 12 >> 10),
  1746→        0x15 => {
  1747→            let rd = (word >> 7) & 0x1f;
  1748→            format!("jirl $r{}, {}", rd, ((word >> 10) as i32) << 12 >> 10)
  1749→        }
  1750→        _ => format!(".word {:08x}", word),
  1751→    }
  1752→}

  1774→/// Compute the stack frame size for an IR function on LoongArch64.
  1775→///
  1776→/// Sums `Alloc` instruction sizes, adds 16 bytes for the ra/fp save pair,
  1777→/// and rounds up to 16-byte alignment.
  1778→fn loongarch64_compute_frame_size(func: &IRFunction) -> usize {
  1779→    let mut total: u32 = 16; // ra + fp save pair
  1780→    for block in &func.blocks {
  1781→        for instr in &block.instructions {
  1782→            if let IRInstr::Alloc { size, .. } = instr {
  1783→                let aligned = (*size).div_ceil(16) * 16;
  1784→                total += aligned;
  1785→            }
  1786→        }
  1787→    }
  1788→    // Round up to 16-byte alignment
  1789→    total = (total + 15) & !15;
  1790→    total as usize
  1791→}

  1830→/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1831→fn map_vreg_to_gpr(
  1832→    vreg_id: u32,
  1833→    arg_index: Option<usize>,
  1834→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1835→) -> Gpr {
  1836→    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1837→        return *gpr;
  1838→    }
  1839→
  1840→    // If this is an argument, map it to the corresponding argument register.
  1841→    if let Some(idx) = arg_index {
  1842→        if let Some(reg) = Gpr::arg_register(idx) {
  1843→            vreg_map.insert(vreg_id, reg);
  1844→            return reg;
  1845→        }
  1846→    }
  1847→
  1848→    // Otherwise, pick the first available allocatable register.
  1849→    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1850→    for &reg in ALLOCATABLE_GPRS {
  1851→        if !used.contains(&reg) {
  1852→            vreg_map.insert(vreg_id, reg);
  1853→            return reg;
  1854→        }
  1855→    }
  1856→
  1857→    // Fallback: use t8 as a scratch register.
  1858→    Gpr::T8
  1859→}

  1869→/// Load a 64-bit immediate into a register using the lu12i.w/ori/lu32i.d/lu52i.d
  1870→/// instruction sequence.
  1871→///
  1872→/// Returns the list of AllocatedInstructions emitted to perform the load.
  1873→fn load_imm_la64(rd: Gpr, imm: i64) -> Vec<AllocatedInstruction> {
  1874→    let mut result = Vec::new();
  1875→    let val = imm as u64;
  1876→
  1877→    // lu12i.w rd, bits[31:12]
  1878→    let hi20 = ((val >> 12) & 0xFFFFF) as i32;
  1879→    result.push(emit_alloc_instr(
  1880→        Instruction::Lu12iW { rd, imm20: hi20 },
  1881→        vec![],
  1882→        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1883→    ));
  1884→
  1885→    // ori rd, rd, bits[11:0]
  1886→    let lo12 = (val & 0xFFF) as u32;
  1887→    result.push(emit_alloc_instr(
  1888→        Instruction::Ori {
  1889→            rd,
  1890→            rj: rd,
  1891→            imm12: lo12,
  1892→        },
  1893→        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1894→        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1895→    ));
  1896→
  1897→    // lu32i.d is actually pcaddi in QEMU — skip it
  1898→    // Instead, handle upper bits differently
  1899→
  1900→    let lower32 = val & 0xFFFFFFFF;
  1901→    let sign_ext = if lower32 & 0x80000000 != 0 {
  1902→        0xFFFFFFFF00000000u64
  1903→    } else {
  1904→        0u64
  1905→    };
  1906→
  1907→    if val >> 32 != sign_ext >> 32 {
  1908→        // Upper bits don't match sign extension
  1909→        if val >> 32 == 0 && lower32 & 0x80000000 != 0 {
  1910→            // Zero-extend: slli.d + srli.d
  1911→            result.push(emit_alloc_instr(
  1912→                Instruction::SlliD { rd, rj: rd, imm8: 32 },
  1913→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1914→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1915→            ));
  1916→            result.push(emit_alloc_instr(
  1917→                Instruction::SrliD { rd, rj: rd, imm8: 32 },
  1918→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1919→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1920→            ));
  1921→        } else {
  1922→            // Full 64-bit: use lu52i.d for bits[63:52] then slli+srli+or for bits[51:32]
  1923→            let hi52 = ((val >> 52) & 0xFFF) as i32;
  1924→            result.push(emit_alloc_instr(
  1925→                Instruction::Lu52iD {
  1926→                    rd,
  1927→                    rj: rd,
  1928→                    imm12: hi52,
  1929→                },
  1930→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1931→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1932→            ));
  1933→            let bits_51_32 = ((val >> 32) & 0xFFFFF) as u32;
  1934→            if bits_51_32 != 0 {
  1935→                let temp = Gpr::T0;
  1936→                let hi_51_32 = ((val >> 32) & 0xFFFFF) as i32;
  1937→                result.push(emit_alloc_instr(
  1938→                    Instruction::Lu12iW { rd: temp, imm20: hi_51_32 },
  1939→                    vec![],
  1940→                    vec![PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1941→                ));
  1942→                result.push(emit_alloc_instr(
  1943→                    Instruction::SlliD { rd: temp, rj: temp, imm8: 32 },
  1944→                    vec![PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1945→                    vec![PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1946→                ));
  1947→                result.push(emit_alloc_instr(
  1948→                    Instruction::SlliD { rd, rj: rd, imm8: 32 },
  1949→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1950→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1951→                ));
  1952→                result.push(emit_alloc_instr(
  1953→                    Instruction::SrliD { rd, rj: rd, imm8: 32 },
  1954→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1955→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1956→                ));
  1957→                result.push(emit_alloc_instr(
  1958→                    Instruction::Or { rd, rj: rd, rk: temp },
  1959→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding()), PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1960→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1961→                ));
  1962→            }
  1963→        }
  1964→    }

  1969→/// Resolve an IRValue to a physical GPR.
  1970→///
  1971→/// - For `IRValue::Register`: looks up in `vreg_map`.
  1972→/// - For `IRValue::Immediate`: loads the value into `scratch` and returns
  1973→///   `(scratch, instructions)`.
  1974→/// - For `IRValue::Address`: loads the address into `scratch`.
  1975→/// - For `IRValue::Label`: loads a placeholder into `scratch`.
  1976→fn resolve_gpr_la64(
  1977→    val: &IRValue,
  1978→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1979→    scratch: Gpr,
  1980→) -> (Gpr, Vec<AllocatedInstruction>) {
  1981→    match val {
  1982→        IRValue::Register(id) => {
  1983→            let reg = map_vreg_to_gpr(*id, None, vreg_map);
  1984→            (reg, Vec::new())
  1985→        }
  1986→        IRValue::Immediate(imm) => {
  1987→            let code = load_imm_la64(scratch, *imm);
  1988→            (scratch, code)
  1989→        }
  1990→        IRValue::Address(addr) => {
  1991→            let code = load_imm_la64(scratch, *addr as i64);
  1992→            (scratch, code)
  1993→        }
  1994→        IRValue::Label(_) => {
  1995→            // Labels need relocation; emit a placeholder load
  1996→            let code = load_imm_la64(scratch, 0);
  1997→            (scratch, code)
  1998→        }
  1999→    }
