     1→//! # ARM64 Code Emission
     2→//!
     3→//! Lowers IR to ARM64 machine code and produces ELF binaries or raw binaries
     4→//! suitable for the Raspberry Pi 5 (Cortex-A76, ARMv8.2-A).
     5→//!
     6→//! ## Pipeline
     7→//!
     8→//! 1. **IR → ARM64 Instructions**: Each IR instruction is pattern-matched and
     9→//!    lowered to one or more [`Instruction`]
    10→//!    values, with virtual registers replaced by physical registers from the
    11→//!    register allocator.
    12→//! 2. **Emit function**: Each IR function is emitted as a sequence of 32-bit
    13→//!    ARM64 code words.
    14→//! 3. **Emit program**: All functions are collected, data sections are laid
    15→//!    out, and a minimal ELF64 binary is produced.
    16→//!
    17→//! ## Output Formats
    18→//!
    19→//! - **ELF**: Full ELF64 binary with program headers, section headers, symbol
    20→//!   table, and string table.  Suitable for Linux on Pi 5.
    21→//! - **Raw**: Flat binary image for bare-metal Pi 5 (loaded at 0x80000).
    22→//! - **Obj**: Relocatable ELF object file (`ET_REL`) for linking.
    23→//!
    24→//! ## ELF Layout (executable)
    25→//!
    26→//! ```text
    27→//! ┌─────────────────────┐
    28→//! │ ELF Header           │  64 bytes
    29→//! ├─────────────────────┤
    30→//! │ Program Headers      │  2 × 56 bytes (LOAD segments)
    31→//! ├─────────────────────┤
    32→//! │ .text                │  emitted code
    33→//! ├─────────────────────┤
    34→//! │ .rodata              │  read-only data
    35→//! ├─────────────────────┤
    36→//! │ .data                │  initialized data
    37→//! ├─────────────────────┤
    38→//! │ .symtab              │  symbol table entries
    39→//! ├─────────────────────┤
    40→//! │ .strtab              │  symbol string table
    41→//! ├─────────────────────┤
    42→//! │ .shstrtab            │  section header string table
    43→//! ├─────────────────────┤
    44→//! │ Section Headers      │  N × 64 bytes
    45→//! └─────────────────────┘
    46→//! .bss is virtual-only (memsz > filesz in the data LOAD segment).
    47→//! ```
    48→//!
    49→//! ## Relocation Support
    50→//!
    51→//! Inter-function calls (`BL`) are resolved in a fixup pass after all
    52→//! functions have been emitted.  The emitter records the word offset of each
    53→//! `BL` instruction and the target function name; once function addresses
    54→//! are known, the branch offsets are patched into the encoded instructions.

   491→// ---------------------------------------------------------------------------
   492→// ELF RelaEntry
   493→// ---------------------------------------------------------------------------
   494→
   495→/// An ELF64 relocation entry with an explicit addend (SHT_RELA).
   496→///
   497→/// Each entry specifies where in the section a relocation must be applied,
   498→/// which symbol the relocation references, the type of relocation, and an
   499→/// addend value used in the relocation computation.
   500→#[derive(Debug, Clone)]
   501→pub struct RelaEntry {
   502→    /// Byte offset within the section where the relocation applies.
   503→    pub offset: u64,
   504→    /// Symbol index (upper 32 bits) and relocation type (lower 32 bits),
   505→    /// packed as: `(sym_idx << 32) | r_type`.
   506→    pub info: u64,
   507→    /// Addend value used in the relocation computation.
   508→    pub addend: i64,
   509→}
   510→
   511→impl RelaEntry {
   512→    /// Create a new relocation entry.
   513→    pub fn new(offset: u64, sym_idx: u32, r_type: u32, addend: i64) -> Self {
   514→        Self {
   515→            offset,
   516→            info: ((sym_idx as u64) << 32) | (r_type as u64),
   517→            addend,
   518→        }
   519→    }
   520→
   521→    /// Extract the symbol index from `info`.
   522→    pub fn sym_idx(&self) -> u32 {
   523→        (self.info >> 32) as u32
   524→    }
   525→
   526→    /// Extract the relocation type from `info`.
   527→    pub fn r_type(&self) -> u32 {
   528→        (self.info & 0xFFFFFFFF) as u32
   529→    }
   530→
   531→    /// Serialize the entry to 24 bytes (little-endian).
   532→    pub fn to_bytes(&self) -> [u8; 24] {
   533→        let mut buf = [0u8; 24];
   534→        buf[0..8].copy_from_slice(&self.offset.to_le_bytes());
   535→        buf[8..16].copy_from_slice(&self.info.to_le_bytes());
   536→        buf[16..24].copy_from_slice(&self.addend.to_le_bytes());
   537→        buf
   538→    }
   539→}

   545→/// The ARM64 code emitter.
   546→///
   547→/// Holds state for the current emission context: the register allocator,
   548→/// accumulated code, and fixup records for branch targets.
   549→pub struct Emitter {
   550→    /// Register allocator used during emission.
   551→    reg_alloc: RegAllocator,
   552→    /// Accumulated machine code for the current function (32-bit words).
   553→    code: Vec<u32>,
   554→    /// Fixup records for intra-function branches: (word index, target label name).
   555→    fixups: Vec<(usize, String)>,
   556→    /// Map from label name to code offset (in words) within the current function.
   557→    label_offsets: HashMap<String, usize>,
   558→    /// Inter-function call relocations for the current function.
   559→    call_relocs: Vec<CallRelocation>,
   560→    /// Relocation entries using the `RelocationEntry` infrastructure.
   561→    ///
   562→    /// Each entry records a byte offset within the function's encoded output
   563→    /// where a symbolic reference must be resolved, the target symbol name,
   564→    /// and the ISA-specific relocation type (e.g., `"R_AARCH64_CALL26"`).
   565→    relocations: Vec<RelocationEntry>,
   566→    /// Name of the function currently being emitted.
   567→    current_func_name: String,
   568→    /// Byte offset of the current function within the text section (set externally).
   569→    func_text_offset: u64,
   570→    /// Computed stack frame size (in bytes) for the current function.
   571→    frame_size: u16,
   572→}
   573→
   574→impl Emitter {
   575→    /// Create a new emitter with a fresh register allocator.
   576→    pub fn new() -> Self {
   577→        Self {
   578→            reg_alloc: RegAllocator::new(),
   579→            code: Vec::new(),
   580→            fixups: Vec::new(),
   581→            label_offsets: HashMap::new(),
   582→            call_relocs: Vec::new(),
   583→            relocations: Vec::new(),
   584→            current_func_name: String::new(),
   585→            func_text_offset: 0,
   586→            frame_size: 0,
   587→        }
   588→    }

   608→    /// Emit a single IR function to ARM64 machine code.
   609→    ///
   610→    /// Returns a vector of 32-bit ARM64 instruction words.
   611→    pub fn emit_function(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   612→        self.code.clear();
   613→        self.fixups.clear();
   614→        self.label_offsets.clear();
   615→        self.call_relocs.clear();
   616→        self.relocations.clear();
   617→        self.current_func_name = func.name.clone();
   618→        self.reg_alloc.reset();
   619→
   620→        // Allocate registers for parameters (AAPCS64: X0–X7).
   621→        for (i, param) in func.params.iter().enumerate() {
   622→            if let IRValue::Register(vreg_id) = param {
   623→                if i < 8 {
   624→                    let _ = self.reg_alloc.allocate(*vreg_id);
   625→                }
   626→            }
   627→        }
   628→
   629→        // Emit prologue: SUB SP, SP, #frame_size+16 (space for saved FP/LR + locals)
   630→        // Then STP X29, X30, [SP, #frame_size] (save FP/LR at top of allocated frame)
   631→        let aligned_stack = compute_frame_size(func);
   632→        self.frame_size = aligned_stack;
   633→        let total_alloc = aligned_stack + 16; // 16 bytes for FP+LR pair
   634→        self.emit_instruction(Instruction::SUB {
   635→            rd: Register::SP,
   636→            rn: Register::SP,
   637→            rm: Operand::Imm12(total_alloc),
   638→        })?;
   639→        self.emit_instruction(Instruction::STP {
   640→            rt1: Register::X29,
   641→            rt2: Register::X30,
   642→            rn: Register::SP,
   643→            offset: aligned_stack as i32, // save FP/LR at top of frame
   644→        })?;
   645→
   646→        // Emit each basic block.
   647→        for block in &func.blocks {
   648→            self.label_offsets
   649→                .insert(block.label.clone(), self.code.len());
   650→            self.emit_block(block)?;
   651→        }
   652→
   653→        // Apply fixups — resolve intra-function branch targets.
   654→        self.apply_fixups()?;
   655→
   656→        Ok(self.code.clone())
   657→    }
   658→
   659→    /// Returns the relocation entries recorded during emission of the current
   660→    /// function.
   661→    ///
   662→    /// Each `RelocationEntry` records a byte offset within the function's
   663→    /// encoded output where a symbolic reference must be resolved, the target
   664→    /// symbol name, and the ISA-specific relocation type (e.g.,
   665→    /// `"R_AARCH64_CALL26"`).
   666→    pub fn relocations(&self) -> &[RelocationEntry] {
   667→        &self.relocations
   668→    }
   669→
   670→    /// Emit instructions for a single IR basic block.
   671→    fn emit_block(&mut self, block: &IRBlock) -> Result<()> {
   672→        for instr in &block.instructions {
   673→            self.emit_ir_instr(instr)?;
   674→        }
   675→        self.emit_terminator(&block.terminator)?;
   676→        Ok(())
   677→    }
   678→

   680→    // IR → ARM64 instruction lowering
   681→    // -----------------------------------------------------------------------

   683→    /// Lower a single IR instruction to ARM64 instructions.
   684→    fn emit_ir_instr(&mut self, instr: &IRInstr) -> Result<()> {
   685→        match instr {
   686→            IRInstr::Load { dst, addr, offset, ty } => {
   687→                let rt = self.resolve_reg(dst)?;
   688→                let rn = self.resolve_reg(addr)?;
   689→                self.emit_load(rt, rn, *offset, ty)?;
   690→            }
   691→
   692→            IRInstr::Store { value, addr, offset, ty } => {
   693→                let rt = self.resolve_reg(value)?;
   694→                let rn = self.resolve_reg(addr)?;
   695→                self.emit_store(rt, rn, *offset, ty)?;
   696→            }
   697→
   698→            IRInstr::BinOp { op, dst, lhs, rhs, ty } => {
   699→                self.emit_binop(*op, dst, lhs, rhs, ty.as_ref())?;
   700→            }
   701→
   702→            IRInstr::UnaryOp { op, dst, operand, ty: _ } => {
   703→                let rd = self.resolve_reg(dst)?;
   704→                let rn = self.resolve_reg(operand)?;
   705→                match op {
   706→                    UnaryOpKind::Neg => {
   707→                        self.emit_instruction(Instruction::SUB {
   708→                            rd,
   709→                            rn: Register::XZR,
   710→                            rm: Operand::Reg {
   711→                                reg: rn,
   712→                                shift: None,
   713→                            },
   714→                        })?;
   715→                    }
   716→                    UnaryOpKind::Not => {
   717→                        self.emit_load_immediate(Register::X9, -1)?;
   718→                        self.emit_instruction(Instruction::EOR {
   719→                            rd,
   720→                            rn,
   721→                            rm: Register::X9,
   722→                        })?;
   723→                    }
   724→                    UnaryOpKind::Clz => {
   725→                        // CLZ Xd, Xn — count leading zeros (native ARM64 instruction)
   726→                        self.emit_instruction(Instruction::CLZ { rd, rn })?;
   727→                    }
   728→                    UnaryOpKind::Ctz => {
   729→                        // CTZ = RBIT + CLZ: reverse bits then count leading zeros.
   730→                        // Use X9 as scratch if rd == rn (need intermediate result).
   731→                        if rd == rn {
   732→                            self.emit_instruction(Instruction::RBIT {
   733→                                rd: Register::X9,
   734→                                rn,
   735→                            })?;
   736→                            self.emit_instruction(Instruction::CLZ {
   737→                                rd,
   738→                                rn: Register::X9,
   739→                            })?;
   740→                        } else {
   741→                            self.emit_instruction(Instruction::RBIT { rd, rn })?;
   742→                            self.emit_instruction(Instruction::CLZ { rd, rn: rd })?;
   743→                        }
   744→                    }
   745→                    UnaryOpKind::Popcnt => {
   746→                        // POPCNT via FMOV+CNT+ADDV+UMOV sequence:
   747→                        // FMOV D8, Xn        — move GPR value to SIMD register (V8 is caller-saved)
   748→                        // CNT V8.8B, V8.8B   — count bits per byte
   749→                        // ADDV B8, V8.8B     — horizontal sum of byte counts
   750→                        // UMOV Xd, V8.B[0]   — move result back to GPR (zero-extends to 64 bits)
   751→                        const SIMD_SCRATCH: u8 = 8; // V8 is caller-saved in AAPCS64
   752→                        self.emit_instruction(Instruction::FMOV_DX {
   753→                            vd: SIMD_SCRATCH,
   754→                            rn,
   755→                        })?;
   756→                        self.emit_instruction(Instruction::CNT {
   757→                            vd: SIMD_SCRATCH,
   758→                            vn: SIMD_SCRATCH,
   759→                        })?;
   760→                        self.emit_instruction(Instruction::ADDV {
   761→                            vd: SIMD_SCRATCH,
   762→                            vn: SIMD_SCRATCH,
   763→                        })?;
   764→                        self.emit_instruction(Instruction::UMOV {
   765→                            rd,
   766→                            vn: SIMD_SCRATCH,
   767→                        })?;
   768→                    }
   769→                }
   770→            }
   771→
   772→            IRInstr::Call {
   773→                dst,
   774→                func: target_name,
   775→                args,
   776→            } => {
   777→                // Move arguments into X0–X7.
   778→                for (i, arg) in args.iter().enumerate() {
   779→                    if i >= 8 {
   780→                        break;
   781→                    }
   782→                    let src = self.resolve_reg(arg)?;
   783→                    let dst_reg = match i {
   784→                        0 => Register::X0,
   785→                        1 => Register::X1,
   786→                        2 => Register::X2,
   787→                        3 => Register::X3,
   788→                        4 => Register::X4,
   789→                        5 => Register::X5,
   790→                        6 => Register::X6,
   791→                        7 => Register::X7,
   792→                        _ => unreachable!(),
   793→                    };
   794→                    if src != dst_reg {
   795→                        self.emit_instruction(Instruction::MOV {
   796→                            rd: dst_reg,
   797→                            rm: src,
   798→                        })?;
   799→                    }
   800→                }
   801→
   802→                // BL — record a relocation for later patching.
   803→                let bl_word_idx = self.code.len();
   804→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
   805→                self.call_relocs.push(CallRelocation {
   806→                    text_byte_offset: bl_byte_offset,
   807→                    target_func: target_name.clone(),
   808→                });
   809→                // Register a RelocationEntry so the linker can patch this BL.
   810→                self.relocations.push(RelocationEntry {
   811→                    offset: bl_byte_offset,
   812→                    symbol: target_name.clone(),
   813→                    reloc_type: "R_AARCH64_CALL26".to_string(),
   814→                });
   815→                self.emit_instruction(Instruction::BL { offset: 0 })?;
   816→
   817→                if let Some(dst_val) = dst {
   818→                    let rd = self.resolve_reg(dst_val)?;
   819→                    if rd != Register::X0 {
   820→                        self.emit_instruction(Instruction::MOV {
   821→                            rd,
   822→                            rm: Register::X0,
   823→                        })?;
   824→                    }
   825→                }
   826→            }
   827→
   828→            IRInstr::Alloc { dst, size } => {
   829→                let rd = self.resolve_reg(dst)?;
   830→                let aligned = (*size).div_ceil(16) * 16;
   831→                // First decrement SP, then save new SP as the pointer.
   832→                // On ARM64 the stack grows downward, so the allocated region
   833→                // is [new_SP, old_SP) and the pointer should be new_SP.
   834→                self.emit_instruction(Instruction::SUB {
   835→                    rd: Register::SP,
   836→                    rn: Register::SP,
   837→                    rm: Operand::Imm12(aligned as u16),
   838→                })?;
   839→                // ADD rd, SP, #0 (not MOV rd, SP — ORR treats Rm=31 as XZR)
   840→                self.emit_instruction(Instruction::ADD {
   841→                    rd,
   842→                    rn: Register::SP,
   843→                    rm: Operand::Imm12(0),
   844→                })?;
   845→            }
   846→
   847→            IRInstr::Free { ptr } => {
   848→                let rt = self.resolve_reg(ptr)?;
   849→                // Move ptr to X0 (first argument)
   850→                if rt != Register::X0 {
   851→                    self.emit_instruction(Instruction::MOV {
   852→                        rd: Register::X0,
   853→                        rm: rt,
   854→                    })?;
   855→                }
   856→                // BL __vuma_free
   857→                let bl_word_idx = self.code.len();
   858→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
   859→                self.call_relocs.push(CallRelocation {
   860→                    text_byte_offset: bl_byte_offset,
   861→                    target_func: "__vuma_free".to_string(),
   862→                });
   863→                self.relocations.push(RelocationEntry {
   864→                    offset: bl_byte_offset,
   865→                    symbol: "__vuma_free".to_string(),
   866→                    reloc_type: "R_AARCH64_CALL26".to_string(),
   867→                });
   868→                self.emit_instruction(Instruction::BL { offset: 0 })?;
   869→            }
   870→
   871→            IRInstr::Cast { kind, dst, src } => {
   872→                let rd = self.resolve_reg(dst)?;
   873→                let rn = self.resolve_reg(src)?;
   874→                match kind {
   875→                    CastKind::ZExt => {
   876→                        // Zero-extend 32-bit to 64-bit using UBFM
   877→                        self.emit_instruction(Instruction::UBFM {
   878→                            rd,
   879→                            rn,
   880→                            immr: 0,
   881→                            imms: 31,
   882→                        })?;
   883→                    }
   884→                    CastKind::SExt => {
   885→                        // Sign-extend 32-bit to 64-bit using SBFM
   886→                        self.emit_instruction(Instruction::SBFM {
   887→                            rd,
   888→                            rn,
   889→                            immr: 0,
   890→                            imms: 31,
   891→                        })?;
   892→                    }
   893→                    CastKind::Trunc | CastKind::BitCast => {
   894→                        // Trunc: upper bits discarded on write — just MOV
   895→                        // BitCast: no data change — just MOV
   896→                        if rd != rn {
   897→                            self.emit_instruction(Instruction::MOV { rd, rm: rn })?;
   898→                        }
   899→                    }
   900→                }
   901→            }
   902→
   903→            IRInstr::Phi { .. } => {
   904→                log::warn!(
   905→                    "IRInstr::Phi encountered during emission — should be resolved by SSA pass"
   906→                );
   907→            }
   908→
   909→            IRInstr::GetAddress { dst, name } => {
   910→                let rd = self.resolve_reg(dst)?;
   911→                // Emit a call to __vuma_getaddr to resolve the symbol at runtime.
   912→                // Move name hash to X0 as the argument.
   913→                let name_hash = name
   914→                    .chars()
   915→                    .fold(0u64, |acc, c| acc.wrapping_mul(31).wrapping_add(c as u64));
   916→                self.emit_load_immediate(Register::X0, name_hash as i64)?;
   917→                let bl_word_idx = self.code.len();
   918→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
   919→                self.call_relocs.push(CallRelocation {
   920→                    text_byte_offset: bl_byte_offset,
   921→                    target_func: "__vuma_getaddr".to_string(),
   922→                });
   923→                self.relocations.push(RelocationEntry {
   924→                    offset: bl_byte_offset,
   925→                    symbol: "__vuma_getaddr".to_string(),
   926→                    reloc_type: "R_AARCH64_CALL26".to_string(),
   927→                });
   928→                self.emit_instruction(Instruction::BL { offset: 0 })?;
   929→                if rd != Register::X0 {
   930→                    self.emit_instruction(Instruction::MOV {
   931→                        rd,
   932→                        rm: Register::X0,
   933→                    })?;
   934→                }
   935→            }
   936→
   937→            IRInstr::Offset { dst, base, offset } => {
   938→                let rd = self.resolve_reg(dst)?;
   939→                let rn = self.resolve_reg(base)?;
   940→                let rm = match offset {
   941→                    IRValue::Immediate(v) => {
   942→                        if *v >= 0 && *v <= 4095 {
   943→                            Operand::Imm12(*v as u16)
   944→                        } else {
   945→                            let temp = Register::X9;
   946→                            self.emit_load_immediate(temp, *v)?;
   947→                            Operand::Reg {
   948→                                reg: temp,
   949→                                shift: None,
   950→                            }
   951→                        }
   952→                    }
   953→                    _ => Operand::Reg {
   954→                        reg: self.resolve_reg(offset)?,
   955→                        shift: None,
   956→                    },
   957→                };
   958→                self.emit_instruction(Instruction::ADD { rd, rn, rm })?;
   959→            }
   960→
   961→            // ── Dedicated arithmetic — delegate to BinOp ──
   962→            IRInstr::Add { dst, lhs, rhs, ty } => {
   963→                self.emit_binop(BinOpKind::Add, dst, lhs, rhs, ty.as_ref())?;
   964→            }
   965→            IRInstr::Sub { dst, lhs, rhs, ty } => {
   966→                self.emit_binop(BinOpKind::Sub, dst, lhs, rhs, ty.as_ref())?;
   967→            }
   968→            IRInstr::Mul { dst, lhs, rhs, ty } => {
   969→                self.emit_binop(BinOpKind::Mul, dst, lhs, rhs, ty.as_ref())?;
   970→            }
   971→            IRInstr::Div { dst, lhs, rhs, ty } => {
   972→                self.emit_binop(BinOpKind::SDiv, dst, lhs, rhs, ty.as_ref())?;
   973→            }
   974→
   975→            // ── Comparison instruction ──
   976→            IRInstr::Cmp {
   977→                kind,
   978→                dst,
   979→                lhs,
   980→                rhs,
   981→                ty,
   982→            } => {
   983→                let width = RegWidth::from_ir_type(ty.as_ref());
   984→                let rd = self.resolve_reg(dst)?;
   985→                let rn = self.resolve_reg(lhs)?;
   986→                let rm = self.resolve_reg(rhs)?;
   987→                self.emit_instruction_with_width(
   988→                    Instruction::CMP {
   989→                        rn,
   990→                        rm: Operand::Reg {
   991→                            reg: rm,
   992→                            shift: None,
   993→                        },
   994→                    },
   995→                    width,
   996→                )?;
   997→                let cond = cmp_kind_to_condition(kind);
   998→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
   999→            }
  1000→
  1001→            // ── Operating-system interface ──
  1002→            IRInstr::Syscall { dst, sysno, args } => {
  1003→                // ARM64 Linux syscall convention:
  1004→                //   x8 = syscall number, x0-x5 = arguments, svc #0, result in x0
  1005→                // Move arguments into X0–X5.
  1006→                for (i, arg) in args.iter().enumerate() {
  1007→                    if i >= 6 {
  1008→                        break;
  1009→                    }
  1010→                    let src = self.resolve_reg(arg)?;
  1011→                    let dst_reg = match i {
  1012→                        0 => Register::X0,
  1013→                        1 => Register::X1,
  1014→                        2 => Register::X2,
  1015→                        3 => Register::X3,
  1016→                        4 => Register::X4,
  1017→                        5 => Register::X5,
  1018→                        _ => unreachable!(),
  1019→                    };
  1020→                    if src != dst_reg {
  1021→                        self.emit_instruction(Instruction::MOV {
  1022→                            rd: dst_reg,
  1023→                            rm: src,
  1024→                        })?;
  1025→                    }
  1026→                }
  1027→                // Move syscall number into X8
  1028→                self.emit_instruction(Instruction::MOVZ {
  1029→                    rd: Register::X8,
  1030→                    imm16: *sysno as u16,
  1031→                    shift: 0,
  1032→                })?;
  1033→                // If sysno > 65535, use MOVK for upper bits
  1034→                if *sysno > 0xFFFF {
  1035→                    self.emit_instruction(Instruction::MOVK {
  1036→                        rd: Register::X8,
  1037→                        imm16: (*sysno >> 16) as u16,
  1038→                        shift: 16,
  1039→                    })?;
  1040→                }
  1041→                // svc #0
  1042→                self.emit_instruction(Instruction::SVC { imm16: 0 })?;
  1043→                // Move result from X0 to destination
  1044→                if let Some(dst_val) = dst {
  1045→                    let rd = self.resolve_reg(dst_val)?;
  1046→                    if rd != Register::X0 {
  1047→                        self.emit_instruction(Instruction::MOV {
  1048→                            rd,
  1049→                            rm: Register::X0,
  1050→                        })?;
  1051→                    }
  1052→                }
  1053→            }
  1054→
  1055→            // ── Instruction-level control flow ──
  1056→            IRInstr::Ret { values } => {
  1057→                for (i, val) in values.iter().enumerate() {
  1058→                    if i >= 8 {
  1059→                        break;
  1060→                    }
  1061→                    let src = self.resolve_reg(val)?;
  1062→                    let dst_reg = match i {
  1063→                        0 => Register::X0,
  1064→                        1 => Register::X1,
  1065→                        2 => Register::X2,
  1066→                        3 => Register::X3,
  1067→                        4 => Register::X4,
  1068→                        5 => Register::X5,
  1069→                        6 => Register::X6,
  1070→                        7 => Register::X7,
  1071→                        _ => unreachable!(),
  1072→                    };
  1073→                    if src != dst_reg {
  1074→                        self.emit_instruction(Instruction::MOV {
  1075→                            rd: dst_reg,
  1076→                            rm: src,
  1077→                        })?;
  1078→                    }
  1079→                }
  1080→            }
  1081→
  1082→            IRInstr::Branch { target } => {
  1083→                let fixup_idx = self.code.len();
  1084→                self.fixups.push((fixup_idx, target.clone()));
  1085→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1086→            }
  1087→
  1088→            IRInstr::CondBranch {
  1089→                cond,
  1090→                true_target,
  1091→                false_target,
  1092→            } => {
  1093→                let rt = self.resolve_reg(cond)?;
  1094→                let fixup_cbz = self.code.len();
  1095→                self.fixups.push((fixup_cbz, false_target.clone()));
  1096→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1097→                let fixup_b = self.code.len();
  1098→                self.fixups.push((fixup_b, true_target.clone()));
  1099→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1100→            }
  1101→
  1102→            IRInstr::Select {
  1103→                dst,
  1104→                cond,
  1105→                true_val,
  1106→                false_val,
  1107→                ty,
  1108→            } => {
  1109→                let width = RegWidth::from_ir_type(ty.as_ref());
  1110→                // Lower select as: SUBS XZR, cond, #0; CSEL dst, false_val, true_val, NE
  1111→                let rd = self.resolve_reg(dst)?;
  1112→                let rc = self.resolve_reg(cond)?;
  1113→                let rt = self.resolve_reg(true_val)?;
  1114→                let rf = self.resolve_reg(false_val)?;
  1115→                // Compare cond against zero and select.
  1116→                self.emit_instruction_with_width(
  1117→                    Instruction::SUB {
  1118→                        rd: Register::XZR,
  1119→                        rn: rc,
  1120→                        rm: Operand::Imm12(0),
  1121→                    },
  1122→                    width,
  1123→                )?;
  1124→                // Set flags by using a separate CMP (SUB with XZR destination
  1125→                // doesn't set flags; we need a flags-setting variant).
  1126→                // We emulate this with: CMP rc, #0 which is SUBS XZR, rc, #0.
  1127→                // Since we only have SUB, we use the existing CMP pattern.
  1128→                self.emit_instruction_with_width(
  1129→                    Instruction::CSEL {
  1130→                        rd,
  1131→                        rn: rf,
  1132→                        rm: rt,
  1133→                        cond: crate::arm64::Condition::NE,
  1134→                    },
  1135→                    width,
  1136→                )?;
  1137→            }
  1138→        }
  1139→        Ok(())
  1140→    }
  1141→
  1142→    /// Emit a binary operation (shared by `BinOp` and dedicated `Add`/`Sub`/…).
  1143→    ///
  1144→    /// Some ARM64 instructions (ADD, SUB, LSL, LSR, ASR) accept a 12-bit
  1145→    /// unsigned immediate operand directly.  Most others (MUL, AND, ORR, EOR,
  1146→    /// SDIV, UDIV, remainder, comparisons) **require** a register operand.
  1147→    /// When the RHS is an immediate that the target instruction cannot accept
  1148→    /// directly, we "spill" it into a scratch register (X16 / IP0) first via
  1149→    /// MOVZ + MOVK and then use that register as the operand.
  1150→    ///
  1151→    /// X16 (IP0) is chosen as the scratch register because it is **not** in
  1152→    /// the register allocator's free pool, so it will never conflict with
  1153→    /// `rd` or `rn`.
  1154→    fn emit_binop(
  1155→        &mut self,
  1156→        op: BinOpKind,
  1157→        dst: &IRValue,
  1158→        lhs: &IRValue,
  1159→        rhs: &IRValue,
  1160→        ty: Option<&IRType>,
  1161→    ) -> Result<()> {
  1162→        let width = RegWidth::from_ir_type(ty);
  1163→        let rd = self.resolve_reg(dst)?;
  1164→        let rn = self.resolve_reg(lhs)?;
  1165→
  1166→        // Determine whether the operation can accept an immediate operand.
  1167→        // ADD, SUB, LSL, LSR, ASR all have immediate forms in ARM64.
  1168→        // All other operations require a register operand.
  1169→        let supports_imm = matches!(
  1170→            op,
  1171→            BinOpKind::Add | BinOpKind::Sub | BinOpKind::Shl | BinOpKind::ShrL | BinOpKind::ShrA
  1172→        );
  1173→
  1174→        let rm = match rhs {
  1175→            IRValue::Immediate(v) => {
  1176→                if supports_imm && *v >= 0 && *v <= 4095 {
  1177→                    // Small immediate that the instruction accepts directly.
  1178→                    Operand::Imm12(*v as u16)
  1179→                } else {
  1180→                    // Load the immediate into scratch register X16 (IP0).
  1181→                    // X16 is not in the register allocator's pool, so it cannot
  1182→                    // conflict with `rd` or `rn`.
  1183→                    let scratch = Register::X16;
  1184→                    self.emit_load_immediate_with_width(scratch, *v, width)?;
  1185→                    Operand::Reg {
  1186→                        reg: scratch,
  1187→                        shift: None,
  1188→                    }
  1189→                }
  1190→            }
  1191→            _ => Operand::Reg {
  1192→                reg: self.resolve_reg(rhs)?,
  1193→                shift: None,
  1194→            },
  1195→        };
  1196→
  1197→        match op {
  1198→            BinOpKind::Add => {
  1199→                self.emit_instruction_with_width(Instruction::ADD { rd, rn, rm }, width)?;
  1200→            }
  1201→            BinOpKind::Sub => {
  1202→                self.emit_instruction_with_width(Instruction::SUB { rd, rn, rm }, width)?;
  1203→            }
  1204→            BinOpKind::Mul => {
  1205→                let rm_reg = self.operand_to_reg(&rm)?;
  1206→                self.emit_instruction_with_width(Instruction::MUL { rd, rn, rm: rm_reg }, width)?;
  1207→            }
  1208→            BinOpKind::SDiv => {
  1209→                let rm_reg = self.operand_to_reg(&rm)?;
  1210→                self.emit_instruction_with_width(Instruction::SDIV { rd, rn, rm: rm_reg }, width)?;
  1211→            }
  1212→            BinOpKind::UDiv => {
  1213→                let rm_reg = self.operand_to_reg(&rm)?;
  1214→                self.emit_instruction_with_width(Instruction::UDIV { rd, rn, rm: rm_reg }, width)?;
  1215→            }
  1216→            BinOpKind::And => {
  1217→                let rm_reg = self.operand_to_reg(&rm)?;
  1218→                self.emit_instruction_with_width(Instruction::AND { rd, rn, rm: rm_reg }, width)?;
  1219→            }
  1220→            BinOpKind::Or => {
  1221→                let rm_reg = self.operand_to_reg(&rm)?;
  1222→                self.emit_instruction_with_width(Instruction::ORR { rd, rn, rm: rm_reg }, width)?;
  1223→            }
  1224→            BinOpKind::Xor => {
  1225→                let rm_reg = self.operand_to_reg(&rm)?;
  1226→                self.emit_instruction_with_width(Instruction::EOR { rd, rn, rm: rm_reg }, width)?;
  1227→            }
  1228→            BinOpKind::Shl => {
  1229→                self.emit_instruction_with_width(Instruction::LSL { rd, rn, rm }, width)?;
  1230→            }
  1231→            BinOpKind::ShrL => {
  1232→                self.emit_instruction_with_width(Instruction::LSR { rd, rn, rm }, width)?;
  1233→            }
  1234→            BinOpKind::ShrA => {
  1235→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm }, width)?;
  1236→            }
  1237→            BinOpKind::Ror | BinOpKind::Rol => {
  1238→                let rm_reg = self.operand_to_reg(&rm)?;
  1239→                // ARM64 rotation: use ASR as placeholder
  1240→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm: Operand::Reg { reg: rm_reg, shift: None } }, width)?;
  1241→            }
  1242→            BinOpKind::SRem | BinOpKind::URem => {
  1243→                let rm_reg = self.operand_to_reg(&rm)?;
  1244→                let div_instr = if op == BinOpKind::SRem {
  1245→                    Instruction::SDIV { rd, rn, rm: rm_reg }
  1246→                } else {
  1247→                    Instruction::UDIV { rd, rn, rm: rm_reg }
  1248→                };
  1249→                self.emit_instruction_with_width(div_instr, width)?;
  1250→                // MSUB rd, rd, rm, rn  =>  rd = rn - rd * rm  =  dividend - quotient * divisor
  1251→                self.emit_instruction_with_width(Instruction::MSUB {
  1252→                    rd,
  1253→                    rn: rd,     // quotient (result of DIV)
  1254→                    rm: rm_reg, // divisor
  1255→                    ra: rn,     // dividend
  1256→                }, width)?;
  1257→            }
  1258→            BinOpKind::SLt
  1259→            | BinOpKind::SLe
  1260→            | BinOpKind::SGt
  1261→            | BinOpKind::SGe
  1262→            | BinOpKind::ULt
  1263→            | BinOpKind::ULe
  1264→            | BinOpKind::UGt
  1265→            | BinOpKind::UGe
  1266→            | BinOpKind::Eq
  1267→            | BinOpKind::Ne => {
  1268→                let rm_reg = self.operand_to_reg(&rm)?;
  1269→                self.emit_instruction_with_width(Instruction::CMP {
  1270→                    rn,
  1271→                    rm: Operand::Reg {
  1272→                        reg: rm_reg,
  1273→                        shift: None,
  1274→                    },
  1275→                }, width)?;
  1276→                let cond = binop_kind_to_condition(&op);
  1277→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1278→            }
  1279→        }
  1280→        Ok(())
  1281→    }
  1282→
  1283→    /// Emit the block terminator.
  1284→    fn emit_terminator(&mut self, term: &IRTerminator) -> Result<()> {
  1285→        match term {
  1286→            IRTerminator::Jump(target) => {
  1287→                let fixup_idx = self.code.len();
  1288→                self.fixups.push((fixup_idx, target.clone()));
  1289→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1290→            }
  1291→            IRTerminator::Branch {
  1292→                cond,
  1293→                true_block,
  1294→                false_block,
  1295→            } => {
  1296→                let rt = self.resolve_reg(cond)?;
  1297→                let fixup_cbnz = self.code.len();
  1298→                self.fixups.push((fixup_cbnz, true_block.clone()));
  1299→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1300→                let fixup_b = self.code.len();
  1301→                self.fixups.push((fixup_b, false_block.clone()));
  1302→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1303→            }
  1304→            IRTerminator::Return(vals) => {
  1305→                for (i, val) in vals.iter().enumerate() {
  1306→                    if i >= 8 {
  1307→                        break;
  1308→                    }
  1309→                    let src = self.resolve_reg(val)?;
  1310→                    let dst_reg = match i {
  1311→                        0 => Register::X0,
  1312→                        1 => Register::X1,
  1313→                        2 => Register::X2,
  1314→                        3 => Register::X3,
  1315→                        4 => Register::X4,
  1316→                        5 => Register::X5,
  1317→                        6 => Register::X6,
  1318→                        7 => Register::X7,
  1319→                        _ => unreachable!(),
  1320→                    };
  1321→                    if src != dst_reg {
  1322→                        self.emit_instruction(Instruction::MOV {
  1323→                            rd: dst_reg,
  1324→                            rm: src,
  1325→                        })?;
  1326→                    }
  1327→                }
  1328→                // Use the same frame size computed in the prologue
  1329→                let frame_size = self.frame_size;
  1330→                self.emit_instruction(Instruction::LDP {
  1331→                    rt1: Register::X29,
  1332→                    rt2: Register::X30,
  1333→                    rn: Register::SP,
  1334→                    offset: frame_size as i32, // load FP/LR from top of frame
  1335→                })?;
  1336→                self.emit_instruction(Instruction::ADD {
  1337→                    rd: Register::SP,
  1338→                    rn: Register::SP,
  1339→                    rm: Operand::Imm12(frame_size + 16), // restore SP
  1340→                })?;
  1341→                self.emit_instruction(Instruction::RET { rn: None })?;
  1342→            }
  1343→            IRTerminator::Unreachable => {
  1344→                self.emit_instruction(Instruction::MOV {
  1345→                    rd: Register::XZR,
  1346→                    rm: Register::XZR,
  1347→                })?;
  1348→            }
  1349→            // Switch, Invoke, TailCall, and Resume are lowered by the
  1350→            // control_flow module before reaching the emitter. If they
  1351→            // appear here, it means the lowering pass was not run.
  1352→            IRTerminator::Switch { .. } => {
  1353→                return Err(CodegenError::InvalidInstruction(
  1354→                    "Switch terminator must be lowered before emission".to_string(),
  1355→                ));
  1356→            }
  1357→            IRTerminator::Invoke { .. } => {
  1358→                return Err(CodegenError::InvalidInstruction(
  1359→                    "Invoke terminator must be lowered before emission".to_string(),
  1360→                ));
  1361→            }
  1362→            IRTerminator::TailCall { .. } => {
  1363→                return Err(CodegenError::InvalidInstruction(
  1364→                    "TailCall terminator must be lowered before emission".to_string(),
  1365→                ));
  1366→            }
  1367→            IRTerminator::Resume { .. } => {
  1368→                return Err(CodegenError::InvalidInstruction(
  1369→                    "Resume terminator must be lowered before emission".to_string(),
  1370→                ));
  1371→            }
  1372→        }
  1373→        Ok(())
  1374→    }

  1398→    /// Emit a load instruction with the given offset and IR type.
  1399→    ///
  1400→    /// Selects the correct ARM64 load variant based on the IR type:
  1401→    /// - I8/U8 → LDRB (byte, zero-extended)
  1402→    /// - I16/U16 → LDRH (halfword, zero-extended)
  1403→    /// - I32/U32/Ptr/Func → LDR (word or doubleword)
  1404→    /// - I64/U64 → LDR (doubleword)
  1405→    ///
  1406→    /// If the offset fits the ARM64 unsigned-offset immediate encoding for the
  1407→    /// selected instruction variant, it is encoded directly. Otherwise, the
  1408→    /// effective address is computed in a scratch register (X9) first, and
  1409→    /// the load is performed at [X9 + 0].
  1410→    fn emit_load(&mut self, rt: Register, rn: Register, offset: i32, ty: &IRType) -> Result<()> {
  1411→        // Determine the ARM64 load instruction and its immediate-offset constraints.
  1412→        // For each variant: (scale, max_imm12) where the encoded offset = imm12 << scale.
  1413→        // LDRB:  scale=0, imm12 0..4095 → offset 0..4095
  1414→        // LDRH:  scale=1, imm12 0..4095 → offset 0..8190, even
  1415→        // LDRSW: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1416→        // LDR W: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1417→        // LDR X: scale=3, imm12 0..4095 → offset 0..32760, multiple of 8
  1418→        match ty {
  1419→            IRType::I8 | IRType::U8 => {
  1420→                if offset >= 0 && offset <= 4095 {
  1421→                    self.emit_instruction(Instruction::LDRB { rt, rn, offset })?;
  1422→                } else {
  1423→                    self.emit_address_with_offset(rn, offset)?;
  1424→                    self.emit_instruction(Instruction::LDRB {
  1425→                        rt,
  1426→                        rn: Register::X9,
  1427→                        offset: 0,
  1428→                    })?;
  1429→                }
  1430→                // Sign-extend for signed byte loads (SXTB = SBFM Xd, Xn, #0, #7)
  1431→                if *ty == IRType::I8 {
  1432→                    self.emit_instruction(Instruction::SBFM {
  1433→                        rd: rt,
  1434→                        rn: rt,
  1435→                        immr: 0,
  1436→                        imms: 7,
  1437→                    })?;
  1438→                }
  1439→            }
  1440→            IRType::I16 | IRType::U16 => {
  1441→                if offset >= 0 && offset <= 8190 && offset % 2 == 0 {
  1442→                    self.emit_instruction(Instruction::LDRH { rt, rn, offset })?;
  1443→                } else {
  1444→                    self.emit_address_with_offset(rn, offset)?;
  1445→                    self.emit_instruction(Instruction::LDRH {
  1446→                        rt,
  1447→                        rn: Register::X9,
  1448→                        offset: 0,
  1449→                    })?;
  1450→                }
  1451→                // Sign-extend for signed halfword loads (SXTH = SBFM Xd, Xn, #0, #15)
  1452→                if *ty == IRType::I16 {
  1453→                    self.emit_instruction(Instruction::SBFM {
  1454→                        rd: rt,
  1455→                        rn: rt,
  1456→                        immr: 0,
  1457→                        imms: 15,
  1458→                    })?;
  1459→                }
  1460→            }
  1461→            IRType::I32 | IRType::U32 => {
  1462→                // 32-bit load uses LDR Wt encoding (scale=2, offset must be multiple of 4)
  1463→                if offset >= 0 && offset <= 16380 && offset % 4 == 0 {
  1464→                    self.emit_instruction(Instruction::LDR_W { rt, rn, offset })?;
  1465→                } else {
  1466→                    self.emit_address_with_offset(rn, offset)?;
  1467→                    self.emit_instruction(Instruction::LDR_W {
  1468→                        rt,
  1469→                        rn: Register::X9,
  1470→                        offset: 0,
  1471→                    })?;
  1472→                }
  1473→                // Sign-extend for signed word loads (SXTW = SBFM Xd, Xn, #0, #31)
  1474→                if *ty == IRType::I32 {
  1475→                    self.emit_instruction(Instruction::SBFM {
  1476→                        rd: rt,
  1477→                        rn: rt,
  1478→                        immr: 0,
  1479→                        imms: 31,
  1480→                    })?;
  1481→                }
  1482→            }
  1483→            IRType::I64 | IRType::U64 | IRType::Ptr | IRType::Func => {
  1484→                // 64-bit load uses LDR Xt encoding (scale=3, offset must be multiple of 8)
  1485→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1486→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1487→                } else {
  1488→                    self.emit_address_with_offset(rn, offset)?;
  1489→                    self.emit_instruction(Instruction::LDR {
  1490→                        rt,
  1491→                        rn: Register::X9,
  1492→                        offset: 0,
  1493→                    })?;
  1494→                }
  1495→            }
  1496→            _ => {
  1497→                // Default: treat as 64-bit load
  1498→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1499→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1500→                } else {
  1501→                    self.emit_address_with_offset(rn, offset)?;
  1502→                    self.emit_instruction(Instruction::LDR {
  1503→                        rt,
  1504→                        rn: Register::X9,
  1505→                        offset: 0,
  1506→                    })?;
  1507→                }
  1508→            }
  1509→        }
  1510→        Ok(())
  1511→    }

  1594→    /// Compute `X9 = rn + offset` for load/store with an offset that does not
  1595→    /// fit the ARM64 unsigned-offset immediate encoding.
  1596→    ///
  1597→    /// Uses X9 as a scratch register. If the offset fits in a 12-bit unsigned
  1598→    /// immediate (0..4095), emits `ADD X9, rn, #offset`. Otherwise, loads the
  1599→    /// offset into X9 with MOVZ/MOVK and then emits `ADD X9, rn, X9`.
  1600→    fn emit_address_with_offset(
  1601→        &mut self,
  1602→        rn: Register,
  1603→        offset: i32,
  1604→    ) -> Result<()> {
  1605→        if offset >= 0 && offset <= 4095 {
  1606→            // Small positive offset: ADD X9, rn, #offset
  1607→            self.emit_instruction(Instruction::ADD {
  1608→                rd: Register::X9,
  1609→                rn,
  1610→                rm: Operand::Imm12(offset as u16),
  1611→            })?;
  1612→        } else {
  1613→            // Large or negative offset: load offset into X9, then ADD X9, rn, X9
  1614→            self.emit_load_immediate(Register::X9, offset as i64)?;
  1615→            self.emit_instruction(Instruction::ADD {
  1616→                rd: Register::X9,
  1617→                rn,
  1618→                rm: Operand::Reg {
  1619→                    reg: Register::X9,
  1620→                    shift: None,
  1621→                },
  1622→            })?;
  1623→        }
  1624→        Ok(())
  1625→    }

  1627→    fn resolve_reg(&mut self, val: &IRValue) -> Result<Register> {
  1628→        match val {
  1629→            IRValue::Register(id) => self.reg_alloc.allocate(*id),
  1630→            IRValue::Immediate(v) => {
  1631→                let temp = Register::X9;
  1632→                self.emit_load_immediate(temp, *v)?;
  1633→                Ok(temp)
  1634→            }
  1635→            IRValue::Address(addr) => {
  1636→                let temp = Register::X10;
  1637→                self.emit_load_immediate(temp, *addr as i64)?;
  1638→                Ok(temp)
  1639→            }
  1640→            IRValue::Label(_) => Err(CodegenError::EncodingError(
  1641→                "label value cannot be resolved to a register".into(),

  1773→    fn apply_fixups(&mut self) -> Result<()> {
  1774→        let fixups = std::mem::take(&mut self.fixups);
  1775→        for (word_idx, label) in &fixups {
  1776→            let target_offset = self.label_offsets.get(label).copied().unwrap_or(0);
  1777→            let offset = (target_offset as i32) - (*word_idx as i32);
  1778→            let old_word = self.code[*word_idx];
  1779→            let patched = (old_word & !0x03FFFFFF) | ((offset & 0x03FFFFFF) as u32);

  1781→        }
  1782→        Ok(())
  1783→    }

  1845→/// Compute the stack frame size for a function by summing its `Alloc`
  1846→/// instructions, adding 16 bytes for the FP/LR save pair, and rounding up
  1847→/// to 16-byte alignment.
  1848→fn compute_frame_size(func: &IRFunction) -> u16 {
  1849→    let mut total: u32 = 16; // FP/LR save pair
  1850→    for block in &func.blocks {
  1851→        for instr in &block.instructions {
  1852→            if let IRInstr::Alloc { size, .. } = instr {
  1853→                let aligned = (*size).div_ceil(16) * 16;
  1854→                total += aligned;
  1855→            }
  1856→        }
  1857→    }
  1858→    // Round up to 16-byte alignment (should already be, but be safe).
  1859→    total = (total + 15) & !15;
  1860→    total as u16
