  2590→    /// Helper: build a minimal IR function with a single instruction and
  2591→    /// a Ret, then run allocate_registers and return the encoded bytes
  2592→    /// for the instruction (skipping prologue).
  2593→    fn isel_single_instr(instr: IRInstr) -> Vec<u8> {
  2594→        let mut func = IRFunction::new("test");
  2595→        // vreg 0 = dst, vreg 1 = lhs (if any), vreg 2 = rhs (if any)
  2596→        func.current_block().instructions.push(instr);
  2597→        func.current_block().instructions.push(IRInstr::Ret {
  2598→            values: vec![IRValue::Register(0)],
  2599→        });
  2600→        let backend = X86_64Backend::new();
  2601→        let allocated = backend.allocate_registers(&func).unwrap();
  2602→        // The encoded bytes include prologue + instruction(s) + epilogue.
  2603→        // Concatenate all instructions and return the full encoded output.
  2604→        let mut bytes = Vec::new();
  2605→        for block in &allocated.blocks {
  2606→            for instr in &block.instructions {
  2607→                bytes.extend_from_slice(&instr.encoded);
  2608→            }
  2609→        }
  2610→        bytes
  2611→    }
