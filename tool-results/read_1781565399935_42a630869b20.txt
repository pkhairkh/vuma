     1→     1→//! # Cross-Backend Consistency Test Suite
     2→     2→//!
     3→     3→//! Compiles the same VUMA programs for all 8 backends and verifies they produce
     4→     4→//! equivalent, structurally valid results.  Each test constructs IR functions
     5→     5→//! directly (bypassing the SCG front-end), runs each backend's
     6→     6→//! `allocate_registers` + `encode_function`, and validates:
     7→     7→//!
     8→     8→//! - Binary output exists and has reasonable size
     9→     9→//! - For Wasm32: the module structure (magic, version, sections)
    10→    10→//! - For ELF backends: the ELF header (magic, class, machine type)
    11→    11→//!
    12→    12→//! # Test Programs
    13→    13→//!
    14→    14→//! | # | Program        | Semantics                                        | Expected result |
    15→    15→//! |---|----------------|--------------------------------------------------|-----------------|
    16→    16→//! | 1 | Simple         | `fn main() -> i64 { return 42; }`               | 42              |
    17→    17→//! | 2 | Arithmetic     | `fn main() -> i64 { return (10+20)*3 - 5; }`    | 85              |
    18→    18→//! | 3 | Memory         | alloc 8B, store 0x42424242, load, return low byte| 66 (0x42)       |
    19→    19→//! | 4 | Function call  | helper() returns 7; main returns helper()        | 7               |

    45→    45→/// Human-readable name for a BackendKind (for assertion messages).
    46→    46→fn backend_name(kind: BackendKind) -> &'static str {
    47→    47→    match kind {
    48→    48→        BackendKind::AArch64 => "aarch64",
    49→    49→        BackendKind::RiscV64 => "riscv64",
    50→    50→        BackendKind::Wasm32 => "wasm32",
    51→    51→        BackendKind::LoongArch64 => "loongarch64",
    52→    52→        BackendKind::X86_64 => "x86_64",
    53→    53→        BackendKind::Arm32 => "arm32",
    54→    54→        BackendKind::Mips64 => "mips64",
    55→    55→        BackendKind::PowerPC64 => "ppc64",
    56→    56→    }
    57→    57→}

    59→    59→/// ELF machine type for a BackendKind (0 for non-ELF targets).
    60→    60→fn elf_machine(kind: BackendKind) -> u16 {
    61→    61→    match kind {
    62→    62→        BackendKind::AArch64 => 183,   // EM_AARCH64
    63→    63→        BackendKind::RiscV64 => 243,   // EM_RISCV
    64→    64→        BackendKind::Wasm32 => 0,      // Not ELF
    65→    65→        BackendKind::LoongArch64 => 258, // EM_LOONGARCH
    66→    66→        BackendKind::X86_64 => 62,     // EM_X86_64
    67→    67→        BackendKind::Arm32 => 40,      // EM_ARM
    68→    68→        BackendKind::Mips64 => 8,      // EM_MIPS
    69→    69→        BackendKind::PowerPC64 => 21,  // EM_PPC64
    70→    70→    }
    71→    71→}

    73→    73→/// Expected output format for a BackendKind.
    74→    74→fn expected_output_format(kind: BackendKind) -> OutputFormat {
    75→    75→    match kind {
    76→    76→        BackendKind::Arm32 => OutputFormat::Elf32,
    77→    77→        BackendKind::Wasm32 => OutputFormat::WasmBinary,
    78→    78→        _ => OutputFormat::Elf64,
    79→    79→    }
    80→    80→}

    82→    82→/// Run the full `allocate_registers` + `encode_program` pipeline for a
    83→    83→/// multi-function program and return the final binary.
    84→    84→fn compile_program(
    85→    85→    backend: &dyn Backend,
    86→    86→    functions: &[IRFunction],
    87→    87→    label: &str,
    88→    88→) -> Vec<u8> {
    89→    89→    let mut allocated_functions = Vec::new();
    90→    90→    for func in functions {
    91→    91→        let allocated = backend
    92→    92→            .allocate_registers(func)
    93→    93→            .unwrap_or_else(|e| {
    94→    94→                panic!(
    95→    95→                    "{}: allocate_registers failed for {} / {}: {}",
    96→    96→                    backend.name(),
    97→    97→                    label,
    98→    98→                    func.name,
    99→    99→                    e
   100→   100→                )
   101→   101→            });
   102→   102→        allocated_functions.push(allocated);
   103→   103→    }
   104→   104→
   105→   105→    let total_code_size: usize = allocated_functions
   106→   106→        .iter()
   107→   107→        .map(|f| f.code_size)
   108→   108→        .sum();
   109→   109→
   110→   110→    let program = AllocatedProgram {
   111→   111→        functions: allocated_functions,
   112→   112→        total_code_size,
   113→   113→        total_data_size: 0,
   114→   114→    };
   115→   115→
   116→   116→    backend
   117→   117→        .encode_program(&program)
   118→   118→        .unwrap_or_else(|e| {
   119→   119→            panic!(
   120→   120→                "{}: encode_program failed for {}: {}",
   121→   121→                backend.name(),
   122→   122→                label,
   123→   123→                e
   124→   124→            )
   125→   125→        })
   126→   126→}

   128→   128→/// Validate the ELF header of a compiled binary for the given backend.
   129→   129→fn validate_elf_header(bytes: &[u8], kind: BackendKind) {
   130→   130→    let name = backend_name(kind);
   131→   131→
   132→   132→    // ELF header must be at least 52 bytes (ELF32) or 64 bytes (ELF64).
   133→   133→    let min_header = match expected_output_format(kind) {
   134→   134→        OutputFormat::Elf32 => 52,
   135→   135→        OutputFormat::Elf64 => 64,
   136→   136→        _ => panic!("validate_elf_header called for non-ELF backend {}", name),
   137→   137→    };
   138→   138→    assert!(
   139→   139→        bytes.len() >= min_header,
   140→   140→        "{}: ELF binary too short ({} bytes, need at least {})",
   141→   141→        name,
   142→   142→        bytes.len(),
   143→   143→        min_header
   144→   144→    );
   145→   145→
   146→   146→    // Magic bytes: 0x7f 'E' 'L' 'F'
   147→   147→    assert_eq!(
   148→   148→        &bytes[0..4],
   149→   149→        &[0x7f, b'E', b'L', b'F'],
   150→   150→        "{}: ELF magic bytes incorrect",
   151→   151→        name
   152→   152→    );
   153→   153→
   154→   154→    // ELF class
   155→   155→    let expected_class = match expected_output_format(kind) {
   156→   156→        OutputFormat::Elf32 => 1u8, // ELFCLASS32
   157→   157→        OutputFormat::Elf64 => 2u8, // ELFCLASS64
   158→   158→        _ => unreachable!(),
   159→   159→    };
   160→   160→    assert_eq!(
   161→   161→        bytes[4], expected_class,
   162→   162→        "{}: ELF class should be {}",
   163→   163→        name, expected_class
   164→   164→    );
   165→   165→
   166→   166→    // ELF version must be EV_CURRENT (1)
   167→   167→    assert_eq!(bytes[6], 1, "{}: ELF version should be EV_CURRENT (1)", name);
   168→   168→
   169→   169→    // Machine type at offset 18..20 — read using the ELF's declared byte order.
   170→   170→    // Byte 5 (ei_data): 1 = little-endian, 2 = big-endian.
   171→   171→    let e_machine = if bytes[5] == 2 {
   172→   172→        u16::from_be_bytes([bytes[18], bytes[19]])
   173→   173→    } else {
   174→   174→        u16::from_le_bytes([bytes[18], bytes[19]])
   175→   175→    };
   176→   176→    assert_eq!(
   177→   177→        e_machine,
   178→   178→        elf_machine(kind),
   179→   179→        "{}: ELF machine type should be {} (got {})",
   180→   180→        name,
   181→   181→        elf_machine(kind),
   182→   182→        e_machine
   183→   183→    );
   184→   184→}

   186→   186→/// Validate the Wasm module structure of a compiled binary.
   187→   187→fn validate_wasm_module(bytes: &[u8]) {
   188→   188→    // Must have at least 8 bytes (magic + version)
   189→   189→    assert!(
   190→   190→        bytes.len() >= 8,
   191→   191→        "wasm32: binary too short ({} bytes, need at least 8)",
   192→   192→        bytes.len()
   193→   193→    );
   194→   194→
   195→   195→    // Magic: 0x00 0x61 0x73 0x6D ("\0asm")
   196→   196→    assert_eq!(
   197→   197→        &bytes[0..4],
   198→   198→        &[0x00, 0x61, 0x73, 0x6D],
   199→   199→        "wasm32: magic bytes should be \\0asm"
   200→   200→    );
   201→   201→
   202→   202→    // Version: 0x01 0x00 0x00 0x00 (version 1)
   203→   203→    assert_eq!(
   204→   204→        &bytes[4..8],
   205→   205→        &[0x01, 0x00, 0x00, 0x00],
   206→   206→        "wasm32: version should be 1"
   207→   207→    );
   208→   208→
   209→   209→    // Verify at least some sections exist after the header
   210→   210→    assert!(
   211→   211→        bytes.len() > 8,
   212→   212→        "wasm32: module should have content after header"
   213→   213→    );
   214→   214→
   215→   215→    // Walk sections and verify they appear in ascending ID order
   216→   216→    let mut offset = 8usize;
   217→   217→    let mut last_section_id: Option<u8> = None;
   218→   218→    while offset < bytes.len() {
   219→   219→        let section_id = bytes[offset];
   220→   220→        offset += 1;
   221→   221→
   222→   222→        // Decode LEB128 size
   223→   223→        let mut size: usize = 0;
   224→   224→        let mut shift: usize = 0;
   225→   225→        loop {
   226→   226→            assert!(offset < bytes.len(), "wasm32: truncated section size");
   227→   227→            let byte = bytes[offset];
   228→   228→            offset += 1;
   229→   229→            size |= ((byte & 0x7F) as usize) << shift;
   230→   230→            shift += 7;
   231→   231→            if byte & 0x80 == 0 {
   232→   232→                break;
   233→   233→            }
   234→   234→        }
   235→   235→
   236→   236→        // Sections must appear in order of ascending ID (except custom sections, ID 0)
   237→   237→        if section_id != 0 {
   238→   238→            if let Some(prev) = last_section_id {
   239→   239→                assert!(
   240→   240→                    section_id > prev,
   241→   241→                    "wasm32: sections out of order ({} after {})",
   242→   242→                    section_id,
   243→   243→                    prev
   244→   244→                );
   245→   245→            }
   246→   246→            last_section_id = Some(section_id);
   247→   247→        }
   248→   248→
   249→   249→        offset += size;
   250→   250→    }
   251→   251→}
   252→   252→
   253→   253→/// Validate a binary produced by any backend: check format-specific structure
   254→   254→/// and that the output has a reasonable minimum size.
   255→   255→fn validate_binary(bytes: &[u8], kind: BackendKind, min_size: usize) {
   256→   256→    let name = backend_name(kind);
   257→   257→
   258→   258→    // Reasonable minimum size (at least a few instructions)
   259→   259→    assert!(
   260→   260→        bytes.len() >= min_size,
   261→   261→        "{}: binary too small ({} bytes, expected at least {})",
   262→   262→        name,
   263→   263→        bytes.len(),
   264→   264→        min_size
   265→   265→    );
   266→   266→
   267→   267→    match expected_output_format(kind) {
   268→   268→        OutputFormat::Elf32 | OutputFormat::Elf64 => validate_elf_header(bytes, kind),
   269→   269→        OutputFormat::WasmBinary => validate_wasm_module(bytes),
   270→   270→        OutputFormat::RawBinary => {
   271→   271→            // No structural validation for raw binaries
   272→   272→        }
   273→   273→    }
   274→   274→}

   280→   280→/// Program 1: Simple — a function that returns 42.
   281→   281→///
   282→   282→/// ```text
   283→   283→/// fn main() -> i64 { return 42; }

   285→   285→fn make_simple_function() -> IRFunction {
   286→   286→    let mut func = IRFunction::new("main");
   287→   287→    func.result_types.push(IRType::I64);
   288→   288→    func.results.push(IRValue::Register(0));
   289→   289→    func.vregs.insert(0, VirtualRegister::new(0, Some("ret_val".to_string())));

   295→   295→/// Program 2: Arithmetic — computes (10 + 20) * 3 - 5 = 85.
   296→   296→///
   297→   297→/// ```text
   298→   298→/// fn main() -> i64 {
   299→   299→///     let a = 10 + 20;   // 30
   300→   300→///     let b = a * 3;     // 90
   301→   301→///     let c = b - 5;     // 85
   302→   302→///     return c;
   303→   303→/// }
   304→   304→/// ```
   305→   305→fn make_arithmetic_function() -> IRFunction {
   306→   306→    let mut func = IRFunction::new("main");
   307→   307→    func.result_types.push(IRType::I64);
   308→   308→    func.results.push(IRValue::Register(2));
   309→   309→    func.vregs.insert(0, VirtualRegister::new(0, Some("a".to_string())));
   310→   310→    func.vregs.insert(1, VirtualRegister::new(1, Some("b".to_string())));
   311→   311→    func.vregs.insert(2, VirtualRegister::new(2, Some("c".to_string())));
   312→   312→
   313→   313→    let block = func.current_block();
   314→   314→
   315→   315→    // a = 10 + 20
   316→   316→    block.push(IRInstr::Add {
   317→   317→        dst: IRValue::Register(0),
   318→   318→        lhs: IRValue::Immediate(10),
   319→   319→        rhs: IRValue::Immediate(20),
   320→   320→        ty: Some(IRType::I64),
   321→   321→    });
   322→   322→
   323→   323→    // b = a * 3
   324→   324→    block.push(IRInstr::BinOp {
   325→   325→        op: BinOpKind::Mul,
   326→   326→        dst: IRValue::Register(1),
   327→   327→        lhs: IRValue::Register(0),
   328→   328→        rhs: IRValue::Immediate(3),
   329→   329→        ty: Some(IRType::I64),
   330→   330→    });
   331→   331→
   332→   332→    // c = b - 5
   333→   333→    block.push(IRInstr::BinOp {
   334→   334→        op: BinOpKind::Sub,
   335→   335→        dst: IRValue::Register(2),
   336→   336→        lhs: IRValue::Register(1),
   337→   337→        rhs: IRValue::Immediate(5),
   338→   338→        ty: Some(IRType::I64),
   339→   339→    });
   340→   340→
   341→   341→    block.terminator = IRTerminator::Return(vec![IRValue::Register(2)]);
   342→   342→    func
   343→   343→}
   344→   344→
   345→   345→/// Program 3: Memory — allocates 8 bytes, writes 0x42424242, reads it back,
   346→   346→/// returns the low byte (0x42 = 66).
   347→   347→///
   348→   348→/// ```text
   349→   349→/// fn main() -> i64 {
   350→   350→///     let ptr = alloc 8;
   351→   351→///     store 0x42424242 at ptr;
   352→   352→///     let val = load ptr as i64;
   353→   353→///     let byte = val & 0xFF;
   354→   354→///     return byte;          // 0x42 = 66
   355→   355→/// }
   356→   356→/// ```
   357→   357→fn make_memory_function() -> IRFunction {
   358→   358→    let mut func = IRFunction::new("main");
   359→   359→    func.result_types.push(IRType::I64);
   360→   360→    func.results.push(IRValue::Register(2));
   361→   361→    func.vregs.insert(0, VirtualRegister::new(0, Some("ptr".to_string())));
   362→   362→    func.vregs.insert(1, VirtualRegister::new(1, Some("val".to_string())));
   363→   363→    func.vregs.insert(2, VirtualRegister::new(2, Some("byte".to_string())));
   364→   364→
   365→   365→    let block = func.current_block();
   366→   366→
   367→   367→    // ptr = alloc 8
   368→   368→    block.push(IRInstr::Alloc {
   369→   369→        dst: IRValue::Register(0),
   370→   370→        size: 8,
   371→   371→    });
   372→   372→
   373→   373→    // store 0x42424242 at ptr + 0
   374→   374→    block.push(IRInstr::Store {
   375→   375→        value: IRValue::Immediate(0x42424242),
   376→   376→        addr: IRValue::Register(0),
   377→   377→        offset: 0,
   378→   378→        ty: IRType::I64,
   379→   379→    });
   380→   380→
   381→   381→    // val = load ptr + 0 as i64
   382→   382→    block.push(IRInstr::Load {
   383→   383→        dst: IRValue::Register(1),
   384→   384→        addr: IRValue::Register(0),
   385→   385→        offset: 0,
   386→   386→        ty: IRType::I64,
   387→   387→    });
   388→   388→
   389→   389→    // byte = val & 0xFF
   390→   390→    block.push(IRInstr::BinOp {
   391→   391→        op: BinOpKind::And,
   392→   392→        dst: IRValue::Register(2),
   393→   393→        lhs: IRValue::Register(1),
   394→   394→        rhs: IRValue::Immediate(0xFF),
   395→   395→        ty: Some(IRType::I64),
   396→   396→    });
   397→   397→
   398→   398→    block.terminator = IRTerminator::Return(vec![IRValue::Register(2)]);
   399→   399→    func
   400→   400→}

   402→   402→/// Program 4: Function call — a helper that returns 7, main calls it
   403→   403→/// and returns the result.
   404→   404→///
   405→   405→/// ```text
   406→   406→/// fn helper() -> i64 { return 7; }
   407→   407→/// fn main() -> i64 { return helper(); }
   408→   408→/// ```
   409→   409→fn make_function_call_program() -> Vec<IRFunction> {
   410→   410→    // Helper function: returns 7
   411→   411→    let mut helper = IRFunction::new("helper");
   412→   412→    helper.result_types.push(IRType::I64);
   413→   413→    helper.results.push(IRValue::Register(0));
   414→   414→    helper
   415→   415→        .vregs
   416→   416→        .insert(0, VirtualRegister::new(0, Some("ret".to_string())));
   417→   417→    helper.current_block().terminator = IRTerminator::Return(vec![IRValue::Immediate(7)]);
   418→   418→
   419→   419→    // Main function: calls helper and returns the result
   420→   420→    let mut main_fn = IRFunction::new("main");
   421→   421→    main_fn.result_types.push(IRType::I64);
   422→   422→    main_fn.results.push(IRValue::Register(0));
   423→   423→    main_fn
   424→   424→        .vregs
   425→   425→        .insert(0, VirtualRegister::new(0, Some("result".to_string())));
   426→   426→
   427→   427→    main_fn.current_block().push(IRInstr::Call {
   428→   428→        dst: Some(IRValue::Register(0)),
   429→   429→        func: "helper".to_string(),
   430→   430→        args: vec![],
   431→   431→        is_extern: false,
   432→   432→    });
   433→   433→    main_fn.current_block().terminator = IRTerminator::Return(vec![IRValue::Register(0)]);
   434→   434→
   435→   435→    vec![helper, main_fn]
   436→   436→}

   442→   442→/// Test 1: Simple program — `fn main() -> i64 { return 42; }`
   443→   443→///
   444→   444→/// Validates that all 8 backends can compile a trivial return-constant
   445→   445→/// function and produce structurally valid output.
   446→   446→#[test]
   447→   447→fn test_cross_backend_simple_return() {
   448→   448→    let func = make_simple_function();
   449→   449→
   450→   450→    for &kind in ALL_BACKENDS {
   451→   451→        let name = backend_name(kind);
   452→   452→        let backend = create_backend(kind).expect("backend creation should succeed");
   453→   453→
   454→   454→        // --- allocate_registers + encode_function ---
   455→   455→        let allocated = backend
   456→   456→            .allocate_registers(&func)
   457→   457→            .unwrap_or_else(|e| panic!("{}: allocate_registers failed: {}", name, e));
   458→   458→
   459→   459→        // The allocated function should have at least one block
   460→   460→        assert!(
   461→   461→            !allocated.blocks.is_empty(),
   462→   462→            "{}: allocated function should have at least one block",
   463→   463→            name
   464→   464→        );
   465→   465→
   466→   466→        // The block should contain at least one instruction
   467→   467→        let total_instrs: usize = allocated
   468→   468→            .blocks
   469→   469→            .iter()
   470→   470→            .map(|b| b.instructions.len())
   471→   471→            .sum();
   472→   472→        assert!(
   473→   473→            total_instrs > 0,
   474→   474→            "{}: allocated function should have instructions",
   475→   475→            name
   476→   476→        );
   477→   477→
   478→   478→        // Encode the function
   479→   479→        let code = backend
   480→   480→            .encode_function(&allocated)
   481→   481→            .unwrap_or_else(|e| panic!("{}: encode_function failed: {}", name, e));
   482→   482→
   483→   483→        // Even a trivial function needs at least 4 bytes of machine code
   484→   484→        assert!(
   485→   485→            code.len() >= 4,
   486→   486→            "{}: encoded function too small ({} bytes)",
   487→   487→            name,
   488→   488→            code.len()
   489→   489→        );
