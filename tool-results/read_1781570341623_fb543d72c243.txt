     1→//! VUMA CLI — Command-line interface for the VUMA compiler.
     2→//!
     3→//! Subcommands:
     4→//! - `vuma build <file>` — Parse + compile to ARM64 ELF (default), save to output file
     5→//! - `vuma run <file>`   — Build + execute (via QEMU aarch64 or native)
     6→//! - `vuma check <file>` — Parse + SCG + BD inference + IVE verification only
     7→//! - `vuma emit <isa> <file>` — Compile to specific ISA
     8→//! - `vuma disasm <file>` — Read binary and disassemble
     9→//! - `vuma verify <file>` — Run IVE 5-invariant verification
    10→//! - `vuma repl` — Interactive REPL (parse expr, print AST)
    11→//! - `vuma lsp`  — Start Language Server (LSP) for IDE/LLM integration
    12→

   336→/// Determine the default output path for a given input file.
   337→fn default_output_path(input: &Path) -> PathBuf {
   338→    let stem = input.file_stem().unwrap_or_default().to_string_lossy();
   339→    let dir = input.parent().unwrap_or(std::path::Path::new("."));
   340→    dir.join(format!("{}.o", stem))
   341→}

   347→/// `vuma build <file>` — Parse + compile to ELF, save to output file.
   348→fn cmd_build(
   349→    cli: &Cli,
   350→    file: &PathBuf,
   351→    output: &Option<PathBuf>,
   352→    target: &TargetArg,
   353→) -> Result<(), String> {
   354→    let source = read_source(file)?;
   355→    let config = make_config(cli, CompileTarget::from(*target));
   356→
   357→    // Initialize telemetry collector if --telemetry is set.
   358→    let mut telemetry = if cli.telemetry {
   359→        let mut tc = TelemetryCollector::new();
   360→        tc.set_opt_level(&format!("{:?}", cli.opt_level));
   361→        tc.set_verification_level(&format!("{:?}", cli.verification));
   362→        tc.set_target(&format!("{:?}", target));
   363→        tc.set_debug_info(cli.debug);
   364→        Some(tc)
   365→    } else {
   366→        None
   367→    };
   368→
   369→    if let Some(ref mut tc) = telemetry { tc.stage_start("compile"); }
   370→
   371→    let compile_result = vuma::compile_with_recovery(&source, Some(file), &config);
   372→
   373→    if let Some(ref mut tc) = telemetry { tc.stage_end("compile"); }
   374→
   375→    match compile_result {
   376→        CompileResult::Success(result) => {
   377→            if let Some(ref mut tc) = telemetry {
   378→                tc.set_scg_node_count(result.scg.node_count());
   379→                tc.set_ir_function_count(result.ir_function_count);
   380→                tc.set_ir_instruction_count(result.ir_instruction_count);
   381→                tc.set_binary_size(result.binary.len());
   382→            }
   383→
   384→            let out_path = output
   385→                .as_ref()
   386→                .cloned()
   387→                .unwrap_or_else(|| default_output_path(file));
   388→            fs::write(&out_path, &result.binary).map_err(|e| {
   389→                format!(
   390→                    "error: cannot write output file '{}': {}",
   391→                    out_path.display(),
   392→                    e
   393→                )
   394→            })?;
   395→            // Make the output file executable on Unix.
   396→            #[cfg(unix)]
   397→            {
   398→                use std::os::unix::fs::PermissionsExt;
   399→                let mut perms = fs::metadata(&out_path)
   400→                    .map_err(|e| format!("error: cannot stat '{}': {}", out_path.display(), e))?
   401→                    .permissions();
   402→                perms.set_mode(0o755);
   403→                fs::set_permissions(&out_path, perms)
   404→                    .map_err(|e| format!("error: cannot chmod '{}': {}", out_path.display(), e))?;
   405→            }
   406→
   407→            if !cli.quiet {
   408→                println!(
   409→                    "Compiled {} -> {} ({} bytes, {} SCG nodes, {} IR instructions)",
   410→                    file.display(),
   411→                    out_path.display(),
   412→                    result.binary.len(),
   413→                    result.scg.node_count(),
   414→                    result.ir_instruction_count,
   415→                );
   416→
   417→                // Print stage timings.
   418→                for (stage, ms) in &result.stage_timings {
   419→                    println!("  {:20} {}ms", stage, ms);
   420→                }
   421→            }
   422→
   423→            // Output telemetry if requested.
   424→            if let Some(tc) = telemetry {
   425→                let report = tc.finalize();
   426→                println!("\n{}", serde_json::to_string_pretty(&report).unwrap());
   427→            }
   428→
   429→            Ok(())
   430→        }
   431→        CompileResult::Partial(partial) => {
   432→            print_errors(&partial.diagnostics);
   433→
   434→            if cli.telemetry {
   435→                if let Some(tc) = telemetry {
   436→                    let report = tc.finalize();
   437→                    eprintln!("\n{}", serde_json::to_string_pretty(&report).unwrap());
   438→                }
   439→            }
   440→
   441→            Err(format!(
   442→                "compilation failed with {} error(s) (last completed stage: {})",
   443→                partial.diagnostics.len(),
   444→                partial.last_completed_stage
   445→                    .map(|s| format!("{:?}", s))
   446→                    .unwrap_or_else(|| "none".to_string())
   447→            ))
   448→        }
   449→    }
   450→}

   452→/// `vuma run <file>` — Build + execute.
   453→fn cmd_run(cli: &Cli, file: &PathBuf, args: &[String]) -> Result<(), String> {
   454→    let source = read_source(file)?;
   455→    let config = make_config(cli, CompileTarget::Linux);
   456→    let result = compile_with_path(&source, Some(file), &config).map_err(|errors| {
   457→        print_errors(&errors);
   458→        global_logger().error("compile", &format!("compilation failed with {} error(s)", errors.len()));
   459→        format!("compilation failed with {} error(s)", errors.len())
   460→    })?;
   461→    let tmp_dir = std::env::temp_dir();
   462→    let exe_path = tmp_dir.join(format!("vuma_run_{}", std::process::id()));
   463→    fs::write(&exe_path, &result.binary).map_err(|e| {
   464→        format!(
   465→            "error: cannot write temporary executable '{}': {}",
   466→            exe_path.display(),
   467→            e
   468→        )
   469→    })?;
   470→
   471→    // Make the file executable on Unix.
   472→    #[cfg(unix)]
   473→    {
   474→        use std::os::unix::fs::PermissionsExt;
   475→        let mut perms = fs::metadata(&exe_path)
   476→            .map_err(|e| format!("error: cannot stat '{}': {}", exe_path.display(), e))?
   477→            .permissions();
   478→        perms.set_mode(0o755);
   479→        fs::set_permissions(&exe_path, perms)
   480→            .map_err(|e| format!("error: cannot chmod '{}': {}", exe_path.display(), e))?;
   481→    }
   482→
   483→    // Try to execute: first natively (if on aarch64), then via qemu-aarch64.
   484→    let native_output = Command::new(&exe_path).args(args).output();
   485→
   486→    let output = match native_output {
   487→        Ok(out) => out,
   488→        Err(_) => {
   489→            // Native execution failed; try qemu-aarch64.
   490→            let qemu_output = Command::new("qemu-aarch64")
   491→                .arg(&exe_path)
   492→                .args(args)
   493→                .output()
   494→                .map_err(|e| {
   495→                    format!(
   496→                        "error: failed to execute binary (neither native nor qemu-aarch64): {}",
   497→                        e
   498→                    )
   499→                })?;
   500→            qemu_output
   501→        }
   502→    };
   503→
   504→    io::stdout()
   505→        .write_all(&output.stdout)
   506→        .map_err(|e| format!("error: failed to write program output: {}", e))?;
   507→    io::stderr()
   508→        .write_all(&output.stderr)
   509→        .map_err(|e| format!("error: failed to write program stderr: {}", e))?;
   510→
   511→    // Clean up temp file.
   512→    let _ = fs::remove_file(&exe_path);
   513→
   514→    let code = output.status.code().unwrap_or(1);
   515→    if code != 0 {
   516→        return Err(format!("program exited with code {}", code));
   517→    }
   518→
   519→    Ok(())
   520→}

   522→/// `vuma check <file>` — Parse + SCG + BD inference + IVE verification only.
   523→fn cmd_check(cli: &Cli, file: &PathBuf) -> Result<(), String> {
   524→    let source = read_source(file)?;
   525→    let mut config = make_config(cli, CompileTarget::Linux);
   526→    // check mode: always run verification, don't skip
   527→    if config.verification_level == VerificationLevel::None {
   528→        config.verification_level = VerificationLevel::Normal;
   529→    }
   530→
   531→    // Run the full compile — the check command doesn't save the binary,
   532→    // it just verifies the program compiles and passes verification.
   533→    let result = compile_with_path(&source, Some(file), &config).map_err(|errors| {
   534→        print_errors(&errors);
   535→        global_logger().error("check", &format!("check failed with {} error(s)", errors.len()));
   536→        format!("check failed with {} error(s)", errors.len())
   537→    })?;
   538→
   539→    if !cli.quiet {
   540→        println!("Check passed for {}", file.display());
   541→        println!("  SCG nodes:      {}", result.scg.node_count());
   542→        println!("  IR functions:   {}", result.ir_function_count);
   543→        println!("  IR instructions: {}", result.ir_instruction_count);
   544→
   545→        if let Some(ref verification) = result.verification {
   546→            println!("  Verification:   {}", verification.overall);
   547→        }
   548→
   549→        // Print stage timings.
   550→        for (stage, ms) in &result.stage_timings {
   551→            println!("  {:20} {}ms", stage, ms);
   552→        }
   553→    }
   554→
   555→    Ok(())
   556→}

   558→/// `vuma emit <isa> <file>` — Compile to a specific ISA target.
   559→///
   560→/// Uses the direct AST → codegen SCG → IR path for better code quality
   561→/// than the main pipeline (which goes through the semantic SCG and loses
   562→/// most program semantics in the SCG→IR bridge).
   563→fn cmd_emit(
   564→    cli: &Cli,
   565→    isa: &IsaArg,
   566→    file: &PathBuf,
   567→    output: &Option<PathBuf>,
   568→) -> Result<(), String> {
   569→    let source = read_source(file)?;
   570→    let backend_kind = BackendKind::from(*isa);
   571→
   572→    // Step 1: Parse source → AST.
   573→    let mut parser = vuma_parser::Parser::new(&source);
   574→    let parse_result = parser.parse_program();
   575→    if parse_result.is_err() {
   576→        return Err(format!("parse error: {:?}", parse_result.errors));
   577→    }
   578→    if !parse_result.errors.is_empty() {
   579→        eprintln!("[emit] WARNING: {} non-fatal parse errors:", parse_result.errors.len());
   580→        for err in &parse_result.errors {
   581→            eprintln!("[emit]   {:?}", err);
   582→        }
   583→    }
   584→    let program = parse_result.value.unwrap();
   585→
   586→    // Step 2: Bridge parser AST → codegen SCG.
   587→    let codegen_scg = bridge_ast_to_codegen_scg(&program);
   588→
   589→    // Step 3: Lower codegen SCG → IR.
   590→    let mut ir_builder = ScgToIr::new();
   591→    let ir_program = ir_builder.convert(&codegen_scg).map_err(|e| {
   592→        format!("IR conversion error: {}", e)
   593→    })?;
   594→
   595→    eprintln!("[emit] IR program has {} functions", ir_program.functions.len());
   596→    for func in &ir_program.functions {
   597→        eprintln!("[emit] Function: {} ({} params, {} vregs)", func.name, func.params.len(), func.vregs.len());
   598→        for block in &func.blocks {
   599→            eprintln!("[emit]   Block: {}", block.label);
   600→            for instr in &block.instructions {
   601→                eprintln!("[emit]     {:?}", instr);
   602→            }
   603→        }
   604→    }
   605→
   606→    // Step 4: Create backend and allocate registers (with fallback).
   607→    let backend = match create_backend(backend_kind) {
   608→        Ok(b) => b,
   609→        Err(e) => {
   610→            let err_msg = format!("{}", e);
   611→            global_logger().warn("emit", &format!("{} backend failed: {}", backend_kind.isa_name(), err_msg));
   612→            // Try fallback to AArch64 if not already trying it
   613→            if backend_kind != BackendKind::AArch64 {
   614→                global_logger().info("emit", "falling back to aarch64 backend");
   615→                create_backend(BackendKind::AArch64).map_err(|e2| {
   616→                    format!("error: cannot create {} backend: {}, aarch64 fallback also failed: {}",
   617→                        backend_kind.isa_name(), err_msg, e2)
   618→                })?
   619→            } else {
   620→                return Err(format!("error: cannot create {} backend: {}", backend_kind.isa_name(), err_msg));
   621→            }
   622→        }
   623→    };
   624→
   625→    let mut allocated_functions = Vec::new();
   626→    for func in &ir_program.functions {
   627→        match backend.allocate_registers(func) {
   628→            Ok(allocated) => allocated_functions.push(allocated),
   629→            Err(e) => {
   630→                eprintln!("warning: register allocation failed for '{}': {}", func.name, e);
   631→            }
   632→        }
   633→    }
   634→
   635→    let out_path = output
   636→        .as_ref()
   637→        .cloned()
   638→        .unwrap_or_else(|| default_output_path(file));
   639→
   640→    // Step 5: Encode and write output.
   641→    if !allocated_functions.is_empty() {
   642→        let allocated_program = vuma_codegen::backend::AllocatedProgram {
   643→            functions: allocated_functions,
   644→            total_code_size: 0,
   645→            total_data_size: 0,
   646→        };
   647→        match backend.encode_program(&allocated_program) {
   648→            Ok(bytes) => {
   649→                fs::write(&out_path, &bytes).map_err(|e| {
   650→                    format!("error: cannot write output file '{}': {}", out_path.display(), e)
   651→                })?;
   652→                // Make the output file executable on Unix.
   653→                #[cfg(unix)]
   654→                {
   655→                    use std::os::unix::fs::PermissionsExt;
   656→                    let mut perms = fs::metadata(&out_path)
   657→                        .map_err(|e| format!("error: cannot stat '{}': {}", out_path.display(), e))?
   658→                        .permissions();
   659→                    perms.set_mode(0o755);
   660→                    fs::set_permissions(&out_path, perms)
   661→                        .map_err(|e| format!("error: cannot chmod '{}': {}", out_path.display(), e))?;
   662→                }
   663→                println!(
   664→                    "Emitted {} -> {} ({} bytes, ISA: {})",
   665→                    file.display(),
   666→                    out_path.display(),
   667→                    bytes.len(),
   668→                    backend.name(),
   669→                );
   670→            }
   671→            Err(e) => {
   672→                use vuma_codegen::backend::BackendError;
   673→                let prefix = match &e {
   674→                    BackendError::UnresolvedRelocation { .. } => "E037",
   675→                    _ => "error",
   676→                };
   677→                return Err(format!("{}: {} encoding failed: {}", prefix, backend.name(), e));
   678→            }
   679→        }
   680→    } else {
   681→        return Err("no functions were successfully allocated".to_string());
   682→    }
   683→
   684→    Ok(())
   685→}
   686→
   687→/// Bridge parser AST → codegen SCG.
   688→///
   689→/// This converts the parser's AST into the codegen's SCG representation,
   690→/// which can then be lowered to IR. This bypasses the main pipeline's
   691→/// semantic SCG which loses most program semantics.
   692→///
   693→/// Key design: expressions are flattened into three-address code (TAC) by
   694→/// introducing temporary variables for sub-expressions. This preserves the
   695→/// full semantics of nested binary operations, function calls, dereferences,
   696→/// and casts.
   697→fn bridge_ast_to_codegen_scg(program: &vuma_parser::ast::Program) -> Scg {
   698→    use vuma_parser::ast::Item;
   699→
   700→    let mut nodes: Vec<ScgNode> = Vec::new();
   701→
   702→    for item in &program.items {
   703→        if let Item::FnDef(fn_def) = item {
   704→            let params: Vec<ScgParam> = fn_def
   705→                .params
   706→                .iter()
   707→                .map(|p| ScgParam {
   708→                    name: p.name.clone(),
   709→                    ty: bridge_type_to_codegen_scg(&p.ty),
   710→                })
   711→                .collect();
   712→
   713→            let results = if let Some(ref ret_ty) = fn_def.return_type {
   714→                vec![bridge_type_to_codegen_scg(&Some(ret_ty.clone()))]
   715→            } else {
   716→                vec![]
   717→            };
   718→
   719→            let mut ctx = BridgeCtx::new();
   720→            let mut body = bridge_block_to_scg_stmts(&fn_def.body, &mut ctx);
   721→
   722→            // Ensure every function ends with a Return statement.
   723→            // If the body doesn't end with a Return, add an implicit one.
   724→            // When the function has a return type and the last statement was an
   725→            // expression, use ctx.last_expr_result as the return value.
   726→            let has_return = body.last().map_or(false, |s| matches!(s, ScgStatement::Return(_)));
   727→            if !has_return {
   728→                let ret_val = if !results.is_empty() {
   729→                    // First check if the last expression was tracked.
   730→                    if let Some(ref expr) = ctx.last_expr_result {
   731→                        Some(expr.clone())
   732→                    } else {
   733→                        // Otherwise, look for the last computation/call result.
   734→                        body.iter().rev().find_map(|s| match s {
   735→                            ScgStatement::Computation(comp) => Some(ScgExpr::Var(comp.dst.clone())),
   736→                            ScgStatement::Call(call) => call.dst.as_ref().map(|d| ScgExpr::Var(d.clone())),
   737→                            _ => None,
   738→                        })
   739→                    }
   740→                } else {
   741→                    None
   742→                };
   743→                body.push(ScgStatement::Return(ret_val.into_iter().collect()));
   744→            }
   745→
   746→            nodes.push(ScgNode::Function(ScgFunction {
   747→                name: fn_def.name.clone(),
   748→                params,
   749→                results,
   750→                body,
   751→            }));
   752→        }
   753→    }
   754→
   755→    Scg { nodes }
   756→}
   757→
   758→/// Context for the AST → codegen SCG bridge, tracking a monotonic temp counter
   759→/// and the last expression result for implicit returns.
   760→struct BridgeCtx {
   761→    temp_counter: u32,
   762→    /// The result of the last expression statement, if any.
   763→    /// Used for implicit return when a function body ends with an expression.
   764→    last_expr_result: Option<ScgExpr>,
   765→}
   766→
   767→impl BridgeCtx {
   768→    fn new() -> Self {
   769→        Self { temp_counter: 0, last_expr_result: None }
   770→    }
   771→
   772→    /// Allocate a unique temporary variable name.
   773→    fn alloc_temp(&mut self) -> String {
   774→        let name = format!("__t{}", self.temp_counter);
   775→        self.temp_counter += 1;
   776→        name
   777→    }
   778→}
   779→
   780→/// Convert a parser type annotation to a codegen SCG type.
   781→fn bridge_type_to_codegen_scg(ty: &Option<vuma_parser::ast::Type>) -> ScgType {
   782→    match ty {
   783→        Some(vuma_parser::ast::Type::BDBase(name)) => match name.as_str() {
   784→            "i8" => ScgType::I8,
   785→            "i16" => ScgType::I16,
   786→            "i32" => ScgType::I32,
   787→            "i64" => ScgType::I64,
   788→            "u8" => ScgType::U8,
   789→            "u16" => ScgType::U16,
   790→            "u32" => ScgType::U32,
   791→            "u64" => ScgType::U64,
   792→            _ => ScgType::I64,
   793→        },
   794→        Some(vuma_parser::ast::Type::Ptr(_)) => ScgType::Ptr,
   795→        Some(vuma_parser::ast::Type::RegionPtr { .. }) => ScgType::Ptr,
   796→        _ => ScgType::Void,
   797→    }
   798→}

   801→/// Convert a parser block into codegen SCG statements, flattening expressions
   802→/// into three-address code with temporaries.
   803→fn bridge_block_to_scg_stmts(block: &vuma_parser::ast::Block, ctx: &mut BridgeCtx) -> Vec<ScgStatement> {
   804→    block
   805→        .statements
   806→        .iter()
   807→        .flat_map(|s| bridge_stmt_to_scg(s, ctx))
   808→        .collect()
   809→}
   810→
   811→/// Flatten an expression into three-address code. Returns the ScgExpr that
   812→/// holds the result, and appends any intermediate computation statements
   813→/// to `stmts`.
   814→///
   815→/// This is the core of the bridge: it recursively decomposes nested
   816→/// expressions into a sequence of simple computation nodes, each operating
   817→/// on at most two operands and producing one result. This preserves the
   818→/// full semantics of the original expression tree.
   819→fn flatten_expr(
   820→    expr: &vuma_parser::ast::Expr,
   821→    stmts: &mut Vec<ScgStatement>,
   822→    ctx: &mut BridgeCtx,
   823→) -> ScgExpr {
   824→    use vuma_parser::ast::{Expr, Lit, UnOp};
   825→
   826→    match expr {
   827→        // ── Leaf expressions: return directly, no flattening needed ──
   828→        Expr::Var { name, .. } => ScgExpr::Var(name.clone()),
   829→        Expr::Lit { value, .. } => match value {
   830→            Lit::Int(n) => ScgExpr::Int(*n),
   831→            Lit::Float(f) => ScgExpr::Float(*f),
   832→            Lit::Bool(b) => ScgExpr::Int(if *b { 1 } else { 0 }),
   833→            Lit::Address(a) => ScgExpr::Int(*a as i64),
   834→            Lit::String(_) => ScgExpr::Int(0),
   835→        },
   836→
   837→        // ── Binary operations: flatten lhs and rhs, then emit one Computation ──
   838→        Expr::BinOp { op, lhs, rhs, .. } => {
   839→            let lhs_expr = flatten_expr(lhs, stmts, ctx);
   840→            let rhs_expr = flatten_expr(rhs, stmts, ctx);
   841→            let dst = ctx.alloc_temp();
   842→            let binop_kind = map_ast_binop(op);
   843→            stmts.push(ScgStatement::Computation(ComputationNode {
   844→                dst: dst.clone(),
   845→                op: binop_kind,
   846→                lhs: lhs_expr,
   847→                rhs: rhs_expr,
   848→                tail_call: false,
   849→            }));
   850→            ScgExpr::Var(dst)
   851→        }
   852→
   853→        // ── Unary operations: flatten operand, then emit one Computation ──
   854→        Expr::UnOp { op, expr: operand, .. } => {
   855→            let operand_expr = flatten_expr(operand, stmts, ctx);
   856→            let dst = ctx.alloc_temp();
   857→            match op {
   858→                UnOp::BitNot => {
   859→                    stmts.push(ScgStatement::Computation(ComputationNode {
   860→                        dst: dst.clone(),
   861→                        op: BinOpKind::Xor,
   862→                        lhs: operand_expr,
   863→                        rhs: ScgExpr::Int(-1),
   864→                        tail_call: false,
   865→                    }));
   866→                }
   867→                UnOp::Neg => {
   868→                    stmts.push(ScgStatement::Computation(ComputationNode {
   869→                        dst: dst.clone(),
   870→                        op: BinOpKind::Sub,
   871→                        lhs: ScgExpr::Int(0),
   872→                        rhs: operand_expr,
   873→                        tail_call: false,
   874→                    }));
   875→                }
   876→                UnOp::Not => {
   877→                    stmts.push(ScgStatement::Computation(ComputationNode {
   878→                        dst: dst.clone(),
   879→                        op: BinOpKind::Eq,
   880→                        lhs: operand_expr,
   881→                        rhs: ScgExpr::Int(0),
   882→                        tail_call: false,
   883→                    }));
   884→                }
   885→                UnOp::Deref => {
   886→                    stmts.push(ScgStatement::Access(AccessNode::Load {
   887→                        dst: dst.clone(),
   888→                        ptr: operand_expr,
   889→                        offset: None,
   890→                    }));
   891→                }
   892→            }
   893→            ScgExpr::Var(dst)
   894→        }
   895→
   896→        // ── Function call: flatten args, emit CallNode ──
   897→        Expr::Call { callee, args, .. } => {
   898→            let func_name = match callee.as_ref() {
   899→                Expr::Var { name, .. } => name.clone(),
   900→                _ => "_unknown".into(),
   901→            };
   902→            let flat_args: Vec<ScgExpr> = args.iter()
   903→                .map(|a| flatten_expr(a, stmts, ctx))
   904→                .collect();
   905→            let dst = ctx.alloc_temp();
   906→            stmts.push(ScgStatement::Call(CallNode {
   907→                dst: Some(dst.clone()),
   908→                func: func_name,
   909→                args: flat_args,
   910→                is_extern: false,
   911→            }));
   912→            ScgExpr::Var(dst)
   913→        }
   914→
   915→        // ── Dereference: flatten the address, emit Load ──
   916→        Expr::Deref { expr, .. } => {
   917→            let addr = flatten_expr(expr, stmts, ctx);
   918→            let dst = ctx.alloc_temp();
   919→            stmts.push(ScgStatement::Access(AccessNode::Load {
   920→                dst: dst.clone(),
   921→                ptr: addr,
   922→                offset: None,
   923→            }));
   924→            ScgExpr::Var(dst)
   925→        }
   926→
   927→        // ── Offset (pointer arithmetic): flatten base and offset, emit Add ──
   928→        Expr::Offset { base, offset, .. } => {
   929→            let base_expr = flatten_expr(base, stmts, ctx);
   930→            let off_expr = flatten_expr(offset, stmts, ctx);
   931→            let dst = ctx.alloc_temp();
   932→            stmts.push(ScgStatement::Computation(ComputationNode {
   933→                dst: dst.clone(),
   934→                op: BinOpKind::Add,
   935→                lhs: base_expr,
   936→                rhs: off_expr,
   937→                tail_call: false,
   938→            }));
   939→            ScgExpr::Var(dst)
   940→        }
   941→
   942→        // ── Cast: flatten operand, pass through ──
   943→        Expr::Cast { expr, .. } => flatten_expr(expr, stmts, ctx),
   944→
   945→        // ── TypeAscription: flatten inner expression ──
   946→        Expr::TypeAscription { expr, .. } => flatten_expr(expr, stmts, ctx),
   947→
   948→        // ── Index: flatten base and index, compute addr, emit Load ──
   949→        Expr::Index { expr, index, .. } => {
   950→            let base_expr = flatten_expr(expr, stmts, ctx);
   951→            let idx_expr = flatten_expr(index, stmts, ctx);
   952→            let addr = ctx.alloc_temp();
   953→            stmts.push(ScgStatement::Computation(ComputationNode {
   954→                dst: addr.clone(),
   955→                op: BinOpKind::Add,
   956→                lhs: base_expr,
   957→                rhs: idx_expr,
   958→                tail_call: false,
   959→            }));
   960→            let dst = ctx.alloc_temp();
   961→            stmts.push(ScgStatement::Access(AccessNode::Load {
   962→                dst: dst.clone(),
   963→                ptr: ScgExpr::Var(addr),
   964→                offset: None,
   965→            }));
   966→            ScgExpr::Var(dst)
   967→        }
   968→
   969→        // ── Range: just flatten start (range is handled by For) ──
   970→        Expr::Range { start, .. } => flatten_expr(start, stmts, ctx),
   971→
   972→        // ── Allocate: emit as a heap allocation call ──
   973→        Expr::Allocate { size, .. } => {
   974→            let size_expr = flatten_expr(size, stmts, ctx);
   975→            let dst = ctx.alloc_temp();
   976→            stmts.push(ScgStatement::Call(CallNode {
   977→                dst: Some(dst.clone()),
   978→                func: "__vuma_alloc".to_string(),
   979→                args: vec![size_expr],
   980→                is_extern: true,
   981→            }));
   982→            ScgExpr::Var(dst)
   983→        }
   984→
   985→        // ── Null → 0 ──
   986→        Expr::Null { .. } => ScgExpr::Int(0),
   987→
   988→        // ── Uninitialized → 0 ──
   989→        Expr::Uninitialized { .. } => ScgExpr::Int(0),
   990→
   991→        // ── Fallback for unsupported expression types ──
   992→        _ => ScgExpr::Int(0),
   993→    }
   994→}

   996→/// Map a VUMA AST BinOp to a codegen BinOpKind.
   997→fn map_ast_binop(op: &vuma_parser::ast::BinOp) -> BinOpKind {
   998→    use vuma_parser::ast::BinOp;
   999→    match op {
  1000→        BinOp::Add => BinOpKind::Add,
  1001→        BinOp::Sub => BinOpKind::Sub,
  1002→        BinOp::Mul => BinOpKind::Mul,
  1003→        BinOp::Div => BinOpKind::SDiv,
  1004→        BinOp::Mod => BinOpKind::SRem,
  1005→        BinOp::And => BinOpKind::And,
  1006→        BinOp::Or => BinOpKind::Or,
  1007→        BinOp::BitAnd => BinOpKind::And,
  1008→        BinOp::BitOr => BinOpKind::Or,
  1009→        BinOp::BitXor => BinOpKind::Xor,
  1010→        BinOp::Shl => BinOpKind::Shl,
  1011→        BinOp::Shr => BinOpKind::ShrL,
  1012→        BinOp::Eq => BinOpKind::Eq,
  1013→        BinOp::Ne => BinOpKind::Ne,
  1014→        BinOp::Lt => BinOpKind::SLt,
  1015→        BinOp::Le => BinOpKind::SLe,
  1016→        BinOp::Gt => BinOpKind::SGt,
  1017→        BinOp::Ge => BinOpKind::SGe,
  1018→    }
  1019→}

  1021→/// Convert a single parser statement into zero or more codegen SCG statements.
  1022→/// Uses `flatten_expr` to decompose nested expressions into three-address code.
  1023→fn bridge_stmt_to_scg(stmt: &vuma_parser::ast::Stmt, ctx: &mut BridgeCtx) -> Vec<ScgStatement> {
  1024→    use vuma_parser::ast::Stmt as PStmt;
  1025→
  1026→    match stmt {
  1027→        // ── let x [: T] = expr ──
  1028→        PStmt::Let(let_stmt) => {
  1029→            let mut stmts = Vec::new();
  1030→
  1031→            // Check if the RHS is an allocate() call → AllocationNode::Stack
  1032→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &let_stmt.value {
  1033→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
  1034→                    if name == "allocate" {
  1035→                        let size: u32 = args.first()
  1036→                            .and_then(|a| {
  1037→                                if let vuma_parser::ast::Expr::Lit { value, .. } = a {
  1038→                                    if let vuma_parser::ast::Lit::Int(n) = value {
  1039→                                        return Some(*n as u32);
  1040→                                    }
  1041→                                }
  1042→                                None
  1043→                            })
  1044→                            .unwrap_or(8);
  1045→                        return vec![ScgStatement::Allocation(AllocationNode::Stack {
  1046→                            name: let_stmt.name.clone(),
  1047→                            size,
  1048→                            ty: ScgType::Ptr,
  1049→                        })];
  1050→                    }
  1051→                    // Other function calls → CallNode (flatten args)
  1052→                    let flat_args: Vec<ScgExpr> = args.iter()
  1053→                        .map(|a| flatten_expr(a, &mut stmts, ctx))
  1054→                        .collect();
  1055→                    stmts.push(ScgStatement::Call(CallNode {
  1056→                        dst: Some(let_stmt.name.clone()),
  1057→                        func: name.clone(),
  1058→                        args: flat_args,
  1059→                        is_extern: false,
  1060→                    }));
  1061→                    return stmts;
  1062→                }
  1063→            }
  1064→
  1065→            // Check if the RHS is an Allocate expression → AllocationNode::Stack
  1066→            if let vuma_parser::ast::Expr::Allocate { size, .. } = &let_stmt.value {
  1067→                let size_val: u32 = match size.as_ref() {
  1068→                    vuma_parser::ast::Expr::Lit { value, .. } => {
  1069→                        if let vuma_parser::ast::Lit::Int(n) = value {
  1070→                            *n as u32
  1071→                        } else {
  1072→                            8
  1073→                        }
  1074→                    }
  1075→                    _ => 8,
  1076→                };
  1077→                return vec![ScgStatement::Allocation(AllocationNode::Stack {
  1078→                    name: let_stmt.name.clone(),
  1079→                    size: size_val,
  1080→                    ty: ScgType::Ptr,
  1081→                })];
  1082→            }
  1083→
  1084→            // General case: flatten the expression and assign to dst
  1085→            let result = flatten_expr(&let_stmt.value, &mut stmts, ctx);
  1086→            match &result {
  1087→                ScgExpr::Var(name) if name == &let_stmt.name => {}
  1088→                _ => {
  1089→                    stmts.push(ScgStatement::Computation(ComputationNode {
  1090→                        dst: let_stmt.name.clone(),
  1091→                        op: BinOpKind::Add,
  1092→                        lhs: result,
  1093→                        rhs: ScgExpr::Int(0),
  1094→                        tail_call: false,
  1095→                    }));
  1096→                }
  1097→            }
  1098→            stmts
  1099→        }
  1100→
  1101→        // ── target = value ──
  1102→        PStmt::Assign(assign_stmt) => {
  1103→            let mut stmts = Vec::new();
  1104→
  1105→            // Detect dereference writes: `*expr = val` → Access::Store
  1106→            if let vuma_parser::ast::AssignTarget::Deref { expr, .. } = &assign_stmt.target {
  1107→                let ptr = flatten_expr(expr, &mut stmts, ctx);
  1108→                let value = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
  1109→                stmts.push(ScgStatement::Access(AccessNode::Store {
  1110→                    ptr,
  1111→                    offset: None,
  1112→                    value,
  1113→                }));
  1114→                return stmts;
  1115→            }
  1116→
  1117→            // Handle Index target: ptr[index] = value
  1118→            if let vuma_parser::ast::AssignTarget::Index { expr, index, .. } = &assign_stmt.target {
  1119→                let base = flatten_expr(expr, &mut stmts, ctx);
  1120→                let idx = flatten_expr(index, &mut stmts, ctx);
  1121→                let addr = ctx.alloc_temp();
  1122→                stmts.push(ScgStatement::Computation(ComputationNode {
  1123→                    dst: addr.clone(),
  1124→                    op: BinOpKind::Add,
  1125→                    lhs: base,
  1126→                    rhs: idx,
  1127→                    tail_call: false,
  1128→                }));
  1129→                let value = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
  1130→                stmts.push(ScgStatement::Access(AccessNode::Store {
  1131→                    ptr: ScgExpr::Var(addr),
  1132→                    offset: None,
  1133→                    value,
  1134→                }));
  1135→                return stmts;
  1136→            }
  1137→
  1138→            let dst = match &assign_stmt.target {
  1139→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
  1140→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
  1141→                vuma_parser::ast::AssignTarget::Deref { .. } => "_deref".into(),
  1142→                vuma_parser::ast::AssignTarget::Index { .. } => "_index".into(),
  1143→            };
  1144→
  1145→            // Detect allocate() expression → AllocationNode::Stack
  1146→            if let vuma_parser::ast::Expr::Allocate { size, .. } = &assign_stmt.value {
  1147→                let size_val: u32 = match size.as_ref() {
  1148→                    vuma_parser::ast::Expr::Lit { value, .. } => {
  1149→                        if let vuma_parser::ast::Lit::Int(n) = value {
  1150→                            *n as u32
  1151→                        } else {
  1152→                            8
  1153→                        }
  1154→                    }
  1155→                    _ => 8,
  1156→                };
  1157→                return vec![ScgStatement::Allocation(AllocationNode::Stack {
  1158→                    name: dst,
  1159→                    size: size_val,
  1160→                    ty: ScgType::Ptr,
  1161→                })];
  1162→            }
  1163→
  1164→            // Detect function calls in assign: `x = foo(args)` → CallNode (flatten args)
  1165→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &assign_stmt.value {
  1166→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
  1167→                    let flat_args: Vec<ScgExpr> = args.iter()
  1168→                        .map(|a| flatten_expr(a, &mut stmts, ctx))
  1169→                        .collect();
  1170→                    stmts.push(ScgStatement::Call(CallNode {
  1171→                        dst: Some(dst),
  1172→                        func: name.clone(),
  1173→                        args: flat_args,
  1174→                        is_extern: false,
  1175→                    }));
  1176→                    return stmts;
  1177→                }
  1178→            }
  1179→
  1180→            // General case: flatten the value expression and assign to dst
  1181→            let result = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
  1182→            match &result {
  1183→                ScgExpr::Var(name) if name == &dst => {}
  1184→                _ => {
  1185→                    stmts.push(ScgStatement::Computation(ComputationNode {
  1186→                        dst,
  1187→                        op: BinOpKind::Add,
  1188→                        lhs: result,
  1189→                        rhs: ScgExpr::Int(0),
  1190→                        tail_call: false,
  1191→                    }));
  1192→                }
  1193→            }
  1194→            stmts
  1195→        }
  1196→
  1197→        // ── target op= value ──
  1198→        PStmt::CompoundAssign(ca_stmt) => {
  1199→            let mut stmts = Vec::new();
  1200→            let dst = match &ca_stmt.target {
  1201→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
  1202→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
  1203→                _ => "_".into(),
  1204→            };
  1205→            let binop = match ca_stmt.op {
  1206→                vuma_parser::ast::CompoundOp::Add => BinOpKind::Add,
  1207→                vuma_parser::ast::CompoundOp::Sub => BinOpKind::Sub,
  1208→                vuma_parser::ast::CompoundOp::Mul => BinOpKind::Mul,
  1209→                vuma_parser::ast::CompoundOp::Div => BinOpKind::SDiv,
  1210→                vuma_parser::ast::CompoundOp::Mod => BinOpKind::SRem,
  1211→                vuma_parser::ast::CompoundOp::BitAnd => BinOpKind::And,
  1212→                vuma_parser::ast::CompoundOp::BitOr => BinOpKind::Or,
  1213→                vuma_parser::ast::CompoundOp::BitXor => BinOpKind::Xor,
  1214→                vuma_parser::ast::CompoundOp::Shl => BinOpKind::Shl,
  1215→                vuma_parser::ast::CompoundOp::Shr => BinOpKind::ShrL,
  1216→            };
  1217→            let rhs = flatten_expr(&ca_stmt.value, &mut stmts, ctx);
  1218→            stmts.push(ScgStatement::Computation(ComputationNode {
  1219→                dst: dst.clone(),
  1220→                op: binop,
  1221→                lhs: ScgExpr::Var(dst),
  1222→                rhs,
  1223→                tail_call: false,
  1224→            }));
  1225→            stmts
  1226→        }
  1227→
  1228→        PStmt::Allocate(_) => vec![],
  1229→        PStmt::Free(_) => vec![],
  1230→
  1231→        PStmt::Cast(cast_stmt) => {
  1232→            let mut stmts = Vec::new();
  1233→            let _ = flatten_expr(&cast_stmt.expr, &mut stmts, ctx);
  1234→            stmts
  1235→        }
  1236→
  1237→        PStmt::Return(ret_stmt) => {
  1238→            let mut stmts = Vec::new();
  1239→            let values = match &ret_stmt.value {
  1240→                Some(expr) => vec![flatten_expr(expr, &mut stmts, ctx)],
  1241→                None => vec![],
  1242→            };
  1243→            stmts.push(ScgStatement::Return(values));
  1244→            stmts
  1245→        }
  1246→
  1247→        PStmt::Expr(expr_stmt) => {
  1248→            let mut stmts = Vec::new();
  1249→            let result = flatten_expr(&expr_stmt.expr, &mut stmts, ctx);
  1250→            ctx.last_expr_result = Some(result);
  1251→            stmts
  1252→        }
  1253→
  1254→        PStmt::If(if_stmt) => {
  1255→            let mut pre_stmts = Vec::new();
  1256→            let cond = flatten_expr(&if_stmt.condition, &mut pre_stmts, ctx);
  1257→            let then_body = bridge_block_to_scg_stmts(&if_stmt.then_block, ctx);
  1258→            let else_body = if_stmt
  1259→                .else_block
  1260→                .as_ref()
  1261→                .map(|b| bridge_block_to_scg_stmts(b, ctx));
  1262→            let mut result = pre_stmts;
  1263→            result.push(ScgStatement::Control(ControlNode::If {
  1264→                cond,
  1265→                then_body,
  1266→                else_body,
  1267→            }));
  1268→            result
  1269→        }
  1270→
  1271→        // while condition { body }
  1272→        // Lowered as: Loop { <compute cond>; if cond { body } else { break } }
  1273→        PStmt::While(while_stmt) => {
  1274→            let then_body = bridge_block_to_scg_stmts(&while_stmt.body, ctx);
  1275→            let mut loop_body = Vec::new();
  1276→            let cond = flatten_expr(&while_stmt.condition, &mut loop_body, ctx);
  1277→            loop_body.push(ScgStatement::Control(ControlNode::If {
  1278→                cond,
  1279→                then_body,
  1280→                else_body: Some(vec![ScgStatement::Control(ControlNode::Break)]),
  1281→            }));
  1282→            vec![ScgStatement::Control(ControlNode::Loop { body: loop_body })]
  1283→        }
  1284→
  1285→        // for name in start..end { body }
  1286→        // Lowered as:
  1287→        //   name = start
  1288→        //   loop { _cond = name < end; if _cond { body; name = name + 1 } else { break } }
  1289→        PStmt::For(for_stmt) => {
  1290→            let mut pre_stmts = Vec::new();
  1291→            let (start_expr, end_expr) = match &for_stmt.iter {
  1292→                vuma_parser::ast::Expr::Range { start, end, .. } => {
  1293→                    let s = flatten_expr(start, &mut pre_stmts, ctx);
  1294→                    let e = flatten_expr(end, &mut pre_stmts, ctx);
  1295→                    (s, e)
  1296→                }
  1297→                other => {
  1298→                    (ScgExpr::Int(0), ScgExpr::Int(0))
  1299→                }
  1300→            };
  1301→
  1302→            let init_stmt = ScgStatement::Computation(ComputationNode {
  1303→                dst: for_stmt.name.clone(),
  1304→                op: BinOpKind::Add,
  1305→                lhs: start_expr,
  1306→                rhs: ScgExpr::Int(0),
  1307→                tail_call: false,
  1308→            });
  1309→
  1310→            let mut loop_body = Vec::new();
  1311→            let cond_temp = format!("{}_cond", for_stmt.name);
  1312→            loop_body.push(ScgStatement::Computation(ComputationNode {
  1313→                dst: cond_temp.clone(),
  1314→                op: BinOpKind::SLt,
  1315→                lhs: ScgExpr::Var(for_stmt.name.clone()),
  1316→                rhs: end_expr.clone(),
  1317→                tail_call: false,
  1318→            }));
  1319→
  1320→            let inner_body = bridge_block_to_scg_stmts(&for_stmt.body, ctx);
  1321→            let mut full_then = inner_body;
  1322→            full_then.push(ScgStatement::Computation(ComputationNode {
  1323→                dst: for_stmt.name.clone(),
  1324→                op: BinOpKind::Add,
  1325→                lhs: ScgExpr::Var(for_stmt.name.clone()),
  1326→                rhs: ScgExpr::Int(1),
  1327→                tail_call: false,
  1328→            }));
  1329→
  1330→            loop_body.push(ScgStatement::Control(ControlNode::If {
  1331→                cond: ScgExpr::Var(cond_temp),
  1332→                then_body: full_then,
  1333→                else_body: Some(vec![ScgStatement::Control(ControlNode::Break)]),
  1334→            }));
  1335→
  1336→            let mut result = pre_stmts;
  1337→            result.push(init_stmt);
  1338→            result.push(ScgStatement::Control(ControlNode::Loop { body: loop_body }));
  1339→            result
  1340→        }
  1341→
  1342→        PStmt::Loop(loop_stmt) => {
  1343→            let body = bridge_block_to_scg_stmts(&loop_stmt.body, ctx);
  1344→            vec![ScgStatement::Control(ControlNode::Loop { body })]
  1345→        }
  1346→
  1347→        PStmt::Break(_) => vec![ScgStatement::Control(ControlNode::Break)],
  1348→        PStmt::Continue(_) => vec![ScgStatement::Control(ControlNode::Continue)],
  1349→        PStmt::BdDirective(_)
  1350→        | PStmt::Sync(_)
  1351→        | PStmt::UnsafeBlock { .. }
  1352→        | PStmt::Match(_)
  1353→        | PStmt::Access(_) => vec![],
  1354→    }
  1355→}
  1356→fn cmd_disasm(file: &PathBuf, isa: &IsaArg, base_addr_str: &str) -> Result<(), String> {
  1357→    let bytes = fs::read(file)
  1358→        .map_err(|e| format!("error: cannot read binary file '{}': {}", file.display(), e))?;
  1359→
  1360→    let base_addr = u64::from_str_radix(base_addr_str.trim_start_matches("0x"), 16)
  1361→        .map_err(|e| format!("error: invalid base address '{}': {}", base_addr_str, e))?;
  1362→

  1366→            "error: cannot create {} backend: {}",
  1367→            backend_kind.isa_name(),

  1371→
  1372→    let instructions = backend.disassemble(&bytes, base_addr);
  1373→
  1374→    println!(
  1375→        "Disassembly of {} ({} bytes, ISA: {}):",
  1376→        file.display(),
  1377→        bytes.len(),
  1378→        backend.name()
  1379→    );
  1380→    for line in &instructions {
  1381→        println!("{}", line);
  1382→    }
  1383→
  1384→    Ok(())
  1385→}

  1439→/// `vuma repl` — Interactive REPL: parse expressions and print AST.
  1440→fn cmd_repl() -> Result<(), String> {
  1441→    println!("VUMA REPL v0.1.0");
  1442→    println!("Type VUMA expressions or statements. Enter ':quit' to exit, ':help' for help.");
  1443→
  1444→    let stdin = io::stdin();
  1445→    let mut input = String::new();
  1446→
  1447→    loop {
  1448→        print!("vuma> ");
  1449→        io::stdout()
  1450→            .flush()
  1451→            .map_err(|e| format!("flush error: {}", e))?;
  1452→
  1453→        input.clear();
  1454→        match stdin.read_line(&mut input) {
  1455→            Ok(0) => break, // EOF
  1456→            Ok(_) => {}
  1457→            Err(e) => return Err(format!("error reading input: {}", e)),
  1458→        }
  1459→
  1460→        let trimmed = input.trim();
  1461→        if trimmed.is_empty() {
  1462→            continue;
  1463→        }
  1464→        if trimmed == ":quit" || trimmed == ":q" {
  1465→            break;
  1466→        }
  1467→        if trimmed == ":help" || trimmed == ":h" {
  1468→            println!("  :quit, :q   — Exit the REPL");
  1469→            println!("  :help, :h   — Show this help");
  1470→            println!("  <expr>      — Parse and display the AST");
  1471→            println!("  <stmt>      — Parse and display the AST");
  1472→            println!("  <program>   — Parse and display the full program AST");
  1473→            continue;
  1474→        }
  1475→
  1476→        // Try to parse as a full program first.
  1477→        let mut parser = vuma_parser::Parser::new(trimmed);
  1478→        let result = parser.parse_program();
  1479→        if !result.has_errors() {
  1480→            // Successfully parsed; display the AST.
  1481→            let program = result.unwrap();
  1482→            println!("{:#?}", program);
  1483→            continue;
  1484→        }
  1485→        // Program parse had errors; try to parse as an expression via
  1486→        // wrapping it in a dummy program context.
  1487→        let program_errors = result.errors.clone();
  1488→        // Since parse_expr is private, we re-parse as a minimal program.
  1489→        let wrapped = format!("fn _repl_expr() {{ {} }}", trimmed);
  1490→        let mut expr_parser = vuma_parser::Parser::new(&wrapped);
  1491→        let expr_result = expr_parser.parse_program();
  1492→        if !expr_result.has_errors() {
  1493→            // Print the function body AST.
  1494→            let program = expr_result.unwrap();
  1495→            if let Some(item) = program.items.first() {
  1496→                println!("{:#?}", item);
  1497→            } else {
  1498→                println!("{:#?}", program);
  1499→            }
  1500→        } else {
  1501→            // Show the original parse errors.
  1502→            for err in &program_errors {
  1503→                eprintln!("parse error: {}", err);
  1504→            }
  1505→        }
  1506→    }
  1507→
  1508→    Ok(())
  1509→}
