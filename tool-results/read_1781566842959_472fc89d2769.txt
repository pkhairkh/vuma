     1‚Üí//! # Foreign Function Interface (FFI) for VUMA
     2‚Üí//!
     3‚Üí//! Provides the FFI infrastructure for calling external C functions and
     4‚Üí//! Linux syscalls from VUMA programs.
     5‚Üí//!
     6‚Üí//! # Overview
     7‚Üí//!
     8‚Üí//! The FFI module handles:
     9‚Üí//!
    10‚Üí//! - **`extern "C"` block syntax** ‚Äî declares external functions in VUMA source
    11‚Üí//! - **Syscall bindings** ‚Äî Linux kernel interfaces: `write`, `read`, `exit`,
    12‚Üí//!   `mmap`, `munmap`, `brk`
    13‚Üí//! - **C library bindings** ‚Äî libc functions: `memcpy`, `memset`, `malloc`,
    14‚Üí//!   `free`
    15‚Üí//! - **Codegen support** ‚Äî extern function calls emit relocations instead of
    16‚Üí//!   local `BL` instructions
    17‚Üí//!
    18‚Üí//! # Example VUMA Source
    19‚Üí//!
    20‚Üí//! ```vuma
    21‚Üí//! extern "C" {
    22‚Üí//!     fn write(fd: i64, buf: Address, count: i64) -> i64;
    23‚Üí//!     fn read(fd: i64, buf: Address, count: i64) -> i64;
    24‚Üí//!     fn exit(code: i64);
    25‚Üí//! }
    26‚Üí//!
    27‚Üí//! fn main() {
    28‚Üí//!     write(1, 0x400000, 13);
    29‚Üí//!     exit(0);
    30‚Üí//! }
    31‚Üí//! ```

    41‚Üí/// An `extern "C" { ... }` block declaring external functions.
    42‚Üí#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
    43‚Üípub struct ExternBlock {

    50‚Üí/// A function declared inside an `extern` block.
    51‚Üí#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]

   223‚Üí// ExternRegistry ‚Äî tracks all known extern functions
   224‚Üí// ---------------------------------------------------------------------------
   225‚Üí
   226‚Üí/// Registry of all known extern functions, built from `extern` blocks
   227‚Üí/// in the source code and the built-in binding tables.
   228‚Üí#[derive(Debug, Clone, Default)]
   229‚Üípub struct ExternRegistry {
   230‚Üí    /// Map from function name to its extern declaration.
   231‚Üí    functions: HashMap<String, ExternFn>,
   232‚Üí    /// Map from function name to its calling convention.
   233‚Üí    conventions: HashMap<String, CallingConvention>,
   234‚Üí}
   235‚Üí
   236‚Üíimpl ExternRegistry {
   237‚Üí    /// Create a new empty registry.
   238‚Üí    pub fn new() -> Self {
   239‚Üí        Self::default()
   240‚Üí    }
   241‚Üí
   242‚Üí    /// Create a registry pre-populated with Linux syscall and C library bindings.
   243‚Üí    pub fn with_default_bindings() -> Self {
   244‚Üí        let mut registry = Self::new();
   245‚Üí        registry.register_block(&linux_syscall_bindings());
   246‚Üí        registry.register_block(&c_library_bindings());
   247‚Üí        registry
   248‚Üí    }
   249‚Üí
   250‚Üí    /// Register all functions from an `extern` block.
   251‚Üí    pub fn register_block(&mut self, block: &ExternBlock) {
   252‚Üí        for func in &block.functions {
   253‚Üí            self.functions.insert(func.name.clone(), func.clone());
   254‚Üí            self.conventions.insert(func.name.clone(), block.convention);
   255‚Üí        }
   256‚Üí    }
   257‚Üí
   258‚Üí    /// Look up an extern function by name.
   259‚Üí    pub fn get(&self, name: &str) -> Option<&ExternFn> {
   260‚Üí        self.functions.get(name)
   261‚Üí    }
   262‚Üí
   263‚Üí    /// Check if a function name is a known extern function.
   264‚Üí    pub fn is_extern(&self, name: &str) -> bool {
   265‚Üí        self.functions.contains_key(name)
   266‚Üí    }
   267‚Üí
   268‚Üí    /// Get the calling convention for a function.
   269‚Üí    pub fn convention(&self, name: &str) -> Option<CallingConvention> {
   270‚Üí        self.conventions.get(name).copied()
   271‚Üí    }
   272‚Üí
   273‚Üí    /// Returns true if a call to this function should be emitted as a
   274‚Üí    /// relocation (external symbol) rather than a local branch.
   275‚Üí    pub fn needs_relocation(&self, name: &str) -> bool {
   276‚Üí        self.is_extern(name)
   277‚Üí    }
   278‚Üí
   279‚Üí    /// List all registered extern function names.
   280‚Üí    pub fn function_names(&self) -> Vec<&str> {
   281‚Üí        let mut names: Vec<&str> = self.functions.keys().map(|s| s.as_str()).collect();
   282‚Üí        names.sort();
   283‚Üí        names
   284‚Üí    }
   285‚Üí
   286‚Üí    /// Returns the set of all registered extern function names.
   287‚Üí    /// Useful for passing to the codegen bridge so that calls to
   288‚Üí    /// extern functions are marked with `is_extern: true`.
   289‚Üí    pub fn function_name_set(&self) -> HashSet<String> {
   290‚Üí        self.functions.keys().cloned().collect()
   291‚Üí    }
   292‚Üí}

   316‚Üí/// A relocation entry emitted for an extern function call.
   317‚Üí///
   318‚Üí/// Instead of emitting a local `BL <symbol>` instruction, the codegen
   319‚Üí/// emits a relocation that the linker will resolve at link time.
   320‚Üí#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
   321‚Üípub struct Relocation {
   322‚Üí    /// Offset in the code section where the relocation applies.
   323‚Üí    pub offset: u64,
   324‚Üí    /// The kind of relocation.
   325‚Üí    pub kind: RelocationKind,
   326‚Üí    /// The external symbol name to resolve.
   327‚Üí    pub symbol: String,
   328‚Üí    /// Addend (typically 0 for function calls).
   329‚Üí    pub addend: i64,
   330‚Üí}
   331‚Üí
   332‚Üí/// The kind of relocation.
   333‚Üí#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq)]
   334‚Üípub enum RelocationKind {
   335‚Üí    /// AArch64: R_AARCH64_CALL26 ‚Äî 26-bit relative call.
   336‚Üí    AArch64Call26,
   337‚Üí    /// AArch64: R_AARCH64_ADR_PREL_PG_HI21 + R_AARCH64_LDST64_ABS_LO12_NC
   338‚Üí    /// for loading a 64-bit address.
   339‚Üí    AArch64AdrpPage,
   340‚Üí    /// x86_64: R_X86_64_PLT32 ‚Äî 32-bit PC-relative PLT call.
   341‚Üí    X86_64Plt32,
   342‚Üí    /// x86_64: R_X86_64_64 ‚Äî absolute 64-bit address.
   343‚Üí    X86_64Abs64,
   344‚Üí    /// RISC-V: R_RISCV_CALL ‚Äî auipc + jalr pair.
   345‚Üí    RiscvCall,
   346‚Üí    /// ARM32: R_ARM_CALL / R_ARM_JUMP24 ‚Äî 24-bit relative branch.
   347‚Üí    Arm32Call,
   348‚Üí    /// ARM32: R_ARM_V4BX ‚Äî branch exchange (for Thumb interworking).
   349‚Üí    Arm32V4bx,
   350‚Üí    /// MIPS64: R_MIPS_26 ‚Äî 26-bit jump target.
   351‚Üí    Mips26,
   352‚Üí    /// MIPS64: R_MIPS_GOT_CALL ‚Äî GOT-based call.
   353‚Üí    MipsGotCall,
   354‚Üí    /// PPC64: R_PPC64_REL24 ‚Äî 24-bit relative branch.
   355‚Üí    Ppc64Rel24,
   356‚Üí    /// PPC64: R_PPC64_REL64 ‚Äî 64-bit relative address.
   357‚Üí    Ppc64Rel64,
   358‚Üí    /// LoongArch64: R_LARCH_B26 ‚Äî 26-bit relative call.
   359‚Üí    LoongArchB26,
   360‚Üí    /// LoongArch64: R_LARCH_CALL36 ‚Äî 36-bit call sequence.
   361‚Üí    LoongArchCall36,
   362‚Üí    /// Generic: 32-bit PC-relative call (fallback).
   363‚Üí    GenericCall32,
   364‚Üí    /// Generic: 64-bit absolute address (fallback).
   365‚Üí    GenericAbs64,
   366‚Üí}

   533‚Üí/// Per-architecture syscall number table for Linux.
   534‚Üí///
   535‚Üí/// Linux assigns different syscall numbers to each architecture. This struct
   536‚Üí/// provides a unified lookup interface so that VUMA codegen can emit the
   537‚Üí/// correct immediate for `svc #0` (AArch64), `syscall` (x86_64), `ecall`
   538‚Üí/// (RISC-V), etc.
   539‚Üí#[derive(Debug, Clone)]
   540‚Üípub struct SyscallTable {
   541‚Üí    /// Architecture this table is for.
   542‚Üí    pub arch: Arch,
   543‚Üí    /// Map from syscall name to its number.
   544‚Üí    numbers: HashMap<SyscallName, u64>,
   545‚Üí}
   546‚Üí
   547‚Üíimpl SyscallTable {
   548‚Üí    /// Create the syscall table for the given architecture.
   549‚Üí    pub fn for_arch(arch: Arch) -> Self {
   550‚Üí        let numbers = match arch {
   551‚Üí            Arch::X86_64 => x86_64_syscalls(),
   552‚Üí            Arch::AArch64 => aarch64_syscalls(),
   553‚Üí            Arch::RiscV64 => riscv64_syscalls(),
   554‚Üí            Arch::Arm32 => arm32_syscalls(),
   555‚Üí            Arch::Mips64 => mips64_syscalls(),
   556‚Üí            Arch::PPC64 => ppc64_syscalls(),
   557‚Üí            Arch::LoongArch64 => loongarch64_syscalls(),
   558‚Üí            Arch::Wasm32 => HashMap::new(), // no native syscalls
   559‚Üí        };
   560‚Üí        Self { arch, numbers }
   561‚Üí    }
   562‚Üí
   563‚Üí    /// Look up the syscall number for a named syscall.
   564‚Üí    pub fn get(&self, name: SyscallName) -> Option<u64> {
   565‚Üí        self.numbers.get(&name).copied()
   566‚Üí    }
   567‚Üí
   568‚Üí    /// Returns the number of syscalls in this table.
   569‚Üí    pub fn len(&self) -> usize {
   570‚Üí        self.numbers.len()
   571‚Üí    }
   572‚Üí
   573‚Üí    /// Returns true if the table is empty.
   574‚Üí    pub fn is_empty(&self) -> bool {
   575‚Üí        self.numbers.is_empty()
   576‚Üí    }
   577‚Üí
   578‚Üí    /// Iterate over all (name, number) pairs.
   579‚Üí    pub fn iter(&self) -> impl Iterator<Item = (SyscallName, u64)> + '_ {
   580‚Üí        self.numbers.iter().map(|(&k, &v)| (k, v))
   581‚Üí    }
   582‚Üí}
