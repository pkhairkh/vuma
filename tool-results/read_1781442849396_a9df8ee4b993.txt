     1→//! # WebAssembly 32-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the Wasm32 target.  WebAssembly is
     4→//! fundamentally different from register-based ISAs: it is a stack machine
     5→//! with structured control flow and a binary format.  This module provides:
     6→//!
     7→//! - `WasmType` — value types (i32, i64, f32, f64)
     8→//! - `WasmInstr` — exhaustive instruction enum covering all ops needed for VUMA IR
     9→//! - LEB128 encoding/decoding (unsigned and signed)
    10→//! - Wasm binary-format encoder that emits complete `.wasm` modules
    11→//! - `Wasm32Backend` — `Backend` implementation that lowers IR to Wasm bytecode
    12→

    14→    AllocatedBlock, AllocatedFunction, AllocatedInstruction, AllocatedProgram, Backend,
    15→    BackendError, Wasm32TargetInfo,
    16→};
    17→use crate::ir::{
    18→    BinOpKind, CastKind, IRFunction, IRInstr, IRTerminator, IRType, IRValue, UnaryOpKind,
    19→};
    20→use std::collections::HashMap;

    22→// ===========================================================================
    23→// Wasm value types
    24→// ===========================================================================
    25→
    26→/// WebAssembly value types.
    27→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
    28→pub enum WasmType {
    29→    I32,
    30→    I64,
    31→    F32,
    32→    F64,
    33→}
    34→
    35→impl WasmType {
    36→    /// Binary encoding of this type in the Wasm format.
    37→    pub fn to_byte(&self) -> u8 {
    38→        match self {
    39→            WasmType::I32 => 0x7F,
    40→            WasmType::I64 => 0x7E,
    41→            WasmType::F32 => 0x7D,
    42→            WasmType::F64 => 0x7C,
    43→        }
    44→    }
    45→
    46→    /// Size in bytes of this Wasm type on the stack.
    47→    pub fn byte_size(&self) -> usize {
    48→        match self {
    49→            WasmType::I32 | WasmType::F32 => 4,
    50→            WasmType::I64 | WasmType::F64 => 8,
    51→        }
    52→    }
    53→
    54→    /// Whether this is an integer type.
    55→    pub fn is_integer(&self) -> bool {
    56→        matches!(self, WasmType::I32 | WasmType::I64)
    57→    }
    58→
    59→    /// Whether this is a floating-point type.
    60→    pub fn is_float(&self) -> bool {
    61→        matches!(self, WasmType::F32 | WasmType::F64)
    62→    }
    63→
    64→    /// Map an IRType to the corresponding WasmType.
    65→    pub fn from_ir_type(ty: &IRType) -> Option<WasmType> {
    66→        match ty {
    67→            IRType::I8
    68→            | IRType::I16
    69→            | IRType::I32
    70→            | IRType::U8
    71→            | IRType::U16
    72→            | IRType::U32
    73→            | IRType::Ptr
    74→            | IRType::Func => Some(WasmType::I32),
    75→            IRType::I64 | IRType::U64 => Some(WasmType::I64),
    76→            IRType::F32 => Some(WasmType::F32),
    77→            IRType::F64 => Some(WasmType::F64),
    78→            IRType::Void | IRType::Struct { .. } | IRType::Array { .. } => None,
    79→        }
    80→    }
    81→}
    82→
    83→impl std::fmt::Display for WasmType {
    84→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    85→        match self {
    86→            WasmType::I32 => write!(f, "i32"),
    87→            WasmType::I64 => write!(f, "i64"),
    88→            WasmType::F32 => write!(f, "f32"),
    89→            WasmType::F64 => write!(f, "f64"),
    90→        }
    91→    }
    92→}

   379→impl WasmInstr {
   380→    /// Encode this instruction into Wasm bytecode, appending bytes to `out`.
   381→    pub fn encode(&self, out: &mut Vec<u8>) {
   382→        match self {
   383→            // ── Control ──────────────────────────────────────────────
   384→            WasmInstr::Block(ty) => {
   385→                out.push(0x02);
   386→                out.push(ty.to_byte());
   387→            }
   388→            WasmInstr::Loop(ty) => {
   389→                out.push(0x03);
   390→                out.push(ty.to_byte());
   391→            }
   392→            WasmInstr::If(ty) => {
   393→                out.push(0x04);
   394→                out.push(ty.to_byte());
   395→            }
   396→            WasmInstr::Else => out.push(0x05),
   397→            WasmInstr::End => out.push(0x0B),
   398→            WasmInstr::Br(idx) => {
   399→                out.push(0x0C);
   400→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   401→            }
   402→            WasmInstr::BrIf(idx) => {
   403→                out.push(0x0D);
   404→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   405→            }
   406→            WasmInstr::BrTable { labels, default } => {
   407→                out.push(0x0E);
   408→                out.extend_from_slice(&encode_unsigned_leb128(labels.len() as u64));
   409→                for &lbl in labels {
   410→                    out.extend_from_slice(&encode_unsigned_leb128(lbl as u64));
   411→                }
   412→                out.extend_from_slice(&encode_unsigned_leb128(*default as u64));
   413→            }
   414→            WasmInstr::Return => out.push(0x0F),
   415→            WasmInstr::Call(idx) => {
   416→                out.push(0x10);
   417→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   418→            }
   419→            WasmInstr::CallIndirect {
   420→                type_idx,
   421→                table_idx,
   422→            } => {
   423→                out.push(0x11);
   424→                out.extend_from_slice(&encode_unsigned_leb128(*type_idx as u64));
   425→                out.extend_from_slice(&encode_unsigned_leb128(*table_idx as u64));
   426→            }
   427→
   428→            // ── Parametric ──────────────────────────────────────────
   429→            WasmInstr::Select => out.push(0x1B),
   430→            WasmInstr::Drop => out.push(0x1A),
   431→
   432→            // ── Variable ────────────────────────────────────────────
   433→            WasmInstr::LocalGet(idx) => {
   434→                out.push(0x20);
   435→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   436→            }
   437→            WasmInstr::LocalSet(idx) => {
   438→                out.push(0x21);
   439→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   440→            }
   441→            WasmInstr::LocalTee(idx) => {
   442→                out.push(0x22);
   443→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   444→            }
   445→            WasmInstr::GlobalGet(idx) => {
   446→                out.push(0x23);
   447→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   448→            }
   449→            WasmInstr::GlobalSet(idx) => {
   450→                out.push(0x24);
   451→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   452→            }
   453→
   454→            // ── Memory ──────────────────────────────────────────────
   455→            WasmInstr::I32Load { align, offset } => {
   456→                out.push(0x28);
   457→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   458→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   459→            }
   460→            WasmInstr::I64Load { align, offset } => {
   461→                out.push(0x29);
   462→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   463→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   464→            }
   465→            WasmInstr::F32Load { align, offset } => {
   466→                out.push(0x2A);
   467→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   468→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   469→            }
   470→            WasmInstr::F64Load { align, offset } => {
   471→                out.push(0x2B);
   472→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   473→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   474→            }
   475→            WasmInstr::I32Store { align, offset } => {
   476→                out.push(0x36);
   477→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   478→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   479→            }
   480→            WasmInstr::I64Store { align, offset } => {
   481→                out.push(0x37);

   534→            }

   766→/// Encode an unsigned 64-bit value as unsigned LEB128.
   767→pub fn encode_unsigned_leb128(value: u64) -> Vec<u8> {
   768→    let mut result = Vec::new();
   769→    let mut val = value;
   770→    loop {
   771→        let mut byte = (val & 0x7F) as u8;
   772→        val >>= 7;
   773→        if val != 0 {
   774→            byte |= 0x80;
   775→        }
   776→        result.push(byte);
   777→        if val == 0 {
   778→            break;
   779→        }
   780→    }
   781→    result
   782→}
   783→
   784→/// Encode a signed 64-bit value as signed LEB128.
   785→pub fn encode_signed_leb128(value: i64) -> Vec<u8> {
   786→    let mut result = Vec::new();
   787→    let mut val = value;
   788→    loop {
   789→        let mut byte = (val & 0x7F) as u8;
   790→        val >>= 7;
   791→        // Sign bit of byte is bit 6
   792→        let sign_bit = (byte & 0x40) != 0;
   793→        // If val is 0 and sign bit is clear, or val is -1 and sign bit is set, we're done
   794→        if (val == 0 && !sign_bit) || (val == -1 && sign_bit) {
   795→            result.push(byte);
   796→            break;
   797→        }
   798→        byte |= 0x80;
   799→        result.push(byte);
   800→    }
   801→    result
   802→}
   803→
   804→/// Decode an unsigned LEB128 value from a byte slice.
   805→/// Returns (decoded_value, number_of_bytes_consumed).
   806→pub fn decode_unsigned_leb128(bytes: &[u8]) -> (u64, usize) {
   807→    let mut result: u64 = 0;
   808→    let mut shift: u32 = 0;
   809→    let mut count = 0;
   810→    for &byte in bytes {
   811→        count += 1;
   812→        result |= ((byte & 0x7F) as u64) << shift;
   813→        if (byte & 0x80) == 0 {
   814→            break;
   815→        }
   816→        shift += 7;
   817→    }
   818→    (result, count)
   819→}
   820→
   821→/// Decode a signed LEB128 value from a byte slice.
   822→/// Returns (decoded_value, number_of_bytes_consumed).
   823→pub fn decode_signed_leb128(bytes: &[u8]) -> (i64, usize) {
   824→    let mut result: i64 = 0;
   825→    let mut shift: u32 = 0;
   826→    let mut count = 0;
   827→    let mut byte: u8;
   828→    loop {
   829→        byte = bytes[count];
   830→        count += 1;
   831→        result |= ((byte & 0x7F) as i64) << shift;
   832→        shift += 7;
   833→        if (byte & 0x80) == 0 {
   834→            break;
   835→        }
   836→    }
   837→    // Sign-extend if the sign bit is set
   838→    if shift < 64 && (byte & 0x40) != 0 {
   839→        result |= !0i64 << shift;
   840→    }
   841→    (result, count)
   842→}

   865→/// A Wasm function type signature: (params) -> (results).
   866→///
   867→/// Supports multi-value returns (Wasm 2.0 feature): the `results` vector
   868→/// may contain more than one type.
   869→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   870→pub struct WasmFuncType {
   871→    pub params: Vec<WasmType>,
   872→    pub results: Vec<WasmType>,
   873→}
   874→
   875→impl WasmFuncType {
   876→    /// Encode this function type in Wasm binary format.
   877→    pub fn encode(&self) -> Vec<u8> {
   878→        let mut out = Vec::new();
   879→        out.push(0x60); // func type tag
   880→        out.extend_from_slice(&encode_unsigned_leb128(self.params.len() as u64));
   881→        for p in &self.params {
   882→            out.push(p.to_byte());
   883→        }
   884→        out.extend_from_slice(&encode_unsigned_leb128(self.results.len() as u64));
   885→        for r in &self.results {
   886→            out.push(r.to_byte());
   887→        }
   888→        out
   889→    }
   890→}

  1095→    /// Encode the complete module into bytes.
  1096→    pub fn encode(self) -> Vec<u8> {
  1097→        let mut module = Vec::new();
  1098→
  1099→        // Magic + version
  1100→        module.extend_from_slice(&WASM_MAGIC);
  1101→        module.extend_from_slice(&WASM_VERSION);
  1102→
  1103→        // Type section
  1104→        if !self.types.is_empty() {
  1105→            let mut section = Vec::new();
  1106→            section.extend_from_slice(&encode_unsigned_leb128(self.types.len() as u64));
  1107→            for ft in &self.types {
  1108→                section.extend_from_slice(&ft.encode());
  1109→            }
  1110→            emit_section(&mut module, SECTION_TYPE, &section);
  1111→        }
  1112→
  1113→        // Import section
  1114→        if !self.imports.is_empty() {
  1115→            let mut section = Vec::new();
  1116→            section.extend_from_slice(&encode_unsigned_leb128(self.imports.len() as u64));
  1117→            for imp in &self.imports {
  1118→                section.extend_from_slice(&encode_unsigned_leb128(imp.module.len() as u64));
  1119→                section.extend_from_slice(imp.module.as_bytes());
  1120→                section.extend_from_slice(&encode_unsigned_leb128(imp.name.len() as u64));
  1121→                section.extend_from_slice(imp.name.as_bytes());
  1122→                match &imp.kind {
  1123→                    WasmImportKind::Function { type_idx } => {
  1124→                        section.push(0x00);
  1125→                        section.extend_from_slice(&encode_unsigned_leb128(*type_idx as u64));
  1126→                    }
  1127→                    WasmImportKind::Table { elem_type, limits } => {
  1128→                        section.push(0x01);
  1129→                        section.push(*elem_type);
  1130→                        section.extend_from_slice(&limits.encode());
  1131→                    }
  1132→                    WasmImportKind::Memory { limits } => {
  1133→                        section.push(0x02);
  1134→                        section.extend_from_slice(&limits.encode());
  1135→                    }
  1136→                    WasmImportKind::Global { val_type, mutable } => {
  1137→                        section.push(0x03);
  1138→                        section.push(val_type.to_byte());
  1139→                        section.push(if *mutable { 0x01 } else { 0x00 });
  1140→                    }
  1141→                }
  1142→            }
  1143→            emit_section(&mut module, SECTION_IMPORT, &section);
  1144→        }
  1145→
  1146→        // Function section
  1147→        if !self.functions.is_empty() {
  1148→            let mut section = Vec::new();
  1149→            section.extend_from_slice(&encode_unsigned_leb128(self.functions.len() as u64));
  1150→            for &type_idx in &self.functions {
  1151→                section.extend_from_slice(&encode_unsigned_leb128(type_idx as u64));
  1152→            }
  1153→            emit_section(&mut module, SECTION_FUNCTION, &section);
  1154→        }
  1155→
  1156→        // Table section
  1157→        if !self.tables.is_empty() {
  1158→            let mut section = Vec::new();
  1159→            section.extend_from_slice(&encode_unsigned_leb128(self.tables.len() as u64));
  1160→            for (elem_type, limits) in &self.tables {
  1161→                section.push(*elem_type);
  1162→                section.extend_from_slice(&limits.encode());
  1163→            }
  1164→            emit_section(&mut module, SECTION_TABLE, &section);
  1165→        }
  1166→
  1167→        // Memory section
  1168→        if !self.memories.is_empty() {
  1169→            let mut section = Vec::new();
  1170→            section.extend_from_slice(&encode_unsigned_leb128(self.memories.len() as u64));
  1171→            for limits in &self.memories {
  1172→                section.extend_from_slice(&limits.encode());
  1173→            }
  1174→            emit_section(&mut module, SECTION_MEMORY, &section);
  1175→        }
  1176→
  1177→        // Global section
  1178→        if !self.globals.is_empty() {
  1179→            let mut section = Vec::new();
  1180→            section.extend_from_slice(&encode_unsigned_leb128(self.globals.len() as u64));
  1181→            for g in &self.globals {
  1182→                section.push(g.val_type.to_byte());
  1183→                section.push(if g.mutable { 0x01 } else { 0x00 });
  1184→                section.extend_from_slice(&g.init_expr);
  1185→            }
  1186→            emit_section(&mut module, SECTION_GLOBAL, &section);
  1187→        }
  1188→
  1189→        // Export section
  1190→        if !self.exports.is_empty() {
  1191→            let mut section = Vec::new();
  1192→            section.extend_from_slice(&encode_unsigned_leb128(self.exports.len() as u64));
  1193→            for exp in &self.exports {
  1194→                section.extend_from_slice(&encode_unsigned_leb128(exp.name.len() as u64));
  1195→                section.extend_from_slice(exp.name.as_bytes());
  1196→                section.push(exp.kind as u8);
  1197→                section.extend_from_slice(&encode_unsigned_leb128(exp.index as u64));
  1198→            }
  1199→            emit_section(&mut module, SECTION_EXPORT, &section);
  1200→        }
  1201→
  1202→        // Start section
  1203→        if let Some(func_idx) = self.start_func {
  1204→            let section = encode_unsigned_leb128(func_idx as u64);
  1205→            emit_section(&mut module, SECTION_START, &section);
  1206→        }
  1207→
  1208→        // Element section
  1209→        if !self.elements.is_empty() {
  1210→            let mut section = Vec::new();
  1211→            section.extend_from_slice(&encode_unsigned_leb128(self.elements.len() as u64));
  1212→            for elem in &self.elements {
  1213→                section.extend_from_slice(&encode_unsigned_leb128(elem.table_index as u64));
  1214→                section.extend_from_slice(&elem.offset_expr);
  1215→                section.extend_from_slice(&encode_unsigned_leb128(elem.func_indices.len() as u64));
  1216→                for &idx in &elem.func_indices {
  1217→                    section.extend_from_slice(&encode_unsigned_leb128(idx as u64));
  1218→                }
  1219→            }
  1220→            emit_section(&mut module, SECTION_ELEMENT, &section);
  1221→        }
  1222→
  1223→        // Code section
  1224→        if !self.code.is_empty() {
  1225→            let mut section = Vec::new();
  1226→            section.extend_from_slice(&encode_unsigned_leb128(self.code.len() as u64));
  1227→            for func_body in &self.code {
  1228→                // Encode the function body: locals + code + end byte
  1229→                let mut body_bytes = Vec::new();
  1230→                body_bytes
  1231→                    .extend_from_slice(&encode_unsigned_leb128(func_body.locals.len() as u64));
  1232→                for (count, ty) in &func_body.locals {
  1233→                    body_bytes.extend_from_slice(&encode_unsigned_leb128(*count as u64));
  1234→                    body_bytes.push(ty.to_byte());
  1235→                }
  1236→                body_bytes.extend_from_slice(&func_body.body);
  1237→                // Function body is size-prefixed
  1238→                let body_size = body_bytes.len() as u64;
  1239→                section.extend_from_slice(&encode_unsigned_leb128(body_size));
  1240→                section.extend_from_slice(&body_bytes);
  1241→            }
  1242→            emit_section(&mut module, SECTION_CODE, &section);
  1243→        }
  1244→
  1245→        // Data section
  1246→        if !self.data.is_empty() {
  1247→            let mut section = Vec::new();
  1248→            section.extend_from_slice(&encode_unsigned_leb128(self.data.len() as u64));
  1249→            for seg in &self.data {
  1250→                section.extend_from_slice(&encode_unsigned_leb128(seg.memory_index as u64));
  1251→                section.extend_from_slice(&seg.offset_expr);
  1252→                section.extend_from_slice(&encode_unsigned_leb128(seg.data.len() as u64));
  1253→                section.extend_from_slice(&seg.data);
  1254→            }
  1255→            emit_section(&mut module, SECTION_DATA, &section);
  1256→        }
  1257→
  1258→        module
  1259→    }
  1260→}
  1261→
  1262→/// Emit a Wasm section: section ID + LEB128 size + content.
  1263→fn emit_section(module: &mut Vec<u8>, section_id: u8, content: &[u8]) {
  1264→    module.push(section_id);
  1265→    module.extend_from_slice(&encode_unsigned_leb128(content.len() as u64));
  1266→    module.extend_from_slice(content);
  1267→}
  1268→
  1269→/// Skip `count` unsigned LEB128 values in the byte slice, advancing `offset`.
  1270→fn skip_leb128(bytes: &[u8], offset: &mut usize, count: usize) {
  1271→    for _ in 0..count {
  1272→        while *offset < bytes.len() && (bytes[*offset] & 0x80) != 0 {
  1273→            *offset += 1;
  1274→        }
  1275→        if *offset < bytes.len() {
  1276→            *offset += 1;
  1277→        }
  1278→    }
  1279→}
  1280→
  1281→/// Skip `count` signed LEB128 values in the byte slice, advancing `offset`.
  1282→fn skip_signed_leb128(bytes: &[u8], offset: &mut usize, count: usize) {
  1283→    // Signed and unsigned LEB128 have the same encoding structure for skipping
  1284→    skip_leb128(bytes, offset, count);
  1285→}
  1286→
  1287→/// Skip a single Wasm instruction starting at `offset`, advancing past it.
  1288→///
  1289→/// This is used by `allocate_registers` to split the body bytecode into
  1290→/// per-instruction `AllocatedInstruction` entries with real opcode names.
  1291→fn skip_one_instruction(bytes: &[u8], offset: &mut usize) {
  1292→    if *offset >= bytes.len() {
  1293→        return;
  1294→    }
  1295→    let byte = bytes[*offset];
  1296→    *offset += 1;
  1297→
  1298→    match byte {
  1299→        // ── Multi-byte opcodes ────────────────────────────────────
  1300→        0xFC => {
  1301→            // Bulk memory / saturated arithmetic prefix
  1302→            skip_leb128(bytes, offset, 1); // sub-opcode
  1303→            let (subop, _) = decode_unsigned_leb128(&bytes[*offset - 1..]);
  1304→            match subop {
  1305→                0x08 => skip_leb128(bytes, offset, 2), // memory.init
  1306→                0x0A => skip_leb128(bytes, offset, 2), // memory.copy
  1307→                0x0B => skip_leb128(bytes, offset, 1), // memory.fill
  1308→                _ => {}                                // other sub-ops may have 0 or more operands
  1309→            }
  1310→        }
  1311→        0xFD => {
  1312→            // SIMD prefix
  1313→            skip_leb128(bytes, offset, 1); // sub-opcode
  1314→            let (subop, _) = decode_unsigned_leb128(&bytes[*offset - 1..]);
  1315→            if subop == 0x0C {
  1316→                // v128.const: skip 16 bytes of payload
  1317→                *offset += 16;
  1318→            }
  1319→        }
  1320→
  1321→        // ── Control flow ─────────────────────────────────────────
  1322→        0x00 => {} // unreachable
  1323→        0x01 => {} // nop
  1324→        0x02..=0x04 => {
  1325→            *offset += 1;
  1326→        } // block/loop/if + blocktype
  1327→        0x05 | 0x0B => {} // else, end
  1328→        0x0C | 0x0D => {
  1329→            skip_leb128(bytes, offset, 1);
  1330→        } // br, br_if
  1331→        0x0E => {
  1332→            // br_table: count + labels + default
  1333→            let (count, size) = decode_unsigned_leb128(&bytes[*offset..]);
  1334→            *offset += size;
  1335→            skip_leb128(bytes, offset, count as usize + 1);
  1336→        }
  1337→        0x0F => {} // return
  1338→        0x10 => {
  1339→            skip_leb128(bytes, offset, 1);
  1340→        } // call
  1341→        0x11 => {
  1342→            skip_leb128(bytes, offset, 2);
  1343→        } // call_indirect
  1344→
  1345→        // ── Parametric ───────────────────────────────────────────
  1346→        0x1A | 0x1B => {} // drop, select
  1347→
  1348→        // ── Variable ─────────────────────────────────────────────
  1349→        0x20..=0x24 => {
  1350→            skip_leb128(bytes, offset, 1);
  1351→        } // local.get/set/tee, global.get/set
  1352→
  1353→        // ── Memory ───────────────────────────────────────────────
  1354→        0x28..=0x3E => {
  1355→            skip_leb128(bytes, offset, 2);
  1356→        } // loads/stores: align + offset
  1357→        0x3F | 0x40 => {
  1358→            skip_leb128(bytes, offset, 1);
  1359→        } // memory.size, memory.grow
  1360→
  1361→        // ── Numeric ──────────────────────────────────────────────
  1362→        0x41 => {
  1363→            skip_signed_leb128(bytes, offset, 1);
  1364→        } // i32.const
  1365→        0x42 => {
  1366→            skip_signed_leb128(bytes, offset, 1);
  1367→        } // i64.const
  1368→        0x43 => {
  1369→            *offset += 4;
  1370→        } // f32.const
  1371→        0x44 => {
  1372→            *offset += 8;
  1373→        } // f64.const
  1374→
  1375→        // All other single-byte opcodes (comparisons, arithmetic, conversions)
  1376→        0x45..=0x4F
  1377→        | 0x50..=0x5A
  1378→        | 0x5B..=0x66
  1379→        | 0x67..=0x78
  1380→        | 0x79..=0x8A
  1381→        | 0x8C
  1382→        | 0x91..=0x95
  1383→        | 0x9A
  1384→        | 0x9F..=0xA3
  1385→        | 0xA7..=0xBF => {}
  1386→
  1387→        _ => {} // Unknown opcode; nothing more to skip
  1388→    }
  1389→}

  1395→/// Context for lowering an IR function to Wasm bytecode.
  1396→struct LoweringContext {
  1397→    /// Map from virtual register ID to local index.
  1398→    vreg_to_local: HashMap<u32, u32>,
  1399→    /// Number of locals (including parameters).
  1400→    num_locals: u32,
  1401→    /// Local declarations (for the function body).
  1402→    locals: Vec<(u32, WasmType)>,
  1403→    /// Map from block label to its label depth index for branch targets.
  1404→    block_labels: HashMap<String, u32>,
  1405→    /// Accumulated Wasm instructions.
  1406→    instrs: Vec<WasmInstr>,
  1407→}
  1408→
  1409→impl LoweringContext {
  1410→    fn new() -> Self {
  1411→        Self {
  1412→            vreg_to_local: HashMap::new(),
  1413→            num_locals: 0,
  1414→            locals: Vec::new(),
  1415→            block_labels: HashMap::new(),
  1416→            instrs: Vec::new(),
  1417→        }
  1418→    }
  1419→
  1420→    /// Allocate a local for a virtual register, returning the local index.
  1421→    fn alloc_local(&mut self, vreg_id: u32, ty: WasmType) -> u32 {
  1422→        let idx = self.num_locals;
  1423→        self.vreg_to_local.insert(vreg_id, idx);
  1424→        self.num_locals += 1;
  1425→        // Try to merge with an existing (count, type) entry
  1426→        if let Some(last) = self.locals.last_mut() {
  1427→            if last.1 == ty {
  1428→                last.0 += 1;
  1429→                return idx;
  1430→            }
  1431→        }
  1432→        self.locals.push((1, ty));
  1433→        idx
  1434→    }
  1435→
  1436→    /// Get the local index for a virtual register.
  1437→    fn get_local(&self, vreg_id: u32) -> Option<u32> {
  1438→        self.vreg_to_local.get(&vreg_id).copied()
  1439→    }
  1440→
  1441→    /// Emit an instruction.
  1442→    fn emit(&mut self, instr: WasmInstr) {
  1443→        self.instrs.push(instr);
  1444→    }
  1445→
  1446→    /// Push a value onto the Wasm stack from an IRValue.
  1447→    fn push_value(&mut self, val: &IRValue, type_hint: Option<&IRType>) {
  1448→        match val {
  1449→            IRValue::Immediate(v) => {
  1450→                let ty = type_hint
  1451→                    .and_then(WasmType::from_ir_type)
  1452→                    .unwrap_or(WasmType::I32);
  1453→                match ty {
  1454→                    WasmType::I32 => self.emit(WasmInstr::I32Const(*v as i32)),
  1455→                    WasmType::I64 => self.emit(WasmInstr::I64Const(*v)),
  1456→                    WasmType::F32 => self.emit(WasmInstr::F32Const(*v as f32)),
  1457→                    WasmType::F64 => self.emit(WasmInstr::F64Const(*v as f64)),
  1458→                }
  1459→            }
  1460→            IRValue::Register(id) => {
  1461→                if let Some(local_idx) = self.get_local(*id) {
  1462→                    self.emit(WasmInstr::LocalGet(local_idx));
  1463→                }
  1464→            }
  1465→            IRValue::Address(addr) => {
  1466→                self.emit(WasmInstr::I32Const(*addr as i32)); // wasm32: pointers are i32
  1467→            }
  1468→            IRValue::Label(_) => {
  1469→                // Labels are handled via block structure; not pushed as values
  1470→            }
  1471→        }
  1472→    }
  1473→
  1474→    /// Store the top of the Wasm stack into a virtual register's local.
  1475→    fn pop_to_vreg(&mut self, vreg_id: u32, ty: WasmType) {
  1476→        if !self.vreg_to_local.contains_key(&vreg_id) {
  1477→            self.alloc_local(vreg_id, ty);
  1478→        }
  1479→        if let Some(local_idx) = self.get_local(vreg_id) {
  1480→            self.emit(WasmInstr::LocalSet(local_idx));
  1481→        }
  1482→    }
  1483→}
  1484→
  1485→/// Determine the Wasm type for an IR BinOp based on the operand types.
  1486→/// Since IR doesn't carry per-instruction types, we infer from context:
  1487→/// 64-bit integer ops use i64, 32-bit use i32, float ops use f32/f64.
  1488→fn wasm_type_for_binop(_op: &BinOpKind, lhs: &IRValue, rhs: &IRValue) -> WasmType {
  1489→    // If either operand is an i64 immediate, use i64
  1490→    if let IRValue::Immediate(v) = lhs {
  1491→        if *v != (*v as i32 as i64) {
  1492→            return WasmType::I64;
  1493→        }
  1494→    }
  1495→    if let IRValue::Immediate(v) = rhs {
  1496→        if *v != (*v as i32 as i64) {
  1497→            return WasmType::I64;
  1498→        }
  1499→    }
  1500→    // Default to i32 for integer ops
  1501→    WasmType::I32
  1502→}
  1503→
  1504→/// Infer the Wasm type of an IR value based on its representation.
  1505→///
  1506→/// For immediates, we check if the value fits in i32; for registers and
  1507→/// addresses, we default to i32 (the Wasm32 pointer type).  This is a
  1508→/// heuristic — the IR does not carry per-value type information — but it
  1509→/// works for the common cases needed during instruction selection.
  1510→fn infer_wasm_type(val: &IRValue) -> WasmType {
  1511→    match val {
  1512→        IRValue::Immediate(v) => {
  1513→            if *v != (*v as i32 as i64) {
  1514→                WasmType::I64
  1515→            } else {
  1516→                WasmType::I32
  1517→            }
  1518→        }
  1519→        IRValue::Register(_) | IRValue::Address(_) | IRValue::Label(_) => WasmType::I32,
  1520→    }
  1521→}
  1522→
  1523→/// Lower an IR function to Wasm bytecode, returning the function body bytes
  1524→/// and local declarations.
  1525→fn lower_function(func: &IRFunction) -> Result<(WasmFuncBody, WasmFuncType), BackendError> {
  1526→    let mut ctx = LoweringContext::new();
  1527→
  1528→    // Assign locals for parameters
  1529→    for (i, param) in func.params.iter().enumerate() {
  1530→        let ty = func
  1531→            .param_types
  1532→            .get(i)
  1533→            .and_then(WasmType::from_ir_type)
  1534→            .unwrap_or(WasmType::I32);
  1535→        if let IRValue::Register(id) = param {
  1536→            let idx = ctx.num_locals;
  1537→            ctx.vreg_to_local.insert(*id, idx);
  1538→            ctx.num_locals += 1;
  1539→            if let Some(last) = ctx.locals.last_mut() {
  1540→                if last.1 == ty {
  1541→                    last.0 += 1;
  1542→                } else {
  1543→                    ctx.locals.push((1, ty));
  1544→                }
  1545→            } else {
  1546→                ctx.locals.push((1, ty));
  1547→            }
  1548→        }
  1549→    }
  1550→
  1551→    // Lower each block
  1552→    for (block_idx, block) in func.blocks.iter().enumerate() {
  1553→        // Emit a block for structured control flow (except the entry block)
  1554→        if block_idx > 0 {
  1555→            ctx.emit(WasmInstr::Block(WasmType::I32)); // placeholder block type
  1556→            ctx.block_labels
  1557→                .insert(block.label.clone(), block_idx as u32);
  1558→        } else {
  1559→            ctx.block_labels.insert(block.label.clone(), 0);
  1560→        }
  1561→
  1562→        // Lower instructions
  1563→        for instr in &block.instructions {
  1564→            lower_instruction(instr, &mut ctx)?;
  1565→        }
  1566→
  1567→        // Lower terminator
  1568→        lower_terminator(&block.terminator, &mut ctx, block_idx)?;
  1569→
  1570→        // End the block
  1571→        if block_idx > 0 {
  1572→            ctx.emit(WasmInstr::End);
  1573→        }
  1574→    }
  1575→
  1576→    // Build the function type
  1577→    let param_types: Vec<WasmType> = func
  1578→        .param_types
  1579→        .iter()
  1580→        .filter_map(WasmType::from_ir_type)
  1581→        .collect();
  1582→    let result_types: Vec<WasmType> = func
  1583→        .result_types
  1584→        .iter()
