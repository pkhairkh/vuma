     1→//! # PowerPC64 Backend
     2→//!
     3→//! Implements the `Backend` trait for the PowerPC 64-bit target (ELFv2 ABI,
     4→//! little-endian default). This module provides:
     5→//!
     6→//! - `Gpr` — General-purpose register enum (R0–R31)
     7→//! - `Fpr` — Floating-point register enum (F0–F31)
     8→//! - `CrField` — Condition register field enum (CR0–CR7)
     9→//! - `Instruction` — PPC64 instruction enum with correct 32-bit encoding
    10→//! - `PPC64Backend` — `Backend` implementation that lowers IR to PPC64
    11→//!   machine code and emits ELF64 binaries
    12→//!
    13→//! ## PowerPC64 Register Convention (ELFv2 ABI)
    14→//!
    15→//! | Register(s) | Role                                  |
    16→//! |-------------|---------------------------------------|
    17→//! | R0          | Volatile (not hardwired zero!)        |
    18→//! | R1          | Stack pointer (SP)                    |
    19→//! | R2          | TOC pointer                           |
    20→//! | R3–R10      | Argument / return registers           |
    21→//! | R11–R12     | Volatile scratch                      |
    22→//! | R13         | Thread pointer                        |
    23→//! | R14–R31     | Callee-saved                          |
    24→//! | F0–F13      | FP argument / return, volatile        |
    25→//! | F14–F31     | FP callee-saved                       |
    26→//! | CR0–CR7     | Condition register fields (4 bits ea) |
    27→//! | LR          | Link register (SPR)                   |
    28→//! | CTR         | Count register (SPR)                  |
    29→//!
    30→//! ## Instruction Encoding
    31→//!
    32→//! All instructions are 32 bits, fixed-width. PPC64 is bi-endian; the default
    33→//! for ppc64le is little-endian byte order for both data and instructions.
    34→//! The primary opcode occupies bits \[0:5\] (MSB-first bit numbering).
    35→//!
    36→//! ## References
    37→//!
    38→//! - Power ISA Version 3.1
    39→//! - <https://openpowerfoundation.org/specifications/isa/>

   418→// ===========================================================================
   419→// Instruction Encoding Helpers
   420→// ===========================================================================
   421→
   422→/// Encode a PPC64 32-bit instruction word and return as big-endian bytes.
   423→fn encode_word(word: u32) -> [u8; 4] {
   424→    word.to_be_bytes()
   425→}
   426→
   427→/// Build a D-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | d[16:31]
   428→fn encode_d_form(opcode: u32, rt: u32, ra: u32, d: i32) -> [u8; 4] {
   429→    let word =
   430→        ((opcode & 0x3F) << 26) | ((rt & 0x1F) << 21) | ((ra & 0x1F) << 16) | ((d as u32) & 0xFFFF);
   431→    encode_word(word)
   432→}
   433→
   434→/// Build a DS-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | ds[16:29] | xo[30:31]
   435→/// The `ds` parameter is the byte offset; the DS field stores ds/4.
   436→fn encode_ds_form(opcode: u32, rt: u32, ra: u32, ds: i32, xo: u32) -> [u8; 4] {
   437→    let word = ((opcode & 0x3F) << 26)
   438→        | ((rt & 0x1F) << 21)
   439→        | ((ra & 0x1F) << 16)
   440→        | (((ds >> 2) as u32) & 0x3FFF) << 2
   441→        | (xo & 0x3);
   442→    encode_word(word)
   443→}
   444→
   445→/// Build an X-form instruction: opcode[0:5] | rS[6:10] | rA[11:15] | rB[16:20] | xo[21:30] | Rc[31]
   446→fn encode_x_form(opcode: u32, rs: u32, ra: u32, rb: u32, xo: u32, rc: u32) -> [u8; 4] {
   447→    let word = ((opcode & 0x3F) << 26)
   448→        | ((rs & 0x1F) << 21)
   449→        | ((ra & 0x1F) << 16)
   450→        | ((rb & 0x1F) << 11)
   451→        | ((xo & 0x3FF) << 1)
   452→        | (rc & 1);
   453→    encode_word(word)
   454→}
   455→
   456→/// Build an XO-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | rB[16:20] | OE[21] | xo[22:30] | Rc[31]
   457→fn encode_xo_form(opcode: u32, rt: u32, ra: u32, rb: u32, oe: u32, xo: u32, rc: u32) -> [u8; 4] {
   458→    let word = ((opcode & 0x3F) << 26)
   459→        | ((rt & 0x1F) << 21)
   460→        | ((ra & 0x1F) << 16)
   461→        | ((rb & 0x1F) << 11)
   462→        | ((oe & 1) << 10)
   463→        | ((xo & 0x1FF) << 1)
   464→        | (rc & 1);
   465→    encode_word(word)
   466→}
   467→
   468→/// Build an M-form instruction: opcode[0:5] | rS[6:10] | rA[11:15] | SH[16:20] | MB[21:25] | ME[26:30] | Rc[31]
   469→fn encode_m_form(opcode: u32, rs: u32, ra: u32, sh: u32, mb: u32, me: u32, rc: u32) -> [u8; 4] {
   470→    let word = ((opcode & 0x3F) << 26)
   471→        | ((rs & 0x1F) << 21)
   472→        | ((ra & 0x1F) << 16)
   473→        | ((sh & 0x1F) << 11)
   474→        | ((mb & 0x1F) << 6)
   475→        | ((me & 0x1F) << 1)
   476→        | (rc & 1);
   477→    encode_word(word)
   478→}
   479→
   480→/// Build an I-form instruction: opcode[0:5] | LI[6:29] | AA[30] | LK[31]
   481→/// LI is a 24-bit signed value (word offset from CIA).
   482→fn encode_i_form(opcode: u32, li: i32, aa: u32, lk: u32) -> [u8; 4] {
   483→    let word =
   484→        ((opcode & 0x3F) << 26) | (((li as u32) & 0x00FF_FFFF) << 2) | ((aa & 1) << 1) | (lk & 1);
   485→    encode_word(word)
   486→}
   487→
   488→/// Build a B-form instruction: opcode[0:5] | BO[6:10] | BI[11:15] | BD[16:29] | AA[30] | LK[31]
   489→fn encode_b_form(opcode: u32, bo: u32, bi: u32, bd: i32, aa: u32, lk: u32) -> [u8; 4] {
   490→    let word = ((opcode & 0x3F) << 26)
   491→        | ((bo & 0x1F) << 21)
   492→        | ((bi & 0x1F) << 16)
   493→        | (((bd as u32) & 0x3FFF) << 2)
   494→        | ((aa & 1) << 1)
   495→        | (lk & 1);
   496→    encode_word(word)
   497→}
   498→
   499→/// Build an XL-form instruction: opcode[0:5] | BO[6:10] | BI[11:15] | 0[16:18] | BH[19:21] | xo[22:30] | LK[31]
   500→/// BH[19:21] in MSB-first bit numbering = normal bits 12:10 = shift by 10.
   501→fn encode_xl_form(opcode: u32, bo: u32, bi: u32, bh: u32, xo: u32, lk: u32) -> [u8; 4] {
   502→    let word = ((opcode & 0x3F) << 26)
   503→        | ((bo & 0x1F) << 21)
   504→        | ((bi & 0x1F) << 16)
   505→        | ((bh & 0x7) << 10)
   506→        | ((xo & 0x3FF) << 1)
   507→        | (lk & 1);
   508→    encode_word(word)
   509→}

   515→/// PowerPC64 instruction representations for code generation.
   516→///
   517→/// Covers key arithmetic, logical, shift/rotate, load/store, compare, branch,
   518→/// move, and system instructions. Each variant captures the operands needed for
   519→/// encoding and disassembly. The `encode()` method produces a 4-byte
   520→/// little-endian machine code word.
   521→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   522→pub enum Instruction {
   523→    // ── Arithmetic ──────────────────────────────────────────────────
   524→    /// Add: `add rT, rA, rB` (XO-form, primary=31, xo=266)
   525→    Add { rt: Gpr, ra: Gpr, rb: Gpr },
   526→    /// Add Immediate: `addi rT, rA, simm16` (D-form, primary=14)
   527→    Addi { rt: Gpr, ra: Gpr, simm: i32 },
   528→    /// Add Immediate Shifted: `addis rT, rA, simm16` (D-form, primary=15)
   529→    Addis { rt: Gpr, ra: Gpr, simm: i32 },
   530→    /// Subtract From: `subf rT, rA, rB` (XO-form, primary=31, xo=40)
   531→    Subf { rt: Gpr, ra: Gpr, rb: Gpr },
   532→    /// Multiply Low Word: `mullw rT, rA, rB` (XO-form, primary=31, xo=235)
   533→    Mullw { rt: Gpr, ra: Gpr, rb: Gpr },

  1403→/// Build a proper ELF64 binary for PPC64LE with 2 LOAD segments.
  1404→///
  1405→/// Produces a static executable with:
  1406→/// - Segment 1: LOAD (PF_R | PF_X) — .text
  1407→/// - Segment 2: LOAD (PF_R | PF_W) — .data / BSS (writable)
  1408→///
  1409→/// Uses little-endian byte order for ppc64le.  The text segment is placed
  1410→/// at a page-aligned offset so that the kernel's ELF loader can mmap() it.
  1411→fn build_ppc64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1412→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (PPC64 typical page size)
  1413→
  1414→    let elf_header_size: u64 = 64;
  1415→    let phdr_size: u64 = 56;
  1416→    let num_phdrs: u64 = 2;
  1417→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1418→    // Page-align the text segment start in the file.
  1419→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1420→    let text_size = code.len() as u64;
  1421→
  1422→    // The data segment starts on the next page after the text.
  1423→    let text_file_end = text_offset + text_size;
  1424→    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1425→    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1426→    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1427→    let entry_point = base_addr + text_offset;
  1428→
  1429→    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1430→
  1431→    // --- e_ident ---
  1432→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1433→    elf.push(2); // ELFCLASS64
  1434→    elf.push(2); // ELFDATA2MSB (big-endian PPC64)
  1435→    elf.push(1); // EV_CURRENT
  1436→    elf.push(3); // ELFOSABI_LINUX
  1437→    elf.push(0); // padding
  1438→    elf.extend_from_slice(&[0u8; 7]); // padding
  1439→
  1440→    // --- ELF header fields ---
  1441→    elf.extend_from_slice(&2u16.to_be_bytes()); // e_type = ET_EXEC
  1442→    elf.extend_from_slice(&21u16.to_be_bytes()); // e_machine = EM_PPC64
  1443→    elf.extend_from_slice(&1u32.to_be_bytes()); // e_version
  1444→    elf.extend_from_slice(&entry_point.to_be_bytes()); // e_entry
  1445→    elf.extend_from_slice(&elf_header_size.to_be_bytes()); // e_phoff
  1446→    elf.extend_from_slice(&0u64.to_be_bytes()); // e_shoff (no section headers)
  1447→    // e_flags: EF_PPC64_ABI_V2 = 0x2 (required for PPC64LE ELFv2 ABI)
  1448→    elf.extend_from_slice(&2u32.to_be_bytes()); // e_flags
  1449→    elf.extend_from_slice(&64u16.to_be_bytes()); // e_ehsize
  1450→    elf.extend_from_slice(&56u16.to_be_bytes()); // e_phentsize
  1451→    elf.extend_from_slice(&2u16.to_be_bytes()); // e_phnum = 2
  1452→    elf.extend_from_slice(&64u16.to_be_bytes()); // e_shentsize
  1453→    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shnum
  1454→    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shstrndx
  1455→
  1456→    // --- Program Header 1: LOAD (PF_R | PF_X) — .text ---
  1457→    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1458→    elf.extend_from_slice(&5u32.to_be_bytes()); // p_flags = PF_R | PF_X
  1459→    elf.extend_from_slice(&text_offset.to_be_bytes()); // p_offset
  1460→    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_vaddr
  1461→    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_paddr
  1462→    elf.extend_from_slice(&text_size.to_be_bytes()); // p_filesz
  1463→    elf.extend_from_slice(&text_size.to_be_bytes()); // p_memsz
  1464→    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1465→
  1466→    // --- Program Header 2: LOAD (PF_R | PF_W) — .data / stack ---
  1467→    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1468→    elf.extend_from_slice(&6u32.to_be_bytes()); // p_flags = PF_R | PF_W
  1469→    elf.extend_from_slice(&data_offset.to_be_bytes()); // p_offset
  1470→    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_vaddr
  1471→    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_paddr
  1472→    elf.extend_from_slice(&0u64.to_be_bytes()); // p_filesz (no initialized data)
  1473→    elf.extend_from_slice(&data_size.to_be_bytes()); // p_memsz (writable pages)
  1474→    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1475→
  1476→    // --- .text section ---
  1477→    // Pad to page-aligned text_offset
  1478→    while (elf.len() as u64) < text_offset {
  1479→        elf.push(0);
  1480→    }
  1481→    elf.extend_from_slice(code);
  1482→
  1483→    // --- Pad to data segment offset (if needed) ---
  1484→    while (elf.len() as u64) < data_offset {
  1485→        elf.push(0);
  1486→    }
  1487→
  1488→    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1489→
  1490→    elf
  1491→}

  1517→/// Compute the stack frame size for an IR function on PPC64.
  1518→///
  1519→/// Sums `Alloc` instruction sizes, adds 32 bytes for the LR/CR save area
  1520→/// (per ELFv2 ABI), and rounds up to 16-byte alignment.
  1521→fn ppc64_compute_frame_size(func: &IRFunction) -> usize {
  1522→    let mut total: u32 = 32; // LR save (8) + CR save (8) + back chain (8) + TOC save (8)
  1523→    for block in &func.blocks {
  1524→        for instr in &block.instructions {
  1525→            if let IRInstr::Alloc { size, .. } = instr {
  1526→                let aligned = (*size).div_ceil(16) * 16;
  1527→                total += aligned;
  1528→            }
  1529→        }
  1530→    }
  1531→    // Round up to 16-byte alignment
  1532→    total = (total + 15) & !15;
  1533→    total as usize
  1534→}

  1536→/// Allocatable GPR registers for PPC64, in priority order.
  1537→///
  1538→/// Order: argument registers first, then volatile temporaries, then callee-saved.
  1539→/// R0 is reserved (not allocatable — in some ISA contexts rA=0 means literal zero).

  1555→    Gpr::R14,

  1557→    Gpr::R16,

  1575→/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1576→fn map_vreg_to_gpr(
  1577→    vreg_id: u32,
  1578→    arg_index: Option<usize>,
  1579→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1580→) -> Gpr {
  1581→    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1582→        return *gpr;
  1583→    }
  1584→
  1585→    // If this is an argument, map it to the corresponding argument register.
  1586→    if let Some(idx) = arg_index {
  1587→        if let Some(reg) = Gpr::arg_register(idx) {
  1588→            vreg_map.insert(vreg_id, reg);
  1589→            return reg;
  1590→        }
  1591→    }
  1592→
  1593→    // Otherwise, pick the first available allocatable register.
  1594→    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1595→    for &reg in ALLOCATABLE_GPRS {
  1596→        if !used.contains(&reg) {
  1597→            vreg_map.insert(vreg_id, reg);
  1598→            return reg;
  1599→        }
  1600→    }
  1601→
  1602→    // Fallback: use R11 as a scratch register.
  1603→    Gpr::R11
  1604→}

  1610→        _ => 0,
  1611→    }
  1612→}

  1628→/// Load a 64-bit immediate value into a GPR, emitting the minimal number of
  1629→/// PPC64 instructions.
  1630→///
  1631→/// Strategy:
  1632→/// - **`[-32768, 32767]`**: `li rd, val` (single `addi rd, r0, val`)
  1633→/// - **`[0, 65535]`**: `ori rd, r0, val`
  1634→/// - **`0xXXXX0000` (low 16 bits zero, fits 32-bit)**: `lis rd, upper16`
  1635→/// - **Other 32-bit**: `lis rd, upper16` + `ori rd, rd, lower16`
  1636→/// - **Full 64-bit**: `lis` + `ori` + `sldi 32` + `oris` + `ori`
  1637→fn load_immediate_ppc64(rd: Gpr, val: i64, out: &mut Vec<AllocatedInstruction>) {
  1638→    let rd_phys = PhysicalReg::new(RegClass::Gpr, rd.encoding());
  1639→    let uval = val as u64;
  1640→
  1641→    if (-32768..=32767).contains(&val) {
  1642→        // li rd, val (addi rd, r0, val)
  1643→        out.push(emit_alloc_instr(
  1644→            Instruction::Li {

  1772→/// Resolve an `IRValue` to a physical `Gpr`, emitting immediate-load
  1773→/// instructions as needed.
  1774→///
  1775→/// - `IRValue::Register(id)` → looks up `id` in `reg_map`
  1776→/// - `IRValue::Immediate(val)` → loads `val` into `scratch` via
  1777→///   `load_immediate_ppc64` and returns `scratch`
  1778→/// - `IRValue::Address(addr)` → loads `addr` into `scratch`
  1779→/// - `IRValue::Label(_)` → loads 0 into `scratch` (placeholder for relocation)
  1780→///
  1781→/// The caller must supply a scratch register that is not live at this point
  1782→/// (e.g., `R11` for the first operand, `R12` for the second).
  1783→fn resolve_gpr_ppc64(
  1784→    val: &IRValue,
  1785→    reg_map: &mut std::collections::HashMap<u32, Gpr>,
  1786→    scratch: Gpr,
  1787→    out: &mut Vec<AllocatedInstruction>,
  1788→) -> Gpr {
  1789→    match val {
  1790→        IRValue::Register(id) => map_vreg_to_gpr(*id, None, reg_map),
  1791→        IRValue::Immediate(imm) => {
  1792→            load_immediate_ppc64(scratch, *imm, out);
  1793→            scratch
  1794→        }
  1795→        IRValue::Address(addr) => {
  1796→            load_immediate_ppc64(scratch, *addr as i64, out);
  1797→            scratch
  1798→        }
  1799→        IRValue::Label(_) => {
  1800→            // Labels need linker relocation; emit li scratch, 0 as placeholder
  1801→            load_immediate_ppc64(scratch, 0, out);
  1802→            scratch
  1803→        }
  1804→    }
  1805→}
