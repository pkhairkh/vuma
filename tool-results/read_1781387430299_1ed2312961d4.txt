     1→     1→//! # MIPS 64-bit Backend
     2→     2→//!
     3→     3→//! Implements the `Backend` trait for the MIPS 64-bit target (N64 ABI,
     4→     4→//! big-endian).  This module provides:
     5→     5→//!
     6→     6→//! - `Gpr` — General-purpose register enum ($0–$31)
     7→     7→//! - `Fpr` — Floating-point register enum ($f0–$f31)
     8→     8→//! - `Instruction` — MIPS64 instruction enum with correct encoding for
     9→     9→//!   R-type, I-type, and J-type formats
    10→    10→//! - `Mips64Backend` — `Backend` implementation that lowers IR to MIPS64
    11→    11→//!   machine code and emits ELF64 binaries
    12→    12→//!
    13→    13→//! ## MIPS64 Register Convention (N64 ABI)
    14→    14→//!
    15→    15→//! | Register(s) | ABI Name | Role                              |
    16→    16→//! |-------------|----------|-----------------------------------|
    17→    17→//! | $0          | $zero    | Hardwired zero                    |
    18→    18→//! | $1          | $at      | Assembler temporary               |
    19→    19→//! | $2–$3       | $v0–$v1  | Return values                     |
    20→    20→//! | $4–$7       | $a0–$a3  | Argument registers                |
    21→    21→//! | $8–$15      | $t0–$t7  | Caller-saved temporaries          |
    22→    22→//! | $16–$23     | $s0–$s7  | Callee-saved                      |
    23→    23→//! | $24–$25     | $t8–$t9  | Caller-saved temporaries          |
    24→    24→//! | $26–$27     | $k0–$k1  | Kernel registers                  |
    25→    25→//! | $28         | $gp      | Global pointer                    |
    26→    26→//! | $29         | $sp      | Stack pointer                     |
    27→    27→//! | $30         | $fp      | Frame pointer (callee-saved)      |
    28→    28→//! | $31         | $ra      | Return address                    |
    29→    29→//!
    30→    30→//! ## Branch Delay Slots
    31→    31→//!
    32→    32→//! MIPS has branch delay slots: the instruction immediately following a branch
    33→    33→//! or jump is always executed before the branch takes effect.  This backend
    34→    34→//! inserts a NOP (0x00000000) in every delay slot for correctness.
    35→    35→//!
    36→    36→//! ## Instruction Encoding
    37→    37→//!
    38→    38→//! All instructions are 32 bits, **big-endian**, with three formats:
    39→    39→//!
    40→    40→//! - **R-type**: `opcode[31:26] | rs[25:21] | rt[20:16] | rd[15:11] | sa[10:6] | funct[5:0]`
    41→    41→//! - **I-type**: `opcode[31:26] | rs[25:21] | rt[20:16] | imm[15:0]`
    42→    42→//! - **J-type**: `opcode[31:26] | target[25:0]`
    43→    43→//!
    44→    44→//! ## References
    45→    45→//!
    46→    46→//! - MIPS64 Architecture for Programmers, Volume II: Instruction Set
    47→    47→//! - <https://www.mips.com/products/architectures/mips64/>

   420→   420→// Instruction Encoding Helpers
   421→   421→// ===========================================================================
   422→   422→
   423→   423→/// Encode an R-type instruction (big-endian).
   424→   424→///
   425→   425→/// Format: `opcode[31:26] | rs[25:21] | rt[20:16] | rd[15:11] | sa[10:6] | funct[5:0]`
   426→   426→fn encode_r_type(opcode: u32, rs: u32, rt: u32, rd: u32, sa: u32, funct: u32) -> [u8; 4] {
   427→   427→    let word = ((opcode & 0x3F) << 26)
   428→   428→        | ((rs & 0x1F) << 21)
   429→   429→        | ((rt & 0x1F) << 16)
   430→   430→        | ((rd & 0x1F) << 11)
   431→   431→        | ((sa & 0x1F) << 6)
   432→   432→        | (funct & 0x3F);
   433→   433→    word.to_be_bytes()
   434→   434→}
   435→   435→
   436→   436→/// Encode an I-type instruction (big-endian).
   437→   437→///
   438→   438→/// Format: `opcode[31:26] | rs[25:21] | rt[20:16] | imm[15:0]`
   439→   439→fn encode_i_type(opcode: u32, rs: u32, rt: u32, imm: u32) -> [u8; 4] {
   440→   440→    let word = ((opcode & 0x3F) << 26) | ((rs & 0x1F) << 21) | ((rt & 0x1F) << 16) | (imm & 0xFFFF);
   441→   441→    word.to_be_bytes()
   442→   442→}
   443→   443→
   444→   444→/// Encode a J-type instruction (big-endian).
   445→   445→///
   446→   446→/// Format: `opcode[31:26] | target[25:0]`
   447→   447→fn encode_j_type(opcode: u32, target: u32) -> [u8; 4] {
   448→   448→    let word = ((opcode & 0x3F) << 26) | (target & 0x03FF_FFFF);
   449→   449→    word.to_be_bytes()
   450→   450→}

  1341→  1341→/// Build a proper ELF64 binary for MIPS64 (big-endian) with 2 LOAD segments.
  1342→  1342→///
  1343→  1343→/// Produces a static executable with:
  1344→  1344→/// - Segment 1: LOAD RX — `.text` section (code)
  1345→  1345→/// - Segment 2: LOAD RW — `.data` / BSS (writable memory for stack/data)
  1346→  1346→///
  1347→  1347→/// All header fields are written in big-endian byte order.
  1348→  1348→/// e_flags is set to EF_MIPS_ARCH_64 (0x60000000) for the MIPS64 N64 ABI.
  1349→  1349→fn build_mips64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1350→  1350→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (MIPS typical page size)
  1351→  1351→
  1352→  1352→    let elf_header_size: u64 = 64;
  1353→  1353→    let phdr_size: u64 = 56;
  1354→  1354→    let num_phdrs: u64 = 2;
  1355→  1355→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1356→  1356→    // Page-align the text segment start in the file.
  1357→  1357→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1358→  1358→    let text_size = code.len() as u64;
  1359→  1359→
  1360→  1360→    // The data segment starts on the next page after the text.
  1361→  1361→    let text_file_end = text_offset + text_size;
  1362→  1362→    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1363→  1363→    let data_offset = data_vaddr - base_addr;
  1364→  1364→    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1365→  1365→    let entry_point = base_addr + text_offset;
  1366→  1366→
  1367→  1367→    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1368→  1368→
  1369→  1369→    // --- e_ident ---
  1370→  1370→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1371→  1371→    elf.push(2); // ELFCLASS64
  1372→  1372→    elf.push(2); // ELFDATA2MSB (big-endian)
  1373→  1373→    elf.push(1); // EV_CURRENT
  1374→  1374→    elf.push(3); // ELFOSABI_LINUX
  1375→  1375→    elf.push(0); // padding
  1376→  1376→    elf.extend_from_slice(&[0u8; 7]); // padding
  1377→  1377→
  1378→  1378→    // --- ELF header fields (big-endian) ---
  1379→  1379→    elf.extend_from_slice(&2u16.to_be_bytes()); // e_type = ET_EXEC
  1380→  1380→    elf.extend_from_slice(&8u16.to_be_bytes()); // e_machine = EM_MIPS
  1381→  1381→    elf.extend_from_slice(&1u32.to_be_bytes()); // e_version
  1382→  1382→    elf.extend_from_slice(&entry_point.to_be_bytes()); // e_entry
  1383→  1383→    elf.extend_from_slice(&elf_header_size.to_be_bytes()); // e_phoff
  1384→  1384→    elf.extend_from_slice(&0u64.to_be_bytes()); // e_shoff (no section headers)
  1385→  1385→    // e_flags: EF_MIPS_ARCH_64 = 0x60000000 (MIPS64 ISA, N64 ABI implied by ELFCLASS64)
  1386→  1386→    elf.extend_from_slice(&0x60000000u32.to_be_bytes()); // e_flags
  1387→  1387→    elf.extend_from_slice(&64u16.to_be_bytes()); // e_ehsize
  1388→  1388→    elf.extend_from_slice(&56u16.to_be_bytes()); // e_phentsize
  1389→  1389→    elf.extend_from_slice(&2u16.to_be_bytes()); // e_phnum = 2
  1390→  1390→    elf.extend_from_slice(&64u16.to_be_bytes()); // e_shentsize
  1391→  1391→    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shnum
  1392→  1392→    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shstrndx
  1393→  1393→
  1394→  1394→    // --- Program Header 1: LOAD (PF_R | PF_X) — .text ---
  1395→  1395→    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1396→  1396→    elf.extend_from_slice(&5u32.to_be_bytes()); // p_flags = PF_R | PF_X
  1397→  1397→    elf.extend_from_slice(&text_offset.to_be_bytes()); // p_offset
  1398→  1398→    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_vaddr
  1399→  1399→    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_paddr
  1400→  1400→    elf.extend_from_slice(&text_size.to_be_bytes()); // p_filesz
  1401→  1401→    elf.extend_from_slice(&text_size.to_be_bytes()); // p_memsz
  1402→  1402→    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1403→  1403→
  1404→  1404→    // --- Program Header 2: LOAD (PF_R | PF_W) — .data / stack ---
  1405→  1405→    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1406→  1406→    elf.extend_from_slice(&6u32.to_be_bytes()); // p_flags = PF_R | PF_W
  1407→  1407→    elf.extend_from_slice(&data_offset.to_be_bytes()); // p_offset
  1408→  1408→    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_vaddr
  1409→  1409→    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_paddr
  1410→  1410→    elf.extend_from_slice(&0u64.to_be_bytes()); // p_filesz (no initialized data)
  1411→  1411→    elf.extend_from_slice(&data_size.to_be_bytes()); // p_memsz (writable pages)
  1412→  1412→    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1413→  1413→
  1414→  1414→    // --- .text section ---
  1415→  1415→    // Pad to page-aligned text_offset
  1416→  1416→    while (elf.len() as u64) < text_offset {
  1417→  1417→        elf.push(0);
  1418→  1418→    }
  1419→  1419→    elf.extend_from_slice(code);
  1420→  1420→
  1421→  1421→    // --- Pad to data segment offset (if needed) ---
  1422→  1422→    while (elf.len() as u64) < data_offset {
  1423→  1423→        elf.push(0);
  1424→  1424→    }
  1425→  1425→
  1426→  1426→    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1427→  1427→
  1428→  1428→    elf
  1429→  1429→}

  1459→  1459→/// Compute the stack frame size for an IR function on MIPS64.
  1460→  1460→///
  1461→  1461→/// Sums `Alloc` instruction sizes, adds 8 bytes for the $ra save slot,
  1462→  1462→/// and rounds up to 16-byte alignment.
  1463→  1463→fn mips64_compute_frame_size(func: &IRFunction) -> usize {
  1464→  1464→    let mut total: u32 = 8; // $ra save slot
  1465→  1465→    for block in &func.blocks {
  1466→  1466→        for instr in &block.instructions {
  1467→  1467→            if let IRInstr::Alloc { size, .. } = instr {
  1468→  1468→                let aligned = (*size).div_ceil(16) * 16;
  1469→  1469→                total += aligned;
  1470→  1470→            }
  1471→  1471→        }
  1472→  1472→    }
  1473→  1473→    // Round up to 16-byte alignment
  1474→  1474→    total = (total + 15) & !15;
  1475→  1475→    total as usize
  1476→  1476→}

  1478→  1478→/// Allocatable GPR registers for MIPS64, in priority order.
  1479→  1479→///
  1480→  1480→/// Order: temporaries first, then argument registers, then callee-saved.
  1481→  1481→const ALLOCATABLE_GPRS: &[Gpr] = &[
  1482→  1482→    // Caller-saved temporaries (highest priority — no save/restore needed)
  1483→  1483→    Gpr::T0,
  1484→  1484→    Gpr::T1,
  1485→  1485→    Gpr::T2,
  1486→  1486→    Gpr::T3,

  1488→  1488→    Gpr::T5,

  1490→  1490→    Gpr::T7,
  1491→  1491→    Gpr::T8,
  1492→  1492→    Gpr::T9,
  1493→  1493→    // Return value registers (also caller-saved)
  1494→  1494→    Gpr::V0,
  1495→  1495→    Gpr::V1,
  1496→  1496→    // Argument registers (also caller-saved)
  1497→  1497→    Gpr::A0,
  1498→  1498→    Gpr::A1,
  1499→  1499→    Gpr::A2,
  1500→  1500→    Gpr::A3,
  1501→  1501→    // Callee-saved (require save/restore)
  1502→  1502→    Gpr::S0,
  1503→  1503→    Gpr::S1,
  1504→  1504→    Gpr::S2,
  1505→  1505→    Gpr::S3,
  1506→  1506→    Gpr::S4,
  1507→  1507→    Gpr::S5,
  1508→  1508→    Gpr::S6,
  1509→  1509→    Gpr::S7,

  1511→  1511→    Gpr::Fp,

  1514→  1514→/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1515→  1515→///
  1516→  1516→/// Argument virtual registers are mapped to a0–a3 first.  Remaining virtual
  1517→  1517→/// registers are assigned from the pool of allocatable registers.
  1518→  1518→fn map_vreg_to_gpr(
  1519→  1519→    vreg_id: u32,
  1520→  1520→    arg_index: Option<usize>,
  1521→  1521→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1522→  1522→) -> Gpr {
  1523→  1523→    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1524→  1524→        return *gpr;
  1525→  1525→    }
  1526→  1526→
  1527→  1527→    // If this is an argument, map it to the corresponding argument register.
  1528→  1528→    if let Some(idx) = arg_index {
  1529→  1529→        if let Some(reg) = Gpr::arg_register(idx) {
  1530→  1530→            vreg_map.insert(vreg_id, reg);
  1531→  1531→            return reg;
  1532→  1532→        }
  1533→  1533→    }
  1534→  1534→
  1535→  1535→    // Otherwise, pick the first available allocatable register.
  1536→  1536→    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1537→  1537→    for &reg in ALLOCATABLE_GPRS {
  1538→  1538→        if !used.contains(&reg) {
  1539→  1539→            vreg_map.insert(vreg_id, reg);
  1540→  1540→            return reg;
  1541→  1541→        }
  1542→  1542→    }
  1543→  1543→
  1544→  1544→    // Fallback: use t9 as a scratch register (will cause issues, but shouldn't
  1545→  1545→    // happen with reasonable register pressure).
  1546→  1546→    Gpr::T9
  1547→  1547→}

  1549→  1549→/// Helper: extract the virtual register ID from an IRValue, if it is a register.
  1550→  1550→fn vreg_id(val: &IRValue) -> u32 {
  1551→  1551→    match val {
  1552→  1552→        IRValue::Register(id) => *id,
  1553→  1553→        _ => 0,
  1554→  1554→    }
  1555→  1555→}

  1557→  1557→/// Resolve an IRValue to a physical GPR, emitting load-immediate instructions
  1558→  1558→/// if the value is an immediate constant.
  1559→  1559→///
  1560→  1560→/// Returns a tuple of `(register, instructions)`.  The `instructions` vector
  1561→  1561→/// contains any AllocatedInstructions needed to load an immediate into a
  1562→  1562→/// register (empty if the value was already in a register).
  1563→  1563→///
  1564→  1564→/// `imm_counter` is used to generate unique virtual register IDs for immediate
  1565→  1565→/// temporaries.  It is incremented each time an immediate is loaded.
  1566→  1566→fn resolve_value(
  1567→  1567→    val: &IRValue,
  1568→  1568→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1569→  1569→    imm_counter: &mut u32,
  1570→  1570→) -> (Gpr, Vec<AllocatedInstruction>) {
  1571→  1571→    if let Some(imm) = val.as_immediate() {
  1572→  1572→        // Allocate a unique vreg ID for this immediate
  1573→  1573→        let vreg_id = *imm_counter;
  1574→  1574→        *imm_counter += 1;
  1575→  1575→        let dst = map_vreg_to_gpr(vreg_id, None, vreg_map);
  1576→  1576→        let mut code = Vec::new();
  1577→  1577→
  1578→  1578→        // Load the 64-bit immediate into dst.
  1579→  1579→        // MIPS64: use lui + ori + dsll + daddiu sequence for full 64-bit.
  1580→  1580→        let u64_val = imm as u64;
  1581→  1581→        let highest = (u64_val >> 48) & 0xFFFF;
  1582→  1582→        let higher  = (u64_val >> 32) & 0xFFFF;
  1583→  1583→        let hi      = (u64_val >> 16) & 0xFFFF;
  1584→  1584→        let lo      = u64_val & 0xFFFF;
  1585→  1585→
  1586→  1586→        if imm >= -32768 && imm <= 32767 {
  1587→  1587→            // Fits in 16-bit sign-extended: addiu dst, $zero, imm
  1588→  1588→            let inst = Instruction::Addiu { rt: dst, rs: Gpr::Zero, imm: imm as i32 };
  1589→  1589→            code.push(AllocatedInstruction {
  1590→  1590→                opcode: "addiu".to_string(),
  1591→  1591→                reads: vec![],
  1592→  1592→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1593→  1593→                encoded: inst.encode().to_vec(),
  1594→  1594→            });
  1595→  1595→        } else if imm >= 0 && imm <= 65535 {
  1596→  1596→            // 16-bit unsigned: ori dst, $zero, imm
  1597→  1597→            let inst = Instruction::Ori { rt: dst, rs: Gpr::Zero, imm: imm as u32 };
  1598→  1598→            code.push(AllocatedInstruction {
  1599→  1599→                opcode: "ori".to_string(),
  1600→  1600→                reads: vec![],
  1601→  1601→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1602→  1602→                encoded: inst.encode().to_vec(),
  1603→  1603→            });
  1604→  1604→        } else if imm >= -2147483648 && imm <= 2147483647 && (imm as i32 >> 16) == 0 || (hi == 0 && (lo as i16 as i32) < 0) {

  1606→  1606→            let lui_val = ((imm as i32 as u32) >> 16) as u32;
  1607→  1607→            let ori_val = (imm as i32 as u32) & 0xFFFF;
  1608→  1608→            let lui_inst = Instruction::Lui { rt: dst, imm: lui_val };
  1609→  1609→            code.push(AllocatedInstruction {
  1610→  1610→                opcode: "lui".to_string(),
  1611→  1611→                reads: vec![],
  1612→  1612→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1613→  1613→                encoded: lui_inst.encode().to_vec(),
  1614→  1614→            });
  1615→  1615→            if ori_val != 0 {
  1616→  1616→                let ori_inst = Instruction::Ori { rt: dst, rs: dst, imm: ori_val };
  1617→  1617→                code.push(AllocatedInstruction {
  1618→  1618→                    opcode: "ori".to_string(),
  1619→  1619→                    reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1620→  1620→                    writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1621→  1621→                    encoded: ori_inst.encode().to_vec(),
  1622→  1622→                });
  1623→  1623→            }
  1624→  1624→        } else if hi != 0 || lo != 0 {
  1625→  1625→            // Full 64-bit immediate load
  1626→  1626→            // lui dst, highest
  1627→  1627→            let lui_inst = Instruction::Lui { rt: dst, imm: highest as u32 };
  1628→  1628→            code.push(AllocatedInstruction {
  1629→  1629→                opcode: "lui".to_string(),
  1630→  1630→                reads: vec![],
  1631→  1631→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1632→  1632→                encoded: lui_inst.encode().to_vec(),
  1633→  1633→            });
  1634→  1634→            // daddiu dst, dst, higher
  1635→  1635→            let daddiu_inst = Instruction::Daddiu { rt: dst, rs: dst, imm: higher as i32 };

  1642→  1642→            // dsll dst, dst, 16
  1643→  1643→            let dsll_inst = Instruction::Dsll { rd: dst, rt: dst, sa: 16 };

  1645→  1645→                opcode: "dsll".to_string(),
  1646→  1646→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],

  1650→  1650→            // daddiu dst, dst, hi

  1656→  1656→                encoded: daddiu_hi.encode().to_vec(),

  1658→  1658→            // dsll dst, dst, 16
  1659→  1659→            let dsll2 = Instruction::Dsll { rd: dst, rt: dst, sa: 16 };
  1660→  1660→            code.push(AllocatedInstruction {
  1661→  1661→                opcode: "dsll".to_string(),
  1662→  1662→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1663→  1663→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1664→  1664→                encoded: dsll2.encode().to_vec(),
  1665→  1665→            });
  1666→  1666→            // daddiu dst, dst, lo
  1667→  1667→            let daddiu_lo = Instruction::Daddiu { rt: dst, rs: dst, imm: lo as i32 };

  1683→  1683→            let daddiu_inst = Instruction::Daddiu { rt: dst, rs: dst, imm: higher as i32 };

  1696→  1696→            });
  1697→  1697→            let daddiu_hi = Instruction::Daddiu { rt: dst, rs: dst, imm: hi as i32 };
  1698→  1698→            code.push(AllocatedInstruction {
  1699→  1699→                opcode: "daddiu".to_string(),
  1700→  1700→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1701→  1701→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1702→  1702→                encoded: daddiu_hi.encode().to_vec(),
  1703→  1703→            });
  1704→  1704→            let dsll2 = Instruction::Dsll { rd: dst, rt: dst, sa: 16 };
  1705→  1705→            code.push(AllocatedInstruction {

  1712→  1712→
  1713→  1713→        (dst, code)
  1714→  1714→    } else {
  1715→  1715→        // Register value — just map it
  1716→  1716→        (map_vreg_to_gpr(vreg_id(val), None, vreg_map), Vec::new())

  1718→  1718→}

  1720→  1720→/// Lower a BinOpKind + operands to MIPS64 instructions.
