   317‚Üí   317‚Üí
   318‚Üí   318‚Üí/// Constrain a value to lie within the inclusive range `[lo, hi]`.
   319‚Üí   319‚Üí///
   320‚Üí   320‚Üí/// Returns `lo` if `x < lo`, `hi` if `x > hi`, or `x` otherwise.
   321‚Üí   321‚Üí/// This is equivalent to `max(lo, min(hi, x))` but expressed as a single
   322‚Üí   322‚Üí/// operation for clarity and efficiency.
   323‚Üí   323‚Üí///
   324‚Üí   324‚Üí/// ## Panics
   325‚Üí   325‚Üí///
   326‚Üí   326‚Üí/// In debug builds, panics if `lo > hi`. In release builds, the behavior
   327‚Üí   327‚Üí/// is `max(lo, min(hi, x))`, which effectively swaps the bounds.
   328‚Üí   328‚Üí///
   329‚Üí   329‚Üí/// ## VUMA Program Equivalent
   330‚Üí   330‚Üí///
   331‚Üí   331‚Üí/// ```vuma
   332‚Üí   332‚Üí/// fn clamp(x: i64, lo: i64, hi: i64) -> i64 {
   333‚Üí   333‚Üí///     if x < lo {
   334‚Üí   334‚Üí///         return lo;
   335‚Üí   335‚Üí///     }
   336‚Üí   336‚Üí///     if x > hi {
   337‚Üí   337‚Üí///         return hi;
   338‚Üí   338‚Üí///     }
   339‚Üí   339‚Üí///     return x;
   340‚Üí   340‚Üí/// }
   341‚Üí   341‚Üí/// ```
   342‚Üí   342‚Üí///
   343‚Üí   343‚Üí/// ## Common Use Cases
   344‚Üí   344‚Üí///
   345‚Üí   345‚Üí/// - Clamping array indices: `clamp(idx, 0, len - 1)`
   346‚Üí   346‚Üí/// - Clamping color channels: `clamp(channel, 0, 255)`
   347‚Üí   347‚Üí/// - Clamping buffer offsets: `clamp(offset, 0, buf_size - 1)`
   348‚Üí   348‚Üí/// - Saturation arithmetic: `clamp(a + b, i64::MIN, i64::MAX)`
   349‚Üí   349‚Üí///
   350‚Üí   350‚Üí/// ## BD Annotations
   351‚Üí   351‚Üí///
   352‚Üí   352‚Üí/// - CapD: { Read, Compare } ‚Äî pure function, no side effects
   353‚Üí   353‚Üí// VUMA-VERIFIED: clamp is correct when lo <= hi
   354‚Üí   354‚Üípub fn clamp(x: i64, lo: i64, hi: i64) -> i64 {
   355‚Üí   355‚Üí    debug_assert!(lo <= hi, "clamp: lo ({}) must be <= hi ({})", lo, hi);
   356‚Üí   356‚Üí    if x < lo {
   357‚Üí   357‚Üí        lo
   358‚Üí   358‚Üí    } else if x > hi {
   359‚Üí   359‚Üí        hi
   360‚Üí   360‚Üí    } else {
   361‚Üí   361‚Üí        x
   362‚Üí   362‚Üí    }
   363‚Üí   363‚Üí}
