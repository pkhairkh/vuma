     1→//! Lexer (tokeniser) for the VUMA language frontend.
     2→//!
     3→//! Transforms raw source text into a flat stream of [`Token`] values, each
     4→//! annotated with a [`Span`] and line/column position for error reporting.
     5→//! The lexer supports **full error recovery**: it never stops on the first
     6→//! error. Instead, it produces [`TokenKind::Error`] tokens and collects
     7→//! [`ParseError`] values that can be retrieved later via [`Lexer::errors`].
     8→//!
     9→//! # Example
    10→//!
    11→//! ```
    12→//! use vuma_parser::lexer::Lexer;
    13→//!
    14→//! let source = "region pool = allocate(1024);";
    15→//! let mut lexer = Lexer::new(source);
    16→//! let tokens = lexer.collect_tokens();
    17→//! for tok in &tokens {
    18→//!     println!("{:?} at {}:{} — '{}'", tok.kind, tok.line + 1, tok.column + 1, tok.lexeme);
    19→//! }
    20→//! ```
    21→
    22→use crate::error::{ParseError, ParseErrorKind, Span};
    23→use serde::{Deserialize, Serialize};

    25→// ---------------------------------------------------------------------------
    26→// Position
    27→// ---------------------------------------------------------------------------
    28→
    29→/// Source position: byte offset, line (0-based), and column (0-based).
    30→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
    31→pub struct Position {
    32→    /// Byte offset into the source text.
    33→    pub offset: usize,
    34→    /// Line number (0-based).
    35→    pub line: usize,
    36→    /// Column number (0-based, measured in Unicode code points).
    37→    pub column: usize,
    38→}
    39→
    40→impl Position {
    41→    /// Create a new position.
    42→    pub fn new(offset: usize, line: usize, column: usize) -> Self {
    43→        Self {
    44→            offset,
    45→            line,
    46→            column,
    47→        }
    48→    }
    49→
    50→    /// The initial position (offset 0, line 0, column 0).
    51→    pub fn zero() -> Self {
    52→        Self::new(0, 0, 0)
    53→    }
    54→}
    55→
    56→impl std::fmt::Display for Position {
    57→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    58→        write!(f, "{}:{}", self.line + 1, self.column + 1)
    59→    }
    60→}

    62→// ---------------------------------------------------------------------------
    63→// Token
    64→// ---------------------------------------------------------------------------
    65→
    66→/// A single lexical token produced by the [`Lexer`].
    67→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    68→pub struct Token {
    69→    /// The classified kind of this token.
    70→    pub kind: TokenKind,
    71→    /// The exact source text that was consumed (for strings this is the
    72→    /// decoded content; for everything else it is the raw source slice).
    73→    pub lexeme: String,
    74→    /// Byte-offset span within the original source.
    75→    pub span: Span,
    76→    /// Line number where this token starts (0-based).
    77→    pub line: usize,
    78→    /// Column number where this token starts (0-based).
    79→    pub column: usize,
    80→}
    81→
    82→impl Token {
    83→    /// Convenience constructor.
    84→    pub fn new(
    85→        kind: TokenKind,
    86→        lexeme: impl Into<String>,
    87→        span: Span,
    88→        line: usize,
    89→        column: usize,
    90→    ) -> Self {
    91→        Self {
    92→            kind,
    93→            lexeme: lexeme.into(),
    94→            span,
    95→            line,
    96→            column,
    97→        }
    98→    }
    99→
   100→    /// True when this token is the end-of-file sentinel.
   101→    pub fn is_eof(&self) -> bool {
   102→        self.kind == TokenKind::Eof
   103→    }
   104→}

   622→
   623→        // Booleans / null
   624→        "true" => Some(TokenKind::True),
   625→        "false" => Some(TokenKind::False),
   626→        "null" => Some(TokenKind::Null),
   627→
   628→        // Type operators
   629→        "sizeof" => Some(TokenKind::Sizeof),
   630→        "alignof" => Some(TokenKind::Alignof),
   631→
   632→        // Option/Result sugar
   633→        "Option" => Some(TokenKind::OptionKw),
   634→        "Some" => Some(TokenKind::SomeKw),
   635→        "None" => Some(TokenKind::NoneKw),
   636→        "Result" => Some(TokenKind::ResultKw),
   637→        "Ok" => Some(TokenKind::OkKw),
   638→        "Err" => Some(TokenKind::ErrKw),
   639→
   640→        _ => None,
   641→    }
   642→}

   644→// ---------------------------------------------------------------------------
   645→// Lexer
   646→// ---------------------------------------------------------------------------
   647→
   648→/// A streaming lexer that converts source text into [`Token`] values.
   649→///
   650→/// The lexer is *pull*-based: the caller invokes [`Lexer::next_token`]
   651→/// repeatedly until a token with [`TokenKind::Eof`] is returned.
   652→///
   653→/// **Error recovery**: the lexer never panics or returns `Err`. When it
   654→/// encounters invalid input it produces [`TokenKind::Error`] tokens and
   655→/// records the error internally. Call [`Lexer::errors`] or
   656→/// [`Lexer::take_errors`] to retrieve accumulated errors.
   657→///
   658→/// Whitespace and non-doc comments are silently skipped. Doc comments
   659→/// (`///` and `//!`) are emitted as tokens for downstream tooling.
   660→pub struct Lexer<'src> {
   661→    /// Full source text.
   662→    source: &'src str,
   663→    /// Characters yet to be consumed.
   664→    chars: std::iter::Peekable<std::str::Chars<'src>>,
   665→    /// Current byte offset into `source`.
   666→    offset: usize,
   667→    /// Current line number (0-based).
   668→    line: usize,
   669→    /// Current column number (0-based).
   670→    column: usize,
   671→    /// Lookahead token already consumed but not yet handed to the caller.
   672→    peeked: Option<Token>,
   673→    /// Cached EOF token used as a safe fallback in [`peek`] when the
   674→    /// peeked buffer is unexpectedly empty (defensive guard against panic).
   675→    eof_token: Token,
   676→    /// Accumulated lexical errors (for error recovery).
   677→    errors: Vec<ParseError>,
   678→}
   679→
   680→impl<'src> Lexer<'src> {
   681→    /// Create a new lexer over the given source text.
   682→    pub fn new(source: &'src str) -> Self {
   683→        Self {
   684→            source,
   685→            chars: source.chars().peekable(),
   686→            offset: 0,
   687→            line: 0,
   688→            column: 0,
   689→            peeked: None,
   690→            eof_token: Token::new(TokenKind::Eof, "", Span::new(0, 0), 0, 0),
   691→            errors: Vec::new(),
   692→        }
   693→    }
   694→
   695→    // -- public API ----------------------------------------------------------
   696→
   697→    /// Consume and return the next token from the source.
   698→    ///
   699→    /// On end of input a token with kind [`TokenKind::Eof`] is returned.
   700→    /// Lexical errors are reported via [`TokenKind::Error`] tokens and
   701→    /// collected internally (see [`Lexer::errors`]).
   702→    pub fn next_token(&mut self) -> Token {
   703→        if let Some(tok) = self.peeked.take() {
   704→            return tok;
   705→        }
   706→        self.advance()
   707→    }
   708→
   709→    /// Peek at the next token without consuming it.
   710→    ///
   711→    /// Successive calls to `peek` return the same token until
   712→    /// [`Lexer::next_token`] is called.
   713→    pub fn peek(&mut self) -> &Token {
   714→        if self.peeked.is_none() {
   715→            self.peeked = Some(self.advance());
   716→        }
   717→        // Defensive: if `peeked` is still `None` after the branch above
   718→        // (should never happen), fall back to the cached EOF token instead
   719→        // of panicking.
   720→        self.peeked.as_ref().unwrap_or(&self.eof_token)
   721→    }
   722→
   723→    /// Consume all remaining tokens and return them (including the final Eof).
   724→    pub fn collect_tokens(&mut self) -> Vec<Token> {
   725→        let mut tokens = Vec::new();
   726→        loop {
   727→            let tok = self.next_token();
   728→            let is_eof = tok.is_eof();
   729→            tokens.push(tok);
   730→            if is_eof {
   731→                break;
   732→            }
   733→        }
   734→        tokens
   735→    }
   736→
   737→    /// Return the accumulated lexical errors.
   738→    pub fn errors(&self) -> &[ParseError] {
   739→        &self.errors
   740→    }
   741→
   742→    /// Take all accumulated errors, leaving the internal buffer empty.
   743→    pub fn take_errors(&mut self) -> Vec<ParseError> {
   744→        std::mem::take(&mut self.errors)
   745→    }
   746→
   747→    /// Return the source text being lexed.
   748→    pub fn source(&self) -> &'src str {
   749→        self.source
   750→    }
   751→
   752→    // -- core advance --------------------------------------------------------
   753→
   754→    /// Core token production (bypasses peeked buffer).
   755→    fn advance(&mut self) -> Token {
   756→        // Loop: skip whitespace and non-doc comments until we find a real
   757→        // token or a doc-comment token.
   758→        loop {
   759→            self.skip_whitespace();
   760→
   761→            let start = self.offset;
   762→            let line = self.line;
   763→            let column = self.column;
   764→
   765→            let ch = match self.chars.peek() {
   766→                Some(&c) => c,
   767→                None => return self.eof_token(start, line, column),
   768→            };
   769→
   770→            // ---- Comment handling ----
   771→            if ch == '/' {
   772→                let second = self.peek_next(1);
   773→                if second == Some('/') {
   774→                    // `//` — line comment or doc comment
   775→                    self.bump(); // consume first /
   776→                    self.bump(); // consume second /
   777→                    let third = self.chars.peek().copied();
   778→                    if third == Some('/') {
   779→                        // `///` outer doc comment — emit as token
   780→                        self.bump(); // consume third /
   781→                        self.consume_to_eol();
   782→                        return self.make_token(TokenKind::DocComment, start, line, column);
   783→                    } else if third == Some('!') {
   784→                        // `//!` module doc comment — emit as token
   785→                        self.bump(); // consume !
   786→                        self.consume_to_eol();
   787→                        return self.make_token(TokenKind::ModuleDoc, start, line, column);
   788→                    } else {
   789→                        // Regular line comment — skip
   790→                        self.consume_to_eol();
   791→                        continue; // loop back for next token
   792→                    }
   793→                } else if second == Some('*') {
   794→                    // `/* */` block comment — skip
   795→                    self.bump(); // consume /
   796→                    self.bump(); // consume *
   797→                    self.skip_block_comment(start);
   798→                    continue; // loop back
   799→                } else {
   800→                    // Just `/` — the division operator; fall through to main match
   801→                    break;
   802→                }
   803→            }
   804→
   805→            // Not whitespace, not a comment — lex the token.
   806→            break;
   807→        }
   808→
   809→        self.lex_token()
   810→    }
   811→
   812→    /// Lex a single token starting at the current position.
   813→    fn lex_token(&mut self) -> Token {
   814→        let start = self.offset;
   815→        let line = self.line;
   816→        let column = self.column;
   817→
   818→        let ch = match self.chars.peek() {
   819→            Some(&c) => c,
   820→            None => return self.eof_token(start, line, column),
   821→        };
   822→
   823→        let kind = match ch {
   824→            // ---- Delimiters ----
   825→            '(' => {
   826→                self.bump();
   827→                TokenKind::LParen
   828→            }
   829→            ')' => {
   830→                self.bump();
   831→                TokenKind::RParen
   832→            }
   833→            '{' => {
   834→                self.bump();
   835→                TokenKind::LBrace
   836→            }
   837→            '}' => {
   838→                self.bump();
   839→                TokenKind::RBrace
   840→            }
   841→            '[' => {
   842→                self.bump();
   843→                TokenKind::LBracket
   844→            }
   845→            ']' => {
   846→                self.bump();
   847→                TokenKind::RBracket
   848→            }
   849→
   850→            // ---- Simple operators ----
   851→            '+' => return self.lex_plus(start, line, column),
   852→            '*' => return self.lex_star(start, line, column),
   853→            '/' => return self.lex_slash(start, line, column),
   854→            '%' => return self.lex_percent(start, line, column),
   855→            '^' => return self.lex_caret(start, line, column),
   856→            '~' => {
   857→                self.bump();
   858→                TokenKind::Tilde
   859→            }
   860→            '@' => {
   861→                self.bump();
   862→                TokenKind::Ampersat
   863→            }
   864→            '#' => {
   865→                self.bump();
   866→                TokenKind::Hash
   867→            }
   868→            '$' => {
   869→                self.bump();
   870→                TokenKind::Dollar
   871→            }
   872→            '?' => {
   873→                self.bump();
   874→                TokenKind::Question
   875→            }
   876→            ';' => {
   877→                self.bump();
   878→                TokenKind::Semicolon
   879→            }
   880→            ',' => {
   881→                self.bump();
   882→                TokenKind::Comma
   883→            }
   884→
   885→            // ---- Multi-char operators ----
   886→            '-' => return self.lex_minus(start, line, column),
   887→            '!' => return self.lex_bang(start, line, column),
   888→            '=' => return self.lex_eq(start, line, column),
   889→            '<' => return self.lex_lt(start, line, column),
   890→            '>' => return self.lex_gt(start, line, column),
   891→            '&' => return self.lex_ampersand(start, line, column),
   892→            '|' => return self.lex_pipe(start, line, column),
   893→            ':' => return self.lex_colon(start, line, column),
   894→            '.' => return self.lex_dot(start, line, column),
   895→
   896→            // ---- Literals ----
   897→            '0'..='9' => return self.lex_number(start, line, column),
   898→            '"' => return self.lex_string(start, line, column),
   899→            '\'' => return self.lex_char(start, line, column),
   900→
   901→            // Byte-string prefix `b"…"`
   902→            'b' => {
   903→                if self.peek_next(1) == Some('"') {
   904→                    self.bump(); // consume 'b'
   905→                    return self.lex_byte_string(start, line, column);
   906→                }
   907→                return self.lex_ident(start, line, column);
   908→            }
   909→
   910→            // Raw-string prefix `r"…"` or `r#"…"#`
   911→            'r' => {
   912→                let second = self.peek_next(1);
   913→                if second == Some('"') || second == Some('#') {
   914→                    self.bump(); // consume 'r'
   915→                    return self.lex_raw_string(start, line, column);
   916→                }
   917→                return self.lex_ident(start, line, column);
   918→            }
   919→
   920→            // Format-string prefix `f"…"`
   921→            'f' => {
   922→                if self.peek_next(1) == Some('"') {
   923→                    self.bump(); // consume 'f'
   924→                    return self.lex_format_string(start, line, column);
   925→                }
   926→                return self.lex_ident(start, line, column);
   927→            }
   928→
   929→            // ---- Identifier, keyword, or underscore ----
   930→            '_' => {
   931→                return self.lex_ident(start, line, column);
   932→            }
   933→            c if c.is_ascii_alphabetic() => {
   934→                return self.lex_ident(start, line, column);
   935→            }
   936→
   937→            // ---- Unknown character ----
   938→            _ => {
   939→                self.bump();
   940→                self.push_error(
   941→                    format!("unexpected character '{}'", ch),
   942→                    Span::new(start, self.offset),
   943→                );
   944→                TokenKind::Error
   945→            }
   946→        };
   947→
   948→        self.make_token(kind, start, line, column)
   949→    }
   950→

  1153→    /// Lex a numeric literal (integer or float).
  1154→    ///
  1155→    /// Handles: decimal, `0x` hex, `0b` binary, `0o` octal, float with
  1156→    /// `.` and/or exponent `e`/`E`.  Underscore separators are allowed.
  1157→    fn lex_number(&mut self, start: usize, line: usize, column: usize) -> Token {
  1158→        let first = self.chars.peek().copied();
  1159→        debug_assert!(first.is_some_and(|c| c.is_ascii_digit()));
  1160→
  1161→        if first == Some('0') {
  1162→            let second = self.peek_next(1);
  1163→            match second {
  1164→                Some('x') | Some('X') => {
  1165→                    // Hex prefix — only if followed by at least one hex digit
  1166→                    let third = self.peek_next(2);
  1167→                    if third.is_some_and(|c| c.is_ascii_hexdigit() || c == '_') {
  1168→                        self.bump(); // consume '0'
  1169→                        self.bump(); // consume 'x'/'X'
  1170→                        return self.lex_hex_digits(start, line, column);
  1171→                    }
  1172→                    // `0x` without digits -> treat `0` as integer
  1173→                    self.bump(); // consume '0'
  1174→                    return self.make_token(TokenKind::Number, start, line, column);
  1175→                }
  1176→                Some('b') | Some('B') => {
  1177→                    let third = self.peek_next(2);
  1178→                    if third.is_some_and(|c| c == '0' || c == '1' || c == '_') {
  1179→                        self.bump();
  1180→                        self.bump(); // consume '0b'
  1181→                        return self.lex_binary_digits(start, line, column);
  1182→                    }
  1183→                    self.bump(); // consume '0'
  1184→                    self.consume_decimal_digits();
  1185→                    return self.check_float_or_int(start, line, column);
  1186→                }
  1187→                Some('o') | Some('O') => {
  1188→                    let third = self.peek_next(2);
  1189→                    if third.is_some_and(|c| ('0'..='7').contains(&c) || c == '_') {
  1190→                        self.bump();
  1191→                        self.bump(); // consume '0o'
  1192→                        return self.lex_octal_digits(start, line, column);
  1193→                    }
  1194→                    self.bump(); // consume '0'
  1195→                    self.consume_decimal_digits();
  1196→                    return self.check_float_or_int(start, line, column);
  1197→                }
  1198→                Some('e') | Some('E') => {
  1199→                    // 0e10 — float with exponent
  1200→                    self.bump(); // consume '0'
  1201→                    self.lex_exponent();
  1202→                    return self.make_token(TokenKind::Float, start, line, column);
  1203→                }
  1204→                Some('.') => {
  1205→                    // 0.xxx — float
  1206→                    let third = self.peek_next(2);
  1207→                    if third.is_some_and(|c| c.is_ascii_digit()) {
  1208→                        self.bump(); // consume '0'
  1209→                        self.bump(); // consume '.'
  1210→                        self.consume_decimal_digits();
  1211→                        self.lex_exponent();
  1212→                        return self.make_token(TokenKind::Float, start, line, column);
  1213→                    }
  1214→                    // 0. -> integer 0 then dot
  1215→                    self.bump(); // consume '0'
  1216→                    return self.make_token(TokenKind::Number, start, line, column);
  1217→                }
  1218→                _ => {
  1219→                    // Plain 0, or 007 etc.
  1220→                    self.bump(); // consume '0'
  1221→                    self.consume_decimal_digits();
  1222→                    return self.check_float_or_int(start, line, column);
  1223→                }
  1224→            }
  1225→        }
  1226→
  1227→        // Non-zero decimal start
  1228→        self.consume_decimal_digits();
  1229→        self.check_float_or_int(start, line, column)
  1230→    }
  1231→
  1232→    /// After consuming the integer part, check for a float suffix (`.` or `e`).
  1233→    fn check_float_or_int(&mut self, start: usize, line: usize, column: usize) -> Token {
  1234→        if self.chars.peek() == Some(&'.') {
  1235→            let next = self.peek_next(1);
  1236→            if next.is_some_and(|c| c.is_ascii_digit()) {
  1237→                self.bump(); // consume '.'
  1238→                self.consume_decimal_digits();
  1239→                self.lex_exponent();
  1240→                return self.make_token(TokenKind::Float, start, line, column);
  1241→            }
  1242→            // `.` not followed by a digit — integer followed by Dot/DotDot
  1243→        }
  1244→        if self.chars.peek() == Some(&'e') || self.chars.peek() == Some(&'E') {
  1245→            self.lex_exponent();
  1246→            return self.make_token(TokenKind::Float, start, line, column);
  1247→        }
  1248→        self.make_token(TokenKind::Number, start, line, column)
  1249→    }

  1251→    fn consume_decimal_digits(&mut self) {
  1252→        while let Some(&c) = self.chars.peek() {
  1253→            if c.is_ascii_digit() || c == '_' {
  1254→                self.bump();
  1255→            } else {
  1256→                break;
  1257→            }
  1258→        }
  1259→    }

  1261→    fn lex_hex_digits(&mut self, start: usize, line: usize, column: usize) -> Token {
  1262→        let digits_start = self.offset;
  1263→        while let Some(&c) = self.chars.peek() {
  1264→            if c.is_ascii_hexdigit() || c == '_' {
  1265→                self.bump();
  1266→            } else {
  1267→                break;
  1268→            }
  1269→        }
  1270→        if self.offset == digits_start {
  1271→            self.push_error("empty hex digits after '0x'", Span::new(start, self.offset));

  1273→        }
  1274→        self.make_token(TokenKind::Address, start, line, column)

  1277→    fn lex_binary_digits(&mut self, start: usize, line: usize, column: usize) -> Token {

  1280→            if c == '0' || c == '1' || c == '_' {
  1281→                self.bump();
  1282→            } else {
  1283→                break;
  1284→            }
  1285→        }
  1286→        if self.offset == digits_start {

  1290→            );
  1291→            return self.make_token(TokenKind::Error, start, line, column);
  1292→        }

  1296→    fn lex_octal_digits(&mut self, start: usize, line: usize, column: usize) -> Token {

  1300→                self.bump();
  1301→            } else {
  1302→                break;

  1315→    /// Consume an optional exponent part: `[eE][+-]?[0-9_]+`.
  1316→    fn lex_exponent(&mut self) {
  1317→        if self.chars.peek() == Some(&'e') || self.chars.peek() == Some(&'E') {
  1318→            self.bump(); // consume 'e'/'E'
  1319→            if self.chars.peek() == Some(&'+') || self.chars.peek() == Some(&'-') {
  1320→                self.bump(); // consume sign
  1321→            }
  1322→            while let Some(&c) = self.chars.peek() {
  1323→                if c.is_ascii_digit() || c == '_' {
  1324→                    self.bump();
  1325→                } else {
  1326→                    break;
  1327→                }
  1328→            }
  1329→        }
  1330→    }

  1332→    /// Lex a double-quoted string literal with escape processing.
  1333→    fn lex_string(&mut self, start: usize, line: usize, column: usize) -> Token {
  1334→        self.bump(); // consume opening "
  1335→        let mut decoded = String::new();
  1336→
  1337→        loop {
  1338→            match self.chars.peek() {
  1339→                Some(&'"') => {
  1340→                    self.bump(); // consume closing "
  1341→                    break;
  1342→                }
  1343→                Some(&'\\') => {
  1344→                    self.bump(); // consume backslash
  1345→                    match self.chars.peek() {
  1346→                        Some(&'n') => {
  1347→                            self.bump();
  1348→                            decoded.push('\n');
  1349→                        }
  1350→                        Some(&'t') => {
  1351→                            self.bump();
  1352→                            decoded.push('\t');
  1353→                        }
  1354→                        Some(&'r') => {
  1355→                            self.bump();
  1356→                            decoded.push('\r');
  1357→                        }
  1358→                        Some(&'\\') => {
  1359→                            self.bump();
  1360→                            decoded.push('\\');
  1361→                        }
  1362→                        Some(&'"') => {
  1363→                            self.bump();
  1364→                            decoded.push('"');
  1365→                        }
  1366→                        Some(&'0') => {
  1367→                            self.bump();
  1368→                            decoded.push('\0');
  1369→                        }
  1370→                        Some(&'x') => {
  1371→                            self.bump(); // consume 'x'
  1372→                            let h1 = self.chars.peek().and_then(|c| c.to_digit(16));
  1373→                            if let Some(h1) = h1 {
  1374→                                self.bump();
  1375→                                let h2 = self.chars.peek().and_then(|c| c.to_digit(16));
  1376→                                if let Some(h2) = h2 {
  1377→                                    self.bump();
  1378→                                    decoded.push(char::from((h1 * 16 + h2) as u8));
  1379→                                } else {
  1380→                                    self.push_error(
  1381→                                        "incomplete hex escape in string",
  1382→                                        Span::new(self.offset.saturating_sub(2), self.offset),
  1383→                                    );
  1384→                                }
  1385→                            } else {
  1386→                                self.push_error(
  1387→                                    "incomplete hex escape in string",
  1388→                                    Span::new(self.offset.saturating_sub(1), self.offset),
  1389→                                );
  1390→                            }
  1391→                        }
  1392→                        Some(&'u') => {
  1393→                            // Unicode escape: \u{XXXX}
  1394→                            self.bump(); // consume 'u'
  1395→                            if self.chars.peek() == Some(&'{') {
  1396→                                self.bump(); // consume '{'
  1397→                                let mut hex_str = String::new();
  1398→                                while let Some(&c) = self.chars.peek() {
  1399→                                    if c.is_ascii_hexdigit() {
  1400→                                        self.bump();
  1401→                                        hex_str.push(c);
  1402→                                    } else {
  1403→                                        break;
  1404→                                    }
  1405→                                }
  1406→                                if self.chars.peek() == Some(&'}') {
  1407→                                    self.bump(); // consume '}'
  1408→                                    if let Ok(code_point) = u32::from_str_radix(&hex_str, 16) {
  1409→                                        if let Some(ch) = char::from_u32(code_point) {
  1410→                                            decoded.push(ch);
  1411→                                        } else {
  1412→                                            self.push_error(
  1413→                                                "invalid unicode code point in string",
  1414→                                                Span::new(start, self.offset),
  1415→                                            );
  1416→                                        }
  1417→                                    } else {
  1418→                                        self.push_error(
  1419→                                            "invalid unicode escape in string",
  1420→                                            Span::new(start, self.offset),
  1421→                                        );
  1422→                                    }
  1423→                                } else {
  1424→                                    self.push_error(
  1425→                                        "expected '}' after unicode escape",
  1426→                                        Span::new(start, self.offset),
  1427→                                    );
  1428→                                }
  1429→                            } else {
  1430→                                self.push_error(
  1431→                                    "expected '{' after \\u escape",
  1432→                                    Span::new(self.offset.saturating_sub(1), self.offset),
  1433→                                );
  1434→                            }
  1435→                        }
  1436→                        Some(&c) => {
  1437→                            self.bump();
  1438→                            decoded.push(c);
  1439→                        }
  1440→                        None => {
  1441→                            self.push_error(
  1442→                                "unterminated escape in string literal",
  1443→                                Span::new(start, self.offset),
  1444→                            );
  1445→                            return self.make_token_with_lexeme(
  1446→                                TokenKind::Error,
  1447→                                decoded,
  1448→                                start,
  1449→                                line,
  1450→                                column,
  1451→                            );
  1452→                        }
  1453→                    }
  1454→                }
  1455→                Some(&'\n') => {
  1456→                    self.push_error(
  1457→                        "unterminated string literal (newline)",
  1458→                        Span::new(start, self.offset),
  1459→                    );
  1460→                    return self.make_token_with_lexeme(
  1461→                        TokenKind::Error,
  1462→                        decoded,
  1463→                        start,
  1464→                        line,
  1465→                        column,
  1466→                    );
  1467→                }
  1468→                Some(&c) => {
  1469→                    self.bump();
  1470→                    decoded.push(c);
  1471→                }
  1472→                None => {
  1473→                    self.push_error("unterminated string literal", Span::new(start, self.offset));
  1474→                    return self.make_token_with_lexeme(
  1475→                        TokenKind::Error,
  1476→                        decoded,
  1477→                        start,
  1478→                        line,
  1479→                        column,
  1480→                    );
  1481→                }
  1482→            }
  1483→        }
  1484→
  1485→        self.make_token_with_lexeme(TokenKind::String, decoded, start, line, column)
  1486→    }

  1488→    /// Lex a character literal: `'c'` or `'\n'` etc.
  1489→    fn lex_char(&mut self, start: usize, line: usize, column: usize) -> Token {
  1490→        self.bump(); // consume opening '
  1491→
  1492→        let decoded = if self.chars.peek() == Some(&'\\') {
  1493→            self.bump(); // consume backslash
  1494→            match self.chars.peek() {
  1495→                Some(&'n') => {
  1496→                    self.bump();
  1497→                    Some('\n')
  1498→                }
  1499→                Some(&'t') => {
  1500→                    self.bump();
  1501→                    Some('\t')
  1502→                }
  1503→                Some(&'r') => {
  1504→                    self.bump();
  1505→                    Some('\r')
  1506→                }
  1507→                Some(&'\\') => {
  1508→                    self.bump();
  1509→                    Some('\\')
  1510→                }
  1511→                Some(&'\'') => {
  1512→                    self.bump();
  1513→                    Some('\'')
  1514→                }
  1515→                Some(&'0') => {
  1516→                    self.bump();
  1517→                    Some('\0')
  1518→                }
  1519→                Some(&'x') => {
  1520→                    self.bump(); // consume 'x'
  1521→                    let h1 = self.chars.peek().and_then(|c| c.to_digit(16));
  1522→                    if let Some(h1) = h1 {
  1523→                        self.bump();
  1524→                        let h2 = self.chars.peek().and_then(|c| c.to_digit(16));
  1525→                        if let Some(h2) = h2 {
  1526→                            self.bump();
  1527→                            Some(char::from((h1 * 16 + h2) as u8))
  1528→                        } else {
  1529→                            self.push_error(
  1530→                                "incomplete hex escape in char literal",
  1531→                                Span::new(self.offset.saturating_sub(2), self.offset),
  1532→                            );
  1533→                            None
  1534→                        }
  1535→                    } else {
  1536→                        self.push_error(
  1537→                            "incomplete hex escape in char literal",
  1538→                            Span::new(self.offset.saturating_sub(1), self.offset),
  1539→                        );
  1540→                        None
  1541→                    }
  1542→                }
  1543→                Some(&c) => {
  1544→                    self.bump();
  1545→                    Some(c)
  1546→                }
  1547→                None => {
  1548→                    self.push_error(
  1549→                        "unterminated escape in char literal",
  1550→                        Span::new(start, self.offset),
  1551→                    );
  1552→                    None
  1553→                }
  1554→            }
  1555→        } else if let Some(&c) = self.chars.peek() {
  1556→            self.bump();
  1557→            Some(c)
  1558→        } else {
  1559→            self.push_error("unterminated char literal", Span::new(start, self.offset));
  1560→            None
  1561→        };
  1562→
  1563→        // Expect closing '
  1564→        if self.chars.peek() == Some(&'\'') {
  1565→            self.bump();
  1566→        } else {
  1567→            self.push_error(
  1568→                "expected closing '\\'' in char literal",
  1569→                Span::new(start, self.offset),
  1570→            );
  1571→        }
  1572→
  1573→        let lexeme = decoded.map(|c| c.to_string()).unwrap_or_default();
  1574→        self.make_token_with_lexeme(TokenKind::Char, lexeme, start, line, column)
  1575→    }

  1577→    /// Lex a byte string literal: `b"…"`.
  1578→    fn lex_byte_string(&mut self, start: usize, line: usize, column: usize) -> Token {

  1594→                }
  1595→                '\\' => {
  1596→                    self.bump(); // consume backslash

  1617→    /// Lex a raw string literal: `r"…"`, `r#"…"#`, `r##"…"##`, etc.
  1618→    fn lex_raw_string(&mut self, start: usize, line: usize, column: usize) -> Token {
  1619→        // 'r' already consumed; count '#' delimiters
  1620→        let mut num_hashes: usize = 0;
  1621→        while self.chars.peek() == Some(&'#') {
  1622→            self.bump();
  1623→            num_hashes += 1;
  1624→        }
  1625→
  1626→        // Expect opening "
  1627→        if self.chars.peek() != Some(&'"') {
  1628→            self.push_error(
  1629→                "expected '\"' after raw string prefix",
  1630→                Span::new(start, self.offset),
  1631→            );
  1632→            return self.make_token(TokenKind::Error, start, line, column);
  1633→        }
  1634→        self.bump(); // consume opening "
  1635→
  1636→        // Read until closing `"` followed by exactly `num_hashes` `#` chars
  1637→        loop {
  1638→            match self.chars.peek() {
  1639→                Some(&'"') => {
  1640→                    self.bump(); // consume '"'
  1641→                    let mut closing_hashes: usize = 0;
  1642→                    while closing_hashes < num_hashes && self.chars.peek() == Some(&'#') {
  1643→                        self.bump();
  1644→                        closing_hashes += 1;
  1645→                    }
  1646→                    if closing_hashes == num_hashes {
  1647→                        break; // found the end
  1648→                    }
  1649→                    // Not enough closing hashes — keep reading
  1650→                }
  1651→                Some(_) => {
  1652→                    self.bump();
  1653→                }
  1654→                None => {
  1655→                    self.push_error(
  1656→                        "unterminated raw string literal",
  1657→                        Span::new(start, self.offset),
  1658→                    );
  1659→                    break;
  1660→                }
  1661→            }
  1662→        }
  1663→
  1664→        self.make_token(TokenKind::RawStr, start, line, column)
  1665→    }

  1667→    /// Lex a format string: `f"…{expr}…"`
  1668→    ///
  1669→    /// The 'f' prefix has already been consumed. The content (between the
  1670→    /// quotes) is stored in the lexeme, including `{` and `}` markers.
  1671→    /// The parser is responsible for splitting into parts.
  1672→    fn lex_format_string(&mut self, start: usize, line: usize, column: usize) -> Token {
  1673→        // 'f' already consumed; expect opening "
  1674→        if self.chars.peek() != Some(&'"') {
  1675→            self.push_error(
  1676→                "expected '\"' after format string prefix 'f'",
  1677→                Span::new(start, self.offset),
  1678→            );
  1679→            return self.make_token(TokenKind::Error, start, line, column);
  1680→        }
  1681→        self.bump(); // consume opening "
  1682→
  1683→        // Read until closing ", handling escape sequences and { } braces
  1684→        loop {
  1685→            match self.chars.peek() {
  1686→                Some(&'"') => {
  1687→                    self.bump(); // consume closing "
  1688→                    break;
  1689→                }
  1690→                Some('\\') => {
  1691→                    self.bump(); // consume backslash
  1692→                    if self.chars.peek().is_some() {
  1693→                        self.bump(); // consume escaped character
  1694→                    }
  1695→                }
  1696→                Some(&'{') => {
  1697→                    self.bump(); // consume {
  1698→                                 // If {{, it's an escaped brace, not an interpolation
  1699→                    if self.chars.peek() == Some(&'}') {
  1700→                        // Empty {} — just consume the } and continue
  1701→                        self.bump();
  1702→                    }
  1703→                    // Otherwise the } will be consumed as part of the expression
  1704→                }
  1705→                Some(&'}') => {
  1706→                    self.bump(); // consume }
  1707→                }
  1708→                Some(_) => {
  1709→                    self.bump();
  1710→                }
  1711→                None => {
  1712→                    self.push_error(
  1713→                        "unterminated format string literal",
  1714→                        Span::new(start, self.offset),
  1715→                    );
  1716→                    break;
  1717→                }
  1718→            }
  1719→        }
  1720→
  1721→        self.make_token(TokenKind::FormatStr, start, line, column)
  1722→    }

  1724→    /// Lex an identifier or keyword.
  1725→    ///
  1726→    /// Also detects Rust-style macro invocations (`println!`, `vec!`, etc.)
  1727→    /// and emits a diagnostic warning, since VUMA does not support the `!`
  1728→    /// macro syntax.
  1729→    fn lex_ident(&mut self, start: usize, line: usize, column: usize) -> Token {
  1730→        self.bump(); // consume first character
  1731→        while let Some(&c) = self.chars.peek() {
  1732→            if c.is_ascii_alphanumeric() || c == '_' {
  1733→                self.bump();
  1734→            } else {
  1735→                break;
  1736→            }
  1737→        }
  1738→        let text = &self.source[start..self.offset];
  1739→        // Special case: a standalone `_` is the Underscore token.
  1740→        if text == "_" {
  1741→            return self.make_token(TokenKind::Underscore, start, line, column);
  1742→        }
  1743→
  1744→        // Detect Rust-style macro invocation: `name!`
  1745→        // The `!` is consumed as part of the identifier lexeme so the parser
  1746→        // can report a clear LLM-mistake diagnostic.
  1747→        if self.chars.peek() == Some(&'!') {
  1748→            let macro_name = text.to_string();

  1786→    /// `/*` has already been consumed.  On unterminated comments, push an
  1787→    /// error but do not stop lexing.
  1788→    fn skip_block_comment(&mut self, comment_start: usize) {
  1789→        let mut depth: usize = 1;
  1790→        while depth > 0 {
  1791→            match self.chars.peek() {
  1792→                Some(&'/') => {
  1793→                    let second = self.peek_next(1);
  1794→                    if second == Some('*') {
  1795→                        self.bump();
  1796→                        self.bump(); // consume /*
  1797→                        depth += 1;
  1798→                    } else {
  1799→                        self.bump();
  1800→                    }
  1801→                }
  1802→                Some(&'*') => {
  1803→                    let second = self.peek_next(1);
  1804→                    if second == Some('/') {
  1805→                        self.bump();
  1806→                        self.bump(); // consume */
  1807→                        depth -= 1;
  1808→                    } else {
  1809→                        self.bump();
  1810→                    }

  1812→                Some(_) => {
