     1→//! # VUMA Language Server Protocol Implementation
     2→//!
     3→//! This module provides a full LSP server for the VUMA language, enabling
     4→//! structured access to compilation results, diagnostics, and code intelligence
     5→//! for LLMs and IDE integrations.
     6→//!
     7→//! ## Capabilities
     8→//!
     9→//! - **JSON-RPC transport** over stdin/stdout (LSP spec compatible)
    10→//! - **TextDocumentSync** — full document sync for `.vuma` files
    11→//! - **Diagnostics** — publish compilation errors with line/column info
    12→//! - **Hover** — show type information for variables and functions
    13→//! - **Go to Definition** — navigate from usage to definition
    14→//! - **Completion** — suggest function names, types, keywords
    15→//! - **Document Symbols** — list all functions, variables in a document
    16→//! - **Semantic Tokens** — highlight keywords, types, functions, variables
    17→//!
    18→//! ## Usage
    19→//!
    20→//! ```rust,no_run
    21→//! use vuma::lsp::LspServer;
    22→//!
    23→//! // Start the LSP server — reads JSON-RPC from stdin, writes to stdout
    24→//! let mut server = LspServer::new();
    25→//! server.run();
    26→//! ```

    46→/// A position in a text document (0-based line and character).
    47→#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
    48→pub struct Position {
    49→    /// Line position (0-based).
    50→    pub line: u32,
    51→    /// Character offset on the line (0-based, UTF-16 code units).
    52→    pub character: u32,
    53→}
    54→
    55→impl Position {
    56→    /// Create a new position.
    57→    pub fn new(line: u32, character: u32) -> Self {
    58→        Self { line, character }
    59→    }
    60→}

    62→/// A range in a text document expressed as (start, end) positions.
    63→#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
    64→pub struct Range {
    65→    /// Start position (inclusive).
    66→    pub start: Position,
    67→    /// End position (exclusive).
    68→    pub end: Position,
    69→}
    70→
    71→impl Range {
    72→    /// Create a new range.
    73→    pub fn new(start: Position, end: Position) -> Self {
    74→        Self { start, end }
    75→    }
    76→
    77→    /// Degenerate range at a single position.
    78→    pub fn at(line: u32, character: u32) -> Self {
    79→        let pos = Position::new(line, character);
    80→        Self { start: pos.clone(), end: pos }
    81→    }
    82→}

   439→/// Read a single JSON-RPC message from stdin.
   440→fn read_message() -> io::Result<Option<String>> {
   441→    let stdin = io::stdin();
   442→    let mut handle = stdin.lock();
   443→
   444→    // Read headers until blank line
   445→    let mut content_length: Option<usize> = None;
   446→    loop {
   447→        let mut line = String::new();
   448→        let bytes = handle.read_line(&mut line)?;
   449→        if bytes == 0 {
   450→            // EOF
   451→            return Ok(None);
   452→        }
   453→        let line = line.trim_end_matches(|c| c == '\r' || c == '\n');
   454→        if line.is_empty() {
   455→            break;
   456→        }
   457→        if let Some(rest) = line.strip_prefix("Content-Length: ") {
   458→            content_length = Some(rest.trim().parse::<usize>().map_err(|e| {
   459→                io::Error::new(io::ErrorKind::InvalidData, e)
   460→            })?);
   461→        }
   462→    }
   463→
   464→    let length = content_length.ok_or_else(|| {
   465→        io::Error::new(io::ErrorKind::InvalidData, "missing Content-Length header")
   466→    })?;
   467→
   468→    let mut buf = vec![0u8; length];
   469→    handle.read_exact(&mut buf)?;
   470→    Ok(Some(String::from_utf8(buf).map_err(|e| {
   471→        io::Error::new(io::ErrorKind::InvalidData, e)
   472→    })?))
   473→}
   474→
   475→/// Write a single JSON-RPC message to stdout.
   476→fn write_message(content: &str) -> io::Result<()> {
   477→    let stdout = io::stdout();
   478→    let mut handle = stdout.lock();
   479→    write!(handle, "Content-Length: {}\r\n\r\n{}", content.len(), content)?;
   480→    handle.flush()?;
   481→    Ok(())
   482→}

   488→/// The VUMA Language Server.
   489→///
   490→/// Maintains open documents, parses them on change, and responds to LSP
   491→/// requests using the `vuma-parser` crate for lexing/parsing and the AST
   492→/// for code intelligence.
   493→pub struct LspServer {
   494→    /// Open documents keyed by URI.
   495→    documents: HashMap<String, VumaDocument>,
   496→    /// Cached parse results per document URI.
   497→    parse_cache: HashMap<String, Option<Program>>,
   498→    /// Cached document info per URI.
   499→    info_cache: HashMap<String, DocumentInfo>,
   500→    /// Next request ID counter (for server-initiated requests).
   501→    next_id: i64,
   502→    /// Whether the server has been initialized.
   503→    initialized: bool,
   504→    /// Whether the server is shutting down.
   505→    shutting_down: bool,
   506→}
   507→
   508→impl LspServer {
   509→    /// Create a new LSP server instance.
   510→    pub fn new() -> Self {
   511→        Self {
   512→            documents: HashMap::new(),
   513→            parse_cache: HashMap::new(),
   514→            info_cache: HashMap::new(),
   515→            next_id: 1,
   516→            initialized: false,
   517→            shutting_down: false,
   518→        }
   519→    }
   520→
   521→    /// Run the main event loop, reading JSON-RPC from stdin and writing to stdout.
   522→    pub fn run(&mut self) {
   523→        loop {
   524→            match read_message() {
   525→                Ok(Some(msg)) => {
   526→                    if let Ok(value) = serde_json::from_str::<Value>(&msg) {
   527→                        self.handle_message(value);
   528→                    }
   529→                }
   530→                Ok(None) => break, // EOF
   531→                Err(e) => {
   532→                    eprintln!("LSP read error: {}", e);
   533→                    break;
   534→                }
   535→            }
   536→            if self.shutting_down {
   537→                break;
   538→            }
   539→        }
   540→    }
   541→
   542→    /// Dispatch an incoming JSON-RPC message.
   543→    fn handle_message(&mut self, value: Value) {
   544→        let id = value.get("id").cloned();
   545→        let method = value.get("method").and_then(|v| v.as_str()).unwrap_or("");
   546→
   547→        // Check if this is a request (has id) or notification (no id)
   548→        let is_request = id.is_some();
   549→
   550→        match method {
   551→            "initialize" => {
   552→                if let Some(ref id_val) = id {
   553→                    let params = value.get("params").cloned().unwrap_or(Value::Null);
   554→                    let result = self.handle_initialize(params);
   555→                    self.send_response(id_val.clone(), Ok(result));
   556→                }
   557→            }
   558→            "initialized" => {
   559→                // Notification — no response needed
   560→                self.initialized = true;
   561→            }
   562→            "shutdown" => {
   563→                if let Some(ref id_val) = id {
   564→                    self.shutting_down = true;
   565→                    self.send_response(id_val.clone(), Ok(Value::Null));
   566→                }
   567→            }
   568→            "exit" => {
   569→                self.shutting_down = true;
   570→            }
   571→            "textDocument/didOpen" => {
   572→                let params = value.get("params").cloned().unwrap_or(Value::Null);
   573→                self.handle_text_document_did_open(params);
   574→            }
   575→            "textDocument/didChange" => {
   576→                let params = value.get("params").cloned().unwrap_or(Value::Null);
   577→                self.handle_text_document_did_change(params);
   578→            }
   579→            "textDocument/didClose" => {
   580→                let params = value.get("params").cloned().unwrap_or(Value::Null);
   581→                self.handle_text_document_did_close(params);
   582→            }
   583→            "textDocument/completion" => {
   584→                if let Some(ref id_val) = id {
   585→                    let params = value.get("params").cloned().unwrap_or(Value::Null);
   586→                    let result = self.handle_text_document_completion(params);
   587→                    self.send_response(id_val.clone(), Ok(result));
   588→                }
   589→            }
   590→            "textDocument/hover" => {
   591→                if let Some(ref id_val) = id {
   592→                    let params = value.get("params").cloned().unwrap_or(Value::Null);
   593→                    let result = self.handle_text_document_hover(params);
   594→                    self.send_response(id_val.clone(), result);
   595→                }
   596→            }
   597→            "textDocument/definition" => {
   598→                if let Some(ref id_val) = id {
   599→                    let params = value.get("params").cloned().unwrap_or(Value::Null);
   600→                    let result = self.handle_text_document_definition(params);
   601→                    self.send_response(id_val.clone(), Ok(result));
   602→                }
   603→            }
   604→            "textDocument/documentSymbol" => {
   605→                if let Some(ref id_val) = id {
   606→                    let params = value.get("params").cloned().unwrap_or(Value::Null);
   607→                    let result = self.handle_text_document_symbols(params);
   608→                    self.send_response(id_val.clone(), Ok(result));
   609→                }
   610→            }
   611→            "textDocument/semanticTokens/full" => {
   612→                if let Some(ref id_val) = id {
   613→                    let params = value.get("params").cloned().unwrap_or(Value::Null);
   614→                    let result = self.handle_text_document_semantic_tokens_full(params);
   615→                    self.send_response(id_val.clone(), Ok(result));
   616→                }
   617→            }
   618→            "$/cancelRequest" => {
   619→                // Ignore cancel requests for now
   620→            }
   621→            _ => {
   622→                if is_request {
   623→                    // Respond with method not found
   624→                    if let Some(ref id_val) = id {
   625→                        self.send_response(

   641→            "jsonrpc": "2.0",
   642→            "id": id,
   643→        });
   644→        match result {
   645→            Ok(val) => {

   647→            }
   648→            Err(err) => {
   649→                response["error"] = err;
   650→            }
   651→        }

   653→        let _ = write_message(&msg);

  1229→                selection_range: range.clone(),
  1230→                detail: None,
  1231→                children: None,
  1232→            });
  1233→        }
  1234→
  1235→        // Regions
  1236→        for (name, range) in &info.regions {
  1237→            symbols.push(DocumentSymbol {
  1238→                name: name.clone(),
  1239→                kind: SymbolKind::Variable,
  1240→                range: range.clone(),
  1241→                selection_range: range.clone(),
  1242→                detail: Some("region".to_string()),
  1243→                children: None,
  1244→            });
  1245→        }
  1246→
  1247→        // Constants
  1248→        for (name, range, ty) in &info.constants {
  1249→            let detail = ty.as_deref().map(|t| format!(": {}", t));
  1250→            symbols.push(DocumentSymbol {
  1251→                name: name.clone(),
  1252→                kind: SymbolKind::Constant,
  1253→                range: range.clone(),
  1254→                selection_range: range.clone(),
  1255→                detail,
  1256→                children: None,
  1257→            });
  1258→        }
  1259→
  1260→        // Traits
  1261→        for (name, range) in &info.traits {
  1262→            symbols.push(DocumentSymbol {
  1263→                name: name.clone(),
  1264→                kind: SymbolKind::Interface,
  1265→                range: range.clone(),
  1266→                selection_range: range.clone(),
  1267→                detail: None,
  1268→                children: None,
  1269→            });
  1270→        }
  1271→
  1272→        serde_json::to_value(&symbols).unwrap_or(json!([]))
  1273→    }

  1275→    /// Handle `textDocument/semanticTokens/full` request.

  1311→    /// Re-parse a document and update caches.
  1312→    fn reparse_document(&mut self, doc: &VumaDocument) {
  1313→        let mut parser = Parser::new(&doc.text);
  1314→        let result = parser.parse_program();
  1315→
  1316→        if let Some(program) = result.value {
  1317→            let info = self.extract_document_info(&program, &doc.text);
  1318→            self.parse_cache.insert(doc.uri.clone(), Some(program));
  1319→            self.info_cache.insert(doc.uri.clone(), info);
  1320→        } else {
  1321→            // Even on parse error, try to get partial results
  1322→            // The parser may have produced partial AST with error recovery
  1323→            self.parse_cache.insert(doc.uri.clone(), None);
  1324→            self.info_cache.insert(doc.uri.clone(), DocumentInfo {
  1325→                functions: vec![],
  1326→                structs: vec![],
  1327→                enums: vec![],
  1328→                regions: vec![],
  1329→                constants: vec![],
  1330→                traits: vec![],
  1331→                variables: vec![],
  1332→                type_names: vec![],
  1333→            });
  1334→        }

  1337→    /// Compute diagnostics for a document by running the lexer and parser.
  1338→    fn compute_diagnostics(&self, doc: &VumaDocument) -> Vec<Diagnostic> {
  1339→        let mut diagnostics = Vec::new();
  1340→
  1341→        // Run the lexer to collect lexical errors
  1342→        let mut lexer = Lexer::new(&doc.text);
  1343→        let _tokens = lexer.collect_tokens();
  1344→        for error in lexer.errors() {
  1345→            let range = self.span_to_range(&doc.text, error.span);
  1346→            diagnostics.push(Diagnostic {
  1347→                range,
  1348→                severity: Some(DiagnosticSeverity::Error),
  1349→                code: None,
  1350→                source: Some("vuma-lsp".to_string()),
  1351→                message: error.message.clone(),
  1352→            });
  1353→        }
  1354→
  1355→        // Run the parser to collect parse errors
  1356→        let mut parser = Parser::new(&doc.text);
  1357→        let _ = parser.parse_program();
  1358→        for error in parser.errors() {
  1359→            let range = self.span_to_range(&doc.text, error.span);
  1360→            diagnostics.push(Diagnostic {
  1361→                range,
  1362→                severity: Some(DiagnosticSeverity::Error),
  1363→                code: None,
  1364→                source: Some("vuma-lsp".to_string()),
  1365→                message: error.message.clone(),
  1366→            });
  1367→        }
  1368→
  1369→        diagnostics
  1370→    }

  1372→    /// Convert a byte-offset Span to an LSP Range.
  1373→    fn span_to_range(&self, text: &str, span: Span) -> Range {
  1374→        let start = self.offset_to_position(text, span.start);
  1375→        let end = self.offset_to_position(text, span.end);
  1376→        Range { start, end }
  1377→    }
  1378→
  1379→    /// Convert a byte offset to an LSP Position.
  1380→    fn offset_to_position(&self, text: &str, offset: usize) -> Position {
  1381→        let mut line = 0u32;
  1382→        let mut col = 0u32;
  1383→        for (i, ch) in text.char_indices() {
  1384→            if i >= offset {
  1385→                break;
  1386→            }
  1387→            if ch == '\n' {
  1388→                line += 1;
  1389→                col = 0;
  1390→            } else {
  1391→                col += 1;
  1392→            }
  1393→        }
  1394→        Position::new(line, col)
  1395→    }

  1397→    /// Extract the word at a given position in the text.
  1398→    fn word_at_position(&self, text: &str, line: usize, character: usize) -> String {
  1399→        let lines: Vec<&str> = text.lines().collect();
  1400→        if line >= lines.len() {
  1401→            return String::new();
  1402→        }
  1403→        let line_text = lines[line];
  1404→        if character as usize >= line_text.len() {
  1405→            return String::new();
  1406→        }
  1407→
  1408→        let chars: Vec<char> = line_text.chars().collect();
  1409→        let char_idx = character as usize;
  1410→
  1411→        // Find start of word
  1412→        let mut start = char_idx;
  1413→        while start > 0 && is_ident_char(chars[start - 1]) {
  1414→            start -= 1;
  1415→        }
  1416→
  1417→        // Find end of word
  1418→        let mut end = char_idx;
  1419→        while end < chars.len() && is_ident_char(chars[end]) {
  1420→            end += 1;
  1421→        }
  1422→
  1423→        if start == end {
  1424→            return String::new();
  1425→        }
  1426→
  1427→        chars[start..end].iter().collect()
  1428→    }

  1430→    /// Extract document info from a parsed program.
  1431→    fn extract_document_info(&self, program: &Program, text: &str) -> DocumentInfo {
  1432→        let mut info = DocumentInfo {
  1433→            functions: vec![],
  1434→            structs: vec![],
  1435→            enums: vec![],
  1436→            regions: vec![],
  1437→            constants: vec![],
  1438→            traits: vec![],
  1439→            variables: vec![],
  1440→            type_names: vec![],
  1441→        };
  1442→
  1443→        for item in &program.items {
  1444→            self.extract_item_info(item, text, &mut info);
  1445→        }
  1446→
  1447→        info
  1448→    }
  1449→
  1450→    /// Extract info from a single top-level item.
  1451→    fn extract_item_info(&self, item: &Item, text: &str, info: &mut DocumentInfo) {
  1452→        match item {
  1453→            Item::FnDef(fndef) => {
  1454→                let range = self.span_to_range(text, fndef.span);
  1455→                let ret_type = fndef.return_type.as_ref().map(|t| format_type(t));
  1456→                info.functions.push((fndef.name.clone(), range, ret_type));
  1457→
  1458→                // Extract parameters as variables
  1459→                for param in &fndef.params {
  1460→                    let range = self.span_to_range(text, param.span);
  1461→                    let ty = param.ty.as_ref().map(|t| format_type(t));
  1462→                    info.variables.push((param.name.clone(), range, ty));
  1463→                }
  1464→
  1465→                // Extract let bindings from function body
  1466→                self.extract_block_info(&fndef.body, text, info);
  1467→            }
  1468→            Item::StructDef(structdef) => {
  1469→                let range = self.span_to_range(text, structdef.span);
  1470→                info.structs.push((structdef.name.clone(), range.clone()));
  1471→                info.type_names.push(structdef.name.clone());
  1472→
  1473→                // Extract fields as variables
  1474→                for field in &structdef.fields {
  1475→                    let range = self.span_to_range(text, field.span);
  1476→                    let ty = format_type(&field.ty);
  1477→                    info.variables.push((field.name.clone(), range, Some(ty)));
  1478→                }
  1479→            }
  1480→            Item::EnumDef(enumdef) => {
  1481→                let range = self.span_to_range(text, enumdef.span);
  1482→                info.enums.push((enumdef.name.clone(), range));
  1483→                info.type_names.push(enumdef.name.clone());
  1484→            }
  1485→            Item::RegionDef(regiondef) => {
  1486→                let range = self.span_to_range(text, regiondef.span);
  1487→                info.regions.push((regiondef.name.clone(), range));
  1488→            }
  1489→            Item::Const(constdef) => {
  1490→                let range = self.span_to_range(text, constdef.span);
  1491→                let ty = constdef.ty.as_ref().map(|t| format_type(t));
  1492→                info.constants.push((constdef.name.clone(), range, ty));
  1493→            }
  1494→            Item::Static(staticdef) => {
  1495→                let range = self.span_to_range(text, staticdef.span);
  1496→                let ty = staticdef.ty.as_ref().map(|t| format_type(t));
  1497→                info.constants.push((staticdef.name.clone(), range, ty));
  1498→            }
  1499→            Item::TraitDef(traitdef) => {
  1500→                let range = self.span_to_range(text, traitdef.span);
  1501→                info.traits.push((traitdef.name.clone(), range));
  1502→                info.type_names.push(traitdef.name.clone());
  1503→
  1504→                // Extract method signatures
  1505→                for method in traitdef.required_methods.iter().chain(traitdef.provided_methods.iter()) {
  1506→                    let range = self.span_to_range(text, method.span);
  1507→                    let ret_type = method.return_type.as_ref().map(|t| format_type(t));
  1508→                    info.functions.push((method.name.clone(), range, ret_type));
  1509→                }
  1510→            }
  1511→            Item::ImplBlock(implblock) => {
  1512→                for method in &implblock.methods {
  1513→                    let range = self.span_to_range(text, method.span);
  1514→                    let ret_type = method.return_type.as_ref().map(|t| format_type(t));
  1515→                    info.functions.push((method.name.clone(), range, ret_type));
  1516→
  1517→                    for param in &method.params {
  1518→                        let range = self.span_to_range(text, param.span);
  1519→                        let ty = param.ty.as_ref().map(|t| format_type(t));
  1520→                        info.variables.push((param.name.clone(), range, ty));
  1521→                    }
  1522→
  1523→                    self.extract_block_info(&method.body, text, info);
  1524→                }
  1525→            }
  1526→            Item::ModuleDef(moduledef) => {
  1527→                for sub_item in &moduledef.items {
  1528→                    self.extract_item_info(sub_item, text, info);
  1529→                }
  1530→            }
  1531→            Item::Stmt(stmt) => {
  1532→                self.extract_stmt_info(stmt, text, info);
  1533→            }
  1534→            _ => {}
  1535→        }
  1536→    }
  1537→
  1538→    /// Extract info from a block of statements.
  1539→    fn extract_block_info(&self, block: &Block, text: &str, info: &mut DocumentInfo) {
  1540→        for stmt in &block.statements {
  1541→            self.extract_stmt_info(stmt, text, info);
  1542→        }
  1543→    }
  1544→
  1545→    /// Extract info from a single statement.
  1546→    fn extract_stmt_info(&self, stmt: &Stmt, text: &str, info: &mut DocumentInfo) {
  1547→        match stmt {
  1548→            Stmt::Let(let_stmt) => {
  1549→                let range = self.span_to_range(text, let_stmt.span);
  1550→                let ty = let_stmt.ty.as_ref().map(|t| format_type(t));
  1551→                info.variables.push((let_stmt.name.clone(), range, ty));
  1552→            }
  1553→            Stmt::If(if_stmt) => {
  1554→                self.extract_block_info(&if_stmt.then_block, text, info);
  1555→                if let Some(else_block) = &if_stmt.else_block {
  1556→                    self.extract_block_info(else_block, text, info);
  1557→                }
  1558→            }
  1559→            Stmt::While(while_stmt) => {
  1560→                self.extract_block_info(&while_stmt.body, text, info);
  1561→            }
  1562→            Stmt::For(for_stmt) => {
  1563→                let range = self.span_to_range(text, for_stmt.span);
  1564→                info.variables.push((for_stmt.name.clone(), range, None));
  1565→                self.extract_block_info(&for_stmt.body, text, info);
  1566→            }
  1567→            Stmt::Loop(loop_stmt) => {
  1568→                self.extract_block_info(&loop_stmt.body, text, info);
  1569→            }
  1570→            Stmt::Match(match_stmt) => {
  1571→                // MatchArm.body is an Expr, not a Block, so no nested stmts to extract
  1572→                let _ = &match_stmt.arms;
  1573→            }
  1574→            Stmt::UnsafeBlock { body, .. } => {
  1575→                self.extract_block_info(body, text, info);
  1576→            }
  1577→            Stmt::Sync(sync_block) => {
  1578→                self.extract_block_info(&sync_block.body, text, info);
  1579→            }
  1580→            _ => {}
  1581→        }
  1582→    }

  1584→    /// Compute semantic tokens for a document by lexing it.
  1585→    fn compute_semantic_tokens(&self, text: &str) -> Vec<u32> {
  1586→        let mut lexer = Lexer::new(text);
  1587→        let tokens = lexer.collect_tokens();
  1588→
  1589→        // Semantic tokens use delta encoding: each token is
  1590→        // [delta_line, delta_start, length, token_type, token_modifiers]
  1591→        let mut data: Vec<u32> = Vec::new();
  1592→        let mut prev_line = 0u32;
  1593→        let mut prev_start = 0u32;
  1594→
  1595→        for token in &tokens {
  1596→            if token.kind == TokenKind::Eof {
  1597→                continue;
  1598→            }
  1599→
  1600→            let token_line = token.line as u32;
  1601→            let token_start = token.column as u32;
  1602→            let length = token.lexeme.len() as u32;
  1603→
  1604→            let token_type = match token.kind {
  1605→                // Keywords
  1606→                TokenKind::Fn | TokenKind::Let | TokenKind::Pub | TokenKind::Crate |
  1607→                TokenKind::Ptr | TokenKind::Region | TokenKind::Alloc | TokenKind::Allocate |
  1608→                TokenKind::Free | TokenKind::Derive | TokenKind::Cast | TokenKind::Read |
  1609→                TokenKind::Write | TokenKind::Sync | TokenKind::If | TokenKind::Else |
  1610→                TokenKind::While | TokenKind::For | TokenKind::Return | TokenKind::Struct |
  1611→                TokenKind::Enum | TokenKind::Match | TokenKind::Unsafe | TokenKind::Safe |
  1612→                TokenKind::Bd | TokenKind::Repd | TokenKind::Capd | TokenKind::Reld |
  1613→                TokenKind::Import | TokenKind::Export | TokenKind::Mod | TokenKind::Use |
  1614→                TokenKind::SelfKw | TokenKind::Super | TokenKind::Async | TokenKind::Await |
  1615→                TokenKind::Spawn | TokenKind::Lock | TokenKind::Unlock | TokenKind::Channel |
  1616→                TokenKind::Send | TokenKind::Recv | TokenKind::True | TokenKind::False |
  1617→                TokenKind::Null | TokenKind::As | TokenKind::Sizeof | TokenKind::Alignof |
  1618→                TokenKind::Break | TokenKind::Continue | TokenKind::Loop | TokenKind::Where |
  1619→                TokenKind::Impl | TokenKind::Trait | TokenKind::Type | TokenKind::Const |
  1620→                TokenKind::Static | TokenKind::Mut | TokenKind::Ref |
  1621→                TokenKind::OptionKw | TokenKind::SomeKw | TokenKind::NoneKw |
  1622→                TokenKind::ResultKw | TokenKind::OkKw | TokenKind::ErrKw => {
  1623→                    SemanticTokenType::Keyword
  1624→                }
  1625→                // Strings
  1626→                TokenKind::String | TokenKind::RawStr | TokenKind::ByteStr |
  1627→                TokenKind::FormatStr | TokenKind::Char => {
  1628→                    SemanticTokenType::String
  1629→                }
  1630→                // Numbers
  1631→                TokenKind::Number | TokenKind::Float | TokenKind::Address => {
  1632→                    SemanticTokenType::Number
  1633→                }
  1634→                // Identifiers — could be variable, function, or type
  1635→                TokenKind::Ident => {
  1636→                    // Heuristic: if it starts with uppercase, it's likely a type
  1637→                    if token.lexeme.chars().next().map_or(false, |c| c.is_uppercase()) {
  1638→                        SemanticTokenType::Type
  1639→                    } else {
  1640→                        SemanticTokenType::Variable
  1641→                    }
  1642→                }
  1643→                // Comments
  1644→                TokenKind::DocComment | TokenKind::ModuleDoc => {
  1645→                    SemanticTokenType::Comment
  1646→                }
  1647→                // Operators
  1648→                TokenKind::Plus | TokenKind::Minus | TokenKind::Star | TokenKind::Slash |
  1649→                TokenKind::Percent | TokenKind::Ampersand | TokenKind::Pipe | TokenKind::Caret |
  1650→                TokenKind::Tilde | TokenKind::Bang | TokenKind::Assign | TokenKind::EqEq |
  1651→                TokenKind::Ne | TokenKind::Lt | TokenKind::Le | TokenKind::Gt | TokenKind::Ge |
  1652→                TokenKind::Shl | TokenKind::Shr | TokenKind::AndAnd | TokenKind::OrOr |
  1653→                TokenKind::Arrow | TokenKind::FatArrow | TokenKind::PathSep | TokenKind::Colon |
  1654→                TokenKind::PlusEq | TokenKind::MinusEq | TokenKind::StarEq | TokenKind::SlashEq |
  1655→                TokenKind::PercentEq | TokenKind::AmpEq | TokenKind::PipeEq | TokenKind::CaretEq |
  1656→                TokenKind::ShlEq | TokenKind::ShrEq => {
  1657→                    SemanticTokenType::Operator
  1658→                }
  1659→                // Skip delimiters and other tokens
  1660→                _ => continue,
  1661→            };
  1662→
  1663→            let delta_line = token_line - prev_line;
  1664→            let delta_start = if delta_line == 0 {
  1665→                token_start - prev_start
  1666→            } else {
  1667→                token_start
  1668→            };
  1669→
  1670→            data.push(delta_line);
  1671→            data.push(delta_start);
  1672→            data.push(length);
  1673→            data.push(token_type.legend_index());
  1674→            data.push(0); // No modifiers
  1675→
  1676→            prev_line = token_line;
  1677→            prev_start = token_start;
  1678→        }
  1679→
  1680→        data
  1681→    }
  1682→}
