     1→//! Adaptive deployment for the Continuous Optimization Runtime.
     2→//!
     3→//! The COR can distribute compiled regions across heterogeneous execution
     4→//! targets: the local process, remote endpoints (e.g. a cloud compute
     5→//! instance), or Raspberry Pi 5 boards running either bare-metal or Linux.
     6→//!
     7→//! # Core types
     8→//!
     9→//! - [`DeploymentTarget`] — where to deploy (Local, Pi5Bare, Pi5Linux, Remote).
    10→//! - [`DeploymentPackage`] — compiled binary + metadata + debug info.
    11→//! - [`DeploymentManager`] — orchestrates deploys, hot-swaps, rollbacks, and
    12→//!   delta transfers.
    13→//!
    14→//! # Hot-swap (Pi 5)
    15→//!
    16→//! On Raspberry Pi 5 targets, the deployment manager supports *hot-swapping*:
    17→//! replacing a region's compiled code while execution continues. A shadow
    18→//! buffer is prepared, then atomically swapped in via a fence. The previous
    19→//! version is retained for potential rollback.
    20→//!
    21→//! # Version management
    22→//!
    23→//! Every successful deployment is recorded in a [`VersionLog`]. The log
    24→//! enables rollback to any previously deployed version of a region.
    25→//!
    26→//! # Delta deployment
    27→//!
    28→//! When re-deploying a region whose previous version is already present on
    29→//! the target, the manager computes a [`DeploymentDelta`] containing only the
    30→//! changed bytes, reducing transfer size.
    31→

    43→/// A target where a compiled region can be deployed for execution.
    44→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
    45→pub enum DeploymentTarget {
    46→    /// Execute the region in the local process.
    47→    Local,
    48→
    49→    /// Deploy to a Raspberry Pi 5 running in **bare-metal** mode.
    50→    ///
    51→    /// Bare-metal mode gives the runtime direct control over the BCM2712
    52→    /// hardware — no OS, no MMU translation, deterministic latency.
    53→    Pi5Bare {
    54→        /// Board identifier (for multi-board setups).
    55→        board_id: String,
    56→        /// Core to pin the region to (0–3 on Cortex-A76).
    57→        core_id: u32,
    58→    },
    59→
    60→    /// Deploy to a Raspberry Pi 5 running **Linux** (typically Raspberry Pi OS).
    61→    ///
    62→    /// Linux mode uses standard POSIX calls and the kernel scheduler.
    63→    Pi5Linux {
    64→        /// Hostname or IP address of the Pi.
    65→        host: String,
    66→        /// Core to pin the region to (0–3), or `None` for no affinity.
    67→        core_affinity: Option<u32>,
    68→    },
    69→
    70→    /// Execute the region on a remote endpoint (e.g. a cloud lambda or
    71→    /// compute server).
    72→    Remote {
    73→        /// The endpoint URI (e.g. `"https://compute.example.com/v1/exec"`).
    74→        endpoint: String,
    75→    },
    76→}

    78→impl DeploymentTarget {
    79→    /// Returns `true` if this target is any Raspberry Pi 5 variant.
    80→    pub fn is_pi5(&self) -> bool {
    81→        matches!(
    82→            self,
    83→            DeploymentTarget::Pi5Bare { .. } | DeploymentTarget::Pi5Linux { .. }
    84→        )
    85→    }
    86→
    87→    /// Returns `true` if this target supports hot-swapping of code.
    88→    ///
    89→    /// Hot-swap is currently supported on Pi 5 bare-metal (direct memory
    90→    /// control) and Pi 5 Linux (via `mmap` + `mprotect` trampoline).
    91→    pub fn supports_hot_swap(&self) -> bool {
    92→        self.is_pi5()
    93→    }
    94→
    95→    /// Returns a human-readable label for the target kind.
    96→    pub fn kind_label(&self) -> &'static str {
    97→        match self {
    98→            DeploymentTarget::Local => "Local",
    99→            DeploymentTarget::Pi5Bare { .. } => "Pi5Bare",
   100→            DeploymentTarget::Pi5Linux { .. } => "Pi5Linux",
   101→            DeploymentTarget::Remote { .. } => "Remote",
   102→        }
   103→    }
   104→}

   107→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   108→        match self {
   109→            DeploymentTarget::Local => write!(f, "Local"),
   110→            DeploymentTarget::Pi5Bare { board_id, core_id } => {

   119→            },
   120→            DeploymentTarget::Remote { endpoint } => write!(f, "Remote({})", endpoint),
   121→        }
   122→    }

   220→/// A delta (diff) between two deployment packages, used for bandwidth-
   221→/// efficient re-deployment.
   222→#[derive(Debug, Clone, Serialize, Deserialize)]
   223→pub struct DeploymentDelta {
   224→    /// The region this delta applies to.
   225→    pub region_id: RegionId,
   226→    /// Version we are diffing **from**.
   227→    pub from_version: PackageVersion,
   228→    /// Version we are diffing **to**.
   229→    pub to_version: PackageVersion,
   230→    /// Patches: (offset_in_code, replacement_bytes).
   231→    /// Offsets that are not listed are unchanged.
   232→    pub patches: Vec<(usize, Vec<u8>)>,
   233→    /// Bytes that should be **deleted** starting at the given offset.
   234→    pub deletions: Vec<(usize, usize)>,
   235→    /// Bytes that should be **inserted** at the given offset.
   236→    pub insertions: Vec<(usize, Vec<u8>)>,
   237→}
   238→
   239→impl DeploymentDelta {
   240→    /// Computes a delta between two code buffers.
   241→    ///
   242→    /// The algorithm performs a simple block-level diff:
   243→    /// - Scans both buffers in [`DELTA_BLOCK_SIZE`] chunks.
   244→    /// - Chunks that differ become patches.
   245→    /// - Trailing bytes in the new buffer become insertions.
   246→    /// - Trailing bytes in the old buffer become deletions.
   247→    pub fn compute(
   248→        region_id: RegionId,
   249→        from_version: PackageVersion,
   250→        to_version: PackageVersion,
   251→        old_code: &[u8],
   252→        new_code: &[u8],
   253→    ) -> Self {
   254→        const BLOCK: usize = 64;
   255→        let min_len = old_code.len().min(new_code.len());
   256→        let mut patches = Vec::new();
   257→
   258→        let mut offset = 0;
   259→        while offset + BLOCK <= min_len {
   260→            if old_code[offset..offset + BLOCK] != new_code[offset..offset + BLOCK] {
   261→                patches.push((offset, new_code[offset..offset + BLOCK].to_vec()));
   262→            }
   263→            offset += BLOCK;
   264→        }
   265→        // Handle the tail (< BLOCK bytes).
   266→        if offset < min_len && old_code[offset..min_len] != new_code[offset..min_len] {
   267→            patches.push((offset, new_code[offset..min_len].to_vec()));
   268→        }
   269→
   270→        let mut deletions = Vec::new();
   271→        let mut insertions = Vec::new();
   272→
   273→        if new_code.len() > old_code.len() {
   274→            insertions.push((old_code.len(), new_code[old_code.len()..].to_vec()));
   275→        } else if old_code.len() > new_code.len() {
   276→            deletions.push((new_code.len(), old_code.len() - new_code.len()));
   277→        }
   278→
   279→        DeploymentDelta {
   280→            region_id,
   281→            from_version,
   282→            to_version,
   283→            patches,
   284→            deletions,
   285→            insertions,
   286→        }
   287→    }
   288→
   289→    /// Applies this delta to an old code buffer, producing the new code
   290→    /// buffer.
   291→    pub fn apply(&self, old_code: &[u8]) -> Result<Vec<u8>, DeploymentError> {
   292→        let mut result = old_code.to_vec();
   293→
   294→        // Apply patches (same-length replacements).
   295→        for (offset, data) in &self.patches {
   296→            let end = offset.checked_add(data.len()).ok_or_else(|| {
   297→                DeploymentError::DeltaApplyFailed(
   298→                    self.region_id,
   299→                    "patch offset + length overflow".to_owned(),
   300→                )
   301→            })?;
   302→            if end > result.len() {
   303→                return Err(DeploymentError::DeltaApplyFailed(
   304→                    self.region_id,
   305→                    format!(
   306→                        "patch at {}..{} exceeds buffer len {}",
   307→                        offset,
   308→                        end,
   309→                        result.len()
   310→                    ),
   311→                ));
   312→            }
   313→            result[*offset..end].copy_from_slice(data);
   314→        }
   315→
   316→        // Apply deletions (truncate tail only — ordered back-to-front).
   317→        let mut sorted_deletions = self.deletions.clone();
   318→        sorted_deletions.sort_by_key(|b| std::cmp::Reverse(b.0));
   319→        for (offset, len) in &sorted_deletions {
   320→            let end = *offset + len;
   321→            if end > result.len() {
   322→                return Err(DeploymentError::DeltaApplyFailed(
   323→                    self.region_id,
   324→                    format!("deletion at {}..{} exceeds buffer", offset, end),
   325→                ));
   326→            }
   327→            result.drain(*offset..*offset + len);
   328→        }
   329→
   330→        // Apply insertions (at specific offsets).
   331→        // Sort by offset descending so later insertions don't shift earlier ones.
   332→        let mut sorted_insertions = self.insertions.clone();
   333→        sorted_insertions.sort_by_key(|b| std::cmp::Reverse(b.0));
   334→        for (offset, data) in &sorted_insertions {
   335→            if *offset > result.len() {
   336→                return Err(DeploymentError::DeltaApplyFailed(
   337→                    self.region_id,
   338→                    format!(
   339→                        "insertion at {} exceeds buffer len {}",
   340→                        offset,
   341→                        result.len()
   342→                    ),
   343→                ));
   344→            }
   345→            result.splice(*offset..*offset, data.iter().copied());
   346→        }
   347→
   348→        Ok(result)
   349→    }
   350→
   351→    /// Returns an estimate of the delta size in bytes (for bandwidth
   352→    /// planning).
   353→    pub fn estimated_size(&self) -> usize {
   354→        self.patches.iter().map(|(_, d)| d.len()).sum::<usize>()
   355→            + self.insertions.iter().map(|(_, d)| d.len()).sum::<usize>()
   356→    }
   357→
   358→    /// Returns true if the delta contains no changes.
   359→    pub fn is_empty(&self) -> bool {
   360→        self.patches.is_empty() && self.deletions.is_empty() && self.insertions.is_empty()
   361→    }
   362→}

   409→/// Tracks deployed versions per region, enabling rollback.
   410→#[derive(Debug, Clone, Default, Serialize, Deserialize)]
   411→pub struct VersionLog {
   412→    /// region_id → ordered list of deployed versions (oldest first).
   413→    records: HashMap<RegionId, Vec<VersionRecord>>,
   414→}
   415→
   416→impl VersionLog {
   417→    /// Creates an empty version log.
   418→    pub fn new() -> Self {
   419→        Self::default()
   420→    }
   421→
   422→    /// Records a deployment in the log.
   423→    pub fn record(&mut self, pkg: &DeploymentPackage, target: &DeploymentTarget) {
   424→        let entry = self.records.entry(pkg.metadata.region_id).or_default();
   425→        entry.push(VersionRecord {
   426→            region_id: pkg.metadata.region_id,
   427→            version: pkg.metadata.version.clone(),
   428→            target: target.clone(),
   429→            checksum: pkg.metadata.checksum,
   430→            code: pkg.code.clone(),
   431→            deployed_at_secs: epoch_secs(),
   432→        });
   433→    }

   435→    /// Returns the latest version record for a region, if any.
   436→    pub fn latest(&self, region_id: RegionId) -> Option<&VersionRecord> {
   437→        self.records.get(&region_id).and_then(|v| v.last())
   438→    }

   440→    /// Returns the version record *before* the latest (i.e. the rollback

   443→        self.records
   444→            .get(&region_id)
   445→            .and_then(|v| v.len().checked_sub(2).map(|i| &v[i]))
   446→    }
   447→
   448→    /// Returns all version records for a region (oldest first).

   450→        self.records
   451→            .get(&region_id)
   452→            .map(|v| v.as_slice())
   453→            .unwrap_or(&[])
   454→    }
   455→
   456→    /// Returns the number of versions recorded for a region.
   457→    pub fn version_count(&self, region_id: RegionId) -> usize {
   458→        self.records.get(&region_id).map(|v| v.len()).unwrap_or(0)
   459→    }
   460→}

   466→/// State machine for a hot-swap operation.
   467→#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
   468→pub enum HotSwapPhase {
   469→    /// No hot-swap in progress.
   470→    Idle,
   471→    /// Shadow buffer is being prepared with the new code.
   472→    PreparingShadow,
   473→    /// Shadow is ready; waiting for a safe point to swap.
   474→    AwaitingSafePoint,
   475→    /// Atomic pointer swap has been performed; old code is quiescing.
   476→    Swapping,
   477→    /// Hot-swap completed successfully.
   478→    Completed,
   479→    /// Hot-swap failed; rolled back to the previous code.
   480→    Failed,
   481→}
   482→
   483→impl std::fmt::Display for HotSwapPhase {
   484→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   485→        match self {
   486→            HotSwapPhase::Idle => write!(f, "Idle"),
   487→            HotSwapPhase::PreparingShadow => write!(f, "PreparingShadow"),
   488→            HotSwapPhase::AwaitingSafePoint => write!(f, "AwaitingSafePoint"),
   489→            HotSwapPhase::Swapping => write!(f, "Swapping"),
   490→            HotSwapPhase::Completed => write!(f, "Completed"),
   491→            HotSwapPhase::Failed => write!(f, "Failed"),
   492→        }
   493→    }
   494→}

   559→/// Computes deployment plans and manages live migrations.
   560→#[derive(Debug)]
   561→pub struct DeploymentPlanner {
   562→    /// Current deployment plan.
   563→    plan: DeploymentPlan,
   564→    /// Runtime configuration.
   565→    config: Config,
   566→}
   567→
   568→impl DeploymentPlanner {
   569→    /// Creates a new deployment planner with the given configuration.
   570→    pub fn new(config: Config) -> Self {
   571→        DeploymentPlanner {
   572→            plan: DeploymentPlan::new(),
   573→            config,
   574→        }
   575→    }
   576→
   577→    /// Computes a new deployment plan based on profile data and the
   578→    /// current configuration.
   579→    ///
   580→    /// # Strategy
   581→    ///
   582→    /// 1. **Hot regions** (high call count) are deployed locally for
   583→    ///    minimal latency.
   584→    /// 2. **Cold regions** are candidates for remote execution.
   585→    /// 3. On Pi 5 targets, regions are round-robin-assigned to cores.
   586→    pub fn compute_deployment_plan(
   587→        &mut self,
   588→        region_ids: &[RegionId],
   589→        profile: &ProfileData,
   590→    ) -> &DeploymentPlan {
   591→        self.plan.regions.clear();
   592→
   593→        let hot_threshold = 100;
   594→
   595→        for &region_id in region_ids {
   596→            let _is_hot = profile
   597→                .call_counts
   598→                .values()
   599→                .any(|&count| count > hot_threshold);
   600→
   601→            let target = if self.config.is_pi5_target() {
   602→                let core_id = (region_id % 4) as u32;
   603→                if self.config.enable_speculative {
   604→                    DeploymentTarget::Pi5Bare {
   605→                        board_id: "default".to_owned(),
   606→                        core_id,
   607→                    }
   608→                } else {
   609→                    DeploymentTarget::Pi5Linux {
   610→                        host: "localhost".to_owned(),
   611→                        core_affinity: Some(core_id),
   612→                    }
   613→                }
   614→            } else {
   615→                DeploymentTarget::Local
   616→            };
   617→
   618→            self.plan.regions.push((region_id, target));
   619→        }
   620→
   621→        &self.plan
   622→    }

   624→    /// Migrates a single region to a new target.
   625→    pub fn migrate_region(
   626→        &mut self,
   627→        region_id: RegionId,
   628→        new_target: DeploymentTarget,
   629→    ) -> Result<(), DeploymentError> {
   630→        let entry = self
   631→            .plan
   632→            .regions
   633→            .iter_mut()
   634→            .find(|(r, _)| *r == region_id)
   635→            .ok_or(DeploymentError::RegionNotFound(region_id))?;
   636→
   637→        let old_target = entry.1.clone();
   638→        entry.1 = new_target.clone();
   639→
   640→        log::info!(
   641→            "Migrated region {} from {} to {}",
   642→            region_id,
   643→            old_target,
   644→            new_target
   645→        );
   646→
   647→        Ok(())
   648→    }

   668→
   669→/// Manages the deployment lifecycle of compiled regions.
   670→///
   671→/// The `DeploymentManager` is the central orchestrator for:
   672→/// - Deploying packages to targets ([`deploy`](DeploymentManager::deploy)).
   673→/// - Hot-swapping running code on Pi 5 targets
   674→///   ([`hot_swap`](DeploymentManager::hot_swap)).
   675→/// - Version tracking and rollback
   676→///   ([`rollback`](DeploymentManager::rollback)).
   677→/// - Delta-based re-deployment
   678→///   ([`deploy_delta`](DeploymentManager::deploy_delta)).
   679→#[derive(Debug)]
   680→pub struct DeploymentManager {
   681→    /// Runtime configuration.
   682→    config: Config,
   683→    /// Version history for all deployed regions.
   684→    version_log: VersionLog,
   685→    /// Hot-swap state per region.
   686→    hot_swap_states: HashMap<RegionId, HotSwapState>,
   687→    /// Deployment planner for computing placement.
   688→    planner: DeploymentPlanner,
   689→}
   690→
   691→impl DeploymentManager {
   692→    /// Creates a new deployment manager with the given configuration.
   693→    pub fn new(config: Config) -> Self {
   694→        let planner = DeploymentPlanner::new(config.clone());
   695→        DeploymentManager {
   696→            config,
   697→            version_log: VersionLog::new(),
   698→            hot_swap_states: HashMap::new(),
   699→            planner,
   700→        }
   701→    }
   702→
   703→    /// Deploys a package to the specified target.
   704→    ///
   705→    /// This is the primary deployment entry point. It:
   706→    /// 1. Validates the package checksum.
   707→    /// 2. If a previous version exists on the target and the target supports
   708→    ///    hot-swap, performs a hot-swap.
   709→    /// 3. Otherwise performs a full deployment.
   710→    /// 4. Records the version in the log.
   711→    ///
   712→    /// # Errors
   713→    ///
   714→    /// Returns [`DeploymentError::ChecksumMismatch`] if the package checksum
   715→    /// is invalid, or [`DeploymentError::TargetUnreachable`] if the target
   716→    /// cannot be reached (simulated).
   717→    pub fn deploy(
   718→        &mut self,
   719→        package: &DeploymentPackage,
   720→        target: &DeploymentTarget,
   721→    ) -> Result<DeploymentResult, DeploymentError> {
   722→        // Step 1: Validate checksum.
   723→        if !package.validate_checksum() {
   724→            return Err(DeploymentError::ChecksumMismatch(
   725→                package.metadata.region_id,
   726→            ));
   727→        }
   728→
   729→        let start = Instant::now();
   730→
   731→        // Step 2: Check if we can hot-swap.
   732→        let region_id = package.metadata.region_id;
   733→        let previous = self.version_log.latest(region_id).cloned();
   734→        let mut hot_swapped = false;
   735→
   736→        if let Some(ref prev) = previous {
   737→            if target.supports_hot_swap() && prev.target == *target {
   738→                // Hot-swap path.
   739→                self.hot_swap(package, target)?;
   740→                hot_swapped = true;
   741→            }
   742→        }
   743→
   744→        // Step 3: Record version.
   745→        self.version_log.record(package, target);
   746→
   747→        let duration = start.elapsed();
   748→
   749→        Ok(DeploymentResult {
   750→            region_id,
   751→            version: package.metadata.version.clone(),
   752→            target: target.clone(),
   753→            duration,
   754→            bytes_transferred: package.metadata.code_size,
   755→            hot_swapped,
   756→            was_delta: false,
   757→        })
   758→    }
   759→
   760→    /// Performs a hot-swap of a running region on a Pi 5 target.
   761→    ///
   762→    /// The hot-swap proceeds through these phases:
   763→    /// 1. **PreparingShadow** — allocate and fill shadow buffer with new code.
   764→    /// 2. **AwaitingSafePoint** — wait until no threads are in the region.
   765→    /// 3. **Swapping** — atomically swap the code pointer.
   766→    /// 4. **Completed** — old code is retained in the version log for
   767→    ///    rollback.
   768→    ///
   769→    /// In this simulation the phases advance immediately. A production
   770→    /// implementation would coordinate with the runtime's safe-point
   771→    /// mechanism.
   772→    pub fn hot_swap(
   773→        &mut self,
   774→        package: &DeploymentPackage,
   775→        target: &DeploymentTarget,
   776→    ) -> Result<HotSwapPhase, DeploymentError> {
   777→        if !target.supports_hot_swap() {
   778→            return Err(DeploymentError::HotSwapNotSupported(
   779→                target.kind_label().to_owned(),
   780→            ));
   781→        }
   782→
   783→        let region_id = package.metadata.region_id;
   784→        let from_version = self
   785→            .version_log
   786→            .latest(region_id)
   787→            .map(|r| r.version.clone())
   788→            .unwrap_or(PackageVersion(0));
   789→
   790→        let state = self
   791→            .hot_swap_states
   792→            .entry(region_id)
   793→            .or_insert_with(|| HotSwapState::new(region_id));
   794→
   795→        // Phase 1: Prepare shadow buffer.
   796→        state.phase = HotSwapPhase::PreparingShadow;
   797→        state.from_version = from_version.clone();
   798→        state.to_version = package.metadata.version.clone();
   799→
   800→        // Phase 2: Await safe point (simulated — immediate).
   801→        state.phase = HotSwapPhase::AwaitingSafePoint;
   802→
   803→        // Phase 3: Atomic swap.
   804→        state.phase = HotSwapPhase::Swapping;
   805→
   806→        // Phase 4: Completed.
   807→        state.phase = HotSwapPhase::Completed;
   808→
   809→        log::info!(
   810→            "Hot-swapped region {} from {} to {} on {}",
   811→            region_id,
   812→            from_version,
   813→            package.metadata.version,
   814→            target
   815→        );
   816→
   817→        Ok(HotSwapPhase::Completed)
   818→    }
   819→
   820→    /// Rolls back a region to the previous deployed version.
   821→    ///
   822→    /// The previous version's code is re-deployed to the same target. The
   823→    /// rollback itself is recorded in the version log.
   824→    pub fn rollback(&mut self, region_id: RegionId) -> Result<DeploymentResult, DeploymentError> {
   825→        let prev_record = self
   826→            .version_log
   827→            .previous(region_id)
   828→            .cloned()
   829→            .ok_or(DeploymentError::NoRollbackTarget(region_id))?;
   830→
   831→        // Determine the new version number: must be strictly greater than
   832→        // the current latest.
   833→        let latest_version = self
   834→            .version_log
   835→            .latest(region_id)
   836→            .map(|r| r.version.0)
   837→            .unwrap_or(0);
   838→
   839→        // Build a package from the previous record.
   840→        let rollback_pkg = DeploymentPackage::new(
   841→            prev_record.code.clone(),
   842→            prev_record.region_id,
   843→            // Bump version: new version = latest + 1.
   844→            PackageVersion(latest_version + 1),
   845→            "rollback",
   846→        );
   847→
   848→        let target = prev_record.target.clone();
   849→        self.deploy(&rollback_pkg, &target)
   850→    }
   851→
   852→    /// Deploys only the delta (diff) between the current deployed version
   853→    /// and the new package.
   854→    ///
   855→    /// This is more bandwidth-efficient than a full deploy when the target
   856→    /// already holds a previous version. The delta is computed, sent, and
   857→    /// applied on the "remote" side (simulated locally).
   858→    pub fn deploy_delta(
   859→        &mut self,
   860→        package: &DeploymentPackage,
   861→        target: &DeploymentTarget,
   862→    ) -> Result<DeploymentResult, DeploymentError> {
   863→        let region_id = package.metadata.region_id;
   864→
   865→        let prev_code = self
   866→            .version_log
   867→            .latest(region_id)
   868→            .map(|r| r.code.clone())
   869→            .ok_or(DeploymentError::NoPreviousVersion(region_id))?;
   870→
   871→        let from_version = self
   872→            .version_log
   873→            .latest(region_id)
   874→            .map(|r| r.version.clone())
   875→            .unwrap_or(PackageVersion(0));
   876→
   877→        // Compute delta.
   878→        let delta = DeploymentDelta::compute(
   879→            region_id,
   880→            from_version,
   881→            package.metadata.version.clone(),
   882→            &prev_code,
   883→            &package.code,
   884→        );
   885→
   886→        // Simulate applying the delta on the remote side.
   887→        let reconstructed = delta
   888→            .apply(&prev_code)
   889→            .map_err(|e| DeploymentError::DeltaApplyFailed(region_id, format!("{:?}", e)))?;
   890→
   891→        // Verify the reconstructed code matches.
   892→        if reconstructed != package.code {
   893→            return Err(DeploymentError::DeltaApplyFailed(
   894→                region_id,
   895→                "reconstructed code does not match expected".to_owned(),
   896→            ));
   897→        }
   898→
   899→        let start = Instant::now();
   900→        let delta_size = delta.estimated_size();
   901→
   902→        // If the delta is not empty, we also try hot-swap on supported
   903→        // targets.
   904→        let mut hot_swapped = false;
   905→        if target.supports_hot_swap() && !delta.is_empty() {
   906→            // For a delta hot-swap, we reconstruct on the target side and
   907→            // then swap. Here we just flag it.
   908→            hot_swapped = true;
   909→        }
   910→
   911→        // Record the version (full code stored for future deltas).
   912→        self.version_log.record(package, target);
   913→
   914→        let duration = start.elapsed();
   915→
   916→        Ok(DeploymentResult {
   917→            region_id,
   918→            version: package.metadata.version.clone(),
   919→            target: target.clone(),
   920→            duration,
   921→            bytes_transferred: delta_size,
   922→            hot_swapped,
   923→            was_delta: true,
   924→        })
   925→    }
   926→

   930→    }

   957→/// Errors that can occur during deployment operations.
   958→#[derive(Debug, Clone, thiserror::Error)]
   959→pub enum DeploymentError {
   960→    /// The specified region was not found in the current plan.
   961→    #[error("Region {0} not found in deployment plan")]
   962→    RegionNotFound(RegionId),
   963→
   964→    /// Migration failed because the target is unreachable.
   965→    #[error("Target unreachable: {0}")]
   966→    TargetUnreachable(String),
   967→
   968→    /// Migration failed due to a state transfer error.
   969→    #[error("State transfer failed for region {0}: {1}")]
   970→    StateTransferFailed(RegionId, String),
   971→
   972→    /// Package checksum validation failed.
   973→    #[error("Checksum mismatch for region {0}")]
   974→    ChecksumMismatch(RegionId),
   975→
   976→    /// Hot-swap is not supported on the given target kind.
   977→    #[error("Hot-swap not supported on target kind: {0}")]
   978→    HotSwapNotSupported(String),
   979→
   980→    /// No previous version exists to roll back to.
   981→    #[error("No rollback target available for region {0}")]
   982→    NoRollbackTarget(RegionId),
   983→
   984→    /// No previous version exists on the target for delta computation.
   985→    #[error("No previous version for delta deployment of region {0}")]
   986→    NoPreviousVersion(RegionId),
   987→
   988→    /// Delta application failed.
   989→    #[error("Delta apply failed for region {0}: {1}")]
   990→    DeltaApplyFailed(RegionId, String),
   991→}

   997→/// Simple CRC32 computation (stub — a real implementation would use the
   998→/// `crc32fast` crate or hardware CRC). We use a basic table-driven CRC32
   999→/// for determinism in tests.
  1000→fn crc32(data: &[u8]) -> u32 {
  1001→    // CRC32 (IEEE 802.3 polynomial: 0xEDB88320 reflected).
  1002→    const POLY: u32 = 0xEDB88320;
  1003→    let mut table = [0u32; 256];
  1004→    for i in 0..256u32 {
  1005→        let mut crc = i;
  1006→        for _ in 0..8 {
  1007→            if crc & 1 != 0 {
  1008→                crc = (crc >> 1) ^ POLY;
  1009→            } else {
  1010→                crc >>= 1;
  1011→            }
  1012→        }
  1013→        table[i as usize] = crc;
  1014→    }
  1015→
  1016→    let mut crc: u32 = 0xFFFF_FFFF;
  1017→    for &byte in data {
  1018→        let index = ((crc ^ byte as u32) & 0xFF) as usize;
  1019→        crc = (crc >> 8) ^ table[index];
  1020→    }
  1021→    !crc
  1022→}

  1024→/// Returns the current time as seconds since the Unix epoch (best-effort).
  1025→fn epoch_secs() -> u64 {
  1026→    std::time::SystemTime::now()
  1027→        .duration_since(std::time::UNIX_EPOCH)
  1028→        .map(|d| d.as_secs())
  1029→        .unwrap_or(0)
  1030→}
