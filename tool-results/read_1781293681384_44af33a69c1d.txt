     1→//! # Multi-Backend Trait Architecture
     2→//!
     3→//! Defines the `TargetInfo` and `Backend` traits that allow VUMA to target
     4→//! multiple instruction set architectures. Each ISA implements these traits
     5→//! to provide target-specific information and code generation.
     6→

    17→// ---------------------------------------------------------------------------
    18→// Endianness
    19→// ---------------------------------------------------------------------------
    20→
    21→/// Byte order of the target architecture.
    22→#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, serde::Serialize, serde::Deserialize)]
    23→pub enum Endianness {
    24→    /// Least-significant byte first (AArch64, RISC-V, x86_64, LoongArch).
    25→    Little,
    26→    /// Most-significant byte first (MIPS64 big-endian, PPC64 big-endian).
    27→    Big,
    28→    /// Bi-endian — the ISA supports both but the default is big-endian (PPC64).
    29→    Bi,
    30→}
    31→
    32→// ---------------------------------------------------------------------------
    33→// OutputFormat
    34→// ---------------------------------------------------------------------------
    35→
    36→/// The output binary format produced by the backend.
    37→#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, serde::Serialize, serde::Deserialize)]
    38→pub enum OutputFormat {
    39→    /// 64-bit ELF (AArch64, RISC-V64, x86_64, LoongArch64, MIPS64, PPC64).
    40→    Elf64,
    41→    /// 32-bit ELF (ARM32).
    42→    Elf32,
    43→    /// WebAssembly binary module (.wasm).
    44→    WasmBinary,
    45→    /// Raw binary blob (bare-metal, no headers).
    46→    RawBinary,
    47→}

    49→// ---------------------------------------------------------------------------
    50→// PhysicalReg
    51→// ---------------------------------------------------------------------------
    52→
    53→/// A physical register identified by class and index.
    54→#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, serde::Serialize, serde::Deserialize)]
    55→pub struct PhysicalReg {
    56→    /// Register class.
    57→    pub class: RegClass,
    58→    /// Register index within its class (0-based).
    59→    pub index: u32,
    60→}
    61→
    62→impl PhysicalReg {
    63→    /// Creates a new physical register identifier with the given class and index.
    64→    pub fn new(class: RegClass, index: u32) -> Self {
    65→        Self { class, index }
    66→    }
    67→}
    68→
    69→impl fmt::Display for PhysicalReg {
    70→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
    71→        write!(f, "{:?}:{}", self.class, self.index)
    72→    }
    73→}

    75→// ---------------------------------------------------------------------------
    76→// RegClass
    77→// ---------------------------------------------------------------------------
    78→
    79→/// Classification of physical registers.
    80→#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, serde::Serialize, serde::Deserialize)]
    81→pub enum RegClass {
    82→    /// General-purpose integer registers (X0-X30 on ARM64, RAX-R15 on x86_64, etc.)
    83→    Gpr,
    84→    /// SIMD / floating-point registers (V0-V31 on ARM64, XMM0-XMM15 on x86_64, etc.)
    85→    SimdFp,
    86→    /// Condition register fields (PPC64 CR0-CR7).
    87→    Condition,
    88→    /// Special-purpose register (TOC pointer on PPC64, etc.)
    89→    Special,
    90→}

   109→// ---------------------------------------------------------------------------
   110→// RelocationEntry
   111→// ---------------------------------------------------------------------------
   112→
   113→/// A relocation entry for patching encoded code at link time.
   114→///
   115→/// Each entry records a byte offset within the function's encoded output where
   116→/// a symbolic reference must be resolved, the name of the target symbol, and
   117→/// the ISA-specific relocation type (e.g., `"R_X86_64_PLT32"`, `"R_X86_64_64"`).
   118→#[derive(Clone, Debug, serde::Serialize, serde::Deserialize)]
   119→pub struct RelocationEntry {
   120→    /// Byte offset within the function's encoded code where the relocation applies.
   121→    pub offset: u64,
   122→    /// Name of the target symbol.
   123→    pub symbol: String,
   124→    /// Relocation type (ISA-specific, e.g., "R_X86_64_PLT32", "R_X86_64_64").
   125→    pub reloc_type: String,
   126→}

   141→    /// Encoded bytes (filled in during encoding phase).
   142→    pub encoded: Vec<u8>,
   143→}

   200→// BackendError
   201→// ---------------------------------------------------------------------------
   202→
   203→/// Error type for backend operations.
   204→#[derive(Debug, Clone, thiserror::Error)]
   205→pub enum BackendError {
   206→    /// The requested feature is not supported by this ISA.
   207→    #[error("[{isa}] unsupported feature: {feature}")]
   208→    UnsupportedFeature {
   209→        /// ISA identifier (e.g., "aarch64", "x86_64").
   210→        isa: &'static str,
   211→        /// Description of the unsupported feature.
   212→        feature: String,
   213→    },
   214→
   215→    /// Register allocation failed.
   216→    #[error("[{isa}] register allocation failed: {reason}")]
   217→    RegisterAllocFailed {
   218→        /// ISA identifier.
   219→        isa: &'static str,
   220→        /// Reason for the allocation failure.
   221→        reason: String,
   222→    },
   223→
   224→    /// Instruction encoding failed.
   225→    #[error("[{isa}] encoding error: {reason}")]
   226→    EncodingError {
   227→        /// ISA identifier.
   228→        isa: &'static str,
   229→        /// Reason for the encoding failure.
   230→        reason: String,
   231→    },
   232→
   233→    /// Invalid instruction for this target.
   234→    #[error("[{isa}] invalid instruction: {details}")]
   235→    InvalidInstruction {
   236→        /// ISA identifier.
   237→        isa: &'static str,
   238→        /// Details about why the instruction is invalid.
   239→        details: String,
   240→    },
   241→
   242→    /// ELF / binary emission error.
   243→    #[error("[{isa}] emission error: {reason}")]
   244→    EmissionError {
   245→        /// ISA identifier.
   246→        isa: &'static str,
   247→        /// Reason for the emission failure.
   248→        reason: String,
   249→    },
   250→
   251→    /// The target cannot handle this type.
   252→    #[error("[{isa}] unsupported type: {ty}")]
   253→    UnsupportedType {
   254→        /// ISA identifier.
   255→        isa: &'static str,
   256→        /// The unsupported type name.
   257→        ty: String,
   258→    },
   259→
   260→    /// Generic backend error.
   261→    #[error("[{isa}] {message}")]
   262→    Other {
   263→        /// ISA identifier.
   264→        isa: &'static str,
   265→        /// Error message.
   266→        message: String,
   267→    },
   268→}

   270→// ---------------------------------------------------------------------------
   271→// TargetInfo trait
   272→// ---------------------------------------------------------------------------
   273→
   274→/// Target-specific information needed during code generation.
   275→///
   276→/// This trait provides a data-driven interface for query target properties.
   277→/// It must be implementable for ALL supported ISAs, including fundamentally
   278→/// different architectures like Wasm (stack machine, no registers) and
   279→/// MIPS (branch delay slots).
   280→///
   281→/// # Object Safety
   282→///
   283→/// This trait is object-safe: all methods take `&self` and return only
   284→/// owned types or references with `'static` lifetime.
   285→pub trait TargetInfo: Send + Sync + 'static {
   286→    // === Identity ===
   287→
   288→    /// ISA name in lowercase (e.g., "aarch64", "riscv64", "wasm32").
   289→    fn isa_name(&self) -> &'static str;
   290→
   291→    /// LLVM-style target triple (e.g., "aarch64-unknown-linux-gnu").
   292→    fn target_triple(&self) -> &'static str;
   293→
   294→    /// ELF `e_machine` value.  Returns 0 for non-ELF targets (Wasm).
   295→    fn elf_machine_type(&self) -> u16;
   296→
   297→    /// Default base address for the .text section.
   298→    fn default_base_address(&self) -> u64;
   299→
   300→    // === Data model ===
   301→
   302→    /// Pointer width in bytes (4 for 32-bit, 8 for 64-bit).
   303→    fn pointer_width(&self) -> usize;
   304→
   305→    /// Size in bytes of `ty` on this target.
   306→    fn size_of(&self, ty: &IRType) -> usize;
   307→
   308→    /// Natural alignment in bytes of `ty` on this target.
   309→    fn alignment_of(&self, ty: &IRType) -> usize;
   310→
   311→    /// Byte order of this target.
   312→    fn endianness(&self) -> Endianness;
   313→
   314→    // === Register architecture ===
   315→
   316→    /// Whether this target has registers at all.  `false` for Wasm (stack machine).
   317→    fn has_registers(&self) -> bool;
   318→
   319→    /// Number of general-purpose registers.  0 for Wasm.
   320→    fn num_gp_regs(&self) -> usize;
   321→
   322→    /// Number of SIMD/FP registers.  0 for Wasm.
   323→    fn num_simd_fp_regs(&self) -> usize;
   324→
   325→    /// Whether the ISA has a hardwired-zero register (RISC-V x0, LoongArch r0).
   326→    fn has_hardwired_zero(&self) -> bool;
   327→
   328→    /// Whether the ISA uses a link register (ARM, RISC-V, MIPS, PPC) rather than
   329→    /// pushing the return address on the stack (x86_64).
   330→    fn has_link_register(&self) -> bool;
   331→
   332→    /// Whether branches have delay slots (MIPS only).
   333→    fn has_branch_delay_slots(&self) -> bool;
   334→
   335→    /// Whether this ISA uses a TOC (Table of Contents) pointer (PPC64 r2).
   336→    fn has_toc_pointer(&self) -> bool;
   337→
   338→    /// Whether this ISA has dedicated condition register fields (PPC64 CR0-CR7).
   339→    fn has_condition_registers(&self) -> bool;
   340→
   341→    // === Calling convention ===
   342→
   343→    /// Name of the calling convention (e.g., "aapcs64", "lp64d", "systemv").
   344→    fn calling_convention_name(&self) -> &'static str;
   345→
   346→    /// Number of integer argument registers.
   347→    fn num_int_arg_regs(&self) -> usize;
   348→
   349→    /// Number of FP/SIMD argument registers.
   350→    fn num_fp_arg_regs(&self) -> usize;
   351→
   352→    /// Required stack alignment in bytes.
   353→    fn stack_alignment(&self) -> usize;
   354→
   355→    // === Instruction encoding ===
   356→
   357→    /// Alignment requirement for instructions in bytes (4 for fixed-width RISCs,
   358→    /// 1 for x86_64 and Wasm).
   359→    fn instruction_alignment(&self) -> usize;
   360→
   361→    /// Minimum and maximum instruction width in bytes.
   362→    /// - Fixed-width 32-bit ISAs: (4, 4)
   363→    /// - x86_64: (1, 15)
   364→    /// - RISC-V with RVC: (2, 4)
   365→    /// - Wasm: (1, ~) but typically (1, 15)
   366→    fn instruction_width_range(&self) -> (usize, usize);
   367→
   368→    // === Output format ===
   369→
   370→    /// Binary format produced by this backend.
   371→    fn output_format(&self) -> OutputFormat;
   372→}
   373→
   374→// ---------------------------------------------------------------------------
   375→// Backend trait
   376→// ---------------------------------------------------------------------------
   377→
   378→/// A code generation backend for a specific target architecture.
   379→///
   380→/// Each supported ISA implements this trait, providing register allocation,
   381→/// instruction encoding, program emission, and disassembly.
   382→///
   383→/// # Object Safety
   384→///
   385→/// This trait is object-safe.
   386→pub trait Backend: Send + Sync + 'static {
   387→    /// Returns a reference to this backend's target info.
   388→    fn target_info(&self) -> &dyn TargetInfo;
   389→
   390→    /// Allocate physical registers for an IR function.
   391→    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError>;
   392→
   393→    /// Encode a single allocated function into machine code bytes.
   394→    fn encode_function(&self, func: &AllocatedFunction) -> Result<Vec<u8>, BackendError>;
   395→
   396→    /// Encode an entire allocated program into its final binary form
   397→    /// (ELF, .wasm, raw binary, etc.).
   398→    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError>;
   399→
   400→    /// Returns the bytes for a minimal return stub (e.g., `RET` on ARM64,
   401→    /// `mov eax, 0; ret` on x86_64, `end` on Wasm).
   402→    fn return_stub(&self) -> Vec<u8>;
   403→
   404→    /// Returns a trampoline that jumps to `entry_addr`.
   405→    fn trampoline(&self, entry_addr: u64) -> Vec<u8>;
   406→
   407→    /// Disassemble `bytes` starting at virtual address `addr`.
   408→    fn disassemble(&self, bytes: &[u8], addr: u64) -> Vec<String>;
   409→
   410→    /// Human-readable name of this backend.
   411→    fn name(&self) -> &'static str;
   412→}

   415→// BackendKind
   416→// ---------------------------------------------------------------------------
   417→
   418→/// Enumeration of all supported backend architectures.
   419→#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, serde::Serialize, serde::Deserialize)]
   420→pub enum BackendKind {
   421→    /// ARM 64-bit (AArch64).
   422→    AArch64,
   423→    /// RISC-V 64-bit.
   424→    RiscV64,
   425→    /// WebAssembly 32-bit.
   426→    Wasm32,
   427→    /// LoongArch 64-bit.
   428→    LoongArch64,
   429→    /// x86-64.
   430→    X86_64,
   431→    /// ARM 32-bit.
   432→    Arm32,
   433→    /// MIPS 64-bit.
   434→    Mips64,
   435→    /// PowerPC 64-bit.
   436→    PowerPC64,
   437→}
   438→
   439→impl BackendKind {
   440→    /// Returns the ISA name string for this backend kind.
   441→    pub fn isa_name(&self) -> &'static str {
   442→        match self {
   443→            BackendKind::AArch64 => "aarch64",
   444→            BackendKind::RiscV64 => "riscv64",
   445→            BackendKind::Wasm32 => "wasm32",
   446→            BackendKind::LoongArch64 => "loongarch64",
   447→            BackendKind::X86_64 => "x86_64",
   448→            BackendKind::Arm32 => "arm32",
   449→            BackendKind::Mips64 => "mips64",
   450→            BackendKind::PowerPC64 => "ppc64",
   451→        }
   452→    }
   453→}
   454→
   455→impl fmt::Display for BackendKind {
   456→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   457→        f.write_str(self.isa_name())
   458→    }
   459→}

  1113→// ---------------------------------------------------------------------------
  1114→// AArch64 Mnemonic Decoder
  1115→// ---------------------------------------------------------------------------
  1116→
  1117→/// Decode a 32-bit AArch64 instruction word into a human-readable mnemonic.
  1118→///
  1119→/// Covers the most common AArch64 instructions: ADD, SUB, MOV, LDR, STR, B,
  1120→/// BL, RET, CMP, B.cond, STP, LDP, NOP, MUL, SDIV, UDIV, AND, ORR, EOR,
  1121→/// plus several additional frequently-encountered encodings.
  1122→fn decode_aarch64(word: u32) -> String {
  1123→    let rd = word & 0x1F;
  1124→    let rn = (word >> 5) & 0x1F;
  1125→    let rt = rd; // alias for load/store destination
  1126→    let rm = (word >> 16) & 0x1F;
  1127→    let imm12 = (word >> 10) & 0xFFF;
  1128→    let cond = word & 0xF;
  1129→
  1130→    // NOP: d503201f
  1131→    if word == 0xD503201F {
  1132→        return "nop".to_string();
  1133→    }
  1134→
  1135→    // RET: d65f03c0
  1136→    if word == 0xD65F03C0 {
  1137→        return "ret".to_string();
  1138→    }
  1139→
  1140→    let _top8 = word >> 24;
  1141→    let _top10 = word >> 22;
  1142→
  1143→    // --- ADD/SUB (immediate): 100100xx ...
  1144→    if (word >> 23) & 0x1FF == 0b1_0010_0010 {
  1145→        // ADD Xd, Xn, #imm12
  1146→        return format!("add x{}, x{}, #{}", rd, rn, imm12);
  1147→    }
  1148→    if (word >> 23) & 0x1FF == 0b1_1010_0010 {
  1149→        // SUB Xd, Xn, #imm12
  1150→        return format!("sub x{}, x{}, #{}", rd, rn, imm12);
  1151→    }
  1152→
  1153→    // --- ADD (shifted register): 1_00_0101_1_xxx ...
  1154→    if (word >> 24) & 0xFF == 0b1000_1011 {
  1155→        // ADD Xd, Xn, Xm
  1156→        return format!("add x{}, x{}, x{}", rd, rn, rm);
  1157→    }
  1158→
  1159→    // --- SUB (shifted register): 1_00_0101_1_xxx ... with S=1 at bit30
  1160→    if (word >> 24) & 0xFF == 0b1101_0110 {
  1161→        // SUB Xd, Xn, Xm (bit 30 set = sub)
  1162→        return format!("sub x{}, x{}, x{}", rd, rn, rm);
  1163→    }
  1164→
  1165→    // --- AND (shifted register): 1_00_0101_0_00_xxx
  1166→    if (word >> 24) & 0xFE == 0b1000_1010 {
  1167→        // Check bit 21-15: opcode[31:21] = 10001010_000
  1168→        if (word >> 21) & 0x7FF == 0b10001010000 {
  1169→            return format!("and x{}, x{}, x{}", rd, rn, rm);
  1170→        }
  1171→    }
  1172→
  1173→    // --- ORR (shifted register): 1_01_0101_0_00_xxx
  1174→    if (word >> 21) & 0x7FF == 0b10101010000 {
  1175→        return format!("orr x{}, x{}, x{}", rd, rn, rm);
  1176→    }
  1177→
  1178→    // --- EOR (shifted register): 1_10_0101_0_00_xxx
  1179→    if (word >> 21) & 0x7FF == 0b11001010000 {
  1180→        return format!("eor x{}, x{}, x{}", rd, rn, rm);
  1181→    }
  1182→
  1183→    // --- MOV (register): alias for ORR Xd, XZR, Xm
  1184→    // ORR Xd, XZR, Xm: 10101010_000_rm_000000_xzr_rd
  1185→    // More general: ORR with Rn=XZR(31)
  1186→    if (word >> 21) & 0x7FF == 0b10101010000 && rn == 31 {
  1187→        return format!("mov x{}, x{}", rd, rm);
  1188→    }
  1189→
  1190→    // --- MUL: MADD Xd, Xn, Xm, XZR
  1191→    // Encoding: 1_00_1101_1_000_Rm_0_01111_Rn_Rd
  1192→    if (word >> 21) & 0x7FF == 0b10011011000 && ((word >> 10) & 0x1F) == 0b01111 {
  1193→        return format!("mul x{}, x{}, x{}", rd, rn, rm);
  1194→    }
  1195→
  1196→    // --- SDIV: 1_00_1101_1_0100_00000_00001_Rn_Rd  (actually 1_00_1101_0100_Rm_00001_Rn_Rd)
  1197→    if (word >> 21) & 0x7FF == 0b10011010100 && (word >> 10) & 0x1F == 0b00001 {
  1198→        return format!("sdiv x{}, x{}, x{}", rd, rn, rm);
  1199→    }
  1200→
  1201→    // --- UDIV: 1_00_1101_0000_Rm_00001_Rn_Rd
  1202→    if (word >> 21) & 0x7FF == 0b10011010000 && (word >> 10) & 0x1F == 0b00001 {
  1203→        return format!("udiv x{}, x{}, x{}", rd, rn, rm);
  1204→    }
  1205→
  1206→    // --- CMP (immediate): SUBS XZR, Xn, #imm12
  1207→    // 11100001_00_xxx_xxx_xxx_xxx_xxx_11111_xxx
  1208→    if (word >> 23) & 0x1FF == 0b1_1110_0010 && rd == 31 {
  1209→        return format!("cmp x{}, #{}", rn, imm12);
  1210→    }
  1211→
  1212→    // --- CMP (register): SUBS XZR, Xn, Xm
  1213→    if (word >> 21) & 0x7FF == 0b11101011000 && rd == 31 {
  1214→        return format!("cmp x{}, x{}", rn, rm);
  1215→    }
  1216→
  1217→    // --- B.cond: 0101010x xxxxxxxxxx xxxxxx cond
  1218→    if (word >> 24) & 0xFF == 0x54 {
  1219→        let cond_name = match cond {
  1220→            0 => "eq",
  1221→            1 => "ne",
  1222→            2 => "cs",
  1223→            3 => "cc",
  1224→            4 => "mi",
  1225→            5 => "pl",
  1226→            6 => "vs",
  1227→            7 => "vc",
  1228→            8 => "hi",
  1229→            9 => "ls",
  1230→            10 => "ge",
  1231→            11 => "lt",
  1232→            12 => "gt",
  1233→            13 => "le",
  1234→            14 => "al",
  1235→            _ => "??",
  1236→        };
  1237→        let imm19 = (word >> 5) & 0x7FFFF;
  1238→        let offset = ((imm19 as i32) << 13) >> 11; // sign-extend and *4
  1239→        return format!("b.{} {:+}", cond_name, offset);
  1240→    }
  1241→
  1242→    // --- B (unconditional): 000101xx xxxxxxxxxxxxxxxxxxxx
  1243→    if (word >> 26) & 0x3F == 0b000101 {
  1244→        let imm26 = word & 0x3FFFFFF;
  1245→        let offset = ((imm26 as i32) << 6) >> 4; // sign-extend and *4
  1246→        return format!("b {:+}", offset);
  1247→    }
  1248→
  1249→    // --- BL: 100101xx xxxxxxxxxxxxxxxxxxxx
  1250→    if (word >> 26) & 0x3F == 0b100101 {
  1251→        let imm26 = word & 0x3FFFFFF;
  1252→        let offset = ((imm26 as i32) << 6) >> 4;
  1253→        return format!("bl {:+}", offset);
  1254→    }
  1255→
  1256→    // --- LDR (unsigned offset): 11111001_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1257→    if (word >> 22) & 0x3FF == 0b1111100101 {
  1258→        let imm12_raw = (word >> 10) & 0xFFF;
  1259→        let offset = imm12_raw * 8; // scale by 8 for 64-bit
  1260→        return format!("ldr x{}, [x{}, #{}]", rt, rn, offset);
  1261→    }
  1262→
  1263→    // --- STR (unsigned offset): 11111000_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1264→    if (word >> 22) & 0x3FF == 0b1111100001 {
  1265→        let imm12_raw = (word >> 10) & 0xFFF;
  1266→        let offset = imm12_raw * 8;
  1267→        return format!("str x{}, [x{}, #{}]", rt, rn, offset);
  1268→    }
  1269→
  1270→    // --- LDP (signed offset, 64-bit): 101_0100_110_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1271→    if (word >> 22) & 0x3FF == 0b1010100110 {
  1272→        let rt2 = (word >> 10) & 0x1F;
  1273→        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1274→        let offset = imm7 * 8;
  1275→        return format!("ldp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1276→    }
  1277→
  1278→    // --- STP (signed offset, 64-bit): 101_0100_010_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1279→    if (word >> 22) & 0x3FF == 0b1010100010 {
  1280→        let rt2 = (word >> 10) & 0x1F;
  1281→        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1282→        let offset = imm7 * 8;
  1283→        return format!("stp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1284→    }
  1285→
  1286→    // --- MOVZ: 110100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1287→    if (word >> 23) & 0x1FF == 0b110100101 {
  1288→        let hw = (word >> 21) & 0x3;
  1289→        let imm16 = (word >> 5) & 0xFFFF;
  1290→        return format!("movz x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1291→    }
  1292→
  1293→    // --- MOVK: 111100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1294→    if (word >> 23) & 0x1FF == 0b111100101 {
  1295→        let hw = (word >> 21) & 0x3;
  1296→        let imm16 = (word >> 5) & 0xFFFF;
  1297→        return format!("movk x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1298→    }
  1299→
  1300→    format!(".word {:08x}", word)
  1301→}

  1330→/// Compute the stack frame size for an IR function.
  1331→///
  1332→/// Replicates the private `compute_frame_size` function in `emit.rs`:
  1333→/// sums `Alloc` instruction sizes, adds 16 bytes for the FP/LR save pair,
  1334→/// and rounds up to 16-byte alignment.
  1335→fn aarch64_compute_frame_size(func: &IRFunction) -> usize {
  1336→    let mut total: u32 = 16; // FP/LR save pair
  1337→    for block in &func.blocks {
  1338→        for instr in &block.instructions {
  1339→            if let IRInstr::Alloc { size, .. } = instr {
  1340→                let aligned = (*size).div_ceil(16) * 16;
  1341→                total += aligned;
  1342→            }
  1343→        }
  1344→    }
  1345→    // Round up to 16-byte alignment
  1346→    total = (total + 15) & !15;
  1347→    total as usize
  1348→}
  1349→
  1350→/// Build a minimal ELF64 binary for AArch64 from raw code bytes.
  1351→///
  1352→/// Produces a static executable with a single LOAD segment containing the
  1353→/// `.text` section.  This is sufficient for programs that have already been
  1354→/// emitted through the `allocate_registers` / `encode_function` pipeline.
  1355→fn build_minimal_aarch64_elf(code: &[u8], base_addr: u64) -> Vec<u8> {
  1356→    // Layout: ELF header (64) | 2 program headers (2×56) | code | [padding] | stack region
  1357→    //
  1358→    // IMPORTANT: Segments must not share a page (4KB = 0x1000), otherwise
  1359→    // the Linux ELF loader will give the second segment's permissions to
  1360→    // the shared page, potentially removing execute permission from code.
  1361→    let elf_header_size: u64 = 64;
  1362→    let phdr_size: u64 = 56;
  1363→    let num_phdrs: u64 = 2;
  1364→    let text_offset = elf_header_size + phdr_size * num_phdrs;
  1365→    let text_size = code.len() as u64;
  1366→    let page_size: u64 = 0x1000;
  1367→    // Round text up to page boundary so data segment starts on a new page
  1368→    let text_aligned = (text_size + page_size - 1) & !(page_size - 1);
  1369→    let entry_point = base_addr + text_offset;
  1370→
  1371→    // Writable data/stack segment starts on the next page after text
  1372→    let data_offset = text_offset + text_aligned;
  1373→    let data_size: u64 = 0; // no initialized data
  1374→    let data_vaddr = base_addr + data_offset;
  1375→    let stack_size: u64 = 0x100000; // 1 MB for stack/BSS
  1376→    let data_memsz = data_size + stack_size;
  1377→
  1378→    let mut elf = Vec::with_capacity(data_offset as usize + code.len());
  1379→
  1380→    // --- e_ident ---
  1381→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1382→    elf.push(2); // ELFCLASS64
  1383→    elf.push(1); // ELFDATA2LSB
  1384→    elf.push(1); // EV_CURRENT
  1385→    elf.push(3); // ELFOSABI_LINUX
  1386→    elf.push(0); // padding
  1387→    elf.extend_from_slice(&[0u8; 7]); // padding
  1388→
  1389→    // --- ELF header fields ---
  1390→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1391→    elf.extend_from_slice(&183u16.to_le_bytes()); // e_machine = EM_AARCH64
  1392→    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1393→    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1394→    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1395→    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1396→    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1397→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1398→    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1399→    elf.extend_from_slice(&(num_phdrs as u16).to_le_bytes()); // e_phnum
  1400→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1401→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1402→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1403→
  1404→    // --- Program Header 1: LOAD (PF_R | PF_X) for text ---
  1405→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1406→    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1407→    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1408→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1409→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1410→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1411→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1412→    elf.extend_from_slice(&page_size.to_le_bytes()); // p_align
  1413→
  1414→    // --- Program Header 2: LOAD (PF_R | PF_W) for data/stack ---
  1415→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1416→    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1417→    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1418→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1419→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1420→    elf.extend_from_slice(&data_size.to_le_bytes()); // p_filesz
  1421→    elf.extend_from_slice(&data_memsz.to_le_bytes()); // p_memsz (includes stack)
  1422→    elf.extend_from_slice(&page_size.to_le_bytes()); // p_align
  1423→
  1424→    // --- Code section ---
  1425→    elf.extend_from_slice(code);
  1426→    // Pad to text_aligned (page boundary)
  1427→    let padding = text_aligned - text_size;
  1428→    elf.extend_from_slice(&vec![0u8; padding as usize]);
  1429→
  1430→    elf
  1431→}

  1433→impl Backend for AArch64Backend {
  1434→    fn target_info(&self) -> &dyn TargetInfo {
  1435→        &self.target_info
  1436→    }
  1437→
  1438→    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError> {
  1439→        // Use the existing Emitter to emit the function, which internally
  1440→        // performs register allocation and instruction encoding.
  1441→        let mut emitter = crate::emit::Emitter::new();
  1442→        let code = emitter
  1443→            .emit_function(func)
  1444→            .map_err(|e| BackendError::RegisterAllocFailed {
  1445→                isa: "aarch64",
  1446→                reason: e.to_string(),
  1447→            })?;
  1448→
  1449→        let func_name = func.name.clone();
  1450→        let frame_size = aarch64_compute_frame_size(func);
  1451→
  1452→        // Convert each 32-bit ARM64 instruction word into an AllocatedInstruction
  1453→        // with its little-endian encoded bytes.
  1454→        let instructions: Vec<AllocatedInstruction> = code
  1455→            .iter()
  1456→            .enumerate()
  1457→            .map(|(i, &word)| AllocatedInstruction {
  1458→                opcode: format!("arm64_{}", i),
  1459→                reads: vec![],
  1460→                writes: vec![],
  1461→                encoded: word.to_le_bytes().to_vec(),
  1462→            })
  1463→            .collect();
  1464→
  1465→        let code_size = instructions.len() * 4;
  1466→
  1467→        // Extract call relocations from the emitter.
  1468→        // These are BL instructions that need to be patched with the correct
  1469→        // offset once all function positions are known.
  1470→        let relocations: Vec<RelocationEntry> = emitter.call_relocs()
  1471→            .iter()
  1472→            .map(|reloc| RelocationEntry {
  1473→                // Convert from text-section-relative to function-relative offset
  1474→                offset: reloc.text_byte_offset().saturating_sub(emitter.func_text_offset()),
  1475→                symbol: reloc.target_func().to_string(),
  1476→                reloc_type: "R_AARCH64_CALL26".to_string(),
  1477→            })
  1478→            .collect();
  1479→
  1480→        Ok(AllocatedFunction {
  1481→            name: func_name,
  1482→            blocks: vec![AllocatedBlock {
  1483→                label: "entry".to_string(),
  1484→                instructions,
  1485→                code_offset: 0,
  1486→            }],
  1487→            frame_size,
  1488→            callee_saved: vec![],
  1489→            spill_slots: 0,
  1490→            code_size,
  1491→            relocations,
  1492→        })
  1493→    }
  1494→
  1495→    fn encode_function(&self, func: &AllocatedFunction) -> Result<Vec<u8>, BackendError> {
  1496→        let mut bytes = Vec::new();
  1497→        for block in &func.blocks {
  1498→            for instr in &block.instructions {
  1499→                bytes.extend_from_slice(&instr.encoded);
  1500→            }
  1501→        }
  1502→        Ok(bytes)
  1503→    }
  1504→
  1505→    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError> {
  1506→        // ARM64 Linux executable layout:
  1507→        // [_start stub (16 bytes)] [user functions] [runtime I/O functions] [hex_table]
  1508→        let start_stub_size: usize = 16; // 4 ARM64 instructions
  1509→
  1510→        // Compute function offsets (after _start stub)
  1511→        let mut func_offsets: HashMap<String, usize> = HashMap::new();
  1512→        let mut current_offset: usize = start_stub_size;
  1513→
  1514→        for func in &program.functions {
  1515→            func_offsets.insert(func.name.clone(), current_offset);
  1516→            let func_size: usize = func.blocks.iter()
  1517→                .flat_map(|b| b.instructions.iter())
  1518→                .map(|i| i.encoded.len())
  1519→                .sum();
  1520→            current_offset += func_size;
  1521→        }
  1522→
  1523→        // Runtime I/O functions: __vuma_write_stdout (6 instructions = 24 bytes)
  1524→        let write_stdout_offset = current_offset;
  1525→        for name in &["print_buf", "write_stdout", "__vuma_write_stdout"] {
  1526→            func_offsets.insert(name.to_string(), write_stdout_offset);
  1527→        }
  1528→        let runtime_write_stdout: Vec<u8> = vec![
  1529→            0xE2, 0x03, 0x01, 0xAA, // mov x2, x1     (x2 = len)
  1530→            0xE1, 0x03, 0x00, 0xAA, // mov x1, x0     (x1 = ptr)
  1531→            0x20, 0x00, 0x80, 0xD2, // mov x0, #1     (fd = stdout)
  1532→            0x08, 0x08, 0x80, 0xD2, // mov x8, #64    (sys_write)
  1533→            0x01, 0x00, 0x00, 0xD4, // svc #0
  1534→            0xC0, 0x03, 0x5F, 0xD6, // ret
  1535→        ];
  1536→        current_offset += runtime_write_stdout.len();
  1537→
  1538→        // Runtime I/O: __vuma_print_hex (30 instructions = 120 bytes)
  1539→        // Converts each byte to two hex characters and writes to stdout.
  1540→        //
  1541→        // Register usage:
  1542→        //   x19 = ptr (callee-saved)
  1543→        //   x20 = len (callee-saved)
  1544→        //   x21 = hex_table pointer (callee-saved)
  1545→        //   x22 = temporary address register (callee-saved)
  1546→        //   x9  = loop counter
  1547→        //   x10 = current byte value
  1548→        //   x11 = nibble value
  1549→        //   x12 = hex character
  1550→        //
  1551→        // Stack layout (64 bytes):
  1552→        //   [sp+0..15]  = x29, x30 (FP/LR)
  1553→        //   [sp+16..31] = x19, x20
  1554→        //   [sp+32..47] = x21, x22
  1555→        //   [sp+48..49] = 2-byte hex output buffer
  1556→        let print_hex_offset = current_offset;
  1557→        for name in &["print_hex", "__vuma_print_hex"] {
  1558→            func_offsets.insert(name.to_string(), print_hex_offset);
  1559→        }
  1560→        let print_hex_code: Vec<u32> = vec![
  1561→            // Prologue
  1562→            0xA9BC7BFD, // stp x29, x30, [sp, #-64]!
  1563→            0x910003FD, // mov x29, sp
  1564→            0xA88153F3, // stp x19, x20, [sp, #16]
  1565→            0xA8825BF5, // stp x21, x22, [sp, #32]
  1566→            // Save args
  1567→            0xAA0003F3, // mov x19, x0 (ptr)
  1568→            0xAA0103F4, // mov x20, x1 (len)
  1569→            // ADR x21, hex_table — PATCHED AT LINK TIME (index 6)
  1570→            0x10000015, // adr x21, <hex_table>  ← placeholder
  1571→            // Loop counter
  1572→            0xD2800009, // mov x9, #0
  1573→
  1574→            // .Lhex_loop:
  1575→            0xEB14013F, // cmp x9, x20
  1576→            0x5400020A, // b.ge .Lhex_done
  1577→
  1578→            // Load byte: add x22, x19, x9; ldrb w10, [x22, #0]
  1579→            0x8B130126, // add x22, x19, x9
  1580→            0x394002CA, // ldrb w10, [x22, #0]
  1581→
  1582→            // High nibble: lsr w11, w10, #4; lookup hex char
  1583→            0x53047D4B, // lsr w11, w10, #4
  1584→            0x8B150166, // add x22, x21, x11
  1585→            0x394002CC, // ldrb w12, [x22, #0]
  1586→            0x3900C3EC, // strb w12, [sp, #48]
  1587→
  1588→            // Low nibble: and w11, w10, #0xF; lookup hex char
  1589→            0x12003D4B, // and w11, w10, #0xF
  1590→            0x8B150166, // add x22, x21, x11
  1591→            0x394002CC, // ldrb w12, [x22, #0]
  1592→            0x3900C7EC, // strb w12, [sp, #49]
  1593→
  1594→            // sys_write(1, sp+48, 2)
  1595→            0x9100C3E1, // add x1, sp, #48
  1596→            0xD2800042, // mov x2, #2
  1597→            0xD2800020, // mov x0, #1 (fd = stdout)
  1598→            0xD2800808, // mov x8, #64 (sys_write)
  1599→            0xD4000001, // svc #0
  1600→
  1601→            // Increment counter
  1602→            0x91000529, // add x9, x9, #1
  1603→
  1604→            // Loop back
  1605→            0x17FFFFEB, // b .Lhex_loop
  1606→
  1607→            // .Lhex_done:
  1608→            // Write newline
  1609→            0xD280014A, // mov x10, #10 ('\n')
  1610→            0x3900C3EA, // strb w10, [sp, #48]
  1611→            0x9100C3E1, // add x1, sp, #48
  1612→            0xD2800022, // mov x2, #1
  1613→            0xD2800020, // mov x0, #1
  1614→            0xD2800808, // mov x8, #64
  1615→            0xD4000001, // svc #0
  1616→
  1617→            // Epilogue
  1618→            0xA8C25BF5, // ldp x21, x22, [sp, #32]
  1619→            0xA8C153F3, // ldp x19, x20, [sp, #16]
  1620→            0xA9447BFD, // ldp x29, x30, [sp], #64
  1621→            0xD65F03C0, // ret
  1622→        ];
  1623→        let mut runtime_print_hex: Vec<u8> = Vec::with_capacity(print_hex_code.len() * 4);
  1624→        for &word in &print_hex_code {
  1625→            runtime_print_hex.extend_from_slice(&word.to_le_bytes());
  1626→        }
  1627→        current_offset += runtime_print_hex.len();
  1628→
  1629→        // Runtime I/O: __vuma_print_int (28 instructions = 112 bytes)
  1630→        // Converts a 64-bit unsigned integer to decimal and writes to stdout
  1631→        // with a trailing newline.
  1632→        //
  1633→        // Register usage:
  1634→        //   x19 = value (callee-saved)
  1635→        //   x20 = buffer pointer (callee-saved)
  1636→        //   x10 = quotient / temporary
  1637→        //   x11 = divisor (#10)
  1638→        //   x12 = remainder / ASCII digit
  1639→        //
  1640→        // Stack layout (48 bytes):
  1641→        //   [sp+0..15]  = x29, x30 (FP/LR)
  1642→        //   [sp+16..31] = x19, x20
  1643→        //   [sp+32..47] = decimal output buffer (16 bytes)
  1644→        let print_int_offset = current_offset;
  1645→        for name in &["print_int", "print", "__vuma_print_int"] {
  1646→            func_offsets.insert(name.to_string(), print_int_offset);
  1647→        }
  1648→        let print_int_code: Vec<u32> = vec![
  1649→            // Prologue
  1650→            0xA9BD7BFD, // stp x29, x30, [sp, #-48]!
  1651→            0x910003FD, // mov x29, sp
  1652→            0xA88153F3, // stp x19, x20, [sp, #16]
  1653→
  1654→            // Save val
  1655→            0xAA0003F3, // mov x19, x0
  1656→

  1836→    fn disassemble(&self, bytes: &[u8], addr: u64) -> Vec<String> {
  1837→        // Mnemonic decoder for AArch64 (4-byte fixed-width instructions).
  1838→        let mut lines = Vec::new();
  1839→        let mut offset = 0usize;
  1840→        let mut pc = addr;
  1841→        while offset + 4 <= bytes.len() {
  1842→            let word = u32::from_le_bytes([
  1843→                bytes[offset],
  1844→                bytes[offset + 1],
  1845→                bytes[offset + 2],
  1846→                bytes[offset + 3],
  1847→            ]);
  1848→            let mnemonic = if let Some(instr) = crate::arm64::Instruction::decode(word) {
  1849→                format!("{}", instr)
  1850→            } else {
  1851→                decode_aarch64(word)
  1852→            };
  1853→            lines.push(format!("{:#010x}:  {:08x}  {}", pc, word, mnemonic));
  1854→            offset += 4;
  1855→            pc += 4;
  1856→        }
  1857→        if offset < bytes.len() {
  1858→            let remaining = &bytes[offset..];
  1859→            lines.push(format!("{:#010x}:  {:02x?}", pc, remaining));
  1860→        }
  1861→        lines
  1862→    }
