   450→        // Version
   451→        buf.extend_from_slice(&DWARF_VERSION.to_le_bytes());
   452→        // Unit type
   453→        buf.push(DW_UT_COMPILE);
   454→        // Address size
   455→        buf.push(ADDRESS_SIZE);
   456→        // Debug abbrev offset (always 0 — we have a single abbrev table)
   457→        buf.extend_from_slice(&0u32.to_le_bytes());
   458→
   459→        // DIE data
   460→        buf.extend_from_slice(&die_buf);
   461→
   462→        buf
   463→    }
   464→
   465→    // -----------------------------------------------------------------------
   466→    // .debug_line
   467→    // -----------------------------------------------------------------------
   468→
   469→    /// Emit the `.debug_line` section containing a DWARF5 line-number program.
   470→    fn emit_debug_line(&self) -> Vec<u8> {
   471→        let mut program = Vec::new();
   472→
   473→        // Sort line entries by offset for a well-formed program.
   474→        let mut entries = self.line_entries.clone();
   475→        entries.sort_by_key(|e| e.offset);
   476→
   477→        // Default_is_stmt: true (beginning of a statement)
   478→        let default_is_stmt: u8 = 1;
   479→        // Line base: -5 (standard DWARF range)
   480→        let line_base: i8 = -5;
   481→        // Line range: 14 (standard DWARF value)
   482→        let line_range: u8 = 14;
   483→        // Opcode base: 13 (standard DWARF5 value)
   484→        let opcode_base: u8 = 13;
   485→        // Standard opcode lengths for opcodes 1..12
   486→        let standard_opcode_lengths: [u8; 12] = [0, 1, 1, 1, 1, 0, 0, 0, 1, 0, 0, 1];
   487→        // Minimum instruction length: 4 (AArch64)
   488→        let min_inst_length: u8 = 4;
   489→        // Maximum operations per instruction: 1
   490→        let max_ops_per_inst: u8 = 1;
   491→
   492→        // -- Build line program opcodes --
   493→        let mut current_line: i32 = 1;
   494→        let mut current_file: u32 = 1;
   495→        let mut current_offset: u64 = 0;
   496→
   497→        for entry in &entries {
   498→            // Set file if different
   499→            if entry.file != current_file {
   500→                program.push(DW_LNS_SET_FILE);
   501→                encode_uleb128(&mut program, entry.file as u64);
   502→                current_file = entry.file;
   503→            }
   504→
   505→            // Advance line if different
   506→            let line_diff = (entry.line as i32) - current_line;
   507→            if line_diff != 0 {
   508→                program.push(DW_LNS_ADVANCE_LINE);
   509→                encode_sleb128(&mut program, i64::from(line_diff));
   510→                current_line = entry.line as i32;
   511→            }
   512→
   513→            // Advance PC
   514→            if entry.offset > current_offset {
   515→                let addr_advance = entry.offset - current_offset;
   516→                // DW_LNS_ADVANCE_PC advances by the operand * min_inst_length
   517→                // Since min_inst_length = 4, operand = addr_advance / 4
   518→                let op_advance = addr_advance / (min_inst_length as u64);
   519→                if op_advance > 0 {
   520→                    program.push(DW_LNS_ADVANCE_PC);
   521→                    encode_uleb128(&mut program, op_advance);
   522→                }
   523→                current_offset = entry.offset;
   524→            }
   525→
   526→            // Copy current row
   527→            program.push(DW_LNS_COPY);
   528→        }
   529→
   530→        // End sequence: advance PC to end and emit DW_LNE_END_SEQUENCE
   531→        // Use the last entry's offset + some delta, or 0 if no entries
   532→        let end_addr = entries.last().map(|e| e.end_offset()).unwrap_or(0);
   533→        if end_addr > current_offset {
   534→            let op_advance = (end_addr - current_offset) / (min_inst_length as u64);
   535→            if op_advance > 0 {
   536→                program.push(DW_LNS_ADVANCE_PC);
   537→                encode_uleb128(&mut program, op_advance);
   538→            }
   539→        }
   540→        // DW_LNE_END_SEQUENCE: extended opcode
   541→        let ext_opcode_bytes = [DW_LNE_END_SEQUENCE];
   542→        encode_uleb128(&mut program, (ext_opcode_bytes.len() as u64) + 1); // length (includes sub-opcode)
   543→        program.push(DW_LNE_END_SEQUENCE);
   544→
   545→        // -- Build directory entry format and table --
   546→        // DWARF5: directory entry format count
   547→        let dir_entry_format_count: u8 = 1;
   548→        // Directory entry format: (DW_LNCT_path, DW_FORM_STRING)
   549→        let dir_entry_format: Vec<u8> = {
   550→            let mut f = Vec::new();
   551→            encode_uleb128(&mut f, 0x01); // DW_LNCT_path
   552→            encode_uleb128(&mut f, DW_FORM_STRING as u64);
   553→            f
   554→        };
   555→
   556→        // Directories: just one — the compilation directory (".")
   557→        let directories_count: u8 = 1;
   558→        let directory_name = b".\0";
   559→
   560→        // -- Build file name entry format and table --
   561→        // DWARF5: file name entry format count
   562→        let file_entry_format_count: u8 = 2;
   563→        // File entry format: (DW_LNCT_path, DW_FORM_STRING), (DW_LNCT_directory_index, DW_FORM_DATA4)
   564→        let file_entry_format: Vec<u8> = {
   565→            let mut f = Vec::new();
   566→            encode_uleb128(&mut f, 0x01); // DW_LNCT_path
   567→            encode_uleb128(&mut f, DW_FORM_STRING as u64);
   568→            encode_uleb128(&mut f, 0x02); // DW_LNCT_directory_index
   569→            encode_uleb128(&mut f, DW_FORM_DATA4 as u64);
   570→            f
   571→        };
   572→
   573→        // File names: just one — the source file
   574→        let file_names_count: u8 = 1;
   575→        let file_name_entry = {
   576→            let mut f = Vec::new();
   577→            f.extend_from_slice(self.source_file.as_bytes());
   578→            f.push(0); // null-terminated
   579→            f.extend_from_slice(&0u32.to_le_bytes()); // directory index 0
   580→            f
   581→        };
   582→
   583→        // -- Assemble the line number program header --
   584→        let mut header = Vec::new();
   585→
   586→        // Version
   587→        header.extend_from_slice(&DWARF_VERSION.to_le_bytes());
   588→
   589→        // Address size
   590→        header.push(ADDRESS_SIZE);
   591→
   592→        // Segment selector size (DWARF5: 0)
   593→        header.push(0);
   594→
   595→        // Header length — will be filled after we know the size
   596→        let header_before_length_field = header.len();
   597→        // Placeholder for 4-byte header length
   598→        header.extend_from_slice(&0u32.to_le_bytes());
   599→
   600→        // Minimum instruction length
   601→        header.push(min_inst_length);
   602→        // Maximum operations per instruction
   603→        header.push(max_ops_per_inst);
   604→        // Default is_stmt
   605→        header.push(default_is_stmt);
   606→        // Line base
   607→        header.push(line_base as u8);
   608→        // Line range
   609→        header.push(line_range);
   610→        // Opcode base
   611→        header.push(opcode_base);
   612→        // Standard opcode lengths
   613→        header.extend_from_slice(&standard_opcode_lengths);
   614→
   615→        // Directory entry format count
   616→        header.push(dir_entry_format_count);
   617→        // Directory entry format
   618→        header.extend_from_slice(&dir_entry_format);
   619→        // Directories count
   620→        encode_uleb128(&mut header, directories_count as u64);
   621→        // Directory entries
   622→        header.extend_from_slice(directory_name);
   623→
   624→        // File name entry format count
   625→        header.push(file_entry_format_count);
   626→        // File name entry format
   627→        header.extend_from_slice(&file_entry_format);
   628→        // File names count
   629→        encode_uleb128(&mut header, file_names_count as u64);
   630→        // File name entries
   631→        header.extend_from_slice(&file_name_entry);
   632→
   633→        // Fix up header_length (bytes after the header_length field itself)
   634→        let header_length = (header.len() - header_before_length_field - 4) as u32;
   635→        let hl_offset = header_before_length_field;
   636→        header[hl_offset..hl_offset + 4].copy_from_slice(&header_length.to_le_bytes());
   637→
   638→        // -- Assemble full .debug_line section --
   639→        let mut buf = Vec::new();
   640→
   641→        // Unit length (4 bytes) — total length of the rest of the unit
   642→        let unit_length = (header.len() + program.len()) as u32;
   643→        buf.extend_from_slice(&unit_length.to_le_bytes());
   644→        buf.extend_from_slice(&header);
   645→        buf.extend_from_slice(&program);
   646→
   647→        buf
   648→    }
   649→}

   652→    fn default() -> Self {
   653→        Self::new()
   654→    }

   658→// Encoding helpers
   659→// ---------------------------------------------------------------------------
   660→
   661→/// Encode an unsigned LEB128 value into `buf`.
   662→fn encode_uleb128(buf: &mut Vec<u8>, mut value: u64) {
   663→    loop {
   664→        let mut byte = (value & 0x7F) as u8;
   665→        value >>= 7;
   666→        if value != 0 {
   667→            byte |= 0x80;
   668→        }
   669→        buf.push(byte);
   670→        if value == 0 {
   671→            break;
   672→        }
   673→    }
   674→}
   675→
   676→/// Encode a signed LEB128 value into `buf`.
   677→fn encode_sleb128(buf: &mut Vec<u8>, mut value: i64) {
   678→    loop {
   679→        let mut byte = (value & 0x7F) as u8;
   680→        value >>= 7;
   681→        let sign_bit_set = (byte & 0x40) != 0;
   682→        let done = value == 0 || (value == -1 && sign_bit_set);
   683→        if !done {
   684→            byte |= 0x80;
   685→        }
   686→        buf.push(byte);
   687→        if done {
   688→            break;
   689→        }
   690→    }
   691→}
   692→
   693→/// Write a null-terminated string into `buf`.
   694→fn write_null_string(buf: &mut Vec<u8>, s: &str) {
   695→    buf.extend_from_slice(s.as_bytes());
   696→    buf.push(0);
   697→}
   698→
   699→/// Build a `DW_OP_fbreg <offset>` expression.
   700→///
   701→/// `DW_OP_fbreg` (0x91) followed by a signed LEB128 offset.
   702→fn build_fbreg_expr(offset: i32) -> Vec<u8> {
   703→    let mut expr = vec![0x91]; // DW_OP_fbreg
   704→    encode_sleb128(&mut expr, offset as i64);
   705→    expr
   706→}

   725→/// Section header type for DWARF debug info sections (progbits).
   726→const SHT_PROGBITS: u32 = 1;
   727→
   728→/// Append DWARF5 debug sections to an ELF binary.
   729→///
   730→/// Given an existing ELF binary and the debug sections, this function inserts
   731→/// the three debug sections (`.debug_abbrev`, `.debug_info`, `.debug_line`)
   732→/// into the ELF, updates the section header table, and patches the section
   733→/// header string table and section count.
   734→///
   735→/// This is a simplified integration that appends sections before the section
   736→/// headers. It assumes the input ELF was produced by `emit_elf` with
   737→/// section headers enabled.
   738→pub fn append_debug_sections_to_elf(elf: &mut Vec<u8>, debug: &DebugSections) {
   739→    // Parse the ELF header to find section header table location and count.
   740→    if elf.len() < 64 {
   741→        return;
   742→    }
   743→
   744→    let e_shoff = u64::from_le_bytes(elf[40..48].try_into().unwrap_or([0; 8]));
   745→    let e_shentsize = u16::from_le_bytes(elf[58..60].try_into().unwrap_or([0; 2])) as u64;
   746→    let e_shnum = u16::from_le_bytes(elf[60..62].try_into().unwrap_or([0; 2])) as usize;
   747→    let e_shstrndx = u16::from_le_bytes(elf[62..64].try_into().unwrap_or([0; 2])) as usize;
   748→
   749→    if e_shoff == 0 || e_shnum == 0 || e_shentsize == 0 {
   750→        return;
   751→    }
   752→
   753→    // Read the .shstrtab section to find its data.
   754→    let shstrtab_shdr_offset = e_shoff as usize + e_shstrndx * e_shentsize as usize;
   755→    if shstrtab_shdr_offset + 64 > elf.len() {
   756→        return;
   757→    }
   758→    let shstrtab_offset = u64::from_le_bytes(
   759→        elf[shstrtab_shdr_offset + 24..shstrtab_shdr_offset + 32]
   760→            .try_into()
   761→            .unwrap_or([0; 8]),
   762→    ) as usize;
   763→    let shstrtab_size = u64::from_le_bytes(
   764→        elf[shstrtab_shdr_offset + 32..shstrtab_shdr_offset + 40]
   765→            .try_into()
   766→            .unwrap_or([0; 8]),
   767→    ) as usize;
   768→
   769→    // Build new shstrtab with debug section names appended.
   770→    let mut new_shstrtab = elf[shstrtab_offset..shstrtab_offset + shstrtab_size].to_vec();
   771→
   772→    // Record name offsets
   773→    let debug_abbrev_name_off = new_shstrtab.len() as u32;
   774→    new_shstrtab.extend_from_slice(b".debug_abbrev\0");
   775→    let debug_info_name_off = new_shstrtab.len() as u32;
   776→    new_shstrtab.extend_from_slice(b".debug_info\0");
   777→    let debug_line_name_off = new_shstrtab.len() as u32;
   778→    new_shstrtab.extend_from_slice(b".debug_line\0");
   779→
   780→    // The new sections will be appended right before the section header table.
   781→    // We need to compute their file offsets.
   782→    let new_sections_start = e_shoff as usize;
   783→
   784→    // Compute aligned sizes for debug sections.
   785→    let debug_abbrev_aligned = align_up(debug.debug_abbrev.len() as u64, 8) as usize;
   786→    let debug_info_aligned = align_up(debug.debug_info.len() as u64, 8) as usize;
   787→    let debug_line_aligned = align_up(debug.debug_line.len() as u64, 8) as usize;
   788→    let new_shstrtab_aligned = align_up(new_shstrtab.len() as u64, 8) as usize;
   789→
   790→    // Build section data to insert.
   791→    let mut new_section_data = Vec::new();
   792→
   793→    let debug_abbrev_file_offset = new_sections_start + new_section_data.len();
   794→    new_section_data.extend_from_slice(&debug.debug_abbrev);
   795→    let pad = debug_abbrev_aligned - debug.debug_abbrev.len();
   796→    new_section_data.extend_from_slice(&vec![0u8; pad]);
   797→
   798→    let debug_info_file_offset = new_sections_start + new_section_data.len();
   799→    new_section_data.extend_from_slice(&debug.debug_info);
   800→    let pad = debug_info_aligned - debug.debug_info.len();
   801→    new_section_data.extend_from_slice(&vec![0u8; pad]);
   802→
   803→    let debug_line_file_offset = new_sections_start + new_section_data.len();
   804→    new_section_data.extend_from_slice(&debug.debug_line);
   805→    let pad = debug_line_aligned - debug.debug_line.len();
   806→    new_section_data.extend_from_slice(&vec![0u8; pad]);
   807→
   808→    // Replace the old shstrtab with the new one.
   809→    // We'll place the new shstrtab at the end of the debug sections.
   810→    let new_shstrtab_file_offset = new_sections_start + new_section_data.len();
   811→    new_section_data.extend_from_slice(&new_shstrtab);
   812→    let pad = new_shstrtab_aligned - new_shstrtab.len();
   813→    new_section_data.extend_from_slice(&vec![0u8; pad]);
   814→
   815→    // Now insert the new section data before the section headers.
   816→    // First, remove the old section headers from the end.
   817→    let old_shdrs: Vec<u8> = elf.drain(e_shoff as usize..).collect();
   818→
   819→    // Insert debug section data.
   820→    elf.extend_from_slice(&new_section_data);
   821→
   822→    // Compute the new section header table offset.
   823→    let new_e_shoff = elf.len() as u64;
   824→
   825→    // Build new section headers: copy old ones, add debug section headers,
   826→    // and update the .shstrtab header.
   827→    let mut new_shdrs = old_shdrs;
   828→
   829→    // Update the .shstrtab section header to point to the new data.
   830→    let shstrtab_shdr_start = e_shstrndx * 64;
   831→    if shstrtab_shdr_start + 64 <= new_shdrs.len() {
   832→        // sh_offset (bytes 24-31)
   833→        new_shdrs[shstrtab_shdr_start + 24..shstrtab_shdr_start + 32]
   834→            .copy_from_slice(&new_shstrtab_file_offset.to_le_bytes());
   835→        // sh_size (bytes 32-39)
   836→        new_shdrs[shstrtab_shdr_start + 32..shstrtab_shdr_start + 40]
   837→            .copy_from_slice(&(new_shstrtab.len() as u64).to_le_bytes());
   838→    }
   839→
   840→    // Add section headers for the three debug sections.
   841→    let new_num_shdrs = e_shnum + 3;
   842→
   843→    // .debug_abbrev section header
   844→    let mut sh = new_shdr(
   845→        SHT_PROGBITS,
   846→        0, // no flags
   847→        0, // no virtual address
   848→        debug_abbrev_file_offset as u64,
   849→        debug.debug_abbrev.len() as u64,
   850→        0,
   851→        0,
   852→        1,
   853→        0,
   854→    );
   855→    sh.name = debug_abbrev_name_off;
   856→    write_filled_shdr(&mut new_shdrs, &sh);
   857→
   858→    // .debug_info section header
   859→    let mut sh = new_shdr(
   860→        SHT_PROGBITS,
   861→        0,
   862→        0,
   863→        debug_info_file_offset as u64,
   864→        debug.debug_info.len() as u64,
   865→        0,
   866→        0,
   867→        1,
   868→        0,
   869→    );
   870→    sh.name = debug_info_name_off;
   871→    write_filled_shdr(&mut new_shdrs, &sh);
   872→
   873→    // .debug_line section header
   874→    let mut sh = new_shdr(
   875→        SHT_PROGBITS,
   876→        0,
   877→        0,
   878→        debug_line_file_offset as u64,
   879→        debug.debug_line.len() as u64,
   880→        0,
   881→        0,
   882→        1,
   883→        0,
   884→    );
   885→    sh.name = debug_line_name_off;
   886→    write_filled_shdr(&mut new_shdrs, &sh);
   887→
   888→    // Append section headers.
   889→    elf.extend_from_slice(&new_shdrs);
   890→
   891→    // Update the ELF header: e_shoff and e_shnum.
   892→    elf[40..48].copy_from_slice(&new_e_shoff.to_le_bytes());
   893→    elf[60..62].copy_from_slice(&(new_num_shdrs as u16).to_le_bytes());
   894→    // e_shstrndx stays the same — the .shstrtab section index hasn't changed.
   895→}

   924→    sh_addralign: u64,

   928→        name: 0,

   941→fn write_filled_shdr(buf: &mut Vec<u8>, sh: &FilledShdr) {
   942→    buf.extend_from_slice(&sh.name.to_le_bytes());
   943→    buf.extend_from_slice(&sh.sh_type.to_le_bytes());
   944→    buf.extend_from_slice(&sh.sh_flags.to_le_bytes());
   945→    buf.extend_from_slice(&sh.sh_addr.to_le_bytes());
   946→    buf.extend_from_slice(&sh.sh_offset.to_le_bytes());
   947→    buf.extend_from_slice(&sh.sh_size.to_le_bytes());
   948→    buf.extend_from_slice(&sh.sh_link.to_le_bytes());
   949→    buf.extend_from_slice(&sh.sh_info.to_le_bytes());
   950→    buf.extend_from_slice(&sh.sh_addralign.to_le_bytes());
   951→    buf.extend_from_slice(&sh.sh_entsize.to_le_bytes());
   952→}

   954→/// Round `value` up to the nearest multiple of `alignment`.

   959→// ---------------------------------------------------------------------------
   960→// ULEB128 / SLEB128 decoding helpers (for tests)
   961→// ---------------------------------------------------------------------------
   962→
   963→/// Decode a ULEB128 value from `buf` starting at `offset`.
   964→/// Returns (value, bytes_consumed).
   965→#[cfg(test)]
   966→fn decode_uleb128(buf: &[u8], offset: usize) -> (u64, usize) {
   967→    let mut value: u64 = 0;
   968→    let mut shift: u32 = 0;
   969→    let mut i = offset;
   970→    loop {
   971→        let byte = buf[i];
   972→        value |= ((byte & 0x7F) as u64) << shift;
   973→        shift += 7;
   974→        i += 1;
   975→        if (byte & 0x80) == 0 {
   976→            break;
   977→        }
   978→    }
   979→    (value, i - offset)
   980→}
   981→
   982→/// Decode a SLEB128 value from `buf` starting at `offset`.
   983→/// Returns (value, bytes_consumed).
   984→#[cfg(test)]
   985→fn decode_sleb128(buf: &[u8], offset: usize) -> (i64, usize) {
   986→    let mut value: i64 = 0;
   987→    let mut shift: u32 = 0;
   988→    let mut i = offset;
   989→    let mut byte: u8;
   990→    loop {
   991→        byte = buf[i];
   992→        value |= ((byte & 0x7F) as i64) << shift;
   993→        shift += 7;
   994→        i += 1;
   995→        if (byte & 0x80) == 0 {
   996→            break;
   997→        }
   998→    }
   999→    // Sign extend
  1000→    if shift < 64 && (byte & 0x40) != 0 {
  1001→        value |= !0 << shift;
  1002→    }
  1003→    (value, i - offset)
  1004→}
