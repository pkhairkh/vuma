     1→//! # ARM64 Code Emission
     2→//!
     3→//! Lowers IR to ARM64 machine code and produces ELF binaries or raw binaries
     4→//! suitable for the AArch64 (Cortex-A76, ARMv8.2-A).
     5→//!
     6→//! ## Pipeline
     7→//!
     8→//! 1. **IR → ARM64 Instructions**: Each IR instruction is pattern-matched and
     9→//!    lowered to one or more [`Instruction`]
    10→//!    values, with virtual registers replaced by physical registers from the
    11→//!    register allocator.
    12→//! 2. **Emit function**: Each IR function is emitted as a sequence of 32-bit
    13→//!    ARM64 code words.
    14→//! 3. **Emit program**: All functions are collected, data sections are laid
    15→//!    out, and a minimal ELF64 binary is produced.
    16→//!
    17→//! ## Output Formats
    18→//!
    19→//! - **ELF**: Full ELF64 binary with program headers, section headers, symbol
    20→//!   table, and string table.  Suitable for Linux on AArch64.
    21→//! - **Raw**: Flat binary image for bare-metal AArch64 (loaded at 0x80000).
    22→//! - **Obj**: Relocatable ELF object file (`ET_REL`) for linking.
    23→//!
    24→//! ## ELF Layout (executable)
    25→//!
    26→//! ```text
    27→//! ┌─────────────────────┐
    28→//! │ ELF Header           │  64 bytes
    29→//! ├─────────────────────┤
    30→//! │ Program Headers      │  3 × 56 bytes (LOAD segments)
    31→//! ├─────────────────────┤
    32→//! │ .rodata              │  read-only data (segment: R)
    33→//! ├─────────────────────┤
    34→//! │ .text                │  emitted code   (segment: R+X)
    35→//! ├─────────────────────┤
    36→//! │ .data                │  initialized data (segment: R+W)
    37→//! ├─────────────────────┤
    38→//! │ .symtab              │  symbol table entries
    39→//! ├─────────────────────┤
    40→//! │ .strtab              │  symbol string table
    41→//! ├─────────────────────┤
    42→//! │ .shstrtab            │  section header string table
    43→//! ├─────────────────────┤
    44→//! │ Section Headers      │  N × 64 bytes
    45→//! └─────────────────────┘
    46→//! .bss is virtual-only (memsz > filesz in the data LOAD segment).
    47→//!
    48→//! Section alignment is target-dependent:
    49→//! - ARM32:    4-byte alignment
    50→//! - AArch64:  16-byte alignment
    51→//! - x86-64:   16-byte alignment
    52→//! - RISC-V:   4-byte alignment
    53→//! - MIPS64:   8-byte alignment
    54→//! - PPC64:    16-byte alignment
    55→//! - LoongArch: 8-byte alignment
    56→//! ```
    57→//!
    58→//! ## Relocation Support
    59→//!
    60→//! Inter-function calls (`BL`) are resolved in a fixup pass after all
    61→//! functions have been emitted.  The emitter records the word offset of each
    62→//! `BL` instruction and the target function name; once function addresses
    63→//! are known, the branch offsets are patched into the encoded instructions.
    64→

   517→// ---------------------------------------------------------------------------
   518→// Inter-function call relocation record
   519→// ---------------------------------------------------------------------------
   520→
   521→/// A record for a `BL` instruction that needs to be patched with the address
   522→/// of a named function.  After all functions are emitted, these records are
   523→/// resolved to compute the correct branch offset.
   524→#[derive(Debug, Clone)]
   525→struct CallRelocation {
   526→    /// Byte offset within the text section where the BL instruction lives.
   527→    text_byte_offset: u64,
   528→    /// Name of the target function.
   529→    target_func: String,
   530→}
   531→
   532→// ---------------------------------------------------------------------------
   533→// ELF RelaEntry
   534→// ---------------------------------------------------------------------------
   535→
   536→/// An ELF64 relocation entry with an explicit addend (SHT_RELA).
   537→///
   538→/// Each entry specifies where in the section a relocation must be applied,
   539→/// which symbol the relocation references, the type of relocation, and an
   540→/// addend value used in the relocation computation.
   541→#[derive(Debug, Clone)]
   542→pub struct RelaEntry {
   543→    /// Byte offset within the section where the relocation applies.
   544→    pub offset: u64,
   545→    /// Symbol index (upper 32 bits) and relocation type (lower 32 bits),
   546→    /// packed as: `(sym_idx << 32) | r_type`.
   547→    pub info: u64,
   548→    /// Addend value used in the relocation computation.
   549→    pub addend: i64,
   550→}
   551→
   552→impl RelaEntry {
   553→    /// Create a new relocation entry.
   554→    pub fn new(offset: u64, sym_idx: u32, r_type: u32, addend: i64) -> Self {
   555→        Self {
   556→            offset,
   557→            info: ((sym_idx as u64) << 32) | (r_type as u64),
   558→            addend,
   559→        }
   560→    }
   561→
   562→    /// Extract the symbol index from `info`.
   563→    pub fn sym_idx(&self) -> u32 {
   564→        (self.info >> 32) as u32
   565→    }
   566→
   567→    /// Extract the relocation type from `info`.
   568→    pub fn r_type(&self) -> u32 {
   569→        (self.info & 0xFFFFFFFF) as u32
   570→    }
   571→
   572→    /// Serialize the entry to 24 bytes (little-endian).
   573→    pub fn to_bytes(&self) -> [u8; 24] {
   574→        let mut buf = [0u8; 24];
   575→        buf[0..8].copy_from_slice(&self.offset.to_le_bytes());
   576→        buf[8..16].copy_from_slice(&self.info.to_le_bytes());
   577→        buf[16..24].copy_from_slice(&self.addend.to_le_bytes());
   578→        buf
   579→    }
   580→}
   581→
   582→// ---------------------------------------------------------------------------
   583→// Emitter
   584→// ---------------------------------------------------------------------------
   585→
   586→/// The ARM64 code emitter.
   587→///
   588→/// Holds state for the current emission context: the register allocator,
   589→/// accumulated code, and fixup records for branch targets.
   590→pub struct Emitter {
   591→    /// Register allocator used during emission.
   592→    reg_alloc: RegAllocator,
   593→    /// Accumulated machine code for the current function (32-bit words).
   594→    code: Vec<u32>,
   595→    /// Fixup records for intra-function branches: (word index, target label name, format).
   596→    fixups: Vec<(usize, String, BranchFormat)>,
   597→    /// Map from label name to code offset (in words) within the current function.
   598→    label_offsets: HashMap<String, usize>,
   599→    /// Inter-function call relocations for the current function.
   600→    call_relocs: Vec<CallRelocation>,
   601→    /// Relocation entries using the `RelocationEntry` infrastructure.
   602→    ///
   603→    /// Each entry records a byte offset within the function's encoded output
   604→    /// where a symbolic reference must be resolved, the target symbol name,
   605→    /// and the ISA-specific relocation type (e.g., `"R_AARCH64_CALL26"`).
   606→    relocations: Vec<RelocationEntry>,
   607→    /// Name of the function currently being emitted.
   608→    current_func_name: String,
   609→    /// Byte offset of the current function within the text section (set externally).
   610→    func_text_offset: u64,
   611→    /// Computed stack frame size (in bytes) for the current function.
   612→    frame_size: u32,
   613→    /// Registers pinned for the current instruction (auto-unpinned after each instruction).
   614→    instr_pinned_regs: Vec<Register>,
   615→}
   616→
   617→impl Emitter {
   618→    /// Create a new emitter with a fresh register allocator.
   619→    pub fn new() -> Self {
   620→        Self {
   621→            reg_alloc: RegAllocator::new(),
   622→            code: Vec::new(),
   623→            fixups: Vec::new(),
   624→            label_offsets: HashMap::new(),
   625→            call_relocs: Vec::new(),
   626→            relocations: Vec::new(),
   627→            current_func_name: String::new(),
   628→            func_text_offset: 0,
   629→            frame_size: 0,
   630→            instr_pinned_regs: Vec::new(),
   631→        }
   632→    }
   633→
   634→    // -----------------------------------------------------------------------
   635→    // Function emission
   636→    // -----------------------------------------------------------------------
   637→
   638→    /// Emit a single IR function to ARM64 machine code.
   639→    ///
   640→    /// For functions with more than `STACK_SLOT_VREG_THRESHOLD` virtual
   641→    /// registers, the stack-slot emitter is used (every vreg gets a stack
   642→    /// slot, scratch registers only).  For simpler functions, the greedy
   643→    /// register allocator is used.
   644→    ///
   645→    /// Returns a vector of 32-bit ARM64 instruction words.
   646→    pub fn emit_function(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   647→        let vreg_count = count_vregs(func);
   648→        if vreg_count > STACK_SLOT_VREG_THRESHOLD {
   649→            self.emit_function_stack_slot(func)
   650→        } else {
   651→            self.emit_function_greedy(func)
   652→        }
   653→    }
   654→
   655→    /// Emit a single IR function using the greedy register allocator.
   656→    ///
   657→    /// This is the original emission strategy — suitable for functions with
   658→    /// a small number of virtual registers.
   659→    fn emit_function_greedy(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   660→        self.code.clear();
   661→        self.fixups.clear();
   662→        self.label_offsets.clear();
   663→        self.call_relocs.clear();
   664→        self.relocations.clear();
   665→        self.current_func_name = func.name.clone();
   666→        self.reg_alloc.reset();
   667→
   668→        // Pre-allocate registers for parameters (AAPCS64: X0–X7).
   669→        // We must explicitly assign parameter virtual registers to the
   670→        // correct argument registers so the calling convention is respected.
   671→        let arg_regs = [
   672→            Register::X0, Register::X1, Register::X2, Register::X3,
   673→            Register::X4, Register::X5, Register::X6, Register::X7,
   674→        ];
   675→        for (i, param) in func.params.iter().enumerate() {
   676→            if let IRValue::Register(vreg_id) = param {
   677→                if i < 8 {
   678→                    self.reg_alloc.preassign(*vreg_id, arg_regs[i]);
   679→                }
   680→            }
   681→        }
   682→
   683→        // Emit prologue:
   684→        // 1. SUB SP, SP, #16          (make room for FP/LR save pair)
   685→        // 2. STP X29, X30, [SP]       (save FP and LR at new SP)
   686→        // 3. ADD X29, SP, #0           (set frame pointer to new SP)
   687→        // 4. SUB SP, SP, #frame_size   (make room for local variables)
   688→        //
   689→        // NOTE: We cannot use pre-indexed STP [SP, #-16]! because the
   690→        // Instruction::STP encoding only supports signed-offset mode.
   691→        // Instead, we explicitly decrement SP with SUB before the STP.
   692→        self.emit_instruction(Instruction::SUB {
   693→            rd: Register::SP,
   694→            rn: Register::SP,
   695→            rm: Operand::Imm12(16),
   696→        })?;
   697→        self.emit_instruction(Instruction::STP {
   698→            rt1: Register::X29,
   699→            rt2: Register::X30,
   700→            rn: Register::SP,
   701→            offset: 0,
   702→        })?;
   703→
   704→        // MOV X29, SP (set frame pointer)
   705→        // IMPORTANT: Cannot use MOV (ORR Xd, XZR, SP) because ORR treats
   706→        // Rm=31 as XZR, yielding zero. Use ADD X29, SP, #0 instead.
   707→        self.emit_instruction(Instruction::ADD {
   708→            rd: Register::X29,
   709→            rn: Register::SP,
   710→            rm: Operand::Imm12(0),
   711→        })?;
   712→
   713→        // Reserve space for spill slots only. Each Alloc instruction handles
   714→        // its own SUB SP, SP, #aligned_size. The epilogue will restore only
   715→        // the spill area reservation. The Alloc areas are NOT restored by the
   716→        // epilogue — they're effectively "leaked" stack space. This is fine
   717→        // because the function's stack frame is restored by the LDP + ADD SP, #16.
   718→        //
   719→        // For spill slots, we need to estimate the maximum number of spills
   720→        // that could occur during emission. Since we can't know the exact
   721→        // count beforehand, we use the compute_frame_size function which
   722→        // estimates based on the vreg count.
   723→        let aligned_stack = compute_frame_size(func);
   724→        // frame_size = only the spill area (NOT the Alloc sizes, since each
   725→        // Alloc instruction does its own SUB SP)
   726→        // Compute the spill-only portion:
   727→        let mut alloc_total: u32 = 0;
   728→        for block in &func.blocks {
   729→            for instr in &block.instructions {
   730→                if let IRInstr::Alloc { size, .. } = instr {
   731→                    let aligned = (*size).div_ceil(16) * 16;
   732→                    alloc_total += aligned;
   733→                }
   734→            }
   735→        }
   736→        // Spill area = total frame - alloc area
   737→        let spill_area = if aligned_stack > alloc_total {
   738→            aligned_stack - alloc_total
   739→        } else {
   740→            // Minimal 16-byte reservation for safety
   741→            16
   742→        };
   743→        let spill_area_aligned = (spill_area + 15) & !15;
   744→        self.frame_size = spill_area_aligned;
   745→        if spill_area_aligned > 0 {
   746→            if spill_area_aligned <= 4095 {
   747→                self.emit_instruction(Instruction::SUB {
   748→                    rd: Register::SP,
   749→                    rn: Register::SP,
   750→                    rm: Operand::Imm12(spill_area_aligned as u16),
   751→                })?;
   752→            } else {
   753→                self.emit_load_immediate(Register::X9, spill_area_aligned as i64)?;
   754→                self.emit_instruction(Instruction::SUB {
   755→                    rd: Register::SP,
   756→                    rn: Register::SP,
   757→                    rm: Operand::Reg {
   758→                        reg: Register::X9,
   759→                        shift: None,
   760→                    },
   761→                })?;
   762→            }
   763→        }
   764→
   765→        // Emit each basic block.
   766→        for block in &func.blocks {
   767→            self.label_offsets
   768→                .insert(block.label.clone(), self.code.len());
   769→            self.emit_block(block)?;
   770→        }
   771→
   772→        // Apply fixups — resolve intra-function branch targets.
   773→        self.apply_fixups()?;
   774→
   775→        Ok(self.code.clone())
   776→    }
   777→
   778→    /// Returns the relocation entries recorded during emission of the current
   779→    /// function.
   780→    ///
   781→    /// Each `RelocationEntry` records a byte offset within the function's
   782→    /// encoded output where a symbolic reference must be resolved, the target
   783→    /// symbol name, and the ISA-specific relocation type (e.g.,
   784→    /// `"R_AARCH64_CALL26"`).
   785→    pub fn relocations(&self) -> &[RelocationEntry] {
   786→        &self.relocations
   787→    }
   788→
   789→    /// Emit instructions for a single IR basic block.
   790→    fn emit_block(&mut self, block: &IRBlock) -> Result<()> {
   791→        for instr in &block.instructions {
   792→            self.emit_ir_instr(instr)?;
   793→            // Auto-unpin all registers that were pinned during this instruction.
   794→            for reg in self.instr_pinned_regs.drain(..) {
   795→                self.reg_alloc.unpin(reg);
   796→            }
   797→        }
   798→        self.emit_terminator(&block.terminator)?;
   799→        Ok(())
   800→    }
   801→
   802→    // -----------------------------------------------------------------------
   803→    // IR → ARM64 instruction lowering
   804→    // -----------------------------------------------------------------------
   805→
   806→    /// Lower a single IR instruction to ARM64 instructions.
   807→    fn emit_ir_instr(&mut self, instr: &IRInstr) -> Result<()> {
   808→        match instr {
   809→            IRInstr::Load { dst, addr, offset, ty } => {
   810→                let rn = self.resolve_reg(addr)?;
   811→                let rt = self.resolve_reg(dst)?;
   812→                self.emit_load(rt, rn, *offset, ty)?;
   813→            }
   814→
   815→            IRInstr::Store { value, addr, offset, ty } => {
   816→                let rt = self.resolve_reg(value)?;
   817→                let rn = self.resolve_reg(addr)?;
   818→                if rt == rn {
   819→                    // Both resolved to the same register — use scratch for value
   820→                    self.emit_instruction(Instruction::MOV {
   821→                        rd: Register::X16,
   822→                        rm: rt,
   823→                    })?;
   824→                    self.emit_store(Register::X16, rn, *offset, ty)?;
   825→                } else {
   826→                    self.emit_store(rt, rn, *offset, ty)?;
   827→                }
   828→            }
   829→
   830→            IRInstr::BinOp { op, dst, lhs, rhs, ty } => {
   831→                self.emit_binop(*op, dst, lhs, rhs, ty.as_ref())?;
   832→            }
   833→
   834→            IRInstr::UnaryOp { op, dst, operand, ty: _ } => {
   835→                let rd = self.resolve_reg(dst)?;
   836→                let rn = self.resolve_reg(operand)?;
   837→                match op {
   838→                    UnaryOpKind::Neg => {
   839→                        self.emit_instruction(Instruction::SUB {
   840→                            rd,
   841→                            rn: Register::XZR,
   842→                            rm: Operand::Reg {
   843→                                reg: rn,
   844→                                shift: None,
   845→                            },
   846→                        })?;
   847→                    }
   848→                    UnaryOpKind::Not => {
   849→                        self.emit_load_immediate(Register::X9, -1)?;
   850→                        self.emit_instruction(Instruction::EOR {
   851→                            rd,
   852→                            rn,
   853→                            rm: Register::X9,
   854→                        })?;
   855→                    }
   856→                    UnaryOpKind::Clz => {
   857→                        // CLZ Xd, Xn — count leading zeros (native ARM64 instruction)
   858→                        self.emit_instruction(Instruction::CLZ { rd, rn })?;
   859→                    }
   860→                    UnaryOpKind::Ctz => {
   861→                        // CTZ = RBIT + CLZ: reverse bits then count leading zeros.
   862→                        // Use X9 as scratch if rd == rn (need intermediate result).
   863→                        if rd == rn {
   864→                            self.emit_instruction(Instruction::RBIT {
   865→                                rd: Register::X9,
   866→                                rn,
   867→                            })?;
   868→                            self.emit_instruction(Instruction::CLZ {
   869→                                rd,
   870→                                rn: Register::X9,
   871→                            })?;
   872→                        } else {
   873→                            self.emit_instruction(Instruction::RBIT { rd, rn })?;
   874→                            self.emit_instruction(Instruction::CLZ { rd, rn: rd })?;
   875→                        }
   876→                    }
   877→                    UnaryOpKind::Popcnt => {
   878→                        // POPCNT via FMOV+CNT+ADDV+UMOV sequence:
   879→                        // FMOV D8, Xn        — move GPR value to SIMD register (V8 is caller-saved)
   880→                        // CNT V8.8B, V8.8B   — count bits per byte
   881→                        // ADDV B8, V8.8B     — horizontal sum of byte counts
   882→                        // UMOV Xd, V8.B[0]   — move result back to GPR (zero-extends to 64 bits)
   883→                        const SIMD_SCRATCH: u8 = 8; // V8 is caller-saved in AAPCS64
   884→                        self.emit_instruction(Instruction::FMOV_DX {
   885→                            vd: SIMD_SCRATCH,
   886→                            rn,
   887→                        })?;
   888→                        self.emit_instruction(Instruction::CNT {
   889→                            vd: SIMD_SCRATCH,
   890→                            vn: SIMD_SCRATCH,
   891→                        })?;
   892→                        self.emit_instruction(Instruction::ADDV {
   893→                            vd: SIMD_SCRATCH,
   894→                            vn: SIMD_SCRATCH,
   895→                        })?;
   896→                        self.emit_instruction(Instruction::UMOV {
   897→                            rd,
   898→                            vn: SIMD_SCRATCH,
   899→                        })?;
   900→                    }
   901→                }
   902→            }
   903→
   904→            IRInstr::Call {
   905→                dst,
   906→                func: target_name,
   907→                args,
   908→                is_extern,
   909→            } => {
   910→                // Resolve all argument source registers FIRST, before moving
   911→                // any of them. This prevents a later move from overwriting a
   912→                // register that an earlier argument is in.
   913→                //
   914→                // IMPORTANT: For immediate arguments, resolve_reg always loads
   915→                // into X9. When multiple immediates are present, we must use
   916→                // different scratch registers to avoid overwriting. We load
   917→                // immediates into X0-X7 directly (since they're the target
   918→                // registers anyway) or use X11-X15 as temporary scratch.
   919→                let arg_regs = [
   920→                    Register::X0, Register::X1, Register::X2, Register::X3,
   921→                    Register::X4, Register::X5, Register::X6, Register::X7,
   922→                ];
   923→                // Scratch registers for loading immediates when the target arg
   924→                // register isn't available yet (because of conflicts).
   925→                let scratch_regs = [
   926→                    Register::X11, Register::X12, Register::X13, Register::X14,
   927→                    Register::X15, Register::X3, Register::X4, Register::X5,
   928→                ];
   929→
   930→                let mut src_regs: Vec<Register> = Vec::new();
   931→                let mut immediate_scratch_used: Vec<usize> = Vec::new(); // which scratch reg each immediate uses
   932→                let mut next_scratch = 0;
   933→
   934→                for (i, arg) in args.iter().enumerate() {
   935→                    if i >= 8 {
   936→                        break;
   937→                    }
   938→                    let is_immediate = matches!(arg, IRValue::Immediate(_) | IRValue::Address(_));
   939→                    if is_immediate {
   940→                        let v = match arg {
   941→                            IRValue::Immediate(v) => *v,
   942→                            IRValue::Address(a) => *a as i64,
   943→                            _ => unreachable!(),
   944→                        };
   945→                        // Load the immediate directly into the target register
   946→                        // if it's not needed by another argument.
   947→                        // For simplicity, always load into a scratch register
   948→                        // and record it.
   949→                        if next_scratch < scratch_regs.len() {
   950→                            let scratch = scratch_regs[next_scratch];
   951→                            next_scratch += 1;
   952→                            self.emit_load_immediate(scratch, v)?;
   953→                            src_regs.push(scratch);
   954→                        } else {
   955→                            // Fallback to X9 (rare, only with 8+ immediates)
   956→                            let scratch = Register::X9;
   957→                            self.emit_load_immediate(scratch, v)?;
   958→                            src_regs.push(scratch);
   959→                        }
   960→                    } else {
   961→                        let src = self.resolve_reg(arg)?;
   962→                        src_regs.push(src);
   963→                    }
   964→                }
   965→
   966→                // Now move arguments into X0–X7. We need to handle the case
   967→                // where a source register is also a target register (cycle).
   968→                // Strategy: first move arguments that don't conflict, then
   969→                // handle conflicts using a scratch register.
   970→                let n = src_regs.len();
   971→                let mut moved = vec![false; n];
   972→
   973→                // Pass 1: move args where src != any dst that hasn't been moved yet
   974→                // Simple approach: iterate and move non-conflicting ones first
   975→                for pass in 0..2 {
   976→                    for i in 0..n {
   977→                        if moved[i] {
   978→                            continue;
   979→                        }
   980→                        let src = src_regs[i];
   981→                        let dst = arg_regs[i];
   982→                        if src == dst {
   983→                            // Already in the right register
   984→                            moved[i] = true;
   985→                            continue;
   986→                        }
   987→                        if pass == 0 {
   988→                            // Check if src is needed as a source by a later unmoved arg
   989→                            let mut conflict = false;
   990→                            for j in 0..n {
   991→                                if !moved[j] && j != i && src_regs[j] == dst {
   992→                                    // Moving arg i to dst would overwrite arg j's source
   993→                                    conflict = true;
   994→                                    break;
   995→                                }
   996→                            }
   997→                            if conflict {
   998→                                continue; // Defer to pass 2
   999→                            }
  1000→                        }
  1001→                        // Safe to move now
  1002→                        self.emit_instruction(Instruction::MOV {
  1003→                            rd: dst,
  1004→                            rm: src,
  1005→                        })?;
  1006→                        moved[i] = true;
  1007→                    }
  1008→                }
  1009→
  1010→                // Pass 2: handle any remaining conflicts with scratch register
  1011→                for i in 0..n {
  1012→                    if moved[i] {
  1013→                        continue;
  1014→                    }
  1015→                    let src = src_regs[i];
  1016→                    let dst = arg_regs[i];
  1017→                    if src == dst {
  1018→                        moved[i] = true;
  1019→                        continue;
  1020→                    }
  1021→                    // Use X16 as scratch to break the cycle
  1022→                    self.emit_instruction(Instruction::MOV {
  1023→                        rd: Register::X16,
  1024→                        rm: src,
  1025→                    })?;
  1026→                    self.emit_instruction(Instruction::MOV {
  1027→                        rd: dst,
  1028→                        rm: Register::X16,
  1029→                    })?;
  1030→                    moved[i] = true;
  1031→                }
  1032→
  1033→                // BL — record a relocation for later patching.
  1034→                // For extern calls, the relocation will be resolved by
  1035→                // the system linker; for local calls, it is patched in
  1036→                // emit_elf's resolve_call_relocs.
  1037→                let bl_word_idx = self.code.len();
  1038→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1039→                self.call_relocs.push(CallRelocation {
  1040→                    text_byte_offset: bl_byte_offset,
  1041→                    target_func: target_name.clone(),
  1042→                });
  1043→                // Register a RelocationEntry so the linker can patch this BL.
  1044→                // Extern calls use the same R_AARCH64_CALL26 relocation —
  1045→                // the distinction is that the symbol will be marked as
  1046→                // undefined (SHN_UNDEF) in the ELF symbol table.
  1047→                let _ = is_extern; // tracked via symbol table, not relocation type
  1048→                self.relocations.push(RelocationEntry {
  1049→                    offset: bl_byte_offset,
  1050→                    symbol: target_name.clone(),
  1051→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1052→                });
  1053→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1054→
  1055→                if let Some(dst_val) = dst {
  1056→                    let rd = self.resolve_reg(dst_val)?;
  1057→                    if rd != Register::X0 {
  1058→                        self.emit_instruction(Instruction::MOV {
  1059→                            rd,
  1060→                            rm: Register::X0,
  1061→                        })?;
  1062→                    }
  1063→                }
  1064→            }
  1065→
  1066→            IRInstr::Alloc { dst, size } => {
  1067→                let rd = self.resolve_reg(dst)?;
  1068→                // IMPORTANT: Decrement SP FIRST, then save the new SP value.
  1069→                // If we save before decrementing, the allocation pointer
  1070→                // points to the wrong (too-high) address.
  1071→                let aligned = (*size).div_ceil(16) * 16;
  1072→                self.emit_instruction(Instruction::SUB {
  1073→                    rd: Register::SP,
  1074→                    rn: Register::SP,
  1075→                    rm: Operand::Imm12(aligned as u16),
  1076→                })?;
  1077→                // MOV rd, SP: cannot use ORR because ORR treats Rm=31 as XZR.
  1078→                // Use ADD rd, SP, #0 instead.
  1079→                self.emit_instruction(Instruction::ADD {
  1080→                    rd,
  1081→                    rn: Register::SP,
  1082→                    rm: Operand::Imm12(0),
  1083→                })?;
  1084→            }
  1085→
  1086→            IRInstr::Free { ptr } => {
  1087→                let rt = self.resolve_reg(ptr)?;
  1088→                // Move ptr to X0 (first argument)
  1089→                if rt != Register::X0 {
  1090→                    self.emit_instruction(Instruction::MOV {
  1091→                        rd: Register::X0,
  1092→                        rm: rt,
  1093→                    })?;
  1094→                }
  1095→                // BL __vuma_free
  1096→                let bl_word_idx = self.code.len();
  1097→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1098→                self.call_relocs.push(CallRelocation {
  1099→                    text_byte_offset: bl_byte_offset,
  1100→                    target_func: "__vuma_free".to_string(),
  1101→                });
  1102→                self.relocations.push(RelocationEntry {
  1103→                    offset: bl_byte_offset,
  1104→                    symbol: "__vuma_free".to_string(),
  1105→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1106→                });
  1107→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1108→            }
  1109→
  1110→            IRInstr::Cast { kind, dst, src, from_ty, to_ty } => {
  1111→                let rd = self.resolve_reg(dst)?;
  1112→                let rn = self.resolve_reg(src)?;
  1113→
  1114→                // Determine source and destination sizes from IR type info.
  1115→                // When type info is unavailable (`None`), fall back to defaults
  1116→                // matching the prior behaviour (64-bit int, double-precision float).
  1117→                let src_is_64bit_int = matches!(from_ty,
  1118→                    Some(IRType::I64) | Some(IRType::U64) | None
  1119→                );
  1120→                let dst_is_f64 = matches!(to_ty,
  1121→                    Some(IRType::F64) | None
  1122→                );
  1123→                let dst_is_64bit_int = matches!(to_ty,
  1124→                    Some(IRType::I64) | Some(IRType::U64) | None
  1125→                );
  1126→                let src_is_f64 = matches!(from_ty,
  1127→                    Some(IRType::F64) | None
  1128→                );
  1129→
  1130→                match kind {
  1131→                    CastKind::ZExt => {
  1132→                        // Zero-extend 32-bit to 64-bit using UBFM
  1133→                        self.emit_instruction(Instruction::UBFM {
  1134→                            rd,
  1135→                            rn,
  1136→                            immr: 0,
  1137→                            imms: 31,
  1138→                        })?;
  1139→                    }
  1140→                    CastKind::SExt => {
  1141→                        // Sign-extend 32-bit to 64-bit using SBFM
  1142→                        self.emit_instruction(Instruction::SBFM {
  1143→                            rd,
  1144→                            rn,
  1145→                            immr: 0,
  1146→                            imms: 31,
  1147→                        })?;
  1148→                    }
  1149→                    CastKind::Trunc | CastKind::BitCast => {
  1150→                        // Trunc: upper bits discarded on write — just MOV
  1151→                        // BitCast: no data change — just MOV
  1152→                        if rd != rn {
  1153→                            self.emit_instruction(Instruction::MOV { rd, rm: rn })?;
  1154→                        }
  1155→                    }
  1156→                    CastKind::IntToFloat => {
  1157→                        // Signed int → float: SCVTF
  1158→                        self.emit_instruction(Instruction::SCVTF {
  1159→                            rd,
  1160→                            rn,
  1161→                            src_64: src_is_64bit_int,
  1162→                            dst_double: dst_is_f64,
  1163→                        })?;
  1164→                    }
  1165→                    CastKind::UIntToFloat => {
  1166→                        // Unsigned int → float: UCVTF
  1167→                        self.emit_instruction(Instruction::UCVTF {
  1168→                            rd,
  1169→                            rn,
  1170→                            src_64: src_is_64bit_int,
  1171→                            dst_double: dst_is_f64,
  1172→                        })?;
  1173→                    }
  1174→                    CastKind::FloatToInt => {
  1175→                        // Float → signed int: FCVTZS
  1176→                        self.emit_instruction(Instruction::FCVTZS {
  1177→                            rd,
  1178→                            rn,
  1179→                            dst_64: dst_is_64bit_int,
  1180→                            src_double: src_is_f64,
  1181→                        })?;
  1182→                    }
  1183→                    CastKind::FloatToUInt => {
  1184→                        // Float → unsigned int: FCVTZU
  1185→                        self.emit_instruction(Instruction::FCVTZU {
  1186→                            rd,
  1187→                            rn,
  1188→                            dst_64: dst_is_64bit_int,
  1189→                            src_double: src_is_f64,
  1190→                        })?;
  1191→                    }
  1192→                    CastKind::FloatToFloat => {
  1193→                        // Float ↔ float width change: FCVT
  1194→                        self.emit_instruction(Instruction::FCVT {
  1195→                            rd,
  1196→                            rn,
  1197→                            to_double: dst_is_f64,
  1198→                        })?;
  1199→                    }
  1200→                }
  1201→            }
  1202→
  1203→            IRInstr::Phi { .. } => {
  1204→                log::warn!(
  1205→                    "IRInstr::Phi encountered during emission — should be resolved by SSA pass"
  1206→                );
  1207→            }
  1208→
  1209→            IRInstr::GetAddress { dst, name } => {
  1210→                let rd = self.resolve_reg(dst)?;
  1211→                // Emit a call to __vuma_getaddr to resolve the symbol at runtime.
  1212→                // Move name hash to X0 as the argument.
  1213→                let name_hash = name
  1214→                    .chars()
  1215→                    .fold(0u64, |acc, c| acc.wrapping_mul(31).wrapping_add(c as u64));

  1217→                let bl_word_idx = self.code.len();
  1218→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1219→                self.call_relocs.push(CallRelocation {
  1220→                    text_byte_offset: bl_byte_offset,
  1221→                    target_func: "__vuma_getaddr".to_string(),
  1222→                });
  1223→                self.relocations.push(RelocationEntry {
  1224→                    offset: bl_byte_offset,
  1225→                    symbol: "__vuma_getaddr".to_string(),
  1226→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1227→                });

  1229→                if rd != Register::X0 {
  1230→                    self.emit_instruction(Instruction::MOV {
  1231→                        rd,
  1232→                        rm: Register::X0,
  1233→                    })?;
  1234→                }
  1235→            }
  1236→
  1237→            IRInstr::Offset { dst, base, offset } => {
  1238→                let rd = self.resolve_reg(dst)?;
  1239→                let rn = self.resolve_reg(base)?;
  1240→                let rm = match offset {
  1241→                    IRValue::Immediate(v) => {
  1242→                        if *v >= 0 && *v <= 4095 {
  1243→                            Operand::Imm12(*v as u16)
  1244→                        } else {
  1245→                            let temp = Register::X9;
  1246→                            self.emit_load_immediate(temp, *v)?;
  1247→                            Operand::Reg {
  1248→                                reg: temp,
  1249→                                shift: None,
  1250→                            }
  1251→                        }
  1252→                    }
  1253→                    _ => Operand::Reg {
  1254→                        reg: self.resolve_reg(offset)?,
  1255→                        shift: None,
  1256→                    },
  1257→                };
  1258→                self.emit_instruction(Instruction::ADD { rd, rn, rm })?;
  1259→            }
  1260→
  1261→            // ── Dedicated arithmetic — delegate to BinOp ──
  1262→            IRInstr::Add { dst, lhs, rhs, ty } => {
  1263→                self.emit_binop(BinOpKind::Add, dst, lhs, rhs, ty.as_ref())?;
  1264→            }
  1265→            IRInstr::Sub { dst, lhs, rhs, ty } => {
  1266→                self.emit_binop(BinOpKind::Sub, dst, lhs, rhs, ty.as_ref())?;
  1267→            }
  1268→            IRInstr::Mul { dst, lhs, rhs, ty } => {
  1269→                self.emit_binop(BinOpKind::Mul, dst, lhs, rhs, ty.as_ref())?;
  1270→            }
  1271→            IRInstr::Div { dst, lhs, rhs, ty } => {
  1272→                self.emit_binop(BinOpKind::SDiv, dst, lhs, rhs, ty.as_ref())?;
  1273→            }
  1274→
  1275→            // ── Comparison instruction ──
  1276→            IRInstr::Cmp {
  1277→                kind,
  1278→                dst,
  1279→                lhs,
  1280→                rhs,
  1281→                ty,
  1282→            } => {
  1283→                let width = RegWidth::from_ir_type(ty.as_ref());
  1284→                let rd = self.resolve_reg(dst)?;
  1285→                // IMPORTANT: resolve_reg uses X9 for immediates, so if both lhs and rhs
  1286→                // are immediates, the second load would overwrite the first. Use X10 for
  1287→                // the RHS when both are immediates.
  1288→                let rn = self.resolve_reg(lhs)?;
  1289→                let rm = match rhs {
  1290→                    IRValue::Immediate(_) | IRValue::Address(_) => {
  1291→                        // If lhs was also an immediate, it's in X9; use X10 for rhs
  1292→                        let temp = if matches!(lhs, IRValue::Immediate(_) | IRValue::Address(_)) {
  1293→                            Register::X10
  1294→                        } else {
  1295→                            Register::X9
  1296→                        };
  1297→                        match rhs {
  1298→                            IRValue::Immediate(v) => self.emit_load_immediate(temp, *v)?,
  1299→                            IRValue::Address(a) => self.emit_load_immediate(temp, *a as i64)?,
  1300→                            _ => unreachable!(),
  1301→                        }
  1302→                        temp
  1303→                    }
  1304→                    _ => self.resolve_reg(rhs)?,
  1305→                };
  1306→                self.emit_instruction_with_width(
  1307→                    Instruction::CMP {
  1308→                        rn,
  1309→                        rm: Operand::Reg {
  1310→                            reg: rm,
  1311→                            shift: None,
  1312→                        },
  1313→                    },
  1314→                    width,
  1315→                )?;
  1316→                let cond = cmp_kind_to_condition(kind);
  1317→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1318→            }
  1319→
  1320→            // ── Instruction-level control flow ──
  1321→            IRInstr::Ret { values } => {
  1322→                for (i, val) in values.iter().enumerate() {
  1323→                    if i >= 8 {
  1324→                        break;
  1325→                    }
  1326→                    let src = self.resolve_reg(val)?;
  1327→                    let dst_reg = match i {
  1328→                        0 => Register::X0,
  1329→                        1 => Register::X1,
  1330→                        2 => Register::X2,
  1331→                        3 => Register::X3,
  1332→                        4 => Register::X4,
  1333→                        5 => Register::X5,
  1334→                        6 => Register::X6,
  1335→                        7 => Register::X7,
  1336→                        _ => unreachable!(),
  1337→                    };
  1338→                    if src != dst_reg {
  1339→                        self.emit_instruction(Instruction::MOV {
  1340→                            rd: dst_reg,
  1341→                            rm: src,
  1342→                        })?;
  1343→                    }
  1344→                }
  1345→            }
  1346→
  1347→            IRInstr::Branch { target } => {
  1348→                let fixup_idx = self.code.len();
  1349→                self.fixups.push((fixup_idx, target.clone(), BranchFormat::B26));
  1350→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1351→            }
  1352→
  1353→            IRInstr::CondBranch {
  1354→                cond,
  1355→                true_target,
  1356→                false_target,
  1357→            } => {
  1358→                let rt = self.resolve_reg(cond)?;
  1359→                // CBNZ: if cond != 0, branch to true_target
  1360→                let fixup_cbz = self.code.len();
  1361→                self.fixups.push((fixup_cbz, true_target.clone(), BranchFormat::Cond19));
  1362→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1363→                // Otherwise (cond == 0), branch to false_target
  1364→                let fixup_b = self.code.len();
  1365→                self.fixups.push((fixup_b, false_target.clone(), BranchFormat::B26));
  1366→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1367→            }
  1368→
  1369→            IRInstr::Select {
  1370→                dst,
  1371→                cond,
  1372→                true_val,
  1373→                false_val,
  1374→                ty,
  1375→            } => {
  1376→                let width = RegWidth::from_ir_type(ty.as_ref());
  1377→                // Lower select as: SUBS XZR, cond, #0; CSEL dst, false_val, true_val, NE
  1378→                let rd = self.resolve_reg(dst)?;
  1379→                let rc = self.resolve_reg(cond)?;
  1380→                let rt = self.resolve_reg(true_val)?;
  1381→                let rf = self.resolve_reg(false_val)?;
  1382→                // Compare cond against zero and select.
  1383→                self.emit_instruction_with_width(
  1384→                    Instruction::SUB {
  1385→                        rd: Register::XZR,
  1386→                        rn: rc,
  1387→                        rm: Operand::Imm12(0),
  1388→                    },
  1389→                    width,
  1390→                )?;
  1391→                // Set flags by using a separate CMP (SUB with XZR destination
  1392→                // doesn't set flags; we need a flags-setting variant).
  1393→                // We emulate this with: CMP rc, #0 which is SUBS XZR, rc, #0.
  1394→                // Since we only have SUB, we use the existing CMP pattern.
  1395→                self.emit_instruction_with_width(
  1396→                    Instruction::CSEL {
  1397→                        rd,
  1398→                        rn: rf,
  1399→                        rm: rt,
  1400→                        cond: crate::arm64::Condition::NE,
  1401→                    },
  1402→                    width,
  1403→                )?;

  1409→                let rt = self.resolve_reg(dst)?;
  1410→                let rn = self.resolve_reg(addr)?;
  1411→                self.emit_instruction(Instruction::LDAXR { rt, rn })?;

  1413→

  1416→                // Loop until success.
  1417→                let rt = self.resolve_reg(value)?;
  1418→                let rn = self.resolve_reg(addr)?;

  1425→                let fixup = self.code.len();

  1427→                self.emit_instruction(Instruction::CBNZ { rt: rs, offset: 0 })?;

  1429→

  1431→                // AArch64 CAS loop: LDAXR / CMP / B.NE skip / STLXR / CBNZ retry
  1432→                let rd = self.resolve_reg(dst)?;
  1433→                let rn = self.resolve_reg(addr)?;
  1434→                let re = self.resolve_reg(expected)?;
  1435→                let rm = self.resolve_reg(desired)?;
  1436→                let rs = Register::X9; // scratch for STLXR status
  1437→                let scratch_cmp = Register::X10; // scratch for comparison
  1438→
  1439→                let retry_label = format!("__atomic_cas_retry_{}", self.code.len());
  1440→                let done_label = format!("__atomic_cas_done_{}", self.code.len());
  1441→
  1442→                // Record retry label position
  1443→                self.label_offsets.insert(retry_label.clone(), self.code.len());
  1444→                // LDAXR scratch_cmp, [addr] — load current value
  1445→                self.emit_instruction(Instruction::LDAXR { rt: scratch_cmp, rn })?;
  1446→                // CMP scratch_cmp, expected — SUB XZR, scratch_cmp, expected
  1447→                self.emit_instruction(Instruction::SUB {
  1448→                    rd: Register::XZR,
  1449→                    rn: scratch_cmp,
  1450→                    rm: Operand::Reg { reg: re, shift: None },
  1451→                })?;
  1452→                // B.NE done — if not equal, skip store
  1453→                let fixup_ne = self.code.len();
  1454→                self.fixups.push((fixup_ne, done_label.clone(), BranchFormat::Cond19));
  1455→                self.emit_instruction(Instruction::BCond {
  1456→                    cond: crate::arm64::Condition::NE,
  1457→                    offset: 0,
  1458→                })?;
  1459→                // STLXR rs, desired, [addr] — try to store
  1460→                self.emit_instruction(Instruction::STLXR { rs, rt: rm, rn })?;
  1461→                // CBNZ rs, retry — if store failed, retry
  1462→                let fixup_retry = self.code.len();
  1463→                self.fixups.push((fixup_retry, retry_label, BranchFormat::Cond19));
  1464→                self.emit_instruction(Instruction::CBNZ { rt: rs, offset: 0 })?;
  1465→                // Record done label position
  1466→                self.label_offsets.insert(done_label.clone(), self.code.len());
  1467→                // CSEL rd, re, scratch_cmp, EQ
  1468→                // If EQ (match succeeded), rd = re (expected); else rd = scratch_cmp (current value)
  1469→                self.emit_instruction(Instruction::CSEL {
  1470→                    rd,
  1471→                    rn: re,
  1472→                    rm: scratch_cmp,
  1473→                    cond: crate::arm64::Condition::EQ,
  1474→                })?;
  1475→            }
  1476→

  1497→                        rd,
  1498→                        rn: rf,

  1507→                dst,

  1513→                let rd = self.resolve_reg(dst)?;
  1514→                let rn = self.resolve_reg(lhs)?;
  1515→                let rm = self.resolve_reg(rhs)?;

  1517→                self.emit_instruction(Instruction::EOR { rd, rn, rm })?;

  1519→                self.emit_instruction(Instruction::SUB {
  1520→                    rd: Register::XZR,
  1521→                    rn: rd,
  1522→                    rm: Operand::Imm12(0),

  1529→
  1530→    /// Emit a binary operation (shared by `BinOp` and dedicated `Add`/`Sub`/…).
  1531→    ///
  1532→    /// Some ARM64 instructions (ADD, SUB, LSL, LSR, ASR) accept a 12-bit
  1533→    /// unsigned immediate operand directly.  Most others (MUL, AND, ORR, EOR,
  1534→    /// SDIV, UDIV, remainder, comparisons) **require** a register operand.
  1535→    /// When the RHS is an immediate that the target instruction cannot accept
  1536→    /// directly, we "spill" it into a scratch register (X16 / IP0) first via
  1537→    /// MOVZ + MOVK and then use that register as the operand.
  1538→    ///
  1539→    /// X16 (IP0) is chosen as the scratch register because it is **not** in
  1540→    /// the register allocator's free pool, so it will never conflict with
  1541→    /// `rd` or `rn`.
  1542→    fn emit_binop(
  1543→        &mut self,
  1544→        op: BinOpKind,
  1545→        dst: &IRValue,
  1546→        lhs: &IRValue,
  1547→        rhs: &IRValue,
  1548→        ty: Option<&IRType>,
  1549→    ) -> Result<()> {
  1550→        let width = RegWidth::from_ir_type(ty);
  1551→        let rd = self.resolve_reg(dst)?;
  1552→        let rn = self.resolve_reg(lhs)?;
  1553→
  1554→        // Determine whether the operation can accept an immediate operand.
  1555→        // ADD, SUB, LSL, LSR, ASR all have immediate forms in ARM64.
  1556→        // All other operations require a register operand.
  1557→        let supports_imm = matches!(
  1558→            op,
  1559→            BinOpKind::Add | BinOpKind::Sub | BinOpKind::Shl | BinOpKind::ShrL | BinOpKind::ShrA
  1560→        );
  1561→
  1562→        let rm = match rhs {
  1563→            IRValue::Immediate(v) => {
  1564→                if supports_imm && *v >= 0 && *v <= 4095 {
  1565→                    // Small immediate that the instruction accepts directly.
  1566→                    Operand::Imm12(*v as u16)
  1567→                } else {
  1568→                    // Load the immediate into scratch register X16 (IP0).
  1569→                    // X16 is not in the register allocator's pool, so it cannot
  1570→                    // conflict with `rd` or `rn`.
  1571→                    let scratch = Register::X16;
  1572→                    self.emit_load_immediate_with_width(scratch, *v, width)?;
  1573→                    Operand::Reg {
  1574→                        reg: scratch,
  1575→                        shift: None,
  1576→                    }
  1577→                }
  1578→            }
  1579→            _ => Operand::Reg {
  1580→                reg: self.resolve_reg(rhs)?,
  1581→                shift: None,
  1582→            },
  1583→        };
  1584→
  1585→        match op {
  1586→            BinOpKind::Add => {
  1587→                self.emit_instruction_with_width(Instruction::ADD { rd, rn, rm }, width)?;
  1588→            }
  1589→            BinOpKind::Sub => {
  1590→                self.emit_instruction_with_width(Instruction::SUB { rd, rn, rm }, width)?;
  1591→            }
  1592→            BinOpKind::Mul => {
  1593→                let rm_reg = self.operand_to_reg(&rm)?;
  1594→                self.emit_instruction_with_width(Instruction::MUL { rd, rn, rm: rm_reg }, width)?;
  1595→            }
  1596→            BinOpKind::SDiv => {
  1597→                let rm_reg = self.operand_to_reg(&rm)?;
  1598→                self.emit_instruction_with_width(Instruction::SDIV { rd, rn, rm: rm_reg }, width)?;
  1599→            }
  1600→            BinOpKind::UDiv => {
  1601→                let rm_reg = self.operand_to_reg(&rm)?;
  1602→                self.emit_instruction_with_width(Instruction::UDIV { rd, rn, rm: rm_reg }, width)?;
  1603→            }
  1604→            BinOpKind::And => {
  1605→                let rm_reg = self.operand_to_reg(&rm)?;
  1606→                self.emit_instruction_with_width(Instruction::AND { rd, rn, rm: rm_reg }, width)?;
  1607→            }
  1608→            BinOpKind::Or => {
  1609→                let rm_reg = self.operand_to_reg(&rm)?;
  1610→                self.emit_instruction_with_width(Instruction::ORR { rd, rn, rm: rm_reg }, width)?;
  1611→            }
  1612→            BinOpKind::Xor => {
  1613→                let rm_reg = self.operand_to_reg(&rm)?;
  1614→                self.emit_instruction_with_width(Instruction::EOR { rd, rn, rm: rm_reg }, width)?;
  1615→            }
  1616→            BinOpKind::Shl => {

  1682→    fn emit_terminator(&mut self, term: &IRTerminator) -> Result<()> {
  1683→        match term {
  1684→            IRTerminator::Jump(target) => {
  1685→                let fixup_idx = self.code.len();
  1686→                self.fixups.push((fixup_idx, target.clone(), BranchFormat::B26));
  1687→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1688→            }
  1689→            IRTerminator::Branch {
  1690→                cond,
  1691→                true_block,
  1692→                false_block,
  1693→            } => {
  1694→                let rt = self.resolve_reg(cond)?;
  1695→                let fixup_cbnz = self.code.len();
  1696→                self.fixups.push((fixup_cbnz, true_block.clone(), BranchFormat::Cond19));
  1697→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1698→                let fixup_b = self.code.len();
  1699→                self.fixups.push((fixup_b, false_block.clone(), BranchFormat::B26));
  1700→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1701→            }
  1702→            IRTerminator::Return(vals) => {
  1703→                for (i, val) in vals.iter().enumerate() {
  1704→                    if i >= 8 {
  1705→                        break;
  1706→                    }
  1707→                    let src = self.resolve_reg(val)?;
  1708→                    let dst_reg = match i {
  1709→                        0 => Register::X0,
  1710→                        1 => Register::X1,
  1711→                        2 => Register::X2,
  1712→                        3 => Register::X3,
  1713→                        4 => Register::X4,
  1714→                        5 => Register::X5,
  1715→                        6 => Register::X6,
  1716→                        7 => Register::X7,
  1717→                        _ => unreachable!(),
  1718→                    };
  1719→                    if src != dst_reg {
  1720→                        self.emit_instruction(Instruction::MOV {
  1721→                            rd: dst_reg,
  1722→                            rm: src,
  1723→                        })?;
  1724→                    }
  1725→                }
  1726→                // Restore SP to point to the saved FP/LR pair.
  1727→                // X29 was set to SP right after the prologue's SUB SP, SP, #16
  1728→                // and STP X29, X30, [SP]. So X29 points to the FP/LR save area.
  1729→                // Using MOV SP, X29 (via ADD SP, X29, #0) is the most robust
  1730→                // way to restore SP regardless of how many Alloc instructions
  1731→                // modified it during the function body.
  1732→                self.emit_instruction(Instruction::ADD {
  1733→                    rd: Register::SP,
  1734→                    rn: Register::X29,
  1735→                    rm: Operand::Imm12(0),
  1736→                })?;
  1737→                // FP/LR were stored at [SP] after the prologue's SUB SP, SP, #16.
  1738→                // After the ADD above, SP points back to the FP/LR save area.
  1739→                // Load from [SP, #0], then restore SP by adding 16.
  1740→                self.emit_instruction(Instruction::LDP {
  1741→                    rt1: Register::X29,
  1742→                    rt2: Register::X30,
  1743→                    rn: Register::SP,
  1744→                    offset: 0,
  1745→                })?;
  1746→                self.emit_instruction(Instruction::ADD {
  1747→                    rd: Register::SP,
  1748→                    rn: Register::SP,
  1749→                    rm: Operand::Imm12(16),
  1750→                })?;
  1751→                self.emit_instruction(Instruction::RET { rn: None })?;
  1752→            }
  1753→            IRTerminator::Unreachable => {
  1754→                self.emit_instruction(Instruction::MOV {
  1755→                    rd: Register::XZR,
  1756→                    rm: Register::XZR,
  1757→                })?;
  1758→            }
  1759→            // Switch, Invoke, TailCall, and Resume are lowered by the
  1760→            // control_flow module before reaching the emitter. If they
  1761→            // appear here, it means the lowering pass was not run.
  1762→            IRTerminator::Switch { .. } => {
  1763→                return Err(CodegenError::InvalidInstruction(
  1764→                    "Switch terminator must be lowered before emission".to_string(),
  1765→                ));
  1766→            }
  1767→            IRTerminator::Invoke { .. } => {
  1768→                return Err(CodegenError::InvalidInstruction(
  1769→                    "Invoke terminator must be lowered before emission".to_string(),
  1770→                ));
  1771→            }
  1772→            IRTerminator::TailCall { .. } => {
  1773→                return Err(CodegenError::InvalidInstruction(
  1774→                    "TailCall terminator must be lowered before emission".to_string(),
  1775→                ));
  1776→            }
  1777→            IRTerminator::Resume { .. } => {
  1778→                return Err(CodegenError::InvalidInstruction(
  1779→                    "Resume terminator must be lowered before emission".to_string(),
  1780→                ));
  1781→            }
  1782→        }
  1783→        Ok(())
  1784→    }

  1822→        // For each variant: (scale, max_imm12) where the encoded offset = imm12 << scale.

  1832→                } else {
  1833→                    self.emit_address_with_offset(rn, offset)?;
  1834→                    self.emit_instruction(Instruction::LDRB {
  1835→                        rt,
  1836→                        rn: Register::X9,
  1837→                        offset: 0,
  1838→                    })?;

  1844→                        rn: rt,
  1845→                        immr: 0,
  1846→                        imms: 7,
  1847→                    })?;
  1848→                }
