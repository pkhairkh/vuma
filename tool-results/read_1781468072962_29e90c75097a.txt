    14в†’    14в†’//! ## LoongArch64 Register Convention (LP64 ABI)

    17в†’    17в†’//! |-------------|----------|-----------------------------------|

    19в†’    19в†’//! | r1          | ra       | Return address                    |
    20в†’    20в†’//! | r2          | tp       | Thread pointer                    |

    24в†’    24в†’//! | r21         | вЂ”        | Reserved                          |
    25в†’    25в†’//! | r22         | fp       | Frame pointer (callee-saved)      |
    26в†’    26в†’//! | r23вЂ“r31     | s0вЂ“s8    | Callee-saved                      |
    27в†’    27в†’//!
    28в†’    28в†’//! ## LoongArch64 FP Register Convention (LP64 ABI)
    29в†’    29в†’//!
    30в†’    30в†’//! | Register(s) | ABI Name | Role                              |
    31в†’    31в†’//! |-------------|----------|-----------------------------------|
    32в†’    32в†’//! | f0вЂ“f7       | fa0вЂ“fa7  | FP argument / return registers    |
    33в†’    33в†’//! | f8вЂ“f23      | ft0вЂ“ft15 | Caller-saved temporaries          |
    34в†’    34в†’//! | f24вЂ“f31     | fs0вЂ“fs7  | Callee-saved                      |

   327в†’   327в†’/// Encode a 2R format instruction.
   328в†’   328в†’///
   329в†’   329в†’/// Format: `opcode[31:10] | rj[9:5] | rd[4:0]`
   330в†’   330в†’fn encode_2r(opcode: u32, rj: u32, rd: u32) -> [u8; 4] {
   331в†’   331в†’    let word = ((opcode & 0x3FF_FFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   332в†’   332в†’    word.to_le_bytes()
   333в†’   333в†’}
   334в†’   334в†’
   335в†’   335в†’/// Encode a 3R format instruction.
   336в†’   336в†’///
   337в†’   337в†’/// Format: `opcode[31:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   338в†’   338в†’fn encode_3r(opcode: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {

   340в†’   340в†’    word.to_le_bytes()
   341в†’   341в†’}

   350в†’   350в†’        | ((rk & 0x1F) << 10)
   351в†’   351в†’        | ((rj & 0x1F) << 5)
   352в†’   352в†’        | (rd & 0x1F);
   353в†’   353в†’    word.to_le_bytes()

   355в†’   355в†’
   356в†’   356в†’/// Encode a 2RI8 format instruction.
   357в†’   357в†’///
   358в†’   358в†’/// Format: `opcode[31:22] | I8[21:14] | rj[9:5] | rd[4:0]`

  1591в†’  1591в†’/// Build a minimal ELF64 binary for LoongArch64 from raw code bytes.
  1592в†’  1592в†’///
  1593в†’  1593в†’/// Produces a static executable with two LOAD segments:
  1594в†’  1594в†’///   1. PF_R | PF_X вЂ” .text (code)
  1595в†’  1595в†’///   2. PF_R | PF_W вЂ” .data / stack (writable memory)
  1596в†’  1596в†’fn build_loongarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1597в†’  1597в†’    const PAGE_SIZE: u64 = 0x10000; // 64 KB (LoongArch64 typical page size)
  1598в†’  1598в†’
  1599в†’  1599в†’    let elf_header_size: u64 = 64;
  1600в†’  1600в†’    let phdr_size: u64 = 56;
  1601в†’  1601в†’    let num_phdrs: u64 = 2;
  1602в†’  1602в†’    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1603в†’  1603в†’    // Page-align the text segment start in the file.
  1604в†’  1604в†’    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1605в†’  1605в†’    let text_size = code.len() as u64;
  1606в†’  1606в†’
  1607в†’  1607в†’    // The data segment starts on the next page after the text.
  1608в†’  1608в†’    let text_file_end = text_offset + text_size;
  1609в†’  1609в†’    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1610в†’  1610в†’    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1611в†’  1611в†’    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1612в†’  1612в†’    let entry_point = base_addr + text_offset;
  1613в†’  1613в†’
  1614в†’  1614в†’    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1615в†’  1615в†’
  1616в†’  1616в†’    // --- e_ident ---
  1617в†’  1617в†’    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1618в†’  1618в†’    elf.push(2); // ELFCLASS64
  1619в†’  1619в†’    elf.push(1); // ELFDATA2LSB
  1620в†’  1620в†’    elf.push(1); // EV_CURRENT
  1621в†’  1621в†’    elf.push(3); // ELFOSABI_LINUX
  1622в†’  1622в†’    elf.push(0); // padding
  1623в†’  1623в†’    elf.extend_from_slice(&[0u8; 7]); // padding
  1624в†’  1624в†’
  1625в†’  1625в†’    // --- ELF header fields ---
  1626в†’  1626в†’    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1627в†’  1627в†’    elf.extend_from_slice(&258u16.to_le_bytes()); // e_machine = EM_LOONGARCH
  1628в†’  1628в†’    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1629в†’  1629в†’    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1630в†’  1630в†’    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1631в†’  1631в†’    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1632в†’  1632в†’    elf.extend_from_slice(&0x43u32.to_le_bytes()); // e_flags = 0x43 (LP64D ABI double-float)
  1633в†’  1633в†’    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1634в†’  1634в†’    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1635в†’  1635в†’    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1636в†’  1636в†’    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1637в†’  1637в†’    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1638в†’  1638в†’    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1639в†’  1639в†’
  1640в†’  1640в†’    // --- Program Header 1: LOAD (PF_R | PF_X) вЂ” .text ---
  1641в†’  1641в†’    // p_filesz = actual code size, p_memsz = page-aligned (QEMU needs this for mapping)
  1642в†’  1642в†’    let text_memsz = ((text_size + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1643в†’  1643в†’    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1644в†’  1644в†’    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1645в†’  1645в†’    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1646в†’  1646в†’    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1647в†’  1647в†’    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1648в†’  1648в†’    elf.extend_from_slice(&text_memsz.to_le_bytes()); // p_filesz (cover full page for QEMU)
  1649в†’  1649в†’    elf.extend_from_slice(&text_memsz.to_le_bytes()); // p_memsz
  1650в†’  1650в†’    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1651в†’  1651в†’
  1652в†’  1652в†’    // --- Program Header 2: LOAD (PF_R | PF_W) вЂ” .data / stack ---
  1653в†’  1653в†’    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1654в†’  1654в†’    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1655в†’  1655в†’    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1656в†’  1656в†’    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1657в†’  1657в†’    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1658в†’  1658в†’    elf.extend_from_slice(&data_size.to_le_bytes()); // p_filesz (write zeros so QEMU can mmap)
  1659в†’  1659в†’    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1660в†’  1660в†’    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1661в†’  1661в†’
  1662в†’  1662в†’    // --- .text section ---
  1663в†’  1663в†’    // Pad to page-aligned text_offset
  1664в†’  1664в†’    while (elf.len() as u64) < text_offset {
  1665в†’  1665в†’        elf.push(0);
  1666в†’  1666в†’    }
  1667в†’  1667в†’    elf.extend_from_slice(code);
  1668в†’  1668в†’
  1669в†’  1669в†’    // --- Pad to data segment offset and write data segment ---
  1670в†’  1670в†’    while (elf.len() as u64) < data_offset {
  1671в†’  1671в†’        elf.push(0);
  1672в†’  1672в†’    }
  1673в†’  1673в†’    // Write data segment (all zeros, but must exist in file for QEMU mmap)
  1674в†’  1674в†’    elf.extend_from_slice(&vec![0u8; data_size as usize]);
  1675в†’  1675в†’
  1676в†’  1676в†’    elf
  1677в†’  1677в†’}

  1683в†’  1683в†’/// Decode a single LoongArch64 32-bit instruction word into a mnemonic string.
  1684в†’  1684в†’fn decode_loongarch64_instruction(word: u32) -> String {
  1685в†’  1685в†’    let opcode = word & 0x7f;
  1686в†’  1686в†’    match opcode {
  1687в†’  1687в†’        0x00 => {
  1688в†’  1688в†’            let rd = (word >> 7) & 0x1f;
  1689в†’  1689в†’            let rj = (word >> 15) & 0x1f;
  1690в†’  1690в†’            let rk = (word >> 10) & 0x1f;

  1692в†’  1692в†’        }
  1693в†’  1693в†’        0x01 => {
  1694в†’  1694в†’            let rd = (word >> 7) & 0x1f;
  1695в†’  1695в†’            let rj = (word >> 15) & 0x1f;
  1696в†’  1696в†’            let rk = (word >> 10) & 0x1f;
  1697в†’  1697в†’            format!("sub.w $r{}, $r{}, $r{}", rd, rj, rk)
  1698в†’  1698в†’        }
  1699в†’  1699в†’        0x02 => {
  1700в†’  1700в†’            let rd = (word >> 7) & 0x1f;
  1701в†’  1701в†’            let rj = (word >> 15) & 0x1f;
  1702в†’  1702в†’            let rk = (word >> 10) & 0x1f;
  1703в†’  1703в†’            format!("mul.w $r{}, $r{}, $r{}", rd, rj, rk)
  1704в†’  1704в†’        }
  1705в†’  1705в†’        0x05 => {
  1706в†’  1706в†’            let rd = (word >> 7) & 0x1f;

  1710в†’  1710в†’        }

  1712в†’  1712в†’            let rd = (word >> 7) & 0x1f;

  1714в†’  1714в†’            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1715в†’  1715в†’            format!("ld.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1716в†’  1716в†’        }
  1717в†’  1717в†’        0x0a => {
  1718в†’  1718в†’            let rd = (word >> 7) & 0x1f;
  1719в†’  1719в†’            let rj = (word >> 15) & 0x1f;
  1720в†’  1720в†’            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1721в†’  1721в†’            format!("st.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1722в†’  1722в†’        }
  1723в†’  1723в†’        0x0c => {
  1724в†’  1724в†’            let rd = (word >> 7) & 0x1f;
  1725в†’  1725в†’            let rj = (word >> 15) & 0x1f;
  1726в†’  1726в†’            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1727в†’  1727в†’            format!("ld.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1728в†’  1728в†’        }
  1729в†’  1729в†’        0x0e => {
  1730в†’  1730в†’            let rd = (word >> 7) & 0x1f;
  1731в†’  1731в†’            let rj = (word >> 15) & 0x1f;
  1732в†’  1732в†’            let si12 = ((word >> 10) as i32) << 20 >> 20;

  1734в†’  1734в†’        }
  1735в†’  1735в†’        0x10 => {
  1736в†’  1736в†’            let rj = (word >> 15) & 0x1f;
  1737в†’  1737в†’            let offs = ((word >> 10) as i32) << 12 >> 10;

  1740в†’  1740в†’        0x11 => {
  1741в†’  1741в†’            let rj = (word >> 15) & 0x1f;

  1743в†’  1743в†’            format!("bne $r{}, {}({})", rj, offs, offs)
  1744в†’  1744в†’        }

  1775в†’  1775в†’///

  1778в†’  1778в†’fn loongarch64_compute_frame_size(func: &IRFunction) -> usize {

  1780в†’  1780в†’    for block in &func.blocks {

  1783в†’  1783в†’                let aligned = (*size).div_ceil(16) * 16;
  1784в†’  1784в†’                total += aligned;
  1785в†’  1785в†’            }

  1869в†’  1869в†’/// Load a 64-bit immediate into a register using the lu12i.w/ori/lu32i.d/lu52i.d
  1870в†’  1870в†’/// instruction sequence.
  1871в†’  1871в†’///
  1872в†’  1872в†’/// Returns the list of AllocatedInstructions emitted to perform the load.
  1873в†’  1873в†’fn load_imm_la64(rd: Gpr, imm: i64) -> Vec<AllocatedInstruction> {
  1874в†’  1874в†’    let mut result = Vec::new();
  1875в†’  1875в†’    let val = imm as u64;
  1876в†’  1876в†’
  1877в†’  1877в†’    // lu12i.w rd, bits[31:12]
  1878в†’  1878в†’    let hi20 = ((val >> 12) & 0xFFFFF) as i32;
  1879в†’  1879в†’    result.push(emit_alloc_instr(
  1880в†’  1880в†’        Instruction::Lu12iW { rd, imm20: hi20 },
  1881в†’  1881в†’        vec![],
  1882в†’  1882в†’        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1883в†’  1883в†’    ));
  1884в†’  1884в†’
  1885в†’  1885в†’    // ori rd, rd, bits[11:0]
  1886в†’  1886в†’    let lo12 = (val & 0xFFF) as u32;
  1887в†’  1887в†’    result.push(emit_alloc_instr(
  1888в†’  1888в†’        Instruction::Ori {
  1889в†’  1889в†’            rd,
  1890в†’  1890в†’            rj: rd,
  1891в†’  1891в†’            imm12: lo12,
  1892в†’  1892в†’        },
  1893в†’  1893в†’        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1894в†’  1894в†’        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1895в†’  1895в†’    ));
  1896в†’  1896в†’
