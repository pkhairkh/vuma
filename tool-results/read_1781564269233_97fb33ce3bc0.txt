     1→//! # ARM64 (AArch64) Instruction Definitions & Instruction Selection
     2→//!
     3→//! Defines the ARM64 register set, condition codes, instruction
     4→//! representations, addressing modes, and the instruction selector that maps
     5→//! SCG/IR node types to ARM64 instructions for the AArch64
     6→//! (Cortex-A76, ARMv8.2-A).
     7→//!
     8→//! ## Instruction Selection (SCG → ARM64)
     9→//!
    10→//! | SCG / IR Node          | ARM64 Instructions                                  |
    11→//! |------------------------|-----------------------------------------------------|
    12→//! | Computation(add)       | ADD, SUB, MUL, SDIV, UDIV                          |
    13→//! | Computation(cmp)       | CMP, CSEL                                           |
    14→//! | Computation(bitwise)   | AND, ORR, EOR, LSL, LSR, ASR                       |
    15→//! | Allocation             | SUB SP, SP, #size / BL malloc                      |
    16→//! | Deallocation           | ADD SP, SP, #size / BL free                        |
    17→//! | Access(read)           | LDR, LDRB, LDRH, LDRSW                             |
    18→//! | Access(write)          | STR, STRB, STRH                                    |
    19→//! | Cast                   | MOV (no-op) / FCVT / SCVTF / FCVTZS / SXTW        |
    20→//! | ControlFlow            | B, B.cond, CBZ, CBNZ, TBZ, TBNZ                   |
    21→//!
    22→//! ## AAPCS64 Calling Convention
    23→//!
    24→//! | Register(s) | Role                                  |
    25→//! |-------------|---------------------------------------|
    26→//! | X0–X7       | Argument / result registers            |
    27→//! | X8          | Indirect result location register      |
    28→//! | X9–X15      | Caller-saved temporary registers       |
    29→//! | X16–X17     | Intra-procedure-call scratch (IP0/IP1) |
    30→//! | X18         | Platform register                      |
    31→//! | X19–X28     | Callee-saved registers                 |
    32→//! | X29         | Frame pointer (FP)                     |
    33→//! | X30         | Link register (LR)                     |
    34→//! | SP          | Stack pointer                          |
    35→//! | XZR         | Zero register                          |
    36→//!
    37→//! ## References
    38→//!
    39→//! - ARM Architecture Reference Manual ARMv8, for ARMv8-A architecture profile
    40→//! - <https://developer.arm.com/documentation/ddi0487/latest>
    41→

    46→// ---------------------------------------------------------------------------
    47→// Register Width
    48→// ---------------------------------------------------------------------------
    49→
    50→/// ARM64 register width — selects between 64-bit X registers and 32-bit W
    51→/// sub-registers.
    52→///
    53→/// On AArch64, the 5-bit register encoding is the same for Xn and Wn; the
    54→/// instruction's **sf** bit (bit 31) or **size** field (bits 31:30) selects
    55→/// the operand width. Using W registers gives automatic 32-bit wrapping
    56→/// arithmetic, which is required for algorithms like SHA-256 that operate on
    57→/// `u32` values.
    58→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
    59→pub enum RegWidth {
    60→    /// 32-bit W registers (W0–W30, WZR, WSP).
    61→    W32,
    62→    /// 64-bit X registers (X0–X30, XZR, SP) — default.
    63→    X64,
    64→}
    65→
    66→impl RegWidth {
    67→    /// Returns `true` if this is a 32-bit register width.
    68→    pub fn is_32bit(&self) -> bool {
    69→        matches!(self, RegWidth::W32)
    70→    }
    71→
    72→    /// Returns the **sf** bit value for this width (bit 31 of the encoding).
    73→    ///
    74→    /// - `X64` → 1
    75→    /// - `W32` → 0
    76→    pub fn sf_bit(&self) -> u32 {
    77→        match self {
    78→            RegWidth::X64 => 1,
    79→            RegWidth::W32 => 0,
    80→        }
    81→    }
    82→
    83→    /// Returns the register size in bits (64 or 32).
    84→    pub fn bits(&self) -> u32 {
    85→        match self {
    86→            RegWidth::X64 => 64,
    87→            RegWidth::W32 => 32,
    88→        }
    89→    }
    90→
    91→    /// Returns the register size in bytes (8 or 4).
    92→    pub fn bytes(&self) -> u32 {
    93→        self.bits() / 8
    94→    }
    95→
    96→    /// Returns the log2 scale for unsigned-offset load/store encoding.
    97→    ///
    98→    /// - `X64` → 3 (offset is divided by 8 for 64-bit LDR/STR)
    99→    /// - `W32` → 2 (offset is divided by 4 for 32-bit LDR/STR)
   100→    pub fn scale(&self) -> u32 {
   101→        match self {
   102→            RegWidth::X64 => 3,
   103→            RegWidth::W32 => 2,
   104→        }
   105→    }
   106→
   107→    /// Returns the mask value used in UBFM/SBFM for the `imms` field
   108→    /// (register size - 1).
   109→    pub fn size_minus_1(&self) -> u32 {
   110→        match self {
   111→            RegWidth::X64 => 63,
   112→            RegWidth::W32 => 31,
   113→        }
   114→    }
   115→
   116→    /// Derive the register width from an optional IR type.
   117→    ///
   118→    /// Returns `W32` for `I32`/`U32` types, `X64` otherwise (including `None`).
   119→    pub fn from_ir_type(ty: Option<&IRType>) -> RegWidth {
   120→        match ty {
   121→            Some(IRType::I32) | Some(IRType::U32) => RegWidth::W32,
   122→            _ => RegWidth::X64,
   123→        }
   124→    }
   125→}
   126→
   127→impl Default for RegWidth {
   128→    fn default() -> Self {
   129→        RegWidth::X64
   130→    }
   131→}
   132→
   133→impl std::fmt::Display for RegWidth {
   134→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   135→        match self {
   136→            RegWidth::W32 => write!(f, "w32"),
   137→            RegWidth::X64 => write!(f, "x64"),
   138→        }
   139→    }
   140→}

   432→/// ARM64 condition codes used in conditional branches and instructions.
   433→///
   434→/// Each condition code tests the NZCV flags set by a preceding comparison or
   435→/// arithmetic instruction.
   436→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
   437→pub enum Condition {
   438→    /// Equal / zero (Z == 1)
   439→    EQ,
   440→    /// Not equal / not zero (Z == 0)
   441→    NE,
   442→    /// Carry set / unsigned higher or same (C == 1)
   443→    CS,
   444→    /// Carry clear / unsigned lower (C == 0)
   445→    CC,
   446→    /// Minus / negative (N == 1)
   447→    MI,
   448→    /// Plus / positive or zero (N == 0)
   449→    PL,
   450→    /// Overflow (V == 1)
   451→    VS,
   452→    /// No overflow (V == 0)
   453→    VC,
   454→    /// Unsigned higher (C == 1 && Z == 0)
   455→    HI,
   456→    /// Unsigned lower or same (C == 0 || Z == 1)
   457→    LS,
   458→    /// Signed greater than or equal (N == V)
   459→    GE,
   460→    /// Signed less than (N != V)
   461→    LT,
   462→    /// Signed greater than (Z == 0 && N == V)
   463→    GT,
   464→    /// Signed less than or equal (Z == 1 || N != V)
   465→    LE,
   466→}
   467→
   468→impl Condition {
   469→    /// Returns the 4-bit condition code encoding used in ARM64 instructions.
   470→    pub fn encoding(&self) -> u32 {
   471→        match self {
   472→            Condition::EQ => 0b0000,
   473→            Condition::NE => 0b0001,
   474→            Condition::CS => 0b0010,
   475→            Condition::CC => 0b0011,
   476→            Condition::MI => 0b0100,
   477→            Condition::PL => 0b0101,
   478→            Condition::VS => 0b0110,
   479→            Condition::VC => 0b0111,
   480→            Condition::HI => 0b1000,
   481→            Condition::LS => 0b1001,
   482→            Condition::GE => 0b1010,
   483→            Condition::LT => 0b1011,
   484→            Condition::GT => 0b1100,
   485→            Condition::LE => 0b1101,
   486→        }
   487→    }
   488→
   489→    /// Returns the standard assembly mnemonic suffix for this condition.
   490→    pub fn asm_suffix(&self) -> &'static str {
   491→        match self {
   492→            Condition::EQ => "eq",
   493→            Condition::NE => "ne",
   494→            Condition::CS => "cs",
   495→            Condition::CC => "cc",
   496→            Condition::MI => "mi",
   497→            Condition::PL => "pl",
   498→            Condition::VS => "vs",
   499→            Condition::VC => "vc",
   500→            Condition::HI => "hi",
   501→            Condition::LS => "ls",
   502→            Condition::GE => "ge",
   503→            Condition::LT => "lt",
   504→            Condition::GT => "gt",
   505→            Condition::LE => "le",
   506→        }
   507→    }
   508→
   509→    /// Returns the condition code for a 4-bit encoding index.
   510→    pub fn from_encoding(idx: u32) -> Option<Condition> {
   511→        match idx {
   512→            0b0000 => Some(Condition::EQ),
   513→            0b0001 => Some(Condition::NE),
   514→            0b0010 => Some(Condition::CS),
   515→            0b0011 => Some(Condition::CC),
   516→            0b0100 => Some(Condition::MI),
   517→            0b0101 => Some(Condition::PL),
   518→            0b0110 => Some(Condition::VS),
   519→            0b0111 => Some(Condition::VC),
   520→            0b1000 => Some(Condition::HI),
   521→            0b1001 => Some(Condition::LS),
   522→            0b1010 => Some(Condition::GE),
   523→            0b1011 => Some(Condition::LT),
   524→            0b1100 => Some(Condition::GT),
   525→            0b1101 => Some(Condition::LE),
   526→            _ => None,
   527→        }
   528→    }
   529→
   530→    /// Returns the inverse (complementary) condition code.
   531→    pub fn invert(&self) -> Condition {
   532→        match self {
   533→            Condition::EQ => Condition::NE,
   534→            Condition::NE => Condition::EQ,
   535→            Condition::CS => Condition::CC,
   536→            Condition::CC => Condition::CS,
   537→            Condition::MI => Condition::PL,
   538→            Condition::PL => Condition::MI,
   539→            Condition::VS => Condition::VC,
   540→            Condition::VC => Condition::VS,
   541→            Condition::HI => Condition::LS,
   542→            Condition::LS => Condition::HI,
   543→            Condition::GE => Condition::LT,
   544→            Condition::LT => Condition::GE,
   545→            Condition::GT => Condition::LE,
   546→            Condition::LE => Condition::GT,
   547→        }
   548→    }
   549→}

   610→    /// Inner shareable domain.
   611→    ISH,
   612→    /// Inner shareable load-load / load-store.
   613→    ISHLD,
   614→    /// Inner shareable store-store.
   615→    ISHST,
   616→    /// Outer shareable domain.
   617→    OSH,
   618→    /// Outer shareable load-load / load-store.
   619→    OSHLD,
   620→    /// Outer shareable store-store.
   621→    OSHST,
   622→}
   623→
   624→impl BarrierOption {
   625→    /// Returns the 4-bit option encoding used in barrier instructions.
   626→    pub fn encoding(&self) -> u32 {
   627→        match self {
   628→            BarrierOption::SY => 0b1111,
   629→            BarrierOption::ST => 0b1110,
   630→            BarrierOption::LD => 0b1101,

   657→// ---------------------------------------------------------------------------
   658→// Memory Size (for typed load/store selection)
   659→// ---------------------------------------------------------------------------
   660→
   661→/// The size of a memory operation, used to select the correct load/store
   662→/// variant (LDR vs LDRB vs LDRH vs LDRSW, etc.).
   663→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
   664→pub enum MemorySize {
   665→    /// 8-bit byte.
   666→    Byte,
   667→    /// 16-bit halfword.
   668→    HalfWord,
   669→    /// 32-bit word.
   670→    Word,
   671→    /// 32-bit word, sign-extended to 64 bits.
   672→    SignedWord,
   673→    /// 64-bit doubleword.
   674→    DoubleWord,
   675→}
   676→
   677→impl MemorySize {
   678→    /// Returns the scale (log2 of byte size) for unsigned-offset encoding.
   679→    pub fn scale(&self) -> u32 {
   680→        match self {
   681→            MemorySize::Byte => 0,
   682→            MemorySize::HalfWord => 1,
   683→            MemorySize::Word | MemorySize::SignedWord => 2,
   684→            MemorySize::DoubleWord => 3,
   685→        }
   686→    }
   687→
   688→    /// Returns the byte size of this memory access.
   689→    pub fn byte_size(&self) -> u32 {
   690→        1u32 << self.scale()
   691→    }
   692→}

   694→// ---------------------------------------------------------------------------
   695→// Addressing Mode
   696→// ---------------------------------------------------------------------------
   697→
   698→/// Addressing mode for load/store instructions.
   699→///
   700→/// Supports the three primary ARM64 addressing patterns:
   701→/// - **Base + offset**: `LDR Xt, [Xn, #offset]`
   702→/// - **Pre-indexed**: `LDR Xt, [Xn, #offset]!` (update base before load)
   703→/// - **Post-indexed**: `LDR Xt, [Xn], #offset` (update base after load)
   704→/// - **Register offset**: `LDR Xt, [Xn, Xm, LSL #scale]`
   705→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   706→pub enum AddressingMode {
   707→    /// Unsigned offset: `[Xn, #offset]` where offset is an unsigned multiple
   708→    /// of the access size.
   709→    UnsignedOffset { base: Register, offset: u32 },
   710→    /// Pre-indexed: `[Xn, #offset]!` — base is updated before the memory
   711→    /// access.
   712→    PreIndex { base: Register, offset: i32 },
   713→    /// Post-indexed: `[Xn], #offset` — base is updated after the memory
   714→    /// access.
   715→    PostIndex { base: Register, offset: i32 },
   716→    /// Register offset: `[Xn, Xm, LSL #scale]` — index register shifted by
   717→    /// the element size.
   718→    RegisterOffset {
   719→        base: Register,
   720→        index: Register,
   721→        shift: Option<(ShiftKind, u32)>,
   722→    },
   723→}
   724→
   725→impl AddressingMode {
   726→    /// Convenience: create an unsigned-offset addressing mode.
   727→    pub fn offset(base: Register, offset: u32) -> Self {
   728→        AddressingMode::UnsignedOffset { base, offset }
   729→    }
   730→
   731→    /// Convenience: create a register-offset addressing mode with optional
   732→    /// shift (used for array indexing with element-size scaling).
   733→    pub fn reg_offset(base: Register, index: Register, shift: Option<(ShiftKind, u32)>) -> Self {
   734→        AddressingMode::RegisterOffset { base, index, shift }
   735→    }
   736→}
   737→
   738→impl std::fmt::Display for AddressingMode {
   739→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   740→        match self {
   741→            AddressingMode::UnsignedOffset { base, offset } => {
   742→                if *offset == 0 {
   743→                    write!(f, "[{}]", base)
   744→                } else {
   745→                    write!(f, "[{}, #{}]", base, offset)
   746→                }
   747→            }
   748→            AddressingMode::PreIndex { base, offset } => {
   749→                write!(f, "[{}, #{}]!", base, offset)
   750→            }
   751→            AddressingMode::PostIndex { base, offset } => {
   752→                write!(f, "[{}], #{}", base, offset)
   753→            }
   754→            AddressingMode::RegisterOffset { base, index, shift } => match shift {
   755→                Some((kind, amount)) => {
   756→                    write!(f, "[{}, {}, {} #{}]", base, index, kind, amount)
   757→                }
   758→                None => {
   759→                    write!(f, "[{}, {}]", base, index)
   760→                }
   761→            },
   762→        }
   763→    }
   764→}

   766→// ---------------------------------------------------------------------------
   767→// Operand (register or immediate)
   768→// ---------------------------------------------------------------------------
   769→
   770→/// A flexible operand — either a register (optionally shifted) or a 12-bit
   771→/// immediate.
   772→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   773→pub enum Operand {
   774→    /// Register operand with optional shift.
   775→    Reg {
   776→        reg: Register,
   777→        shift: Option<(ShiftKind, u32)>,
   778→    },
   779→    /// 12-bit unsigned immediate (0–4095) for arithmetic / logical ops.
   780→    Imm12(u16),
   781→}
   782→
   783→impl Operand {
   784→    /// Create a plain register operand with no shift.
   785→    pub fn reg(r: Register) -> Self {
   786→        Operand::Reg {
   787→            reg: r,
   788→            shift: None,
   789→        }
   790→    }
   791→
   792→    /// Create a shifted register operand.
   793→    pub fn shifted(r: Register, kind: ShiftKind, amount: u32) -> Self {
   794→        Operand::Reg {
   795→            reg: r,
   796→            shift: Some((kind, amount)),
   797→        }
   798→    }
   799→
   800→    /// Extract the register, if this is a register operand.
   801→    pub fn as_reg(&self) -> Option<Register> {
   802→        match self {
   803→            Operand::Reg { reg, .. } => Some(*reg),
   804→            Operand::Imm12(_) => None,
   805→        }
   806→    }
   807→}

   809→// ---------------------------------------------------------------------------
   810→// Instruction
   811→// ---------------------------------------------------------------------------
   812→
   813→/// ARM64 instruction representations for code generation.
   814→///
   815→/// Each variant captures the operands needed for both encoding and
   816→/// disassembly. The `encode()` method produces a 32-bit machine code word;
   817→/// the `Display` impl produces a human-readable assembly line.
   818→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   819→#[allow(non_camel_case_types)]
   820→pub enum Instruction {
   821→    // ---- Arithmetic ----
   822→    /// Add: `ADD Rd, Rn, Rm` or `ADD Rd, Rn, #imm`
   823→    ADD {
   824→        rd: Register,
   825→        rn: Register,
   826→        rm: Operand,
   827→    },
   828→    /// Subtract: `SUB Rd, Rn, Rm` or `SUB Rd, Rn, #imm`
   829→    SUB {
   830→        rd: Register,
   831→        rn: Register,
   832→        rm: Operand,
   833→    },
   834→    /// Multiply: `MUL Rd, Rn, Rm`
   835→    MUL {
   836→        rd: Register,
   837→        rn: Register,
   838→        rm: Register,
   839→    },
   840→    /// Signed divide: `SDIV Rd, Rn, Rm`
   841→    SDIV {
   842→        rd: Register,
   843→        rn: Register,
   844→        rm: Register,
   845→    },
   846→    /// Unsigned divide: `UDIV Rd, Rn, Rm`
   847→    UDIV {
   848→        rd: Register,
   849→        rn: Register,
   850→        rm: Register,
   851→    },
   852→
   853→    // ---- Bitwise / Shift ----
   854→    /// Bitwise AND: `AND Rd, Rn, Rm`
   855→    AND {
   856→        rd: Register,
   857→        rn: Register,
   858→        rm: Register,
   859→    },
   860→    /// Bitwise OR: `ORR Rd, Rn, Rm`
   861→    ORR {
   862→        rd: Register,
   863→        rn: Register,
   864→        rm: Register,
   865→    },
   866→    /// Bitwise exclusive OR: `EOR Rd, Rn, Rm`
   867→    EOR {
   868→        rd: Register,
   869→        rn: Register,
   870→        rm: Register,
   871→    },
   872→    /// Logical shift left: `LSL Rd, Rn, Rm` or `LSL Rd, Rn, #imm`
   873→    LSL {
   874→        rd: Register,
   875→        rn: Register,
   876→        rm: Operand,
   877→    },
   878→    /// Logical shift right: `LSR Rd, Rn, Rm` or `LSR Rd, Rn, #imm`
   879→    LSR {
   880→        rd: Register,
   881→        rn: Register,
   882→        rm: Operand,
   883→    },
   884→    /// Arithmetic shift right: `ASR Rd, Rn, Rm` or `ASR Rd, Rn, #imm`
   885→    ASR {
   886→        rd: Register,
   887→        rn: Register,
   888→        rm: Operand,
   889→    },
   890→    /// Extract and rotate: `EXTR Rd, Rn, Rm, #lsb`
   891→    ///
   892→    /// On AArch64, `ROR Rd, Rn, #amount` is encoded as `EXTR Rd, Rn, Rn, #amount`.
   893→    /// `ROL Rd, Rn, #amount` is encoded as `EXTR Rd, Rn, Rn, #(regsize - amount)`.
   894→    EXTR {
   895→        rd: Register,
   896→        rn: Register,
   897→        rm: Register,
   898→        imm6: u32,
   899→    },
   900→    /// Rotate right variable: `RORV Rd, Rn, Rm`
   901→    ///
   902→    /// `ROLV` does not exist on AArch64; rotate-left-by-register is
   903→    /// synthesized as `SUB tmp, XZR, Rm; RORV Rd, Rn, tmp`.
   904→    RORV {
   905→        rd: Register,
   906→        rn: Register,
   907→        rm: Register,
   908→    },
   909→
   910→    // ---- Load / Store (64-bit) ----
   911→    /// Load register (64-bit): `LDR Xt, [addr]`
   912→    LDR {
   913→        rt: Register,
   914→        rn: Register,
   915→        offset: i32,
   916→    },
   917→    /// Store register (64-bit): `STR Xt, [addr]`
   918→    STR {
   919→        rt: Register,
   920→        rn: Register,
   921→        offset: i32,
   922→    },
   923→
   924→    // ---- Load / Store (32-bit) ----
   925→    /// Load register (32-bit, zero-extended): `LDR Wt, [addr]`
   926→    LDR_W {
   927→        rt: Register,
   928→        rn: Register,
   929→        offset: i32,
   930→    },
   931→    /// Store register (32-bit): `STR Wt, [addr]`
   932→    STR_W {
   933→        rt: Register,
   934→        rn: Register,
   935→        offset: i32,
   936→    },
   937→
   938→    // ---- Load / Store (sub-word) ----
   939→    /// Load byte (zero-extended): `LDRB Wt, [addr]`
   940→    LDRB {
   941→        rt: Register,
   942→        rn: Register,
   943→        offset: i32,
   944→    },
   945→    /// Load halfword (zero-extended): `LDRH Wt, [addr]`
   946→    LDRH {
   947→        rt: Register,
   948→        rn: Register,
   949→        offset: i32,
   950→    },
   951→    /// Load word (sign-extended to 64-bit): `LDRSW Xt, [addr]`
   952→    LDRSW {
   953→        rt: Register,
   954→        rn: Register,
   955→        offset: i32,
   956→    },
   957→    /// Store byte: `STRB Wt, [addr]`
   958→    STRB {
   959→        rt: Register,
   960→        rn: Register,
   961→        offset: i32,
   962→    },
   963→    /// Store halfword: `STRH Wt, [addr]`
   964→    STRH {
   965→        rt: Register,
   966→        rn: Register,
   967→        offset: i32,
   968→    },
   969→
   970→    // ---- Load / Store Pair ----
   971→    /// Load pair: `LDP Rt1, Rt2, [Rn, #offset]`
   972→    LDP {
   973→        rt1: Register,
   974→        rt2: Register,
   975→        rn: Register,
   976→        offset: i32,
   977→    },
   978→    /// Store pair: `STP Rt1, Rt2, [Rn, #offset]`
   979→    STP {
   980→        rt1: Register,
   981→        rt2: Register,
   982→        rn: Register,
   983→        offset: i32,
   984→    },
   985→
   986→    // ---- Atomic ----
   987→    /// Load-exclusive register: `LDXR Rt, [Rn]`
   988→    LDXR { rt: Register, rn: Register },
   989→    /// Store-exclusive register: `STXR Rs, Rt, [Rn]`
   990→    STXR {
   991→        rs: Register,
   992→        rt: Register,
   993→        rn: Register,
   994→    },
   995→    /// Load-acquire exclusive register: `LDAXR Rt, [Rn]`
   996→    /// Used in atomic CAS loops per ARMv8-A acquire-release semantics.
   997→    LDAXR { rt: Register, rn: Register },
   998→    /// Store-release exclusive register: `STLXR Rs, Rt, [Rn]`
   999→    /// Used in atomic CAS loops per ARMv8-A acquire-release semantics.
  1000→    STLXR {
  1001→        rs: Register,
  1002→        rt: Register,
  1003→        rn: Register,
  1004→    },
  1005→    /// Compare-and-swap: `CAS Rs, Rt, [Rn]`
  1006→    CAS {
  1007→        rs: Register,
  1008→        rt: Register,
  1009→        rn: Register,
  1010→    },
  1011→
  1012→    // ---- Acquire/Release Load/Store ----
  1013→    /// Load-acquire: `LDAR Rt, [Rn]` — ensures subsequent memory ops are
  1014→    /// observed after this load. Used for SyncEdge::AtomicAcquireRelease.
  1015→    LDAR { rt: Register, rn: Register },
  1016→    /// Store-release: `STLR Rt, [Rn]` — ensures all prior memory ops are
  1017→    /// globally visible before this store. Used for SyncEdge::AtomicAcquireRelease.
  1018→    STLR { rt: Register, rn: Register },
  1019→
  1020→    // ---- Branch ----
  1021→    /// Unconditional branch: `B offset`
  1022→    B { offset: i32 },
  1023→    /// Branch with link: `BL offset`
  1024→    BL { offset: i32 },
  1025→    /// Branch to register: `BR Rn`
  1026→    BR { rn: Register },
  1027→    /// Branch with link to register: `BLR Rn`
  1028→    BLR { rn: Register },
  1029→    /// Return: `RET {Rn}`
  1030→    RET { rn: Option<Register> },
  1031→    /// Conditional branch: `B.cond offset`
  1032→    BCond { cond: Condition, offset: i32 },
  1033→    /// Compare and branch on zero: `CBZ Rt, offset`
  1034→    CBZ { rt: Register, offset: i32 },
  1035→    /// Compare and branch on non-zero: `CBNZ Rt, offset`
  1036→    CBNZ { rt: Register, offset: i32 },
  1037→    /// Test bit and branch on zero: `TBZ Rt, bit, offset`
  1038→    TBZ { rt: Register, bit: u32, offset: i32 },
  1039→    /// Test bit and branch on non-zero: `TBNZ Rt, bit, offset`
  1040→    TBNZ { rt: Register, bit: u32, offset: i32 },
  1041→
  1042→    // ---- Compare / Conditional Select ----
  1043→    /// Compare (subtract, discard result): `CMP Rn, Rm` or `CMP Rn, #imm`
  1044→    CMP { rn: Register, rm: Operand },
  1045→    /// Compare negative: `CMN Rn, Rm` or `CMN Rn, #imm`
  1046→    CMN { rn: Register, rm: Operand },
  1047→    /// Test (bitwise AND, discard result): `TST Rn, Rm`
  1048→    TST { rn: Register, rm: Register },

  1056→

  1069→

  1107→        rn: Register,
  1108→        dst_64: bool,

  1117→        rn: Register,

  1122→    ///
  1123→    /// - `dst_64`: true if destination is 64-bit integer (Xd), false for 32-bit (Wd)

  1126→        rd: Register,
  1127→        rn: Register,
  1128→        dst_64: bool,
  1129→        src_double: bool,
  1130→    },
  1131→    /// Float convert (single ↔ double): `FCVT Sd, Dn` or `FCVT Dd, Sn`
  1132→    FCVT {
  1133→        rd: Register,
  1134→        rn: Register,
  1135→        to_double: bool,
  1136→    },
  1137→

  1139→    /// Data memory barrier: `DMB option`
  1140→    DMB { option: BarrierOption },
  1141→    /// Data synchronization barrier: `DSB option`
  1142→    DSB { option: BarrierOption },
  1143→    /// Instruction synchronization barrier: `ISB`

  1169→    /// Move GPR to FP double register: `FMOV Dd, Xn`

  1176→    // ---- SIMD integer ----
  1177→    /// Population count per byte: `CNT Vd.8B, Vn.8B`
  1178→    /// `vd` and `vn` are 5-bit SIMD/FP register indices (0–31).
  1179→    CNT { vd: u8, vn: u8 },
  1180→    /// Add across vector: `ADDV Bd, Vn.8B`
  1181→    /// `vd` and `vn` are 5-bit SIMD/FP register indices (0–31).
  1182→    ADDV { vd: u8, vn: u8 },
  1183→    /// Unsigned move vector element to GPR: `UMOV Wd, Vn.B[0]`
  1184→    /// `vn` is the 5-bit SIMD/FP source register index (0–31).
  1185→    /// Only index=0 is supported (sufficient for POPCNT result extraction).
  1186→    UMOV { rd: Register, vn: u8 },
  1187→
  1188→    // ---- System ----
  1189→    /// Supervisor call: `SVC #imm16`
  1190→    SVC { imm16: u16 },
  1191→    /// No-operation: `NOP`
  1192→    NOP,
  1193→}
  1194→
  1195→// ---------------------------------------------------------------------------
  1196→// Instruction — encode
  1197→// ---------------------------------------------------------------------------
  1198→
  1199→impl Instruction {
  1200→    /// Encode this instruction into a 32-bit ARM64 machine-code word.
  1201→    ///
  1202→    /// Encoding follows the ARM Architecture Reference Manual (ARMv8-A).
  1203→    /// Paths that require multi-instruction sequences or are not yet
  1204→    /// implemented return `Err(CodegenError::EncodingError)`.
  1205→    pub fn encode(&self) -> Result<u32> {
  1206→        match self {
  1207→            // ---- ADD (shifted register): 1 0 0 0 1 0 1 1 shift 0 Rm imm6 Rn Rd ----
  1208→            Instruction::ADD { rd, rn, rm } => match rm {
  1209→                Operand::Reg { reg, shift } => {
  1210→                    let (hw, imm6) = shift.map(|(k, v)| (k.encoding(), v)).unwrap_or((0, 0));
  1211→                    // ADD Xd, Xn, Xm, shift: sf=1, op=00, S=0, 01011, shift, 0, Rm, imm6, Rn, Rd
  1212→                    // Base = 0x8B000000
  1213→                    Ok(0x8B000000u32
  1214→                        | (hw << 22)
  1215→                        | (reg.encoding() << 16)
  1216→                        | (imm6 << 10)
  1217→                        | (rn.encoding() << 5)
  1218→                        | rd.encoding())
  1219→                }
  1220→                Operand::Imm12(imm) => {
  1221→                    // ADD Xd, Xn, #imm: sf=1, op=0, S=0, 10001, shift=0, imm12, Rn, Rd
  1222→                    // Base = 0x91000000
  1223→                    Ok(0x91000000u32
  1224→                        | ((*imm as u32) << 10)
  1225→                        | (rn.encoding() << 5)
  1226→                        | rd.encoding())
  1227→                }
  1228→            },
  1229→
  1230→            // ---- SUB (shifted register): 1 1 0 0 1 0 1 1 shift 0 Rm imm6 Rn Rd ----
  1231→            Instruction::SUB { rd, rn, rm } => match rm {
  1232→                Operand::Reg { reg, shift } => {
  1233→                    let (hw, imm6) = shift.map(|(k, v)| (k.encoding(), v)).unwrap_or((0, 0));
  1234→                    Ok(0xCB000000u32
  1235→                        | (hw << 22)
  1236→                        | (reg.encoding() << 16)
  1237→                        | (imm6 << 10)
  1238→                        | (rn.encoding() << 5)
  1239→                        | rd.encoding())
  1240→                }
  1241→                Operand::Imm12(imm) => Ok(0xD1000000u32
  1242→                    | ((*imm as u32) << 10)
  1243→                    | (rn.encoding() << 5)
  1244→                    | rd.encoding()),
  1245→            },
  1246→
  1247→            // ---- MUL: alias for MADD Rd, Rn, Rm, XZR ----
  1248→            Instruction::MUL { rd, rn, rm } => Ok(0x9B007C00u32
  1249→                | (rm.encoding() << 16)
  1250→                | (rn.encoding() << 5)
  1251→                | rd.encoding()),
  1252→
  1253→            // ---- SDIV ----
  1254→            // SDIV Xd, Xn, Xm: sf=1, 00, 1101, 011, Rm, 000011, Rn, Rd
  1255→            // Base = 0x9AC00C00
  1256→            Instruction::SDIV { rd, rn, rm } => Ok(0x9AC00C00u32
  1257→                | (rm.encoding() << 16)
  1258→                | (rn.encoding() << 5)
  1259→                | rd.encoding()),
  1260→
  1261→            // ---- UDIV ----
  1262→            // UDIV Xd, Xn, Xm: sf=1, 00, 1101, 011, Rm, 000010, Rn, Rd
  1263→            // Base = 0x9AC00800
  1264→            Instruction::UDIV { rd, rn, rm } => Ok(0x9AC00800u32
  1265→                | (rm.encoding() << 16)
  1266→                | (rn.encoding() << 5)
  1267→                | rd.encoding()),
  1268→
  1269→            // ---- AND (shifted register) ----
  1270→            // AND Xd, Xn, Xm: sf=1, opc=00, 01010, N=0, shift=00, Rm, imm6=0, Rn, Rd
  1271→            // Base = 0x8A000000
  1272→            Instruction::AND { rd, rn, rm } => Ok(0x8A000000u32
  1273→                | (rm.encoding() << 16)
  1274→                | (rn.encoding() << 5)
  1275→                | rd.encoding()),
  1276→
  1277→            // ---- ORR (shifted register) ----
  1278→            // ORR Xd, Xn, Xm: sf=1, opc=01, 01010, N=0, shift=00, Rm, imm6=0, Rn, Rd
  1279→            // Base = 0xAA000000
  1280→            Instruction::ORR { rd, rn, rm } => Ok(0xAA000000u32
  1281→                | (rm.encoding() << 16)
  1282→                | (rn.encoding() << 5)
  1283→                | rd.encoding()),
  1284→
  1285→            // ---- EOR (shifted register) ----
  1286→            // EOR Xd, Xn, Xm: sf=1, opc=10, 01010, N=0, shift=00, Rm, imm6=0, Rn, Rd
  1287→            // Base = 0xCA000000
  1288→            Instruction::EOR { rd, rn, rm } => Ok(0xCA000000u32
  1289→                | (rm.encoding() << 16)
  1290→                | (rn.encoding() << 5)
  1291→                | rd.encoding()),
  1292→
  1293→            // ---- LSL / LSR / ASR (shifted register or immediate) ----
  1294→            // LSL Xd, Xn, Xm uses data processing shifted register format (same as ADD)
  1295→            // Base = 0x8B000000 (sf=1, op=00, S=0, 01011, shift, 0, Rm, imm6=0, Rn, Rd)
  1296→            Instruction::LSL { rd, rn, rm } => match rm {
  1297→                Operand::Reg { reg, shift: _ } => Ok(0x8B000000u32
  1298→                    | (ShiftKind::LSL.encoding() << 22)
  1299→                    | (reg.encoding() << 16)
  1300→                    | (rn.encoding() << 5)
  1301→                    | rd.encoding()),
  1302→                // LSL Xd, Xn, #shift = UBFM Xd, Xn, #(64-shift), #(63-shift)
  1303→                // UBFM: sf=1, opc=10, 100110, N=1, immr, imms, Rn, Rd
  1304→                // Base = 0xD3400000
  1305→                Operand::Imm12(imm) => {
  1306→                    let shift_val = *imm as u32;
  1307→                    let immr = (64 - shift_val) & 63;
  1308→                    let imms = 63 - shift_val;
  1309→                    Ok(0xD3400000u32
  1310→                        | (immr << 16)
  1311→                        | (imms << 10)
  1312→                        | (rn.encoding() << 5)
  1313→                        | rd.encoding())
  1314→                }
  1315→            },
  1316→
  1317→            // LSR Xd, Xn, Xm: same base as ADD shifted
  1318→            Instruction::LSR { rd, rn, rm } => match rm {
  1319→                Operand::Reg { reg, shift: _ } => Ok(0x8B000000u32
  1320→                    | (ShiftKind::LSR.encoding() << 22)
  1321→                    | (reg.encoding() << 16)
  1322→                    | (rn.encoding() << 5)
  1323→                    | rd.encoding()),
  1324→                // LSR Xd, Xn, #shift = UBFM Xd, Xn, #shift, #63
  1325→                // UBFM: sf=1, opc=10, 100110, N=1, immr, imms, Rn, Rd
  1326→                // Base = 0xD3400000
  1327→                Operand::Imm12(imm) => Ok(0xD3400000u32
  1328→                    | ((*imm as u32) << 16)
  1329→                    | (63u32 << 10)
  1330→                    | (rn.encoding() << 5)
  1331→                    | rd.encoding()),
  1332→            },
  1333→
  1334→            // ASR Xd, Xn, Xm: same base as ADD shifted
  1335→            Instruction::ASR { rd, rn, rm } => match rm {
  1336→                Operand::Reg { reg, shift: _ } => Ok(0x8B000000u32
  1337→                    | (ShiftKind::ASR.encoding() << 22)
  1338→                    | (reg.encoding() << 16)
  1339→                    | (rn.encoding() << 5)
  1340→                    | rd.encoding()),
  1341→                // ASR Xd, Xn, #shift = SBFM Xd, Xn, #shift, #63
  1342→                // SBFM: sf=1, opc=00, 100110, N=1, immr, imms, Rn, Rd
  1343→                // Base = 0x93400000
  1344→                Operand::Imm12(imm) => Ok(0x93400000u32
  1345→                    | ((*imm as u32) << 16)
  1346→                    | (63u32 << 10)
  1347→                    | (rn.encoding() << 5)
  1348→                    | rd.encoding()),
  1349→            },
  1350→
  1351→            // ---- EXTR (extract): 1 00 100111 1 0 Rm imm6 Rn Rd (64-bit) ----
  1352→            // 64-bit base = 0x93C00000
  1353→            // Used for ROR Rd, Rn, #amount (= EXTR Rd, Rn, Rn, #amount)
  1354→            // and for ROL Rd, Rn, #amount (= EXTR Rd, Rn, Rn, #(64 - amount))
  1355→            Instruction::EXTR { rd, rn, rm, imm6 } => Ok(0x93C00000u32
  1356→                | (rm.encoding() << 16)
  1357→                | ((imm6 & 0x3F) << 10)
  1358→                | (rn.encoding() << 5)
  1359→                | rd.encoding()),
  1360→
  1361→            // ---- RORV (rotate right variable): 1 00 11010 11 0 Rm 001011 Rn Rd ----
  1362→            // 64-bit base = 0x9AC02C00
  1363→            Instruction::RORV { rd, rn, rm } => Ok(0x9AC02C00u32
  1364→                | (rm.encoding() << 16)
  1365→                | (rn.encoding() << 5)
  1366→                | rd.encoding()),
  1367→
  1368→            // ---- LDR (unsigned offset, 64-bit) ----
  1369→            Instruction::LDR { rt, rn, offset } => {
  1370→                if *offset >= 0 && *offset % 8 == 0 {
  1371→                    let imm12 = (*offset as u32) / 8;
  1372→                    if imm12 <= 4095 {
  1373→                        Ok(0b1111_1001_0100_0000_0000_0000_0000_0000
  1374→                            | (imm12 << 10)
  1375→                            | (rn.encoding() << 5)
  1376→                            | rt.encoding())
  1377→                    } else {
  1378→                        Err(CodegenError::EncodingError(format!(
  1379→                            "LDR offset {} too large (imm12 = {} exceeds 4095); use address computation instead",
  1380→                            offset, imm12
  1381→                        )))
  1382→                    }
  1383→                } else {
  1384→                    Err(CodegenError::EncodingError(format!(
  1385→                        "LDR with offset {} not supported (must be non-negative multiple of 8, 0..32760); use address computation instead",
  1386→                        offset
  1387→                    )))
  1388→                }
  1389→            }
  1390→

  1392→            Instruction::STR { rt, rn, offset } => {

  1395→                    if imm12 <= 4095 {
  1396→                        // STR Xt, [Xn, #pimm]: 11_111_0_00_01_pimm_Rn_Rt

  1399→                            | (imm12 << 10)
  1400→                            | (rn.encoding() << 5)
  1401→                            | rt.encoding())

  1406→                        )))

  1411→                        offset
  1412→                    )))
