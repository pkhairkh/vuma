  2549→                        IRType::I8 => {

  2553→                        IRType::U8 => {
  2554→                            if (-32768..=32767).contains(&off) { code.extend_from_slice(&Instruction::Lbu { rt: Gpr::T0, base: Gpr::T3, offset: off }.encode()); code.extend_from_slice(&encode_nop()); }
  2555→                            else { code.extend(ss_load_imm(Gpr::T4, off as i64)); code.extend_from_slice(&Instruction::Daddu { rd: Gpr::T4, rs: Gpr::T3, rt: Gpr::T4 }.encode()); code.extend_from_slice(&encode_nop()); code.extend_from_slice(&Instruction::Lbu { rt: Gpr::T0, base: Gpr::T4, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2556→                        }
  2557→                        IRType::I32 | IRType::U32 => {

  2560→                        }
  2561→                        _ => {
  2562→                            if (-32768..=32767).contains(&off) { code.extend_from_slice(&Instruction::Ld { rt: Gpr::T0, base: Gpr::T3, offset: off }.encode()); code.extend_from_slice(&encode_nop()); }

  2564→                        }
  2565→                    }
  2566→                    code.extend(ss_sd(Gpr::T0, dst_off));
  2567→                }
  2568→

  2570→                IRInstr::Store { value, addr, offset, ty } => {
  2571→                    code.extend(ss_load_value(addr, &vreg_stack_slots, Gpr::T3));
  2572→                    code.extend(ss_load_value(value, &vreg_stack_slots, Gpr::T0));
  2573→                    let off = *offset as i32;
  2574→                    match ty {
  2575→                        IRType::I8 | IRType::U8 => {
  2576→                            if (-32768..=32767).contains(&off) { code.extend_from_slice(&Instruction::Sb { rt: Gpr::T0, base: Gpr::T3, offset: off }.encode()); code.extend_from_slice(&encode_nop()); }

  2583→                        _ => {
  2584→                            if (-32768..=32767).contains(&off) { code.extend_from_slice(&Instruction::Sd { rt: Gpr::T0, base: Gpr::T3, offset: off }.encode()); code.extend_from_slice(&encode_nop()); }
  2585→                            else { code.extend(ss_load_imm(Gpr::T4, off as i64)); code.extend_from_slice(&Instruction::Daddu { rd: Gpr::T4, rs: Gpr::T3, rt: Gpr::T4 }.encode()); code.extend_from_slice(&encode_nop()); code.extend_from_slice(&Instruction::Sd { rt: Gpr::T0, base: Gpr::T4, offset: 0 }.encode()); code.extend_from_slice(&encode_nop()); }
  2586→                        }
  2587→                    }
  2588→                }
  2589→
  2590→                // ── Alloc ──
  2591→                IRInstr::Alloc { dst, size: _ } => {

  2599→                    } else {
  2600→                        code.extend(ss_load_imm(Gpr::T0, alloc_off as i64));
  2601→                        code.extend_from_slice(&Instruction::Dsubu { rd: Gpr::T0, rs: Gpr::Fp, rt: Gpr::T0 }.encode());
  2602→                        code.extend_from_slice(&encode_nop());
  2603→                    }
  2604→                    code.extend(ss_sd(Gpr::T0, dst_off));
  2605→                }
  2606→
  2607→                // ── Free ──
  2608→                IRInstr::Free { ptr: _ } => { /* Stack allocations freed by epilogue */ }
  2609→

  2611→                IRInstr::Cast { kind, dst, src } => {

  2613→                    let dst_off = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);

  2618→                            code.extend_from_slice(&Instruction::Dsll { rd: Gpr::T0, rt: Gpr::T0, sa: 32 }.encode()); code.extend_from_slice(&encode_nop());

  2620→                        }
  2621→                        CastKind::SExt => {
  2622→                            // Sign-extend: already sign-extended in 64-bit MIPS load
  2623→                        }
  2624→                        CastKind::BitCast | CastKind::Trunc => { /* no-op */ }
  2625→                    }
  2626→                    code.extend(ss_sd(Gpr::T0, dst_off));
  2627→                }
  2628→
  2629→                // ── Select ──
  2630→                IRInstr::Select { dst, cond, true_val, false_val, ty: _ } => {
  2631→                    let dst_id = dst.as_register().unwrap_or(0);
  2632→                    let dst_off = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
  2633→                    code.extend(ss_load_value(false_val, &vreg_stack_slots, Gpr::T0));
  2634→                    code.extend(ss_load_value(true_val, &vreg_stack_slots, Gpr::T1));
  2635→                    code.extend(ss_load_value(cond, &vreg_stack_slots, Gpr::T2));
  2636→                    // MOVN $t0, $t1, $t2 (if $t2 != 0, $t0 = $t1)
  2637→                    code.extend_from_slice(&Instruction::Movn { rd: Gpr::T0, rs: Gpr::T1, rt: Gpr::T2 }.encode());
  2638→                    code.extend_from_slice(&encode_nop());
  2639→                    code.extend(ss_sd(Gpr::T0, dst_off));
  2640→                }
  2641→
  2642→                // ── Offset ──
  2643→                IRInstr::Offset { dst, base, offset } => {
  2644→                    let dst_id = dst.as_register().unwrap_or(0);
  2645→                    let dst_off = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
  2646→                    code.extend(ss_load_value(base, &vreg_stack_slots, Gpr::T0));
  2647→                    code.extend(ss_load_value(offset, &vreg_stack_slots, Gpr::T1));
  2648→                    code.extend_from_slice(&Instruction::Daddu { rd: Gpr::T0, rs: Gpr::T0, rt: Gpr::T1 }.encode());
  2649→                    code.extend_from_slice(&encode_nop());
  2650→                    code.extend(ss_sd(Gpr::T0, dst_off));

  2652→

  2654→                IRInstr::GetAddress { dst, name: _ } => {

  2656→                    let dst_off = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
  2657→                    code.extend(ss_load_imm(Gpr::T0, 0));
  2658→                    code.extend(ss_sd(Gpr::T0, dst_off));

  2660→
  2661→                // ── Ret ──
  2662→                IRInstr::Ret { values } => {
  2663→                    if let Some(val) = values.first() {
  2664→                        code.extend(ss_load_value(val, &vreg_stack_slots, Gpr::V0));
  2665→                    }
  2666→                    // Epilogue
  2667→                    code.extend_from_slice(&Instruction::Ld { rt: Gpr::Fp, base: Gpr::Sp, offset: (frame_size - 16) as i32 }.encode()); code.extend_from_slice(&encode_nop());
  2668→                    code.extend_from_slice(&Instruction::Ld { rt: Gpr::Ra, base: Gpr::Sp, offset: (frame_size - 8) as i32 }.encode()); code.extend_from_slice(&encode_nop());
  2669→                    code.extend_from_slice(&Instruction::Daddiu { rt: Gpr::Sp, rs: Gpr::Sp, imm: frame_size as i32 }.encode()); code.extend_from_slice(&encode_nop());
  2670→                    code.extend_from_slice(&Instruction::Jr { rs: Gpr::Ra }.encode()); code.extend_from_slice(&encode_nop());
  2671→                }
  2672→
  2673→                // ── Branch ──
  2674→                IRInstr::Branch { target } => {
  2675→                    let fixup_offset = code.len();
  2676→                    branch_fixups.push((fixup_offset, target.clone(), 0)); // B instruction offset
  2677→                    code.extend_from_slice(&Instruction::Beq { rs: Gpr::Zero, rt: Gpr::Zero, offset: 0 }.encode()); // placeholder
  2678→                    code.extend_from_slice(&encode_nop()); // delay slot
  2679→                }
  2680→
  2681→                // ── CondBranch ──
  2682→                IRInstr::CondBranch { cond, true_target, false_target } => {
  2683→                    code.extend(ss_load_value(cond, &vreg_stack_slots, Gpr::T0));
  2684→                    // BNE $t0, $zero, true_target
  2685→                    let true_fixup = code.len();
  2686→                    branch_fixups.push((true_fixup, true_target.clone(), 0));
  2687→                    code.extend_from_slice(&Instruction::Bne { rs: Gpr::T0, rt: Gpr::Zero, offset: 0 }.encode()); // placeholder
  2688→                    code.extend_from_slice(&encode_nop()); // delay slot
  2689→                    // BEQ $zero, $zero, false_target  (unconditional branch)
  2690→                    let false_fixup = code.len();
  2691→                    branch_fixups.push((false_fixup, false_target.clone(), 0));
  2692→                    code.extend_from_slice(&Instruction::Beq { rs: Gpr::Zero, rt: Gpr::Zero, offset: 0 }.encode()); // placeholder
  2693→                    code.extend_from_slice(&encode_nop()); // delay slot
  2694→                }
  2695→
  2696→                // ── Call ──
  2697→                IRInstr::Call { dst, func: target_func, args } => {
  2698→                    // Load arguments into $a0-$a3
  2699→                    let arg_regs = [Gpr::A0, Gpr::A1, Gpr::A2, Gpr::A3];
  2700→                    for (i, arg) in args.iter().enumerate() {
  2701→                        if i < 4 {
  2702→                            code.extend(ss_load_value(arg, &vreg_stack_slots, arg_regs[i]));
  2703→                        }
  2704→                    }
  2705→                    // JAL — record relocation
  2706→                    let jal_byte_offset = code.len() as u64;
  2707→                    relocations.push(RelocationEntry {
  2708→                        offset: jal_byte_offset,
  2709→                        symbol: target_func.clone(),
  2710→                        reloc_type: "R_MIPS_26".to_string(),
  2711→                    });
  2712→                    code.extend_from_slice(&Instruction::Jal { target: 0 }.encode()); // placeholder
  2713→                    code.extend_from_slice(&encode_nop()); // delay slot
  2714→
  2715→                    // Move return value from $v0 to dst stack slot
  2716→                    if let Some(d) = dst {
  2717→                        let dst_id = d.as_register().unwrap_or(0);
  2718→                        let dst_off = vreg_stack_slots.get(&dst_id).copied().unwrap_or(0);
  2719→                        code.extend(ss_sd(Gpr::V0, dst_off));
  2720→                    }
  2721→                }
  2722→
  2723→                // ── Phi ──
  2724→                IRInstr::Phi { .. } => { /* no-op */ }
  2725→            }
  2726→        }
  2727→    }
  2728→
  2729→    // ── Phase 5: Apply branch fixups ──
  2730→    for (fixup_offset, label, _adj) in &branch_fixups {
  2731→        if let Some(&target_offset) = label_offsets.get(label) {
  2732→            let branch_offset = (target_offset as i32 - *fixup_offset as i32) / 4 - 1; // -1 for delay slot
  2733→            let existing = u32::from_be_bytes([code[*fixup_offset], code[*fixup_offset+1], code[*fixup_offset+2], code[*fixup_offset+3]]);
  2734→            let imm16 = (branch_offset as u32) & 0xFFFF;
  2735→            let patched = (existing & 0xFFFF0000) | imm16;
  2736→            code[*fixup_offset..*fixup_offset+4].copy_from_slice(&patched.to_be_bytes());
  2737→        }
  2738→    }
  2739→
  2740→    // Convert code bytes to AllocatedInstructions
  2741→    let instructions: Vec<AllocatedInstruction> = code.chunks_exact(4)
  2742→        .map(|chunk| AllocatedInstruction {
  2743→            opcode: "mips64".to_string(),
  2744→            reads: vec![],
  2745→            writes: vec![],
  2746→            encoded: chunk.to_vec(),
  2747→        })
  2748→        .collect();
  2749→
  2750→    let code_size = code.len();
  2751→
