     1→//! SCG Serialization / Deserialization
     2→//!
     3→//! This module provides three serialization formats for the Semantic Computation Graph:
     4→//!
     5→//! - **Binary**: A compact, versioned binary format suitable for persistent storage
     6→//!   and network transfer. The format includes a magic header and version number
     7→//!   for forward/backward compatibility.
     8→//! - **JSON**: A human-readable JSON representation using serde, ideal for debugging
     9→//!   and interoperability.
    10→//! - **DOT**: A Graphviz DOT format for visualizing the graph structure.
    11→//!
    12→//! # Binary Format (Version 1)
    13→//!
    14→//! ```text
    15→//! [4B]  Magic: "VSCG"
    16→//! [4B]  Version: u32 LE
    17→//! [8B]  Next node ID counter: u64 LE
    18→//! [8B]  Next edge ID counter: u64 LE
    19→//! [4B]  Node count: u32 LE
    20→//! [4B]  Edge count: u32 LE
    21→//! [4B]  Region count: u32 LE
    22→//! --- Nodes (repeated Node count times) ---
    23→//! --- Edges (repeated Edge count times) ---
    24→//! --- Regions (repeated Region count times) ---
    25→//! ```

   100→// ── Error type ──────────────────────────────────────────────────────────
   101→
   102→/// Errors that can occur during SCG deserialization.
   103→#[derive(Debug, Clone, PartialEq)]
   104→pub enum DeserializeError {
   105→    /// The data does not start with the expected magic bytes.
   106→    InvalidMagic {
   107→        /// The expected magic bytes.
   108→        expected: Vec<u8>,
   109→        /// The magic bytes actually found.
   110→        found: Vec<u8>,
   111→    },
   112→    /// The format version is not supported by this implementation.
   113→    UnsupportedVersion {
   114→        /// The version number found in the data.
   115→        version: u32,
   116→        /// The minimum version supported by this implementation.
   117→        min_supported: u32,
   118→    },
   119→    /// An unexpected end of input was reached.
   120→    UnexpectedEof {
   121→        /// Description of what was being read when EOF was encountered.
   122→        context: String,
   123→    },
   124→    /// A value was out of the valid range for its type.
   125→    InvalidValue {
   126→        /// Name of the field that had the invalid value.
   127→        field: String,
   128→        /// String representation of the invalid value.
   129→        value: String,
   130→    },
   131→    /// A string could not be decoded as valid UTF-8.
   132→    InvalidUtf8 {
   133→        /// Description of what was being read when the invalid UTF-8 was encountered.
   134→        context: String,
   135→        /// The underlying UTF-8 error message.
   136→        source: String,
   137→    },
   138→    /// An I/O error occurred during deserialization.
   139→    IoError(String),
   140→    /// A JSON serialization/deserialization error.
   141→    JsonError(String),
   142→    /// An internal consistency error in the serialized data.
   143→    ConsistencyError(String),
   144→}
   145→
   146→impl std::fmt::Display for DeserializeError {
   147→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   148→        match self {
   149→            DeserializeError::InvalidMagic { expected, found } => {
   150→                write!(
   151→                    f,
   152→                    "invalid magic: expected {:?}, found {:?}",
   153→                    expected, found
   154→                )
   155→            }
   156→            DeserializeError::UnsupportedVersion {
   157→                version,
   158→                min_supported,
   159→            } => {
   160→                write!(
   161→                    f,
   162→                    "unsupported format version {} (minimum supported: {})",
   163→                    version, min_supported
   164→                )
   165→            }
   166→            DeserializeError::UnexpectedEof { context } => {
   167→                write!(f, "unexpected end of input: {}", context)
   168→            }
   169→            DeserializeError::InvalidValue { field, value } => {
   170→                write!(f, "invalid value for '{}': {}", field, value)
   171→            }
   172→            DeserializeError::InvalidUtf8 { context, source } => {
   173→                write!(f, "invalid UTF-8 in '{}': {}", context, source)
   174→            }
   175→            DeserializeError::IoError(msg) => write!(f, "I/O error: {}", msg),
   176→            DeserializeError::JsonError(msg) => write!(f, "JSON error: {}", msg),
   177→            DeserializeError::ConsistencyError(msg) => write!(f, "consistency error: {}", msg),
   178→        }
   179→    }
   180→}
   181→
   182→impl std::error::Error for DeserializeError {}

   184→impl From<serde_json::Error> for DeserializeError {
   185→    fn from(e: serde_json::Error) -> Self {
   186→        DeserializeError::JsonError(e.to_string())
   187→    }
   188→}

   190→// ── Binary Reader ───────────────────────────────────────────────────────
   191→
   192→/// Helper for reading structured binary data from a byte slice.
   193→struct BinaryReader<'a> {
   194→    data: &'a [u8],
   195→    pos: usize,
   196→}
   197→
   198→impl<'a> BinaryReader<'a> {
   199→    fn new(data: &'a [u8]) -> Self {
   200→        Self { data, pos: 0 }
   201→    }
   202→
   203→    #[allow(dead_code)] // part of BinaryReader API for future serialization needs
   204→    fn remaining(&self) -> usize {
   205→        self.data.len().saturating_sub(self.pos)
   206→    }
   207→
   208→    fn read_bytes(&mut self, n: usize, context: &str) -> Result<&'a [u8], DeserializeError> {
   209→        if self.pos + n > self.data.len() {
   210→            return Err(DeserializeError::UnexpectedEof {
   211→                context: context.to_string(),
   212→            });
   213→        }
   214→        let slice = &self.data[self.pos..self.pos + n];
   215→        self.pos += n;
   216→        Ok(slice)
   217→    }
   218→
   219→    fn read_u8(&mut self, context: &str) -> Result<u8, DeserializeError> {
   220→        let bytes = self.read_bytes(1, context)?;
   221→        Ok(bytes[0])
   222→    }
   223→
   224→    fn read_bool(&mut self, context: &str) -> Result<bool, DeserializeError> {
   225→        let val = self.read_u8(context)?;
   226→        Ok(val != 0)
   227→    }
   228→
   229→    fn read_u32_le(&mut self, context: &str) -> Result<u32, DeserializeError> {
   230→        let bytes = self.read_bytes(4, context)?;
   231→        Ok(u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]))
   232→    }
   233→
   234→    fn read_u64_le(&mut self, context: &str) -> Result<u64, DeserializeError> {
   235→        let bytes = self.read_bytes(8, context)?;
   236→        Ok(u64::from_le_bytes([
   237→            bytes[0], bytes[1], bytes[2], bytes[3], bytes[4], bytes[5], bytes[6], bytes[7],
   238→        ]))
   239→    }
   240→
   241→    fn read_string(&mut self, context: &str) -> Result<String, DeserializeError> {
   242→        let len = self.read_u32_le(&format!("{}.length", context))? as usize;
   243→        if len == 0 {
   244→            return Ok(String::new());
   245→        }
   246→        let bytes = self.read_bytes(len, context)?;
   247→        std::str::from_utf8(bytes)
   248→            .map(|s| s.to_string())
   249→            .map_err(|e| DeserializeError::InvalidUtf8 {
   250→                context: context.to_string(),
   251→                source: e.to_string(),
   252→            })
   253→    }
   254→
   255→    fn read_optional_string(&mut self, context: &str) -> Result<Option<String>, DeserializeError> {
   256→        let has = self.read_bool(&format!("{}.has", context))?;
   257→        if has {
   258→            self.read_string(context).map(Some)
   259→        } else {
   260→            Ok(None)
   261→        }
   262→    }
   263→
   264→    fn read_optional_u64(&mut self, context: &str) -> Result<Option<u64>, DeserializeError> {
   265→        let has = self.read_bool(&format!("{}.has", context))?;
   266→        if has {
   267→            self.read_u64_le(context).map(Some)
   268→        } else {
   269→            Ok(None)
   270→        }
   271→    }
   272→}

   274→// ── Binary Writer ───────────────────────────────────────────────────────
   275→
   276→/// Helper for writing structured binary data into a byte buffer.
   277→struct BinaryWriter {
   278→    buf: Vec<u8>,
   279→}
   280→
   281→impl BinaryWriter {
   282→    fn new() -> Self {
   283→        Self {
   284→            buf: Vec::with_capacity(256),
   285→        }
   286→    }
   287→
   288→    fn write_bytes(&mut self, data: &[u8]) {
   289→        self.buf.extend_from_slice(data);
   290→    }
   291→
   292→    #[allow(dead_code)] // part of BinaryWriter API for future serialization needs
   293→    fn write_u8(&mut self, val: u8) {
   294→        self.buf.push(val);
   295→    }
   296→
   297→    fn write_bool(&mut self, val: bool) {
   298→        self.buf.push(if val { 1 } else { 0 });
   299→    }
   300→
   301→    fn write_u32_le(&mut self, val: u32) {
   302→        self.buf.extend_from_slice(&val.to_le_bytes());
   303→    }
   304→
   305→    fn write_u64_le(&mut self, val: u64) {
   306→        self.buf.extend_from_slice(&val.to_le_bytes());
   307→    }
   308→
   309→    fn write_string(&mut self, s: &str) {
   310→        let bytes = s.as_bytes();
   311→        self.write_u32_le(bytes.len() as u32);
   312→        self.buf.extend_from_slice(bytes);
   313→    }
   314→
   315→    fn write_optional_string(&mut self, opt: &Option<String>) {
   316→        match opt {
   317→            Some(s) => {
   318→                self.write_bool(true);
   319→                self.write_string(s);
   320→            }
   321→            None => {
   322→                self.write_bool(false);
   323→            }
   324→        }
   325→    }
   326→
   327→    fn write_optional_u64(&mut self, opt: &Option<u64>) {
   328→        match opt {
   329→            Some(v) => {
   330→                self.write_bool(true);
   331→                self.write_u64_le(*v);
   332→            }
   333→            None => {
   334→                self.write_bool(false);
   335→            }
   336→        }
   337→    }
   338→
   339→    fn into_vec(self) -> Vec<u8> {
   340→        self.buf
   341→    }
   342→}

   522→// ── Intermediate serializable representation ────────────────────────────
   523→
   524→/// Intermediate representation used for JSON serialization.
   525→///
   526→/// This struct flattens the SCG's internal petgraph representation into
   527→/// a simple structure that can be serialized with serde.
   528→#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
   529→struct SerializedSCG {
   530→    /// Format version for forward/backward compatibility.
   531→    version: u32,
   532→    /// All nodes in the graph.
   533→    nodes: Vec<NodeData>,
   534→    /// All edges in the graph.
   535→    edges: Vec<EdgeData>,
   536→    /// All regions in the graph.
   537→    regions: Vec<SCGRegion>,
   538→    /// The next node ID counter.
   539→    next_node_id: u64,
   540→    /// The next edge ID counter.
   541→    next_edge_id: u64,
   542→}
   543→
   544→// ── Extract SCG into intermediate representation ────────────────────────
   545→
   546→fn scg_to_serialized(scg: &SCG) -> SerializedSCG {
   547→    let nodes: Vec<NodeData> = scg.nodes().cloned().collect();
   548→    let edges: Vec<EdgeData> = scg.edges().cloned().collect();
   549→    let regions: Vec<SCGRegion> = scg.regions().cloned().collect();
   550→
   551→    // Compute the next_node_id and next_edge_id from the SCG.
   552→    // We derive these from the maximum observed IDs + 1, since SCG doesn't
   553→    // expose the counters directly.
   554→    let next_node_id = nodes.iter().map(|n| n.id.as_u64() + 1).max().unwrap_or(0);
   555→    let next_edge_id = edges.iter().map(|e| e.id.as_u64() + 1).max().unwrap_or(0);
   556→
   557→    SerializedSCG {
   558→        version: FORMAT_VERSION,
   559→        nodes,
   560→        edges,
   561→        regions,
   562→        next_node_id,
   563→        next_edge_id,
   564→    }
   565→}
   566→
   567→/// Reconstruct an SCG from the intermediate representation.
   568→fn serialized_to_scg(s: SerializedSCG) -> Result<SCG, DeserializeError> {
   569→    let mut scg = SCG::new();
   570→
   571→    // Add nodes first (using their pre-assigned IDs)
   572→    for node_data in &s.nodes {
   573→        scg.add_node_with_id(
   574→            node_data.id,
   575→            node_data.node_type.clone(),
   576→            node_data.payload.clone(),
   577→            node_data.program_point.clone(),
   578→        )
   579→        .map_err(|e| DeserializeError::ConsistencyError(e.to_string()))?;
   580→
   581→        // Restore annotation if present
   582→        if let Some(ref ann) = node_data.annotation {
   583→            if let Some(nd) = scg.get_node_mut(node_data.id) {
   584→                nd.annotation = Some(ann.clone());
   585→            }
   586→        }
   587→    }
   588→
   589→    // Add edges (using their pre-assigned IDs)
   590→    for edge_data in &s.edges {
   591→        scg.add_edge_with_id(
   592→            edge_data.id,
   593→            edge_data.source,
   594→            edge_data.target,
   595→            edge_data.kind.clone(),
   596→        )
   597→        .map_err(|e| DeserializeError::ConsistencyError(e.to_string()))?;
   598→
   599→        // Restore label if present
   600→        if let Some(ref label) = edge_data.label {
   601→            if let Some(ed) = scg.get_edge_mut(edge_data.id) {
   602→                ed.label = Some(label.clone());
   603→            }
   604→        }
   605→    }
   606→
   607→    // Add regions
   608→    for region in s.regions {
   609→        scg.add_region(region);
   610→    }
   611→
   612→    Ok(scg)
   613→}
   614→
   615→// ── Binary serialization ────────────────────────────────────────────────
   616→
   617→/// Serializes an SCG to a versioned binary format.
   618→///
   619→/// The binary format starts with a magic header (`VSCG`), a version number,
   620→/// and then the graph contents encoded as little-endian integers and
   621→/// length-prefixed strings.
   622→///
   623→/// # Example
   624→///
   625→/// ```
   626→/// use vuma_scg::*;
   627→/// let mut scg = SCG::new();
   628→/// let id = scg.add_node(
   629→///     NodeType::Computation,
   630→///     NodePayload::Computation(ComputationNode {
   631→///         kind: ComputationKind::Other("add".to_string()),
   632→///         result_type: None,
   633→///         tail_call: false }),
   634→///     ProgramPoint { file: None, line: None, column: None, offset: None },
   635→/// );
   636→/// let bytes = vuma_scg::serialize::serialize_scg(&scg);
   637→/// assert!(&bytes[..4] == b"VSCG");
   638→/// ```
   639→pub fn serialize_scg(scg: &SCG) -> Vec<u8> {
   640→    let s = scg_to_serialized(scg);
   641→    let mut w = BinaryWriter::new();
   642→
   643→    // Header
   644→    w.write_bytes(MAGIC);
   645→    w.write_u32_le(FORMAT_VERSION);
   646→    w.write_u64_le(s.next_node_id);
   647→    w.write_u64_le(s.next_edge_id);
   648→    w.write_u32_le(s.nodes.len() as u32);
   649→    w.write_u32_le(s.edges.len() as u32);
   650→    w.write_u32_le(s.regions.len() as u32);
   651→
   652→    // Nodes
   653→    for node in &s.nodes {
   654→        write_node(&mut w, node);
   655→    }
   656→
   657→    // Edges
   658→    for edge in &s.edges {
   659→        write_edge(&mut w, edge);
   660→    }
   661→
   662→    // Regions
   663→    for region in &s.regions {
   664→        write_region(&mut w, region);
   665→    }
   666→
   667→    w.into_vec()
   668→}
   669→
   670→/// Deserializes an SCG from the versioned binary format.
   671→///
   672→/// Returns an error if the magic bytes don't match, the version is not
   673→/// supported, or the data is malformed.
   674→pub fn deserialize_scg(data: &[u8]) -> Result<SCG, DeserializeError> {
   675→    let mut reader = BinaryReader::new(data);
   676→
   677→    // Read and validate magic
   678→    let magic = reader.read_bytes(4, "magic")?;
   679→    if magic != MAGIC {
   680→        return Err(DeserializeError::InvalidMagic {
   681→            expected: MAGIC.to_vec(),
   682→            found: magic.to_vec(),
   683→        });
   684→    }
   685→
   686→    // Read and validate version
   687→    let version = reader.read_u32_le("version")?;
   688→    if version < MIN_SUPPORTED_VERSION {
   689→        return Err(DeserializeError::UnsupportedVersion {
   690→            version,
   691→            min_supported: MIN_SUPPORTED_VERSION,
   692→        });
   693→    }
   694→    // Forward compatibility: we could handle unknown future versions here
   695→    // by reading only the fields we understand. For now, we require
   696→    // version == FORMAT_VERSION for strictness, but allow future versions
   697→    // with a warning-style approach (we just try to read the v1 layout).
   698→
   699→    // Read counters and counts
   700→    let next_node_id = reader.read_u64_le("next_node_id")?;
   701→    let next_edge_id = reader.read_u64_le("next_edge_id")?;
   702→    let node_count = reader.read_u32_le("node_count")? as usize;
   703→    let edge_count = reader.read_u32_le("edge_count")? as usize;
   704→    let region_count = reader.read_u32_le("region_count")? as usize;
   705→
   706→    // Read nodes
   707→    let mut nodes = Vec::with_capacity(node_count);
   708→    for i in 0..node_count {
   709→        nodes.push(read_node(&mut reader, i)?);
   710→    }
   711→
   712→    // Read edges
   713→    let mut edges = Vec::with_capacity(edge_count);
   714→    for i in 0..edge_count {
   715→        edges.push(read_edge(&mut reader, i)?);
   716→    }
   717→
   718→    // Read regions
   719→    let mut regions = Vec::with_capacity(region_count);
   720→    for i in 0..region_count {
   721→        regions.push(read_region(&mut reader, i)?);
   722→    }
   723→
   724→    let serialized = SerializedSCG {
   725→        version,
   726→        nodes,
   727→        edges,
   728→        regions,
   729→        next_node_id,
   730→        next_edge_id,
   731→    };
   732→
   733→    serialized_to_scg(serialized)
   734→}
   735→
   736→// ── Binary: Node serialization ──────────────────────────────────────────
   737→
   738→fn write_node(w: &mut BinaryWriter, node: &NodeData) {
   739→    w.write_u64_le(node.id.as_u64());
   740→    w.write_u32_le(node_type_to_tag(&node.node_type));
   741→
   742→    // Annotation (BDReference)
   743→    write_optional_bd_reference(w, &node.annotation);
   744→
   745→    // Program point
   746→    write_program_point(w, &node.program_point);
   747→
   748→    // Payload
   749→    write_payload(w, &node.payload);
   750→}
   751→
   752→fn read_node(reader: &mut BinaryReader, index: usize) -> Result<NodeData, DeserializeError> {
   753→    let ctx = || format!("node[{}]", index);
   754→
   755→    let id = NodeId::new(reader.read_u64_le(&ctx())?);
   756→    let node_type = {
   757→        let tag = reader.read_u32_le(&format!("{}.node_type", ctx()))?;
   758→        tag_to_node_type(tag)?
   759→    };
   760→    let annotation = read_optional_bd_reference(reader, &format!("{}.annotation", ctx()))?;
   761→    let program_point = read_program_point(reader, &format!("{}.program_point", ctx()))?;
   762→    let payload = read_payload(reader, &node_type, &format!("{}.payload", ctx()))?;
   763→
   764→    Ok(NodeData {
   765→        id,
   766→        node_type,
   767→        annotation,
   768→        program_point,
   769→        payload,
   770→    })
   771→}
   772→
   773→fn write_optional_bd_reference(w: &mut BinaryWriter, opt: &Option<BDReference>) {
   774→    match opt {
   775→        Some(bd) => {
   776→            w.write_bool(true);
   777→            w.write_u64_le(bd.bd_id);
   778→            w.write_optional_u64(&bd.version);
   779→        }
   780→        None => {
   781→            w.write_bool(false);
   782→        }
   783→    }
   784→}
   785→
   786→fn read_optional_bd_reference(
   787→    reader: &mut BinaryReader,
   788→    context: &str,
   789→) -> Result<Option<BDReference>, DeserializeError> {
   790→    let has = reader.read_bool(&format!("{}.has", context))?;
   791→    if has {
   792→        let bd_id = reader.read_u64_le(&format!("{}.bd_id", context))?;
   793→        let version = reader.read_optional_u64(&format!("{}.version", context))?;
   794→        Ok(Some(BDReference { bd_id, version }))
   795→    } else {
   796→        Ok(None)
   797→    }
   798→}
   799→
   800→fn write_program_point(w: &mut BinaryWriter, pp: &ProgramPoint) {
   801→    w.write_optional_string(&pp.file);
   802→    w.write_optional_u64(&pp.line);
   803→    w.write_optional_u64(&pp.column);
   804→    w.write_optional_u64(&pp.offset);
   805→}
   806→
   807→fn read_program_point(
   808→    reader: &mut BinaryReader,
   809→    context: &str,
   810→) -> Result<ProgramPoint, DeserializeError> {
   811→    let file = reader.read_optional_string(&format!("{}.file", context))?;
   812→    let line = reader.read_optional_u64(&format!("{}.line", context))?;
   813→    let column = reader.read_optional_u64(&format!("{}.column", context))?;
   814→    let offset = reader.read_optional_u64(&format!("{}.offset", context))?;
   815→    Ok(ProgramPoint {
   816→        file,
   817→        line,
   818→        column,
   819→        offset,
   820→    })
   821→}
   822→
   823→fn write_payload(w: &mut BinaryWriter, payload: &NodePayload) {
   824→    match payload {
   825→        NodePayload::Computation(c) => {
   826→            w.write_u32_le(NODE_TYPE_COMPUTATION);
   827→            w.write_string(&c.kind.label());
   828→            w.write_optional_string(&c.result_type);
   829→        }
   830→        NodePayload::Allocation(a) => {
   831→            w.write_u32_le(NODE_TYPE_ALLOCATION);
   832→            w.write_u64_le(a.size);
   833→            w.write_u64_le(a.align);
   834→            w.write_u64_le(a.region_id.as_u64());
   835→            w.write_optional_string(&a.type_name);
   836→        }
   837→        NodePayload::Deallocation(d) => {
   838→            w.write_u32_le(NODE_TYPE_DEALLOCATION);
   839→            w.write_u64_le(d.allocation_node.as_u64());
   840→            w.write_u64_le(d.region_id.as_u64());
   841→        }
   842→        NodePayload::Access(a) => {
   843→            w.write_u32_le(NODE_TYPE_ACCESS);
   844→            w.write_u32_le(access_mode_to_tag(&a.mode));
   845→            w.write_u64_le(a.region_id.as_u64());
   846→            w.write_optional_u64(&a.offset);
   847→            w.write_optional_u64(&a.access_size);
   848→        }
   849→        NodePayload::Cast(c) => {
   850→            w.write_u32_le(NODE_TYPE_CAST);
   851→            w.write_string(&c.from_type);
   852→            w.write_string(&c.to_type);
   853→            w.write_bool(c.is_lossless);
   854→        }
   855→        NodePayload::Effect(e) => {
   856→            w.write_u32_le(NODE_TYPE_EFFECT);
   857→            w.write_string(&e.effect_kind);
   858→            w.write_bool(e.is_observable);
   859→        }
   860→        NodePayload::Control(c) => {
   861→            w.write_u32_le(NODE_TYPE_CONTROL);
   862→            w.write_u32_le(control_kind_to_tag(&c.kind));
   863→            w.write_optional_string(&c.label);
   864→        }
   865→        NodePayload::Phantom(p) => {
   866→            w.write_u32_le(NODE_TYPE_PHANTOM);
   867→            w.write_string(&p.purpose);
   868→        }
   869→        NodePayload::VTable(v) => {
   870→            w.write_u32_le(NODE_TYPE_VTABLE);
   871→            w.write_string(&v.trait_name);
   872→            w.write_string(&v.concrete_type);
   873→            w.write_u64_le(v.method_entries.len() as u64);
   874→            for &entry_id in &v.method_entries {
   875→                w.write_u64_le(entry_id.as_u64());
   876→            }
   877→        }
   878→        NodePayload::ClosureEnv(e) => {
   879→            w.write_u32_le(NODE_TYPE_CLOSURE_ENV);
   880→            w.write_u64_le(e.captured_vars.len() as u64);
   881→            for var in &e.captured_vars {
   882→                w.write_string(var);
   883→            }
   884→            w.write_u64_le(e.capture_modes.len() as u64);
   885→            for &mode in &e.capture_modes {
   886→                w.write_bool(mode);
   887→            }
   888→            w.write_optional_u64(&e.closure_entry.map(|id| id.as_u64()));
   889→        }
   890→        NodePayload::StructDef(s) => {
   891→            w.write_u32_le(NODE_TYPE_STRUCT_DEF);
   892→            w.write_string(&s.name);
   893→            w.write_u64_le(s.fields.len() as u64);
   894→            for f in &s.fields {
   895→                w.write_string(&f.name);
   896→                w.write_string(&f.type_name);
   897→                w.write_u64_le(f.offset);
   898→                w.write_u64_le(f.size);
   899→            }
   900→            w.write_u64_le(s.total_size);
   901→            w.write_u64_le(s.alignment);
   902→        }
   903→        NodePayload::EnumDef(e) => {
   904→            w.write_u32_le(NODE_TYPE_ENUM_DEF);
   905→            w.write_string(&e.name);
   906→            w.write_u64_le(e.variants.len() as u64);
   907→            for v in &e.variants {
   908→                w.write_string(&v.name);
   909→                w.write_u64_le(v.discriminant);
   910→                w.write_optional_string(&v.payload_type);
   911→                w.write_u64_le(v.payload_size);
   912→            }
   913→            w.write_string(&e.tag_type);
   914→            w.write_u64_le(e.tag_size);
   915→            w.write_u64_le(e.max_payload_size);
   916→            w.write_u64_le(e.total_size);
   917→            w.write_u64_le(e.alignment);
   918→        }
   919→        NodePayload::Match(m) => {
   920→            w.write_u32_le(NODE_TYPE_MATCH);
   921→            w.write_string(&m.subject);
   922→            w.write_string(&m.subject_type);
   923→            w.write_u64_le(m.arms.len() as u64);
   924→            for arm in &m.arms {
   925→                write_match_pattern(w, &arm.pattern);
   926→                w.write_optional_string(&arm.guard);
   927→                w.write_u64_le(arm.body.len() as u64);
   928→                for stmt in &arm.body {
   929→                    w.write_string(stmt);
   930→                }
   931→            }
   932→        }
   933→        NodePayload::ConstantTime(ct) => {
   934→            w.write_u32_le(NODE_TYPE_CONSTANT_TIME);
   935→            w.write_u32_le(match ct.op {
   936→                ConstantTimeOp::CtSelect => 0,
   937→                ConstantTimeOp::CtEq => 1,
   938→            });
   939→            w.write_string(&ct.dst);
   940→            w.write_u64_le(ct.operands.len() as u64);
   941→            for op in &ct.operands {
   942→                w.write_string(op);
   943→            }
   944→        }
   945→    }
   946→}
   947→
   948→fn write_match_pattern(w: &mut BinaryWriter, pattern: &MatchPatternInfo) {
   949→    match pattern {
   950→        MatchPatternInfo::Wildcard => {
   951→            w.write_u32_le(0);
   952→        }
   953→        MatchPatternInfo::Lit(v) => {
   954→            w.write_u32_le(1);
   955→            w.write_u64_le(*v as u64);
   956→        }
   957→        MatchPatternInfo::Ident(name) => {
   958→            w.write_u32_le(2);
   959→            w.write_string(name);
   960→        }
   961→        MatchPatternInfo::Enum { variant, binding } => {
   962→            w.write_u32_le(3);
   963→            w.write_string(variant);
   964→            w.write_optional_string(binding);
   965→        }
   966→        MatchPatternInfo::Struct { name, fields } => {
   967→            w.write_u32_le(4);
   968→            w.write_string(name);
   969→            w.write_u64_le(fields.len() as u64);
   970→            for f in fields {
   971→                w.write_string(f);
   972→            }
   973→        }
   974→        MatchPatternInfo::Or(patterns) => {
   975→            w.write_u32_le(5);
   976→            w.write_u64_le(patterns.len() as u64);
   977→            for p in patterns {
   978→                write_match_pattern(w, p);
   979→            }
   980→        }
   981→    }
   982→}
   983→
   984→fn read_payload(
   985→    reader: &mut BinaryReader,
   986→    expected_type: &NodeType,
   987→    context: &str,
   988→) -> Result<NodePayload, DeserializeError> {
   989→    let tag = reader.read_u32_le(&format!("{}.tag", context))?;
   990→    let payload_type = tag_to_node_type(tag)?;
   991→
   992→    // Validate consistency between node_type and payload tag
   993→    if &payload_type != expected_type {
   994→        return Err(DeserializeError::ConsistencyError(format!(
   995→            "node_type {:?} does not match payload tag {:?}",
   996→            expected_type, payload_type
   997→        )));
   998→    }
   999→
  1000→    match payload_type {
  1001→        NodeType::Computation => {
  1002→            let operation = reader.read_string(&format!("{}.operation", context))?;
  1003→            let result_type = reader.read_optional_string(&format!("{}.result_type", context))?;
  1004→            Ok(NodePayload::Computation(ComputationNode {
  1005→                kind: ComputationKind::Other(operation),
  1006→                result_type,
  1007→                tail_call: false,
  1008→            }))
  1009→        }
  1010→        NodeType::Allocation => {
  1011→            let size = reader.read_u64_le(&format!("{}.size", context))?;
  1012→            let align = reader.read_u64_le(&format!("{}.align", context))?;
  1013→            let region_id = RegionId::new(reader.read_u64_le(&format!("{}.region_id", context))?);
  1014→            let type_name = reader.read_optional_string(&format!("{}.type_name", context))?;
  1015→            Ok(NodePayload::Allocation(AllocationNode {
  1016→                size,
  1017→                align,
  1018→                region_id,
  1019→                type_name,
  1020→            }))
  1021→        }
  1022→        NodeType::Deallocation => {
  1023→            let allocation_node =
  1024→                NodeId::new(reader.read_u64_le(&format!("{}.allocation_node", context))?);
  1025→            let region_id = RegionId::new(reader.read_u64_le(&format!("{}.region_id", context))?);
  1026→            Ok(NodePayload::Deallocation(DeallocationNode {
  1027→                allocation_node,
  1028→                region_id,
  1029→            }))
  1030→        }
  1031→        NodeType::Access => {
  1032→            let mode_tag = reader.read_u32_le(&format!("{}.mode", context))?;
  1033→            let mode = tag_to_access_mode(mode_tag)?;
  1034→            let region_id = RegionId::new(reader.read_u64_le(&format!("{}.region_id", context))?);
  1035→            let offset = reader.read_optional_u64(&format!("{}.offset", context))?;
  1036→            let access_size = reader.read_optional_u64(&format!("{}.access_size", context))?;
  1037→            Ok(NodePayload::Access(AccessNode {
  1038→                mode,
  1039→                region_id,
  1040→                offset,
  1041→                access_size,
  1042→            }))
  1043→        }
  1044→        NodeType::Cast => {
  1045→            let from_type = reader.read_string(&format!("{}.from_type", context))?;
  1046→            let to_type = reader.read_string(&format!("{}.to_type", context))?;
  1047→            let is_lossless = reader.read_bool(&format!("{}.is_lossless", context))?;
  1048→            Ok(NodePayload::Cast(CastNode {
  1049→                from_type,
  1050→                to_type,
  1051→                is_lossless,
  1052→            }))
  1053→        }
  1054→        NodeType::Effect => {
  1055→            let effect_kind = reader.read_string(&format!("{}.effect_kind", context))?;
  1056→            let is_observable = reader.read_bool(&format!("{}.is_observable", context))?;
  1057→            Ok(NodePayload::Effect(EffectNode {
  1058→                effect_kind,
  1059→                is_observable,
  1060→            }))
  1061→        }
  1062→        NodeType::Control => {
  1063→            let kind_tag = reader.read_u32_le(&format!("{}.kind", context))?;
  1064→            let kind = tag_to_control_kind(kind_tag)?;
  1065→            let label = reader.read_optional_string(&format!("{}.label", context))?;
  1066→            Ok(NodePayload::Control(ControlNode { kind, label }))
  1067→        }
  1068→        NodeType::Phantom => {
  1069→            let purpose = reader.read_string(&format!("{}.purpose", context))?;
  1070→            Ok(NodePayload::Phantom(PhantomNode { purpose }))
  1071→        }
  1072→        NodeType::VTable => {
  1073→            let trait_name = reader.read_string(&format!("{}.trait_name", context))?;
  1074→            let concrete_type = reader.read_string(&format!("{}.concrete_type", context))?;
  1075→            let count = reader.read_u64_le(&format!("{}.method_count", context))?;
  1076→            let mut method_entries = Vec::with_capacity(count as usize);
  1077→            for i in 0..count as usize {
  1078→                method_entries.push(NodeId::new(
  1079→                    reader.read_u64_le(&format!("{}.method[{}]", context, i))?,
  1080→                ));
  1081→            }
  1082→            Ok(NodePayload::VTable(VTableNode {
  1083→                trait_name,
  1084→                concrete_type,
  1085→                method_entries,
  1086→            }))
  1087→        }
  1088→        NodeType::ClosureEnv => {
  1089→            let var_count = reader.read_u64_le(&format!("{}.var_count", context))?;
  1090→            let mut captured_vars = Vec::with_capacity(var_count as usize);
  1091→            for i in 0..var_count as usize {
  1092→                captured_vars
  1093→                    .push(reader.read_string(&format!("{}.captured_var[{}]", context, i))?);
  1094→            }
  1095→            let mode_count = reader.read_u64_le(&format!("{}.mode_count", context))?;
  1096→            let mut capture_modes = Vec::with_capacity(mode_count as usize);
  1097→            for i in 0..mode_count as usize {
  1098→                capture_modes.push(reader.read_bool(&format!("{}.capture_mode[{}]", context, i))?);
  1099→            }
  1100→            let closure_entry = reader
  1101→                .read_optional_u64(&format!("{}.closure_entry", context))?
  1102→                .map(NodeId::new);
  1103→            Ok(NodePayload::ClosureEnv(ClosureEnvNode {
  1104→                captured_vars,
  1105→                capture_modes,
  1106→                closure_entry,
  1107→            }))
  1108→        }
  1109→        NodeType::StructDef => {
  1110→            let name = reader.read_string(&format!("{}.name", context))?;
  1111→            let field_count = reader.read_u64_le(&format!("{}.field_count", context))?;
  1112→            let mut fields = Vec::with_capacity(field_count as usize);
  1113→            for i in 0..field_count as usize {
  1114→                let f_name = reader.read_string(&format!("{}.field[{}].name", context, i))?;
  1115→                let f_type = reader.read_string(&format!("{}.field[{}].type", context, i))?;
  1116→                let f_offset = reader.read_u64_le(&format!("{}.field[{}].offset", context, i))?;
  1117→                let f_size = reader.read_u64_le(&format!("{}.field[{}].size", context, i))?;
  1118→                fields.push(StructFieldInfo {
  1119→                    name: f_name,
  1120→                    type_name: f_type,
  1121→                    offset: f_offset,
  1122→                    size: f_size,
  1123→                });
  1124→            }
  1125→            let total_size = reader.read_u64_le(&format!("{}.total_size", context))?;
  1126→            let alignment = reader.read_u64_le(&format!("{}.alignment", context))?;
  1127→            Ok(NodePayload::StructDef(StructDefNode {
  1128→                name,
  1129→                fields,
  1130→                total_size,
  1131→                alignment,
  1132→            }))
  1133→        }
  1134→        NodeType::EnumDef => {
  1135→            let name = reader.read_string(&format!("{}.name", context))?;
  1136→            let variant_count = reader.read_u64_le(&format!("{}.variant_count", context))?;
  1137→            let mut variants = Vec::with_capacity(variant_count as usize);
  1138→            for i in 0..variant_count as usize {
  1139→                let v_name = reader.read_string(&format!("{}.variant[{}].name", context, i))?;
  1140→                let discriminant = reader.read_u64_le(&format!("{}.variant[{}].discriminant", context, i))?;
  1141→                let payload_type = reader.read_optional_string(&format!("{}.variant[{}].payload_type", context, i))?;
  1142→                let payload_size = reader.read_u64_le(&format!("{}.variant[{}].payload_size", context, i))?;
  1143→                variants.push(EnumVariantInfo {
  1144→                    name: v_name,
  1145→                    discriminant,
  1146→                    payload_type,
  1147→                    payload_size,
  1148→                });
  1149→            }
  1150→            let tag_type = reader.read_string(&format!("{}.tag_type", context))?;
  1151→            let tag_size = reader.read_u64_le(&format!("{}.tag_size", context))?;
  1152→            let max_payload_size = reader.read_u64_le(&format!("{}.max_payload_size", context))?;
  1153→            let total_size = reader.read_u64_le(&format!("{}.total_size", context))?;
  1154→            let alignment = reader.read_u64_le(&format!("{}.alignment", context))?;
  1155→            Ok(NodePayload::EnumDef(EnumDefNode {
  1156→                name,
  1157→                variants,
  1158→                tag_type,
  1159→                tag_size,
  1160→                max_payload_size,
  1161→                total_size,
  1162→                alignment,
  1163→            }))
  1164→        }
  1165→        NodeType::Match => {
  1166→            let subject = reader.read_string(&format!("{}.subject", context))?;
  1167→            let subject_type = reader.read_string(&format!("{}.subject_type", context))?;
  1168→            let arm_count = reader.read_u64_le(&format!("{}.arm_count", context))?;
  1169→            let mut arms = Vec::with_capacity(arm_count as usize);
  1170→            for i in 0..arm_count as usize {
  1171→                let pattern = read_match_pattern(reader, &format!("{}.arm[{}].pattern", context, i))?;
  1172→                let guard = reader.read_optional_string(&format!("{}.arm[{}].guard", context, i))?;
  1173→                let body_count = reader.read_u64_le(&format!("{}.arm[{}].body_count", context, i))?;
  1174→                let mut body = Vec::with_capacity(body_count as usize);
  1175→                for j in 0..body_count as usize {
  1176→                    body.push(reader.read_string(&format!("{}.arm[{}].body[{}]", context, i, j))?);
  1177→                }
  1178→                arms.push(MatchArmInfo {
  1179→                    pattern,
  1180→                    guard,
  1181→                    body,
  1182→                });
  1183→            }
  1184→            Ok(NodePayload::Match(MatchNode {
  1185→                subject,
  1186→                arms,
  1187→                subject_type,
  1188→            }))
  1189→        }
  1190→        NodeType::ConstantTime => {
  1191→            let op_tag = reader.read_u32_le(&format!("{}.op", context))?;
  1192→            let op = match op_tag {
  1193→                0 => ConstantTimeOp::CtSelect,
  1194→                1 => ConstantTimeOp::CtEq,
  1195→                _ => return Err(DeserializeError::InvalidValue {
  1196→                    field: format!("{}.op", context),
  1197→                    value: format!("{}", op_tag),
  1198→                }),
  1199→            };
  1200→            let dst = reader.read_string(&format!("{}.dst", context))?;
  1201→            let operand_count = reader.read_u64_le(&format!("{}.operand_count", context))?;
  1202→            let mut operands = Vec::with_capacity(operand_count as usize);
  1203→            for i in 0..operand_count as usize {
  1204→                operands.push(reader.read_string(&format!("{}.operand[{}]", context, i))?);
  1205→            }
  1206→            Ok(NodePayload::ConstantTime(ConstantTimeNode {
  1207→                op,
  1208→                dst,
  1209→                operands,
  1210→            }))
  1211→        }
  1212→    }
  1213→}
  1214→
  1215→fn read_match_pattern(
  1216→    reader: &mut BinaryReader,
  1217→    context: &str,
  1218→) -> Result<MatchPatternInfo, DeserializeError> {
  1219→    let tag = reader.read_u32_le(&format!("{}.tag", context))?;
  1220→    match tag {
  1221→        0 => Ok(MatchPatternInfo::Wildcard),
  1222→        1 => {
  1223→            let v = reader.read_u64_le(&format!("{}.value", context))? as i64;
  1224→            Ok(MatchPatternInfo::Lit(v))
  1225→        }
  1226→        2 => {
  1227→            let name = reader.read_string(&format!("{}.name", context))?;
  1228→            Ok(MatchPatternInfo::Ident(name))
  1229→        }
  1230→        3 => {
  1231→            let variant = reader.read_string(&format!("{}.variant", context))?;
  1232→            let binding = reader.read_optional_string(&format!("{}.binding", context))?;
  1233→            Ok(MatchPatternInfo::Enum { variant, binding })
  1234→        }
  1235→        4 => {
  1236→            let name = reader.read_string(&format!("{}.name", context))?;
  1237→            let field_count = reader.read_u64_le(&format!("{}.field_count", context))?;
  1238→            let mut fields = Vec::with_capacity(field_count as usize);
  1239→            for i in 0..field_count as usize {
  1240→                fields.push(reader.read_string(&format!("{}.field[{}]", context, i))?);
  1241→            }
  1242→            Ok(MatchPatternInfo::Struct { name, fields })
  1243→        }
  1244→        5 => {
  1245→            let pattern_count = reader.read_u64_le(&format!("{}.pattern_count", context))?;
  1246→            let mut patterns = Vec::with_capacity(pattern_count as usize);
  1247→            for i in 0..pattern_count as usize {
  1248→                patterns.push(read_match_pattern(reader, &format!("{}.pattern[{}]", context, i))?);
  1249→            }
  1250→            Ok(MatchPatternInfo::Or(patterns))
  1251→        }
  1252→        _ => Err(DeserializeError::InvalidValue {
  1253→            field: format!("{}.tag", context),
  1254→            value: format!("{}", tag),
  1255→        }),
  1256→    }
  1257→}

  1281→    Ok(EdgeData {

  1292→fn write_region(w: &mut BinaryWriter, region: &SCGRegion) {
  1293→    w.write_u64_le(region.id.as_u64());
  1294→    w.write_u32_le(region.nodes.len() as u32);
  1295→    for node_id in &region.nodes {
  1296→        w.write_u64_le(node_id.as_u64());
  1297→    }
  1298→    w.write_u32_le(region.scope_level);
  1299→    w.write_bool(region.security_boundary);
  1300→    w.write_u32_le(deployment_target_to_tag(&region.deployment_target));
  1301→    if let DeploymentTarget::Custom(ref name) = region.deployment_target {
  1302→        w.write_string(name);
  1303→    }
  1304→}

  1306→fn read_region(reader: &mut BinaryReader, index: usize) -> Result<SCGRegion, DeserializeError> {
  1307→    let ctx = || format!("region[{}]", index);
  1308→
  1309→    let id = RegionId::new(reader.read_u64_le(&ctx())?);
  1310→    let node_count = reader.read_u32_le(&format!("{}.node_count", ctx()))? as usize;
  1311→    let mut nodes = hashbrown::HashSet::with_capacity(node_count);
  1312→    for j in 0..node_count {
  1313→        let nid = NodeId::new(reader.read_u64_le(&format!("{}.nodes[{}]", ctx(), j))?);
  1314→        nodes.insert(nid);
  1315→    }
  1316→    let scope_level = reader.read_u32_le(&format!("{}.scope_level", ctx()))?;
  1317→    let security_boundary = reader.read_bool(&format!("{}.security_boundary", ctx()))?;
  1318→    let dt_tag = reader.read_u32_le(&format!("{}.deployment_target", ctx()))?;
  1319→    let deployment_target = tag_to_deployment_target(dt_tag, reader)?;
  1320→
  1321→    Ok(SCGRegion {
  1322→        id,
  1323→        nodes,
  1324→        scope_level,
  1325→        security_boundary,
  1326→        deployment_target,
  1327→    })
  1328→}

  1352→// ── DOT (Graphviz) serialization ────────────────────────────────────────
  1353→
  1354→/// Serializes an SCG to Graphviz DOT format for visualization.
  1355→///
  1356→/// Produces a directed graph (`digraph`) where:
  1357→/// - Nodes are rendered with labels showing their type and key payload info.
  1358→/// - Edges are styled by kind (solid for data flow, dashed for control flow,
  1359→///   dotted for derivation, bold for annotation).
  1360→/// - Regions are rendered as subgraph clusters.
  1361→///
  1362→/// # Example
  1363→///
  1364→/// ```
  1365→/// use vuma_scg::*;
  1366→/// let mut scg = SCG::new();
  1367→/// let n1 = scg.add_node(
  1368→///     NodeType::Computation,
  1369→///     NodePayload::Computation(ComputationNode {
  1370→///         kind: ComputationKind::Other("add".to_string()),
  1371→///         result_type: None,
  1372→///         tail_call: false }),
  1373→///     ProgramPoint { file: None, line: None, column: None, offset: None },
  1374→/// );
  1375→/// let dot = vuma_scg::serialize::serialize_scg_dot(&scg);
  1376→/// assert!(dot.contains("digraph SCG"));
  1377→/// ```
  1378→pub fn serialize_scg_dot(scg: &SCG) -> String {
  1379→    let mut out = String::with_capacity(4096);
  1380→
  1381→    // Graph header
  1382→    out.push_str("digraph SCG {\n");
  1383→    out.push_str("    rankdir=TB;\n");
  1384→    out.push_str("    node [shape=record, fontname=\"monospace\"];\n");
  1385→    out.push_str("    edge [fontname=\"monospace\"];\n");
  1386→    out.push('\n');
  1387→
  1388→    // Group nodes by region for subgraph clustering
  1389→    let mut node_to_region: std::collections::HashMap<NodeId, RegionId> =
  1390→        std::collections::HashMap::new();
  1391→    for region in scg.regions() {
  1392→        for node_id in region.iter_nodes() {
  1393→            node_to_region.insert(*node_id, region.id);
  1394→        }
  1395→    }
  1396→
  1397→    // Render regions as subgraph clusters
  1398→    for region in scg.regions() {
  1399→        let cluster_name = format!("cluster_region_{}", region.id.as_u64());
  1400→        out.push_str(&format!(
  1401→            "    subgraph {} {{\n",
  1402→            dot_escape_id(&cluster_name)
  1403→        ));
  1404→        let boundary_label = if region.security_boundary {
  1405→            " [SECURITY BOUNDARY]"
  1406→        } else {
  1407→            ""
  1408→        };
  1409→        out.push_str(&format!(
  1410→            "        label=\"Region {} ({}){}\";\n",
  1411→            region.id.as_u64(),
  1412→            region.deployment_target,
  1413→            boundary_label
  1414→        ));
  1415→        out.push_str(&format!(
  1416→            "        style=dashed;\n        color={};\n",
  1417→            if region.security_boundary {
  1418→                "red"
  1419→            } else {
  1420→                "blue"
  1421→            }
  1422→        ));
  1423→
  1424→        for node_id in region.iter_nodes() {
  1425→            if let Some(node_data) = scg.get_node(*node_id) {
  1426→                out.push_str(&format!("        {}\n", format_dot_node(node_data)));
  1427→            }
  1428→        }
  1429→        out.push_str("    }\n\n");
  1430→    }
  1431→
  1432→    // Render nodes not in any region
  1433→    let mut unassigned: Vec<&NodeData> = Vec::new();
  1434→    for node_data in scg.nodes() {
  1435→        if !node_to_region.contains_key(&node_data.id) {
  1436→            unassigned.push(node_data);
  1437→        }
  1438→    }
  1439→    if !unassigned.is_empty() {
  1440→        out.push_str("    subgraph cluster_unassigned {\n");
  1441→        out.push_str("        label=\"Unassigned\";\n        style=dotted;\n        color=gray;\n");
  1442→        for node_data in &unassigned {
  1443→            out.push_str(&format!("        {}\n", format_dot_node(node_data)));
  1444→        }
  1445→        out.push_str("    }\n\n");
  1446→    }
  1447→
  1448→    // Render edges
  1449→    for edge_data in scg.edges() {
  1450→        let style = match edge_data.kind {
  1451→            EdgeKind::DataFlow => "solid",
  1452→            EdgeKind::ControlFlow => "dashed",
  1453→            EdgeKind::Derivation => "dotted",
  1454→            EdgeKind::Annotation => "bold",
  1455→            EdgeKind::Dispatch => "bold dashed",
  1456→            EdgeKind::Call { .. } => "bold solid",
  1457→            EdgeKind::Return { .. } => "bold dotted",
  1458→        };
  1459→        let color = match edge_data.kind {
  1460→            EdgeKind::DataFlow => "black",
  1461→            EdgeKind::ControlFlow => "blue",
  1462→            EdgeKind::Derivation => "gray",
  1463→            EdgeKind::Annotation => "purple",
  1464→            EdgeKind::Dispatch => "red",
  1465→            EdgeKind::Call { .. } => "green",
  1466→            EdgeKind::Return { .. } => "orange",
  1467→        };
  1468→        let kind_str = format!("{}", edge_data.kind);
  1469→        let label = edge_data.label.as_deref().unwrap_or(&kind_str);
  1470→        out.push_str(&format!(
  1471→            "    {} -> {} [style={}, color={}, label=\"{}\"];\n",
  1472→            dot_escape_id(&format!("n{}", edge_data.source.as_u64())),
  1473→            dot_escape_id(&format!("n{}", edge_data.target.as_u64())),
  1474→            style,
  1475→            color,
  1476→            dot_escape_label(label),
  1477→        ));
  1478→    }
  1479→
  1480→    out.push_str("}\n");
  1481→    out
  1482→}
  1483→
  1484→/// Formats a single node for DOT output.
  1485→fn format_dot_node(node: &NodeData) -> String {
  1486→    let label = format_node_label(node);
  1487→    format!(
  1488→        "{} [label=\"{}\"];",
  1489→        dot_escape_id(&format!("n{}", node.id.as_u64())),
  1490→        dot_escape_label(&label)
  1491→    )
  1492→}
  1493→
  1494→/// Creates a human-readable label for a node.
  1495→fn format_node_label(node: &NodeData) -> String {
  1496→    let type_info = match &node.payload {
  1497→        NodePayload::Computation(c) => {
  1498→            let rt = c
  1499→                .result_type
  1500→                .as_deref()
  1501→                .map(|t| format!(":{}", t))
  1502→                .unwrap_or_default();
  1503→            format!("{}{}", c.kind.label(), rt)
  1504→        }
  1505→        NodePayload::Allocation(a) => {
  1506→            let tn = a
  1507→                .type_name
  1508→                .as_deref()
  1509→                .map(|t| format!(" {}", t))
  1510→                .unwrap_or_default();
  1511→            format!("alloc {}B align={}{}", a.size, a.align, tn)
  1512→        }
  1513→        NodePayload::Deallocation(_) => "dealloc".to_string(),
  1514→        NodePayload::Access(a) => {
  1515→            let offset = a.offset.map(|o| format!("+{}", o)).unwrap_or_default();
  1516→            format!("{:?}{} @r{}", a.mode, offset, a.region_id.as_u64())
  1517→        }
  1518→        NodePayload::Cast(c) => format!("{} -> {}", c.from_type, c.to_type),
  1519→        NodePayload::Effect(e) => format!("eff({})", e.effect_kind),
  1520→        NodePayload::Control(c) => {
  1521→            let lbl = c
  1522→                .label
  1523→                .as_deref()
  1524→                .map(|l| format!(" {}", l))
  1525→                .unwrap_or_default();
  1526→            format!("{:?}{}", c.kind, lbl)
  1527→        }
  1528→        NodePayload::Phantom(p) => format!("phantom({})", p.purpose),
  1529→        NodePayload::VTable(v) => format!("vtable({} for {})", v.trait_name, v.concrete_type),
  1530→        NodePayload::ClosureEnv(e) => format!("closure_env({:?})", e.captured_vars),
  1531→        NodePayload::StructDef(s) => format!("struct_def({})", s.name),
  1532→        NodePayload::EnumDef(e) => format!("enum_def({})", e.name),
  1533→        NodePayload::Match(m) => format!("match({})", m.subject),
