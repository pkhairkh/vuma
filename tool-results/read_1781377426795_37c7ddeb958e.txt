     1→//! # MIPS 64-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the MIPS 64-bit target (N64 ABI,
     4→//! big-endian).  This module provides:
     5→//!
     6→//! - `Gpr` — General-purpose register enum ($0–$31)
     7→//! - `Fpr` — Floating-point register enum ($f0–$f31)
     8→//! - `Instruction` — MIPS64 instruction enum with correct encoding for
     9→//!   R-type, I-type, and J-type formats
    10→//! - `Mips64Backend` — `Backend` implementation that lowers IR to MIPS64
    11→//!   machine code and emits ELF64 binaries
    12→//!
    13→//! ## MIPS64 Register Convention (N64 ABI)
    14→//!
    15→//! | Register(s) | ABI Name | Role                              |
    16→//! |-------------|----------|-----------------------------------|
    17→//! | $0          | $zero    | Hardwired zero                    |
    18→//! | $1          | $at      | Assembler temporary               |
    19→//! | $2–$3       | $v0–$v1  | Return values                     |
    20→//! | $4–$7       | $a0–$a3  | Argument registers                |
    21→//! | $8–$15      | $t0–$t7  | Caller-saved temporaries          |
    22→//! | $16–$23     | $s0–$s7  | Callee-saved                      |
    23→//! | $24–$25     | $t8–$t9  | Caller-saved temporaries          |
    24→//! | $26–$27     | $k0–$k1  | Kernel registers                  |
    25→//! | $28         | $gp      | Global pointer                    |
    26→//! | $29         | $sp      | Stack pointer                     |
    27→//! | $30         | $fp      | Frame pointer (callee-saved)      |
    28→//! | $31         | $ra      | Return address                    |
    29→//!
    30→//! ## Branch Delay Slots
    31→//!
    32→//! MIPS has branch delay slots: the instruction immediately following a branch
    33→//! or jump is always executed before the branch takes effect.  This backend
    34→//! inserts a NOP (0x00000000) in every delay slot for correctness.
    35→//!
    36→//! ## Instruction Encoding
    37→//!
    38→//! All instructions are 32 bits, **big-endian**, with three formats:
    39→//!
    40→//! - **R-type**: `opcode[31:26] | rs[25:21] | rt[20:16] | rd[15:11] | sa[10:6] | funct[5:0]`
    41→//! - **I-type**: `opcode[31:26] | rs[25:21] | rt[20:16] | imm[15:0]`
    42→//! - **J-type**: `opcode[31:26] | target[25:0]`
    43→//!
    44→//! ## References
    45→//!
    46→//! - MIPS64 Architecture for Programmers, Volume II: Instruction Set
    47→//! - <https://www.mips.com/products/architectures/mips64/>

   174→    S0 = 16,
   175→    S1 = 17,

   192→impl Gpr {
   193→    /// Returns the 5-bit encoding index for this register.
   194→    pub fn encoding(&self) -> u32 {
   195→        *self as u32
   196→    }
   197→
   198→    /// Returns `true` if this register is available for register allocation.
   199→    ///
   200→    /// Zero ($0), At ($1), K0–K1 ($26–$27), Gp ($28), Sp ($29), and Ra ($31)
   201→    /// are reserved.
   202→    pub fn is_allocatable(&self) -> bool {
   203→        !matches!(
   204→            self,
   205→            Gpr::Zero | Gpr::At | Gpr::K0 | Gpr::K1 | Gpr::Gp | Gpr::Sp | Gpr::Ra
   206→        )
   207→    }
   208→
   209→    /// Returns `true` if this register is callee-saved (s0–s7, fp).
   210→    pub fn is_callee_saved(&self) -> bool {
   211→        matches!(
   212→            self,
   213→            Gpr::S0 | Gpr::S1 | Gpr::S2 | Gpr::S3 | Gpr::S4 | Gpr::S5 | Gpr::S6 | Gpr::S7 | Gpr::Fp
   214→        )
   215→    }
   216→
   217→    /// Returns `true` if this register is an argument register (a0–a3).
   218→    pub fn is_arg_reg(&self) -> bool {
   219→        matches!(self, Gpr::A0 | Gpr::A1 | Gpr::A2 | Gpr::A3)
   220→    }
   221→

   223→    pub fn asm_name(&self) -> &'static str {
   224→        match self {

   226→            Gpr::At => "$at",
   227→            Gpr::V0 => "$v0",
   228→            Gpr::V1 => "$v1",
   229→            Gpr::A0 => "$a0",
   230→            Gpr::A1 => "$a1",
   231→            Gpr::A2 => "$a2",
   232→            Gpr::A3 => "$a3",
   233→            Gpr::T0 => "$t0",
   234→            Gpr::T1 => "$t1",
   235→            Gpr::T2 => "$t2",
   236→            Gpr::T3 => "$t3",
   237→            Gpr::T4 => "$t4",
   238→            Gpr::T5 => "$t5",
   239→            Gpr::T6 => "$t6",
   240→            Gpr::T7 => "$t7",
   241→            Gpr::S0 => "$s0",
   242→            Gpr::S1 => "$s1",
   243→            Gpr::S2 => "$s2",
   244→            Gpr::S3 => "$s3",
   245→            Gpr::S4 => "$s4",
   246→            Gpr::S5 => "$s5",
   247→            Gpr::S6 => "$s6",
   248→            Gpr::S7 => "$s7",
   249→            Gpr::T8 => "$t8",
   250→            Gpr::T9 => "$t9",
   251→            Gpr::K0 => "$k0",
   252→            Gpr::K1 => "$k1",
   253→            Gpr::Gp => "$gp",
   254→            Gpr::Sp => "$sp",
   255→            Gpr::Fp => "$fp",
   256→            Gpr::Ra => "$ra",
   257→        }
   258→    }
   259→
   260→    /// Returns the Gpr for a given argument index (0–3). Returns `None` for

   420→// Instruction Encoding Helpers
   421→// ===========================================================================
   422→
   423→/// Encode an R-type instruction (big-endian).
   424→///
   425→/// Format: `opcode[31:26] | rs[25:21] | rt[20:16] | rd[15:11] | sa[10:6] | funct[5:0]`
   426→fn encode_r_type(opcode: u32, rs: u32, rt: u32, rd: u32, sa: u32, funct: u32) -> [u8; 4] {
   427→    let word = ((opcode & 0x3F) << 26)
   428→        | ((rs & 0x1F) << 21)
   429→        | ((rt & 0x1F) << 16)
   430→        | ((rd & 0x1F) << 11)
   431→        | ((sa & 0x1F) << 6)
   432→        | (funct & 0x3F);
   433→    word.to_be_bytes()
   434→}
   435→
   436→/// Encode an I-type instruction (big-endian).
   437→///
   438→/// Format: `opcode[31:26] | rs[25:21] | rt[20:16] | imm[15:0]`
   439→fn encode_i_type(opcode: u32, rs: u32, rt: u32, imm: u32) -> [u8; 4] {
   440→    let word = ((opcode & 0x3F) << 26) | ((rs & 0x1F) << 21) | ((rt & 0x1F) << 16) | (imm & 0xFFFF);
   441→    word.to_be_bytes()
   442→}
   443→
   444→/// Encode a J-type instruction (big-endian).
   445→///
   446→/// Format: `opcode[31:26] | target[25:0]`
   447→fn encode_j_type(opcode: u32, target: u32) -> [u8; 4] {
   448→    let word = ((opcode & 0x3F) << 26) | (target & 0x03FF_FFFF);
   449→    word.to_be_bytes()
   450→}
   451→
   452→/// Encode a NOP instruction (0x00000000).
   453→fn encode_nop() -> [u8; 4] {
   454→    0x00000000u32.to_be_bytes()
   455→}

   457→// ===========================================================================
   458→// Instruction Enum
   459→// ===========================================================================
   460→
   461→/// MIPS64 instruction representations for code generation.
   462→///
   463→/// Covers key R-type, I-type, and J-type instructions from the MIPS64 ISA.
   464→/// Each variant captures the operands needed for encoding and disassembly.
   465→/// The `encode()` method produces a 4-byte **big-endian** machine code word.
   466→///
   467→/// Branch delay slots are handled by the `has_delay_slot()` method: when it
   468→/// returns `true`, the caller must insert a NOP after the instruction.
   469→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   470→pub enum Instruction {
   471→    // ── R-type: Arithmetic (32-bit) ────────────────────────────────────
   472→    /// Add: `add rd, rs, rt`
   473→    Add { rd: Gpr, rs: Gpr, rt: Gpr },
   474→    /// Add Unsigned: `addu rd, rs, rt`
   475→    Addu { rd: Gpr, rs: Gpr, rt: Gpr },
   476→    /// Subtract: `sub rd, rs, rt`
   477→    Sub { rd: Gpr, rs: Gpr, rt: Gpr },
   478→    /// Subtract Unsigned: `subu rd, rs, rt`
   479→    Subu { rd: Gpr, rs: Gpr, rt: Gpr },
   480→
   481→    // ── R-type: Logical ────────────────────────────────────────────────
   482→    /// AND: `and rd, rs, rt`
   483→    And { rd: Gpr, rs: Gpr, rt: Gpr },
   484→    /// OR: `or rd, rs, rt`
   485→    Or { rd: Gpr, rs: Gpr, rt: Gpr },
   486→    /// XOR: `xor rd, rs, rt`
   487→    Xor { rd: Gpr, rs: Gpr, rt: Gpr },
   488→    /// NOR: `nor rd, rs, rt`
   489→    Nor { rd: Gpr, rs: Gpr, rt: Gpr },
   490→
   491→    // ── R-type: Set on Less Than ───────────────────────────────────────
   492→    /// Set on Less Than (signed): `slt rd, rs, rt`
   493→    Slt { rd: Gpr, rs: Gpr, rt: Gpr },
   494→    /// Set on Less Than (unsigned): `sltu rd, rs, rt`
   495→    Sltu { rd: Gpr, rs: Gpr, rt: Gpr },
   496→
   497→    // ── R-type: Shift (immediate sa) ───────────────────────────────────
   498→    /// Shift Left Logical: `sll rd, rt, sa`
   499→    Sll { rd: Gpr, rt: Gpr, sa: u32 },
   500→    /// Shift Right Logical: `srl rd, rt, sa`
   501→    Srl { rd: Gpr, rt: Gpr, sa: u32 },
   502→    /// Shift Right Arithmetic: `sra rd, rt, sa`
   503→    Sra { rd: Gpr, rt: Gpr, sa: u32 },
   504→
   505→    // ── R-type: Shift (variable) ───────────────────────────────────────
   506→    /// Shift Left Logical Variable: `sllv rd, rt, rs`
   507→    Sllv { rd: Gpr, rt: Gpr, rs: Gpr },
   508→    /// Shift Right Logical Variable: `srlv rd, rt, rs`
   509→    Srlv { rd: Gpr, rt: Gpr, rs: Gpr },
   510→    /// Shift Right Arithmetic Variable: `srav rd, rt, rs`
   511→    Srav { rd: Gpr, rt: Gpr, rs: Gpr },
   512→
   513→    // ── R-type: Multiply/Divide (32-bit) ───────────────────────────────
   514→    /// Multiply (signed): `mult rs, rt` → HI:LO
   515→    Mult { rs: Gpr, rt: Gpr },
   516→    /// Multiply (unsigned): `multu rs, rt` → HI:LO
   517→    Multu { rs: Gpr, rt: Gpr },
   518→    /// Divide (signed): `div rs, rt` → LO = quotient, HI = remainder
   519→    Div { rs: Gpr, rt: Gpr },
   520→    /// Divide (unsigned): `divu rs, rt` → LO = quotient, HI = remainder
   521→    Divu { rs: Gpr, rt: Gpr },
   522→

   660→    /// Jump: `j target`
   661→    J { target: u32 },
   662→    /// Jump and Link: `jal target`
   663→    Jal { target: u32 },
   664→
   665→    // ── No-op ──────────────────────────────────────────────────────────
   666→    /// No-operation (encoded as `sll $zero, $zero, 0` = 0x00000000).
   667→    Nop,
   668→}

   670→impl Instruction {
   671→    /// Encode this instruction into a 4-byte **big-endian** machine code word.
   672→    ///
   673→    /// Encoding follows the MIPS64 ISA Specification.

   675→        match self {

   678→                OPC_SPECIAL,

   683→                FN_ADD,
   684→            ),

   698→                0,
   699→                FN_SUB,
   700→            ),

   707→                FN_SUBU,
   708→            ),

   713→                rs.encoding(),
   714→                rt.encoding(),
   715→                rd.encoding(),
   716→                0,
   717→                FN_AND,

   760→            ),

   762→            // ── R-type: Shift (immediate sa) ───────────────────────────
   763→            Instruction::Sll { rd, rt, sa } => encode_r_type(

   816→                encode_r_type(OPC_SPECIAL, rs.encoding(), rt.encoding(), 0, 0, FN_MULT)
   817→            }
   818→            Instruction::Multu { rs, rt } => {
   819→                encode_r_type(OPC_SPECIAL, rs.encoding(), rt.encoding(), 0, 0, FN_MULTU)
   820→            }

   832→            // ── R-type: Arithmetic (64-bit) ────────────────────────────

   868→                OPC_SPECIAL,
   869→                0,
   870→                rt.encoding(),

   873→                FN_DSLL,

   885→                0,

   887→                rd.encoding(),
   888→                *sa & 0x1F,
   889→                FN_DSRA,
   890→            ),
   891→
   892→            // ── R-type: Shift (64-bit, variable) ──────────────────────
   893→            Instruction::Dsllv { rd, rt, rs } => encode_r_type(
   894→                OPC_SPECIAL,
   895→                rs.encoding(),
   896→                rt.encoding(),
   897→                rd.encoding(),
   898→                0,
   899→                FN_DSLLV,

   919→            Instruction::Dmult { rs, rt } => {

   921→            }

   924→            }

   930→            }
   931→
   932→            // ── R-type: Conditional Move ───────────────────────────────
   933→            Instruction::Movz { rd, rs, rt } => encode_r_type(
   934→                OPC_SPECIAL,
   935→                rs.encoding(),
   936→                rt.encoding(),
   937→                rd.encoding(),
   938→                0,
   939→                FN_MOVZ,
   940→            ),
   941→            Instruction::Movn { rd, rs, rt } => encode_r_type(
   942→                OPC_SPECIAL,
   943→                rs.encoding(),
   944→                rt.encoding(),
   945→                rd.encoding(),
   946→                0,
   947→                FN_MOVN,
   948→            ),
   949→
   950→            // ── R-type: Jump Register ──────────────────────────────────
   951→            Instruction::Jr { rs } => encode_r_type(OPC_SPECIAL, rs.encoding(), 0, 0, 0, FN_JR),
   952→            Instruction::Jalr { rd, rs } => {
   953→                encode_r_type(OPC_SPECIAL, rs.encoding(), 0, rd.encoding(), 0, FN_JALR)
   954→            }
   955→
   956→            // ── R-type: System ─────────────────────────────────────────
   957→            Instruction::Syscall { code } => {
   958→                // syscall: bits[25:6] = code, funct = 0x0C
   959→                let word = (OPC_SPECIAL << 26) | ((*code & 0xFFFFF) << 6) | FN_SYSCALL;
   960→                word.to_be_bytes()
   961→            }
   962→            Instruction::Break { code } => {
   963→                // break: bits[25:6] = code, funct = 0x0D
   964→                let word = (OPC_SPECIAL << 26) | ((*code & 0xFFFFF) << 6) | FN_BREAK;
   965→                word.to_be_bytes()
   966→            }
   967→
   968→            // ── I-type: Immediate Arithmetic (32-bit) ─────────────────
   969→            Instruction::Addi { rt, rs, imm } => encode_i_type(
   970→                OPC_ADDI,
   971→                rs.encoding(),
   972→                rt.encoding(),
   973→                (*imm as u32) & 0xFFFF,
   974→            ),
   975→            Instruction::Addiu { rt, rs, imm } => encode_i_type(
   976→                OPC_ADDIU,

   978→                rt.encoding(),

   980→            ),
   981→
   982→            // ── I-type: Immediate Logical ──────────────────────────────

   984→                encode_i_type(OPC_ANDI, rs.encoding(), rt.encoding(), *imm & 0xFFFF)
   985→            }
   986→            Instruction::Ori { rt, rs, imm } => {
   987→                encode_i_type(OPC_ORI, rs.encoding(), rt.encoding(), *imm & 0xFFFF)
   988→            }
   989→            Instruction::Xori { rt, rs, imm } => {
   990→                encode_i_type(OPC_XORI, rs.encoding(), rt.encoding(), *imm & 0xFFFF)
   991→            }
   992→
   993→            // ── I-type: Set on Less Than Immediate ─────────────────────
   994→            Instruction::Slti { rt, rs, imm } => encode_i_type(
   995→                OPC_SLTI,
   996→                rs.encoding(),
   997→                rt.encoding(),
   998→                (*imm as u32) & 0xFFFF,
   999→            ),
  1000→            Instruction::Sltiu { rt, rs, imm } => encode_i_type(
  1001→                OPC_SLTIU,
  1002→                rs.encoding(),
  1003→                rt.encoding(),
  1004→                (*imm as u32) & 0xFFFF,
  1005→            ),
  1006→
  1007→            // ── I-type: Upper Immediate ────────────────────────────────
  1008→            Instruction::Lui { rt, imm } => encode_i_type(OPC_LUI, 0, rt.encoding(), *imm & 0xFFFF),
  1009→
  1010→            // ── I-type: Immediate Arithmetic (64-bit) ─────────────────
  1011→            Instruction::Daddi { rt, rs, imm } => encode_i_type(
  1012→                OPC_DADDI,
  1013→                rs.encoding(),
  1014→                rt.encoding(),
  1015→                (*imm as u32) & 0xFFFF,
  1016→            ),
  1017→            Instruction::Daddiu { rt, rs, imm } => encode_i_type(
  1018→                OPC_DADDIU,
  1019→                rs.encoding(),
  1020→                rt.encoding(),
  1021→                (*imm as u32) & 0xFFFF,
  1022→            ),
  1023→
  1024→            // ── I-type: Branch ─────────────────────────────────────────
  1025→            Instruction::Beq { rs, rt, offset } => {
  1026→                // Offset is in words, shifted left 2 by the hardware.
  1027→                let off_words = (*offset >> 2) as u32;
  1028→                encode_i_type(OPC_BEQ, rs.encoding(), rt.encoding(), off_words & 0xFFFF)
  1029→            }
  1030→            Instruction::Bne { rs, rt, offset } => {
  1031→                let off_words = (*offset >> 2) as u32;

  1341→/// Build a proper ELF64 binary for MIPS64 (big-endian) with 2 LOAD segments.
  1342→///
  1343→/// Produces a static executable with:
  1344→/// - Segment 1: LOAD RX — `.text` section (code)
  1345→/// - Segment 2: LOAD RW — `.data` / BSS (writable memory for stack/data)
  1346→///
  1347→/// All header fields are written in big-endian byte order.
  1348→/// e_flags is set to EF_MIPS_ARCH_64 (0x60000000) for the MIPS64 N64 ABI.
  1349→fn build_mips64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1350→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (MIPS typical page size)
  1351→
  1352→    let elf_header_size: u64 = 64;
  1353→    let phdr_size: u64 = 56;
  1354→    let num_phdrs: u64 = 2;
  1355→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1356→    // Page-align the text segment start in the file.
  1357→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1358→    let text_size = code.len() as u64;
  1359→
  1360→    // The data segment starts on the next page after the text.
  1361→    let text_file_end = text_offset + text_size;
  1362→    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1363→    let data_offset = data_vaddr - base_addr;
  1364→    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1365→    let entry_point = base_addr + text_offset;
  1366→
  1367→    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1368→
  1369→    // --- e_ident ---
  1370→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1371→    elf.push(2); // ELFCLASS64
  1372→    elf.push(2); // ELFDATA2MSB (big-endian)
  1373→    elf.push(1); // EV_CURRENT
  1374→    elf.push(3); // ELFOSABI_LINUX
  1375→    elf.push(0); // padding
  1376→    elf.extend_from_slice(&[0u8; 7]); // padding
  1377→
  1378→    // --- ELF header fields (big-endian) ---
  1379→    elf.extend_from_slice(&2u16.to_be_bytes()); // e_type = ET_EXEC
  1380→    elf.extend_from_slice(&8u16.to_be_bytes()); // e_machine = EM_MIPS
  1381→    elf.extend_from_slice(&1u32.to_be_bytes()); // e_version
  1382→    elf.extend_from_slice(&entry_point.to_be_bytes()); // e_entry
  1383→    elf.extend_from_slice(&elf_header_size.to_be_bytes()); // e_phoff
  1384→    elf.extend_from_slice(&0u64.to_be_bytes()); // e_shoff (no section headers)
  1385→    // e_flags: EF_MIPS_ARCH_64 = 0x60000000 (MIPS64 ISA, N64 ABI implied by ELFCLASS64)
  1386→    elf.extend_from_slice(&0x60000000u32.to_be_bytes()); // e_flags
  1387→    elf.extend_from_slice(&64u16.to_be_bytes()); // e_ehsize
  1388→    elf.extend_from_slice(&56u16.to_be_bytes()); // e_phentsize
  1389→    elf.extend_from_slice(&2u16.to_be_bytes()); // e_phnum = 2
  1390→    elf.extend_from_slice(&64u16.to_be_bytes()); // e_shentsize
  1391→    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shnum
  1392→    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shstrndx
  1393→
  1394→    // --- Program Header 1: LOAD (PF_R | PF_X) — .text ---
  1395→    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1396→    elf.extend_from_slice(&5u32.to_be_bytes()); // p_flags = PF_R | PF_X
  1397→    elf.extend_from_slice(&text_offset.to_be_bytes()); // p_offset
  1398→    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_vaddr
  1399→    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_paddr
  1400→    elf.extend_from_slice(&text_size.to_be_bytes()); // p_filesz
  1401→    elf.extend_from_slice(&text_size.to_be_bytes()); // p_memsz
  1402→    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1403→
  1404→    // --- Program Header 2: LOAD (PF_R | PF_W) — .data / stack ---
  1405→    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1406→    elf.extend_from_slice(&6u32.to_be_bytes()); // p_flags = PF_R | PF_W
  1407→    elf.extend_from_slice(&data_offset.to_be_bytes()); // p_offset
  1408→    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_vaddr
  1409→    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_paddr
  1410→    elf.extend_from_slice(&0u64.to_be_bytes()); // p_filesz (no initialized data)
  1411→    elf.extend_from_slice(&data_size.to_be_bytes()); // p_memsz (writable pages)
  1412→    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1413→
  1414→    // --- .text section ---
  1415→    // Pad to page-aligned text_offset
  1416→    while (elf.len() as u64) < text_offset {
  1417→        elf.push(0);
  1418→    }
  1419→    elf.extend_from_slice(code);
  1420→
  1421→    // --- Pad to data segment offset (if needed) ---
  1422→    while (elf.len() as u64) < data_offset {
  1423→        elf.push(0);
  1424→    }
  1425→
  1426→    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1427→
  1428→    elf
  1429→}

  1459→/// Compute the stack frame size for an IR function on MIPS64.
  1460→///
  1461→/// Sums `Alloc` instruction sizes, adds 8 bytes for the $ra save slot,
  1462→/// and rounds up to 16-byte alignment.
  1463→fn mips64_compute_frame_size(func: &IRFunction) -> usize {
  1464→    let mut total: u32 = 8; // $ra save slot
  1465→    for block in &func.blocks {
  1466→        for instr in &block.instructions {
  1467→            if let IRInstr::Alloc { size, .. } = instr {
  1468→                let aligned = (*size).div_ceil(16) * 16;
  1469→                total += aligned;
  1470→            }
  1471→        }
  1472→    }
  1473→    // Round up to 16-byte alignment
  1474→    total = (total + 15) & !15;
  1475→    total as usize
  1476→}
  1477→
  1478→/// Allocatable GPR registers for MIPS64, in priority order.
  1479→///
  1480→/// Order: temporaries first, then argument registers, then callee-saved.
  1481→const ALLOCATABLE_GPRS: &[Gpr] = &[
  1482→    // Caller-saved temporaries (highest priority — no save/restore needed)
  1483→    Gpr::T0,
  1484→    Gpr::T1,
  1485→    Gpr::T2,
  1486→    Gpr::T3,
  1487→    Gpr::T4,
  1488→    Gpr::T5,
  1489→    Gpr::T6,
  1490→    Gpr::T7,
  1491→    Gpr::T8,
  1492→    Gpr::T9,
  1493→    // Return value registers (also caller-saved)
  1494→    Gpr::V0,
  1495→    Gpr::V1,
  1496→    // Argument registers (also caller-saved)
  1497→    Gpr::A0,
  1498→    Gpr::A1,
  1499→    Gpr::A2,
  1500→    Gpr::A3,
  1501→    // Callee-saved (require save/restore)
  1502→    Gpr::S0,
  1503→    Gpr::S1,
  1504→    Gpr::S2,
  1505→    Gpr::S3,
  1506→    Gpr::S4,
  1507→    Gpr::S5,
  1508→    Gpr::S6,
  1509→    Gpr::S7,
  1510→    // Frame pointer is last — we prefer not to use it
  1511→    Gpr::Fp,
  1512→];
  1513→
  1514→/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1515→///
  1516→/// Argument virtual registers are mapped to a0–a3 first.  Remaining virtual
  1517→/// registers are assigned from the pool of allocatable registers.
  1518→fn map_vreg_to_gpr(
  1519→    vreg_id: u32,
  1520→    arg_index: Option<usize>,
  1521→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1522→) -> Gpr {
  1523→    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1524→        return *gpr;
  1525→    }
  1526→
  1527→    // If this is an argument, map it to the corresponding argument register.
  1528→    if let Some(idx) = arg_index {
  1529→        if let Some(reg) = Gpr::arg_register(idx) {
  1530→            vreg_map.insert(vreg_id, reg);
  1531→            return reg;
  1532→        }
  1533→    }
  1534→
  1535→    // Otherwise, pick the first available allocatable register.
  1536→    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1537→    for &reg in ALLOCATABLE_GPRS {
  1538→        if !used.contains(&reg) {
  1539→            vreg_map.insert(vreg_id, reg);
  1540→            return reg;
  1541→        }
  1542→    }
  1543→
  1544→    // Fallback: use t9 as a scratch register (will cause issues, but shouldn't
  1545→    // happen with reasonable register pressure).
  1546→    Gpr::T9
  1547→}
  1548→
  1549→/// Helper: extract the virtual register ID from an IRValue, if it is a register.
  1550→fn vreg_id(val: &IRValue) -> u32 {
  1551→    match val {
  1552→        IRValue::Register(id) => *id,
  1553→        _ => 0,
  1554→    }
  1555→}
  1556→
  1557→/// Resolve an IRValue to a physical GPR, emitting load-immediate instructions
  1558→/// if the value is an immediate constant.
  1559→///
  1560→/// Returns a tuple of `(register, instructions)`.  The `instructions` vector
  1561→/// contains any AllocatedInstructions needed to load an immediate into a
  1562→/// register (empty if the value was already in a register).
  1563→///
  1564→/// `imm_counter` is used to generate unique virtual register IDs for immediate
  1565→/// temporaries.  It is incremented each time an immediate is loaded.
  1566→fn resolve_value(
  1567→    val: &IRValue,
  1568→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1569→    imm_counter: &mut u32,
  1570→) -> (Gpr, Vec<AllocatedInstruction>) {
  1571→    if let Some(imm) = val.as_immediate() {
  1572→        // Allocate a unique vreg ID for this immediate
  1573→        let vreg_id = *imm_counter;
  1574→        *imm_counter += 1;
  1575→        let dst = map_vreg_to_gpr(vreg_id, None, vreg_map);
  1576→        let mut code = Vec::new();
  1577→
  1578→        // Load the 64-bit immediate into dst.
  1579→        // MIPS64: use lui + ori + dsll + daddiu sequence for full 64-bit.
  1580→        let u64_val = imm as u64;
  1581→        let highest = (u64_val >> 48) & 0xFFFF;
  1582→        let higher  = (u64_val >> 32) & 0xFFFF;
  1583→        let hi      = (u64_val >> 16) & 0xFFFF;
  1584→        let lo      = u64_val & 0xFFFF;
  1585→
  1586→        if imm >= -32768 && imm <= 32767 {
  1587→            // Fits in 16-bit sign-extended: addiu dst, $zero, imm
  1588→            let inst = Instruction::Addiu { rt: dst, rs: Gpr::Zero, imm: imm as i32 };
  1589→            code.push(AllocatedInstruction {
  1590→                opcode: "addiu".to_string(),
  1591→                reads: vec![],
  1592→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1593→                encoded: inst.encode().to_vec(),
  1594→            });
  1595→        } else if imm >= 0 && imm <= 65535 {
  1596→            // 16-bit unsigned: ori dst, $zero, imm
  1597→            let inst = Instruction::Ori { rt: dst, rs: Gpr::Zero, imm: imm as u32 };
  1598→            code.push(AllocatedInstruction {
  1599→                opcode: "ori".to_string(),
  1600→                reads: vec![],
  1601→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1602→                encoded: inst.encode().to_vec(),
  1603→            });
  1604→        } else if imm >= -2147483648 && imm <= 2147483647 && (imm as i32 >> 16) == 0 || (hi == 0 && (lo as i16 as i32) < 0) {
  1605→            // 32-bit value: lui dst, upper; ori dst, dst, lower
  1606→            let lui_val = ((imm as i32 as u32) >> 16) as u32;
  1607→            let ori_val = (imm as i32 as u32) & 0xFFFF;
  1608→            let lui_inst = Instruction::Lui { rt: dst, imm: lui_val };
  1609→            code.push(AllocatedInstruction {
  1610→                opcode: "lui".to_string(),
  1611→                reads: vec![],
  1612→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1613→                encoded: lui_inst.encode().to_vec(),
  1614→            });
  1615→            if ori_val != 0 {
  1616→                let ori_inst = Instruction::Ori { rt: dst, rs: dst, imm: ori_val };
  1617→                code.push(AllocatedInstruction {
  1618→                    opcode: "ori".to_string(),
  1619→                    reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1620→                    writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1621→                    encoded: ori_inst.encode().to_vec(),
  1622→                });
  1623→            }
  1624→        } else if hi != 0 || lo != 0 {
  1625→            // Full 64-bit immediate load
  1626→            // lui dst, highest
  1627→            let lui_inst = Instruction::Lui { rt: dst, imm: highest as u32 };
  1628→            code.push(AllocatedInstruction {
  1629→                opcode: "lui".to_string(),
  1630→                reads: vec![],
  1631→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1632→                encoded: lui_inst.encode().to_vec(),
  1633→            });
  1634→            // daddiu dst, dst, higher
  1635→            let daddiu_inst = Instruction::Daddiu { rt: dst, rs: dst, imm: higher as i32 };
  1636→            code.push(AllocatedInstruction {
  1637→                opcode: "daddiu".to_string(),
  1638→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1639→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1640→                encoded: daddiu_inst.encode().to_vec(),
  1641→            });
  1642→            // dsll dst, dst, 16
  1643→            let dsll_inst = Instruction::Dsll { rd: dst, rt: dst, sa: 16 };
  1644→            code.push(AllocatedInstruction {
  1645→                opcode: "dsll".to_string(),
  1646→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1647→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1648→                encoded: dsll_inst.encode().to_vec(),
  1649→            });
  1650→            // daddiu dst, dst, hi
  1651→            let daddiu_hi = Instruction::Daddiu { rt: dst, rs: dst, imm: hi as i32 };
  1652→            code.push(AllocatedInstruction {
  1653→                opcode: "daddiu".to_string(),
  1654→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1655→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1656→                encoded: daddiu_hi.encode().to_vec(),
  1657→            });
  1658→            // dsll dst, dst, 16
  1659→            let dsll2 = Instruction::Dsll { rd: dst, rt: dst, sa: 16 };
  1660→            code.push(AllocatedInstruction {
  1661→                opcode: "dsll".to_string(),
  1662→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1663→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1664→                encoded: dsll2.encode().to_vec(),
  1665→            });
  1666→            // daddiu dst, dst, lo
  1667→            let daddiu_lo = Instruction::Daddiu { rt: dst, rs: dst, imm: lo as i32 };
  1668→            code.push(AllocatedInstruction {
  1669→                opcode: "daddiu".to_string(),
  1670→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1671→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1672→                encoded: daddiu_lo.encode().to_vec(),
  1673→            });
  1674→        } else {
  1675→            // 32-bit value in the upper 32 bits
  1676→            let lui_inst = Instruction::Lui { rt: dst, imm: highest as u32 };
  1677→            code.push(AllocatedInstruction {
  1678→                opcode: "lui".to_string(),
  1679→                reads: vec![],
  1680→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1681→                encoded: lui_inst.encode().to_vec(),
  1682→            });
  1683→            let daddiu_inst = Instruction::Daddiu { rt: dst, rs: dst, imm: higher as i32 };
  1684→            code.push(AllocatedInstruction {
  1685→                opcode: "daddiu".to_string(),
  1686→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1687→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1688→                encoded: daddiu_inst.encode().to_vec(),
  1689→            });
  1690→            let dsll_inst = Instruction::Dsll { rd: dst, rt: dst, sa: 16 };
  1691→            code.push(AllocatedInstruction {
  1692→                opcode: "dsll".to_string(),
  1693→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1694→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1695→                encoded: dsll_inst.encode().to_vec(),
  1696→            });
  1697→            let daddiu_hi = Instruction::Daddiu { rt: dst, rs: dst, imm: hi as i32 };
  1698→            code.push(AllocatedInstruction {
  1699→                opcode: "daddiu".to_string(),
  1700→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1701→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1702→                encoded: daddiu_hi.encode().to_vec(),
  1703→            });
  1704→            let dsll2 = Instruction::Dsll { rd: dst, rt: dst, sa: 16 };
  1705→            code.push(AllocatedInstruction {
  1706→                opcode: "dsll".to_string(),
  1707→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1708→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1709→                encoded: dsll2.encode().to_vec(),
  1710→            });
  1711→        }
  1712→
  1713→        (dst, code)
  1714→    } else {
  1715→        // Register value — just map it
  1716→        (map_vreg_to_gpr(vreg_id(val), None, vreg_map), Vec::new())
  1717→    }
  1718→}

  1720→/// Lower a BinOpKind + operands to MIPS64 instructions.
  1721→fn lower_binop(
  1722→    op: &BinOpKind,
  1723→    dst: &IRValue,
  1724→    lhs: &IRValue,
  1725→    rhs: &IRValue,
  1726→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1727→    imm_counter: &mut u32,
  1728→) -> Vec<AllocatedInstruction> {
  1729→    let mut result = Vec::new();
  1730→    let dst_reg = map_vreg_to_gpr(vreg_id(dst), None, vreg_map);
  1731→    let (lhs_reg, lhs_load) = resolve_value(lhs, vreg_map, imm_counter);
  1732→    let (rhs_reg, rhs_load) = resolve_value(rhs, vreg_map, imm_counter);
  1733→    result.extend(lhs_load);
  1734→    result.extend(rhs_load);
  1735→
  1736→    let mips_inst = match op {
  1737→        BinOpKind::Add => Instruction::Daddu {
  1738→            rd: dst_reg,
  1739→            rs: lhs_reg,
  1740→            rt: rhs_reg,
  1741→        },
  1742→        BinOpKind::Sub => Instruction::Dsubu {
  1743→            rd: dst_reg,
  1744→            rs: lhs_reg,
  1745→            rt: rhs_reg,
  1746→        },
  1747→        BinOpKind::Mul => Instruction::Dmult {
  1748→            rs: lhs_reg,
  1749→            rt: rhs_reg,
  1750→        },
  1751→        BinOpKind::SDiv => Instruction::Ddiv {
  1752→            rs: lhs_reg,
  1753→            rt: rhs_reg,
  1754→        },
  1755→        BinOpKind::UDiv => Instruction::Ddivu {
  1756→            rs: lhs_reg,
  1757→            rt: rhs_reg,
  1758→        },
  1759→        BinOpKind::SRem => Instruction::Ddiv {
  1760→            rs: lhs_reg,
  1761→            rt: rhs_reg,
  1762→        },
  1763→        BinOpKind::URem => Instruction::Ddivu {
  1764→            rs: lhs_reg,
  1765→            rt: rhs_reg,
  1766→        },
  1767→        BinOpKind::And => Instruction::And {
  1768→            rd: dst_reg,
  1769→            rs: lhs_reg,
  1770→            rt: rhs_reg,
  1771→        },
  1772→        BinOpKind::Or => Instruction::Or {
  1773→            rd: dst_reg,
  1774→            rs: lhs_reg,
  1775→            rt: rhs_reg,
  1776→        },
  1777→        BinOpKind::Xor => Instruction::Xor {
  1778→            rd: dst_reg,
  1779→            rs: lhs_reg,
  1780→            rt: rhs_reg,
  1781→        },
  1782→        BinOpKind::Shl => Instruction::Dsllv {
  1783→            rd: dst_reg,
  1784→            rt: lhs_reg,
  1785→            rs: rhs_reg,
  1786→        },
  1787→        BinOpKind::ShrL => Instruction::Dsrlv {
  1788→            rd: dst_reg,
  1789→            rt: lhs_reg,
  1790→            rs: rhs_reg,
  1791→        },
  1792→        BinOpKind::ShrA => Instruction::Dsrav {
  1793→            rd: dst_reg,
  1794→            rt: lhs_reg,
  1795→            rs: rhs_reg,
  1796→        },
  1797→        BinOpKind::Ror | BinOpKind::Rol => Instruction::Dsrlv {
  1798→            rd: dst_reg,
  1799→            rt: lhs_reg,
  1800→            rs: rhs_reg,
  1801→        },
  1802→        BinOpKind::SLt => Instruction::Slt {
  1803→            rd: dst_reg,
  1804→            rs: lhs_reg,
  1805→            rt: rhs_reg,
  1806→        },
  1807→        BinOpKind::SLe => {
  1808→            // slt rd, rhs, lhs ; xori rd, rd, 1  (rd = !(rhs < lhs) = lhs <= rhs)
  1809→            result.push(AllocatedInstruction {
  1810→                opcode: "slt".to_string(),
  1811→                reads: vec![
  1812→                    PhysicalReg::new(RegClass::Gpr, rhs_reg.encoding()),
  1813→                    PhysicalReg::new(RegClass::Gpr, lhs_reg.encoding()),
  1814→                ],
  1815→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())],
  1816→                encoded: Instruction::Slt {
  1817→                    rd: dst_reg,
  1818→                    rs: rhs_reg,
  1819→                    rt: lhs_reg,
  1820→                }
  1821→                .encode()
  1822→                .to_vec(),
  1823→            });
  1824→            result.push(AllocatedInstruction {
  1825→                opcode: "xori".to_string(),
  1826→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())],
  1827→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())],
  1828→                encoded: Instruction::Xori {
  1829→                    rt: dst_reg,
  1830→                    rs: dst_reg,
  1831→                    imm: 1,
  1832→                }
  1833→                .encode()
  1834→                .to_vec(),
  1835→            });
  1836→            return result;
  1837→        }
  1838→        BinOpKind::SGt => {
  1839→            // slt rd, rhs, lhs (rd = rhs < lhs = lhs > rhs)
  1840→            result.push(AllocatedInstruction {
  1841→                opcode: "slt".to_string(),
  1842→                reads: vec![
  1843→                    PhysicalReg::new(RegClass::Gpr, rhs_reg.encoding()),
  1844→                    PhysicalReg::new(RegClass::Gpr, lhs_reg.encoding()),
  1845→                ],
  1846→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())],
  1847→                encoded: Instruction::Slt {
  1848→                    rd: dst_reg,
  1849→                    rs: rhs_reg,
  1850→                    rt: lhs_reg,
  1851→                }
  1852→                .encode()
  1853→                .to_vec(),
  1854→            });
  1855→            return result;
  1856→        }
  1857→        BinOpKind::SGe => {
  1858→            // slt rd, lhs, rhs ; xori rd, rd, 1 (rd = !(lhs < rhs) = lhs >= rhs)
  1859→            result.push(AllocatedInstruction {
  1860→                opcode: "slt".to_string(),
  1861→                reads: vec![
  1862→                    PhysicalReg::new(RegClass::Gpr, lhs_reg.encoding()),
  1863→                    PhysicalReg::new(RegClass::Gpr, rhs_reg.encoding()),
  1864→                ],
  1865→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())],
  1866→                encoded: Instruction::Slt {
  1867→                    rd: dst_reg,
  1868→                    rs: lhs_reg,
  1869→                    rt: rhs_reg,
  1870→                }
  1871→                .encode()
  1872→                .to_vec(),
  1873→            });
  1874→            result.push(AllocatedInstruction {
  1875→                opcode: "xori".to_string(),
  1876→                reads: vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())],
  1877→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())],
  1878→                encoded: Instruction::Xori {
  1879→                    rt: dst_reg,
  1880→                    rs: dst_reg,
  1881→                    imm: 1,
  1882→                }
  1883→                .encode()
  1884→                .to_vec(),
  1885→            });
  1886→            return result;
  1887→        }
  1888→        BinOpKind::ULt => Instruction::Sltu {
  1889→            rd: dst_reg,
  1890→            rs: lhs_reg,
  1891→            rt: rhs_reg,
  1892→        },
  1893→        BinOpKind::ULe => {
  1894→            result.push(AllocatedInstruction {
  1895→                opcode: "sltu".to_string(),
  1896→                reads: vec![
  1897→                    PhysicalReg::new(RegClass::Gpr, rhs_reg.encoding()),
  1898→                    PhysicalReg::new(RegClass::Gpr, lhs_reg.encoding()),
  1899→                ],
  1900→                writes: vec![PhysicalReg::new(RegClass::Gpr, dst_reg.encoding())],
  1901→                encoded: Instruction::Sltu {
  1902→                    rd: dst_reg,
  1903→                    rs: rhs_reg,
  1904→                    rt: lhs_reg,
  1905→                }
  1906→                .encode()
  1907→                .to_vec(),
