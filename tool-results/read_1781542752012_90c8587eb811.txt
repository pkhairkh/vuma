     1→//! # VUMA Compilation Pipeline
     2→//!
     3→//! The full compilation pipeline that wires together every workspace crate:
     4→//!
     5→//! ```text
     6→//! Source → Parse → AST → SCG → BD Inference → MSG Construction
     7→//!        → IVE Verification → SCG Transforms → IR Lowering
     8→//!        → Register Allocation → ARM64 Codegen → ELF Emission
     9→//! ```
    10→//!
    11→//! ## Quick Start
    12→//!
    13→//! ```rust
    14→//! use vuma::pipeline::{compile, CompileConfig, CompileTarget, OptLevel, VerificationLevel};
    15→//!
    16→//! let source = r#"
    17→//!     region buf = allocate(256);
    18→//!     fn main() {
    19→//!         ptr = buf + 64;
    20→//!         header = ptr as *NodeHeader;
    21→//!     }
    22→//! "#;
    23→//!
    24→//! let config = CompileConfig::default();
    25→//! let output = compile(source, &config);
    26→//! match output {
    27→//!     Ok(out) => println!("Compiled {} bytes, {} SCG nodes", out.binary.len(), out.scg.node_count()),
    28→//!     Err(errors) => {
    29→//!         for err in &errors {
    30→//!             eprintln!("{}", err);
    31→//!         }
    32→//!     }
    33→//! }
    34→//! ```

    96→/// Optimization level.
    97→#[derive(
    98→    Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize, Default,

   100→pub enum OptLevel {
   101→    /// No optimisation — fastest compilation, best debuggability.
   102→    O0,

   104→    O1,
   105→    /// Full optimisations (DCE, CSE, constant folding, inlining).
   106→    #[default]

   108→    /// Aggressive optimisations (O2 + inlining of larger functions).

   114→        match self {
   115→            OptLevel::O0 => write!(f, "O0"),
   116→            OptLevel::O1 => write!(f, "O1"),
   117→            OptLevel::O2 => write!(f, "O2"),
   118→            OptLevel::O3 => write!(f, "O3"),
   119→        }

   150→/// Full compilation configuration.
   151→#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
   152→pub struct CompileConfig {
   153→    /// Target platform.
   154→    pub target: CompileTarget,
   155→    /// Optimisation level.
   156→    pub opt_level: OptLevel,
   157→    /// Verification thoroughness.
   158→    pub verification_level: VerificationLevel,
   159→    /// Entry-point function name (default: "main" for hosted, "_start" for bare).
   160→    pub entry_name: String,
   161→    /// Include debug info in the output.
   162→    pub debug_info: bool,
   163→    /// Stop compilation at the first error.
   164→    pub stop_on_first_error: bool,
   165→    /// Maximum inline size (number of SCG nodes) for the inlining pass.
   166→    pub max_inline_size: usize,
   167→    /// Enable memory safety checks (use-after-free, double-free, leaks, etc.).
   168→    pub memory_safety: bool,
   169→    /// Enable runtime bounds checks for array accesses (--safe flag).
   170→    pub runtime_bounds_checks: bool,
   171→    /// Force section headers in the ELF output (--sections flag).
   172→    pub section_headers: bool,
   173→}
   174→
   175→impl CompileConfig {
   176→    /// Fast-compilation debug configuration.
   177→    pub fn debug() -> Self {
   178→        Self {
   179→            opt_level: OptLevel::O0,
   180→            debug_info: true,
   181→            verification_level: VerificationLevel::Quick,
   182→            ..Self::default()
   183→        }
   184→    }
   185→
   186→    /// Release configuration with full optimisation and exhaustive verification.
   187→    pub fn release() -> Self {
   188→        Self {
   189→            opt_level: OptLevel::O3,
   190→            verification_level: VerificationLevel::Exhaustive,
   191→            ..Self::default()
   192→        }
   193→    }
   194→
   195→    /// Returns the emit config for this compile config.
   196→    fn emit_config(&self) -> EmitConfig {
   197→        match self.target {
   198→            CompileTarget::Linux => {
   199→                let mut cfg = EmitConfig::linux_elf();
   200→                cfg.section_headers = cfg.section_headers || self.section_headers;
   201→                cfg
   202→            }
   203→            CompileTarget::Wasm32 => EmitConfig::wasm_binary(),
   204→        }
   205→    }
   206→}
   207→

   212→            opt_level: OptLevel::O2,
   213→            verification_level: VerificationLevel::Normal,
   214→            entry_name: "main".to_string(),
   215→            debug_info: false,
   216→            stop_on_first_error: true,
   217→            max_inline_size: 50,
   218→            memory_safety: true,
   219→            runtime_bounds_checks: false,
   220→            section_headers: false,
   221→        }
   222→    }
   223→}

   229→/// A unified error type for the VUMA compilation pipeline.
   230→///
   231→/// Each variant captures the pipeline stage where the error occurred
   232→/// and the underlying cause.
   233→#[derive(Debug, Clone)]
   234→pub enum VumaError {
   235→    /// Error during lexing or parsing.
   236→    Parse {
   237→        /// The parse errors.
   238→        errors: Vec<ParseError>,
   239→    },
   240→    /// Error converting AST to SCG.
   241→    AstToScg {
   242→        /// Error message.
   243→        message: String,
   244→    },
   245→    /// SCG validation failed.
   246→    ScgValidation {
   247→        /// Validation error messages.
   248→        errors: Vec<String>,
   249→    },
   250→    /// SCG → MSG conversion error.
   251→    ScgToMsg {
   252→        /// The conversion error.
   253→        error: ConversionError,
   254→    },
   255→    /// BD inference error.
   256→    BdInference {
   257→        /// Node ID where inference failed, if known.
   258→        node_id: Option<u64>,
   259→        /// Error message.
   260→        message: String,
   261→    },
   262→    /// IVE verification failure (one or more invariants violated).
   263→    Verification {
   264→        /// The aggregated verification result.
   265→        result: AggregatedResult,
   266→    },
   267→    /// SCG transformation pass error.
   268→    Transform {
   269→        /// Name of the pass that failed.
   270→        pass_name: String,
   271→        /// Error messages from the pass.
   272→        errors: Vec<String>,
   273→    },
   274→    /// IR lowering / codegen error.
   275→    Codegen {
   276→        /// The codegen error.
   277→        error: CodegenError,
   278→    },
   279→    /// Register allocation failure.
   280→    RegisterAlloc {
   281→        /// Error message.
   282→        message: String,
   283→    },
   284→    /// ELF emission failure.
   285→    Emission {
   286→        /// Error message.
   287→        message: String,
   288→    },
   289→    /// COR initialization failure.
   290→    CorInit {
   291→        /// Error message.
   292→        message: String,
   293→    },
   294→    /// Module resolution error (import not found, circular import, etc.).
   295→    ModuleResolution {
   296→        /// The resolution errors.
   297→        errors: Vec<ResolveError>,
   298→    },
   299→    /// A collection of errors accumulated across stages.
   300→    Multi {
   301→        /// The collected errors.
   302→        errors: Vec<VumaError>,
   303→    },
   304→    /// Backend failed; fallback to next available backend was attempted.
   305→    BackendFallback {
   306→        /// Name of the backend that failed.
   307→        failed_backend: String,
   308→        /// Name of the fallback backend that was tried (if any).
   309→        fallback_backend: Option<String>,
   310→        /// Error message from the failed backend.
   311→        error: String,
   312→    },
   313→    /// Internal panic caught during compilation (crash recovery).
   314→    PanicCaught {
   315→        /// The pipeline stage where the panic occurred.
   316→        stage: String,
   317→        /// The panic message.
   318→        message: String,
   319→    },
   320→}
   321→
   322→impl VumaError {
   323→    /// Returns the pipeline stage that produced this error.
   324→    pub fn stage(&self) -> &'static str {
   325→        match self {
   326→            VumaError::Parse { .. } => "parse",
   327→            VumaError::AstToScg { .. } => "ast-to-scg",
   328→            VumaError::ScgValidation { .. } => "scg-validation",
   329→            VumaError::ScgToMsg { .. } => "scg-to-msg",
   330→            VumaError::BdInference { .. } => "bd-inference",
   331→            VumaError::Verification { .. } => "ive-verification",
   332→            VumaError::Transform { .. } => "scg-transform",
   333→            VumaError::Codegen { .. } => "codegen",
   334→            VumaError::RegisterAlloc { .. } => "register-alloc",
   335→            VumaError::Emission { .. } => "elf-emission",
   336→            VumaError::CorInit { .. } => "cor-init",
   337→            VumaError::ModuleResolution { .. } => "module-resolution",
   338→            VumaError::Multi { .. } => "multi",
   339→            VumaError::BackendFallback { .. } => "backend-fallback",
   340→            VumaError::PanicCaught { .. } => "panic-caught",
   341→        }
   342→    }
   343→}
   344→
   345→impl fmt::Display for VumaError {
   346→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   347→        match self {
   348→            VumaError::Parse { errors } => {
   349→                write!(f, "[parse] {} error(s):", errors.len())?;
   350→                for e in errors {
   351→                    write!(f, "\n  - {}", e)?;
   352→                }
   353→                Ok(())
   354→            }
   355→            VumaError::AstToScg { message } => write!(f, "[ast-to-scg] {}", message),
   356→            VumaError::ScgValidation { errors } => {
   357→                write!(f, "[scg-validation] {} error(s):", errors.len())?;
   358→                for e in errors {
   359→                    write!(f, "\n  - {}", e)?;
   360→                }
   361→                Ok(())
   362→            }
   363→            VumaError::ScgToMsg { error } => write!(f, "[scg-to-msg] {}", error),
   364→            VumaError::BdInference { node_id, message } => {
   365→                write!(f, "[bd-inference] {}", message)?;
   366→                if let Some(id) = node_id {
   367→                    write!(f, " (node {})", id)?;
   368→                }
   369→                Ok(())
   370→            }
   371→            VumaError::Verification { result } => {
   372→                write!(f, "[ive-verification] verdict: {}", result.overall)
   373→            }
   374→            VumaError::Transform { pass_name, errors } => {
   375→                write!(f, "[scg-transform:{}] {} error(s)", pass_name, errors.len())
   376→            }
   377→            VumaError::Codegen { error } => write!(f, "[codegen] {}", error),
   378→            VumaError::RegisterAlloc { message } => write!(f, "[register-alloc] {}", message),
   379→            VumaError::Emission { message } => write!(f, "[elf-emission] {}", message),
   380→            VumaError::CorInit { message } => write!(f, "[cor-init] {}", message),
   381→            VumaError::ModuleResolution { errors } => {
   382→                write!(f, "[module-resolution] {} error(s):", errors.len())?;
   383→                for e in errors {
   384→                    write!(f, "\n  - {}", e)?;
   385→                }
   386→                Ok(())
   387→            }
   388→            VumaError::Multi { errors } => {
   389→                write!(f, "multiple errors ({}):", errors.len())?;
   390→                for (i, e) in errors.iter().enumerate() {
   391→                    write!(f, "\n{}. {}", i + 1, e)?;
   392→                }
   393→                Ok(())
   394→            }
   395→            VumaError::BackendFallback { failed_backend, fallback_backend, error } => {
   396→                write!(f, "[backend-fallback] {} failed: {}", failed_backend, error)?;
   397→                if let Some(fb) = fallback_backend {
   398→                    write!(f, ", attempting fallback to {}", fb)?;
   399→                }
   400→                Ok(())
   401→            }
   402→            VumaError::PanicCaught { stage, message } => {
   403→                write!(f, "[panic-caught] panic in stage '{}': {}", stage, message)
   404→            }
   405→        }
   406→    }
   407→}

   409→impl std::error::Error for VumaError {}

   468→/// Result of a compilation attempt with crash recovery.
   469→///
   470→/// On success, contains the full [`CompilationOutput`].
   471→/// On failure, contains a [`PartialCompilationOutput`] with whatever
   472→/// intermediate results were produced, plus all diagnostics.
   473→#[derive(Debug)]
   474→pub enum CompileResult {
   475→    /// Compilation succeeded.
   476→    Success(CompilationOutput),
   477→    /// Compilation failed, but partial results are available.
   478→    Partial(PartialCompilationOutput),
   479→}
   480→
   481→impl CompileResult {
   482→    /// Returns true if compilation succeeded.
   483→    pub fn is_success(&self) -> bool {
   484→        matches!(self, CompileResult::Success(_))
   485→    }
   486→
   487→    /// Returns the diagnostics (empty on success).
   488→    pub fn diagnostics(&self) -> &[VumaError] {
   489→        match self {
   490→            CompileResult::Success(_) => &[],
   491→            CompileResult::Partial(p) => &p.diagnostics,
   492→        }
   493→    }
   494→}

   513→/// A fingerprint of a source file, used to detect changes for
   514→/// incremental compilation.
   515→#[derive(Debug, Clone, PartialEq, Eq)]
   516→pub struct SourceFingerprint {
   517→    /// A hash of the source text.
   518→    pub hash: u64,
   519→    /// Byte length of the source.
   520→    pub len: usize,
   521→}
   522→
   523→impl SourceFingerprint {
   524→    /// Compute a fingerprint from source text.
   525→    pub fn from_source(source: &str) -> Self {
   526→        // Simple FNV-1a hash — sufficient for change detection.
   527→        let mut hash: u64 = 0xcbf29ce484222325;
   528→        for byte in source.bytes() {
   529→            hash ^= byte as u64;
   530→            hash = hash.wrapping_mul(0x100000001b3);
   531→        }
   532→        Self {
   533→            hash,
   534→            len: source.len(),
   535→        }
   536→    }
   537→}

   559→/// Identifies a pipeline stage for incremental invalidation.
   560→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
   561→pub enum PipelineStage {
   562→    /// Lexing + parsing.
   563→    Parse,
   564→    /// AST → SCG conversion.
   565→    AstToScg,
   566→    /// SCG validation.
   567→    ScgValidation,
   568→    /// BD inference.
   569→    BdInference,
   570→    /// SCG → MSG construction.
   571→    MsgConstruction,
   572→    /// IVE verification.
   573→    IveVerification,
   574→    /// SCG transformation passes.
   575→    ScgTransforms,
   576→    /// IR lowering (SCG → IR).
   577→    IrLowering,
   578→    /// Register allocation.
   579→    RegisterAlloc,
   580→    /// ARM64 code emission.
   581→    CodeEmission,
   582→    /// COR (Continuous Optimization Runtime) initialization.
   583→    CorInit,
   584→}

   586→impl PipelineStage {
   587→    /// All stages in order.
   588→    pub fn all() -> &'static [PipelineStage; 11] {
   589→        &[

   591→            PipelineStage::AstToScg,
   592→            PipelineStage::ScgValidation,
   593→            PipelineStage::BdInference,
   594→            PipelineStage::MsgConstruction,
   595→            PipelineStage::IveVerification,
   596→            PipelineStage::ScgTransforms,
   597→            PipelineStage::IrLowering,
   598→            PipelineStage::RegisterAlloc,
   599→            PipelineStage::CodeEmission,

   603→
   604→    /// Returns all stages from (and including) the given stage onwards.
   605→    pub fn from(stage: PipelineStage) -> Vec<PipelineStage> {
   606→        PipelineStage::all()
   607→            .iter()
   608→            .filter(|&&s| s >= stage)
   609→            .copied()
   610→            .collect()
   611→    }
   612→}

   637→
   638→/// Pre-computed edge index for efficient graph traversal during bridge
   639→/// conversion. Built once from all edges in the SCG and then queried
   640→/// by node ID and edge kind.
   641→struct EdgeIndex {
   642→    /// Outgoing edges keyed by source node.
   643→    outgoing: HashMap<NodeId, Vec<EdgeData>>,
   644→    /// Incoming edges keyed by target node.
   645→    incoming: HashMap<NodeId, Vec<EdgeData>>,
   646→}
   647→
   648→impl EdgeIndex {
   649→    /// Build the edge index from all edges in the SCG.
   650→    fn build(scg: &SCG) -> Self {
   651→        let mut outgoing: HashMap<NodeId, Vec<EdgeData>> = HashMap::new();
   652→        let mut incoming: HashMap<NodeId, Vec<EdgeData>> = HashMap::new();
   653→        for edge in scg.edges() {
   654→            outgoing.entry(edge.source).or_default().push(edge.clone());
   655→            incoming.entry(edge.target).or_default().push(edge.clone());
   656→        }
   657→        Self { outgoing, incoming }
   658→    }
   659→
   660→    /// Get outgoing ControlFlow edges from a node.
   661→    fn outgoing_cf(&self, id: NodeId) -> Vec<&EdgeData> {
   662→        self.outgoing
   663→            .get(&id)
   664→            .map(|edges| {
   665→                edges
   666→                    .iter()
   667→                    .filter(|e| e.kind == EdgeKind::ControlFlow)
   668→                    .collect()
   669→            })
   670→            .unwrap_or_default()
   671→    }
   672→
   673→    /// Get incoming DataFlow edges to a node.
   674→    fn incoming_df(&self, id: NodeId) -> Vec<&EdgeData> {
   675→        self.incoming
   676→            .get(&id)
   677→            .map(|edges| {
   678→                edges
   679→                    .iter()
   680→                    .filter(|e| e.kind == EdgeKind::DataFlow)
   681→                    .collect()
   682→            })
   683→            .unwrap_or_default()
   684→    }
   685→
   686→    /// Get outgoing DataFlow edges from a node.
   687→    fn outgoing_df(&self, id: NodeId) -> Vec<&EdgeData> {
   688→        self.outgoing
   689→            .get(&id)
   690→            .map(|edges| {
   691→                edges
   692→                    .iter()
   693→                    .filter(|e| e.kind == EdgeKind::DataFlow)
   694→                    .collect()
   695→            })
   696→            .unwrap_or_default()
   697→    }
   698→}

   702→/// Generate a variable name for a node with a given prefix.
   703→fn node_var(id: NodeId, _prefix: &str) -> String {
   704→    // Must match the naming convention in resolve_df_input so that
   705→    // source references (v_{source_id}) resolve correctly in the
   706→    // codegen IR builder's names map.
   707→    format!("v_{}", id.as_u64())
   708→}

   710→/// Resolve a DataFlow input for a node, returning a `ScgExpr` referencing
   711→/// the variable produced by the source node of the DataFlow edge at the
   712→/// given position.
   713→///
   714→/// If the source node is a Control node (FunctionEntry, etc.) that does not
   715→/// produce a named variable, falls back to `ScgExpr::Int(0)` to avoid
   716→/// referencing a non-existent variable in the codegen IR.
   717→fn resolve_df_input(
   718→    node_id: NodeId,
   719→    position: usize,
   720→    edge_idx: &EdgeIndex,
   721→    scg: &SCG,
   722→) -> ScgExpr {
   723→    let df_inputs = edge_idx.incoming_df(node_id);
   724→    if position < df_inputs.len() {
   725→        let source = df_inputs[position].source;
   726→        // Check if the source node is a productive node that defines a variable.
   727→        // Control nodes (FunctionEntry, Branch, etc.) and Phantom nodes do not
   728→        // produce named variables — skip them to avoid UnknownVariable errors.
   729→        if let Some(src_data) = scg.get_node(source) {
   730→            match &src_data.payload {
   731→                // These node types do NOT produce named variables in the
   732→                // codegen SCG.  Control/Phantom are structural; Deallocation
   733→                // and Effect are lowered to runtime calls (no dst vreg).
   734→                NodePayload::Control(_)
   735→                | NodePayload::Phantom(_)
   736→                | NodePayload::Deallocation(_)
   737→                | NodePayload::Effect(_)
   738→                | NodePayload::VTable(_)
   739→                | NodePayload::ClosureEnv(_) => {
   740→                    // Non-productive source — use 0 as placeholder
   741→                    ScgExpr::Int(0)
   742→                }
   743→                _ => ScgExpr::Var(format!("v_{}", source.as_u64())),
   744→            }
   745→        } else {
   746→            ScgExpr::Int(0)
   747→        }
   748→    } else {
   749→        // No DataFlow edge at this position — use 0 as placeholder
   750→        ScgExpr::Int(0)
   751→    }
   752→}
   753→
   754→/// Resolve the condition expression for a Branch node by looking at its
   755→/// incoming DataFlow edges.
   756→fn resolve_branch_cond(branch_id: NodeId, edge_idx: &EdgeIndex, scg: &SCG) -> ScgExpr {
   757→    resolve_df_input(branch_id, 0, edge_idx, scg)
   758→}

   762→/// Find the `FunctionReturn` node reachable from a `FunctionEntry` via
   763→/// ControlFlow edges, using BFS.
   764→fn find_function_return(entry_id: NodeId, scg: &SCG, edge_idx: &EdgeIndex) -> Option<NodeId> {
   765→    let mut visited = HashSet::new();
   766→    let mut queue = VecDeque::new();
   767→    queue.push_back(entry_id);
   768→    visited.insert(entry_id);
   769→
   770→    while let Some(current) = queue.pop_front() {
   771→        for edge in edge_idx.outgoing_cf(current) {
   772→            let target = edge.target;
   773→            if visited.contains(&target) {
   774→                continue;
   775→            }
   776→            visited.insert(target);
   777→            if let Some(node) = scg.get_node(target) {
   778→                if let NodePayload::Control(c) = &node.payload {
   779→                    if c.kind == ControlKind::FunctionReturn {
   780→                        return Some(target);
   781→                    }
   782→                }
   783→            }
   784→            queue.push_back(target);
   785→        }
   786→    }
   787→
   788→    None
   789→}
   790→
   791→/// Find all `Join` nodes reachable from `start` via ControlFlow edges,
   792→/// stopping at the first Join encountered on each path (Joins are
   793→/// convergence points, not passed through during search).
   794→fn find_reachable_joins(start: NodeId, scg: &SCG, edge_idx: &EdgeIndex) -> Vec<NodeId> {
   795→    let mut joins = Vec::new();
   796→    let mut visited = HashSet::new();
   797→    let mut queue = VecDeque::new();
   798→    queue.push_back(start);
   799→    visited.insert(start);
   800→
   801→    let max_steps = 500;
   802→    let mut steps = 0;
   803→
   804→    while let Some(current) = queue.pop_front() {
   805→        steps += 1;
   806→        if steps > max_steps {
   807→            break;
   808→        }
   809→
   810→        if let Some(node) = scg.get_node(current) {
   811→            if let NodePayload::Control(c) = &node.payload {
   812→                if c.kind == ControlKind::Join {
   813→                    joins.push(current);
   814→                    continue; // Don't walk past Join
   815→                }
   816→            }
   817→        }
   818→
   819→        for edge in edge_idx.outgoing_cf(current) {
   820→            let target = edge.target;
   821→            if !visited.contains(&target) {
   822→                visited.insert(target);
   823→                queue.push_back(target);
   824→            }
   825→        }
   826→    }
   827→
   828→    joins
   829→}

   831→/// Find the `Join` node where a Branch's then and else arms converge.
   832→fn find_join_for_branch(
   833→    then_start: NodeId,
   834→    else_start: Option<NodeId>,
   835→    scg: &SCG,
   836→    edge_idx: &EdgeIndex,
   837→) -> Option<NodeId> {
   838→    let then_joins = find_reachable_joins(then_start, scg, edge_idx);
   839→
   840→    if let Some(else_start) = else_start {
   841→        let else_joins = find_reachable_joins(else_start, scg, edge_idx);
   842→        // Find the first Join reachable from both arms
   843→        for jid in &then_joins {
   844→            if else_joins.contains(jid) {
   845→                return Some(*jid);
   846→            }
   847→        }
   848→    }
   849→
   850→    // Fallback: first Join reachable from then_start
   851→    then_joins.into_iter().next()
   852→}

   854→/// Resolve a Branch node's then/else targets and Join convergence point.
   855→///
   856→/// Looks for labeled ControlFlow edges ("then", "else", "else_fallthrough")
   857→/// and falls back to positional ordering if labels are missing.
   858→fn resolve_branch(
   859→    branch_id: NodeId,
   860→    scg: &SCG,
   861→    edge_idx: &EdgeIndex,
   862→) -> (NodeId, Option<NodeId>, Option<NodeId>) {
   863→    let cf_edges = edge_idx.outgoing_cf(branch_id);
   864→
   865→    // Look for labeled edges
   866→    let then_target = cf_edges
   867→        .iter()
   868→        .find(|e| e.label.as_deref() == Some("then"))
   869→        .map(|e| e.target)
   870→        .or_else(|| cf_edges.first().map(|e| e.target));
   871→
   872→    let else_target = cf_edges
   873→        .iter()
   874→        .find(|e| {
   875→            e.label.as_deref() == Some("else") || e.label.as_deref() == Some("else_fallthrough")
   876→        })
   877→        .map(|e| e.target)
   878→        .or_else(|| {
   879→            // If there are exactly 2 CF edges and one is "then", the other is "else"
   880→            if cf_edges.len() == 2 {
   881→                let then = then_target?;
   882→                cf_edges.iter().find(|e| e.target != then).map(|e| e.target)
   883→            } else {
   884→                None
   885→            }
   886→        });
   887→
   888→    let then_tgt = then_target.unwrap_or(branch_id);
   889→    let join = find_join_for_branch(then_tgt, else_target, scg, edge_idx);
   890→
   891→    (then_tgt, else_target, join)
   892→}

   894→/// Resolve a LoopHeader node's body and exit targets.
   895→///
   896→/// Classifies outgoing ControlFlow edges: edges targeting a `LoopExit`
   897→/// node are the exit; all other edges are the loop body.
   898→fn resolve_loop(header_id: NodeId, scg: &SCG, edge_idx: &EdgeIndex) -> (NodeId, Option<NodeId>) {
   899→    let cf_edges = edge_idx.outgoing_cf(header_id);
   900→
   901→    let mut body_target = None;
   902→    let mut exit_target = None;
   903→
   904→    for edge in &cf_edges {
   905→        if let Some(target_node) = scg.get_node(edge.target) {
   906→            if let NodePayload::Control(c) = &target_node.payload {
   907→                if c.kind == ControlKind::LoopExit {
   908→                    exit_target = Some(edge.target);
   909→                    continue;
   910→                }
   911→            }
   912→        }
   913→        if body_target.is_none() {
   914→            body_target = Some(edge.target);
   915→        }
   916→    }
   917→
   918→    // Fallbacks
   919→    if body_target.is_none() {
   920→        body_target = cf_edges.first().map(|e| e.target);
   921→    }
   922→    if exit_target.is_none() && cf_edges.len() > 1 {
   923→        exit_target = cf_edges.get(1).map(|e| e.target);
   924→    }
   925→
   926→    (body_target.unwrap_or(header_id), exit_target)
   927→}

   929→// ── Match/switch case-value extraction ──────────────────────────────────
   930→
   931→/// Extract the integer case value for a match/switch arm from the branch
   932→/// condition and surrounding SCG context.
   933→///
   934→/// A match arm `42 => body` produces a branch whose condition is typically
   935→/// the result of an equality comparison `disc == 42`. This function traces
   936→/// back through the DataFlow edges to find the constant operand, which is
   937→/// the case value from the AST's MatchArm pattern.
   938→///
   939→/// Extraction strategy (in priority order):
   940→///
   941→/// 1. If `cond` is already an `ScgExpr::Int(n)`, return `n`.
   942→/// 2. Trace the first DataFlow edge of the Branch back to its source node.
   943→///    If the source is a Computation node with an equality operation
   944→///    (`eq` / `==`), inspect its second DataFlow input — the RHS of the
   945→///    comparison is the case value. If that RHS source node is a
   946→///    Computation whose `operation` string parses as an integer, use it.
   947→/// 3. Try to parse an integer from the control node's label string.
   948→///    Recognised formats: `"match disc == 42"`, `"case 2: 42"`.
   949→/// 4. Fall back to `arm_index` — each arm in a match expression receives
   950→///    a distinct fallback value so that unknown case values don't collide.
   951→fn extract_case_value(
   952→    branch_id: NodeId,
   953→    cond: &ScgExpr,
   954→    ctrl_label: Option<&str>,
   955→    scg: &SCG,
   956→    edge_idx: &EdgeIndex,
   957→    arm_index: usize,
   958→) -> i64 {
   959→    // Strategy 1: direct integer condition.
   960→    if let ScgExpr::Int(n) = cond {
   961→        return *n;
   962→    }
   963→
   964→    // Strategy 2: trace back through the equality comparison node.
   965→    let df_inputs = edge_idx.incoming_df(branch_id);
   966→    if let Some(df_edge) = df_inputs.first() {
   967→        let cond_source = df_edge.source;
   968→        if let Some(source_node) = scg.get_node(cond_source) {
   969→            if let NodePayload::Computation(comp) = &source_node.payload {
   970→                let op_label = comp.kind.label();
   971→                let is_eq = op_label == "eq" || op_label == "==";
   972→                if is_eq {
   973→                    // The RHS of the equality is the case value.
   974→                    let rhs_inputs = edge_idx.incoming_df(cond_source);
   975→                    if rhs_inputs.len() >= 2 {
   976→                        let rhs_source = rhs_inputs[1].source;
   977→                        if let Some(rhs_node) = scg.get_node(rhs_source) {
   978→                            // The RHS node might be a Computation whose
   979→                            // operation string is a literal integer.
   980→                            if let NodePayload::Computation(rhs_comp) = &rhs_node.payload {
   981→                                if let Ok(val) = rhs_comp.kind.label().parse::<i64>() {
   982→                                    return val;
   983→                                }
   984→                            }
   985→                        }
   986→                    }
   987→                }
   988→            }
   989→        }
   990→    }
   991→
   992→    // Strategy 3: parse from the control node label.
   993→    if let Some(label) = ctrl_label {
   994→        // Format: "match <disc> == <value>"
   995→        if let Some(idx) = label.find("==") {
   996→            let after_eq = label[idx + 2..].trim();
   997→            // Take the first token (stop at whitespace / punctuation)
   998→            let token = after_eq
   999→                .split(|c: char| c.is_whitespace() || c == ')')
  1000→                .next()
  1001→                .unwrap_or(after_eq);
  1002→            if let Ok(val) = token.parse::<i64>() {
  1003→                return val;
  1004→            }
  1005→        }
  1006→        // Format: "case <idx>: <value>"
  1007→        if label.starts_with("case ") {
  1008→            let parts: Vec<&str> = label.splitn(2, ':').collect();
  1009→            if parts.len() == 2 {
  1010→                let value_str = parts[1].trim();
  1011→                let token = value_str
  1012→                    .split(|c: char| c.is_whitespace() || c == ')')
  1013→                    .next()
  1014→                    .unwrap_or(value_str);
  1015→                if let Ok(val) = token.parse::<i64>() {
  1016→                    return val;
  1017→                }
  1018→            }
  1019→        }
  1020→    }
  1021→
  1022→    // Strategy 4: fallback to arm_index so each arm gets a distinct value.
  1023→    arm_index as i64
  1024→}

  1026→// ── Control flow walk ──────────────────────────────────────────────────
  1027→
  1028→/// Walk control flow starting from `start`, producing `ScgStatement`s.
  1029→///
  1030→/// Stops when reaching a node in `stop_at` (does NOT consume that node;
  1031→/// the caller is responsible for handling it). Adds processed nodes to
  1032→/// `consumed`.
  1033→///
  1034→/// # Control Flow Reconstruction
  1035→///
  1036→/// - **Branch → If**: Follows "then"/"else" labeled CF edges, walks each
  1037→///   arm until reaching a Join, produces `ControlNode::If`.
  1038→/// - **LoopHeader → Loop**: Follows body/exit CF edges, walks the body
  1039→///   until back-edge or LoopExit, produces `ControlNode::Loop`.
  1040→/// - **Jump("break")**: Produces `ControlNode::Break`.
  1041→/// - **Jump("continue")**: Produces `ControlNode::Continue`.
  1042→/// - **FunctionReturn**: Produces `ScgStatement::Return`.
  1043→fn walk_control_flow(
  1044→    start: NodeId,
  1045→    scg: &SCG,
  1046→    edge_idx: &EdgeIndex,
  1047→    consumed: &mut HashSet<NodeId>,
  1048→    stop_at: &HashSet<NodeId>,
  1049→) -> Vec<ScgStatement> {
  1050→    let mut stmts = Vec::new();
  1051→    let mut current = Some(start);
  1052→
  1053→    while let Some(node_id) = current {
  1054→        // Stop if we've reached a merge point
  1055→        if stop_at.contains(&node_id) {
  1056→            break;
  1057→        }
  1058→        // Skip already-consumed nodes
  1059→        if consumed.contains(&node_id) {
  1060→            break;
  1061→        }
  1062→        consumed.insert(node_id);
  1063→
  1064→        let node_data = match scg.get_node(node_id) {
  1065→            Some(n) => n,
  1066→            None => break,
  1067→        };
  1068→
  1069→        match &node_data.payload {
  1070→            // ── Control nodes ──────────────────────────────────────
  1071→            NodePayload::Control(ctrl) => match ctrl.kind {
  1072→                ControlKind::Branch => {
  1073→                    let (then_tgt, else_tgt, join_node) = resolve_branch(node_id, scg, edge_idx);
  1074→                    let cond = resolve_branch_cond(node_id, edge_idx, scg);
  1075→
  1076→                    // Check if this is a match/switch branch (label starts
  1077→                    // with "match") vs a simple if/else. For match branches,
  1078→                    // we look for multiple Branch→Join diamonds that share
  1079→                    // the same Join and collapse them into a Switch node.
  1080→                    let is_match = ctrl
  1081→                        .label
  1082→                        .as_ref()
  1083→                        .map(|l| l.starts_with("match"))
  1084→                        .unwrap_or(false);
  1085→
  1086→                    if is_match {
  1087→                        // For match/switch, collect all arms that lead to
  1088→                        // the same Join node. Each arm is a then/else pair
  1089→                        // where the then branch is the matched case.
  1090→                        // We walk the then arm to find the case value and body.
  1091→                        let mut arms = Vec::new();
  1092→                        let mut default_body = Vec::new();
  1093→
  1094→                        // Build stop-at for both arms (includes Join)
  1095→                        let mut arm_stop = stop_at.clone();
  1096→                        if let Some(join) = join_node {
  1097→                            arm_stop.insert(join);
  1098→                        }
  1099→
  1100→                        // Generate a simple switch from the then arm
  1101→                        // with a discriminant expression.
  1102→                        let then_body_stmts =
  1103→                            walk_control_flow(then_tgt, scg, edge_idx, consumed, &arm_stop);
  1104→
  1105→                        // Extract the case value from the AST's MatchArm pattern.
  1106→                        // The branch condition for a match arm is typically
  1107→                        // `disc == value`. We trace back through the DataFlow
  1108→                        // edges to find the constant being compared against.
  1109→                        let case_value = extract_case_value(
  1110→                            node_id,
  1111→                            &cond,
  1112→                            ctrl.label.as_deref(),
  1113→                            scg,
  1114→                            edge_idx,
  1115→                            arms.len(),
  1116→                        );
  1117→                        arms.push(SwitchArm {
  1118→                            value: case_value,
  1119→                            body: then_body_stmts,
  1120→                        });
  1121→
  1122→                        if let Some(tgt) = else_tgt {
  1123→                            let else_stmts =
  1124→                                walk_control_flow(tgt, scg, edge_idx, consumed, &arm_stop);
  1125→                            default_body = else_stmts;
  1126→                        }
  1127→
  1128→                        // Use the first operand of the condition as discriminant
  1129→                        let disc = if let ScgExpr::Var(_) = &cond {
  1130→                            cond.clone()
  1131→                        } else {
  1132→                            ScgExpr::Var("disc".to_string())
  1133→                        };
  1134→
  1135→                        stmts.push(ScgStatement::Control(ControlNode::Switch {
  1136→                            discriminant: disc,
  1137→                            arms,
  1138→                            default_body,
  1139→                        }));
  1140→                    } else {
  1141→                        // Standard if/else
  1142→                        let mut arm_stop = stop_at.clone();
  1143→                        if let Some(join) = join_node {
  1144→                            arm_stop.insert(join);
  1145→                        }
  1146→
  1147→                        let then_body =
  1148→                            walk_control_flow(then_tgt, scg, edge_idx, consumed, &arm_stop);
  1149→
  1150→                        let else_body = else_tgt
  1151→                            .map(|tgt| walk_control_flow(tgt, scg, edge_idx, consumed, &arm_stop));
  1152→
  1153→                        stmts.push(ScgStatement::Control(ControlNode::If {
  1154→                            cond,
  1155→                            then_body,
  1156→                            else_body,
  1157→                        }));
  1158→                    }
  1159→
  1160→                    // Continue from the Join
  1161→                    if let Some(join) = join_node {
  1162→                        consumed.insert(join);
  1163→                        current = edge_idx.outgoing_cf(join).first().map(|e| e.target);
  1164→                    } else {
  1165→                        current = None;
  1166→                    }
  1167→                    continue;
  1168→                }
  1169→
  1170→                ControlKind::LoopHeader => {
  1171→                    let (body_tgt, exit_tgt) = resolve_loop(node_id, scg, edge_idx);
  1172→
  1173→                    // Stop the body walk at back-edges (LoopHeader) and LoopExit
  1174→                    let mut loop_stop = stop_at.clone();
  1175→                    loop_stop.insert(node_id); // back-edge target
  1176→                    if let Some(exit) = exit_tgt {
  1177→                        loop_stop.insert(exit);
  1178→                    }
  1179→
  1180→                    let body = walk_control_flow(body_tgt, scg, edge_idx, consumed, &loop_stop);
  1181→
  1182→                    stmts.push(ScgStatement::Control(ControlNode::Loop { body }));
  1183→
  1184→                    // Continue from the LoopExit
  1185→                    if let Some(exit) = exit_tgt {
  1186→                        consumed.insert(exit);
  1187→                        current = edge_idx.outgoing_cf(exit).first().map(|e| e.target);
  1188→                    } else {
  1189→                        current = None;
  1190→                    }
  1191→                    continue;
  1192→                }
  1193→
  1194→                ControlKind::Jump => match ctrl.label.as_deref() {
  1195→                    Some("break") => {
  1196→                        stmts.push(ScgStatement::Control(ControlNode::Break));
  1197→                        current = None;
  1198→                        continue;
  1199→                    }
  1200→                    Some("continue") => {
  1201→                        stmts.push(ScgStatement::Control(ControlNode::Continue));
  1202→                        current = None;
  1203→                        continue;
  1204→                    }
  1205→                    _ => {
  1206→                        // Unconditional jump — follow the CF edge
  1207→                        let target = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1208→                        if let Some(tgt) = target {
  1209→                            if !consumed.contains(&tgt) && !stop_at.contains(&tgt) {
  1210→                                current = Some(tgt);
  1211→                                continue;
  1212→                            }
  1213→                        }
  1214→                        current = None;
  1215→                        continue;
  1216→                    }
  1217→                },
  1218→
  1219→                ControlKind::FunctionReturn => {
  1220→                    stmts.push(ScgStatement::Return(vec![]));
  1221→                    current = None;
  1222→                    continue;
  1223→                }
  1224→
  1225→                ControlKind::Join | ControlKind::LoopExit => {
  1226→                    // Structural nodes handled by Branch/LoopHeader.
  1227→                    // Pass through to the next node.
  1228→                    current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1229→                    continue;
  1230→                }
  1231→
  1232→                ControlKind::FunctionEntry => {
  1233→                    // Shouldn't appear inside a function body; pass through
  1234→                    current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1235→                    continue;
  1236→                }
  1237→
  1238→                ControlKind::Switch | ControlKind::SwitchCase => {
  1239→                    // Switch/switch-case nodes are handled like Branch
  1240→                    current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1241→                    continue;
  1242→                }
  1243→
  1244→                ControlKind::ClosureEntry | ControlKind::ClosureReturn => {
  1245→                    // Closure entry/return handled like function entry/return
  1246→                    current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1247→                    continue;
  1248→                }
  1249→
  1250→                ControlKind::FuturePoll
  1251→                | ControlKind::WakerRegistration
  1252→                | ControlKind::StateTransition => {
  1253→                    // Async state machine nodes: pass through
  1254→                    current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1255→                    continue;
  1256→                }
  1257→            },
  1258→
  1259→            // ── Non-control nodes: convert to statements ───────────
  1260→            _ => {
  1261→                if let Some(stmt) = convert_node_to_statement(node_id, node_data, edge_idx, scg) {
  1262→                    stmts.push(stmt);
  1263→                }
  1264→
  1265→                // Continue to the next node via ControlFlow
  1266→                current = edge_idx.outgoing_cf(node_id).first().map(|e| e.target);
  1267→            }
  1268→        }
  1269→    }
  1270→
  1271→    stmts
  1272→}
  1273→
  1274→// ── Node-to-statement conversion ───────────────────────────────────────
  1275→
  1276→/// Convert a non-control SCG node into an `ScgStatement`.
  1277→///
  1278→/// Handles all node types except `Control` (which is handled by
  1279→/// `walk_control_flow`) and `Phantom` (which is skipped).
  1280→fn convert_node_to_statement(
  1281→    node_id: NodeId,
  1282→    node_data: &NodeData,
  1283→    edge_idx: &EdgeIndex,
  1284→    scg: &SCG,
  1285→) -> Option<ScgStatement> {
  1286→    match &node_data.payload {
  1287→        NodePayload::Allocation(alloc) => {
  1288→            let ty = alloc
  1289→                .type_name
  1290→                .as_deref()
  1291→                .and_then(parse_scg_type)
  1292→                .unwrap_or(ScgType::U8);
  1293→            Some(ScgStatement::Allocation(AllocationNode::Stack {
  1294→                name: node_var(node_id, "alloc"),
  1295→                size: alloc.size as u32,
  1296→                ty,
  1297→            }))
  1298→        }
  1299→
  1300→        NodePayload::Access(access) => match access.mode {
  1301→            AccessMode::Read => Some(ScgStatement::Access(AccessNode::Load {
  1302→                dst: node_var(node_id, "val"),
  1303→                ptr: resolve_df_input(node_id, 0, edge_idx, scg),
  1304→                offset: access.offset.map(|o| ScgExpr::Int(o as i64)),
  1305→            })),
  1306→            AccessMode::Write | AccessMode::ReadWrite => {
  1307→                Some(ScgStatement::Access(AccessNode::Store {
  1308→                    ptr: resolve_df_input(node_id, 0, edge_idx, scg),
  1309→                    offset: access.offset.map(|o| ScgExpr::Int(o as i64)),
  1310→                    value: resolve_df_input(node_id, 1, edge_idx, scg),
  1311→                }))
  1312→            }
  1313→        },
  1314→
  1315→        NodePayload::Computation(comp) => {
  1316→            let op_label = comp.kind.label();
  1317→            let op = parse_binop(&op_label).unwrap_or(IrBinOpKind::Add);
  1318→            Some(ScgStatement::Computation(ComputationNode {
  1319→                dst: node_var(node_id, "comp"),
  1320→                op,
  1321→                lhs: resolve_df_input(node_id, 0, edge_idx, scg),
  1322→                rhs: resolve_df_input(node_id, 1, edge_idx, scg),
  1323→                tail_call: false,
  1324→            }))
  1325→        }
  1326→
  1327→        NodePayload::Cast(cast) => {
  1328→            let to_ty = parse_scg_type(&cast.to_type).unwrap_or(ScgType::Ptr);
  1329→            let from_ty = parse_scg_type(&cast.from_type).unwrap_or(ScgType::Ptr);
  1330→            Some(ScgStatement::Cast(CastNode {
  1331→                dst: node_var(node_id, "cast"),
  1332→                src: resolve_df_input(node_id, 0, edge_idx, scg),
  1333→                kind: if cast.is_lossless {
  1334→                    CodegenCastKind::ZExt
  1335→                } else {
  1336→                    CodegenCastKind::BitCast
  1337→                },
  1338→                from_ty,
  1339→                to_ty,
  1340→            }))
  1341→        }
  1342→
  1343→        NodePayload::Deallocation(_dealloc) => {
  1344→            // Lower deallocation as a proper runtime call rather than
  1345→            // a semantic no-op (`*ptr = 0`).  This ensures the memory
  1346→            // is actually freed at runtime and is more correct than
  1347→            // simply zeroing the pointer.
  1348→            Some(ScgStatement::Call(CallNode {
  1349→                dst: None,
  1350→                func: "__vuma_dealloc".to_string(),
  1351→                args: vec![resolve_df_input(node_id, 0, edge_idx, scg)],
  1352→                is_extern: true,
  1353→            }))
  1354→        }
  1355→
  1356→        NodePayload::Effect(eff) => Some(ScgStatement::Call(CallNode {
  1357→            dst: Some(node_var(node_id, "eff")),
  1358→            func: eff.effect_kind.clone(),
  1359→            args: vec![],
  1360→            is_extern: false,
  1361→        })),
  1362→
  1363→        NodePayload::Phantom(_) => None,
  1364→
  1365→        NodePayload::Control(_) => {
  1366→            // Control nodes are handled by walk_control_flow
  1367→            None
  1368→        }
  1369→
  1370→        NodePayload::VTable(_) | NodePayload::ClosureEnv(_) => {
  1371→            // VTable and ClosureEnv are structural nodes; no IR statement
  1372→            None
  1373→        }
  1374→
  1375→        NodePayload::StructDef(_) | NodePayload::EnumDef(_) | NodePayload::Match(_) 
  1376→        | NodePayload::ConstantTime(_) => {
  1377→            // Type definitions, match nodes, and constant-time ops are
  1378→            // not lowered to IR statements in the bridge
  1379→            None
  1380→        }
  1381→    }
  1382→}
  1383→
  1384→// ── Function parameter extraction ──────────────────────────────────────
  1385→
  1386→/// Extract function parameters from DataFlow edges leaving the
  1387→/// FunctionEntry node.
  1388→///
  1389→/// Parameter types are inferred from the target node's payload, which
  1390→/// has been refined by BD inference (via `refine_scg_types_with_bd`).
  1391→/// If no type info is available, defaults to `ScgType::I64`.
  1392→fn extract_function_params(entry_id: NodeId, scg: &SCG, edge_idx: &EdgeIndex) -> Vec<ScgParam> {
  1393→    let df_edges = edge_idx.outgoing_df(entry_id);
  1394→    let mut params = Vec::new();
  1395→
  1396→    for (i, edge) in df_edges.iter().enumerate() {
  1397→        let (name, ty) = if let Some(target_node) = scg.get_node(edge.target) {
  1398→            match &target_node.payload {
  1399→                NodePayload::Allocation(alloc) => {
  1400→                    let name = alloc
  1401→                        .type_name
  1402→                        .clone()
  1403→                        .unwrap_or_else(|| format!("param_{}", i));
  1404→                    let ty = alloc
  1405→                        .type_name
  1406→                        .as_deref()
  1407→                        .and_then(parse_scg_type)
  1408→                        .unwrap_or(ScgType::I64);
  1409→                    (name, ty)
  1410→                }
  1411→                NodePayload::Computation(comp) => {
  1412→                    let name = comp
  1413→                        .result_type
  1414→                        .clone()
  1415→                        .unwrap_or_else(|| format!("param_{}", i));
  1416→                    let ty = comp
  1417→                        .result_type
  1418→                        .as_deref()
  1419→                        .and_then(parse_scg_type)
  1420→                        .unwrap_or(ScgType::I64);
  1421→                    (name, ty)
  1422→                }
  1423→                NodePayload::Cast(cast) => {
  1424→                    let name = format!("param_{}", i);
  1425→                    let ty = parse_scg_type(&cast.to_type).unwrap_or(ScgType::I64);
  1426→                    (name, ty)
  1427→                }
  1428→                _ => (format!("param_{}", i), ScgType::I64),
  1429→            }
  1430→        } else {
  1431→            (format!("param_{}", i), ScgType::I64)
  1432→        };
  1433→
  1434→        params.push(ScgParam { name, ty });
  1435→    }
  1436→
  1437→    params
  1438→}
  1439→
  1440→// ── Type parsing helper ────────────────────────────────────────────────
  1441→
  1442→/// Parse a type string into a `ScgType`.
  1443→fn parse_scg_type(type_str: &str) -> Option<ScgType> {
  1444→    match type_str {
  1445→        "i8" | "I8" => Some(ScgType::I8),
  1446→        "i16" | "I16" => Some(ScgType::I16),
  1447→        "i32" | "I32" => Some(ScgType::I32),
  1448→        "i64" | "I64" => Some(ScgType::I64),
  1449→        "u8" | "U8" => Some(ScgType::U8),
  1450→        "u16" | "U16" => Some(ScgType::U16),
  1451→        "u32" | "U32" => Some(ScgType::U32),
  1452→        "u64" | "U64" => Some(ScgType::U64),
  1453→        "ptr" | "*void" | "*u8" | "*i8" => Some(ScgType::Ptr),
  1454→        "void" => Some(ScgType::Void),
  1455→        _ => None,
  1456→    }
  1457→}
  1458→
  1459→// ── BD type refinement ─────────────────────────────────────────────────
  1460→
  1461→/// Map a BD `RepD` to the codegen's `ScgType`.
  1462→///
  1463→/// Uses the RepD's size and kind to pick the most specific `ScgType`:
  1464→/// - Pointer RepDs → `ScgType::Ptr`
  1465→/// - Byte RepDs → integer types by size (u8, u16, u32, u64)
  1466→/// - Struct/Array/Enum/Union → `ScgType::Ptr` (passed by reference)
  1467→/// - Generic → `ScgType::I64` (fallback)
  1468→fn repd_to_scg_type(repd: &RepD) -> ScgType {
  1469→    match repd {
  1470→        RepD::Ptr(_) | RepD::Func(_) => ScgType::Ptr,
  1471→        RepD::Byte(byte_rep) => match byte_rep.size {
  1472→            1 => ScgType::U8,
  1473→            2 => ScgType::U16,
  1474→            4 => ScgType::U32,
  1475→            _ => ScgType::U64,
  1476→        },
  1477→        RepD::Struct(_) | RepD::Array(_) | RepD::Enum(_) | RepD::Union(_) => ScgType::Ptr,
  1478→        RepD::Generic { .. } => ScgType::I64,
  1479→    }
  1480→}
  1481→
  1482→/// Convert a `ScgType` to its canonical string name for storing in SCG
  1483→/// node payloads (e.g., `AllocationNode.type_name`, `CastNode.from_type`).
  1484→fn scg_type_to_name(ty: &ScgType) -> &'static str {
  1485→    match ty {
  1486→        ScgType::I8 => "i8",
  1487→        ScgType::I16 => "i16",
  1488→        ScgType::I32 => "i32",
  1489→        ScgType::I64 => "i64",
  1490→        ScgType::U8 => "u8",
  1491→        ScgType::U16 => "u16",
  1492→        ScgType::U32 => "u32",
  1493→        ScgType::U64 => "u64",
  1494→        ScgType::Ptr => "ptr",
  1495→        ScgType::Void => "void",
  1496→    }
  1497→}
  1498→
  1499→/// Refine SCG node type metadata using BD inference results.
  1500→///
  1501→/// After BD inference, each node's `RepD` describes the actual memory
  1502→/// representation.  This function maps those RepDs back to `ScgType`s
  1503→/// and stores the result in the SCG node payloads so that downstream
  1504→/// bridge code (`convert_node_to_statement`, `extract_function_params`)
  1505→/// can pick up the refined types instead of using defaults.
  1506→///
  1507→/// # What is refined
  1508→///
  1509→/// - **Allocation nodes**: `type_name` is set if it was previously `None`.
  1510→/// - **Cast nodes**: `from_type` / `to_type` are updated if they couldn't
  1511→///   previously be parsed by `parse_scg_type`.
  1512→/// - **Computation nodes**: `result_type` is set if it was previously `None`.
  1513→pub fn refine_scg_types_with_bd(scg: &mut SCG, bd_results: &[(NodeId, BD)]) {
  1514→    let bd_map: HashMap<NodeId, &BD> = bd_results.iter().map(|(id, bd)| (*id, bd)).collect();
  1515→
  1516→    let node_ids: Vec<_> = scg.node_ids().collect();
  1517→    for node_id in node_ids {
  1518→        let Some(bd) = bd_map.get(&node_id) else {
  1519→            continue;
  1520→        };
  1521→        let inferred_type = repd_to_scg_type(&bd.repd);
  1522→        let type_name = scg_type_to_name(&inferred_type);
  1523→
  1524→        if let Some(node) = scg.get_node_mut(node_id) {
  1525→            match &mut node.payload {
  1526→                NodePayload::Allocation(alloc)
  1527→                    // Update type_name if it was previously unset.
  1528→                    if alloc.type_name.is_none() => {
  1529→                        alloc.type_name = Some(type_name.to_string());
  1530→                    }
  1531→                NodePayload::Cast(cast) => {
  1532→                    // Update from_type / to_type if they couldn't previously
  1533→                    // be parsed by `parse_scg_type` (i.e., they were opaque
  1534→                    // type names from the AST that don't map directly).
  1535→                    if parse_scg_type(&cast.from_type).is_none() {
  1536→                        cast.from_type = type_name.to_string();
  1537→                    }
  1538→                    if parse_scg_type(&cast.to_type).is_none() {
  1539→                        cast.to_type = type_name.to_string();
  1540→                    }
  1541→                }
  1542→                NodePayload::Computation(comp)
  1543→                    if comp.result_type.is_none() => {
  1544→                        comp.result_type = Some(type_name.to_string());
  1545→                    }
  1546→                _ => {}
  1547→            }
  1548→        }
  1549→    }
  1550→}
  1551→
  1552→// ── Entry-point detection ──────────────────────────────────────────────
  1553→
  1554→/// Find entry-point nodes (no incoming ControlFlow edges) for a function
  1555→/// that lacks an explicit FunctionEntry node.
  1556→fn find_entry_points(scg: &SCG, edge_idx: &EdgeIndex) -> Vec<NodeId> {
  1557→    let mut entry_points = Vec::new();
  1558→
  1559→    for node_id in scg.node_ids() {
  1560→        let has_incoming_cf = edge_idx
  1561→            .incoming
  1562→            .get(&node_id)
  1563→            .map(|edges| edges.iter().any(|e| e.kind == EdgeKind::ControlFlow))
  1564→            .unwrap_or(false);
  1565→
  1566→        if !has_incoming_cf {
  1567→            if let Some(node_data) = scg.get_node(node_id) {
  1568→                // Skip Phantom nodes
  1569→                if matches!(node_data.node_type, NodeType::Phantom) {
  1570→                    continue;
  1571→                }
  1572→                entry_points.push(node_id);
  1573→            }
  1574→        }
  1575→    }
  1576→
  1577→    // If no entry points found, use the first node
  1578→    if entry_points.is_empty() {
  1579→        if let Some(first_id) = scg.node_ids().next() {
  1580→            entry_points.push(first_id);
  1581→        }
  1582→    }
  1583→
  1584→    entry_points
  1585→}
  1586→
  1587→// ── Main bridge function ───────────────────────────────────────────────
  1588→
  1589→/// Convert a `vuma_scg::SCG` into the codegen's stub `Scg` type.
  1590→///
  1591→/// This function reconstructs real control flow (if/else, loops, function
  1592→/// boundaries, break/continue) from the SCG's graph structure, instead of
  1593→/// just flattening everything into a single linear "main" function.
  1594→///
  1595→/// # Algorithm
  1596→///
  1597→/// 1. **Phase 1: Function boundary detection** — Group nodes by
  1598→///    FunctionEntry→FunctionReturn regions.
  1599→/// 2. **Phase 2: Control flow reconstruction** — Within each function,
  1600→///    detect Branch+Join diamonds (if/else) and LoopHeader+LoopExit
  1601→///    patterns (loops).
  1602→/// 3. **Phase 3: Statement generation** — Convert non-control nodes into
  1603→///    ScgStatements with DataFlow-based variable naming.
  1604→pub fn bridge_scg_to_codegen(scg: &SCG) -> Scg {
  1605→    let edge_idx = EdgeIndex::build(scg);
  1606→    let mut consumed: HashSet<NodeId> = HashSet::new();
  1607→    let mut scg_nodes: Vec<ScgNode> = Vec::new();
  1608→
  1609→    // ── Phase 1: Function boundary detection ─────────────────────
  1610→    let function_entries: Vec<(NodeId, String)> = scg
  1611→        .nodes()
  1612→        .filter_map(|n| {
  1613→            if let NodePayload::Control(c) = &n.payload {
  1614→                if c.kind == ControlKind::FunctionEntry {
  1615→                    let name = c.label.clone().unwrap_or_else(|| "unknown".to_string());
  1616→                    return Some((n.id, name));
  1617→                }
  1618→            }
  1619→            None
  1620→        })
  1621→        .collect();
  1622→
  1623→    if !function_entries.is_empty() {
  1624→        // Process each function defined by a FunctionEntry node
  1625→        for (entry_id, func_name) in &function_entries {
  1626→            consumed.insert(*entry_id);
  1627→
  1628→            let return_node = find_function_return(*entry_id, scg, &edge_idx);
  1629→            let params = extract_function_params(*entry_id, scg, &edge_idx);
  1630→
  1631→            let mut body = if let Some(first_cf) = edge_idx.outgoing_cf(*entry_id).first() {
  1632→                let mut stop_at = HashSet::new();
  1633→                if let Some(ret) = return_node {
  1634→                    stop_at.insert(ret);
  1635→                }
  1636→                walk_control_flow(first_cf.target, scg, &edge_idx, &mut consumed, &stop_at)
  1637→            } else {
  1638→                vec![]
  1639→            };
  1640→
  1641→            // Add return statement if the function has a FunctionReturn
  1642→            if let Some(ret) = return_node {
  1643→                consumed.insert(ret);
  1644→            }
  1645→            if !body.iter().any(|s| matches!(s, ScgStatement::Return(_))) {
  1646→                body.push(ScgStatement::Return(vec![]));
  1647→            }
  1648→
  1649→            scg_nodes.push(ScgNode::Function(ScgFunction {
  1650→                name: func_name.clone(),
  1651→                params,
  1652→                results: vec![],
  1653→                body,
  1654→            }));
  1655→        }
  1656→    } else {
  1657→        // No FunctionEntry nodes — find entry points and walk control flow
  1658→        let entry_points = find_entry_points(scg, &edge_idx);
  1659→
  1660→        let mut body = Vec::new();
  1661→        for start in &entry_points {
  1662→            let stop_at = HashSet::new();
  1663→            let mut partial = walk_control_flow(*start, scg, &edge_idx, &mut consumed, &stop_at);
  1664→            body.append(&mut partial);
  1665→        }
  1666→
  1667→        // Process any remaining unconsumed nodes (connected only via DataFlow)

  1768→/// Compile VUMA source code with the given configuration.
  1769→///
  1770→/// This is the main entry point for the VUMA compilation pipeline.
  1771→/// It runs all stages in order, collecting errors and producing a
  1772→/// [`CompilationOutput`] on success.
  1773→///
  1774→/// # Pipeline Stages
  1775→///
  1776→/// 1. **Parse** — lex and parse source into an AST
  1777→/// 2. **AST → SCG** — convert the AST into a Semantic Computation Graph
  1778→/// 3. **SCG Validation** — verify the SCG is well-formed
  1779→/// 4. **BD Inference** — infer behavioral descriptions from the SCG
  1780→/// 5. **MSG Construction** — build the Memory State Graph from the SCG
  1781→/// 6. **IVE Verification** — verify the five core VUMA invariants
  1782→/// 7. **SCG Transforms** — run optimisation passes (DCE, CSE, etc.)
  1783→/// 8. **IR Lowering** — lower the SCG to an intermediate representation
  1784→/// 9. **Register Allocation** — assign physical ARM64 registers
  1785→/// 10. **Code Emission** — generate ARM64 machine code and ELF binary
  1786→pub fn compile(source: &str, config: &CompileConfig) -> Result<CompilationOutput, Vec<VumaError>> {
  1787→    compile_with_path(source, None, config)
  1788→}
