   272тЖТ   272тЖТ
   273тЖТ   273тЖТ/// Target-specific information needed during code generation.
   274тЖТ   274тЖТ///
   275тЖТ   275тЖТ/// This trait provides a data-driven interface for query target properties.
   276тЖТ   276тЖТ/// It must be implementable for ALL supported ISAs, including fundamentally
   277тЖТ   277тЖТ/// different architectures like Wasm (stack machine, no registers) and
   278тЖТ   278тЖТ/// MIPS (branch delay slots).
   279тЖТ   279тЖТ///
   280тЖТ   280тЖТ/// # Object Safety
   281тЖТ   281тЖТ///
   282тЖТ   282тЖТ/// This trait is object-safe: all methods take `&self` and return only
   283тЖТ   283тЖТ/// owned types or references with `'static` lifetime.
   284тЖТ   284тЖТpub trait TargetInfo: Send + Sync + 'static {
   285тЖТ   285тЖТ    // === Identity ===
   286тЖТ   286тЖТ
   287тЖТ   287тЖТ    /// ISA name in lowercase (e.g., "aarch64", "riscv64", "wasm32").
   288тЖТ   288тЖТ    fn isa_name(&self) -> &'static str;
   289тЖТ   289тЖТ
   290тЖТ   290тЖТ    /// LLVM-style target triple (e.g., "aarch64-unknown-linux-gnu").
   291тЖТ   291тЖТ    fn target_triple(&self) -> &'static str;
   292тЖТ   292тЖТ
   293тЖТ   293тЖТ    /// ELF `e_machine` value.  Returns 0 for non-ELF targets (Wasm).
   294тЖТ   294тЖТ    fn elf_machine_type(&self) -> u16;
   295тЖТ   295тЖТ
   296тЖТ   296тЖТ    /// Default base address for the .text section.
   297тЖТ   297тЖТ    fn default_base_address(&self) -> u64;
   298тЖТ   298тЖТ
   299тЖТ   299тЖТ    // === Data model ===
   300тЖТ   300тЖТ
   301тЖТ   301тЖТ    /// Pointer width in bytes (4 for 32-bit, 8 for 64-bit).
   302тЖТ   302тЖТ    fn pointer_width(&self) -> usize;
   303тЖТ   303тЖТ
   304тЖТ   304тЖТ    /// Size in bytes of `ty` on this target.
   305тЖТ   305тЖТ    fn size_of(&self, ty: &IRType) -> usize;
   306тЖТ   306тЖТ
   307тЖТ   307тЖТ    /// Natural alignment in bytes of `ty` on this target.
   308тЖТ   308тЖТ    fn alignment_of(&self, ty: &IRType) -> usize;
   309тЖТ   309тЖТ
   310тЖТ   310тЖТ    /// Byte order of this target.
   311тЖТ   311тЖТ    fn endianness(&self) -> Endianness;
   312тЖТ   312тЖТ
   313тЖТ   313тЖТ    // === Register architecture ===
   314тЖТ   314тЖТ
   315тЖТ   315тЖТ    /// Whether this target has registers at all.  `false` for Wasm (stack machine).
   316тЖТ   316тЖТ    fn has_registers(&self) -> bool;
   317тЖТ   317тЖТ
   318тЖТ   318тЖТ    /// Number of general-purpose registers.  0 for Wasm.
   319тЖТ   319тЖТ    fn num_gp_regs(&self) -> usize;
   320тЖТ   320тЖТ
   321тЖТ   321тЖТ    /// Number of SIMD/FP registers.  0 for Wasm.
   322тЖТ   322тЖТ    fn num_simd_fp_regs(&self) -> usize;
   323тЖТ   323тЖТ
   324тЖТ   324тЖТ    /// Whether the ISA has a hardwired-zero register (RISC-V x0, LoongArch r0).
   325тЖТ   325тЖТ    fn has_hardwired_zero(&self) -> bool;
   326тЖТ   326тЖТ
   327тЖТ   327тЖТ    /// Whether the ISA uses a link register (ARM, RISC-V, MIPS, PPC) rather than
   328тЖТ   328тЖТ    /// pushing the return address on the stack (x86_64).
   329тЖТ   329тЖТ    fn has_link_register(&self) -> bool;
   330тЖТ   330тЖТ
   331тЖТ   331тЖТ    /// Whether branches have delay slots (MIPS only).
   332тЖТ   332тЖТ    fn has_branch_delay_slots(&self) -> bool;
   333тЖТ   333тЖТ
   334тЖТ   334тЖТ    /// Whether this ISA uses a TOC (Table of Contents) pointer (PPC64 r2).
   335тЖТ   335тЖТ    fn has_toc_pointer(&self) -> bool;
   336тЖТ   336тЖТ
   337тЖТ   337тЖТ    /// Whether this ISA has dedicated condition register fields (PPC64 CR0-CR7).
   338тЖТ   338тЖТ    fn has_condition_registers(&self) -> bool;
   339тЖТ   339тЖТ
   340тЖТ   340тЖТ    // === Calling convention ===
   341тЖТ   341тЖТ
   342тЖТ   342тЖТ    /// Name of the calling convention (e.g., "aapcs64", "lp64d", "systemv").
   343тЖТ   343тЖТ    fn calling_convention_name(&self) -> &'static str;
   344тЖТ   344тЖТ
   345тЖТ   345тЖТ    /// Number of integer argument registers.
   346тЖТ   346тЖТ    fn num_int_arg_regs(&self) -> usize;
   347тЖТ   347тЖТ
   348тЖТ   348тЖТ    /// Number of FP/SIMD argument registers.
   349тЖТ   349тЖТ    fn num_fp_arg_regs(&self) -> usize;
   350тЖТ   350тЖТ
   351тЖТ   351тЖТ    /// Required stack alignment in bytes.
   352тЖТ   352тЖТ    fn stack_alignment(&self) -> usize;
   353тЖТ   353тЖТ
   354тЖТ   354тЖТ    // === Instruction encoding ===
   355тЖТ   355тЖТ
   356тЖТ   356тЖТ    /// Alignment requirement for instructions in bytes (4 for fixed-width RISCs,
   357тЖТ   357тЖТ    /// 1 for x86_64 and Wasm).
   358тЖТ   358тЖТ    fn instruction_alignment(&self) -> usize;
   359тЖТ   359тЖТ
   360тЖТ   360тЖТ    /// Minimum and maximum instruction width in bytes.
   361тЖТ   361тЖТ    /// - Fixed-width 32-bit ISAs: (4, 4)
   362тЖТ   362тЖТ    /// - x86_64: (1, 15)
   363тЖТ   363тЖТ    /// - RISC-V with RVC: (2, 4)
   364тЖТ   364тЖТ    /// - Wasm: (1, ~) but typically (1, 15)
   365тЖТ   365тЖТ    fn instruction_width_range(&self) -> (usize, usize);

   391тЖТ   391тЖТ

   394тЖТ   394тЖТ
   395тЖТ   395тЖТ    /// Encode an entire allocated program into its final binary form

   398тЖТ   398тЖТ

   402тЖТ   402тЖТ

   408тЖТ   408тЖТ

  1101тЖТ  1101тЖТ    fn instruction_alignment(&self) -> usize {

  1103тЖТ  1103тЖТ    }
  1104тЖТ  1104тЖТ    fn instruction_width_range(&self) -> (usize, usize) {
  1105тЖТ  1105тЖТ        (4, 4)
  1106тЖТ  1106тЖТ    }

  1114тЖТ  1114тЖТ// ---------------------------------------------------------------------------

  1116тЖТ  1116тЖТ/// Decode a 32-bit AArch64 instruction word into a human-readable mnemonic.
  1117тЖТ  1117тЖТ///
  1118тЖТ  1118тЖТ/// Covers the most common AArch64 instructions: ADD, SUB, MOV, LDR, STR, B,
  1119тЖТ  1119тЖТ/// BL, RET, CMP, B.cond, STP, LDP, NOP, MUL, SDIV, UDIV, AND, ORR, EOR,
  1120тЖТ  1120тЖТ/// plus several additional frequently-encountered encodings.
  1121тЖТ  1121тЖТfn decode_aarch64(word: u32) -> String {
  1122тЖТ  1122тЖТ    let rd = word & 0x1F;
  1123тЖТ  1123тЖТ    let rn = (word >> 5) & 0x1F;
  1124тЖТ  1124тЖТ    let rt = rd; // alias for load/store destination
  1125тЖТ  1125тЖТ    let rm = (word >> 16) & 0x1F;
  1126тЖТ  1126тЖТ    let imm12 = (word >> 10) & 0xFFF;
  1127тЖТ  1127тЖТ    let cond = word & 0xF;
  1128тЖТ  1128тЖТ
  1129тЖТ  1129тЖТ    // NOP: d503201f
  1130тЖТ  1130тЖТ    if word == 0xD503201F {
  1131тЖТ  1131тЖТ        return "nop".to_string();
  1132тЖТ  1132тЖТ    }
  1133тЖТ  1133тЖТ
  1134тЖТ  1134тЖТ    // RET: d65f03c0
  1135тЖТ  1135тЖТ    if word == 0xD65F03C0 {
  1136тЖТ  1136тЖТ        return "ret".to_string();
  1137тЖТ  1137тЖТ    }
  1138тЖТ  1138тЖТ
  1139тЖТ  1139тЖТ    let _top8 = word >> 24;
  1140тЖТ  1140тЖТ    let _top10 = word >> 22;
  1141тЖТ  1141тЖТ
  1142тЖТ  1142тЖТ    // --- ADD/SUB (immediate): 100100xx ...
  1143тЖТ  1143тЖТ    if (word >> 23) & 0x1FF == 0b1_0010_0010 {
  1144тЖТ  1144тЖТ        // ADD Xd, Xn, #imm12
  1145тЖТ  1145тЖТ        return format!("add x{}, x{}, #{}", rd, rn, imm12);
  1146тЖТ  1146тЖТ    }
  1147тЖТ  1147тЖТ    if (word >> 23) & 0x1FF == 0b1_1010_0010 {
  1148тЖТ  1148тЖТ        // SUB Xd, Xn, #imm12
  1149тЖТ  1149тЖТ        return format!("sub x{}, x{}, #{}", rd, rn, imm12);
  1150тЖТ  1150тЖТ    }
  1151тЖТ  1151тЖТ
  1152тЖТ  1152тЖТ    // --- ADD (shifted register): 1_00_0101_1_xxx ...
  1153тЖТ  1153тЖТ    if (word >> 24) & 0xFF == 0b1000_1011 {
  1154тЖТ  1154тЖТ        // ADD Xd, Xn, Xm
  1155тЖТ  1155тЖТ        return format!("add x{}, x{}, x{}", rd, rn, rm);
  1156тЖТ  1156тЖТ    }
  1157тЖТ  1157тЖТ
  1158тЖТ  1158тЖТ    // --- SUB (shifted register): 1_00_0101_1_xxx ... with S=1 at bit30
  1159тЖТ  1159тЖТ    if (word >> 24) & 0xFF == 0b1101_0110 {
  1160тЖТ  1160тЖТ        // SUB Xd, Xn, Xm (bit 30 set = sub)
  1161тЖТ  1161тЖТ        return format!("sub x{}, x{}, x{}", rd, rn, rm);
  1162тЖТ  1162тЖТ    }
  1163тЖТ  1163тЖТ
  1164тЖТ  1164тЖТ    // --- AND (shifted register): 1_00_0101_0_00_xxx
  1165тЖТ  1165тЖТ    if (word >> 24) & 0xFE == 0b1000_1010 {
  1166тЖТ  1166тЖТ        // Check bit 21-15: opcode[31:21] = 10001010_000
  1167тЖТ  1167тЖТ        if (word >> 21) & 0x7FF == 0b10001010000 {
  1168тЖТ  1168тЖТ            return format!("and x{}, x{}, x{}", rd, rn, rm);
  1169тЖТ  1169тЖТ        }
  1170тЖТ  1170тЖТ    }
  1171тЖТ  1171тЖТ
  1172тЖТ  1172тЖТ    // --- ORR (shifted register): 1_01_0101_0_00_xxx
  1173тЖТ  1173тЖТ    if (word >> 21) & 0x7FF == 0b10101010000 {
  1174тЖТ  1174тЖТ        return format!("orr x{}, x{}, x{}", rd, rn, rm);
  1175тЖТ  1175тЖТ    }
  1176тЖТ  1176тЖТ
  1177тЖТ  1177тЖТ    // --- EOR (shifted register): 1_10_0101_0_00_xxx
  1178тЖТ  1178тЖТ    if (word >> 21) & 0x7FF == 0b11001010000 {
  1179тЖТ  1179тЖТ        return format!("eor x{}, x{}, x{}", rd, rn, rm);
  1180тЖТ  1180тЖТ    }
  1181тЖТ  1181тЖТ
  1182тЖТ  1182тЖТ    // --- MOV (register): alias for ORR Xd, XZR, Xm
  1183тЖТ  1183тЖТ    // ORR Xd, XZR, Xm: 10101010_000_rm_000000_xzr_rd
  1184тЖТ  1184тЖТ    // More general: ORR with Rn=XZR(31)
  1185тЖТ  1185тЖТ    if (word >> 21) & 0x7FF == 0b10101010000 && rn == 31 {
  1186тЖТ  1186тЖТ        return format!("mov x{}, x{}", rd, rm);
  1187тЖТ  1187тЖТ    }
  1188тЖТ  1188тЖТ
  1189тЖТ  1189тЖТ    // --- MUL: MADD Xd, Xn, Xm, XZR
  1190тЖТ  1190тЖТ    // Encoding: 1_00_1101_1_000_Rm_0_01111_Rn_Rd
  1191тЖТ  1191тЖТ    if (word >> 21) & 0x7FF == 0b10011011000 && ((word >> 10) & 0x1F) == 0b01111 {
  1192тЖТ  1192тЖТ        return format!("mul x{}, x{}, x{}", rd, rn, rm);
  1193тЖТ  1193тЖТ    }
  1194тЖТ  1194тЖТ
  1195тЖТ  1195тЖТ    // --- SDIV: 1_00_1101_1_0100_00000_00001_Rn_Rd  (actually 1_00_1101_0100_Rm_00001_Rn_Rd)
  1196тЖТ  1196тЖТ    if (word >> 21) & 0x7FF == 0b10011010100 && (word >> 10) & 0x1F == 0b00001 {
  1197тЖТ  1197тЖТ        return format!("sdiv x{}, x{}, x{}", rd, rn, rm);
  1198тЖТ  1198тЖТ    }
  1199тЖТ  1199тЖТ
  1200тЖТ  1200тЖТ    // --- UDIV: 1_00_1101_0000_Rm_00001_Rn_Rd
  1201тЖТ  1201тЖТ    if (word >> 21) & 0x7FF == 0b10011010000 && (word >> 10) & 0x1F == 0b00001 {
  1202тЖТ  1202тЖТ        return format!("udiv x{}, x{}, x{}", rd, rn, rm);
  1203тЖТ  1203тЖТ    }
  1204тЖТ  1204тЖТ
  1205тЖТ  1205тЖТ    // --- CMP (immediate): SUBS XZR, Xn, #imm12
  1206тЖТ  1206тЖТ    // 11100001_00_xxx_xxx_xxx_xxx_xxx_11111_xxx
  1207тЖТ  1207тЖТ    if (word >> 23) & 0x1FF == 0b1_1110_0010 && rd == 31 {
  1208тЖТ  1208тЖТ        return format!("cmp x{}, #{}", rn, imm12);
  1209тЖТ  1209тЖТ    }
  1210тЖТ  1210тЖТ
  1211тЖТ  1211тЖТ    // --- CMP (register): SUBS XZR, Xn, Xm
  1212тЖТ  1212тЖТ    if (word >> 21) & 0x7FF == 0b11101011000 && rd == 31 {
  1213тЖТ  1213тЖТ        return format!("cmp x{}, x{}", rn, rm);
  1214тЖТ  1214тЖТ    }
  1215тЖТ  1215тЖТ
  1216тЖТ  1216тЖТ    // --- B.cond: 0101010x xxxxxxxxxx xxxxxx cond
  1217тЖТ  1217тЖТ    if (word >> 24) & 0xFF == 0x54 {
  1218тЖТ  1218тЖТ        let cond_name = match cond {
  1219тЖТ  1219тЖТ            0 => "eq",
  1220тЖТ  1220тЖТ            1 => "ne",
  1221тЖТ  1221тЖТ            2 => "cs",
  1222тЖТ  1222тЖТ            3 => "cc",
  1223тЖТ  1223тЖТ            4 => "mi",
  1224тЖТ  1224тЖТ            5 => "pl",
  1225тЖТ  1225тЖТ            6 => "vs",
  1226тЖТ  1226тЖТ            7 => "vc",
  1227тЖТ  1227тЖТ            8 => "hi",
  1228тЖТ  1228тЖТ            9 => "ls",
  1229тЖТ  1229тЖТ            10 => "ge",
  1230тЖТ  1230тЖТ            11 => "lt",
  1231тЖТ  1231тЖТ            12 => "gt",
  1232тЖТ  1232тЖТ            13 => "le",
  1233тЖТ  1233тЖТ            14 => "al",
  1234тЖТ  1234тЖТ            _ => "??",
  1235тЖТ  1235тЖТ        };
  1236тЖТ  1236тЖТ        let imm19 = (word >> 5) & 0x7FFFF;
  1237тЖТ  1237тЖТ        let offset = ((imm19 as i32) << 13) >> 11; // sign-extend and *4
  1238тЖТ  1238тЖТ        return format!("b.{} {:+}", cond_name, offset);
  1239тЖТ  1239тЖТ    }
  1240тЖТ  1240тЖТ
  1241тЖТ  1241тЖТ    // --- B (unconditional): 000101xx xxxxxxxxxxxxxxxxxxxx
  1242тЖТ  1242тЖТ    if (word >> 26) & 0x3F == 0b000101 {
  1243тЖТ  1243тЖТ        let imm26 = word & 0x3FFFFFF;
  1244тЖТ  1244тЖТ        let offset = ((imm26 as i32) << 6) >> 4; // sign-extend and *4
  1245тЖТ  1245тЖТ        return format!("b {:+}", offset);
  1246тЖТ  1246тЖТ    }
  1247тЖТ  1247тЖТ
  1248тЖТ  1248тЖТ    // --- BL: 100101xx xxxxxxxxxxxxxxxxxxxx
  1249тЖТ  1249тЖТ    if (word >> 26) & 0x3F == 0b100101 {
  1250тЖТ  1250тЖТ        let imm26 = word & 0x3FFFFFF;
  1251тЖТ  1251тЖТ        let offset = ((imm26 as i32) << 6) >> 4;
  1252тЖТ  1252тЖТ        return format!("bl {:+}", offset);
  1253тЖТ  1253тЖТ    }
  1254тЖТ  1254тЖТ
  1255тЖТ  1255тЖТ    // --- LDR (unsigned offset): 11111001_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1256тЖТ  1256тЖТ    if (word >> 22) & 0x3FF == 0b1111100101 {
  1257тЖТ  1257тЖТ        let imm12_raw = (word >> 10) & 0xFFF;
  1258тЖТ  1258тЖТ        let offset = imm12_raw * 8; // scale by 8 for 64-bit
  1259тЖТ  1259тЖТ        return format!("ldr x{}, [x{}, #{}]", rt, rn, offset);
  1260тЖТ  1260тЖТ    }
  1261тЖТ  1261тЖТ
  1262тЖТ  1262тЖТ    // --- STR (unsigned offset): 11111000_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1263тЖТ  1263тЖТ    if (word >> 22) & 0x3FF == 0b1111100001 {
  1264тЖТ  1264тЖТ        let imm12_raw = (word >> 10) & 0xFFF;
  1265тЖТ  1265тЖТ        let offset = imm12_raw * 8;
  1266тЖТ  1266тЖТ        return format!("str x{}, [x{}, #{}]", rt, rn, offset);
  1267тЖТ  1267тЖТ    }
  1268тЖТ  1268тЖТ
  1269тЖТ  1269тЖТ    // --- LDP (signed offset, 64-bit): 101_0100_110_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1270тЖТ  1270тЖТ    if (word >> 22) & 0x3FF == 0b1010100110 {
  1271тЖТ  1271тЖТ        let rt2 = (word >> 10) & 0x1F;
  1272тЖТ  1272тЖТ        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1273тЖТ  1273тЖТ        let offset = imm7 * 8;
  1274тЖТ  1274тЖТ        return format!("ldp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1275тЖТ  1275тЖТ    }
  1276тЖТ  1276тЖТ
  1277тЖТ  1277тЖТ    // --- STP (signed offset, 64-bit): 101_0100_010_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1278тЖТ  1278тЖТ    if (word >> 22) & 0x3FF == 0b1010100010 {
  1279тЖТ  1279тЖТ        let rt2 = (word >> 10) & 0x1F;
  1280тЖТ  1280тЖТ        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1281тЖТ  1281тЖТ        let offset = imm7 * 8;
  1282тЖТ  1282тЖТ        return format!("stp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1283тЖТ  1283тЖТ    }
  1284тЖТ  1284тЖТ
  1285тЖТ  1285тЖТ    // --- MOVZ: 110100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1286тЖТ  1286тЖТ    if (word >> 23) & 0x1FF == 0b110100101 {
  1287тЖТ  1287тЖТ        let hw = (word >> 21) & 0x3;
  1288тЖТ  1288тЖТ        let imm16 = (word >> 5) & 0xFFFF;
  1289тЖТ  1289тЖТ        return format!("movz x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1290тЖТ  1290тЖТ    }
  1291тЖТ  1291тЖТ
  1292тЖТ  1292тЖТ    // --- MOVK: 111100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1293тЖТ  1293тЖТ    if (word >> 23) & 0x1FF == 0b111100101 {
  1294тЖТ  1294тЖТ        let hw = (word >> 21) & 0x3;
  1295тЖТ  1295тЖТ        let imm16 = (word >> 5) & 0xFFFF;
  1296тЖТ  1296тЖТ        return format!("movk x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1297тЖТ  1297тЖТ    }
  1298тЖТ  1298тЖТ
  1299тЖТ  1299тЖТ    format!(".word {:08x}", word)
  1300тЖТ  1300тЖТ}

  1329тЖТ  1329тЖТ/// Compute the stack frame size for an IR function.
  1330тЖТ  1330тЖТ///
  1331тЖТ  1331тЖТ/// Replicates the private `compute_frame_size` function in `emit.rs`:
  1332тЖТ  1332тЖТ/// sums `Alloc` instruction sizes, adds 16 bytes for the FP/LR save pair,
  1333тЖТ  1333тЖТ/// and rounds up to 16-byte alignment.
  1334тЖТ  1334тЖТfn aarch64_compute_frame_size(func: &IRFunction) -> usize {
  1335тЖТ  1335тЖТ    let mut total: u32 = 16; // FP/LR save pair
  1336тЖТ  1336тЖТ    for block in &func.blocks {
  1337тЖТ  1337тЖТ        for instr in &block.instructions {
  1338тЖТ  1338тЖТ            if let IRInstr::Alloc { size, .. } = instr {
  1339тЖТ  1339тЖТ                let aligned = (*size).div_ceil(16) * 16;
  1340тЖТ  1340тЖТ                total += aligned;
  1341тЖТ  1341тЖТ            }
  1342тЖТ  1342тЖТ        }
  1343тЖТ  1343тЖТ    }
  1344тЖТ  1344тЖТ    // Round up to 16-byte alignment
  1345тЖТ  1345тЖТ    total = (total + 15) & !15;
  1346тЖТ  1346тЖТ    total as usize
  1347тЖТ  1347тЖТ}

  1349тЖТ  1349тЖТ/// Build a minimal ELF64 binary for AArch64 from raw code bytes.
  1350тЖТ  1350тЖТ///
  1351тЖТ  1351тЖТ/// Produces a static executable with a single LOAD segment containing the
  1352тЖТ  1352тЖТ/// `.text` section.  This is sufficient for programs that have already been
  1353тЖТ  1353тЖТ/// emitted through the `allocate_registers` / `encode_function` pipeline.
  1354тЖТ  1354тЖТfn build_minimal_aarch64_elf(code: &[u8], base_addr: u64) -> Vec<u8> {
  1355тЖТ  1355тЖТ    // Layout: ELF header (64) | 1 program header (56) | code
  1356тЖТ  1356тЖТ    let elf_header_size: u64 = 64;
  1357тЖТ  1357тЖТ    let phdr_size: u64 = 56;
  1358тЖТ  1358тЖТ    let text_offset = elf_header_size + phdr_size;
  1359тЖТ  1359тЖТ    let text_size = code.len() as u64;
  1360тЖТ  1360тЖТ    let entry_point = base_addr + text_offset;
  1361тЖТ  1361тЖТ
  1362тЖТ  1362тЖТ    let mut elf = Vec::with_capacity(text_offset as usize + code.len());
  1363тЖТ  1363тЖТ
  1364тЖТ  1364тЖТ    // --- e_ident ---
  1365тЖТ  1365тЖТ    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1366тЖТ  1366тЖТ    elf.push(2); // ELFCLASS64
  1367тЖТ  1367тЖТ    elf.push(1); // ELFDATA2LSB
  1368тЖТ  1368тЖТ    elf.push(1); // EV_CURRENT
  1369тЖТ  1369тЖТ    elf.push(3); // ELFOSABI_LINUX
  1370тЖТ  1370тЖТ    elf.push(0); // padding
  1371тЖТ  1371тЖТ    elf.extend_from_slice(&[0u8; 7]); // padding
  1372тЖТ  1372тЖТ
  1373тЖТ  1373тЖТ    // --- ELF header fields ---
  1374тЖТ  1374тЖТ    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1375тЖТ  1375тЖТ    elf.extend_from_slice(&183u16.to_le_bytes()); // e_machine = EM_AARCH64
  1376тЖТ  1376тЖТ    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1377тЖТ  1377тЖТ    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1378тЖТ  1378тЖТ    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1379тЖТ  1379тЖТ    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1380тЖТ  1380тЖТ    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1381тЖТ  1381тЖТ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1382тЖТ  1382тЖТ    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1383тЖТ  1383тЖТ    elf.extend_from_slice(&1u16.to_le_bytes()); // e_phnum
  1384тЖТ  1384тЖТ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1385тЖТ  1385тЖТ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1386тЖТ  1386тЖТ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1387тЖТ  1387тЖТ
  1388тЖТ  1388тЖТ    // --- Program Header (single LOAD segment: PF_R | PF_X) ---
  1389тЖТ  1389тЖТ    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1390тЖТ  1390тЖТ    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X

  1411тЖТ  1411тЖТ        // performs register allocation and instruction encoding.
  1412тЖТ  1412тЖТ        let mut emitter = crate::emit::Emitter::new();
  1413тЖТ  1413тЖТ        let code = emitter
  1414тЖТ  1414тЖТ            .emit_function(func)
  1415тЖТ  1415тЖТ            .map_err(|e| BackendError::RegisterAllocFailed {
  1416тЖТ  1416тЖТ                isa: "aarch64",
  1417тЖТ  1417тЖТ                reason: e.to_string(),
  1418тЖТ  1418тЖТ            })?;
  1419тЖТ  1419тЖТ
  1420тЖТ  1420тЖТ        let func_name = func.name.clone();
  1421тЖТ  1421тЖТ        let frame_size = aarch64_compute_frame_size(func);
  1422тЖТ  1422тЖТ
  1423тЖТ  1423тЖТ        // Convert each 32-bit ARM64 instruction word into an AllocatedInstruction
  1424тЖТ  1424тЖТ        // with its little-endian encoded bytes.
  1425тЖТ  1425тЖТ        let instructions: Vec<AllocatedInstruction> = code
  1426тЖТ  1426тЖТ            .iter()
  1427тЖТ  1427тЖТ            .enumerate()

  1429тЖТ  1429тЖТ                opcode: format!("arm64_{}", i),
  1430тЖТ  1430тЖТ                reads: vec![],
  1431тЖТ  1431тЖТ                writes: vec![],
  1432тЖТ  1432тЖТ                encoded: word.to_le_bytes().to_vec(),
  1433тЖТ  1433тЖТ            })
  1434тЖТ  1434тЖТ            .collect();
  1435тЖТ  1435тЖТ
  1436тЖТ  1436тЖТ        let code_size = instructions.len() * 4;
  1437тЖТ  1437тЖТ
  1438тЖТ  1438тЖТ        Ok(AllocatedFunction {
  1439тЖТ  1439тЖТ            name: func_name,
  1440тЖТ  1440тЖТ            blocks: vec![AllocatedBlock {
  1441тЖТ  1441тЖТ                label: "entry".to_string(),
  1442тЖТ  1442тЖТ                instructions,
  1443тЖТ  1443тЖТ                code_offset: 0,
  1444тЖТ  1444тЖТ            }],
  1445тЖТ  1445тЖТ            frame_size,
  1446тЖТ  1446тЖТ            callee_saved: vec![],
  1447тЖТ  1447тЖТ            spill_slots: 0,
  1448тЖТ  1448тЖТ            code_size,
  1449тЖТ  1449тЖТ            relocations: Vec::new(),
  1450тЖТ  1450тЖТ        })
  1451тЖТ  1451тЖТ    }

  1454тЖТ  1454тЖТ        let mut bytes = Vec::new();
  1455тЖТ  1455тЖТ        for block in &func.blocks {
  1456тЖТ  1456тЖТ            for instr in &block.instructions {
  1457тЖТ  1457тЖТ                bytes.extend_from_slice(&instr.encoded);
  1458тЖТ  1458тЖТ            }

  1461тЖТ  1461тЖТ    }

  1465тЖТ  1465тЖТ        let mut all_code = Vec::new();

  1471тЖТ  1471тЖТ            }

  1474тЖТ  1474тЖТ        // Wrap in a minimal ELF64 binary for AArch64.

  1495тЖТ  1495тЖТ    fn disassemble(&self, bytes: &[u8], addr: u64) -> Vec<String> {
  1496тЖТ  1496тЖТ        // Mnemonic decoder for AArch64 (4-byte fixed-width instructions).
  1497тЖТ  1497тЖТ        let mut lines = Vec::new();
  1498тЖТ  1498тЖТ        let mut offset = 0usize;
  1499тЖТ  1499тЖТ        let mut pc = addr;
  1500тЖТ  1500тЖТ        while offset + 4 <= bytes.len() {
  1501тЖТ  1501тЖТ            let word = u32::from_le_bytes([
  1502тЖТ  1502тЖТ                bytes[offset],
  1503тЖТ  1503тЖТ                bytes[offset + 1],
  1504тЖТ  1504тЖТ                bytes[offset + 2],
  1505тЖТ  1505тЖТ                bytes[offset + 3],
  1506тЖТ  1506тЖТ            ]);
  1507тЖТ  1507тЖТ            let mnemonic = if let Some(instr) = crate::arm64::Instruction::decode(word) {
  1508тЖТ  1508тЖТ                format!("{}", instr)
  1509тЖТ  1509тЖТ            } else {
  1510тЖТ  1510тЖТ                decode_aarch64(word)
  1511тЖТ  1511тЖТ            };
  1512тЖТ  1512тЖТ            lines.push(format!("{:#010x}:  {:08x}  {}", pc, word, mnemonic));
  1513тЖТ  1513тЖТ            offset += 4;
  1514тЖТ  1514тЖТ            pc += 4;
  1515тЖТ  1515тЖТ        }
  1516тЖТ  1516тЖТ        if offset < bytes.len() {
  1517тЖТ  1517тЖТ            let remaining = &bytes[offset..];
  1518тЖТ  1518тЖТ            lines.push(format!("{:#010x}:  {:02x?}", pc, remaining));
  1519тЖТ  1519тЖТ        }
  1520тЖТ  1520тЖТ        lines
  1521тЖТ  1521тЖТ    }
