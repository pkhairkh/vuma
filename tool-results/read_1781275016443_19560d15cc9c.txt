     1в†’     1в†’//! # MIPS 64-bit Backend
     2в†’     2в†’//!
     3в†’     3в†’//! Implements the `Backend` trait for the MIPS 64-bit target (N64 ABI,
     4в†’     4в†’//! big-endian).  This module provides:
     5в†’     5в†’//!
     6в†’     6в†’//! - `Gpr` вЂ” General-purpose register enum ($0вЂ“$31)
     7в†’     7в†’//! - `Fpr` вЂ” Floating-point register enum ($f0вЂ“$f31)
     8в†’     8в†’//! - `Instruction` вЂ” MIPS64 instruction enum with correct encoding for
     9в†’     9в†’//!   R-type, I-type, and J-type formats
    10в†’    10в†’//! - `Mips64Backend` вЂ” `Backend` implementation that lowers IR to MIPS64
    11в†’    11в†’//!   machine code and emits ELF64 binaries
    12в†’    12в†’//!
    13в†’    13в†’//! ## MIPS64 Register Convention (N64 ABI)
    14в†’    14в†’//!
    15в†’    15в†’//! | Register(s) | ABI Name | Role                              |
    16в†’    16в†’//! |-------------|----------|-----------------------------------|
    17в†’    17в†’//! | $0          | $zero    | Hardwired zero                    |
    18в†’    18в†’//! | $1          | $at      | Assembler temporary               |
    19в†’    19в†’//! | $2вЂ“$3       | $v0вЂ“$v1  | Return values                     |
    20в†’    20в†’//! | $4вЂ“$7       | $a0вЂ“$a3  | Argument registers                |
    21в†’    21в†’//! | $8вЂ“$15      | $t0вЂ“$t7  | Caller-saved temporaries          |
    22в†’    22в†’//! | $16вЂ“$23     | $s0вЂ“$s7  | Callee-saved                      |
    23в†’    23в†’//! | $24вЂ“$25     | $t8вЂ“$t9  | Caller-saved temporaries          |
    24в†’    24в†’//! | $26вЂ“$27     | $k0вЂ“$k1  | Kernel registers                  |
    25в†’    25в†’//! | $28         | $gp      | Global pointer                    |
    26в†’    26в†’//! | $29         | $sp      | Stack pointer                     |
    27в†’    27в†’//! | $30         | $fp      | Frame pointer (callee-saved)      |
    28в†’    28в†’//! | $31         | $ra      | Return address                    |
    29в†’    29в†’//!
    30в†’    30в†’//! ## Branch Delay Slots
    31в†’    31в†’//!
    32в†’    32в†’//! MIPS has branch delay slots: the instruction immediately following a branch
    33в†’    33в†’//! or jump is always executed before the branch takes effect.  This backend
    34в†’    34в†’//! inserts a NOP (0x00000000) in every delay slot for correctness.
    35в†’    35в†’//!
    36в†’    36в†’//! ## Instruction Encoding
    37в†’    37в†’//!
    38в†’    38в†’//! All instructions are 32 bits, **big-endian**, with three formats:
    39в†’    39в†’//!
    40в†’    40в†’//! - **R-type**: `opcode[31:26] | rs[25:21] | rt[20:16] | rd[15:11] | sa[10:6] | funct[5:0]`
    41в†’    41в†’//! - **I-type**: `opcode[31:26] | rs[25:21] | rt[20:16] | imm[15:0]`
    42в†’    42в†’//! - **J-type**: `opcode[31:26] | target[25:0]`
    43в†’    43в†’//!
    44в†’    44в†’//! ## References
    45в†’    45в†’//!
    46в†’    46в†’//! - MIPS64 Architecture for Programmers, Volume II: Instruction Set
    47в†’    47в†’//! - <https://www.mips.com/products/architectures/mips64/>

  1340в†’  1340в†’/// Build a minimal ELF64 binary for MIPS64 (big-endian) from raw code bytes.
  1341в†’  1341в†’///
  1342в†’  1342в†’/// Produces a static executable with a single LOAD segment containing the
  1343в†’  1343в†’/// `.text` section.  The ELF header and program header are written in
  1344в†’  1344в†’/// big-endian byte order.
  1345в†’  1345в†’fn build_minimal_mips64_elf(code: &[u8], base_addr: u64) -> Vec<u8> {
  1346в†’  1346в†’    // Layout: ELF header (64) | 1 program header (56) | code
  1347в†’  1347в†’    let elf_header_size: u64 = 64;
  1348в†’  1348в†’    let phdr_size: u64 = 56;
  1349в†’  1349в†’    let text_offset = elf_header_size + phdr_size;
  1350в†’  1350в†’    let text_size = code.len() as u64;
  1351в†’  1351в†’    let entry_point = base_addr + text_offset;
  1352в†’  1352в†’
  1353в†’  1353в†’    let mut elf = Vec::with_capacity(text_offset as usize + code.len());
  1354в†’  1354в†’
  1355в†’  1355в†’    // --- e_ident ---
  1356в†’  1356в†’    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1357в†’  1357в†’    elf.push(2); // ELFCLASS64
  1358в†’  1358в†’    elf.push(2); // ELFDATA2MSB (big-endian)
  1359в†’  1359в†’    elf.push(1); // EV_CURRENT
  1360в†’  1360в†’    elf.push(3); // ELFOSABI_LINUX
  1361в†’  1361в†’    elf.push(0); // padding
  1362в†’  1362в†’    elf.extend_from_slice(&[0u8; 7]); // padding
  1363в†’  1363в†’
  1364в†’  1364в†’    // --- ELF header fields (big-endian) ---
  1365в†’  1365в†’    elf.extend_from_slice(&2u16.to_be_bytes()); // e_type = ET_EXEC
  1366в†’  1366в†’    elf.extend_from_slice(&8u16.to_be_bytes()); // e_machine = EM_MIPS
  1367в†’  1367в†’    elf.extend_from_slice(&1u32.to_be_bytes()); // e_version
  1368в†’  1368в†’    elf.extend_from_slice(&entry_point.to_be_bytes()); // e_entry
  1369в†’  1369в†’    elf.extend_from_slice(&elf_header_size.to_be_bytes()); // e_phoff
  1370в†’  1370в†’    elf.extend_from_slice(&0u64.to_be_bytes()); // e_shoff (no section headers)
  1371в†’  1371в†’                                                // e_flags: MIPS flags вЂ” MIPS64 N64 ABI.
  1372в†’  1372в†’                                                // EF_MIPS_ARCH_64 = 0x6000 (MIPS64 ISA)
  1373в†’  1373в†’                                                // EF_MIPS_ABI_O64 = 0x00002000 (O64 ABI вЂ” closest to N64 in ELF flags)
  1374в†’  1374в†’    elf.extend_from_slice(&0x8000u32.to_be_bytes()); // e_flags (MIPS64)

  1391в†’  1391в†’
  1392в†’  1392в†’    // --- Code section ---
  1393в†’  1393в†’    elf.extend_from_slice(code);

  1395в†’  1395в†’    elf
  1396в†’  1396в†’}

  1426в†’  1426в†’/// Compute the stack frame size for an IR function on MIPS64.
  1427в†’  1427в†’///
  1428в†’  1428в†’/// Sums `Alloc` instruction sizes, adds 8 bytes for the $ra save slot,
  1429в†’  1429в†’/// and rounds up to 16-byte alignment.
  1430в†’  1430в†’fn mips64_compute_frame_size(func: &IRFunction) -> usize {
  1431в†’  1431в†’    let mut total: u32 = 16; // $ra save slot + $fp save slot
  1432в†’  1432в†’    for block in &func.blocks {
  1433в†’  1433в†’        for instr in &block.instructions {
  1434в†’  1434в†’            if let IRInstr::Alloc { size, .. } = instr {
  1435в†’  1435в†’                let aligned = (*size).div_ceil(16) * 16;
  1436в†’  1436в†’                total += aligned;
  1437в†’  1437в†’            }
  1438в†’  1438в†’        }
  1439в†’  1439в†’    }
  1440в†’  1440в†’    // Round up to 16-byte alignment
  1441в†’  1441в†’    total = (total + 15) & !15;
  1442в†’  1442в†’    total as usize
  1443в†’  1443в†’}

  1479в†’  1479в†’/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1480в†’  1480в†’///
  1481в†’  1481в†’/// Argument virtual registers are mapped to a0вЂ“a3 first.  Remaining virtual
  1482в†’  1482в†’/// registers are assigned from the pool of allocatable registers.
  1483в†’  1483в†’fn map_vreg_to_gpr(
  1484в†’  1484в†’    vreg_id: u32,
  1485в†’  1485в†’    arg_index: Option<usize>,
  1486в†’  1486в†’    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1487в†’  1487в†’) -> Gpr {
  1488в†’  1488в†’    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1489в†’  1489в†’        return *gpr;
  1490в†’  1490в†’    }
  1491в†’  1491в†’
  1492в†’  1492в†’    // If this is an argument, map it to the corresponding argument register.
  1493в†’  1493в†’    if let Some(idx) = arg_index {
  1494в†’  1494в†’        if let Some(reg) = Gpr::arg_register(idx) {
  1495в†’  1495в†’            vreg_map.insert(vreg_id, reg);
  1496в†’  1496в†’            return reg;
  1497в†’  1497в†’        }
  1498в†’  1498в†’    }
  1499в†’  1499в†’
  1500в†’  1500в†’    // Otherwise, pick the first available allocatable register.
  1501в†’  1501в†’    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1502в†’  1502в†’    for &reg in ALLOCATABLE_GPRS {
  1503в†’  1503в†’        if !used.contains(&reg) {
  1504в†’  1504в†’            vreg_map.insert(vreg_id, reg);
  1505в†’  1505в†’            return reg;
  1506в†’  1506в†’        }
  1507в†’  1507в†’    }
  1508в†’  1508в†’
  1509в†’  1509в†’    // Fallback: use t9 as a scratch register (will cause issues, but shouldn't
  1510в†’  1510в†’    // happen with reasonable register pressure).
  1511в†’  1511в†’    Gpr::T9
  1512в†’  1512в†’}
