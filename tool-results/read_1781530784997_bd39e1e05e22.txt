     1‚Üí     1‚Üí//! # Intermediate Representation (IR)
     2‚Üí     2‚Üí//!
     3‚Üí     3‚Üí//! Defines the IR types used as the central representation between the SCG
     4‚Üí     4‚Üí//! (Semantic Computation Graph) front-end and the ARM64 code emitter.
     5‚Üí     5‚Üí//!
     6‚Üí     6‚Üí//! ## Hierarchy
     7‚Üí     7‚Üí//!
     8‚Üí     8‚Üí//! ```text
     9‚Üí     9‚Üí//! IRProgram
    10‚Üí    10‚Üí//!  ‚îú‚îÄ‚îÄ Vec<IRFunction>
    11‚Üí    11‚Üí//!  ‚îÇ    ‚îú‚îÄ‚îÄ name, params, results, param_types, result_types
    12‚Üí    12‚Üí//!  ‚îÇ    ‚îî‚îÄ‚îÄ Vec<IRBlock>
    13‚Üí    13‚Üí//!  ‚îÇ         ‚îú‚îÄ‚îÄ label
    14‚Üí    14‚Üí//!  ‚îÇ         ‚îú‚îÄ‚îÄ Vec<IRInstr>
    15‚Üí    15‚Üí//!  ‚îÇ         ‚îî‚îÄ‚îÄ IRTerminator
    16‚Üí    16‚Üí//!  ‚îî‚îÄ‚îÄ Vec<DataSection>
    17‚Üí    17‚Üí//! ```
    18‚Üí    18‚Üí//!
    19‚Üí    19‚Üí//! ## Type System
    20‚Üí    20‚Üí//!
    21‚Üí    21‚Üí//! The IR includes a type system (`IRType`) that models the ARM64 LP64 data
    22‚Üí    22‚Üí//! model, with functions for computing sizes, alignments, AAPCS64 argument
    23‚Üí    23‚Üí//! classification, calling-convention layout, and stack-frame layout.
    24‚Üí    24‚Üí//!
    25‚Üí    25‚Üí//! The IR is intentionally low-level but target-independent.  It uses virtual
    26‚Üí    26‚Üí//! registers (`IRValue::Register(id)`) that are later mapped to physical
    27‚Üí    27‚Üí//! ARM64 registers by the register allocator.
    28‚Üí    28‚Üí
    29‚Üí    29‚Üíuse std::collections::{HashMap, HashSet};
    30‚Üí    30‚Üíuse std::fmt;

    36‚Üí    36‚Üí/// A type in the IR type system.
    37‚Üí    37‚Üí///
    38‚Üí    38‚Üí/// Target-independent type representation.  The size of `Ptr` and `Func`
    39‚Üí    39‚Üí/// depends on the target's pointer width (4 bytes on 32-bit targets like
    40‚Üí    40‚Üí/// Wasm32/ARM32, 8 bytes on 64-bit targets like ARM64/x86_64).  All other
    41‚Üí    41‚Üí/// types have fixed sizes regardless of target.
    42‚Üí    42‚Üí#[derive(Debug, Clone, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
    43‚Üí    43‚Üípub enum IRType {
    44‚Üí    44‚Üí    /// Signed 8-bit integer.
    45‚Üí    45‚Üí    I8,
    46‚Üí    46‚Üí    /// Signed 16-bit integer.
    47‚Üí    47‚Üí    I16,
    48‚Üí    48‚Üí    /// Signed 32-bit integer.
    49‚Üí    49‚Üí    I32,
    50‚Üí    50‚Üí    /// Signed 64-bit integer.
    51‚Üí    51‚Üí    I64,
    52‚Üí    52‚Üí    /// Unsigned 8-bit integer.
    53‚Üí    53‚Üí    U8,
    54‚Üí    54‚Üí    /// Unsigned 16-bit integer.
    55‚Üí    55‚Üí    U16,
    56‚Üí    56‚Üí    /// Unsigned 32-bit integer.
    57‚Üí    57‚Üí    U32,
    58‚Üí    58‚Üí    /// Unsigned 64-bit integer.
    59‚Üí    59‚Üí    U64,
    60‚Üí    60‚Üí    /// 32-bit IEEE 754 floating-point.
    61‚Üí    61‚Üí    F32,
    62‚Üí    62‚Üí    /// 64-bit IEEE 754 floating-point.
    63‚Üí    63‚Üí    F64,
    64‚Üí    64‚Üí    /// Opaque pointer.  Size is target-dependent: 4 bytes on 32-bit targets
    65‚Üí    65‚Üí    /// (Wasm32, ARM32), 8 bytes on 64-bit targets (ARM64, x86_64, etc.).
    66‚Üí    66‚Üí    Ptr,
    67‚Üí    67‚Üí    /// Void type (size 0, only valid as a function return type).
    68‚Üí    68‚Üí    Void,
    69‚Üí    69‚Üí    /// Function pointer (pointer-sized).
    70‚Üí    70‚Üí    Func,
    71‚Üí    71‚Üí    /// Struct type with a name and ordered fields.
    72‚Üí    72‚Üí    Struct {
    73‚Üí    73‚Üí        /// Struct name (may be empty for anonymous structs).
    74‚Üí    74‚Üí        name: String,
    75‚Üí    75‚Üí        /// Fields in declaration order.
    76‚Üí    76‚Üí        fields: Vec<IRType>,
    77‚Üí    77‚Üí    },
    78‚Üí    78‚Üí    /// Array type with an element type and element count.
    79‚Üí    79‚Üí    Array {
    80‚Üí    80‚Üí        /// Element type.
    81‚Üí    81‚Üí        element: Box<IRType>,
    82‚Üí    82‚Üí        /// Number of elements.
    83‚Üí    83‚Üí        count: usize,
    84‚Üí    84‚Üí    },
    85‚Üí    85‚Üí}

    87‚Üí    87‚Üíimpl IRType {
    88‚Üí    88‚Üí    /// Returns `true` if this is an integer type (signed or unsigned).
    89‚Üí    89‚Üí    pub fn is_integer(&self) -> bool {
    90‚Üí    90‚Üí        matches!(
    91‚Üí    91‚Üí            self,
    92‚Üí    92‚Üí            IRType::I8
    93‚Üí    93‚Üí                | IRType::I16
    94‚Üí    94‚Üí                | IRType::I32
    95‚Üí    95‚Üí                | IRType::I64
    96‚Üí    96‚Üí                | IRType::U8
    97‚Üí    97‚Üí                | IRType::U16
    98‚Üí    98‚Üí                | IRType::U32
    99‚Üí    99‚Üí                | IRType::U64
   100‚Üí   100‚Üí        )
   101‚Üí   101‚Üí    }
   102‚Üí   102‚Üí
   103‚Üí   103‚Üí    /// Returns `true` if this is a floating-point type.
   104‚Üí   104‚Üí    pub fn is_float(&self) -> bool {
   105‚Üí   105‚Üí        matches!(self, IRType::F32 | IRType::F64)
   106‚Üí   106‚Üí    }
   107‚Üí   107‚Üí
   108‚Üí   108‚Üí    /// Returns `true` if this type is passed by value in registers under
   109‚Üí   109‚Üí    /// AAPCS64 (integer or FP primitive, pointer, func, or small struct/array).
   110‚Üí   110‚Üí    pub fn is_register_passable(&self) -> bool {
   111‚Üí   111‚Üí        match self {
   112‚Üí   112‚Üí            IRType::I8
   113‚Üí   113‚Üí            | IRType::I16
   114‚Üí   114‚Üí            | IRType::I32
   115‚Üí   115‚Üí            | IRType::I64
   116‚Üí   116‚Üí            | IRType::U8
   117‚Üí   117‚Üí            | IRType::U16
   118‚Üí   118‚Üí            | IRType::U32
   119‚Üí   119‚Üí            | IRType::U64
   120‚Üí   120‚Üí            | IRType::F32
   121‚Üí   121‚Üí            | IRType::F64
   122‚Üí   122‚Üí            | IRType::Ptr
   123‚Üí   123‚Üí            | IRType::Func => true,
   124‚Üí   124‚Üí            IRType::Void => false,
   125‚Üí   125‚Üí            IRType::Struct { fields, .. } => size_of(self) <= 16 && !fields.is_empty(),
   126‚Üí   126‚Üí            IRType::Array { .. } => size_of(self) <= 16,
   127‚Üí   127‚Üí        }
   128‚Üí   128‚Üí    }
   129‚Üí   129‚Üí
   130‚Üí   130‚Üí    /// Returns `true` if this is a Homogeneous Floating-point Aggregate (HFA)
   131‚Üí   131‚Üí    /// ‚Äî a struct or array of 1‚Äì4 identical floating-point members.
   132‚Üí   132‚Üí    pub fn is_hfa(&self) -> bool {
   133‚Üí   133‚Üí        match self {
   134‚Üí   134‚Üí            IRType::Struct { fields, .. } => {
   135‚Üí   135‚Üí                if fields.is_empty() || fields.len() > 4 {
   136‚Üí   136‚Üí                    return false;
   137‚Üí   137‚Üí                }
   138‚Üí   138‚Üí                let first = &fields[0];
   139‚Üí   139‚Üí                if !first.is_float() {
   140‚Üí   140‚Üí                    return false;
   141‚Üí   141‚Üí                }
   142‚Üí   142‚Üí                fields.iter().all(|f| f == first)
   143‚Üí   143‚Üí            }
   144‚Üí   144‚Üí            IRType::Array { element, count } => {
   145‚Üí   145‚Üí                if *count == 0 || *count > 4 {
   146‚Üí   146‚Üí                    return false;
   147‚Üí   147‚Üí                }
   148‚Üí   148‚Üí                element.is_float()
   149‚Üí   149‚Üí            }
   150‚Üí   150‚Üí            _ => false,
   151‚Üí   151‚Üí        }
   152‚Üí   152‚Üí    }
   153‚Üí   153‚Üí
   154‚Üí   154‚Üí    /// If this is an HFA, returns the element type and count.
   155‚Üí   155‚Üí    pub fn hfa_info(&self) -> Option<(&IRType, usize)> {
   156‚Üí   156‚Üí        if !self.is_hfa() {
   157‚Üí   157‚Üí            return None;
   158‚Üí   158‚Üí        }
   159‚Üí   159‚Üí        match self {
   160‚Üí   160‚Üí            IRType::Struct { fields, .. } => Some((&fields[0], fields.len())),
   161‚Üí   161‚Üí            IRType::Array { element, count } => Some((element, *count)),
   162‚Üí   162‚Üí            _ => None,
   163‚Üí   163‚Üí        }
   164‚Üí   164‚Üí    }
   165‚Üí   165‚Üí}

   202‚Üí   202‚Üí/// Returns the byte size of `t` for a target with the given pointer width.
   203‚Üí   203‚Üí///
   204‚Üí   204‚Üí/// - Integers: their bit-width / 8
   205‚Üí   205‚Üí/// - Floats: 4 (f32) or 8 (f64)
   206‚Üí   206‚Üí/// - Pointers / function pointers: `ptr_width` (4 or 8)
   207‚Üí   207‚Üí/// - Void: 0
   208‚Üí   208‚Üí/// - Structs: sum of field sizes with inter-field padding for alignment,
   209‚Üí   209‚Üí///   rounded up to the struct's own alignment
   210‚Üí   210‚Üí/// - Arrays: `size_of(element) * count`
   211‚Üí   211‚Üípub fn size_of_with_ptr_width(t: &IRType, ptr_width: usize) -> usize {
   212‚Üí   212‚Üí    match t {
   213‚Üí   213‚Üí        IRType::I8 | IRType::U8 => 1,
   214‚Üí   214‚Üí        IRType::I16 | IRType::U16 => 2,
   215‚Üí   215‚Üí        IRType::I32 | IRType::U32 => 4,
   216‚Üí   216‚Üí        IRType::I64 | IRType::U64 => 8,
   217‚Üí   217‚Üí        IRType::Ptr | IRType::Func => ptr_width,
   218‚Üí   218‚Üí        IRType::F32 => 4,
   219‚Üí   219‚Üí        IRType::F64 => 8,
   220‚Üí   220‚Üí        IRType::Void => 0,
   221‚Üí   221‚Üí        IRType::Struct { fields, .. } => {
   222‚Üí   222‚Üí            let mut offset = 0usize;
   223‚Üí   223‚Üí            for field in fields {
   224‚Üí   224‚Üí                let field_align = alignment_of_with_ptr_width(field, ptr_width);
   225‚Üí   225‚Üí                // Align the current offset to the field alignment.
   226‚Üí   226‚Üí                offset = (offset + field_align - 1) & !(field_align - 1);
   227‚Üí   227‚Üí                offset += size_of_with_ptr_width(field, ptr_width);
   228‚Üí   228‚Üí            }
   229‚Üí   229‚Üí            // Round up to struct alignment.
   230‚Üí   230‚Üí            let struct_align = alignment_of_with_ptr_width(t, ptr_width);
   231‚Üí   231‚Üí            if struct_align > 0 {
   232‚Üí   232‚Üí                (offset + struct_align - 1) & !(struct_align - 1)
   233‚Üí   233‚Üí            } else {
   234‚Üí   234‚Üí                0
   235‚Üí   235‚Üí            }
   236‚Üí   236‚Üí        }
   237‚Üí   237‚Üí        IRType::Array { element, count } => size_of_with_ptr_width(element, ptr_width) * count,
   238‚Üí   238‚Üí    }
   239‚Üí   239‚Üí}
   240‚Üí   240‚Üí
   241‚Üí   241‚Üí/// Returns the natural alignment of `t` for a target with the given pointer
   242‚Üí   242‚Üí/// width.
   243‚Üí   243‚Üí///
   244‚Üí   244‚Üí/// - Primitives: their size
   245‚Üí   245‚Üí/// - Pointers / function pointers: `ptr_width`
   246‚Üí   246‚Üí/// - Void: 1 (by convention, so that `size_of` math works for empty structs)
   247‚Üí   247‚Üí/// - Structs: maximum alignment of any field
   248‚Üí   248‚Üí/// - Arrays: alignment of the element type
   249‚Üí   249‚Üípub fn alignment_of_with_ptr_width(t: &IRType, ptr_width: usize) -> usize {
   250‚Üí   250‚Üí    match t {
   251‚Üí   251‚Üí        IRType::I8 | IRType::U8 => 1,
   252‚Üí   252‚Üí        IRType::I16 | IRType::U16 => 2,
   253‚Üí   253‚Üí        IRType::I32 | IRType::U32 => 4,
   254‚Üí   254‚Üí        IRType::I64 | IRType::U64 => 8,
   255‚Üí   255‚Üí        IRType::Ptr | IRType::Func => ptr_width,
   256‚Üí   256‚Üí        IRType::F32 => 4,
   257‚Üí   257‚Üí        IRType::F64 => 8,
   258‚Üí   258‚Üí        IRType::Void => 1,
   259‚Üí   259‚Üí        IRType::Struct { fields, .. } => {
   260‚Üí   260‚Üí            if fields.is_empty() {
   261‚Üí   261‚Üí                return 1;
   262‚Üí   262‚Üí            }
   263‚Üí   263‚Üí            fields
   264‚Üí   264‚Üí                .iter()
   265‚Üí   265‚Üí                .map(|f| alignment_of_with_ptr_width(f, ptr_width))
   266‚Üí   266‚Üí                .max()
   267‚Üí   267‚Üí                .unwrap_or(1)
   268‚Üí   268‚Üí        }
   269‚Üí   269‚Üí        IRType::Array { element, .. } => alignment_of_with_ptr_width(element, ptr_width),
   270‚Üí   270‚Üí    }
   271‚Üí   271‚Üí}
   272‚Üí   272‚Üí
   273‚Üí   273‚Üí/// Returns the byte size of `t` assuming a 64-bit target (ARM64 LP64).
   274‚Üí   274‚Üí///
   275‚Üí   275‚Üí/// This is a convenience wrapper around [`size_of_with_ptr_width`] that
   276‚Üí   276‚Üí/// assumes 8-byte pointers.  For target-correct sizes, use
   277‚Üí   277‚Üí/// `TargetInfo::size_of` or [`size_of_with_ptr_width`] with the appropriate
   278‚Üí   278‚Üí/// pointer width.
   279‚Üí   279‚Üípub fn size_of(t: &IRType) -> usize {
   280‚Üí   280‚Üí    size_of_with_ptr_width(t, 8)
   281‚Üí   281‚Üí}
   282‚Üí   282‚Üí
   283‚Üí   283‚Üí/// Returns the natural alignment of `t` assuming a 64-bit target (ARM64 LP64).
   284‚Üí   284‚Üí///
   285‚Üí   285‚Üí/// This is a convenience wrapper around [`alignment_of_with_ptr_width`] that
   286‚Üí   286‚Üí/// assumes 8-byte pointers.  For target-correct alignments, use
   287‚Üí   287‚Üí/// `TargetInfo::alignment_of` or [`alignment_of_with_ptr_width`] with the
   288‚Üí   288‚Üí/// appropriate pointer width.
   289‚Üí   289‚Üípub fn alignment_of(t: &IRType) -> usize {
   290‚Üí   290‚Üí    alignment_of_with_ptr_width(t, 8)
   291‚Üí   291‚Üí}

   322‚Üí   322‚Üí/// Classifies a single argument or return-value type according to AAPCS64.
   323‚Üí   323‚Üí///
   324‚Üí   324‚Üí/// ## Rules (simplified)
   325‚Üí   325‚Üí///
   326‚Üí   326‚Üí/// - Integer types, pointers, and function pointers ‚Üí `Integer`
   327‚Üí   327‚Üí/// - Floating-point types ‚Üí `FP`
   328‚Üí   328‚Üí/// - Void ‚Üí `Integer` (only valid for return; caller should check)
   329‚Üí   329‚Üí/// - Homogeneous FP aggregates (1‚Äì4 same-type FP members) ‚Üí `FP`
   330‚Üí   330‚Üí/// - Structs / arrays ‚â§ 16 bytes (not HFA) ‚Üí `Integer`
   331‚Üí   331‚Üí/// - Structs / arrays > 16 bytes ‚Üí `Indirect`
   332‚Üí   332‚Üípub fn classify_arg(t: &IRType) -> ArgClass {
   333‚Üí   333‚Üí    match t {
   334‚Üí   334‚Üí        // Integer types ‚Üí Integer class
   335‚Üí   335‚Üí        IRType::I8
   336‚Üí   336‚Üí        | IRType::I16
   337‚Üí   337‚Üí        | IRType::I32
   338‚Üí   338‚Üí        | IRType::I64
   339‚Üí   339‚Üí        | IRType::U8
   340‚Üí   340‚Üí        | IRType::U16
   341‚Üí   341‚Üí        | IRType::U32
   342‚Üí   342‚Üí        | IRType::U64
   343‚Üí   343‚Üí        | IRType::Ptr
   344‚Üí   344‚Üí        | IRType::Func => ArgClass::Integer,
   345‚Üí   345‚Üí
   346‚Üí   346‚Üí        // Floating-point ‚Üí FP class
   347‚Üí   347‚Üí        IRType::F32 | IRType::F64 => ArgClass::FP,
   348‚Üí   348‚Üí
   349‚Üí   349‚Üí        // Void ‚Äî only meaningful as a return type; classify as Integer
   350‚Üí   350‚Üí        // (the caller should treat a void return as "no return value").
   351‚Üí   351‚Üí        IRType::Void => ArgClass::Integer,
   352‚Üí   352‚Üí
   353‚Üí   353‚Üí        // Struct / Array: check HFA first, then size.
   354‚Üí   354‚Üí        IRType::Struct { .. } | IRType::Array { .. } => {
   355‚Üí   355‚Üí            if t.is_hfa() {
   356‚Üí   356‚Üí                ArgClass::FP
   357‚Üí   357‚Üí            } else {
   358‚Üí   358‚Üí                let sz = size_of(t);
   359‚Üí   359‚Üí                if sz <= 16 {
   360‚Üí   360‚Üí                    ArgClass::Integer
   361‚Üí   361‚Üí                } else {
   362‚Üí   362‚Üí                    ArgClass::Indirect
   363‚Üí   363‚Üí                }
   364‚Üí   364‚Üí            }
   365‚Üí   365‚Üí        }
   366‚Üí   366‚Üí    }
   367‚Üí   367‚Üí}

   373‚Üí   373‚Üí/// Which kind of register an argument or return value occupies.

   375‚Üí   375‚Üípub enum RegisterClass {

   391‚Üí   391‚Üí/// Describes where a single argument or return value is placed.
   392‚Üí   392‚Üí#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   393‚Üí   393‚Üípub struct ArgLocation {
   394‚Üí   394‚Üí    /// Index of the argument (0-based) in the argument list.
   395‚Üí   395‚Üí    pub index: usize,
   396‚Üí   396‚Üí    /// AAPCS64 classification of this argument.
   397‚Üí   397‚Üí    pub class: ArgClass,
   398‚Üí   398‚Üí    /// Register class and index, if passed in a register.
   399‚Üí   399‚Üí    /// For `ArgClass::Integer` ‚Üí `(RegisterClass::X, n)` where n is 0‚Äì7.
   400‚Üí   400‚Üí    /// For `ArgClass::FP` ‚Üí `(RegisterClass::V, n)` where n is 0‚Äì7.
   401‚Üí   401‚Üí    /// `None` for stack or indirect arguments.
   402‚Üí   402‚Üí    pub register: Option<(RegisterClass, u32)>,
   403‚Üí   403‚Üí    /// Byte offset from SP (for stack arguments) or from the indirect
   404‚Üí   404‚Üí    /// pointer, if applicable.  `None` for register arguments.
   405‚Üí   405‚Üí    pub stack_offset: Option<i32>,
   406‚Üí   406‚Üí}

   408‚Üí   408‚Üí/// Describes the location of the return value.
   409‚Üí   409‚Üí#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   410‚Üí   410‚Üípub struct RetLocation {
   411‚Üí   411‚Üí    /// AAPCS64 classification of the return.
   412‚Üí   412‚Üí    pub class: ArgClass,
   413‚Üí   413‚Üí    /// Register(s) used for the return value.
   414‚Üí   414‚Üí    /// - Void: empty
   415‚Üí   415‚Üí    /// - Integer ‚â§ 8 bytes: `[(X, 0)]`
   416‚Üí   416‚Üí    /// - Integer 9‚Äì16 bytes (struct): `[(X, 0), (X, 1)]`
   417‚Üí   417‚Üí    /// - FP (HFA 1 member): `[(V, 0)]`, etc.
   418‚Üí   418‚Üí    /// - Indirect: `[(X, 8)]` ‚Äî X8 holds the caller-allocated address
   419‚Üí   419‚Üí    pub registers: Vec<(RegisterClass, u32)>,
   420‚Üí   420‚Üí}

   422‚Üí   422‚Üí/// Complete calling-convention information for a function signature.
   423‚Üí   423‚Üí#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   424‚Üí   424‚Üípub struct CallingConvInfo {
   425‚Üí   425‚Üí    /// Location for each argument.
   426‚Üí   426‚Üí    pub arg_locations: Vec<ArgLocation>,
   427‚Üí   427‚Üí    /// Location for the return value.
   428‚Üí   428‚Üí    pub ret_location: RetLocation,
   429‚Üí   429‚Üí    /// Total bytes of stack argument space that the *caller* must reserve
   430‚Üí   430‚Üí    /// for arguments that don't fit in registers.
   431‚Üí   431‚Üí    pub stack_args_size: usize,
   432‚Üí   432‚Üí}
   433‚Üí   433‚Üí
   434‚Üí   434‚Üí/// Computes calling-convention information for a function signature.
   435‚Üí   435‚Üí///
   436‚Üí   436‚Üí/// Walks the argument types, classifies each, and assigns registers (X0‚ÄìX7
   437‚Üí   437‚Üí/// for integer/pointer, V0‚ÄìV7 for FP) or stack slots.  Also classifies the
   438‚Üí   438‚Üí/// return value.
   439‚Üí   439‚Üí///
   440‚Üí   440‚Üí/// ## Return-value handling
   441‚Üí   441‚Üí///
   442‚Üí   442‚Üí/// - `void` ‚Üí no registers
   443‚Üí   443‚Üí/// - Integer/pointer ‚â§ 8 bytes ‚Üí X0
   444‚Üí   444‚Üí/// - Small struct (‚â§16 bytes, integer-like) ‚Üí X0, X1
   445‚Üí   445‚Üí/// - FP primitive ‚Üí V0
   446‚Üí   446‚Üí/// - HFA with N members ‚Üí V0..V(N-1)
   447‚Üí   447‚Üí/// - Large type (>16 bytes, not HFA) ‚Üí indirect via X8
   448‚Üí   448‚Üípub fn compute_calling_conv(args: &[IRType], ret: &IRType) -> CallingConvInfo {
   449‚Üí   449‚Üí    let mut x_reg_idx: u32 = 0;
   450‚Üí   450‚Üí    let mut v_reg_idx: u32 = 0;
   451‚Üí   451‚Üí    let mut stack_offset: i32 = 0;
   452‚Üí   452‚Üí    let mut arg_locations = Vec::with_capacity(args.len());
   453‚Üí   453‚Üí
   454‚Üí   454‚Üí    // If the return value is indirect, X8 is used for the return-area
   455‚Üí   455‚Üí    // pointer.  Per AAPCS64, this is passed as a hidden first argument in X8.
   456‚Üí   456‚Üí    // We model it by setting x_reg_idx = 1 so that the first explicit
   457‚Üí   457‚Üí    // argument starts at X0 (if not indirect) or X1 (since X8 is reserved).
   458‚Üí   458‚Üí    // Actually, AAPCS64 says X8 is *separate* ‚Äî it does not consume an
   459‚Üí   459‚Üí    // argument register slot.  The integer argument registers X0‚ÄìX7 are
   460‚Üí   460‚Üí    // independent of X8.  So we don't need to adjust x_reg_idx.
   461‚Üí   461‚Üí    let indirect_ret = classify_arg(ret) == ArgClass::Indirect;
   462‚Üí   462‚Üí
   463‚Üí   463‚Üí    for (i, arg_type) in args.iter().enumerate() {
   464‚Üí   464‚Üí        let class = classify_arg(arg_type);
   465‚Üí   465‚Üí        match class {
   466‚Üí   466‚Üí            ArgClass::Integer => {
   467‚Üí   467‚Üí                if x_reg_idx < 8 {
   468‚Üí   468‚Üí                    arg_locations.push(ArgLocation {
   469‚Üí   469‚Üí                        index: i,
   470‚Üí   470‚Üí                        class,
   471‚Üí   471‚Üí                        register: Some((RegisterClass::X, x_reg_idx)),
   472‚Üí   472‚Üí                        stack_offset: None,
   473‚Üí   473‚Üí                    });
   474‚Üí   474‚Üí                    x_reg_idx += 1;
   475‚Üí   475‚Üí                } else {
   476‚Üí   476‚Üí                    // Spill to stack ‚Äî each stack arg is 8-byte aligned.
   477‚Üí   477‚Üí                    arg_locations.push(ArgLocation {
   478‚Üí   478‚Üí                        index: i,
   479‚Üí   479‚Üí                        class: ArgClass::Stack,
   480‚Üí   480‚Üí                        register: None,
   481‚Üí   481‚Üí                        stack_offset: Some(stack_offset),
   482‚Üí   482‚Üí                    });
   483‚Üí   483‚Üí                    stack_offset += 8;
   484‚Üí   484‚Üí                }
   485‚Üí   485‚Üí            }
   486‚Üí   486‚Üí            ArgClass::FP => {
   487‚Üí   487‚Üí                // HFA: may consume multiple V registers.
   488‚Üí   488‚Üí                if let Some((elem_ty, count)) = arg_type.hfa_info() {
   489‚Üí   489‚Üí                    let elem_size = size_of(elem_ty);
   490‚Üí   490‚Üí                    if v_reg_idx + count as u32 <= 8 {
   491‚Üí   491‚Üí                        // HFA fits in V registers ‚Äî store the first register;
   492‚Üí   492‚Üí                        // the remaining are consecutive.
   493‚Üí   493‚Üí                        arg_locations.push(ArgLocation {
   494‚Üí   494‚Üí                            index: i,
   495‚Üí   495‚Üí                            class,
   496‚Üí   496‚Üí                            register: Some((RegisterClass::V, v_reg_idx)),
   497‚Üí   497‚Üí                            stack_offset: None,
   498‚Üí   498‚Üí                        });
   499‚Üí   499‚Üí                        v_reg_idx += count as u32;
   500‚Üí   500‚Üí                    } else {
   501‚Üí   501‚Üí                        // HFA spills to stack ‚Äî each member is naturally aligned.
   502‚Üí   502‚Üí                        let mut off = stack_offset;
   503‚Üí   503‚Üí                        for _ in 0..count {
   504‚Üí   504‚Üí                            let align = alignment_of(elem_ty) as i32;
   505‚Üí   505‚Üí                            off = (off + align - 1) & !(align - 1);
   506‚Üí   506‚Üí                            off += elem_size as i32;
   507‚Üí   507‚Üí                        }
   508‚Üí   508‚Üí                        arg_locations.push(ArgLocation {
   509‚Üí   509‚Üí                            index: i,
   510‚Üí   510‚Üí                            class: ArgClass::Stack,
   511‚Üí   511‚Üí                            register: None,
   512‚Üí   512‚Üí                            stack_offset: Some(stack_offset),
   513‚Üí   513‚Üí                        });
   514‚Üí   514‚Üí                        stack_offset = off;
   515‚Üí   515‚Üí                    }
   516‚Üí   516‚Üí                } else {
   517‚Üí   517‚Üí                    // Single FP value.
   518‚Üí   518‚Üí                    if v_reg_idx < 8 {
   519‚Üí   519‚Üí                        arg_locations.push(ArgLocation {
   520‚Üí   520‚Üí                            index: i,
   521‚Üí   521‚Üí                            class,
   522‚Üí   522‚Üí                            register: Some((RegisterClass::V, v_reg_idx)),
   523‚Üí   523‚Üí                            stack_offset: None,
   524‚Üí   524‚Üí                        });
   525‚Üí   525‚Üí                        v_reg_idx += 1;
   526‚Üí   526‚Üí                    } else {
   527‚Üí   527‚Üí                        let align = alignment_of(arg_type) as i32;
   528‚Üí   528‚Üí                        stack_offset = (stack_offset + align - 1) & !(align - 1);
   529‚Üí   529‚Üí                        arg_locations.push(ArgLocation {
   530‚Üí   530‚Üí                            index: i,
   531‚Üí   531‚Üí                            class: ArgClass::Stack,
   532‚Üí   532‚Üí                            register: None,
   533‚Üí   533‚Üí                            stack_offset: Some(stack_offset),
   534‚Üí   534‚Üí                        });
   535‚Üí   535‚Üí                        stack_offset += size_of(arg_type) as i32;
   536‚Üí   536‚Üí                    }
   537‚Üí   537‚Üí                }
   538‚Üí   538‚Üí            }
   539‚Üí   539‚Üí            ArgClass::Indirect => {
   540‚Üí   540‚Üí                // Pass a pointer to the value in an X register (or on stack).
   541‚Üí   541‚Üí                if x_reg_idx < 8 {
   542‚Üí   542‚Üí                    arg_locations.push(ArgLocation {
   543‚Üí   543‚Üí                        index: i,
   544‚Üí   544‚Üí                        class,
   545‚Üí   545‚Üí                        register: Some((RegisterClass::X, x_reg_idx)),
   546‚Üí   546‚Üí                        stack_offset: None,
   547‚Üí   547‚Üí                    });
   548‚Üí   548‚Üí                    x_reg_idx += 1;
   549‚Üí   549‚Üí                } else {
   550‚Üí   550‚Üí                    arg_locations.push(ArgLocation {
   551‚Üí   551‚Üí                        index: i,
   552‚Üí   552‚Üí                        class: ArgClass::Stack,
   553‚Üí   553‚Üí                        register: None,
   554‚Üí   554‚Üí                        stack_offset: Some(stack_offset),
   555‚Üí   555‚Üí                    });
   556‚Üí   556‚Üí                    stack_offset += 8;
   557‚Üí   557‚Üí                }
   558‚Üí   558‚Üí            }
   559‚Üí   559‚Üí            ArgClass::Stack => {
   560‚Üí   560‚Üí                let align = alignment_of(arg_type) as i32;
   561‚Üí   561‚Üí                stack_offset = (stack_offset + align - 1) & !(align - 1);
   562‚Üí   562‚Üí                arg_locations.push(ArgLocation {
   563‚Üí   563‚Üí                    index: i,
   564‚Üí   564‚Üí                    class,
   565‚Üí   565‚Üí                    register: None,
   566‚Üí   566‚Üí                    stack_offset: Some(stack_offset),
   567‚Üí   567‚Üí                });
   568‚Üí   568‚Üí                stack_offset += size_of(arg_type) as i32;
   569‚Üí   569‚Üí            }
   570‚Üí   570‚Üí        }
   571‚Üí   571‚Üí    }
   572‚Üí   572‚Üí
   573‚Üí   573‚Üí    // Align stack args size to 16 bytes (AAPCS64 requirement).
   574‚Üí   574‚Üí    let stack_args_size = ((stack_offset as usize) + 15) & !15;
   575‚Üí   575‚Üí
   576‚Üí   576‚Üí    // Compute return-value location.
   577‚Üí   577‚Üí    let ret_class = classify_arg(ret);
   578‚Üí   578‚Üí    let ret_location = match ret_class {
   579‚Üí   579‚Üí        ArgClass::Integer => {
   580‚Üí   580‚Üí            if *ret == IRType::Void {
   581‚Üí   581‚Üí                RetLocation {
   582‚Üí   582‚Üí                    class: ret_class,
   583‚Üí   583‚Üí                    registers: vec![],
   584‚Üí   584‚Üí                }
   585‚Üí   585‚Üí            } else {
   586‚Üí   586‚Üí                let sz = size_of(ret);
   587‚Üí   587‚Üí                if sz <= 8 {
   588‚Üí   588‚Üí                    RetLocation {
   589‚Üí   589‚Üí                        class: ret_class,
   590‚Üí   590‚Üí                        registers: vec![(RegisterClass::X, 0)],
   591‚Üí   591‚Üí                    }
   592‚Üí   592‚Üí                } else {
   593‚Üí   593‚Üí                    // Struct ‚â§ 16 bytes returned in X0 + X1.
   594‚Üí   594‚Üí                    RetLocation {
   595‚Üí   595‚Üí                        class: ret_class,
   596‚Üí   596‚Üí                        registers: vec![(RegisterClass::X, 0), (RegisterClass::X, 1)],
   597‚Üí   597‚Üí                    }
   598‚Üí   598‚Üí                }
   599‚Üí   599‚Üí            }
   600‚Üí   600‚Üí        }
   601‚Üí   601‚Üí        ArgClass::FP => {
   602‚Üí   602‚Üí            if let Some((_, count)) = ret.hfa_info() {
   603‚Üí   603‚Üí                let regs: Vec<(RegisterClass, u32)> =
   604‚Üí   604‚Üí                    (0..count as u32).map(|i| (RegisterClass::V, i)).collect();
   605‚Üí   605‚Üí                RetLocation {
   606‚Üí   606‚Üí                    class: ret_class,
   607‚Üí   607‚Üí                    registers: regs,
   608‚Üí   608‚Üí                }
   609‚Üí   609‚Üí            } else {
   610‚Üí   610‚Üí                RetLocation {
   611‚Üí   611‚Üí                    class: ret_class,
   612‚Üí   612‚Üí                    registers: vec![(RegisterClass::V, 0)],
   613‚Üí   613‚Üí                }
   614‚Üí   614‚Üí            }
   615‚Üí   615‚Üí        }
   616‚Üí   616‚Üí        ArgClass::Indirect => RetLocation {
   617‚Üí   617‚Üí            class: ret_class,
   618‚Üí   618‚Üí            registers: vec![(RegisterClass::X, 8)],
   619‚Üí   619‚Üí        },
   620‚Üí   620‚Üí        ArgClass::Stack => {
   621‚Üí   621‚Üí            // Stack return is unusual; model as indirect for now.
   622‚Üí   622‚Üí            RetLocation {
   623‚Üí   623‚Üí                class: ArgClass::Indirect,
   624‚Üí   624‚Üí                registers: vec![(RegisterClass::X, 8)],
   625‚Üí   625‚Üí            }
   626‚Üí   626‚Üí        }
   627‚Üí   627‚Üí    };
   628‚Üí   628‚Üí
   629‚Üí   629‚Üí    let _ = indirect_ret; // X8 reservation is implicit via ret_location.
   630‚Üí   630‚Üí
   631‚Üí   631‚Üí    CallingConvInfo {
   632‚Üí   632‚Üí        arg_locations,
   633‚Üí   633‚Üí        ret_location,
   634‚Üí   634‚Üí        stack_args_size,
   635‚Üí   635‚Üí    }
   636‚Üí   636‚Üí}

   640‚Üí   640‚Üí// ---------------------------------------------------------------------------
   641‚Üí   641‚Üí
   642‚Üí   642‚Üí/// A named slot in the stack frame.
   643‚Üí   643‚Üí#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   644‚Üí   644‚Üípub struct StackSlot {

   655‚Üí   655‚Üí}

   657‚Üí   657‚Üí/// Complete stack-frame layout for a function.
   658‚Üí   658‚Üí///
   659‚Üí   659‚Üí/// The stack frame layout follows the AAPCS64 convention:
   660‚Üí   660‚Üí///
   661‚Üí   661‚Üí/// ```text
   662‚Üí   662‚Üí/// Higher addresses
   663‚Üí   663‚Üí///   ‚îå‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îê
   664‚Üí   664‚Üí///   ‚îÇ Incoming stack args  ‚îÇ  FP+16, FP+24, ‚Ä¶
   665‚Üí   665‚Üí///   ‚îú‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚î§
   666‚Üí   666‚Üí///   ‚îÇ Saved FP (X29)       ‚îÇ  FP+0
   667‚Üí   667‚Üí///   ‚îÇ Saved LR (X30)       ‚îÇ  FP+8
   668‚Üí   668‚Üí///   ‚îú‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚î§
   669‚Üí   669‚Üí///   ‚îÇ Callee-saved regs    ‚îÇ  FP-8, FP-16, ‚Ä¶
   670‚Üí   670‚Üí///   ‚îÇ (X19..X28)           ‚îÇ
   671‚Üí   671‚Üí///   ‚îú‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚î§
   672‚Üí   672‚Üí///   ‚îÇ Local variables      ‚îÇ  from Alloc instructions
   673‚Üí   673‚Üí///   ‚îÇ (aligned)            ‚îÇ
   674‚Üí   674‚Üí///   ‚îú‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚î§
   675‚Üí   675‚Üí///   ‚îÇ Outgoing stack args  ‚îÇ  bottom of frame (for calls)
   676‚Üí   676‚Üí///   ‚îÇ (for nested calls)   ‚îÇ
   677‚Üí   677‚Üí///   ‚îî‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îò  ‚Üê SP
   678‚Üí   678‚Üí/// Lower addresses
   679‚Üí   679‚Üí/// ```
   680‚Üí   680‚Üí#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   681‚Üí   681‚Üípub struct StackLayout {
   682‚Üí   682‚Üí    /// Total stack frame size in bytes (always 16-byte aligned).
   683‚Üí   683‚Üí    pub total_size: usize,
   684‚Üí   684‚Üí    /// Slots for callee-saved registers.
   685‚Üí   685‚Üí    pub callee_save_slots: Vec<StackSlot>,
   686‚Üí   686‚Üí    /// Slots for local variables (from `Alloc` instructions).
   687‚Üí   687‚Üí    pub local_slots: Vec<StackSlot>,
   688‚Üí   688‚Üí    /// Slot for outgoing stack arguments to callees (single contiguous area).
   689‚Üí   689‚Üí    pub outgoing_args_slot: Option<StackSlot>,
   690‚Üí   690‚Üí    /// Slot for the saved FP (X29) ‚Äî always at FP+0.
   691‚Üí   691‚Üí    pub fp_slot: StackSlot,
   692‚Üí   692‚Üí    /// Slot for the saved LR (X30) ‚Äî always at FP+8.
   693‚Üí   693‚Üí    pub lr_slot: StackSlot,
   694‚Üí   694‚Üí    /// Number of callee-saved registers that are spilled.
   695‚Üí   695‚Üí    pub callee_saves_count: usize,
   696‚Üí   696‚Üí}

   698‚Üí   698‚Üí/// Computes the stack-frame layout for an IR function.
   699‚Üí   699‚Üí///
   700‚Üí   700‚Üí/// This scans the function body for `Alloc` instructions (local variables) and
   701‚Üí   701‚Üí/// `Call` instructions (outgoing stack arguments), and lays out the frame
   702‚Üí   702‚Üí/// according to AAPCS64 conventions.
   703‚Üí   703‚Üí///
   704‚Üí   704‚Üí/// ## Parameters
   705‚Üí   705‚Üí///
   706‚Üí   706‚Üí/// - `func`: The IR function to compute the layout for.
   707‚Üí   707‚Üí/// - `callee_saves_count`: How many callee-saved registers (X19‚ÄìX28) the
   708‚Üí   708‚Üí///   function uses.  Set by the register allocator.  Each consumes 8 bytes.
   709‚Üí   709‚Üí/// - `call_arg_types`: For each `Call` instruction in the function, the
   710‚Üí   710‚Üí///   argument types.  This is used to compute the outgoing stack-argument
   711‚Üí   711‚Üí///   area size.  If empty or if calls have no stack args, no outgoing area
   712‚Üí   712‚Üí///   is needed.
   713‚Üí   713‚Üípub fn compute_stack_layout(func: &IRFunction) -> StackLayout {
   714‚Üí   714‚Üí    compute_stack_layout_with_info(func, 0, &[])
   715‚Üí   715‚Üí}
   716‚Üí   716‚Üí
   717‚Üí   717‚Üí/// Computes the stack-frame layout with additional information from the
   718‚Üí   718‚Üí/// register allocator and call-site type information.
   719‚Üí   719‚Üí///
   720‚Üí   720‚Üí/// See [`compute_stack_layout`] for the simplified version.
   721‚Üí   721‚Üípub fn compute_stack_layout_with_info(
   722‚Üí   722‚Üí    func: &IRFunction,
   723‚Üí   723‚Üí    callee_saves_count: usize,
   724‚Üí   724‚Üí    call_arg_types: &[Vec<IRType>],
   725‚Üí   725‚Üí) -> StackLayout {
   726‚Üí   726‚Üí    // Fixed slots: saved FP at FP+0, saved LR at FP+8.
   727‚Üí   727‚Üí    let fp_slot = StackSlot {
   728‚Üí   728‚Üí        name: "saved_fp".to_string(),
   729‚Üí   729‚Üí        offset: 0,
   730‚Üí   730‚Üí        size: 8,
   731‚Üí   731‚Üí        alignment: 8,
   732‚Üí   732‚Üí    };
   733‚Üí   733‚Üí    let lr_slot = StackSlot {
   734‚Üí   734‚Üí        name: "saved_lr".to_string(),
   735‚Üí   735‚Üí        offset: 8,
   736‚Üí   736‚Üí        size: 8,
   737‚Üí   737‚Üí        alignment: 8,
   738‚Üí   738‚Üí    };
   739‚Üí   739‚Üí
   740‚Üí   740‚Üí    // Current negative offset from FP, growing downward.
   741‚Üí   741‚Üí    let mut offset: i32 = 0;
   742‚Üí   742‚Üí
   743‚Üí   743‚Üí    // --- Callee-saved registers ---
   744‚Üí   744‚Üí    let mut callee_save_slots = Vec::new();
   745‚Üí   745‚Üí    for i in 0..callee_saves_count {
   746‚Üí   746‚Üí        offset -= 8;
   747‚Üí   747‚Üí        callee_save_slots.push(StackSlot {
   748‚Üí   748‚Üí            name: format!("callee_save_{}", i),
   749‚Üí   749‚Üí            offset,
   750‚Üí   750‚Üí            size: 8,
   751‚Üí   751‚Üí            alignment: 8,
   752‚Üí   752‚Üí        });
   753‚Üí   753‚Üí    }
   754‚Üí   754‚Üí
   755‚Üí   755‚Üí    // --- Local variables (from Alloc instructions) ---
   756‚Üí   756‚Üí    let mut local_slots = Vec::new();
   757‚Üí   757‚Üí    let mut alloc_index = 0;
   758‚Üí   758‚Üí    for block in &func.blocks {
   759‚Üí   759‚Üí        for instr in &block.instructions {
   760‚Üí   760‚Üí            if let IRInstr::Alloc { dst, size } = instr {
   761‚Üí   761‚Üí                let size = *size as usize;
   762‚Üí   762‚Üí                let align = if size >= 8 {
   763‚Üí   763‚Üí                    8usize
   764‚Üí   764‚Üí                } else {
   765‚Üí   765‚Üí                    size.next_power_of_two()
   766‚Üí   766‚Üí                };
   767‚Üí   767‚Üí                // Align offset.
   768‚Üí   768‚Üí                offset = (offset - align as i32) & !(align as i32 - 1);
   769‚Üí   769‚Üí                offset -= size as i32;
   770‚Üí   770‚Üí                // Re-align offset to the start of the slot.
   771‚Üí   771‚Üí                offset &= !(align as i32 - 1);
   772‚Üí   772‚Üí                let name = match dst {
   773‚Üí   773‚Üí                    IRValue::Register(id) => format!("local_%v{}", id),
   774‚Üí   774‚Üí                    _ => format!("local_{}", alloc_index),
   775‚Üí   775‚Üí                };
   776‚Üí   776‚Üí                local_slots.push(StackSlot {
   777‚Üí   777‚Üí                    name,
   778‚Üí   778‚Üí                    offset,
   779‚Üí   779‚Üí                    size,
   780‚Üí   780‚Üí                    alignment: align,
   781‚Üí   781‚Üí                });
   782‚Üí   782‚Üí                alloc_index += 1;
   783‚Üí   783‚Üí            }
   784‚Üí   784‚Üí        }
   785‚Üí   785‚Üí    }
   786‚Üí   786‚Üí
   787‚Üí   787‚Üí    // --- Outgoing stack arguments ---
   788‚Üí   788‚Üí    // Compute the maximum stack-args size across all call sites.
   789‚Üí   789‚Üí    let mut max_outgoing_stack = 0usize;
   790‚Üí   790‚Üí    for arg_types in call_arg_types {
   791‚Üí   791‚Üí        let cc = compute_calling_conv(arg_types, &IRType::Void);
   792‚Üí   792‚Üí        max_outgoing_stack = max_outgoing_stack.max(cc.stack_args_size);
   793‚Üí   793‚Üí    }
   794‚Üí   794‚Üí
   795‚Üí   795‚Üí    let outgoing_args_slot = if max_outgoing_stack > 0 {
   796‚Üí   796‚Üí        // Align to 16 bytes.
   797‚Üí   797‚Üí        let size = (max_outgoing_stack + 15) & !15;
   798‚Üí   798‚Üí        offset -= size as i32;
   799‚Üí   799‚Üí        Some(StackSlot {
   800‚Üí   800‚Üí            name: "outgoing_args".to_string(),
   801‚Üí   801‚Üí            offset,
   802‚Üí   802‚Üí            size,
   803‚Üí   803‚Üí            alignment: 16,
   804‚Üí   804‚Üí        })
   805‚Üí   805‚Üí    } else {
   806‚Üí   806‚Üí        None
   807‚Üí   807‚Üí    };
   808‚Üí   808‚Üí
   809‚Üí   809‚Üí    // Total frame size: absolute value of the lowest offset, plus 16 for
   810‚Üí   810‚Üí    // FP/LR pair.  The frame pointer is at the saved-FP position; everything
   811‚Üí   811‚Üí    // below is at negative offsets from FP.
   812‚Üí   812‚Üí    let total_raw = ((-offset) as usize) + 16; // 16 = FP + LR
   813‚Üí   813‚Üí    let total_size = (total_raw + 15) & !15; // 16-byte align.
   814‚Üí   814‚Üí
   815‚Üí   815‚Üí    StackLayout {
   816‚Üí   816‚Üí        total_size,
   817‚Üí   817‚Üí        callee_save_slots,
   818‚Üí   818‚Üí        local_slots,
   819‚Üí   819‚Üí        outgoing_args_slot,
   820‚Üí   820‚Üí        fp_slot,
   821‚Üí   821‚Üí        lr_slot,
   822‚Üí   822‚Üí        callee_saves_count,
   823‚Üí   823‚Üí    }
   824‚Üí   824‚Üí}

  1418‚Üí  1418‚Üí    /// Returns the set of virtual-register IDs that this instruction uses
  1419‚Üí  1419‚Üí    /// (reads from).
  1420‚Üí  1420‚Üí    pub fn used_regs(&self) -> Vec<u32> {
  1421‚Üí  1421‚Üí        match self {
  1422‚Üí  1422‚Üí            IRInstr::Load { addr, .. } => addr.as_register().into_iter().collect(),
  1423‚Üí  1423‚Üí            IRInstr::Store { value, addr, .. } => {
  1424‚Üí  1424‚Üí                let mut r = value.as_register().into_iter().collect::<Vec<_>>();

  1427‚Üí  1427‚Üí            }
  1428‚Üí  1428‚Üí            IRInstr::BinOp { lhs, rhs, .. }
  1429‚Üí  1429‚Üí            | IRInstr::Add { lhs, rhs, .. }
  1430‚Üí  1430‚Üí            | IRInstr::Sub { lhs, rhs, .. }
  1431‚Üí  1431‚Üí            | IRInstr::Mul { lhs, rhs, .. }

  1438‚Üí  1438‚Üí            IRInstr::UnaryOp { operand, .. } => operand.as_register().into_iter().collect(),

  1444‚Üí  1444‚Üí                .iter()
  1445‚Üí  1445‚Üí                .filter_map(|(v, _)| v.as_register())
  1446‚Üí  1446‚Üí                .collect(),
  1447‚Üí  1447‚Üí            IRInstr::Offset { base, offset, .. } => {
  1448‚Üí  1448‚Üí                let mut r = base.as_register().into_iter().collect::<Vec<_>>();

  1455‚Üí  1455‚Üí                false_val, ty: _,

  1457‚Üí  1457‚Üí            } => {
  1458‚Üí  1458‚Üí                let mut r = cond.as_register().into_iter().collect::<Vec<_>>();

  1460‚Üí  1460‚Üí                r.extend(false_val.as_register());

  1462‚Üí  1462‚Üí            }

  1467‚Üí  1467‚Üí    }
