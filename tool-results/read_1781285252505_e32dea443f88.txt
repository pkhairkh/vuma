     1тЖТ//! # x86_64 Backend
     2тЖТ//!
     3тЖТ//! Implements the `Backend` trait for the x86_64 target (SystemV ABI).
     4тЖТ//! This module provides:
     5тЖТ//!
     6тЖТ//! - `Gpr` тАФ General-purpose register enum (RAXтАУR15)
     7тЖТ//! - `Xmm` тАФ SSE/SIMD register enum (XMM0тАУXMM15)
     8тЖТ//! - REX prefix generation
     9тЖТ//! - ModR/M + SIB byte encoding
    10тЖТ//! - Instruction encoding for all key x86_64 instructions
    11тЖТ//! - `X86_64Backend` тАФ `Backend` implementation that lowers IR to x86_64 machine code
    12тЖТ//!
    13тЖТ//! ## x86_64 Register Convention (SystemV ABI)
    14тЖТ//!
    15тЖТ//! | Register | Role                    | Callee-saved |
    16тЖТ//! |----------|-------------------------|-------------|
    17тЖТ//! | RAX      | Return value / scratch  | No          |
    18тЖТ//! | RCX      | 4th int arg / scratch   | No          |
    19тЖТ//! | RDX      | 3rd int arg / scratch   | No          |
    20тЖТ//! | RBX      | Callee-saved            | Yes         |
    21тЖТ//! | RSP      | Stack pointer           | (special)   |
    22тЖТ//! | RBP      | Frame pointer           | Yes         |
    23тЖТ//! | RSI      | 2nd int arg / scratch   | No          |
    24тЖТ//! | RDI      | 1st int arg / scratch   | No          |
    25тЖТ//! | R8тАУR9    | 5th/6th int arg         | No          |
    26тЖТ//! | R10тАУR11  | Scratch                 | No          |
    27тЖТ//! | R12тАУR15  | Callee-saved            | Yes         |
    28тЖТ//!
    29тЖТ//! ## References
    30тЖТ//!
    31тЖТ//! - Intel 64 and IA-32 Architectures Software Developer's Manual, Volumes 2A/2B
    32тЖТ//! - System V Application Binary Interface, AMD64 Architecture Processor Supplement

   372тЖТ// REX Prefix
   373тЖТ// ===========================================================================
   374тЖТ
   375тЖТ/// Generate a REX prefix byte.
   376тЖТ///
   377тЖТ/// - `w`: REX.W тАФ 64-bit operand size
   378тЖТ/// - `r`: Extension of the ModR/M `reg` field (for R8тАУR15 / XMM8тАУXMM15)
   379тЖТ/// - `x`: Extension of the SIB `index` field
   380тЖТ/// - `b`: Extension of the ModR/M `rm` field or SIB `base` field
   381тЖТ///
   382тЖТ/// Returns `None` if no REX byte is needed (all bits are 0).
   383тЖТfn rex_prefix(w: bool, r: bool, x: bool, b: bool) -> Option<u8> {
   384тЖТ    let byte = 0x40 | (w as u8) << 3 | (r as u8) << 2 | (x as u8) << 1 | b as u8;
   385тЖТ    if byte > 0x40 {
   386тЖТ        Some(byte)
   387тЖТ    } else {
   388тЖТ        None
   389тЖТ    }
   390тЖТ}
   391тЖТ
   392тЖТ// ===========================================================================
   393тЖТ// ModR/M + SIB Encoding
   394тЖТ// ===========================================================================
   395тЖТ
   396тЖТ/// Encode a ModR/M byte.
   397тЖТ///
   398тЖТ/// - `mod_bits`: 2-bit mod field (0=mem, 1=mem+disp8, 2=mem+disp32, 3=reg)
   399тЖТ/// - `reg`: 3-bit reg field (register or opcode extension)
   400тЖТ/// - `rm`: 3-bit r/m field (register or memory operand)
   401тЖТfn modrm(mod_bits: u8, reg: u8, rm: u8) -> u8 {
   402тЖТ    (mod_bits & 3) << 6 | (reg & 7) << 3 | (rm & 7)
   403тЖТ}
   404тЖТ
   405тЖТ/// Encode a SIB byte.
   406тЖТ///
   407тЖТ/// - `scale`: 2-bit scale factor (0=1, 1=2, 2=4, 3=8)
   408тЖТ/// - `index`: 3-bit index register
   409тЖТ/// - `base`: 3-bit base register
   410тЖТfn sib(scale: u8, index: u8, base: u8) -> u8 {
   411тЖТ    (scale & 3) << 6 | (index & 7) << 3 | (base & 7)
   412тЖТ}

   418тЖТ/// Emit a REX.W prefix plus opcode, then a ModR/M byte for reg-reg operations.
   419тЖТ///
   420тЖТ/// This is the common pattern for 64-bit ALU instructions: REX.W + opcode + ModR/M(mod=3, reg, rm).
   421тЖТfn emit_rexw_reg_reg(code: &mut Vec<u8>, opcode: u8, reg: Gpr, rm: Gpr) {
   422тЖТ    let r = reg.needs_rex();
   423тЖТ    let b = rm.needs_rex();
   424тЖТ    if let Some(rex) = rex_prefix(true, r, false, b) {
   425тЖТ        code.push(rex);
   426тЖТ    } else {
   427тЖТ        // REX.W is still needed for 64-bit operations even if r and b are 0
   428тЖТ        code.push(0x48);
   429тЖТ    }
   430тЖТ    code.push(opcode);
   431тЖТ    code.push(modrm(3, reg.encoding() & 7, rm.encoding() & 7));
   432тЖТ}
   433тЖТ
   434тЖТ/// Emit a REX.W prefix (always), then opcode, then ModR/M for reg-reg with
   435тЖТ/// specific reg field (opcode extension) and rm register.
   436тЖТfn emit_rexw_opext_reg(code: &mut Vec<u8>, opcode: u8, opext: u8, rm: Gpr) {
   437тЖТ    let b = rm.needs_rex();
   438тЖТ    // Always emit REX.W for 64-bit
   439тЖТ    let rex = 0x48 | (b as u8);
   440тЖТ    code.push(rex);
   441тЖТ    code.push(opcode);
   442тЖТ    code.push(modrm(3, opext & 7, rm.encoding() & 7));
   443тЖТ}

   481тЖТ/// Encode MOV r64, [r64+offset] (REX.W + 8B /r + displacement)
   482тЖТ///
   483тЖТ/// Handles special cases:
   484тЖТ/// - RSP/R12 as base: SIB byte required
   485тЖТ/// - RBP/R13 as base: mod=1 with disp8=0 even for zero offset
   486тЖТ/// - Offset fits in i8: disp8; otherwise: disp32
   487тЖТpub fn encode_mov_reg_mem(dst: Gpr, base: Gpr, offset: i32) -> Vec<u8> {
   488тЖТ    let mut code = Vec::with_capacity(8);
   489тЖТ    let r = dst.needs_rex();
   490тЖТ    let b = base.needs_rex();
   491тЖТ
   492тЖТ    if let Some(rex) = rex_prefix(true, r, false, b) {
   493тЖТ        code.push(rex);
   494тЖТ    } else {
   495тЖТ        code.push(0x48);
   496тЖТ    }
   497тЖТ    code.push(0x8B);
   498тЖТ
   499тЖТ    let needs_sib = base == Gpr::Rsp || base == Gpr::R12;
   500тЖТ    let needs_disp8_for_zero = base == Gpr::Rbp || base == Gpr::R13;
   501тЖТ
   502тЖТ    if offset == 0 && !needs_disp8_for_zero && !needs_sib {
   503тЖТ        // mod=00, no displacement
   504тЖТ        code.push(modrm(0, dst.encoding() & 7, base.encoding() & 7));
   505тЖТ    } else if needs_sib {
   506тЖТ        // SIB byte required: base = RSP(4), index = RSP(4) means "no index"
   507тЖТ        if offset == 0 {
   508тЖТ            code.push(modrm(0, dst.encoding() & 7, 4));
   509тЖТ            code.push(sib(0, 4, base.encoding() & 7));
   510тЖТ        } else if (-128..=127).contains(&offset) {
   511тЖТ            code.push(modrm(1, dst.encoding() & 7, 4));
   512тЖТ            code.push(sib(0, 4, base.encoding() & 7));
   513тЖТ            code.push(offset as u8);
   514тЖТ        } else {
   515тЖТ            code.push(modrm(2, dst.encoding() & 7, 4));
   516тЖТ            code.push(sib(0, 4, base.encoding() & 7));
   517тЖТ            code.extend_from_slice(&offset.to_le_bytes());
   518тЖТ        }
   519тЖТ    } else if (-128..=127).contains(&offset) {
   520тЖТ        // mod=01, disp8
   521тЖТ        code.push(modrm(1, dst.encoding() & 7, base.encoding() & 7));
   522тЖТ        code.push(offset as u8);
   523тЖТ    } else {
   524тЖТ        // mod=10, disp32
   525тЖТ        code.push(modrm(2, dst.encoding() & 7, base.encoding() & 7));
   526тЖТ        code.extend_from_slice(&offset.to_le_bytes());
   527тЖТ    }
   528тЖТ
   529тЖТ    code
   530тЖТ}

  1041тЖТ/// Encode a memory operand (ModR/M + optional SIB + displacement) for [base + offset].
  1042тЖТ/// Appends the ModR/M byte, SIB byte (if needed), and displacement to `code`.
  1043тЖТfn encode_mem_operand(code: &mut Vec<u8>, reg: u8, base: Gpr, offset: i32) {
  1044тЖТ    let needs_sib = base == Gpr::Rsp || base == Gpr::R12;
  1045тЖТ    let needs_disp8_for_zero = base == Gpr::Rbp || base == Gpr::R13;
  1046тЖТ
  1047тЖТ    if offset == 0 && !needs_disp8_for_zero && !needs_sib {
  1048тЖТ        // mod=00, no displacement
  1049тЖТ        code.push(modrm(0, reg, base.encoding() & 7));
  1050тЖТ    } else if needs_sib {
  1051тЖТ        // SIB byte required: base = RSP(4), index = RSP(4) means "no index"
  1052тЖТ        if offset == 0 {
  1053тЖТ            code.push(modrm(0, reg, 4));
  1054тЖТ            code.push(sib(0, 4, base.encoding() & 7));
  1055тЖТ        } else if (-128..=127).contains(&offset) {
  1056тЖТ            code.push(modrm(1, reg, 4));
  1057тЖТ            code.push(sib(0, 4, base.encoding() & 7));
  1058тЖТ            code.push(offset as u8);
  1059тЖТ        } else {
  1060тЖТ            code.push(modrm(2, reg, 4));
  1061тЖТ            code.push(sib(0, 4, base.encoding() & 7));
  1062тЖТ            code.extend_from_slice(&offset.to_le_bytes());
  1063тЖТ        }
  1064тЖТ    } else if (-128..=127).contains(&offset) {
  1065тЖТ        // mod=01, disp8
  1066тЖТ        code.push(modrm(1, reg, base.encoding() & 7));
  1067тЖТ        code.push(offset as u8);
  1068тЖТ    } else {
  1069тЖТ        // mod=10, disp32
  1070тЖТ        code.push(modrm(2, reg, base.encoding() & 7));
  1071тЖТ        code.extend_from_slice(&offset.to_le_bytes());
  1072тЖТ    }
  1073тЖТ}

  1076тЖТpub fn encode_movzx_reg8_mem(dst: Gpr, base: Gpr, offset: i32) -> Vec<u8> {

  1209тЖТ/// Decode x86_64 bytes into mnemonic strings with (offset, mnemonic) pairs.
  1210тЖТ///
  1211тЖТ/// Handles the top 20+ most common x86_64 instructions including mov, add, sub,
  1212тЖТ/// push, pop, call, ret, jmp, cmp, test, lea, xor, and, or, shl, shr, nop,
  1213тЖТ/// mul, div, imul.
