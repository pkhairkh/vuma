     1вҶ’     1вҶ’//! Structured JSON diagnostics output for LLM integration.
     2вҶ’     2вҶ’//!
     3вҶ’     3вҶ’//! This module provides machine-readable diagnostic messages that can be

     9вҶ’     9вҶ’//! Each diagnostic carries a structured error code (`"E001"`, `"W001"`, etc.),
    10вҶ’    10вҶ’//! a severity level, a human-readable message, a source identifier (which
    11вҶ’    11вҶ’//! compiler stage produced it), a precise source location, optional related
    12вҶ’    12вҶ’//! information, and quick-fix suggestions.
    13вҶ’    13вҶ’//!
    14вҶ’    14вҶ’//! # Example JSON output
    15вҶ’    15вҶ’//!
    16вҶ’    16вҶ’//! ```json
    17вҶ’    17вҶ’//! [

    21вҶ’    21вҶ’//!     "message": "undefined variable `foo`",

   407вҶ’   407вҶ’pub fn from_parse_error(
   408вҶ’   408вҶ’    err: &ParseError,

   466вҶ’   466вҶ’/// diagnostics will have [`DiagnosticSourceLocation::unknown()`] for variants

   472вҶ’   472вҶ’            // Parse errors need source text for location computation;
   473вҶ’   473вҶ’            // without it, we produce diagnostics with unknown locations.
   474вҶ’   474вҶ’            errors

   482вҶ’   482вҶ’                        "parser",

   484вҶ’   484вҶ’                    );

   486вҶ’   486вҶ’                        diag = diag.with_suggestion(format!("did you mean '{}'?", suggestion));
   487вҶ’   487вҶ’                    }

   491вҶ’   491вҶ’        }

   493вҶ’   493вҶ’            vec![VumaDiagnostic::new(
   494вҶ’   494вҶ’                "E001",

   497вҶ’   497вҶ’                "scg",
   498вҶ’   498вҶ’                DiagnosticSourceLocation::unknown(),
   499вҶ’   499вҶ’            )]
   500вҶ’   500вҶ’        }
   501вҶ’   501вҶ’        VumaError::ScgValidation { errors } => errors
   502вҶ’   502вҶ’            .iter()
   503вҶ’   503вҶ’            .map(|msg| {
   504вҶ’   504вҶ’                VumaDiagnostic::new(
