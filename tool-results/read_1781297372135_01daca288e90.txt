     1→//! # Multi-Backend Trait Architecture
     2→//!
     3→//! Defines the `TargetInfo` and `Backend` traits that allow VUMA to target
     4→//! multiple instruction set architectures. Each ISA implements these traits
     5→//! to provide target-specific information and code generation.
     6→

     8→use crate::ir::{IRFunction, IRInstr, IRType};
     9→use crate::loongarch64::LoongArch64Backend;
    10→use crate::mips64::Mips64Backend;
    11→use crate::ppc64::PPC64Backend;
    12→use crate::riscv64::RiscV64Backend;

    16→// ---------------------------------------------------------------------------
    17→// Endianness
    18→// ---------------------------------------------------------------------------
    19→
    20→/// Byte order of the target architecture.
    21→#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, serde::Serialize, serde::Deserialize)]
    22→pub enum Endianness {
    23→    /// Least-significant byte first (AArch64, RISC-V, x86_64, LoongArch).
    24→    Little,
    25→    /// Most-significant byte first (MIPS64 big-endian, PPC64 big-endian).
    26→    Big,
    27→    /// Bi-endian — the ISA supports both but the default is big-endian (PPC64).
    28→    Bi,
    29→}
    30→
    31→// ---------------------------------------------------------------------------
    32→// OutputFormat
    33→// ---------------------------------------------------------------------------
    34→
    35→/// The output binary format produced by the backend.
    36→#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, serde::Serialize, serde::Deserialize)]
    37→pub enum OutputFormat {
    38→    /// 64-bit ELF (AArch64, RISC-V64, x86_64, LoongArch64, MIPS64, PPC64).
    39→    Elf64,
    40→    /// 32-bit ELF (ARM32).
    41→    Elf32,
    42→    /// WebAssembly binary module (.wasm).
    43→    WasmBinary,
    44→    /// Raw binary blob (bare-metal, no headers).
    45→    RawBinary,
    46→}
    47→
    48→// ---------------------------------------------------------------------------
    49→// PhysicalReg
    50→// ---------------------------------------------------------------------------
    51→
    52→/// A physical register identified by class and index.
    53→#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, serde::Serialize, serde::Deserialize)]
    54→pub struct PhysicalReg {
    55→    /// Register class.
    56→    pub class: RegClass,
    57→    /// Register index within its class (0-based).
    58→    pub index: u32,
    59→}
    60→
    61→impl PhysicalReg {
    62→    /// Creates a new physical register identifier with the given class and index.
    63→    pub fn new(class: RegClass, index: u32) -> Self {
    64→        Self { class, index }
    65→    }
    66→}
    67→
    68→impl fmt::Display for PhysicalReg {
    69→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
    70→        write!(f, "{:?}:{}", self.class, self.index)
    71→    }
    72→}

    74→// ---------------------------------------------------------------------------
    75→// RegClass
    76→// ---------------------------------------------------------------------------
    77→
    78→/// Classification of physical registers.
    79→#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, serde::Serialize, serde::Deserialize)]
    80→pub enum RegClass {
    81→    /// General-purpose integer registers (X0-X30 on ARM64, RAX-R15 on x86_64, etc.)
    82→    Gpr,
    83→    /// SIMD / floating-point registers (V0-V31 on ARM64, XMM0-XMM15 on x86_64, etc.)
    84→    SimdFp,
    85→    /// Condition register fields (PPC64 CR0-CR7).
    86→    Condition,
    87→    /// Special-purpose register (TOC pointer on PPC64, etc.)
    88→    Special,
    89→}

   108→// ---------------------------------------------------------------------------
   109→// RelocationEntry
   110→// ---------------------------------------------------------------------------
   111→
   112→/// A relocation entry for patching encoded code at link time.
   113→///
   114→/// Each entry records a byte offset within the function's encoded output where
   115→/// a symbolic reference must be resolved, the name of the target symbol, and
   116→/// the ISA-specific relocation type (e.g., `"R_X86_64_PLT32"`, `"R_X86_64_64"`).
   117→#[derive(Clone, Debug, serde::Serialize, serde::Deserialize)]
   118→pub struct RelocationEntry {
   119→    /// Byte offset within the function's encoded code where the relocation applies.
   120→    pub offset: u64,
   121→    /// Name of the target symbol.
   122→    pub symbol: String,
   123→    /// Relocation type (ISA-specific, e.g., "R_X86_64_PLT32", "R_X86_64_64").
   124→    pub reloc_type: String,
   125→}

   140→    /// Encoded bytes (filled in during encoding phase).
   141→    pub encoded: Vec<u8>,
   142→}

   199→// BackendError
   200→// ---------------------------------------------------------------------------
   201→
   202→/// Error type for backend operations.
   203→#[derive(Debug, Clone, thiserror::Error)]
   204→pub enum BackendError {
   205→    /// The requested feature is not supported by this ISA.
   206→    #[error("[{isa}] unsupported feature: {feature}")]
   207→    UnsupportedFeature {
   208→        /// ISA identifier (e.g., "aarch64", "x86_64").
   209→        isa: &'static str,
   210→        /// Description of the unsupported feature.
   211→        feature: String,
   212→    },
   213→
   214→    /// Register allocation failed.
   215→    #[error("[{isa}] register allocation failed: {reason}")]
   216→    RegisterAllocFailed {
   217→        /// ISA identifier.
   218→        isa: &'static str,
   219→        /// Reason for the allocation failure.
   220→        reason: String,
   221→    },
   222→
   223→    /// Instruction encoding failed.
   224→    #[error("[{isa}] encoding error: {reason}")]
   225→    EncodingError {
   226→        /// ISA identifier.
   227→        isa: &'static str,
   228→        /// Reason for the encoding failure.
   229→        reason: String,
   230→    },
   231→
   232→    /// Invalid instruction for this target.
   233→    #[error("[{isa}] invalid instruction: {details}")]
   234→    InvalidInstruction {
   235→        /// ISA identifier.
   236→        isa: &'static str,
   237→        /// Details about why the instruction is invalid.
   238→        details: String,
   239→    },
   240→
   241→    /// ELF / binary emission error.
   242→    #[error("[{isa}] emission error: {reason}")]
   243→    EmissionError {
   244→        /// ISA identifier.
   245→        isa: &'static str,
   246→        /// Reason for the emission failure.
   247→        reason: String,
   248→    },
   249→
   250→    /// The target cannot handle this type.
   251→    #[error("[{isa}] unsupported type: {ty}")]
   252→    UnsupportedType {
   253→        /// ISA identifier.
   254→        isa: &'static str,
   255→        /// The unsupported type name.
   256→        ty: String,
   257→    },
   258→
   259→    /// Generic backend error.
   260→    #[error("[{isa}] {message}")]
   261→    Other {
   262→        /// ISA identifier.
   263→        isa: &'static str,
   264→        /// Error message.
   265→        message: String,
   266→    },
   267→}

   269→// ---------------------------------------------------------------------------
   270→// TargetInfo trait
   271→// ---------------------------------------------------------------------------
   272→
   273→/// Target-specific information needed during code generation.
   274→///
   275→/// This trait provides a data-driven interface for query target properties.
   276→/// It must be implementable for ALL supported ISAs, including fundamentally
   277→/// different architectures like Wasm (stack machine, no registers) and
   278→/// MIPS (branch delay slots).
   279→///
   280→/// # Object Safety
   281→///
   282→/// This trait is object-safe: all methods take `&self` and return only
   283→/// owned types or references with `'static` lifetime.
   284→pub trait TargetInfo: Send + Sync + 'static {
   285→    // === Identity ===
   286→
   287→    /// ISA name in lowercase (e.g., "aarch64", "riscv64", "wasm32").
   288→    fn isa_name(&self) -> &'static str;
   289→
   290→    /// LLVM-style target triple (e.g., "aarch64-unknown-linux-gnu").
   291→    fn target_triple(&self) -> &'static str;
   292→
   293→    /// ELF `e_machine` value.  Returns 0 for non-ELF targets (Wasm).
   294→    fn elf_machine_type(&self) -> u16;
   295→
   296→    /// Default base address for the .text section.
   297→    fn default_base_address(&self) -> u64;
   298→
   299→    // === Data model ===
   300→
   301→    /// Pointer width in bytes (4 for 32-bit, 8 for 64-bit).
   302→    fn pointer_width(&self) -> usize;
   303→
   304→    /// Size in bytes of `ty` on this target.
   305→    fn size_of(&self, ty: &IRType) -> usize;
   306→
   307→    /// Natural alignment in bytes of `ty` on this target.
   308→    fn alignment_of(&self, ty: &IRType) -> usize;
   309→
   310→    /// Byte order of this target.
   311→    fn endianness(&self) -> Endianness;
   312→
   313→    // === Register architecture ===
   314→
   315→    /// Whether this target has registers at all.  `false` for Wasm (stack machine).
   316→    fn has_registers(&self) -> bool;
   317→
   318→    /// Number of general-purpose registers.  0 for Wasm.
   319→    fn num_gp_regs(&self) -> usize;
   320→
   321→    /// Number of SIMD/FP registers.  0 for Wasm.
   322→    fn num_simd_fp_regs(&self) -> usize;
   323→
   324→    /// Whether the ISA has a hardwired-zero register (RISC-V x0, LoongArch r0).
   325→    fn has_hardwired_zero(&self) -> bool;
   326→
   327→    /// Whether the ISA uses a link register (ARM, RISC-V, MIPS, PPC) rather than
   328→    /// pushing the return address on the stack (x86_64).
   329→    fn has_link_register(&self) -> bool;
   330→
   331→    /// Whether branches have delay slots (MIPS only).
   332→    fn has_branch_delay_slots(&self) -> bool;
   333→
   334→    /// Whether this ISA uses a TOC (Table of Contents) pointer (PPC64 r2).
   335→    fn has_toc_pointer(&self) -> bool;
   336→
   337→    /// Whether this ISA has dedicated condition register fields (PPC64 CR0-CR7).
   338→    fn has_condition_registers(&self) -> bool;
   339→
   340→    // === Calling convention ===
   341→
   342→    /// Name of the calling convention (e.g., "aapcs64", "lp64d", "systemv").
   343→    fn calling_convention_name(&self) -> &'static str;
   344→
   345→    /// Number of integer argument registers.
   346→    fn num_int_arg_regs(&self) -> usize;
   347→
   348→    /// Number of FP/SIMD argument registers.
   349→    fn num_fp_arg_regs(&self) -> usize;
   350→
   351→    /// Required stack alignment in bytes.
   352→    fn stack_alignment(&self) -> usize;
   353→
   354→    // === Instruction encoding ===
   355→
   356→    /// Alignment requirement for instructions in bytes (4 for fixed-width RISCs,
   357→    /// 1 for x86_64 and Wasm).
   358→    fn instruction_alignment(&self) -> usize;
   359→
   360→    /// Minimum and maximum instruction width in bytes.
   361→    /// - Fixed-width 32-bit ISAs: (4, 4)
   362→    /// - x86_64: (1, 15)
   363→    /// - RISC-V with RVC: (2, 4)
   364→    /// - Wasm: (1, ~) but typically (1, 15)
   365→    fn instruction_width_range(&self) -> (usize, usize);
   366→
   367→    // === Output format ===
   368→
   369→    /// Binary format produced by this backend.
   370→    fn output_format(&self) -> OutputFormat;
   371→}

   373→// ---------------------------------------------------------------------------
   374→// Backend trait
   375→// ---------------------------------------------------------------------------
   376→
   377→/// A code generation backend for a specific target architecture.
   378→///
   379→/// Each supported ISA implements this trait, providing register allocation,
   380→/// instruction encoding, program emission, and disassembly.
   381→///
   382→/// # Object Safety
   383→///
   384→/// This trait is object-safe.
   385→pub trait Backend: Send + Sync + 'static {
   386→    /// Returns a reference to this backend's target info.
   387→    fn target_info(&self) -> &dyn TargetInfo;
   388→
   389→    /// Allocate physical registers for an IR function.
   390→    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError>;
   391→
   392→    /// Encode a single allocated function into machine code bytes.
   393→    fn encode_function(&self, func: &AllocatedFunction) -> Result<Vec<u8>, BackendError>;
   394→
   395→    /// Encode an entire allocated program into its final binary form
   396→    /// (ELF, .wasm, raw binary, etc.).
   397→    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError>;
   398→
   399→    /// Returns the bytes for a minimal return stub (e.g., `RET` on ARM64,
   400→    /// `mov eax, 0; ret` on x86_64, `end` on Wasm).
   401→    fn return_stub(&self) -> Vec<u8>;
   402→
   403→    /// Returns a trampoline that jumps to `entry_addr`.
   404→    fn trampoline(&self, entry_addr: u64) -> Vec<u8>;
   405→
   406→    /// Disassemble `bytes` starting at virtual address `addr`.
   407→    fn disassemble(&self, bytes: &[u8], addr: u64) -> Vec<String>;
   408→
   409→    /// Human-readable name of this backend.
   410→    fn name(&self) -> &'static str;
   411→}

   413→// ---------------------------------------------------------------------------
   414→// BackendKind
   415→// ---------------------------------------------------------------------------
   416→
   417→/// Enumeration of all supported backend architectures.
   418→#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, serde::Serialize, serde::Deserialize)]
   419→pub enum BackendKind {
   420→    /// ARM 64-bit (AArch64).
   421→    AArch64,
   422→    /// RISC-V 64-bit.
   423→    RiscV64,
   424→    /// WebAssembly 32-bit.
   425→    Wasm32,
   426→    /// LoongArch 64-bit.
   427→    LoongArch64,
   428→    /// x86-64.
   429→    X86_64,
   430→    /// ARM 32-bit.
   431→    Arm32,
   432→    /// MIPS 64-bit.
   433→    Mips64,
   434→    /// PowerPC 64-bit.
   435→    PowerPC64,
   436→}
   437→
   438→impl BackendKind {
   439→    /// Returns the ISA name string for this backend kind.
   440→    pub fn isa_name(&self) -> &'static str {
   441→        match self {
   442→            BackendKind::AArch64 => "aarch64",
   443→            BackendKind::RiscV64 => "riscv64",
   444→            BackendKind::Wasm32 => "wasm32",
   445→            BackendKind::LoongArch64 => "loongarch64",
   446→            BackendKind::X86_64 => "x86_64",
   447→            BackendKind::Arm32 => "arm32",
   448→            BackendKind::Mips64 => "mips64",
   449→            BackendKind::PowerPC64 => "ppc64",
   450→        }
   451→    }
   452→}
   453→
   454→impl fmt::Display for BackendKind {
   455→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   456→        f.write_str(self.isa_name())
   457→    }
   458→}

  1112→// ---------------------------------------------------------------------------
  1113→// AArch64 Mnemonic Decoder
  1114→// ---------------------------------------------------------------------------
  1115→
  1116→/// Decode a 32-bit AArch64 instruction word into a human-readable mnemonic.
  1117→///
  1118→/// Covers the most common AArch64 instructions: ADD, SUB, MOV, LDR, STR, B,
  1119→/// BL, RET, CMP, B.cond, STP, LDP, NOP, MUL, SDIV, UDIV, AND, ORR, EOR,
  1120→/// plus several additional frequently-encountered encodings.
  1121→fn decode_aarch64(word: u32) -> String {
  1122→    let rd = word & 0x1F;
  1123→    let rn = (word >> 5) & 0x1F;
  1124→    let rt = rd; // alias for load/store destination
  1125→    let rm = (word >> 16) & 0x1F;
  1126→    let imm12 = (word >> 10) & 0xFFF;
  1127→    let cond = word & 0xF;
  1128→
  1129→    // NOP: d503201f
  1130→    if word == 0xD503201F {
  1131→        return "nop".to_string();
  1132→    }
  1133→
  1134→    // RET: d65f03c0
  1135→    if word == 0xD65F03C0 {
  1136→        return "ret".to_string();
  1137→    }
  1138→
  1139→    let _top8 = word >> 24;
  1140→    let _top10 = word >> 22;
  1141→
  1142→    // --- ADD/SUB (immediate): 100100xx ...
  1143→    if (word >> 23) & 0x1FF == 0b1_0010_0010 {
  1144→        // ADD Xd, Xn, #imm12
  1145→        return format!("add x{}, x{}, #{}", rd, rn, imm12);
  1146→    }
  1147→    if (word >> 23) & 0x1FF == 0b1_1010_0010 {
  1148→        // SUB Xd, Xn, #imm12
  1149→        return format!("sub x{}, x{}, #{}", rd, rn, imm12);
  1150→    }
  1151→
  1152→    // --- ADD (shifted register): 1_00_0101_1_xxx ...
  1153→    if (word >> 24) & 0xFF == 0b1000_1011 {
  1154→        // ADD Xd, Xn, Xm
  1155→        return format!("add x{}, x{}, x{}", rd, rn, rm);
  1156→    }
  1157→
  1158→    // --- SUB (shifted register): 1_00_0101_1_xxx ... with S=1 at bit30
  1159→    if (word >> 24) & 0xFF == 0b1101_0110 {
  1160→        // SUB Xd, Xn, Xm (bit 30 set = sub)
  1161→        return format!("sub x{}, x{}, x{}", rd, rn, rm);
  1162→    }
  1163→
  1164→    // --- AND (shifted register): 1_00_0101_0_00_xxx
  1165→    if (word >> 24) & 0xFE == 0b1000_1010 {
  1166→        // Check bit 21-15: opcode[31:21] = 10001010_000
  1167→        if (word >> 21) & 0x7FF == 0b10001010000 {
  1168→            return format!("and x{}, x{}, x{}", rd, rn, rm);
  1169→        }
  1170→    }
  1171→
  1172→    // --- ORR (shifted register): 1_01_0101_0_00_xxx
  1173→    if (word >> 21) & 0x7FF == 0b10101010000 {
  1174→        return format!("orr x{}, x{}, x{}", rd, rn, rm);
  1175→    }
  1176→
  1177→    // --- EOR (shifted register): 1_10_0101_0_00_xxx
  1178→    if (word >> 21) & 0x7FF == 0b11001010000 {
  1179→        return format!("eor x{}, x{}, x{}", rd, rn, rm);
  1180→    }
  1181→
  1182→    // --- MOV (register): alias for ORR Xd, XZR, Xm
  1183→    // ORR Xd, XZR, Xm: 10101010_000_rm_000000_xzr_rd
  1184→    // More general: ORR with Rn=XZR(31)
  1185→    if (word >> 21) & 0x7FF == 0b10101010000 && rn == 31 {
  1186→        return format!("mov x{}, x{}", rd, rm);
  1187→    }
  1188→
  1189→    // --- MUL: MADD Xd, Xn, Xm, XZR
  1190→    // Encoding: 1_00_1101_1_000_Rm_0_01111_Rn_Rd
  1191→    if (word >> 21) & 0x7FF == 0b10011011000 && ((word >> 10) & 0x1F) == 0b01111 {
  1192→        return format!("mul x{}, x{}, x{}", rd, rn, rm);
  1193→    }
  1194→
  1195→    // --- SDIV: 1_00_1101_1_0100_00000_00001_Rn_Rd  (actually 1_00_1101_0100_Rm_00001_Rn_Rd)
  1196→    if (word >> 21) & 0x7FF == 0b10011010100 && (word >> 10) & 0x1F == 0b00001 {
  1197→        return format!("sdiv x{}, x{}, x{}", rd, rn, rm);
  1198→    }
  1199→
  1200→    // --- UDIV: 1_00_1101_0000_Rm_00001_Rn_Rd
  1201→    if (word >> 21) & 0x7FF == 0b10011010000 && (word >> 10) & 0x1F == 0b00001 {
  1202→        return format!("udiv x{}, x{}, x{}", rd, rn, rm);
  1203→    }
  1204→
  1205→    // --- CMP (immediate): SUBS XZR, Xn, #imm12
  1206→    // 11100001_00_xxx_xxx_xxx_xxx_xxx_11111_xxx
  1207→    if (word >> 23) & 0x1FF == 0b1_1110_0010 && rd == 31 {
  1208→        return format!("cmp x{}, #{}", rn, imm12);
  1209→    }
  1210→
  1211→    // --- CMP (register): SUBS XZR, Xn, Xm
  1212→    if (word >> 21) & 0x7FF == 0b11101011000 && rd == 31 {
  1213→        return format!("cmp x{}, x{}", rn, rm);
  1214→    }
  1215→
  1216→    // --- B.cond: 0101010x xxxxxxxxxx xxxxxx cond
  1217→    if (word >> 24) & 0xFF == 0x54 {
  1218→        let cond_name = match cond {
  1219→            0 => "eq",
  1220→            1 => "ne",
  1221→            2 => "cs",
  1222→            3 => "cc",
  1223→            4 => "mi",
  1224→            5 => "pl",
  1225→            6 => "vs",
  1226→            7 => "vc",
  1227→            8 => "hi",
  1228→            9 => "ls",
  1229→            10 => "ge",
  1230→            11 => "lt",
  1231→            12 => "gt",
  1232→            13 => "le",
  1233→            14 => "al",
  1234→            _ => "??",
  1235→        };
  1236→        let imm19 = (word >> 5) & 0x7FFFF;
  1237→        let offset = ((imm19 as i32) << 13) >> 11; // sign-extend and *4
  1238→        return format!("b.{} {:+}", cond_name, offset);
  1239→    }
  1240→
  1241→    // --- B (unconditional): 000101xx xxxxxxxxxxxxxxxxxxxx
  1242→    if (word >> 26) & 0x3F == 0b000101 {
  1243→        let imm26 = word & 0x3FFFFFF;
  1244→        let offset = ((imm26 as i32) << 6) >> 4; // sign-extend and *4
  1245→        return format!("b {:+}", offset);
  1246→    }
  1247→
  1248→    // --- BL: 100101xx xxxxxxxxxxxxxxxxxxxx
  1249→    if (word >> 26) & 0x3F == 0b100101 {
  1250→        let imm26 = word & 0x3FFFFFF;
  1251→        let offset = ((imm26 as i32) << 6) >> 4;
  1252→        return format!("bl {:+}", offset);
  1253→    }
  1254→
  1255→    // --- LDR (unsigned offset): 11111001_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1256→    if (word >> 22) & 0x3FF == 0b1111100101 {
  1257→        let imm12_raw = (word >> 10) & 0xFFF;
  1258→        let offset = imm12_raw * 8; // scale by 8 for 64-bit
  1259→        return format!("ldr x{}, [x{}, #{}]", rt, rn, offset);
  1260→    }
  1261→
  1262→    // --- STR (unsigned offset): 11111000_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1263→    if (word >> 22) & 0x3FF == 0b1111100001 {
  1264→        let imm12_raw = (word >> 10) & 0xFFF;
  1265→        let offset = imm12_raw * 8;
  1266→        return format!("str x{}, [x{}, #{}]", rt, rn, offset);
  1267→    }
  1268→
  1269→    // --- LDP (signed offset, 64-bit): 101_0100_110_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1270→    if (word >> 22) & 0x3FF == 0b1010100110 {
  1271→        let rt2 = (word >> 10) & 0x1F;
  1272→        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1273→        let offset = imm7 * 8;
  1274→        return format!("ldp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1275→    }
  1276→
  1277→    // --- STP (signed offset, 64-bit): 101_0100_010_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1278→    if (word >> 22) & 0x3FF == 0b1010100010 {
  1279→        let rt2 = (word >> 10) & 0x1F;
  1280→        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1281→        let offset = imm7 * 8;
  1282→        return format!("stp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1283→    }
  1284→
  1285→    // --- MOVZ: 110100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1286→    if (word >> 23) & 0x1FF == 0b110100101 {
  1287→        let hw = (word >> 21) & 0x3;
  1288→        let imm16 = (word >> 5) & 0xFFFF;
  1289→        return format!("movz x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1290→    }
  1291→
  1292→    // --- MOVK: 111100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1293→    if (word >> 23) & 0x1FF == 0b111100101 {
  1294→        let hw = (word >> 21) & 0x3;
  1295→        let imm16 = (word >> 5) & 0xFFFF;
  1296→        return format!("movk x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1297→    }
  1298→
  1299→    format!(".word {:08x}", word)
  1300→}

  1306→/// AArch64 (ARM64) code generation backend.
  1307→///
  1308→/// Wraps the existing ARM64 emitter, register allocator, and instruction
  1309→/// encoding behind the `Backend` trait.
  1310→pub struct AArch64Backend {
  1311→    target_info: AArch64TargetInfo,
  1312→}

  1314→impl AArch64Backend {
  1315→    /// Create a new AArch64 backend.
  1316→    pub fn new() -> Self {
  1317→        Self {
  1318→            target_info: AArch64TargetInfo,
  1319→        }
  1320→    }
  1321→}

  1325→        Self::new()
  1326→    }

  1329→/// Compute the stack frame size for an IR function.
  1330→///
  1331→/// Replicates the private `compute_frame_size` function in `emit.rs`:
  1332→/// sums `Alloc` instruction sizes, adds 16 bytes for the FP/LR save pair,
  1333→/// and rounds up to 16-byte alignment.
  1334→fn aarch64_compute_frame_size(func: &IRFunction) -> usize {
  1335→    let mut total: u32 = 16; // FP/LR save pair
  1336→    for block in &func.blocks {
  1337→        for instr in &block.instructions {
  1338→            if let IRInstr::Alloc { size, .. } = instr {
  1339→                let aligned = (*size).div_ceil(16) * 16;
  1340→                total += aligned;
  1341→            }
  1342→        }
  1343→    }
  1344→    // Round up to 16-byte alignment
  1345→    total = (total + 15) & !15;
  1346→    total as usize
  1347→}
  1348→
  1349→/// Build a minimal ELF64 binary for AArch64 from raw code bytes.
  1350→///
  1351→/// Produces a static executable with a single LOAD segment containing the
  1352→/// `.text` section.  This is sufficient for programs that have already been
  1353→/// emitted through the `allocate_registers` / `encode_function` pipeline.
  1354→fn build_minimal_aarch64_elf(code: &[u8], base_addr: u64) -> Vec<u8> {
  1355→    // Layout: ELF header (64) | 1 program header (56) | code
  1356→    let elf_header_size: u64 = 64;
  1357→    let phdr_size: u64 = 56;
  1358→    let text_offset = elf_header_size + phdr_size;
  1359→    let text_size = code.len() as u64;
  1360→    let entry_point = base_addr + text_offset;
  1361→
  1362→    let mut elf = Vec::with_capacity(text_offset as usize + code.len());
  1363→
  1364→    // --- e_ident ---
  1365→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1366→    elf.push(2); // ELFCLASS64
  1367→    elf.push(1); // ELFDATA2LSB
  1368→    elf.push(1); // EV_CURRENT
  1369→    elf.push(3); // ELFOSABI_LINUX
  1370→    elf.push(0); // padding
  1371→    elf.extend_from_slice(&[0u8; 7]); // padding
  1372→
  1373→    // --- ELF header fields ---
  1374→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1375→    elf.extend_from_slice(&183u16.to_le_bytes()); // e_machine = EM_AARCH64
  1376→    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1377→    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1378→    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1379→    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1380→    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1381→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1382→    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1383→    elf.extend_from_slice(&1u16.to_le_bytes()); // e_phnum
  1384→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1385→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1386→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1387→
  1388→    // --- Program Header (single LOAD segment: PF_R | PF_X) ---
  1389→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1390→    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1391→    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1392→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1393→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1394→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1395→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1396→    elf.extend_from_slice(&16u64.to_le_bytes()); // p_align
  1397→
  1398→    // --- Code section ---
  1399→    elf.extend_from_slice(code);
  1400→
  1401→    elf
  1402→}

  1404→impl Backend for AArch64Backend {
  1405→    fn target_info(&self) -> &dyn TargetInfo {
  1406→        &self.target_info
  1407→    }
  1408→
  1409→    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError> {
  1410→        // Use the existing Emitter to emit the function, which internally
  1411→        // performs register allocation and instruction encoding.
  1412→        let mut emitter = crate::emit::Emitter::new();
  1413→        let code = emitter
  1414→            .emit_function(func)
  1415→            .map_err(|e| BackendError::RegisterAllocFailed {
  1416→                isa: "aarch64",
  1417→                reason: e.to_string(),
  1418→            })?;
  1419→
  1420→        let func_name = func.name.clone();
  1421→        let frame_size = aarch64_compute_frame_size(func);
  1422→
  1423→        // Convert each 32-bit ARM64 instruction word into an AllocatedInstruction
  1424→        // with its little-endian encoded bytes.
  1425→        let instructions: Vec<AllocatedInstruction> = code
  1426→            .iter()
  1427→            .enumerate()
  1428→            .map(|(i, &word)| AllocatedInstruction {
  1429→                opcode: format!("arm64_{}", i),
  1430→                reads: vec![],
  1431→                writes: vec![],
  1432→                encoded: word.to_le_bytes().to_vec(),
  1433→            })
  1434→            .collect();
  1435→
  1436→        let code_size = instructions.len() * 4;
  1437→
  1438→        Ok(AllocatedFunction {
  1439→            name: func_name,
  1440→            blocks: vec![AllocatedBlock {
  1441→                label: "entry".to_string(),
  1442→                instructions,
  1443→                code_offset: 0,
  1444→            }],
  1445→            frame_size,
  1446→            callee_saved: vec![],
  1447→            spill_slots: 0,
  1448→            code_size,
  1449→            relocations: Vec::new(),
  1450→        })
  1451→    }
  1452→
  1453→    fn encode_function(&self, func: &AllocatedFunction) -> Result<Vec<u8>, BackendError> {
  1454→        let mut bytes = Vec::new();
  1455→        for block in &func.blocks {
  1456→            for instr in &block.instructions {
  1457→                bytes.extend_from_slice(&instr.encoded);
  1458→            }
  1459→        }
  1460→        Ok(bytes)
  1461→    }
  1462→
  1463→    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError> {
  1464→        // Collect all encoded bytes from every function and block.
  1465→        let mut all_code = Vec::new();
  1466→        for func in &program.functions {
  1467→            for block in &func.blocks {
  1468→                for instr in &block.instructions {
  1469→                    all_code.extend_from_slice(&instr.encoded);
  1470→                }
  1471→            }
  1472→        }
  1473→
  1474→        // Wrap in a minimal ELF64 binary for AArch64.
  1475→        Ok(build_minimal_aarch64_elf(&all_code, 0x400000))
  1476→    }
  1477→
  1478→    fn return_stub(&self) -> Vec<u8> {
  1479→        // ARM64 RET instruction: 0xD65F03C0
  1480→        vec![0xC0, 0x03, 0x5F, 0xD6]
  1481→    }
  1482→
  1483→    fn trampoline(&self, entry_addr: u64) -> Vec<u8> {
  1484→        // LDR X16, [PC, #8] ; BR X16 ; <8 bytes address>
  1485→        let mut code = Vec::with_capacity(16);
  1486→        // LDR X16, [PC, #8] = 0x58000050
  1487→        code.extend_from_slice(&0x58000050u32.to_le_bytes());
  1488→        // BR X16 = 0xD61F0200
  1489→        code.extend_from_slice(&0xD61F0200u32.to_le_bytes());
  1490→        // 64-bit address
  1491→        code.extend_from_slice(&entry_addr.to_le_bytes());
  1492→        code
  1493→    }
  1494→
  1495→    fn disassemble(&self, bytes: &[u8], addr: u64) -> Vec<String> {
  1496→        // Mnemonic decoder for AArch64 (4-byte fixed-width instructions).
  1497→        let mut lines = Vec::new();
  1498→        let mut offset = 0usize;
  1499→        let mut pc = addr;
  1500→        while offset + 4 <= bytes.len() {
  1501→            let word = u32::from_le_bytes([
  1502→                bytes[offset],
  1503→                bytes[offset + 1],
  1504→                bytes[offset + 2],
  1505→                bytes[offset + 3],
  1506→            ]);
  1507→            let mnemonic = if let Some(instr) = crate::arm64::Instruction::decode(word) {
  1508→                format!("{}", instr)
  1509→            } else {
  1510→                decode_aarch64(word)
  1511→            };
  1512→            lines.push(format!("{:#010x}:  {:08x}  {}", pc, word, mnemonic));
  1513→            offset += 4;
  1514→            pc += 4;
  1515→        }
  1516→        if offset < bytes.len() {
  1517→            let remaining = &bytes[offset..];
  1518→            lines.push(format!("{:#010x}:  {:02x?}", pc, remaining));
  1519→        }
  1520→        lines
  1521→    }
  1522→
  1523→    fn name(&self) -> &'static str {
  1524→        "aarch64"
  1525→    }
  1526→}

  1557→    /// Helper: validate that a TargetInfo impl returns internally consistent values.
  1558→    fn validate_target_info(info: &dyn TargetInfo) {
  1559→        // If there are no registers, register counts must be zero.
  1560→        if !info.has_registers() {
  1561→            assert_eq!(
  1562→                info.num_gp_regs(),
  1563→                0,
  1564→                "{}: has_registers=false but num_gp_regs != 0",
  1565→                info.isa_name()
  1566→            );
  1567→            assert_eq!(
  1568→                info.num_simd_fp_regs(),
  1569→                0,
  1570→                "{}: has_registers=false but num_simd_fp_regs != 0",
  1571→                info.isa_name()
  1572→            );
  1573→            assert_eq!(
  1574→                info.num_int_arg_regs(),
  1575→                0,
  1576→                "{}: has_registers=false but num_int_arg_regs != 0",
  1577→                info.isa_name()
  1578→            );
  1579→            assert_eq!(
  1580→                info.num_fp_arg_regs(),
  1581→                0,
  1582→                "{}: has_registers=false but num_fp_arg_regs != 0",
  1583→                info.isa_name()
  1584→            );
  1585→        }
  1586→
  1587→        // Pointer width must be 4 or 8.
  1588→        assert!(
  1589→            info.pointer_width() == 4 || info.pointer_width() == 8,
  1590→            "{}: pointer_width must be 4 or 8, got {}",
  1591→            info.isa_name(),
  1592→            info.pointer_width()
  1593→        );
  1594→
  1595→        // Stack alignment must be a power of 2 and at least 8.
  1596→        let sa = info.stack_alignment();
  1597→        assert!(
  1598→            sa >= 8,
  1599→            "{}: stack_alignment must be >= 8, got {}",
  1600→            info.isa_name(),
  1601→            sa
  1602→        );
  1603→        assert!(
  1604→            sa.is_power_of_two(),
  1605→            "{}: stack_alignment must be a power of 2, got {}",
  1606→            info.isa_name(),
  1607→            sa
  1608→        );
  1609→
  1610→        // Instruction alignment must be 1, 2, or 4.
  1611→        let ia = info.instruction_alignment();
  1612→        assert!(
  1613→            ia == 1 || ia == 2 || ia == 4,
  1614→            "{}: instruction_alignment must be 1, 2, or 4, got {}",
  1615→            info.isa_name(),
  1616→            ia
  1617→        );
  1618→
  1619→        // Width range must be sane.
  1620→        let (min_w, max_w) = info.instruction_width_range();
  1621→        assert!(
  1622→            min_w >= 1,
  1623→            "{}: min instruction width must be >= 1",
  1624→            info.isa_name()
  1625→        );
  1626→        assert!(
  1627→            max_w >= min_w,
  1628→            "{}: max instruction width must be >= min",
  1629→            info.isa_name()
  1630→        );
  1631→
  1632→        // Only MIPS has branch delay slots.
  1633→        if info.has_branch_delay_slots() {
  1634→            assert_eq!(
  1635→                info.isa_name(),
  1636→                "mips64",
  1637→                "Only MIPS64 should have branch delay slots"
  1638→            );
  1639→        }
  1640→
  1641→        // Only PPC64 has a TOC pointer.
  1642→        if info.has_toc_pointer() {
  1643→            assert_eq!(
  1644→                info.isa_name(),
  1645→                "ppc64",
  1646→                "Only PPC64 should have a TOC pointer"
  1647→            );
  1648→        }
  1649→
  1650→        // Only PPC64 has condition registers.
  1651→        if info.has_condition_registers() {
  1652→            assert_eq!(
  1653→                info.isa_name(),
  1654→                "ppc64",
  1655→                "Only PPC64 should have condition registers"
  1656→            );
  1657→        }
  1658→
  1659→        // size_of and alignment_of for basic types.
  1660→        let ptr_size = info.size_of(&IRType::Ptr);
  1661→        assert_eq!(
  1662→            ptr_size,
  1663→            info.pointer_width(),
  1664→            "{}: Ptr size must match pointer_width",
  1665→            info.isa_name()
  1666→        );
  1667→    }
