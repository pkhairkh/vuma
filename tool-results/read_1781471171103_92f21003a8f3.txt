     1→//! SCG Serialization / Deserialization
     2→//!
     3→//! This module provides three serialization formats for the Semantic Computation Graph:
     4→//!
     5→//! - **Binary**: A compact, versioned binary format suitable for persistent storage
     6→//!   and network transfer. The format includes a magic header and version number
     7→//!   for forward/backward compatibility.
     8→//! - **JSON**: A human-readable JSON representation using serde, ideal for debugging
     9→//!   and interoperability.
    10→//! - **DOT**: A Graphviz DOT format for visualizing the graph structure.
    11→//!
    12→//! # Binary Format (Version 1)
    13→//!
    14→//! ```text
    15→//! [4B]  Magic: "VSCG"
    16→//! [4B]  Version: u32 LE
    17→//! [8B]  Next node ID counter: u64 LE
    18→//! [8B]  Next edge ID counter: u64 LE
    19→//! [4B]  Node count: u32 LE
    20→//! [4B]  Edge count: u32 LE
    21→//! [4B]  Region count: u32 LE
    22→//! --- Nodes (repeated Node count times) ---
    23→//! --- Edges (repeated Edge count times) ---
    24→//! --- Regions (repeated Region count times) ---
    25→//! ```

    94→// ── Error type ──────────────────────────────────────────────────────────
    95→
    96→/// Errors that can occur during SCG deserialization.
    97→#[derive(Debug, Clone, PartialEq)]
    98→pub enum DeserializeError {
    99→    /// The data does not start with the expected magic bytes.
   100→    InvalidMagic {
   101→        /// The expected magic bytes.
   102→        expected: Vec<u8>,
   103→        /// The magic bytes actually found.
   104→        found: Vec<u8>,
   105→    },
   106→    /// The format version is not supported by this implementation.
   107→    UnsupportedVersion {
   108→        /// The version number found in the data.
   109→        version: u32,
   110→        /// The minimum version supported by this implementation.
   111→        min_supported: u32,
   112→    },
   113→    /// An unexpected end of input was reached.
   114→    UnexpectedEof {
   115→        /// Description of what was being read when EOF was encountered.
   116→        context: String,
   117→    },
   118→    /// A value was out of the valid range for its type.
   119→    InvalidValue {
   120→        /// Name of the field that had the invalid value.
   121→        field: String,
   122→        /// String representation of the invalid value.
   123→        value: String,
   124→    },
   125→    /// A string could not be decoded as valid UTF-8.
   126→    InvalidUtf8 {
   127→        /// Description of what was being read when the invalid UTF-8 was encountered.
   128→        context: String,
   129→        /// The underlying UTF-8 error message.
   130→        source: String,
   131→    },
   132→    /// An I/O error occurred during deserialization.
   133→    IoError(String),
   134→    /// A JSON serialization/deserialization error.
   135→    JsonError(String),
   136→    /// An internal consistency error in the serialized data.
   137→    ConsistencyError(String),
   138→}
   139→
   140→impl std::fmt::Display for DeserializeError {
   141→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   142→        match self {
   143→            DeserializeError::InvalidMagic { expected, found } => {
   144→                write!(
   145→                    f,
   146→                    "invalid magic: expected {:?}, found {:?}",
   147→                    expected, found
   148→                )
   149→            }
   150→            DeserializeError::UnsupportedVersion {
   151→                version,
   152→                min_supported,
   153→            } => {
   154→                write!(
   155→                    f,
   156→                    "unsupported format version {} (minimum supported: {})",
   157→                    version, min_supported
   158→                )
   159→            }
   160→            DeserializeError::UnexpectedEof { context } => {
   161→                write!(f, "unexpected end of input: {}", context)
   162→            }
   163→            DeserializeError::InvalidValue { field, value } => {
   164→                write!(f, "invalid value for '{}': {}", field, value)
   165→            }
   166→            DeserializeError::InvalidUtf8 { context, source } => {
   167→                write!(f, "invalid UTF-8 in '{}': {}", context, source)
   168→            }
   169→            DeserializeError::IoError(msg) => write!(f, "I/O error: {}", msg),
   170→            DeserializeError::JsonError(msg) => write!(f, "JSON error: {}", msg),
   171→            DeserializeError::ConsistencyError(msg) => write!(f, "consistency error: {}", msg),
   172→        }
   173→    }
   174→}

   184→// ── Binary Reader ───────────────────────────────────────────────────────
   185→
   186→/// Helper for reading structured binary data from a byte slice.
   187→struct BinaryReader<'a> {
   188→    data: &'a [u8],
   189→    pos: usize,
   190→}
   191→
   192→impl<'a> BinaryReader<'a> {
   193→    fn new(data: &'a [u8]) -> Self {
   194→        Self { data, pos: 0 }
   195→    }
   196→
   197→    #[allow(dead_code)] // part of BinaryReader API for future serialization needs
   198→    fn remaining(&self) -> usize {
   199→        self.data.len().saturating_sub(self.pos)
   200→    }
   201→
   202→    fn read_bytes(&mut self, n: usize, context: &str) -> Result<&'a [u8], DeserializeError> {
   203→        if self.pos + n > self.data.len() {
   204→            return Err(DeserializeError::UnexpectedEof {
   205→                context: context.to_string(),
   206→            });
   207→        }
   208→        let slice = &self.data[self.pos..self.pos + n];
   209→        self.pos += n;
   210→        Ok(slice)
   211→    }
   212→
   213→    fn read_u8(&mut self, context: &str) -> Result<u8, DeserializeError> {
   214→        let bytes = self.read_bytes(1, context)?;
   215→        Ok(bytes[0])
   216→    }
   217→
   218→    fn read_bool(&mut self, context: &str) -> Result<bool, DeserializeError> {
   219→        let val = self.read_u8(context)?;
   220→        Ok(val != 0)
   221→    }
   222→
   223→    fn read_u32_le(&mut self, context: &str) -> Result<u32, DeserializeError> {
   224→        let bytes = self.read_bytes(4, context)?;
   225→        Ok(u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]))
   226→    }
   227→
   228→    fn read_u64_le(&mut self, context: &str) -> Result<u64, DeserializeError> {
   229→        let bytes = self.read_bytes(8, context)?;
   230→        Ok(u64::from_le_bytes([
   231→            bytes[0], bytes[1], bytes[2], bytes[3], bytes[4], bytes[5], bytes[6], bytes[7],
   232→        ]))
   233→    }
   234→
   235→    fn read_string(&mut self, context: &str) -> Result<String, DeserializeError> {
   236→        let len = self.read_u32_le(&format!("{}.length", context))? as usize;
   237→        if len == 0 {
   238→            return Ok(String::new());
   239→        }
   240→        let bytes = self.read_bytes(len, context)?;
   241→        std::str::from_utf8(bytes)
   242→            .map(|s| s.to_string())
   243→            .map_err(|e| DeserializeError::InvalidUtf8 {
   244→                context: context.to_string(),
   245→                source: e.to_string(),
   246→            })
   247→    }
   248→
   249→    fn read_optional_string(&mut self, context: &str) -> Result<Option<String>, DeserializeError> {
   250→        let has = self.read_bool(&format!("{}.has", context))?;
   251→        if has {
   252→            self.read_string(context).map(Some)
   253→        } else {
   254→            Ok(None)
   255→        }
   256→    }
   257→
   258→    fn read_optional_u64(&mut self, context: &str) -> Result<Option<u64>, DeserializeError> {
   259→        let has = self.read_bool(&format!("{}.has", context))?;
   260→        if has {
   261→            self.read_u64_le(context).map(Some)
   262→        } else {
   263→            Ok(None)
   264→        }
   265→    }
   266→}

   268→// ── Binary Writer ───────────────────────────────────────────────────────
   269→
   270→/// Helper for writing structured binary data into a byte buffer.
   271→struct BinaryWriter {
   272→    buf: Vec<u8>,
   273→}
   274→
   275→impl BinaryWriter {
   276→    fn new() -> Self {
   277→        Self {
   278→            buf: Vec::with_capacity(256),
   279→        }
   280→    }
   281→
   282→    fn write_bytes(&mut self, data: &[u8]) {
   283→        self.buf.extend_from_slice(data);
   284→    }
   285→
   286→    #[allow(dead_code)] // part of BinaryWriter API for future serialization needs
   287→    fn write_u8(&mut self, val: u8) {
   288→        self.buf.push(val);
   289→    }
   290→
   291→    fn write_bool(&mut self, val: bool) {
   292→        self.buf.push(if val { 1 } else { 0 });
   293→    }
   294→
   295→    fn write_u32_le(&mut self, val: u32) {
   296→        self.buf.extend_from_slice(&val.to_le_bytes());
   297→    }
   298→
   299→    fn write_u64_le(&mut self, val: u64) {
   300→        self.buf.extend_from_slice(&val.to_le_bytes());
   301→    }
   302→
   303→    fn write_string(&mut self, s: &str) {
   304→        let bytes = s.as_bytes();
   305→        self.write_u32_le(bytes.len() as u32);
   306→        self.buf.extend_from_slice(bytes);
   307→    }
   308→
   309→    fn write_optional_string(&mut self, opt: &Option<String>) {
   310→        match opt {
   311→            Some(s) => {
   312→                self.write_bool(true);
   313→                self.write_string(s);
   314→            }
   315→            None => {
   316→                self.write_bool(false);
   317→            }
   318→        }
   319→    }
   320→
   321→    fn write_optional_u64(&mut self, opt: &Option<u64>) {
   322→        match opt {
   323→            Some(v) => {
   324→                self.write_bool(true);
   325→                self.write_u64_le(*v);
   326→            }
   327→            None => {
   328→                self.write_bool(false);
   329→            }
   330→        }
   331→    }
   332→
   333→    fn into_vec(self) -> Vec<u8> {
   334→        self.buf
   335→    }
   336→}

   509→
   510→/// Intermediate representation used for JSON serialization.
   511→///
   512→/// This struct flattens the SCG's internal petgraph representation into
   513→/// a simple structure that can be serialized with serde.
   514→#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
   515→struct SerializedSCG {
   516→    /// Format version for forward/backward compatibility.
   517→    version: u32,
   518→    /// All nodes in the graph.
   519→    nodes: Vec<NodeData>,
   520→    /// All edges in the graph.
   521→    edges: Vec<EdgeData>,
   522→    /// All regions in the graph.
   523→    regions: Vec<SCGRegion>,
   524→    /// The next node ID counter.
   525→    next_node_id: u64,
   526→    /// The next edge ID counter.
   527→    next_edge_id: u64,
   528→}
   529→
   530→// ── Extract SCG into intermediate representation ────────────────────────
   531→
   532→fn scg_to_serialized(scg: &SCG) -> SerializedSCG {
   533→    let nodes: Vec<NodeData> = scg.nodes().cloned().collect();
   534→    let edges: Vec<EdgeData> = scg.edges().cloned().collect();
   535→    let regions: Vec<SCGRegion> = scg.regions().cloned().collect();
   536→
   537→    // Compute the next_node_id and next_edge_id from the SCG.
   538→    // We derive these from the maximum observed IDs + 1, since SCG doesn't
   539→    // expose the counters directly.
   540→    let next_node_id = nodes.iter().map(|n| n.id.as_u64() + 1).max().unwrap_or(0);
   541→    let next_edge_id = edges.iter().map(|e| e.id.as_u64() + 1).max().unwrap_or(0);
   542→
   543→    SerializedSCG {
   544→        version: FORMAT_VERSION,
   545→        nodes,
   546→        edges,
   547→        regions,
   548→        next_node_id,
   549→        next_edge_id,
   550→    }
   551→}
   552→
   553→/// Reconstruct an SCG from the intermediate representation.
   554→fn serialized_to_scg(s: SerializedSCG) -> Result<SCG, DeserializeError> {
   555→    let mut scg = SCG::new();
   556→
   557→    // Add nodes first (using their pre-assigned IDs)
   558→    for node_data in &s.nodes {
   559→        scg.add_node_with_id(
   560→            node_data.id,
   561→            node_data.node_type.clone(),
   562→            node_data.payload.clone(),
   563→            node_data.program_point.clone(),
   564→        )
   565→        .map_err(|e| DeserializeError::ConsistencyError(e.to_string()))?;
   566→
   567→        // Restore annotation if present
   568→        if let Some(ref ann) = node_data.annotation {
   569→            if let Some(nd) = scg.get_node_mut(node_data.id) {
   570→                nd.annotation = Some(ann.clone());
   571→            }
   572→        }
   573→    }
   574→
   575→    // Add edges (using their pre-assigned IDs)
   576→    for edge_data in &s.edges {
   577→        scg.add_edge_with_id(
   578→            edge_data.id,
   579→            edge_data.source,
   580→            edge_data.target,
   581→            edge_data.kind.clone(),
   582→        )
   583→        .map_err(|e| DeserializeError::ConsistencyError(e.to_string()))?;
   584→
   585→        // Restore label if present
   586→        if let Some(ref label) = edge_data.label {
   587→            if let Some(ed) = scg.get_edge_mut(edge_data.id) {
   588→                ed.label = Some(label.clone());
   589→            }
   590→        }
   591→    }
   592→
   593→    // Add regions
   594→    for region in s.regions {
   595→        scg.add_region(region);
   596→    }
   597→
   598→    Ok(scg)
   599→}

   601→// ── Binary serialization ────────────────────────────────────────────────

   603→/// Serializes an SCG to a versioned binary format.
   604→///
   605→/// The binary format starts with a magic header (`VSCG`), a version number,
   606→/// and then the graph contents encoded as little-endian integers and
   607→/// length-prefixed strings.
   608→///
   609→/// # Example
   610→///
   611→/// ```
   612→/// use vuma_scg::*;
   613→/// let mut scg = SCG::new();
   614→/// let id = scg.add_node(
   615→///     NodeType::Computation,
   616→///     NodePayload::Computation(ComputationNode {
   617→///         operation: "add".to_string(),
   618→///         result_type: None,
   619→///         tail_call: false }),
   620→///     ProgramPoint { file: None, line: None, column: None, offset: None },
   621→/// );
   622→/// let bytes = vuma_scg::serialize::serialize_scg(&scg);
   623→/// assert!(&bytes[..4] == b"VSCG");
   624→/// ```
   625→pub fn serialize_scg(scg: &SCG) -> Vec<u8> {
   626→    let s = scg_to_serialized(scg);
   627→    let mut w = BinaryWriter::new();
   628→
   629→    // Header
   630→    w.write_bytes(MAGIC);
   631→    w.write_u32_le(FORMAT_VERSION);
   632→    w.write_u64_le(s.next_node_id);
   633→    w.write_u64_le(s.next_edge_id);
   634→    w.write_u32_le(s.nodes.len() as u32);
   635→    w.write_u32_le(s.edges.len() as u32);
   636→    w.write_u32_le(s.regions.len() as u32);
   637→
   638→    // Nodes
   639→    for node in &s.nodes {
   640→        write_node(&mut w, node);
   641→    }
   642→
   643→    // Edges
   644→    for edge in &s.edges {
   645→        write_edge(&mut w, edge);
   646→    }
   647→
   648→    // Regions
   649→    for region in &s.regions {
   650→        write_region(&mut w, region);
   651→    }
   652→
   653→    w.into_vec()
   654→}
   655→
   656→/// Deserializes an SCG from the versioned binary format.
   657→///
   658→/// Returns an error if the magic bytes don't match, the version is not
   659→/// supported, or the data is malformed.
   660→pub fn deserialize_scg(data: &[u8]) -> Result<SCG, DeserializeError> {
   661→    let mut reader = BinaryReader::new(data);
   662→
   663→    // Read and validate magic
   664→    let magic = reader.read_bytes(4, "magic")?;
   665→    if magic != MAGIC {
   666→        return Err(DeserializeError::InvalidMagic {
   667→            expected: MAGIC.to_vec(),
   668→            found: magic.to_vec(),
   669→        });
   670→    }
   671→
   672→    // Read and validate version
   673→    let version = reader.read_u32_le("version")?;
   674→    if version < MIN_SUPPORTED_VERSION {
   675→        return Err(DeserializeError::UnsupportedVersion {
   676→            version,
   677→            min_supported: MIN_SUPPORTED_VERSION,
   678→        });
   679→    }
   680→    // Forward compatibility: we could handle unknown future versions here
   681→    // by reading only the fields we understand. For now, we require
   682→    // version == FORMAT_VERSION for strictness, but allow future versions
   683→    // with a warning-style approach (we just try to read the v1 layout).
   684→
   685→    // Read counters and counts
   686→    let next_node_id = reader.read_u64_le("next_node_id")?;
   687→    let next_edge_id = reader.read_u64_le("next_edge_id")?;
   688→    let node_count = reader.read_u32_le("node_count")? as usize;
   689→    let edge_count = reader.read_u32_le("edge_count")? as usize;
   690→    let region_count = reader.read_u32_le("region_count")? as usize;
   691→
   692→    // Read nodes
   693→    let mut nodes = Vec::with_capacity(node_count);
   694→    for i in 0..node_count {
   695→        nodes.push(read_node(&mut reader, i)?);
   696→    }
   697→
   698→    // Read edges
   699→    let mut edges = Vec::with_capacity(edge_count);
   700→    for i in 0..edge_count {
   701→        edges.push(read_edge(&mut reader, i)?);
   702→    }
   703→
   704→    // Read regions
   705→    let mut regions = Vec::with_capacity(region_count);
   706→    for i in 0..region_count {
   707→        regions.push(read_region(&mut reader, i)?);
   708→    }
   709→
   710→    let serialized = SerializedSCG {
   711→        version,
   712→        nodes,
   713→        edges,
   714→        regions,
   715→        next_node_id,
   716→        next_edge_id,
   717→    };
   718→
   719→    serialized_to_scg(serialized)
   720→}

   724→fn write_node(w: &mut BinaryWriter, node: &NodeData) {
   725→    w.write_u64_le(node.id.as_u64());
   726→    w.write_u32_le(node_type_to_tag(&node.node_type));
   727→
   728→    // Annotation (BDReference)
   729→    write_optional_bd_reference(w, &node.annotation);
   730→
   731→    // Program point
   732→    write_program_point(w, &node.program_point);
   733→
   734→    // Payload
   735→    write_payload(w, &node.payload);
   736→}
   737→
   738→fn read_node(reader: &mut BinaryReader, index: usize) -> Result<NodeData, DeserializeError> {
   739→    let ctx = || format!("node[{}]", index);
   740→
   741→    let id = NodeId::new(reader.read_u64_le(&ctx())?);
   742→    let node_type = {
   743→        let tag = reader.read_u32_le(&format!("{}.node_type", ctx()))?;
   744→        tag_to_node_type(tag)?
   745→    };
   746→    let annotation = read_optional_bd_reference(reader, &format!("{}.annotation", ctx()))?;
   747→    let program_point = read_program_point(reader, &format!("{}.program_point", ctx()))?;
   748→    let payload = read_payload(reader, &node_type, &format!("{}.payload", ctx()))?;
   749→
   750→    Ok(NodeData {
   751→        id,
   752→        node_type,
   753→        annotation,
   754→        program_point,
   755→        payload,
   756→    })
   757→}
   758→
   759→fn write_optional_bd_reference(w: &mut BinaryWriter, opt: &Option<BDReference>) {
   760→    match opt {
   761→        Some(bd) => {
   762→            w.write_bool(true);
   763→            w.write_u64_le(bd.bd_id);
   764→            w.write_optional_u64(&bd.version);
   765→        }
   766→        None => {
   767→            w.write_bool(false);
   768→        }
   769→    }
   770→}
   771→
   772→fn read_optional_bd_reference(
   773→    reader: &mut BinaryReader,
   774→    context: &str,
   775→) -> Result<Option<BDReference>, DeserializeError> {
   776→    let has = reader.read_bool(&format!("{}.has", context))?;
   777→    if has {
   778→        let bd_id = reader.read_u64_le(&format!("{}.bd_id", context))?;
   779→        let version = reader.read_optional_u64(&format!("{}.version", context))?;
   780→        Ok(Some(BDReference { bd_id, version }))
   781→    } else {
   782→        Ok(None)
   783→    }
   784→}

   786→fn write_program_point(w: &mut BinaryWriter, pp: &ProgramPoint) {
   787→    w.write_optional_string(&pp.file);
   788→    w.write_optional_u64(&pp.line);
   789→    w.write_optional_u64(&pp.column);
   790→    w.write_optional_u64(&pp.offset);
   791→}

   793→fn read_program_point(
   794→    reader: &mut BinaryReader,
   795→    context: &str,
   796→) -> Result<ProgramPoint, DeserializeError> {
   797→    let file = reader.read_optional_string(&format!("{}.file", context))?;
   798→    let line = reader.read_optional_u64(&format!("{}.line", context))?;
   799→    let column = reader.read_optional_u64(&format!("{}.column", context))?;
   800→    let offset = reader.read_optional_u64(&format!("{}.offset", context))?;
   801→    Ok(ProgramPoint {
   802→        file,
   803→        line,
   804→        column,
   805→        offset,
   806→    })
   807→}

   809→fn write_payload(w: &mut BinaryWriter, payload: &NodePayload) {
   810→    match payload {
   811→        NodePayload::Computation(c) => {
   812→            w.write_u32_le(NODE_TYPE_COMPUTATION);
   813→            w.write_string(&c.operation);
   814→            w.write_optional_string(&c.result_type);
   815→        }
   816→        NodePayload::Allocation(a) => {
   817→            w.write_u32_le(NODE_TYPE_ALLOCATION);
   818→            w.write_u64_le(a.size);
   819→            w.write_u64_le(a.align);
   820→            w.write_u64_le(a.region_id.as_u64());
   821→            w.write_optional_string(&a.type_name);
   822→        }
   823→        NodePayload::Deallocation(d) => {
   824→            w.write_u32_le(NODE_TYPE_DEALLOCATION);
   825→            w.write_u64_le(d.allocation_node.as_u64());
   826→            w.write_u64_le(d.region_id.as_u64());
   827→        }
   828→        NodePayload::Access(a) => {
   829→            w.write_u32_le(NODE_TYPE_ACCESS);
   830→            w.write_u32_le(access_mode_to_tag(&a.mode));
   831→            w.write_u64_le(a.region_id.as_u64());
   832→            w.write_optional_u64(&a.offset);
   833→            w.write_optional_u64(&a.access_size);
   834→        }
   835→        NodePayload::Cast(c) => {
   836→            w.write_u32_le(NODE_TYPE_CAST);
   837→            w.write_string(&c.from_type);
   838→            w.write_string(&c.to_type);
   839→            w.write_bool(c.is_lossless);
   840→        }
   841→        NodePayload::Effect(e) => {
   842→            w.write_u32_le(NODE_TYPE_EFFECT);
   843→            w.write_string(&e.effect_kind);
   844→            w.write_bool(e.is_observable);
   845→        }
   846→        NodePayload::Control(c) => {
   847→            w.write_u32_le(NODE_TYPE_CONTROL);
   848→            w.write_u32_le(control_kind_to_tag(&c.kind));
   849→            w.write_optional_string(&c.label);
   850→        }
   851→        NodePayload::Phantom(p) => {
   852→            w.write_u32_le(NODE_TYPE_PHANTOM);
   853→            w.write_string(&p.purpose);
   854→        }
   855→        NodePayload::VTable(v) => {
   856→            w.write_u32_le(NODE_TYPE_VTABLE);
   857→            w.write_string(&v.trait_name);
   858→            w.write_string(&v.concrete_type);
   859→            w.write_u64_le(v.method_entries.len() as u64);
   860→            for &entry_id in &v.method_entries {
   861→                w.write_u64_le(entry_id.as_u64());
   862→            }
   863→        }
   864→        NodePayload::ClosureEnv(e) => {
   865→            w.write_u32_le(NODE_TYPE_CLOSURE_ENV);
   866→            w.write_u64_le(e.captured_vars.len() as u64);
   867→            for var in &e.captured_vars {
   868→                w.write_string(var);
   869→            }
   870→            w.write_u64_le(e.capture_modes.len() as u64);
   871→            for &mode in &e.capture_modes {
   872→                w.write_bool(mode);
   873→            }
   874→            w.write_optional_u64(&e.closure_entry.map(|id| id.as_u64()));
   875→        }
   876→    }
   877→}
   878→
   879→fn read_payload(
   880→    reader: &mut BinaryReader,
   881→    expected_type: &NodeType,
   882→    context: &str,
   883→) -> Result<NodePayload, DeserializeError> {
   884→    let tag = reader.read_u32_le(&format!("{}.tag", context))?;
   885→    let payload_type = tag_to_node_type(tag)?;
   886→
   887→    // Validate consistency between node_type and payload tag
   888→    if &payload_type != expected_type {
   889→        return Err(DeserializeError::ConsistencyError(format!(
   890→            "node_type {:?} does not match payload tag {:?}",
   891→            expected_type, payload_type
   892→        )));
   893→    }
   894→
   895→    match payload_type {
   896→        NodeType::Computation => {
   897→            let operation = reader.read_string(&format!("{}.operation", context))?;
   898→            let result_type = reader.read_optional_string(&format!("{}.result_type", context))?;
   899→            Ok(NodePayload::Computation(ComputationNode {
   900→                operation,
   901→                result_type,
   902→                tail_call: false,
   903→            }))
   904→        }
   905→        NodeType::Allocation => {
   906→            let size = reader.read_u64_le(&format!("{}.size", context))?;
   907→            let align = reader.read_u64_le(&format!("{}.align", context))?;
   908→            let region_id = RegionId::new(reader.read_u64_le(&format!("{}.region_id", context))?);
   909→            let type_name = reader.read_optional_string(&format!("{}.type_name", context))?;
   910→            Ok(NodePayload::Allocation(AllocationNode {
   911→                size,
   912→                align,
   913→                region_id,
   914→                type_name,
   915→            }))
   916→        }
   917→        NodeType::Deallocation => {
   918→            let allocation_node =
   919→                NodeId::new(reader.read_u64_le(&format!("{}.allocation_node", context))?);
   920→            let region_id = RegionId::new(reader.read_u64_le(&format!("{}.region_id", context))?);
   921→            Ok(NodePayload::Deallocation(DeallocationNode {
   922→                allocation_node,
   923→                region_id,
   924→            }))
   925→        }
   926→        NodeType::Access => {
   927→            let mode_tag = reader.read_u32_le(&format!("{}.mode", context))?;
   928→            let mode = tag_to_access_mode(mode_tag)?;
   929→            let region_id = RegionId::new(reader.read_u64_le(&format!("{}.region_id", context))?);
   930→            let offset = reader.read_optional_u64(&format!("{}.offset", context))?;
   931→            let access_size = reader.read_optional_u64(&format!("{}.access_size", context))?;
   932→            Ok(NodePayload::Access(AccessNode {
   933→                mode,
   934→                region_id,
   935→                offset,
   936→                access_size,
   937→            }))
   938→        }
   939→        NodeType::Cast => {
   940→            let from_type = reader.read_string(&format!("{}.from_type", context))?;
   941→            let to_type = reader.read_string(&format!("{}.to_type", context))?;
   942→            let is_lossless = reader.read_bool(&format!("{}.is_lossless", context))?;
   943→            Ok(NodePayload::Cast(CastNode {
   944→                from_type,
   945→                to_type,
   946→                is_lossless,
   947→            }))
   948→        }
   949→        NodeType::Effect => {
   950→            let effect_kind = reader.read_string(&format!("{}.effect_kind", context))?;
   951→            let is_observable = reader.read_bool(&format!("{}.is_observable", context))?;
   952→            Ok(NodePayload::Effect(EffectNode {
   953→                effect_kind,
   954→                is_observable,
   955→            }))
   956→        }
   957→        NodeType::Control => {
   958→            let kind_tag = reader.read_u32_le(&format!("{}.kind", context))?;
   959→            let kind = tag_to_control_kind(kind_tag)?;
   960→            let label = reader.read_optional_string(&format!("{}.label", context))?;
   961→            Ok(NodePayload::Control(ControlNode { kind, label }))
   962→        }
   963→        NodeType::Phantom => {
   964→            let purpose = reader.read_string(&format!("{}.purpose", context))?;
   965→            Ok(NodePayload::Phantom(PhantomNode { purpose }))
   966→        }
   967→        NodeType::VTable => {
   968→            let trait_name = reader.read_string(&format!("{}.trait_name", context))?;
   969→            let concrete_type = reader.read_string(&format!("{}.concrete_type", context))?;
   970→            let count = reader.read_u64_le(&format!("{}.method_count", context))?;
   971→            let mut method_entries = Vec::with_capacity(count as usize);
   972→            for i in 0..count as usize {
   973→                method_entries.push(NodeId::new(
   974→                    reader.read_u64_le(&format!("{}.method[{}]", context, i))?,
   975→                ));
   976→            }
   977→            Ok(NodePayload::VTable(VTableNode {
   978→                trait_name,
   979→                concrete_type,
   980→                method_entries,
   981→            }))
   982→        }
   983→        NodeType::ClosureEnv => {
   984→            let var_count = reader.read_u64_le(&format!("{}.var_count", context))?;
   985→            let mut captured_vars = Vec::with_capacity(var_count as usize);
   986→            for i in 0..var_count as usize {
   987→                captured_vars
   988→                    .push(reader.read_string(&format!("{}.captured_var[{}]", context, i))?);
   989→            }
   990→            let mode_count = reader.read_u64_le(&format!("{}.mode_count", context))?;
   991→            let mut capture_modes = Vec::with_capacity(mode_count as usize);
   992→            for i in 0..mode_count as usize {
   993→                capture_modes.push(reader.read_bool(&format!("{}.capture_mode[{}]", context, i))?);
   994→            }
   995→            let closure_entry = reader
   996→                .read_optional_u64(&format!("{}.closure_entry", context))?
   997→                .map(NodeId::new);
   998→            Ok(NodePayload::ClosureEnv(ClosureEnvNode {
   999→                captured_vars,
  1000→                capture_modes,
  1001→                closure_entry,
  1002→            }))
  1003→        }
  1004→    }
  1005→}

  1012→    w.write_u64_le(edge.target.as_u64());
  1013→    w.write_u32_le(edge_kind_to_tag(&edge.kind));
  1014→    w.write_optional_string(&edge.label);
  1015→}

  1017→fn read_edge(reader: &mut BinaryReader, index: usize) -> Result<EdgeData, DeserializeError> {

  1024→        let tag = reader.read_u32_le(&format!("{}.kind", ctx()))?;
  1025→        tag_to_edge_kind(tag)?
  1026→    };
  1027→    let label = reader.read_optional_string(&format!("{}.label", ctx()))?;
  1028→
  1029→    Ok(EdgeData {
  1030→        id,
  1031→        source,
  1032→        target,
  1033→        kind,

  1040→fn write_region(w: &mut BinaryWriter, region: &SCGRegion) {
  1041→    w.write_u64_le(region.id.as_u64());
  1042→    w.write_u32_le(region.nodes.len() as u32);
  1043→    for node_id in &region.nodes {
  1044→        w.write_u64_le(node_id.as_u64());
  1045→    }
  1046→    w.write_u32_le(region.scope_level);
  1047→    w.write_bool(region.security_boundary);
  1048→    w.write_u32_le(deployment_target_to_tag(&region.deployment_target));
  1049→    if let DeploymentTarget::Custom(ref name) = region.deployment_target {
  1050→        w.write_string(name);
  1051→    }
  1052→}

  1054→fn read_region(reader: &mut BinaryReader, index: usize) -> Result<SCGRegion, DeserializeError> {
  1055→    let ctx = || format!("region[{}]", index);
  1056→
  1057→    let id = RegionId::new(reader.read_u64_le(&ctx())?);
  1058→    let node_count = reader.read_u32_le(&format!("{}.node_count", ctx()))? as usize;
  1059→    let mut nodes = hashbrown::HashSet::with_capacity(node_count);
  1060→    for j in 0..node_count {
  1061→        let nid = NodeId::new(reader.read_u64_le(&format!("{}.nodes[{}]", ctx(), j))?);
  1062→        nodes.insert(nid);
  1063→    }
  1064→    let scope_level = reader.read_u32_le(&format!("{}.scope_level", ctx()))?;
  1065→    let security_boundary = reader.read_bool(&format!("{}.security_boundary", ctx()))?;
  1066→    let dt_tag = reader.read_u32_le(&format!("{}.deployment_target", ctx()))?;
  1067→    let deployment_target = tag_to_deployment_target(dt_tag, reader)?;
  1068→
  1069→    Ok(SCGRegion {
  1070→        id,
  1071→        nodes,
  1072→        scope_level,
  1073→        security_boundary,
  1074→        deployment_target,
  1075→    })
  1076→}

  1080→/// Serializes an SCG to a JSON string for debugging and interoperability.
  1081→///
  1082→/// The JSON representation uses the same intermediate format as the binary
  1083→/// serialization, but rendered as human-readable JSON.
  1084→pub fn serialize_scg_json(scg: &SCG) -> String {
  1085→    let s = scg_to_serialized(scg);
  1086→    serde_json::to_string_pretty(&s).unwrap_or_else(|e| {
  1087→        // Fallback: return an error object if serialization fails
  1088→        format!(r#"{{"error": "serialization failed: {}"}}"#, e)
  1089→    })
  1090→}

  1092→/// Deserializes an SCG from a JSON string.

  1094→/// Returns an error if the JSON is malformed or contains inconsistent data.
  1095→pub fn deserialize_scg_json(json: &str) -> Result<SCG, DeserializeError> {
  1096→    let s: SerializedSCG = serde_json::from_str(json)?;
  1097→    serialized_to_scg(s)
  1098→}

  1100→// ── DOT (Graphviz) serialization ────────────────────────────────────────
  1101→
  1102→/// Serializes an SCG to Graphviz DOT format for visualization.
  1103→///
  1104→/// Produces a directed graph (`digraph`) where:
  1105→/// - Nodes are rendered with labels showing their type and key payload info.
  1106→/// - Edges are styled by kind (solid for data flow, dashed for control flow,
  1107→///   dotted for derivation, bold for annotation).
  1108→/// - Regions are rendered as subgraph clusters.
  1109→///
  1110→/// # Example
  1111→///
  1112→/// ```
  1113→/// use vuma_scg::*;
  1114→/// let mut scg = SCG::new();
  1115→/// let n1 = scg.add_node(
  1116→///     NodeType::Computation,
  1117→///     NodePayload::Computation(ComputationNode {
  1118→///         operation: "add".to_string(),
  1119→///         result_type: None,
  1120→///         tail_call: false }),
  1121→///     ProgramPoint { file: None, line: None, column: None, offset: None },
  1122→/// );
  1123→/// let dot = vuma_scg::serialize::serialize_scg_dot(&scg);
  1124→/// assert!(dot.contains("digraph SCG"));
  1125→/// ```
  1126→pub fn serialize_scg_dot(scg: &SCG) -> String {
  1127→    let mut out = String::with_capacity(4096);
  1128→
  1129→    // Graph header
  1130→    out.push_str("digraph SCG {\n");
  1131→    out.push_str("    rankdir=TB;\n");
  1132→    out.push_str("    node [shape=record, fontname=\"monospace\"];\n");
  1133→    out.push_str("    edge [fontname=\"monospace\"];\n");
  1134→    out.push('\n');
  1135→
  1136→    // Group nodes by region for subgraph clustering
  1137→    let mut node_to_region: std::collections::HashMap<NodeId, RegionId> =
  1138→        std::collections::HashMap::new();
  1139→    for region in scg.regions() {
  1140→        for node_id in region.iter_nodes() {
  1141→            node_to_region.insert(*node_id, region.id);
  1142→        }
  1143→    }
  1144→
  1145→    // Render regions as subgraph clusters
  1146→    for region in scg.regions() {
  1147→        let cluster_name = format!("cluster_region_{}", region.id.as_u64());
  1148→        out.push_str(&format!(
  1149→            "    subgraph {} {{\n",
  1150→            dot_escape_id(&cluster_name)
  1151→        ));
  1152→        let boundary_label = if region.security_boundary {
  1153→            " [SECURITY BOUNDARY]"
  1154→        } else {
  1155→            ""
  1156→        };
  1157→        out.push_str(&format!(
  1158→            "        label=\"Region {} ({}){}\";\n",
  1159→            region.id.as_u64(),
  1160→            region.deployment_target,
  1161→            boundary_label
  1162→        ));
  1163→        out.push_str(&format!(
  1164→            "        style=dashed;\n        color={};\n",
  1165→            if region.security_boundary {
  1166→                "red"
  1167→            } else {
  1168→                "blue"
  1169→            }
  1170→        ));
  1171→
  1172→        for node_id in region.iter_nodes() {
  1173→            if let Some(node_data) = scg.get_node(*node_id) {
  1174→                out.push_str(&format!("        {}\n", format_dot_node(node_data)));
  1175→            }
  1176→        }
  1177→        out.push_str("    }\n\n");
  1178→    }
  1179→
  1180→    // Render nodes not in any region
  1181→    let mut unassigned: Vec<&NodeData> = Vec::new();
  1182→    for node_data in scg.nodes() {
  1183→        if !node_to_region.contains_key(&node_data.id) {
  1184→            unassigned.push(node_data);
  1185→        }
  1186→    }
  1187→    if !unassigned.is_empty() {
  1188→        out.push_str("    subgraph cluster_unassigned {\n");
  1189→        out.push_str("        label=\"Unassigned\";\n        style=dotted;\n        color=gray;\n");
  1190→        for node_data in &unassigned {
  1191→            out.push_str(&format!("        {}\n", format_dot_node(node_data)));
  1192→        }
  1193→        out.push_str("    }\n\n");
  1194→    }
  1195→
  1196→    // Render edges
  1197→    for edge_data in scg.edges() {
  1198→        let style = match edge_data.kind {
  1199→            EdgeKind::DataFlow => "solid",
  1200→            EdgeKind::ControlFlow => "dashed",
  1201→            EdgeKind::Derivation => "dotted",
  1202→            EdgeKind::Annotation => "bold",
  1203→            EdgeKind::Dispatch => "bold dashed",
  1204→            EdgeKind::Call { .. } => "bold solid",
  1205→            EdgeKind::Return { .. } => "bold dotted",
  1206→        };
  1207→        let color = match edge_data.kind {
  1208→            EdgeKind::DataFlow => "black",
  1209→            EdgeKind::ControlFlow => "blue",
  1210→            EdgeKind::Derivation => "gray",
  1211→            EdgeKind::Annotation => "purple",
  1212→            EdgeKind::Dispatch => "red",
  1213→            EdgeKind::Call { .. } => "green",
  1214→            EdgeKind::Return { .. } => "orange",
  1215→        };
  1216→        let kind_str = format!("{}", edge_data.kind);
  1217→        let label = edge_data.label.as_deref().unwrap_or(&kind_str);
  1218→        out.push_str(&format!(
  1219→            "    {} -> {} [style={}, color={}, label=\"{}\"];\n",
  1220→            dot_escape_id(&format!("n{}", edge_data.source.as_u64())),
  1221→            dot_escape_id(&format!("n{}", edge_data.target.as_u64())),
  1222→            style,
  1223→            color,
  1224→            dot_escape_label(label),
  1225→        ));
  1226→    }
  1227→
  1228→    out.push_str("}\n");
  1229→    out
  1230→}
  1231→
  1232→/// Formats a single node for DOT output.
  1233→fn format_dot_node(node: &NodeData) -> String {
  1234→    let label = format_node_label(node);
  1235→    format!(
  1236→        "{} [label=\"{}\"];",
  1237→        dot_escape_id(&format!("n{}", node.id.as_u64())),
  1238→        dot_escape_label(&label)
  1239→    )
  1240→}
  1241→
  1242→/// Creates a human-readable label for a node.
  1243→fn format_node_label(node: &NodeData) -> String {
  1244→    let type_info = match &node.payload {
  1245→        NodePayload::Computation(c) => {
  1246→            let rt = c
  1247→                .result_type
  1248→                .as_deref()
  1249→                .map(|t| format!(":{}", t))
  1250→                .unwrap_or_default();
  1251→            format!("{}{}", c.operation, rt)
  1252→        }
  1253→        NodePayload::Allocation(a) => {
  1254→            let tn = a
  1255→                .type_name
  1256→                .as_deref()
  1257→                .map(|t| format!(" {}", t))
  1258→                .unwrap_or_default();
  1259→            format!("alloc {}B align={}{}", a.size, a.align, tn)
  1260→        }
  1261→        NodePayload::Deallocation(_) => "dealloc".to_string(),
  1262→        NodePayload::Access(a) => {
  1263→            let offset = a.offset.map(|o| format!("+{}", o)).unwrap_or_default();
  1264→            format!("{:?}{} @r{}", a.mode, offset, a.region_id.as_u64())
  1265→        }
  1266→        NodePayload::Cast(c) => format!("{} -> {}", c.from_type, c.to_type),
  1267→        NodePayload::Effect(e) => format!("eff({})", e.effect_kind),
  1268→        NodePayload::Control(c) => {
  1269→            let lbl = c
  1270→                .label
  1271→                .as_deref()
  1272→                .map(|l| format!(" {}", l))
  1273→                .unwrap_or_default();
  1274→            format!("{:?}{}", c.kind, lbl)
  1275→        }
  1276→        NodePayload::Phantom(p) => format!("phantom({})", p.purpose),
  1277→        NodePayload::VTable(v) => format!("vtable({} for {})", v.trait_name, v.concrete_type),
  1278→        NodePayload::ClosureEnv(e) => format!("closure_env({:?})", e.captured_vars),
  1279→    };
  1280→
  1281→    let ann = node
  1282→        .annotation
  1283→        .as_ref()
  1284→        .map(|bd| format!(" [BD:{}]", bd.bd_id))
  1285→        .unwrap_or_default();
  1286→
  1287→    format!(
  1288→        "n{}: {}\\n{}{}",
  1289→        node.id.as_u64(),
  1290→        node.node_type,
  1291→        type_info,
  1292→        ann
  1293→    )
  1294→}
  1295→
  1296→/// Escapes a string for use as a DOT identifier.
  1297→fn dot_escape_id(s: &str) -> String {
  1298→    s.replace('-', "_")
  1299→}
  1300→
  1301→/// Escapes a string for use as a DOT label (inside double quotes).
  1302→fn dot_escape_label(s: &str) -> String {
  1303→    s.replace('\\', "\\\\")
  1304→        .replace('"', "\\\"")
  1305→        .replace('\n', "\\n")
  1306→        .replace('{', "\\{")
  1307→        .replace('}', "\\}")
  1308→        .replace('<', "\\<")
  1309→        .replace('>', "\\>")
  1310→        .replace('|', "\\|")
  1311→}
