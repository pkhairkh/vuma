     1→//! Profile-guided optimization engine for the Continuous Optimization Runtime.
     2→//!
     3→//! This module defines the [`OptimizationEngine`] and the [`OptimizationPass`]
     4→//! trait, along with four concrete passes that apply profile-guided
     5→//! transformations to the Semantic Computation Graph (SCG):
     6→//!
     7→//! - [`HotPathInlining`] — inlines frequently-called function nodes.
     8→//! - [`ColdPathOutline`] — moves rarely-executed code to separate out-of-line
     9→//!   functions, improving instruction-cache utilisation on hot paths.
    10→//! - [`LoopOptimization`] — unrolls hot loops and, where possible, emits
    11→//!   SIMD/vectorized instructions.
    12→//! - [`MemoryOptimization`] — inserts prefetch hints and aligns data to
    13→//!   cache-line boundaries, targeting the Raspberry Pi 5 L1/L2 hierarchy.
    14→//!
    15→//! The top-level function [`apply_optimizations`] runs all registered passes
    16→//! over the SCG and returns an [`OptimizationResult`] summarising what was
    17→//! done.
    18→
    19→use crate::config::{Config, TargetArch};
    20→use crate::profile::ProfileData;
    21→use crate::types::{EdgeId, NodeId, NodeKind, SCG};
    22→use std::collections::HashMap;
    23→
    24→// ---------------------------------------------------------------------------
    25→// ProfileReport — digest of profile data for the optimiser
    26→// ---------------------------------------------------------------------------
    27→
    28→/// A summarised profile report consumed by optimisation passes.
    29→///
    30→/// `ProfileReport` is constructed from the raw [`ProfileData`] and provides
    31→/// pre-classified hot/cold node lists, loop back-edge frequencies, and
    32→/// allocation hotspots so that each pass does not have to recompute these
    33→/// from scratch.
    34→#[derive(Debug, Clone)]
    35→pub struct ProfileReport {
    36→    /// Nodes sorted by call count, descending (hottest first).
    37→    pub hot_nodes: Vec<(NodeId, u64)>,
    38→    /// Nodes whose call count is below the cold threshold.
    39→    pub cold_nodes: Vec<(NodeId, u64)>,
    40→    /// Loop back-edges and their traversal counts.
    41→    /// Each entry is `(edge_id, source_node, target_node, weight)`.
    42→    pub loop_back_edges: Vec<(EdgeId, NodeId, NodeId, u64)>,
    43→    /// Allocation hotspots by region class.
    44→    pub allocation_hotspots: Vec<(String, u64)>,
    45→    /// Total number of profile samples recorded.
    46→    pub total_samples: u64,
    47→}
    48→
    49→impl ProfileReport {
    50→    /// Hotness threshold: nodes called more than this are "hot".
    51→    const HOT_THRESHOLD: u64 = 100;
    52→    /// Coldness threshold: nodes called fewer than this are "cold".
    53→    const COLD_THRESHOLD: u64 = 5;
    54→
    55→    /// Builds a `ProfileReport` from raw [`ProfileData`].
    56→    ///
    57→    /// The constructor sorts nodes into hot/cold buckets and identifies loop
    58→    /// back-edges (edges whose `weight` significantly exceeds the average
    59→    /// edge weight in the SCG).
    60→    pub fn from_profile_data(profile: &ProfileData, scg: &SCG) -> Self {
    61→        let mut all_nodes: Vec<(NodeId, u64)> =
    62→            profile.call_counts.iter().map(|(&n, &c)| (n, c)).collect();
    63→        all_nodes.sort_by_key(|b| std::cmp::Reverse(b.1));
    64→
    65→        let hot_nodes: Vec<(NodeId, u64)> = all_nodes
    66→            .iter()
    67→            .filter(|(_, c)| *c > Self::HOT_THRESHOLD)
    68→            .cloned()
    69→            .collect();
    70→
    71→        let cold_nodes: Vec<(NodeId, u64)> = all_nodes
    72→            .iter()
    73→            .filter(|(_, c)| *c < Self::COLD_THRESHOLD)
    74→            .cloned()
    75→            .collect();
    76→
    77→        // Identify loop back-edges: edges where source > target (simple
    78→        // heuristic) or where the weight is at least 10× the median.
    79→        let avg_weight = if scg.edges.is_empty() {
    80→            1
    81→        } else {
    82→            let total: u64 = scg.edges.values().map(|e| e.weight).sum();
    83→            total / scg.edges.len().max(1) as u64
    84→        };
    85→
    86→        let loop_back_edges: Vec<(EdgeId, NodeId, NodeId, u64)> = scg
    87→            .edges
    88→            .values()
    89→            .filter(|e| e.weight > avg_weight * 10 || e.target <= e.source)
    90→            .map(|e| (e.id, e.source, e.target, e.weight))
    91→            .collect();
    92→
    93→        let allocation_hotspots: Vec<(String, u64)> = profile
    94→            .allocation_stats
    95→            .by_region_class
    96→            .iter()
    97→            .filter(|(_, count)| **count > 50)
    98→            .map(|(k, &v)| (k.clone(), v))
    99→            .collect();
   100→
   101→        let total_samples: u64 = profile.call_counts.values().sum();
   102→
   103→        ProfileReport {
   104→            hot_nodes,
   105→            cold_nodes,
   106→            loop_back_edges,
   107→            allocation_hotspots,
   108→            total_samples,
   109→        }
   110→    }
   111→
   112→    /// Returns `true` if the given node is classified as hot.
   113→    pub fn is_hot(&self, node_id: NodeId) -> bool {
   114→        self.hot_nodes.iter().any(|(n, _)| *n == node_id)
   115→    }
   116→
   117→    /// Returns `true` if the given node is classified as cold.
   118→    pub fn is_cold(&self, node_id: NodeId) -> bool {
   119→        self.cold_nodes.iter().any(|(n, _)| *n == node_id)
   120→    }
   121→
   122→    /// Returns the call count for the given node, or 0 if not profiled.
   123→    pub fn call_count(&self, node_id: NodeId) -> u64 {
   124→        self.hot_nodes
   125→            .iter()
   126→            .chain(self.cold_nodes.iter())
   127→            .find(|(n, _)| *n == node_id)
   128→            .map(|(_, c)| *c)
   129→            .unwrap_or(0)
   130→    }
   131→}

   133→// ---------------------------------------------------------------------------
   134→// Transformation record
   135→// ---------------------------------------------------------------------------
   136→
   137→/// The kind of optimisation transformation applied to a node.
   138→#[derive(Debug, Clone, PartialEq, Eq)]
   139→pub enum TransformationKind {
   140→    /// The node was inlined at its call site(s).
   141→    Inlined,
   142→    /// The node was outlined to a separate cold function.
   143→    Outlined,
   144→    /// A loop was unrolled by the given factor.
   145→    LoopUnrolled {
   146→        /// Unroll factor applied.
   147→        factor: u32,
   148→    },
   149→    /// A loop was vectorized using SIMD instructions.
   150→    LoopVectorized,
   151→    /// A prefetch instruction was inserted before a memory access.
   152→    PrefetchInserted,
   153→    /// Data was aligned to a cache-line boundary.
   154→    CacheLineAligned {
   155→        /// Alignment in bytes.
   156→        alignment: usize,
   157→    },
   158→}
   159→
   160→impl std::fmt::Display for TransformationKind {
   161→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   162→        match self {
   163→            TransformationKind::Inlined => write!(f, "Inlined"),
   164→            TransformationKind::Outlined => write!(f, "Outlined"),
   165→            TransformationKind::LoopUnrolled { factor } => {
   166→                write!(f, "LoopUnrolled(×{})", factor)
   167→            }
   168→            TransformationKind::LoopVectorized => write!(f, "LoopVectorized"),
   169→            TransformationKind::PrefetchInserted => write!(f, "PrefetchInserted"),
   170→            TransformationKind::CacheLineAligned { alignment } => {
   171→                write!(f, "CacheLineAligned({}B)", alignment)
   172→            }
   173→        }
   174→    }
   175→}
   176→
   177→/// A record of a single optimisation applied to a specific node.
   178→#[derive(Debug, Clone)]
   179→pub struct Transformation {
   180→    /// The kind of transformation.
   181→    pub kind: TransformationKind,
   182→    /// The node that was transformed.
   183→    pub target_node: NodeId,
   184→    /// Human-readable explanation of why the transformation was applied.
   185→    pub description: String,
   186→}

   188→// ---------------------------------------------------------------------------
   189→// PassResult — output of a single optimisation pass
   190→// ---------------------------------------------------------------------------
   191→
   192→/// The result of running one [`OptimizationPass`].
   193→#[derive(Debug, Clone)]
   194→pub struct PassResult {
   195→    /// Name of the pass that produced this result.
   196→    pub pass_name: String,
   197→    /// Individual transformations applied.
   198→    pub transformations: Vec<Transformation>,
   199→    /// Estimated speedup factor (1.0 = no improvement).
   200→    pub estimated_speedup: f64,
   201→}
   202→
   203→impl PassResult {
   204→    /// Creates an empty result for the given pass name.
   205→    pub fn empty(pass_name: &str) -> Self {
   206→        PassResult {
   207→            pass_name: pass_name.to_owned(),
   208→            transformations: Vec::new(),
   209→            estimated_speedup: 1.0,
   210→        }
   211→    }
   212→
   213→    /// Returns the number of transformations applied.
   214→    pub fn count(&self) -> usize {
   215→        self.transformations.len()
   216→    }
   217→}

   219→// ---------------------------------------------------------------------------
   220→// OptimizationResult — aggregate output of all passes
   221→// ---------------------------------------------------------------------------
   222→
   223→/// The aggregate result of applying all optimisation passes.
   224→#[derive(Debug, Clone)]
   225→pub struct OptimizationResult {
   226→    /// Per-pass results, in execution order.
   227→    pub pass_results: Vec<PassResult>,
   228→    /// Total number of transformations across all passes.
   229→    pub total_transformations: usize,
   230→    /// Combined estimated speedup factor (1.0 = no improvement).
   231→    pub estimated_speedup: f64,
   232→}
   233→
   234→impl OptimizationResult {
   235→    /// Creates a result from the list of per-pass results.
   236→    pub fn from_pass_results(pass_results: Vec<PassResult>) -> Self {
   237→        let total_transformations: usize = pass_results.iter().map(|r| r.count()).sum();
   238→        // Combine speedups multiplicatively (a simplification).
   239→        let estimated_speedup: f64 = pass_results
   240→            .iter()
   241→            .fold(1.0, |acc, r| acc * r.estimated_speedup);
   242→
   243→        OptimizationResult {
   244→            pass_results,
   245→            total_transformations,
   246→            estimated_speedup,
   247→        }
   248→    }
   249→
   250→    /// Returns an empty result (no passes run, no transformations).
   251→    pub fn empty() -> Self {
   252→        OptimizationResult {
   253→            pass_results: Vec::new(),
   254→            total_transformations: 0,
   255→            estimated_speedup: 1.0,
   256→        }
   257→    }
   258→}

   260→// ---------------------------------------------------------------------------
   261→// OptimizationPass trait
   262→// ---------------------------------------------------------------------------
   263→
   264→/// A single optimisation pass that can be applied to the SCG.
   265→///
   266→/// Implementations inspect the profile report, identify optimisation
   267→/// opportunities, and mutate the SCG accordingly. Each pass returns a
   268→/// [`PassResult`] describing what was done.
   269→pub trait OptimizationPass {
   270→    /// Human-readable name of the pass (used in logging and results).
   271→    fn name(&self) -> &str;
   272→
   273→    /// Apply the pass to the SCG, guided by the profile report.
   274→    ///
   275→    /// Returns a [`PassResult`] describing transformations applied and an
   276→    /// estimated speedup factor.
   277→    fn apply(&self, scg: &mut SCG, profile: &ProfileReport) -> PassResult;
   278→}

   280→// ---------------------------------------------------------------------------
   281→// HotPathInlining
   282→// ---------------------------------------------------------------------------
   283→
   284→/// Inlines hot call sites into their callers.
   285→///
   286→/// When a [`Call`](NodeKind::Call) node is identified as hot by the profile
   287→/// report and its compiled code is small enough (below `max_inline_size`),
   288→/// this pass marks the node as inlined, eliminating the call overhead and
   289→/// enabling further intra-procedural optimisations.
   290→pub struct HotPathInlining {
   291→    /// Maximum code size (in bytes) of a callee that is safe to inline.
   292→    /// Larger functions are not inlined to avoid code bloat.
   293→    pub max_inline_size: usize,
   294→}
   295→
   296→impl HotPathInlining {
   297→    /// Default maximum inline size: 256 bytes (roughly 64 AArch64 instructions).
   298→    pub const DEFAULT_MAX_INLINE_SIZE: usize = 256;
   299→
   300→    /// Creates a new `HotPathInlining` pass with default settings.
   301→    pub fn new() -> Self {
   302→        HotPathInlining {
   303→            max_inline_size: Self::DEFAULT_MAX_INLINE_SIZE,
   304→        }
   305→    }
   306→
   307→    /// Creates a new `HotPathInlining` pass with a custom max inline size.
   308→    pub fn with_max_inline_size(mut self, size: usize) -> Self {
   309→        self.max_inline_size = size;
   310→        self
   311→    }
   312→}
   313→
   314→impl Default for HotPathInlining {
   315→    fn default() -> Self {
   316→        Self::new()
   317→    }
   318→}
   319→
   320→impl OptimizationPass for HotPathInlining {
   321→    fn name(&self) -> &str {
   322→        "HotPathInlining"
   323→    }
   324→
   325→    fn apply(&self, scg: &mut SCG, profile: &ProfileReport) -> PassResult {
   326→        let mut result = PassResult::empty(self.name());
   327→
   328→        for &(node_id, call_count) in &profile.hot_nodes {
   329→            let node = match scg.get_node_mut(node_id) {
   330→                Some(n) => n,
   331→                None => continue,
   332→            };
   333→
   334→            // Only inline Call nodes that haven't already been inlined.
   335→            if node.kind != NodeKind::Call || node.is_inlined {
   336→                continue;
   337→            }
   338→
   339→            // Only inline if the callee is small enough.
   340→            if node.code_size > self.max_inline_size {
   341→                log::debug!(
   342→                    "HotPathInlining: skipping node {} — code_size {} exceeds limit {}",
   343→                    node_id,
   344→                    node.code_size,
   345→                    self.max_inline_size,
   346→                );
   347→                continue;
   348→            }
   349→
   350→            let saved_call_overhead = call_count as f64 * 5.0; // ~5 cycles per call
   351→            result.estimated_speedup += saved_call_overhead / profile.total_samples.max(1) as f64;
   352→
   353→            node.is_inlined = true;
   354→            result.transformations.push(Transformation {
   355→                kind: TransformationKind::Inlined,
   356→                target_node: node_id,
   357→                description: format!(
   358→                    "Inlined hot call node (called {}×, code_size={}B)",
   359→                    call_count, node.code_size,
   360→                ),
   361→            });
   362→        }
   363→
   364→        // Ensure speedup is at least 1.0 (no regression).
   365→        result.estimated_speedup = result.estimated_speedup.max(1.0);
   366→
   367→        log::info!(
   368→            "HotPathInlining: {} transformations, estimated speedup {:.3}×",
   369→            result.count(),
   370→            result.estimated_speedup,
   371→        );
   372→
   373→        result
   374→    }
   375→}

   377→// ---------------------------------------------------------------------------
   378→// ColdPathOutline
   379→// ---------------------------------------------------------------------------
   380→
   381→/// Moves cold code to separate out-of-line functions.
   382→///
   383→/// When a node is classified as cold by the profile report and resides on
   384→/// a path that also contains hot nodes, outlining it shrinks the hot path
   385→/// and improves instruction-cache utilisation.
   386→pub struct ColdPathOutline {
   387→    /// Whether to also outline nodes that share an edge with a hot node
   388→    /// even if they are not directly cold themselves (rarely-taken branches).
   389→    pub outline_adjacent_to_hot: bool,
   390→}
   391→
   392→impl ColdPathOutline {
   393→    /// Creates a new `ColdPathOutline` pass with default settings.
   394→    pub fn new() -> Self {
   395→        ColdPathOutline {
   396→            outline_adjacent_to_hot: true,
   397→        }
   398→    }
   399→}
   400→
   401→impl Default for ColdPathOutline {
   402→    fn default() -> Self {
   403→        Self::new()
   404→    }
   405→}
   406→
   407→impl OptimizationPass for ColdPathOutline {
   408→    fn name(&self) -> &str {
   409→        "ColdPathOutline"
   410→    }
   411→
   412→    fn apply(&self, scg: &mut SCG, profile: &ProfileReport) -> PassResult {
   413→        let mut result = PassResult::empty(self.name());
   414→
   415→        // Collect node IDs to outline (we cannot borrow scg mutably while
   416→        // iterating over profile.cold_nodes which may reference nodes not
   417→        // in the graph, so we collect first and mutate second).
   418→        let mut to_outline: Vec<(NodeId, String)> = Vec::new();
   419→
   420→        for &(node_id, call_count) in &profile.cold_nodes {
   421→            let node = match scg.get_node(node_id) {
   422→                Some(n) => n,
   423→                None => continue,
   424→            };
   425→
   426→            // Only outline nodes that are on the same path as hot nodes.
   427→            let adjacent_to_hot = if self.outline_adjacent_to_hot {
   428→                node.incoming_edges
   429→                    .iter()
   430→                    .chain(node.outgoing_edges.iter())
   431→                    .any(|&eid| {
   432→                        scg.get_edge(eid)
   433→                            .is_some_and(|e| profile.is_hot(e.source) || profile.is_hot(e.target))
   434→                    })
   435→            } else {
   436→                false
   437→            };
   438→
   439→            if !adjacent_to_hot && !profile.is_hot(node_id) {
   440→                // Cold node not adjacent to any hot node — less benefit.
   441→                continue;
   442→            }
   443→
   444→            to_outline.push((
   445→                node_id,
   446→                format!(
   447→                    "Outlined cold node (called {}×, adjacent to hot path)",
   448→                    call_count,
   449→                ),
   450→            ));
   451→        }
   452→
   453→        // Also outline Branch nodes on cold paths adjacent to hot nodes.
   454→        if self.outline_adjacent_to_hot {
   455→            for &node_id in scg.nodes.keys() {
   456→                let node = match scg.get_node(node_id) {
   457→                    Some(n) => n,
   458→                    None => continue,
   459→                };
   460→                if node.kind != NodeKind::Branch || node.is_outlined {
   461→                    continue;
   462→                }
   463→                // Check if this branch is adjacent to a hot node but is itself cold.
   464→                let adjacent_hot = node
   465→                    .incoming_edges
   466→                    .iter()
   467→                    .chain(node.outgoing_edges.iter())
   468→                    .any(|&eid| {
   469→                        scg.get_edge(eid)
   470→                            .is_some_and(|e| profile.is_hot(e.source) || profile.is_hot(e.target))
   471→                    });
   472→                if adjacent_hot && profile.is_cold(node_id) {
   473→                    to_outline.push((
   474→                        node_id,
   475→                        "Outlined cold branch adjacent to hot path".to_owned(),
   476→                    ));
   477→                }
   478→            }
   479→        }
   480→
   481→        for (node_id, description) in to_outline {
   482→            if let Some(node) = scg.get_node_mut(node_id) {
   483→                node.is_outlined = true;
   484→                result.transformations.push(Transformation {
   485→                    kind: TransformationKind::Outlined,
   486→                    target_node: node_id,
   487→                    description,
   488→                });
   489→            }
   490→        }
   491→
   492→        // Estimated speedup: each outlined node frees icache space.
   493→        // Rough model: 2% per outlined node, capped at 20%.
   494→        let outline_benefit = (result.count() as f64 * 0.02).min(0.20);
   495→        result.estimated_speedup = 1.0 + outline_benefit;
   496→
   497→        log::info!(
   498→            "ColdPathOutline: {} transformations, estimated speedup {:.3}×",
   499→            result.count(),
   500→            result.estimated_speedup,
   501→        );
   502→
   503→        result
   504→    }
   505→}

   507→// ---------------------------------------------------------------------------
   508→// LoopOptimization
   509→// ---------------------------------------------------------------------------
   510→
   511→/// Unrolls hot loops and, where profitable, vectorises them.
   512→///
   513→/// Loop unrolling reduces branch overhead and exposes instruction-level
   514→/// parallelism. Vectorization replaces scalar loop bodies with SIMD
   515→/// operations, which is especially effective on the Cortex-A76 (Pi 5)
   516→/// with its 128-bit NEON unit.
   517→pub struct LoopOptimization {
   518→    /// Minimum trip count (from profile) to consider a loop "hot" enough
   519→    /// for unrolling.
   520→    pub min_trip_count: u64,
   521→    /// Maximum unroll factor to apply.
   522→    pub max_unroll_factor: u32,
   523→    /// Whether to attempt vectorisation.
   524→    pub enable_vectorization: bool,
   525→}
   526→
   527→impl LoopOptimization {
   528→    /// Creates a new `LoopOptimization` pass with default settings.
   529→    pub fn new() -> Self {
   530→        LoopOptimization {
   531→            min_trip_count: 100,
   532→            max_unroll_factor: 8,
   533→            enable_vectorization: true,
   534→        }
   535→    }
   536→
   537→    /// Creates a pass with a custom minimum trip count.
   538→    pub fn with_min_trip_count(mut self, count: u64) -> Self {
   539→        self.min_trip_count = count;
   540→        self
   541→    }
   542→
   543→    /// Determines the best unroll factor for a loop with the given trip
   544→    /// count. Chooses the largest power-of-two ≤ `max_unroll_factor` that
   545→    /// does not exceed the trip count.
   546→    fn best_unroll_factor(&self, trip_count: u64) -> u32 {
   547→        let mut factor: u32 = 1;
   548→        let mut candidate = 2;
   549→        while candidate <= self.max_unroll_factor && (candidate as u64) <= trip_count {
   550→            factor = candidate;
   551→            candidate *= 2;
   552→        }
   553→        factor
   554→    }
   555→
   556→    /// Heuristic: determines whether a loop body is vectorizable.
   557→    ///
   558→    /// In the full implementation this would analyse data dependencies,
   559→    /// stride patterns, and type widths. For now we use a simple heuristic:
   560→    /// a loop is vectorizable if it has a Memory node among its successors
   561→    /// and is not already vectorized.
   562→    fn is_vectorizable(&self, scg: &SCG, loop_node_id: NodeId) -> bool {
   563→        let loop_node = match scg.get_node(loop_node_id) {
   564→            Some(n) => n,
   565→            None => return false,
   566→        };
   567→        if loop_node.is_vectorized {
   568→            return false;
   569→        }
   570→        // Check if any successor reachable through outgoing edges is a
   571→        // Memory node (indicating a load/store pattern).
   572→        let mut has_memory = false;
   573→        for &eid in &loop_node.outgoing_edges {
   574→            if let Some(edge) = scg.get_edge(eid) {
   575→                if let Some(target) = scg.get_node(edge.target) {
   576→                    if target.kind == NodeKind::Memory {
   577→                        has_memory = true;
   578→                        break;
   579→                    }
   580→                }
   581→            }
   582→        }
   583→        has_memory
   584→    }
   585→}
   586→
   587→impl Default for LoopOptimization {
   588→    fn default() -> Self {
   589→        Self::new()
   590→    }
   591→}
   592→
   593→impl OptimizationPass for LoopOptimization {
   594→    fn name(&self) -> &str {
   595→        "LoopOptimization"
   596→    }
   597→
   598→    fn apply(&self, scg: &mut SCG, profile: &ProfileReport) -> PassResult {
   599→        let mut result = PassResult::empty(self.name());
   600→
   601→        // Build a map from loop-header node ID to estimated trip count
   602→        // derived from back-edge weights.
   603→        let mut loop_trip_counts: HashMap<NodeId, u64> = HashMap::new();
   604→        for &(_eid, _src, target, weight) in &profile.loop_back_edges {
   605→            let entry = loop_trip_counts.entry(target).or_insert(0);
   606→            *entry = (*entry).max(weight);
   607→        }
   608→
   609→        // Also consider Loop-kind nodes that are hot in the profile.
   610→        for &(node_id, call_count) in &profile.hot_nodes {
   611→            let node = match scg.get_node(node_id) {
   612→                Some(n) => n,
   613→                None => continue,
   614→            };
   615→            if node.kind == NodeKind::Loop || node.kind == NodeKind::LoopHeader {
   616→                let entry = loop_trip_counts.entry(node_id).or_insert(0);
   617→                *entry = (*entry).max(call_count);
   618→            }
   619→        }
   620→
   621→        // Apply unrolling and vectorisation.
   622→        let loop_nodes: Vec<NodeId> = loop_trip_counts.keys().cloned().collect();
   623→        for loop_node_id in loop_nodes {
   624→            let trip_count = loop_trip_counts[&loop_node_id];
   625→            if trip_count < self.min_trip_count {
   626→                continue;
   627→            }
   628→
   629→            let node = match scg.get_node_mut(loop_node_id) {
   630→                Some(n) => n,
   631→                None => continue,
   632→            };
   633→
   634→            if node.kind != NodeKind::Loop && node.kind != NodeKind::LoopHeader {
   635→                continue;
   636→            }
   637→
   638→            // Unroll.
   639→            let factor = self.best_unroll_factor(trip_count);
   640→            if factor > 1 && node.unroll_factor < factor {
   641→                let old_factor = node.unroll_factor;
   642→                node.unroll_factor = factor;
   643→                result.transformations.push(Transformation {
   644→                    kind: TransformationKind::LoopUnrolled { factor },
   645→                    target_node: loop_node_id,
   646→                    description: format!(
   647→                        "Unrolled loop (trip={}, factor {}→{})",
   648→                        trip_count, old_factor, factor,
   649→                    ),
   650→                });
   651→            }
   652→
   653→            // Vectorise (requires mutable borrow again, but we already
   654→            // released the first one — we need to reborrow).
   655→            if self.enable_vectorization && self.is_vectorizable(scg, loop_node_id) {
   656→                let node = scg.get_node_mut(loop_node_id).unwrap();
   657→                if !node.is_vectorized {
   658→                    node.is_vectorized = true;
   659→                    result.transformations.push(Transformation {
   660→                        kind: TransformationKind::LoopVectorized,
   661→                        target_node: loop_node_id,
   662→                        description: format!("Vectorized loop (trip={}, NEON/SIMD)", trip_count,),
   663→                    });
   664→                }
   665→            }
   666→        }
   667→
   668→        // Speedup model: unrolling gives ~factor/2 improvement up to 2×;
   669→        // vectorisation gives ~4× for memory-bound loops.
   670→        let unroll_speedup: f64 = result
   671→            .transformations
   672→            .iter()
   673→            .filter(|t| matches!(t.kind, TransformationKind::LoopUnrolled { .. }))
   674→            .map(|t| {
   675→                if let TransformationKind::LoopUnrolled { factor } = t.kind {
   676→                    (factor as f64 / 2.0).min(2.0)
   677→                } else {
   678→                    1.0
   679→                }
   680→            })
   681→            .product::<f64>()
   682→            .max(1.0);
   683→
   684→        let vec_count = result
   685→            .transformations
   686→            .iter()
   687→            .filter(|t| matches!(t.kind, TransformationKind::LoopVectorized))
   688→            .count();
   689→
   690→        // Each vectorised loop adds ~4× speedup for that loop body.
   691→        let vec_speedup = if vec_count > 0 {
   692→            1.0 + vec_count as f64 * 0.3
   693→        } else {
   694→            1.0
   695→        };
   696→
   697→        result.estimated_speedup = (unroll_speedup * vec_speedup).max(1.0);
   698→
   699→        log::info!(
   700→            "LoopOptimization: {} transformations, estimated speedup {:.3}×",
   701→            result.count(),
   702→            result.estimated_speedup,
   703→        );
   704→
   705→        result
   706→    }
   707→}

   709→// ---------------------------------------------------------------------------
   710→// MemoryOptimization
   711→// ---------------------------------------------------------------------------
   712→
   713→/// Inserts prefetch hints and aligns data to cache-line boundaries.
   714→///
   715→/// On the Raspberry Pi 5 (BCM2712, Cortex-A76):
   716→///
   717→/// - **L1 data cache**: 64 KB per core, 64-byte cache lines, 4-way set
   718→///   associative.
   719→/// - **L2 cache**: 512 KB shared (per core pair), 64-byte cache lines.
   720→///
   721→/// This pass:
   722→///
   723→/// 1. Marks [`Memory`](NodeKind::Memory) nodes on hot paths with prefetch
   724→///    instructions so data is loaded into L1 before it is needed.
   725→/// 2. Aligns memory accesses to 64-byte boundaries to avoid cross-cache-line
   726→///    loads that waste bandwidth.
   727→pub struct MemoryOptimization {
   728→    /// Cache-line size in bytes. Default: 64 (Pi 5 / Cortex-A76).
   729→    pub cache_line_size: usize,
   730→    /// L1 data cache size in bytes. Default: 65_536 (64 KB).
   731→    pub l1d_size: usize,
   732→    /// L2 cache size in bytes. Default: 524_288 (512 KB).
   733→    pub l2_size: usize,
   734→    /// Whether to target the Pi 5 specifically.
   735→    pub target_pi5: bool,
   736→}
   737→
   738→impl MemoryOptimization {
   739→    /// Creates a new `MemoryOptimization` pass targeting the given architecture.
   740→    pub fn new(target_arch: TargetArch) -> Self {
   741→        let target_pi5 = matches!(target_arch, TargetArch::ArmV8A);
   742→        MemoryOptimization {
   743→            cache_line_size: 64,
   744→            l1d_size: 65_536,
   745→            l2_size: 524_288,
   746→            target_pi5,
   747→        }
   748→    }
   749→
   750→    /// Creates a pass with default Pi 5 parameters.
   751→    pub fn for_pi5() -> Self {
   752→        Self::new(TargetArch::ArmV8A)
   753→    }
   754→
   755→    /// Creates a pass with default x86-64 parameters.
   756→    pub fn for_x86_64() -> Self {
   757→        let mut pass = Self::new(TargetArch::X86_64);
   758→        pass.l2_size = 1_048_576; // 1 MB typical L2
   759→        pass
   760→    }
   761→}
   762→
   763→impl OptimizationPass for MemoryOptimization {
   764→    fn name(&self) -> &str {
   765→        "MemoryOptimization"
   766→    }
   767→
   768→    fn apply(&self, scg: &mut SCG, profile: &ProfileReport) -> PassResult {
   769→        let mut result = PassResult::empty(self.name());
   770→
   771→        // Collect (node_id, is_hot) for Memory nodes.
   772→        let memory_nodes: Vec<(NodeId, bool)> = scg
   773→            .nodes
   774→            .iter()
   775→            .filter(|(_, n)| n.kind == NodeKind::Memory)
   776→            .map(|(&id, _)| (id, profile.is_hot(id)))
   777→            .collect();
   778→
   779→        for (node_id, is_hot) in memory_nodes {
   780→            let node = match scg.get_node_mut(node_id) {
   781→                Some(n) => n,
   782→                None => continue,
   783→            };
   784→
   785→            // Insert prefetch for hot memory accesses.
   786→            if is_hot && !node.has_prefetch {
   787→                node.has_prefetch = true;
   788→                result.transformations.push(Transformation {
   789→                    kind: TransformationKind::PrefetchInserted,
   790→                    target_node: node_id,
   791→                    description: format!(
   792→                        "Prefetch inserted for hot memory node (cache-line {}B, L1D {}KB)",
   793→                        self.cache_line_size,
   794→                        self.l1d_size / 1024,
   795→                    ),
   796→                });
   797→            }
   798→
   799→            // Align to cache-line boundary on Pi 5 or for hot paths.
   800→            if (self.target_pi5 || is_hot) && node.alignment < self.cache_line_size {
   801→                let old_alignment = node.alignment;
   802→                node.alignment = self.cache_line_size;
   803→                result.transformations.push(Transformation {
   804→                    kind: TransformationKind::CacheLineAligned {
   805→                        alignment: self.cache_line_size,
   806→                    },
   807→                    target_node: node_id,
   808→                    description: format!(
   809→                        "Aligned memory node {}→{}B (Pi 5 L1/L2 cache-line)",
   810→                        old_alignment, self.cache_line_size,
   811→                    ),
   812→                });
   813→            }
   814→        }
   815→
   816→        // Speedup model: prefetch reduces latency ~20% per access;
   817→        // alignment avoids cross-line penalties ~10%.
   818→        let prefetch_count = result
   819→            .transformations
   820→            .iter()
   821→            .filter(|t| matches!(t.kind, TransformationKind::PrefetchInserted))
   822→            .count();
   823→        let align_count = result
   824→            .transformations
   825→            .iter()
   826→            .filter(|t| matches!(t.kind, TransformationKind::CacheLineAligned { .. }))
   827→            .count();
   828→
   829→        let prefetch_benefit = (prefetch_count as f64 * 0.05).min(0.30);
   830→        let align_benefit = (align_count as f64 * 0.02).min(0.15);
   831→        result.estimated_speedup = 1.0 + prefetch_benefit + align_benefit;
   832→
   833→        log::info!(
   834→            "MemoryOptimization: {} transformations, estimated speedup {:.3}×",
   835→            result.count(),
   836→            result.estimated_speedup,
   837→        );
   838→
   839→        result
   840→    }
   841→}

   843→// ---------------------------------------------------------------------------
   844→// OptimizationEngine
   845→// ---------------------------------------------------------------------------
   846→
   847→/// Applies a sequence of profile-guided optimisation passes to the SCG.
   848→///
   849→/// The engine holds a collection of [`OptimizationPass`] implementations and
   850→/// a [`Config`]. Calling [`OptimizationEngine::run`] executes each pass in
   851→/// order and returns an aggregate [`OptimizationResult`].
   852→pub struct OptimizationEngine {
   853→    passes: Vec<Box<dyn OptimizationPass>>,
   854→    config: Config,
   855→}
   856→
   857→impl std::fmt::Debug for OptimizationEngine {
   858→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   859→        f.debug_struct("OptimizationEngine")
   860→            .field("pass_count", &self.passes.len())
   861→            .field(
   862→                "pass_names",
   863→                &self.passes.iter().map(|p| p.name()).collect::<Vec<_>>(),
   864→            )
   865→            .field("config", &self.config)
   866→            .finish()
   867→    }
   868→}
   869→
   870→impl OptimizationEngine {
   871→    /// Creates a new optimisation engine with the default set of passes
   872→    /// appropriate for the given configuration.
   873→    pub fn new(config: Config) -> Self {
   874→        let mut engine = OptimizationEngine {
   875→            passes: Vec::new(),
   876→            config,
   877→        };
   878→        engine.register_default_passes();
   879→        engine
   880→    }
   881→
   882→    /// Creates an engine with no passes registered (for custom setups).
   883→    pub fn empty(config: Config) -> Self {
   884→        OptimizationEngine {
   885→            passes: Vec::new(),
   886→            config,
   887→        }
   888→    }
   889→
   890→    /// Registers the default set of passes based on the current config.
   891→    fn register_default_passes(&mut self) {
   892→        self.passes.push(Box::new(HotPathInlining::new()));
   893→
   894→        self.passes.push(Box::new(ColdPathOutline::new()));
   895→
   896→        self.passes.push(Box::new(LoopOptimization::new()));
   897→
   898→        self.passes
   899→            .push(Box::new(MemoryOptimization::new(self.config.target_arch)));
   900→    }
   901→
   902→    /// Adds a custom optimisation pass.
   903→    pub fn add_pass(&mut self, pass: Box<dyn OptimizationPass>) {
   904→        self.passes.push(pass);
   905→    }
   906→
   907→    /// Removes all registered passes.
   908→    pub fn clear_passes(&mut self) {
   909→        self.passes.clear();
   910→    }
   911→
   912→    /// Returns the number of registered passes.
   913→    pub fn pass_count(&self) -> usize {
   914→        self.passes.len()
   915→    }
   916→
   917→    /// Runs all registered passes over the SCG and returns the aggregate
   918→    /// result.
   919→    pub fn run(&self, scg: &mut SCG, profile: &ProfileReport) -> OptimizationResult {
   920→        let mut pass_results = Vec::with_capacity(self.passes.len());
   921→
   922→        for pass in &self.passes {
   923→            log::debug!("OptimizationEngine: running pass '{}'", pass.name());
   924→            let result = pass.apply(scg, profile);
   925→            pass_results.push(result);
   926→        }
   927→
   928→        OptimizationResult::from_pass_results(pass_results)
   929→    }
   930→}
   931→
   932→// ---------------------------------------------------------------------------
   933→// apply_optimizations — top-level convenience function
   934→// ---------------------------------------------------------------------------
   935→
   936→/// Applies all default optimisation passes to the SCG, guided by the
   937→/// profile report.
   938→///
   939→/// This is the main entry point for the optimisation pipeline. It creates
   940→/// an [`OptimizationEngine`] with the default passes for the given
   941→/// configuration and runs them over the SCG.
   942→///
   943→/// # Arguments
   944→///
   945→/// * `scg` – The Semantic Computation Graph to optimise (mutated in place).
   946→/// * `profile` – The profile report guiding optimisation decisions.
   947→///
   948→/// # Returns
   949→///
   950→/// An [`OptimizationResult`] summarising all transformations.
   951→pub fn apply_optimizations(scg: &mut SCG, profile: &ProfileReport) -> OptimizationResult {
   952→    apply_optimizations_with_config(scg, profile, &Config::default())
   953→}
   954→
   955→/// Applies all default optimisation passes with the given configuration.
   956→pub fn apply_optimizations_with_config(
   957→    scg: &mut SCG,
   958→    profile: &ProfileReport,
   959→    config: &Config,
   960→) -> OptimizationResult {
   961→    let engine = OptimizationEngine::new(config.clone());
   962→    engine.run(scg, profile)
   963→}
