     1→//! Security Model — lattice-based information-flow control, taint tracking,
     2→//! capability-to-hardware mapping, declassification, and boundary enforcement.
     3→//!
     4→//! This module implements the VUMA Security Model as specified in
     5→//! `VUMA-SPEC-SEC-001` (security-model-spec.md). The central invariant is the
     6→//! **no-downgrade rule**: data may flow from a lower security level to an equal
     7→//! or higher level, but *never* downward without explicit declassification.
     8→//!
     9→//! # Architecture
    10→//!
    11→//! | Concept               | Type                     | Description                                    |
    12→//! |-----------------------|--------------------------|------------------------------------------------|
    13→//! | **Security Level**    | [`SecurityLevel`]        | 5-level lattice: Public…TopSecret              |
    14→//! | **Flow Policy**       | [`FlowPolicy`]           | FreeFlow / NoDowngrade / NoFlow                |
    15→//! | **Taint Label**       | [`TaintLabel`]           | Lightweight taint label (source set)           |
    16→//! | **Taint**             | [`TaintStatus`]          | Source-tracking with sanitization              |
    17→//! | **Taint Tracker**     | [`TaintTracker`]         | Propagate taint through derivation chains      |
    18→//! | **Security Relation** | [`SecurityRel`]          | Per-value security metadata (part of RelD)     |
    19→//! | **Boundary**          | [`SecurityBoundary`]     | Region-pair boundary with crossing rules       |
    20→//! | **Declassification**  | [`DeclassificationRecord`]| Audit-trail for every downgrade event         |
    21→//! | **HW Mapping**        | [`Arm64SecurityMapping`] | CapD → PAC/BTI/MTE for ARM64                  |
    22→//! | **Verifier**          | [`SecurityVerifier`]     | Whole-program invariant checker                |

    31→// ---------------------------------------------------------------------------
    32→// Security Level Lattice
    33→// ---------------------------------------------------------------------------
    34→
    35→/// The five security levels in VUMA's mandatory access control lattice.
    36→///
    37→/// The ordering is total: `Public < Internal < Confidential < Secret < TopSecret`.
    38→///
    39→/// The pair `(L, ≤)` forms a lattice with:
    40→/// - **Join** (lub) = `max(l1, l2)` — the more restrictive level.
    41→/// - **Meet** (glb) = `min(l1, l2)` — the less restrictive level.
    42→/// - **Top** = `TopSecret`, **Bottom** = `Public`.
    43→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize, PartialOrd, Ord)]
    44→pub enum SecurityLevel {
    45→    /// Bottom element — unrestricted, openly observable.
    46→    Public = 0,
    47→    /// Internal to the organisation / process — not for external release.
    48→    Internal = 1,
    49→    /// Confidential — restricted to authorised personnel.
    50→    Confidential = 2,
    51→    /// Secret — highly restricted, significant damage if disclosed.
    52→    Secret = 3,
    53→    /// Top element — exceptionally grave damage if disclosed.
    54→    TopSecret = 4,
    55→}
    56→
    57→impl SecurityLevel {
    58→    /// All levels in ascending order.
    59→    pub const ALL: [SecurityLevel; 5] = [
    60→        SecurityLevel::Public,
    61→        SecurityLevel::Internal,
    62→        SecurityLevel::Confidential,
    63→        SecurityLevel::Secret,
    64→        SecurityLevel::TopSecret,
    65→    ];
    66→
    67→    /// The bottom element of the lattice.
    68→    pub const BOTTOM: SecurityLevel = SecurityLevel::Public;
    69→
    70→    /// The top element of the lattice.
    71→    pub const TOP: SecurityLevel = SecurityLevel::TopSecret;
    72→
    73→    /// **Join** (least upper bound) — returns the more restrictive of two levels.
    74→    ///
    75→    /// ```
    76→    /// use vuma_core::security::SecurityLevel;
    77→    /// assert_eq!(SecurityLevel::Public.join(SecurityLevel::Secret), SecurityLevel::Secret);
    78→    /// ```
    79→    pub fn join(self, other: SecurityLevel) -> SecurityLevel {
    80→        self.max(other)
    81→    }
    82→
    83→    /// **Meet** (greatest lower bound) — returns the less restrictive of two levels.
    84→    ///
    85→    /// ```
    86→    /// use vuma_core::security::SecurityLevel;
    87→    /// assert_eq!(SecurityLevel::Confidential.meet(SecurityLevel::Secret), SecurityLevel::Confidential);
    88→    /// ```
    89→    pub fn meet(self, other: SecurityLevel) -> SecurityLevel {
    90→        self.min(other)
    91→    }
    92→
    93→    /// Returns `true` if information may flow from `self` to `target`
    94→    /// (i.e. `self ≤ target`).
    95→    pub fn can_flow_to(self, target: SecurityLevel) -> bool {
    96→        self <= target
    97→    }
    98→
    99→    /// Numeric rank (0..4) useful for array indexing.
   100→    pub fn rank(self) -> u8 {
   101→        self as u8
   102→    }
   103→}
   104→
   105→impl fmt::Display for SecurityLevel {
   106→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   107→        match self {
   108→            SecurityLevel::Public => write!(f, "Public"),
   109→            SecurityLevel::Internal => write!(f, "Internal"),
   110→            SecurityLevel::Confidential => write!(f, "Confidential"),
   111→            SecurityLevel::Secret => write!(f, "Secret"),
   112→            SecurityLevel::TopSecret => write!(f, "TopSecret"),
   113→        }
   114→    }
   115→}

   168→/// The taint status of a value.
   169→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   170→pub enum TaintStatus {
   171→    /// Value is not tainted.
   172→    Clean,
   173→    /// Value is tainted with the given set of sources.
   174→    Tainted {
   175→        /// The set of taint sources this value carries.
   176→        sources: HashSet<TaintSource>,
   177→        /// Whether the taint is sanitizable (can be removed via verified
   178→        /// sanitization).
   179→        sanitizable: bool,
   180→    },
   181→}
   182→
   183→impl TaintStatus {
   184→    /// Create a clean (non-tainted) status.
   185→    pub fn clean() -> Self {
   186→        TaintStatus::Clean
   187→    }
   188→
   189→    /// Create a tainted status from a single source.
   190→    pub fn tainted(source: TaintSource, sanitizable: bool) -> Self {
   191→        TaintStatus::Tainted {
   192→            sources: [source].into_iter().collect(),
   193→            sanitizable,
   194→        }
   195→    }
   196→
   197→    /// Create a tainted status from multiple sources.
   198→    pub fn tainted_multi(sources: HashSet<TaintSource>, sanitizable: bool) -> Self {
   199→        TaintStatus::Tainted {
   200→            sources,
   201→            sanitizable,
   202→        }
   203→    }
   204→
   205→    /// Returns `true` if the value is tainted.
   206→    pub fn is_tainted(&self) -> bool {
   207→        matches!(self, TaintStatus::Tainted { .. })
   208→    }
   209→
   210→    /// Returns the taint sources, or an empty set if clean.
   211→    pub fn sources(&self) -> HashSet<TaintSource> {
   212→        match self {
   213→            TaintStatus::Clean => HashSet::new(),
   214→            TaintStatus::Tainted { sources, .. } => sources.clone(),
   215→        }
   216→    }
   217→
   218→    /// Propagate taint: union the sources of two taint statuses.
   219→    ///
   220→    /// If either is tainted, the result is tainted with the union of sources.
   221→    /// The result is sanitizable only if both inputs are sanitizable.
   222→    pub fn propagate(&self, other: &TaintStatus) -> TaintStatus {
   223→        match (self, other) {
   224→            (TaintStatus::Clean, TaintStatus::Clean) => TaintStatus::Clean,
   225→            (
   226→                TaintStatus::Clean,
   227→                TaintStatus::Tainted {
   228→                    sources,
   229→                    sanitizable,
   230→                },
   231→            )
   232→            | (
   233→                TaintStatus::Tainted {
   234→                    sources,
   235→                    sanitizable,
   236→                },
   237→                TaintStatus::Clean,
   238→            ) => TaintStatus::Tainted {
   239→                sources: sources.clone(),
   240→                sanitizable: *sanitizable,
   241→            },
   242→            (
   243→                TaintStatus::Tainted {
   244→                    sources: s1,
   245→                    sanitizable: san1,
   246→                },
   247→                TaintStatus::Tainted {
   248→                    sources: s2,
   249→                    sanitizable: san2,
   250→                },
   251→            ) => TaintStatus::Tainted {
   252→                sources: s1.union(s2).copied().collect(),
   253→                sanitizable: *san1 && *san2,
   254→            },
   255→        }
   256→    }
   257→
   258→    /// Sanitize the value: remove taint if sanitizable.
   259→    ///
   260→    /// Returns `Ok(Clean)` if the taint was sanitizable, or `Err(self)` if not.
   261→    pub fn sanitize(self) -> Result<TaintStatus, TaintStatus> {
   262→        match self {
   263→            TaintStatus::Clean => Ok(TaintStatus::Clean),
   264→            TaintStatus::Tainted {
   265→                sanitizable: true, ..
   266→            } => Ok(TaintStatus::Clean),
   267→            t @ TaintStatus::Tainted {
   268→                sanitizable: false, ..
   269→            } => Err(t),
   270→        }
   271→    }
   272→
   273→    /// Effective security level adjustment for tainted data.
   274→    /// Tainted data is treated at the join of its explicit level and `Internal`.
   275→    pub fn effective_level(&self, explicit: SecurityLevel) -> SecurityLevel {
   276→        match self {
   277→            TaintStatus::Clean => explicit,
   278→            TaintStatus::Tainted { .. } => explicit.join(SecurityLevel::Internal),
   279→        }
   280→    }
   281→}
   282→
   283→impl fmt::Display for TaintStatus {
   284→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   285→        match self {
   286→            TaintStatus::Clean => write!(f, "Clean"),
   287→            TaintStatus::Tainted {
   288→                sources,
   289→                sanitizable,
   290→            } => {
   291→                let srcs: Vec<_> = sources.iter().map(|s| s.to_string()).collect();
   292→                write!(
   293→                    f,
   294→                    "Tainted({}{})",
   295→                    srcs.join(","),
   296→                    if *sanitizable { "" } else { ";unsanitizable" }
   297→                )
   298→            }
   299→        }
   300→    }
   301→}

   307→/// A lightweight taint label — the set of taint sources attached to a value.
   308→///
   309→/// Unlike [`TaintStatus`], which also tracks sanitizability, `TaintLabel`
   310→/// is the minimal information needed for taint propagation along data-flow
   311→/// edges. It is the label that flows through the SCG's DataFlow edges
   312→/// during the IVE's fixed-point taint computation.
   313→///
   314→/// A `TaintLabel` is essentially a set of [`TaintSource`] values. The empty
   315→/// set represents "Clean" (no taint). Taint propagation is set union.
   316→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   317→pub struct TaintLabel {
   318→    /// The set of taint sources. Empty means clean.
   319→    sources: HashSet<TaintSource>,
   320→}
   321→
   322→impl TaintLabel {
   323→    /// Create a clean (empty) label.
   324→    pub fn clean() -> Self {
   325→        TaintLabel {
   326→            sources: HashSet::new(),
   327→        }
   328→    }
   329→
   330→    /// Create a label from a single source.
   331→    pub fn from_source(source: TaintSource) -> Self {
   332→        TaintLabel {
   333→            sources: [source].into_iter().collect(),
   334→        }
   335→    }
   336→
   337→    /// Create a label from multiple sources.
   338→    pub fn from_sources(sources: HashSet<TaintSource>) -> Self {
   339→        TaintLabel { sources }
   340→    }
   341→
   342→    /// Returns `true` if the label is clean (no taint sources).
   343→    pub fn is_clean(&self) -> bool {
   344→        self.sources.is_empty()
   345→    }
   346→
   347→    /// Returns `true` if the label is tainted (has at least one source).
   348→    pub fn is_tainted(&self) -> bool {
   349→        !self.sources.is_empty()
   350→    }
   351→
   352→    /// Returns the set of taint sources.
   353→    pub fn sources(&self) -> &HashSet<TaintSource> {
   354→        &self.sources
   355→    }
   356→
   357→    /// Propagate (join) two labels: the result carries the union of sources.
   358→    ///
   359→    /// This is the lattice join operation for taint labels.
   360→    pub fn join(&self, other: &TaintLabel) -> TaintLabel {
   361→        TaintLabel {
   362→            sources: self.sources.union(&other.sources).copied().collect(),
   363→        }
   364→    }
   365→
   366→    /// Check if this label contains a specific source.
   367→    pub fn contains(&self, source: &TaintSource) -> bool {
   368→        self.sources.contains(source)
   369→    }
   370→
   371→    /// Convert to a `TaintStatus` with the given sanitizability.
   372→    pub fn to_status(&self, sanitizable: bool) -> TaintStatus {
   373→        if self.sources.is_empty() {
   374→            TaintStatus::Clean
   375→        } else {
   376→            TaintStatus::Tainted {
   377→                sources: self.sources.clone(),
   378→                sanitizable,
   379→            }
   380→        }
   381→    }
   382→}
   383→
   384→impl fmt::Display for TaintLabel {
   385→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   386→        if self.sources.is_empty() {
   387→            write!(f, "Clean")
   388→        } else {
   389→            let srcs: Vec<_> = self.sources.iter().map(|s| s.to_string()).collect();
   390→            write!(f, "Tainted({})", srcs.join(","))
   391→        }
   392→    }
   393→}
   394→
   395→impl Default for TaintLabel {
   396→    fn default() -> Self {
   397→        Self::clean()
   398→    }
   399→}

   405→/// A taint tracker that propagates taint labels through a graph of value
   406→/// nodes connected by derivation chains.
   407→///
   408→/// The tracker maintains a map from [`NodeId`] to [`TaintLabel`] and
   409→/// supports incremental propagation: when a new edge is added, only the
   410→/// affected downstream nodes are updated.
   411→///
   412→/// # Fixed-Point Propagation
   413→///
   414→/// The `propagate` method performs a fixed-point computation: it iterates
   415→/// over all edges, joining source labels into destination labels, until no
   416→/// labels change. This is equivalent to the IVE's taint propagation over
   417→/// the SCG's DataFlow edges.
   418→#[derive(Debug, Clone, Default)]
   419→pub struct TaintTracker {
   420→    /// Taint labels indexed by node ID.
   421→    labels: HashMap<NodeId, TaintLabel>,
   422→    /// Data-flow edges: (source, destination).
   423→    edges: Vec<(NodeId, NodeId)>,
   424→}
   425→
   426→impl TaintTracker {
   427→    /// Create a new, empty taint tracker.
   428→    pub fn new() -> Self {
   429→        Self::default()
   430→    }
   431→
   432→    /// Set the taint label for a node.
   433→    pub fn set_label(&mut self, node: NodeId, label: TaintLabel) {
   434→        self.labels.insert(node, label);
   435→    }
   436→
   437→    /// Get the taint label for a node (Clean if not set).
   438→    pub fn get_label(&self, node: NodeId) -> TaintLabel {
   439→        self.labels.get(&node).cloned().unwrap_or_default()
   440→    }
   441→
   442→    /// Add a data-flow edge from `src` to `dst`.
   443→    pub fn add_edge(&mut self, src: NodeId, dst: NodeId) {
   444→        self.edges.push((src, dst));
   445→    }
   446→
   447→    /// Propagate taint labels along all edges until a fixed point is reached.
   448→    ///
   449→    /// Returns the number of iterations performed.
   450→    pub fn propagate(&mut self) -> usize {
   451→        let mut iterations = 0;
   452→        loop {
   453→            let mut changed = false;
   454→            for &(src, dst) in &self.edges {
   455→                let src_label = self.get_label(src);
   456→                let dst_label = self.get_label(dst);
   457→                let joined = src_label.join(&dst_label);
   458→                if joined != dst_label {
   459→                    self.labels.insert(dst, joined);
   460→                    changed = true;
   461→                }
   462→            }
   463→            iterations += 1;
   464→            if !changed {
   465→                break;
   466→            }
   467→        }
   468→        iterations
   469→    }
   470→
   471→    /// Propagate taint through a derivation chain.
   472→    ///
   473→    /// Given a chain of derivations and a taint map, walks the chain from
   474→    /// root to leaf, joining all taint labels.
   475→    pub fn propagate_chain(
   476→        derivation: &Derivation,
   477→        taint_map: &HashMap<DerivationId, TaintLabel>,
   478→        lookup: impl Fn(DerivationId) -> Option<Derivation>,
   479→    ) -> TaintLabel {
   480→        let chain = derivation.trace(lookup);
   481→        let mut accumulated = TaintLabel::clean();
   482→        for d in &chain {
   483→            if let Some(label) = taint_map.get(&d.id) {
   484→                accumulated = accumulated.join(label);
   485→            }
   486→        }
   487→        accumulated
   488→    }
   489→
   490→    /// Returns the number of nodes with taint labels.
   491→    pub fn node_count(&self) -> usize {
   492→        self.labels.len()
   493→    }
   494→
   495→    /// Returns the number of edges.
   496→    pub fn edge_count(&self) -> usize {
   497→        self.edges.len()
   498→    }
   499→
   500→    /// Returns all tainted nodes (nodes with non-empty taint labels).
   501→    pub fn tainted_nodes(&self) -> Vec<(NodeId, TaintLabel)> {
   502→        self.labels
   503→            .iter()
   504→            .filter(|(_, label)| label.is_tainted())
   505→            .map(|(&id, label)| (id, label.clone()))
   506→            .collect()
   507→    }
   508→}

   546→// ---------------------------------------------------------------------------
   547→// Security Relation (part of RelD)
   548→// ---------------------------------------------------------------------------
   549→
   550→/// The security metadata carried by every value as part of its Relational
   551→/// Descriptor (RelD).
   552→///
   553→/// This is the core data structure of the security model. It records the
   554→/// security classification, flow policy, taint status, and (if applicable)
   555→/// the provenance of any declassification that has been applied.
   556→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   557→pub struct SecurityRel {
   558→    /// The security classification of this value.
   559→    pub level: SecurityLevel,
   560→    /// The flow policy governing information flow.
   561→    pub flow: FlowPolicy,
   562→    /// The taint status (taint sources, sanitizability).
   563→    pub taint: TaintStatus,
   564→    /// If this value has been declassified, records the audit trail.
   565→    pub declassification: Option<DeclassificationRecord>,
   566→}
   567→
   568→impl SecurityRel {
   569→    /// Create a default SecurityRel at the given level with NoDowngrade flow
   570→    /// and Clean taint.
   571→    pub fn at(level: SecurityLevel) -> Self {
   572→        Self {
   573→            level,
   574→            flow: FlowPolicy::NoDowngrade,
   575→            taint: TaintStatus::Clean,
   576→            declassification: None,
   577→        }
   578→    }
   579→
   580→    /// Create a SecurityRel for untrusted (tainted) input.
   581→    ///
   582→    /// Tainted data is created at the given explicit level with `NoDowngrade`
   583→    /// flow. The effective level is the join of the explicit level and
   584→    /// `Internal`.
   585→    pub fn for_untrusted(level: SecurityLevel, source: TaintSource) -> Self {
   586→        Self {
   587→            level,
   588→            flow: FlowPolicy::NoDowngrade,
   589→            taint: TaintStatus::tainted(source, true),
   590→            declassification: None,
   591→        }
   592→    }
   593→
   594→    /// Create a SecurityRel for cryptographic key material (NoFlow).
   595→    pub fn for_key_material(level: SecurityLevel) -> Self {
   596→        Self {
   597→            level,
   598→            flow: FlowPolicy::NoFlow,
   599→            taint: TaintStatus::Clean,
   600→            declassification: None,
   601→        }
   602→    }
   603→
   604→    /// The effective security level, accounting for taint.
   605→    ///
   606→    /// Tainted data is treated at the join of its explicit level and Internal.
   607→    pub fn effective_level(&self) -> SecurityLevel {
   608→        self.taint.effective_level(self.level)
   609→    }
   610→
   611→    /// Check whether information can flow from `self` to `target`.
   612→    ///
   613→    /// Returns `Ok(())` if the flow is permitted, or a `SecurityViolation`
   614→    /// describing why it is not.
   615→    pub fn check_flow_to(&self, target: &SecurityRel) -> Result<(), SecurityViolation> {
   616→        // NoFlow values cannot flow at all.
   617→        if self.flow == FlowPolicy::NoFlow {
   618→            return Err(SecurityViolation::NoFlowViolation {
   619→                src_level: self.level,
   620→                dst_level: target.level,
   621→            });
   622→        }
   623→
   624→        // Check the no-downgrade rule using effective levels.
   625→        let src_eff = self.effective_level();
   626→        let dst_eff = target.effective_level();
   627→
   628→        if !src_eff.can_flow_to(dst_eff) {
   629→            // Allow if the target has a declassification gate that covers
   630→            // this exact downgrade — but that check is done by the
   631→            // SecurityVerifier at the boundary level, not here.
   632→            return Err(SecurityViolation::InformationLeak {
   633→                src_level: src_eff,
   634→                dst_level: dst_eff,
   635→            });
   636→        }
   637→
   638→        // NoDowngrade policy on the source blocks any downward flow
   639→        // even if the target level would otherwise allow it.
   640→        if self.flow == FlowPolicy::NoDowngrade && self.level > target.level {
   641→            return Err(SecurityViolation::DowngradeBlocked {
   642→                src_level: self.level,
   643→                dst_level: target.level,
   644→            });
   645→        }
   646→
   647→        Ok(())
   648→    }
   649→
   650→    /// Join two SecurityRels — used when combining values (e.g. arithmetic).
   651→    ///
   652→    /// The result has the join of the levels, the most restrictive flow policy,
   653→    /// and the union of taint sources.
   654→    pub fn join(&self, other: &SecurityRel) -> SecurityRel {
   655→        SecurityRel {
   656→            level: self.level.join(other.level),
   657→            flow: self.flow.more_restrictive(other.flow),
   658→            taint: self.taint.propagate(&other.taint),
   659→            declassification: None,
   660→        }
   661→    }
   662→}
   663→
   664→impl fmt::Display for SecurityRel {
   665→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   666→        write!(f, "SecurityRel({} {} {}", self.level, self.flow, self.taint)?;
   667→        if let Some(ref dec) = self.declassification {
   668→            write!(f, " declassified_by={}", dec.gate_function)?;
   669→        }
   670→        write!(f, ")")
   671→    }
   672→}
   673→
   674→// ---------------------------------------------------------------------------
   675→// Flow Policy helpers
   676→// ---------------------------------------------------------------------------
   677→
   678→impl FlowPolicy {
   679→    /// Returns the more restrictive of two flow policies.
   680→    ///
   681→    /// Ordering: `NoFlow > NoDowngrade > FreeFlow`.
   682→    pub fn more_restrictive(self, other: FlowPolicy) -> FlowPolicy {
   683→        match (self, other) {
   684→            (FlowPolicy::NoFlow, _) | (_, FlowPolicy::NoFlow) => FlowPolicy::NoFlow,
   685→            (FlowPolicy::NoDowngrade, _) | (_, FlowPolicy::NoDowngrade) => FlowPolicy::NoDowngrade,
   686→            (FlowPolicy::FreeFlow, FlowPolicy::FreeFlow) => FlowPolicy::FreeFlow,
   687→        }
   688→    }
   689→}

   695→/// A violation of the security model detected during verification.
   696→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   697→pub enum SecurityViolation {
   698→    /// Information would leak from a higher level to a lower level.
   699→    InformationLeak {
   700→        src_level: SecurityLevel,
   701→        dst_level: SecurityLevel,
   702→    },
   703→    /// A NoDowngrade policy prevents the flow.
   704→    DowngradeBlocked {
   705→        src_level: SecurityLevel,
   706→        dst_level: SecurityLevel,
   707→    },
   708→    /// A NoFlow value attempted to cross a boundary.
   709→    NoFlowViolation {
   710→        src_level: SecurityLevel,
   711→        dst_level: SecurityLevel,
   712→    },
   713→    /// Tainted data reached a sink that requires clean data.
   714→    TaintedDataAtSink {
   715→        sources: HashSet<TaintSource>,
   716→        sink: String,
   717→    },
   718→    /// A boundary crossing was attempted without the required capabilities.
   719→    MissingCapabilityForCrossing {
   720→        capability: SecurityCapability,
   721→        boundary_id: BoundaryId,
   722→    },
   723→    /// A declassification was attempted without a valid proof.
   724→    DeclassificationWithoutProof {
   725→        from_level: SecurityLevel,
   726→        to_level: SecurityLevel,
   727→    },
   728→    /// Capability monotonicity violated — a capability was added.
   729→    CapabilityMonotonicityViolation { added: SecurityCapability },
   730→    /// An untrusted source value carries the Execute capability.
   731→    ExecuteOnUntrusted { source: TaintSource },
   732→    /// Implicit flow across a boundary.
   733→    ImplicitFlowAcrossBoundary {
   734→        condition_level: SecurityLevel,
   735→        target_level: SecurityLevel,
   736→    },
   737→}
   738→
   739→impl fmt::Display for SecurityViolation {
   740→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   741→        match self {
   742→            SecurityViolation::InformationLeak {
   743→                src_level,
   744→                dst_level,
   745→            } => {
   746→                write!(f, "information leak: {} → {}", src_level, dst_level)
   747→            }
   748→            SecurityViolation::DowngradeBlocked {
   749→                src_level,
   750→                dst_level,
   751→            } => {
   752→                write!(f, "downgrade blocked: {} → {}", src_level, dst_level)
   753→            }
   754→            SecurityViolation::NoFlowViolation {
   755→                src_level,
   756→                dst_level,
   757→            } => {
   758→                write!(f, "NoFlow violation: {} → {}", src_level, dst_level)
   759→            }
   760→            SecurityViolation::TaintedDataAtSink { sources, sink } => {
   761→                let srcs: Vec<_> = sources.iter().map(|s| s.to_string()).collect();
   762→                write!(f, "tainted data ({}) at sink '{}'", srcs.join(","), sink)
   763→            }
   764→            SecurityViolation::MissingCapabilityForCrossing {
   765→                capability,
   766→                boundary_id,
   767→            } => {
   768→                write!(
   769→                    f,
   770→                    "missing {} for crossing boundary {}",
   771→                    capability, boundary_id
   772→                )
   773→            }
   774→            SecurityViolation::DeclassificationWithoutProof {
   775→                from_level,
   776→                to_level,
   777→            } => {
   778→                write!(
   779→                    f,
   780→                    "declassification without proof: {} → {}",
   781→                    from_level, to_level
   782→                )
   783→            }
   784→            SecurityViolation::CapabilityMonotonicityViolation { added } => {
   785→                write!(f, "capability monotonicity violated: {} added", added)
   786→            }
   787→            SecurityViolation::ExecuteOnUntrusted { source } => {
   788→                write!(f, "Execute capability on untrusted source: {}", source)
   789→            }
   790→            SecurityViolation::ImplicitFlowAcrossBoundary {
   791→                condition_level,
   792→                target_level,
   793→            } => {
   794→                write!(
   795→                    f,
   796→                    "implicit flow across boundary: condition at {} influences {}",
   797→                    condition_level, target_level
   798→                )
   799→            }
   800→        }
   801→    }
   802→}

   818→/// A proof that a declassification is safe.
   819→///
   820→/// A declassification is only valid if:
   821→/// 1. It is performed through a verified gate function.
   822→/// 2. The gate function has been verified to produce output safe at the
   823→///    target level (output independence, no side channels, completeness).
   824→/// 3. The gate is the designated gate for the boundary being crossed.
   825→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   826→pub struct DeclassificationProof {
   827→    /// The gate function that performs the declassification.
   828→    pub gate: GateFunctionId,
   829→    /// The source security level being declassified from.
   830→    pub from_level: SecurityLevel,
   831→    /// The target security level being declassified to.
   832→    pub to_level: SecurityLevel,
   833→    /// Whether the gate function has been verified for output independence.
   834→    pub output_independence_verified: bool,
   835→    /// Whether the gate function has been verified for no side channels.
   836→    pub no_side_channels_verified: bool,
   837→    /// Whether the gate function has been verified for completeness.
   838→    pub completeness_verified: bool,
   839→    /// The boundary this declassification applies to (if any).
   840→    pub boundary_id: Option<BoundaryId>,
   841→}
   842→
   843→impl DeclassificationProof {
   844→    /// Create a new declassification proof.
   845→    pub fn new(gate: GateFunctionId, from: SecurityLevel, to: SecurityLevel) -> Self {
   846→        Self {
   847→            gate,
   848→            from_level: from,
   849→            to_level: to,
   850→            output_independence_verified: false,
   851→            no_side_channels_verified: false,
   852→            completeness_verified: false,
   853→            boundary_id: None,
   854→        }
   855→    }
   856→
   857→    /// Returns `true` if all proof obligations are satisfied.
   858→    pub fn is_valid(&self) -> bool {
   859→        self.output_independence_verified
   860→            && self.no_side_channels_verified
   861→            && self.completeness_verified
   862→            && self.from_level > self.to_level
   863→    }
   864→
   865→    /// Mark all proof obligations as verified.
   866→    pub fn verify_all(&mut self) {
   867→        self.output_independence_verified = true;
   868→        self.no_side_channels_verified = true;
   869→        self.completeness_verified = true;
   870→    }
   871→}

   873→/// An audit-trail record for a declassification event.

   875→/// Every declassification is logged at runtime, enabling post-incident

   883→    /// The new (lower) security level.
   884→    pub to_level: SecurityLevel,
   885→    /// Where in the source code the declassification occurred.
   886→    pub source_location: ProgramPoint,
   887→    /// The proof that was validated before declassification.
   888→    pub proof: DeclassificationProof,
   889→}

   905→/// A security boundary between two adjacent SCG regions.
   906→///
   907→/// A boundary `B = (R_high, R_low)` enforces that data and control flow
   908→/// crossing from the higher-level region to the lower-level region must
   909→/// satisfy the information-flow rule or go through a declassification gate.
   910→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   911→pub struct SecurityBoundary {
   912→    /// Unique identifier.
   913→    pub id: BoundaryId,
   914→    /// The region with the higher security level.
   915→    pub region_high: RegionId,
   916→    /// The region with the lower security level.
   917→    pub region_low: RegionId,
   918→    /// The security level of the high region.
   919→    pub level_high: SecurityLevel,
   920→    /// The security level of the low region.
   921→    pub level_low: SecurityLevel,
   922→    /// Capabilities required for data to cross this boundary.
   923→    pub cross_permissions: HashSet<SecurityCapability>,
   924→    /// The declassification gate for this boundary, if any.
   925→    pub declassification_gate: Option<GateFunctionId>,
   926→}
   927→
   928→impl SecurityBoundary {
   929→    /// Create a new security boundary.
   930→    ///
   931→    /// # Panics
   932→    ///
   933→    /// In debug builds, panics if `level_high <= level_low`.
   934→    pub fn new(
   935→        id: BoundaryId,
   936→        region_high: RegionId,
   937→        region_low: RegionId,
   938→        level_high: SecurityLevel,
   939→        level_low: SecurityLevel,
   940→    ) -> Self {
   941→        debug_assert!(
   942→            level_high > level_low,
   943→            "SecurityBoundary requires level_high > level_low"
   944→        );
   945→        Self {
   946→            id,
   947→            region_high,
   948→            region_low,
   949→            level_high,
   950→            level_low,
   951→            cross_permissions: HashSet::new(),
   952→            declassification_gate: None,
   953→        }
   954→    }
   955→
   956→    /// Check whether a read from `src_region` to `dst_region` is permitted.
   957→    ///
   958→    /// - High→Low: level must be ≤ level_low, or declassification gate required.
   959→    /// - Low→High: always permitted (information flows upward).
   960→    pub fn check_read_across(
   961→        &self,
   962→        src_region: RegionId,
   963→        value_level: SecurityLevel,
   964→    ) -> Result<(), SecurityViolation> {
   965→        if src_region == self.region_high {
   966→            // High → Low read: information flows downward
   967→            if value_level.can_flow_to(self.level_low) || self.declassification_gate.is_some() {
   968→                Ok(())
   969→            } else {
   970→                Err(SecurityViolation::InformationLeak {
   971→                    src_level: value_level,
   972→                    dst_level: self.level_low,
   973→                })
   974→            }
   975→        } else {
   976→            // Low → High: always permitted
   977→            Ok(())
   978→        }
   979→    }
   980→
   981→    /// Check whether a write from `src_region` to `dst_region` is permitted.
   982→    ///
   983→    /// - High→Low: integrity concern — data is being written into low region.
   984→    /// - Low→High: integrity concern — untrusted data injected into high region.
   985→    pub fn check_write_across(
   986→        &self,
   987→        src_region: RegionId,
   988→        value_level: SecurityLevel,
   989→    ) -> Result<(), SecurityViolation> {
   990→        if src_region == self.region_high {
   991→            // High → Low write
   992→            if value_level.can_flow_to(self.level_low) {
   993→                Ok(())
   994→            } else {
   995→                Err(SecurityViolation::InformationLeak {
   996→                    src_level: value_level,
   997→                    dst_level: self.level_low,
   998→                })
   999→            }
  1000→        } else {
  1001→            // Low → High write: potential integrity violation
  1002→            // In VUMA this is flagged unless the target CapD includes Write
  1003→            // and the source value has been validated. For the model-level
  1004→            // check we just verify the level is appropriate.
  1005→            Ok(())
  1006→        }
  1007→    }
  1008→
  1009→    /// Check whether a control-flow crossing requires specific capabilities.
  1010→    pub fn check_control_flow_across(
  1011→        &self,
  1012→        caller_caps: &HashSet<SecurityCapability>,
  1013→    ) -> Result<(), SecurityViolation> {
  1014→        for cap in &self.cross_permissions {
  1015→            if !caller_caps.contains(cap) {
  1016→                return Err(SecurityViolation::MissingCapabilityForCrossing {
  1017→                    capability: *cap,
  1018→                    boundary_id: self.id,
  1019→                });
  1020→            }
  1021→        }
  1022→        Ok(())
  1023→    }
  1024→}

  1122→    /// Map a capability to the corresponding ARM64 feature.
  1123→    ///
  1124→    /// Returns the hardware feature(s) that enforce the given VUMA capability.
  1125→    pub fn capability_to_hw(&self, cap: SecurityCapability) -> Vec<Arm64Feature> {
  1126→        match cap {
  1127→            SecurityCapability::DerivePtr => {
  1128→                if self.pac_enabled {
  1129→                    vec![Arm64Feature::Pac]
  1130→                } else {
  1131→                    vec![]
  1132→                }
  1133→            }
  1134→            SecurityCapability::Execute => {
  1135→                if self.bti_enabled {
  1136→                    vec![Arm64Feature::Bti]
  1137→                } else {
  1138→                    vec![]
  1139→                }
  1140→            }
  1141→            SecurityCapability::Read | SecurityCapability::Write => {
  1142→                if self.mte_enabled {
  1143→                    vec![Arm64Feature::Mte]
  1144→                } else {
  1145→                    vec![]
  1146→                }
  1147→            }
  1148→            SecurityCapability::Send => {
  1149→                // Send is enforced at the software level; PAC+BTI provide
  1150→                // defense-in-depth against bypass.
  1151→                let mut features = vec![];
  1152→                if self.pac_enabled {
  1153→                    features.push(Arm64Feature::Pac);
  1154→                }
  1155→                if self.bti_enabled {
  1156→                    features.push(Arm64Feature::Bti);
  1157→                }
  1158→                features
  1159→            }
  1160→        }
  1161→    }
  1162→
  1163→    /// Map a capability set to the full set of ARM64 features required.
  1164→    pub fn capabilities_to_hw(&self, caps: &HashSet<SecurityCapability>) -> HashSet<Arm64Feature> {
  1165→        caps.iter()
  1166→            .flat_map(|c| self.capability_to_hw(*c))
  1167→            .collect()
  1168→    }

  1190→    /// Generate the BTI landing pad pseudocode for a function entry.
  1191→    pub fn emit_bti_landing_pad(&self, has_execute: bool) -> &'static str {
  1192→        if self.bti_enabled && has_execute {
  1193→            "bti c  // permit indirect calls"
  1194→        } else if self.bti_enabled {

  1245→/// Taint propagation along a derivation chain.
  1246→///
  1247→/// Given a derivation and a lookup function, this walks the chain from the
  1248→/// current derivation back to the root region, propagating taint forward.
  1249→/// If any derivation in the chain carries taint, the result is tainted with
  1250→/// the union of all sources.
  1251→pub fn propagate_taint_through_chain<F>(
  1252→    derivation: &Derivation,
  1253→    taint_map: &HashMap<DerivationId, TaintStatus>,
  1254→    lookup: F,
  1255→) -> TaintStatus
  1256→where
  1257→    F: Fn(DerivationId) -> Option<Derivation>,
  1258→{
  1259→    let chain = derivation.trace(lookup);
  1260→    let mut accumulated = TaintStatus::Clean;
  1261→
  1262→    for d in &chain {
  1263→        if let Some(taint) = taint_map.get(&d.id) {
  1264→            accumulated = accumulated.propagate(taint);
  1265→        }
  1266→    }
  1267→
  1268→    accumulated
  1269→}

  1275→/// Result of a security verification pass.
  1276→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
  1277→pub struct VerificationResult {
  1278→    /// Total number of checks performed.
  1279→    pub total_checks: usize,
  1280→    /// Number of checks that passed.
  1281→    pub passed: usize,
  1282→    /// Violations detected.
  1283→    pub violations: Vec<SecurityViolation>,
  1284→}
  1285→
  1286→impl Default for VerificationResult {
  1287→    fn default() -> Self {
  1288→        Self::new()
  1289→    }
  1290→}
  1291→
  1292→impl VerificationResult {
  1293→    /// Create an empty result.
  1294→    pub fn new() -> Self {
  1295→        Self {
  1296→            total_checks: 0,
  1297→            passed: 0,
  1298→            violations: Vec::new(),
  1299→        }
  1300→    }
  1301→
  1302→    /// Record a passing check.
  1303→    pub fn pass(&mut self) {
  1304→        self.total_checks += 1;
  1305→        self.passed += 1;
  1306→    }
  1307→
  1308→    /// Record a failing check with the given violation.
  1309→    pub fn fail(&mut self, violation: SecurityViolation) {
  1310→        self.total_checks += 1;
  1311→        self.violations.push(violation);
  1312→    }
  1313→
  1314→    /// Returns `true` if all checks passed (no violations).
  1315→    pub fn all_passed(&self) -> bool {
  1316→        self.violations.is_empty()
  1317→    }
  1318→}
  1319→
  1320→impl fmt::Display for VerificationResult {
  1321→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
  1322→        write!(
  1323→            f,
  1324→            "VerificationResult({}/{} passed, {} violations)",
  1325→            self.passed,
  1326→            self.total_checks,
  1327→            self.violations.len()
  1328→        )
  1329→    }
  1330→}

  1348→/// A data-flow edge in the security graph.

  1353→    /// Destination node.

  1357→    /// The boundary this edge crosses, if any.
  1358→    pub boundary: Option<BoundaryId>,

  1361→/// The **Security Verifier** — checks all security invariants over a
  1362→/// collection of security-annotated nodes, edges, and boundaries.
  1363→///
  1364→/// The verifier enforces:
  1365→/// - The no-downgrade information-flow rule (level check).
  1366→/// - Taint-at-sink checks.
  1367→/// - Boundary crossing rules (B1–B3 from the spec).
  1368→/// - Capability monotonicity.
  1369→/// - Execute-on-untrusted prevention.
  1370→/// - Declassification proof requirements.
  1371→#[derive(Debug, Clone, Default)]
  1372→pub struct SecurityVerifier {
  1373→    /// Security-annotated nodes.
  1374→    pub nodes: HashMap<NodeId, SecNode>,
  1375→    /// Data-flow edges.
  1376→    pub edges: Vec<SecEdge>,
  1377→    /// Security boundaries.
  1378→    pub boundaries: HashMap<BoundaryId, SecurityBoundary>,
  1379→    /// Valid declassification proofs.
  1380→    pub declassification_proofs: HashMap<GateFunctionId, DeclassificationProof>,
  1381→}
  1382→
  1383→impl SecurityVerifier {
  1384→    /// Create a new, empty verifier.
  1385→    pub fn new() -> Self {
  1386→        Self::default()
  1387→    }
  1388→
  1389→    /// Add a node to the verifier.
  1390→    pub fn add_node(&mut self, node: SecNode) {
  1391→        self.nodes.insert(node.id, node);
  1392→    }
  1393→
  1394→    /// Add a data-flow edge.
  1395→    pub fn add_edge(&mut self, edge: SecEdge) {
  1396→        self.edges.push(edge);
  1397→    }
  1398→
  1399→    /// Add a security boundary.
  1400→    pub fn add_boundary(&mut self, boundary: SecurityBoundary) {
  1401→        self.boundaries.insert(boundary.id, boundary);
  1402→    }
  1403→
  1404→    /// Register a valid declassification proof.
  1405→    pub fn register_declassification_proof(&mut self, proof: DeclassificationProof) {
  1406→        self.declassification_proofs.insert(proof.gate, proof);
  1407→    }
  1408→
  1409→    /// Run all security checks and return the aggregated result.
  1410→    pub fn verify(&self) -> VerificationResult {
  1411→        let mut result = VerificationResult::new();
  1412→
  1413→        // 1. Check all data-flow edges for information-flow violations.
  1414→        self.check_information_flow(&mut result);
  1415→
  1416→        // 2. Check tainted data at sinks.
  1417→        self.check_taint_at_sinks(&mut result);
  1418→
  1419→        // 3. Check boundary crossing rules.
  1420→        self.check_boundary_crossings(&mut result);
  1421→
  1422→        // 4. Check capability monotonicity.
  1423→        self.check_capability_monotonicity(&mut result);
  1424→
  1425→        // 5. Check Execute on untrusted sources.
  1426→        self.check_execute_on_untrusted(&mut result);
  1427→
  1428→        // 6. Check declassification proofs for any declassified values.
  1429→        self.check_declassification_proofs(&mut result);
  1430→
  1431→        result
  1432→    }
  1433→
  1434→    /// Check the no-downgrade information-flow rule on all edges.
  1435→    fn check_information_flow(&self, result: &mut VerificationResult) {
  1436→        for edge in &self.edges {
  1437→            let from_node = match self.nodes.get(&edge.from) {
  1438→                Some(n) => n,
  1439→                None => continue,
  1440→            };
  1441→            let to_node = match self.nodes.get(&edge.to) {
  1442→                Some(n) => n,
  1443→                None => continue,
  1444→            };
  1445→
  1446→            match from_node.security.check_flow_to(&to_node.security) {
  1447→                Ok(()) => result.pass(),
  1448→                Err(v) => result.fail(v),
  1449→            }
  1450→        }
  1451→    }
  1452→
  1453→    /// Check that tainted data does not reach clean-only sinks.
  1454→    ///
  1455→    /// Sinks are nodes whose `SecurityRel` is Clean and whose capabilities
  1456→    /// do not include `Send` (indicating they are security-critical consumers
  1457→    /// like system calls, database writes, eval).
  1458→    fn check_taint_at_sinks(&self, result: &mut VerificationResult) {
  1459→        for edge in &self.edges {
  1460→            let from_node = match self.nodes.get(&edge.from) {
  1461→                Some(n) => n,
  1462→                None => continue,
  1463→            };
  1464→            let to_node = match self.nodes.get(&edge.to) {
  1465→                Some(n) => n,
  1466→                None => continue,
  1467→            };
  1468→
  1469→            if from_node.security.taint.is_tainted() && !to_node.security.taint.is_tainted() {
  1470→                // Tainted data flowing to a clean node — this is a potential
  1471→                // taint-at-sink violation. We flag it as a taint propagation
  1472→                // warning if the destination is a sink (no Send capability).
  1473→                if !to_node.capabilities.contains(&SecurityCapability::Send) {
  1474→                    result.fail(SecurityViolation::TaintedDataAtSink {
  1475→                        sources: from_node.security.taint.sources(),
  1476→                        sink: format!("node {}", to_node.id),
  1477→                    });
  1478→                } else {
  1479→                    result.pass();
  1480→                }
  1481→            } else {
  1482→                result.pass();
  1483→            }
  1484→        }
  1485→    }
  1486→
  1487→    /// Check boundary crossing rules (B1–B3).
  1488→    fn check_boundary_crossings(&self, result: &mut VerificationResult) {
  1489→        for edge in &self.edges {
  1490→            let Some(boundary_id) = edge.boundary else {
  1491→                continue;
  1492→            };
  1493→            let Some(boundary) = self.boundaries.get(&boundary_id) else {
  1494→                continue;
  1495→            };
  1496→            let from_node = match self.nodes.get(&edge.from) {
  1497→                Some(n) => n,
  1498→                None => continue,
  1499→            };
  1500→            let to_node = match self.nodes.get(&edge.to) {
  1501→                Some(n) => n,
  1502→                None => continue,
  1503→            };
  1504→
  1505→            // Determine direction.
  1506→            let from_region = from_node.region;
  1507→            let to_region = to_node.region;
  1508→
  1509→            // Only check if the edge actually crosses the boundary.
  1510→            let crosses_high_to_low =
  1511→                from_region == Some(boundary.region_high) && to_region == Some(boundary.region_low);
  1512→            let crosses_low_to_high =
  1513→                from_region == Some(boundary.region_low) && to_region == Some(boundary.region_high);
  1514→
  1515→            if !crosses_high_to_low && !crosses_low_to_high {
  1516→                result.pass();
  1517→                continue;
  1518→            }
  1519→
  1520→            if edge.implicit {
  1521→                // Implicit flow across boundary.
  1522→                if from_node.security.effective_level() > boundary.level_low {
  1523→                    result.fail(SecurityViolation::ImplicitFlowAcrossBoundary {
  1524→                        condition_level: from_node.security.effective_level(),
  1525→                        target_level: boundary.level_low,
  1526→                    });
  1527→                    continue;
  1528→                }
  1529→            }
  1530→
  1531→            if crosses_high_to_low {
  1532→                // B1: Read-across (High → Low).
  1533→                match boundary
  1534→                    .check_read_across(boundary.region_high, from_node.security.effective_level())
  1535→                {
  1536→                    Ok(()) => result.pass(),
  1537→                    Err(v) => result.fail(v),
  1538→                }
  1539→            } else {
  1540→                // Low → High: check control-flow capabilities (B3).
  1541→                match boundary.check_control_flow_across(&from_node.capabilities) {
  1542→                    Ok(()) => result.pass(),
  1543→                    Err(v) => result.fail(v),
  1544→                }
  1545→            }
  1546→        }
  1547→    }
  1548→
  1549→    /// Check that capabilities only decrease over time (monotonicity).
  1550→    ///
  1551→    /// This is a structural check: along each edge, the destination's
  1552→    /// capabilities must be a subset of the source's capabilities.
  1553→    fn check_capability_monotonicity(&self, result: &mut VerificationResult) {
  1554→        for edge in &self.edges {
  1555→            let from_node = match self.nodes.get(&edge.from) {
  1556→                Some(n) => n,
  1557→                None => continue,
  1558→            };
  1559→            let to_node = match self.nodes.get(&edge.to) {
  1560→                Some(n) => n,
  1561→                None => continue,
  1562→            };
  1563→
  1564→            // Check that to_node's capabilities are a subset of from_node's.
  1565→            let added: Vec<_> = to_node
  1566→                .capabilities
  1567→                .difference(&from_node.capabilities)
  1568→                .copied()
  1569→                .collect();
  1570→
  1571→            if added.is_empty() {
  1572→                result.pass();
  1573→            } else {
  1574→                for cap in added {
  1575→                    result.fail(SecurityViolation::CapabilityMonotonicityViolation { added: cap });
  1576→                }
  1577→            }
  1578→        }
  1579→    }
  1580→
  1581→    /// Check that no value from an untrusted source carries Execute.
  1582→    fn check_execute_on_untrusted(&self, result: &mut VerificationResult) {
  1583→        for node in self.nodes.values() {
  1584→            if node.capabilities.contains(&SecurityCapability::Execute)
  1585→                && node.security.taint.is_tainted()
  1586→            {
  1587→                for source in node.security.taint.sources() {
  1588→                    result.fail(SecurityViolation::ExecuteOnUntrusted { source });
  1589→                }
  1590→            } else {
  1591→                result.pass();
  1592→            }
  1593→        }
  1594→    }
  1595→
  1596→    /// Check that declassified values have valid proofs.
  1597→    fn check_declassification_proofs(&self, result: &mut VerificationResult) {
  1598→        for node in self.nodes.values() {
  1599→            if let Some(ref dec) = node.security.declassification {
  1600→                // The gate must have a registered, valid proof.
  1601→                if let Some(proof) = self.declassification_proofs.get(&dec.gate_function) {
  1602→                    if proof.is_valid() {
  1603→                        result.pass();
  1604→                    } else {
  1605→                        result.fail(SecurityViolation::DeclassificationWithoutProof {
  1606→                            from_level: dec.from_level,
  1607→                            to_level: dec.to_level,
  1608→                        });
  1609→                    }
  1610→                } else {
  1611→                    result.fail(SecurityViolation::DeclassificationWithoutProof {
  1612→                        from_level: dec.from_level,
  1613→                        to_level: dec.to_level,
  1614→                    });
  1615→                }
  1616→            }
  1617→        }
  1618→    }
  1619→}
