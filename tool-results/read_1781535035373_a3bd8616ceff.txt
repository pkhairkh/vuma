     1‚Üí//! # DWARF5 Debug Info Generation
     2‚Üí//!
     3‚Üí//! Produces DWARF version 5 debug information sections for the VUMA AArch64
     4‚Üí//! backend.  The emitted sections can be appended to an ELF binary so that
     5‚Üí//! tools such as `readelf`, `objdump`, and `gdb` can decode the program's
     6‚Üí//! structure, function boundaries, local variables, and source-line mapping.
     7‚Üí//!
     8‚Üí//! ## Sections Generated
     9‚Üí//!
    10‚Üí//! | Section          | Contents                                          |
    11‚Üí//! |------------------|---------------------------------------------------|
    12‚Üí//! | `.debug_abbrev`  | Abbreviation tables (tag + attribute encodings)   |
    13‚Üí//! | `.debug_info`    | Compilation unit DIEs (subprograms, variables)    |
    14‚Üí//! | `.debug_line`    | Line-number program (DWARF5 standard opcodes)     |
    15‚Üí//!
    16‚Üí//! ## DWARF5 Encoding
    17‚Üí//!
    18‚Üí//! - Abbreviation codes: `DW_TAG_COMPILE_UNIT`, `DW_TAG_SUBPROGRAM`,
    19‚Üí//!   `DW_TAG_VARIABLE`
    20‚Üí//! - Attributes: `DW_AT_NAME`, `DW_AT_LOW_PC`, `DW_AT_HIGH_PC`,
    21‚Üí//!   `DW_AT_TYPE`, `DW_AT_LOCATION`
    22‚Üí//! - Forms: `DW_FORM_STRING`, `DW_FORM_ADDR`, `DW_FORM_DATA4`,
    23‚Üí//!   `DW_FORM_EXPRLOC`
    24‚Üí//! - Line-number opcodes: `DW_LNS_COPY`, `DW_LNS_ADVANCE_PC`,
    25‚Üí//!   `DW_LNS_ADVANCE_LINE`, `DW_LNS_SET_FILE`, `DW_LNE_END_SEQUENCE`
    26‚Üí
    27‚Üí// ---------------------------------------------------------------------------
    28‚Üí// DWARF5 Constants
    29‚Üí// ---------------------------------------------------------------------------
    30‚Üí
    31‚Üí/// DWARF version number (5).
    32‚Üíconst DWARF_VERSION: u16 = 5;
    33‚Üí
    34‚Üí/// DWARF5 unit type: compile unit.
    35‚Üíconst DW_UT_COMPILE: u8 = 0x01;
    36‚Üí
    37‚Üí/// DWARF5 address size for AArch64 (64-bit).
    38‚Üíconst ADDRESS_SIZE: u8 = 8;
    39‚Üí
    40‚Üí// -- Tags --
    41‚Üí
    42‚Üí/// Tag for a compilation unit.
    43‚Üíconst DW_TAG_COMPILE_UNIT: u16 = 0x11;
    44‚Üí/// Tag for a subprogram (function).
    45‚Üíconst DW_TAG_SUBPROGRAM: u16 = 0x2E;
    46‚Üí/// Tag for a variable.
    47‚Üíconst DW_TAG_VARIABLE: u16 = 0x34;
    48‚Üí
    49‚Üí// -- Attributes --
    50‚Üí
    51‚Üí/// Name attribute.
    52‚Üíconst DW_AT_NAME: u16 = 0x03;
    53‚Üí/// Low PC (start address) attribute.
    54‚Üíconst DW_AT_LOW_PC: u16 = 0x11;
    55‚Üí/// High PC (end address) attribute.
    56‚Üíconst DW_AT_HIGH_PC: u16 = 0x12;
    57‚Üí/// Language attribute.
    58‚Üíconst DW_AT_LANGUAGE: u16 = 0x13;
    59‚Üí/// Producer attribute.
    60‚Üíconst DW_AT_PRODUCER: u16 = 0x25;
    61‚Üí/// Type attribute.
    62‚Üíconst DW_AT_TYPE: u16 = 0x49;
    63‚Üí/// Location attribute.
    64‚Üíconst DW_AT_LOCATION: u16 = 0x02;
    65‚Üí/// Statement list attribute (offset into .debug_line).
    66‚Üíconst DW_AT_STMT_LIST: u16 = 0x10;
    67‚Üí
    68‚Üí// -- Forms --
    69‚Üí
    70‚Üí/// Null form (terminator).
    71‚Üíconst DW_FORM_NONE: u16 = 0x00;
    72‚Üí/// Inline null-terminated string.
    73‚Üíconst DW_FORM_STRING: u16 = 0x08;
    74‚Üí/// Address (8 bytes for 64-bit).
    75‚Üíconst DW_FORM_ADDR: u16 = 0x01;
    76‚Üí/// 4-byte unsigned data.
    77‚Üíconst DW_FORM_DATA4: u16 = 0x06;
    78‚Üí/// 2-byte unsigned data.
    79‚Üíconst DW_FORM_DATA2: u16 = 0x05;
    80‚Üí/// Expression location (ULEB128 length + bytes).
    81‚Üíconst DW_FORM_EXPRLOC: u16 = 0x18;
    82‚Üí/// 4-byte offset into .debug_line.
    83‚Üíconst DW_FORM_SEC_OFFSET: u16 = 0x17;
    84‚Üí
    85‚Üí// -- Children --
    86‚Üí
    87‚Üí/// DIE has children.
    88‚Üíconst DW_CHILDREN_YES: u8 = 0x01;
    89‚Üí/// DIE has no children.
    90‚Üíconst DW_CHILDREN_NO: u8 = 0x00;
    91‚Üí
    92‚Üí// -- Line Number Standard Opcodes --
    93‚Üí
    94‚Üí/// Copy current row to matrix.
    95‚Üíconst DW_LNS_COPY: u8 = 0x01;
    96‚Üí/// Advance PC by operation advance * minimum_instruction_length.
    97‚Üíconst DW_LNS_ADVANCE_PC: u8 = 0x02;
    98‚Üí/// Advance line by signed LEB128 increment.
    99‚Üíconst DW_LNS_ADVANCE_LINE: u8 = 0x03;
   100‚Üí/// Set the file register to the value in the unsigned LEB128 operand.
   101‚Üíconst DW_LNS_SET_FILE: u8 = 0x04;
   102‚Üí
   103‚Üí// -- Line Number Extended Opcodes --
   104‚Üí
   105‚Üí/// End sequence of line number entries.
   106‚Üíconst DW_LNE_END_SEQUENCE: u8 = 0x01;
   107‚Üí
   108‚Üí// -- Language codes --
   109‚Üí
   110‚Üí/// VUMA user language code (reserved range 0x8001‚Äì0xFFFF).
   111‚Üíconst DW_LANG_VUMA: u16 = 0x8001;
   112‚Üí
   113‚Üí// ---------------------------------------------------------------------------

   116‚Üí
   117‚Üí/// A recorded subprogram (function boundary).
   118‚Üí#[derive(Debug, Clone)]
   119‚Üípub struct Subprogram {
   120‚Üí    /// Function name.
   121‚Üí    pub name: String,
   122‚Üí    /// Byte offset (relative to text section start) where the function begins.
   123‚Üí    pub start_offset: u64,
   124‚Üí    /// Byte offset (relative to text section start) where the function ends.
   125‚Üí    pub end_offset: u64,
   126‚Üí}
   127‚Üí
   128‚Üí/// A recorded local variable.
   129‚Üí#[derive(Debug, Clone)]
   130‚Üípub struct Variable {
   131‚Üí    /// Variable name.
   132‚Üí    pub name: String,
   133‚Üí    /// DWARF type name (e.g. "int", "i64").
   134‚Üí    pub type_name: String,
   135‚Üí    /// Stack offset for the variable (used in the DW_OP_fbreg expression).
   136‚Üí    pub offset: i32,
   137‚Üí    /// Register number the variable is in (AArch64 register encoding, 0‚Äì31).
   138‚Üí    pub register: u8,
   139‚Üí}
   140‚Üí
   141‚Üí/// A line-number table entry.
   142‚Üí#[derive(Debug, Clone)]
   143‚Üípub struct LineEntry {
   144‚Üí    /// Byte offset within the text section.
   145‚Üí    pub offset: u64,
   146‚Üí    /// Source file index (1-based in the file table).
   147‚Üí    pub file: u32,
   148‚Üí    /// Line number (1-based).
   149‚Üí    pub line: u32,
   150‚Üí    /// Column number (0 = no column info).
   151‚Üí    pub column: u32,
   152‚Üí}

   157‚Üí
   158‚Üí/// Accumulates debug information during codegen and emits DWARF5 sections.
   159‚Üí///
   160‚Üí/// # Example
   161‚Üí///
   162‚Üí/// ```
   163‚Üí/// use vuma_codegen::dwarf::DwarfBuilder;
   164‚Üí///
   165‚Üí/// let mut db = DwarfBuilder::new();
   166‚Üí/// db.add_compile_unit("test.vuma", "vuma-codegen 0.1");
   167‚Üí/// db.add_subprogram("main", 0, 64);
   168‚Üí/// db.add_variable("x", "i64", -8, 0);
   169‚Üí/// db.add_line_entry(0, 1, 1, 0);
   170‚Üí/// db.add_line_entry(16, 1, 2, 4);
   171‚Üí/// let sections = db.emit_debug_sections();
   172‚Üí/// assert!(!sections.debug_abbrev.is_empty());
   173‚Üí/// assert!(!sections.debug_info.is_empty());
   174‚Üí/// assert!(!sections.debug_line.is_empty());
   175‚Üí/// ```
   176‚Üí#[derive(Debug, Clone)]
   177‚Üípub struct DwarfBuilder {
   178‚Üí    /// Source file path for the compilation unit.
   179‚Üí    source_file: String,
   180‚Üí    /// Producer string (compiler name + version).
   181‚Üí    producer: String,
   182‚Üí    /// Recorded subprograms.
   183‚Üí    subprograms: Vec<Subprogram>,
   184‚Üí    /// Recorded local variables.
   185‚Üí    variables: Vec<Variable>,
   186‚Üí    /// Recorded line-number entries.
   187‚Üí    line_entries: Vec<LineEntry>,
   188‚Üí}
   189‚Üí
   190‚Üí/// The three DWARF5 debug sections emitted by [`DwarfBuilder::emit_debug_sections`].
   191‚Üí#[derive(Debug, Clone)]
   192‚Üípub struct DebugSections {
   193‚Üí    /// `.debug_abbrev` ‚Äî abbreviation table.
   194‚Üí    pub debug_abbrev: Vec<u8>,
   195‚Üí    /// `.debug_info` ‚Äî compilation unit DIEs.
   196‚Üí    pub debug_info: Vec<u8>,
   197‚Üí    /// `.debug_line` ‚Äî line number program.
   198‚Üí    pub debug_line: Vec<u8>,
   199‚Üí}
   200‚Üí
   201‚Üíimpl DwarfBuilder {
   202‚Üí    /// Create a new, empty `DwarfBuilder`.
   203‚Üí    pub fn new() -> Self {
   204‚Üí        Self {
   205‚Üí            source_file: String::new(),
   206‚Üí            producer: String::new(),
   207‚Üí            subprograms: Vec::new(),
   208‚Üí            variables: Vec::new(),
   209‚Üí            line_entries: Vec::new(),
   210‚Üí        }
   211‚Üí    }
   212‚Üí
   213‚Üí    /// Record the top-level compilation unit.
   214‚Üí    pub fn add_compile_unit(&mut self, source_file: &str, producer: &str) {
   215‚Üí        self.source_file = source_file.to_string();
   216‚Üí        self.producer = producer.to_string();
   217‚Üí    }
   218‚Üí
   219‚Üí    /// Record a subprogram (function boundary).
   220‚Üí    ///
   221‚Üí    /// `start_offset` and `end_offset` are byte offsets relative to the
   222‚Üí    /// beginning of the text section.
   223‚Üí    pub fn add_subprogram(&mut self, name: &str, start_offset: u64, end_offset: u64) {
   224‚Üí        self.subprograms.push(Subprogram {
   225‚Üí            name: name.to_string(),
   226‚Üí            start_offset,
   227‚Üí            end_offset,
   228‚Üí        });
   229‚Üí    }
   230‚Üí
   231‚Üí    /// Record a local variable.
   232‚Üí    ///
   233‚Üí    /// `offset` is the stack offset (negative for below the frame pointer).
   234‚Üí    /// `register` is the AArch64 register encoding number (0‚Äì31).
   235‚Üí    pub fn add_variable(&mut self, name: &str, type_name: &str, offset: i32, register: u8) {
   236‚Üí        self.variables.push(Variable {
   237‚Üí            name: name.to_string(),
   238‚Üí            type_name: type_name.to_string(),
   239‚Üí            offset,
   240‚Üí            register,
   241‚Üí        });
   242‚Üí    }
   243‚Üí
   244‚Üí    /// Record a line-number table entry.
   245‚Üí    ///
   246‚Üí    /// `offset` is the byte offset within the text section.
   247‚Üí    /// `file` is a 1-based file index.
   248‚Üí    /// `line` is a 1-based line number.
   249‚Üí    /// `column` is a 0-based column number (0 = whole line).
   250‚Üí    pub fn add_line_entry(&mut self, offset: u64, file: u32, line: u32, column: u32) {
   251‚Üí        self.line_entries.push(LineEntry {
   252‚Üí            offset,
   253‚Üí            file,
   254‚Üí            line,
   255‚Üí            column,
   256‚Üí        });
   257‚Üí    }
   258‚Üí
   259‚Üí    /// Emit all three DWARF5 debug sections.
   260‚Üí    ///
   261‚Üí    /// Returns a [`DebugSections`] containing `.debug_abbrev`, `.debug_info`,
   262‚Üí    /// and `.debug_line`.
   263‚Üí    pub fn emit_debug_sections(&self) -> DebugSections {
   264‚Üí        let debug_abbrev = self.emit_debug_abbrev();
   265‚Üí        let debug_line = self.emit_debug_line();
   266‚Üí        let debug_info = self.emit_debug_info(&debug_abbrev, &debug_line);
   267‚Üí        DebugSections {
   268‚Üí            debug_abbrev,
   269‚Üí            debug_info,
   270‚Üí            debug_line,
   271‚Üí        }
   272‚Üí    }
   273‚Üí
   274‚Üí    // -----------------------------------------------------------------------
   275‚Üí    // .debug_abbrev
   276‚Üí    // -----------------------------------------------------------------------
   277‚Üí
   278‚Üí    /// Emit the `.debug_abbrev` section.
   279‚Üí    ///
   280‚Üí    /// Contains three abbreviation entries:
   281‚Üí    /// - Abbrev 1: `DW_TAG_COMPILE_UNIT` (has children)
   282‚Üí    ///   - `DW_AT_NAME` / `DW_FORM_STRING`
   283‚Üí    ///   - `DW_AT_LANGUAGE` / `DW_FORM_DATA2`
   284‚Üí    ///   - `DW_AT_PRODUCER` / `DW_FORM_STRING`
   285‚Üí    ///   - `DW_AT_LOW_PC` / `DW_FORM_ADDR`
   286‚Üí    ///   - `DW_AT_HIGH_PC` / `DW_FORM_DATA4`
   287‚Üí    ///   - `DW_AT_STMT_LIST` / `DW_FORM_SEC_OFFSET`
   288‚Üí    ///
   289‚Üí    /// - Abbrev 2: `DW_TAG_SUBPROGRAM` (has children)
   290‚Üí    ///   - `DW_AT_NAME` / `DW_FORM_STRING`
   291‚Üí    ///   - `DW_AT_LOW_PC` / `DW_FORM_ADDR`
   292‚Üí    ///   - `DW_AT_HIGH_PC` / `DW_FORM_DATA4`
   293‚Üí    ///
   294‚Üí    /// - Abbrev 3: `DW_TAG_VARIABLE` (no children)
   295‚Üí    ///   - `DW_AT_NAME` / `DW_FORM_STRING`
   296‚Üí    ///   - `DW_AT_TYPE` / `DW_FORM_STRING`
   297‚Üí    ///   - `DW_AT_LOCATION` / `DW_FORM_EXPRLOC`
   298‚Üí    fn emit_debug_abbrev(&self) -> Vec<u8> {
   299‚Üí        let mut buf = Vec::new();
   300‚Üí
   301‚Üí        // Abbrev 1: DW_TAG_COMPILE_UNIT
   302‚Üí        encode_uleb128(&mut buf, 1); // abbreviation code
   303‚Üí        encode_uleb128(&mut buf, DW_TAG_COMPILE_UNIT as u64);
   304‚Üí        buf.push(DW_CHILDREN_YES);
   305‚Üí        // DW_AT_NAME, DW_FORM_STRING
   306‚Üí        encode_uleb128(&mut buf, DW_AT_NAME as u64);
   307‚Üí        encode_uleb128(&mut buf, DW_FORM_STRING as u64);
   308‚Üí        // DW_AT_LANGUAGE, DW_FORM_DATA2
   309‚Üí        encode_uleb128(&mut buf, DW_AT_LANGUAGE as u64);
   310‚Üí        encode_uleb128(&mut buf, DW_FORM_DATA2 as u64);
   311‚Üí        // DW_AT_PRODUCER, DW_FORM_STRING
   312‚Üí        encode_uleb128(&mut buf, DW_AT_PRODUCER as u64);
   313‚Üí        encode_uleb128(&mut buf, DW_FORM_STRING as u64);
   314‚Üí        // DW_AT_LOW_PC, DW_FORM_ADDR
   315‚Üí        encode_uleb128(&mut buf, DW_AT_LOW_PC as u64);
   316‚Üí        encode_uleb128(&mut buf, DW_FORM_ADDR as u64);
   317‚Üí        // DW_AT_HIGH_PC, DW_FORM_DATA4
   318‚Üí        encode_uleb128(&mut buf, DW_AT_HIGH_PC as u64);
   319‚Üí        encode_uleb128(&mut buf, DW_FORM_DATA4 as u64);
   320‚Üí        // DW_AT_STMT_LIST, DW_FORM_SEC_OFFSET
   321‚Üí        encode_uleb128(&mut buf, DW_AT_STMT_LIST as u64);
   322‚Üí        encode_uleb128(&mut buf, DW_FORM_SEC_OFFSET as u64);
   323‚Üí        // Terminator
   324‚Üí        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   325‚Üí        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   326‚Üí
   327‚Üí        // Abbrev 2: DW_TAG_SUBPROGRAM
   328‚Üí        encode_uleb128(&mut buf, 2);
   329‚Üí        encode_uleb128(&mut buf, DW_TAG_SUBPROGRAM as u64);
   330‚Üí        buf.push(DW_CHILDREN_YES);
   331‚Üí        // DW_AT_NAME, DW_FORM_STRING
   332‚Üí        encode_uleb128(&mut buf, DW_AT_NAME as u64);
   333‚Üí        encode_uleb128(&mut buf, DW_FORM_STRING as u64);
   334‚Üí        // DW_AT_LOW_PC, DW_FORM_ADDR
   335‚Üí        encode_uleb128(&mut buf, DW_AT_LOW_PC as u64);
   336‚Üí        encode_uleb128(&mut buf, DW_FORM_ADDR as u64);
   337‚Üí        // DW_AT_HIGH_PC, DW_FORM_DATA4
   338‚Üí        encode_uleb128(&mut buf, DW_AT_HIGH_PC as u64);
   339‚Üí        encode_uleb128(&mut buf, DW_FORM_DATA4 as u64);
   340‚Üí        // Terminator
   341‚Üí        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   342‚Üí        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   343‚Üí
   344‚Üí        // Abbrev 3: DW_TAG_VARIABLE
   345‚Üí        encode_uleb128(&mut buf, 3);
   346‚Üí        encode_uleb128(&mut buf, DW_TAG_VARIABLE as u64);
   347‚Üí        buf.push(DW_CHILDREN_NO);
   348‚Üí        // DW_AT_NAME, DW_FORM_STRING
   349‚Üí        encode_uleb128(&mut buf, DW_AT_NAME as u64);
   350‚Üí        encode_uleb128(&mut buf, DW_FORM_STRING as u64);
   351‚Üí        // DW_AT_TYPE, DW_FORM_STRING (simplified: type name as string)
   352‚Üí        encode_uleb128(&mut buf, DW_AT_TYPE as u64);
   353‚Üí        encode_uleb128(&mut buf, DW_FORM_STRING as u64);
   354‚Üí        // DW_AT_LOCATION, DW_FORM_EXPRLOC
   355‚Üí        encode_uleb128(&mut buf, DW_AT_LOCATION as u64);
   356‚Üí        encode_uleb128(&mut buf, DW_FORM_EXPRLOC as u64);
   357‚Üí        // Terminator
   358‚Üí        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   359‚Üí        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   360‚Üí
   361‚Üí        // End of abbreviation table
   362‚Üí        encode_uleb128(&mut buf, 0);
   363‚Üí
   364‚Üí        buf
   365‚Üí    }
   366‚Üí
   367‚Üí    // -----------------------------------------------------------------------
   368‚Üí    // .debug_info
   369‚Üí    // -----------------------------------------------------------------------
   370‚Üí
   371‚Üí    /// Emit the `.debug_info` section.
   372‚Üí    ///
   373‚Üí    /// Contains a single compilation unit DIE with nested subprogram and
   374‚Üí    /// variable DIEs.
   375‚Üí    fn emit_debug_info(&self, _debug_abbrev: &[u8], _debug_line: &[u8]) -> Vec<u8> {
   376‚Üí        let mut die_buf = Vec::new();
   377‚Üí
   378‚Üí        // -- Compile Unit DIE (abbrev 1) --
   379‚Üí        encode_uleb128(&mut die_buf, 1); // abbrev code
   380‚Üí                                         // DW_AT_NAME (DW_FORM_STRING)
   381‚Üí        write_null_string(&mut die_buf, &self.source_file);
   382‚Üí        // DW_AT_LANGUAGE (DW_FORM_DATA2)
   383‚Üí        die_buf.extend_from_slice(&DW_LANG_VUMA.to_le_bytes());
   384‚Üí        // DW_AT_PRODUCER (DW_FORM_STRING)
   385‚Üí        write_null_string(&mut die_buf, &self.producer);
   386‚Üí        // DW_AT_LOW_PC (DW_FORM_ADDR)
   387‚Üí        let cu_low_pc = self
   388‚Üí            .subprograms
   389‚Üí            .iter()
   390‚Üí            .map(|s| s.start_offset)
   391‚Üí            .min()
   392‚Üí            .unwrap_or(0);
   393‚Üí        die_buf.extend_from_slice(&cu_low_pc.to_le_bytes());
   394‚Üí        // DW_AT_HIGH_PC (DW_FORM_DATA4) ‚Äî offset from low_pc
   395‚Üí        let cu_high_pc: u32 = self
   396‚Üí            .subprograms
   397‚Üí            .iter()
   398‚Üí            .map(|s| s.end_offset)
   399‚Üí            .max()
   400‚Üí            .unwrap_or(0) as u32
   401‚Üí            - cu_low_pc as u32;
   402‚Üí        die_buf.extend_from_slice(&cu_high_pc.to_le_bytes());
   403‚Üí        // DW_AT_STMT_LIST (DW_FORM_SEC_OFFSET) ‚Äî offset into .debug_line
   404‚Üí        die_buf.extend_from_slice(&0u32.to_le_bytes());
   405‚Üí
   406‚Üí        // -- Subprogram DIEs (abbrev 2) --
   407‚Üí        for sub in &self.subprograms {
   408‚Üí            encode_uleb128(&mut die_buf, 2); // abbrev code
   409‚Üí                                             // DW_AT_NAME (DW_FORM_STRING)
   410‚Üí            write_null_string(&mut die_buf, &sub.name);
   411‚Üí            // DW_AT_LOW_PC (DW_FORM_ADDR)
   412‚Üí            die_buf.extend_from_slice(&sub.start_offset.to_le_bytes());
   413‚Üí            // DW_AT_HIGH_PC (DW_FORM_DATA4) ‚Äî offset from low_pc
   414‚Üí            let size = (sub.end_offset - sub.start_offset) as u32;
   415‚Üí            die_buf.extend_from_slice(&size.to_le_bytes());
   416‚Üí
   417‚Üí            // -- Variable DIEs (abbrev 3) ‚Äî nested inside subprogram --
   418‚Üí            for var in &self.variables {
   419‚Üí                encode_uleb128(&mut die_buf, 3); // abbrev code
   420‚Üí                                                 // DW_AT_NAME (DW_FORM_STRING)
   421‚Üí                write_null_string(&mut die_buf, &var.name);
   422‚Üí                // DW_AT_TYPE (DW_FORM_STRING)
   423‚Üí                write_null_string(&mut die_buf, &var.type_name);
   424‚Üí                // DW_AT_LOCATION (DW_FORM_EXPRLOC)
   425‚Üí                // DW_OP_fbreg <offset>  =  0x91 + signed LEB128
   426‚Üí                let expr = build_fbreg_expr(var.offset);
   427‚Üí                encode_uleb128(&mut die_buf, expr.len() as u64);
   428‚Üí                die_buf.extend_from_slice(&expr);
   429‚Üí            }
   430‚Üí
   431‚Üí            // Null terminator for subprogram children
   432‚Üí            encode_uleb128(&mut die_buf, 0);
   433‚Üí        }
   434‚Üí
   435‚Üí        // Null terminator for compile_unit children
   436‚Üí        encode_uleb128(&mut die_buf, 0);
   437‚Üí
   438‚Üí        // Now build the full .debug_info section with the compilation unit
   439‚Üí        // header.
   440‚Üí        let mut buf = Vec::new();
   441‚Üí
   442‚Üí        // Unit length (4-byte length, DWARF5 format) ‚Äî we fill this in after.
   443‚Üí        let unit_length = (die_buf.len() as u32)
   444‚Üí            // header fields after the unit_length itself:
   445‚Üí            + 2  // version
   446‚Üí            + 1  // unit_type
   447‚Üí            + 1  // address_size
   448‚Üí            + 4; // debug_abbrev_offset
   449‚Üí        buf.extend_from_slice(&unit_length.to_le_bytes());
   450‚Üí        // Version
   451‚Üí        buf.extend_from_slice(&DWARF_VERSION.to_le_bytes());
   452‚Üí        // Unit type
   453‚Üí        buf.push(DW_UT_COMPILE);
   454‚Üí        // Address size
   455‚Üí        buf.push(ADDRESS_SIZE);
   456‚Üí        // Debug abbrev offset (always 0 ‚Äî we have a single abbrev table)
   457‚Üí        buf.extend_from_slice(&0u32.to_le_bytes());
   458‚Üí
   459‚Üí        // DIE data
   460‚Üí        buf.extend_from_slice(&die_buf);
   461‚Üí
   462‚Üí        buf
   463‚Üí    }
   464‚Üí
   465‚Üí    // -----------------------------------------------------------------------
   466‚Üí    // .debug_line
   467‚Üí    // -----------------------------------------------------------------------
   468‚Üí
   469‚Üí    /// Emit the `.debug_line` section containing a DWARF5 line-number program.
   470‚Üí    fn emit_debug_line(&self) -> Vec<u8> {
   471‚Üí        let mut program = Vec::new();
   472‚Üí
   473‚Üí        // Sort line entries by offset for a well-formed program.
   474‚Üí        let mut entries = self.line_entries.clone();
   475‚Üí        entries.sort_by_key(|e| e.offset);
   476‚Üí
   477‚Üí        // Default_is_stmt: true (beginning of a statement)
   478‚Üí        let default_is_stmt: u8 = 1;
   479‚Üí        // Line base: -5 (standard DWARF range)
   480‚Üí        let line_base: i8 = -5;
   481‚Üí        // Line range: 14 (standard DWARF value)
   482‚Üí        let line_range: u8 = 14;
   483‚Üí        // Opcode base: 13 (standard DWARF5 value)
   484‚Üí        let opcode_base: u8 = 13;
   485‚Üí        // Standard opcode lengths for opcodes 1..12
   486‚Üí        let standard_opcode_lengths: [u8; 12] = [0, 1, 1, 1, 1, 0, 0, 0, 1, 0, 0, 1];
   487‚Üí        // Minimum instruction length: 4 (AArch64)
   488‚Üí        let min_inst_length: u8 = 4;
   489‚Üí        // Maximum operations per instruction: 1
   490‚Üí        let max_ops_per_inst: u8 = 1;
   491‚Üí
   492‚Üí        // -- Build line program opcodes --
   493‚Üí        let mut current_line: i32 = 1;
   494‚Üí        let mut current_file: u32 = 1;
   495‚Üí        let mut current_offset: u64 = 0;
   496‚Üí
   497‚Üí        for entry in &entries {
   498‚Üí            // Set file if different
   499‚Üí            if entry.file != current_file {
   500‚Üí                program.push(DW_LNS_SET_FILE);
   501‚Üí                encode_uleb128(&mut program, entry.file as u64);
   502‚Üí                current_file = entry.file;
   503‚Üí            }
   504‚Üí
   505‚Üí            // Advance line if different
   506‚Üí            let line_diff = (entry.line as i32) - current_line;
   507‚Üí            if line_diff != 0 {
   508‚Üí                program.push(DW_LNS_ADVANCE_LINE);
   509‚Üí                encode_sleb128(&mut program, i64::from(line_diff));
   510‚Üí                current_line = entry.line as i32;
   511‚Üí            }
   512‚Üí
   513‚Üí            // Advance PC
   514‚Üí            if entry.offset > current_offset {
   515‚Üí                let addr_advance = entry.offset - current_offset;
   516‚Üí                // DW_LNS_ADVANCE_PC advances by the operand * min_inst_length
   517‚Üí                // Since min_inst_length = 4, operand = addr_advance / 4
   518‚Üí                let op_advance = addr_advance / (min_inst_length as u64);
   519‚Üí                if op_advance > 0 {
   520‚Üí                    program.push(DW_LNS_ADVANCE_PC);
   521‚Üí                    encode_uleb128(&mut program, op_advance);
   522‚Üí                }
   523‚Üí                current_offset = entry.offset;
   524‚Üí            }
   525‚Üí
   526‚Üí            // Copy current row
   527‚Üí            program.push(DW_LNS_COPY);
   528‚Üí        }
   529‚Üí
   530‚Üí        // End sequence: advance PC to end and emit DW_LNE_END_SEQUENCE
   531‚Üí        // Use the last entry's offset + some delta, or 0 if no entries
   532‚Üí        let end_addr = entries.last().map(|e| e.end_offset()).unwrap_or(0);
   533‚Üí        if end_addr > current_offset {
   534‚Üí            let op_advance = (end_addr - current_offset) / (min_inst_length as u64);
   535‚Üí            if op_advance > 0 {
   536‚Üí                program.push(DW_LNS_ADVANCE_PC);
   537‚Üí                encode_uleb128(&mut program, op_advance);
   538‚Üí            }
   539‚Üí        }
   540‚Üí        // DW_LNE_END_SEQUENCE: extended opcode
   541‚Üí        let ext_opcode_bytes = [DW_LNE_END_SEQUENCE];
   542‚Üí        encode_uleb128(&mut program, (ext_opcode_bytes.len() as u64) + 1); // length (includes sub-opcode)
   543‚Üí        program.push(DW_LNE_END_SEQUENCE);
   544‚Üí
   545‚Üí        // -- Build directory entry format and table --
   546‚Üí        // DWARF5: directory entry format count
   547‚Üí        let dir_entry_format_count: u8 = 1;
   548‚Üí        // Directory entry format: (DW_LNCT_path, DW_FORM_STRING)
   549‚Üí        let dir_entry_format: Vec<u8> = {
   550‚Üí            let mut f = Vec::new();
   551‚Üí            encode_uleb128(&mut f, 0x01); // DW_LNCT_path
   552‚Üí            encode_uleb128(&mut f, DW_FORM_STRING as u64);
   553‚Üí            f
   554‚Üí        };
   555‚Üí
   556‚Üí        // Directories: just one ‚Äî the compilation directory (".")
   557‚Üí        let directories_count: u8 = 1;
   558‚Üí        let directory_name = b".\0";
   559‚Üí
   560‚Üí        // -- Build file name entry format and table --
   561‚Üí        // DWARF5: file name entry format count
   562‚Üí        let file_entry_format_count: u8 = 2;
   563‚Üí        // File entry format: (DW_LNCT_path, DW_FORM_STRING), (DW_LNCT_directory_index, DW_FORM_DATA4)
   564‚Üí        let file_entry_format: Vec<u8> = {
   565‚Üí            let mut f = Vec::new();
   566‚Üí            encode_uleb128(&mut f, 0x01); // DW_LNCT_path
   567‚Üí            encode_uleb128(&mut f, DW_FORM_STRING as u64);
   568‚Üí            encode_uleb128(&mut f, 0x02); // DW_LNCT_directory_index
   569‚Üí            encode_uleb128(&mut f, DW_FORM_DATA4 as u64);
   570‚Üí            f
   571‚Üí        };
   572‚Üí
   573‚Üí        // File names: just one ‚Äî the source file
   574‚Üí        let file_names_count: u8 = 1;
   575‚Üí        let file_name_entry = {
   576‚Üí            let mut f = Vec::new();
   577‚Üí            f.extend_from_slice(self.source_file.as_bytes());
   578‚Üí            f.push(0); // null-terminated
   579‚Üí            f.extend_from_slice(&0u32.to_le_bytes()); // directory index 0
   580‚Üí            f
   581‚Üí        };
   582‚Üí
   583‚Üí        // -- Assemble the line number program header --
   584‚Üí        let mut header = Vec::new();
   585‚Üí
   586‚Üí        // Version
   587‚Üí        header.extend_from_slice(&DWARF_VERSION.to_le_bytes());
   588‚Üí
   589‚Üí        // Address size
   590‚Üí        header.push(ADDRESS_SIZE);
   591‚Üí
   592‚Üí        // Segment selector size (DWARF5: 0)
   593‚Üí        header.push(0);
   594‚Üí
   595‚Üí        // Header length ‚Äî will be filled after we know the size
   596‚Üí        let header_before_length_field = header.len();
   597‚Üí        // Placeholder for 4-byte header length
   598‚Üí        header.extend_from_slice(&0u32.to_le_bytes());
   599‚Üí
   600‚Üí        // Minimum instruction length
   601‚Üí        header.push(min_inst_length);
   602‚Üí        // Maximum operations per instruction
   603‚Üí        header.push(max_ops_per_inst);
   604‚Üí        // Default is_stmt
   605‚Üí        header.push(default_is_stmt);
   606‚Üí        // Line base
   607‚Üí        header.push(line_base as u8);
   608‚Üí        // Line range
   609‚Üí        header.push(line_range);
   610‚Üí        // Opcode base
   611‚Üí        header.push(opcode_base);
   612‚Üí        // Standard opcode lengths
   613‚Üí        header.extend_from_slice(&standard_opcode_lengths);
   614‚Üí
   615‚Üí        // Directory entry format count
   616‚Üí        header.push(dir_entry_format_count);
   617‚Üí        // Directory entry format
   618‚Üí        header.extend_from_slice(&dir_entry_format);
   619‚Üí        // Directories count
   620‚Üí        encode_uleb128(&mut header, directories_count as u64);
   621‚Üí        // Directory entries
   622‚Üí        header.extend_from_slice(directory_name);
   623‚Üí
   624‚Üí        // File name entry format count
   625‚Üí        header.push(file_entry_format_count);
   626‚Üí        // File name entry format
   627‚Üí        header.extend_from_slice(&file_entry_format);
   628‚Üí        // File names count
   629‚Üí        encode_uleb128(&mut header, file_names_count as u64);
   630‚Üí        // File name entries
   631‚Üí        header.extend_from_slice(&file_name_entry);
   632‚Üí
   633‚Üí        // Fix up header_length (bytes after the header_length field itself)
   634‚Üí        let header_length = (header.len() - header_before_length_field - 4) as u32;
   635‚Üí        let hl_offset = header_before_length_field;
   636‚Üí        header[hl_offset..hl_offset + 4].copy_from_slice(&header_length.to_le_bytes());
   637‚Üí
   638‚Üí        // -- Assemble full .debug_line section --
   639‚Üí        let mut buf = Vec::new();
   640‚Üí
   641‚Üí        // Unit length (4 bytes) ‚Äî total length of the rest of the unit
   642‚Üí        let unit_length = (header.len() + program.len()) as u32;
   643‚Üí        buf.extend_from_slice(&unit_length.to_le_bytes());
   644‚Üí        buf.extend_from_slice(&header);
   645‚Üí        buf.extend_from_slice(&program);
   646‚Üí
   647‚Üí        buf
   648‚Üí    }
   649‚Üí}

   653‚Üí        Self::new()

   660‚Üí
   661‚Üí/// Encode an unsigned LEB128 value into `buf`.
   662‚Üífn encode_uleb128(buf: &mut Vec<u8>, mut value: u64) {
   663‚Üí    loop {
   664‚Üí        let mut byte = (value & 0x7F) as u8;
   665‚Üí        value >>= 7;
   666‚Üí        if value != 0 {
   667‚Üí            byte |= 0x80;
   668‚Üí        }
   669‚Üí        buf.push(byte);
   670‚Üí        if value == 0 {
   671‚Üí            break;
   672‚Üí        }
   673‚Üí    }
   674‚Üí}
   675‚Üí
   676‚Üí/// Encode a signed LEB128 value into `buf`.
   677‚Üífn encode_sleb128(buf: &mut Vec<u8>, mut value: i64) {
   678‚Üí    loop {
   679‚Üí        let mut byte = (value & 0x7F) as u8;
   680‚Üí        value >>= 7;
   681‚Üí        let sign_bit_set = (byte & 0x40) != 0;
   682‚Üí        let done = value == 0 || (value == -1 && sign_bit_set);
   683‚Üí        if !done {
   684‚Üí            byte |= 0x80;
   685‚Üí        }
   686‚Üí        buf.push(byte);
   687‚Üí        if done {
   688‚Üí            break;
   689‚Üí        }
   690‚Üí    }
   691‚Üí}
   692‚Üí
   693‚Üí/// Write a null-terminated string into `buf`.
   694‚Üífn write_null_string(buf: &mut Vec<u8>, s: &str) {
   695‚Üí    buf.extend_from_slice(s.as_bytes());
   696‚Üí    buf.push(0);
   697‚Üí}
   698‚Üí
   699‚Üí/// Build a `DW_OP_fbreg <offset>` expression.
   700‚Üí///
   701‚Üí/// `DW_OP_fbreg` (0x91) followed by a signed LEB128 offset.
   702‚Üífn build_fbreg_expr(offset: i32) -> Vec<u8> {
   703‚Üí    let mut expr = vec![0x91]; // DW_OP_fbreg
   704‚Üí    encode_sleb128(&mut expr, offset as i64);
   705‚Üí    expr
   706‚Üí}

   712‚Üíimpl LineEntry {
   713‚Üí    /// Compute a reasonable end-offset for this line entry.

   718‚Üí    }

   721‚Üí// ---------------------------------------------------------------------------
   722‚Üí// ELF Integration
   723‚Üí// ---------------------------------------------------------------------------
   724‚Üí
   725‚Üí/// Section header type for DWARF debug info sections (progbits).
   726‚Üíconst SHT_PROGBITS: u32 = 1;
   727‚Üí
   728‚Üí/// Append DWARF5 debug sections to an ELF binary.
   729‚Üí///
   730‚Üí/// Given an existing ELF binary and the debug sections, this function inserts
   731‚Üí/// the three debug sections (`.debug_abbrev`, `.debug_info`, `.debug_line`)
   732‚Üí/// into the ELF, updates the section header table, and patches the section
   733‚Üí/// header string table and section count.
   734‚Üí///
   735‚Üí/// This is a simplified integration that appends sections before the section
   736‚Üí/// headers. It assumes the input ELF was produced by `emit_elf` with
   737‚Üí/// section headers enabled.
   738‚Üípub fn append_debug_sections_to_elf(elf: &mut Vec<u8>, debug: &DebugSections) {
   739‚Üí    // Parse the ELF header to find section header table location and count.
   740‚Üí    if elf.len() < 64 {
   741‚Üí        return;
   742‚Üí    }
   743‚Üí
   744‚Üí    let e_shoff = u64::from_le_bytes(elf[40..48].try_into().unwrap_or([0; 8]));
   745‚Üí    let e_shentsize = u16::from_le_bytes(elf[58..60].try_into().unwrap_or([0; 2])) as u64;
   746‚Üí    let e_shnum = u16::from_le_bytes(elf[60..62].try_into().unwrap_or([0; 2])) as usize;
   747‚Üí    let e_shstrndx = u16::from_le_bytes(elf[62..64].try_into().unwrap_or([0; 2])) as usize;
   748‚Üí
   749‚Üí    if e_shoff == 0 || e_shnum == 0 || e_shentsize == 0 {
   750‚Üí        return;
   751‚Üí    }
   752‚Üí
   753‚Üí    // Read the .shstrtab section to find its data.
   754‚Üí    let shstrtab_shdr_offset = e_shoff as usize + e_shstrndx * e_shentsize as usize;
   755‚Üí    if shstrtab_shdr_offset + 64 > elf.len() {
   756‚Üí        return;
   757‚Üí    }
   758‚Üí    let shstrtab_offset = u64::from_le_bytes(
   759‚Üí        elf[shstrtab_shdr_offset + 24..shstrtab_shdr_offset + 32]
   760‚Üí            .try_into()
   761‚Üí            .unwrap_or([0; 8]),
   762‚Üí    ) as usize;
   763‚Üí    let shstrtab_size = u64::from_le_bytes(
   764‚Üí        elf[shstrtab_shdr_offset + 32..shstrtab_shdr_offset + 40]
   765‚Üí            .try_into()
   766‚Üí            .unwrap_or([0; 8]),
   767‚Üí    ) as usize;
   768‚Üí
   769‚Üí    // Build new shstrtab with debug section names appended.
   770‚Üí    let mut new_shstrtab = elf[shstrtab_offset..shstrtab_offset + shstrtab_size].to_vec();
   771‚Üí
   772‚Üí    // Record name offsets
   773‚Üí    let debug_abbrev_name_off = new_shstrtab.len() as u32;
   774‚Üí    new_shstrtab.extend_from_slice(b".debug_abbrev\0");
   775‚Üí    let debug_info_name_off = new_shstrtab.len() as u32;
   776‚Üí    new_shstrtab.extend_from_slice(b".debug_info\0");
   777‚Üí    let debug_line_name_off = new_shstrtab.len() as u32;
   778‚Üí    new_shstrtab.extend_from_slice(b".debug_line\0");
   779‚Üí
   780‚Üí    // The new sections will be appended right before the section header table.
   781‚Üí    // We need to compute their file offsets.
   782‚Üí    let new_sections_start = e_shoff as usize;
   783‚Üí
   784‚Üí    // Compute aligned sizes for debug sections.
   785‚Üí    let debug_abbrev_aligned = align_up(debug.debug_abbrev.len() as u64, 8) as usize;
   786‚Üí    let debug_info_aligned = align_up(debug.debug_info.len() as u64, 8) as usize;
   787‚Üí    let debug_line_aligned = align_up(debug.debug_line.len() as u64, 8) as usize;
   788‚Üí    let new_shstrtab_aligned = align_up(new_shstrtab.len() as u64, 8) as usize;
   789‚Üí
   790‚Üí    // Build section data to insert.
   791‚Üí    let mut new_section_data = Vec::new();
   792‚Üí
   793‚Üí    let debug_abbrev_file_offset = new_sections_start + new_section_data.len();
   794‚Üí    new_section_data.extend_from_slice(&debug.debug_abbrev);
   795‚Üí    let pad = debug_abbrev_aligned - debug.debug_abbrev.len();
   796‚Üí    new_section_data.extend_from_slice(&vec![0u8; pad]);
   797‚Üí
   798‚Üí    let debug_info_file_offset = new_sections_start + new_section_data.len();
   799‚Üí    new_section_data.extend_from_slice(&debug.debug_info);
   800‚Üí    let pad = debug_info_aligned - debug.debug_info.len();
   801‚Üí    new_section_data.extend_from_slice(&vec![0u8; pad]);
   802‚Üí
   803‚Üí    let debug_line_file_offset = new_sections_start + new_section_data.len();
   804‚Üí    new_section_data.extend_from_slice(&debug.debug_line);
   805‚Üí    let pad = debug_line_aligned - debug.debug_line.len();
   806‚Üí    new_section_data.extend_from_slice(&vec![0u8; pad]);
   807‚Üí
   808‚Üí    // Replace the old shstrtab with the new one.
   809‚Üí    // We'll place the new shstrtab at the end of the debug sections.
   810‚Üí    let new_shstrtab_file_offset = new_sections_start + new_section_data.len();
   811‚Üí    new_section_data.extend_from_slice(&new_shstrtab);
   812‚Üí    let pad = new_shstrtab_aligned - new_shstrtab.len();
   813‚Üí    new_section_data.extend_from_slice(&vec![0u8; pad]);
   814‚Üí
   815‚Üí    // Now insert the new section data before the section headers.
   816‚Üí    // First, remove the old section headers from the end.
   817‚Üí    let old_shdrs: Vec<u8> = elf.drain(e_shoff as usize..).collect();
   818‚Üí
   819‚Üí    // Insert debug section data.
   820‚Üí    elf.extend_from_slice(&new_section_data);
   821‚Üí
   822‚Üí    // Compute the new section header table offset.
   823‚Üí    let new_e_shoff = elf.len() as u64;
   824‚Üí
   825‚Üí    // Build new section headers: copy old ones, add debug section headers,
   826‚Üí    // and update the .shstrtab header.
   827‚Üí    let mut new_shdrs = old_shdrs;
   828‚Üí
   829‚Üí    // Update the .shstrtab section header to point to the new data.
   830‚Üí    let shstrtab_shdr_start = e_shstrndx * 64;
   831‚Üí    if shstrtab_shdr_start + 64 <= new_shdrs.len() {
   832‚Üí        // sh_offset (bytes 24-31)
   833‚Üí        new_shdrs[shstrtab_shdr_start + 24..shstrtab_shdr_start + 32]
   834‚Üí            .copy_from_slice(&new_shstrtab_file_offset.to_le_bytes());
   835‚Üí        // sh_size (bytes 32-39)
   836‚Üí        new_shdrs[shstrtab_shdr_start + 32..shstrtab_shdr_start + 40]
   837‚Üí            .copy_from_slice(&(new_shstrtab.len() as u64).to_le_bytes());
   838‚Üí    }
   839‚Üí
   840‚Üí    // Add section headers for the three debug sections.
   841‚Üí    let new_num_shdrs = e_shnum + 3;
   842‚Üí
   843‚Üí    // .debug_abbrev section header
   844‚Üí    let mut sh = new_shdr(
   845‚Üí        SHT_PROGBITS,
   846‚Üí        0, // no flags
   847‚Üí        0, // no virtual address
   848‚Üí        debug_abbrev_file_offset as u64,
   849‚Üí        debug.debug_abbrev.len() as u64,
   850‚Üí        0,
   851‚Üí        0,
   852‚Üí        1,
   853‚Üí        0,
   854‚Üí    );
   855‚Üí    sh.name = debug_abbrev_name_off;
   856‚Üí    write_filled_shdr(&mut new_shdrs, &sh);
   857‚Üí
   858‚Üí    // .debug_info section header
   859‚Üí    let mut sh = new_shdr(
   860‚Üí        SHT_PROGBITS,
   861‚Üí        0,
   862‚Üí        0,
   863‚Üí        debug_info_file_offset as u64,
   864‚Üí        debug.debug_info.len() as u64,
   865‚Üí        0,
   866‚Üí        0,
   867‚Üí        1,
   868‚Üí        0,
   869‚Üí    );
   870‚Üí    sh.name = debug_info_name_off;
   871‚Üí    write_filled_shdr(&mut new_shdrs, &sh);
   872‚Üí
   873‚Üí    // .debug_line section header
   874‚Üí    let mut sh = new_shdr(
   875‚Üí        SHT_PROGBITS,
   876‚Üí        0,
   877‚Üí        0,
   878‚Üí        debug_line_file_offset as u64,
   879‚Üí        debug.debug_line.len() as u64,
   880‚Üí        0,
   881‚Üí        0,
   882‚Üí        1,
   883‚Üí        0,
   884‚Üí    );
   885‚Üí    sh.name = debug_line_name_off;
   886‚Üí    write_filled_shdr(&mut new_shdrs, &sh);
   887‚Üí
   888‚Üí    // Append section headers.
   889‚Üí    elf.extend_from_slice(&new_shdrs);
   890‚Üí
   891‚Üí    // Update the ELF header: e_shoff and e_shnum.
   892‚Üí    elf[40..48].copy_from_slice(&new_e_shoff.to_le_bytes());
   893‚Üí    elf[60..62].copy_from_slice(&(new_num_shdrs as u16).to_le_bytes());
   894‚Üí    // e_shstrndx stays the same ‚Äî the .shstrtab section index hasn't changed.
   895‚Üí}

   962‚Üí
   963‚Üí/// Decode a ULEB128 value from `buf` starting at `offset`.
   964‚Üí/// Returns (value, bytes_consumed).
   965‚Üí#[cfg(test)]
   966‚Üífn decode_uleb128(buf: &[u8], offset: usize) -> (u64, usize) {
   967‚Üí    let mut value: u64 = 0;
   968‚Üí    let mut shift: u32 = 0;
   969‚Üí    let mut i = offset;
   970‚Üí    loop {
   971‚Üí        let byte = buf[i];
   972‚Üí        value |= ((byte & 0x7F) as u64) << shift;
   973‚Üí        shift += 7;
   974‚Üí        i += 1;
   975‚Üí        if (byte & 0x80) == 0 {
   976‚Üí            break;
   977‚Üí        }
   978‚Üí    }
   979‚Üí    (value, i - offset)
   980‚Üí}
   981‚Üí
   982‚Üí/// Decode a SLEB128 value from `buf` starting at `offset`.
   983‚Üí/// Returns (value, bytes_consumed).
   984‚Üí#[cfg(test)]
   985‚Üífn decode_sleb128(buf: &[u8], offset: usize) -> (i64, usize) {
   986‚Üí    let mut value: i64 = 0;
   987‚Üí    let mut shift: u32 = 0;
   988‚Üí    let mut i = offset;
   989‚Üí    let mut byte: u8;
   990‚Üí    loop {
   991‚Üí        byte = buf[i];
   992‚Üí        value |= ((byte & 0x7F) as i64) << shift;
   993‚Üí        shift += 7;
   994‚Üí        i += 1;
   995‚Üí        if (byte & 0x80) == 0 {
   996‚Üí            break;
   997‚Üí        }
   998‚Üí    }
   999‚Üí    // Sign extend
  1000‚Üí    if shift < 64 && (byte & 0x40) != 0 {
  1001‚Üí        value |= !0 << shift;
  1002‚Üí    }
  1003‚Üí    (value, i - offset)
  1004‚Üí}
