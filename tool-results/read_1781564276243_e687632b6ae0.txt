   315‚Üí   315‚Üí/// Target-specific information needed during code generation.
   316‚Üí   316‚Üí///
   317‚Üí   317‚Üí/// This trait provides a data-driven interface for query target properties.
   318‚Üí   318‚Üí/// It must be implementable for ALL supported ISAs, including fundamentally
   319‚Üí   319‚Üí/// different architectures like Wasm (stack machine, no registers) and
   320‚Üí   320‚Üí/// MIPS (branch delay slots).
   321‚Üí   321‚Üí///
   322‚Üí   322‚Üí/// # Object Safety
   323‚Üí   323‚Üí///
   324‚Üí   324‚Üí/// This trait is object-safe: all methods take `&self` and return only
   325‚Üí   325‚Üí/// owned types or references with `'static` lifetime.
   326‚Üí   326‚Üípub trait TargetInfo: Send + Sync + 'static {
   327‚Üí   327‚Üí    // === Identity ===
   328‚Üí   328‚Üí
   329‚Üí   329‚Üí    /// ISA name in lowercase (e.g., "aarch64", "riscv64", "wasm32").
   330‚Üí   330‚Üí    fn isa_name(&self) -> &'static str;
   331‚Üí   331‚Üí

  1147‚Üí  1147‚Üí// AArch64 Mnemonic Decoder
  1148‚Üí  1148‚Üí// ---------------------------------------------------------------------------
  1149‚Üí  1149‚Üí
  1150‚Üí  1150‚Üí/// Decode a 32-bit AArch64 instruction word into a human-readable mnemonic.
  1151‚Üí  1151‚Üí///
  1152‚Üí  1152‚Üí/// Covers the most common AArch64 instructions: ADD, SUB, MOV, LDR, STR, B,
  1153‚Üí  1153‚Üí/// BL, RET, CMP, B.cond, STP, LDP, NOP, MUL, SDIV, UDIV, AND, ORR, EOR,
  1154‚Üí  1154‚Üí/// plus several additional frequently-encountered encodings.
  1155‚Üí  1155‚Üífn decode_aarch64(word: u32) -> String {
  1156‚Üí  1156‚Üí    let rd = word & 0x1F;
  1157‚Üí  1157‚Üí    let rn = (word >> 5) & 0x1F;
  1158‚Üí  1158‚Üí    let rt = rd; // alias for load/store destination
  1159‚Üí  1159‚Üí    let rm = (word >> 16) & 0x1F;
  1160‚Üí  1160‚Üí    let imm12 = (word >> 10) & 0xFFF;
  1161‚Üí  1161‚Üí    let cond = word & 0xF;
  1162‚Üí  1162‚Üí
  1163‚Üí  1163‚Üí    // NOP: d503201f
  1164‚Üí  1164‚Üí    if word == 0xD503201F {
  1165‚Üí  1165‚Üí        return "nop".to_string();
  1166‚Üí  1166‚Üí    }
  1167‚Üí  1167‚Üí
  1168‚Üí  1168‚Üí    // RET: d65f03c0
  1169‚Üí  1169‚Üí    if word == 0xD65F03C0 {
  1170‚Üí  1170‚Üí        return "ret".to_string();
  1171‚Üí  1171‚Üí    }
  1172‚Üí  1172‚Üí
  1173‚Üí  1173‚Üí    let _top8 = word >> 24;
  1174‚Üí  1174‚Üí    let _top10 = word >> 22;
  1175‚Üí  1175‚Üí
  1176‚Üí  1176‚Üí    // --- ADD/SUB (immediate): 100100xx ...
  1177‚Üí  1177‚Üí    if (word >> 23) & 0x1FF == 0b1_0010_0010 {
  1178‚Üí  1178‚Üí        // ADD Xd, Xn, #imm12
  1179‚Üí  1179‚Üí        return format!("add x{}, x{}, #{}", rd, rn, imm12);
  1180‚Üí  1180‚Üí    }
  1181‚Üí  1181‚Üí    if (word >> 23) & 0x1FF == 0b1_1010_0010 {
  1182‚Üí  1182‚Üí        // SUB Xd, Xn, #imm12
  1183‚Üí  1183‚Üí        return format!("sub x{}, x{}, #{}", rd, rn, imm12);
  1184‚Üí  1184‚Üí    }
  1185‚Üí  1185‚Üí
  1186‚Üí  1186‚Üí    // --- ADD (shifted register): 1_00_0101_1_xxx ...
  1187‚Üí  1187‚Üí    if (word >> 24) & 0xFF == 0b1000_1011 {
  1188‚Üí  1188‚Üí        // ADD Xd, Xn, Xm
  1189‚Üí  1189‚Üí        return format!("add x{}, x{}, x{}", rd, rn, rm);
  1190‚Üí  1190‚Üí    }
  1191‚Üí  1191‚Üí
  1192‚Üí  1192‚Üí    // --- SUB (shifted register): 1_00_0101_1_xxx ... with S=1 at bit30
  1193‚Üí  1193‚Üí    if (word >> 24) & 0xFF == 0b1101_0110 {
  1194‚Üí  1194‚Üí        // SUB Xd, Xn, Xm (bit 30 set = sub)
  1195‚Üí  1195‚Üí        return format!("sub x{}, x{}, x{}", rd, rn, rm);
  1196‚Üí  1196‚Üí    }
  1197‚Üí  1197‚Üí
  1198‚Üí  1198‚Üí    // --- AND (shifted register): 1_00_0101_0_00_xxx
  1199‚Üí  1199‚Üí    if (word >> 24) & 0xFE == 0b1000_1010 {
  1200‚Üí  1200‚Üí        // Check bit 21-15: opcode[31:21] = 10001010_000
  1201‚Üí  1201‚Üí        if (word >> 21) & 0x7FF == 0b10001010000 {
  1202‚Üí  1202‚Üí            return format!("and x{}, x{}, x{}", rd, rn, rm);
  1203‚Üí  1203‚Üí        }
  1204‚Üí  1204‚Üí    }
  1205‚Üí  1205‚Üí
  1206‚Üí  1206‚Üí    // --- ORR (shifted register): 1_01_0101_0_00_xxx
  1207‚Üí  1207‚Üí    if (word >> 21) & 0x7FF == 0b10101010000 {
  1208‚Üí  1208‚Üí        return format!("orr x{}, x{}, x{}", rd, rn, rm);
  1209‚Üí  1209‚Üí    }
  1210‚Üí  1210‚Üí
  1211‚Üí  1211‚Üí    // --- EOR (shifted register): 1_10_0101_0_00_xxx
  1212‚Üí  1212‚Üí    if (word >> 21) & 0x7FF == 0b11001010000 {
  1213‚Üí  1213‚Üí        return format!("eor x{}, x{}, x{}", rd, rn, rm);
  1214‚Üí  1214‚Üí    }
  1215‚Üí  1215‚Üí
  1216‚Üí  1216‚Üí    // --- MOV (register): alias for ORR Xd, XZR, Xm
  1217‚Üí  1217‚Üí    // ORR Xd, XZR, Xm: 10101010_000_rm_000000_xzr_rd
  1218‚Üí  1218‚Üí    // More general: ORR with Rn=XZR(31)
  1219‚Üí  1219‚Üí    if (word >> 21) & 0x7FF == 0b10101010000 && rn == 31 {
  1220‚Üí  1220‚Üí        return format!("mov x{}, x{}", rd, rm);
  1221‚Üí  1221‚Üí    }
  1222‚Üí  1222‚Üí
  1223‚Üí  1223‚Üí    // --- MUL: MADD Xd, Xn, Xm, XZR
  1224‚Üí  1224‚Üí    // Encoding: 1_00_1101_1_000_Rm_0_01111_Rn_Rd
  1225‚Üí  1225‚Üí    if (word >> 21) & 0x7FF == 0b10011011000 && ((word >> 10) & 0x1F) == 0b01111 {
  1226‚Üí  1226‚Üí        return format!("mul x{}, x{}, x{}", rd, rn, rm);
  1227‚Üí  1227‚Üí    }
  1228‚Üí  1228‚Üí
  1229‚Üí  1229‚Üí    // --- SDIV: 1_00_1101_1_0100_00000_00001_Rn_Rd  (actually 1_00_1101_0100_Rm_00001_Rn_Rd)
  1230‚Üí  1230‚Üí    if (word >> 21) & 0x7FF == 0b10011010100 && (word >> 10) & 0x1F == 0b00001 {
  1231‚Üí  1231‚Üí        return format!("sdiv x{}, x{}, x{}", rd, rn, rm);
  1232‚Üí  1232‚Üí    }
  1233‚Üí  1233‚Üí
  1234‚Üí  1234‚Üí    // --- UDIV: 1_00_1101_0000_Rm_00001_Rn_Rd
  1235‚Üí  1235‚Üí    if (word >> 21) & 0x7FF == 0b10011010000 && (word >> 10) & 0x1F == 0b00001 {
  1236‚Üí  1236‚Üí        return format!("udiv x{}, x{}, x{}", rd, rn, rm);
  1237‚Üí  1237‚Üí    }
  1238‚Üí  1238‚Üí
  1239‚Üí  1239‚Üí    // --- CMP (immediate): SUBS XZR, Xn, #imm12
  1240‚Üí  1240‚Üí    // 11100001_00_xxx_xxx_xxx_xxx_xxx_11111_xxx
  1241‚Üí  1241‚Üí    if (word >> 23) & 0x1FF == 0b1_1110_0010 && rd == 31 {
  1242‚Üí  1242‚Üí        return format!("cmp x{}, #{}", rn, imm12);
  1243‚Üí  1243‚Üí    }
  1244‚Üí  1244‚Üí
  1245‚Üí  1245‚Üí    // --- CMP (register): SUBS XZR, Xn, Xm
  1246‚Üí  1246‚Üí    if (word >> 21) & 0x7FF == 0b11101011000 && rd == 31 {
  1247‚Üí  1247‚Üí        return format!("cmp x{}, x{}", rn, rm);
  1248‚Üí  1248‚Üí    }
  1249‚Üí  1249‚Üí
  1250‚Üí  1250‚Üí    // --- B.cond: 0101010x xxxxxxxxxx xxxxxx cond
  1251‚Üí  1251‚Üí    if (word >> 24) & 0xFF == 0x54 {
  1252‚Üí  1252‚Üí        let cond_name = match cond {
  1253‚Üí  1253‚Üí            0 => "eq",
  1254‚Üí  1254‚Üí            1 => "ne",
  1255‚Üí  1255‚Üí            2 => "cs",
  1256‚Üí  1256‚Üí            3 => "cc",
  1257‚Üí  1257‚Üí            4 => "mi",
  1258‚Üí  1258‚Üí            5 => "pl",
  1259‚Üí  1259‚Üí            6 => "vs",
  1260‚Üí  1260‚Üí            7 => "vc",
  1261‚Üí  1261‚Üí            8 => "hi",
  1262‚Üí  1262‚Üí            9 => "ls",
  1263‚Üí  1263‚Üí            10 => "ge",
  1264‚Üí  1264‚Üí            11 => "lt",
  1265‚Üí  1265‚Üí            12 => "gt",
  1266‚Üí  1266‚Üí            13 => "le",
  1267‚Üí  1267‚Üí            14 => "al",
  1268‚Üí  1268‚Üí            _ => "??",
  1269‚Üí  1269‚Üí        };
  1270‚Üí  1270‚Üí        let imm19 = (word >> 5) & 0x7FFFF;
  1271‚Üí  1271‚Üí        let offset = ((imm19 as i32) << 13) >> 11; // sign-extend and *4
  1272‚Üí  1272‚Üí        return format!("b.{} {:+}", cond_name, offset);
  1273‚Üí  1273‚Üí    }
  1274‚Üí  1274‚Üí
  1275‚Üí  1275‚Üí    // --- B (unconditional): 000101xx xxxxxxxxxxxxxxxxxxxx
  1276‚Üí  1276‚Üí    if (word >> 26) & 0x3F == 0b000101 {
  1277‚Üí  1277‚Üí        let imm26 = word & 0x3FFFFFF;
  1278‚Üí  1278‚Üí        let offset = ((imm26 as i32) << 6) >> 4; // sign-extend and *4
  1279‚Üí  1279‚Üí        return format!("b {:+}", offset);
  1280‚Üí  1280‚Üí    }
  1281‚Üí  1281‚Üí
  1282‚Üí  1282‚Üí    // --- BL: 100101xx xxxxxxxxxxxxxxxxxxxx
  1283‚Üí  1283‚Üí    if (word >> 26) & 0x3F == 0b100101 {
  1284‚Üí  1284‚Üí        let imm26 = word & 0x3FFFFFF;
  1285‚Üí  1285‚Üí        let offset = ((imm26 as i32) << 6) >> 4;
  1286‚Üí  1286‚Üí        return format!("bl {:+}", offset);
  1287‚Üí  1287‚Üí    }
  1288‚Üí  1288‚Üí
  1289‚Üí  1289‚Üí    // --- LDR (unsigned offset): 11111001_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1290‚Üí  1290‚Üí    if (word >> 22) & 0x3FF == 0b1111100101 {
  1291‚Üí  1291‚Üí        let imm12_raw = (word >> 10) & 0xFFF;
  1292‚Üí  1292‚Üí        let offset = imm12_raw * 8; // scale by 8 for 64-bit
  1293‚Üí  1293‚Üí        return format!("ldr x{}, [x{}, #{}]", rt, rn, offset);
  1294‚Üí  1294‚Üí    }
  1295‚Üí  1295‚Üí
  1296‚Üí  1296‚Üí    // --- STR (unsigned offset): 11111000_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1297‚Üí  1297‚Üí    if (word >> 22) & 0x3FF == 0b1111100001 {
  1298‚Üí  1298‚Üí        let imm12_raw = (word >> 10) & 0xFFF;
  1299‚Üí  1299‚Üí        let offset = imm12_raw * 8;
  1300‚Üí  1300‚Üí        return format!("str x{}, [x{}, #{}]", rt, rn, offset);
  1301‚Üí  1301‚Üí    }
  1302‚Üí  1302‚Üí
  1303‚Üí  1303‚Üí    // --- LDP (signed offset, 64-bit): 101_0100_110_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1304‚Üí  1304‚Üí    if (word >> 22) & 0x3FF == 0b1010100110 {
  1305‚Üí  1305‚Üí        let rt2 = (word >> 10) & 0x1F;
  1306‚Üí  1306‚Üí        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1307‚Üí  1307‚Üí        let offset = imm7 * 8;
  1308‚Üí  1308‚Üí        return format!("ldp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1309‚Üí  1309‚Üí    }
  1310‚Üí  1310‚Üí
  1311‚Üí  1311‚Üí    // --- STP (signed offset, 64-bit): 101_0100_010_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1312‚Üí  1312‚Üí    if (word >> 22) & 0x3FF == 0b1010100010 {
  1313‚Üí  1313‚Üí        let rt2 = (word >> 10) & 0x1F;
  1314‚Üí  1314‚Üí        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1315‚Üí  1315‚Üí        let offset = imm7 * 8;
  1316‚Üí  1316‚Üí        return format!("stp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1317‚Üí  1317‚Üí    }
  1318‚Üí  1318‚Üí
  1319‚Üí  1319‚Üí    // --- MOVZ: 110100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1320‚Üí  1320‚Üí    if (word >> 23) & 0x1FF == 0b110100101 {
  1321‚Üí  1321‚Üí        let hw = (word >> 21) & 0x3;
  1322‚Üí  1322‚Üí        let imm16 = (word >> 5) & 0xFFFF;
  1323‚Üí  1323‚Üí        return format!("movz x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1324‚Üí  1324‚Üí    }
  1325‚Üí  1325‚Üí
  1326‚Üí  1326‚Üí    // --- MOVK: 111100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1327‚Üí  1327‚Üí    if (word >> 23) & 0x1FF == 0b111100101 {
  1328‚Üí  1328‚Üí        let hw = (word >> 21) & 0x3;
  1329‚Üí  1329‚Üí        let imm16 = (word >> 5) & 0xFFFF;
  1330‚Üí  1330‚Üí        return format!("movk x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1331‚Üí  1331‚Üí    }
  1332‚Üí  1332‚Üí
  1333‚Üí  1333‚Üí    format!(".word {:08x}", word)
  1334‚Üí  1334‚Üí}

  1363‚Üí  1363‚Üí/// Compute the stack frame size for an IR function.
  1364‚Üí  1364‚Üí///
  1365‚Üí  1365‚Üí/// Replicates the private `compute_frame_size` function in `emit.rs`:
  1366‚Üí  1366‚Üí/// sums `Alloc` instruction sizes and rounds up to 16-byte alignment.
  1367‚Üí  1367‚Üí/// NOTE: Does NOT include 16 bytes for FP/LR pair (handled by prologue separately).
  1368‚Üí  1368‚Üífn aarch64_compute_frame_size(func: &IRFunction) -> usize {
  1369‚Üí  1369‚Üí    let mut total: u32 = 0; // Alloc sizes only; FP/LR handled separately
  1370‚Üí  1370‚Üí    for block in &func.blocks {
  1371‚Üí  1371‚Üí        for instr in &block.instructions {
  1372‚Üí  1372‚Üí            if let IRInstr::Alloc { size, .. } = instr {
  1373‚Üí  1373‚Üí                let aligned = (*size).div_ceil(16) * 16;
  1374‚Üí  1374‚Üí                total += aligned;
  1375‚Üí  1375‚Üí            }
  1376‚Üí  1376‚Üí        }
  1377‚Üí  1377‚Üí    }
  1378‚Üí  1378‚Üí    // Round up to 16-byte alignment
  1379‚Üí  1379‚Üí    total = (total + 15) & !15;
  1380‚Üí  1380‚Üí    total as usize
  1381‚Üí  1381‚Üí}

  1383‚Üí  1383‚Üí/// Build an ELF64 binary for AArch64 Linux with 2 LOAD segments.
  1384‚Üí  1384‚Üí///
  1385‚Üí  1385‚Üí/// Segment 1: PF_R | PF_X ‚Äî contains .text (code)
  1386‚Üí  1386‚Üí/// Segment 2: PF_R | PF_W ‚Äî contains .data + stack space (writable)
  1387‚Üí  1387‚Üí///
  1388‚Üí  1388‚Üí/// The two segments are separated by page alignment (4KB) to ensure
  1389‚Üí  1389‚Üí/// the kernel maps them with different permissions. Without this,
  1390‚Üí  1390‚Üí/// a single PF_R|PF_W|PF_X segment is insecure and may cause
  1391‚Üí  1391‚Üí/// QEMU/Linux to reject the executable.
  1392‚Üí  1392‚Üífn build_aarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1393‚Üí  1393‚Üí    const PAGE_SIZE: u64 = 0x1000; // 4 KB
  1394‚Üí  1394‚Üí
  1395‚Üí  1395‚Üí    let elf_header_size: u64 = 64;
  1396‚Üí  1396‚Üí    let phdr_size: u64 = 56;
  1397‚Üí  1397‚Üí    let num_phdrs: u64 = 2;
  1398‚Üí  1398‚Üí    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1399‚Üí  1399‚Üí    // Page-align the text segment start in the file.  The kernel's ELF
  1400‚Üí  1400‚Üí    // loader mmap()s each LOAD segment, and the file offset must be
  1401‚Üí  1401‚Üí    // congruent with the virtual address modulo the page size.  The
  1402‚Üí  1402‚Üí    // simplest way to guarantee this is to place the text at a
  1403‚Üí  1403‚Üí    // page-aligned file offset, with vaddr = base_addr.
  1404‚Üí  1404‚Üí    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1405‚Üí  1405‚Üí    let text_size = code.len() as u64;
  1406‚Üí  1406‚Üí
  1407‚Üí  1407‚Üí    // The data segment starts on the next page after the text.
  1408‚Üí  1408‚Üí    let text_file_end = text_offset + text_size;
  1409‚Üí  1409‚Üí    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1410‚Üí  1410‚Üí    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1411‚Üí  1411‚Üí    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1412‚Üí  1412‚Üí    let entry_point = base_addr + text_offset;
  1413‚Üí  1413‚Üí
  1414‚Üí  1414‚Üí    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1415‚Üí  1415‚Üí
  1416‚Üí  1416‚Üí    // --- e_ident ---
  1417‚Üí  1417‚Üí    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1418‚Üí  1418‚Üí    elf.push(2); // ELFCLASS64
  1419‚Üí  1419‚Üí    elf.push(1); // ELFDATA2LSB
  1420‚Üí  1420‚Üí    elf.push(1); // EV_CURRENT
  1421‚Üí  1421‚Üí    elf.push(3); // ELFOSABI_LINUX
  1422‚Üí  1422‚Üí    elf.push(0); // padding
  1423‚Üí  1423‚Üí    elf.extend_from_slice(&[0u8; 7]); // padding
  1424‚Üí  1424‚Üí
  1425‚Üí  1425‚Üí    // --- ELF header fields ---
  1426‚Üí  1426‚Üí    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1427‚Üí  1427‚Üí    elf.extend_from_slice(&183u16.to_le_bytes()); // e_machine = EM_AARCH64
  1428‚Üí  1428‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1429‚Üí  1429‚Üí    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1430‚Üí  1430‚Üí    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1431‚Üí  1431‚Üí    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1432‚Üí  1432‚Üí    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1433‚Üí  1433‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1434‚Üí  1434‚Üí    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1435‚Üí  1435‚Üí    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1436‚Üí  1436‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1437‚Üí  1437‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1438‚Üí  1438‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1439‚Üí  1439‚Üí
  1440‚Üí  1440‚Üí    // --- Program Header 1: LOAD (PF_R | PF_X) ‚Äî .text ---
  1441‚Üí  1441‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1442‚Üí  1442‚Üí    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1443‚Üí  1443‚Üí    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1444‚Üí  1444‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1445‚Üí  1445‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1446‚Üí  1446‚Üí    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1447‚Üí  1447‚Üí    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1448‚Üí  1448‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1449‚Üí  1449‚Üí
  1450‚Üí  1450‚Üí    // --- Program Header 2: LOAD (PF_R | PF_W) ‚Äî .data / stack ---
  1451‚Üí  1451‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1452‚Üí  1452‚Üí    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1453‚Üí  1453‚Üí    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1454‚Üí  1454‚Üí    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1455‚Üí  1455‚Üí    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1456‚Üí  1456‚Üí    elf.extend_from_slice(&0u64.to_le_bytes()); // p_filesz (no initialized data)
  1457‚Üí  1457‚Üí    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1458‚Üí  1458‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1459‚Üí  1459‚Üí
  1460‚Üí  1460‚Üí    // --- .text section ---
  1461‚Üí  1461‚Üí    // Pad to page-aligned text_offset
  1462‚Üí  1462‚Üí    while (elf.len() as u64) < text_offset {
  1463‚Üí  1463‚Üí        elf.push(0);
  1464‚Üí  1464‚Üí    }
  1465‚Üí  1465‚Üí    elf.extend_from_slice(code);
  1466‚Üí  1466‚Üí
  1467‚Üí  1467‚Üí    // --- Pad to data segment offset (if needed) ---
  1468‚Üí  1468‚Üí    while (elf.len() as u64) < data_offset {
  1469‚Üí  1469‚Üí        elf.push(0);
  1470‚Üí  1470‚Üí    }
  1471‚Üí  1471‚Üí
  1472‚Üí  1472‚Üí    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1473‚Üí  1473‚Üí
  1474‚Üí  1474‚Üí    elf
  1475‚Üí  1475‚Üí}

  1477‚Üí  1477‚Üí/// Build ARM64 runtime I/O functions using Linux SVC syscalls.
  1478‚Üí  1478‚Üí///
  1479‚Üí  1479‚Üí/// Provides:
  1480‚Üí  1480‚Üí/// - `__vuma_print_hex`: Print X0 as 8 hex digits to stdout (FD=1)
  1481‚Üí  1481‚Üí///   Uses sys_write (X8=64) via SVC #0.
  1482‚Üí  1482‚Üí///   Saves/restores X1,X2,X3,X8,X9,X10 on stack.
  1483‚Üí  1483‚Üí///
  1484‚Üí  1484‚Üí/// - `__vuma_print_int`: Print X0 as a decimal integer to stdout (FD=1)
  1485‚Üí  1485‚Üí///   Converts digit-by-digit into a stack buffer, then sys_write.
  1486‚Üí  1486‚Üí///
  1487‚Üí  1487‚Üí/// - `__vuma_print_newline`: Print a newline character to stdout.
  1488‚Üí  1488‚Üí///
  1489‚Üí  1489‚Üí/// All functions follow AAPCS64: X0 is the argument, X1-X7 are caller-saved,
  1490‚Üí  1490‚Üí/// X8 is the indirect result register / syscall number, X19-X28 are callee-saved.
  1491‚Üí  1491‚Üífn build_aarch64_runtime() -> Vec<u8> {
  1492‚Üí  1492‚Üí    let mut code = Vec::new();
  1493‚Üí  1493‚Üí
  1494‚Üí  1494‚Üí    // ‚îÄ‚îÄ __vuma_print_hex ‚îÄ‚îÄ
  1495‚Üí  1495‚Üí    // Input: X0 = 64-bit value to print as 8 hex digits
  1496‚Üí  1496‚Üí    // Clobbers: X1, X2, X3, X8, X9, X10
  1497‚Üí  1497‚Üí    // Stack frame: 48 bytes (8 for pair save + 8 for hex chars + 32 padding)
  1498‚Üí  1498‚Üí
  1499‚Üí  1499‚Üí    // Prologue: save callee-saved and link register
  1500‚Üí  1500‚Üí    // STP X29, X30, [SP, #-48]!  (pre-indexed, 48-byte frame)
  1501‚Üí  1501‚Üí    code.extend_from_slice(&0xA9BF7BFDu32.to_le_bytes());
  1502‚Üí  1502‚Üí    // ADD X29, SP, #0
  1503‚Üí  1503‚Üí    code.extend_from_slice(&0x910003FDu32.to_le_bytes());
  1504‚Üí  1504‚Üí
  1505‚Üí  1505‚Üí    // Save X9, X10 on stack at [SP, #16]
  1506‚Üí  1506‚Üí    // STP X9, X10, [SP, #16]
  1507‚Üí  1507‚Üí    code.extend_from_slice(&0xA90127E9u32.to_le_bytes());
  1508‚Üí  1508‚Üí
  1509‚Üí  1509‚Üí    // Save X1, X2 on stack at [SP, #32]
  1510‚Üí  1510‚Üí    // STP X1, X2, [SP, #32]
  1511‚Üí  1511‚Üí    code.extend_from_slice(&0xA9020441u32.to_le_bytes());
  1512‚Üí  1512‚Üí
  1513‚Üí  1513‚Üí    // Save X3, X8 on stack at [SP, #40]  (offset 40 = 5*8)
  1514‚Üí  1514‚Üí    // Wait, 40 = 5*8 but STP needs offset/8 for signed-offset. 40/8 = 5.
  1515‚Üí  1515‚Üí    // STP X3, X8, [SP, #40]
  1516‚Üí  1516‚Üí    code.extend_from_slice(&0xA9052C63u32.to_le_bytes());
  1517‚Üí  1517‚Üí
  1518‚Üí  1518‚Üí    // Hex lookup table: we'll use ADD + LDRB pattern.
  1519‚Üí  1519‚Üí    // Instead, compute hex digit directly:
  1520‚Üí  1520‚Üí    // For each nibble: digit = nibble; if digit < 10 then char = digit + '0' else char = digit - 10 + 'a'
  1521‚Üí  1521‚Üí
  1522‚Üí  1522‚Üí    // We'll write 8 hex chars starting at SP+16 (reuse save area after we load saves)
  1523‚Üí  1523‚Üí    // Actually, let's use SP+16 as the char buffer (8 bytes).
  1524‚Üí  1524‚Üí    // First, save the important regs, then overwrite the buffer area.
  1525‚Üí  1525‚Üí
  1526‚Üí  1526‚Üí    // Process 8 nibbles from most significant to least significant
  1527‚Üí  1527‚Üí    // X0 = value, iterate 8 times shifting right by 28, 24, 20, ..., 4, 0
  1528‚Üí  1528‚Üí
  1529‚Üí  1529‚Üí    // X9 = char buffer pointer = SP
  1530‚Üí  1530‚Üí    // X10 = loop counter (0..8)
  1531‚Üí  1531‚Üí
  1532‚Üí  1532‚Üí    // ADD X9, SP, #16  (buffer starts at SP+16)
  1533‚Üí  1533‚Üí    code.extend_from_slice(&0x910043E9u32.to_le_bytes());
  1534‚Üí  1534‚Üí    // MOV X10, #0
  1535‚Üí  1535‚Üí    code.extend_from_slice(&0xD280001Au32.to_le_bytes());
  1536‚Üí  1536‚Üí    // MOV X3, #0  (bit shift amount, starts at 28)
  1537‚Üí  1537‚Üí    code.extend_from_slice(&0xD280003Du32.to_le_bytes()); // MOVZ X3, #28>>4... wait
  1538‚Üí  1538‚Üí    // MOV X3, #28
  1539‚Üí  1539‚Üí    code.extend_from_slice(&0xD2800383u32.to_le_bytes()); // MOVZ X3, #28
  1540‚Üí  1540‚Üí
  1541‚Üí  1541‚Üí    // Loop label:
  1542‚Üí  1542‚Üí    let loop_offset = code.len();
  1543‚Üí  1543‚Üí
  1544‚Üí  1544‚Üí    // Extract nibble: X2 = (X0 >> X3) & 0xF
  1545‚Üí  1545‚Üí    // LSR X2, X0, X3
  1546‚Üí  1546‚Üí    code.extend_from_slice(&0x9AC02482u32.to_le_bytes()); // LSR X2, X0, X3
  1547‚Üí  1547‚Üí    // AND X2, X2, #0xF
