     1→//! # VUMA REPL — Interactive Read-Eval-Print Loop
     2→//!
     3→//! The [`VumaRepl`] struct provides an interactive shell for the VUMA language.
     4→//! It parses VUMA expressions, builds the Semantic Computation Graph (SCG),
     5→//! converts it to a Memory State Graph (MSG), and runs IVE verification.
     6→//!
     7→//! ## Interactive Commands
     8→//!
     9→//! | Command          | Description                                        |
    10→//! |------------------|----------------------------------------------------|
    11→//! | `:help`          | Show available commands                             |
    12→//! | `:load <file>`   | Load and evaluate a VUMA source file               |
    13→//! | `:verify`        | Run IVE verification on the current SCG            |
    14→//! | `:show scg`      | Display the current SCG summary                    |
    15→//! | `:show msg`      | Display the current MSG summary                    |
    16→//! | `:show bd`       | Display behavioural descriptors for all nodes      |
    17→//! | `:compile`       | Run the full pipeline: parse → SCG → MSG → verify  |
    18→//! | `:profile`       | Show profiling data from the last verification     |
    19→//! | `:quit`          | Exit the REPL                                      |
    20→//!
    21→//! ## Expression Evaluation
    22→//!
    23→//! Simple arithmetic and literal expressions are evaluated immediately.
    24→//! Definitions (`let x = …`, `fn f() { … }`, etc.) persist across inputs
    25→//! so that subsequent expressions can reference previously-defined names.
    26→//!
    27→//! ## Error Display
    28→//!
    29→//! Parse errors and runtime errors are displayed with source context,
    30→//! showing the offending line and a caret pointing to the error location.
    31→//!
    32→//! ## History
    33→//!
    34→//! The REPL maintains an in-memory history buffer. Use `:history` to list
    35→//! previous inputs. Arrow-key navigation (up/down) is supported in the
    36→//! interactive loop.
    37→

    58→/// Errors that can occur during REPL operation.
    59→#[derive(Debug)]
    60→pub enum ReplError {
    61→    /// A parse error from the VUMA frontend.
    62→    Parse(ParseError),
    63→    /// An SCG construction error.
    64→    ScgConstruction(String),
    65→    /// An MSG conversion error.
    66→    MsgConversion(scg_to_msg::ConversionError),
    67→    /// An I/O error.
    68→    Io(io::Error),
    69→    /// A general REPL error with a message.
    70→    General(String),
    71→    /// Multiple parse errors.
    72→    ParseErrors(Vec<ParseError>),
    73→}
    74→
    75→impl fmt::Display for ReplError {
    76→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
    77→        match self {
    78→            ReplError::Parse(e) => write!(f, "parse error: {e}"),
    79→            ReplError::ScgConstruction(msg) => write!(f, "SCG construction error: {msg}"),
    80→            ReplError::MsgConversion(e) => write!(f, "MSG conversion error: {e}"),
    81→            ReplError::Io(e) => write!(f, "I/O error: {e}"),
    82→            ReplError::General(msg) => write!(f, "{msg}"),
    83→            ReplError::ParseErrors(errors) => {
    84→                write!(f, "{} parse error(s):", errors.len())?;
    85→                for e in errors {
    86→                    write!(f, "\n  {e}")?;
    87→                }
    88→                Ok(())
    89→            }
    90→        }
    91→    }
    92→}
    93→
    94→impl std::error::Error for ReplError {
    95→    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
    96→        match self {
    97→            ReplError::Parse(e) => Some(e),
    98→            ReplError::MsgConversion(e) => Some(e),
    99→            ReplError::Io(e) => Some(e),
   100→            _ => None,
   101→        }
   102→    }
   103→}
   104→
   105→impl From<ParseError> for ReplError {
   106→    fn from(e: ParseError) -> Self {
   107→        ReplError::Parse(e)
   108→    }
   109→}
   110→
   111→impl From<scg_to_msg::ConversionError> for ReplError {
   112→    fn from(e: scg_to_msg::ConversionError) -> Self {
   113→        ReplError::MsgConversion(e)
   114→    }
   115→}
   116→
   117→impl From<io::Error> for ReplError {
   118→    fn from(e: io::Error) -> Self {
   119→        ReplError::Io(e)
   120→    }
   121→}

   127→/// The result of processing a single REPL input.
   128→#[derive(Debug)]
   129→pub enum ReplResult {
   130→    /// The input was processed successfully; optional output text.
   131→    Ok(Option<String>),
   132→    /// The input resulted in a value that can be displayed.
   133→    Value(String),
   134→    /// A verification result.
   135→    Verification(AggregatedResult),
   136→    /// The user requested to quit.
   137→    Quit,
   138→}
   139→
   140→impl fmt::Display for ReplResult {
   141→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   142→        match self {
   143→            ReplResult::Ok(Some(msg)) => write!(f, "{msg}"),
   144→            ReplResult::Ok(None) => Ok(()),
   145→            ReplResult::Value(v) => write!(f, "{v}"),
   146→            ReplResult::Verification(r) => {
   147→                let report = DiagnosticsReport::from_aggregated(r);
   148→                write!(f, "{report}")
   149→            }
   150→            ReplResult::Quit => write!(f, "Goodbye."),
   151→        }
   152→    }
   153→}

   196→// ---------------------------------------------------------------------------
   197→// Expression Evaluator (simple immediate evaluation)
   198→// ---------------------------------------------------------------------------
   199→
   200→/// A simple immediate evaluator for arithmetic expressions.
   201→///
   202→/// This evaluator can compute integer arithmetic expressions without
   203→/// needing to go through the full SCG pipeline. It supports:
   204→/// - Integer literals
   205→/// - Variable references (from the REPL's definition map)
   206→/// - Basic arithmetic: +, -, *, /
   207→/// - Parenthesised expressions
   208→struct SimpleEvaluator {
   209→    /// Variable bindings from previous definitions.
   210→    vars: HashMap<String, i64>,
   211→}
   212→
   213→impl SimpleEvaluator {
   214→    fn new(vars: HashMap<String, i64>) -> Self {
   215→        Self { vars }
   216→    }
   217→
   218→    /// Attempt to evaluate a simple expression string.
   219→    /// Returns `None` if the expression is too complex for this evaluator.
   220→    fn eval(&self, input: &str) -> Option<i64> {
   221→        let trimmed = input.trim();
   222→        // Remove trailing semicolons.
   223→        let trimmed = trimmed.trim_end_matches(';').trim();
   224→        let tokens = self.tokenize(trimmed)?;
   225→        self.eval_tokens(&tokens)
   226→    }
   227→
   228→    /// Tokenize the input into a simple token stream.
   229→    fn tokenize(&self, input: &str) -> Option<Vec<Token>> {
   230→        let mut tokens = Vec::new();
   231→        let mut chars = input.chars().peekable();
   232→        while let Some(&ch) = chars.peek() {
   233→            match ch {
   234→                ' ' | '\t' => {
   235→                    chars.next();
   236→                }
   237→                '(' => {
   238→                    tokens.push(Token::LParen);
   239→                    chars.next();
   240→                }
   241→                ')' => {
   242→                    tokens.push(Token::RParen);
   243→                    chars.next();
   244→                }
   245→                '+' => {
   246→                    tokens.push(Token::Plus);
   247→                    chars.next();
   248→                }
   249→                '-' => {
   250→                    tokens.push(Token::Minus);
   251→                    chars.next();
   252→                }
   253→                '*' => {
   254→                    tokens.push(Token::Star);
   255→                    chars.next();
   256→                }
   257→                '/' => {
   258→                    tokens.push(Token::Slash);
   259→                    chars.next();
   260→                }
   261→                '0'..='9' => {
   262→                    let mut num = String::new();
   263→                    while let Some(&d) = chars.peek() {
   264→                        if d.is_ascii_digit() {
   265→                            num.push(d);
   266→                            chars.next();
   267→                        } else {
   268→                            break;
   269→                        }
   270→                    }
   271→                    tokens.push(Token::Num(num.parse::<i64>().ok()?));
   272→                }
   273→                _ if ch.is_alphanumeric() || ch == '_' => {
   274→                    let mut name = String::new();
   275→                    while let Some(&c) = chars.peek() {
   276→                        if c.is_alphanumeric() || c == '_' {
   277→                            name.push(c);
   278→                            chars.next();
   279→                        } else {
   280→                            break;
   281→                        }
   282→                    }
   283→                    if let Some(&val) = self.vars.get(&name) {
   284→                        tokens.push(Token::Num(val));
   285→                    } else {
   286→                        return None; // unknown variable
   287→                    }
   288→                }
   289→                _ => return None, // unsupported character
   290→            }
   291→        }
   292→        Some(tokens)
   293→    }
   294→
   295→    /// Evaluate a token stream using recursive descent.
   296→    fn eval_tokens(&self, tokens: &[Token]) -> Option<i64> {
   297→        let mut pos = 0;
   298→        let result = self.parse_additive(tokens, &mut pos)?;
   299→        if pos == tokens.len() {
   300→            Some(result)
   301→        } else {
   302→            None
   303→        }
   304→    }
   305→
   306→    fn parse_additive(&self, tokens: &[Token], pos: &mut usize) -> Option<i64> {
   307→        let mut left = self.parse_multiplicative(tokens, pos)?;
   308→        while *pos < tokens.len() {
   309→            match &tokens[*pos] {
   310→                Token::Plus => {
   311→                    *pos += 1;
   312→                    let right = self.parse_multiplicative(tokens, pos)?;
   313→                    left += right;
   314→                }
   315→                Token::Minus => {
   316→                    *pos += 1;
   317→                    let right = self.parse_multiplicative(tokens, pos)?;
   318→                    left -= right;
   319→                }
   320→                _ => break,
   321→            }
   322→        }
   323→        Some(left)
   324→    }
   325→
   326→    fn parse_multiplicative(&self, tokens: &[Token], pos: &mut usize) -> Option<i64> {
   327→        let mut left = self.parse_unary(tokens, pos)?;
   328→        while *pos < tokens.len() {
   329→            match &tokens[*pos] {
   330→                Token::Star => {
   331→                    *pos += 1;
   332→                    let right = self.parse_unary(tokens, pos)?;
   333→                    left *= right;
   334→                }
   335→                Token::Slash => {
   336→                    *pos += 1;
   337→                    let right = self.parse_unary(tokens, pos)?;
   338→                    if right == 0 {
   339→                        return None;
   340→                    }
   341→                    left /= right;
   342→                }
   343→                _ => break,
   344→            }
   345→        }
   346→        Some(left)
   347→    }
   348→
   349→    fn parse_unary(&self, tokens: &[Token], pos: &mut usize) -> Option<i64> {
   350→        if *pos < tokens.len() {
   351→            match &tokens[*pos] {
   352→                Token::Minus => {
   353→                    *pos += 1;
   354→                    let val = self.parse_primary(tokens, pos)?;
   355→                    Some(-val)
   356→                }
   357→                Token::Plus => {
   358→                    *pos += 1;
   359→                    self.parse_primary(tokens, pos)
   360→                }
   361→                _ => self.parse_primary(tokens, pos),
   362→            }
   363→        } else {
   364→            None
   365→        }
   366→    }
   367→
   368→    fn parse_primary(&self, tokens: &[Token], pos: &mut usize) -> Option<i64> {
   369→        if *pos >= tokens.len() {
   370→            return None;
   371→        }
   372→        match &tokens[*pos] {
   373→            Token::Num(n) => {
   374→                *pos += 1;
   375→                Some(*n)
   376→            }
   377→            Token::LParen => {
   378→                *pos += 1;
   379→                let result = self.parse_additive(tokens, pos)?;
   380→                if *pos < tokens.len() && matches!(&tokens[*pos], Token::RParen) {
   381→                    *pos += 1;
   382→                    Some(result)
   383→                } else {
   384→                    None
   385→                }
   386→            }
   387→            _ => None,
   388→        }
   389→    }
   390→}
   391→
   392→/// Simple token type for the arithmetic evaluator.
   393→#[derive(Debug)]
   394→enum Token {
   395→    Num(i64),
   396→    Plus,
   397→    Minus,
   398→    Star,
   399→    Slash,
   400→    LParen,
   401→    RParen,
   402→}

   408→/// Format a parse error with source context.
   409→fn format_error_with_context(source: &str, span: &Span, message: &str) -> String {
   410→    let loc = offset_to_location(source, span.start, None);
   411→    let line_start = source[..span.start].rfind('\n').map(|i| i + 1).unwrap_or(0);
   412→    let line_end = source[span.start..]
   413→        .find('\n')
   414→        .map(|i| span.start + i)
   415→        .unwrap_or(source.len());
   416→    let line_text = &source[line_start..line_end];
   417→
   418→    let column = span.start - line_start;
   419→    let caret_width = if span.end > span.start {
   420→        (span.end - span.start).min(line_end - span.start)
   421→    } else {
   422→        1
   423→    };
   424→
   425→    let mut result = String::new();
   426→    result.push_str(&format!("error: {message}\n"));
   427→    result.push_str(&format!(
   428→        "  --> {}:{}:{}\n",
   429→        loc.line, loc.column, span.start
   430→    ));
   431→    result.push_str("   |\n");
   432→    result.push_str(&format!("{:3}| {}\n", loc.line, line_text));
   433→    result.push_str("   | ");
   434→    result.push_str(&" ".repeat(column));
   435→    result.push_str(&"^".repeat(caret_width));
   436→    result.push('\n');
   437→    result
   438→}

   444→/// Generate a human-readable summary of the SCG.
   445→fn format_scg_summary(scg: &SCG) -> String {
   446→    let mut type_counts: HashMap<String, usize> = HashMap::new();
   447→    let mut edge_counts: HashMap<String, usize> = HashMap::new();
   448→
   449→    for node in scg.nodes() {
   450→        let label = format!("{:?}", node.node_type);
   451→        *type_counts.entry(label).or_insert(0) += 1;
   452→    }
   453→
   454→    for edge in scg.edges() {
   455→        let label = format!("{:?}", edge.kind);
   456→        *edge_counts.entry(label).or_insert(0) += 1;
   457→    }
   458→
   459→    let mut result = String::new();
   460→    result.push_str(&format!(
   461→        "SCG Summary ({} nodes, {} edges, {} regions)\n",
   462→        scg.node_count(),
   463→        scg.edge_count(),
   464→        scg.region_count()
   465→    ));
   466→
   467→    if !type_counts.is_empty() {
   468→        result.push_str("  Node types:\n");
   469→        let mut entries: Vec<_> = type_counts.into_iter().collect();
   470→        entries.sort_by_key(|b| std::cmp::Reverse(b.1));
   471→        for (kind, count) in entries {
   472→            result.push_str(&format!("    {kind}: {count}\n"));
   473→        }
   474→    }
   475→
   476→    if !edge_counts.is_empty() {
   477→        result.push_str("  Edge types:\n");
   478→        let mut entries: Vec<_> = edge_counts.into_iter().collect();
   479→        entries.sort_by_key(|b| std::cmp::Reverse(b.1));
   480→        for (kind, count) in entries {
   481→            result.push_str(&format!("    {kind}: {count}\n"));
   482→        }
   483→    }
   484→
   485→    // List regions.
   486→    for region in scg.regions() {
   487→        result.push_str(&format!(
   488→            "  Region {} (deployment: {}, nodes: {}, scope: {})\n",
   489→            region.id,
   490→            region.deployment_target,
   491→            region.node_count(),
   492→            region.scope_level,
   493→        ));
   494→    }
   495→
   496→    result
   497→}

   499→/// Generate a BD (Behavioural Descriptor) display for all nodes in the SCG.
   500→fn format_bd_display(scg: &SCG, inference_engine: &InferenceEngine) -> String {
   501→    let mut result = String::new();
   502→    result.push_str("Behavioural Descriptors:\n");
   503→
   504→    if scg.node_count() == 0 {
   505→        result.push_str("  (no nodes)\n");
   506→        return result;
   507→    }
   508→
   509→    for node in scg.nodes() {
   510→        let bd = inference_engine.infer_bd(scg, node.id).unwrap_or_else(|_| {
   511→            vuma_bd::descriptor::BD::new(
   512→                vuma_bd::repd::RepD::Byte(vuma_bd::repd::ByteRep { size: 0, align: 0 }),
   513→                vuma_bd::capd::CapD::empty(),
   514→                vuma_bd::reld::RelD::empty(),
   515→            )
   516→        });
   517→        result.push_str(&format!(
   518→            "  Node {} ({:?}): {}\n",
   519→            node.id, node.node_type, bd
   520→        ));
   521→    }
   522→
   523→    result
   524→}

   526→// ---------------------------------------------------------------------------
   527→// VumaRepl
   528→// ---------------------------------------------------------------------------
   529→
   530→/// The VUMA Read-Eval-Print Loop.
   531→///
   532→/// `VumaRepl` maintains incremental state across expressions: previous
   533→/// definitions are accumulated in a source buffer and re-parsed as a whole
   534→/// so that later expressions can reference earlier bindings.
   535→///
   536→/// # Usage
   537→///
   538→/// ```rust
   539→/// use vuma_core::repl::VumaRepl;
   540→///
   541→/// let mut repl = VumaRepl::new();
   542→///
   543→/// // Process a single line programmatically.
   544→/// let result = repl.process_line("let x = 42;");
   545→/// match result {
   546→///     Ok(output) => println!("{output}"),
   547→///     Err(e) => eprintln!("Error: {e}"),
   548→/// }
   549→///
   550→/// // Run the interactive loop (reads from stdin).
   551→/// // repl.run();
   552→/// ```
   553→pub struct VumaRepl {
   554→    /// Accumulated source text (all previous definitions + current expression).
   555→    source_buffer: String,
   556→    /// The current SCG built from the accumulated source.
   557→    scg: SCG,
   558→    /// The current MSG converted from the SCG.
   559→    msg: Option<MSG>,
   560→    /// The AST-to-SCG converter (maintains variable scopes).
   561→    converter: AstToScg,
   562→    /// IVE inference engine.
   563→    inference_engine: InferenceEngine,
   564→    /// IVE verification engine (reserved for individual invariant checks).
   565→    _verification_engine: VerificationEngine,
   566→    /// IVE invariant aggregator.
   567→    aggregator: InvariantAggregator,
   568→    /// Command input history.
   569→    history: Vec<String>,
   570→    /// Current position in history for up/down navigation.
   571→    history_cursor: usize,
   572→    /// Profiling data.
   573→    profile: ReplProfile,
   574→    /// Whether the REPL should keep running.
   575→    running: bool,
   576→    /// Last verification result.
   577→    last_verification: Option<AggregatedResult>,
   578→    /// Variable values for simple expression evaluation.
   579→    simple_vars: HashMap<String, i64>,
   580→    /// Currently loaded file path (if any).
   581→    loaded_file: Option<String>,
   582→}
   583→
   584→impl VumaRepl {
   585→    /// Create a new REPL instance.
   586→    pub fn new() -> Self {
   587→        Self {
   588→            source_buffer: String::new(),
   589→            scg: SCG::new(),
   590→            msg: None,
   591→            converter: AstToScg::new(),
   592→            inference_engine: InferenceEngine::new(),
   593→            _verification_engine: VerificationEngine::new(),
   594→            aggregator: InvariantAggregator::new(),
   595→            history: Vec::new(),
   596→            history_cursor: 0,
   597→            profile: ReplProfile::default(),
   598→            running: true,
   599→            last_verification: None,
   600→            simple_vars: HashMap::new(),
   601→            loaded_file: None,
   602→        }
   603→    }
   604→
   605→    /// Create a REPL with verbose IVE output.
   606→    pub fn with_verbose() -> Self {
   607→        Self {
   608→            inference_engine: InferenceEngine::new().with_verbose(true),
   609→            _verification_engine: VerificationEngine::new().with_verbose(true),
   610→            aggregator: InvariantAggregator::new().with_verbose(true),
   611→            ..Self::new()
   612→        }
   613→    }
   614→
   615→    // -----------------------------------------------------------------------
   616→    // Core: process a line of input
   617→    // -----------------------------------------------------------------------
   618→
   619→    /// Process a single line of input and return the result.
   620→    ///
   621→    /// This is the primary API for programmatic use. The line may be a
   622→    /// VUMA expression/statement or a REPL command prefixed with `:`.
   623→    pub fn process_line(&mut self, line: &str) -> Result<ReplResult, ReplError> {
   624→        let trimmed = line.trim();
   625→
   626→        // Record in history.
   627→        if !trimmed.is_empty() {
   628→            self.history.push(trimmed.to_string());
   629→            self.history_cursor = self.history.len();
   630→        }
   631→
   632→        // Handle empty input.
   633→        if trimmed.is_empty() {
   634→            return Ok(ReplResult::Ok(None));
   635→        }
   636→
   637→        // Handle REPL commands.
   638→        if trimmed.starts_with(':') {
   639→            return self.handle_command(trimmed);
   640→        }
   641→
   642→        // Try simple expression evaluation first.
   643→        let evaluator = SimpleEvaluator::new(self.simple_vars.clone());
   644→        if let Some(value) = evaluator.eval(trimmed) {
   645→            self.profile.expressions_processed += 1;
   646→            return Ok(ReplResult::Value(value.to_string()));
   647→        }
   648→
   649→        // Fall through to full VUMA parsing.
   650→        self.evaluate_vuma(trimmed)
   651→    }

   654→    // VUMA expression evaluation

   657→    /// Evaluate a VUMA expression/statement through the full pipeline.
   658→    fn evaluate_vuma(&mut self, input: &str) -> Result<ReplResult, ReplError> {
   659→        // Append the input to the source buffer.
   660→        let prev_len = self.source_buffer.len();
   661→        if !self.source_buffer.is_empty() && !self.source_buffer.ends_with('\n') {
   662→            self.source_buffer.push('\n');
   663→        }
   664→        self.source_buffer.push_str(input);
   665→        if !input.ends_with(';') && !input.ends_with('}') {
   666→            self.source_buffer.push(';');
   667→        }
   668→
   669→        // Parse.
   670→        let parse_start = Instant::now();
   671→        let mut parser = Parser::new(&self.source_buffer);
   672→        let result = parser.parse_program();
   673→        if result.has_errors() {
   674→            // Roll back the source buffer on parse error.
   675→            let errors = result.errors.clone();
   676→            self.source_buffer.truncate(prev_len);
   677→            self.profile.parse_errors += errors.len();
   678→            self.profile.parse_time_ms += parse_start.elapsed().as_millis() as u64;
   679→            return Err(ReplError::ParseErrors(errors));
   680→        }
   681→        let program = result.unwrap();
   682→        self.profile.parse_time_ms += parse_start.elapsed().as_millis() as u64;
   683→
   684→        // Build SCG.
   685→        let scg_start = Instant::now();
   686→        // We rebuild the SCG from scratch each time to keep it consistent
   687→        // with the full source buffer. In a production implementation we
   688→        // would use incremental SCG updates.
   689→        let mut converter = AstToScg::new();
   690→        let scg = match converter.convert(&program) {
   691→            Ok(s) => s,
   692→            Err(e) => {
   693→                self.source_buffer.truncate(prev_len);
   694→                self.profile.scg_time_ms += scg_start.elapsed().as_millis() as u64;
   695→                return Err(ReplError::Parse(e));
   696→            }
   697→        };
   698→        self.scg = scg;
   699→        self.converter = converter;
   700→        self.profile.scg_time_ms += scg_start.elapsed().as_millis() as u64;
   701→
   702→        // Convert to MSG.
   703→        let msg_start = Instant::now();
   704→        match scg_to_msg::scg_to_msg(&self.scg) {
   705→            Ok(msg) => self.msg = Some(msg),
   706→            Err(e) => {
   707→                // MSG conversion failure is non-fatal for the REPL; we just
   708→                // don't have an MSG available.
   709→                log::warn!("MSG conversion failed: {e}");
   710→                self.msg = None;
   711→            }
   712→        }
   713→        self.profile.msg_time_ms += msg_start.elapsed().as_millis() as u64;
   714→
   715→        // Extract simple variable bindings for immediate evaluation.
   716→        self.extract_simple_bindings(&program);
   717→
   718→        self.profile.expressions_processed += 1;
   719→
   720→        // Build a short description of what was added.
   721→        let node_count = self.scg.node_count();
   722→        let edge_count = self.scg.edge_count();
   723→        Ok(ReplResult::Ok(Some(format!(
   724→            "OK (SCG: {node_count} nodes, {edge_count} edges)"
   725→        ))))
   726→    }
   727→
   728→    /// Extract simple integer bindings from let statements for immediate eval.
   729→    fn extract_simple_bindings(&mut self, program: &vuma_parser::ast::Program) {
   730→        for item in &program.items {
   731→            match item {
   732→                Item::Stmt(Stmt::Let(l)) => {
   733→                    if let Expr::Lit {
   734→                        value: Lit::Int(n), ..
   735→                    } = &l.value
   736→                    {
   737→                        self.simple_vars.insert(l.name.clone(), *n);
   738→                    }
   739→                }
   740→                Item::Const(c) => {
   741→                    if let Expr::Lit {
   742→                        value: Lit::Int(n), ..
   743→                    } = &c.value
   744→                    {
   745→                        self.simple_vars.insert(c.name.clone(), *n);
   746→                    }
   747→                }
   748→                _ => {}
   749→            }
   750→        }
   751→    }

   757→    /// Handle a REPL command (input starting with `:`).
   758→    fn handle_command(&mut self, input: &str) -> Result<ReplResult, ReplError> {
   759→        let parts: Vec<&str> = input.splitn(2, ' ').collect();
   760→        let cmd = parts[0];
   761→        let arg = parts.get(1).map(|s| s.trim()).unwrap_or("");
   762→
   763→        match cmd {
   764→            ":help" => Ok(ReplResult::Ok(Some(self.help_text()))),
   765→            ":load" => self.cmd_load(arg),
   766→            ":verify" => self.cmd_verify(),
   767→            ":show" => self.cmd_show(arg),
   768→            ":compile" => self.cmd_compile(),
   769→            ":profile" => Ok(ReplResult::Ok(Some(format!("{}", self.profile)))),
   770→            ":history" => Ok(ReplResult::Ok(Some(self.format_history()))),
   771→            ":reset" => self.cmd_reset(),
   772→            ":quit" | ":q" | ":exit" => {
   773→                self.running = false;
   774→                Ok(ReplResult::Quit)
   775→            }
   776→            _ => Ok(ReplResult::Ok(Some(format!(
   777→                "Unknown command: {cmd}. Type :help for available commands."
   778→            )))),
   779→        }
   780→    }

   810→    /// Handle the `:load <file>` command.
   811→    fn cmd_load(&mut self, path: &str) -> Result<ReplResult, ReplError> {
   812→        if path.is_empty() {
   813→            return Ok(ReplResult::Ok(Some("Usage: :load <file>".to_string())));
   814→        }
   815→
   816→        let source = std::fs::read_to_string(path)
   817→            .map_err(|e| ReplError::General(format!("Cannot read '{}': {}", path, e)))?;
   818→
   819→        self.loaded_file = Some(path.to_string());
   820→
   821→        // Replace the source buffer with the file content.
   822→        self.source_buffer = source.clone();
   823→
   824→        // Parse and build SCG.
   825→        let parse_start = Instant::now();
   826→        let mut parser = Parser::new(&self.source_buffer);
   827→        let result = parser.parse_program();
   828→        if result.has_errors() {
   829→            let errors = result.errors.clone();
   830→            self.profile.parse_errors += errors.len();
   831→            self.profile.parse_time_ms += parse_start.elapsed().as_millis() as u64;
   832→            return Err(ReplError::ParseErrors(errors));
   833→        }
   834→        let program = result.unwrap();
   835→        self.profile.parse_time_ms += parse_start.elapsed().as_millis() as u64;
   836→
   837→        // Build SCG.
   838→        let scg_start = Instant::now();
   839→        let mut converter = AstToScg::new();
   840→        let scg = converter.convert(&program).map_err(|e| {
   841→            self.profile.scg_time_ms += scg_start.elapsed().as_millis() as u64;
   842→            ReplError::Parse(e)
   843→        })?;
   844→        self.scg = scg;
   845→        self.converter = converter;
   846→        self.profile.scg_time_ms += scg_start.elapsed().as_millis() as u64;
   847→
   848→        // Convert to MSG.
   849→        let msg_start = Instant::now();
   850→        match scg_to_msg::scg_to_msg(&self.scg) {
   851→            Ok(msg) => self.msg = Some(msg),
   852→            Err(e) => {
   853→                log::warn!("MSG conversion failed: {e}");
   854→                self.msg = None;
   855→            }
   856→        }
   857→        self.profile.msg_time_ms += msg_start.elapsed().as_millis() as u64;
   858→
   859→        // Extract simple bindings.
   860→        self.extract_simple_bindings(&program);
   861→
   862→        self.profile.expressions_processed += 1;
   863→
   864→        Ok(ReplResult::Ok(Some(format!(
   865→            "Loaded '{}' (SCG: {} nodes, {} edges)",
   866→            path,
   867→            self.scg.node_count(),
   868→            self.scg.edge_count()
   869→        ))))
   870→    }

   872→    /// Handle the `:verify` command.
   873→    fn cmd_verify(&mut self) -> Result<ReplResult, ReplError> {
   874→        let verify_start = Instant::now();
   875→
   876→        // Create verification input from the current SCG.
   877→        let input = VerificationInput::from_scg(self.scg.clone());
   878→
   879→        let result = self.aggregator.verify_all(&input);
   880→        self.profile.verify_time_ms += verify_start.elapsed().as_millis() as u64;
   881→        self.profile.verification_runs += 1;
   882→
   883→        self.last_verification = Some(result.clone());
   884→
   885→        Ok(ReplResult::Verification(result))
   886→    }

   888→    /// Handle the `:show` command.

   902→            _ => Ok(ReplResult::Ok(Some(format!(

   905→        }

   908→    /// Handle the `:compile` command — full pipeline.
   909→    fn cmd_compile(&mut self) -> Result<ReplResult, ReplError> {
   910→        let mut output = String::new();
   911→
   912→        // Step 1: Parse (already done if we have source).
   913→        if self.source_buffer.is_empty() {
   914→            return Ok(ReplResult::Ok(Some(
   915→                "No source to compile. Enter some VUMA code first.".to_string(),
   916→            )));
   917→        }
   918→
   919→        output.push_str(&format!("Source: {} bytes\n", self.source_buffer.len()));
   920→
   921→        // Step 2: Rebuild SCG.
   922→        let scg_start = Instant::now();
   923→        let mut parser = Parser::new(&self.source_buffer);
   924→        let result = parser.parse_program();
   925→        if result.has_errors() {
   926→            return Err(ReplError::ParseErrors(result.errors.clone()));
   927→        }
   928→        let program = result.unwrap();
   929→        let mut converter = AstToScg::new();
   930→        let scg = converter.convert(&program).map_err(ReplError::Parse)?;
   931→        self.scg = scg;
   932→        self.converter = converter;
   933→        self.profile.scg_time_ms += scg_start.elapsed().as_millis() as u64;
   934→
   935→        output.push_str(&format!(
   936→            "SCG: {} nodes, {} edges, {} regions\n",
   937→            self.scg.node_count(),
   938→            self.scg.edge_count(),
   939→            self.scg.region_count()
   940→        ));
   941→
   942→        // Step 3: Convert to MSG.
   943→        let msg_start = Instant::now();
   944→        match scg_to_msg::scg_to_msg(&self.scg) {
   945→            Ok(msg) => {
   946→                self.msg = Some(msg);
   947→                output.push_str(&format!("MSG: {}\n", self.msg.as_ref().unwrap()));
   948→            }
   949→            Err(e) => {
   950→                output.push_str(&format!("MSG conversion failed: {e}\n"));
   951→                self.msg = None;
   952→            }
   953→        }
   954→        self.profile.msg_time_ms += msg_start.elapsed().as_millis() as u64;
   955→
   956→        // Step 4: Verify.
   957→        let verify_start = Instant::now();
   958→        let input = VerificationInput::from_scg(self.scg.clone());
   959→        let result = self.aggregator.verify_all(&input);
   960→        self.profile.verify_time_ms += verify_start.elapsed().as_millis() as u64;
   961→        self.profile.verification_runs += 1;
   962→        self.last_verification = Some(result.clone());
   963→
   964→        output.push_str(&format!(
   965→            "Verification: {} ({}ms)\n",
   966→            result.overall, result.total_elapsed_ms
   967→        ));
   968→
   969→        self.profile.expressions_processed += 1;
   970→
   971→        Ok(ReplResult::Ok(Some(output)))
   972→    }

   974→    /// Handle the `:reset` command.

  1072→    /// Run the interactive REPL loop.
  1073→    ///
  1074→    /// Reads lines from stdin, processes them, and prints results.
  1075→    /// Supports basic up/down arrow key history navigation via ANSI
  1076→    /// escape sequences.
  1077→    pub fn run(&mut self) -> Result<(), ReplError> {
  1078→        println!("VUMA REPL v0.1.0");
  1079→        println!("Type :help for available commands.\n");
  1080→
  1081→        while self.running {
  1082→            print!("vuma> ");
  1083→            io::stdout().flush().map_err(ReplError::Io)?;
  1084→
  1085→            let mut input = String::new();
  1086→            match io::stdin().read_line(&mut input) {
  1087→                Ok(0) => {
  1088→                    // EOF.
  1089→                    self.running = false;
  1090→                    println!("Goodbye.");
  1091→                    break;
  1092→                }
  1093→                Ok(_) => {}
  1094→                Err(e) => return Err(ReplError::Io(e)),
  1095→            }
  1096→
  1097→            let line = input.trim_end();
  1098→
  1099→            match self.process_line(line) {
  1100→                Ok(ReplResult::Quit) => {
  1101→                    println!("Goodbye.");
  1102→                    break;
  1103→                }
  1104→                Ok(result) => {
  1105→                    if !result.to_string().is_empty() {
  1106→                        println!("{}", result);
  1107→                    }
  1108→                }
  1109→                Err(e) => match &e {
  1110→                    ReplError::Parse(pe) => {
  1111→                        let ctx = format_error_with_context(
  1112→                            &self.source_buffer,
  1113→                            &pe.span,
  1114→                            &pe.to_string(),
  1115→                        );
  1116→                        eprintln!("{ctx}");
  1117→                    }
  1118→                    ReplError::ParseErrors(errors) => {
  1119→                        for pe in errors {
  1120→                            let ctx = format_error_with_context(
  1121→                                &self.source_buffer,
  1122→                                &pe.span,
  1123→                                &pe.to_string(),
  1124→                            );
  1125→                            eprintln!("{ctx}");
  1126→                        }
  1127→                    }
  1128→                    _ => eprintln!("Error: {e}"),
  1129→                },
  1130→            }
  1131→        }
  1132→
  1133→        Ok(())
  1134→    }
  1135→}
