     1→//! # LoongArch 64-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the LoongArch 64-bit target (LP64 ABI).
     4→//! This module provides:
     5→//!
     6→//! - `Gpr` — General-purpose register enum (r0–r31)
     7→//! - `Fpr` — Floating-point register enum (f0–f31)
     8→//! - `Instruction` — LoongArch64 instruction enum with correct encoding
     9→//! - Encoding helpers for all 9 instruction formats (2R, 3R, 4R, 2RI8, 2RI12,
    10→//!   2RI14, 2RI16, 1RI21, I26)
    11→//! - `LoongArch64Backend` — `Backend` implementation that lowers IR to LoongArch64
    12→//!   machine code and emits ELF64 binaries
    13→//!
    14→//! ## LoongArch64 Register Convention (LP64 ABI)
    15→//!
    16→//! | Register(s) | ABI Name | Role                              |
    17→//! |-------------|----------|-----------------------------------|
    18→//! | r0          | zero     | Hardwired zero                    |
    19→//! | r1          | ra       | Return address                    |
    20→//! | r2          | tp       | Thread pointer                    |
    21→//! | r3          | sp       | Stack pointer                     |
    22→//! | r4–r11      | a0–a7    | Argument / return registers       |
    23→//! | r12–r20     | t0–t8    | Caller-saved temporaries          |
    24→//! | r21         | —        | Reserved                          |
    25→//! | r22         | fp       | Frame pointer (callee-saved)      |
    26→//! | r23–r31     | s0–s8    | Callee-saved                      |
    27→//!
    28→//! ## LoongArch64 FP Register Convention (LP64 ABI)
    29→//!
    30→//! | Register(s) | ABI Name | Role                              |
    31→//! |-------------|----------|-----------------------------------|
    32→//! | f0–f7       | fa0–fa7  | FP argument / return registers    |
    33→//! | f8–f23      | ft0–ft15 | Caller-saved temporaries          |
    34→//! | f24–f31     | fs0–fs7  | Callee-saved                      |
    35→//!
    36→//! ## Instruction Formats
    37→//!
    38→//! All instructions are 32 bits, little-endian, with no branch delay slots.
    39→//! Nine formats: 2R, 3R, 4R, 2RI8, 2RI12, 2RI14, 2RI16, 1RI21, I26.
    40→//!
    41→//! ## References
    42→//!
    43→//! - LoongArch Reference Manual, Volume 1: Basic Architecture
    44→//! - <https://loongson.github.io/LoongArch-Documentation/>
    45→

    95→impl Gpr {
    96→    /// Returns the 5-bit encoding index for this register.
    97→    pub fn encoding(&self) -> u32 {
    98→        *self as u32
    99→    }
   100→
   101→    /// Returns `true` if this register is available for register allocation.
   102→    ///
   103→    /// R0 (zero), Ra, Tp, and Sp are reserved.
   104→    pub fn is_allocatable(&self) -> bool {
   105→        !matches!(self, Gpr::R0 | Gpr::Ra | Gpr::Tp | Gpr::Sp)
   106→    }
   107→
   108→    /// Returns `true` if this register is callee-saved (fp, s0–s8).
   109→    pub fn is_callee_saved(&self) -> bool {
   110→        matches!(
   111→            self,
   112→            Gpr::Fp
   113→                | Gpr::S0
   114→                | Gpr::S1
   115→                | Gpr::S2
   116→                | Gpr::S3
   117→                | Gpr::S4
   118→                | Gpr::S5
   119→                | Gpr::S6
   120→                | Gpr::S7
   121→                | Gpr::S8
   122→        )
   123→    }
   124→
   125→    /// Returns `true` if this register is an argument register (a0–a7).
   126→    pub fn is_arg_reg(&self) -> bool {
   127→        matches!(
   128→            self,
   129→            Gpr::A0 | Gpr::A1 | Gpr::A2 | Gpr::A3 | Gpr::A4 | Gpr::A5 | Gpr::A6 | Gpr::A7
   130→        )
   131→    }

   153→            Gpr::T5 => "$t5",
   154→            Gpr::T6 => "$t6",
   155→            Gpr::T7 => "$t7",

   171→    /// Returns the Gpr for a given argument index (0–7). Returns `None` for
   172→    /// indices >= 8.
   173→    pub fn arg_register(index: usize) -> Option<Gpr> {
   174→        match index {
   175→            0 => Some(Gpr::A0),
   176→            1 => Some(Gpr::A1),
   177→            2 => Some(Gpr::A2),

   183→            _ => None,
   184→        }
   185→    }
   186→}

   323→// ===========================================================================
   324→// Instruction Encoding Helpers
   325→// ===========================================================================
   326→
   327→/// Encode a 2R format instruction.
   328→///
   329→/// Format: `opcode[31:10] | rj[9:5] | rd[4:0]`
   330→fn encode_2r(opcode: u32, rj: u32, rd: u32) -> [u8; 4] {
   331→    let word = ((opcode & 0x3FF_FFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   332→    word.to_le_bytes()
   333→}
   334→
   335→/// Encode a 3R format instruction.
   336→///
   337→/// Format: `opcode[31:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   338→fn encode_3r(opcode: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   339→    let word = ((opcode & 0x1FFFF) << 15) | ((rk & 0x1F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   340→    word.to_le_bytes()
   341→}
   342→
   343→/// Encode a 4R format instruction.
   344→///
   345→/// Format: `opcode[31:20] | ra[19:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   346→#[allow(dead_code)]
   347→fn encode_4r(opcode: u32, ra: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   348→    let word = ((opcode & 0xFFF) << 20)
   349→        | ((ra & 0x1F) << 15)
   350→        | ((rk & 0x1F) << 10)
   351→        | ((rj & 0x1F) << 5)
   352→        | (rd & 0x1F);
   353→    word.to_le_bytes()
   354→}
   355→
   356→/// Encode a 2RI8 format instruction.
   357→///
   358→/// Format: `opcode[31:22] | I8[21:14] | rj[9:5] | rd[4:0]`
   359→fn encode_2ri8(opcode: u32, imm8: u32, rj: u32, rd: u32) -> [u8; 4] {
   360→    let word = ((opcode & 0x3FF) << 22) | ((imm8 & 0xFF) << 14) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   361→    word.to_le_bytes()
   362→}
   363→
   364→/// Encode a 2RI12 format instruction.
   365→///
   366→/// Format: `opcode[31:22] | I12[21:10] | rj[9:5] | rd[4:0]`
   367→fn encode_2ri12(opcode: u32, imm12: u32, rj: u32, rd: u32) -> [u8; 4] {
   368→    let word =
   369→        ((opcode & 0x3FF) << 22) | ((imm12 & 0xFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   370→    word.to_le_bytes()
   371→}
   372→
   373→/// Encode a 2RI14 format instruction.
   374→///
   375→/// Format: `opcode[31:24] | I14[23:10] | rj[9:5] | rd[4:0]`
   376→fn encode_2ri14(opcode: u32, imm14: u32, rj: u32, rd: u32) -> [u8; 4] {
   377→    let word =
   378→        ((opcode & 0xFF) << 24) | ((imm14 & 0x3FFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   379→    word.to_le_bytes()
   380→}
   381→
   382→/// Encode a 2RI16 format instruction.
   383→///
   384→/// Format: `opcode[31:26] | I16[25:10] | rj[9:5] | rd[4:0]`
   385→fn encode_2ri16(opcode: u32, imm16: u32, rj: u32, rd: u32) -> [u8; 4] {
   386→    let word =
   387→        ((opcode & 0x3F) << 26) | ((imm16 & 0xFFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   388→    word.to_le_bytes()
   389→}
   390→
   391→/// Encode a 1RI21 format instruction.
   392→///
   393→/// Format: `opcode[31:26] | I21[25:5] | rj[9:5]`
   394→///
   395→/// The 21-bit immediate is split: bits[20:16] in bits[25:21], bits[15:0] in
   396→/// bits[20:5].
   397→fn encode_1ri21(opcode: u32, imm21: u32, rj: u32) -> [u8; 4] {
   398→    let word = ((opcode & 0x3F) << 26)
   399→        | ((imm21 & 0x1F) << 21)
   400→        | ((imm21 >> 5) & 0xFFFF) << 5
   401→        | (rj & 0x1F);
   402→    word.to_le_bytes()
   403→}
   404→
   405→/// Encode an I26 format instruction.
   406→///
   407→/// Format: `opcode[31:26] | I26[25:0]`
   408→///
   409→/// The 26-bit immediate is split: bits[25:16] in bits[25:16], bits[15:0] in
   410→/// bits[9:0] with bits[15:10] in bits[25:20] and bits[9:0] in bits[9:0].
   411→fn encode_i26(opcode: u32, imm26: u32) -> [u8; 4] {
   412→    let hi10 = (imm26 >> 16) & 0x3FF;
   413→    let lo16 = imm26 & 0xFFFF;
   414→    let word = ((opcode & 0x3F) << 26) | (hi10 << 16) | lo16;
   415→    word.to_le_bytes()
   416→}

   848→impl Instruction {
   849→    /// Encode this instruction into a 4-byte little-endian machine code word.
   850→    ///
   851→    /// Encoding follows the LoongArch ISA Specification.
   852→    pub fn encode(&self) -> [u8; 4] {
   853→        match self {
   854→            // ── Arithmetic (3R) ────────────────────────────────────
   855→            Instruction::AddW { rd, rj, rk } => {
   856→                encode_3r(OPC_ADD_W, rk.encoding(), rj.encoding(), rd.encoding())
   857→            }
   858→            Instruction::AddD { rd, rj, rk } => {
   859→                encode_3r(OPC_ADD_D, rk.encoding(), rj.encoding(), rd.encoding())
   860→            }
   861→            Instruction::SubW { rd, rj, rk } => {
   862→                encode_3r(OPC_SUB_W, rk.encoding(), rj.encoding(), rd.encoding())
   863→            }
   864→            Instruction::SubD { rd, rj, rk } => {
   865→                encode_3r(OPC_SUB_D, rk.encoding(), rj.encoding(), rd.encoding())
   866→            }
   867→            Instruction::Slt { rd, rj, rk } => {
   868→                encode_3r(OPC_SLT, rk.encoding(), rj.encoding(), rd.encoding())
   869→            }
   870→            Instruction::Sltu { rd, rj, rk } => {
   871→                encode_3r(OPC_SLTU, rk.encoding(), rj.encoding(), rd.encoding())
   872→            }
   873→            Instruction::MulW { rd, rj, rk } => {
   874→                encode_3r(OPC_MUL_W, rk.encoding(), rj.encoding(), rd.encoding())
   875→            }
   876→            Instruction::MulD { rd, rj, rk } => {
   877→                encode_3r(OPC_MUL_D, rk.encoding(), rj.encoding(), rd.encoding())
   878→            }
   879→            Instruction::DivW { rd, rj, rk } => {
   880→                encode_3r(OPC_DIV_W, rk.encoding(), rj.encoding(), rd.encoding())
   881→            }
   882→            Instruction::ModW { rd, rj, rk } => {
   883→                encode_3r(OPC_MOD_W, rk.encoding(), rj.encoding(), rd.encoding())
   884→            }
   885→            Instruction::DivD { rd, rj, rk } => {
   886→                encode_3r(OPC_DIV_D, rk.encoding(), rj.encoding(), rd.encoding())
   887→            }
   888→            Instruction::ModD { rd, rj, rk } => {
   889→                encode_3r(OPC_MOD_D, rk.encoding(), rj.encoding(), rd.encoding())
   890→            }
   891→
   892→            // ── Logical (3R) ──────────────────────────────────────
   893→            Instruction::And { rd, rj, rk } => {
   894→                encode_3r(OPC_AND, rk.encoding(), rj.encoding(), rd.encoding())
   895→            }
   896→            Instruction::Or { rd, rj, rk } => {
   897→                encode_3r(OPC_OR, rk.encoding(), rj.encoding(), rd.encoding())
   898→            }
   899→            Instruction::Xor { rd, rj, rk } => {
   900→                encode_3r(OPC_XOR, rk.encoding(), rj.encoding(), rd.encoding())
   901→            }
   902→            Instruction::Nor { rd, rj, rk } => {
   903→                encode_3r(OPC_NOR, rk.encoding(), rj.encoding(), rd.encoding())
   904→            }
   905→            Instruction::Andn { rd, rj, rk } => {
   906→                encode_3r(OPC_ANDN, rk.encoding(), rj.encoding(), rd.encoding())
   907→            }
   908→            Instruction::Orn { rd, rj, rk } => {
   909→                encode_3r(OPC_ORN, rk.encoding(), rj.encoding(), rd.encoding())
   910→            }
   911→

  1528→/// Build a minimal ELF64 binary for LoongArch64 from raw code bytes.
  1529→///
  1530→/// Produces a static executable with two LOAD segments:
  1531→///   1. PF_R | PF_X — .text (code)
  1532→///   2. PF_R | PF_W — .data / stack (writable memory)
  1533→fn build_loongarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1534→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (LoongArch64 typical page size)
  1535→
  1536→    let elf_header_size: u64 = 64;
  1537→    let phdr_size: u64 = 56;
  1538→    let num_phdrs: u64 = 2;
  1539→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1540→    // Page-align the text segment start in the file.
  1541→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1542→    let text_size = code.len() as u64;
  1543→
  1544→    // The data segment starts on the next page after the text.
  1545→    let text_file_end = text_offset + text_size;
  1546→    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1547→    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1548→    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1549→    let entry_point = base_addr + text_offset;
  1550→
  1551→    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1552→
  1553→    // --- e_ident ---
  1554→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1555→    elf.push(2); // ELFCLASS64
  1556→    elf.push(1); // ELFDATA2LSB
  1557→    elf.push(1); // EV_CURRENT
  1558→    elf.push(3); // ELFOSABI_LINUX
  1559→    elf.push(0); // padding
  1560→    elf.extend_from_slice(&[0u8; 7]); // padding
  1561→
  1562→    // --- ELF header fields ---
  1563→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1564→    elf.extend_from_slice(&258u16.to_le_bytes()); // e_machine = EM_LOONGARCH
  1565→    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1566→    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1567→    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1568→    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1569→    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1570→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1571→    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1572→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1573→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1574→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1575→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1576→
  1577→    // --- Program Header 1: LOAD (PF_R | PF_X) — .text ---
  1578→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1579→    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1580→    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1581→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1582→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1583→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1584→    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1585→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1586→
  1587→    // --- Program Header 2: LOAD (PF_R | PF_W) — .data / stack ---
  1588→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1589→    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1590→    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1591→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1592→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1593→    elf.extend_from_slice(&0u64.to_le_bytes()); // p_filesz (no initialized data)
  1594→    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1595→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1596→
  1597→    // --- .text section ---
  1598→    // Pad to page-aligned text_offset
  1599→    while (elf.len() as u64) < text_offset {
  1600→        elf.push(0);
  1601→    }
  1602→    elf.extend_from_slice(code);
  1603→
  1604→    // --- Pad to data segment offset (if needed) ---
  1605→    while (elf.len() as u64) < data_offset {
  1606→        elf.push(0);
  1607→    }
  1608→
  1609→    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1610→
  1611→    elf
  1612→}

  1618→/// Decode a single LoongArch64 32-bit instruction word into a mnemonic string.
  1619→fn decode_loongarch64_instruction(word: u32) -> String {
  1620→    let opcode = word & 0x7f;
  1621→    match opcode {
  1622→        0x00 => {
  1623→            let rd = (word >> 7) & 0x1f;
  1624→            let rj = (word >> 15) & 0x1f;
  1625→            let rk = (word >> 10) & 0x1f;
  1626→            format!("add.w $r{}, $r{}, $r{}", rd, rj, rk)
  1627→        }
  1628→        0x01 => {
  1629→            let rd = (word >> 7) & 0x1f;
  1630→            let rj = (word >> 15) & 0x1f;
  1631→            let rk = (word >> 10) & 0x1f;
  1632→            format!("sub.w $r{}, $r{}, $r{}", rd, rj, rk)
  1633→        }
  1634→        0x02 => {
  1635→            let rd = (word >> 7) & 0x1f;
  1636→            let rj = (word >> 15) & 0x1f;
  1637→            let rk = (word >> 10) & 0x1f;
  1638→            format!("mul.w $r{}, $r{}, $r{}", rd, rj, rk)
  1639→        }
  1640→        0x05 => {
  1641→            let rd = (word >> 7) & 0x1f;
  1642→            let _rj = (word >> 15) & 0x1f; // lu12i.w does not use rj in the disassembly
  1643→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1644→            format!("lu12i.w $r{}, {}({})", rd, si12, si12)
  1645→        }
  1646→        0x08 => {
  1647→            let rd = (word >> 7) & 0x1f;
  1648→            let rj = (word >> 15) & 0x1f;
  1649→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1650→            format!("ld.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1651→        }
  1652→        0x0a => {
  1653→            let rd = (word >> 7) & 0x1f;
  1654→            let rj = (word >> 15) & 0x1f;
  1655→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1656→            format!("st.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1657→        }
  1658→        0x0c => {
  1659→            let rd = (word >> 7) & 0x1f;
  1660→            let rj = (word >> 15) & 0x1f;
  1661→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1662→            format!("ld.d $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1663→        }
  1664→        0x0e => {
  1665→            let rd = (word >> 7) & 0x1f;
  1666→            let rj = (word >> 15) & 0x1f;
  1667→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1668→            format!("st.d $r{}, $r{}, {}({})", rd, rj, si12, si12)

  1709→/// Compute the stack frame size for an IR function on LoongArch64.
  1710→///
  1711→/// Sums `Alloc` instruction sizes, adds 16 bytes for the ra/fp save pair,
  1712→/// and rounds up to 16-byte alignment.
  1713→fn loongarch64_compute_frame_size(func: &IRFunction) -> usize {
  1714→    let mut total: u32 = 16; // ra + fp save pair
  1715→    for block in &func.blocks {
  1716→        for instr in &block.instructions {
  1717→            if let IRInstr::Alloc { size, .. } = instr {
  1718→                let aligned = (*size).div_ceil(16) * 16;
  1719→                total += aligned;
  1720→            }
  1721→        }
  1722→    }
  1723→    // Round up to 16-byte alignment
  1724→    total = (total + 15) & !15;
  1725→    total as usize
  1726→}

  1765→/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1766→fn map_vreg_to_gpr(
  1767→    vreg_id: u32,
  1768→    arg_index: Option<usize>,
  1769→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1770→) -> Gpr {
  1771→    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1772→        return *gpr;
  1773→    }
  1774→
  1775→    // If this is an argument, map it to the corresponding argument register.
  1776→    if let Some(idx) = arg_index {
  1777→        if let Some(reg) = Gpr::arg_register(idx) {
  1778→            vreg_map.insert(vreg_id, reg);
  1779→            return reg;
  1780→        }
  1781→    }
  1782→
  1783→    // Otherwise, pick the first available allocatable register.
  1784→    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1785→    for &reg in ALLOCATABLE_GPRS {
  1786→        if !used.contains(&reg) {
  1787→            vreg_map.insert(vreg_id, reg);
  1788→            return reg;
  1789→        }
  1790→    }
  1791→
  1792→    // Fallback: use t8 as a scratch register.
  1793→    Gpr::T8
  1794→}

  1804→/// Load a 64-bit immediate into a register using the lu12i.w/ori/lu32i.d/lu52i.d
  1805→/// instruction sequence.
  1806→///

  1808→fn load_imm_la64(rd: Gpr, imm: i64) -> Vec<AllocatedInstruction> {

  1856→///
  1857→/// - For `IRValue::Register`: looks up in `vreg_map`.
  1858→/// - For `IRValue::Immediate`: loads the value into `scratch` and returns
  1859→///   `(scratch, instructions)`.
  1860→/// - For `IRValue::Address`: loads the address into `scratch`.

  1862→fn resolve_gpr_la64(

  1864→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,

  1884→        }
  1885→    }
  1886→}
