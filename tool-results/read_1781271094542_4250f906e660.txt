     1→//! # RISC-V 64-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the RISC-V 64-bit target (RV64GC,
     4→//! LP64D ABI).  This module provides:
     5→//!
     6→//! - `Gpr` — General-purpose register enum (x0–x31)
     7→//! - `Fpr` — Floating-point register enum (f0–f31)
     8→//! - `Instruction` — RV64I + M + F/D instruction enum with correct encoding
     9→//! - Encoding helpers for R/I/S/B/U/J-type instruction formats
    10→//! - `RiscV64Backend` — `Backend` implementation that lowers IR to RISC-V machine code
    11→//!
    12→//! ## RISC-V Register Convention (LP64D ABI)
    13→//!
    14→//! | Register(s) | ABI Name | Role                              |
    15→//! |-------------|----------|-----------------------------------|
    16→//! | x0          | zero     | Hardwired zero                    |
    17→//! | x1          | ra       | Return address                    |
    18→//! | x2          | sp       | Stack pointer                     |
    19→//! | x3          | gp       | Global pointer                    |
    20→//! | x4          | tp       | Thread pointer                    |
    21→//! | x5–x7       | t0–t2    | Caller-saved temporaries          |
    22→//! | x8          | s0/fp    | Callee-saved / frame pointer      |
    23→//! | x9          | s1       | Callee-saved                      |
    24→//! | x10–x17     | a0–a7    | Argument / return registers       |
    25→//! | x18–x27     | s2–s11   | Callee-saved                      |
    26→//! | x28–x31     | t3–t6    | Caller-saved temporaries          |
    27→//!
    28→//! ## References
    29→//!
    30→//! - RISC-V Instruction Set Manual, Volume I: User-Level ISA, Document 20191213
    31→//! - <https://riscv.org/specifications/>

   403→/// Encode an R-type instruction.
   404→///
   405→/// Format: `funct7[31:25] | rs2[24:20] | rs1[19:15] | funct3[14:12] | rd[11:7] | opcode[6:0]`
   406→fn encode_r_type(funct7: u32, rs2: u32, rs1: u32, funct3: u32, rd: u32, opcode: u32) -> [u8; 4] {
   407→    let word = ((funct7 & 0x7F) << 25)
   408→        | ((rs2 & 0x1F) << 20)
   409→        | ((rs1 & 0x1F) << 15)
   410→        | ((funct3 & 0x7) << 12)
   411→        | ((rd & 0x1F) << 7)
   412→        | (opcode & 0x7F);
   413→    word.to_le_bytes()
   414→}
   415→
   416→/// Encode an I-type instruction.
   417→///
   418→/// Format: `imm[31:20] | rs1[19:15] | funct3[14:12] | rd[11:7] | opcode[6:0]`
   419→fn encode_i_type(imm: u32, rs1: u32, funct3: u32, rd: u32, opcode: u32) -> [u8; 4] {
   420→    let word = ((imm & 0xFFF) << 20)
   421→        | ((rs1 & 0x1F) << 15)
   422→        | ((funct3 & 0x7) << 12)
   423→        | ((rd & 0x1F) << 7)
   424→        | (opcode & 0x7F);
   425→    word.to_le_bytes()
   426→}
   427→
   428→/// Encode an S-type instruction.
   429→///
   430→/// Format: `imm[11:5][31:25] | rs2[24:20] | rs1[19:15] | funct3[14:12] | imm[4:0][11:7] | opcode[6:0]`
   431→fn encode_s_type(imm: u32, rs2: u32, rs1: u32, funct3: u32, opcode: u32) -> [u8; 4] {
   432→    let imm_lo = imm & 0x1F;
   433→    let imm_hi = (imm >> 5) & 0x7F;
   434→    let word = (imm_hi << 25)
   435→        | ((rs2 & 0x1F) << 20)
   436→        | ((rs1 & 0x1F) << 15)
   437→        | ((funct3 & 0x7) << 12)
   438→        | (imm_lo << 7)
   439→        | (opcode & 0x7F);
   440→    word.to_le_bytes()
   441→}
   442→
   443→/// Encode a B-type instruction.
   444→///
   445→/// Format: `imm[12|10:5][31:25] | rs2[24:20] | rs1[19:15] | funct3[14:12] | imm[4:1|11][11:7] | opcode[6:0]`
   446→///
   447→/// The immediate is a signed 13-bit byte offset (bit 0 is always 0).
   448→fn encode_b_type(imm: i32, rs2: u32, rs1: u32, funct3: u32, opcode: u32) -> [u8; 4] {
   449→    // B-type immediate bit layout:
   450→    // bit 31:   imm[12]
   451→    // bits 30:25: imm[10:5]
   452→    // bits 11:8:  imm[4:1]
   453→    // bit 7:    imm[11]
   454→    let imm_u = imm as u32;
   455→    let word = (((imm_u >> 12) & 0x1) << 31)
   456→        | (((imm_u >> 5) & 0x3F) << 25)
   457→        | ((rs2 & 0x1F) << 20)
   458→        | ((rs1 & 0x1F) << 15)
   459→        | ((funct3 & 0x7) << 12)
   460→        | (((imm_u >> 1) & 0xF) << 8)
   461→        | (((imm_u >> 11) & 0x1) << 7)
   462→        | (opcode & 0x7F);
   463→    word.to_le_bytes()
   464→}
   465→
   466→/// Encode a U-type instruction.
   467→///
   468→/// Format: `imm[31:12] | rd[11:7] | opcode[6:0]`
   469→///
   470→/// The immediate is the upper 20 bits; the lower 12 bits are zero.
   471→fn encode_u_type(imm: u32, rd: u32, opcode: u32) -> [u8; 4] {
   472→    let word = (imm & 0xFFFFF000) | ((rd & 0x1F) << 7) | (opcode & 0x7F);
   473→    word.to_le_bytes()
   474→}
   475→
   476→/// Encode a J-type instruction.
   477→///
   478→/// Format: `imm[20|10:1|11|19:12][31:12] | rd[11:7] | opcode[6:0]`
   479→///
   480→/// The immediate is a signed 21-bit byte offset (bit 0 is always 0).
   481→fn encode_j_type(imm: i32, rd: u32, opcode: u32) -> [u8; 4] {
   482→    // J-type immediate bit layout:
   483→    // bit 31:    imm[20]
   484→    // bits 30:21: imm[10:1]
   485→    // bit 20:    imm[11]
   486→    // bits 19:12: imm[19:12]
   487→    let imm_u = imm as u32;
   488→    let word = (((imm_u >> 20) & 0x1) << 31)
   489→        | (((imm_u >> 1) & 0x3FF) << 21)
   490→        | (((imm_u >> 11) & 0x1) << 20)
   491→        | (((imm_u >> 12) & 0xFF) << 12)
   492→        | ((rd & 0x1F) << 7)
   493→        | (opcode & 0x7F);
   494→    word.to_le_bytes()
   495→}

   497→// ===========================================================================
   498→// Instruction Enum
   499→// ===========================================================================
   500→
   501→/// RISC-V 64-bit instruction representations for code generation.
   502→///
   503→/// Covers RV64I base, M extension, and key F/D extension instructions.
   504→/// Each variant captures the operands needed for encoding and disassembly.
   505→/// The `encode()` method produces a 4-byte little-endian machine code word.
   506→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   507→pub enum Instruction {
   508→    // ── RV64I: Upper Immediate ────────────────────────────────────────
   509→    /// Load Upper Immediate: `lui rd, imm`
   510→    Lui { rd: Gpr, imm: u32 },
   511→    /// Add Upper Immediate to PC: `auipc rd, imm`
   512→    Auipc { rd: Gpr, imm: u32 },
   513→
   514→    // ── RV64I: Jumps ─────────────────────────────────────────────────
   515→    /// Jump and Link: `jal rd, offset`
   516→    Jal { rd: Gpr, offset: i32 },
   517→    /// Jump and Link Register: `jalr rd, rs1, imm`
   518→    Jalr { rd: Gpr, rs1: Gpr, imm: i32 },
   519→
   520→    // ── RV64I: Branches ──────────────────────────────────────────────
   521→    /// Branch if Equal: `beq rs1, rs2, offset`
   522→    Beq { rs1: Gpr, rs2: Gpr, offset: i32 },
   523→    /// Branch if Not Equal: `bne rs1, rs2, offset`
   524→    Bne { rs1: Gpr, rs2: Gpr, offset: i32 },
   525→    /// Branch if Less Than (signed): `blt rs1, rs2, offset`
   526→    Blt { rs1: Gpr, rs2: Gpr, offset: i32 },
   527→    /// Branch if Greater or Equal (signed): `bge rs1, rs2, offset`
   528→    Bge { rs1: Gpr, rs2: Gpr, offset: i32 },
   529→    /// Branch if Less Than (unsigned): `bltu rs1, rs2, offset`
   530→    Bltu { rs1: Gpr, rs2: Gpr, offset: i32 },

   691→impl Instruction {
   692→    /// Encode this instruction into a 4-byte little-endian machine code word.
   693→    ///
   694→    /// Encoding follows the RISC-V ISA Specification.
   695→    pub fn encode(&self) -> [u8; 4] {
   696→        match self {
   697→            // ── Upper Immediate ──────────────────────────────────────
   698→            Instruction::Lui { rd, imm } => encode_u_type(*imm, rd.encoding(), OP_LUI),
   699→            Instruction::Auipc { rd, imm } => encode_u_type(*imm, rd.encoding(), OP_AUIPC),
   700→
   701→            // ── Jumps ───────────────────────────────────────────────
   702→            Instruction::Jal { rd, offset } => encode_j_type(*offset, rd.encoding(), OP_JAL),
   703→            Instruction::Jalr { rd, rs1, imm } => encode_i_type(
   704→                (*imm as u32) & 0xFFF,
   705→                rs1.encoding(),
   706→                0b000,
   707→                rd.encoding(),
   708→                OP_JALR,
   709→            ),
   710→
   711→            // ── Branches ────────────────────────────────────────────
   712→            Instruction::Beq { rs1, rs2, offset } => {
   713→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b000, OP_BRANCH)
   714→            }
   715→            Instruction::Bne { rs1, rs2, offset } => {
   716→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b001, OP_BRANCH)
   717→            }
   718→            Instruction::Blt { rs1, rs2, offset } => {
   719→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b100, OP_BRANCH)
   720→            }
   721→            Instruction::Bge { rs1, rs2, offset } => {
   722→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b101, OP_BRANCH)
   723→            }
   724→            Instruction::Bltu { rs1, rs2, offset } => {
   725→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b110, OP_BRANCH)
   726→            }
   727→            Instruction::Bgeu { rs1, rs2, offset } => {
   728→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b111, OP_BRANCH)
   729→            }
   730→
   731→            // ── Loads ───────────────────────────────────────────────
   732→            Instruction::Lb { rd, rs1, imm } => encode_i_type(
   733→                (*imm as u32) & 0xFFF,
   734→                rs1.encoding(),
   735→                0b000,
   736→                rd.encoding(),
   737→                OP_LOAD,
   738→            ),
   739→            Instruction::Lh { rd, rs1, imm } => encode_i_type(
   740→                (*imm as u32) & 0xFFF,
   741→                rs1.encoding(),
   742→                0b001,
   743→                rd.encoding(),
   744→                OP_LOAD,
   745→            ),
   746→            Instruction::Lw { rd, rs1, imm } => encode_i_type(
   747→                (*imm as u32) & 0xFFF,
   748→                rs1.encoding(),
   749→                0b010,
   750→                rd.encoding(),
   751→                OP_LOAD,
   752→            ),
   753→            Instruction::Ld { rd, rs1, imm } => encode_i_type(
   754→                (*imm as u32) & 0xFFF,
   755→                rs1.encoding(),
   756→                0b011,
   757→                rd.encoding(),
   758→                OP_LOAD,
   759→            ),
   760→            Instruction::Lbu { rd, rs1, imm } => encode_i_type(
   761→                (*imm as u32) & 0xFFF,
   762→                rs1.encoding(),
   763→                0b100,
   764→                rd.encoding(),
   765→                OP_LOAD,
   766→            ),
   767→            Instruction::Lhu { rd, rs1, imm } => encode_i_type(
   768→                (*imm as u32) & 0xFFF,
   769→                rs1.encoding(),
   770→                0b101,
   771→                rd.encoding(),
   772→                OP_LOAD,
   773→            ),
   774→            Instruction::Lwu { rd, rs1, imm } => encode_i_type(
   775→                (*imm as u32) & 0xFFF,
   776→                rs1.encoding(),
   777→                0b110,
   778→                rd.encoding(),
   779→                OP_LOAD,
   780→            ),
   781→
   782→            // ── Stores ──────────────────────────────────────────────
   783→            Instruction::Sb { rs1, rs2, imm } => encode_s_type(
   784→                (*imm as u32) & 0xFFF,
   785→                rs2.encoding(),
   786→                rs1.encoding(),
   787→                0b000,
   788→                OP_STORE,
   789→            ),
   790→            Instruction::Sh { rs1, rs2, imm } => encode_s_type(
   791→                (*imm as u32) & 0xFFF,
   792→                rs2.encoding(),
   793→                rs1.encoding(),
   794→                0b001,
   795→                OP_STORE,
   796→            ),
   797→            Instruction::Sw { rs1, rs2, imm } => encode_s_type(
   798→                (*imm as u32) & 0xFFF,
   799→                rs2.encoding(),
   800→                rs1.encoding(),
   801→                0b010,
   802→                OP_STORE,
   803→            ),
   804→            Instruction::Sd { rs1, rs2, imm } => encode_s_type(
   805→                (*imm as u32) & 0xFFF,
   806→                rs2.encoding(),
   807→                rs1.encoding(),
   808→                0b011,
   809→                OP_STORE,
   810→            ),
   811→
   812→            // ── Immediate Arithmetic ────────────────────────────────
   813→            Instruction::Addi { rd, rs1, imm } => encode_i_type(
   814→                (*imm as u32) & 0xFFF,
   815→                rs1.encoding(),
   816→                0b000,
   817→                rd.encoding(),
   818→                OP_IMM,
   819→            ),
   820→            Instruction::Slti { rd, rs1, imm } => encode_i_type(
   821→                (*imm as u32) & 0xFFF,
   822→                rs1.encoding(),
   823→                0b010,
   824→                rd.encoding(),
   825→                OP_IMM,
   826→            ),
   827→            Instruction::Sltiu { rd, rs1, imm } => encode_i_type(
   828→                (*imm as u32) & 0xFFF,
   829→                rs1.encoding(),
   830→                0b011,
   831→                rd.encoding(),
   832→                OP_IMM,
   833→            ),
   834→            Instruction::Xori { rd, rs1, imm } => encode_i_type(
   835→                (*imm as u32) & 0xFFF,
   836→                rs1.encoding(),
   837→                0b100,
   838→                rd.encoding(),
   839→                OP_IMM,
   840→            ),
   841→            Instruction::Ori { rd, rs1, imm } => encode_i_type(
   842→                (*imm as u32) & 0xFFF,
   843→                rs1.encoding(),
   844→                0b110,
   845→                rd.encoding(),
   846→                OP_IMM,
   847→            ),
   848→            Instruction::Andi { rd, rs1, imm } => encode_i_type(
   849→                (*imm as u32) & 0xFFF,
   850→                rs1.encoding(),
   851→                0b111,
   852→                rd.encoding(),
   853→                OP_IMM,
   854→            ),
   855→            Instruction::Slli { rd, rs1, shamt } => {
   856→                // funct7 = 0b0000000, funct3 = 0b001
   857→                // For RV64I, shamt is 6 bits (bits [25:20])
   858→                let funct7_and_shamt = (*shamt & 0x3F) << 20;
   859→                let word = funct7_and_shamt
   860→                    | (rs1.encoding() << 15)
   861→                    | (0b001 << 12)
   862→                    | (rd.encoding() << 7)
   863→                    | OP_IMM;
   864→                word.to_le_bytes()
   865→            }
   866→            Instruction::Srli { rd, rs1, shamt } => {
   867→                // funct7 = 0b0000000, funct3 = 0b101
   868→                let funct7_and_shamt = (*shamt & 0x3F) << 20;
   869→                let word = funct7_and_shamt
   870→                    | (rs1.encoding() << 15)
   871→                    | (0b101 << 12)
   872→                    | (rd.encoding() << 7)
   873→                    | OP_IMM;
   874→                word.to_le_bytes()
   875→            }
   876→            Instruction::Srai { rd, rs1, shamt } => {
   877→                // funct7 = 0b0100000, funct3 = 0b101
   878→                let funct7_and_shamt = (0b0100000u32 << 25) | ((*shamt & 0x3F) << 20);
   879→                let word = funct7_and_shamt
   880→                    | (rs1.encoding() << 15)
   881→                    | (0b101 << 12)
   882→                    | (rd.encoding() << 7)
   883→                    | OP_IMM;
   884→                word.to_le_bytes()
   885→            }
   886→
   887→            // ── Register Arithmetic ─────────────────────────────────
   888→            Instruction::Add { rd, rs1, rs2 } => encode_r_type(
   889→                0b0000000,
   890→                rs2.encoding(),
   891→                rs1.encoding(),
   892→                0b000,
   893→                rd.encoding(),
   894→                OP_REG,
   895→            ),
   896→            Instruction::Sub { rd, rs1, rs2 } => encode_r_type(
   897→                0b0100000,
   898→                rs2.encoding(),
   899→                rs1.encoding(),
   900→                0b000,
   901→                rd.encoding(),
   902→                OP_REG,
   903→            ),
   904→            Instruction::Sll { rd, rs1, rs2 } => encode_r_type(
   905→                0b0000000,
   906→                rs2.encoding(),
   907→                rs1.encoding(),
   908→                0b001,
   909→                rd.encoding(),
   910→                OP_REG,
   911→            ),
   912→            Instruction::Slt { rd, rs1, rs2 } => encode_r_type(
   913→                0b0000000,
   914→                rs2.encoding(),
   915→                rs1.encoding(),
   916→                0b010,
   917→                rd.encoding(),
   918→                OP_REG,
   919→            ),
   920→            Instruction::Sltu { rd, rs1, rs2 } => encode_r_type(
   921→                0b0000000,
   922→                rs2.encoding(),
   923→                rs1.encoding(),
   924→                0b011,
   925→                rd.encoding(),
   926→                OP_REG,
   927→            ),
   928→            Instruction::Xor { rd, rs1, rs2 } => encode_r_type(
   929→                0b0000000,
   930→                rs2.encoding(),
   931→                rs1.encoding(),
   932→                0b100,
   933→                rd.encoding(),
   934→                OP_REG,
   935→            ),
   936→            Instruction::Srl { rd, rs1, rs2 } => encode_r_type(
   937→                0b0000000,
   938→                rs2.encoding(),
   939→                rs1.encoding(),
   940→                0b101,
   941→                rd.encoding(),
   942→                OP_REG,
   943→            ),
   944→            Instruction::Sra { rd, rs1, rs2 } => encode_r_type(
   945→                0b0100000,
   946→                rs2.encoding(),
   947→                rs1.encoding(),
   948→                0b101,
   949→                rd.encoding(),
   950→                OP_REG,
   951→            ),
   952→            Instruction::Or { rd, rs1, rs2 } => encode_r_type(
   953→                0b0000000,
   954→                rs2.encoding(),
   955→                rs1.encoding(),
   956→                0b110,
   957→                rd.encoding(),
   958→                OP_REG,
   959→            ),
   960→            Instruction::And { rd, rs1, rs2 } => encode_r_type(
   961→                0b0000000,
   962→                rs2.encoding(),
   963→                rs1.encoding(),
   964→                0b111,
   965→                rd.encoding(),
   966→                OP_REG,
   967→            ),
   968→
   969→            // ── Word-level Arithmetic (RV64) ────────────────────────
   970→            Instruction::Addw { rd, rs1, rs2 } => encode_r_type(

   981→                rs1.encoding(),
   982→                0b000,
   983→                rd.encoding(),
   984→                OP_REG32,
   985→            ),
   986→            Instruction::Sllw { rd, rs1, rs2 } => encode_r_type(
   987→                0b0000000,
   988→                rs2.encoding(),
   989→                rs1.encoding(),
   990→                0b001,
   991→                rd.encoding(),
   992→                OP_REG32,
   993→            ),
   994→            Instruction::Srlw { rd, rs1, rs2 } => encode_r_type(
   995→                0b0000000,
   996→                rs2.encoding(),
   997→                rs1.encoding(),
   998→                0b101,
   999→                rd.encoding(),
  1000→                OP_REG32,
  1001→            ),
  1002→            Instruction::Sraw { rd, rs1, rs2 } => encode_r_type(
  1003→                0b0100000,
  1004→                rs2.encoding(),
  1005→                rs1.encoding(),
  1006→                0b101,
  1007→                rd.encoding(),
  1008→                OP_REG32,
  1009→            ),
  1010→            Instruction::Addiw { rd, rs1, imm } => encode_i_type(
  1011→                (*imm as u32) & 0xFFF,
  1012→                rs1.encoding(),
  1013→                0b000,
  1014→                rd.encoding(),
  1015→                OP_IMM32,
  1016→            ),
  1017→            Instruction::Slliw { rd, rs1, shamt } => {
  1018→                // funct7 = 0b0000000, funct3 = 0b001, shamt is 5 bits
  1019→                let funct7_and_shamt = (*shamt & 0x1F) << 20;
  1020→                let word = funct7_and_shamt
  1021→                    | (rs1.encoding() << 15)
  1022→                    | (0b001 << 12)
  1023→                    | (rd.encoding() << 7)
  1024→                    | OP_IMM32;
  1025→                word.to_le_bytes()
  1026→            }
  1027→            Instruction::Srliw { rd, rs1, shamt } => {
  1028→                let funct7_and_shamt = (*shamt & 0x1F) << 20;
  1029→                let word = funct7_and_shamt
  1030→                    | (rs1.encoding() << 15)
  1031→                    | (0b101 << 12)
  1032→                    | (rd.encoding() << 7)
  1033→                    | OP_IMM32;
  1034→                word.to_le_bytes()
  1035→            }
  1036→            Instruction::Sraiw { rd, rs1, shamt } => {
  1037→                let funct7_and_shamt = (0b0100000u32 << 25) | ((*shamt & 0x1F) << 20);
  1038→                let word = funct7_and_shamt
  1039→                    | (rs1.encoding() << 15)
  1040→                    | (0b101 << 12)
  1041→                    | (rd.encoding() << 7)
  1042→                    | OP_IMM32;
  1043→                word.to_le_bytes()
  1044→            }
  1045→
  1046→            // ── M Extension ─────────────────────────────────────────
  1047→            Instruction::Mul { rd, rs1, rs2 } => encode_r_type(
  1048→                0b0000001,
  1049→                rs2.encoding(),
  1050→                rs1.encoding(),
  1051→                0b000,
  1052→                rd.encoding(),
  1053→                OP_REG,
  1054→            ),
  1055→            Instruction::Mulh { rd, rs1, rs2 } => encode_r_type(
  1056→                0b0000001,
  1057→                rs2.encoding(),
  1058→                rs1.encoding(),
  1059→                0b001,
  1060→                rd.encoding(),
  1061→                OP_REG,
  1062→            ),
  1063→            Instruction::Mulhsu { rd, rs1, rs2 } => encode_r_type(
  1064→                0b0000001,
  1065→                rs2.encoding(),
  1066→                rs1.encoding(),
  1067→                0b010,
  1068→                rd.encoding(),
  1069→                OP_REG,
  1070→            ),
  1071→            Instruction::Mulhu { rd, rs1, rs2 } => encode_r_type(
  1072→                0b0000001,
  1073→                rs2.encoding(),
  1074→                rs1.encoding(),
  1075→                0b011,
  1076→                rd.encoding(),
  1077→                OP_REG,
  1078→            ),
  1079→            Instruction::Div { rd, rs1, rs2 } => encode_r_type(
  1080→                0b0000001,
  1081→                rs2.encoding(),
  1082→                rs1.encoding(),
  1083→                0b100,
  1084→                rd.encoding(),
  1085→                OP_REG,
  1086→            ),
  1087→            Instruction::Divu { rd, rs1, rs2 } => encode_r_type(
  1088→                0b0000001,
  1089→                rs2.encoding(),
  1090→                rs1.encoding(),
  1091→                0b101,
  1092→                rd.encoding(),
  1093→                OP_REG,
  1094→            ),
  1095→            Instruction::Rem { rd, rs1, rs2 } => encode_r_type(
  1096→                0b0000001,
  1097→                rs2.encoding(),
  1098→                rs1.encoding(),
  1099→                0b110,
  1100→                rd.encoding(),
  1101→                OP_REG,
  1102→            ),
  1103→            Instruction::Remu { rd, rs1, rs2 } => encode_r_type(
  1104→                0b0000001,
  1105→                rs2.encoding(),
  1106→                rs1.encoding(),
  1107→                0b111,
  1108→                rd.encoding(),
  1109→                OP_REG,
  1110→            ),
  1111→
  1112→            // ── F/D Extension: Load/Store ───────────────────────────
  1113→            Instruction::Flw { rd, rs1, imm } => encode_i_type(
  1114→                (*imm as u32) & 0xFFF,
  1115→                rs1.encoding(),
  1116→                0b010,
  1117→                rd.encoding(),
  1118→                OP_LOAD,
  1119→            ),
  1120→            Instruction::Fld { rd, rs1, imm } => encode_i_type(
  1121→                (*imm as u32) & 0xFFF,
  1122→                rs1.encoding(),
  1123→                0b011,
  1124→                rd.encoding(),
  1125→                OP_LOAD,
  1126→            ),
  1127→            Instruction::Fsw { rs1, rs2, imm } => encode_s_type(
  1128→                (*imm as u32) & 0xFFF,
  1129→                rs2.encoding(),
  1130→                rs1.encoding(),
  1131→                0b010,
  1132→                OP_STORE,
  1133→            ),
  1134→            Instruction::Fsd { rs1, rs2, imm } => encode_s_type(

  1136→                rs2.encoding(),
  1137→                rs1.encoding(),
  1138→                0b011,
  1139→                OP_STORE,
  1140→            ),
  1141→
  1142→            // ── F/D Extension: Arithmetic ───────────────────────────
  1143→            Instruction::FaddD { rd, rs1, rs2 } => {
  1144→                // FADD.D: funct7=0b0000001, rm=0b111 (dynamic), opcode=OP_FP
  1145→                // Actually: funct7[6:0] = 0000001, rs2, rs1, rm=111, rd, opcode
  1146→                encode_r_type(
  1147→                    0b0000001,
  1148→                    rs2.encoding(),
  1149→                    rs1.encoding(),
  1150→                    0b111,
  1151→                    rd.encoding(),
  1152→                    OP_FP,
  1153→                )
  1154→            }

  1159→                0b111,
  1160→                rd.encoding(),
  1161→                OP_FP,
  1162→            ),
  1163→            Instruction::FmulD { rd, rs1, rs2 } => encode_r_type(

  1165→                rs2.encoding(),
  1166→                rs1.encoding(),
  1167→                0b111,
  1168→                rd.encoding(),
  1169→                OP_FP,
  1170→            ),

  1173→                rs2.encoding(),
  1174→                rs1.encoding(),
  1175→                0b111,
  1176→                rd.encoding(),
  1177→                OP_FP,
  1178→            ),
  1179→            Instruction::FmvD { rd, rs1 } => {

  1181→                encode_r_type(
  1182→                    0b0010001,
  1183→                    rs1.encoding(),
  1184→                    rs1.encoding(),
  1185→                    0b000,
  1186→                    rd.encoding(),
  1187→                    OP_FP,
  1188→                )
  1189→            }
  1190→
  1191→            // ── Zicsr Extension ──────────────────────────────────────
  1192→            Instruction::Csrrw { rd, csr, rs1 } => {
  1193→                // CSRRW: I-type, funct3=0b001, opcode=SYSTEM
  1194→                encode_i_type(*csr, rs1.encoding(), 0b001, rd.encoding(), OP_SYSTEM)
  1195→            }
  1196→            Instruction::Csrrs { rd, csr, rs1 } => {
  1197→                // CSRRS: I-type, funct3=0b010, opcode=SYSTEM
  1198→                encode_i_type(*csr, rs1.encoding(), 0b010, rd.encoding(), OP_SYSTEM)
  1199→            }
  1200→            Instruction::Csrrc { rd, csr, rs1 } => {
  1201→                // CSRRC: I-type, funct3=0b011, opcode=SYSTEM
  1202→                encode_i_type(*csr, rs1.encoding(), 0b011, rd.encoding(), OP_SYSTEM)
  1203→            }
  1204→            Instruction::Csrrwi { rd, csr, uimm } => {
  1205→                // CSRRWI: I-type, funct3=0b101, opcode=SYSTEM
  1206→                encode_i_type(*csr, *uimm & 0x1F, 0b101, rd.encoding(), OP_SYSTEM)
  1207→            }
  1208→            Instruction::Csrrsi { rd, csr, uimm } => {
  1209→                // CSRRSI: I-type, funct3=0b110, opcode=SYSTEM
  1210→                encode_i_type(*csr, *uimm & 0x1F, 0b110, rd.encoding(), OP_SYSTEM)
  1211→            }
  1212→            Instruction::Csrrci { rd, csr, uimm } => {
  1213→                // CSRRCI: I-type, funct3=0b111, opcode=SYSTEM
  1214→                encode_i_type(*csr, *uimm & 0x1F, 0b111, rd.encoding(), OP_SYSTEM)
  1215→            }
  1216→
  1217→            // ── Zifencei Extension ──────────────────────────────────
  1218→            Instruction::FenceI => {
  1219→                // FENCE.I: opcode=MISC-MEM, funct3=0b001, rd=0, rs1=0, imm=0
  1220→                encode_i_type(0, 0, 0b001, 0, OP_MISC_MEM)
  1221→            }
  1222→
  1223→            // ── System / Misc ───────────────────────────────────────
  1224→            Instruction::Ecall => {
  1225→                // ECALL = 0x00000073
  1226→                0x00000073u32.to_le_bytes()
  1227→            }
  1228→            Instruction::Ebreak => {
  1229→                // EBREAK = 0x00100073
  1230→                0x00100073u32.to_le_bytes()
  1231→            }
  1232→            Instruction::Fence { pred, succ } => {
  1233→                // FENCE: opcode=MISC-MEM, funct3=0b000, rd=0, rs1=0
  1234→                let imm = ((*pred & 0xF) << 4) | (*succ & 0xF);
  1235→                encode_i_type(imm, 0, 0b000, 0, OP_MISC_MEM)
  1236→            }
  1237→            Instruction::Nop => {
  1238→                // NOP = ADDI x0, x0, 0 = 0x00000013
  1239→                encode_i_type(0, 0, 0b000, 0, OP_IMM)
  1240→            }
  1241→        }
  1242→    }

  1327→    /// Decode a 32-bit RISC-V machine-code word into an `Instruction`.
  1328→    ///
  1329→    /// Returns `None` for encodings not yet covered by the decoder.
  1330→    /// Covers all instruction classes defined in this backend.
  1331→    pub fn decode(word: u32) -> Option<Instruction> {
  1332→        let opcode = word & 0x7F;
  1333→        let rd = (word >> 7) & 0x1F;
  1334→        let funct3 = (word >> 12) & 0x7;
  1335→        let rs1 = (word >> 15) & 0x1F;
  1336→        let rs2 = (word >> 20) & 0x1F;
  1337→        let funct7 = (word >> 25) & 0x7F;
  1338→
  1339→        match opcode {
  1340→            // ── LUI ────────────────────────────────────────────────
  1341→            0b0110111 => {
  1342→                let rd_reg = Gpr::from_encoding(rd)?;
  1343→                let imm = word & 0xFFFFF000;
  1344→                Some(Instruction::Lui { rd: rd_reg, imm })
  1345→            }
  1346→
  1347→            // ── AUIPC ──────────────────────────────────────────────
  1348→            0b0010111 => {
  1349→                let rd_reg = Gpr::from_encoding(rd)?;
  1350→                let imm = word & 0xFFFFF000;
  1351→                Some(Instruction::Auipc { rd: rd_reg, imm })
  1352→            }
  1353→
  1354→            // ── JAL ────────────────────────────────────────────────
  1355→            0b1101111 => {
  1356→                let rd_reg = Gpr::from_encoding(rd)?;
  1357→                let imm20 = ((word >> 31) & 1) << 20
  1358→                    | ((word >> 12) & 0xFF) << 12
  1359→                    | ((word >> 20) & 1) << 11
  1360→                    | ((word >> 21) & 0x3FF) << 1;
  1361→                let offset = ((imm20 << 11) as i32) >> 11;
  1362→                Some(Instruction::Jal { rd: rd_reg, offset })
  1363→            }
  1364→
  1365→            // ── JALR ───────────────────────────────────────────────
  1366→            0b1100111 => {
  1367→                let rd_reg = Gpr::from_encoding(rd)?;
  1368→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1369→                let imm = (((word >> 20) as i32) << 20) >> 20;
  1370→                Some(Instruction::Jalr {
  1371→                    rd: rd_reg,
  1372→                    rs1: rs1_reg,
  1373→                    imm,
  1374→                })
  1375→            }
  1376→
  1377→            // ── BRANCH ─────────────────────────────────────────────
  1378→            0b1100011 => {
  1379→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1380→                let rs2_reg = Gpr::from_encoding(rs2)?;
  1381→                let imm12 = ((word >> 31) & 1) << 12
  1382→                    | ((word >> 7) & 1) << 11
  1383→                    | ((word >> 25) & 0x3F) << 5
  1384→                    | ((word >> 8) & 0xF) << 1;
  1385→                let offset = ((imm12 << 19) as i32) >> 19;
  1386→                match funct3 {
  1387→                    0b000 => Some(Instruction::Beq {
  1388→                        rs1: rs1_reg,
  1389→                        rs2: rs2_reg,
  1390→                        offset,
  1391→                    }),
  1392→                    0b001 => Some(Instruction::Bne {
  1393→                        rs1: rs1_reg,
  1394→                        rs2: rs2_reg,
  1395→                        offset,
  1396→                    }),
  1397→                    0b100 => Some(Instruction::Blt {
  1398→                        rs1: rs1_reg,
  1399→                        rs2: rs2_reg,
  1400→                        offset,
  1401→                    }),
  1402→                    0b101 => Some(Instruction::Bge {
  1403→                        rs1: rs1_reg,
  1404→                        rs2: rs2_reg,
  1405→                        offset,
  1406→                    }),
  1407→                    0b110 => Some(Instruction::Bltu {
  1408→                        rs1: rs1_reg,
  1409→                        rs2: rs2_reg,
  1410→                        offset,
  1411→                    }),
  1412→                    0b111 => Some(Instruction::Bgeu {
  1413→                        rs1: rs1_reg,
  1414→                        rs2: rs2_reg,
  1415→                        offset,
  1416→                    }),
  1417→                    _ => None,
  1418→                }
  1419→            }
  1420→
  1421→            // ── LOAD ───────────────────────────────────────────────
  1422→            0b0000011 => {
  1423→                let rd_reg = Gpr::from_encoding(rd)?;
  1424→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1425→                let imm = (((word >> 20) as i32) << 20) >> 20;
  1426→                match funct3 {
  1427→                    0b000 => Some(Instruction::Lb {
  1428→                        rd: rd_reg,
  1429→                        rs1: rs1_reg,
  1430→                        imm,
  1431→                    }),
  1432→                    0b001 => Some(Instruction::Lh {
  1433→                        rd: rd_reg,
  1434→                        rs1: rs1_reg,
  1435→                        imm,
  1436→                    }),
  1437→                    0b010 => Some(Instruction::Lw {
  1438→                        rd: rd_reg,
  1439→                        rs1: rs1_reg,
  1440→                        imm,
  1441→                    }),
  1442→                    0b011 => Some(Instruction::Ld {
  1443→                        rd: rd_reg,
  1444→                        rs1: rs1_reg,
  1445→                        imm,
  1446→                    }),
  1447→                    0b100 => Some(Instruction::Lbu {
  1448→                        rd: rd_reg,
  1449→                        rs1: rs1_reg,
  1450→                        imm,
  1451→                    }),
  1452→                    0b101 => Some(Instruction::Lhu {
  1453→                        rd: rd_reg,
  1454→                        rs1: rs1_reg,
  1455→                        imm,
  1456→                    }),
  1457→                    0b110 => Some(Instruction::Lwu {
  1458→                        rd: rd_reg,
  1459→                        rs1: rs1_reg,
  1460→                        imm,
  1461→                    }),
  1462→                    _ => None,
  1463→                }
  1464→            }
  1465→
  1466→            // ── STORE ──────────────────────────────────────────────
  1467→            0b0100011 => {
  1468→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1469→                let rs2_reg = Gpr::from_encoding(rs2)?;
  1470→                let imm_lo = (word >> 7) & 0x1F;
  1471→                let imm_hi = (word >> 25) & 0x7F;
  1472→                let imm_raw = (imm_hi << 5) | imm_lo;
  1473→                let imm = ((imm_raw as i32) << 20) >> 20;
  1474→                match funct3 {
  1475→                    0b000 => Some(Instruction::Sb {
  1476→                        rs1: rs1_reg,
  1477→                        rs2: rs2_reg,
  1478→                        imm,
  1479→                    }),
  1480→                    0b001 => Some(Instruction::Sh {
  1481→                        rs1: rs1_reg,
  1482→                        rs2: rs2_reg,
  1483→                        imm,
  1484→                    }),
  1485→                    0b010 => Some(Instruction::Sw {
  1486→                        rs1: rs1_reg,
  1487→                        rs2: rs2_reg,
  1488→                        imm,
  1489→                    }),
  1490→                    0b011 => Some(Instruction::Sd {
  1491→                        rs1: rs1_reg,
  1492→                        rs2: rs2_reg,
  1493→                        imm,
  1494→                    }),
  1495→                    _ => None,
  1496→                }
  1497→            }
  1498→
  1499→            // ── OP-IMM (RV64I) ─────────────────────────────────────
  1500→            0b0010011 => {
  1501→                let rd_reg = Gpr::from_encoding(rd)?;
  1502→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1503→                let imm = (((word >> 20) as i32) << 20) >> 20;
  1504→                let shamt = (word >> 20) & 0x3F;
  1505→                match funct3 {
  1506→                    0b000 => Some(Instruction::Addi {
  1507→                        rd: rd_reg,
  1508→                        rs1: rs1_reg,
  1509→                        imm,
  1510→                    }),
  1511→                    0b010 => Some(Instruction::Slti {
  1512→                        rd: rd_reg,
  1513→                        rs1: rs1_reg,
  1514→                        imm,
  1515→                    }),
  1516→                    0b011 => Some(Instruction::Sltiu {
  1517→                        rd: rd_reg,
  1518→                        rs1: rs1_reg,
  1519→                        imm,
  1520→                    }),
  1521→                    0b100 => Some(Instruction::Xori {
  1522→                        rd: rd_reg,
  1523→                        rs1: rs1_reg,
  1524→                        imm,
  1525→                    }),
  1526→                    0b110 => Some(Instruction::Ori {
  1527→                        rd: rd_reg,
  1528→                        rs1: rs1_reg,
  1529→                        imm,
  1530→                    }),
  1531→                    0b111 => Some(Instruction::Andi {
  1532→                        rd: rd_reg,
  1533→                        rs1: rs1_reg,
  1534→                        imm,
  1535→                    }),
  1536→                    0b001 => Some(Instruction::Slli {
  1537→                        rd: rd_reg,
  1538→                        rs1: rs1_reg,
  1539→                        shamt,
  1540→                    }),
  1541→                    0b101 => {
  1542→                        if funct7 == 0b0100000 {
  1543→                            Some(Instruction::Srai {
  1544→                                rd: rd_reg,
  1545→                                rs1: rs1_reg,
  1546→                                shamt,

  1548→                        } else {
  1549→                            Some(Instruction::Srli {
  1550→                                rd: rd_reg,
  1551→                                rs1: rs1_reg,
  1552→                                shamt,
  1553→                            })
  1554→                        }
  1555→                    }
  1556→                    _ => None,
  1557→                }
  1558→            }
  1559→
  1560→            // ── OP (RV64I register-register) ────────────────────────
  1561→            0b0110011 => {
  1562→                let rd_reg = Gpr::from_encoding(rd)?;
  1563→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1564→                let rs2_reg = Gpr::from_encoding(rs2)?;
  1565→                match (funct7, funct3) {
  1566→                    (0b0000000, 0b000) => Some(Instruction::Add {
  1567→                        rd: rd_reg,
  1568→                        rs1: rs1_reg,
  1569→                        rs2: rs2_reg,
  1570→                    }),
  1571→                    (0b0100000, 0b000) => Some(Instruction::Sub {
  1572→                        rd: rd_reg,
  1573→                        rs1: rs1_reg,
  1574→                        rs2: rs2_reg,
  1575→                    }),
  1576→                    (0b0000000, 0b001) => Some(Instruction::Sll {
  1577→                        rd: rd_reg,
  1578→                        rs1: rs1_reg,
  1579→                        rs2: rs2_reg,
  1580→                    }),
  1581→                    (0b0000000, 0b010) => Some(Instruction::Slt {
  1582→                        rd: rd_reg,
  1583→                        rs1: rs1_reg,
  1584→                        rs2: rs2_reg,
  1585→                    }),
  1586→                    (0b0000000, 0b011) => Some(Instruction::Sltu {
  1587→                        rd: rd_reg,
  1588→                        rs1: rs1_reg,
  1589→                        rs2: rs2_reg,
  1590→                    }),
  1591→                    (0b0000000, 0b100) => Some(Instruction::Xor {
  1592→                        rd: rd_reg,
  1593→                        rs1: rs1_reg,
  1594→                        rs2: rs2_reg,
  1595→                    }),
  1596→                    (0b0000000, 0b101) => Some(Instruction::Srl {
  1597→                        rd: rd_reg,
  1598→                        rs1: rs1_reg,
  1599→                        rs2: rs2_reg,
  1600→                    }),
  1601→                    (0b0100000, 0b101) => Some(Instruction::Sra {
  1602→                        rd: rd_reg,
  1603→                        rs1: rs1_reg,
  1604→                        rs2: rs2_reg,
  1605→                    }),
  1606→                    (0b0000000, 0b110) => Some(Instruction::Or {
  1607→                        rd: rd_reg,
  1608→                        rs1: rs1_reg,
  1609→                        rs2: rs2_reg,
  1610→                    }),
  1611→                    (0b0000000, 0b111) => Some(Instruction::And {
  1612→                        rd: rd_reg,
  1613→                        rs1: rs1_reg,
  1614→                        rs2: rs2_reg,
  1615→                    }),
  1616→                    // M extension
  1617→                    (0b0000001, 0b000) => Some(Instruction::Mul {
  1618→                        rd: rd_reg,
  1619→                        rs1: rs1_reg,
  1620→                        rs2: rs2_reg,
  1621→                    }),
  1622→                    (0b0000001, 0b001) => Some(Instruction::Mulh {
  1623→                        rd: rd_reg,
  1624→                        rs1: rs1_reg,
  1625→                        rs2: rs2_reg,
  1626→                    }),
  1627→                    (0b0000001, 0b010) => Some(Instruction::Mulhsu {
  1628→                        rd: rd_reg,
  1629→                        rs1: rs1_reg,
  1630→                        rs2: rs2_reg,
  1631→                    }),
  1632→                    (0b0000001, 0b011) => Some(Instruction::Mulhu {
  1633→                        rd: rd_reg,
  1634→                        rs1: rs1_reg,
  1635→                        rs2: rs2_reg,
  1636→                    }),
  1637→                    (0b0000001, 0b100) => Some(Instruction::Div {
  1638→                        rd: rd_reg,
  1639→                        rs1: rs1_reg,
  1640→                        rs2: rs2_reg,
  1641→                    }),
  1642→                    (0b0000001, 0b101) => Some(Instruction::Divu {
  1643→                        rd: rd_reg,
  1644→                        rs1: rs1_reg,
  1645→                        rs2: rs2_reg,
  1646→                    }),
  1647→                    (0b0000001, 0b110) => Some(Instruction::Rem {
  1648→                        rd: rd_reg,
  1649→                        rs1: rs1_reg,
  1650→                        rs2: rs2_reg,

  1658→                }
  1659→            }
  1660→

  1662→            0b0011011 => {
  1663→                let rd_reg = Gpr::from_encoding(rd)?;
  1664→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1665→                let imm = (((word >> 20) as i32) << 20) >> 20;
  1666→                let shamt = (word >> 20) & 0x1F;
  1667→                match funct3 {
  1668→                    0b000 => Some(Instruction::Addiw {
  1669→                        rd: rd_reg,
  1670→                        rs1: rs1_reg,
  1671→                        imm,
  1672→                    }),
  1673→                    0b001 => Some(Instruction::Slliw {
  1674→                        rd: rd_reg,
  1675→                        rs1: rs1_reg,
  1676→                        shamt,
  1677→                    }),
  1678→                    0b101 => {
  1679→                        if funct7 == 0b0100000 {
  1680→                            Some(Instruction::Sraiw {
  1681→                                rd: rd_reg,
  1682→                                rs1: rs1_reg,
  1683→                                shamt,
  1684→                            })
  1685→                        } else {
  1686→                            Some(Instruction::Srliw {
  1687→                                rd: rd_reg,
  1688→                                rs1: rs1_reg,
  1689→                                shamt,
  1690→                            })
  1691→                        }
  1692→                    }
  1693→                    _ => None,
  1694→                }
  1695→            }
  1696→
  1697→            // ── OP-32 (RV64) ───────────────────────────────────────
  1698→            0b0111011 => {
  1699→                let rd_reg = Gpr::from_encoding(rd)?;
  1700→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1701→                let rs2_reg = Gpr::from_encoding(rs2)?;
  1702→                match (funct7, funct3) {
  1703→                    (0b0000000, 0b000) => Some(Instruction::Addw {
  1704→                        rd: rd_reg,
  1705→                        rs1: rs1_reg,
  1706→                        rs2: rs2_reg,
  1707→                    }),
  1708→                    (0b0100000, 0b000) => Some(Instruction::Subw {
  1709→                        rd: rd_reg,
  1710→                        rs1: rs1_reg,
  1711→                        rs2: rs2_reg,
  1712→                    }),
  1713→                    (0b0000000, 0b001) => Some(Instruction::Sllw {
  1714→                        rd: rd_reg,
  1715→                        rs1: rs1_reg,
  1716→                        rs2: rs2_reg,
  1717→                    }),
  1718→                    (0b0000000, 0b101) => Some(Instruction::Srlw {
  1719→                        rd: rd_reg,
  1720→                        rs1: rs1_reg,
  1721→                        rs2: rs2_reg,
  1722→                    }),
  1723→                    (0b0100000, 0b101) => Some(Instruction::Sraw {
  1724→                        rd: rd_reg,
  1725→                        rs1: rs1_reg,

  1728→                    _ => None,
  1729→                }
  1730→            }
  1731→
  1732→            // ── SYSTEM ─────────────────────────────────────────────
  1733→            0b1110011 => {
  1734→                if word == 0x00000073 {
  1735→                    Some(Instruction::Ecall)
  1736→                } else if word == 0x00100073 {
  1737→                    Some(Instruction::Ebreak)
  1738→                } else {
  1739→                    let csr = (word >> 20) & 0xFFF;
  1740→                    let rd_reg = Gpr::from_encoding(rd)?;
  1741→                    let rs1_reg = Gpr::from_encoding(rs1)?;
  1742→                    match funct3 {
  1743→                        0b001 => Some(Instruction::Csrrw {
  1744→                            rd: rd_reg,
  1745→                            csr,
  1746→                            rs1: rs1_reg,
  1747→                        }),
  1748→                        0b010 => Some(Instruction::Csrrs {
  1749→                            rd: rd_reg,
  1750→                            csr,
  1751→                            rs1: rs1_reg,
  1752→                        }),
  1753→                        0b011 => Some(Instruction::Csrrc {
  1754→                            rd: rd_reg,
  1755→                            csr,
  1756→                            rs1: rs1_reg,
  1757→                        }),
  1758→                        0b101 => Some(Instruction::Csrrwi {
  1759→                            rd: rd_reg,
  1760→                            csr,
  1761→                            uimm: rs1,
  1762→                        }),
  1763→                        0b110 => Some(Instruction::Csrrsi {
  1764→                            rd: rd_reg,
  1765→                            csr,
  1766→                            uimm: rs1,
  1767→                        }),
  1768→                        0b111 => Some(Instruction::Csrrci {
  1769→                            rd: rd_reg,
  1770→                            csr,
  1771→                            uimm: rs1,
  1772→                        }),
  1773→                        _ => None,
  1774→                    }
  1775→                }
  1776→            }
  1777→
  1778→            // ── MISC-MEM (FENCE / FENCE.I) ─────────────────────────
  1779→            0b0001111 => {
  1780→                if funct3 == 0b001 {
  1781→                    Some(Instruction::FenceI)
  1782→                } else {
  1783→                    let imm = (word >> 20) & 0xFF;
  1784→                    let pred = (imm >> 4) & 0xF;
  1785→                    let succ = imm & 0xF;
  1786→                    Some(Instruction::Fence { pred, succ })
  1787→                }
  1788→            }
  1789→
  1790→            // ── FP (opcode=0x53) ───────────────────────────────────
  1791→            0b1010011 => {
  1792→                let rd_fpr = Fpr::from_encoding(rd)?;
  1793→                let rs1_fpr = Fpr::from_encoding(rs1)?;
  1794→                let rs2_fpr = Fpr::from_encoding(rs2)?;
  1795→                match (funct7, funct3) {
  1796→                    (0b0000001, 0b111) => Some(Instruction::FaddD {
  1797→                        rd: rd_fpr,
  1798→                        rs1: rs1_fpr,
  1799→                        rs2: rs2_fpr,
  1800→                    }),
  1801→                    (0b0000101, 0b111) => Some(Instruction::FsubD {
  1802→                        rd: rd_fpr,
  1803→                        rs1: rs1_fpr,
  1804→                        rs2: rs2_fpr,
  1805→                    }),
  1806→                    (0b0001001, 0b111) => Some(Instruction::FmulD {
  1807→                        rd: rd_fpr,
  1808→                        rs1: rs1_fpr,
  1809→                        rs2: rs2_fpr,
  1810→                    }),
  1811→                    (0b0001101, 0b111) => Some(Instruction::FdivD {
  1812→                        rd: rd_fpr,
  1813→                        rs1: rs1_fpr,
  1814→                        rs2: rs2_fpr,
  1815→                    }),
  1816→                    (0b0010001, 0b000) if rs1 == rs2 => Some(Instruction::FmvD {
  1817→                        rd: rd_fpr,
  1818→                        rs1: rs1_fpr,
  1819→                    }),
  1820→                    _ => None,
  1821→                }
  1822→            }
  1823→
  1824→            _ => None,
  1825→        }
  1826→    }

  1941→/// Build a minimal ELF64 binary for RISC-V 64-bit from raw code bytes.
  1942→///
  1943→/// Produces a static executable with a single LOAD segment containing the
  1944→/// `.text` section.  Uses EM_RISCV (243) as the machine type.
  1945→fn build_minimal_riscv64_elf(code: &[u8], base_addr: u64) -> Vec<u8> {
  1946→    // Layout: ELF header (64) | 1 program header (56) | code
