   160тЖТ   160тЖТ    /// Decode a single LoongArch64 instruction from 4 little-endian bytes.
   161тЖТ   161тЖТ    pub fn decode(bytes: &[u8]) -> Result<Self, DecodeError> {
   162тЖТ   162тЖТ        if bytes.len() < 4 {
   163тЖТ   163тЖТ            return Err(DecodeError::Truncated {
   164тЖТ   164тЖТ                needed: 4,
   165тЖТ   165тЖТ                available: bytes.len(),
   166тЖТ   166тЖТ            });
   167тЖТ   167тЖТ        }
   168тЖТ   168тЖТ        let word = u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]);
   169тЖТ   169тЖТ
   170тЖТ   170тЖТ        // Extract common fields
   171тЖТ   171тЖТ        let rd = word & 0x1F;
   172тЖТ   172тЖТ        let rj = (word >> 5) & 0x1F;
   173тЖТ   173тЖТ        let rk = (word >> 10) & 0x1F;
   174тЖТ   174тЖТ
   175тЖТ   175тЖТ        // 3R format: opcode in bits [31:15]
   176тЖТ   176тЖТ        let opc_3r = (word >> 15) & 0x1FFFF;
   177тЖТ   177тЖТ
   178тЖТ   178тЖТ        // 2RI12 format: opcode in bits [31:22]
   179тЖТ   179тЖТ        let opc_2ri12 = (word >> 22) & 0x3FF;
   180тЖТ   180тЖТ        let imm12_raw = (word >> 10) & 0xFFF;
   181тЖТ   181тЖТ
   182тЖТ   182тЖТ        // 2RI16 format: opcode in bits [31:26]
   183тЖТ   183тЖТ        let opc_2ri16 = (word >> 26) & 0x3F;
   184тЖТ   184тЖТ        let imm16_raw = (word >> 10) & 0xFFFF;
   185тЖТ   185тЖТ
   186тЖТ   186тЖТ        // 1RI21 format: opcode in bits [31:26]
   187тЖТ   187тЖТ        let opc_1ri21 = (word >> 26) & 0x3F;
   188тЖТ   188тЖТ
   189тЖТ   189тЖТ        // I26 format: opcode in bits [31:26]
   190тЖТ   190тЖТ        let opc_i26 = (word >> 26) & 0x3F;
   191тЖТ   191тЖТ
   192тЖТ   192тЖТ        // 2RI8 format: opcode in bits [31:22]
   193тЖТ   193тЖТ        let _opc_2ri8 = (word >> 22) & 0x3FF;
   194тЖТ   194тЖТ        let imm8_raw = (word >> 14) & 0xFF;
   195тЖТ   195тЖТ
   196тЖТ   196тЖТ        // 2R format: opcode in bits [31:10]

   716тЖТ   716тЖТ                return Ok(Instruction::B {
   717тЖТ   717тЖТ                    offs26: offs26 as i32,
   718тЖТ   718тЖТ                })
   719тЖТ   719тЖТ            }
   720тЖТ   720тЖТ            0x15 => {
   721тЖТ   721тЖТ                return Ok(Instruction::Bl {
   722тЖТ   722тЖТ                    offs26: offs26 as i32,
   723тЖТ   723тЖТ                })
   724тЖТ   724тЖТ            }
   725тЖТ   725тЖТ            _ => {}
   726тЖТ   726тЖТ        }
   727тЖТ   727тЖТ
   728тЖТ   728тЖТ        // 1RI21 format (beqz/bnez)
   729тЖТ   729тЖТ        let imm21_hi5 = (word >> 21) & 0x1F;
   730тЖТ   730тЖТ        let imm21_lo16 = (word >> 5) & 0xFFFF;
   731тЖТ   731тЖТ        let imm21 = (imm21_hi5 << 16) | imm21_lo16;
   732тЖТ   732тЖТ        match opc_1ri21 {
   733тЖТ   733тЖТ            0x1C => {
   734тЖТ   734тЖТ                return Ok(Instruction::Beqz {
   735тЖТ   735тЖТ                    rj: gpr_from_bits(rj),
   736тЖТ   736тЖТ                    offs21: sign_extend_21(imm21),
   737тЖТ   737тЖТ                })
   738тЖТ   738тЖТ            }
   739тЖТ   739тЖТ            0x1D => {
   740тЖТ   740тЖТ                return Ok(Instruction::Bnez {
   741тЖТ   741тЖТ                    rj: gpr_from_bits(rj),
   742тЖТ   742тЖТ                    offs21: sign_extend_21(imm21),
   743тЖТ   743тЖТ                })
   744тЖТ   744тЖТ            }
   745тЖТ   745тЖТ            _ => {}
   746тЖТ   746тЖТ        }
   747тЖТ   747тЖТ
   748тЖТ   748тЖТ        // 2R format (ext.w.h, ext.w.b)
   749тЖТ   749тЖТ        match opc_2r {
   750тЖТ   750тЖТ            0x000005A => {
   751тЖТ   751тЖТ                return Ok(Instruction::ExtWH {
   752тЖТ   752тЖТ                    rd: gpr_from_bits(rd),
   753тЖТ   753тЖТ                    rj: gpr_from_bits(rj),
   754тЖТ   754тЖТ                })
   755тЖТ   755тЖТ            }
   756тЖТ   756тЖТ            0x000005B => {
   757тЖТ   757тЖТ                return Ok(Instruction::ExtWB {
   758тЖТ   758тЖТ                    rd: gpr_from_bits(rd),
   759тЖТ   759тЖТ                    rj: gpr_from_bits(rj),
   760тЖТ   760тЖТ                })
   761тЖТ   761тЖТ            }
   762тЖТ   762тЖТ            // FP Move: fmov.s fd, fj
   763тЖТ   763тЖТ            0x000004E => {
   764тЖТ   764тЖТ                return Ok(Instruction::FmovS {
   765тЖТ   765тЖТ                    fd: fpr_from_bits(rd),

   770тЖТ   770тЖТ            0x000004F => {
   771тЖТ   771тЖТ                return Ok(Instruction::FmovD {
   772тЖТ   772тЖТ                    fd: fpr_from_bits(rd),
   773тЖТ   773тЖТ                    fj: fpr_from_bits(rj),
   774тЖТ   774тЖТ                })
   775тЖТ   775тЖТ            }
   776тЖТ   776тЖТ            // FP Move: movfr2gr.d rd, fj
   777тЖТ   777тЖТ            0x0000052 => {
   778тЖТ   778тЖТ                return Ok(Instruction::FmovGr2FprD {
   779тЖТ   779тЖТ                    rd: gpr_from_bits(rd),

   789тЖТ   789тЖТ            }
   790тЖТ   790тЖТ            _ => {}
   791тЖТ   791тЖТ        }
   792тЖТ   792тЖТ
   793тЖТ   793тЖТ        // 4R format (FP Compare: fcmp.cond.s/d)
   794тЖТ   794тЖТ        let opc_4r = (word >> 20) & 0xFFF;
   795тЖТ   795тЖТ        let cond = ((word >> 15) & 0x1F) as u8;

   797тЖТ   797тЖТ        match opc_4r {

   803тЖТ   803тЖТ                    cd,
   804тЖТ   804тЖТ                })
   805тЖТ   805тЖТ            }
   806тЖТ   806тЖТ            0x0C5 => {
   807тЖТ   807тЖТ                return Ok(Instruction::FCmpD {
   808тЖТ   808тЖТ                    cond,
   809тЖТ   809тЖТ                    fj: fpr_from_bits(rj),
   810тЖТ   810тЖТ                    fk: fpr_from_bits(rk),
   811тЖТ   811тЖТ                    cd,
   812тЖТ   812тЖТ                })
   813тЖТ   813тЖТ            }
   814тЖТ   814тЖТ            _ => {}
