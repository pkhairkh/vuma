     1→     1→//! # VUMA REPL — Interactive Read-Eval-Print Loop
     2→     2→//!
     3→     3→//! The [`VumaRepl`] struct provides an interactive shell for the VUMA language.
     4→     4→//! It parses VUMA expressions, builds the Semantic Computation Graph (SCG),
     5→     5→//! converts it to a Memory State Graph (MSG), and runs IVE verification.
     6→     6→//!
     7→     7→//! ## Interactive Commands
     8→     8→//!
     9→     9→//! | Command          | Description                                        |
    10→    10→//! |------------------|----------------------------------------------------|
    11→    11→//! | `:help`          | Show available commands                             |
    12→    12→//! | `:load <file>`   | Load and evaluate a VUMA source file               |
    13→    13→//! | `:verify`        | Run IVE verification on the current SCG            |
    14→    14→//! | `:show scg`      | Display the current SCG summary                    |
    15→    15→//! | `:show msg`      | Display the current MSG summary                    |
    16→    16→//! | `:show bd`       | Display behavioural descriptors for all nodes      |
    17→    17→//! | `:compile`       | Run the full pipeline: parse → SCG → MSG → verify  |
    18→    18→//! | `:profile`       | Show profiling data from the last verification     |
    19→    19→//! | `:quit`          | Exit the REPL                                      |
    20→    20→//!
    21→    21→//! ## Expression Evaluation
    22→    22→//!
    23→    23→//! Simple arithmetic and literal expressions are evaluated immediately.
    24→    24→//! Definitions (`let x = …`, `fn f() { … }`, etc.) persist across inputs
    25→    25→//! so that subsequent expressions can reference previously-defined names.
    26→    26→//!
    27→    27→//! ## Error Display
    28→    28→//!
    29→    29→//! Parse errors and runtime errors are displayed with source context,
    30→    30→//! showing the offending line and a caret pointing to the error location.
    31→    31→//!
    32→    32→//! ## History
    33→    33→//!
    34→    34→//! The REPL maintains an in-memory history buffer. Use `:history` to list
    35→    35→//! previous inputs. Arrow-key navigation (up/down) is supported in the
    36→    36→//! interactive loop.

    58→    58→/// Errors that can occur during REPL operation.
    59→    59→#[derive(Debug)]
    60→    60→pub enum ReplError {
    61→    61→    /// A parse error from the VUMA frontend.
    62→    62→    Parse(ParseError),
    63→    63→    /// An SCG construction error.
    64→    64→    ScgConstruction(String),
    65→    65→    /// An MSG conversion error.
    66→    66→    MsgConversion(scg_to_msg::ConversionError),
    67→    67→    /// An I/O error.
    68→    68→    Io(io::Error),
    69→    69→    /// A general REPL error with a message.
    70→    70→    General(String),
    71→    71→    /// Multiple parse errors.
    72→    72→    ParseErrors(Vec<ParseError>),
    73→    73→}
    74→    74→
    75→    75→impl fmt::Display for ReplError {
    76→    76→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
    77→    77→        match self {
    78→    78→            ReplError::Parse(e) => write!(f, "parse error: {e}"),
    79→    79→            ReplError::ScgConstruction(msg) => write!(f, "SCG construction error: {msg}"),
    80→    80→            ReplError::MsgConversion(e) => write!(f, "MSG conversion error: {e}"),
    81→    81→            ReplError::Io(e) => write!(f, "I/O error: {e}"),
    82→    82→            ReplError::General(msg) => write!(f, "{msg}"),
    83→    83→            ReplError::ParseErrors(errors) => {
    84→    84→                write!(f, "{} parse error(s):", errors.len())?;
    85→    85→                for e in errors {
    86→    86→                    write!(f, "\n  {e}")?;
    87→    87→                }
    88→    88→                Ok(())
    89→    89→            }
    90→    90→        }
    91→    91→    }
    92→    92→}
    93→    93→

   102→   102→    }
   103→   103→}

   127→   127→/// The result of processing a single REPL input.
   128→   128→#[derive(Debug)]
   129→   129→pub enum ReplResult {
   130→   130→    /// The input was processed successfully; optional output text.
   131→   131→    Ok(Option<String>),
   132→   132→    /// The input resulted in a value that can be displayed.
   133→   133→    Value(String),
   134→   134→    /// A verification result.
   135→   135→    Verification(AggregatedResult),
   136→   136→    /// The user requested to quit.
   137→   137→    Quit,
   138→   138→}

   140→   140→impl fmt::Display for ReplResult {
   141→   141→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   142→   142→        match self {
   143→   143→            ReplResult::Ok(Some(msg)) => write!(f, "{msg}"),
   144→   144→            ReplResult::Ok(None) => Ok(()),
   145→   145→            ReplResult::Value(v) => write!(f, "{v}"),
   146→   146→            ReplResult::Verification(r) => {
   147→   147→                let report = DiagnosticsReport::from_aggregated(r);
   148→   148→                write!(f, "{report}")
   149→   149→            }
   150→   150→            ReplResult::Quit => write!(f, "Goodbye."),
   151→   151→        }
   152→   152→    }
   153→   153→}

   200→   200→/// A simple immediate evaluator for arithmetic expressions.
   201→   201→///
   202→   202→/// This evaluator can compute integer arithmetic expressions without
   203→   203→/// needing to go through the full SCG pipeline. It supports:
   204→   204→/// - Integer literals
   205→   205→/// - Variable references (from the REPL's definition map)
   206→   206→/// - Basic arithmetic: +, -, *, /
   207→   207→/// - Parenthesised expressions
   208→   208→struct SimpleEvaluator {
   209→   209→    /// Variable bindings from previous definitions.
   210→   210→    vars: HashMap<String, i64>,
   211→   211→}
   212→   212→
   213→   213→impl SimpleEvaluator {
   214→   214→    fn new(vars: HashMap<String, i64>) -> Self {
   215→   215→        Self { vars }
   216→   216→    }
   217→   217→
   218→   218→    /// Attempt to evaluate a simple expression string.
   219→   219→    /// Returns `None` if the expression is too complex for this evaluator.
   220→   220→    fn eval(&self, input: &str) -> Option<i64> {
   221→   221→        let trimmed = input.trim();
   222→   222→        // Remove trailing semicolons.
   223→   223→        let trimmed = trimmed.trim_end_matches(';').trim();
   224→   224→        let tokens = self.tokenize(trimmed)?;
   225→   225→        self.eval_tokens(&tokens)
   226→   226→    }
   227→   227→
   228→   228→    /// Tokenize the input into a simple token stream.
   229→   229→    fn tokenize(&self, input: &str) -> Option<Vec<Token>> {
   230→   230→        let mut tokens = Vec::new();
   231→   231→        let mut chars = input.chars().peekable();
   232→   232→        while let Some(&ch) = chars.peek() {
   233→   233→            match ch {
   234→   234→                ' ' | '\t' => {
   235→   235→                    chars.next();
   236→   236→                }
   237→   237→                '(' => {
   238→   238→                    tokens.push(Token::LParen);
   239→   239→                    chars.next();
   240→   240→                }
   241→   241→                ')' => {
   242→   242→                    tokens.push(Token::RParen);
   243→   243→                    chars.next();
   244→   244→                }
   245→   245→                '+' => {
   246→   246→                    tokens.push(Token::Plus);
   247→   247→                    chars.next();
   248→   248→                }
   249→   249→                '-' => {
   250→   250→                    tokens.push(Token::Minus);
   251→   251→                    chars.next();
   252→   252→                }
   253→   253→                '*' => {
   254→   254→                    tokens.push(Token::Star);
   255→   255→                    chars.next();
   256→   256→                }
   257→   257→                '/' => {
   258→   258→                    tokens.push(Token::Slash);
   259→   259→                    chars.next();
   260→   260→                }
   261→   261→                '0'..='9' => {
   262→   262→                    let mut num = String::new();
   263→   263→                    while let Some(&d) = chars.peek() {
   264→   264→                        if d.is_ascii_digit() {
   265→   265→                            num.push(d);
   266→   266→                            chars.next();
   267→   267→                        } else {
   268→   268→                            break;
   269→   269→                        }
   270→   270→                    }
   271→   271→                    tokens.push(Token::Num(num.parse::<i64>().ok()?));
   272→   272→                }
   273→   273→                _ if ch.is_alphanumeric() || ch == '_' => {
   274→   274→                    let mut name = String::new();
   275→   275→                    while let Some(&c) = chars.peek() {
   276→   276→                        if c.is_alphanumeric() || c == '_' {
   277→   277→                            name.push(c);
   278→   278→                            chars.next();
   279→   279→                        } else {
   280→   280→                            break;
   281→   281→                        }
   282→   282→                    }
   283→   283→                    if let Some(&val) = self.vars.get(&name) {
   284→   284→                        tokens.push(Token::Num(val));
   285→   285→                    } else {
   286→   286→                        return None; // unknown variable
   287→   287→                    }
   288→   288→                }
   289→   289→                _ => return None, // unsupported character
   290→   290→            }
   291→   291→        }
   292→   292→        Some(tokens)
   293→   293→    }
   294→   294→
   295→   295→    /// Evaluate a token stream using recursive descent.
   296→   296→    fn eval_tokens(&self, tokens: &[Token]) -> Option<i64> {
   297→   297→        let mut pos = 0;
   298→   298→        let result = self.parse_additive(tokens, &mut pos)?;
   299→   299→        if pos == tokens.len() {
   300→   300→            Some(result)
   301→   301→        } else {
   302→   302→            None
   303→   303→        }
   304→   304→    }
   305→   305→
   306→   306→    fn parse_additive(&self, tokens: &[Token], pos: &mut usize) -> Option<i64> {
   307→   307→        let mut left = self.parse_multiplicative(tokens, pos)?;
   308→   308→        while *pos < tokens.len() {
   309→   309→            match &tokens[*pos] {
   310→   310→                Token::Plus => {
   311→   311→                    *pos += 1;
   312→   312→                    let right = self.parse_multiplicative(tokens, pos)?;
   313→   313→                    left += right;
   314→   314→                }
   315→   315→                Token::Minus => {
   316→   316→                    *pos += 1;
   317→   317→                    let right = self.parse_multiplicative(tokens, pos)?;
   318→   318→                    left -= right;
   319→   319→                }
   320→   320→                _ => break,
   321→   321→            }
   322→   322→        }
   323→   323→        Some(left)
   324→   324→    }
   325→   325→
   326→   326→    fn parse_multiplicative(&self, tokens: &[Token], pos: &mut usize) -> Option<i64> {
   327→   327→        let mut left = self.parse_unary(tokens, pos)?;
   328→   328→        while *pos < tokens.len() {
   329→   329→            match &tokens[*pos] {
   330→   330→                Token::Star => {
   331→   331→                    *pos += 1;
   332→   332→                    let right = self.parse_unary(tokens, pos)?;
   333→   333→                    left *= right;
   334→   334→                }
   335→   335→                Token::Slash => {
   336→   336→                    *pos += 1;
   337→   337→                    let right = self.parse_unary(tokens, pos)?;
   338→   338→                    if right == 0 {
   339→   339→                        return None;
   340→   340→                    }
   341→   341→                    left /= right;
   342→   342→                }
   343→   343→                _ => break,
   344→   344→            }
   345→   345→        }
   346→   346→        Some(left)
   347→   347→    }
   348→   348→
   349→   349→    fn parse_unary(&self, tokens: &[Token], pos: &mut usize) -> Option<i64> {
   350→   350→        if *pos < tokens.len() {
   351→   351→            match &tokens[*pos] {
   352→   352→                Token::Minus => {
   353→   353→                    *pos += 1;
   354→   354→                    let val = self.parse_primary(tokens, pos)?;
   355→   355→                    Some(-val)
   356→   356→                }
   357→   357→                Token::Plus => {
   358→   358→                    *pos += 1;
   359→   359→                    self.parse_primary(tokens, pos)
   360→   360→                }
   361→   361→                _ => self.parse_primary(tokens, pos),
   362→   362→            }
   363→   363→        } else {
   364→   364→            None
   365→   365→        }
   366→   366→    }
   367→   367→
   368→   368→    fn parse_primary(&self, tokens: &[Token], pos: &mut usize) -> Option<i64> {
   369→   369→        if *pos >= tokens.len() {
   370→   370→            return None;
   371→   371→        }
   372→   372→        match &tokens[*pos] {
   373→   373→            Token::Num(n) => {
   374→   374→                *pos += 1;
   375→   375→                Some(*n)
   376→   376→            }
   377→   377→            Token::LParen => {
   378→   378→                *pos += 1;
   379→   379→                let result = self.parse_additive(tokens, pos)?;
   380→   380→                if *pos < tokens.len() && matches!(&tokens[*pos], Token::RParen) {
   381→   381→                    *pos += 1;
   382→   382→                    Some(result)
   383→   383→                } else {
   384→   384→                    None
   385→   385→                }
   386→   386→            }
   387→   387→            _ => None,
   388→   388→        }
   389→   389→    }
   390→   390→}
   391→   391→
   392→   392→/// Simple token type for the arithmetic evaluator.
   393→   393→#[derive(Debug)]
   394→   394→enum Token {
   395→   395→    Num(i64),
   396→   396→    Plus,
   397→   397→    Minus,
   398→   398→    Star,
   399→   399→    Slash,
   400→   400→    LParen,
   401→   401→    RParen,
   402→   402→}

   408→   408→/// Format a parse error with source context.
   409→   409→fn format_error_with_context(source: &str, span: &Span, message: &str) -> String {
   410→   410→    let loc = offset_to_location(source, span.start, None);
   411→   411→    let line_start = source[..span.start].rfind('\n').map(|i| i + 1).unwrap_or(0);
   412→   412→    let line_end = source[span.start..]
   413→   413→        .find('\n')
   414→   414→        .map(|i| span.start + i)
   415→   415→        .unwrap_or(source.len());
   416→   416→    let line_text = &source[line_start..line_end];
   417→   417→
   418→   418→    let column = span.start - line_start;
   419→   419→    let caret_width = if span.end > span.start {
   420→   420→        (span.end - span.start).min(line_end - span.start)
   421→   421→    } else {
   422→   422→        1
   423→   423→    };
   424→   424→
   425→   425→    let mut result = String::new();
   426→   426→    result.push_str(&format!("error: {message}\n"));
   427→   427→    result.push_str(&format!(
   428→   428→        "  --> {}:{}:{}\n",
   429→   429→        loc.line, loc.column, span.start
   430→   430→    ));
   431→   431→    result.push_str("   |\n");
   432→   432→    result.push_str(&format!("{:3}| {}\n", loc.line, line_text));
   433→   433→    result.push_str("   | ");
   434→   434→    result.push_str(&" ".repeat(column));
   435→   435→    result.push_str(&"^".repeat(caret_width));
   436→   436→    result.push('\n');
   437→   437→    result
   438→   438→}

   444→   444→/// Generate a human-readable summary of the SCG.
   445→   445→fn format_scg_summary(scg: &SCG) -> String {
   446→   446→    let mut type_counts: HashMap<String, usize> = HashMap::new();
   447→   447→    let mut edge_counts: HashMap<String, usize> = HashMap::new();
   448→   448→
   449→   449→    for node in scg.nodes() {
   450→   450→        let label = format!("{:?}", node.node_type);
   451→   451→        *type_counts.entry(label).or_insert(0) += 1;
   452→   452→    }
   453→   453→
   454→   454→    for edge in scg.edges() {
   455→   455→        let label = format!("{:?}", edge.kind);
   456→   456→        *edge_counts.entry(label).or_insert(0) += 1;
   457→   457→    }
   458→   458→
   459→   459→    let mut result = String::new();
   460→   460→    result.push_str(&format!(
   461→   461→        "SCG Summary ({} nodes, {} edges, {} regions)\n",
   462→   462→        scg.node_count(),
   463→   463→        scg.edge_count(),
   464→   464→        scg.region_count()
   465→   465→    ));
   466→   466→
   467→   467→    if !type_counts.is_empty() {
   468→   468→        result.push_str("  Node types:\n");
   469→   469→        let mut entries: Vec<_> = type_counts.into_iter().collect();
   470→   470→        entries.sort_by_key(|b| std::cmp::Reverse(b.1));
   471→   471→        for (kind, count) in entries {
   472→   472→            result.push_str(&format!("    {kind}: {count}\n"));
   473→   473→        }
   474→   474→    }
   475→   475→
   476→   476→    if !edge_counts.is_empty() {
   477→   477→        result.push_str("  Edge types:\n");
   478→   478→        let mut entries: Vec<_> = edge_counts.into_iter().collect();
   479→   479→        entries.sort_by_key(|b| std::cmp::Reverse(b.1));
   480→   480→        for (kind, count) in entries {
   481→   481→            result.push_str(&format!("    {kind}: {count}\n"));
   482→   482→        }
   483→   483→    }
   484→   484→
   485→   485→    // List regions.
   486→   486→    for region in scg.regions() {
   487→   487→        result.push_str(&format!(
   488→   488→            "  Region {} (deployment: {}, nodes: {}, scope: {})\n",
   489→   489→            region.id,
   490→   490→            region.deployment_target,
   491→   491→            region.node_count(),
   492→   492→            region.scope_level,
   493→   493→        ));
   494→   494→    }
   495→   495→
   496→   496→    result
   497→   497→}

   499→   499→/// Generate a BD (Behavioural Descriptor) display for all nodes in the SCG.
   500→   500→fn format_bd_display(scg: &SCG, inference_engine: &InferenceEngine) -> String {
   501→   501→    let mut result = String::new();
   502→   502→    result.push_str("Behavioural Descriptors:\n");
   503→   503→
   504→   504→    if scg.node_count() == 0 {
   505→   505→        result.push_str("  (no nodes)\n");
   506→   506→        return result;
   507→   507→    }
   508→   508→
   509→   509→    for node in scg.nodes() {
   510→   510→        let bd = inference_engine.infer_bd(scg, node.id).unwrap_or_else(|_| {
   511→   511→            vuma_bd::descriptor::BD::new(
   512→   512→                vuma_bd::repd::RepD::Byte(vuma_bd::repd::ByteRep { size: 0, align: 0 }),
   513→   513→                vuma_bd::capd::CapD::empty(),
   514→   514→                vuma_bd::reld::RelD::empty(),
   515→   515→            )
   516→   516→        });
   517→   517→        result.push_str(&format!(
   518→   518→            "  Node {} ({:?}): {}\n",
   519→   519→            node.id, node.node_type, bd
   520→   520→        ));
   521→   521→    }
   522→   522→
   523→   523→    result
   524→   524→}
   525→   525→
   526→   526→// ---------------------------------------------------------------------------
   527→   527→// VumaRepl
   528→   528→// ---------------------------------------------------------------------------
   529→   529→
   530→   530→/// The VUMA Read-Eval-Print Loop.
   531→   531→///
   532→   532→/// `VumaRepl` maintains incremental state across expressions: previous
   533→   533→/// definitions are accumulated in a source buffer and re-parsed as a whole
   534→   534→/// so that later expressions can reference earlier bindings.
   535→   535→///
   536→   536→/// # Usage
   537→   537→///
   538→   538→/// ```rust
   539→   539→/// use vuma_core::repl::VumaRepl;
   540→   540→///
   541→   541→/// let mut repl = VumaRepl::new();
   542→   542→///
   543→   543→/// // Process a single line programmatically.
   544→   544→/// let result = repl.process_line("let x = 42;");
   545→   545→/// match result {
   546→   546→///     Ok(output) => println!("{output}"),
   547→   547→///     Err(e) => eprintln!("Error: {e}"),
   548→   548→/// }
   549→   549→///
   550→   550→/// // Run the interactive loop (reads from stdin).
   551→   551→/// // repl.run();
   552→   552→/// ```
   553→   553→pub struct VumaRepl {
   554→   554→    /// Accumulated source text (all previous definitions + current expression).
   555→   555→    source_buffer: String,
   556→   556→    /// The current SCG built from the accumulated source.
   557→   557→    scg: SCG,
   558→   558→    /// The current MSG converted from the SCG.
   559→   559→    msg: Option<MSG>,
   560→   560→    /// The AST-to-SCG converter (maintains variable scopes).
   561→   561→    converter: AstToScg,
   562→   562→    /// IVE inference engine.
   563→   563→    inference_engine: InferenceEngine,
   564→   564→    /// IVE verification engine (reserved for individual invariant checks).
   565→   565→    _verification_engine: VerificationEngine,
   566→   566→    /// IVE invariant aggregator.
   567→   567→    aggregator: InvariantAggregator,
   568→   568→    /// Command input history.
   569→   569→    history: Vec<String>,
   570→   570→    /// Current position in history for up/down navigation.
   571→   571→    history_cursor: usize,
   572→   572→    /// Profiling data.
   573→   573→    profile: ReplProfile,
   574→   574→    /// Whether the REPL should keep running.
   575→   575→    running: bool,
   576→   576→    /// Last verification result.
   577→   577→    last_verification: Option<AggregatedResult>,
   578→   578→    /// Variable values for simple expression evaluation.
   579→   579→    simple_vars: HashMap<String, i64>,
   580→   580→    /// Currently loaded file path (if any).
   581→   581→    loaded_file: Option<String>,
   582→   582→}
   583→   583→
   584→   584→impl VumaRepl {
   585→   585→    /// Create a new REPL instance.
   586→   586→    pub fn new() -> Self {
   587→   587→        Self {
   588→   588→            source_buffer: String::new(),

   592→   592→            inference_engine: InferenceEngine::new(),
   593→   593→            _verification_engine: VerificationEngine::new(),
   594→   594→            aggregator: InvariantAggregator::new(),
   595→   595→            history: Vec::new(),
   596→   596→            history_cursor: 0,
   597→   597→            profile: ReplProfile::default(),
   598→   598→            running: true,
   599→   599→            last_verification: None,
   600→   600→            simple_vars: HashMap::new(),
   601→   601→            loaded_file: None,
   602→   602→        }
   603→   603→    }
   604→   604→

   606→   606→    pub fn with_verbose() -> Self {
   607→   607→        Self {
   608→   608→            inference_engine: InferenceEngine::new().with_verbose(true),
   609→   609→            _verification_engine: VerificationEngine::new().with_verbose(true),

   612→   612→        }

   615→   615→    // -----------------------------------------------------------------------
   616→   616→    // Core: process a line of input
   617→   617→    // -----------------------------------------------------------------------
   618→   618→
   619→   619→    /// Process a single line of input and return the result.
   620→   620→    ///
   621→   621→    /// This is the primary API for programmatic use. The line may be a
   622→   622→    /// VUMA expression/statement or a REPL command prefixed with `:`.
   623→   623→    pub fn process_line(&mut self, line: &str) -> Result<ReplResult, ReplError> {
   624→   624→        let trimmed = line.trim();
   625→   625→
   626→   626→        // Record in history.
   627→   627→        if !trimmed.is_empty() {
   628→   628→            self.history.push(trimmed.to_string());
   629→   629→            self.history_cursor = self.history.len();
   630→   630→        }
   631→   631→
   632→   632→        // Handle empty input.
   633→   633→        if trimmed.is_empty() {
   634→   634→            return Ok(ReplResult::Ok(None));
   635→   635→        }
   636→   636→
   637→   637→        // Handle REPL commands.
   638→   638→        if trimmed.starts_with(':') {
   639→   639→            return self.handle_command(trimmed);
   640→   640→        }
   641→   641→
   642→   642→        // Try simple expression evaluation first.
   643→   643→        let evaluator = SimpleEvaluator::new(self.simple_vars.clone());
   644→   644→        if let Some(value) = evaluator.eval(trimmed) {
   645→   645→            self.profile.expressions_processed += 1;
   646→   646→            return Ok(ReplResult::Value(value.to_string()));
   647→   647→        }
   648→   648→
   649→   649→        // Fall through to full VUMA parsing.
   650→   650→        self.evaluate_vuma(trimmed)
   651→   651→    }
   652→   652→
   653→   653→    // -----------------------------------------------------------------------
   654→   654→    // VUMA expression evaluation
   655→   655→    // -----------------------------------------------------------------------
   656→   656→
   657→   657→    /// Evaluate a VUMA expression/statement through the full pipeline.
   658→   658→    fn evaluate_vuma(&mut self, input: &str) -> Result<ReplResult, ReplError> {
   659→   659→        // Append the input to the source buffer.
   660→   660→        let prev_len = self.source_buffer.len();
   661→   661→        if !self.source_buffer.is_empty() && !self.source_buffer.ends_with('\n') {
   662→   662→            self.source_buffer.push('\n');
   663→   663→        }
   664→   664→        self.source_buffer.push_str(input);
   665→   665→        if !input.ends_with(';') && !input.ends_with('}') {
   666→   666→            self.source_buffer.push(';');
   667→   667→        }
   668→   668→
   669→   669→        // Parse.
   670→   670→        let parse_start = Instant::now();
   671→   671→        let mut parser = Parser::new(&self.source_buffer);
   672→   672→        let result = parser.parse_program();
   673→   673→        if result.has_errors() {
   674→   674→            // Roll back the source buffer on parse error.
   675→   675→            let errors = result.errors.clone();
   676→   676→            self.source_buffer.truncate(prev_len);
   677→   677→            self.profile.parse_errors += errors.len();
   678→   678→            self.profile.parse_time_ms += parse_start.elapsed().as_millis() as u64;
   679→   679→            return Err(ReplError::ParseErrors(errors));
   680→   680→        }
   681→   681→        let program = result.unwrap();
   682→   682→        self.profile.parse_time_ms += parse_start.elapsed().as_millis() as u64;
   683→   683→
   684→   684→        // Build SCG.
   685→   685→        let scg_start = Instant::now();
   686→   686→        // We rebuild the SCG from scratch each time to keep it consistent
   687→   687→        // with the full source buffer. In a production implementation we
   688→   688→        // would use incremental SCG updates.
   689→   689→        let mut converter = AstToScg::new();
   690→   690→        let scg = match converter.convert(&program) {
   691→   691→            Ok(s) => s,
   692→   692→            Err(e) => {
   693→   693→                self.source_buffer.truncate(prev_len);
   694→   694→                self.profile.scg_time_ms += scg_start.elapsed().as_millis() as u64;
   695→   695→                return Err(ReplError::Parse(e));
   696→   696→            }
   697→   697→        };
   698→   698→        self.scg = scg;
   699→   699→        self.converter = converter;
   700→   700→        self.profile.scg_time_ms += scg_start.elapsed().as_millis() as u64;
   701→   701→
   702→   702→        // Convert to MSG.
   703→   703→        let msg_start = Instant::now();
   704→   704→        match scg_to_msg::scg_to_msg(&self.scg) {
   705→   705→            Ok(msg) => self.msg = Some(msg),
   706→   706→            Err(e) => {
   707→   707→                // MSG conversion failure is non-fatal for the REPL; we just
   708→   708→                // don't have an MSG available.
   709→   709→                log::warn!("MSG conversion failed: {e}");
   710→   710→                self.msg = None;
   711→   711→            }
   712→   712→        }
   713→   713→        self.profile.msg_time_ms += msg_start.elapsed().as_millis() as u64;
   714→   714→
   715→   715→        // Extract simple variable bindings for immediate evaluation.
   716→   716→        self.extract_simple_bindings(&program);
   717→   717→
   718→   718→        self.profile.expressions_processed += 1;
   719→   719→
   720→   720→        // Build a short description of what was added.
   721→   721→        let node_count = self.scg.node_count();
   722→   722→        let edge_count = self.scg.edge_count();
   723→   723→        Ok(ReplResult::Ok(Some(format!(
   724→   724→            "OK (SCG: {node_count} nodes, {edge_count} edges)"
   725→   725→        ))))
   726→   726→    }
   727→   727→
   728→   728→    /// Extract simple integer bindings from let statements for immediate eval.
   729→   729→    fn extract_simple_bindings(&mut self, program: &vuma_parser::ast::Program) {
   730→   730→        for item in &program.items {
   731→   731→            match item {
   732→   732→                Item::Stmt(Stmt::Let(l)) => {
   733→   733→                    if let Expr::Lit {
   734→   734→                        value: Lit::Int(n), ..
   735→   735→                    } = &l.value
   736→   736→                    {
   737→   737→                        self.simple_vars.insert(l.name.clone(), *n);
   738→   738→                    }
   739→   739→                }
   740→   740→                Item::Const(c) => {
   741→   741→                    if let Expr::Lit {
   742→   742→                        value: Lit::Int(n), ..
   743→   743→                    } = &c.value
   744→   744→                    {
   745→   745→                        self.simple_vars.insert(c.name.clone(), *n);
   746→   746→                    }
   747→   747→                }
   748→   748→                _ => {}
   749→   749→            }
   750→   750→        }
   751→   751→    }
   752→   752→

   754→   754→    // Command handling

   756→   756→
   757→   757→    /// Handle a REPL command (input starting with `:`).
   758→   758→    fn handle_command(&mut self, input: &str) -> Result<ReplResult, ReplError> {
   759→   759→        let parts: Vec<&str> = input.splitn(2, ' ').collect();
   760→   760→        let cmd = parts[0];
   761→   761→        let arg = parts.get(1).map(|s| s.trim()).unwrap_or("");
   762→   762→
   763→   763→        match cmd {
   764→   764→            ":help" => Ok(ReplResult::Ok(Some(self.help_text()))),
   765→   765→            ":load" => self.cmd_load(arg),
   766→   766→            ":verify" => self.cmd_verify(),
   767→   767→            ":show" => self.cmd_show(arg),
   768→   768→            ":compile" => self.cmd_compile(),
   769→   769→            ":profile" => Ok(ReplResult::Ok(Some(format!("{}", self.profile)))),
   770→   770→            ":history" => Ok(ReplResult::Ok(Some(self.format_history()))),
   771→   771→            ":reset" => self.cmd_reset(),
   772→   772→            ":quit" | ":q" | ":exit" => {
   773→   773→                self.running = false;
   774→   774→                Ok(ReplResult::Quit)
   775→   775→            }
   776→   776→            _ => Ok(ReplResult::Ok(Some(format!(
   777→   777→                "Unknown command: {cmd}. Type :help for available commands."
   778→   778→            )))),
   779→   779→        }
   780→   780→    }

   810→   810→    /// Handle the `:load <file>` command.
   811→   811→    fn cmd_load(&mut self, path: &str) -> Result<ReplResult, ReplError> {
   812→   812→        if path.is_empty() {
   813→   813→            return Ok(ReplResult::Ok(Some("Usage: :load <file>".to_string())));
   814→   814→        }
   815→   815→
   816→   816→        let source = std::fs::read_to_string(path)
   817→   817→            .map_err(|e| ReplError::General(format!("Cannot read '{}': {}", path, e)))?;
   818→   818→
   819→   819→        self.loaded_file = Some(path.to_string());
   820→   820→
   821→   821→        // Replace the source buffer with the file content.
   822→   822→        self.source_buffer = source.clone();
   823→   823→
   824→   824→        // Parse and build SCG.
   825→   825→        let parse_start = Instant::now();
   826→   826→        let mut parser = Parser::new(&self.source_buffer);
   827→   827→        let result = parser.parse_program();
   828→   828→        if result.has_errors() {
   829→   829→            let errors = result.errors.clone();
   830→   830→            self.profile.parse_errors += errors.len();
   831→   831→            self.profile.parse_time_ms += parse_start.elapsed().as_millis() as u64;
   832→   832→            return Err(ReplError::ParseErrors(errors));
   833→   833→        }
   834→   834→        let program = result.unwrap();
   835→   835→        self.profile.parse_time_ms += parse_start.elapsed().as_millis() as u64;
   836→   836→
   837→   837→        // Build SCG.
   838→   838→        let scg_start = Instant::now();
   839→   839→        let mut converter = AstToScg::new();
   840→   840→        let scg = converter.convert(&program).map_err(|e| {
   841→   841→            self.profile.scg_time_ms += scg_start.elapsed().as_millis() as u64;
   842→   842→            ReplError::Parse(e)
   843→   843→        })?;
   844→   844→        self.scg = scg;
   845→   845→        self.converter = converter;
   846→   846→        self.profile.scg_time_ms += scg_start.elapsed().as_millis() as u64;
   847→   847→
   848→   848→        // Convert to MSG.
   849→   849→        let msg_start = Instant::now();
   850→   850→        match scg_to_msg::scg_to_msg(&self.scg) {
   851→   851→            Ok(msg) => self.msg = Some(msg),
   852→   852→            Err(e) => {
   853→   853→                log::warn!("MSG conversion failed: {e}");
   854→   854→                self.msg = None;
   855→   855→            }
   856→   856→        }
   857→   857→        self.profile.msg_time_ms += msg_start.elapsed().as_millis() as u64;
   858→   858→
   859→   859→        // Extract simple bindings.
   860→   860→        self.extract_simple_bindings(&program);
   861→   861→
   862→   862→        self.profile.expressions_processed += 1;
   863→   863→
   864→   864→        Ok(ReplResult::Ok(Some(format!(
   865→   865→            "Loaded '{}' (SCG: {} nodes, {} edges)",
   866→   866→            path,
   867→   867→            self.scg.node_count(),
   868→   868→            self.scg.edge_count()
   869→   869→        ))))
   870→   870→    }
   871→   871→
   872→   872→    /// Handle the `:verify` command.
   873→   873→    fn cmd_verify(&mut self) -> Result<ReplResult, ReplError> {
   874→   874→        let verify_start = Instant::now();
   875→   875→
   876→   876→        // Create verification input from the current SCG.
   877→   877→        let input = VerificationInput::from_scg(self.scg.clone());
   878→   878→
   879→   879→        let result = self.aggregator.verify_all(&input);
   880→   880→        self.profile.verify_time_ms += verify_start.elapsed().as_millis() as u64;
   881→   881→        self.profile.verification_runs += 1;
   882→   882→
   883→   883→        self.last_verification = Some(result.clone());
   884→   884→
   885→   885→        Ok(ReplResult::Verification(result))
   886→   886→    }

   888→   888→    /// Handle the `:show` command.

   892→   892→            "msg" => match &self.msg {
   893→   893→                Some(msg) => Ok(ReplResult::Ok(Some(format!("{msg}")))),

   898→   898→            "bd" => Ok(ReplResult::Ok(Some(format_bd_display(
   899→   899→                &self.scg,
   900→   900→                &self.inference_engine,
   901→   901→            )))),
   902→   902→            _ => Ok(ReplResult::Ok(Some(format!(
   903→   903→                "Unknown show target: '{what}'. Use :show scg, :show msg, or :show bd"
   904→   904→            )))),
   905→   905→        }

   908→   908→    /// Handle the `:compile` command — full pipeline.
   909→   909→    fn cmd_compile(&mut self) -> Result<ReplResult, ReplError> {
   910→   910→        let mut output = String::new();
   911→   911→
   912→   912→        // Step 1: Parse (already done if we have source).
   913→   913→        if self.source_buffer.is_empty() {
   914→   914→            return Ok(ReplResult::Ok(Some(
   915→   915→                "No source to compile. Enter some VUMA code first.".to_string(),
   916→   916→            )));
   917→   917→        }
   918→   918→
   919→   919→        output.push_str(&format!("Source: {} bytes\n", self.source_buffer.len()));
   920→   920→
   921→   921→        // Step 2: Rebuild SCG.
   922→   922→        let scg_start = Instant::now();
   923→   923→        let mut parser = Parser::new(&self.source_buffer);
   924→   924→        let result = parser.parse_program();
   925→   925→        if result.has_errors() {
   926→   926→            return Err(ReplError::ParseErrors(result.errors.clone()));
   927→   927→        }
   928→   928→        let program = result.unwrap();
   929→   929→        let mut converter = AstToScg::new();
   930→   930→        let scg = converter.convert(&program).map_err(ReplError::Parse)?;
   931→   931→        self.scg = scg;
   932→   932→        self.converter = converter;
   933→   933→        self.profile.scg_time_ms += scg_start.elapsed().as_millis() as u64;
   934→   934→
   935→   935→        output.push_str(&format!(
   936→   936→            "SCG: {} nodes, {} edges, {} regions\n",
   937→   937→            self.scg.node_count(),
   938→   938→            self.scg.edge_count(),
   939→   939→            self.scg.region_count()
   940→   940→        ));
   941→   941→
   942→   942→        // Step 3: Convert to MSG.
   943→   943→        let msg_start = Instant::now();
   944→   944→        match scg_to_msg::scg_to_msg(&self.scg) {
   945→   945→            Ok(msg) => {
   946→   946→                self.msg = Some(msg);
   947→   947→                output.push_str(&format!("MSG: {}\n", self.msg.as_ref().unwrap()));
   948→   948→            }
   949→   949→            Err(e) => {
   950→   950→                output.push_str(&format!("MSG conversion failed: {e}\n"));
   951→   951→                self.msg = None;
   952→   952→            }
   953→   953→        }
   954→   954→        self.profile.msg_time_ms += msg_start.elapsed().as_millis() as u64;
   955→   955→
   956→   956→        // Step 4: Verify.
   957→   957→        let verify_start = Instant::now();
   958→   958→        let input = VerificationInput::from_scg(self.scg.clone());
   959→   959→        let result = self.aggregator.verify_all(&input);

   964→   964→        output.push_str(&format!(

  1072→  1072→    /// Run the interactive REPL loop.
  1073→  1073→    ///
  1074→  1074→    /// Reads lines from stdin, processes them, and prints results.
  1075→  1075→    /// Supports basic up/down arrow key history navigation via ANSI
  1076→  1076→    /// escape sequences.
  1077→  1077→    pub fn run(&mut self) -> Result<(), ReplError> {
  1078→  1078→        println!("VUMA REPL v0.1.0");
  1079→  1079→        println!("Type :help for available commands.\n");
  1080→  1080→
  1081→  1081→        while self.running {
  1082→  1082→            print!("vuma> ");
  1083→  1083→            io::stdout().flush().map_err(ReplError::Io)?;
  1084→  1084→
  1085→  1085→            let mut input = String::new();
  1086→  1086→            match io::stdin().read_line(&mut input) {
  1087→  1087→                Ok(0) => {
  1088→  1088→                    // EOF.
  1089→  1089→                    self.running = false;
  1090→  1090→                    println!("Goodbye.");
  1091→  1091→                    break;
  1092→  1092→                }
  1093→  1093→                Ok(_) => {}
  1094→  1094→                Err(e) => return Err(ReplError::Io(e)),
  1095→  1095→            }
  1096→  1096→
  1097→  1097→            let line = input.trim_end();
  1098→  1098→
  1099→  1099→            match self.process_line(line) {
  1100→  1100→                Ok(ReplResult::Quit) => {
  1101→  1101→                    println!("Goodbye.");
  1102→  1102→                    break;
  1103→  1103→                }
  1104→  1104→                Ok(result) => {
  1105→  1105→                    if !result.to_string().is_empty() {
  1106→  1106→                        println!("{}", result);
  1107→  1107→                    }
  1108→  1108→                }
  1109→  1109→                Err(e) => match &e {
  1110→  1110→                    ReplError::Parse(pe) => {
  1111→  1111→                        let ctx = format_error_with_context(
  1112→  1112→                            &self.source_buffer,
  1113→  1113→                            &pe.span,
  1114→  1114→                            &pe.to_string(),
  1115→  1115→                        );
  1116→  1116→                        eprintln!("{ctx}");
  1117→  1117→                    }
  1118→  1118→                    ReplError::ParseErrors(errors) => {
  1119→  1119→                        for pe in errors {
  1120→  1120→                            let ctx = format_error_with_context(
  1121→  1121→                                &self.source_buffer,
  1122→  1122→                                &pe.span,
  1123→  1123→                                &pe.to_string(),
  1124→  1124→                            );
  1125→  1125→                            eprintln!("{ctx}");
  1126→  1126→                        }
  1127→  1127→                    }
  1128→  1128→                    _ => eprintln!("Error: {e}"),
  1129→  1129→                },
  1130→  1130→            }
  1131→  1131→        }
  1132→  1132→
  1133→  1133→        Ok(())
  1134→  1134→    }
  1135→  1135→}
