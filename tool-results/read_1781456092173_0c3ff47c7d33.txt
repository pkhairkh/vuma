     1→//! # LoongArch 64-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the LoongArch 64-bit target (LP64 ABI).
     4→//! This module provides:
     5→//!
     6→//! - `Gpr` — General-purpose register enum (r0–r31)
     7→//! - `Fpr` — Floating-point register enum (f0–f31)
     8→//! - `Instruction` — LoongArch64 instruction enum with correct encoding
     9→//! - Encoding helpers for all 9 instruction formats (2R, 3R, 4R, 2RI8, 2RI12,
    10→//!   2RI14, 2RI16, 1RI21, I26)
    11→//! - `LoongArch64Backend` — `Backend` implementation that lowers IR to LoongArch64
    12→//!   machine code and emits ELF64 binaries
    13→//!
    14→//! ## LoongArch64 Register Convention (LP64 ABI)
    15→//!
    16→//! | Register(s) | ABI Name | Role                              |
    17→//! |-------------|----------|-----------------------------------|
    18→//! | r0          | zero     | Hardwired zero                    |
    19→//! | r1          | ra       | Return address                    |
    20→//! | r2          | tp       | Thread pointer                    |
    21→//! | r3          | sp       | Stack pointer                     |
    22→//! | r4–r11      | a0–a7    | Argument / return registers       |
    23→//! | r12–r20     | t0–t8    | Caller-saved temporaries          |
    24→//! | r21         | —        | Reserved                          |
    25→//! | r22         | fp       | Frame pointer (callee-saved)      |
    26→//! | r23–r31     | s0–s8    | Callee-saved                      |
    27→//!
    28→//! ## LoongArch64 FP Register Convention (LP64 ABI)
    29→//!
    30→//! | Register(s) | ABI Name | Role                              |
    31→//! |-------------|----------|-----------------------------------|
    32→//! | f0–f7       | fa0–fa7  | FP argument / return registers    |
    33→//! | f8–f23      | ft0–ft15 | Caller-saved temporaries          |
    34→//! | f24–f31     | fs0–fs7  | Callee-saved                      |
    35→//!
    36→//! ## Instruction Formats
    37→//!
    38→//! All instructions are 32 bits, little-endian, with no branch delay slots.
    39→//! Nine formats: 2R, 3R, 4R, 2RI8, 2RI12, 2RI14, 2RI16, 1RI21, I26.
    40→//!
    41→//! ## References
    42→//!
    43→//! - LoongArch Reference Manual, Volume 1: Basic Architecture
    44→//! - <https://loongson.github.io/LoongArch-Documentation/>
    45→

    59→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]

    61→    R0 = 0,
    62→    Ra = 1,
    63→    Tp = 2,

    77→    T4 = 16,
    78→    T5 = 17,
    79→    T6 = 18,
    80→    T7 = 19,

    95→impl Gpr {
    96→    /// Returns the 5-bit encoding index for this register.
    97→    pub fn encoding(&self) -> u32 {
    98→        *self as u32
    99→    }
   100→
   101→    /// Returns `true` if this register is available for register allocation.
   102→    ///
   103→    /// R0 (zero), Ra, Tp, and Sp are reserved.
   104→    pub fn is_allocatable(&self) -> bool {
   105→        !matches!(self, Gpr::R0 | Gpr::Ra | Gpr::Tp | Gpr::Sp)
   106→    }
   107→
   108→    /// Returns `true` if this register is callee-saved (fp, s0–s8).
   109→    pub fn is_callee_saved(&self) -> bool {
   110→        matches!(
   111→            self,
   112→            Gpr::Fp
   113→                | Gpr::S0
   114→                | Gpr::S1
   115→                | Gpr::S2
   116→                | Gpr::S3
   117→                | Gpr::S4
   118→                | Gpr::S5
   119→                | Gpr::S6
   120→                | Gpr::S7
   121→                | Gpr::S8
   122→        )
   123→    }
   124→
   125→    /// Returns `true` if this register is an argument register (a0–a7).
   126→    pub fn is_arg_reg(&self) -> bool {
   127→        matches!(
   128→            self,
   129→            Gpr::A0 | Gpr::A1 | Gpr::A2 | Gpr::A3 | Gpr::A4 | Gpr::A5 | Gpr::A6 | Gpr::A7
   130→        )
   131→    }
   132→

   134→    pub fn asm_name(&self) -> &'static str {

   136→            Gpr::R0 => "$r0",
   137→            Gpr::Ra => "$ra",
   138→            Gpr::Tp => "$tp",
   139→            Gpr::Sp => "$sp",
   140→            Gpr::A0 => "$a0",
   141→            Gpr::A1 => "$a1",
   142→            Gpr::A2 => "$a2",
   143→            Gpr::A3 => "$a3",
   144→            Gpr::A4 => "$a4",
   145→            Gpr::A5 => "$a5",
   146→            Gpr::A6 => "$a6",
   147→            Gpr::A7 => "$a7",
   148→            Gpr::T0 => "$t0",
   149→            Gpr::T1 => "$t1",
   150→            Gpr::T2 => "$t2",
   151→            Gpr::T3 => "$t3",
   152→            Gpr::T4 => "$t4",
   153→            Gpr::T5 => "$t5",
   154→            Gpr::T6 => "$t6",
   155→            Gpr::T7 => "$t7",
   156→            Gpr::T8 => "$t8",
   157→            Gpr::R21 => "$r21",
   158→            Gpr::Fp => "$fp",
   159→            Gpr::S0 => "$s0",
   160→            Gpr::S1 => "$s1",
   161→            Gpr::S2 => "$s2",
   162→            Gpr::S3 => "$s3",
   163→            Gpr::S4 => "$s4",
   164→            Gpr::S5 => "$s5",
   165→            Gpr::S6 => "$s6",
   166→            Gpr::S7 => "$s7",
   167→            Gpr::S8 => "$s8",
   168→        }
   169→    }

   171→    /// Returns the Gpr for a given argument index (0–7). Returns `None` for
   172→    /// indices >= 8.
   173→    pub fn arg_register(index: usize) -> Option<Gpr> {
   174→        match index {
   175→            0 => Some(Gpr::A0),
   176→            1 => Some(Gpr::A1),
   177→            2 => Some(Gpr::A2),
   178→            3 => Some(Gpr::A3),
   179→            4 => Some(Gpr::A4),

   182→            7 => Some(Gpr::A7),
   183→            _ => None,
   184→        }
   185→    }
   186→}

   323→// ===========================================================================
   324→// Instruction Encoding Helpers
   325→// ===========================================================================
   326→
   327→/// Encode a 2R format instruction.
   328→///
   329→/// Format: `opcode[31:10] | rj[9:5] | rd[4:0]`
   330→fn encode_2r(opcode: u32, rj: u32, rd: u32) -> [u8; 4] {
   331→    let word = ((opcode & 0x3FF_FFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   332→    word.to_le_bytes()
   333→}
   334→
   335→/// Encode a 3R format instruction.
   336→///
   337→/// Format: `opcode[31:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   338→fn encode_3r(opcode: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   339→    let word = ((opcode & 0x1FFFF) << 15) | ((rk & 0x1F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   340→    word.to_le_bytes()
   341→}
   342→
   343→/// Encode a 4R format instruction.
   344→///
   345→/// Format: `opcode[31:20] | ra[19:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   346→#[allow(dead_code)]
   347→fn encode_4r(opcode: u32, ra: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   348→    let word = ((opcode & 0xFFF) << 20)
   349→        | ((ra & 0x1F) << 15)
   350→        | ((rk & 0x1F) << 10)
   351→        | ((rj & 0x1F) << 5)
   352→        | (rd & 0x1F);
   353→    word.to_le_bytes()
   354→}
   355→
   356→/// Encode a 2RI8 format instruction.
   357→///
   358→/// Format: `opcode[31:22] | I8[21:14] | rj[9:5] | rd[4:0]`
   359→#[allow(dead_code)]
   360→fn encode_2ri8(opcode: u32, imm8: u32, rj: u32, rd: u32) -> [u8; 4] {
   361→    let word = ((opcode & 0x3FF) << 22) | ((imm8 & 0xFF) << 14) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   362→    word.to_le_bytes()
   363→}
   364→
   365→/// Encode a reg2i5 format instruction (17-bit opcode).
   366→///
   367→/// Format: `opcode[31:15] | I5[14:10] | rj[9:5] | rd[4:0]`
   368→fn encode_reg2i5(opcode: u32, imm5: u32, rj: u32, rd: u32) -> [u8; 4] {
   369→    let word = ((opcode & 0x1FFFF) << 15) | ((imm5 & 0x1F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   370→    word.to_le_bytes()
   371→}
   372→
   373→/// Encode a reg2i6 format instruction (16-bit opcode).
   374→///
   375→/// Format: `opcode[31:16] | I6[15:10] | rj[9:5] | rd[4:0]`
   376→fn encode_reg2i6(opcode: u32, imm6: u32, rj: u32, rd: u32) -> [u8; 4] {
   377→    let word = ((opcode & 0xFFFF) << 16) | ((imm6 & 0x3F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   378→    word.to_le_bytes()
   379→}
   380→
   381→/// Encode a 2RI12 format instruction.
   382→///
   383→/// Format: `opcode[31:22] | I12[21:10] | rj[9:5] | rd[4:0]`
   384→fn encode_2ri12(opcode: u32, imm12: u32, rj: u32, rd: u32) -> [u8; 4] {
   385→    let word =
   386→        ((opcode & 0x3FF) << 22) | ((imm12 & 0xFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   387→    word.to_le_bytes()
   388→}
   389→
   390→/// Encode a 2RI14 format instruction.
   391→///
   392→/// Format: `opcode[31:24] | I14[23:10] | rj[9:5] | rd[4:0]`
   393→fn encode_2ri14(opcode: u32, imm14: u32, rj: u32, rd: u32) -> [u8; 4] {
   394→    let word =
   395→        ((opcode & 0xFF) << 24) | ((imm14 & 0x3FFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   396→    word.to_le_bytes()
   397→}
   398→
   399→/// Encode a 2RI16 format instruction.
   400→///
   401→/// Format: `opcode[31:26] | I16[25:10] | rj[9:5] | rd[4:0]`
   402→fn encode_2ri16(opcode: u32, imm16: u32, rj: u32, rd: u32) -> [u8; 4] {
   403→    let word =
   404→        ((opcode & 0x3F) << 26) | ((imm16 & 0xFFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   405→    word.to_le_bytes()
   406→}
   407→
   408→/// Encode a 1RI21 format instruction (used for BEQZ, BNEZ).
   409→///
   410→/// Format: `opcode[31:26] | offs21[15:0] in bits[25:10] | offs21[20:16] in bits[9:5] | rj[4:0]`
   411→///
   412→/// Note: the offset bits are SWAPPED compared to a linear layout.
   413→/// The lower 16 bits of the offset go in the higher position (bits[25:10]),
   414→/// and the upper 5 bits go in the lower position (bits[9:5]).
   415→fn encode_1ri21(opcode: u32, imm21: u32, rj: u32) -> [u8; 4] {
   416→    // 1RI21 format (BEQZ/BNEZ): opcode[31:26] | offs[15:0] at [25:10] | rj[9:5] | offs[20:16] at [4:0]
   417→    let word = ((opcode & 0x3F) << 26)
   418→        | ((imm21 & 0xFFFF) << 10)       // offs[15:0] at bits 25:10
   419→        | ((rj & 0x1F) << 5)             // rj at bits 9:5
   420→        | ((imm21 >> 16) & 0x1F);        // offs[20:16] at bits 4:0
   421→    word.to_le_bytes()
   422→}
   423→
   424→/// Encode an I26 format instruction (used for B, BL).
   425→///
   426→/// Format: `opcode[31:26] | offs26[15:0] in bits[25:10] | offs26[25:16] in bits[9:0]`
   427→///
   428→/// Note: the offset bits are SWAPPED compared to a linear layout.
   429→/// The lower 16 bits of the offset go in the higher position (bits[25:10]),
   430→/// and the upper 10 bits go in the lower position (bits[9:0]).
   431→fn encode_i26(opcode: u32, imm26: u32) -> [u8; 4] {
   432→    let word = ((opcode & 0x3F) << 26)
   433→        | ((imm26 & 0xFFFF) << 10)
   434→        | ((imm26 >> 16) & 0x3FF);
   435→    word.to_le_bytes()
   436→}

   873→impl Instruction {
   874→    /// Encode this instruction into a 4-byte little-endian machine code word.
   875→    ///
   876→    /// Encoding follows the LoongArch ISA Specification.
   877→    pub fn encode(&self) -> [u8; 4] {
   878→        match self {
   879→            // ── Arithmetic (3R) ────────────────────────────────────
   880→            Instruction::AddW { rd, rj, rk } => {
   881→                encode_3r(OPC_ADD_W, rk.encoding(), rj.encoding(), rd.encoding())
   882→            }
   883→            Instruction::AddD { rd, rj, rk } => {
   884→                encode_3r(OPC_ADD_D, rk.encoding(), rj.encoding(), rd.encoding())
   885→            }
   886→            Instruction::SubW { rd, rj, rk } => {
   887→                encode_3r(OPC_SUB_W, rk.encoding(), rj.encoding(), rd.encoding())
   888→            }
   889→            Instruction::SubD { rd, rj, rk } => {
   890→                encode_3r(OPC_SUB_D, rk.encoding(), rj.encoding(), rd.encoding())
   891→            }
   892→            Instruction::Slt { rd, rj, rk } => {
   893→                encode_3r(OPC_SLT, rk.encoding(), rj.encoding(), rd.encoding())
   894→            }
   895→            Instruction::Sltu { rd, rj, rk } => {
   896→                encode_3r(OPC_SLTU, rk.encoding(), rj.encoding(), rd.encoding())
   897→            }
   898→            Instruction::MulW { rd, rj, rk } => {
   899→                encode_3r(OPC_MUL_W, rk.encoding(), rj.encoding(), rd.encoding())
   900→            }
   901→            Instruction::MulD { rd, rj, rk } => {

  1559→/// Build a minimal ELF64 binary for LoongArch64 from raw code bytes.
  1560→///
  1561→/// Produces a static executable with two LOAD segments:
  1562→///   1. PF_R | PF_X — .text (code)
  1563→///   2. PF_R | PF_W — .data / stack (writable memory)
  1564→fn build_loongarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1565→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (LoongArch64 typical page size)
  1566→
  1567→    let elf_header_size: u64 = 64;
  1568→    let phdr_size: u64 = 56;
  1569→    let num_phdrs: u64 = 2;
  1570→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1571→    // Page-align the text segment start in the file.
  1572→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1573→    let text_size = code.len() as u64;
  1574→
  1575→    // The data segment starts on the next page after the text.
  1576→    let text_file_end = text_offset + text_size;
  1577→    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1578→    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1579→    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1580→    let entry_point = base_addr + text_offset;
  1581→
  1582→    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1583→
  1584→    // --- e_ident ---
  1585→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1586→    elf.push(2); // ELFCLASS64
  1587→    elf.push(1); // ELFDATA2LSB
  1588→    elf.push(1); // EV_CURRENT
  1589→    elf.push(3); // ELFOSABI_LINUX
  1590→    elf.push(0); // padding
  1591→    elf.extend_from_slice(&[0u8; 7]); // padding
  1592→
  1593→    // --- ELF header fields ---
  1594→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1595→    elf.extend_from_slice(&258u16.to_le_bytes()); // e_machine = EM_LOONGARCH
  1596→    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1597→    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1598→    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1599→    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1600→    elf.extend_from_slice(&0x43u32.to_le_bytes()); // e_flags = 0x43 (LP64D ABI double-float)
  1601→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1602→    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1603→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1604→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1605→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1606→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1607→
  1608→    // --- Program Header 1: LOAD (PF_R | PF_X) — .text ---
  1609→    // p_filesz = actual code size, p_memsz = page-aligned (QEMU needs this for mapping)
  1610→    let text_memsz = ((text_size + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1611→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1612→    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1613→    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1614→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1615→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1616→    elf.extend_from_slice(&text_memsz.to_le_bytes()); // p_filesz (cover full page for QEMU)
  1617→    elf.extend_from_slice(&text_memsz.to_le_bytes()); // p_memsz
  1618→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1619→
  1620→    // --- Program Header 2: LOAD (PF_R | PF_W) — .data / stack ---
  1621→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1622→    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1623→    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1624→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1625→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1626→    elf.extend_from_slice(&data_size.to_le_bytes()); // p_filesz (write zeros so QEMU can mmap)
  1627→    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1628→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1629→
  1630→    // --- .text section ---
  1631→    // Pad to page-aligned text_offset
  1632→    while (elf.len() as u64) < text_offset {
  1633→        elf.push(0);
  1634→    }
  1635→    elf.extend_from_slice(code);
  1636→
  1637→    // --- Pad to data segment offset and write data segment ---
  1638→    while (elf.len() as u64) < data_offset {
  1639→        elf.push(0);
  1640→    }
  1641→    // Write data segment (all zeros, but must exist in file for QEMU mmap)
  1642→    elf.extend_from_slice(&vec![0u8; data_size as usize]);
  1643→
  1644→    elf
  1645→}

  1651→/// Decode a single LoongArch64 32-bit instruction word into a mnemonic string.
  1652→fn decode_loongarch64_instruction(word: u32) -> String {
  1653→    let opcode = word & 0x7f;
  1654→    match opcode {

  1742→/// Compute the stack frame size for an IR function on LoongArch64.
  1743→///
  1744→/// Sums `Alloc` instruction sizes, adds 16 bytes for the ra/fp save pair,
  1745→/// and rounds up to 16-byte alignment.
  1746→fn loongarch64_compute_frame_size(func: &IRFunction) -> usize {
  1747→    let mut total: u32 = 16; // ra + fp save pair
  1748→    for block in &func.blocks {
  1749→        for instr in &block.instructions {
  1750→            if let IRInstr::Alloc { size, .. } = instr {
  1751→                let aligned = (*size).div_ceil(16) * 16;
  1752→                total += aligned;
  1753→            }
  1754→        }
  1755→    }
  1756→    // Round up to 16-byte alignment
  1757→    total = (total + 15) & !15;
  1758→    total as usize
  1759→}

  1798→/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1799→fn map_vreg_to_gpr(
  1800→    vreg_id: u32,
  1801→    arg_index: Option<usize>,
  1802→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1803→) -> Gpr {
  1804→    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1805→        return *gpr;
  1806→    }
  1807→
  1808→    // If this is an argument, map it to the corresponding argument register.
  1809→    if let Some(idx) = arg_index {
  1810→        if let Some(reg) = Gpr::arg_register(idx) {
  1811→            vreg_map.insert(vreg_id, reg);
  1812→            return reg;
  1813→        }
  1814→    }
  1815→
  1816→    // Otherwise, pick the first available allocatable register.
  1817→    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1818→    for &reg in ALLOCATABLE_GPRS {
  1819→        if !used.contains(&reg) {
  1820→            vreg_map.insert(vreg_id, reg);
  1821→            return reg;
  1822→        }
  1823→    }
  1824→
  1825→    // Fallback: use t8 as a scratch register.
  1826→    Gpr::T8
  1827→}

  1837→/// Load a 64-bit immediate into a register using the lu12i.w/ori/lu32i.d/lu52i.d
  1838→/// instruction sequence.
  1839→///
  1840→/// Returns the list of AllocatedInstructions emitted to perform the load.
  1841→fn load_imm_la64(rd: Gpr, imm: i64) -> Vec<AllocatedInstruction> {
  1842→    let mut result = Vec::new();
  1843→    let val = imm as u64;
  1844→
  1845→    // lu12i.w rd, bits[31:12]
  1846→    let hi20 = ((val >> 12) & 0xFFFFF) as i32;

  1848→        Instruction::Lu12iW { rd, imm20: hi20 },
  1849→        vec![],
  1850→        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1851→    ));
  1852→
  1853→    // ori rd, rd, bits[11:0]
  1854→    let lo12 = (val & 0xFFF) as u32;
  1855→    result.push(emit_alloc_instr(
  1856→        Instruction::Ori {
  1857→            rd,
  1858→            rj: rd,
  1859→            imm12: lo12,
  1860→        },

  1862→        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1863→    ));

  1866→    // Instead, handle upper bits differently

  1869→    let sign_ext = if lower32 & 0x80000000 != 0 {
  1870→        0xFFFFFFFF00000000u64
  1871→    } else {
  1872→        0u64
  1873→    };

  1877→        if val >> 32 == 0 && lower32 & 0x80000000 != 0 {

  1881→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1882→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1883→            ));

  1885→                Instruction::SrliD { rd, rj: rd, imm8: 32 },
  1886→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],

  1889→        } else {

  1891→            let hi52 = ((val >> 52) & 0xFFF) as i32;
  1892→            result.push(emit_alloc_instr(

  1894→                    rd,
  1895→                    rj: rd,

  1911→                    Instruction::SlliD { rd: temp, rj: temp, imm8: 32 },
  1912→                    vec![PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1913→                    vec![PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1914→                ));
  1915→                result.push(emit_alloc_instr(

  1917→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1918→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],

  1937→/// Resolve an IRValue to a physical GPR.
  1938→///
  1939→/// - For `IRValue::Register`: looks up in `vreg_map`.
  1940→/// - For `IRValue::Immediate`: loads the value into `scratch` and returns
  1941→///   `(scratch, instructions)`.
  1942→/// - For `IRValue::Address`: loads the address into `scratch`.
  1943→/// - For `IRValue::Label`: loads a placeholder into `scratch`.
  1944→fn resolve_gpr_la64(
  1945→    val: &IRValue,
  1946→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1947→    scratch: Gpr,
  1948→) -> (Gpr, Vec<AllocatedInstruction>) {
  1949→    match val {
  1950→        IRValue::Register(id) => {
  1951→            let reg = map_vreg_to_gpr(*id, None, vreg_map);
  1952→            (reg, Vec::new())
  1953→        }
  1954→        IRValue::Immediate(imm) => {
  1955→            let code = load_imm_la64(scratch, *imm);
  1956→            (scratch, code)
  1957→        }
  1958→        IRValue::Address(addr) => {
  1959→            let code = load_imm_la64(scratch, *addr as i64);
  1960→            (scratch, code)
  1961→        }
  1962→        IRValue::Label(_) => {
  1963→            // Labels need relocation; emit a placeholder load
  1964→            let code = load_imm_la64(scratch, 0);
  1965→            (scratch, code)
  1966→        }
  1967→    }
  1968→}
