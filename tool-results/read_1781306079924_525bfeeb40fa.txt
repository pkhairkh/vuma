     1→//! # RISC-V 64-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the RISC-V 64-bit target (RV64GC,
     4→//! LP64D ABI).  This module provides:
     5→//!
     6→//! - `Gpr` — General-purpose register enum (x0–x31)
     7→//! - `Fpr` — Floating-point register enum (f0–f31)
     8→//! - `Instruction` — RV64I + M + F/D instruction enum with correct encoding
     9→//! - Encoding helpers for R/I/S/B/U/J-type instruction formats
    10→//! - `RiscV64Backend` — `Backend` implementation that lowers IR to RISC-V machine code
    11→//!
    12→//! ## RISC-V Register Convention (LP64D ABI)
    13→//!
    14→//! | Register(s) | ABI Name | Role                              |
    15→//! |-------------|----------|-----------------------------------|
    16→//! | x0          | zero     | Hardwired zero                    |
    17→//! | x1          | ra       | Return address                    |
    18→//! | x2          | sp       | Stack pointer                     |
    19→//! | x3          | gp       | Global pointer                    |
    20→//! | x4          | tp       | Thread pointer                    |
    21→//! | x5–x7       | t0–t2    | Caller-saved temporaries          |
    22→//! | x8          | s0/fp    | Callee-saved / frame pointer      |
    23→//! | x9          | s1       | Callee-saved                      |
    24→//! | x10–x17     | a0–a7    | Argument / return registers       |
    25→//! | x18–x27     | s2–s11   | Callee-saved                      |
    26→//! | x28–x31     | t3–t6    | Caller-saved temporaries          |
    27→//!
    28→//! ## References
    29→//!
    30→//! - RISC-V Instruction Set Manual, Volume I: User-Level ISA, Document 20191213
    31→//! - <https://riscv.org/specifications/>

   404→/// Encode an R-type instruction.
   405→///
   406→/// Format: `funct7[31:25] | rs2[24:20] | rs1[19:15] | funct3[14:12] | rd[11:7] | opcode[6:0]`
   407→fn encode_r_type(funct7: u32, rs2: u32, rs1: u32, funct3: u32, rd: u32, opcode: u32) -> [u8; 4] {
   408→    let word = ((funct7 & 0x7F) << 25)
   409→        | ((rs2 & 0x1F) << 20)
   410→        | ((rs1 & 0x1F) << 15)
   411→        | ((funct3 & 0x7) << 12)
   412→        | ((rd & 0x1F) << 7)
   413→        | (opcode & 0x7F);
   414→    word.to_le_bytes()
   415→}
   416→
   417→/// Encode an I-type instruction.
   418→///
   419→/// Format: `imm[31:20] | rs1[19:15] | funct3[14:12] | rd[11:7] | opcode[6:0]`
   420→fn encode_i_type(imm: u32, rs1: u32, funct3: u32, rd: u32, opcode: u32) -> [u8; 4] {
   421→    let word = ((imm & 0xFFF) << 20)
   422→        | ((rs1 & 0x1F) << 15)
   423→        | ((funct3 & 0x7) << 12)
   424→        | ((rd & 0x1F) << 7)
   425→        | (opcode & 0x7F);
   426→    word.to_le_bytes()
   427→}
   428→
   429→/// Encode an S-type instruction.
   430→///
   431→/// Format: `imm[11:5][31:25] | rs2[24:20] | rs1[19:15] | funct3[14:12] | imm[4:0][11:7] | opcode[6:0]`
   432→fn encode_s_type(imm: u32, rs2: u32, rs1: u32, funct3: u32, opcode: u32) -> [u8; 4] {
   433→    let imm_lo = imm & 0x1F;
   434→    let imm_hi = (imm >> 5) & 0x7F;
   435→    let word = (imm_hi << 25)
   436→        | ((rs2 & 0x1F) << 20)
   437→        | ((rs1 & 0x1F) << 15)
   438→        | ((funct3 & 0x7) << 12)
   439→        | (imm_lo << 7)
   440→        | (opcode & 0x7F);
   441→    word.to_le_bytes()
   442→}
   443→
   444→/// Encode a B-type instruction.
   445→///
   446→/// Format: `imm[12|10:5][31:25] | rs2[24:20] | rs1[19:15] | funct3[14:12] | imm[4:1|11][11:7] | opcode[6:0]`
   447→///
   448→/// The immediate is a signed 13-bit byte offset (bit 0 is always 0).
   449→fn encode_b_type(imm: i32, rs2: u32, rs1: u32, funct3: u32, opcode: u32) -> [u8; 4] {
   450→    // B-type immediate bit layout:
   451→    // bit 31:   imm[12]
   452→    // bits 30:25: imm[10:5]
   453→    // bits 11:8:  imm[4:1]
   454→    // bit 7:    imm[11]
   455→    let imm_u = imm as u32;
   456→    let word = (((imm_u >> 12) & 0x1) << 31)
   457→        | (((imm_u >> 5) & 0x3F) << 25)
   458→        | ((rs2 & 0x1F) << 20)
   459→        | ((rs1 & 0x1F) << 15)
   460→        | ((funct3 & 0x7) << 12)
   461→        | (((imm_u >> 1) & 0xF) << 8)
   462→        | (((imm_u >> 11) & 0x1) << 7)
   463→        | (opcode & 0x7F);
   464→    word.to_le_bytes()
   465→}
   466→
   467→/// Encode a U-type instruction.
   468→///
   469→/// Format: `imm[31:12] | rd[11:7] | opcode[6:0]`
   470→///
   471→/// The immediate is the upper 20 bits; the lower 12 bits are zero.
   472→fn encode_u_type(imm: u32, rd: u32, opcode: u32) -> [u8; 4] {
   473→    let word = (imm & 0xFFFFF000) | ((rd & 0x1F) << 7) | (opcode & 0x7F);
   474→    word.to_le_bytes()
   475→}
   476→
   477→/// Encode a J-type instruction.
   478→///
   479→/// Format: `imm[20|10:1|11|19:12][31:12] | rd[11:7] | opcode[6:0]`
   480→///
   481→/// The immediate is a signed 21-bit byte offset (bit 0 is always 0).
   482→fn encode_j_type(imm: i32, rd: u32, opcode: u32) -> [u8; 4] {
   483→    // J-type immediate bit layout:
   484→    // bit 31:    imm[20]
   485→    // bits 30:21: imm[10:1]
   486→    // bit 20:    imm[11]
   487→    // bits 19:12: imm[19:12]
   488→    let imm_u = imm as u32;
   489→    let word = (((imm_u >> 20) & 0x1) << 31)
   490→        | (((imm_u >> 1) & 0x3FF) << 21)
   491→        | (((imm_u >> 11) & 0x1) << 20)
   492→        | (((imm_u >> 12) & 0xFF) << 12)
   493→        | ((rd & 0x1F) << 7)
   494→        | (opcode & 0x7F);
   495→    word.to_le_bytes()
   496→}

   499→// Instruction Enum
   500→// ===========================================================================
   501→
   502→/// RISC-V 64-bit instruction representations for code generation.
   503→///
   504→/// Covers RV64I base, M extension, and key F/D extension instructions.
   505→/// Each variant captures the operands needed for encoding and disassembly.
   506→/// The `encode()` method produces a 4-byte little-endian machine code word.
   507→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   508→pub enum Instruction {
   509→    // ── RV64I: Upper Immediate ────────────────────────────────────────
   510→    /// Load Upper Immediate: `lui rd, imm`
   511→    Lui { rd: Gpr, imm: u32 },
   512→    /// Add Upper Immediate to PC: `auipc rd, imm`
   513→    Auipc { rd: Gpr, imm: u32 },
   514→
   515→    // ── RV64I: Jumps ─────────────────────────────────────────────────
   516→    /// Jump and Link: `jal rd, offset`
   517→    Jal { rd: Gpr, offset: i32 },
   518→    /// Jump and Link Register: `jalr rd, rs1, imm`
   519→    Jalr { rd: Gpr, rs1: Gpr, imm: i32 },
   520→
   521→    // ── RV64I: Branches ──────────────────────────────────────────────
   522→    /// Branch if Equal: `beq rs1, rs2, offset`
   523→    Beq { rs1: Gpr, rs2: Gpr, offset: i32 },
   524→    /// Branch if Not Equal: `bne rs1, rs2, offset`
   525→    Bne { rs1: Gpr, rs2: Gpr, offset: i32 },
   526→    /// Branch if Less Than (signed): `blt rs1, rs2, offset`
   527→    Blt { rs1: Gpr, rs2: Gpr, offset: i32 },
   528→    /// Branch if Greater or Equal (signed): `bge rs1, rs2, offset`
   529→    Bge { rs1: Gpr, rs2: Gpr, offset: i32 },
   530→    /// Branch if Less Than (unsigned): `bltu rs1, rs2, offset`

   691→
   692→impl Instruction {
   693→    /// Encode this instruction into a 4-byte little-endian machine code word.
   694→    ///
   695→    /// Encoding follows the RISC-V ISA Specification.
   696→    pub fn encode(&self) -> [u8; 4] {
   697→        match self {
   698→            // ── Upper Immediate ──────────────────────────────────────
   699→            Instruction::Lui { rd, imm } => encode_u_type(*imm, rd.encoding(), OP_LUI),
   700→            Instruction::Auipc { rd, imm } => encode_u_type(*imm, rd.encoding(), OP_AUIPC),
   701→
   702→            // ── Jumps ───────────────────────────────────────────────
   703→            Instruction::Jal { rd, offset } => encode_j_type(*offset, rd.encoding(), OP_JAL),
   704→            Instruction::Jalr { rd, rs1, imm } => encode_i_type(
   705→                (*imm as u32) & 0xFFF,
   706→                rs1.encoding(),
   707→                0b000,
   708→                rd.encoding(),
   709→                OP_JALR,
   710→            ),
   711→
   712→            // ── Branches ────────────────────────────────────────────
   713→            Instruction::Beq { rs1, rs2, offset } => {
   714→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b000, OP_BRANCH)
   715→            }
   716→            Instruction::Bne { rs1, rs2, offset } => {
   717→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b001, OP_BRANCH)
   718→            }
   719→            Instruction::Blt { rs1, rs2, offset } => {
   720→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b100, OP_BRANCH)
   721→            }
   722→            Instruction::Bge { rs1, rs2, offset } => {
   723→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b101, OP_BRANCH)
   724→            }
   725→            Instruction::Bltu { rs1, rs2, offset } => {
   726→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b110, OP_BRANCH)
   727→            }
   728→            Instruction::Bgeu { rs1, rs2, offset } => {
   729→                encode_b_type(*offset, rs2.encoding(), rs1.encoding(), 0b111, OP_BRANCH)
   730→            }
   731→
   732→            // ── Loads ───────────────────────────────────────────────
   733→            Instruction::Lb { rd, rs1, imm } => encode_i_type(
   734→                (*imm as u32) & 0xFFF,
   735→                rs1.encoding(),
   736→                0b000,
   737→                rd.encoding(),
   738→                OP_LOAD,
   739→            ),
   740→            Instruction::Lh { rd, rs1, imm } => encode_i_type(
   741→                (*imm as u32) & 0xFFF,
   742→                rs1.encoding(),
   743→                0b001,
   744→                rd.encoding(),
   745→                OP_LOAD,
   746→            ),
   747→            Instruction::Lw { rd, rs1, imm } => encode_i_type(
   748→                (*imm as u32) & 0xFFF,
   749→                rs1.encoding(),
   750→                0b010,
   751→                rd.encoding(),
   752→                OP_LOAD,
   753→            ),
   754→            Instruction::Ld { rd, rs1, imm } => encode_i_type(
   755→                (*imm as u32) & 0xFFF,
   756→                rs1.encoding(),
   757→                0b011,
   758→                rd.encoding(),
   759→                OP_LOAD,
   760→            ),
   761→            Instruction::Lbu { rd, rs1, imm } => encode_i_type(
   762→                (*imm as u32) & 0xFFF,
   763→                rs1.encoding(),
   764→                0b100,
   765→                rd.encoding(),
   766→                OP_LOAD,
   767→            ),
   768→            Instruction::Lhu { rd, rs1, imm } => encode_i_type(
   769→                (*imm as u32) & 0xFFF,
   770→                rs1.encoding(),
   771→                0b101,
   772→                rd.encoding(),
   773→                OP_LOAD,
   774→            ),
   775→            Instruction::Lwu { rd, rs1, imm } => encode_i_type(
   776→                (*imm as u32) & 0xFFF,
   777→                rs1.encoding(),
   778→                0b110,
   779→                rd.encoding(),
   780→                OP_LOAD,
   781→            ),
   782→
   783→            // ── Stores ──────────────────────────────────────────────
   784→            Instruction::Sb { rs1, rs2, imm } => encode_s_type(
   785→                (*imm as u32) & 0xFFF,
   786→                rs2.encoding(),
   787→                rs1.encoding(),
   788→                0b000,
   789→                OP_STORE,
   790→            ),
   791→            Instruction::Sh { rs1, rs2, imm } => encode_s_type(
   792→                (*imm as u32) & 0xFFF,
   793→                rs2.encoding(),
   794→                rs1.encoding(),
   795→                0b001,
   796→                OP_STORE,
   797→            ),
   798→            Instruction::Sw { rs1, rs2, imm } => encode_s_type(
   799→                (*imm as u32) & 0xFFF,
   800→                rs2.encoding(),
   801→                rs1.encoding(),
   802→                0b010,
   803→                OP_STORE,
   804→            ),
   805→            Instruction::Sd { rs1, rs2, imm } => encode_s_type(
   806→                (*imm as u32) & 0xFFF,
   807→                rs2.encoding(),
   808→                rs1.encoding(),
   809→                0b011,
   810→                OP_STORE,
   811→            ),
   812→
   813→            // ── Immediate Arithmetic ────────────────────────────────
   814→            Instruction::Addi { rd, rs1, imm } => encode_i_type(
   815→                (*imm as u32) & 0xFFF,
   816→                rs1.encoding(),
   817→                0b000,
   818→                rd.encoding(),
   819→                OP_IMM,
   820→            ),
   821→            Instruction::Slti { rd, rs1, imm } => encode_i_type(
   822→                (*imm as u32) & 0xFFF,
   823→                rs1.encoding(),
   824→                0b010,
   825→                rd.encoding(),
   826→                OP_IMM,
   827→            ),
   828→            Instruction::Sltiu { rd, rs1, imm } => encode_i_type(
   829→                (*imm as u32) & 0xFFF,
   830→                rs1.encoding(),
   831→                0b011,
   832→                rd.encoding(),
   833→                OP_IMM,
   834→            ),
   835→            Instruction::Xori { rd, rs1, imm } => encode_i_type(
   836→                (*imm as u32) & 0xFFF,
   837→                rs1.encoding(),
   838→                0b100,
   839→                rd.encoding(),
   840→                OP_IMM,
   841→            ),
   842→            Instruction::Ori { rd, rs1, imm } => encode_i_type(
   843→                (*imm as u32) & 0xFFF,
   844→                rs1.encoding(),
   845→                0b110,
   846→                rd.encoding(),
   847→                OP_IMM,
   848→            ),
   849→            Instruction::Andi { rd, rs1, imm } => encode_i_type(
   850→                (*imm as u32) & 0xFFF,
   851→                rs1.encoding(),
   852→                0b111,
   853→                rd.encoding(),
   854→                OP_IMM,
   855→            ),
   856→            Instruction::Slli { rd, rs1, shamt } => {
   857→                // funct7 = 0b0000000, funct3 = 0b001
   858→                // For RV64I, shamt is 6 bits (bits [25:20])
   859→                let funct7_and_shamt = (*shamt & 0x3F) << 20;
   860→                let word = funct7_and_shamt
   861→                    | (rs1.encoding() << 15)
   862→                    | (0b001 << 12)
   863→                    | (rd.encoding() << 7)
   864→                    | OP_IMM;
   865→                word.to_le_bytes()
   866→            }
   867→            Instruction::Srli { rd, rs1, shamt } => {
   868→                // funct7 = 0b0000000, funct3 = 0b101
   869→                let funct7_and_shamt = (*shamt & 0x3F) << 20;
   870→                let word = funct7_and_shamt
   871→                    | (rs1.encoding() << 15)
   872→                    | (0b101 << 12)
   873→                    | (rd.encoding() << 7)
   874→                    | OP_IMM;
   875→                word.to_le_bytes()
   876→            }
   877→            Instruction::Srai { rd, rs1, shamt } => {
   878→                // funct7 = 0b0100000, funct3 = 0b101
   879→                let funct7_and_shamt = (0b0100000u32 << 25) | ((*shamt & 0x3F) << 20);
   880→                let word = funct7_and_shamt
   881→                    | (rs1.encoding() << 15)
   882→                    | (0b101 << 12)
   883→                    | (rd.encoding() << 7)
   884→                    | OP_IMM;
   885→                word.to_le_bytes()
   886→            }
   887→
   888→            // ── Register Arithmetic ─────────────────────────────────
   889→            Instruction::Add { rd, rs1, rs2 } => encode_r_type(
   890→                0b0000000,
   891→                rs2.encoding(),
   892→                rs1.encoding(),
   893→                0b000,
   894→                rd.encoding(),
   895→                OP_REG,
   896→            ),
   897→            Instruction::Sub { rd, rs1, rs2 } => encode_r_type(
   898→                0b0100000,
   899→                rs2.encoding(),
   900→                rs1.encoding(),
   901→                0b000,
   902→                rd.encoding(),
   903→                OP_REG,
   904→            ),
   905→            Instruction::Sll { rd, rs1, rs2 } => encode_r_type(
   906→                0b0000000,
   907→                rs2.encoding(),
   908→                rs1.encoding(),
   909→                0b001,
   910→                rd.encoding(),
   911→                OP_REG,
   912→            ),
   913→            Instruction::Slt { rd, rs1, rs2 } => encode_r_type(
   914→                0b0000000,
   915→                rs2.encoding(),
   916→                rs1.encoding(),
   917→                0b010,
   918→                rd.encoding(),
   919→                OP_REG,
   920→            ),
   921→            Instruction::Sltu { rd, rs1, rs2 } => encode_r_type(
   922→                0b0000000,
   923→                rs2.encoding(),
   924→                rs1.encoding(),
   925→                0b011,
   926→                rd.encoding(),
   927→                OP_REG,
   928→            ),
   929→            Instruction::Xor { rd, rs1, rs2 } => encode_r_type(
   930→                0b0000000,
   931→                rs2.encoding(),
   932→                rs1.encoding(),
   933→                0b100,
   934→                rd.encoding(),
   935→                OP_REG,
   936→            ),
   937→            Instruction::Srl { rd, rs1, rs2 } => encode_r_type(
   938→                0b0000000,
   939→                rs2.encoding(),
   940→                rs1.encoding(),
   941→                0b101,
   942→                rd.encoding(),
   943→                OP_REG,
   944→            ),
   945→            Instruction::Sra { rd, rs1, rs2 } => encode_r_type(
   946→                0b0100000,
   947→                rs2.encoding(),
   948→                rs1.encoding(),
   949→                0b101,
   950→                rd.encoding(),
   951→                OP_REG,
   952→            ),
   953→            Instruction::Or { rd, rs1, rs2 } => encode_r_type(
   954→                0b0000000,
   955→                rs2.encoding(),
   956→                rs1.encoding(),
   957→                0b110,
   958→                rd.encoding(),
   959→                OP_REG,
   960→            ),
   961→            Instruction::And { rd, rs1, rs2 } => encode_r_type(
   962→                0b0000000,
   963→                rs2.encoding(),
   964→                rs1.encoding(),
   965→                0b111,
   966→                rd.encoding(),
   967→                OP_REG,
   968→            ),
   969→
   970→            // ── Word-level Arithmetic (RV64) ────────────────────────
   971→            Instruction::Addw { rd, rs1, rs2 } => encode_r_type(
   972→                0b0000000,
   973→                rs2.encoding(),
   974→                rs1.encoding(),
   975→                0b000,
   976→                rd.encoding(),
   977→                OP_REG32,
   978→            ),
   979→            Instruction::Subw { rd, rs1, rs2 } => encode_r_type(
   980→                0b0100000,
   981→                rs2.encoding(),
   982→                rs1.encoding(),
   983→                0b000,
   984→                rd.encoding(),
   985→                OP_REG32,
   986→            ),
   987→            Instruction::Sllw { rd, rs1, rs2 } => encode_r_type(
   988→                0b0000000,
   989→                rs2.encoding(),
   990→                rs1.encoding(),
   991→                0b001,
   992→                rd.encoding(),
   993→                OP_REG32,
   994→            ),
   995→            Instruction::Srlw { rd, rs1, rs2 } => encode_r_type(
   996→                0b0000000,
   997→                rs2.encoding(),
   998→                rs1.encoding(),
   999→                0b101,
  1000→                rd.encoding(),
  1001→                OP_REG32,
  1002→            ),
  1003→            Instruction::Sraw { rd, rs1, rs2 } => encode_r_type(
  1004→                0b0100000,
  1005→                rs2.encoding(),
  1006→                rs1.encoding(),
  1007→                0b101,
  1008→                rd.encoding(),
  1009→                OP_REG32,
  1010→            ),
  1011→            Instruction::Addiw { rd, rs1, imm } => encode_i_type(
  1012→                (*imm as u32) & 0xFFF,
  1013→                rs1.encoding(),
  1014→                0b000,
  1015→                rd.encoding(),
  1016→                OP_IMM32,
  1017→            ),
  1018→            Instruction::Slliw { rd, rs1, shamt } => {
  1019→                // funct7 = 0b0000000, funct3 = 0b001, shamt is 5 bits
  1020→                let funct7_and_shamt = (*shamt & 0x1F) << 20;
  1021→                let word = funct7_and_shamt
  1022→                    | (rs1.encoding() << 15)
  1023→                    | (0b001 << 12)
  1024→                    | (rd.encoding() << 7)
  1025→                    | OP_IMM32;
  1026→                word.to_le_bytes()
  1027→            }
  1028→            Instruction::Srliw { rd, rs1, shamt } => {
  1029→                let funct7_and_shamt = (*shamt & 0x1F) << 20;
  1030→                let word = funct7_and_shamt
  1031→                    | (rs1.encoding() << 15)
  1032→                    | (0b101 << 12)
  1033→                    | (rd.encoding() << 7)
  1034→                    | OP_IMM32;
  1035→                word.to_le_bytes()
  1036→            }
  1037→            Instruction::Sraiw { rd, rs1, shamt } => {
  1038→                let funct7_and_shamt = (0b0100000u32 << 25) | ((*shamt & 0x1F) << 20);
  1039→                let word = funct7_and_shamt
  1040→                    | (rs1.encoding() << 15)
  1041→                    | (0b101 << 12)
  1042→                    | (rd.encoding() << 7)
  1043→                    | OP_IMM32;
  1044→                word.to_le_bytes()
  1045→            }
  1046→
  1047→            // ── M Extension ─────────────────────────────────────────
  1048→            Instruction::Mul { rd, rs1, rs2 } => encode_r_type(
  1049→                0b0000001,
  1050→                rs2.encoding(),
  1051→                rs1.encoding(),
  1052→                0b000,
  1053→                rd.encoding(),
  1054→                OP_REG,
  1055→            ),
  1056→            Instruction::Mulh { rd, rs1, rs2 } => encode_r_type(
  1057→                0b0000001,
  1058→                rs2.encoding(),
  1059→                rs1.encoding(),
  1060→                0b001,
  1061→                rd.encoding(),
  1062→                OP_REG,
  1063→            ),
  1064→            Instruction::Mulhsu { rd, rs1, rs2 } => encode_r_type(
  1065→                0b0000001,
  1066→                rs2.encoding(),
  1067→                rs1.encoding(),
  1068→                0b010,
  1069→                rd.encoding(),
  1070→                OP_REG,
  1071→            ),
  1072→            Instruction::Mulhu { rd, rs1, rs2 } => encode_r_type(
  1073→                0b0000001,
  1074→                rs2.encoding(),
  1075→                rs1.encoding(),
  1076→                0b011,
  1077→                rd.encoding(),
  1078→                OP_REG,
  1079→            ),
  1080→            Instruction::Div { rd, rs1, rs2 } => encode_r_type(
  1081→                0b0000001,
  1082→                rs2.encoding(),
  1083→                rs1.encoding(),
  1084→                0b100,
  1085→                rd.encoding(),
  1086→                OP_REG,
  1087→            ),
  1088→            Instruction::Divu { rd, rs1, rs2 } => encode_r_type(
  1089→                0b0000001,
  1090→                rs2.encoding(),
  1091→                rs1.encoding(),
  1092→                0b101,
  1093→                rd.encoding(),
  1094→                OP_REG,
  1095→            ),
  1096→            Instruction::Rem { rd, rs1, rs2 } => encode_r_type(
  1097→                0b0000001,
  1098→                rs2.encoding(),
  1099→                rs1.encoding(),
  1100→                0b110,
  1101→                rd.encoding(),
  1102→                OP_REG,
  1103→            ),
  1104→            Instruction::Remu { rd, rs1, rs2 } => encode_r_type(
  1105→                0b0000001,
  1106→                rs2.encoding(),
  1107→                rs1.encoding(),
  1108→                0b111,
  1109→                rd.encoding(),
  1110→                OP_REG,
  1111→            ),
  1112→
  1113→            // ── F/D Extension: Load/Store ───────────────────────────
  1114→            Instruction::Flw { rd, rs1, imm } => encode_i_type(
  1115→                (*imm as u32) & 0xFFF,
  1116→                rs1.encoding(),
  1117→                0b010,
  1118→                rd.encoding(),
  1119→                OP_LOAD,
  1120→            ),
  1121→            Instruction::Fld { rd, rs1, imm } => encode_i_type(
  1122→                (*imm as u32) & 0xFFF,
  1123→                rs1.encoding(),
  1124→                0b011,
  1125→                rd.encoding(),
  1126→                OP_LOAD,
  1127→            ),
  1128→            Instruction::Fsw { rs1, rs2, imm } => encode_s_type(

  1130→                rs2.encoding(),

  1132→                0b010,
  1133→                OP_STORE,
  1134→            ),
  1135→            Instruction::Fsd { rs1, rs2, imm } => encode_s_type(

  1139→                0b011,

  1141→            ),
  1142→

  1144→            Instruction::FaddD { rd, rs1, rs2 } => {

  1146→                // Actually: funct7[6:0] = 0000001, rs2, rs1, rm=111, rd, opcode
  1147→                encode_r_type(
  1148→                    0b0000001,
  1149→                    rs2.encoding(),

  1152→                    rd.encoding(),

  1155→            }

  1160→                0b111,

  1163→            ),

  1175→                rs1.encoding(),
  1176→                0b111,
  1177→                rd.encoding(),
  1178→                OP_FP,
  1179→            ),
  1180→            Instruction::FmvD { rd, rs1 } => {

  1182→                encode_r_type(
  1183→                    0b0010001,
  1184→                    rs1.encoding(),
  1185→                    rs1.encoding(),
  1186→                    0b000,
  1187→                    rd.encoding(),
  1188→                    OP_FP,
  1189→                )
  1190→            }
  1191→
  1192→            // ── Zicsr Extension ──────────────────────────────────────
  1193→            Instruction::Csrrw { rd, csr, rs1 } => {
  1194→                // CSRRW: I-type, funct3=0b001, opcode=SYSTEM
  1195→                encode_i_type(*csr, rs1.encoding(), 0b001, rd.encoding(), OP_SYSTEM)
  1196→            }
  1197→            Instruction::Csrrs { rd, csr, rs1 } => {
  1198→                // CSRRS: I-type, funct3=0b010, opcode=SYSTEM
  1199→                encode_i_type(*csr, rs1.encoding(), 0b010, rd.encoding(), OP_SYSTEM)
  1200→            }
  1201→            Instruction::Csrrc { rd, csr, rs1 } => {
  1202→                // CSRRC: I-type, funct3=0b011, opcode=SYSTEM
  1203→                encode_i_type(*csr, rs1.encoding(), 0b011, rd.encoding(), OP_SYSTEM)
  1204→            }
  1205→            Instruction::Csrrwi { rd, csr, uimm } => {
  1206→                // CSRRWI: I-type, funct3=0b101, opcode=SYSTEM
  1207→                encode_i_type(*csr, *uimm & 0x1F, 0b101, rd.encoding(), OP_SYSTEM)
  1208→            }
  1209→            Instruction::Csrrsi { rd, csr, uimm } => {
  1210→                // CSRRSI: I-type, funct3=0b110, opcode=SYSTEM
  1211→                encode_i_type(*csr, *uimm & 0x1F, 0b110, rd.encoding(), OP_SYSTEM)
  1212→            }
  1213→            Instruction::Csrrci { rd, csr, uimm } => {
  1214→                // CSRRCI: I-type, funct3=0b111, opcode=SYSTEM
  1215→                encode_i_type(*csr, *uimm & 0x1F, 0b111, rd.encoding(), OP_SYSTEM)
  1216→            }
  1217→
  1218→            // ── Zifencei Extension ──────────────────────────────────
  1219→            Instruction::FenceI => {
  1220→                // FENCE.I: opcode=MISC-MEM, funct3=0b001, rd=0, rs1=0, imm=0
  1221→                encode_i_type(0, 0, 0b001, 0, OP_MISC_MEM)
  1222→            }
  1223→
  1224→            // ── System / Misc ───────────────────────────────────────
  1225→            Instruction::Ecall => {
  1226→                // ECALL = 0x00000073
  1227→                0x00000073u32.to_le_bytes()
  1228→            }
  1229→            Instruction::Ebreak => {
  1230→                // EBREAK = 0x00100073
  1231→                0x00100073u32.to_le_bytes()
  1232→            }
  1233→            Instruction::Fence { pred, succ } => {
  1234→                // FENCE: opcode=MISC-MEM, funct3=0b000, rd=0, rs1=0
  1235→                let imm = ((*pred & 0xF) << 4) | (*succ & 0xF);
  1236→                encode_i_type(imm, 0, 0b000, 0, OP_MISC_MEM)
  1237→            }
  1238→            Instruction::Nop => {
  1239→                // NOP = ADDI x0, x0, 0 = 0x00000013
  1240→                encode_i_type(0, 0, 0b000, 0, OP_IMM)
  1241→            }
  1242→        }
  1243→    }

  1328→    /// Decode a 32-bit RISC-V machine-code word into an `Instruction`.
  1329→    ///
  1330→    /// Returns `None` for encodings not yet covered by the decoder.
  1331→    /// Covers all instruction classes defined in this backend.
  1332→    pub fn decode(word: u32) -> Option<Instruction> {
  1333→        let opcode = word & 0x7F;
  1334→        let rd = (word >> 7) & 0x1F;
  1335→        let funct3 = (word >> 12) & 0x7;
  1336→        let rs1 = (word >> 15) & 0x1F;
  1337→        let rs2 = (word >> 20) & 0x1F;
  1338→        let funct7 = (word >> 25) & 0x7F;
  1339→
  1340→        match opcode {
  1341→            // ── LUI ────────────────────────────────────────────────
  1342→            0b0110111 => {
  1343→                let rd_reg = Gpr::from_encoding(rd)?;
  1344→                let imm = word & 0xFFFFF000;
  1345→                Some(Instruction::Lui { rd: rd_reg, imm })
  1346→            }
  1347→
  1348→            // ── AUIPC ──────────────────────────────────────────────
  1349→            0b0010111 => {
  1350→                let rd_reg = Gpr::from_encoding(rd)?;
  1351→                let imm = word & 0xFFFFF000;
  1352→                Some(Instruction::Auipc { rd: rd_reg, imm })
  1353→            }
  1354→
  1355→            // ── JAL ────────────────────────────────────────────────
  1356→            0b1101111 => {
  1357→                let rd_reg = Gpr::from_encoding(rd)?;
  1358→                let imm20 = ((word >> 31) & 1) << 20
  1359→                    | ((word >> 12) & 0xFF) << 12
  1360→                    | ((word >> 20) & 1) << 11
  1361→                    | ((word >> 21) & 0x3FF) << 1;
  1362→                let offset = ((imm20 << 11) as i32) >> 11;
  1363→                Some(Instruction::Jal { rd: rd_reg, offset })
  1364→            }
  1365→
  1366→            // ── JALR ───────────────────────────────────────────────
  1367→            0b1100111 => {
  1368→                let rd_reg = Gpr::from_encoding(rd)?;
  1369→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1370→                let imm = (((word >> 20) as i32) << 20) >> 20;
  1371→                Some(Instruction::Jalr {
  1372→                    rd: rd_reg,
  1373→                    rs1: rs1_reg,
  1374→                    imm,
  1375→                })
  1376→            }
  1377→
  1378→            // ── BRANCH ─────────────────────────────────────────────
  1379→            0b1100011 => {
  1380→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1381→                let rs2_reg = Gpr::from_encoding(rs2)?;
  1382→                let imm12 = ((word >> 31) & 1) << 12
  1383→                    | ((word >> 7) & 1) << 11
  1384→                    | ((word >> 25) & 0x3F) << 5
  1385→                    | ((word >> 8) & 0xF) << 1;
  1386→                let offset = ((imm12 << 19) as i32) >> 19;
  1387→                match funct3 {
  1388→                    0b000 => Some(Instruction::Beq {
  1389→                        rs1: rs1_reg,
  1390→                        rs2: rs2_reg,
  1391→                        offset,
  1392→                    }),
  1393→                    0b001 => Some(Instruction::Bne {
  1394→                        rs1: rs1_reg,
  1395→                        rs2: rs2_reg,
  1396→                        offset,
  1397→                    }),
  1398→                    0b100 => Some(Instruction::Blt {
  1399→                        rs1: rs1_reg,
  1400→                        rs2: rs2_reg,
  1401→                        offset,
  1402→                    }),
  1403→                    0b101 => Some(Instruction::Bge {
  1404→                        rs1: rs1_reg,
  1405→                        rs2: rs2_reg,
  1406→                        offset,
  1407→                    }),
  1408→                    0b110 => Some(Instruction::Bltu {
  1409→                        rs1: rs1_reg,
  1410→                        rs2: rs2_reg,
  1411→                        offset,
  1412→                    }),
  1413→                    0b111 => Some(Instruction::Bgeu {
  1414→                        rs1: rs1_reg,
  1415→                        rs2: rs2_reg,
  1416→                        offset,
  1417→                    }),
  1418→                    _ => None,
  1419→                }
  1420→            }
  1421→
  1422→            // ── LOAD ───────────────────────────────────────────────
  1423→            0b0000011 => {
  1424→                let rd_reg = Gpr::from_encoding(rd)?;
  1425→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1426→                let imm = (((word >> 20) as i32) << 20) >> 20;
  1427→                match funct3 {
  1428→                    0b000 => Some(Instruction::Lb {
  1429→                        rd: rd_reg,
  1430→                        rs1: rs1_reg,
  1431→                        imm,
  1432→                    }),
  1433→                    0b001 => Some(Instruction::Lh {
  1434→                        rd: rd_reg,
  1435→                        rs1: rs1_reg,
  1436→                        imm,
  1437→                    }),
  1438→                    0b010 => Some(Instruction::Lw {
  1439→                        rd: rd_reg,
  1440→                        rs1: rs1_reg,
  1441→                        imm,
  1442→                    }),
  1443→                    0b011 => Some(Instruction::Ld {
  1444→                        rd: rd_reg,
  1445→                        rs1: rs1_reg,
  1446→                        imm,
  1447→                    }),
  1448→                    0b100 => Some(Instruction::Lbu {
  1449→                        rd: rd_reg,
  1450→                        rs1: rs1_reg,
  1451→                        imm,
  1452→                    }),
  1453→                    0b101 => Some(Instruction::Lhu {
  1454→                        rd: rd_reg,
  1455→                        rs1: rs1_reg,
  1456→                        imm,
  1457→                    }),
  1458→                    0b110 => Some(Instruction::Lwu {
  1459→                        rd: rd_reg,
  1460→                        rs1: rs1_reg,
  1461→                        imm,
  1462→                    }),
  1463→                    _ => None,
  1464→                }
  1465→            }
  1466→
  1467→            // ── STORE ──────────────────────────────────────────────
  1468→            0b0100011 => {
  1469→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1470→                let rs2_reg = Gpr::from_encoding(rs2)?;
  1471→                let imm_lo = (word >> 7) & 0x1F;
  1472→                let imm_hi = (word >> 25) & 0x7F;
  1473→                let imm_raw = (imm_hi << 5) | imm_lo;
  1474→                let imm = ((imm_raw as i32) << 20) >> 20;
  1475→                match funct3 {
  1476→                    0b000 => Some(Instruction::Sb {
  1477→                        rs1: rs1_reg,
  1478→                        rs2: rs2_reg,
  1479→                        imm,
  1480→                    }),
  1481→                    0b001 => Some(Instruction::Sh {
  1482→                        rs1: rs1_reg,
  1483→                        rs2: rs2_reg,
  1484→                        imm,
  1485→                    }),
  1486→                    0b010 => Some(Instruction::Sw {
  1487→                        rs1: rs1_reg,
  1488→                        rs2: rs2_reg,
  1489→                        imm,
  1490→                    }),
  1491→                    0b011 => Some(Instruction::Sd {
  1492→                        rs1: rs1_reg,
  1493→                        rs2: rs2_reg,
  1494→                        imm,
  1495→                    }),
  1496→                    _ => None,
  1497→                }
  1498→            }
  1499→
  1500→            // ── OP-IMM (RV64I) ─────────────────────────────────────
  1501→            0b0010011 => {
  1502→                let rd_reg = Gpr::from_encoding(rd)?;
  1503→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1504→                let imm = (((word >> 20) as i32) << 20) >> 20;
  1505→                let shamt = (word >> 20) & 0x3F;
  1506→                match funct3 {
  1507→                    0b000 => Some(Instruction::Addi {
  1508→                        rd: rd_reg,
  1509→                        rs1: rs1_reg,

  1511→                    }),
  1512→                    0b010 => Some(Instruction::Slti {
  1513→                        rd: rd_reg,
  1514→                        rs1: rs1_reg,

  1516→                    }),
  1517→                    0b011 => Some(Instruction::Sltiu {
  1518→                        rd: rd_reg,
  1519→                        rs1: rs1_reg,
  1520→                        imm,
  1521→                    }),
  1522→                    0b100 => Some(Instruction::Xori {
  1523→                        rd: rd_reg,
  1524→                        rs1: rs1_reg,
  1525→                        imm,
  1526→                    }),
  1527→                    0b110 => Some(Instruction::Ori {
  1528→                        rd: rd_reg,
  1529→                        rs1: rs1_reg,
  1530→                        imm,
  1531→                    }),
  1532→                    0b111 => Some(Instruction::Andi {
  1533→                        rd: rd_reg,
  1534→                        rs1: rs1_reg,
  1535→                        imm,
  1536→                    }),
  1537→                    0b001 => Some(Instruction::Slli {
  1538→                        rd: rd_reg,
  1539→                        rs1: rs1_reg,
  1540→                        shamt,
  1541→                    }),
  1542→                    0b101 => {

  1555→                        }
  1556→                    }
  1557→                    _ => None,
  1558→                }
  1559→            }
  1560→
  1561→            // ── OP (RV64I register-register) ────────────────────────
  1562→            0b0110011 => {
  1563→                let rd_reg = Gpr::from_encoding(rd)?;
  1564→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1565→                let rs2_reg = Gpr::from_encoding(rs2)?;

  1568→                        rd: rd_reg,
  1569→                        rs1: rs1_reg,
  1570→                        rs2: rs2_reg,
  1571→                    }),

  1573→                        rd: rd_reg,
  1574→                        rs1: rs1_reg,
  1575→                        rs2: rs2_reg,
  1576→                    }),

  1579→                        rs1: rs1_reg,

  1584→                        rs1: rs1_reg,
  1585→                        rs2: rs2_reg,
  1586→                    }),
  1587→                    (0b0000000, 0b011) => Some(Instruction::Sltu {
  1588→                        rd: rd_reg,
  1589→                        rs1: rs1_reg,

  1593→                        rd: rd_reg,
  1594→                        rs1: rs1_reg,
  1595→                        rs2: rs2_reg,
  1596→                    }),
  1597→                    (0b0000000, 0b101) => Some(Instruction::Srl {
  1598→                        rd: rd_reg,
  1599→                        rs1: rs1_reg,
  1600→                        rs2: rs2_reg,
  1601→                    }),
  1602→                    (0b0100000, 0b101) => Some(Instruction::Sra {
  1603→                        rd: rd_reg,
  1604→                        rs1: rs1_reg,
  1605→                        rs2: rs2_reg,
  1606→                    }),

  1609→                        rs1: rs1_reg,
  1610→                        rs2: rs2_reg,
  1611→                    }),

  1614→                        rs1: rs1_reg,
  1615→                        rs2: rs2_reg,
  1616→                    }),

  1619→                        rd: rd_reg,
  1620→                        rs1: rs1_reg,

  1630→                        rs1: rs1_reg,
  1631→                        rs2: rs2_reg,
  1632→                    }),
  1633→                    (0b0000001, 0b011) => Some(Instruction::Mulhu {

  1635→                        rs1: rs1_reg,
  1636→                        rs2: rs2_reg,
  1637→                    }),
  1638→                    (0b0000001, 0b100) => Some(Instruction::Div {
  1639→                        rd: rd_reg,
  1640→                        rs1: rs1_reg,
  1641→                        rs2: rs2_reg,
  1642→                    }),
  1643→                    (0b0000001, 0b101) => Some(Instruction::Divu {
  1644→                        rd: rd_reg,
  1645→                        rs1: rs1_reg,
  1646→                        rs2: rs2_reg,

  1648→                    (0b0000001, 0b110) => Some(Instruction::Rem {
  1649→                        rd: rd_reg,
  1650→                        rs1: rs1_reg,

  1667→                let shamt = (word >> 20) & 0x1F;

  1669→                    0b000 => Some(Instruction::Addiw {
  1670→                        rd: rd_reg,
  1671→                        rs1: rs1_reg,
  1672→                        imm,
  1673→                    }),
  1674→                    0b001 => Some(Instruction::Slliw {
  1675→                        rd: rd_reg,
  1676→                        rs1: rs1_reg,
  1677→                        shamt,
  1678→                    }),
  1679→                    0b101 => {
  1680→                        if funct7 == 0b0100000 {
  1681→                            Some(Instruction::Sraiw {
  1682→                                rd: rd_reg,
  1683→                                rs1: rs1_reg,
  1684→                                shamt,
  1685→                            })
  1686→                        } else {
  1687→                            Some(Instruction::Srliw {
  1688→                                rd: rd_reg,
  1689→                                rs1: rs1_reg,
  1690→                                shamt,

  1693→                    }
  1694→                    _ => None,
  1695→                }
  1696→            }
  1697→

  1699→            0b0111011 => {
  1700→                let rd_reg = Gpr::from_encoding(rd)?;
  1701→                let rs1_reg = Gpr::from_encoding(rs1)?;
  1702→                let rs2_reg = Gpr::from_encoding(rs2)?;

  1704→                    (0b0000000, 0b000) => Some(Instruction::Addw {
  1705→                        rd: rd_reg,

  1707→                        rs2: rs2_reg,

  1710→                        rd: rd_reg,
  1711→                        rs1: rs1_reg,

  1713→                    }),

  1715→                        rd: rd_reg,

  1736→                    Some(Instruction::Ecall)
  1737→                } else if word == 0x00100073 {
  1738→                    Some(Instruction::Ebreak)

  1751→                            csr,

  1780→            0b0001111 => {

  1782→                    Some(Instruction::FenceI)
  1783→                } else {

  1785→                    let pred = (imm >> 4) & 0xF;
  1786→                    let succ = imm & 0xF;

  1788→                }
  1789→            }
  1790→

  1792→            0b1010011 => {
  1793→                let rd_fpr = Fpr::from_encoding(rd)?;
  1794→                let rs1_fpr = Fpr::from_encoding(rs1)?;
  1795→                let rs2_fpr = Fpr::from_encoding(rs2)?;

  1821→                    _ => None,
  1822→                }
  1823→            }
  1824→
  1825→            _ => None,
  1826→        }

  1941→
  1942→/// Build a minimal ELF64 binary for RISC-V 64-bit from raw code bytes.
  1943→///
  1944→/// Produces a static executable with 2 LOAD segments:
