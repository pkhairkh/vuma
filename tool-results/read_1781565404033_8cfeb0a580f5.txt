     1тЖТ//! # DWARF Debug Info Generation
     2тЖТ//!
     3тЖТ//! Produces DWARF version 4 debug information sections for all VUMA
     4тЖТ//! backends.  The emitted sections can be appended to an ELF binary so that
     5тЖТ//! tools such as `readelf`, `objdump`, and `gdb` can decode the program's
     6тЖТ//! structure, function boundaries, local variables, source-line mapping,
     7тЖТ//! and call frame information for stack unwinding.
     8тЖТ//!
     9тЖТ//! ## Sections Generated
    10тЖТ//!
    11тЖТ//! | Section          | Contents                                          |
    12тЖТ//! |------------------|---------------------------------------------------|
    13тЖТ//! | `.debug_abbrev`  | Abbreviation tables (tag + attribute encodings)   |
    14тЖТ//! | `.debug_info`    | Compilation unit DIEs (subprograms, variables)    |
    15тЖТ//! | `.debug_line`    | Line-number program (DWARF standard opcodes)      |
    16тЖТ//! | `.debug_frame`   | Call frame information (CIE + FDE entries)        |
    17тЖТ//!
    18тЖТ//! ## Multi-Backend Support
    19тЖТ//!
    20тЖТ//! The `DwarfBuilder` is parameterised by address size to support all
    21тЖТ//! eight VUMA backends:
    22тЖТ//!
    23тЖТ//! | Backend      | Address Size | Min Inst Length |
    24тЖТ//! |--------------|-------------|-----------------|
    25тЖТ//! | x86_64       | 8 bytes     | 1               |
    26тЖТ//! | AArch64      | 8 bytes     | 4               |
    27тЖТ//! | RISC-V 64    | 8 bytes     | 2               |
    28тЖТ//! | ARM32        | 4 bytes     | 2               |
    29тЖТ//! | MIPS64       | 8 bytes     | 4               |
    30тЖТ//! | PPC64        | 8 bytes     | 4               |
    31тЖТ//! | LoongArch64  | 8 bytes     | 4               |
    32тЖТ//! | Wasm32       | 4 bytes     | 1               |
    33тЖТ//!
    34тЖТ//! ## DWARF v4 Encoding
    35тЖТ//!
    36тЖТ//! - Abbreviation codes: `DW_TAG_COMPILE_UNIT`, `DW_TAG_SUBPROGRAM`,
    37тЖТ//!   `DW_TAG_VARIABLE`
    38тЖТ//! - Attributes: `DW_AT_NAME`, `DW_AT_LOW_PC`, `DW_AT_HIGH_PC`,
    39тЖТ//!   `DW_AT_TYPE`, `DW_AT_LOCATION`
    40тЖТ//! - Forms: `DW_FORM_STRING`, `DW_FORM_ADDR`, `DW_FORM_DATA4`,
    41тЖТ//!   `DW_FORM_EXPRLOC`
    42тЖТ//! - Line-number opcodes: `DW_LNS_COPY`, `DW_LNS_ADVANCE_PC`,
    43тЖТ//!   `DW_LNS_ADVANCE_LINE`, `DW_LNS_SET_FILE`, `DW_LNE_END_SEQUENCE`
    44тЖТ//! - Call frame: CIE with DW_CFA_def_cfa / DW_CFA_offset,
    45тЖТ//!   FDE per function with initial_location and address_range
    46тЖТ

    75тЖТconst DW_AT_HIGH_PC: u16 = 0x12;
    76тЖТ/// Language attribute.
    77тЖТconst DW_AT_LANGUAGE: u16 = 0x13;

    79тЖТconst DW_AT_PRODUCER: u16 = 0x25;

    83тЖТconst DW_AT_LOCATION: u16 = 0x02;

    98тЖТconst DW_FORM_DATA2: u16 = 0x05;

   100тЖТconst DW_FORM_EXPRLOC: u16 = 0x18;

   102тЖТconst DW_FORM_SEC_OFFSET: u16 = 0x17;

   105тЖТ

   107тЖТconst DW_CHILDREN_YES: u8 = 0x01;
   108тЖТ/// DIE has no children.
   109тЖТconst DW_CHILDREN_NO: u8 = 0x00;

   113тЖТ/// Copy current row to matrix.
   114тЖТconst DW_LNS_COPY: u8 = 0x01;

   116тЖТconst DW_LNS_ADVANCE_PC: u8 = 0x02;

   118тЖТconst DW_LNS_ADVANCE_LINE: u8 = 0x03;

   122тЖТ// -- Line Number Extended Opcodes --
   123тЖТ
   124тЖТ/// End sequence of line number entries.

   132тЖТ// -- Call Frame Information (CFI) constants --

   202тЖТ/// A register saved in the prologue (used for `.debug_frame` generation).
   203тЖТ#[derive(Debug, Clone)]
   204тЖТpub struct SavedRegister {
   205тЖТ    /// DWARF register number (architecture-specific encoding).
   206тЖТ    pub reg: u8,
   207тЖТ    /// Offset from the CFA where the register is saved.
   208тЖТ    pub cfa_offset: i32,

   211тЖТ/// A Frame Description Entry (FDE) for a single function.
   212тЖТ///
   213тЖТ/// Each FDE describes the call frame for one function, referencing the
   214тЖТ/// Common Information Entry (CIE) that shares the prologue pattern.
   215тЖТ#[derive(Debug, Clone)]
   216тЖТpub struct FrameDescriptorEntry {
   217тЖТ    /// Byte offset of this function within the text section.
   218тЖТ    pub initial_location: u64,
   219тЖТ    /// Size of the function in bytes.
   220тЖТ    pub address_range: u64,
   221тЖТ}
   222тЖТ
   223тЖТ/// A Common Information Entry (CIE) shared across all FDEs.
   224тЖТ///
   225тЖТ/// The CIE describes the prologue pattern that is common to all functions
   226тЖТ/// in the compilation unit: which register is the frame pointer, which
   227тЖТ/// registers are saved, and the stack pointer offset.
   228тЖТ#[derive(Debug, Clone)]
   229тЖТpub struct CommonInformationEntry {
   230тЖТ    /// DWARF register number for the stack pointer (CFA base register).
   231тЖТ    /// e.g. 31 for AArch64 SP (x31), 7 for x86_64 RSP.
   232тЖТ    pub cfa_reg: u8,
   233тЖТ    /// Offset of the CFA from the stack pointer register.
   234тЖТ    /// e.g. 0 means CFA = SP, 16 means CFA = SP + 16.
   235тЖТ    pub cfa_offset: u32,
   236тЖТ    /// Registers saved in the prologue and their offsets from CFA.
   237тЖТ    /// e.g. SavedRegister { reg: 30, cfa_offset: -8 } for AArch64 LR (x30).
   238тЖТ    pub saved_regs: Vec<SavedRegister>,
   239тЖТ    /// Code alignment factor (minimum instruction length in bytes).
   240тЖТ    pub code_alignment_factor: u8,
   241тЖТ    /// Data alignment factor (byte size of a stack slot, typically -4 or -8).
   242тЖТ    pub data_alignment_factor: i8,
   243тЖТ    /// Return address register number (e.g. 30 for AArch64 LR).
   244тЖТ    pub return_address_reg: u8,
   245тЖТ}

   251тЖТ/// Accumulates debug information during codegen and emits DWARF v4 sections.
   252тЖТ///
   253тЖТ/// The builder is parameterised by address size (4 for ARM32/Wasm32,
   254тЖТ/// 8 for all 64-bit targets) and minimum instruction length to correctly
   255тЖТ/// generate DWARF sections for each supported backend.
   256тЖТ///
   257тЖТ/// # Example
   258тЖТ///
   259тЖТ/// ```
   260тЖТ/// use vuma_codegen::dwarf::DwarfBuilder;
   261тЖТ///
   262тЖТ/// let mut db = DwarfBuilder::new(); // defaults to 64-bit, min_inst_length=4
   263тЖТ/// db.add_compile_unit("test.vuma", "vuma-codegen 0.1");
   264тЖТ/// db.add_subprogram("main", 0, 64);
   265тЖТ/// db.add_variable("x", "i64", -8, 0);
   266тЖТ/// db.add_line_entry(0, 1, 1, 0);
   267тЖТ/// db.add_line_entry(16, 1, 2, 4);
   268тЖТ/// db.set_cie_aarch64();
   269тЖТ/// db.add_fde("main", 0, 64);
   270тЖТ/// let sections = db.emit_debug_sections();
   271тЖТ/// assert!(!sections.debug_abbrev.is_empty());
   272тЖТ/// assert!(!sections.debug_info.is_empty());
   273тЖТ/// assert!(!sections.debug_line.is_empty());
   274тЖТ/// assert!(!sections.debug_frame.is_empty());
   275тЖТ/// ```
   276тЖТ///
   277тЖТ/// For 32-bit targets:
   278тЖТ///
   279тЖТ/// ```
   280тЖТ/// use vuma_codegen::dwarf::DwarfBuilder;
   281тЖТ///
   282тЖТ/// let mut db = DwarfBuilder::new_32bit(2); // ARM32: 4-byte addr, min_inst=2
   283тЖТ/// db.add_compile_unit("test.vuma", "vuma-codegen 0.1");
   284тЖТ/// db.add_subprogram("main", 0, 32);
   285тЖТ/// db.set_cie_arm32();
   286тЖТ/// db.add_fde("main", 0, 32);
   287тЖТ/// let sections = db.emit_debug_sections();
   288тЖТ/// assert!(!sections.debug_info.is_empty());
   289тЖТ/// assert!(!sections.debug_frame.is_empty());
   290тЖТ/// ```
   291тЖТ#[derive(Debug, Clone)]
   292тЖТpub struct DwarfBuilder {
   293тЖТ    /// Source file path for the compilation unit.
   294тЖТ    source_file: String,
   295тЖТ    /// Producer string (compiler name + version).
   296тЖТ    producer: String,
   297тЖТ    /// Recorded subprograms.
   298тЖТ    subprograms: Vec<Subprogram>,
   299тЖТ    /// Recorded local variables.
   300тЖТ    variables: Vec<Variable>,
   301тЖТ    /// Recorded line-number entries.
   302тЖТ    line_entries: Vec<LineEntry>,
   303тЖТ    /// Common Information Entry for call frame info.
   304тЖТ    cie: Option<CommonInformationEntry>,
   305тЖТ    /// Frame Description Entries (one per function).
   306тЖТ    fdes: Vec<FrameDescriptorEntry>,
   307тЖТ    /// Address size in bytes: 4 for ARM32/Wasm32, 8 for all 64-bit targets.
   308тЖТ    address_size: u8,
   309тЖТ    /// Minimum instruction length in bytes for the target.
   310тЖТ    min_inst_length: u8,
   311тЖТ}

   389тЖТ    /// | Backend      | address_size | min_inst_length |
   390тЖТ    /// |--------------|-------------|-----------------|
   391тЖТ    /// | x86_64       | 8           | 1               |
   392тЖТ    /// | AArch64      | 8           | 4               |
   393тЖТ    /// | RISC-V 64    | 8           | 2               |

   401тЖТ        let (addr_size, min_inst) = match backend {
   402тЖТ            BackendKind::X86_64       => (8, 1),
   403тЖТ            BackendKind::AArch64      => (8, 4),

   408тЖТ            BackendKind::LoongArch64  => (8, 4),

   651тЖТ    /// Emit the `.debug_abbrev` section.
   652тЖТ    ///
   653тЖТ    /// Contains three abbreviation entries:
   654тЖТ    /// - Abbrev 1: `DW_TAG_COMPILE_UNIT` (has children)
   655тЖТ    ///   - `DW_AT_NAME` / `DW_FORM_STRING`
   656тЖТ    ///   - `DW_AT_LANGUAGE` / `DW_FORM_DATA2`
   657тЖТ    ///   - `DW_AT_PRODUCER` / `DW_FORM_STRING`
   658тЖТ    ///   - `DW_AT_LOW_PC` / `DW_FORM_ADDR`
   659тЖТ    ///   - `DW_AT_HIGH_PC` / `DW_FORM_DATA4`
   660тЖТ    ///   - `DW_AT_STMT_LIST` / `DW_FORM_SEC_OFFSET`
   661тЖТ    ///
   662тЖТ    /// - Abbrev 2: `DW_TAG_SUBPROGRAM` (has children)
   663тЖТ    ///   - `DW_AT_NAME` / `DW_FORM_STRING`
   664тЖТ    ///   - `DW_AT_LOW_PC` / `DW_FORM_ADDR`
   665тЖТ    ///   - `DW_AT_HIGH_PC` / `DW_FORM_DATA4`
   666тЖТ    ///
   667тЖТ    /// - Abbrev 3: `DW_TAG_VARIABLE` (no children)
   668тЖТ    ///   - `DW_AT_NAME` / `DW_FORM_STRING`
   669тЖТ    ///   - `DW_AT_TYPE` / `DW_FORM_STRING`
   670тЖТ    ///   - `DW_AT_LOCATION` / `DW_FORM_EXPRLOC`
   671тЖТ    fn emit_debug_abbrev(&self) -> Vec<u8> {
   672тЖТ        let mut buf = Vec::new();
   673тЖТ
   674тЖТ        // Abbrev 1: DW_TAG_COMPILE_UNIT
   675тЖТ        encode_uleb128(&mut buf, 1); // abbreviation code
   676тЖТ        encode_uleb128(&mut buf, DW_TAG_COMPILE_UNIT as u64);
   677тЖТ        buf.push(DW_CHILDREN_YES);
   678тЖТ        // DW_AT_NAME, DW_FORM_STRING
   679тЖТ        encode_uleb128(&mut buf, DW_AT_NAME as u64);
   680тЖТ        encode_uleb128(&mut buf, DW_FORM_STRING as u64);
   681тЖТ        // DW_AT_LANGUAGE, DW_FORM_DATA2
   682тЖТ        encode_uleb128(&mut buf, DW_AT_LANGUAGE as u64);
   683тЖТ        encode_uleb128(&mut buf, DW_FORM_DATA2 as u64);
   684тЖТ        // DW_AT_PRODUCER, DW_FORM_STRING
   685тЖТ        encode_uleb128(&mut buf, DW_AT_PRODUCER as u64);
   686тЖТ        encode_uleb128(&mut buf, DW_FORM_STRING as u64);
   687тЖТ        // DW_AT_LOW_PC, DW_FORM_ADDR
   688тЖТ        encode_uleb128(&mut buf, DW_AT_LOW_PC as u64);
   689тЖТ        encode_uleb128(&mut buf, DW_FORM_ADDR as u64);
   690тЖТ        // DW_AT_HIGH_PC, DW_FORM_DATA4
   691тЖТ        encode_uleb128(&mut buf, DW_AT_HIGH_PC as u64);
   692тЖТ        encode_uleb128(&mut buf, DW_FORM_DATA4 as u64);
   693тЖТ        // DW_AT_STMT_LIST, DW_FORM_SEC_OFFSET
   694тЖТ        encode_uleb128(&mut buf, DW_AT_STMT_LIST as u64);
   695тЖТ        encode_uleb128(&mut buf, DW_FORM_SEC_OFFSET as u64);
   696тЖТ        // Terminator
   697тЖТ        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   698тЖТ        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   699тЖТ
   700тЖТ        // Abbrev 2: DW_TAG_SUBPROGRAM
   701тЖТ        encode_uleb128(&mut buf, 2);
   702тЖТ        encode_uleb128(&mut buf, DW_TAG_SUBPROGRAM as u64);
   703тЖТ        buf.push(DW_CHILDREN_YES);
   704тЖТ        // DW_AT_NAME, DW_FORM_STRING
   705тЖТ        encode_uleb128(&mut buf, DW_AT_NAME as u64);
   706тЖТ        encode_uleb128(&mut buf, DW_FORM_STRING as u64);
   707тЖТ        // DW_AT_LOW_PC, DW_FORM_ADDR
   708тЖТ        encode_uleb128(&mut buf, DW_AT_LOW_PC as u64);
   709тЖТ        encode_uleb128(&mut buf, DW_FORM_ADDR as u64);
   710тЖТ        // DW_AT_HIGH_PC, DW_FORM_DATA4
   711тЖТ        encode_uleb128(&mut buf, DW_AT_HIGH_PC as u64);
   712тЖТ        encode_uleb128(&mut buf, DW_FORM_DATA4 as u64);
   713тЖТ        // Terminator
   714тЖТ        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   715тЖТ        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   716тЖТ
   717тЖТ        // Abbrev 3: DW_TAG_VARIABLE
   718тЖТ        encode_uleb128(&mut buf, 3);
   719тЖТ        encode_uleb128(&mut buf, DW_TAG_VARIABLE as u64);
   720тЖТ        buf.push(DW_CHILDREN_NO);
   721тЖТ        // DW_AT_NAME, DW_FORM_STRING
   722тЖТ        encode_uleb128(&mut buf, DW_AT_NAME as u64);
   723тЖТ        encode_uleb128(&mut buf, DW_FORM_STRING as u64);
   724тЖТ        // DW_AT_TYPE, DW_FORM_STRING (simplified: type name as string)
   725тЖТ        encode_uleb128(&mut buf, DW_AT_TYPE as u64);
   726тЖТ        encode_uleb128(&mut buf, DW_FORM_STRING as u64);
   727тЖТ        // DW_AT_LOCATION, DW_FORM_EXPRLOC
   728тЖТ        encode_uleb128(&mut buf, DW_AT_LOCATION as u64);
   729тЖТ        encode_uleb128(&mut buf, DW_FORM_EXPRLOC as u64);
   730тЖТ        // Terminator
   731тЖТ        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   732тЖТ        encode_uleb128(&mut buf, DW_FORM_NONE as u64);
   733тЖТ
   734тЖТ        // End of abbreviation table
   735тЖТ        encode_uleb128(&mut buf, 0);
   736тЖТ
   737тЖТ        buf
   738тЖТ    }

   743тЖТ
   744тЖТ    /// Emit the `.debug_info` section.
   745тЖТ    ///
   746тЖТ    /// Contains a single compilation unit DIE with nested subprogram and
   747тЖТ    /// variable DIEs.
   748тЖТ    fn emit_debug_info(&self, _debug_abbrev: &[u8], _debug_line: &[u8]) -> Vec<u8> {
   749тЖТ        let mut die_buf = Vec::new();
   750тЖТ
   751тЖТ        // -- Compile Unit DIE (abbrev 1) --
   752тЖТ        encode_uleb128(&mut die_buf, 1); // abbrev code
   753тЖТ                                         // DW_AT_NAME (DW_FORM_STRING)
   754тЖТ        write_null_string(&mut die_buf, &self.source_file);
   755тЖТ        // DW_AT_LANGUAGE (DW_FORM_DATA2)
   756тЖТ        die_buf.extend_from_slice(&DW_LANG_VUMA.to_le_bytes());
   757тЖТ        // DW_AT_PRODUCER (DW_FORM_STRING)
   758тЖТ        write_null_string(&mut die_buf, &self.producer);
   759тЖТ        // DW_AT_LOW_PC (DW_FORM_ADDR)
   760тЖТ        let cu_low_pc = self
   761тЖТ            .subprograms
   762тЖТ            .iter()
   763тЖТ            .map(|s| s.start_offset)
   764тЖТ            .min()
   765тЖТ            .unwrap_or(0);
   766тЖТ        write_address(&mut die_buf, cu_low_pc, self.address_size);
   767тЖТ        // DW_AT_HIGH_PC (DW_FORM_DATA4) тАФ offset from low_pc
   768тЖТ        let cu_high_pc: u32 = self
   769тЖТ            .subprograms
   770тЖТ            .iter()
   771тЖТ            .map(|s| s.end_offset)
   772тЖТ            .max()
   773тЖТ            .unwrap_or(0) as u32
   774тЖТ            - cu_low_pc as u32;
   775тЖТ        die_buf.extend_from_slice(&cu_high_pc.to_le_bytes());
   776тЖТ        // DW_AT_STMT_LIST (DW_FORM_SEC_OFFSET) тАФ offset into .debug_line
   777тЖТ        die_buf.extend_from_slice(&0u32.to_le_bytes());
   778тЖТ
   779тЖТ        // -- Subprogram DIEs (abbrev 2) --
   780тЖТ        for sub in &self.subprograms {
   781тЖТ            encode_uleb128(&mut die_buf, 2); // abbrev code
   782тЖТ                                             // DW_AT_NAME (DW_FORM_STRING)
   783тЖТ            write_null_string(&mut die_buf, &sub.name);
   784тЖТ            // DW_AT_LOW_PC (DW_FORM_ADDR)
   785тЖТ            write_address(&mut die_buf, sub.start_offset, self.address_size);
   786тЖТ            // DW_AT_HIGH_PC (DW_FORM_DATA4) тАФ offset from low_pc
   787тЖТ            let size = (sub.end_offset - sub.start_offset) as u32;
   788тЖТ            die_buf.extend_from_slice(&size.to_le_bytes());
   789тЖТ
   790тЖТ            // -- Variable DIEs (abbrev 3) тАФ nested inside subprogram --
   791тЖТ            for var in &self.variables {
   792тЖТ                encode_uleb128(&mut die_buf, 3); // abbrev code
   793тЖТ                                                 // DW_AT_NAME (DW_FORM_STRING)
   794тЖТ                write_null_string(&mut die_buf, &var.name);
   795тЖТ                // DW_AT_TYPE (DW_FORM_STRING)
   796тЖТ                write_null_string(&mut die_buf, &var.type_name);
   797тЖТ                // DW_AT_LOCATION (DW_FORM_EXPRLOC)
   798тЖТ                // DW_OP_fbreg <offset>  =  0x91 + signed LEB128
   799тЖТ                let expr = build_fbreg_expr(var.offset);
   800тЖТ                encode_uleb128(&mut die_buf, expr.len() as u64);
   801тЖТ                die_buf.extend_from_slice(&expr);
   802тЖТ            }
   803тЖТ
   804тЖТ            // Null terminator for subprogram children
   805тЖТ            encode_uleb128(&mut die_buf, 0);
   806тЖТ        }
   807тЖТ
   808тЖТ        // Null terminator for compile_unit children
   809тЖТ        encode_uleb128(&mut die_buf, 0);
   810тЖТ
   811тЖТ        // Now build the full .debug_info section with the compilation unit
   812тЖТ        // header.
   813тЖТ        //
   814тЖТ        // DWARF v4 .debug_info header:
   815тЖТ        //   unit_length       : 4 bytes
   816тЖТ        //   version           : 2 bytes (4)
   817тЖТ        //   debug_abbrev_offset: 4 bytes
   818тЖТ        //   address_size      : 1 byte
   819тЖТ        let mut buf = Vec::new();
   820тЖТ
   821тЖТ        let unit_length = (die_buf.len() as u32)
   822тЖТ            // header fields after the unit_length itself:
   823тЖТ            + 2  // version
   824тЖТ            + 4  // debug_abbrev_offset
   825тЖТ            + 1; // address_size
   826тЖТ        buf.extend_from_slice(&unit_length.to_le_bytes());
   827тЖТ        // Version
   828тЖТ        buf.extend_from_slice(&DWARF_VERSION.to_le_bytes());
   829тЖТ        // Debug abbrev offset (always 0 тАФ we have a single abbrev table)
   830тЖТ        buf.extend_from_slice(&0u32.to_le_bytes());
   831тЖТ        // Address size
   832тЖТ        buf.push(self.address_size);
   833тЖТ
   834тЖТ        // DIE data
   835тЖТ        buf.extend_from_slice(&die_buf);
   836тЖТ
   837тЖТ        buf
   838тЖТ    }
   839тЖТ
   840тЖТ    // -----------------------------------------------------------------------
   841тЖТ    // .debug_line
   842тЖТ    // -----------------------------------------------------------------------
   843тЖТ
   844тЖТ    /// Emit the `.debug_line` section containing a DWARF v4 line-number program.
   845тЖТ    fn emit_debug_line(&self) -> Vec<u8> {
   846тЖТ        let mut program = Vec::new();
   847тЖТ
   848тЖТ        // Sort line entries by offset for a well-formed program.
   849тЖТ        let mut entries = self.line_entries.clone();
   850тЖТ        entries.sort_by_key(|e| e.offset);
   851тЖТ
   852тЖТ        // Default_is_stmt: true (beginning of a statement)
   853тЖТ        let default_is_stmt: u8 = 1;
   854тЖТ        // Line base: -5 (standard DWARF range)
   855тЖТ        let line_base: i8 = -5;
   856тЖТ        // Line range: 14 (standard DWARF value)
   857тЖТ        let line_range: u8 = 14;
   858тЖТ        // Opcode base: 13 (standard DWARF value)
   859тЖТ        let opcode_base: u8 = 13;
   860тЖТ        // Standard opcode lengths for opcodes 1..12
   861тЖТ        let standard_opcode_lengths: [u8; 12] = [0, 1, 1, 1, 1, 0, 0, 0, 1, 0, 0, 1];
   862тЖТ        // Minimum instruction length: target-dependent
   863тЖТ        let min_inst_length = self.min_inst_length;
   864тЖТ
   865тЖТ        // -- Build line program opcodes --
   866тЖТ        let mut current_line: i32 = 1;
   867тЖТ        let mut current_file: u32 = 1;
   868тЖТ        let mut current_offset: u64 = 0;
   869тЖТ
   870тЖТ        for entry in &entries {
   871тЖТ            // Set file if different
   872тЖТ            if entry.file != current_file {
   873тЖТ                program.push(DW_LNS_SET_FILE);
   874тЖТ                encode_uleb128(&mut program, entry.file as u64);
   875тЖТ                current_file = entry.file;
   876тЖТ            }
   877тЖТ
   878тЖТ            // Advance line if different
   879тЖТ            let line_diff = (entry.line as i32) - current_line;
   880тЖТ            if line_diff != 0 {
   881тЖТ                program.push(DW_LNS_ADVANCE_LINE);
   882тЖТ                encode_sleb128(&mut program, i64::from(line_diff));
   883тЖТ                current_line = entry.line as i32;
   884тЖТ            }
   885тЖТ
   886тЖТ            // Advance PC
   887тЖТ            if entry.offset > current_offset {
   888тЖТ                let addr_advance = entry.offset - current_offset;
   889тЖТ                // DW_LNS_ADVANCE_PC advances by the operand * min_inst_length
   890тЖТ                let op_advance = addr_advance / (min_inst_length as u64);
   891тЖТ                if op_advance > 0 {
   892тЖТ                    program.push(DW_LNS_ADVANCE_PC);
   893тЖТ                    encode_uleb128(&mut program, op_advance);
   894тЖТ                }
   895тЖТ                current_offset = entry.offset;
   896тЖТ            }
   897тЖТ
   898тЖТ            // Copy current row
   899тЖТ            program.push(DW_LNS_COPY);
   900тЖТ        }
   901тЖТ
   902тЖТ        // End sequence: advance PC to end and emit DW_LNE_END_SEQUENCE
   903тЖТ        let end_addr = entries.last().map(|e| e.end_offset()).unwrap_or(0);
   904тЖТ        if end_addr > current_offset {
   905тЖТ            let op_advance = (end_addr - current_offset) / (min_inst_length as u64);
   906тЖТ            if op_advance > 0 {
   907тЖТ                program.push(DW_LNS_ADVANCE_PC);
   908тЖТ                encode_uleb128(&mut program, op_advance);
   909тЖТ            }
   910тЖТ        }
   911тЖТ        // DW_LNE_END_SEQUENCE: extended opcode
   912тЖТ        let ext_opcode_bytes = [DW_LNE_END_SEQUENCE];
   913тЖТ        encode_uleb128(&mut program, (ext_opcode_bytes.len() as u64) + 1); // length (includes sub-opcode)
   914тЖТ        program.push(DW_LNE_END_SEQUENCE);
   915тЖТ
   916тЖТ        // -- Build DWARF v4 directory and file tables --
   917тЖТ        //
   918тЖТ        // DWARF v4 uses a simple null-terminated list format:
   919тЖТ        //   include_directories: sequence of null-terminated strings,
   920тЖТ        //       terminated by a single null byte (empty string).
   921тЖТ        //   file_names: sequence of entries, each containing:
   922тЖТ        //       name (null-terminated string),
   923тЖТ        //       directory index (ULEB128),
   924тЖТ        //       time of last modification (ULEB128),
   925тЖТ        //       length in bytes (ULEB128),
   926тЖТ        //     terminated by a single null byte.
   927тЖТ
   928тЖТ        let directory_table = b".\0"; // one directory: "."
   929тЖТ
   930тЖТ        let mut file_table = Vec::new();
   931тЖТ        // File entry: name, dir_index, time, size
   932тЖТ        file_table.extend_from_slice(self.source_file.as_bytes());
   933тЖТ        file_table.push(0); // null-terminated name
   934тЖТ        encode_uleb128(&mut file_table, 0); // directory index 0
   935тЖТ        encode_uleb128(&mut file_table, 0); // time (unknown)
   936тЖТ        encode_uleb128(&mut file_table, 0); // size (unknown)
   937тЖТ
   938тЖТ        // -- Assemble the DWARF v4 line number program header --
   939тЖТ        let mut header = Vec::new();
   940тЖТ
   941тЖТ        // Version (2 bytes)
   942тЖТ        header.extend_from_slice(&DWARF_VERSION.to_le_bytes());
   943тЖТ
   944тЖТ        // Header length тАФ will be filled after we know the size
   945тЖТ        let header_before_length_field = header.len();
   946тЖТ        // Placeholder for 4-byte header length
   947тЖТ        header.extend_from_slice(&0u32.to_le_bytes());
   948тЖТ
   949тЖТ        // Minimum instruction length
   950тЖТ        header.push(min_inst_length);
   951тЖТ        // Default is_stmt
   952тЖТ        header.push(default_is_stmt);
   953тЖТ        // Line base
   954тЖТ        header.push(line_base as u8);
   955тЖТ        // Line range
   956тЖТ        header.push(line_range);
   957тЖТ        // Opcode base
   958тЖТ        header.push(opcode_base);
   959тЖТ        // Standard opcode lengths
   960тЖТ        header.extend_from_slice(&standard_opcode_lengths);
   961тЖТ
   962тЖТ        // Include directories (DWARF v4 format)
   963тЖТ        header.extend_from_slice(directory_table);
   964тЖТ        header.push(0); // terminator (empty string)
   965тЖТ
   966тЖТ        // File names (DWARF v4 format)
   967тЖТ        header.extend_from_slice(&file_table);
   968тЖТ        header.push(0); // terminator (empty string = null byte)
   969тЖТ
   970тЖТ        // Fix up header_length (bytes after the header_length field itself)
   971тЖТ        let header_length = (header.len() - header_before_length_field - 4) as u32;
   972тЖТ        let hl_offset = header_before_length_field;
   973тЖТ        header[hl_offset..hl_offset + 4].copy_from_slice(&header_length.to_le_bytes());
   974тЖТ
   975тЖТ        // -- Assemble full .debug_line section --
   976тЖТ        let mut buf = Vec::new();
   977тЖТ
   978тЖТ        // Unit length (4 bytes) тАФ total length of the rest of the unit
   979тЖТ        let unit_length = (header.len() + program.len()) as u32;
   980тЖТ        buf.extend_from_slice(&unit_length.to_le_bytes());
   981тЖТ        buf.extend_from_slice(&header);
   982тЖТ        buf.extend_from_slice(&program);
   983тЖТ
   984тЖТ        buf
   985тЖТ    }
   986тЖТ
   987тЖТ    // -----------------------------------------------------------------------
   988тЖТ    // .debug_frame
   989тЖТ    // -----------------------------------------------------------------------
   990тЖТ
   991тЖТ    /// Emit the `.debug_frame` section containing a CIE and FDEs.
   992тЖТ    ///
   993тЖТ    /// The `.debug_frame` section describes how to unwind the stack at every
   994тЖТ    /// point in the program.  It consists of:
   995тЖТ    ///
   996тЖТ    /// 1. A **Common Information Entry (CIE)** that describes the prologue
   997тЖТ    ///    pattern shared by all functions: which register is the CFA base,
   998тЖТ    ///    which registers are saved, and the alignment factors.
   999тЖТ    ///
  1000тЖТ    /// 2. One **Frame Description Entry (FDE)** per function, referencing the
  1001тЖТ    ///    CIE and specifying the function's address range.
  1002тЖТ    ///
  1003тЖТ    /// If no CIE has been configured (via `set_cie_aarch64`, etc.), this
  1004тЖТ    /// method returns an empty section.
  1005тЖТ    fn emit_debug_frame(&self) -> Vec<u8> {
  1006тЖТ        let cie_ref = match &self.cie {
  1007тЖТ            Some(c) => c,
  1008тЖТ            None => return Vec::new(),
  1009тЖТ        };
  1010тЖТ
  1011тЖТ        let mut buf = Vec::new();
  1012тЖТ
  1013тЖТ        // ---- CIE (Common Information Entry) ----
  1014тЖТ        //
  1015тЖТ        // Format (DWARF v4, .debug_frame):
  1016тЖТ        //   length            : 4 bytes (32-bit DWARF) тАФ length of remaining CIE data
  1017тЖТ        //   CIE_id            : 4 bytes (0xFFFFFFFF)
  1018тЖТ        //   version           : 1 byte  (1 for DWARF v4 CFI)
  1019тЖТ        //   augmentation      : null-terminated string ("")
  1020тЖТ        //   address_size      : 1 byte  (not in DWARF v2; present in v4)
  1021тЖТ        //   segment_size      : 1 byte  (not in DWARF v2; present in v4)
  1022тЖТ        //   code_alignment    : ULEB128
  1023тЖТ        //   data_alignment    : SLEB128
  1024тЖТ        //   return_address_reg: ULEB128
  1025тЖТ        //   initial_instructions: CFI instructions
  1026тЖТ
  1027тЖТ        let mut cie_body = Vec::new();
  1028тЖТ
  1029тЖТ        // Version
  1030тЖТ        cie_body.push(DW_CIE_VERSION);
  1031тЖТ
  1032тЖТ        // Augmentation string (empty)
  1033тЖТ        cie_body.push(0);
  1034тЖТ
  1035тЖТ        // Address size (DWARF v4 .debug_frame)
  1036тЖТ        cie_body.push(self.address_size);
  1037тЖТ
  1038тЖТ        // Segment selector size (0)
  1039тЖТ        cie_body.push(0);
  1040тЖТ
  1041тЖТ        // Code alignment factor
  1042тЖТ        encode_uleb128(&mut cie_body, cie_ref.code_alignment_factor as u64);
  1043тЖТ
  1044тЖТ        // Data alignment factor
  1045тЖТ        encode_sleb128(&mut cie_body, cie_ref.data_alignment_factor as i64);
  1046тЖТ
  1047тЖТ        // Return address register
  1048тЖТ        encode_uleb128(&mut cie_body, cie_ref.return_address_reg as u64);
  1049тЖТ
  1050тЖТ        // Initial CFI instructions for the CIE:
  1051тЖТ        // DW_CFA_def_cfa: defines CFA = register + offset
  1052тЖТ        cie_body.push(DW_CFA_DEF_CFA);
  1053тЖТ        encode_uleb128(&mut cie_body, cie_ref.cfa_reg as u64);
  1054тЖТ        encode_uleb128(&mut cie_body, cie_ref.cfa_offset as u64);
  1055тЖТ
  1056тЖТ        // DW_CFA_offset for each saved register
  1057тЖТ        for saved in &cie_ref.saved_regs {
  1058тЖТ            // DW_CFA_offset encodes: (opcode | reg) + ULEB128(offset / data_alignment)
  1059тЖТ            // The opcode is (DW_CFA_OFFSET | (reg & 0x3F))
  1060тЖТ            // The offset factored by data_alignment_factor.
  1061тЖТ            // Since data_alignment_factor is negative (e.g., -8), we need
  1062тЖТ            // offset / |data_alignment_factor| as a positive ULEB128.
  1063тЖТ            let abs_data_align = cie_ref.data_alignment_factor.unsigned_abs() as i32;
  1064тЖТ            let factored_offset = if abs_data_align != 0 {
  1065тЖТ                saved.cfa_offset.abs() / abs_data_align
  1066тЖТ            } else {
  1067тЖТ                saved.cfa_offset.abs()
  1068тЖТ            };
  1069тЖТ            let reg_low6 = saved.reg & 0x3F;
  1070тЖТ            cie_body.push(DW_CFA_OFFSET | reg_low6);
  1071тЖТ            encode_uleb128(&mut cie_body, factored_offset as u64);
  1072тЖТ        }
  1073тЖТ
  1074тЖТ        // Compute CIE length: everything after the length field itself
  1075тЖТ        // length (4) + CIE_id (4) are not counted in the length field
  1076тЖТ        let cie_length = (cie_body.len() + 4) as u32; // +4 for CIE_id
  1077тЖТ
  1078тЖТ        // Write CIE to output
  1079тЖТ        buf.extend_from_slice(&cie_length.to_le_bytes());
  1080тЖТ        // CIE_id = 0xFFFFFFFF (32-bit DWARF)
  1081тЖТ        buf.extend_from_slice(&CIE_ID_32.to_le_bytes());
  1082тЖТ        // CIE body
  1083тЖТ        buf.extend_from_slice(&cie_body);
  1084тЖТ
  1085тЖТ        // ---- FDEs (Frame Description Entries) ----
  1086тЖТ        //
  1087тЖТ        // Format (DWARF v4, .debug_frame):
  1088тЖТ        //   length            : 4 bytes
  1089тЖТ        //   CIE_pointer       : 4 bytes (offset from start of .debug_frame to CIE)
  1090тЖТ        //   initial_location  : address_size bytes
  1091тЖТ        //   address_range     : address_size bytes
  1092тЖТ        //   instructions      : CFI instructions (if any per-function overrides)
  1093тЖТ
  1094тЖТ        let cie_offset_from_start: u32 = 0; // CIE is the first entry
  1095тЖТ
  1096тЖТ        for fde in &self.fdes {
  1097тЖТ            let mut fde_body = Vec::new();
  1098тЖТ
  1099тЖТ            // CIE pointer (offset to the CIE from the beginning of .debug_frame)
  1100тЖТ            fde_body.extend_from_slice(&cie_offset_from_start.to_le_bytes());
  1101тЖТ
  1102тЖТ            // Initial location (address-size dependent)
  1103тЖТ            write_address(&mut fde_body, fde.initial_location, self.address_size);
  1104тЖТ
  1105тЖТ            // Address range (address-size dependent)
  1106тЖТ            write_address(&mut fde_body, fde.address_range, self.address_size);
  1107тЖТ
  1108тЖТ            // Per-function CFI instructions (none for now тАФ the CIE covers
  1109тЖТ            // the prologue pattern and GDB/LLDB can infer the rest from
  1110тЖТ            // the default CFA rule). Future work: emit DW_CFA_advance_loc
  1111тЖТ            // and DW_CFA_def_cfa_offset for dynamic frame size changes
  1112тЖТ            // within the function body.
  1113тЖТ
  1114тЖТ            // Compute FDE length: everything after the length field itself
  1115тЖТ            // length (4) not counted
  1116тЖТ            let fde_length = fde_body.len() as u32;
  1117тЖТ
  1118тЖТ            // Write FDE to output
  1119тЖТ            buf.extend_from_slice(&fde_length.to_le_bytes());
  1120тЖТ            buf.extend_from_slice(&fde_body);
  1121тЖТ        }
  1122тЖТ
  1123тЖТ        buf
  1124тЖТ    }
  1125тЖТ}

  1137тЖТ/// Encode an unsigned LEB128 value into `buf`.
  1138тЖТfn encode_uleb128(buf: &mut Vec<u8>, mut value: u64) {
  1139тЖТ    loop {
  1140тЖТ        let mut byte = (value & 0x7F) as u8;
  1141тЖТ        value >>= 7;
  1142тЖТ        if value != 0 {
  1143тЖТ            byte |= 0x80;
  1144тЖТ        }
  1145тЖТ        buf.push(byte);
  1146тЖТ        if value == 0 {
  1147тЖТ            break;
  1148тЖТ        }
  1149тЖТ    }
  1150тЖТ}
  1151тЖТ
  1152тЖТ/// Encode a signed LEB128 value into `buf`.
  1153тЖТfn encode_sleb128(buf: &mut Vec<u8>, mut value: i64) {
  1154тЖТ    loop {
  1155тЖТ        let mut byte = (value & 0x7F) as u8;
  1156тЖТ        value >>= 7;
  1157тЖТ        let sign_bit_set = (byte & 0x40) != 0;
  1158тЖТ        let done = value == 0 || (value == -1 && sign_bit_set);
  1159тЖТ        if !done {
  1160тЖТ            byte |= 0x80;
  1161тЖТ        }
  1162тЖТ        buf.push(byte);
  1163тЖТ        if done {
  1164тЖТ            break;
  1165тЖТ        }
  1166тЖТ    }
  1167тЖТ}
  1168тЖТ
  1169тЖТ/// Write a null-terminated string into `buf`.
  1170тЖТfn write_null_string(buf: &mut Vec<u8>, s: &str) {
  1171тЖТ    buf.extend_from_slice(s.as_bytes());
  1172тЖТ    buf.push(0);
  1173тЖТ}
  1174тЖТ
  1175тЖТ/// Write an address value into `buf` with the given address size (4 or 8 bytes).
  1176тЖТfn write_address(buf: &mut Vec<u8>, addr: u64, address_size: u8) {
  1177тЖТ    match address_size {
  1178тЖТ        4 => buf.extend_from_slice(&(addr as u32).to_le_bytes()),
  1179тЖТ        _ => buf.extend_from_slice(&addr.to_le_bytes()),
  1180тЖТ    }
  1181тЖТ}

  1183тЖТ/// Build a `DW_OP_fbreg <offset>` expression.
  1184тЖТ///
  1185тЖТ/// `DW_OP_fbreg` (0x91) followed by a signed LEB128 offset.
  1186тЖТfn build_fbreg_expr(offset: i32) -> Vec<u8> {
  1187тЖТ    let mut expr = vec![0x91]; // DW_OP_fbreg
  1188тЖТ    encode_sleb128(&mut expr, offset as i64);
  1189тЖТ    expr
  1190тЖТ}

  1196тЖТimpl LineEntry {

  1209тЖТ/// Section header type for DWARF debug info sections (progbits).
  1210тЖТconst SHT_PROGBITS: u32 = 1;
  1211тЖТ
  1212тЖТ/// Append DWARF debug sections to an ELF binary.
  1213тЖТ///
  1214тЖТ/// Given an existing ELF binary and the debug sections, this function inserts
  1215тЖТ/// the four debug sections (`.debug_abbrev`, `.debug_info`, `.debug_line`,
  1216тЖТ/// `.debug_frame`) into the ELF, updates the section header table, and patches
  1217тЖТ/// the section header string table and section count.
  1218тЖТ///
  1219тЖТ/// If `.debug_frame` is empty (no CIE configured), it is simply omitted from
  1220тЖТ/// the ELF, keeping the binary compact.
  1221тЖТ///
  1222тЖТ/// This is a simplified integration that appends sections before the section
  1223тЖТ/// headers. It assumes the input ELF was produced by `emit_elf` with
  1224тЖТ/// section headers enabled.
  1225тЖТpub fn append_debug_sections_to_elf(elf: &mut Vec<u8>, debug: &DebugSections) {
  1226тЖТ    // Parse the ELF header to find section header table location and count.
  1227тЖТ    if elf.len() < 64 {
  1228тЖТ        return;
  1229тЖТ    }
  1230тЖТ
  1231тЖТ    let e_shoff = u64::from_le_bytes(elf[40..48].try_into().unwrap_or([0; 8]));
  1232тЖТ    let e_shentsize = u16::from_le_bytes(elf[58..60].try_into().unwrap_or([0; 2])) as u64;
  1233тЖТ    let e_shnum = u16::from_le_bytes(elf[60..62].try_into().unwrap_or([0; 2])) as usize;
  1234тЖТ    let e_shstrndx = u16::from_le_bytes(elf[62..64].try_into().unwrap_or([0; 2])) as usize;
  1235тЖТ
  1236тЖТ    if e_shoff == 0 || e_shnum == 0 || e_shentsize == 0 {
  1237тЖТ        return;
  1238тЖТ    }
  1239тЖТ
  1240тЖТ    // Read the .shstrtab section to find its data.
  1241тЖТ    let shstrtab_shdr_offset = e_shoff as usize + e_shstrndx * e_shentsize as usize;
  1242тЖТ    if shstrtab_shdr_offset + 64 > elf.len() {
  1243тЖТ        return;
  1244тЖТ    }
  1245тЖТ    let shstrtab_offset = u64::from_le_bytes(
  1246тЖТ        elf[shstrtab_shdr_offset + 24..shstrtab_shdr_offset + 32]
  1247тЖТ            .try_into()
  1248тЖТ            .unwrap_or([0; 8]),
  1249тЖТ    ) as usize;
  1250тЖТ    let shstrtab_size = u64::from_le_bytes(
  1251тЖТ        elf[shstrtab_shdr_offset + 32..shstrtab_shdr_offset + 40]
  1252тЖТ            .try_into()
  1253тЖТ            .unwrap_or([0; 8]),
  1254тЖТ    ) as usize;
  1255тЖТ
  1256тЖТ    // Build new shstrtab with debug section names appended.
  1257тЖТ    let mut new_shstrtab = elf[shstrtab_offset..shstrtab_offset + shstrtab_size].to_vec();
  1258тЖТ
  1259тЖТ    // Record name offsets
  1260тЖТ    let debug_abbrev_name_off = new_shstrtab.len() as u32;
  1261тЖТ    new_shstrtab.extend_from_slice(b".debug_abbrev\0");
  1262тЖТ    let debug_info_name_off = new_shstrtab.len() as u32;
  1263тЖТ    new_shstrtab.extend_from_slice(b".debug_info\0");
  1264тЖТ    let debug_line_name_off = new_shstrtab.len() as u32;
  1265тЖТ    new_shstrtab.extend_from_slice(b".debug_line\0");
  1266тЖТ    let debug_frame_name_off = new_shstrtab.len() as u32;
  1267тЖТ    new_shstrtab.extend_from_slice(b".debug_frame\0");
  1268тЖТ
  1269тЖТ    // The new sections will be appended right before the section header table.
  1270тЖТ    let new_sections_start = e_shoff as usize;
  1271тЖТ
  1272тЖТ    // Compute aligned sizes for debug sections.
  1273тЖТ    let debug_abbrev_aligned = align_up(debug.debug_abbrev.len() as u64, 8) as usize;
  1274тЖТ    let debug_info_aligned = align_up(debug.debug_info.len() as u64, 8) as usize;
  1275тЖТ    let debug_line_aligned = align_up(debug.debug_line.len() as u64, 8) as usize;
  1276тЖТ    let debug_frame_aligned = align_up(debug.debug_frame.len() as u64, 8) as usize;
  1277тЖТ    let new_shstrtab_aligned = align_up(new_shstrtab.len() as u64, 8) as usize;
  1278тЖТ
  1279тЖТ    // Count how many debug sections we'll add (debug_frame may be empty).
  1280тЖТ    let include_debug_frame = !debug.debug_frame.is_empty();
  1281тЖТ    let num_new_sections = if include_debug_frame { 4 } else { 3 };
  1282тЖТ
  1283тЖТ    // Build section data to insert.
  1284тЖТ    let mut new_section_data = Vec::new();
  1285тЖТ
  1286тЖТ    let debug_abbrev_file_offset = new_sections_start + new_section_data.len();
  1287тЖТ    new_section_data.extend_from_slice(&debug.debug_abbrev);
  1288тЖТ    let pad = debug_abbrev_aligned - debug.debug_abbrev.len();
  1289тЖТ    new_section_data.extend_from_slice(&vec![0u8; pad]);
  1290тЖТ
  1291тЖТ    let debug_info_file_offset = new_sections_start + new_section_data.len();
  1292тЖТ    new_section_data.extend_from_slice(&debug.debug_info);
  1293тЖТ    let pad = debug_info_aligned - debug.debug_info.len();
  1294тЖТ    new_section_data.extend_from_slice(&vec![0u8; pad]);
  1295тЖТ
  1296тЖТ    let debug_line_file_offset = new_sections_start + new_section_data.len();
  1297тЖТ    new_section_data.extend_from_slice(&debug.debug_line);
  1298тЖТ    let pad = debug_line_aligned - debug.debug_line.len();
  1299тЖТ    new_section_data.extend_from_slice(&vec![0u8; pad]);
  1300тЖТ
  1301тЖТ    let debug_frame_file_offset = if include_debug_frame {
  1302тЖТ        let off = new_sections_start + new_section_data.len();
  1303тЖТ        new_section_data.extend_from_slice(&debug.debug_frame);
  1304тЖТ        let pad = debug_frame_aligned - debug.debug_frame.len();
  1305тЖТ        new_section_data.extend_from_slice(&vec![0u8; pad]);
  1306тЖТ        Some(off)
  1307тЖТ    } else {
  1308тЖТ        None
  1309тЖТ    };
  1310тЖТ
  1311тЖТ    // Replace the old shstrtab with the new one.
  1312тЖТ    let new_shstrtab_file_offset = new_sections_start + new_section_data.len();
  1313тЖТ    new_section_data.extend_from_slice(&new_shstrtab);
  1314тЖТ    let pad = new_shstrtab_aligned - new_shstrtab.len();
  1315тЖТ    new_section_data.extend_from_slice(&vec![0u8; pad]);
  1316тЖТ
  1317тЖТ    // Now insert the new section data before the section headers.
  1318тЖТ    let old_shdrs: Vec<u8> = elf.drain(e_shoff as usize..).collect();
  1319тЖТ
  1320тЖТ    // Insert debug section data.
  1321тЖТ    elf.extend_from_slice(&new_section_data);
  1322тЖТ
  1323тЖТ    // Compute the new section header table offset.
  1324тЖТ    let new_e_shoff = elf.len() as u64;
  1325тЖТ
  1326тЖТ    // Build new section headers: copy old ones, add debug section headers,
  1327тЖТ    // and update the .shstrtab header.
  1328тЖТ    let mut new_shdrs = old_shdrs;
  1329тЖТ
  1330тЖТ    // Update the .shstrtab section header to point to the new data.
  1331тЖТ    let shstrtab_shdr_start = e_shstrndx * 64;
  1332тЖТ    if shstrtab_shdr_start + 64 <= new_shdrs.len() {
  1333тЖТ        // sh_offset (bytes 24-31)
  1334тЖТ        new_shdrs[shstrtab_shdr_start + 24..shstrtab_shdr_start + 32]
  1335тЖТ            .copy_from_slice(&new_shstrtab_file_offset.to_le_bytes());
  1336тЖТ        // sh_size (bytes 32-39)
  1337тЖТ        new_shdrs[shstrtab_shdr_start + 32..shstrtab_shdr_start + 40]
  1338тЖТ            .copy_from_slice(&(new_shstrtab.len() as u64).to_le_bytes());
  1339тЖТ    }
  1340тЖТ
  1341тЖТ    // Add section headers for the debug sections.
  1342тЖТ    let new_num_shdrs = e_shnum + num_new_sections;
  1343тЖТ
  1344тЖТ    // .debug_abbrev section header
  1345тЖТ    let mut sh = new_shdr(
  1346тЖТ        SHT_PROGBITS,
  1347тЖТ        0, // no flags
  1348тЖТ        0, // no virtual address
  1349тЖТ        debug_abbrev_file_offset as u64,
  1350тЖТ        debug.debug_abbrev.len() as u64,
  1351тЖТ        0,
  1352тЖТ        0,
  1353тЖТ        1,
  1354тЖТ        0,
  1355тЖТ    );
  1356тЖТ    sh.name = debug_abbrev_name_off;
  1357тЖТ    write_filled_shdr(&mut new_shdrs, &sh);
  1358тЖТ
  1359тЖТ    // .debug_info section header
  1360тЖТ    let mut sh = new_shdr(
  1361тЖТ        SHT_PROGBITS,
  1362тЖТ        0,
  1363тЖТ        0,
  1364тЖТ        debug_info_file_offset as u64,
  1365тЖТ        debug.debug_info.len() as u64,
  1366тЖТ        0,
  1367тЖТ        0,
  1368тЖТ        1,
  1369тЖТ        0,
  1370тЖТ    );
  1371тЖТ    sh.name = debug_info_name_off;
  1372тЖТ    write_filled_shdr(&mut new_shdrs, &sh);
  1373тЖТ
  1374тЖТ    // .debug_line section header
  1375тЖТ    let mut sh = new_shdr(
  1376тЖТ        SHT_PROGBITS,
  1377тЖТ        0,
  1378тЖТ        0,
  1379тЖТ        debug_line_file_offset as u64,
  1380тЖТ        debug.debug_line.len() as u64,
  1381тЖТ        0,
  1382тЖТ        0,
  1383тЖТ        1,
  1384тЖТ        0,
  1385тЖТ    );
  1386тЖТ    sh.name = debug_line_name_off;
  1387тЖТ    write_filled_shdr(&mut new_shdrs, &sh);
  1388тЖТ
  1389тЖТ    // .debug_frame section header (only if we have frame data)
  1390тЖТ    if let Some(frame_offset) = debug_frame_file_offset {
  1391тЖТ        let mut sh = new_shdr(
  1392тЖТ            SHT_PROGBITS,
  1393тЖТ            0,
  1394тЖТ            0,
  1395тЖТ            frame_offset as u64,
  1396тЖТ            debug.debug_frame.len() as u64,
  1397тЖТ            0,
  1398тЖТ            0,
  1399тЖТ            4, // debug_frame typically has 4-byte alignment
  1400тЖТ            0,
  1401тЖТ        );
  1402тЖТ        sh.name = debug_frame_name_off;
  1403тЖТ        write_filled_shdr(&mut new_shdrs, &sh);
  1404тЖТ    }
  1405тЖТ
  1406тЖТ    // Append section headers.
  1407тЖТ    elf.extend_from_slice(&new_shdrs);
  1408тЖТ
  1409тЖТ    // Update the ELF header: e_shoff and e_shnum.
  1410тЖТ    elf[40..48].copy_from_slice(&new_e_shoff.to_le_bytes());
  1411тЖТ    elf[60..62].copy_from_slice(&(new_num_shdrs as u16).to_le_bytes());
  1412тЖТ    // e_shstrndx stays the same тАФ the .shstrtab section index hasn't changed.
  1413тЖТ}

  1481тЖТ/// Decode a ULEB128 value from `buf` starting at `offset`.
  1482тЖТ/// Returns (value, bytes_consumed).
  1483тЖТ#[cfg(test)]
  1484тЖТfn decode_uleb128(buf: &[u8], offset: usize) -> (u64, usize) {
  1485тЖТ    let mut value: u64 = 0;
  1486тЖТ    let mut shift: u32 = 0;
  1487тЖТ    let mut i = offset;
  1488тЖТ    loop {
  1489тЖТ        let byte = buf[i];
  1490тЖТ        value |= ((byte & 0x7F) as u64) << shift;
  1491тЖТ        shift += 7;
  1492тЖТ        i += 1;
  1493тЖТ        if (byte & 0x80) == 0 {
  1494тЖТ            break;
  1495тЖТ        }
  1496тЖТ    }
  1497тЖТ    (value, i - offset)
  1498тЖТ}
  1499тЖТ
  1500тЖТ/// Decode a SLEB128 value from `buf` starting at `offset`.
  1501тЖТ/// Returns (value, bytes_consumed).
  1502тЖТ#[cfg(test)]
  1503тЖТfn decode_sleb128(buf: &[u8], offset: usize) -> (i64, usize) {
  1504тЖТ    let mut value: i64 = 0;
  1505тЖТ    let mut shift: u32 = 0;
  1506тЖТ    let mut i = offset;
  1507тЖТ    let mut byte: u8;
  1508тЖТ    loop {
  1509тЖТ        byte = buf[i];
  1510тЖТ        value |= ((byte & 0x7F) as i64) << shift;
  1511тЖТ        shift += 7;
  1512тЖТ        i += 1;
  1513тЖТ        if (byte & 0x80) == 0 {
  1514тЖТ            break;
  1515тЖТ        }
  1516тЖТ    }
  1517тЖТ    // Sign extend
  1518тЖТ    if shift < 64 && (byte & 0x40) != 0 {
  1519тЖТ        value |= !0 << shift;
  1520тЖТ    }
  1521тЖТ    (value, i - offset)
  1522тЖТ}
