     1→//! SCG Diff Algorithm
     2→//!
     3→//! This module provides diff computation, application, and three-way merging
     4→//! for Semantic Computation Graphs. It is used by:
     5→//! - **COR** (Change-Oriented Recompilation) for incremental recompilation
     6→//! - **Projection system** for visualizing changes between program versions
     7→//! - **IVE** (Incremental Verifier Engine) for incremental re-verification
     8→//!
     9→//! # Overview
    10→//!
    11→//! The diff algorithm compares two SCGs and produces a structured representation
    12→//! of the differences, including added/removed/modified nodes, edges, and regions.
    13→//! A minimal edit script can be computed that transforms one graph into another.
    14→//! Three-way merge enables combining independent changes from two branches
    15→//! relative to a common base.
    16→
    17→use hashbrown::{HashMap, HashSet};
    18→use serde::{Deserialize, Serialize};
    19→
    20→use crate::callgraph::CallGraph;
    21→use crate::edge::{EdgeData, EdgeId};
    22→use crate::graph::SCG;
    23→use crate::node::{NodeData, NodeId};
    24→use crate::region::{RegionId, SCGRegion};
    25→
    26→// ── Diff Entry ──────────────────────────────────────────────────────────────
    27→
    28→/// A single atomic change between two SCG versions.
    29→///
    30→/// Each variant captures one specific kind of graph modification, making
    31→/// diffs inspectable, serializable, and reversible.
    32→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    33→pub enum DiffEntry {
    34→    /// A node was added in the new graph.
    35→    NodeAdded(NodeData),
    36→    /// A node was removed from the old graph.
    37→    NodeRemoved(NodeId),
    38→    /// A node exists in both graphs but its data changed.
    39→    NodeModified {
    40→        /// The stable identifier of the modified node.
    41→        id: NodeId,
    42→        /// The node data in the old graph.
    43→        old: NodeData,
    44→        /// The node data in the new graph.
    45→        new: NodeData,
    46→    },
    47→    /// An edge was added in the new graph.
    48→    EdgeAdded(EdgeData),
    49→    /// An edge was removed from the old graph.
    50→    EdgeRemoved(EdgeId),
    51→    /// An edge exists in both graphs but its data changed.
    52→    EdgeModified {
    53→        /// The stable identifier of the modified edge.
    54→        id: EdgeId,
    55→        /// The edge data in the old graph.
    56→        old: EdgeData,
    57→        /// The edge data in the new graph.
    58→        new: EdgeData,
    59→    },
    60→    /// A region was added in the new graph.
    61→    RegionAdded(SCGRegion),
    62→    /// A region was removed from the old graph.
    63→    RegionRemoved(RegionId),
    64→    /// A region exists in both graphs but its data changed.
    65→    RegionModified {
    66→        /// The stable identifier of the modified region.
    67→        id: RegionId,
    68→        /// The region data in the old graph.
    69→        old: SCGRegion,
    70→        /// The region data in the new graph.
    71→        new: SCGRegion,
    72→    },
    73→}
    74→
    75→impl DiffEntry {
    76→    /// Returns `true` if this entry represents an addition (node, edge, or region).
    77→    pub fn is_addition(&self) -> bool {
    78→        matches!(
    79→            self,
    80→            DiffEntry::NodeAdded(_) | DiffEntry::EdgeAdded(_) | DiffEntry::RegionAdded(_)
    81→        )
    82→    }
    83→
    84→    /// Returns `true` if this entry represents a removal (node, edge, or region).
    85→    pub fn is_removal(&self) -> bool {
    86→        matches!(
    87→            self,
    88→            DiffEntry::NodeRemoved(_) | DiffEntry::EdgeRemoved(_) | DiffEntry::RegionRemoved(_)
    89→        )
    90→    }
    91→
    92→    /// Returns `true` if this entry represents a modification (node, edge, or region).
    93→    pub fn is_modification(&self) -> bool {
    94→        matches!(
    95→            self,
    96→            DiffEntry::NodeModified { .. }
    97→                | DiffEntry::EdgeModified { .. }
    98→                | DiffEntry::RegionModified { .. }
    99→        )
   100→    }
   101→
   102→    /// Returns a human-readable description of this diff entry.
   103→    pub fn describe(&self) -> String {
   104→        match self {
   105→            DiffEntry::NodeAdded(data) => format!("node {} added ({})", data.id, data.node_type),
   106→            DiffEntry::NodeRemoved(id) => format!("node {} removed", id),
   107→            DiffEntry::NodeModified { id, old, new } => {
   108→                format!(
   109→                    "node {} modified ({} -> {})",
   110→                    id, old.node_type, new.node_type
   111→                )
   112→            }
   113→            DiffEntry::EdgeAdded(data) => {
   114→                format!(
   115→                    "edge {} added ({} -> {}, {})",
   116→                    data.id, data.source, data.target, data.kind
   117→                )
   118→            }
   119→            DiffEntry::EdgeRemoved(id) => format!("edge {} removed", id),
   120→            DiffEntry::EdgeModified { id, old, new } => {
   121→                format!("edge {} modified ({} -> {})", id, old.kind, new.kind)
   122→            }
   123→            DiffEntry::RegionAdded(data) => format!("region {} added", data.id),
   124→            DiffEntry::RegionRemoved(id) => format!("region {} removed", id),
   125→            DiffEntry::RegionModified { id, .. } => format!("region {} modified", id),
   126→        }
   127→    }
   128→}

   130→// ── SCG Diff ────────────────────────────────────────────────────────────────
   131→
   132→/// The complete difference between two SCG versions.
   133→///
   134→/// An `SCGDiff` contains an ordered list of `DiffEntry` items that
   135→/// collectively describe all changes needed to transform the old graph
   136→/// into the new graph.
   137→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   138→pub struct SCGDiff {
   139→    /// The ordered list of diff entries.
   140→    entries: Vec<DiffEntry>,
   141→    /// Summary statistics for quick inspection.
   142→    stats: DiffStats,
   143→}
   144→
   145→/// Summary statistics for a diff.
   146→#[derive(Debug, Clone, Copy, PartialEq, Eq, Default, Serialize, Deserialize)]
   147→pub struct DiffStats {
   148→    /// Number of nodes added.
   149→    pub nodes_added: usize,
   150→    /// Number of nodes removed.
   151→    pub nodes_removed: usize,
   152→    /// Number of nodes modified.
   153→    pub nodes_modified: usize,
   154→    /// Number of edges added.
   155→    pub edges_added: usize,
   156→    /// Number of edges removed.
   157→    pub edges_removed: usize,
   158→    /// Number of edges modified.
   159→    pub edges_modified: usize,
   160→    /// Number of regions added.
   161→    pub regions_added: usize,
   162→    /// Number of regions removed: usize,
   163→    pub regions_removed: usize,
   164→    /// Number of regions modified.
   165→    pub regions_modified: usize,
   166→}
   167→
   168→impl DiffStats {
   169→    /// Returns the total number of changes.
   170→    pub fn total_changes(&self) -> usize {
   171→        self.nodes_added
   172→            + self.nodes_removed
   173→            + self.nodes_modified
   174→            + self.edges_added
   175→            + self.edges_removed
   176→            + self.edges_modified
   177→            + self.regions_added
   178→            + self.regions_removed
   179→            + self.regions_modified
   180→    }
   181→
   182→    /// Returns `true` if there are no changes.
   183→    pub fn is_empty(&self) -> bool {
   184→        self.total_changes() == 0
   185→    }
   186→}
   187→
   188→impl SCGDiff {
   189→    /// Creates a new empty diff.
   190→    pub fn empty() -> Self {
   191→        Self {
   192→            entries: Vec::new(),
   193→            stats: DiffStats::default(),
   194→        }
   195→    }
   196→
   197→    /// Creates a diff from a list of entries, computing stats automatically.
   198→    pub fn from_entries(entries: Vec<DiffEntry>) -> Self {
   199→        let stats = compute_stats(&entries);
   200→        Self { entries, stats }
   201→    }
   202→
   203→    /// Returns the list of diff entries.
   204→    pub fn entries(&self) -> &[DiffEntry] {
   205→        &self.entries
   206→    }
   207→
   208→    /// Returns the diff statistics.
   209→    pub fn stats(&self) -> DiffStats {
   210→        self.stats
   211→    }
   212→
   213→    /// Returns `true` if the diff contains no entries.
   214→    pub fn is_empty(&self) -> bool {
   215→        self.entries.is_empty()
   216→    }
   217→
   218→    /// Returns the number of entries in the diff.
   219→    pub fn len(&self) -> usize {
   220→        self.entries.len()
   221→    }
   222→
   223→    /// Returns an iterator over entries that only affect nodes.
   224→    pub fn node_entries(&self) -> impl Iterator<Item = &DiffEntry> {
   225→        self.entries.iter().filter(|e| {
   226→            matches!(
   227→                e,
   228→                DiffEntry::NodeAdded(_)
   229→                    | DiffEntry::NodeRemoved(_)
   230→                    | DiffEntry::NodeModified { .. }
   231→            )
   232→        })
   233→    }
   234→
   235→    /// Returns an iterator over entries that only affect edges.
   236→    pub fn edge_entries(&self) -> impl Iterator<Item = &DiffEntry> {
   237→        self.entries.iter().filter(|e| {
   238→            matches!(
   239→                e,
   240→                DiffEntry::EdgeAdded(_)
   241→                    | DiffEntry::EdgeRemoved(_)
   242→                    | DiffEntry::EdgeModified { .. }
   243→            )
   244→        })
   245→    }
   246→
   247→    /// Returns an iterator over entries that only affect regions.
   248→    pub fn region_entries(&self) -> impl Iterator<Item = &DiffEntry> {
   249→        self.entries.iter().filter(|e| {
   250→            matches!(
   251→                e,
   252→                DiffEntry::RegionAdded(_)
   253→                    | DiffEntry::RegionRemoved(_)
   254→                    | DiffEntry::RegionModified { .. }
   255→            )
   256→        })
   257→    }
   258→
   259→    /// Returns the set of function names affected by this diff.
   260→    ///
   261→    /// An affected function is one that contains at least one changed node
   262→    /// (added, removed, or modified). This is computed by inspecting the
   263→    /// old and new graphs to determine which functions the changed nodes
   264→    /// belong to.
   265→    ///
   266→    /// This is particularly useful for LLMs that need to understand which
   267→    /// parts of a program were impacted by a change.
   268→    pub fn affected_functions(&self, old: &SCG, new: &SCG) -> AffectedFunctions {
   269→        let old_cg = CallGraph::build(old);
   270→        let new_cg = CallGraph::build(new);
   271→
   272→        let mut affected = AffectedFunctions {
   273→            modified: Vec::new(),
   274→            added: Vec::new(),
   275→            removed: Vec::new(),
   276→        };
   277→
   278→        // Track all changed node IDs
   279→        let mut changed_node_ids: HashSet<NodeId> = HashSet::new();
   280→        for entry in &self.entries {
   281→            match entry {
   282→                DiffEntry::NodeAdded(data) => {
   283→                    changed_node_ids.insert(data.id);
   284→                }
   285→                DiffEntry::NodeRemoved(id) => {
   286→                    changed_node_ids.insert(*id);
   287→                }
   288→                DiffEntry::NodeModified { id, .. } => {
   289→                    changed_node_ids.insert(*id);
   290→                }
   291→                DiffEntry::EdgeAdded(data) => {
   292→                    changed_node_ids.insert(data.source);
   293→                    changed_node_ids.insert(data.target);
   294→                }
   295→                DiffEntry::EdgeRemoved(_) => {
   296→                    // Edge removal doesn't necessarily mean function change
   297→                }
   298→                DiffEntry::EdgeModified { id, .. } => {
   299→                    // Look up the edge to find its endpoints
   300→                    if let Some(edge) = old.get_edge(*id).or_else(|| new.get_edge(*id)) {
   301→                        changed_node_ids.insert(edge.source);
   302→                        changed_node_ids.insert(edge.target);
   303→                    }
   304→                }
   305→                DiffEntry::RegionAdded(_) | DiffEntry::RegionRemoved(_) => {}
   306→                DiffEntry::RegionModified { .. } => {}
   307→            }
   308→        }
   309→
   310→        // For each changed node, find which function it belongs to
   311→        let mut old_func_names: HashSet<String> = HashSet::new();
   312→        let mut new_func_names: HashSet<String> = HashSet::new();
   313→
   314→        for node_id in &changed_node_ids {
   315→            // Try to find the function in the new graph first
   316→            if let Some(entry_id) = new_cg.find_enclosing_function(new, *node_id) {
   317→                let fid = crate::callgraph::FunctionId(entry_id);
   318→                if let Some(name) = new_cg.function_label(&fid) {
   319→                    new_func_names.insert(name.to_string());
   320→                } else {
   321→                    new_func_names.insert(format!("func_{}", entry_id.as_u64()));
   322→                }
   323→            }
   324→            // Also try the old graph
   325→            if let Some(entry_id) = old_cg.find_enclosing_function(old, *node_id) {
   326→                let fid = crate::callgraph::FunctionId(entry_id);
   327→                if let Some(name) = old_cg.function_label(&fid) {
   328→                    old_func_names.insert(name.to_string());
   329→                } else {
   330→                    old_func_names.insert(format!("func_{}", entry_id.as_u64()));
   331→                }
   332→            }
   333→        }
   334→
   335→        // Check for functions that were entirely added or removed
   336→        let old_fids: HashSet<_> = old_cg.functions().copied().collect();
   337→        let new_fids: HashSet<_> = new_cg.functions().copied().collect();
   338→
   339→        for fid in &new_fids - &old_fids {
   340→            if let Some(name) = new_cg.function_label(&fid) {
   341→                affected.added.push(name.to_string());
   342→            } else {
   343→                affected.added.push(format!("func_{}", fid.0.as_u64()));
   344→            }
   345→        }
   346→        for fid in &old_fids - &new_fids {
   347→            if let Some(name) = old_cg.function_label(&fid) {
   348→                affected.removed.push(name.to_string());
   349→            } else {
   350→                affected.removed.push(format!("func_{}", fid.0.as_u64()));
   351→            }
   352→        }
   353→
   354→        // Modified functions = those in both old and new that have changed nodes
   355→        let all_names: HashSet<String> = old_func_names
   356→            .union(&new_func_names)
   357→            .cloned()
   358→            .collect();
   359→        for name in &all_names {
   360→            if !affected.added.contains(name) && !affected.removed.contains(name) {
   361→                affected.modified.push(name.clone());
   362→            }
   363→        }
   364→
   365→        // Sort for deterministic output
   366→        affected.modified.sort();
   367→        affected.added.sort();
   368→        affected.removed.sort();
   369→
   370→        affected
   371→    }
   372→
   373→    /// Produces a clean, LLM-friendly JSON representation of this diff.
   374→    ///
   375→    /// Includes summary statistics, affected functions, and a list of
   376→    /// changes with human-readable descriptions.
   377→    pub fn to_json(&self, old: &SCG, new: &SCG) -> String {
   378→        let affected = self.affected_functions(old, new);
   379→        let llm_diff = LlmDiff {
   380→            summary: self.stats,
   381→            affected_functions: affected,
   382→            changes: self
   383→                .entries
   384→                .iter()
   385→                .map(|e| LlmDiffChange {
   386→                    kind: match e {
   387→                        DiffEntry::NodeAdded(_) => "node_added".to_string(),
   388→                        DiffEntry::NodeRemoved(_) => "node_removed".to_string(),
   389→                        DiffEntry::NodeModified { .. } => "node_modified".to_string(),
   390→                        DiffEntry::EdgeAdded(_) => "edge_added".to_string(),
   391→                        DiffEntry::EdgeRemoved(_) => "edge_removed".to_string(),
   392→                        DiffEntry::EdgeModified { .. } => "edge_modified".to_string(),
   393→                        DiffEntry::RegionAdded(_) => "region_added".to_string(),
   394→                        DiffEntry::RegionRemoved(_) => "region_removed".to_string(),
   395→                        DiffEntry::RegionModified { .. } => "region_modified".to_string(),
   396→                    },
   397→                    description: e.describe(),
   398→                })
   399→                .collect(),
   400→        };
   401→        serde_json::to_string_pretty(&llm_diff)
   402→            .unwrap_or_else(|e| format!(r#"{{"error": "diff JSON serialization failed: {}"}}"#, e))
   403→    }
   404→
   405→    /// Produces a human/LLM-readable text representation of this diff.
   406→    pub fn to_text(&self, old: &SCG, new: &SCG) -> String {
   407→        let affected = self.affected_functions(old, new);
   408→        let mut out = String::with_capacity(2048);
   409→
   410→        out.push_str("=== SCG Diff ===\n\n");
   411→
   412→        // Summary
   413→        out.push_str(&format!(
   414→            "Changes: {} added, {} removed, {} modified nodes; {} added, {} removed, {} modified edges; {} added, {} removed, {} modified regions\n",
   415→            self.stats.nodes_added, self.stats.nodes_removed, self.stats.nodes_modified,
   416→            self.stats.edges_added, self.stats.edges_removed, self.stats.edges_modified,
   417→            self.stats.regions_added, self.stats.regions_removed, self.stats.regions_modified,
   418→        ));
   419→        out.push_str(&format!("Total changes: {}\n\n", self.stats.total_changes()));
   420→
   421→        // Affected functions
   422→        if !affected.modified.is_empty() {
   423→            out.push_str(&format!(
   424→                "Modified functions: {}\n",
   425→                affected.modified.join(", ")
   426→            ));
   427→        }
   428→        if !affected.added.is_empty() {
   429→            out.push_str(&format!(
   430→                "Added functions: {}\n",
   431→                affected.added.join(", ")
   432→            ));
   433→        }
   434→        if !affected.removed.is_empty() {
   435→            out.push_str(&format!(
   436→                "Removed functions: {}\n",
   437→                affected.removed.join(", ")
   438→            ));
   439→        }
   440→        if !affected.modified.is_empty() || !affected.added.is_empty() || !affected.removed.is_empty() {
   441→            out.push('\n');
   442→        }
   443→
   444→        // Changes
   445→        out.push_str("--- Changes ---\n\n");
   446→        for entry in &self.entries {
   447→            out.push_str(&format!("  {}\n", entry.describe()));
   448→        }
   449→
   450→        out
   451→    }
   452→}

   456→/// Information about which functions are affected by a diff.
   457→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   458→pub struct AffectedFunctions {
   459→    /// Functions that were modified (had nodes changed within them).
   460→    pub modified: Vec<String>,
   461→    /// Functions that were entirely added.
   462→    pub added: Vec<String>,
   463→    /// Functions that were entirely removed.
   464→    pub removed: Vec<String>,
   465→}
   466→
   467→impl AffectedFunctions {
   468→    /// Returns `true` if no functions are affected.
   469→    pub fn is_empty(&self) -> bool {
   470→        self.modified.is_empty() && self.added.is_empty() && self.removed.is_empty()
   471→    }
   472→}

   496→/// This is an alias for [`diff_scg`] that provides a more intuitive name

   502→/// Compute stats from a slice of diff entries.
   503→fn compute_stats(entries: &[DiffEntry]) -> DiffStats {
   504→    let mut stats = DiffStats::default();
   505→    for entry in entries {
   506→        match entry {
   507→            DiffEntry::NodeAdded(_) => stats.nodes_added += 1,
   508→            DiffEntry::NodeRemoved(_) => stats.nodes_removed += 1,
   509→            DiffEntry::NodeModified { .. } => stats.nodes_modified += 1,
   510→            DiffEntry::EdgeAdded(_) => stats.edges_added += 1,
   511→            DiffEntry::EdgeRemoved(_) => stats.edges_removed += 1,
   512→            DiffEntry::EdgeModified { .. } => stats.edges_modified += 1,
   513→            DiffEntry::RegionAdded(_) => stats.regions_added += 1,
   514→            DiffEntry::RegionRemoved(_) => stats.regions_removed += 1,
   515→            DiffEntry::RegionModified { .. } => stats.regions_modified += 1,
   516→        }
   517→    }
   518→    stats
   519→}

   523→/// Errors that can occur when applying a diff to an SCG.
   524→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   525→pub enum DiffError {
   526→    /// A node referenced in the diff was not found in the target graph.
   527→    NodeNotFound(NodeId),
   528→    /// An edge referenced in the diff was not found in the target graph.
   529→    EdgeNotFound(EdgeId),
   530→    /// A region referenced in the diff was not found in the target graph.
   531→    RegionNotFound(RegionId),
   532→    /// A node with the given ID already exists (cannot add duplicate).
   533→    DuplicateNode(NodeId),
   534→    /// An edge with the given ID already exists (cannot add duplicate).
   535→    DuplicateEdge(EdgeId),
   536→    /// An edge's source or target node does not exist.
   537→    InvalidEdgeEndpoints {
   538→        /// The source node of the invalid edge.
   539→        source: NodeId,
   540→        /// The target node of the invalid edge.
   541→        target: NodeId,
   542→    },
   543→    /// The diff cannot be applied in its current state (e.g., dependency
   544→    /// ordering issue).
   545→    CannotApply(String),
   546→}
   547→
   548→impl std::fmt::Display for DiffError {
   549→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   550→        match self {
   551→            DiffError::NodeNotFound(id) => write!(f, "diff error: node not found: {id}"),
   552→            DiffError::EdgeNotFound(id) => write!(f, "diff error: edge not found: {id}"),
   553→            DiffError::RegionNotFound(id) => write!(f, "diff error: region not found: {id}"),
   554→            DiffError::DuplicateNode(id) => write!(f, "diff error: duplicate node: {id}"),
   555→            DiffError::DuplicateEdge(id) => write!(f, "diff error: duplicate edge: {id}"),
   556→            DiffError::InvalidEdgeEndpoints { source, target } => {
   557→                write!(
   558→                    f,
   559→                    "diff error: invalid edge endpoints: source={source}, target={target}"
   560→                )
   561→            }
   562→            DiffError::CannotApply(msg) => write!(f, "diff error: cannot apply: {msg}"),
   563→        }
   564→    }
   565→}

   569→// ── Diff Computation ────────────────────────────────────────────────────────
   570→
   571→/// Computes the diff between two SCGs.
   572→///
   573→/// The diff captures all changes needed to transform `old` into `new`.
   574→/// Nodes and edges are matched by their stable `NodeId`/`EdgeId` identifiers.
   575→/// Regions are matched by `RegionId`.
   576→///
   577→/// # Algorithm
   578→///
   579→/// 1. **Nodes**: Identify nodes present only in old (removed), only in new
   580→///    (added), or in both (check for modifications by comparing `NodeData`).
   581→/// 2. **Edges**: Identify edges present only in old (removed), only in new
   582→///    (added), or in both (check for modifications by comparing `EdgeData`).
   583→/// 3. **Regions**: Identify regions present only in old (removed), only in
   584→///    new (added), or in both (check for modifications by comparing `SCGRegion`).
   585→///
   586→/// The ordering of entries follows: removed nodes, removed edges, removed
   587→/// regions, modified nodes, modified edges, modified regions, added regions,
   588→/// added edges, added nodes. This ordering ensures safe application:
   589→/// removals happen before additions, so no dangling references exist.
   590→pub fn diff_scg(old: &SCG, new: &SCG) -> SCGDiff {
   591→    let mut entries = Vec::new();
   592→
   593→    // ── Collect node IDs from both graphs ──
   594→    let old_node_ids: HashSet<NodeId> = old.node_ids().collect();
   595→    let new_node_ids: HashSet<NodeId> = new.node_ids().collect();
   596→
   597→    // Removed nodes (in old but not in new)
   598→    for id in &old_node_ids - &new_node_ids {
   599→        entries.push(DiffEntry::NodeRemoved(id));
   600→    }
   601→
   602→    // Modified nodes (in both, but data differs)
   603→    for id in &old_node_ids & &new_node_ids {
   604→        let old_data = old.get_node(id).unwrap();
   605→        let new_data = new.get_node(id).unwrap();
   606→        if old_data != new_data {
   607→            entries.push(DiffEntry::NodeModified {
   608→                id,
   609→                old: old_data.clone(),
   610→                new: new_data.clone(),
   611→            });
   612→        }
   613→    }
   614→
   615→    // ── Collect edge IDs from both graphs ──
   616→    let old_edge_ids: HashSet<EdgeId> = old.edge_ids().collect();
   617→    let new_edge_ids: HashSet<EdgeId> = new.edge_ids().collect();
   618→
   619→    // Removed edges (in old but not in new)
   620→    for id in &old_edge_ids - &new_edge_ids {
   621→        entries.push(DiffEntry::EdgeRemoved(id));
   622→    }
   623→
   624→    // Modified edges (in both, but data differs)
   625→    for id in &old_edge_ids & &new_edge_ids {
   626→        let old_data = old.get_edge(id).unwrap();
   627→        let new_data = new.get_edge(id).unwrap();
   628→        if old_data != new_data {
   629→            entries.push(DiffEntry::EdgeModified {
   630→                id,
   631→                old: old_data.clone(),
   632→                new: new_data.clone(),
   633→            });
   634→        }
   635→    }
   636→
   637→    // ── Collect region IDs from both graphs ──
   638→    let old_region_ids: HashSet<RegionId> = old.regions().map(|r| r.id).collect();
   639→    let new_region_ids: HashSet<RegionId> = new.regions().map(|r| r.id).collect();
   640→
   641→    // Removed regions (in old but not in new)
   642→    for id in &old_region_ids - &new_region_ids {
   643→        entries.push(DiffEntry::RegionRemoved(id));
   644→    }
   645→
   646→    // Modified regions (in both, but data differs)
   647→    for id in &old_region_ids & &new_region_ids {
   648→        let old_region = old.get_region(id).unwrap();
   649→        let new_region = new.get_region(id).unwrap();
   650→        if old_region != new_region {
   651→            entries.push(DiffEntry::RegionModified {
   652→                id,
   653→                old: old_region.clone(),
   654→                new: new_region.clone(),
   655→            });
   656→        }
   657→    }
   658→
   659→    // Added regions (in new but not in old) — must come before added nodes
   660→    for id in &new_region_ids - &old_region_ids {
   661→        let region = new.get_region(id).unwrap();
   662→        entries.push(DiffEntry::RegionAdded(region.clone()));
   663→    }
   664→
   665→    // Added edges (in new but not in old)
   666→    for id in &new_edge_ids - &old_edge_ids {
   667→        let edge_data = new.get_edge(id).unwrap();
   668→        entries.push(DiffEntry::EdgeAdded(edge_data.clone()));
   669→    }
   670→
   671→    // Added nodes (in new but not in old)
   672→    for id in &new_node_ids - &old_node_ids {
   673→        let node_data = new.get_node(id).unwrap();
   674→        entries.push(DiffEntry::NodeAdded(node_data.clone()));
   675→    }
   676→
   677→    SCGDiff::from_entries(entries)
   678→}

   682→/// Applies a diff to an SCG, mutating it in place.
   683→///
   684→/// The diff entries are applied in order. The expected ordering is:
   685→/// 1. Removals (nodes, edges, regions) — clean up old state
   686→/// 2. Modifications — update existing elements
   687→/// 3. Additions (regions, nodes, edges) — add new state
   688→///
   689→/// If any entry cannot be applied (e.g., a node to remove doesn't exist,
   690→/// or a node to add already exists), an error is returned and the graph
   691→/// may be partially modified.
   692→pub fn apply_diff(scg: &mut SCG, diff: &SCGDiff) -> Result<(), DiffError> {
   693→    for entry in &diff.entries {
   694→        apply_entry(scg, entry)?;
   695→    }
   696→    Ok(())
   697→}
   698→
   699→/// Applies a single diff entry to an SCG.
   700→fn apply_entry(scg: &mut SCG, entry: &DiffEntry) -> Result<(), DiffError> {
   701→    match entry {
   702→        DiffEntry::NodeRemoved(id) => {
   703→            scg.remove_node(*id)
   704→                .map_err(|_| DiffError::NodeNotFound(*id))?;
   705→            Ok(())
   706→        }
   707→        DiffEntry::NodeAdded(data) => {
   708→            if scg.get_node(data.id).is_some() {
   709→                return Err(DiffError::DuplicateNode(data.id));
   710→            }
   711→            scg.add_node_with_id(
   712→                data.id,
   713→                data.node_type.clone(),
   714→                data.payload.clone(),
   715→                data.program_point.clone(),
   716→            )
   717→            .map_err(|_| DiffError::DuplicateNode(data.id))?;
   718→            // Copy annotation if present
   719→            if let Some(ref ann) = data.annotation {
   720→                if let Some(node) = scg.get_node_mut(data.id) {
   721→                    node.annotation = Some(ann.clone());
   722→                }
   723→            }
   724→            Ok(())
   725→        }
   726→        DiffEntry::NodeModified { id, new, .. } => {
   727→            let node = scg.get_node_mut(*id).ok_or(DiffError::NodeNotFound(*id))?;
   728→            node.node_type = new.node_type.clone();
   729→            node.payload = new.payload.clone();
   730→            node.annotation = new.annotation.clone();
   731→            node.program_point = new.program_point.clone();
   732→            Ok(())
   733→        }
   734→        DiffEntry::EdgeRemoved(id) => {
   735→            scg.remove_edge(*id)
   736→                .map_err(|_| DiffError::EdgeNotFound(*id))?;
   737→            Ok(())
   738→        }
   739→        DiffEntry::EdgeAdded(data) => {
   740→            if scg.get_edge(data.id).is_some() {
   741→                return Err(DiffError::DuplicateEdge(data.id));
   742→            }
   743→            // Verify endpoints exist
   744→            if scg.get_node(data.source).is_none() {
   745→                return Err(DiffError::InvalidEdgeEndpoints {
   746→                    source: data.source,
   747→                    target: data.target,
   748→                });
   749→            }
   750→            if scg.get_node(data.target).is_none() {
   751→                return Err(DiffError::InvalidEdgeEndpoints {
   752→                    source: data.source,
   753→                    target: data.target,
   754→                });
   755→            }
   756→            scg.add_edge_with_id(data.id, data.source, data.target, data.kind.clone())
   757→                .map_err(|_| DiffError::DuplicateEdge(data.id))?;
   758→            // Copy label if present
   759→            if let Some(ref label) = data.label {
   760→                if let Some(edge) = scg.get_edge_mut(data.id) {
   761→                    edge.label = Some(label.clone());
   762→                }
   763→            }
   764→            Ok(())
   765→        }
   766→        DiffEntry::EdgeModified { id, new, .. } => {
   767→            let edge = scg.get_edge_mut(*id).ok_or(DiffError::EdgeNotFound(*id))?;
   768→            edge.kind = new.kind.clone();
   769→            edge.label = new.label.clone();
   770→            // Note: source/target changes on an existing edge are not
   771→            // supported through modification. Use remove + add instead.
   772→            Ok(())
   773→        }
   774→        DiffEntry::RegionRemoved(id) => {
   775→            scg.remove_region(*id);
   776→            Ok(())
   777→        }
   778→        DiffEntry::RegionAdded(data) => {
   779→            if scg.get_region(data.id).is_some() {
   780→                return Err(DiffError::RegionNotFound(data.id));
   781→            }
   782→            scg.add_region(data.clone());
   783→            Ok(())
   784→        }
   785→        DiffEntry::RegionModified { id, new, .. } => {
   786→            let region = scg
   787→                .get_region_mut(*id)
   788→                .ok_or(DiffError::RegionNotFound(*id))?;
   789→            region.scope_level = new.scope_level;
   790→            region.security_boundary = new.security_boundary;
   791→            region.deployment_target = new.deployment_target.clone();
   792→            region.nodes = new.nodes.clone();
   793→            Ok(())
   794→        }
   795→    }
   796→}

   798→// ── Edit Script ─────────────────────────────────────────────────────────────
   799→
   800→/// Computes a minimal edit script to transform `old` into `new`.
   801→///
   802→/// The edit script is an ordered sequence of `DiffEntry` items that, when
   803→/// applied sequentially to `old`, produces a graph equivalent to `new`.
   804→///
   805→/// The ordering is carefully chosen to ensure safe application:
   806→/// 1. Remove edges (to disconnect nodes before removal)
   807→/// 2. Remove nodes (leaf-first to avoid dangling references)
   808→/// 3. Remove regions
   809→/// 4. Modify existing nodes, edges, and regions
   810→/// 5. Add regions (before nodes so regions exist for node assignment)
   811→/// 6. Add nodes
   812→/// 7. Add edges (after nodes so endpoints exist)
   813→pub fn compute_edit_script(old: &SCG, new: &SCG) -> Vec<DiffEntry> {
   814→    let raw_diff = diff_scg(old, new);
   815→    let mut script = Vec::with_capacity(raw_diff.len());
   816→
   817→    // Phase 1: Removals — edges first, then nodes, then regions
   818→    for entry in raw_diff.edge_entries() {
   819→        if let DiffEntry::EdgeRemoved(id) = entry {
   820→            script.push(DiffEntry::EdgeRemoved(*id));
   821→        }
   822→    }
   823→    for entry in raw_diff.node_entries() {
   824→        if let DiffEntry::NodeRemoved(id) = entry {
   825→            script.push(DiffEntry::NodeRemoved(*id));
   826→        }
   827→    }
   828→    for entry in raw_diff.region_entries() {
   829→        if let DiffEntry::RegionRemoved(id) = entry {
   830→            script.push(DiffEntry::RegionRemoved(*id));
   831→        }
   832→    }
   833→
   834→    // Phase 2: Modifications — nodes, then edges, then regions
   835→    for entry in raw_diff.node_entries() {
   836→        if let DiffEntry::NodeModified { id, old, new } = entry {
   837→            script.push(DiffEntry::NodeModified {
   838→                id: *id,
   839→                old: old.clone(),
   840→                new: new.clone(),
   841→            });
   842→        }
   843→    }
   844→    for entry in raw_diff.edge_entries() {
   845→        if let DiffEntry::EdgeModified { id, old, new } = entry {
   846→            script.push(DiffEntry::EdgeModified {
   847→                id: *id,
   848→                old: old.clone(),
   849→                new: new.clone(),
   850→            });
   851→        }
   852→    }
   853→    for entry in raw_diff.region_entries() {
   854→        if let DiffEntry::RegionModified { id, old, new } = entry {
   855→            script.push(DiffEntry::RegionModified {
   856→                id: *id,
   857→                old: old.clone(),
   858→                new: new.clone(),
   859→            });
   860→        }
   861→    }
   862→
   863→    // Phase 3: Additions — regions first, then nodes, then edges
   864→    for entry in raw_diff.region_entries() {
   865→        if let DiffEntry::RegionAdded(data) = entry {
   866→            script.push(DiffEntry::RegionAdded(data.clone()));
   867→        }
   868→    }
   869→    for entry in raw_diff.node_entries() {
   870→        if let DiffEntry::NodeAdded(data) = entry {
   871→            script.push(DiffEntry::NodeAdded(data.clone()));
   872→        }
   873→    }
   874→    for entry in raw_diff.edge_entries() {
   875→        if let DiffEntry::EdgeAdded(data) = entry {
   876→            script.push(DiffEntry::EdgeAdded(data.clone()));
   877→        }
   878→    }
   879→
   880→    script
   881→}

   885→/// A conflict encountered during three-way merge.
   886→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   887→pub struct MergeConflict {
   888→    /// Conflicts involving nodes.
   889→    pub node_conflicts: Vec<NodeConflict>,
   890→    /// Conflicts involving edges.
   891→    pub edge_conflicts: Vec<EdgeConflict>,
   892→    /// Conflicts involving regions.
   893→    pub region_conflicts: Vec<RegionConflict>,
   894→}
   895→
   896→impl MergeConflict {
   897→    /// Creates an empty merge conflict set.
   898→    pub fn empty() -> Self {
   899→        Self {
   900→            node_conflicts: Vec::new(),
   901→            edge_conflicts: Vec::new(),
   902→            region_conflicts: Vec::new(),
   903→        }
   904→    }
   905→
   906→    /// Returns `true` if there are no conflicts.
   907→    pub fn is_empty(&self) -> bool {
   908→        self.node_conflicts.is_empty()
   909→            && self.edge_conflicts.is_empty()
   910→            && self.region_conflicts.is_empty()
   911→    }
   912→
   913→    /// Returns the total number of conflicts across all categories.
   914→    pub fn total_conflicts(&self) -> usize {
   915→        self.node_conflicts.len() + self.edge_conflicts.len() + self.region_conflicts.len()
   916→    }
   917→}
   918→
   919→impl std::fmt::Display for MergeConflict {
   920→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   921→        write!(
   922→            f,
   923→            "merge conflict: {} node, {} edge, {} region conflicts",
   924→            self.node_conflicts.len(),
   925→            self.edge_conflicts.len(),
   926→            self.region_conflicts.len()
   927→        )
   928→    }
   929→}

   972→/// Performs a three-way merge of SCGs.
   973→///
   974→/// Given a common `base` graph and two divergent versions (`ours` and `theirs`),
   975→/// this function produces a merged graph that incorporates changes from both
   976→/// branches. When both branches modify the same element differently, a conflict
   977→/// is reported.
   978→///
   979→/// # Merge Rules
   980→///
   981→/// For each element (node, edge, region):
   982→/// - **Unchanged in both**: Keep as-is from base.
   983→/// - **Changed in ours only**: Apply ours' version.
   984→/// - **Changed in theirs only**: Apply theirs' version.
   985→/// - **Changed identically in both**: Apply either (they agree).
   986→/// - **Changed differently in both**: Conflict — include in `MergeConflict`.
   987→/// - **Added in ours only**: Include ours' addition.
   988→/// - **Added in theirs only**: Include theirs' addition.
   989→/// - **Added in both (same data)**: Include either.
   990→/// - **Added in both (different data)**: Conflict.
   991→/// - **Removed in ours only**: Remove.
   992→/// - **Removed in theirs only**: Remove.
   993→/// - **Removed in one, modified in other**: Conflict.
   994→/// - **Removed in both**: Remove.
   995→pub fn three_way_merge(base: &SCG, ours: &SCG, theirs: &SCG) -> Result<SCG, MergeConflict> {
   996→    let diff_ours = diff_scg(base, ours);
   997→    let diff_theirs = diff_scg(base, theirs);
   998→
   999→    let mut conflict = MergeConflict::empty();
  1000→
  1001→    // Start with a clone of the base graph
  1002→    let mut merged = base.clone();
  1003→
  1004→    // ── Merge Nodes ──
  1005→    let ours_node_changes = collect_node_changes(&diff_ours);
  1006→    let theirs_node_changes = collect_node_changes(&diff_theirs);
  1007→
  1008→    let all_node_ids: HashSet<NodeId> = ours_node_changes
  1009→        .keys()
  1010→        .chain(theirs_node_changes.keys())
  1011→        .copied()
  1012→        .collect();
  1013→
  1014→    for node_id in &all_node_ids {
  1015→        let ours_change = ours_node_changes.get(node_id);
  1016→        let theirs_change = theirs_node_changes.get(node_id);
  1017→
  1018→        match (ours_change, theirs_change) {
  1019→            (None, None) => { /* No change in either — keep base */ }
  1020→            (Some(oc), None) => {
  1021→                // Only ours changed — apply ours' change
  1022→                apply_node_change(&mut merged, node_id, oc);
  1023→            }
  1024→            (None, Some(tc)) => {
  1025→                // Only theirs changed — apply theirs' change
  1026→                apply_node_change(&mut merged, node_id, tc);
  1027→            }
  1028→            (Some(oc), Some(tc)) => {
  1029→                // Both changed — check if they agree
  1030→                if oc == tc {
  1031→                    apply_node_change(&mut merged, node_id, oc);
  1032→                } else {
  1033→                    conflict.node_conflicts.push(NodeConflict {
  1034→                        id: *node_id,
  1035→                        base: base.get_node(*node_id).cloned(),
  1036→                        ours: change_to_node_data(oc),
  1037→                        theirs: change_to_node_data(tc),
  1038→                    });
  1039→                }
  1040→            }
  1041→        }
  1042→    }
  1043→
  1044→    // ── Merge Edges ──
  1045→    let ours_edge_changes = collect_edge_changes(&diff_ours);
  1046→    let theirs_edge_changes = collect_edge_changes(&diff_theirs);
  1047→
  1048→    let all_edge_ids: HashSet<EdgeId> = ours_edge_changes
  1049→        .keys()
  1050→        .chain(theirs_edge_changes.keys())
  1051→        .copied()
  1052→        .collect();
  1053→
  1054→    for edge_id in &all_edge_ids {
  1055→        let ours_change = ours_edge_changes.get(edge_id);
  1056→        let theirs_change = theirs_edge_changes.get(edge_id);
  1057→
  1058→        match (ours_change, theirs_change) {
  1059→            (None, None) => {}
  1060→            (Some(oc), None) => {
  1061→                apply_edge_change(&mut merged, edge_id, oc);
  1062→            }
  1063→            (None, Some(tc)) => {
  1064→                apply_edge_change(&mut merged, edge_id, tc);
  1065→            }
  1066→            (Some(oc), Some(tc)) => {
  1067→                if oc == tc {
  1068→                    apply_edge_change(&mut merged, edge_id, oc);
  1069→                } else {
  1070→                    conflict.edge_conflicts.push(EdgeConflict {
  1071→                        id: *edge_id,
  1072→                        base: base.get_edge(*edge_id).cloned(),
  1073→                        ours: change_to_edge_data(oc),
  1074→                        theirs: change_to_edge_data(tc),
  1075→                    });
  1076→                }
  1077→            }
  1078→        }
  1079→    }
  1080→
  1081→    // ── Merge Regions ──
  1082→    let ours_region_changes = collect_region_changes(&diff_ours);
  1083→    let theirs_region_changes = collect_region_changes(&diff_theirs);
  1084→
  1085→    let all_region_ids: HashSet<RegionId> = ours_region_changes
  1086→        .keys()
  1087→        .chain(theirs_region_changes.keys())
  1088→        .copied()
  1089→        .collect();
  1090→
  1091→    for region_id in &all_region_ids {
  1092→        let ours_change = ours_region_changes.get(region_id);
  1093→        let theirs_change = theirs_region_changes.get(region_id);
  1094→
  1095→        match (ours_change, theirs_change) {
  1096→            (None, None) => {}
  1097→            (Some(oc), None) => {
  1098→                apply_region_change(&mut merged, region_id, oc);
  1099→            }
  1100→            (None, Some(tc)) => {
  1101→                apply_region_change(&mut merged, region_id, tc);
  1102→            }
  1103→            (Some(oc), Some(tc)) => {
  1104→                if oc == tc {
  1105→                    apply_region_change(&mut merged, region_id, oc);
  1106→                } else {
  1107→                    conflict.region_conflicts.push(RegionConflict {
  1108→                        id: *region_id,
  1109→                        base: base.get_region(*region_id).cloned(),
  1110→                        ours: change_to_region_data(oc),
  1111→                        theirs: change_to_region_data(tc),
  1112→                    });
  1113→                }
  1114→            }
  1115→        }
  1116→    }
  1117→
  1118→    if conflict.is_empty() {
  1119→        Ok(merged)
  1120→    } else {
  1121→        Err(conflict)
  1122→    }
  1123→}

  1125→// ── Change Tracking Helpers ─────────────────────────────────────────────────
  1126→
  1127→/// Represents a change to a single element, used for merge conflict detection.
  1128→#[derive(Debug, Clone, PartialEq)]
  1129→enum ElementChange<T> {
  1130→    /// The element was added with the given data.
  1131→    Added(T),
  1132→    /// The element was removed.
  1133→    Removed,
  1134→    /// The element was modified to the given data.
  1135→    Modified(T),
  1136→}
  1137→
  1138→/// Collect node-level changes from a diff into a map keyed by NodeId.
  1139→fn collect_node_changes(diff: &SCGDiff) -> HashMap<NodeId, ElementChange<NodeData>> {
  1140→    let mut changes = HashMap::new();
  1141→    for entry in diff.entries() {
  1142→        match entry {
  1143→            DiffEntry::NodeAdded(data) => {
  1144→                changes.insert(data.id, ElementChange::Added(data.clone()));
  1145→            }
  1146→            DiffEntry::NodeRemoved(id) => {
  1147→                changes.insert(*id, ElementChange::Removed);
  1148→            }
  1149→            DiffEntry::NodeModified { id, new, .. } => {
  1150→                changes.insert(*id, ElementChange::Modified(new.clone()));
  1151→            }
  1152→            _ => {}
  1153→        }
  1154→    }
  1155→    changes
  1156→}
  1157→
  1158→/// Collect edge-level changes from a diff into a map keyed by EdgeId.
  1159→fn collect_edge_changes(diff: &SCGDiff) -> HashMap<EdgeId, ElementChange<EdgeData>> {
  1160→    let mut changes = HashMap::new();
  1161→    for entry in diff.entries() {
  1162→        match entry {
  1163→            DiffEntry::EdgeAdded(data) => {
  1164→                changes.insert(data.id, ElementChange::Added(data.clone()));
  1165→            }
  1166→            DiffEntry::EdgeRemoved(id) => {
  1167→                changes.insert(*id, ElementChange::Removed);
  1168→            }
  1169→            DiffEntry::EdgeModified { id, new, .. } => {
  1170→                changes.insert(*id, ElementChange::Modified(new.clone()));
  1171→            }
  1172→            _ => {}
  1173→        }
  1174→    }
  1175→    changes
  1176→}

  1198→/// Apply a node change to the merged graph.
  1199→fn apply_node_change(scg: &mut SCG, id: &NodeId, change: &ElementChange<NodeData>) {
  1200→    match change {
  1201→        ElementChange::Added(data) => {
  1202→            if scg.get_node(*id).is_none() {
  1203→                let _ = scg.add_node_with_id(
  1204→                    data.id,
  1205→                    data.node_type.clone(),
  1206→                    data.payload.clone(),
  1207→                    data.program_point.clone(),
  1208→                );
  1209→                if let Some(ref ann) = data.annotation {
  1210→                    if let Some(node) = scg.get_node_mut(*id) {
  1211→                        node.annotation = Some(ann.clone());
  1212→                    }
  1213→                }
  1214→            }
  1215→        }
  1216→        ElementChange::Removed => {
  1217→            let _ = scg.remove_node(*id);
  1218→        }
  1219→        ElementChange::Modified(data) => {
  1220→            if let Some(node) = scg.get_node_mut(*id) {
  1221→                node.node_type = data.node_type.clone();
  1222→                node.payload = data.payload.clone();
  1223→                node.annotation = data.annotation.clone();
  1224→                node.program_point = data.program_point.clone();
  1225→            }
  1226→        }
  1227→    }
  1228→}
  1229→
  1230→/// Apply an edge change to the merged graph.
  1231→fn apply_edge_change(scg: &mut SCG, id: &EdgeId, change: &ElementChange<EdgeData>) {
  1232→    match change {
  1233→        ElementChange::Added(data) => {
  1234→            if scg.get_edge(*id).is_none() {
  1235→                let _ = scg.add_edge_with_id(data.id, data.source, data.target, data.kind.clone());
  1236→                if let Some(ref label) = data.label {
  1237→                    if let Some(edge) = scg.get_edge_mut(*id) {
  1238→                        edge.label = Some(label.clone());
  1239→                    }
  1240→                }
  1241→            }
  1242→        }
  1243→        ElementChange::Removed => {
  1244→            let _ = scg.remove_edge(*id);
  1245→        }
  1246→        ElementChange::Modified(data) => {
  1247→            if let Some(edge) = scg.get_edge_mut(*id) {
  1248→                edge.kind = data.kind.clone();
  1249→                edge.label = data.label.clone();
  1250→            }
  1251→        }
  1252→    }
  1253→}
  1254→
  1255→/// Apply a region change to the merged graph.
  1256→fn apply_region_change(scg: &mut SCG, id: &RegionId, change: &ElementChange<SCGRegion>) {
  1257→    match change {
  1258→        ElementChange::Added(data) => {
  1259→            if scg.get_region(*id).is_none() {
  1260→                scg.add_region(data.clone());
  1261→            }
  1262→        }
  1263→        ElementChange::Removed => {
  1264→            scg.remove_region(*id);
  1265→        }
  1266→        ElementChange::Modified(data) => {
  1267→            if let Some(region) = scg.get_region_mut(*id) {
  1268→                region.scope_level = data.scope_level;
  1269→                region.security_boundary = data.security_boundary;
  1270→                region.deployment_target = data.deployment_target.clone();
  1271→                region.nodes = data.nodes.clone();
  1272→            }
  1273→        }
  1274→    }
  1275→}
  1276→
  1277→/// Extract node data from a change, if available.
  1278→fn change_to_node_data(change: &ElementChange<NodeData>) -> Option<NodeData> {
  1279→    match change {
  1280→        ElementChange::Added(data) | ElementChange::Modified(data) => Some(data.clone()),
  1281→        ElementChange::Removed => None,
  1282→    }
  1283→}
  1284→
  1285→/// Extract edge data from a change, if available.
  1286→fn change_to_edge_data(change: &ElementChange<EdgeData>) -> Option<EdgeData> {
  1287→    match change {
  1288→        ElementChange::Added(data) | ElementChange::Modified(data) => Some(data.clone()),
  1289→        ElementChange::Removed => None,
  1290→    }
  1291→}
  1292→
  1293→/// Extract region data from a change, if available.
  1294→fn change_to_region_data(change: &ElementChange<SCGRegion>) -> Option<SCGRegion> {
  1295→    match change {
  1296→        ElementChange::Added(data) | ElementChange::Modified(data) => Some(data.clone()),
  1297→        ElementChange::Removed => None,
  1298→    }
  1299→}
