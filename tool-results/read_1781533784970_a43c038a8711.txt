     1→//! Structured JSON diagnostics output for LLM integration.
     2→//!
     3→//! This module provides machine-readable diagnostic messages that can be
     4→//! serialized to JSON, enabling LLM-based tools and IDE integrations to
     5→//! programmatically consume compiler errors, warnings, and hints.
     6→//!
     7→//! # Architecture
     8→//!
     9→//! Each diagnostic carries a structured error code (`"E001"`, `"W001"`, etc.),
    10→//! a severity level, a human-readable message, a source identifier (which
    11→//! compiler stage produced it), a precise source location, optional related
    12→//! information, and quick-fix suggestions.
    13→//!
    14→//! # Example JSON output
    15→//!
    16→//! ```json
    17→//! [
    18→//!   {
    19→//!     "code": "E002",
    20→//!     "severity": "error",
    21→//!     "message": "undefined variable `foo`",
    22→//!     "source": "parser",
    23→//!     "location": {
    24→//!       "file": "main.vu",
    25→//!       "start_line": 10,
    26→//!       "start_col": 4,
    27→//!       "end_line": 10,
    28→//!       "end_col": 7
    29→//!     },
    30→//!     "related": [],
    31→//!     "suggestions": ["did you mean `for`?"]
    32→//!   }
    33→//! ]
    34→//! ```
    35→//!
    36→//! # Integration
    37→//!
    38→//! Use [`from_parse_error`] to convert parser errors, [`from_vuma_error`] for
    39→//! pipeline-level errors, and [`from_codegen_error`] for code-generation
    40→//! failures.
    41→
    42→use serde::{Deserialize, Serialize};
    43→use std::fmt;
    44→
    45→use vuma_codegen::CodegenError;
    46→use vuma_parser::{offset_to_location, ParseError, ParseErrorKind, Severity as ParserSeverity};
    47→
    48→use crate::pipeline::VumaError;

    50→// ═══════════════════════════════════════════════════════════════════════════
    51→// DiagnosticSeverity
    52→// ═══════════════════════════════════════════════════════════════════════════
    53→
    54→/// Severity level for a machine-readable diagnostic.
    55→///
    56→/// Unlike the parser's `Severity` (which has `Note`), this uses `Hint` for
    57→/// LLM-consumable suggestions and `Info` for informational notes.
    58→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
    59→#[serde(rename_all = "lowercase")]
    60→pub enum DiagnosticSeverity {
    61→    /// A hard error — the program cannot be compiled.
    62→    Error,
    63→    /// A warning — suspicious code that is technically valid.
    64→    Warning,
    65→    /// An informational note, typically providing context.
    66→    Info,
    67→    /// A hint or quick-fix suggestion.
    68→    Hint,
    69→}
    70→
    71→impl fmt::Display for DiagnosticSeverity {
    72→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
    73→        match self {
    74→            DiagnosticSeverity::Error => write!(f, "error"),
    75→            DiagnosticSeverity::Warning => write!(f, "warning"),
    76→            DiagnosticSeverity::Info => write!(f, "info"),
    77→            DiagnosticSeverity::Hint => write!(f, "hint"),
    78→        }
    79→    }
    80→}
    81→
    82→impl From<ParserSeverity> for DiagnosticSeverity {
    83→    fn from(s: ParserSeverity) -> Self {
    84→        match s {
    85→            ParserSeverity::Error => DiagnosticSeverity::Error,
    86→            ParserSeverity::Warning => DiagnosticSeverity::Warning,
    87→            ParserSeverity::Note => DiagnosticSeverity::Info,
    88→        }
    89→    }
    90→}

    93→// DiagnosticSourceLocation
    94→// ═══════════════════════════════════════════════════════════════════════════
    95→
    96→/// Machine-readable source location with line/column ranges.
    97→///
    98→/// All positions are 1-based (as is conventional for compiler diagnostics)
    99→/// rather than the 0-based positions used internally by the parser.
   100→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   101→pub struct DiagnosticSourceLocation {
   102→    /// File path or module name.
   103→    pub file: String,
   104→    /// Start line (1-based).
   105→    pub start_line: u32,
   106→    /// Start column (1-based).
   107→    pub start_col: u32,
   108→    /// End line (1-based).
   109→    pub end_line: u32,
   110→    /// End column (1-based, exclusive).
   111→    pub end_col: u32,
   112→}
   113→
   114→impl DiagnosticSourceLocation {
   115→    /// Create a location spanning a single point (zero-width).
   116→    pub fn point(file: impl Into<String>, line: u32, col: u32) -> Self {
   117→        Self {
   118→            file: file.into(),
   119→            start_line: line,
   120→            start_col: col,
   121→            end_line: line,
   122→            end_col: col,
   123→        }
   124→    }
   125→
   126→    /// Create a location spanning a range within a single line.
   127→    pub fn range(
   128→        file: impl Into<String>,
   129→        line: u32,
   130→        start_col: u32,
   131→        end_col: u32,
   132→    ) -> Self {
   133→        Self {
   134→            file: file.into(),
   135→            start_line: line,
   136→            start_col,
   137→            end_line: line,
   138→            end_col,
   139→        }
   140→    }
   141→
   142→    /// Create a synthetic "unknown" location for diagnostics that cannot be
   143→    /// tied to a specific source position.
   144→    pub fn unknown() -> Self {
   145→        Self {
   146→            file: String::new(),
   147→            start_line: 0,
   148→            start_col: 0,
   149→            end_line: 0,
   150→            end_col: 0,
   151→        }
   152→    }
   153→
   154→    /// Returns `true` if this location represents an unknown / synthetic
   155→    /// position (all fields zero, no file).
   156→    pub fn is_unknown(&self) -> bool {
   157→        self.file.is_empty() && self.start_line == 0 && self.start_col == 0
   158→    }
   159→}
   160→
   161→impl fmt::Display for DiagnosticSourceLocation {
   162→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   163→        if self.is_unknown() {
   164→            write!(f, "<unknown>")
   165→        } else if self.start_line == self.end_line && self.start_col == self.end_col {
   166→            write!(f, "{}:{}:{}", self.file, self.start_line, self.start_col)
   167→        } else {
   168→            write!(
   169→                f,
   170→                "{}:{}:{}-{}:{}",
   171→                self.file, self.start_line, self.start_col, self.end_line, self.end_col
   172→            )
   173→        }
   174→    }
   175→}

   191→impl RelatedInfo {

   194→        Self {
   195→            location,
   196→            message: message.into(),
   197→        }
   198→    }
   199→}

   202→// Diagnostic codes
   203→// ═══════════════════════════════════════════════════════════════════════════
   204→
   205→/// Returns the diagnostic code string for a given [`ParseErrorKind`].
   206→///
   207→/// Mapping:
   208→///
   209→/// | Code  | Kind                    | Description                |
   210→/// |-------|-------------------------|----------------------------|
   211→/// | E001  | InvalidSyntax           | Syntax error               |
   212→/// | E002  | UndefinedReference / UndefinedVariable | Undefined variable |
   213→/// | E003  | TypeMismatch            | Type mismatch              |
   214→/// | E004  | DuplicateDefinition     | Duplicate definition       |
   215→/// | E005  | —                       | Invalid argument count     |
   216→/// | E006  | —                       | Invalid type               |
   217→/// | E007  | —                       | Missing return             |
   218→/// | E008  | —                       | Unreachable code           |
   219→/// | W001  | —                       | Unused variable            |
   220→/// | W002  | —                       | Implicit type conversion   |
   221→/// | W003  | —                       | Large constant (perf hint) |
   222→/// | E009  | UnexpectedToken         | Unexpected token           |
   223→/// | E010  | ExpectedToken           | Expected token             |
   224→/// | E011  | RegionError             | Region error               |
   225→/// | E012  | BDAnnotationError       | BD annotation error        |
   226→/// | E013  | InvalidCompoundOp       | Invalid compound operator  |
   227→/// | E014  | MissingSemicolon        | Missing semicolon          |
   228→/// | E015  | InvalidAddress          | Invalid address literal    |
   229→pub fn code_for_parse_error_kind(kind: &ParseErrorKind) -> &'static str {
   230→    match kind {
   231→        ParseErrorKind::InvalidSyntax => "E001",
   232→        ParseErrorKind::UndefinedReference | ParseErrorKind::UndefinedVariable => "E002",
   233→        ParseErrorKind::TypeMismatch => "E003",
   234→        ParseErrorKind::DuplicateDefinition => "E004",
   235→        ParseErrorKind::UnexpectedToken => "E009",
   236→        ParseErrorKind::ExpectedToken => "E010",
   237→        ParseErrorKind::RegionError => "E011",
   238→        ParseErrorKind::BDAnnotationError => "E012",
   239→        ParseErrorKind::InvalidCompoundOp => "E013",
   240→        ParseErrorKind::MissingSemicolon => "E014",
   241→        ParseErrorKind::InvalidAddress => "E015",
   242→        ParseErrorKind::LlmMistake => "E021",
   243→        ParseErrorKind::CStyleForLoop => "E022",
   244→        ParseErrorKind::UnknownType => "E023",
   245→    }
   246→}
   247→
   248→/// Returns the diagnostic code string for a given [`CodegenError`].
   249→pub fn code_for_codegen_error(err: &CodegenError) -> &'static str {
   250→    match err {
   251→        CodegenError::InvalidInstruction(_) => "E016",
   252→        CodegenError::RegisterAllocFailed(_) => "E017",
   253→        CodegenError::EncodingError(_) => "E018",
   254→        CodegenError::TranslationError(_) => "E019",
   255→        CodegenError::ElfError(_) => "E020",
   256→        CodegenError::UnknownVariable { .. } => "E002",
   257→        CodegenError::WasmSectionNotFound { .. } => "E016",
   258→    }
   259→}
   260→
   261→/// Human-readable description of a diagnostic code.
   262→pub fn code_description(code: &str) -> &'static str {
   263→    match code {
   264→        "E001" => "Syntax error",
   265→        "E002" => "Undefined variable",
   266→        "E003" => "Type mismatch",
   267→        "E004" => "Duplicate definition",
   268→        "E005" => "Invalid argument count",
   269→        "E006" => "Invalid type",
   270→        "E007" => "Missing return",
   271→        "E008" => "Unreachable code",
   272→        "E009" => "Unexpected token",
   273→        "E010" => "Expected token",
   274→        "E011" => "Region error",
   275→        "E012" => "BD annotation error",
   276→        "E013" => "Invalid compound assignment operator",
   277→        "E014" => "Missing semicolon",
   278→        "E015" => "Invalid address literal",
   279→        "E016" => "Invalid instruction",
   280→        "E017" => "Register allocation failed",
   281→        "E018" => "Encoding error",
   282→        "E019" => "IR translation error",
   283→        "E021" => "LLM code mismatch (Rust/C syntax in VUMA)",
   284→        "E022" => "C-style for loop (use range-based for instead)",
   285→        "E023" => "Unknown type (use VUMA sized integer types)",
   286→        "E020" => "ELF emission error",
   287→        "W001" => "Unused variable",
   288→        "W002" => "Implicit type conversion",
   289→        "W003" => "Large constant (performance hint)",
   290→        _ => "Unknown diagnostic code",
   291→    }
   292→}

   294→// ═══════════════════════════════════════════════════════════════════════════
   295→// VumaDiagnostic
   296→// ═══════════════════════════════════════════════════════════════════════════
   297→
   298→/// A machine-readable diagnostic with structured fields for LLM consumption.
   299→///
   300→/// This is the primary output type for JSON diagnostics. Every field is
   301→/// serializable and carries enough information for an LLM or IDE to
   302→/// understand, categorize, and potentially auto-fix the issue.
   303→#[derive(Debug, Clone, Serialize, Deserialize)]
   304→pub struct VumaDiagnostic {
   305→    /// Structured diagnostic code, e.g. `"E001"`, `"W001"`.
   306→    pub code: String,
   307→    /// Severity level.
   308→    pub severity: DiagnosticSeverity,
   309→    /// Human-readable message (suitable for display).
   310→    pub message: String,
   311→    /// Compiler stage that produced this diagnostic
   312→    /// (`"parser"`, `"scg"`, `"ir"`, `"codegen"`, etc.).
   313→    pub source: String,
   314→    /// Precise source location.
   315→    pub location: DiagnosticSourceLocation,
   316→    /// Related source locations with explanatory messages.
   317→    pub related: Vec<RelatedInfo>,
   318→    /// Quick-fix suggestions (e.g. `"did you mean 'fn'?"`,
   319→    /// `"add semicolon"`).
   320→    pub suggestions: Vec<String>,
   321→}
   322→
   323→impl VumaDiagnostic {
   324→    /// Create a new diagnostic with all fields.
   325→    pub fn new(
   326→        code: impl Into<String>,
   327→        severity: DiagnosticSeverity,
   328→        message: impl Into<String>,
   329→        source: impl Into<String>,
   330→        location: DiagnosticSourceLocation,
   331→    ) -> Self {
   332→        Self {
   333→            code: code.into(),
   334→            severity,
   335→            message: message.into(),
   336→            source: source.into(),
   337→            location,
   338→            related: Vec::new(),
   339→            suggestions: Vec::new(),
   340→        }
   341→    }
   342→
   343→    /// Add a related-info entry.
   344→    pub fn with_related(mut self, info: RelatedInfo) -> Self {
   345→        self.related.push(info);
   346→        self
   347→    }
   348→
   349→    /// Add a quick-fix suggestion.
   350→    pub fn with_suggestion(mut self, suggestion: impl Into<String>) -> Self {
   351→        self.suggestions.push(suggestion.into());
   352→        self
   353→    }
   354→
   355→    /// Serialize this diagnostic as a JSON string.
   356→    pub fn to_json(&self) -> String {
   357→        serde_json::to_string(self).unwrap_or_else(|_| {
   358→            r#"{"code":"E000","severity":"error","message":"JSON serialization failed","source":"diagnostics","location":{"file":"","start_line":0,"start_col":0,"end_line":0,"end_col":0},"related":[],"suggestions":[]}"#.to_string()
   359→        })
   360→    }
   361→
   362→    /// Serialize this diagnostic as a pretty-printed JSON string.
   363→    pub fn to_json_pretty(&self) -> String {
   364→        serde_json::to_string_pretty(self).unwrap_or_else(|_| self.to_json())
   365→    }
   366→}
   367→
   368→impl fmt::Display for VumaDiagnostic {
   369→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   370→        write!(
   371→            f,
   372→            "{}[{}]: {} ({} at {})",
   373→            self.severity, self.code, self.message, self.source, self.location
   374→        )?;
   375→        for related in &self.related {
   376→            write!(f, "\n  note: {} at {}", related.message, related.location)?;
   377→        }
   378→        for suggestion in &self.suggestions {
   379→            write!(f, "\n  help: {}", suggestion)?;
   380→        }
   381→        Ok(())
   382→    }
   383→}

   389→/// Serialize a slice of diagnostics as a JSON array.

   394→/// Serialize a slice of diagnostics as a pretty-printed JSON array.

   399→// ═══════════════════════════════════════════════════════════════════════════
   400→// Integration: ParseError → VumaDiagnostic
   401→// ═══════════════════════════════════════════════════════════════════════════
   402→
   403→/// Convert a [`ParseError`] to a [`VumaDiagnostic`].
   404→///
   405→/// `source` is the full source text (used to compute line/column from byte
   406→/// offsets). `file` is the optional file path.
   407→pub fn from_parse_error(
   408→    err: &ParseError,
   409→    source: &str,
   410→    file: Option<&str>,
   411→) -> VumaDiagnostic {
   412→    let code = code_for_parse_error_kind(&err.kind).to_string();
   413→    let loc = offset_to_location(source, err.span.start, file);
   414→
   415→    // Compute end position from span end.
   416→    let end_loc = if err.span.end > err.span.start && err.span.end <= source.len() {
   417→        offset_to_location(source, err.span.end, file)
   418→    } else {
   419→        loc.clone()
   420→    };
   421→
   422→    let diag_loc = DiagnosticSourceLocation {
   423→        file: loc.file.clone().unwrap_or_default(),
   424→        start_line: loc.line as u32 + 1, // convert 0-based to 1-based
   425→        start_col: loc.column as u32 + 1,
   426→        end_line: end_loc.line as u32 + 1,
   427→        end_col: end_loc.column as u32 + 1,
   428→    };
   429→
   430→    let mut diag = VumaDiagnostic::new(
   431→        code,
   432→        DiagnosticSeverity::Error,
   433→        &err.message,
   434→        "parser",
   435→        diag_loc,
   436→    );
   437→
   438→    if let Some(ref suggestion) = err.suggestion {
   439→        diag = diag.with_suggestion(format!("did you mean '{}'?", suggestion));
   440→    }
   441→
   442→    // Store the line text as a hint if available.
   443→    if let Some(ref line_text) = loc.line_text {
   444→        diag = diag.with_suggestion(format!("source: {}", line_text.trim()));
   445→    }
   446→
   447→    diag
   448→}
   449→
   450→/// Convert multiple [`ParseError`]s into [`VumaDiagnostic`]s.
   451→pub fn from_parse_errors(
   452→    errors: &[ParseError],
   453→    source: &str,
   454→    file: Option<&str>,
   455→) -> Vec<VumaDiagnostic> {
   456→    errors.iter().map(|e| from_parse_error(e, source, file)).collect()
   457→}
   458→
   459→// ═══════════════════════════════════════════════════════════════════════════
   460→// Integration: VumaError → Vec<VumaDiagnostic>
   461→// ═══════════════════════════════════════════════════════════════════════════
   462→
   463→/// Convert a [`VumaError`] to one or more [`VumaDiagnostic`]s.
   464→///
   465→/// Since `VumaError` does not carry the original source text, the resulting
   466→/// diagnostics will have [`DiagnosticSourceLocation::unknown()`] for variants
   467→/// that don't carry location information. For parse errors, prefer
   468→/// [`from_parse_error`] which can compute precise locations.
   469→pub fn from_vuma_error(err: &VumaError) -> Vec<VumaDiagnostic> {
   470→    match err {
   471→        VumaError::Parse { errors } => {
   472→            // Parse errors need source text for location computation;
   473→            // without it, we produce diagnostics with unknown locations.
   474→            errors
   475→                .iter()
   476→                .map(|pe| {
   477→                    let code = code_for_parse_error_kind(&pe.kind).to_string();
   478→                    let mut diag = VumaDiagnostic::new(
   479→                        code,
   480→                        DiagnosticSeverity::Error,
   481→                        &pe.message,
   482→                        "parser",
   483→                        DiagnosticSourceLocation::unknown(),
   484→                    );
   485→                    if let Some(ref suggestion) = pe.suggestion {
   486→                        diag = diag.with_suggestion(format!("did you mean '{}'?", suggestion));
   487→                    }
   488→                    diag
   489→                })
   490→                .collect()
   491→        }
   492→        VumaError::AstToScg { message } => {
   493→            vec![VumaDiagnostic::new(
   494→                "E001",
   495→                DiagnosticSeverity::Error,
   496→                message,
   497→                "scg",
   498→                DiagnosticSourceLocation::unknown(),
   499→            )]
   500→        }
   501→        VumaError::ScgValidation { errors } => errors
   502→            .iter()
   503→            .map(|msg| {
   504→                VumaDiagnostic::new(
   505→                    "E001",
   506→                    DiagnosticSeverity::Error,
   507→                    msg,
   508→                    "scg",
   509→                    DiagnosticSourceLocation::unknown(),
   510→                )
   511→            })
   512→            .collect(),
   513→        VumaError::ScgToMsg { error } => {
   514→            vec![VumaDiagnostic::new(
   515→                "E001",
   516→                DiagnosticSeverity::Error,
   517→                error.to_string(),
   518→                "msg",
   519→                DiagnosticSourceLocation::unknown(),
   520→            )]
   521→        }
   522→        VumaError::BdInference { node_id, message } => {
   523→            let mut diag = VumaDiagnostic::new(
   524→                "E001",
   525→                DiagnosticSeverity::Error,
   526→                message,
   527→                "bd",
   528→                DiagnosticSourceLocation::unknown(),
   529→            );
   530→            if let Some(id) = node_id {
   531→                diag = diag.with_suggestion(format!("at SCG node {}", id));
   532→            }
   533→            vec![diag]
   534→        }
   535→        VumaError::Verification { result } => {
   536→            vec![VumaDiagnostic::new(
   537→                "E001",
   538→                DiagnosticSeverity::Error,
   539→                format!("verification failed: {}", result.overall),
   540→                "ive",
   541→                DiagnosticSourceLocation::unknown(),
   542→            )]
   543→        }
   544→        VumaError::Transform { pass_name, errors } => errors
   545→            .iter()
   546→            .map(|msg| {
   547→                VumaDiagnostic::new(
   548→                    "E001",
   549→                    DiagnosticSeverity::Error,
   550→                    msg,
   551→                    format!("scg-transform:{}", pass_name),
   552→                    DiagnosticSourceLocation::unknown(),
   553→                )
   554→            })
   555→            .collect(),
   556→        VumaError::Codegen { error } => {
   557→            vec![from_codegen_error(error)]
   558→        }
   559→        VumaError::RegisterAlloc { message } => {
   560→            vec![VumaDiagnostic::new(
   561→                "E017",
   562→                DiagnosticSeverity::Error,
   563→                message,
   564→                "register-alloc",
   565→                DiagnosticSourceLocation::unknown(),
   566→            )]
   567→        }
   568→        VumaError::Emission { message } => {
   569→            vec![VumaDiagnostic::new(
   570→                "E020",
   571→                DiagnosticSeverity::Error,
   572→                message,
   573→                "elf-emission",
   574→                DiagnosticSourceLocation::unknown(),
   575→            )]
   576→        }
   577→        VumaError::CorInit { message } => {
   578→            vec![VumaDiagnostic::new(
   579→                "E001",
   580→                DiagnosticSeverity::Error,
   581→                message,
   582→                "cor-init",
   583→                DiagnosticSourceLocation::unknown(),
   584→            )]
   585→        }
   586→        VumaError::Multi { errors } => errors.iter().flat_map(from_vuma_error).collect(),
   587→        VumaError::ModuleResolution { errors } => errors
   588→            .iter()
   589→            .flat_map(|e| {
   590→                vec![VumaDiagnostic::new(
   591→                    "E001",
   592→                    DiagnosticSeverity::Error,
   593→                    e.to_string(),
   594→                    "module-resolution",
   595→                    DiagnosticSourceLocation::unknown(),
   596→                )]
   597→            })
   598→            .collect(),
   599→    }
   600→}

   606→/// Convert a [`CodegenError`] to a [`VumaDiagnostic`].
   607→pub fn from_codegen_error(err: &CodegenError) -> VumaDiagnostic {
   608→    let code = code_for_codegen_error(err).to_string();
   609→    let message = err.to_string();
   610→
   611→    let (source, suggestion) = match err {
   612→        CodegenError::UnknownVariable { name } => (
   613→            "codegen",
   614→            Some(format!("declare '{}' before use", name)),
   615→        ),
   616→        CodegenError::RegisterAllocFailed(_) => (
   617→            "register-alloc",
   618→            Some("reduce register pressure by splitting live ranges".to_string()),
   619→        ),
   620→        CodegenError::TranslationError(_) => (
   621→            "ir",
   622→            None,
   623→        ),
   624→        CodegenError::EncodingError(_) => (
   625→            "codegen",
   626→            None,
   627→        ),
   628→        CodegenError::ElfError(_) => (
   629→            "emit",
   630→            None,
   631→        ),
   632→        CodegenError::InvalidInstruction(_) => (
   633→            "codegen",
   634→            None,
   635→        ),
   636→        CodegenError::WasmSectionNotFound { section } => (
   637→            "codegen",
   638→            Some(format!("ensure '{}' section is generated", section)),
   639→        ),
   640→    };
   641→
   642→    let mut diag = VumaDiagnostic::new(
   643→        code,
   644→        DiagnosticSeverity::Error,
   645→        message,
   646→        source,
   647→        DiagnosticSourceLocation::unknown(),
   648→    );
   649→
   650→    if let Some(s) = suggestion {
   651→        diag = diag.with_suggestion(s);
   652→    }
   653→
   654→    diag
   655→}
