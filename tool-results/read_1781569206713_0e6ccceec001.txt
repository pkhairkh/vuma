     1→     1→//! VUMA CLI — Command-line interface for the VUMA compiler.

     4→     4→//! - `vuma build <file>` — Parse + compile to ARM64 ELF (default), save to output file

   336→   336→/// Determine the default output path for a given input file.
   337→   337→fn default_output_path(input: &Path) -> PathBuf {
   338→   338→    let stem = input.file_stem().unwrap_or_default().to_string_lossy();
   339→   339→    let dir = input.parent().unwrap_or(std::path::Path::new("."));
   340→   340→    dir.join(format!("{}.o", stem))
   341→   341→}

   347→   347→/// `vuma build <file>` — Parse + compile to ELF, save to output file.
   348→   348→fn cmd_build(
   349→   349→    cli: &Cli,
   350→   350→    file: &PathBuf,
   351→   351→    output: &Option<PathBuf>,
   352→   352→    target: &TargetArg,
   353→   353→) -> Result<(), String> {
   354→   354→    let source = read_source(file)?;
   355→   355→    let config = make_config(cli, CompileTarget::from(*target));
   356→   356→
   357→   357→    // Initialize telemetry collector if --telemetry is set.
   358→   358→    let mut telemetry = if cli.telemetry {
   359→   359→        let mut tc = TelemetryCollector::new();
   360→   360→        tc.set_opt_level(&format!("{:?}", cli.opt_level));
   361→   361→        tc.set_verification_level(&format!("{:?}", cli.verification));
   362→   362→        tc.set_target(&format!("{:?}", target));
   363→   363→        tc.set_debug_info(cli.debug);
   364→   364→        Some(tc)
   365→   365→    } else {
   366→   366→        None
   367→   367→    };
   368→   368→
   369→   369→    if let Some(ref mut tc) = telemetry { tc.stage_start("compile"); }
   370→   370→
   371→   371→    let compile_result = vuma::compile_with_recovery(&source, Some(file), &config);
   372→   372→
   373→   373→    if let Some(ref mut tc) = telemetry { tc.stage_end("compile"); }
   374→   374→
   375→   375→    match compile_result {
   376→   376→        CompileResult::Success(result) => {
   377→   377→            if let Some(ref mut tc) = telemetry {
   378→   378→                tc.set_scg_node_count(result.scg.node_count());
   379→   379→                tc.set_ir_function_count(result.ir_function_count);
   380→   380→                tc.set_ir_instruction_count(result.ir_instruction_count);
   381→   381→                tc.set_binary_size(result.binary.len());
   382→   382→            }
   383→   383→
   384→   384→            let out_path = output
   385→   385→                .as_ref()
   386→   386→                .cloned()
   387→   387→                .unwrap_or_else(|| default_output_path(file));
   388→   388→            fs::write(&out_path, &result.binary).map_err(|e| {
   389→   389→                format!(
   390→   390→                    "error: cannot write output file '{}': {}",
   391→   391→                    out_path.display(),
   392→   392→                    e
   393→   393→                )
   394→   394→            })?;
   395→   395→            // Make the output file executable on Unix.
   396→   396→            #[cfg(unix)]
   397→   397→            {
   398→   398→                use std::os::unix::fs::PermissionsExt;
   399→   399→                let mut perms = fs::metadata(&out_path)
   400→   400→                    .map_err(|e| format!("error: cannot stat '{}': {}", out_path.display(), e))?
   401→   401→                    .permissions();
   402→   402→                perms.set_mode(0o755);
   403→   403→                fs::set_permissions(&out_path, perms)
   404→   404→                    .map_err(|e| format!("error: cannot chmod '{}': {}", out_path.display(), e))?;
   405→   405→            }
   406→   406→
   407→   407→            if !cli.quiet {
   408→   408→                println!(
   409→   409→                    "Compiled {} -> {} ({} bytes, {} SCG nodes, {} IR instructions)",
   410→   410→                    file.display(),
   411→   411→                    out_path.display(),
   412→   412→                    result.binary.len(),
   413→   413→                    result.scg.node_count(),
   414→   414→                    result.ir_instruction_count,
   415→   415→                );
   416→   416→
   417→   417→                // Print stage timings.
   418→   418→                for (stage, ms) in &result.stage_timings {
   419→   419→                    println!("  {:20} {}ms", stage, ms);
   420→   420→                }
   421→   421→            }
   422→   422→
   423→   423→            // Output telemetry if requested.
   424→   424→            if let Some(tc) = telemetry {
   425→   425→                let report = tc.finalize();
   426→   426→                println!("\n{}", serde_json::to_string_pretty(&report).unwrap());
   427→   427→            }
   428→   428→
   429→   429→            Ok(())
   430→   430→        }
   431→   431→        CompileResult::Partial(partial) => {
   432→   432→            print_errors(&partial.diagnostics);
   433→   433→
   434→   434→            if cli.telemetry {
   435→   435→                if let Some(tc) = telemetry {
   436→   436→                    let report = tc.finalize();
   437→   437→                    eprintln!("\n{}", serde_json::to_string_pretty(&report).unwrap());
   438→   438→                }
   439→   439→            }
   440→   440→
   441→   441→            Err(format!(
   442→   442→                "compilation failed with {} error(s) (last completed stage: {})",
   443→   443→                partial.diagnostics.len(),
   444→   444→                partial.last_completed_stage
   445→   445→                    .map(|s| format!("{:?}", s))
   446→   446→                    .unwrap_or_else(|| "none".to_string())
   447→   447→            ))
   448→   448→        }
   449→   449→    }
   450→   450→}

   452→   452→/// `vuma run <file>` — Build + execute.
   453→   453→fn cmd_run(cli: &Cli, file: &PathBuf, args: &[String]) -> Result<(), String> {
   454→   454→    let source = read_source(file)?;
   455→   455→    let config = make_config(cli, CompileTarget::Linux);
   456→   456→    let result = compile_with_path(&source, Some(file), &config).map_err(|errors| {
   457→   457→        print_errors(&errors);
   458→   458→        global_logger().error("compile", &format!("compilation failed with {} error(s)", errors.len()));
   459→   459→        format!("compilation failed with {} error(s)", errors.len())
   460→   460→    })?;
   461→   461→    let tmp_dir = std::env::temp_dir();
   462→   462→    let exe_path = tmp_dir.join(format!("vuma_run_{}", std::process::id()));
   463→   463→    fs::write(&exe_path, &result.binary).map_err(|e| {
   464→   464→        format!(
   465→   465→            "error: cannot write temporary executable '{}': {}",
   466→   466→            exe_path.display(),
   467→   467→            e
   468→   468→        )
   469→   469→    })?;
   470→   470→
   471→   471→    // Make the file executable on Unix.
   472→   472→    #[cfg(unix)]
   473→   473→    {
   474→   474→        use std::os::unix::fs::PermissionsExt;
   475→   475→        let mut perms = fs::metadata(&exe_path)
   476→   476→            .map_err(|e| format!("error: cannot stat '{}': {}", exe_path.display(), e))?
   477→   477→            .permissions();
   478→   478→        perms.set_mode(0o755);
   479→   479→        fs::set_permissions(&exe_path, perms)
   480→   480→            .map_err(|e| format!("error: cannot chmod '{}': {}", exe_path.display(), e))?;
   481→   481→    }
   482→   482→
   483→   483→    // Try to execute: first natively (if on aarch64), then via qemu-aarch64.
   484→   484→    let native_output = Command::new(&exe_path).args(args).output();
   485→   485→
   486→   486→    let output = match native_output {
   487→   487→        Ok(out) => out,
   488→   488→        Err(_) => {
   489→   489→            // Native execution failed; try qemu-aarch64.
   490→   490→            let qemu_output = Command::new("qemu-aarch64")
   491→   491→                .arg(&exe_path)
   492→   492→                .args(args)
   493→   493→                .output()
   494→   494→                .map_err(|e| {
   495→   495→                    format!(
   496→   496→                        "error: failed to execute binary (neither native nor qemu-aarch64): {}",
   497→   497→                        e
   498→   498→                    )
   499→   499→                })?;
   500→   500→            qemu_output
   501→   501→        }
   502→   502→    };
   503→   503→
   504→   504→    io::stdout()
   505→   505→        .write_all(&output.stdout)
   506→   506→        .map_err(|e| format!("error: failed to write program output: {}", e))?;
   507→   507→    io::stderr()
   508→   508→        .write_all(&output.stderr)
   509→   509→        .map_err(|e| format!("error: failed to write program stderr: {}", e))?;
   510→   510→
   511→   511→    // Clean up temp file.
   512→   512→    let _ = fs::remove_file(&exe_path);
   513→   513→
   514→   514→    let code = output.status.code().unwrap_or(1);
   515→   515→    if code != 0 {
   516→   516→        return Err(format!("program exited with code {}", code));
   517→   517→    }
   518→   518→
   519→   519→    Ok(())
   520→   520→}

   522→   522→/// `vuma check <file>` — Parse + SCG + BD inference + IVE verification only.
   523→   523→fn cmd_check(cli: &Cli, file: &PathBuf) -> Result<(), String> {

   529→   529→    }

   534→   534→        print_errors(&errors);

   558→   558→/// `vuma emit <isa> <file>` — Compile to a specific ISA target.
   559→   559→///
   560→   560→/// Uses the direct AST → codegen SCG → IR path for better code quality
   561→   561→/// than the main pipeline (which goes through the semantic SCG and loses
   562→   562→/// most program semantics in the SCG→IR bridge).
   563→   563→fn cmd_emit(
   564→   564→    cli: &Cli,
   565→   565→    isa: &IsaArg,
   566→   566→    file: &PathBuf,
   567→   567→    output: &Option<PathBuf>,
   568→   568→) -> Result<(), String> {
   569→   569→    let source = read_source(file)?;
   570→   570→    let backend_kind = BackendKind::from(*isa);
   571→   571→
   572→   572→    // Step 1: Parse source → AST.
   573→   573→    let mut parser = vuma_parser::Parser::new(&source);
   574→   574→    let parse_result = parser.parse_program();
   575→   575→    if parse_result.is_err() {
   576→   576→        return Err(format!("parse error: {:?}", parse_result.errors));
   577→   577→    }
   578→   578→    if !parse_result.errors.is_empty() {
   579→   579→        eprintln!("[emit] WARNING: {} non-fatal parse errors:", parse_result.errors.len());
   580→   580→        for err in &parse_result.errors {
   581→   581→            eprintln!("[emit]   {:?}", err);
   582→   582→        }
   583→   583→    }
   584→   584→    let program = parse_result.value.unwrap();
   585→   585→
   586→   586→    // Step 2: Bridge parser AST → codegen SCG.
   587→   587→    let codegen_scg = bridge_ast_to_codegen_scg(&program);
   588→   588→
   589→   589→    // Step 3: Lower codegen SCG → IR.
   590→   590→    let mut ir_builder = ScgToIr::new();
   591→   591→    let ir_program = ir_builder.convert(&codegen_scg).map_err(|e| {
   592→   592→        format!("IR conversion error: {}", e)
   593→   593→    })?;
   594→   594→
   595→   595→    eprintln!("[emit] IR program has {} functions", ir_program.functions.len());
   596→   596→    for func in &ir_program.functions {
   597→   597→        eprintln!("[emit] Function: {} ({} params, {} vregs)", func.name, func.params.len(), func.vregs.len());
   598→   598→        for block in &func.blocks {
   599→   599→            eprintln!("[emit]   Block: {}", block.label);
   600→   600→            for instr in &block.instructions {
   601→   601→                eprintln!("[emit]     {:?}", instr);
   602→   602→            }
   603→   603→        }
   604→   604→    }
   605→   605→
   606→   606→    // Step 4: Create backend and allocate registers (with fallback).
   607→   607→    let backend = match create_backend(backend_kind) {
   608→   608→        Ok(b) => b,
   609→   609→        Err(e) => {
   610→   610→            let err_msg = format!("{}", e);
   611→   611→            global_logger().warn("emit", &format!("{} backend failed: {}", backend_kind.isa_name(), err_msg));
   612→   612→            // Try fallback to AArch64 if not already trying it
   613→   613→            if backend_kind != BackendKind::AArch64 {
   614→   614→                global_logger().info("emit", "falling back to aarch64 backend");
   615→   615→                create_backend(BackendKind::AArch64).map_err(|e2| {
   616→   616→                    format!("error: cannot create {} backend: {}, aarch64 fallback also failed: {}",
   617→   617→                        backend_kind.isa_name(), err_msg, e2)
   618→   618→                })?
   619→   619→            } else {
   620→   620→                return Err(format!("error: cannot create {} backend: {}", backend_kind.isa_name(), err_msg));
   621→   621→            }
   622→   622→        }
   623→   623→    };
   624→   624→
   625→   625→    let mut allocated_functions = Vec::new();
   626→   626→    for func in &ir_program.functions {
   627→   627→        match backend.allocate_registers(func) {
   628→   628→            Ok(allocated) => allocated_functions.push(allocated),
   629→   629→            Err(e) => {
   630→   630→                eprintln!("warning: register allocation failed for '{}': {}", func.name, e);
   631→   631→            }
   632→   632→        }
   633→   633→    }
   634→   634→
   635→   635→    let out_path = output
   636→   636→        .as_ref()
   637→   637→        .cloned()
   638→   638→        .unwrap_or_else(|| default_output_path(file));
   639→   639→
   640→   640→    // Step 5: Encode and write output.
   641→   641→    if !allocated_functions.is_empty() {
   642→   642→        let allocated_program = vuma_codegen::backend::AllocatedProgram {
   643→   643→            functions: allocated_functions,
   644→   644→            total_code_size: 0,
   645→   645→            total_data_size: 0,
   646→   646→        };
   647→   647→        match backend.encode_program(&allocated_program) {
   648→   648→            Ok(bytes) => {
   649→   649→                fs::write(&out_path, &bytes).map_err(|e| {
   650→   650→                    format!("error: cannot write output file '{}': {}", out_path.display(), e)
   651→   651→                })?;
   652→   652→                // Make the output file executable on Unix.
   653→   653→                #[cfg(unix)]
   654→   654→                {
   655→   655→                    use std::os::unix::fs::PermissionsExt;
   656→   656→                    let mut perms = fs::metadata(&out_path)
   657→   657→                        .map_err(|e| format!("error: cannot stat '{}': {}", out_path.display(), e))?
   658→   658→                        .permissions();
   659→   659→                    perms.set_mode(0o755);
   660→   660→                    fs::set_permissions(&out_path, perms)
   661→   661→                        .map_err(|e| format!("error: cannot chmod '{}': {}", out_path.display(), e))?;
   662→   662→                }
   663→   663→                println!(
   664→   664→                    "Emitted {} -> {} ({} bytes, ISA: {})",
   665→   665→                    file.display(),
   666→   666→                    out_path.display(),
   667→   667→                    bytes.len(),
   668→   668→                    backend.name(),
   669→   669→                );
   670→   670→            }
   671→   671→            Err(e) => {
   672→   672→                use vuma_codegen::backend::BackendError;
   673→   673→                let prefix = match &e {
   674→   674→                    BackendError::UnresolvedRelocation { .. } => "E037",
   675→   675→                    _ => "error",
   676→   676→                };
   677→   677→                return Err(format!("{}: {} encoding failed: {}", prefix, backend.name(), e));
   678→   678→            }
   679→   679→        }
   680→   680→    } else {
   681→   681→        return Err("no functions were successfully allocated".to_string());
   682→   682→    }
   683→   683→
   684→   684→    Ok(())
   685→   685→}
   686→   686→
   687→   687→/// Bridge parser AST → codegen SCG.
   688→   688→///
   689→   689→/// This converts the parser's AST into the codegen's SCG representation,
   690→   690→/// which can then be lowered to IR. This bypasses the main pipeline's
   691→   691→/// semantic SCG which loses most program semantics.
   692→   692→///
   693→   693→/// Key design: expressions are flattened into three-address code (TAC) by
   694→   694→/// introducing temporary variables for sub-expressions. This preserves the
   695→   695→/// full semantics of nested binary operations, function calls, dereferences,
   696→   696→/// and casts.
   697→   697→fn bridge_ast_to_codegen_scg(program: &vuma_parser::ast::Program) -> Scg {
   698→   698→    use vuma_parser::ast::Item;
   699→   699→
   700→   700→    let mut nodes: Vec<ScgNode> = Vec::new();
   701→   701→
   702→   702→    for item in &program.items {
   703→   703→        if let Item::FnDef(fn_def) = item {
   704→   704→            let params: Vec<ScgParam> = fn_def
   705→   705→                .params
   706→   706→                .iter()
   707→   707→                .map(|p| ScgParam {
   708→   708→                    name: p.name.clone(),
   709→   709→                    ty: bridge_type_to_codegen_scg(&p.ty),
   710→   710→                })
   711→   711→                .collect();
   712→   712→
   713→   713→            let results = if let Some(ref ret_ty) = fn_def.return_type {
   714→   714→                vec![bridge_type_to_codegen_scg(&Some(ret_ty.clone()))]
   715→   715→            } else {
   716→   716→                vec![]
   717→   717→            };
   718→   718→
   719→   719→            let mut ctx = BridgeCtx::new();
   720→   720→            let mut body = bridge_block_to_scg_stmts(&fn_def.body, &mut ctx);
   721→   721→
   722→   722→            // Ensure every function ends with a Return statement.
   723→   723→            // If the body doesn't end with a Return, add an implicit one.
   724→   724→            // When the function has a return type and the last statement was an
   725→   725→            // expression, use ctx.last_expr_result as the return value.
   726→   726→            let has_return = body.last().map_or(false, |s| matches!(s, ScgStatement::Return(_)));
   727→   727→            if !has_return {
   728→   728→                let ret_val = if !results.is_empty() {
   729→   729→                    // First check if the last expression was tracked.
   730→   730→                    if let Some(ref expr) = ctx.last_expr_result {
   731→   731→                        Some(expr.clone())
   732→   732→                    } else {
   733→   733→                        // Otherwise, look for the last computation/call result.
   734→   734→                        body.iter().rev().find_map(|s| match s {
   735→   735→                            ScgStatement::Computation(comp) => Some(ScgExpr::Var(comp.dst.clone())),
   736→   736→                            ScgStatement::Call(call) => call.dst.as_ref().map(|d| ScgExpr::Var(d.clone())),
   737→   737→                            _ => None,
   738→   738→                        })
   739→   739→                    }
   740→   740→                } else {
   741→   741→                    None
   742→   742→                };
   743→   743→                body.push(ScgStatement::Return(ret_val.into_iter().collect()));
   744→   744→            }
   745→   745→
   746→   746→            nodes.push(ScgNode::Function(ScgFunction {
   747→   747→                name: fn_def.name.clone(),
   748→   748→                params,
   749→   749→                results,
   750→   750→                body,
   751→   751→            }));
   752→   752→        }
   753→   753→    }
   754→   754→
   755→   755→    Scg { nodes }
   756→   756→}
   757→   757→
   758→   758→/// Context for the AST → codegen SCG bridge, tracking a monotonic temp counter
   759→   759→/// and the last expression result for implicit returns.
   760→   760→struct BridgeCtx {
   761→   761→    temp_counter: u32,
   762→   762→    /// The result of the last expression statement, if any.
   763→   763→    /// Used for implicit return when a function body ends with an expression.
   764→   764→    last_expr_result: Option<ScgExpr>,
   765→   765→}
   766→   766→
   767→   767→impl BridgeCtx {
   768→   768→    fn new() -> Self {
   769→   769→        Self { temp_counter: 0, last_expr_result: None }
   770→   770→    }
   771→   771→
   772→   772→    /// Allocate a unique temporary variable name.
   773→   773→    fn alloc_temp(&mut self) -> String {
   774→   774→        let name = format!("__t{}", self.temp_counter);
   775→   775→        self.temp_counter += 1;
   776→   776→        name
   777→   777→    }
   778→   778→}
   779→   779→
   780→   780→/// Convert a parser type annotation to a codegen SCG type.
   781→   781→fn bridge_type_to_codegen_scg(ty: &Option<vuma_parser::ast::Type>) -> ScgType {
   782→   782→    match ty {
   783→   783→        Some(vuma_parser::ast::Type::BDBase(name)) => match name.as_str() {
   784→   784→            "i8" => ScgType::I8,
   785→   785→            "i16" => ScgType::I16,
   786→   786→            "i32" => ScgType::I32,
   787→   787→            "i64" => ScgType::I64,
   788→   788→            "u8" => ScgType::U8,
   789→   789→            "u16" => ScgType::U16,
   790→   790→            "u32" => ScgType::U32,
   791→   791→            "u64" => ScgType::U64,
   792→   792→            _ => ScgType::I64,
   793→   793→        },
   794→   794→        Some(vuma_parser::ast::Type::Ptr(_)) => ScgType::Ptr,
   795→   795→        Some(vuma_parser::ast::Type::RegionPtr { .. }) => ScgType::Ptr,
   796→   796→        _ => ScgType::Void,
   797→   797→    }
   798→   798→}
   799→   799→
   800→   800→
   801→   801→/// Convert a parser block into codegen SCG statements, flattening expressions
   802→   802→/// into three-address code with temporaries.
   803→   803→fn bridge_block_to_scg_stmts(block: &vuma_parser::ast::Block, ctx: &mut BridgeCtx) -> Vec<ScgStatement> {
   804→   804→    block
   805→   805→        .statements
   806→   806→        .iter()
   807→   807→        .flat_map(|s| bridge_stmt_to_scg(s, ctx))
   808→   808→        .collect()
   809→   809→}
   810→   810→
   811→   811→/// Flatten an expression into three-address code. Returns the ScgExpr that
   812→   812→/// holds the result, and appends any intermediate computation statements
   813→   813→/// to `stmts`.
   814→   814→///
   815→   815→/// This is the core of the bridge: it recursively decomposes nested
   816→   816→/// expressions into a sequence of simple computation nodes, each operating
   817→   817→/// on at most two operands and producing one result. This preserves the
   818→   818→/// full semantics of the original expression tree.
   819→   819→fn flatten_expr(
   820→   820→    expr: &vuma_parser::ast::Expr,
   821→   821→    stmts: &mut Vec<ScgStatement>,
   822→   822→    ctx: &mut BridgeCtx,
   823→   823→) -> ScgExpr {
   824→   824→    use vuma_parser::ast::{Expr, Lit, UnOp};
   825→   825→
   826→   826→    match expr {
   827→   827→        // ── Leaf expressions: return directly, no flattening needed ──
   828→   828→        Expr::Var { name, .. } => ScgExpr::Var(name.clone()),
   829→   829→        Expr::Lit { value, .. } => match value {
   830→   830→            Lit::Int(n) => ScgExpr::Int(*n),
   831→   831→            Lit::Float(f) => ScgExpr::Float(*f),
   832→   832→            Lit::Bool(b) => ScgExpr::Int(if *b { 1 } else { 0 }),
   833→   833→            Lit::Address(a) => ScgExpr::Int(*a as i64),
   834→   834→            Lit::String(_) => ScgExpr::Int(0),
   835→   835→        },
   836→   836→
   837→   837→        // ── Binary operations: flatten lhs and rhs, then emit one Computation ──
   838→   838→        Expr::BinOp { op, lhs, rhs, .. } => {
   839→   839→            let lhs_expr = flatten_expr(lhs, stmts, ctx);
   840→   840→            let rhs_expr = flatten_expr(rhs, stmts, ctx);
   841→   841→            let dst = ctx.alloc_temp();
   842→   842→            let binop_kind = map_ast_binop(op);
   843→   843→            stmts.push(ScgStatement::Computation(ComputationNode {
   844→   844→                dst: dst.clone(),
   845→   845→                op: binop_kind,
   846→   846→                lhs: lhs_expr,
   847→   847→                rhs: rhs_expr,
   848→   848→                tail_call: false,
   849→   849→            }));
   850→   850→            ScgExpr::Var(dst)
   851→   851→        }
   852→   852→
   853→   853→        // ── Unary operations: flatten operand, then emit one Computation ──
   854→   854→        Expr::UnOp { op, expr: operand, .. } => {
   855→   855→            let operand_expr = flatten_expr(operand, stmts, ctx);
   856→   856→            let dst = ctx.alloc_temp();
   857→   857→            match op {
   858→   858→                UnOp::BitNot => {
   859→   859→                    stmts.push(ScgStatement::Computation(ComputationNode {
   860→   860→                        dst: dst.clone(),
   861→   861→                        op: BinOpKind::Xor,
   862→   862→                        lhs: operand_expr,
   863→   863→                        rhs: ScgExpr::Int(-1),
   864→   864→                        tail_call: false,
   865→   865→                    }));
   866→   866→                }
   867→   867→                UnOp::Neg => {
   868→   868→                    stmts.push(ScgStatement::Computation(ComputationNode {
   869→   869→                        dst: dst.clone(),
   870→   870→                        op: BinOpKind::Sub,
   871→   871→                        lhs: ScgExpr::Int(0),
   872→   872→                        rhs: operand_expr,
   873→   873→                        tail_call: false,
   874→   874→                    }));
   875→   875→                }
   876→   876→                UnOp::Not => {
   877→   877→                    stmts.push(ScgStatement::Computation(ComputationNode {
   878→   878→                        dst: dst.clone(),
   879→   879→                        op: BinOpKind::Eq,
   880→   880→                        lhs: operand_expr,
   881→   881→                        rhs: ScgExpr::Int(0),
   882→   882→                        tail_call: false,
   883→   883→                    }));
   884→   884→                }
   885→   885→                UnOp::Deref => {
   886→   886→                    stmts.push(ScgStatement::Access(AccessNode::Load {
   887→   887→                        dst: dst.clone(),
   888→   888→                        ptr: operand_expr,
   889→   889→                        offset: None,
   890→   890→                    }));
   891→   891→                }
   892→   892→            }
   893→   893→            ScgExpr::Var(dst)
   894→   894→        }
   895→   895→
   896→   896→        // ── Function call: flatten args, emit CallNode ──
   897→   897→        Expr::Call { callee, args, .. } => {
   898→   898→            let func_name = match callee.as_ref() {
   899→   899→                Expr::Var { name, .. } => name.clone(),
   900→   900→                _ => "_unknown".into(),
   901→   901→            };
   902→   902→            let flat_args: Vec<ScgExpr> = args.iter()
   903→   903→                .map(|a| flatten_expr(a, stmts, ctx))
   904→   904→                .collect();
   905→   905→            let dst = ctx.alloc_temp();
   906→   906→            stmts.push(ScgStatement::Call(CallNode {
   907→   907→                dst: Some(dst.clone()),
   908→   908→                func: func_name,
   909→   909→                args: flat_args,
   910→   910→                is_extern: false,
   911→   911→            }));
   912→   912→            ScgExpr::Var(dst)
   913→   913→        }
   914→   914→
   915→   915→        // ── Dereference: flatten the address, emit Load ──
   916→   916→        Expr::Deref { expr, .. } => {
   917→   917→            let addr = flatten_expr(expr, stmts, ctx);
   918→   918→            let dst = ctx.alloc_temp();
   919→   919→            stmts.push(ScgStatement::Access(AccessNode::Load {
   920→   920→                dst: dst.clone(),
   921→   921→                ptr: addr,
   922→   922→                offset: None,
   923→   923→            }));
   924→   924→            ScgExpr::Var(dst)
   925→   925→        }
   926→   926→
   927→   927→        // ── Offset (pointer arithmetic): flatten base and offset, emit Add ──
   928→   928→        Expr::Offset { base, offset, .. } => {
   929→   929→            let base_expr = flatten_expr(base, stmts, ctx);
   930→   930→            let off_expr = flatten_expr(offset, stmts, ctx);
   931→   931→            let dst = ctx.alloc_temp();
   932→   932→            stmts.push(ScgStatement::Computation(ComputationNode {
   933→   933→                dst: dst.clone(),
   934→   934→                op: BinOpKind::Add,
   935→   935→                lhs: base_expr,
   936→   936→                rhs: off_expr,
   937→   937→                tail_call: false,
   938→   938→            }));
   939→   939→            ScgExpr::Var(dst)
   940→   940→        }
   941→   941→
   942→   942→        // ── Cast: flatten operand, pass through ──
   943→   943→        Expr::Cast { expr, .. } => flatten_expr(expr, stmts, ctx),
   944→   944→
   945→   945→        // ── TypeAscription: flatten inner expression ──
   946→   946→        Expr::TypeAscription { expr, .. } => flatten_expr(expr, stmts, ctx),
   947→   947→
   948→   948→        // ── Index: flatten base and index, compute addr, emit Load ──
   949→   949→        Expr::Index { expr, index, .. } => {
   950→   950→            let base_expr = flatten_expr(expr, stmts, ctx);
   951→   951→            let idx_expr = flatten_expr(index, stmts, ctx);
   952→   952→            let addr = ctx.alloc_temp();
   953→   953→            stmts.push(ScgStatement::Computation(ComputationNode {
   954→   954→                dst: addr.clone(),
   955→   955→                op: BinOpKind::Add,
   956→   956→                lhs: base_expr,
   957→   957→                rhs: idx_expr,
   958→   958→                tail_call: false,
   959→   959→            }));
   960→   960→            let dst = ctx.alloc_temp();
   961→   961→            stmts.push(ScgStatement::Access(AccessNode::Load {
   962→   962→                dst: dst.clone(),
   963→   963→                ptr: ScgExpr::Var(addr),
   964→   964→                offset: None,
   965→   965→            }));
   966→   966→            ScgExpr::Var(dst)
   967→   967→        }
   968→   968→
   969→   969→        // ── Range: just flatten start (range is handled by For) ──
   970→   970→        Expr::Range { start, .. } => flatten_expr(start, stmts, ctx),
   971→   971→
   972→   972→        // ── Allocate: emit as a heap allocation call ──
   973→   973→        Expr::Allocate { size, .. } => {
   974→   974→            let size_expr = flatten_expr(size, stmts, ctx);
   975→   975→            let dst = ctx.alloc_temp();
   976→   976→            stmts.push(ScgStatement::Call(CallNode {
   977→   977→                dst: Some(dst.clone()),
   978→   978→                func: "__vuma_alloc".to_string(),
   979→   979→                args: vec![size_expr],
   980→   980→                is_extern: true,
   981→   981→            }));
   982→   982→            ScgExpr::Var(dst)
   983→   983→        }
   984→   984→
   985→   985→        // ── Null → 0 ──
   986→   986→        Expr::Null { .. } => ScgExpr::Int(0),
   987→   987→
   988→   988→        // ── Uninitialized → 0 ──
   989→   989→        Expr::Uninitialized { .. } => ScgExpr::Int(0),
   990→   990→
   991→   991→        // ── Fallback for unsupported expression types ──
   992→   992→        _ => ScgExpr::Int(0),
   993→   993→    }
   994→   994→}

   996→   996→/// Map a VUMA AST BinOp to a codegen BinOpKind.
   997→   997→fn map_ast_binop(op: &vuma_parser::ast::BinOp) -> BinOpKind {
   998→   998→    use vuma_parser::ast::BinOp;
   999→   999→    match op {
  1000→  1000→        BinOp::Add => BinOpKind::Add,
  1001→  1001→        BinOp::Sub => BinOpKind::Sub,
  1002→  1002→        BinOp::Mul => BinOpKind::Mul,
  1003→  1003→        BinOp::Div => BinOpKind::SDiv,
  1004→  1004→        BinOp::Mod => BinOpKind::SRem,
  1005→  1005→        BinOp::And => BinOpKind::And,
  1006→  1006→        BinOp::Or => BinOpKind::Or,
  1007→  1007→        BinOp::BitAnd => BinOpKind::And,
  1008→  1008→        BinOp::BitOr => BinOpKind::Or,
  1009→  1009→        BinOp::BitXor => BinOpKind::Xor,
  1010→  1010→        BinOp::Shl => BinOpKind::Shl,
  1011→  1011→        BinOp::Shr => BinOpKind::ShrL,
  1012→  1012→        BinOp::Eq => BinOpKind::Eq,
  1013→  1013→        BinOp::Ne => BinOpKind::Ne,
  1014→  1014→        BinOp::Lt => BinOpKind::SLt,
  1015→  1015→        BinOp::Le => BinOpKind::SLe,
  1016→  1016→        BinOp::Gt => BinOpKind::SGt,
  1017→  1017→        BinOp::Ge => BinOpKind::SGe,
  1018→  1018→    }
  1019→  1019→}

  1021→  1021→/// Convert a single parser statement into zero or more codegen SCG statements.
  1022→  1022→/// Uses `flatten_expr` to decompose nested expressions into three-address code.
  1023→  1023→fn bridge_stmt_to_scg(stmt: &vuma_parser::ast::Stmt, ctx: &mut BridgeCtx) -> Vec<ScgStatement> {
  1024→  1024→    use vuma_parser::ast::Stmt as PStmt;
  1025→  1025→
  1026→  1026→    match stmt {
  1027→  1027→        // ── let x [: T] = expr ──
  1028→  1028→        PStmt::Let(let_stmt) => {
  1029→  1029→            let mut stmts = Vec::new();
  1030→  1030→
  1031→  1031→            // Check if the RHS is an allocate() call → AllocationNode::Stack
  1032→  1032→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &let_stmt.value {
  1033→  1033→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
  1034→  1034→                    if name == "allocate" {
  1035→  1035→                        let size: u32 = args.first()
  1036→  1036→                            .and_then(|a| {
  1037→  1037→                                if let vuma_parser::ast::Expr::Lit { value, .. } = a {
  1038→  1038→                                    if let vuma_parser::ast::Lit::Int(n) = value {
  1039→  1039→                                        return Some(*n as u32);
  1040→  1040→                                    }
  1041→  1041→                                }
  1042→  1042→                                None
  1043→  1043→                            })
  1044→  1044→                            .unwrap_or(8);
  1045→  1045→                        return vec![ScgStatement::Allocation(AllocationNode::Stack {
  1046→  1046→                            name: let_stmt.name.clone(),
  1047→  1047→                            size,
  1048→  1048→                            ty: ScgType::Ptr,
  1049→  1049→                        })];
  1050→  1050→                    }
  1051→  1051→                    // Other function calls → CallNode (flatten args)
  1052→  1052→                    let flat_args: Vec<ScgExpr> = args.iter()
  1053→  1053→                        .map(|a| flatten_expr(a, &mut stmts, ctx))
  1054→  1054→                        .collect();
  1055→  1055→                    stmts.push(ScgStatement::Call(CallNode {
  1056→  1056→                        dst: Some(let_stmt.name.clone()),
  1057→  1057→                        func: name.clone(),
  1058→  1058→                        args: flat_args,
  1059→  1059→                        is_extern: false,
  1060→  1060→                    }));
  1061→  1061→                    return stmts;
  1062→  1062→                }
  1063→  1063→            }
  1064→  1064→
  1065→  1065→            // Check if the RHS is an Allocate expression → AllocationNode::Stack
  1066→  1066→            if let vuma_parser::ast::Expr::Allocate { size, .. } = &let_stmt.value {
  1067→  1067→                let size_val: u32 = match size.as_ref() {
  1068→  1068→                    vuma_parser::ast::Expr::Lit { value, .. } => {
  1069→  1069→                        if let vuma_parser::ast::Lit::Int(n) = value {
  1070→  1070→                            *n as u32
  1071→  1071→                        } else {
  1072→  1072→                            8
  1073→  1073→                        }
  1074→  1074→                    }
  1075→  1075→                    _ => 8,
  1076→  1076→                };
  1077→  1077→                return vec![ScgStatement::Allocation(AllocationNode::Stack {
  1078→  1078→                    name: let_stmt.name.clone(),
  1079→  1079→                    size: size_val,
  1080→  1080→                    ty: ScgType::Ptr,
  1081→  1081→                })];
  1082→  1082→            }
  1083→  1083→
  1084→  1084→            // General case: flatten the expression and assign to dst
  1085→  1085→            let result = flatten_expr(&let_stmt.value, &mut stmts, ctx);
  1086→  1086→            match &result {
  1087→  1087→                ScgExpr::Var(name) if name == &let_stmt.name => {}
  1088→  1088→                _ => {
  1089→  1089→                    stmts.push(ScgStatement::Computation(ComputationNode {
  1090→  1090→                        dst: let_stmt.name.clone(),
  1091→  1091→                        op: BinOpKind::Add,
  1092→  1092→                        lhs: result,
  1093→  1093→                        rhs: ScgExpr::Int(0),
  1094→  1094→                        tail_call: false,
  1095→  1095→                    }));
  1096→  1096→                }
  1097→  1097→            }
  1098→  1098→            stmts
  1099→  1099→        }
  1100→  1100→
  1101→  1101→        // ── target = value ──
  1102→  1102→        PStmt::Assign(assign_stmt) => {
  1103→  1103→            let mut stmts = Vec::new();
  1104→  1104→
  1105→  1105→            // Detect dereference writes: `*expr = val` → Access::Store
  1106→  1106→            if let vuma_parser::ast::AssignTarget::Deref { expr, .. } = &assign_stmt.target {
  1107→  1107→                let ptr = flatten_expr(expr, &mut stmts, ctx);
  1108→  1108→                let value = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
  1109→  1109→                stmts.push(ScgStatement::Access(AccessNode::Store {
  1110→  1110→                    ptr,
  1111→  1111→                    offset: None,
  1112→  1112→                    value,
  1113→  1113→                }));
  1114→  1114→                return stmts;
  1115→  1115→            }
  1116→  1116→
  1117→  1117→            // Handle Index target: ptr[index] = value
  1118→  1118→            if let vuma_parser::ast::AssignTarget::Index { expr, index, .. } = &assign_stmt.target {
  1119→  1119→                let base = flatten_expr(expr, &mut stmts, ctx);
  1120→  1120→                let idx = flatten_expr(index, &mut stmts, ctx);
  1121→  1121→                let addr = ctx.alloc_temp();
  1122→  1122→                stmts.push(ScgStatement::Computation(ComputationNode {
  1123→  1123→                    dst: addr.clone(),
  1124→  1124→                    op: BinOpKind::Add,
  1125→  1125→                    lhs: base,
  1126→  1126→                    rhs: idx,
  1127→  1127→                    tail_call: false,
  1128→  1128→                }));
  1129→  1129→                let value = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
  1130→  1130→                stmts.push(ScgStatement::Access(AccessNode::Store {
  1131→  1131→                    ptr: ScgExpr::Var(addr),
  1132→  1132→                    offset: None,
  1133→  1133→                    value,
  1134→  1134→                }));
  1135→  1135→                return stmts;
  1136→  1136→            }
  1137→  1137→
  1138→  1138→            let dst = match &assign_stmt.target {
  1139→  1139→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
  1140→  1140→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
  1141→  1141→                vuma_parser::ast::AssignTarget::Deref { .. } => "_deref".into(),
  1142→  1142→                vuma_parser::ast::AssignTarget::Index { .. } => "_index".into(),
  1143→  1143→            };
  1144→  1144→
  1145→  1145→            // Detect allocate() expression → AllocationNode::Stack
  1146→  1146→            if let vuma_parser::ast::Expr::Allocate { size, .. } = &assign_stmt.value {
  1147→  1147→                let size_val: u32 = match size.as_ref() {
  1148→  1148→                    vuma_parser::ast::Expr::Lit { value, .. } => {
  1149→  1149→                        if let vuma_parser::ast::Lit::Int(n) = value {
  1150→  1150→                            *n as u32
  1151→  1151→                        } else {
  1152→  1152→                            8
  1153→  1153→                        }
  1154→  1154→                    }
  1155→  1155→                    _ => 8,
  1156→  1156→                };
  1157→  1157→                return vec![ScgStatement::Allocation(AllocationNode::Stack {
  1158→  1158→                    name: dst,
  1159→  1159→                    size: size_val,
  1160→  1160→                    ty: ScgType::Ptr,
  1161→  1161→                })];
  1162→  1162→            }
  1163→  1163→
  1164→  1164→            // Detect function calls in assign: `x = foo(args)` → CallNode (flatten args)
  1165→  1165→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &assign_stmt.value {
  1166→  1166→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
  1167→  1167→                    let flat_args: Vec<ScgExpr> = args.iter()
  1168→  1168→                        .map(|a| flatten_expr(a, &mut stmts, ctx))
  1169→  1169→                        .collect();
  1170→  1170→                    stmts.push(ScgStatement::Call(CallNode {
  1171→  1171→                        dst: Some(dst),
  1172→  1172→                        func: name.clone(),
  1173→  1173→                        args: flat_args,
  1174→  1174→                        is_extern: false,
  1175→  1175→                    }));
  1176→  1176→                    return stmts;
  1177→  1177→                }
  1178→  1178→            }
  1179→  1179→
  1180→  1180→            // General case: flatten the value expression and assign to dst
  1181→  1181→            let result = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
  1182→  1182→            match &result {
  1183→  1183→                ScgExpr::Var(name) if name == &dst => {}
  1184→  1184→                _ => {
  1185→  1185→                    stmts.push(ScgStatement::Computation(ComputationNode {
  1186→  1186→                        dst,
  1187→  1187→                        op: BinOpKind::Add,
  1188→  1188→                        lhs: result,
  1189→  1189→                        rhs: ScgExpr::Int(0),
  1190→  1190→                        tail_call: false,
  1191→  1191→                    }));
  1192→  1192→                }
  1193→  1193→            }
  1194→  1194→            stmts
  1195→  1195→        }
  1196→  1196→
  1197→  1197→        // ── target op= value ──
  1198→  1198→        PStmt::CompoundAssign(ca_stmt) => {
  1199→  1199→            let mut stmts = Vec::new();
  1200→  1200→            let dst = match &ca_stmt.target {
  1201→  1201→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
  1202→  1202→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
  1203→  1203→                _ => "_".into(),
  1204→  1204→            };
  1205→  1205→            let binop = match ca_stmt.op {
  1206→  1206→                vuma_parser::ast::CompoundOp::Add => BinOpKind::Add,
  1207→  1207→                vuma_parser::ast::CompoundOp::Sub => BinOpKind::Sub,
  1208→  1208→                vuma_parser::ast::CompoundOp::Mul => BinOpKind::Mul,
  1209→  1209→                vuma_parser::ast::CompoundOp::Div => BinOpKind::SDiv,
  1210→  1210→                vuma_parser::ast::CompoundOp::Mod => BinOpKind::SRem,
  1211→  1211→                vuma_parser::ast::CompoundOp::BitAnd => BinOpKind::And,
  1212→  1212→                vuma_parser::ast::CompoundOp::BitOr => BinOpKind::Or,
  1213→  1213→                vuma_parser::ast::CompoundOp::BitXor => BinOpKind::Xor,
  1214→  1214→                vuma_parser::ast::CompoundOp::Shl => BinOpKind::Shl,
  1215→  1215→                vuma_parser::ast::CompoundOp::Shr => BinOpKind::ShrL,
  1216→  1216→            };
  1217→  1217→            let rhs = flatten_expr(&ca_stmt.value, &mut stmts, ctx);
  1218→  1218→            stmts.push(ScgStatement::Computation(ComputationNode {
  1219→  1219→                dst: dst.clone(),
  1220→  1220→                op: binop,
  1221→  1221→                lhs: ScgExpr::Var(dst),
  1222→  1222→                rhs,
  1223→  1223→                tail_call: false,
  1224→  1224→            }));
  1225→  1225→            stmts
  1226→  1226→        }
  1227→  1227→
  1228→  1228→        PStmt::Allocate(_) => vec![],
  1229→  1229→        PStmt::Free(_) => vec![],
  1230→  1230→
  1231→  1231→        PStmt::Cast(cast_stmt) => {
  1232→  1232→            let mut stmts = Vec::new();
  1233→  1233→            let _ = flatten_expr(&cast_stmt.expr, &mut stmts, ctx);
  1234→  1234→            stmts
  1235→  1235→        }
  1236→  1236→
  1237→  1237→        PStmt::Return(ret_stmt) => {
  1238→  1238→            let mut stmts = Vec::new();
  1239→  1239→            let values = match &ret_stmt.value {
  1240→  1240→                Some(expr) => vec![flatten_expr(expr, &mut stmts, ctx)],
  1241→  1241→                None => vec![],
  1242→  1242→            };
  1243→  1243→            stmts.push(ScgStatement::Return(values));
  1244→  1244→            stmts
  1245→  1245→        }
  1246→  1246→
  1247→  1247→        PStmt::Expr(expr_stmt) => {
  1248→  1248→            let mut stmts = Vec::new();
  1249→  1249→            let result = flatten_expr(&expr_stmt.expr, &mut stmts, ctx);
  1250→  1250→            ctx.last_expr_result = Some(result);
  1251→  1251→            stmts
  1252→  1252→        }
  1253→  1253→
  1254→  1254→        PStmt::If(if_stmt) => {
  1255→  1255→            let mut pre_stmts = Vec::new();
  1256→  1256→            let cond = flatten_expr(&if_stmt.condition, &mut pre_stmts, ctx);
  1257→  1257→            let then_body = bridge_block_to_scg_stmts(&if_stmt.then_block, ctx);
  1258→  1258→            let else_body = if_stmt
  1259→  1259→                .else_block
  1260→  1260→                .as_ref()
  1261→  1261→                .map(|b| bridge_block_to_scg_stmts(b, ctx));
  1262→  1262→            let mut result = pre_stmts;
  1263→  1263→            result.push(ScgStatement::Control(ControlNode::If {
  1264→  1264→                cond,
  1265→  1265→                then_body,
  1266→  1266→                else_body,
  1267→  1267→            }));
  1268→  1268→            result
  1269→  1269→        }
  1270→  1270→
  1271→  1271→        // while condition { body }
  1272→  1272→        // Lowered as: Loop { <compute cond>; if cond { body } else { break } }
  1273→  1273→        PStmt::While(while_stmt) => {
  1274→  1274→            let then_body = bridge_block_to_scg_stmts(&while_stmt.body, ctx);
  1275→  1275→            let mut loop_body = Vec::new();
  1276→  1276→            let cond = flatten_expr(&while_stmt.condition, &mut loop_body, ctx);
  1277→  1277→            loop_body.push(ScgStatement::Control(ControlNode::If {
  1278→  1278→                cond,
  1279→  1279→                then_body,
  1280→  1280→                else_body: Some(vec![ScgStatement::Control(ControlNode::Break)]),
  1281→  1281→            }));
  1282→  1282→            vec![ScgStatement::Control(ControlNode::Loop { body: loop_body })]
  1283→  1283→        }
  1284→  1284→
  1285→  1285→        // for name in start..end { body }
  1286→  1286→        // Lowered as:
  1287→  1287→        //   name = start
  1288→  1288→        //   loop { _cond = name < end; if _cond { body; name = name + 1 } else { break } }
  1289→  1289→        PStmt::For(for_stmt) => {
  1290→  1290→            let mut pre_stmts = Vec::new();
  1291→  1291→            let (start_expr, end_expr) = match &for_stmt.iter {
  1292→  1292→                vuma_parser::ast::Expr::Range { start, end, .. } => {
  1293→  1293→                    let s = flatten_expr(start, &mut pre_stmts, ctx);
  1294→  1294→                    let e = flatten_expr(end, &mut pre_stmts, ctx);
  1295→  1295→                    (s, e)
  1296→  1296→                }
  1297→  1297→                other => {
  1298→  1298→                    (ScgExpr::Int(0), ScgExpr::Int(0))
  1299→  1299→                }
  1300→  1300→            };
  1301→  1301→
  1302→  1302→            let init_stmt = ScgStatement::Computation(ComputationNode {
  1303→  1303→                dst: for_stmt.name.clone(),
  1304→  1304→                op: BinOpKind::Add,
  1305→  1305→                lhs: start_expr,
  1306→  1306→                rhs: ScgExpr::Int(0),
  1307→  1307→                tail_call: false,
  1308→  1308→            });
  1309→  1309→
  1310→  1310→            let mut loop_body = Vec::new();
  1311→  1311→            let cond_temp = format!("{}_cond", for_stmt.name);
  1312→  1312→            loop_body.push(ScgStatement::Computation(ComputationNode {
  1313→  1313→                dst: cond_temp.clone(),
  1314→  1314→                op: BinOpKind::SLt,
  1315→  1315→                lhs: ScgExpr::Var(for_stmt.name.clone()),
  1316→  1316→                rhs: end_expr.clone(),
  1317→  1317→                tail_call: false,
  1318→  1318→            }));
  1319→  1319→
  1320→  1320→            let inner_body = bridge_block_to_scg_stmts(&for_stmt.body, ctx);
  1321→  1321→            let mut full_then = inner_body;
  1322→  1322→            full_then.push(ScgStatement::Computation(ComputationNode {
  1323→  1323→                dst: for_stmt.name.clone(),
  1324→  1324→                op: BinOpKind::Add,
  1325→  1325→                lhs: ScgExpr::Var(for_stmt.name.clone()),
  1326→  1326→                rhs: ScgExpr::Int(1),
  1327→  1327→                tail_call: false,
  1328→  1328→            }));
  1329→  1329→
  1330→  1330→            loop_body.push(ScgStatement::Control(ControlNode::If {
  1331→  1331→                cond: ScgExpr::Var(cond_temp),
  1332→  1332→                then_body: full_then,
  1333→  1333→                else_body: Some(vec![ScgStatement::Control(ControlNode::Break)]),
  1334→  1334→            }));
  1335→  1335→
  1336→  1336→            let mut result = pre_stmts;
  1337→  1337→            result.push(init_stmt);
  1338→  1338→            result.push(ScgStatement::Control(ControlNode::Loop { body: loop_body }));
  1339→  1339→            result
  1340→  1340→        }
  1341→  1341→
  1342→  1342→        PStmt::Loop(loop_stmt) => {
  1343→  1343→            let body = bridge_block_to_scg_stmts(&loop_stmt.body, ctx);
  1344→  1344→            vec![ScgStatement::Control(ControlNode::Loop { body })]
  1345→  1345→        }
  1346→  1346→
  1347→  1347→        PStmt::Break(_) => vec![ScgStatement::Control(ControlNode::Break)],
  1348→  1348→        PStmt::Continue(_) => vec![ScgStatement::Control(ControlNode::Continue)],
  1349→  1349→        PStmt::BdDirective(_)
  1350→  1350→        | PStmt::Sync(_)
  1351→  1351→        | PStmt::UnsafeBlock { .. }
  1352→  1352→        | PStmt::Match(_)
  1353→  1353→        | PStmt::Access(_) => vec![],
  1354→  1354→    }
  1355→  1355→}
  1356→  1356→fn cmd_disasm(file: &PathBuf, isa: &IsaArg, base_addr_str: &str) -> Result<(), String> {
  1357→  1357→    let bytes = fs::read(file)
  1358→  1358→        .map_err(|e| format!("error: cannot read binary file '{}': {}", file.display(), e))?;

  1362→  1362→

  1366→  1366→            "error: cannot create {} backend: {}",
  1367→  1367→            backend_kind.isa_name(),
  1368→  1368→            e
  1369→  1369→        )
  1370→  1370→    })?;
  1371→  1371→
  1372→  1372→    let instructions = backend.disassemble(&bytes, base_addr);
  1373→  1373→
  1374→  1374→    println!(
  1375→  1375→        "Disassembly of {} ({} bytes, ISA: {}):",
  1376→  1376→        file.display(),
  1377→  1377→        bytes.len(),
  1378→  1378→        backend.name()
  1379→  1379→    );
  1380→  1380→    for line in &instructions {
  1381→  1381→        println!("{}", line);
  1382→  1382→    }
  1383→  1383→
  1384→  1384→    Ok(())
  1385→  1385→}

  1387→  1387→/// `vuma verify <file>` — Run IVE 5-invariant verification.

  1422→  1422→                println!(

  1429→  1429→            }
  1430→  1430→        }
  1431→  1431→        None => {

  1433→  1433→        }

  1436→  1436→    Ok(())
  1437→  1437→}

  1439→  1439→/// `vuma repl` — Interactive REPL: parse expressions and print AST.
  1440→  1440→fn cmd_repl() -> Result<(), String> {
  1441→  1441→    println!("VUMA REPL v0.1.0");
  1442→  1442→    println!("Type VUMA expressions or statements. Enter ':quit' to exit, ':help' for help.");
  1443→  1443→
  1444→  1444→    let stdin = io::stdin();
  1445→  1445→    let mut input = String::new();
  1446→  1446→
  1447→  1447→    loop {
  1448→  1448→        print!("vuma> ");
  1449→  1449→        io::stdout()
  1450→  1450→            .flush()
  1451→  1451→            .map_err(|e| format!("flush error: {}", e))?;
  1452→  1452→
  1453→  1453→        input.clear();
  1454→  1454→        match stdin.read_line(&mut input) {
  1455→  1455→            Ok(0) => break, // EOF
  1456→  1456→            Ok(_) => {}
  1457→  1457→            Err(e) => return Err(format!("error reading input: {}", e)),
  1458→  1458→        }
  1459→  1459→
  1460→  1460→        let trimmed = input.trim();
  1461→  1461→        if trimmed.is_empty() {
  1462→  1462→            continue;
  1463→  1463→        }
  1464→  1464→        if trimmed == ":quit" || trimmed == ":q" {
  1465→  1465→            break;
  1466→  1466→        }
  1467→  1467→        if trimmed == ":help" || trimmed == ":h" {
  1468→  1468→            println!("  :quit, :q   — Exit the REPL");
  1469→  1469→            println!("  :help, :h   — Show this help");
  1470→  1470→            println!("  <expr>      — Parse and display the AST");
  1471→  1471→            println!("  <stmt>      — Parse and display the AST");
  1472→  1472→            println!("  <program>   — Parse and display the full program AST");
  1473→  1473→            continue;
  1474→  1474→        }
  1475→  1475→
  1476→  1476→        // Try to parse as a full program first.
  1477→  1477→        let mut parser = vuma_parser::Parser::new(trimmed);
  1478→  1478→        let result = parser.parse_program();
  1479→  1479→        if !result.has_errors() {
  1480→  1480→            // Successfully parsed; display the AST.
  1481→  1481→            let program = result.unwrap();
  1482→  1482→            println!("{:#?}", program);
  1483→  1483→            continue;
  1484→  1484→        }
  1485→  1485→        // Program parse had errors; try to parse as an expression via
  1486→  1486→        // wrapping it in a dummy program context.
  1487→  1487→        let program_errors = result.errors.clone();
  1488→  1488→        // Since parse_expr is private, we re-parse as a minimal program.
  1489→  1489→        let wrapped = format!("fn _repl_expr() {{ {} }}", trimmed);
  1490→  1490→        let mut expr_parser = vuma_parser::Parser::new(&wrapped);
  1491→  1491→        let expr_result = expr_parser.parse_program();
  1492→  1492→        if !expr_result.has_errors() {
  1493→  1493→            // Print the function body AST.
  1494→  1494→            let program = expr_result.unwrap();
  1495→  1495→            if let Some(item) = program.items.first() {
  1496→  1496→                println!("{:#?}", item);
  1497→  1497→            } else {
  1498→  1498→                println!("{:#?}", program);
  1499→  1499→            }
  1500→  1500→        } else {
  1501→  1501→            // Show the original parse errors.
  1502→  1502→            for err in &program_errors {
  1503→  1503→                eprintln!("parse error: {}", err);
  1504→  1504→            }
  1505→  1505→        }
  1506→  1506→    }
  1507→  1507→
  1508→  1508→    Ok(())
  1509→  1509→}
