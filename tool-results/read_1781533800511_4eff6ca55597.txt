     1→//! # Proof Checker
     2→//!
     3→//! Verifies that a formal proof is valid: every step follows from previously
     4→//! established facts using the stated rule, and there is no circular reasoning.
     5→
     6→use serde::{Deserialize, Serialize};
     7→use thiserror::Error;
     8→
     9→use crate::proof::{Conclusion, Fact, FactId, Proof, ProofStep};
    10→use crate::tactics::ProofResult;
    11→
    12→// ---------------------------------------------------------------------------
    13→// ProofCache
    14→// ---------------------------------------------------------------------------
    15→
    16→/// Cache for proof checking results, keyed by a goal fingerprint.
    17→///
    18→/// The cache avoids re-checking goals that have already been verified.
    19→/// The key is typically a hash (fingerprint) of the goal's invariant name,
    20→/// target, and context. The value is the cached [`ProofResult`].
    21→pub type ProofCache = std::collections::HashMap<u64, ProofResult>;
    22→
    23→// ---------------------------------------------------------------------------
    24→// CheckResult
    25→// ---------------------------------------------------------------------------
    26→
    27→/// The result of checking a proof.
    28→#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
    29→pub enum CheckResult {
    30→    /// Every step in the proof is valid and the conclusion follows.
    31→    Valid,
    32→
    33→    /// The proof is invalid at the given step index for the stated reason.
    34→    Invalid {
    35→        /// Zero-based index of the offending step.
    36→        step: usize,
    37→        /// Human-readable explanation of why the step is invalid.
    38→        reason: String,
    39→    },
    40→
    41→    /// The proof has not reached a definitive conclusion (still `Inconclusive`).
    42→    Incomplete,
    43→}
    44→
    45→impl std::fmt::Display for CheckResult {
    46→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    47→        match self {
    48→            CheckResult::Valid => write!(f, "valid"),
    49→            CheckResult::Invalid { step, reason } => {
    50→                write!(f, "invalid at step {}: {}", step, reason)
    51→            }
    52→            CheckResult::Incomplete => write!(f, "incomplete"),
    53→        }
    54→    }
    55→}
    56→
    57→// ---------------------------------------------------------------------------
    58→// Checker errors (internal)
    59→// ---------------------------------------------------------------------------
    60→
    61→/// Internal errors during proof checking (distinct from `CheckResult::Invalid`,
    62→/// which is a normal outcome).
    63→#[derive(Debug, Clone, Error)]
    64→pub enum CheckerError {
    65→    /// A referenced fact id was not found in the established fact set.
    66→    #[error("fact id {id} not found in established facts")]
    67→    FactNotFound { id: FactId },
    68→
    69→    /// A sub-proof could not be checked.
    70→    #[error("sub-proof check failed: {0}")]
    71→    SubProofFailed(String),
    72→}
    73→
    74→// ---------------------------------------------------------------------------
    75→// ProofChecker
    76→// ---------------------------------------------------------------------------
    77→
    78→/// Checks formal proofs for validity.
    79→///
    80→/// The checker walks through each step of a proof, maintains a set of
    81→/// established facts, and verifies that:
    82→///
    83→/// 1. **Inference steps**: The stated rule can be applied to the referenced
    84→///    premises to produce the claimed conclusion.
    85→/// 2. **Circular reasoning**: No fact is used before it is established (i.e.
    86→///    a step may only reference fact ids that appear in earlier steps).
    87→/// 3. **Conclusion consistency**: If the proof concludes `Proven`, at least
    88→///    one step must establish the goal; if `Refuted`, a contradiction must
    89→///    be present.
    90→///
    91→/// # Example
    92→///
    93→/// ```ignore
    94→/// use vuma_proof::checker::ProofChecker;
    95→/// use vuma_proof::proof::Proof;
    96→///
    97→/// let proof = /* construct a proof */;
    98→/// let checker = ProofChecker::new();
    99→/// match checker.check(&proof) {
   100→///     Ok(CheckResult::Valid) => println!("Proof is valid!"),
   101→///     Ok(CheckResult::Invalid { step, reason }) => {
   102→///         println!("Invalid at step {}: {}", step, reason);
   103→///     }
   104→///     Ok(CheckResult::Incomplete) => println!("Proof is incomplete."),
   105→///     Err(e) => println!("Checker error: {}", e),
   106→/// }
   107→/// ```
   108→#[derive(Debug, Default)]
   109→pub struct ProofChecker {
   110→    /// If true, check for circular reasoning (fact ids used before being
   111→    /// established). Enabled by default.
   112→    check_circular: bool,
   113→}
   114→
   115→impl ProofChecker {
   116→    /// Create a new proof checker with default settings.
   117→    pub fn new() -> Self {
   118→        Self {
   119→            check_circular: true,
   120→        }
   121→    }
   122→
   123→    /// Disable circular-reasoning checking (useful for incremental checks).
   124→    pub fn without_circular_checks(mut self) -> Self {
   125→        self.check_circular = false;
   126→        self
   127→    }
   128→
   129→    /// Check a proof for validity.
   130→    ///
   131→    /// Returns `Ok(CheckResult)` on success (where the result itself may be
   132→    /// `Valid`, `Invalid`, or `Incomplete`), or `Err(CheckerError)` if the
   133→    /// checker encounters an internal error.
   134→    pub fn check(&self, proof: &Proof) -> Result<CheckResult, CheckerError> {
   135→        // Set of fact ids that have been established so far.
   136→        let mut established: std::collections::HashSet<FactId> = std::collections::HashSet::new();
   137→
   138→        for (step_idx, step) in proof.steps.iter().enumerate() {
   139→            match step {
   140→                ProofStep::Assume { fact } => {
   141→                    // Assumptions are always valid as long as the id is fresh.
   142→                    if established.contains(&fact.id) {
   143→                        return Ok(CheckResult::Invalid {
   144→                            step: step_idx,
   145→                            reason: format!(
   146→                                "fact id {} is already established (duplicate)",
   147→                                fact.id
   148→                            ),
   149→                        });
   150→                    }
   151→                    established.insert(fact.id);
   152→                }
   153→
   154→                ProofStep::Infer {
   155→                    from,
   156→                    rule,
   157→                    conclusion,
   158→                } => {
   159→                    // 1. Verify all referenced facts are established.
   160→                    for &fid in from {
   161→                        if self.check_circular && !established.contains(&fid) {
   162→                            return Ok(CheckResult::Invalid {
   163→                                step: step_idx,
   164→                                reason: format!(
   165→                                    "fact id {} is referenced but not yet established (circular reasoning)",
   166→                                    fid
   167→                                ),
   168→                            });
   169→                        }
   170→                    }
   171→
   172→                    // 2. Verify the conclusion id is not already used.
   173→                    if established.contains(&conclusion.id) {
   174→                        return Ok(CheckResult::Invalid {
   175→                            step: step_idx,
   176→                            reason: format!(
   177→                                "conclusion fact id {} is already established (duplicate)",
   178→                                conclusion.id
   179→                            ),
   180→                        });
   181→                    }
   182→
   183→                    // 3. Collect premise facts and apply the rule.
   184→                    let premise_facts: Vec<Fact> = from
   185→                        .iter()
   186→                        .map(|&fid| {
   187→                            proof
   188→                                .find_fact(fid)
   189→                                .cloned()
   190→                                .ok_or(CheckerError::FactNotFound { id: fid })
   191→                        })
   192→                        .collect::<Result<Vec<_>, _>>()?;
   193→
   194→                    let expected_conclusion = match rule.apply(&premise_facts) {
   195→                        Ok(f) => f,
   196→                        Err(e) => {
   197→                            return Ok(CheckResult::Invalid {
   198→                                step: step_idx,
   199→                                reason: format!("rule application failed: {}", e),
   200→                            });
   201→                        }
   202→                    };
   203→
   204→                    // 4. Verify the conclusion matches what the rule produces.
   205→                    // We compare statements for backward compatibility, and
   206→                    // also verify structured judgments match when both sides
   207→                    // have them.
   208→                    if expected_conclusion.statement != conclusion.statement {
   209→                        log::debug!(
   210→                            "Step {}: rule {} produced '{}' but proof claims '{}'",
   211→                            step_idx,
   212→                            rule.name(),
   213→                            expected_conclusion.statement,
   214→                            conclusion.statement,
   215→                        );
   216→                        return Ok(CheckResult::Invalid {
   217→                            step: step_idx,
   218→                            reason: format!(
   219→                                "conclusion mismatch: rule {} produces '{}' but step claims '{}'",
   220→                                rule.name(),
   221→                                expected_conclusion.statement,
   222→                                conclusion.statement
   223→                            ),
   224→                        });
   225→                    }
   226→
   227→                    // 4b. When both the expected and actual conclusion carry
   228→                    //     structured judgments, verify they are structurally
   229→                    //     equal. This prevents a proof from claiming one
   230→                    //     judgment while the rule produces another with the
   231→                    //     same string representation.
   232→                    if let (Some(expected_j), Some(actual_j)) =
   233→                        (&expected_conclusion.judgment, &conclusion.judgment)
   234→                    {
   235→                        if expected_j != actual_j {
   236→                            log::debug!(
   237→                                "Step {}: judgment mismatch — rule produced {:?} but step claims {:?}",
   238→                                step_idx,
   239→                                expected_j,
   240→                                actual_j,
   241→                            );
   242→                            return Ok(CheckResult::Invalid {
   243→                                step: step_idx,
   244→                                reason: format!(
   245→                                    "judgment mismatch: rule {} produces judgment {:?} but step claims {:?}",
   246→                                    rule.name(),
   247→                                    expected_j,
   248→                                    actual_j
   249→                                ),
   250→                            });
   251→                        }
   252→                    }
   253→
   254→                    established.insert(conclusion.id);
   255→                }
   256→
   257→                ProofStep::CaseSplit { cases } => {
   258→                    // Every case must be valid.
   259→                    for (case_idx, case) in cases.iter().enumerate() {
   260→                        match self.check(case)? {
   261→                            CheckResult::Valid => {}
   262→                            CheckResult::Invalid { step: s, reason } => {
   263→                                return Ok(CheckResult::Invalid {
   264→                                    step: step_idx,
   265→                                    reason: format!(
   266→                                        "case {} is invalid at sub-step {}: {}",
   267→                                        case_idx, s, reason
   268→                                    ),
   269→                                });
   270→                            }
   271→                            CheckResult::Incomplete => {
   272→                                return Ok(CheckResult::Invalid {
   273→                                    step: step_idx,
   274→                                    reason: format!("case {} is incomplete", case_idx),
   275→                                });
   276→                            }
   277→                        }
   278→                    }
   279→                }
   280→
   281→                ProofStep::Induction {
   282→                    base,
   283→                    step: ind_step,
   284→                } => {
   285→                    // Both the base case and the inductive step must be valid.
   286→                    match self.check(base)? {
   287→                        CheckResult::Valid => {}
   288→                        CheckResult::Invalid { step: s, reason } => {
   289→                            return Ok(CheckResult::Invalid {
   290→                                step: step_idx,
   291→                                reason: format!(
   292→                                    "induction base case invalid at sub-step {}: {}",
   293→                                    s, reason
   294→                                ),
   295→                            });
   296→                        }
   297→                        CheckResult::Incomplete => {
   298→                            return Ok(CheckResult::Invalid {
   299→                                step: step_idx,
   300→                                reason: "induction base case is incomplete".into(),
   301→                            });
   302→                        }
   303→                    }
   304→
   305→                    match self.check(ind_step)? {
   306→                        CheckResult::Valid => {}
   307→                        CheckResult::Invalid { step: s, reason } => {
   308→                            return Ok(CheckResult::Invalid {
   309→                                step: step_idx,
   310→                                reason: format!(
   311→                                    "induction step invalid at sub-step {}: {}",
   312→                                    s, reason
   313→                                ),
   314→                            });
   315→                        }
   316→                        CheckResult::Incomplete => {
   317→                            return Ok(CheckResult::Invalid {
   318→                                step: step_idx,
   319→                                reason: "induction step is incomplete".into(),
   320→                            });
   321→                        }
   322→                    }
   323→                }
   324→
   325→                ProofStep::Contradiction {
   326→                    assumption,
   327→                    negation,
   328→                } => {
   329→                    // Both facts must be established.
   330→                    if self.check_circular && !established.contains(assumption) {
   331→                        return Ok(CheckResult::Invalid {
   332→                            step: step_idx,
   333→                            reason: format!("assumption fact id {} is not established", assumption),
   334→                        });
   335→                    }
   336→                    if self.check_circular && !established.contains(negation) {
   337→                        return Ok(CheckResult::Invalid {
   338→                            step: step_idx,
   339→                            reason: format!("negation fact id {} is not established", negation),
   340→                        });
   341→                    }
   342→                    // A contradiction step is valid if both facts exist.
   343→                    // The proof author is responsible for ensuring they are
   344→                    // logically contradictory.
   345→                }
   346→
   347→                ProofStep::ByDefinition { definition: _ } => {
   348→                    // By-definition steps are always valid by fiat; they are
   349→                    // essentially definitional expansions that don't introduce
   350→                    // new facts with ids.
   351→                }
   352→            }
   353→        }
   354→
   355→        // After processing all steps, check the conclusion.
   356→        match proof.conclusion {
   357→            Conclusion::Proven => {
   358→                // A proof that claims to be proven must have at least one step.
   359→                if proof.steps.is_empty() {
   360→                    return Ok(CheckResult::Invalid {
   361→                        step: 0,
   362→                        reason: "proof claims Proven but has no steps".into(),
   363→                    });
   364→                }
   365→                Ok(CheckResult::Valid)
   366→            }
   367→            Conclusion::Refuted => {
   368→                // A refutation must contain a Contradiction step somewhere.
   369→                let has_contradiction = proof
   370→                    .steps
   371→                    .iter()
   372→                    .any(|s| matches!(s, ProofStep::Contradiction { .. }));
   373→                if !has_contradiction {
   374→                    return Ok(CheckResult::Invalid {
   375→                        step: 0,
   376→                        reason: "proof claims Refuted but contains no Contradiction step".into(),
   377→                    });
   378→                }
   379→                Ok(CheckResult::Valid)
   380→            }
   381→            Conclusion::Inconclusive => Ok(CheckResult::Incomplete),
   382→        }
   383→    }
   384→}

   519→    // -- check_proof_cached tests -----------------------------------------------
   520→
   521→    /// Checks a proof goal against the checker, using a cache to avoid
   522→    /// redundant work. If the goal's fingerprint is already in the cache,
   523→    /// the cached result is returned directly. Otherwise the proof is
   524→    /// checked, the result is stored in the cache, and the result is
   525→    /// returned.
   526→    pub fn check_proof_cached(goal: &Goal, cache: &mut ProofCache) -> ProofResult {
   527→        // Compute a simple fingerprint from the goal.
   528→        let fingerprint = goal_fingerprint(goal);
   529→
   530→        if let Some(cached) = cache.get(&fingerprint) {
   531→            return cached.clone();
   532→        }
   533→
   534→        // Build a trivial proof from the goal and check it.
   535→        let mut proof = Proof::new(goal.clone());
   536→        proof.add_step(ProofStep::Assume {
   537→            fact: Fact::assumption(1, goal.invariant.to_string()),
   538→        });
   539→        proof.conclude(Conclusion::Proven);
   540→
   541→        let checker = ProofChecker::new();
   542→        let result = match checker.check(&proof) {
   543→            Ok(CheckResult::Valid) => ProofResult::Discharged,
   544→            Ok(CheckResult::Invalid { reason, .. }) => ProofResult::Failed(reason),
   545→            Ok(CheckResult::Incomplete) => ProofResult::SubGoals(vec![goal.clone()]),
   546→            Err(e) => ProofResult::Failed(e.to_string()),
   547→        };
   548→
   549→        cache.insert(fingerprint, result.clone());
   550→        result
   551→    }
   552→
   553→    /// Computes a simple fingerprint for a goal using FNV-1a.
   554→    fn goal_fingerprint(goal: &Goal) -> u64 {
   555→        let mut hash: u64 = 0xcbf29ce484222325;
   556→        for byte in goal.invariant.to_string().bytes() {
   557→            hash ^= byte as u64;
   558→            hash = hash.wrapping_mul(0x100000001b3);
   559→        }
   560→        for byte in goal.context.scope.bytes() {
   561→            hash ^= byte as u64;
   562→            hash = hash.wrapping_mul(0x100000001b3);
   563→        }
   564→        hash
   565→    }
