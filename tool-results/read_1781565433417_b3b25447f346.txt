    53→        BackendKind::RiscV64 => "riscv64",
    54→        BackendKind::Wasm32 => "wasm32",
    55→        BackendKind::LoongArch64 => "loongarch64",

    59→        BackendKind::PowerPC64 => "ppc64",

   137→/// Run the full codegen pipeline (IR build → regalloc → encode) for a given backend.
   138→fn compile_scg_for_backend(backend: &dyn Backend, scg: &Scg, label: &str) -> Vec<u8> {
   139→    // Lower SCG → IR
   140→    let mut builder = IRBuilder::new();
   141→    let ir_program = builder
   142→        .build(scg)
   143→        .unwrap_or_else(|e| panic!("{}: IR build failed for {}: {}", backend.name(), label, e));
   144→
   145→    // Register allocation + encode
   146→    let mut allocated_functions = Vec::new();
   147→    for func in &ir_program.functions {
   148→        let allocated = backend
   149→            .allocate_registers(func)
   150→            .unwrap_or_else(|e| {
   151→                panic!(
   152→                    "{}: allocate_registers failed for {} / {}: {}",
   153→                    backend.name(),
   154→                    label,
   155→                    func.name,
   156→                    e
   157→                )
   158→            });
   159→        allocated_functions.push(allocated);
   160→    }
   161→
   162→    let total_code_size: usize = allocated_functions.iter().map(|f| f.code_size).sum();
   163→    let program = AllocatedProgram {
   164→        functions: allocated_functions,
   165→        total_code_size,
   166→        total_data_size: 0,
   167→    };
   168→
   169→    backend
   170→        .encode_program(&program)
   171→        .unwrap_or_else(|e| {
   172→            panic!(
   173→                "{}: encode_program failed for {}: {}",
   174→                backend.name(),
   175→                label,
   176→                e
   177→            )
   178→        })
   179→}

   181→/// Validate an ELF header for the given backend.
   182→fn validate_elf_header(bytes: &[u8], kind: BackendKind) {
   183→    let name = backend_name(kind);
   184→    let min_header = match expected_output_format(kind) {
   185→        OutputFormat::Elf32 => 52,
   186→        OutputFormat::Elf64 => 64,
   187→        _ => panic!("validate_elf_header called for non-ELF backend {}", name),
   188→    };
   189→    assert!(
   190→        bytes.len() >= min_header,
   191→        "{}: ELF binary too short ({} bytes, need at least {})",
   192→        name,
   193→        bytes.len(),
   194→        min_header
   195→    );
   196→    assert_eq!(
   197→        &bytes[0..4],
   198→        &[0x7f, b'E', b'L', b'F'],
   199→        "{}: ELF magic bytes incorrect",
   200→        name
   201→    );
   202→    let expected_class = match expected_output_format(kind) {
   203→        OutputFormat::Elf32 => 1u8,
   204→        OutputFormat::Elf64 => 2u8,
   205→        _ => unreachable!(),
   206→    };
   207→    assert_eq!(
   208→        bytes[4], expected_class,
   209→        "{}: ELF class should be {}",
   210→        name, expected_class
   211→    );
   212→    assert_eq!(bytes[6], 1, "{}: ELF version should be EV_CURRENT (1)", name);
   213→    let e_machine = if bytes[5] == 2 {
   214→        u16::from_be_bytes([bytes[18], bytes[19]])
   215→    } else {
   216→        u16::from_le_bytes([bytes[18], bytes[19]])
   217→    };
   218→    assert_eq!(
   219→        e_machine,
   220→        elf_machine(kind),
   221→        "{}: ELF machine type should be {} (got {})",
   222→        name,
   223→        elf_machine(kind),
   224→        e_machine
   225→    );
   226→}
   227→
   228→/// Validate a Wasm module header.
   229→fn validate_wasm_module(bytes: &[u8]) {
   230→    assert!(
   231→        bytes.len() >= 8,
   232→        "wasm32: binary too short ({} bytes, need at least 8)",
   233→        bytes.len()
   234→    );
   235→    assert_eq!(
   236→        &bytes[0..4],
   237→        &[0x00, 0x61, 0x73, 0x6D],
   238→        "wasm32: magic bytes should be \\0asm"
   239→    );
   240→    assert_eq!(
   241→        &bytes[4..8],
   242→        &[0x01, 0x00, 0x00, 0x00],
   243→        "wasm32: version should be 1"
   244→    );
   245→}
   246→
   247→/// Validate a binary for any backend (format-specific checks + minimum size).
   248→fn validate_binary(bytes: &[u8], kind: BackendKind, min_size: usize) {
   249→    let name = backend_name(kind);
   250→    assert!(
   251→        bytes.len() >= min_size,
   252→        "{}: binary too small ({} bytes, expected at least {})",
   253→        name,
   254→        bytes.len(),
   255→        min_size
   256→    );
   257→    match expected_output_format(kind) {
   258→        OutputFormat::Elf32 | OutputFormat::Elf64 => validate_elf_header(bytes, kind),
   259→        OutputFormat::WasmBinary => validate_wasm_module(bytes),
   260→        OutputFormat::RawBinary => { /* no structural validation */ }
   261→    }
   262→}

   422→/// Test: sha256d compiled for all 8 backends — each must produce valid output.
   423→#[test]
   424→fn test_full_pipeline_sha256d_all_backends() {
   425→    let sha256d_source = include_str!("../../../../examples/sha256d.vuma");
   426→
   427→    // Parse + AST → SCG (shared front-end)
   428→    let mut parser = Parser::new(sha256d_source);
   429→    let parse_output = parser.parse_program();
   430→    if parse_output.has_errors() {
   431→        // If sha256d.vuma can't parse in this environment, skip gracefully
   432→        eprintln!(
   433→            "Skipping sha256d all-backends test — parse errors: {:?}",
   434→            parse_output.errors
   435→        );
   436→        return;
   437→    }
   438→    let ast = parse_output.unwrap();
   439→    let mut converter = AstToScg::new();
   440→    let scg = match converter.convert(&ast) {
   441→        Ok(s) => s,
   442→        Err(e) => {
   443→            eprintln!("Skipping sha256d all-backends test — SCG error: {}", e);
   444→            return;
   445→        }
   446→    };
   447→
   448→    let codegen_scg = vuma::pipeline::bridge_scg_to_codegen(&scg);
   449→    let mut builder = IRBuilder::new();
   450→    let ir_program = match builder.build(&codegen_scg) {
   451→        Ok(ir) => ir,
   452→        Err(e) => {
   453→            eprintln!("Skipping sha256d all-backends test — IR error: {}", e);
   454→            return;
   455→        }
   456→    };
   457→
   458→    for &kind in ALL_BACKENDS {
   459→        let name = backend_name(kind);
   460→        let backend = match create_backend(kind) {
   461→            Ok(b) => b,
   462→            Err(e) => {
   463→                eprintln!("{}: backend creation failed (skipping): {}", name, e);
   464→                continue;
   465→            }
   466→        };
   467→
   468→        let mut allocated_functions = Vec::new();
   469→        let mut skip = false;
   470→        for func in &ir_program.functions {
   471→            match backend.allocate_registers(func) {
   472→                Ok(allocated) => allocated_functions.push(allocated),
   473→                Err(e) => {
   474→                    eprintln!(
   475→                        "{}: regalloc failed for {} (skipping): {}",
   476→                        name, func.name, e
   477→                    );
   478→                    skip = true;
   479→                    break;
   480→                }
   481→            }
   482→        }
   483→        if skip {
   484→            continue;
   485→        }
   486→
   487→        let total_code_size: usize = allocated_functions.iter().map(|f| f.code_size).sum();
   488→        let program = AllocatedProgram {
   489→            functions: allocated_functions,
   490→            total_code_size,
   491→            total_data_size: 0,
   492→        };
   493→
   494→        match backend.encode_program(&program) {
   495→            Ok(binary) => {
   496→                assert!(
   497→                    !binary.is_empty(),
   498→                    "{}: sha256d binary should be non-empty",
   499→                    name
   500→                );
   501→                // Validate format
   502→                match expected_output_format(kind) {
   503→                    OutputFormat::Elf32 | OutputFormat::Elf64 => {
   504→                        assert_eq!(
   505→                            &binary[0..4],
   506→                            &[0x7f, b'E', b'L', b'F'],
   507→                            "{}: should produce valid ELF",
   508→                            name
   509→                        );
   510→                    }
   511→                    OutputFormat::WasmBinary => {
   512→                        assert_eq!(
   513→                            &binary[0..4],
   514→                            &[0x00, 0x61, 0x73, 0x6D],
   515→                            "wasm32: should produce valid Wasm"
   516→                        );
   517→                    }
   518→                    OutputFormat::RawBinary => {}
   519→                }
   520→            }
   521→            Err(e) => {
   522→                eprintln!(
   523→                    "{}: encode_program failed for sha256d (skipping): {}",
   524→                    name, e
   525→                );
   526→            }
   527→        }
   528→    }
   529→}
