     1→//! # PowerPC64 Backend
     2→//!
     3→//! Implements the `Backend` trait for the PowerPC 64-bit target (ELFv2 ABI,
     4→//! little-endian default). This module provides:
     5→//!
     6→//! - `Gpr` — General-purpose register enum (R0–R31)
     7→//! - `Fpr` — Floating-point register enum (F0–F31)
     8→//! - `CrField` — Condition register field enum (CR0–CR7)
     9→//! - `Instruction` — PPC64 instruction enum with correct 32-bit encoding
    10→//! - `PPC64Backend` — `Backend` implementation that lowers IR to PPC64
    11→//!   machine code and emits ELF64 binaries
    12→//!
    13→//! ## PowerPC64 Register Convention (ELFv2 ABI)
    14→//!
    15→//! | Register(s) | Role                                  |
    16→//! |-------------|---------------------------------------|
    17→//! | R0          | Volatile (not hardwired zero!)        |
    18→//! | R1          | Stack pointer (SP)                    |
    19→//! | R2          | TOC pointer                           |
    20→//! | R3–R10      | Argument / return registers           |
    21→//! | R11–R12     | Volatile scratch                      |
    22→//! | R13         | Thread pointer                        |
    23→//! | R14–R31     | Callee-saved                          |
    24→//! | F0–F13      | FP argument / return, volatile        |
    25→//! | F14–F31     | FP callee-saved                       |
    26→//! | CR0–CR7     | Condition register fields (4 bits ea) |
    27→//! | LR          | Link register (SPR)                   |
    28→//! | CTR         | Count register (SPR)                  |
    29→//!
    30→//! ## Instruction Encoding
    31→//!
    32→//! All instructions are 32 bits, fixed-width. PPC64 is bi-endian; the default
    33→//! for ppc64le is little-endian byte order for both data and instructions.
    34→//! The primary opcode occupies bits \[0:5\] (MSB-first bit numbering).
    35→//!
    36→//! ## References
    37→//!
    38→//! - Power ISA Version 3.1
    39→//! - <https://openpowerfoundation.org/specifications/isa/>

   419→// Instruction Encoding Helpers
   420→// ===========================================================================
   421→
   422→/// Encode a PPC64 32-bit instruction word and return as big-endian bytes.
   423→fn encode_word(word: u32) -> [u8; 4] {
   424→    word.to_be_bytes()
   425→}
   426→
   427→/// Build a D-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | d[16:31]
   428→fn encode_d_form(opcode: u32, rt: u32, ra: u32, d: i32) -> [u8; 4] {
   429→    let word =
   430→        ((opcode & 0x3F) << 26) | ((rt & 0x1F) << 21) | ((ra & 0x1F) << 16) | ((d as u32) & 0xFFFF);
   431→    encode_word(word)
   432→}
   433→
   434→/// Build a DS-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | ds[16:29] | xo[30:31]
   435→/// The `ds` parameter is the byte offset; the DS field stores ds/4.
   436→fn encode_ds_form(opcode: u32, rt: u32, ra: u32, ds: i32, xo: u32) -> [u8; 4] {
   437→    let word = ((opcode & 0x3F) << 26)
   438→        | ((rt & 0x1F) << 21)
   439→        | ((ra & 0x1F) << 16)
   440→        | (((ds >> 2) as u32) & 0x3FFF) << 2
   441→        | (xo & 0x3);
   442→    encode_word(word)
   443→}
   444→
   445→/// Build an X-form instruction: opcode[0:5] | rS[6:10] | rA[11:15] | rB[16:20] | xo[21:30] | Rc[31]
   446→fn encode_x_form(opcode: u32, rs: u32, ra: u32, rb: u32, xo: u32, rc: u32) -> [u8; 4] {
   447→    let word = ((opcode & 0x3F) << 26)
   448→        | ((rs & 0x1F) << 21)
   449→        | ((ra & 0x1F) << 16)
   450→        | ((rb & 0x1F) << 11)
   451→        | ((xo & 0x3FF) << 1)
   452→        | (rc & 1);
   453→    encode_word(word)
   454→}
   455→
   456→/// Build an XO-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | rB[16:20] | OE[21] | xo[22:30] | Rc[31]
   457→fn encode_xo_form(opcode: u32, rt: u32, ra: u32, rb: u32, oe: u32, xo: u32, rc: u32) -> [u8; 4] {
   458→    let word = ((opcode & 0x3F) << 26)
   459→        | ((rt & 0x1F) << 21)
   460→        | ((ra & 0x1F) << 16)
   461→        | ((rb & 0x1F) << 11)
   462→        | ((oe & 1) << 10)
   463→        | ((xo & 0x1FF) << 1)
   464→        | (rc & 1);
   465→    encode_word(word)
   466→}
   467→
   468→/// Build an M-form instruction: opcode[0:5] | rS[6:10] | rA[11:15] | SH[16:20] | MB[21:25] | ME[26:30] | Rc[31]
   469→fn encode_m_form(opcode: u32, rs: u32, ra: u32, sh: u32, mb: u32, me: u32, rc: u32) -> [u8; 4] {
   470→    let word = ((opcode & 0x3F) << 26)
   471→        | ((rs & 0x1F) << 21)
   472→        | ((ra & 0x1F) << 16)
   473→        | ((sh & 0x1F) << 11)
   474→        | ((mb & 0x1F) << 6)
   475→        | ((me & 0x1F) << 1)
   476→        | (rc & 1);
   477→    encode_word(word)
   478→}
   479→
   480→/// Build an I-form instruction: opcode[0:5] | LI[6:29] | AA[30] | LK[31]
   481→fn encode_i_form(opcode: u32, li: i32, aa: u32, lk: u32) -> [u8; 4] {
   482→    let word =
   483→        ((opcode & 0x3F) << 26) | (((li as u32) & 0x03FF_FFFF) << 2) | ((aa & 1) << 1) | (lk & 1);
   484→    encode_word(word)
   485→}
   486→
   487→/// Build a B-form instruction: opcode[0:5] | BO[6:10] | BI[11:15] | BD[16:29] | AA[30] | LK[31]
   488→fn encode_b_form(opcode: u32, bo: u32, bi: u32, bd: i32, aa: u32, lk: u32) -> [u8; 4] {
   489→    let word = ((opcode & 0x3F) << 26)
   490→        | ((bo & 0x1F) << 21)
   491→        | ((bi & 0x1F) << 16)
   492→        | (((bd as u32) & 0x3FFF) << 2)
   493→        | ((aa & 1) << 1)
   494→        | (lk & 1);
   495→    encode_word(word)
   496→}
   497→
   498→/// Build an XL-form instruction: opcode[0:5] | BO[6:10] | BI[11:15] | 0[16:18] | BH[19:21] | xo[21:30] | LK[31]
   499→fn encode_xl_form(opcode: u32, bo: u32, bi: u32, bh: u32, xo: u32, lk: u32) -> [u8; 4] {
   500→    let word = ((opcode & 0x3F) << 26)
   501→        | ((bo & 0x1F) << 21)
   502→        | ((bi & 0x1F) << 16)
   503→        | ((bh & 0x7) << 11)
   504→        | ((xo & 0x3FF) << 1)
   505→        | (lk & 1);
   506→    encode_word(word)
   507→}

   513→/// PowerPC64 instruction representations for code generation.
   514→///
   515→/// Covers key arithmetic, logical, shift/rotate, load/store, compare, branch,
   516→/// move, and system instructions. Each variant captures the operands needed for
   517→/// encoding and disassembly. The `encode()` method produces a 4-byte
   518→/// little-endian machine code word.
   519→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   520→pub enum Instruction {
   521→    // ── Arithmetic ──────────────────────────────────────────────────
   522→    /// Add: `add rT, rA, rB` (XO-form, primary=31, xo=266)
   523→    Add { rt: Gpr, ra: Gpr, rb: Gpr },
   524→    /// Add Immediate: `addi rT, rA, simm16` (D-form, primary=14)
   525→    Addi { rt: Gpr, ra: Gpr, simm: i32 },
   526→    /// Add Immediate Shifted: `addis rT, rA, simm16` (D-form, primary=15)
   527→    Addis { rt: Gpr, ra: Gpr, simm: i32 },
   528→    /// Subtract From: `subf rT, rA, rB` (XO-form, primary=31, xo=40)
   529→    Subf { rt: Gpr, ra: Gpr, rb: Gpr },
   530→    /// Multiply Low Word: `mullw rT, rA, rB` (XO-form, primary=31, xo=235)
   531→    Mullw { rt: Gpr, ra: Gpr, rb: Gpr },
   532→    /// Multiply Low Doubleword: `mulld rT, rA, rB` (XO-form, primary=31, xo=233)
   533→    Mulld { rt: Gpr, ra: Gpr, rb: Gpr },
   534→    /// Multiply High Word: `mulhw rT, rA, rB` (X-form, primary=31, xo=75)
   535→    Mulhw { rt: Gpr, ra: Gpr, rb: Gpr },

   708→impl Instruction {
   709→    /// Encode this instruction into a 4-byte little-endian machine code word.
   710→    ///
   711→    /// Encoding follows the Power ISA Version 3.1.
   712→    pub fn encode(&self) -> [u8; 4] {
   713→        match self {
   714→            // ── Arithmetic ──────────────────────────────────────
   715→            Instruction::Add { rt, ra, rb } => {
   716→                // ADD rT, rA, rB: primary=31, OE=0, xo=266, Rc=0
   717→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 266, 0)
   718→            }
   719→            Instruction::Addi { rt, ra, simm } => {
   720→                // ADDI rT, rA, simm16: primary=14
   721→                encode_d_form(14, rt.encoding(), ra.encoding(), *simm)
   722→            }
   723→            Instruction::Addis { rt, ra, simm } => {
   724→                // ADDIS rT, rA, simm16: primary=15
   725→                encode_d_form(15, rt.encoding(), ra.encoding(), *simm)
   726→            }
   727→            Instruction::Subf { rt, ra, rb } => {
   728→                // SUBF rT, rA, rB: primary=31, OE=0, xo=40, Rc=0
   729→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 40, 0)
   730→            }
   731→            Instruction::Mullw { rt, ra, rb } => {
   732→                // MULLW rT, rA, rB: primary=31, OE=0, xo=235, Rc=0
   733→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 235, 0)
   734→            }
   735→            Instruction::Mulld { rt, ra, rb } => {
   736→                // MULLD rT, rA, rB: primary=31, OE=0, xo=233, Rc=0
   737→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 233, 0)
   738→            }
   739→            Instruction::Mulhw { rt, ra, rb } => {
   740→                // MULHW rT, rA, rB: primary=31, xo=75, Rc=0
   741→                encode_x_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 75, 0)
   742→            }
   743→            Instruction::Mulhd { rt, ra, rb } => {
   744→                // MULHD rT, rA, rB: primary=31, xo=73, Rc=0
   745→                encode_x_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 73, 0)
   746→            }
   747→            Instruction::Divw { rt, ra, rb } => {
   748→                // DIVW rT, rA, rB: primary=31, OE=0, xo=491, Rc=0
   749→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 491, 0)
   750→            }
   751→            Instruction::Divd { rt, ra, rb } => {
   752→                // DIVD rT, rA, rB: primary=31, OE=0, xo=459, Rc=0
   753→                encode_xo_form(31, rt.encoding(), ra.encoding(), rb.encoding(), 0, 459, 0)
   754→            }
   755→            Instruction::Neg { rt, ra } => {
   756→                // NEG rT, rA: primary=31, OE=0, xo=104, Rc=0, rB=0
   757→                encode_xo_form(31, rt.encoding(), ra.encoding(), 0, 0, 104, 0)
   758→            }
   759→            Instruction::Cntlzd { ra, rs } => {
   760→                // CNTLZD rA, rS: primary=31, xo=58, Rc=0, rB=0
   761→                encode_x_form(31, rs.encoding(), ra.encoding(), 0, 58, 0)
   762→            }
   763→            Instruction::Popcntd { ra, rs } => {
   764→                // POPCNTD rA, rS: primary=31, xo=506, Rc=0, rB=0
   765→                encode_x_form(31, rs.encoding(), ra.encoding(), 0, 506, 0)
   766→            }
   767→            Instruction::Extsw { ra, rs } => {
   768→                // EXTSW rA, rS: primary=31, xo=986, Rc=0, rB=0
   769→                encode_x_form(31, rs.encoding(), ra.encoding(), 0, 986, 0)
   770→            }
   771→
   772→            // ── Logical ────────────────────────────────────────
   773→            Instruction::And { ra, rs, rb } => {
   774→                // AND rA, rS, rB: primary=31, xo=28, Rc=0
   775→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 28, 0)
   776→            }
   777→            Instruction::Andi { ra, rs, uimm } => {
   778→                // ANDI. rA, rS, uimm16: primary=28, Rc=1 (always)
   779→                let word = ((28u32 & 0x3F) << 26)
   780→                    | ((rs.encoding() & 0x1F) << 21)
   781→                    | ((ra.encoding() & 0x1F) << 16)
   782→                    | (uimm & 0xFFFF);
   783→                encode_word(word)
   784→            }
   785→            Instruction::Or { ra, rs, rb } => {
   786→                // OR rA, rS, rB: primary=31, xo=444, Rc=0
   787→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 444, 0)
   788→            }
   789→            Instruction::Ori { ra, rs, uimm } => {
   790→                // ORI rA, rS, uimm16: primary=24
   791→                let word = ((24u32 & 0x3F) << 26)
   792→                    | ((rs.encoding() & 0x1F) << 21)
   793→                    | ((ra.encoding() & 0x1F) << 16)
   794→                    | (uimm & 0xFFFF);
   795→                encode_word(word)
   796→            }
   797→            Instruction::Xor { ra, rs, rb } => {
   798→                // XOR rA, rS, rB: primary=31, xo=316, Rc=0
   799→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 316, 0)
   800→            }
   801→            Instruction::Xori { ra, rs, uimm } => {
   802→                // XORI rA, rS, uimm16: primary=26
   803→                let word = ((26u32 & 0x3F) << 26)
   804→                    | ((rs.encoding() & 0x1F) << 21)
   805→                    | ((ra.encoding() & 0x1F) << 16)
   806→                    | (uimm & 0xFFFF);
   807→                encode_word(word)
   808→            }
   809→            Instruction::Nor { ra, rs, rb } => {
   810→                // NOR rA, rS, rB: primary=31, xo=124, Rc=0
   811→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 124, 0)
   812→            }
   813→            Instruction::Andc { ra, rs, rb } => {
   814→                // ANDC rA, rS, rB: primary=31, xo=60, Rc=0
   815→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 60, 0)
   816→            }
   817→            Instruction::Orc { ra, rs, rb } => {
   818→                // ORC rA, rS, rB: primary=31, xo=412, Rc=0
   819→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 412, 0)
   820→            }
   821→            Instruction::Eqv { ra, rs, rb } => {
   822→                // EQV rA, rS, rB: primary=31, xo=284, Rc=0
   823→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 284, 0)
   824→            }
   825→
   826→            // ── Shift / Rotate ─────────────────────────────────
   827→            Instruction::Sld { ra, rs, rb } => {
   828→                // SLD rA, rS, rB: primary=31, xo=27, Rc=0
   829→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 27, 0)
   830→            }
   831→            Instruction::Srd { ra, rs, rb } => {
   832→                // SRD rA, rS, rB: primary=31, xo=539, Rc=0
   833→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 539, 0)
   834→            }
   835→            Instruction::Srad { ra, rs, rb } => {
   836→                // SRAD rA, rS, rB: primary=31, xo=794, Rc=0
   837→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 794, 0)
   838→            }
   839→            Instruction::Slw { ra, rs, rb } => {
   840→                // SLW rA, rS, rB: primary=31, xo=24, Rc=0
   841→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 24, 0)
   842→            }
   843→            Instruction::Srw { ra, rs, rb } => {
   844→                // SRW rA, rS, rB: primary=31, xo=536, Rc=0
   845→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 536, 0)
   846→            }
   847→            Instruction::Sraw { ra, rs, rb } => {
   848→                // SRAW rA, rS, rB: primary=31, xo=792, Rc=0
   849→                encode_x_form(31, rs.encoding(), ra.encoding(), rb.encoding(), 792, 0)
   850→            }
   851→            Instruction::Rldcl { ra, rs, rb, mb } => {
   852→                // RLDCL rA, rS, rB, MB: primary=31, rB, MB, xo=8, sh=0, Rc=0
   853→                // Encoding: [0:5]=31, [6:10]=rS, [11:15]=rA, [16:20]=rB,
   854→                //           [21:25]=MB, [26:30]=xo=8, [31]=Rc=0
   855→                // Note: for RLDCL the mb field occupies bits [21:25] (5 bits,
   856→                // representing the beginning mask position encoded as
   857→                // mb[5] in bit 1 and mb[0:4] in bits [21:25]).
   858→                // Simplified: we treat mb as 5-bit field in bits [21:25]
   859→                let word = (31u32 << 26)
   860→                    | (rs.encoding() << 21)
   861→                    | (ra.encoding() << 16)
   862→                    | (rb.encoding() << 11)
   863→                    | ((mb & 0x1F) << 6)
   864→                    | (8 << 1);
   865→                encode_word(word)
   866→            }
   867→            Instruction::Rldcr { ra, rs, rb, me } => {
   868→                // RLDCR rA, rS, rB, ME: primary=31, rB, ME, xo=9, Rc=0
   869→                let word = (31u32 << 26)
   870→                    | (rs.encoding() << 21)
   871→                    | (ra.encoding() << 16)
   872→                    | (rb.encoding() << 11)
   873→                    | ((me & 0x1F) << 6)
   874→                    | (9 << 1);
   875→                encode_word(word)
   876→            }
   877→            Instruction::Rlwinm { ra, rs, sh, mb, me } => {
   878→                // RLWINM rA, rS, SH, MB, ME: primary=21
   879→                encode_m_form(21, rs.encoding(), ra.encoding(), *sh, *mb, *me, 0)
   880→            }
   881→            Instruction::Rlwimi { ra, rs, sh, mb, me } => {
   882→                // RLWIMI rA, rS, SH, MB, ME: primary=20
   883→                encode_m_form(20, rs.encoding(), ra.encoding(), *sh, *mb, *me, 0)
   884→            }
   885→
   886→            // ── Load / Store ───────────────────────────────────
   887→            Instruction::Ld { rt, ra, ds } => {
   888→                // LD rT, ds(rA): primary=58, xo=0
   889→                encode_ds_form(58, rt.encoding(), ra.encoding(), *ds, 0)
   890→            }
   891→            Instruction::Lwa { rt, ra, ds } => {
   892→                // LWA rT, ds(rA): primary=58, xo=2
   893→                encode_ds_form(58, rt.encoding(), ra.encoding(), *ds, 2)
   894→            }
   895→            Instruction::Lwz { rt, ra, d } => {
   896→                // LWZ rT, d(rA): primary=32
   897→                encode_d_form(32, rt.encoding(), ra.encoding(), *d)
   898→            }
   899→            Instruction::Lwzu { rt, ra, d } => {
   900→                // LWZU rT, d(rA): primary=33
   901→                encode_d_form(33, rt.encoding(), ra.encoding(), *d)
   902→            }
   903→            Instruction::Std { rs, ra, ds } => {
   904→                // STD rS, ds(rA): primary=62, xo=0
   905→                encode_ds_form(62, rs.encoding(), ra.encoding(), *ds, 0)
   906→            }
   907→            Instruction::Stw { rs, ra, d } => {
   908→                // STW rS, d(rA): primary=36
   909→                encode_d_form(36, rs.encoding(), ra.encoding(), *d)
   910→            }
   911→            Instruction::Stwu { rs, ra, d } => {
   912→                // STWU rS, d(rA): primary=37
   913→                encode_d_form(37, rs.encoding(), ra.encoding(), *d)
   914→            }
   915→            Instruction::Stdu { rs, ra, ds } => {
   916→                // STDU rS, ds(rA): primary=62, xo=1
   917→                encode_ds_form(62, rs.encoding(), ra.encoding(), *ds, 1)
   918→            }
   919→            Instruction::Lbz { rt, ra, d } => {
   920→                // LBZ rT, d(rA): primary=34
   921→                encode_d_form(34, rt.encoding(), ra.encoding(), *d)
   922→            }
   923→            Instruction::Lhz { rt, ra, d } => {
   924→                // LHZ rT, d(rA): primary=40
   925→                encode_d_form(40, rt.encoding(), ra.encoding(), *d)
   926→            }
   927→            Instruction::Stb { rs, ra, d } => {
   928→                // STB rS, d(rA): primary=38
   929→                encode_d_form(38, rs.encoding(), ra.encoding(), *d)
   930→            }
   931→            Instruction::Sth { rs, ra, d } => {
   932→                // STH rS, d(rA): primary=44
   933→                encode_d_form(44, rs.encoding(), ra.encoding(), *d)
   934→            }
   935→            Instruction::Lfd { ft, ra, d } => {
   936→                // LFD fT, d(rA): primary=50
   937→                encode_d_form(50, ft.encoding(), ra.encoding(), *d)
   938→            }
   939→            Instruction::Stfd { fs, ra, d } => {
   940→                // STFD fS, d(rA): primary=54
   941→                encode_d_form(54, fs.encoding(), ra.encoding(), *d)
   942→            }
   943→            Instruction::Lfs { ft, ra, d } => {
   944→                // LFS fT, d(rA): primary=48
   945→                encode_d_form(48, ft.encoding(), ra.encoding(), *d)
   946→            }
   947→            Instruction::Stfs { fs, ra, d } => {
   948→                // STFS fS, d(rA): primary=52
   949→                encode_d_form(52, fs.encoding(), ra.encoding(), *d)
   950→            }
   951→
   952→            // ── Compare ────────────────────────────────────────
   953→            Instruction::Cmp { bf, l, ra, rb } => {
   954→                // CMP crf, l, rA, rB: primary=31, xo=0, Rc=0
   955→                // bits [6:8] = bf, [9]=l, [11:15]=rA, [16:20]=rB
   956→                let word = (31u32 << 26)
   957→                    | ((bf.encoding() & 0x7) << 23)
   958→                    | ((*l & 1) << 21)
   959→                    | (ra.encoding() << 16)
   960→                    | (rb.encoding() << 11);
   961→                encode_word(word)
   962→            }
   963→            Instruction::Cmpi { bf, l, ra, simm } => {
   964→                // CMPI crf, l, rA, simm16: primary=11
   965→                let word = (11u32 << 26)
   966→                    | ((bf.encoding() & 0x7) << 23)
   967→                    | ((*l & 1) << 21)
   968→                    | (ra.encoding() << 16)
   969→                    | ((*simm as u32) & 0xFFFF);
   970→                encode_word(word)
   971→            }
   972→            Instruction::Cmpl { bf, l, ra, rb } => {
   973→                // CMPL crf, l, rA, rB: primary=31, xo=32, Rc=0
   974→                let word = (31u32 << 26)
   975→                    | ((bf.encoding() & 0x7) << 23)
   976→                    | ((*l & 1) << 21)
   977→                    | (ra.encoding() << 16)
   978→                    | (rb.encoding() << 11)
   979→                    | (32 << 1);
   980→                encode_word(word)
   981→            }
   982→            Instruction::Cmpli { bf, l, ra, uimm } => {
   983→                // CMPLI crf, l, rA, uimm16: primary=10
   984→                let word = (10u32 << 26)
   985→                    | ((bf.encoding() & 0x7) << 23)
   986→                    | ((*l & 1) << 21)
   987→                    | (ra.encoding() << 16)
   988→                    | (uimm & 0xFFFF);
   989→                encode_word(word)
   990→            }
   991→
   992→            // ── Branch ─────────────────────────────────────────
   993→            Instruction::B { li } => encode_i_form(18, *li, 0, 0),
   994→            Instruction::Ba { li } => encode_i_form(18, *li, 1, 0),
   995→            Instruction::Bl { li } => encode_i_form(18, *li, 0, 1),
   996→            Instruction::Bla { li } => encode_i_form(18, *li, 1, 1),
   997→            Instruction::Bc { bo, bi, bd } => encode_b_form(16, *bo, *bi, *bd, 0, 0),
   998→            Instruction::Bca { bo, bi, bd } => encode_b_form(16, *bo, *bi, *bd, 1, 0),
   999→            Instruction::Bclr { bo, bi, bh } => encode_xl_form(19, *bo, *bi, *bh, 16, 0),
  1000→            Instruction::Bcctr { bo, bi, bh } => encode_xl_form(19, *bo, *bi, *bh, 528, 0),
  1001→            Instruction::Bctar { bo, bi, bh } => encode_xl_form(19, *bo, *bi, *bh, 560, 0),
  1002→
  1003→            // ── Move ───────────────────────────────────────────
  1004→            Instruction::Mr { ra, rs } => {
  1005→                // MR rA, rS = OR rA, rS, rS: primary=31, xo=444, Rc=0
  1006→                encode_x_form(31, rs.encoding(), ra.encoding(), rs.encoding(), 444, 0)
  1007→            }
  1008→            Instruction::Li { rt, simm } => {
  1009→                // LI rT, simm = ADDI rT, R0, simm: primary=14, rA=0
  1010→                encode_d_form(14, rt.encoding(), 0, *simm)
  1011→            }
  1012→            Instruction::Lis { rt, simm } => {
  1013→                // LIS rT, simm = ADDIS rT, R0, simm: primary=15, rA=0
  1014→                encode_d_form(15, rt.encoding(), 0, *simm)
  1015→            }
  1016→
  1017→            // ── System ─────────────────────────────────────────
  1018→            Instruction::Sc => {
  1019→                // SC: primary=17, bits [6:29]=0, bit 30=1 (SVC field)
  1020→                // Full encoding: 0x44000002
  1021→                encode_word(0x44000002)
  1022→            }
  1023→            Instruction::Nop => {
  1024→                // NOP = ORI r0, r0, 0: primary=24, rS=0, rA=0, uimm16=0
  1025→                encode_word(0x60000000)
  1026→            }
  1027→            Instruction::Trap => {
  1028→                // TRAP = TW 31, r0, r0: primary=31, rS=31, rA=0, rB=0, xo=4, Rc=0
  1029→                encode_x_form(31, 31, 0, 0, 4, 0)
  1030→            }
  1031→        }
  1032→    }

  1202→/// Build a proper ELF64 binary for PPC64LE with 2 LOAD segments.
  1203→///
  1204→/// Produces a static executable with:
  1205→/// - Segment 1: LOAD (PF_R | PF_X) — .text
  1206→/// - Segment 2: LOAD (PF_R | PF_W) — .data / BSS (writable)
  1207→///
  1208→/// Uses little-endian byte order for ppc64le.  The text segment is placed
  1209→/// at a page-aligned offset so that the kernel's ELF loader can mmap() it.
  1210→fn build_ppc64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1211→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (PPC64 typical page size)
  1212→
  1213→    let elf_header_size: u64 = 64;
  1214→    let phdr_size: u64 = 56;
  1215→    let num_phdrs: u64 = 2;
  1216→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1217→    // Page-align the text segment start in the file.
  1218→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1219→    let text_size = code.len() as u64;
  1220→
  1221→    // The data segment starts on the next page after the text.
  1222→    let text_file_end = text_offset + text_size;
  1223→    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1224→    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1225→    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1226→    let entry_point = base_addr + text_offset;
  1227→
  1228→    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1229→
  1230→    // --- e_ident ---
  1231→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1232→    elf.push(2); // ELFCLASS64
  1233→    elf.push(2); // ELFDATA2MSB (big-endian PPC64)
  1234→    elf.push(1); // EV_CURRENT
  1235→    elf.push(3); // ELFOSABI_LINUX
  1236→    elf.push(0); // padding
  1237→    elf.extend_from_slice(&[0u8; 7]); // padding
  1238→
  1239→    // --- ELF header fields ---
  1240→    elf.extend_from_slice(&2u16.to_be_bytes()); // e_type = ET_EXEC
  1241→    elf.extend_from_slice(&21u16.to_be_bytes()); // e_machine = EM_PPC64
  1242→    elf.extend_from_slice(&1u32.to_be_bytes()); // e_version
  1243→    elf.extend_from_slice(&entry_point.to_be_bytes()); // e_entry
  1244→    elf.extend_from_slice(&elf_header_size.to_be_bytes()); // e_phoff
  1245→    elf.extend_from_slice(&0u64.to_be_bytes()); // e_shoff (no section headers)
  1246→    // e_flags: EF_PPC64_ABI_V2 = 0x2 (required for PPC64LE ELFv2 ABI)
  1247→    elf.extend_from_slice(&2u32.to_be_bytes()); // e_flags
  1248→    elf.extend_from_slice(&64u16.to_be_bytes()); // e_ehsize
  1249→    elf.extend_from_slice(&56u16.to_be_bytes()); // e_phentsize
  1250→    elf.extend_from_slice(&2u16.to_be_bytes()); // e_phnum = 2
  1251→    elf.extend_from_slice(&64u16.to_be_bytes()); // e_shentsize
  1252→    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shnum
  1253→    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shstrndx
  1254→
  1255→    // --- Program Header 1: LOAD (PF_R | PF_X) — .text ---
  1256→    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1257→    elf.extend_from_slice(&5u32.to_be_bytes()); // p_flags = PF_R | PF_X
  1258→    elf.extend_from_slice(&text_offset.to_be_bytes()); // p_offset
  1259→    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_vaddr
  1260→    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_paddr
  1261→    elf.extend_from_slice(&text_size.to_be_bytes()); // p_filesz
  1262→    elf.extend_from_slice(&text_size.to_be_bytes()); // p_memsz
  1263→    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1264→
  1265→    // --- Program Header 2: LOAD (PF_R | PF_W) — .data / stack ---
  1266→    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1267→    elf.extend_from_slice(&6u32.to_be_bytes()); // p_flags = PF_R | PF_W
  1268→    elf.extend_from_slice(&data_offset.to_be_bytes()); // p_offset
  1269→    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_vaddr
  1270→    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_paddr
  1271→    elf.extend_from_slice(&0u64.to_be_bytes()); // p_filesz (no initialized data)
  1272→    elf.extend_from_slice(&data_size.to_be_bytes()); // p_memsz (writable pages)
  1273→    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1274→
  1275→    // --- .text section ---
  1276→    // Pad to page-aligned text_offset
  1277→    while (elf.len() as u64) < text_offset {
  1278→        elf.push(0);
  1279→    }
  1280→    elf.extend_from_slice(code);
  1281→
  1282→    // --- Pad to data segment offset (if needed) ---
  1283→    while (elf.len() as u64) < data_offset {
  1284→        elf.push(0);
  1285→    }
  1286→
  1287→    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1288→
  1289→    elf
  1290→}

  1316→/// Compute the stack frame size for an IR function on PPC64.
  1317→///
  1318→/// Sums `Alloc` instruction sizes, adds 32 bytes for the LR/CR save area
  1319→/// (per ELFv2 ABI), and rounds up to 16-byte alignment.
  1320→fn ppc64_compute_frame_size(func: &IRFunction) -> usize {
  1321→    let mut total: u32 = 32; // LR save (8) + CR save (8) + back chain (8) + TOC save (8)
  1322→    for block in &func.blocks {
  1323→        for instr in &block.instructions {
  1324→            if let IRInstr::Alloc { size, .. } = instr {
  1325→                let aligned = (*size).div_ceil(16) * 16;
  1326→                total += aligned;
  1327→            }
  1328→        }
  1329→    }
  1330→    // Round up to 16-byte alignment
  1331→    total = (total + 15) & !15;
  1332→    total as usize
  1333→}

  1335→/// Allocatable GPR registers for PPC64, in priority order.
  1336→///
  1337→/// Order: argument registers first, then volatile temporaries, then callee-saved.
  1338→/// R0 is reserved (not allocatable — in some ISA contexts rA=0 means literal zero).
  1339→/// R11 is reserved as a scratch register for instruction lowering.
  1340→/// R12 is the first allocatable temporary.
  1341→const ALLOCATABLE_GPRS: &[Gpr] = &[
  1342→    // Argument registers (also volatile)
  1343→    Gpr::R3,
  1344→    Gpr::R4,
  1345→    Gpr::R5,
  1346→    Gpr::R6,
  1347→    Gpr::R7,
  1348→    Gpr::R8,
  1349→    Gpr::R9,
  1350→    Gpr::R10,
  1351→    // Volatile temporary
  1352→    Gpr::R12,
  1353→    // Callee-saved (require save/restore)
  1354→    Gpr::R14,
  1355→    Gpr::R15,
  1356→    Gpr::R16,
  1357→    Gpr::R17,
  1358→    Gpr::R18,
  1359→    Gpr::R19,
  1360→    Gpr::R20,
  1361→    Gpr::R21,
  1362→    Gpr::R22,
  1363→    Gpr::R23,
  1364→    Gpr::R24,
  1365→    Gpr::R25,
  1366→    Gpr::R26,
  1367→    Gpr::R27,
  1368→    Gpr::R28,
  1369→    Gpr::R29,
  1370→    Gpr::R30,
  1371→    Gpr::R31,
  1372→];

  1374→/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1375→fn map_vreg_to_gpr(
  1376→    vreg_id: u32,
  1377→    arg_index: Option<usize>,
  1378→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1379→) -> Gpr {
  1380→    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1381→        return *gpr;
  1382→    }
  1383→
  1384→    // If this is an argument, map it to the corresponding argument register.
  1385→    if let Some(idx) = arg_index {
  1386→        if let Some(reg) = Gpr::arg_register(idx) {
  1387→            vreg_map.insert(vreg_id, reg);
  1388→            return reg;
  1389→        }
  1390→    }
  1391→
  1392→    // Otherwise, pick the first available allocatable register.
  1393→    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1394→    for &reg in ALLOCATABLE_GPRS {
  1395→        if !used.contains(&reg) {
  1396→            vreg_map.insert(vreg_id, reg);
  1397→            return reg;
  1398→        }
  1399→    }
  1400→
  1401→    // Fallback: use R11 as a scratch register.
  1402→    Gpr::R11
  1403→}
  1404→
  1405→/// Helper: extract the virtual register ID from an IRValue, if it is a register.
  1406→fn vreg_id(val: &IRValue) -> u32 {
  1407→    match val {
  1408→        IRValue::Register(id) => *id,
  1409→        _ => 0,
  1410→    }
  1411→}

  1413→/// Emit a single AllocatedInstruction from a PPC64 Instruction.
  1414→fn emit_alloc_instr(
  1415→    inst: Instruction,
  1416→    reads: Vec<PhysicalReg>,

  1418→) -> AllocatedInstruction {
  1419→    AllocatedInstruction {
  1420→        opcode: inst.mnemonic().to_string(),

  1422→        writes,
  1423→        encoded: inst.encode().to_vec(),
  1424→    }

  1427→/// Load a 64-bit immediate value into a GPR, emitting the minimal number of
  1428→/// PPC64 instructions.
  1429→///
  1430→/// Strategy:
  1431→/// - **`[-32768, 32767]`**: `li rd, val` (single `addi rd, r0, val`)
  1432→/// - **`[0, 65535]`**: `ori rd, r0, val`
  1433→/// - **`0xXXXX0000` (low 16 bits zero, fits 32-bit)**: `lis rd, upper16`
  1434→/// - **Other 32-bit**: `lis rd, upper16` + `ori rd, rd, lower16`
  1435→/// - **Full 64-bit**: `lis` + `ori` + `sldi 32` + `oris` + `ori`
  1436→fn load_immediate_ppc64(rd: Gpr, val: i64, out: &mut Vec<AllocatedInstruction>) {
  1437→    let rd_phys = PhysicalReg::new(RegClass::Gpr, rd.encoding());
  1438→    let uval = val as u64;
  1439→
  1440→    if (-32768..=32767).contains(&val) {
  1441→        // li rd, val (addi rd, r0, val)
  1442→        out.push(emit_alloc_instr(
  1443→            Instruction::Li {
  1444→                rt: rd,
  1445→                simm: val as i32,
  1446→            },
  1447→            vec![],
  1448→            vec![rd_phys],
  1449→        ));
  1450→    } else if uval <= 0xFFFF {
  1451→        // ori rd, r0, val (unsigned 16-bit)
  1452→        out.push(emit_alloc_instr(
  1453→            Instruction::Ori {
  1454→                ra: rd,
  1455→                rs: Gpr::R0,
  1456→                uimm: uval as u32 & 0xFFFF,
  1457→            },
  1458→            vec![],
  1459→            vec![rd_phys],
  1460→        ));
  1461→    } else if uval & 0xFFFF == 0 && (uval >> 32) == 0 {
  1462→        // lis rd, upper16 (0xXXXX0000)
  1463→        out.push(emit_alloc_instr(
  1464→            Instruction::Lis {
  1465→                rt: rd,
  1466→                simm: ((uval >> 16) & 0xFFFF) as i16 as i32,
  1467→            },
  1468→            vec![],
  1469→            vec![rd_phys],
  1470→        ));
  1471→    } else if uval >> 32 == 0 {
  1472→        // 32-bit value: lis + ori
  1473→        let hi16 = ((uval >> 16) & 0xFFFF) as u32;
  1474→        let lo16 = (uval & 0xFFFF) as u32;
  1475→        out.push(emit_alloc_instr(
  1476→            Instruction::Lis {
  1477→                rt: rd,
  1478→                simm: hi16 as i16 as i32,
  1479→            },
  1480→            vec![],
  1481→            vec![rd_phys],
  1482→        ));
  1483→        if lo16 != 0 {
  1484→            out.push(emit_alloc_instr(
  1485→                Instruction::Ori {
  1486→                    ra: rd,
  1487→                    rs: rd,
  1488→                    uimm: lo16,
  1489→                },
  1490→                vec![rd_phys],
  1491→                vec![rd_phys],
  1492→            ));
  1493→        }
  1494→    } else {
  1495→        // Full 64-bit: lis + ori + sldi 32 + oris + ori
  1496→        let hi32 = (uval >> 32) as u32;
  1497→        let lo32 = (uval & 0xFFFF_FFFF) as u32;
  1498→
  1499→        // lis rd, upper16(hi32)
  1500→        out.push(emit_alloc_instr(
  1501→            Instruction::Lis {
  1502→                rt: rd,
  1503→                simm: ((hi32 >> 16) & 0xFFFF) as i16 as i32,
  1504→            },
  1505→            vec![],
  1506→            vec![rd_phys],
  1507→        ));
  1508→
  1509→        // ori rd, rd, lower16(hi32)
  1510→        if hi32 & 0xFFFF != 0 {
  1511→            out.push(emit_alloc_instr(
  1512→                Instruction::Ori {
  1513→                    ra: rd,
  1514→                    rs: rd,
  1515→                    uimm: hi32 & 0xFFFF,
  1516→                },
  1517→                vec![rd_phys],
  1518→                vec![rd_phys],
  1519→            ));
  1520→        }
  1521→
  1522→        // sldi rd, rd, 32 = rldicr rd, rd, 32, 31
  1523→        // MD-form: [0:5]=30, [6:10]=rS, [11:15]=rA, [16:20]=SH[0:4],
  1524→        // [21:25]=ME[0:4], [26]=SH[5], [27]=ME[5], [28:30]=xo(=2), [31]=Rc(=0)
  1525→        let sh: u32 = 32;
  1526→        let me: u32 = 63 - sh;
  1527→        let sldi_word: u32 = (30u32 << 26)
  1528→            | (rd.encoding() << 21)
  1529→            | (rd.encoding() << 16)
  1530→            | ((sh & 0x1F) << 11)
  1531→            | ((me & 0x1F) << 6)
  1532→            | (((sh >> 5) & 1) << 5)
  1533→            | (((me >> 5) & 1) << 4)
  1534→            | (2u32 << 1);
  1535→        out.push(AllocatedInstruction {
  1536→            opcode: "rldicr".to_string(),
  1537→            reads: vec![rd_phys],
  1538→            writes: vec![rd_phys],
  1539→            encoded: encode_word(sldi_word).to_vec(),
  1540→        });
  1541→
  1542→        // oris rd, rd, upper16(lo32) — primary opcode 25, D-form
  1543→        if (lo32 >> 16) & 0xFFFF != 0 {
  1544→            let oris_word: u32 = (25u32 << 26)
  1545→                | (rd.encoding() << 21)
  1546→                | (rd.encoding() << 16)
  1547→                | ((lo32 >> 16) & 0xFFFF);
  1548→            out.push(AllocatedInstruction {
  1549→                opcode: "oris".to_string(),
  1550→                reads: vec![rd_phys],
  1551→                writes: vec![rd_phys],
  1552→                encoded: encode_word(oris_word).to_vec(),
  1553→            });
  1554→        }
  1555→
  1556→        // ori rd, rd, lower16(lo32)
  1557→        if lo32 & 0xFFFF != 0 {
  1558→            out.push(emit_alloc_instr(
  1559→                Instruction::Ori {
  1560→                    ra: rd,
  1561→                    rs: rd,
  1562→                    uimm: lo32 & 0xFFFF,
  1563→                },
  1564→                vec![rd_phys],
  1565→                vec![rd_phys],
  1566→            ));
  1567→        }
  1568→    }
  1569→}
  1570→
  1571→/// Resolve an `IRValue` to a physical `Gpr`, emitting immediate-load
  1572→/// instructions as needed.
  1573→///
  1574→/// - `IRValue::Register(id)` → looks up `id` in `reg_map`
  1575→/// - `IRValue::Immediate(val)` → loads `val` into `scratch` via
  1576→///   `load_immediate_ppc64` and returns `scratch`
  1577→/// - `IRValue::Address(addr)` → loads `addr` into `scratch`
  1578→/// - `IRValue::Label(_)` → loads 0 into `scratch` (placeholder for relocation)
  1579→///
  1580→/// The caller must supply a scratch register that is not live at this point
  1581→/// (e.g., `R11` for the first operand, `R12` for the second).
  1582→fn resolve_gpr_ppc64(
  1583→    val: &IRValue,
  1584→    reg_map: &mut std::collections::HashMap<u32, Gpr>,
  1585→    scratch: Gpr,
  1586→    out: &mut Vec<AllocatedInstruction>,
  1587→) -> Gpr {
  1588→    match val {
  1589→        IRValue::Register(id) => map_vreg_to_gpr(*id, None, reg_map),
  1590→        IRValue::Immediate(imm) => {
  1591→            load_immediate_ppc64(scratch, *imm, out);
  1592→            scratch
  1593→        }
  1594→        IRValue::Address(addr) => {
  1595→            load_immediate_ppc64(scratch, *addr as i64, out);
  1596→            scratch
  1597→        }
  1598→        IRValue::Label(_) => {
  1599→            // Labels need linker relocation; emit li scratch, 0 as placeholder
  1600→            load_immediate_ppc64(scratch, 0, out);
  1601→            scratch
  1602→        }
  1603→    }
  1604→}

  1606→/// Lower a comparison kind to PPC64 instructions, producing 0 or 1 in `dst`.
  1607→fn lower_cmp_ppc64(kind: &CmpKind, dst: Gpr, lhs: Gpr, rhs: Gpr) -> Vec<AllocatedInstruction> {
  1608→    let mut result = Vec::new();
  1609→    let (cmp_inst, cond_bit) = match kind {
  1610→        CmpKind::Eq => (
  1611→            Instruction::Cmp {
  1612→                bf: CrField::CR0,
  1613→                l: 1,
  1614→                ra: lhs,
  1615→                rb: rhs,
  1616→            },
  1617→            2,
  1618→        ),
  1619→        CmpKind::Ne => (
  1620→            Instruction::Cmp {
  1621→                bf: CrField::CR0,
  1622→                l: 1,
  1623→                ra: lhs,
  1624→                rb: rhs,
  1625→            },
  1626→            2,
  1627→        ),
  1628→        CmpKind::SLt => (
  1629→            Instruction::Cmp {
  1630→                bf: CrField::CR0,
  1631→                l: 1,
  1632→                ra: lhs,
  1633→                rb: rhs,
  1634→            },
  1635→            0,
  1636→        ),
  1637→        CmpKind::SLe => (
  1638→            Instruction::Cmp {
  1639→                bf: CrField::CR0,
  1640→                l: 1,
  1641→                ra: lhs,
  1642→                rb: rhs,
  1643→            },
  1644→            0,
  1645→        ),
  1646→        CmpKind::SGt => (
  1647→            Instruction::Cmp {
  1648→                bf: CrField::CR0,
  1649→                l: 1,
  1650→                ra: lhs,
  1651→                rb: rhs,
  1652→            },
  1653→            1,
  1654→        ),
  1655→        CmpKind::SGe => (
  1656→            Instruction::Cmp {
  1657→                bf: CrField::CR0,
  1658→                l: 1,
  1659→                ra: lhs,
  1660→                rb: rhs,
  1661→            },
  1662→            1,
  1663→        ),
  1664→        CmpKind::ULt => (
  1665→            Instruction::Cmpl {
  1666→                bf: CrField::CR0,
  1667→                l: 1,
  1668→                ra: lhs,
  1669→                rb: rhs,
  1670→            },
  1671→            0,
  1672→        ),
  1673→        CmpKind::ULe => (
  1674→            Instruction::Cmpl {
  1675→                bf: CrField::CR0,
  1676→                l: 1,
  1677→                ra: lhs,
  1678→                rb: rhs,
  1679→            },
  1680→            0,
  1681→        ),
  1682→        CmpKind::UGt => (
  1683→            Instruction::Cmpl {
  1684→                bf: CrField::CR0,
  1685→                l: 1,
  1686→                ra: lhs,
  1687→                rb: rhs,
  1688→            },
  1689→            1,
  1690→        ),
  1691→        CmpKind::UGe => (
  1692→            Instruction::Cmpl {
  1693→                bf: CrField::CR0,
  1694→                l: 1,
  1695→                ra: lhs,
  1696→                rb: rhs,
  1697→            },
  1698→            1,
  1699→        ),
  1700→    };
  1701→    result.push(emit_alloc_instr(
  1702→        cmp_inst.clone(),
  1703→        vec![
  1704→            PhysicalReg::new(RegClass::Gpr, lhs.encoding()),
  1705→            PhysicalReg::new(RegClass::Gpr, rhs.encoding()),
  1706→        ],
  1707→        vec![],
  1708→    ));
  1709→    // li dst, 0; then conditionally set to 1
  1710→    result.push(emit_alloc_instr(
  1711→        Instruction::Li { rt: dst, simm: 0 },
  1712→        vec![],
  1713→        vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1714→    ));
  1715→    let (bo, bi) = match kind {
  1716→        // Pattern: li dst,0; bc <skip li dst,1>; li dst,1
  1717→        // Branch is taken → dst stays 0; fall-through → dst becomes 1.
  1718→        // So we want branch taken when result should be 0.
  1719→        CmpKind::Eq => (4, 2 + CrField::CR0.encoding() * 4),  // branch if cr0.eq FALSE → dst=1 when eq
  1720→        CmpKind::Ne => (12, 2 + CrField::CR0.encoding() * 4), // branch if cr0.eq TRUE  → dst=0 when eq, dst=1 when ne
  1721→        CmpKind::SLt | CmpKind::ULt => (4, CrField::CR0.encoding() * 4),   // branch if cr0.lt FALSE
  1722→        CmpKind::SLe | CmpKind::ULe => (12, 1 + CrField::CR0.encoding() * 4), // branch if cr0.gt TRUE
  1723→        CmpKind::SGt | CmpKind::UGt => (4, 1 + CrField::CR0.encoding() * 4), // branch if cr0.gt FALSE
  1724→        CmpKind::SGe | CmpKind::UGe => (12, CrField::CR0.encoding() * 4),   // branch if cr0.lt TRUE
  1725→    };
  1726→    let _ = cond_bit;
  1727→    result.push(emit_alloc_instr(
  1728→        Instruction::Li { rt: dst, simm: 1 },
  1729→        vec![],
  1730→        vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1731→    ));
  1732→    // Patch: use a simplified approach — li dst,1 is always emitted but should be conditional.
  1733→    // For simplicity, use select-like pattern:
  1734→    // cmp; li dst, 0; bne cr0, +8; li dst, 1
  1735→    // Actually let's redo this more correctly:
  1736→    result.clear();
  1737→    result.push(emit_alloc_instr(
  1738→        cmp_inst,
  1739→        vec![
  1740→            PhysicalReg::new(RegClass::Gpr, lhs.encoding()),
  1741→            PhysicalReg::new(RegClass::Gpr, rhs.encoding()),
  1742→        ],
  1743→        vec![],
  1744→    ));
  1745→    result.push(emit_alloc_instr(
  1746→        Instruction::Li { rt: dst, simm: 0 },
  1747→        vec![],
  1748→        vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1749→    ));
  1750→    result.push(emit_alloc_instr(
  1751→        Instruction::Bc { bo, bi, bd: 2 },
  1752→        vec![],
  1753→        vec![],
  1754→    ));
  1755→    result.push(emit_alloc_instr(
  1756→        Instruction::Li { rt: dst, simm: 1 },
  1757→        vec![],
  1758→        vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1759→    ));
  1760→    result
  1761→}

  1763→/// Lower an IR instruction to a sequence of PPC64 AllocatedInstructions,
  1764→/// plus any relocations that need to be patched later.
  1765→///
  1766→/// `alloc_offset` tracks the next free stack-slot offset (starting after the
  1767→/// 32-byte mandatory save area) so that `Alloc` instructions can compute
  1768→/// addresses within the already-allocated frame instead of double-decrementing
  1769→/// SP.
  1770→fn lower_ir_instr_ppc64(
  1771→    instr: &IRInstr,
  1772→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1773→    alloc_offset: &mut i32,
  1774→) -> (Vec<AllocatedInstruction>, Vec<crate::backend::RelocationEntry>) {
  1775→    let mut result = Vec::new();
  1776→    let mut relocations = Vec::new();
  1777→
  1778→    match instr {
  1779→        IRInstr::BinOp { op, dst, lhs, rhs, .. } => {
  1780→            let d = map_vreg_to_gpr(vreg_id(dst), None, vreg_map);
  1781→            let l = resolve_gpr_ppc64(lhs, vreg_map, Gpr::R0, &mut result);
