  3933→    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError> {

  3936→        // ── Phase 1: Collect all vreg IDs and compute stack layout ──
  3937→
  3938→        let mut all_vreg_ids: std::collections::HashSet<u32> =
  3939→            std::collections::HashSet::new();

  3941→            all_vreg_ids.insert(id);
  3942→        }

  3944→            if let Some(id) = param.as_register() {
  3945→                all_vreg_ids.insert(id);

  3952→                }
  3953→                for id in instr.used_regs() {
  3954→                    all_vreg_ids.insert(id);
  3955→                }
  3956→            }
  3957→        }
  3958→        for val in &func.results {
  3959→            if let Some(id) = val.as_register() {
  3960→                all_vreg_ids.insert(id);
  3961→            }
  3962→        }
  3963→
  3964→        // Identify Alloc vregs and their sizes
  3965→        let mut stack_alloc_vregs: std::collections::HashSet<u32> =
  3966→            std::collections::HashSet::new();
  3967→        let mut alloc_sizes: HashMap<u32, i32> = HashMap::new();
  3968→        for block in &func.blocks {
  3969→            for instr in &block.instructions {
  3970→                if let IRInstr::Alloc { dst, size } = instr {
  3971→                    if let Some(id) = dst.as_register() {
  3972→                        stack_alloc_vregs.insert(id);
  3973→                        let aligned_size = ((*size as i32 + 15) & !15) as i32;
  3974→                        alloc_sizes.insert(id, aligned_size);
  3975→                    }
  3976→                }
  3977→            }
  3978→        }
  3979→
  3980→        // ── Stack Layout ──
  3981→        let mut alloc_offsets: HashMap<u32, i32> = HashMap::new();
  3982→        let mut current_offset: i32 = 16; // skip RA+S0 save area (16 bytes)
  3983→
  3984→        let mut alloc_vreg_ids: Vec<u32> = stack_alloc_vregs.iter().copied().collect();
  3985→        alloc_vreg_ids.sort();
  3986→        for &id in &alloc_vreg_ids {
  3987→            let size = alloc_sizes[&id];
  3988→            current_offset += size;
  3989→            alloc_offsets.insert(id, current_offset);
  3990→        }
  3991→
  3992→        let mut vreg_stack_slots: HashMap<u32, i32> = HashMap::new();
  3993→        let mut all_vreg_ids_sorted: Vec<u32> = all_vreg_ids.iter().copied().collect();
  3994→        all_vreg_ids_sorted.sort();
  3995→        for &id in &all_vreg_ids_sorted {
  3996→            current_offset += 8;
  3997→            vreg_stack_slots.insert(id, current_offset);
  3998→        }
  3999→

  4001→        let fs = frame_size as i32;
  4002→
  4003→        // ── Phase 2: Emit prologue ──
  4004→
  4005→        let mut instructions: Vec<AllocatedInstruction> = Vec::new();
  4006→        let mut relocations: Vec<RelocationEntry> = Vec::new();
  4007→
  4008→        // Prologue: addi sp, sp, -frame_size; sd ra, fs-8(sp); sd s0, fs-16(sp); addi s0, sp, fs
  4009→        if fs >= -2048 && fs <= 2047 {
  4010→            instructions.push(AllocatedInstruction {
  4011→                opcode: "addi".to_string(),
  4012→                reads: vec![PhysicalReg::new(RegClass::Gpr, Gpr::Sp.encoding())],
  4013→                writes: vec![PhysicalReg::new(RegClass::Gpr, Gpr::Sp.encoding())],
  4014→                encoded: Instruction::Addi { rd: Gpr::Sp, rs1: Gpr::Sp, imm: -fs }.encode().to_vec(),
  4015→            });
  4016→        } else {
  4017→            let mut prologue = Vec::new();
  4018→            prologue.extend(ss_load_imm(Gpr::T0, fs as i64));
  4019→            prologue.extend(Instruction::Sub { rd: Gpr::Sp, rs1: Gpr::Sp, rs2: Gpr::T0 }.encode());
  4020→            instructions.push(AllocatedInstruction {
  4021→                opcode: "sub".to_string(),
  4022→                reads: vec![PhysicalReg::new(RegClass::Gpr, Gpr::Sp.encoding())],
  4023→                writes: vec![PhysicalReg::new(RegClass::Gpr, Gpr::Sp.encoding())],
  4024→                encoded: prologue,
  4025→            });
  4026→        }
  4027→
  4028→        // Save RA and S0
  4029→        if fs - 8 >= -2048 {
  4030→            instructions.push(AllocatedInstruction {
  4031→                opcode: "sd".to_string(),
  4032→                reads: vec![PhysicalReg::new(RegClass::Gpr, Gpr::Ra.encoding()), PhysicalReg::new(RegClass::Gpr, Gpr::Sp.encoding())],
  4033→                writes: vec![],
  4034→                encoded: Instruction::Sd { rs1: Gpr::Sp, rs2: Gpr::Ra, imm: fs - 8 }.encode().to_vec(),
  4035→            });
  4036→        } else {
  4037→            let mut code = Vec::new();
  4038→            code.extend(ss_load_imm(Gpr::T0, (fs - 8) as i64));
  4039→            code.extend(Instruction::Add { rd: Gpr::T0, rs1: Gpr::Sp, rs2: Gpr::T0 }.encode());
  4040→            code.extend(Instruction::Sd { rs1: Gpr::T0, rs2: Gpr::Ra, imm: 0 }.encode());

  4042→                opcode: "sd".to_string(),
  4043→                reads: vec![PhysicalReg::new(RegClass::Gpr, Gpr::Ra.encoding()), PhysicalReg::new(RegClass::Gpr, Gpr::Sp.encoding())],
  4044→                writes: vec![],
  4045→                encoded: code,
  4046→            });
  4047→        }
  4048→
  4049→        if fs - 16 >= -2048 {
  4050→            instructions.push(AllocatedInstruction {
  4051→                opcode: "sd".to_string(),
  4052→                reads: vec![PhysicalReg::new(RegClass::Gpr, Gpr::S0.encoding()), PhysicalReg::new(RegClass::Gpr, Gpr::Sp.encoding())],
  4053→                writes: vec![],
  4054→                encoded: Instruction::Sd { rs1: Gpr::Sp, rs2: Gpr::S0, imm: fs - 16 }.encode().to_vec(),
  4055→            });
  4056→        } else {
  4057→            let mut code = Vec::new();
  4058→            code.extend(ss_load_imm(Gpr::T0, (fs - 16) as i64));
  4059→            code.extend(Instruction::Add { rd: Gpr::T0, rs1: Gpr::Sp, rs2: Gpr::T0 }.encode());
  4060→            code.extend(Instruction::Sd { rs1: Gpr::T0, rs2: Gpr::S0, imm: 0 }.encode());
  4061→            instructions.push(AllocatedInstruction {
  4062→                opcode: "sd".to_string(),
  4063→                reads: vec![PhysicalReg::new(RegClass::Gpr, Gpr::S0.encoding()), PhysicalReg::new(RegClass::Gpr, Gpr::Sp.encoding())],
  4064→                writes: vec![],
  4065→                encoded: code,
  4066→            });
  4067→        }
  4068→
  4069→        // Set frame pointer: addi s0, sp, frame_size
  4070→        if fs >= -2048 && fs <= 2047 {
  4071→            instructions.push(AllocatedInstruction {
  4072→                opcode: "addi".to_string(),
  4073→                reads: vec![PhysicalReg::new(RegClass::Gpr, Gpr::Sp.encoding())],
  4074→                writes: vec![PhysicalReg::new(RegClass::Gpr, Gpr::S0.encoding())],
  4075→                encoded: Instruction::Addi { rd: Gpr::S0, rs1: Gpr::Sp, imm: fs }.encode().to_vec(),
  4076→            });
  4077→        } else {
  4078→            let mut code = Vec::new();
  4079→            code.extend(ss_load_imm(Gpr::T0, fs as i64));
  4080→            code.extend(Instruction::Add { rd: Gpr::S0, rs1: Gpr::Sp, rs2: Gpr::T0 }.encode());
  4081→            instructions.push(AllocatedInstruction {
  4082→                opcode: "add".to_string(),
  4083→                reads: vec![PhysicalReg::new(RegClass::Gpr, Gpr::Sp.encoding())],
  4084→                writes: vec![PhysicalReg::new(RegClass::Gpr, Gpr::S0.encoding())],
  4085→                encoded: code,
  4086→            });
  4087→        }
  4088→
  4089→        // Store function parameters from A0-A7 to their stack slots
  4090→        let arg_regs = [Gpr::A0, Gpr::A1, Gpr::A2, Gpr::A3, Gpr::A4, Gpr::A5, Gpr::A6, Gpr::A7];
  4091→        for (i, param) in func.params.iter().enumerate() {
  4092→            if let Some(id) = param.as_register() {
  4093→                if i < 8 {
  4094→                    let offset = vreg_stack_slots.get(&id).copied().unwrap_or(0);
  4095→                    let store_code = ss_store_to_slot(arg_regs[i], offset);
  4096→                    instructions.push(AllocatedInstruction {
  4097→                        opcode: "sd".to_string(),
  4098→                        reads: vec![PhysicalReg::new(RegClass::Gpr, arg_regs[i].encoding())],
  4099→                        writes: vec![],
  4100→                        encoded: store_code,
  4101→                    });
  4102→                }
  4103→            }
  4104→        }
  4105→
  4106→        // ── Phase 3: Emit body with branch fixup tracking ──
  4107→
  4108→        let mut current_byte_offset: u64 = instructions.iter().map(|i| i.encoded.len() as u64).sum();
  4109→        let mut label_offsets: HashMap<String, u64> = HashMap::new();

  4111→        // Branch fixup: records a branch instruction that needs its offset patched
  4112→        struct BranchFixup {

  4114→            offset_in_encoded: usize, // Byte offset within the instruction's encoded bytes
  4115→            abs_byte_offset: u64,     // Absolute byte offset of the branch in the function
  4116→            target_label: String,     // Target block label
  4117→            is_jal: bool,             // true for JAL, false for BNE
  4118→            jal_rd: Gpr,              // For JAL: rd field
  4119→            bne_rs1: Gpr,             // For BNE: rs1 field
  4120→            bne_rs2: Gpr,             // For BNE: rs2 field
  4121→        }
  4122→        let mut branch_fixups: Vec<BranchFixup> = Vec::new();
  4123→
  4124→        for block in &func.blocks {
  4125→            // Record the byte offset for this block's label
  4126→            label_offsets.insert(block.label.clone(), current_byte_offset);
  4127→

  4129→                let encoded: Vec<u8> = match instr {
  4130→                    // ── BinOp (generic) ──────────────────────────────────────
  4131→                    IRInstr::BinOp { op, dst, lhs, rhs, .. } => {

  4134→                        let mut code = Vec::new();
  4135→
  4136→                        match op {

  4138→                                code.extend(ss_load_value(lhs, &vreg_stack_slots, Gpr::T1));
  4139→                                code.extend(ss_load_value(rhs, &vreg_stack_slots, Gpr::T2));
  4140→                                code.extend(Instruction::Sub { rd: Gpr::T4, rs1: Gpr::Zero, rs2: Gpr::T2 }.encode());

  4143→                                    code.extend(Instruction::Srl { rd: Gpr::T0, rs1: Gpr::T1, rs2: Gpr::T2 }.encode());

  4145→                                } else {
  4146→                                    code.extend(Instruction::Sll { rd: Gpr::T0, rs1: Gpr::T1, rs2: Gpr::T2 }.encode());
  4147→                                    code.extend(Instruction::Srl { rd: Gpr::T4, rs1: Gpr::T1, rs2: Gpr::T4 }.encode());

  4149→                                code.extend(Instruction::Or { rd: Gpr::T0, rs1: Gpr::T0, rs2: Gpr::T4 }.encode());

  4151→                            _ => {
