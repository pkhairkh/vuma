     1→     1→//! # PowerPC64 Backend
     2→     2→//!
     3→     3→//! Implements the `Backend` trait for the PowerPC 64-bit target (ELFv2 ABI,
     4→     4→//! little-endian default). This module provides:
     5→     5→//!
     6→     6→//! - `Gpr` — General-purpose register enum (R0–R31)
     7→     7→//! - `Fpr` — Floating-point register enum (F0–F31)
     8→     8→//! - `CrField` — Condition register field enum (CR0–CR7)
     9→     9→//! - `Instruction` — PPC64 instruction enum with correct 32-bit encoding
    10→    10→//! - `PPC64Backend` — `Backend` implementation that lowers IR to PPC64
    11→    11→//!   machine code and emits ELF64 binaries
    12→    12→//!
    13→    13→//! ## PowerPC64 Register Convention (ELFv2 ABI)
    14→    14→//!
    15→    15→//! | Register(s) | Role                                  |
    16→    16→//! |-------------|---------------------------------------|
    17→    17→//! | R0          | Volatile (not hardwired zero!)        |
    18→    18→//! | R1          | Stack pointer (SP)                    |
    19→    19→//! | R2          | TOC pointer                           |
    20→    20→//! | R3–R10      | Argument / return registers           |
    21→    21→//! | R11–R12     | Volatile scratch                      |
    22→    22→//! | R13         | Thread pointer                        |
    23→    23→//! | R14–R31     | Callee-saved                          |
    24→    24→//! | F0–F13      | FP argument / return, volatile        |
    25→    25→//! | F14–F31     | FP callee-saved                       |
    26→    26→//! | CR0–CR7     | Condition register fields (4 bits ea) |
    27→    27→//! | LR          | Link register (SPR)                   |
    28→    28→//! | CTR         | Count register (SPR)                  |
    29→    29→//!
    30→    30→//! ## Instruction Encoding
    31→    31→//!
    32→    32→//! All instructions are 32 bits, fixed-width. PPC64 is bi-endian; the default
    33→    33→//! for ppc64le is little-endian byte order for both data and instructions.
    34→    34→//! The primary opcode occupies bits \[0:5\] (MSB-first bit numbering).
    35→    35→//!
    36→    36→//! ## References
    37→    37→//!
    38→    38→//! - Power ISA Version 3.1
    39→    39→//! - <https://openpowerfoundation.org/specifications/isa/>
    40→    40→

   418→   418→// ===========================================================================
   419→   419→// Instruction Encoding Helpers
   420→   420→// ===========================================================================
   421→   421→
   422→   422→/// Encode a PPC64 32-bit instruction word and return as big-endian bytes.
   423→   423→fn encode_word(word: u32) -> [u8; 4] {
   424→   424→    word.to_be_bytes()
   425→   425→}
   426→   426→
   427→   427→/// Build a D-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | d[16:31]
   428→   428→fn encode_d_form(opcode: u32, rt: u32, ra: u32, d: i32) -> [u8; 4] {
   429→   429→    let word =
   430→   430→        ((opcode & 0x3F) << 26) | ((rt & 0x1F) << 21) | ((ra & 0x1F) << 16) | ((d as u32) & 0xFFFF);
   431→   431→    encode_word(word)
   432→   432→}
   433→   433→
   434→   434→/// Build a DS-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | ds[16:29] | xo[30:31]
   435→   435→/// The `ds` parameter is the byte offset; the DS field stores ds/4.
   436→   436→fn encode_ds_form(opcode: u32, rt: u32, ra: u32, ds: i32, xo: u32) -> [u8; 4] {
   437→   437→    let word = ((opcode & 0x3F) << 26)
   438→   438→        | ((rt & 0x1F) << 21)
   439→   439→        | ((ra & 0x1F) << 16)
   440→   440→        | (((ds >> 2) as u32) & 0x3FFF) << 2
   441→   441→        | (xo & 0x3);
   442→   442→    encode_word(word)
   443→   443→}
   444→   444→
   445→   445→/// Build an X-form instruction: opcode[0:5] | rS[6:10] | rA[11:15] | rB[16:20] | xo[21:30] | Rc[31]
   446→   446→fn encode_x_form(opcode: u32, rs: u32, ra: u32, rb: u32, xo: u32, rc: u32) -> [u8; 4] {
   447→   447→    let word = ((opcode & 0x3F) << 26)
   448→   448→        | ((rs & 0x1F) << 21)
   449→   449→        | ((ra & 0x1F) << 16)
   450→   450→        | ((rb & 0x1F) << 11)
   451→   451→        | ((xo & 0x3FF) << 1)
   452→   452→        | (rc & 1);
   453→   453→    encode_word(word)
   454→   454→}
   455→   455→
   456→   456→/// Build an XO-form instruction: opcode[0:5] | rT[6:10] | rA[11:15] | rB[16:20] | OE[21] | xo[22:30] | Rc[31]
   457→   457→fn encode_xo_form(opcode: u32, rt: u32, ra: u32, rb: u32, oe: u32, xo: u32, rc: u32) -> [u8; 4] {
   458→   458→    let word = ((opcode & 0x3F) << 26)
   459→   459→        | ((rt & 0x1F) << 21)
   460→   460→        | ((ra & 0x1F) << 16)
   461→   461→        | ((rb & 0x1F) << 11)
   462→   462→        | ((oe & 1) << 10)
   463→   463→        | ((xo & 0x1FF) << 1)
   464→   464→        | (rc & 1);
   465→   465→    encode_word(word)
   466→   466→}
   467→   467→
   468→   468→/// Build an M-form instruction: opcode[0:5] | rS[6:10] | rA[11:15] | SH[16:20] | MB[21:25] | ME[26:30] | Rc[31]
   469→   469→fn encode_m_form(opcode: u32, rs: u32, ra: u32, sh: u32, mb: u32, me: u32, rc: u32) -> [u8; 4] {
   470→   470→    let word = ((opcode & 0x3F) << 26)
   471→   471→        | ((rs & 0x1F) << 21)
   472→   472→        | ((ra & 0x1F) << 16)
   473→   473→        | ((sh & 0x1F) << 11)
   474→   474→        | ((mb & 0x1F) << 6)
   475→   475→        | ((me & 0x1F) << 1)
   476→   476→        | (rc & 1);
   477→   477→    encode_word(word)
   478→   478→}
   479→   479→
   480→   480→/// Build an I-form instruction: opcode[0:5] | LI[6:29] | AA[30] | LK[31]
   481→   481→fn encode_i_form(opcode: u32, li: i32, aa: u32, lk: u32) -> [u8; 4] {
   482→   482→    let word =
   483→   483→        ((opcode & 0x3F) << 26) | (((li as u32) & 0x03FF_FFFF) << 2) | ((aa & 1) << 1) | (lk & 1);
   484→   484→    encode_word(word)
   485→   485→}
   486→   486→
   487→   487→/// Build a B-form instruction: opcode[0:5] | BO[6:10] | BI[11:15] | BD[16:29] | AA[30] | LK[31]
   488→   488→fn encode_b_form(opcode: u32, bo: u32, bi: u32, bd: i32, aa: u32, lk: u32) -> [u8; 4] {
   489→   489→    let word = ((opcode & 0x3F) << 26)
   490→   490→        | ((bo & 0x1F) << 21)
   491→   491→        | ((bi & 0x1F) << 16)
   492→   492→        | (((bd as u32) & 0x3FFF) << 2)
   493→   493→        | ((aa & 1) << 1)
   494→   494→        | (lk & 1);
   495→   495→    encode_word(word)
   496→   496→}
   497→   497→
   498→   498→/// Build an XL-form instruction: opcode[0:5] | BO[6:10] | BI[11:15] | 0[16:18] | BH[19:21] | xo[21:30] | LK[31]
   499→   499→fn encode_xl_form(opcode: u32, bo: u32, bi: u32, bh: u32, xo: u32, lk: u32) -> [u8; 4] {
   500→   500→    let word = ((opcode & 0x3F) << 26)
   501→   501→        | ((bo & 0x1F) << 21)
   502→   502→        | ((bi & 0x1F) << 16)
   503→   503→        | ((bh & 0x7) << 11)
   504→   504→        | ((xo & 0x3FF) << 1)
   505→   505→        | (lk & 1);
   506→   506→    encode_word(word)
   507→   507→}

  1218→  1218→/// Build a proper ELF64 binary for PPC64LE with 2 LOAD segments.
  1219→  1219→///
  1220→  1220→/// Produces a static executable with:
  1221→  1221→/// - Segment 1: LOAD (PF_R | PF_X) — .text
  1222→  1222→/// - Segment 2: LOAD (PF_R | PF_W) — .data / BSS (writable)
  1223→  1223→///
  1224→  1224→/// Uses little-endian byte order for ppc64le.  The text segment is placed
  1225→  1225→/// at a page-aligned offset so that the kernel's ELF loader can mmap() it.
  1226→  1226→fn build_ppc64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1227→  1227→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (PPC64 typical page size)
  1228→  1228→
  1229→  1229→    let elf_header_size: u64 = 64;
  1230→  1230→    let phdr_size: u64 = 56;
  1231→  1231→    let num_phdrs: u64 = 2;
  1232→  1232→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1233→  1233→    // Page-align the text segment start in the file.
  1234→  1234→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1235→  1235→    let text_size = code.len() as u64;
  1236→  1236→
  1237→  1237→    // The data segment starts on the next page after the text.
  1238→  1238→    let text_file_end = text_offset + text_size;
  1239→  1239→    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1240→  1240→    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1241→  1241→    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1242→  1242→    let entry_point = base_addr + text_offset;
  1243→  1243→
  1244→  1244→    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1245→  1245→
  1246→  1246→    // --- e_ident ---
  1247→  1247→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1248→  1248→    elf.push(2); // ELFCLASS64
  1249→  1249→    elf.push(2); // ELFDATA2MSB (big-endian PPC64)
  1250→  1250→    elf.push(1); // EV_CURRENT
  1251→  1251→    elf.push(3); // ELFOSABI_LINUX
  1252→  1252→    elf.push(0); // padding
  1253→  1253→    elf.extend_from_slice(&[0u8; 7]); // padding
  1254→  1254→
  1255→  1255→    // --- ELF header fields ---
  1256→  1256→    elf.extend_from_slice(&2u16.to_be_bytes()); // e_type = ET_EXEC
  1257→  1257→    elf.extend_from_slice(&21u16.to_be_bytes()); // e_machine = EM_PPC64
  1258→  1258→    elf.extend_from_slice(&1u32.to_be_bytes()); // e_version
  1259→  1259→    elf.extend_from_slice(&entry_point.to_be_bytes()); // e_entry
  1260→  1260→    elf.extend_from_slice(&elf_header_size.to_be_bytes()); // e_phoff
  1261→  1261→    elf.extend_from_slice(&0u64.to_be_bytes()); // e_shoff (no section headers)
  1262→  1262→    // e_flags: EF_PPC64_ABI_V2 = 0x2 (required for PPC64LE ELFv2 ABI)
  1263→  1263→    elf.extend_from_slice(&2u32.to_be_bytes()); // e_flags
  1264→  1264→    elf.extend_from_slice(&64u16.to_be_bytes()); // e_ehsize
  1265→  1265→    elf.extend_from_slice(&56u16.to_be_bytes()); // e_phentsize
  1266→  1266→    elf.extend_from_slice(&2u16.to_be_bytes()); // e_phnum = 2
  1267→  1267→    elf.extend_from_slice(&64u16.to_be_bytes()); // e_shentsize
  1268→  1268→    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shnum
  1269→  1269→    elf.extend_from_slice(&0u16.to_be_bytes()); // e_shstrndx
  1270→  1270→
  1271→  1271→    // --- Program Header 1: LOAD (PF_R | PF_X) — .text ---
  1272→  1272→    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1273→  1273→    elf.extend_from_slice(&5u32.to_be_bytes()); // p_flags = PF_R | PF_X
  1274→  1274→    elf.extend_from_slice(&text_offset.to_be_bytes()); // p_offset
  1275→  1275→    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_vaddr
  1276→  1276→    elf.extend_from_slice(&(base_addr + text_offset).to_be_bytes()); // p_paddr
  1277→  1277→    elf.extend_from_slice(&text_size.to_be_bytes()); // p_filesz
  1278→  1278→    elf.extend_from_slice(&text_size.to_be_bytes()); // p_memsz
  1279→  1279→    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1280→  1280→
  1281→  1281→    // --- Program Header 2: LOAD (PF_R | PF_W) — .data / stack ---
  1282→  1282→    elf.extend_from_slice(&1u32.to_be_bytes()); // p_type = PT_LOAD
  1283→  1283→    elf.extend_from_slice(&6u32.to_be_bytes()); // p_flags = PF_R | PF_W
  1284→  1284→    elf.extend_from_slice(&data_offset.to_be_bytes()); // p_offset
  1285→  1285→    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_vaddr
  1286→  1286→    elf.extend_from_slice(&data_vaddr.to_be_bytes()); // p_paddr
  1287→  1287→    elf.extend_from_slice(&0u64.to_be_bytes()); // p_filesz (no initialized data)
  1288→  1288→    elf.extend_from_slice(&data_size.to_be_bytes()); // p_memsz (writable pages)
  1289→  1289→    elf.extend_from_slice(&PAGE_SIZE.to_be_bytes()); // p_align
  1290→  1290→
  1291→  1291→    // --- .text section ---
  1292→  1292→    // Pad to page-aligned text_offset
  1293→  1293→    while (elf.len() as u64) < text_offset {
  1294→  1294→        elf.push(0);
  1295→  1295→    }
  1296→  1296→    elf.extend_from_slice(code);
  1297→  1297→
  1298→  1298→    // --- Pad to data segment offset (if needed) ---
  1299→  1299→    while (elf.len() as u64) < data_offset {
  1300→  1300→        elf.push(0);
  1301→  1301→    }
  1302→  1302→
  1303→  1303→    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1304→  1304→
  1305→  1305→    elf
  1306→  1306→}

  1332→  1332→/// Compute the stack frame size for an IR function on PPC64.
  1333→  1333→///
  1334→  1334→/// Sums `Alloc` instruction sizes, adds 32 bytes for the LR/CR save area
  1335→  1335→/// (per ELFv2 ABI), and rounds up to 16-byte alignment.
  1336→  1336→fn ppc64_compute_frame_size(func: &IRFunction) -> usize {
  1337→  1337→    let mut total: u32 = 32; // LR save (8) + CR save (8) + back chain (8) + TOC save (8)
  1338→  1338→    for block in &func.blocks {
  1339→  1339→        for instr in &block.instructions {
  1340→  1340→            if let IRInstr::Alloc { size, .. } = instr {
  1341→  1341→                let aligned = (*size).div_ceil(16) * 16;
  1342→  1342→                total += aligned;
  1343→  1343→            }
  1344→  1344→        }
  1345→  1345→    }
  1346→  1346→    // Round up to 16-byte alignment
  1347→  1347→    total = (total + 15) & !15;
  1348→  1348→    total as usize
  1349→  1349→}

  1390→  1390→/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1391→  1391→fn map_vreg_to_gpr(
  1392→  1392→    vreg_id: u32,
  1393→  1393→    arg_index: Option<usize>,
  1394→  1394→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1395→  1395→) -> Gpr {
  1396→  1396→    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1397→  1397→        return *gpr;
  1398→  1398→    }
  1399→  1399→
  1400→  1400→    // If this is an argument, map it to the corresponding argument register.
  1401→  1401→    if let Some(idx) = arg_index {
  1402→  1402→        if let Some(reg) = Gpr::arg_register(idx) {
  1403→  1403→            vreg_map.insert(vreg_id, reg);
  1404→  1404→            return reg;
  1405→  1405→        }
  1406→  1406→    }
  1407→  1407→
  1408→  1408→    // Otherwise, pick the first available allocatable register.
  1409→  1409→    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1410→  1410→    for &reg in ALLOCATABLE_GPRS {
  1411→  1411→        if !used.contains(&reg) {
  1412→  1412→            vreg_map.insert(vreg_id, reg);
  1413→  1413→            return reg;
  1414→  1414→        }
  1415→  1415→    }
  1416→  1416→
  1417→  1417→    // Fallback: use R11 as a scratch register.
  1418→  1418→    Gpr::R11
  1419→  1419→}

  1447→  1447→/// - **`[-32768, 32767]`**: `li rd, val` (single `addi rd, r0, val`)
  1448→  1448→/// - **`[0, 65535]`**: `ori rd, r0, val`
  1449→  1449→/// - **`0xXXXX0000` (low 16 bits zero, fits 32-bit)**: `lis rd, upper16`
  1450→  1450→/// - **Other 32-bit**: `lis rd, upper16` + `ori rd, rd, lower16`
  1451→  1451→/// - **Full 64-bit**: `lis` + `ori` + `sldi 32` + `oris` + `ori`
  1452→  1452→fn load_immediate_ppc64(rd: Gpr, val: i64, out: &mut Vec<AllocatedInstruction>) {
  1453→  1453→    let rd_phys = PhysicalReg::new(RegClass::Gpr, rd.encoding());
  1454→  1454→    let uval = val as u64;
  1455→  1455→
  1456→  1456→    if (-32768..=32767).contains(&val) {
  1457→  1457→        // li rd, val (addi rd, r0, val)
  1458→  1458→        out.push(emit_alloc_instr(
  1459→  1459→            Instruction::Li {
  1460→  1460→                rt: rd,
  1461→  1461→                simm: val as i32,
  1462→  1462→            },
  1463→  1463→            vec![],
  1464→  1464→            vec![rd_phys],
  1465→  1465→        ));
  1466→  1466→    } else if uval <= 0xFFFF {
  1467→  1467→        // ori rd, r0, val (unsigned 16-bit)
  1468→  1468→        out.push(emit_alloc_instr(
  1469→  1469→            Instruction::Ori {
  1470→  1470→                ra: rd,
  1471→  1471→                rs: Gpr::R0,
  1472→  1472→                uimm: uval as u32 & 0xFFFF,
  1473→  1473→            },
  1474→  1474→            vec![],
  1475→  1475→            vec![rd_phys],
  1476→  1476→        ));
  1477→  1477→    } else if uval & 0xFFFF == 0 && (uval >> 32) == 0 {
  1478→  1478→        // lis rd, upper16 (0xXXXX0000)
  1479→  1479→        out.push(emit_alloc_instr(
  1480→  1480→            Instruction::Lis {
  1481→  1481→                rt: rd,
  1482→  1482→                simm: ((uval >> 16) & 0xFFFF) as i16 as i32,
  1483→  1483→            },
  1484→  1484→            vec![],
  1485→  1485→            vec![rd_phys],
  1486→  1486→        ));
  1487→  1487→    } else if uval >> 32 == 0 {
  1488→  1488→        // 32-bit value: lis + ori
  1489→  1489→        let hi16 = ((uval >> 16) & 0xFFFF) as u32;
  1490→  1490→        let lo16 = (uval & 0xFFFF) as u32;
  1491→  1491→        out.push(emit_alloc_instr(
  1492→  1492→            Instruction::Lis {
  1493→  1493→                rt: rd,
  1494→  1494→                simm: hi16 as i16 as i32,
  1495→  1495→            },
  1496→  1496→            vec![],
  1497→  1497→            vec![rd_phys],
  1498→  1498→        ));
  1499→  1499→        if lo16 != 0 {
  1500→  1500→            out.push(emit_alloc_instr(
  1501→  1501→                Instruction::Ori {
  1502→  1502→                    ra: rd,
  1503→  1503→                    rs: rd,
  1504→  1504→                    uimm: lo16,
  1505→  1505→                },
  1506→  1506→                vec![rd_phys],
  1507→  1507→                vec![rd_phys],
  1508→  1508→            ));
  1509→  1509→        }
  1510→  1510→    } else {
  1511→  1511→        // Full 64-bit: lis + ori + sldi 32 + oris + ori
  1512→  1512→        let hi32 = (uval >> 32) as u32;
  1513→  1513→        let lo32 = (uval & 0xFFFF_FFFF) as u32;
  1514→  1514→
  1515→  1515→        // lis rd, upper16(hi32)
  1516→  1516→        out.push(emit_alloc_instr(
  1517→  1517→            Instruction::Lis {
  1518→  1518→                rt: rd,
  1519→  1519→                simm: ((hi32 >> 16) & 0xFFFF) as i16 as i32,
  1520→  1520→            },
  1521→  1521→            vec![],
  1522→  1522→            vec![rd_phys],
  1523→  1523→        ));
  1524→  1524→
  1525→  1525→        // ori rd, rd, lower16(hi32)
  1526→  1526→        if hi32 & 0xFFFF != 0 {
  1527→  1527→            out.push(emit_alloc_instr(
  1528→  1528→                Instruction::Ori {
  1529→  1529→                    ra: rd,
  1530→  1530→                    rs: rd,
  1531→  1531→                    uimm: hi32 & 0xFFFF,
  1532→  1532→                },
  1533→  1533→                vec![rd_phys],
  1534→  1534→                vec![rd_phys],
  1535→  1535→            ));
  1536→  1536→        }
  1537→  1537→
  1538→  1538→        // sldi rd, rd, 32 = rldicr rd, rd, 32, 31
  1539→  1539→        // MD-form: [0:5]=30, [6:10]=rS, [11:15]=rA, [16:20]=SH[0:4],
  1540→  1540→        // [21:25]=ME[0:4], [26]=SH[5], [27]=ME[5], [28:30]=xo(=2), [31]=Rc(=0)
  1541→  1541→        let sh: u32 = 32;
  1542→  1542→        let me: u32 = 63 - sh;
  1543→  1543→        let sldi_word: u32 = (30u32 << 26)
  1544→  1544→            | (rd.encoding() << 21)
  1545→  1545→            | (rd.encoding() << 16)
  1546→  1546→            | ((sh & 0x1F) << 11)
  1547→  1547→            | ((me & 0x1F) << 6)
  1548→  1548→            | (((sh >> 5) & 1) << 5)
  1549→  1549→            | (((me >> 5) & 1) << 4)
  1550→  1550→            | (2u32 << 1);
  1551→  1551→        out.push(AllocatedInstruction {
  1552→  1552→            opcode: "rldicr".to_string(),
  1553→  1553→            reads: vec![rd_phys],
  1554→  1554→            writes: vec![rd_phys],
  1555→  1555→            encoded: encode_word(sldi_word).to_vec(),
  1556→  1556→        });
  1557→  1557→
  1558→  1558→        // oris rd, rd, upper16(lo32) — primary opcode 25, D-form
  1559→  1559→        if (lo32 >> 16) & 0xFFFF != 0 {
  1560→  1560→            let oris_word: u32 = (25u32 << 26)
  1561→  1561→                | (rd.encoding() << 21)
  1562→  1562→                | (rd.encoding() << 16)
  1563→  1563→                | ((lo32 >> 16) & 0xFFFF);
  1564→  1564→            out.push(AllocatedInstruction {
  1565→  1565→                opcode: "oris".to_string(),
  1566→  1566→                reads: vec![rd_phys],
  1567→  1567→                writes: vec![rd_phys],
  1568→  1568→                encoded: encode_word(oris_word).to_vec(),
  1569→  1569→            });
  1570→  1570→        }
  1571→  1571→
  1572→  1572→        // ori rd, rd, lower16(lo32)
  1573→  1573→        if lo32 & 0xFFFF != 0 {
  1574→  1574→            out.push(emit_alloc_instr(
  1575→  1575→                Instruction::Ori {
  1576→  1576→                    ra: rd,
  1577→  1577→                    rs: rd,
  1578→  1578→                    uimm: lo32 & 0xFFFF,
  1579→  1579→                },
  1580→  1580→                vec![rd_phys],
  1581→  1581→                vec![rd_phys],
  1582→  1582→            ));
  1583→  1583→        }
  1584→  1584→    }
  1585→  1585→}

  1587→  1587→/// Resolve an `IRValue` to a physical `Gpr`, emitting immediate-load
  1588→  1588→/// instructions as needed.
  1589→  1589→///
  1590→  1590→/// - `IRValue::Register(id)` → looks up `id` in `reg_map`
  1591→  1591→/// - `IRValue::Immediate(val)` → loads `val` into `scratch` via
  1592→  1592→///   `load_immediate_ppc64` and returns `scratch`
  1593→  1593→/// - `IRValue::Address(addr)` → loads `addr` into `scratch`
  1594→  1594→/// - `IRValue::Label(_)` → loads 0 into `scratch` (placeholder for relocation)
  1595→  1595→///
  1596→  1596→/// The caller must supply a scratch register that is not live at this point
  1597→  1597→/// (e.g., `R11` for the first operand, `R12` for the second).
  1598→  1598→fn resolve_gpr_ppc64(
  1599→  1599→    val: &IRValue,
  1600→  1600→    reg_map: &mut std::collections::HashMap<u32, Gpr>,
  1601→  1601→    scratch: Gpr,
  1602→  1602→    out: &mut Vec<AllocatedInstruction>,
  1603→  1603→) -> Gpr {
  1604→  1604→    match val {
  1605→  1605→        IRValue::Register(id) => map_vreg_to_gpr(*id, None, reg_map),
  1606→  1606→        IRValue::Immediate(imm) => {
  1607→  1607→            load_immediate_ppc64(scratch, *imm, out);
  1608→  1608→            scratch
  1609→  1609→        }
  1610→  1610→        IRValue::Address(addr) => {
  1611→  1611→            load_immediate_ppc64(scratch, *addr as i64, out);
  1612→  1612→            scratch
  1613→  1613→        }
  1614→  1614→        IRValue::Label(_) => {
  1615→  1615→            // Labels need linker relocation; emit li scratch, 0 as placeholder
  1616→  1616→            load_immediate_ppc64(scratch, 0, out);
  1617→  1617→            scratch
  1618→  1618→        }
  1619→  1619→    }
  1620→  1620→}

  1622→  1622→/// Lower a comparison kind to PPC64 instructions, producing 0 or 1 in `dst`.
  1623→  1623→fn lower_cmp_ppc64(kind: &CmpKind, dst: Gpr, lhs: Gpr, rhs: Gpr) -> Vec<AllocatedInstruction> {
  1624→  1624→    let mut result = Vec::new();
  1625→  1625→    let (cmp_inst, cond_bit) = match kind {
  1626→  1626→        CmpKind::Eq => (
  1627→  1627→            Instruction::Cmp {
  1628→  1628→                bf: CrField::CR0,
  1629→  1629→                l: 1,
  1630→  1630→                ra: lhs,
  1631→  1631→                rb: rhs,
  1632→  1632→            },
  1633→  1633→            2,
  1634→  1634→        ),
  1635→  1635→        CmpKind::Ne => (
  1636→  1636→            Instruction::Cmp {
  1637→  1637→                bf: CrField::CR0,
  1638→  1638→                l: 1,
  1639→  1639→                ra: lhs,
  1640→  1640→                rb: rhs,
  1641→  1641→            },
  1642→  1642→            2,
  1643→  1643→        ),
  1644→  1644→        CmpKind::SLt => (
  1645→  1645→            Instruction::Cmp {
  1646→  1646→                bf: CrField::CR0,
  1647→  1647→                l: 1,
  1648→  1648→                ra: lhs,
  1649→  1649→                rb: rhs,
  1650→  1650→            },
  1651→  1651→            0,
  1652→  1652→        ),
  1653→  1653→        CmpKind::SLe => (
  1654→  1654→            Instruction::Cmp {
  1655→  1655→                bf: CrField::CR0,
  1656→  1656→                l: 1,
  1657→  1657→                ra: lhs,
  1658→  1658→                rb: rhs,
  1659→  1659→            },
  1660→  1660→            0,
  1661→  1661→        ),
  1662→  1662→        CmpKind::SGt => (
  1663→  1663→            Instruction::Cmp {
  1664→  1664→                bf: CrField::CR0,
  1665→  1665→                l: 1,
  1666→  1666→                ra: lhs,
  1667→  1667→                rb: rhs,
  1668→  1668→            },
  1669→  1669→            1,
  1670→  1670→        ),
  1671→  1671→        CmpKind::SGe => (
  1672→  1672→            Instruction::Cmp {
  1673→  1673→                bf: CrField::CR0,
  1674→  1674→                l: 1,
  1675→  1675→                ra: lhs,
  1676→  1676→                rb: rhs,
  1677→  1677→            },
  1678→  1678→            1,
  1679→  1679→        ),
  1680→  1680→        CmpKind::ULt => (
  1681→  1681→            Instruction::Cmpl {
  1682→  1682→                bf: CrField::CR0,
  1683→  1683→                l: 1,
  1684→  1684→                ra: lhs,
  1685→  1685→                rb: rhs,
  1686→  1686→            },
  1687→  1687→            0,
  1688→  1688→        ),
  1689→  1689→        CmpKind::ULe => (
  1690→  1690→            Instruction::Cmpl {
  1691→  1691→                bf: CrField::CR0,
  1692→  1692→                l: 1,
  1693→  1693→                ra: lhs,
  1694→  1694→                rb: rhs,
  1695→  1695→            },
  1696→  1696→            0,
  1697→  1697→        ),
  1698→  1698→        CmpKind::UGt => (
  1699→  1699→            Instruction::Cmpl {
  1700→  1700→                bf: CrField::CR0,
  1701→  1701→                l: 1,
  1702→  1702→                ra: lhs,
  1703→  1703→                rb: rhs,
  1704→  1704→            },
  1705→  1705→            1,
  1706→  1706→        ),
  1707→  1707→        CmpKind::UGe => (
  1708→  1708→            Instruction::Cmpl {
  1709→  1709→                bf: CrField::CR0,
  1710→  1710→                l: 1,
  1711→  1711→                ra: lhs,
  1712→  1712→                rb: rhs,
  1713→  1713→            },
  1714→  1714→            1,
  1715→  1715→        ),
  1716→  1716→    };
  1717→  1717→    result.push(emit_alloc_instr(
  1718→  1718→        cmp_inst.clone(),
  1719→  1719→        vec![
  1720→  1720→            PhysicalReg::new(RegClass::Gpr, lhs.encoding()),
  1721→  1721→            PhysicalReg::new(RegClass::Gpr, rhs.encoding()),
  1722→  1722→        ],
  1723→  1723→        vec![],
  1724→  1724→    ));
  1725→  1725→    // li dst, 0; then conditionally set to 1
  1726→  1726→    result.push(emit_alloc_instr(
  1727→  1727→        Instruction::Li { rt: dst, simm: 0 },
  1728→  1728→        vec![],
  1729→  1729→        vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1730→  1730→    ));
  1731→  1731→    let (bo, bi) = match kind {
  1732→  1732→        // Pattern: li dst,0; bc <skip li dst,1>; li dst,1
  1733→  1733→        // Branch is taken → dst stays 0; fall-through → dst becomes 1.
  1734→  1734→        // So we want branch taken when result should be 0.
  1735→  1735→        CmpKind::Eq => (4, 2 + CrField::CR0.encoding() * 4),  // branch if cr0.eq FALSE → dst=1 when eq
  1736→  1736→        CmpKind::Ne => (12, 2 + CrField::CR0.encoding() * 4), // branch if cr0.eq TRUE  → dst=0 when eq, dst=1 when ne
  1737→  1737→        CmpKind::SLt | CmpKind::ULt => (4, CrField::CR0.encoding() * 4),   // branch if cr0.lt FALSE
  1738→  1738→        CmpKind::SLe | CmpKind::ULe => (12, 1 + CrField::CR0.encoding() * 4), // branch if cr0.gt TRUE
  1739→  1739→        CmpKind::SGt | CmpKind::UGt => (4, 1 + CrField::CR0.encoding() * 4), // branch if cr0.gt FALSE
  1740→  1740→        CmpKind::SGe | CmpKind::UGe => (12, CrField::CR0.encoding() * 4),   // branch if cr0.lt TRUE
  1741→  1741→    };
  1742→  1742→    let _ = cond_bit;
  1743→  1743→    result.push(emit_alloc_instr(
  1744→  1744→        Instruction::Li { rt: dst, simm: 1 },
  1745→  1745→        vec![],
  1746→  1746→        vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1747→  1747→    ));
  1748→  1748→    // Patch: use a simplified approach — li dst,1 is always emitted but should be conditional.
  1749→  1749→    // For simplicity, use select-like pattern:
  1750→  1750→    // cmp; li dst, 0; bne cr0, +8; li dst, 1
  1751→  1751→    // Actually let's redo this more correctly:
  1752→  1752→    result.clear();
  1753→  1753→    result.push(emit_alloc_instr(
  1754→  1754→        cmp_inst,
  1755→  1755→        vec![
  1756→  1756→            PhysicalReg::new(RegClass::Gpr, lhs.encoding()),
  1757→  1757→            PhysicalReg::new(RegClass::Gpr, rhs.encoding()),
  1758→  1758→        ],
  1759→  1759→        vec![],
  1760→  1760→    ));
  1761→  1761→    result.push(emit_alloc_instr(
  1762→  1762→        Instruction::Li { rt: dst, simm: 0 },
  1763→  1763→        vec![],
  1764→  1764→        vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1765→  1765→    ));
  1766→  1766→    result.push(emit_alloc_instr(
  1767→  1767→        Instruction::Bc { bo, bi, bd: 2 },
  1768→  1768→        vec![],
  1769→  1769→        vec![],
  1770→  1770→    ));
  1771→  1771→    result.push(emit_alloc_instr(
  1772→  1772→        Instruction::Li { rt: dst, simm: 1 },
  1773→  1773→        vec![],
  1774→  1774→        vec![PhysicalReg::new(RegClass::Gpr, dst.encoding())],
  1775→  1775→    ));
  1776→  1776→    result
  1777→  1777→}

  1779→  1779→/// Lower an IR instruction to a sequence of PPC64 AllocatedInstructions,
  1780→  1780→/// plus any relocations that need to be patched later.
  1781→  1781→///
  1782→  1782→/// `alloc_offset` tracks the next free stack-slot offset (starting after the
  1783→  1783→/// 32-byte mandatory save area) so that `Alloc` instructions can compute
  1784→  1784→/// addresses within the already-allocated frame instead of double-decrementing
  1785→  1785→/// SP.
  1786→  1786→fn lower_ir_instr_ppc64(
  1787→  1787→    instr: &IRInstr,
  1788→  1788→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1789→  1789→    alloc_offset: &mut i32,

  1792→  1792→    let mut relocations = Vec::new();

  1795→  1795→        IRInstr::BinOp { op, dst, lhs, rhs, .. } => {
  1796→  1796→            let d = map_vreg_to_gpr(vreg_id(dst), None, vreg_map);

  1798→  1798→            let r = resolve_gpr_ppc64(rhs, vreg_map, Gpr::R11, &mut result);

  1809→  1809→                            PhysicalReg::new(RegClass::Gpr, r.encoding()),

  1812→  1812→                    ));

  1817→  1817→                            rt: d,
  1818→  1818→                            ra: r,
  1819→  1819→                            rb: l,
  1820→  1820→                        },
  1821→  1821→                        vec![
  1822→  1822→                            PhysicalReg::new(RegClass::Gpr, r.encoding()),
  1823→  1823→                            PhysicalReg::new(RegClass::Gpr, l.encoding()),
  1824→  1824→                        ],

  1829→  1829→                    result.push(emit_alloc_instr(
  1830→  1830→                        Instruction::Mulld {
  1831→  1831→                            rt: d,
  1832→  1832→                            ra: l,
  1833→  1833→                            rb: r,
  1834→  1834→                        },
  1835→  1835→                        vec![
  1836→  1836→                            PhysicalReg::new(RegClass::Gpr, l.encoding()),
  1837→  1837→                            PhysicalReg::new(RegClass::Gpr, r.encoding()),
  1838→  1838→                        ],
  1839→  1839→                        vec![PhysicalReg::new(RegClass::Gpr, d.encoding())],
  1840→  1840→                    ));
  1841→  1841→                }
  1842→  1842→                BinOpKind::SDiv => {
  1843→  1843→                    result.push(emit_alloc_instr(
  1844→  1844→                        Instruction::Divd {
  1845→  1845→                            rt: d,
  1846→  1846→                            ra: l,
  1847→  1847→                            rb: r,
  1848→  1848→                        },
  1849→  1849→                        vec![
  1850→  1850→                            PhysicalReg::new(RegClass::Gpr, l.encoding()),
  1851→  1851→                            PhysicalReg::new(RegClass::Gpr, r.encoding()),
  1852→  1852→                        ],
  1853→  1853→                        vec![PhysicalReg::new(RegClass::Gpr, d.encoding())],
  1854→  1854→                    ));
  1855→  1855→                }
  1856→  1856→                BinOpKind::UDiv => {
  1857→  1857→                    result.push(emit_alloc_instr(
  1858→  1858→                        Instruction::Divdu {
  1859→  1859→                            rt: d,
  1860→  1860→                            ra: l,
  1861→  1861→                            rb: r,
  1862→  1862→                        },
  1863→  1863→                        vec![
  1864→  1864→                            PhysicalReg::new(RegClass::Gpr, l.encoding()),
  1865→  1865→                            PhysicalReg::new(RegClass::Gpr, r.encoding()),
  1866→  1866→                        ],
  1867→  1867→                        vec![PhysicalReg::new(RegClass::Gpr, d.encoding())],
  1868→  1868→                    ));
  1869→  1869→                }
