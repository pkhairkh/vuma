     1→//! # PowerPC64 Mnemonic Disassembler
     2→//!
     3→//! Decodes PowerPC64 32-bit little-endian machine code into `Instruction`
     4→//! instances. Covers the D-form, X-form, XO-form, I-form, B-form, and
     5→//! XL-form instructions lowered by the VUMA ISel. Display is already
     6→//! provided by the parent module.
     7→
     8→use super::CrField;
     9→use super::Fpr;
    10→use super::Gpr;

    12→
    13→// ---------------------------------------------------------------------------
    14→// Decode error
    15→// ---------------------------------------------------------------------------
    16→
    17→/// Error produced when PPC64 decoding fails.
    18→#[derive(Debug, Clone, PartialEq, Eq)]
    19→pub enum DecodeError {
    20→    /// The byte slice is too short.
    21→    Truncated { needed: usize, available: usize },
    22→    /// The instruction encoding is not recognised.
    23→    UnknownEncoding { word: u32 },
    24→}
    25→
    26→impl std::fmt::Display for DecodeError {
    27→    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    28→        match self {
    29→            DecodeError::Truncated { needed, available } => {
    30→                write!(f, "truncated: need {needed} bytes, have {available}")
    31→            }
    32→            DecodeError::UnknownEncoding { word } => {
    33→                write!(f, "unknown PPC64 encoding: 0x{word:08x}")
    34→            }
    35→        }
    36→    }
    37→}
    38→
    39→impl std::error::Error for DecodeError {}

   121→fn cr_from_bits(bits: u32) -> CrField {
   122→    match bits {
   123→        0 => CrField::CR0,
   124→        1 => CrField::CR1,
   125→        2 => CrField::CR2,
   126→        3 => CrField::CR3,
   127→        4 => CrField::CR4,
   128→        5 => CrField::CR5,
   129→        6 => CrField::CR6,
   130→        7 => CrField::CR7,
   131→        _ => CrField::CR0,
   132→    }
   133→}

   135→fn sign_extend_16(val: u32) -> i32 {
   136→    if val & 0x8000 != 0 {
   137→        (val | 0xFFFF0000) as i32
   138→    } else {
   139→        val as i32
   140→    }
   141→}

   147→impl Instruction {
   148→    /// Decode a single PowerPC64 instruction from 4 little-endian bytes.
   149→    pub fn decode(bytes: &[u8]) -> Result<Self, DecodeError> {
   150→        if bytes.len() < 4 {
   151→            return Err(DecodeError::Truncated {
   152→                needed: 4,
   153→                available: bytes.len(),
   154→            });
   155→        }
   156→        let word = u32::from_be_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]);
   157→        let primary = (word >> 26) & 0x3F;
   158→        let rt = (word >> 21) & 0x1F;
   159→        let ra = (word >> 16) & 0x1F;
   160→        let rb = (word >> 11) & 0x1F;
   161→        let xo_xform = (word >> 1) & 0x3FF;
   162→        let rc = word & 1;
   163→        let d = sign_extend_16(word & 0xFFFF);
   164→        let ds_raw = sign_extend_16((word >> 2) & 0x3FFF) * 4; // DS field is ds/4
   165→
   166→        // D-form instructions
   167→        match primary {
   168→            // ADDI (primary=14)
   169→            14 => {
   170→                if ra == 0 {
   171→                    return Ok(Instruction::Li {
   172→                        rt: gpr_from_bits(rt),
   173→                        simm: d,
   174→                    });
   175→                }
   176→                return Ok(Instruction::Addi {
   177→                    rt: gpr_from_bits(rt),
   178→                    ra: gpr_from_bits(ra),
   179→                    simm: d,
   180→                });
   181→            }
   182→            // ADDIS (primary=15)
   183→            15 => {
   184→                if ra == 0 {
   185→                    return Ok(Instruction::Lis {
   186→                        rt: gpr_from_bits(rt),
   187→                        simm: d,
   188→                    });
   189→                }
   190→                return Ok(Instruction::Addis {
   191→                    rt: gpr_from_bits(rt),
   192→                    ra: gpr_from_bits(ra),
   193→                    simm: d,
   194→                });
   195→            }
   196→            // ORI (primary=24)
   197→            24 => {
   198→                let uimm = word & 0xFFFF;
   199→                // NOP: ori r0, r0, 0
   200→                if rt == 0 && ra == 0 && uimm == 0 {
   201→                    return Ok(Instruction::Nop);
   202→                }
   203→                return Ok(Instruction::Ori {
   204→                    ra: gpr_from_bits(ra),
   205→                    rs: gpr_from_bits(rt),
   206→                    uimm,
   207→                });
   208→            }
   209→            // XORI (primary=26)
   210→            26 => {
   211→                let uimm = word & 0xFFFF;
   212→                return Ok(Instruction::Xori {
   213→                    ra: gpr_from_bits(ra),
   214→                    rs: gpr_from_bits(rt),
   215→                    uimm,
   216→                });
   217→            }
   218→            // ANDI. (primary=28)
   219→            28 => {
   220→                let uimm = word & 0xFFFF;
   221→                return Ok(Instruction::Andi {
   222→                    ra: gpr_from_bits(ra),
   223→                    rs: gpr_from_bits(rt),
   224→                    uimm,
   225→                });
   226→            }
   227→            // CMPI (primary=11)
   228→            11 => {
   229→                let bf = (word >> 23) & 0x7;
   230→                let l = (word >> 22) & 1; // l field at MSB-first bit 9 = normal bit 22
   231→                return Ok(Instruction::Cmpi {
   232→                    bf: cr_from_bits(bf),
   233→                    l,
   234→                    ra: gpr_from_bits(ra),
   235→                    simm: d,
   236→                });
   237→            }
   238→            // CMPLI (primary=10)
   239→            10 => {
   240→                let bf = (word >> 23) & 0x7;
   241→                let l = (word >> 22) & 1; // l field at MSB-first bit 9 = normal bit 22
   242→                let uimm = word & 0xFFFF;
   243→                return Ok(Instruction::Cmpli {
   244→                    bf: cr_from_bits(bf),
   245→                    l,
   246→                    ra: gpr_from_bits(ra),
   247→                    uimm,
   248→                });
   249→            }
   250→
   251→            // Load/store D-form
   252→            // LD (primary=58, DS-form)
   253→            58 => {
   254→                let xo = word & 0x3;
   255→                match xo {
   256→                    0 => {
   257→                        return Ok(Instruction::Ld {
   258→                            rt: gpr_from_bits(rt),
   259→                            ra: gpr_from_bits(ra),
   260→                            ds: ds_raw,
   261→                        })
   262→                    }
   263→                    2 => {
   264→                        return Ok(Instruction::Lwa {
   265→                            rt: gpr_from_bits(rt),
   266→                            ra: gpr_from_bits(ra),
   267→                            ds: ds_raw,
   268→                        })
   269→                    }
   270→                    _ => {}
   271→                }
   272→            }
   273→            // LWZ (primary=32)
   274→            32 => {
   275→                return Ok(Instruction::Lwz {
   276→                    rt: gpr_from_bits(rt),
   277→                    ra: gpr_from_bits(ra),
   278→                    d,
   279→                })
   280→            }
   281→            // LWZU (primary=33)
   282→            33 => {
   283→                return Ok(Instruction::Lwzu {
   284→                    rt: gpr_from_bits(rt),
   285→                    ra: gpr_from_bits(ra),
   286→                    d,
   287→                })
   288→            }
   289→            // LBZ (primary=34)
   290→            34 => {
   291→                return Ok(Instruction::Lbz {
   292→                    rt: gpr_from_bits(rt),
   293→                    ra: gpr_from_bits(ra),
   294→                    d,
   295→                })
   296→            }
   297→            // STW (primary=36)
   298→            36 => {
   299→                return Ok(Instruction::Stw {
   300→                    rs: gpr_from_bits(rt),
   301→                    ra: gpr_from_bits(ra),
   302→                    d,
   303→                })
   304→            }
   305→            // STWU (primary=37)
   306→            37 => {
   307→                return Ok(Instruction::Stwu {
   308→                    rs: gpr_from_bits(rt),
   309→                    ra: gpr_from_bits(ra),
   310→                    d,
   311→                })
   312→            }
   313→            // STB (primary=38)
   314→            38 => {
   315→                return Ok(Instruction::Stb {
   316→                    rs: gpr_from_bits(rt),
   317→                    ra: gpr_from_bits(ra),
   318→                    d,
   319→                })
   320→            }
   321→            // LHZ (primary=40)
   322→            40 => {
   323→                return Ok(Instruction::Lhz {
   324→                    rt: gpr_from_bits(rt),
   325→                    ra: gpr_from_bits(ra),
   326→                    d,
   327→                })
   328→            }
   329→            // STH (primary=44)
   330→            44 => {
   331→                return Ok(Instruction::Sth {
   332→                    rs: gpr_from_bits(rt),
   333→                    ra: gpr_from_bits(ra),
   334→                    d,
   335→                })
   336→            }
   337→            // STD (primary=62, DS-form)
   338→            62 => {
   339→                let xo = word & 0x3;
   340→                match xo {
   341→                    0 => {
   342→                        return Ok(Instruction::Std {
   343→                            rs: gpr_from_bits(rt),
   344→                            ra: gpr_from_bits(ra),
   345→                            ds: ds_raw,
   346→                        })
   347→                    }
   348→                    1 => {
   349→                        return Ok(Instruction::Stdu {
   350→                            rs: gpr_from_bits(rt),
   351→                            ra: gpr_from_bits(ra),
   352→                            ds: ds_raw,
   353→                        })
   354→                    }
   355→                    _ => {}
   356→                }
   357→            }
   358→            // FP load/store D-form
   359→            // LFS (primary=48)
   360→            48 => {
   361→                return Ok(Instruction::Lfs {
   362→                    ft: fpr_from_bits(rt),
   363→                    ra: gpr_from_bits(ra),
   364→                    d,
   365→                })
   366→            }
   367→            // LFD (primary=50)
   368→            50 => {
   369→                return Ok(Instruction::Lfd {
   370→                    ft: fpr_from_bits(rt),
   371→                    ra: gpr_from_bits(ra),
   372→                    d,
   373→                })
   374→            }
   375→            // STFS (primary=52)
   376→            52 => {
   377→                return Ok(Instruction::Stfs {
   378→                    fs: fpr_from_bits(rt),
   379→                    ra: gpr_from_bits(ra),
   380→                    d,
   381→                })
   382→            }
   383→            // STFD (primary=54)
   384→            54 => {
   385→                return Ok(Instruction::Stfd {
   386→                    fs: fpr_from_bits(rt),
   387→                    ra: gpr_from_bits(ra),
   388→                    d,
   389→                })
   390→            }
   391→
   392→            // I-form branches
   393→            // B/BL/BA/BLA (primary=18)
   394→            18 => {
   395→                let li_raw = (word >> 2) & 0x3FFFFFF;
   396→                let li = sign_extend_26(li_raw);
   397→                let aa = (word >> 1) & 1;
   398→                let lk = word & 1;
   399→                match (aa, lk) {
   400→                    (0, 0) => return Ok(Instruction::B { li }),
   401→                    (1, 0) => return Ok(Instruction::Ba { li }),
   402→                    (0, 1) => return Ok(Instruction::Bl { li }),
   403→                    (1, 1) => return Ok(Instruction::Bla { li }),
   404→                    _ => unreachable!(),
   405→                }
   406→            }
   407→
   408→            // SC (primary=17)
   409→            17 => return Ok(Instruction::Sc),
   410→
   411→            // M-form: RLWINM (primary=21), RLWIMI (primary=20)
   412→            21 => {
   413→                let sh = (word >> 11) & 0x1F;
   414→                let mb = (word >> 6) & 0x1F;
   415→                let me = (word >> 1) & 0x1F;
   416→                return Ok(Instruction::Rlwinm {
   417→                    ra: gpr_from_bits(ra),
   418→                    rs: gpr_from_bits(rt),
   419→                    sh,
   420→                    mb,
   421→                    me,
   422→                });
   423→            }
   424→            20 => {
   425→                let sh = (word >> 11) & 0x1F;
   426→                let mb = (word >> 6) & 0x1F;
   427→                let me = (word >> 1) & 0x1F;
   428→                return Ok(Instruction::Rlwimi {
   429→                    ra: gpr_from_bits(ra),
   430→                    rs: gpr_from_bits(rt),
   431→                    sh,
   432→                    mb,
   433→                    me,
   434→                });
   435→            }
   436→
   437→            _ => {}
   438→        }
   439→
   440→        // X-form and XO-form (primary=31)
   441→        if primary == 31 {
   442→            // Check for trap first: TW 31, r0, r0 → word = 0x7FE00008
   443→            if word == 0x7FE0_0008 {
   444→                return Ok(Instruction::Trap);
   445→            }
   446→
   447→            match xo_xform {
   448→                // ADD (xo=266)
   449→                266 => {
   450→                    return Ok(Instruction::Add {
   451→                        rt: gpr_from_bits(rt),
   452→                        ra: gpr_from_bits(ra),
   453→                        rb: gpr_from_bits(rb),
   454→                    })
   455→                }
   456→                // SUBF (xo=40)
   457→                40 => {
   458→                    return Ok(Instruction::Subf {
   459→                        rt: gpr_from_bits(rt),
   460→                        ra: gpr_from_bits(ra),
   461→                        rb: gpr_from_bits(rb),
   462→                    })
   463→                }
   464→                // MULLW (xo=235)
   465→                235 => {
   466→                    return Ok(Instruction::Mullw {
   467→                        rt: gpr_from_bits(rt),
   468→                        ra: gpr_from_bits(ra),
   469→                        rb: gpr_from_bits(rb),
   470→                    })
   471→                }
   472→                // MULLD (xo=233)
   473→                233 => {
   474→                    return Ok(Instruction::Mulld {
   475→                        rt: gpr_from_bits(rt),
   476→                        ra: gpr_from_bits(ra),
   477→                        rb: gpr_from_bits(rb),
   478→                    })
   479→                }
   480→                // MULHW (xo=75)
   481→                75 => {
   482→                    return Ok(Instruction::Mulhw {
   483→                        rt: gpr_from_bits(rt),
   484→                        ra: gpr_from_bits(ra),
   485→                        rb: gpr_from_bits(rb),
   486→                    })
   487→                }
   488→                // MULHD (xo=73)
   489→                73 => {
   490→                    return Ok(Instruction::Mulhd {
   491→                        rt: gpr_from_bits(rt),
   492→                        ra: gpr_from_bits(ra),
   493→                        rb: gpr_from_bits(rb),
   494→                    })
   495→                }
   496→                // DIVW (xo=491)
   497→                491 => {
   498→                    return Ok(Instruction::Divw {
   499→                        rt: gpr_from_bits(rt),
   500→                        ra: gpr_from_bits(ra),
   501→                        rb: gpr_from_bits(rb),
   502→                    })
   503→                }
   504→                // DIVD (xo=459)
   505→                459 => {
   506→                    return Ok(Instruction::Divd {
   507→                        rt: gpr_from_bits(rt),
   508→                        ra: gpr_from_bits(ra),
   509→                        rb: gpr_from_bits(rb),
   510→                    })
   511→                }
   512→                // NEG (xo=104)
   513→                104 => {
   514→                    return Ok(Instruction::Neg {
   515→                        rt: gpr_from_bits(rt),
   516→                        ra: gpr_from_bits(ra),
   517→                    })
   518→                }
   519→
   520→                // AND (xo=28)
   521→                28 => {
   522→                    if ra == rt && rb == rt && rc == 0 {
   523→                        // MR pseudo: and ra, rs, rs
   524→                        return Ok(Instruction::Mr {
   525→                            ra: gpr_from_bits(ra),
   526→                            rs: gpr_from_bits(rt),
   527→                        });
   528→                    }
   529→                    return Ok(Instruction::And {
   530→                        ra: gpr_from_bits(ra),
   531→                        rs: gpr_from_bits(rt),
   532→                        rb: gpr_from_bits(rb),
   533→                    });
   534→                }
   535→                // OR (xo=444)
   536→                444 => {
   537→                    if rb == rt && rc == 0 {
   538→                        return Ok(Instruction::Mr {
   539→                            ra: gpr_from_bits(ra),
   540→                            rs: gpr_from_bits(rt),
   541→                        });
   542→                    }
   543→                    return Ok(Instruction::Or {
   544→                        ra: gpr_from_bits(ra),
   545→                        rs: gpr_from_bits(rt),
   546→                        rb: gpr_from_bits(rb),
   547→                    });
   548→                }
   549→                // XOR (xo=316)
   550→                316 => {
   551→                    return Ok(Instruction::Xor {
   552→                        ra: gpr_from_bits(ra),
   553→                        rs: gpr_from_bits(rt),
   554→                        rb: gpr_from_bits(rb),
   555→                    })
   556→                }
   557→                // NOR (xo=124)
   558→                124 => {
   559→                    return Ok(Instruction::Nor {
   560→                        ra: gpr_from_bits(ra),
   561→                        rs: gpr_from_bits(rt),
   562→                        rb: gpr_from_bits(rb),
   563→                    })
   564→                }
   565→                // ANDC (xo=60)
   566→                60 => {
   567→                    return Ok(Instruction::Andc {
   568→                        ra: gpr_from_bits(ra),
   569→                        rs: gpr_from_bits(rt),
   570→                        rb: gpr_from_bits(rb),
   571→                    })
   572→                }
   573→                // ORC (xo=412)
   574→                412 => {
   575→                    return Ok(Instruction::Orc {
   576→                        ra: gpr_from_bits(ra),
   577→                        rs: gpr_from_bits(rt),
   578→                        rb: gpr_from_bits(rb),
   579→                    })
   580→                }
   581→                // EQV (xo=284)
   582→                284 => {
   583→                    return Ok(Instruction::Eqv {
   584→                        ra: gpr_from_bits(ra),
   585→                        rs: gpr_from_bits(rt),
   586→                        rb: gpr_from_bits(rb),
   587→                    })
   588→                }
   589→
   590→                // Shifts
   591→                // SLD (xo=27)
   592→                27 => {
   593→                    return Ok(Instruction::Sld {
   594→                        ra: gpr_from_bits(ra),
   595→                        rs: gpr_from_bits(rt),
   596→                        rb: gpr_from_bits(rb),
   597→                    })
   598→                }
   599→                // SRD (xo=539)
   600→                539 => {
   601→                    return Ok(Instruction::Srd {
   602→                        ra: gpr_from_bits(ra),
   603→                        rs: gpr_from_bits(rt),
   604→                        rb: gpr_from_bits(rb),
   605→                    })
   606→                }
   607→                // SRAD (xo=794)
   608→                794 => {
   609→                    return Ok(Instruction::Srad {
   610→                        ra: gpr_from_bits(ra),
   611→                        rs: gpr_from_bits(rt),
   612→                        rb: gpr_from_bits(rb),
   613→                    })
   614→                }
   615→                // SLW (xo=24)
   616→                24 => {
   617→                    return Ok(Instruction::Slw {
   618→                        ra: gpr_from_bits(ra),
   619→                        rs: gpr_from_bits(rt),
   620→                        rb: gpr_from_bits(rb),
   621→                    })
   622→                }
   623→                // SRW (xo=536)
   624→                536 => {
   625→                    return Ok(Instruction::Srw {
   626→                        ra: gpr_from_bits(ra),
   627→                        rs: gpr_from_bits(rt),
   628→                        rb: gpr_from_bits(rb),
   629→                    })
   630→                }
   631→                // SRAW (xo=792)
   632→                792 => {
   633→                    return Ok(Instruction::Sraw {
   634→                        ra: gpr_from_bits(ra),
   635→                        rs: gpr_from_bits(rt),
   636→                        rb: gpr_from_bits(rb),
   637→                    })
   638→                }
   639→
   640→                // CNTLZD (xo=58)
   641→                58 => {
   642→                    return Ok(Instruction::Cntlzd {
   643→                        ra: gpr_from_bits(ra),
   644→                        rs: gpr_from_bits(rt),
   645→                    })
   646→                }
   647→                // POPCNTD (xo=506)
   648→                506 => {
   649→                    return Ok(Instruction::Popcntd {
   650→                        ra: gpr_from_bits(ra),
   651→                        rs: gpr_from_bits(rt),
   652→                    })
   653→                }
   654→                // EXTSW (xo=986)
   655→                986 => {
   656→                    return Ok(Instruction::Extsw {
   657→                        ra: gpr_from_bits(ra),
   658→                        rs: gpr_from_bits(rt),
   659→                    })
   660→                }
   661→                // EXTSB (xo=954)
   662→                954 => {
   663→                    return Ok(Instruction::Extsb {
   664→                        ra: gpr_from_bits(ra),
   665→                        rs: gpr_from_bits(rt),
   666→                    })
   667→                }
   668→                // EXTSH (xo=922)
   669→                922 => {
   670→                    return Ok(Instruction::Extsh {
   671→                        ra: gpr_from_bits(ra),
   672→                        rs: gpr_from_bits(rt),
   673→                    })
   674→                }
   675→
   676→                // LDARX (xo=84)
   677→                84 => {
   678→                    return Ok(Instruction::Ldarx {
   679→                        rt: gpr_from_bits(rt),
   680→                        ra: gpr_from_bits(ra),
   681→                        rb: gpr_from_bits(rb),
   682→                    })
   683→                }
   684→                // LWARX (xo=20)
   685→                20 => {
   686→                    return Ok(Instruction::Lwarx {
   687→                        rt: gpr_from_bits(rt),
   688→                        ra: gpr_from_bits(ra),
   689→                        rb: gpr_from_bits(rb),
   690→                    })
   691→                }
   692→                // LBARX (xo=52)
   693→                52 => {
   694→                    return Ok(Instruction::Lbarx {
   695→                        rt: gpr_from_bits(rt),
   696→                        ra: gpr_from_bits(ra),
   697→                        rb: gpr_from_bits(rb),
   698→                    })
   699→                }
   700→                // LHARX (xo=116)
   701→                116 => {
   702→                    return Ok(Instruction::Lharx {
   703→                        rt: gpr_from_bits(rt),
   704→                        ra: gpr_from_bits(ra),
   705→                        rb: gpr_from_bits(rb),
   706→                    })
   707→                }
   708→                // STDCX. (xo=214, Rc=1)
   709→                214 => {
   710→                    return Ok(Instruction::Stdcx {
   711→                        rs: gpr_from_bits(rt),
   712→                        ra: gpr_from_bits(ra),
   713→                        rb: gpr_from_bits(rb),
   714→                    })
   715→                }
   716→                // STWCX. (xo=150, Rc=1)
   717→                150 => {
   718→                    return Ok(Instruction::Stwcx {
   719→                        rs: gpr_from_bits(rt),
   720→                        ra: gpr_from_bits(ra),
   721→                        rb: gpr_from_bits(rb),
   722→                    })
   723→                }
   724→                // STBCX. (xo=694, Rc=1)
   725→                694 => {
   726→                    return Ok(Instruction::Stbcx {
   727→                        rs: gpr_from_bits(rt),
   728→                        ra: gpr_from_bits(ra),
   729→                        rb: gpr_from_bits(rb),
   730→                    })
   731→                }
   732→                // STHCX. (xo=726, Rc=1)
   733→                726 => {
   734→                    return Ok(Instruction::Sthcx {
   735→                        rs: gpr_from_bits(rt),
   736→                        ra: gpr_from_bits(ra),
   737→                        rb: gpr_from_bits(rb),
   738→                    })
   739→                }
   740→                // SYNC / LWSYNC (xo=598)
   741→                598 => {
   742→                    // The L field (sync variant) is encoded in the rt position:
   743→                    // rt=0 → heavyweight sync, rt=1 → lwsync
   744→                    if rt == 0 {
   745→                        return Ok(Instruction::Sync);
   746→                    } else {
   747→                        return Ok(Instruction::Lwsync);
   748→                    }
   749→                }
   750→
   751→                // CMP (xo=0)
   752→                0 => {
   753→                    let bf = (word >> 23) & 0x7;
   754→                    let l = (word >> 22) & 1; // l field at MSB-first bit 9 = normal bit 22
   755→                    return Ok(Instruction::Cmp {
   756→                        bf: cr_from_bits(bf),
   757→                        l,
   758→                        ra: gpr_from_bits(ra),
   759→                        rb: gpr_from_bits(rb),
   760→                    });
   761→                }
   762→                // CMPL (xo=32)
   763→                32 => {
   764→                    let bf = (word >> 23) & 0x7;
   765→                    let l = (word >> 22) & 1; // l field at MSB-first bit 9 = normal bit 22
   766→                    return Ok(Instruction::Cmpl {
   767→                        bf: cr_from_bits(bf),
   768→                        l,
   769→                        ra: gpr_from_bits(ra),
   770→                        rb: gpr_from_bits(rb),
   771→                    });
   772→                }
   773→
   774→                _ => {}
   775→            }
   776→        }
   777→
   778→        // B-form (primary=16)
   779→        if primary == 16 {
   780→            let bo = (word >> 21) & 0x1F;
   781→            let bi = (word >> 16) & 0x1F;
   782→            let bd = sign_extend_14((word >> 2) & 0x3FFF);
   783→            let aa = (word >> 1) & 1;
   784→            let lk = word & 1;
   785→            match (aa, lk) {
   786→                (0, 0) => return Ok(Instruction::Bc { bo, bi, bd }),
   787→                (1, 0) => return Ok(Instruction::Bca { bo, bi, bd }),
   788→                _ => {}
   789→            }
   790→        }
   791→
   792→        // MD-form (primary=30) — RLDCL, RLDCR
   793→        if primary == 30 {
   794→            let rs = (word >> 21) & 0x1F;
   795→            let ra = (word >> 16) & 0x1F;
   796→            let rb = (word >> 11) & 0x1F;
   797→            let mb_lo = (word >> 6) & 0x1F;  // MB[0:4]
   798→            let mb5 = (word >> 5) & 1;       // MB[5]
   799→            let mb = mb_lo | (mb5 << 5);
   800→            let me_lo = mb_lo; // same field positions for ME
   801→            let me5 = mb5;
   802→            let me = me_lo | (me5 << 5);
   803→            let xo_md = (word >> 1) & 0xF; // 4-bit XO in MD-form
   804→            match xo_md {
   805→                8 => return Ok(Instruction::Rldcl {
   806→                    ra: gpr_from_bits(ra),
   807→                    rs: gpr_from_bits(rs),
   808→                    rb: gpr_from_bits(rb),
   809→                    mb,
   810→                }),
   811→                9 => return Ok(Instruction::Rldcr {
   812→                    ra: gpr_from_bits(ra),
   813→                    rs: gpr_from_bits(rs),
   814→                    rb: gpr_from_bits(rb),
   815→                    me,
   816→                }),
   817→                _ => {}
   818→            }
   819→        }
   820→
   821→        // XL-form (primary=19)
   822→        if primary == 19 {
   823→            let bo = (word >> 21) & 0x1F;
   824→            let bi = (word >> 16) & 0x1F;
   825→            let bh = (word >> 10) & 0x7; // BH at MSB-first bits [19:21] = normal shift 10
   826→            match xo_xform {
   827→                16 => return Ok(Instruction::Bclr { bo, bi, bh }),
   828→                150 => return Ok(Instruction::Isync), // ISYNC: primary=19, xo=150
   829→                528 => return Ok(Instruction::Bcctr { bo, bi, bh }),
   830→                560 => return Ok(Instruction::Bctar { bo, bi, bh }),
   831→                _ => {}
   832→            }
   833→        }
   834→
   835→        Err(DecodeError::UnknownEncoding { word })
   836→    }
   837→}
   838→
   839→fn sign_extend_14(val: u32) -> i32 {
   840→    if val & 0x2000 != 0 {
   841→        (val | 0xFFFFC000) as i32
   842→    } else {
   843→        val as i32
   844→    }
   845→}
   846→
   847→fn sign_extend_26(val: u32) -> i32 {
   848→    if val & 0x2000000 != 0 {
   849→        (val | 0xFC000000) as i32
   850→    } else {
   851→        val as i32
   852→    }
   853→}
