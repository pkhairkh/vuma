     1→     1→//! # WebAssembly 32-bit Backend
     2→     2→//!
     3→     3→//! Implements the `Backend` trait for the Wasm32 target.  WebAssembly is
     4→     4→//! fundamentally different from register-based ISAs: it is a stack machine
     5→     5→//! with structured control flow and a binary format.  This module provides:
     6→     6→//!
     7→     7→//! - `WasmType` — value types (i32, i64, f32, f64)
     8→     8→//! - `WasmInstr` — exhaustive instruction enum covering all ops needed for VUMA IR
     9→     9→//! - LEB128 encoding/decoding (unsigned and signed)
    10→    10→//! - Wasm binary-format encoder that emits complete `.wasm` modules
    11→    11→//! - `Wasm32Backend` — `Backend` implementation that lowers IR to Wasm bytecode
    12→    12→

    24→    24→// ===========================================================================
    25→    25→// Wasm value types
    26→    26→// ===========================================================================
    27→    27→
    28→    28→/// WebAssembly value types.
    29→    29→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
    30→    30→pub enum WasmType {
    31→    31→    I32,
    32→    32→    I64,
    33→    33→    F32,
    34→    34→    F64,
    35→    35→}
    36→    36→
    37→    37→impl WasmType {
    38→    38→    /// Binary encoding of this type in the Wasm format.
    39→    39→    pub fn to_byte(&self) -> u8 {
    40→    40→        match self {
    41→    41→            WasmType::I32 => 0x7F,
    42→    42→            WasmType::I64 => 0x7E,
    43→    43→            WasmType::F32 => 0x7D,
    44→    44→            WasmType::F64 => 0x7C,
    45→    45→        }
    46→    46→    }
    47→    47→
    48→    48→    /// Size in bytes of this Wasm type on the stack.
    49→    49→    pub fn byte_size(&self) -> usize {
    50→    50→        match self {
    51→    51→            WasmType::I32 | WasmType::F32 => 4,
    52→    52→            WasmType::I64 | WasmType::F64 => 8,
    53→    53→        }
    54→    54→    }
    55→    55→
    56→    56→    /// Whether this is an integer type.
    57→    57→    pub fn is_integer(&self) -> bool {
    58→    58→        matches!(self, WasmType::I32 | WasmType::I64)
    59→    59→    }
    60→    60→
    61→    61→    /// Whether this is a floating-point type.
    62→    62→    pub fn is_float(&self) -> bool {
    63→    63→        matches!(self, WasmType::F32 | WasmType::F64)
    64→    64→    }
    65→    65→
    66→    66→    /// Map an IRType to the corresponding WasmType.
    67→    67→    pub fn from_ir_type(ty: &IRType) -> Option<WasmType> {
    68→    68→        match ty {
    69→    69→            IRType::I8
    70→    70→            | IRType::I16
    71→    71→            | IRType::I32
    72→    72→            | IRType::U8
    73→    73→            | IRType::U16
    74→    74→            | IRType::U32
    75→    75→            | IRType::Ptr
    76→    76→            | IRType::Func => Some(WasmType::I32),
    77→    77→            IRType::I64 | IRType::U64 => Some(WasmType::I64),
    78→    78→            IRType::F32 => Some(WasmType::F32),
    79→    79→            IRType::F64 => Some(WasmType::F64),
    80→    80→            IRType::Void | IRType::Struct { .. } | IRType::Array { .. } => None,
    81→    81→        }
    82→    82→    }
    83→    83→
    84→    84→    /// Decode a WasmType from its binary encoding byte.
    85→    85→    pub fn from_byte(byte: u8) -> Option<WasmType> {
    86→    86→        match byte {
    87→    87→            0x7F => Some(WasmType::I32),
    88→    88→            0x7E => Some(WasmType::I64),
    89→    89→            0x7D => Some(WasmType::F32),
    90→    90→            0x7C => Some(WasmType::F64),
    91→    91→            _ => None,
    92→    92→        }
    93→    93→    }
    94→    94→}

   102→   102→            WasmType::F64 => write!(f, "f64"),
   103→   103→        }
   104→   104→    }
   105→   105→}

   400→   400→impl WasmInstr {
   401→   401→    /// Encode this instruction into Wasm bytecode, appending bytes to `out`.
   402→   402→    pub fn encode(&self, out: &mut Vec<u8>) {
   403→   403→        match self {
   404→   404→            // ── Control ──────────────────────────────────────────────
   405→   405→            WasmInstr::Block(ty) => {
   406→   406→                out.push(0x02);
   407→   407→                match ty {
   408→   408→                    Some(t) => out.push(t.to_byte()),
   409→   409→                    None => out.push(0x40), // void block type
   410→   410→                }
   411→   411→            }
   412→   412→            WasmInstr::Loop(ty) => {
   413→   413→                out.push(0x03);
   414→   414→                match ty {
   415→   415→                    Some(t) => out.push(t.to_byte()),
   416→   416→                    None => out.push(0x40), // void loop type
   417→   417→                }
   418→   418→            }
   419→   419→            WasmInstr::If(ty) => {
   420→   420→                out.push(0x04);
   421→   421→                match ty {
   422→   422→                    Some(t) => out.push(t.to_byte()),
   423→   423→                    None => out.push(0x40), // void if type
   424→   424→                }
   425→   425→            }
   426→   426→            WasmInstr::Else => out.push(0x05),
   427→   427→            WasmInstr::End => out.push(0x0B),
   428→   428→            WasmInstr::Br(idx) => {
   429→   429→                out.push(0x0C);
   430→   430→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   431→   431→            }
   432→   432→            WasmInstr::BrIf(idx) => {
   433→   433→                out.push(0x0D);
   434→   434→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   435→   435→            }
   436→   436→            WasmInstr::BrTable { labels, default } => {
   437→   437→                out.push(0x0E);
   438→   438→                out.extend_from_slice(&encode_unsigned_leb128(labels.len() as u64));
   439→   439→                for &lbl in labels {
   440→   440→                    out.extend_from_slice(&encode_unsigned_leb128(lbl as u64));
   441→   441→                }
   442→   442→                out.extend_from_slice(&encode_unsigned_leb128(*default as u64));
   443→   443→            }
   444→   444→            WasmInstr::Return => out.push(0x0F),
   445→   445→            WasmInstr::Call(idx) => {
   446→   446→                out.push(0x10);
   447→   447→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   448→   448→            }
   449→   449→            WasmInstr::CallIndirect {
   450→   450→                type_idx,
   451→   451→                table_idx,
   452→   452→            } => {
   453→   453→                out.push(0x11);
   454→   454→                out.extend_from_slice(&encode_unsigned_leb128(*type_idx as u64));
   455→   455→                out.extend_from_slice(&encode_unsigned_leb128(*table_idx as u64));
   456→   456→            }
   457→   457→
   458→   458→            // ── Parametric ──────────────────────────────────────────
   459→   459→            WasmInstr::Select => out.push(0x1B),
   460→   460→            WasmInstr::Drop => out.push(0x1A),
   461→   461→
   462→   462→            // ── Variable ────────────────────────────────────────────
   463→   463→            WasmInstr::LocalGet(idx) => {
   464→   464→                out.push(0x20);
   465→   465→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   466→   466→            }
   467→   467→            WasmInstr::LocalSet(idx) => {
   468→   468→                out.push(0x21);
   469→   469→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   470→   470→            }
   471→   471→            WasmInstr::LocalTee(idx) => {
   472→   472→                out.push(0x22);
   473→   473→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   474→   474→            }
   475→   475→            WasmInstr::GlobalGet(idx) => {
   476→   476→                out.push(0x23);
   477→   477→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   478→   478→            }
   479→   479→            WasmInstr::GlobalSet(idx) => {
   480→   480→                out.push(0x24);
   481→   481→                out.extend_from_slice(&encode_unsigned_leb128(*idx as u64));
   482→   482→            }
   483→   483→
   484→   484→            // ── Memory ──────────────────────────────────────────────
   485→   485→            WasmInstr::I32Load { align, offset } => {
   486→   486→                out.push(0x28);
   487→   487→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   488→   488→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   489→   489→            }
   490→   490→            WasmInstr::I64Load { align, offset } => {
   491→   491→                out.push(0x29);
   492→   492→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   493→   493→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   494→   494→            }
   495→   495→            WasmInstr::F32Load { align, offset } => {
   496→   496→                out.push(0x2A);
   497→   497→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   498→   498→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   499→   499→            }
   500→   500→            WasmInstr::F64Load { align, offset } => {
   501→   501→                out.push(0x2B);
   502→   502→                out.extend_from_slice(&encode_unsigned_leb128(*align as u64));
   503→   503→                out.extend_from_slice(&encode_unsigned_leb128(*offset as u64));
   504→   504→            }
   505→   505→            WasmInstr::I32Store { align, offset } => {

   804→   804→/// Encode an unsigned 64-bit value as unsigned LEB128.
   805→   805→pub fn encode_unsigned_leb128(value: u64) -> Vec<u8> {
   806→   806→    let mut result = Vec::new();
   807→   807→    let mut val = value;
   808→   808→    loop {
   809→   809→        let mut byte = (val & 0x7F) as u8;
   810→   810→        val >>= 7;
   811→   811→        if val != 0 {
   812→   812→            byte |= 0x80;
   813→   813→        }
   814→   814→        result.push(byte);
   815→   815→        if val == 0 {
   816→   816→            break;
   817→   817→        }
   818→   818→    }
   819→   819→    result
   820→   820→}

   822→   822→/// Encode a signed 64-bit value as signed LEB128.
   823→   823→pub fn encode_signed_leb128(value: i64) -> Vec<u8> {
   824→   824→    let mut result = Vec::new();
   825→   825→    let mut val = value;
   826→   826→    loop {
   827→   827→        let mut byte = (val & 0x7F) as u8;
   828→   828→        val >>= 7;
   829→   829→        // Sign bit of byte is bit 6
   830→   830→        let sign_bit = (byte & 0x40) != 0;
   831→   831→        // If val is 0 and sign bit is clear, or val is -1 and sign bit is set, we're done
   832→   832→        if (val == 0 && !sign_bit) || (val == -1 && sign_bit) {
   833→   833→            result.push(byte);
   834→   834→            break;
   835→   835→        }
   836→   836→        byte |= 0x80;
   837→   837→        result.push(byte);
   838→   838→    }
   839→   839→    result
   840→   840→}

   842→   842→/// Decode an unsigned LEB128 value from a byte slice.
   843→   843→/// Returns (decoded_value, number_of_bytes_consumed).
   844→   844→pub fn decode_unsigned_leb128(bytes: &[u8]) -> (u64, usize) {
   845→   845→    let mut result: u64 = 0;
   846→   846→    let mut shift: u32 = 0;
   847→   847→    let mut count = 0;
   848→   848→    for &byte in bytes {
   849→   849→        count += 1;
   850→   850→        result |= ((byte & 0x7F) as u64) << shift;
   851→   851→        if (byte & 0x80) == 0 {
   852→   852→            break;
   853→   853→        }
   854→   854→        shift += 7;
   855→   855→    }
   856→   856→    (result, count)
   857→   857→}

   859→   859→/// Decode a signed LEB128 value from a byte slice.
   860→   860→/// Returns (decoded_value, number_of_bytes_consumed).
   861→   861→pub fn decode_signed_leb128(bytes: &[u8]) -> (i64, usize) {
   862→   862→    let mut result: i64 = 0;
   863→   863→    let mut shift: u32 = 0;
   864→   864→    let mut count = 0;
   865→   865→    let mut byte: u8;
   866→   866→    loop {
   867→   867→        byte = bytes[count];
   868→   868→        count += 1;
   869→   869→        result |= ((byte & 0x7F) as i64) << shift;
   870→   870→        shift += 7;
   871→   871→        if (byte & 0x80) == 0 {
   872→   872→            break;
   873→   873→        }
   874→   874→    }
   875→   875→    // Sign-extend if the sign bit is set
   876→   876→    if shift < 64 && (byte & 0x40) != 0 {
   877→   877→        result |= !0i64 << shift;
   878→   878→    }
   879→   879→    (result, count)
   880→   880→}

   903→   903→/// A Wasm function type signature: (params) -> (results).
   904→   904→///
   905→   905→/// Supports multi-value returns (Wasm 2.0 feature): the `results` vector
   906→   906→/// may contain more than one type.
   907→   907→#[derive(Debug, Clone, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
   908→   908→pub struct WasmFuncType {
   909→   909→    pub params: Vec<WasmType>,
   910→   910→    pub results: Vec<WasmType>,
   911→   911→}

   913→   913→impl WasmFuncType {
   914→   914→    /// Encode this function type in Wasm binary format.
   915→   915→    pub fn encode(&self) -> Vec<u8> {
   916→   916→        let mut out = Vec::new();
   917→   917→        out.push(0x60); // func type tag
   918→   918→        out.extend_from_slice(&encode_unsigned_leb128(self.params.len() as u64));
   919→   919→        for p in &self.params {
   920→   920→            out.push(p.to_byte());
   921→   921→        }
   922→   922→        out.extend_from_slice(&encode_unsigned_leb128(self.results.len() as u64));
   923→   923→        for r in &self.results {
   924→   924→            out.push(r.to_byte());
   925→   925→        }
   926→   926→        out
   927→   927→    }
   928→   928→}

  1147→  1147→
  1148→  1148→    /// Encode the complete module into bytes.
  1149→  1149→    pub fn encode(self) -> Vec<u8> {
  1150→  1150→        let mut module = Vec::new();
  1151→  1151→
  1152→  1152→        // Magic + version
  1153→  1153→        module.extend_from_slice(&WASM_MAGIC);
  1154→  1154→        module.extend_from_slice(&WASM_VERSION);
  1155→  1155→
  1156→  1156→        // Type section
  1157→  1157→        if !self.types.is_empty() {
  1158→  1158→            let mut section = Vec::new();
  1159→  1159→            section.extend_from_slice(&encode_unsigned_leb128(self.types.len() as u64));
  1160→  1160→            for ft in &self.types {
  1161→  1161→                section.extend_from_slice(&ft.encode());
  1162→  1162→            }
  1163→  1163→            emit_section(&mut module, SECTION_TYPE, &section);
  1164→  1164→        }
  1165→  1165→
  1166→  1166→        // Import section
  1167→  1167→        if !self.imports.is_empty() {
  1168→  1168→            let mut section = Vec::new();
  1169→  1169→            section.extend_from_slice(&encode_unsigned_leb128(self.imports.len() as u64));
  1170→  1170→            for imp in &self.imports {
  1171→  1171→                section.extend_from_slice(&encode_unsigned_leb128(imp.module.len() as u64));
  1172→  1172→                section.extend_from_slice(imp.module.as_bytes());
  1173→  1173→                section.extend_from_slice(&encode_unsigned_leb128(imp.name.len() as u64));
  1174→  1174→                section.extend_from_slice(imp.name.as_bytes());
  1175→  1175→                match &imp.kind {
  1176→  1176→                    WasmImportKind::Function { type_idx } => {
  1177→  1177→                        section.push(0x00);
  1178→  1178→                        section.extend_from_slice(&encode_unsigned_leb128(*type_idx as u64));
  1179→  1179→                    }
  1180→  1180→                    WasmImportKind::Table { elem_type, limits } => {
  1181→  1181→                        section.push(0x01);
  1182→  1182→                        section.push(*elem_type);
  1183→  1183→                        section.extend_from_slice(&limits.encode());
  1184→  1184→                    }
  1185→  1185→                    WasmImportKind::Memory { limits } => {
  1186→  1186→                        section.push(0x02);
  1187→  1187→                        section.extend_from_slice(&limits.encode());
  1188→  1188→                    }
  1189→  1189→                    WasmImportKind::Global { val_type, mutable } => {
  1190→  1190→                        section.push(0x03);
  1191→  1191→                        section.push(val_type.to_byte());
  1192→  1192→                        section.push(if *mutable { 0x01 } else { 0x00 });
  1193→  1193→                    }
  1194→  1194→                }
  1195→  1195→            }
  1196→  1196→            emit_section(&mut module, SECTION_IMPORT, &section);
  1197→  1197→        }
  1198→  1198→
  1199→  1199→        // Function section
  1200→  1200→        if !self.functions.is_empty() {
  1201→  1201→            let mut section = Vec::new();
  1202→  1202→            section.extend_from_slice(&encode_unsigned_leb128(self.functions.len() as u64));
  1203→  1203→            for &type_idx in &self.functions {
  1204→  1204→                section.extend_from_slice(&encode_unsigned_leb128(type_idx as u64));
  1205→  1205→            }
  1206→  1206→            emit_section(&mut module, SECTION_FUNCTION, &section);
  1207→  1207→        }
  1208→  1208→
  1209→  1209→        // Table section
  1210→  1210→        if !self.tables.is_empty() {
  1211→  1211→            let mut section = Vec::new();
  1212→  1212→            section.extend_from_slice(&encode_unsigned_leb128(self.tables.len() as u64));
  1213→  1213→            for (elem_type, limits) in &self.tables {
  1214→  1214→                section.push(*elem_type);
  1215→  1215→                section.extend_from_slice(&limits.encode());
  1216→  1216→            }
  1217→  1217→            emit_section(&mut module, SECTION_TABLE, &section);
  1218→  1218→        }
  1219→  1219→
  1220→  1220→        // Memory section
  1221→  1221→        if !self.memories.is_empty() {
  1222→  1222→            let mut section = Vec::new();
  1223→  1223→            section.extend_from_slice(&encode_unsigned_leb128(self.memories.len() as u64));
  1224→  1224→            for limits in &self.memories {
  1225→  1225→                section.extend_from_slice(&limits.encode());
  1226→  1226→            }
  1227→  1227→            emit_section(&mut module, SECTION_MEMORY, &section);
  1228→  1228→        }
  1229→  1229→
  1230→  1230→        // Global section
  1231→  1231→        if !self.globals.is_empty() {
  1232→  1232→            let mut section = Vec::new();
  1233→  1233→            section.extend_from_slice(&encode_unsigned_leb128(self.globals.len() as u64));
  1234→  1234→            for g in &self.globals {
  1235→  1235→                section.push(g.ty.to_byte());
  1236→  1236→                section.push(if g.mutable { 0x01 } else { 0x00 });
  1237→  1237→                // Emit init expr: <const opcode> <LEB128 value> end
  1238→  1238→                match g.ty {
  1239→  1239→                    WasmType::I32 => {
  1240→  1240→                        section.push(0x41); // i32.const
  1241→  1241→                        section.extend_from_slice(&encode_signed_leb128(g.init_value));
  1242→  1242→                    }
  1243→  1243→                    WasmType::I64 => {
  1244→  1244→                        section.push(0x42); // i64.const
  1245→  1245→                        section.extend_from_slice(&encode_signed_leb128(g.init_value));
  1246→  1246→                    }
  1247→  1247→                    WasmType::F32 => {
  1248→  1248→                        section.push(0x43); // f32.const
  1249→  1249→                        section.extend_from_slice(&(g.init_value as f32).to_le_bytes());
  1250→  1250→                    }
  1251→  1251→                    WasmType::F64 => {
  1252→  1252→                        section.push(0x44); // f64.const
  1253→  1253→                        section.extend_from_slice(&(g.init_value as f64).to_le_bytes());
  1254→  1254→                    }
  1255→  1255→                }
  1256→  1256→                section.push(0x0B); // end
  1257→  1257→            }
  1258→  1258→            emit_section(&mut module, SECTION_GLOBAL, &section);
  1259→  1259→        }
  1260→  1260→
  1261→  1261→        // Export section
  1262→  1262→        if !self.exports.is_empty() {
  1263→  1263→            let mut section = Vec::new();
  1264→  1264→            section.extend_from_slice(&encode_unsigned_leb128(self.exports.len() as u64));
  1265→  1265→            for exp in &self.exports {
  1266→  1266→                section.extend_from_slice(&encode_unsigned_leb128(exp.name.len() as u64));
  1267→  1267→                section.extend_from_slice(exp.name.as_bytes());
  1268→  1268→                section.push(exp.kind as u8);
  1269→  1269→                section.extend_from_slice(&encode_unsigned_leb128(exp.index as u64));
  1270→  1270→            }
  1271→  1271→            emit_section(&mut module, SECTION_EXPORT, &section);
  1272→  1272→        }
  1273→  1273→
  1274→  1274→        // Start section
  1275→  1275→        if let Some(func_idx) = self.start_func {
  1276→  1276→            let section = encode_unsigned_leb128(func_idx as u64);
  1277→  1277→            emit_section(&mut module, SECTION_START, &section);
  1278→  1278→        }
  1279→  1279→
  1280→  1280→        // Element section
  1281→  1281→        if !self.elements.is_empty() {
  1282→  1282→            let mut section = Vec::new();
  1283→  1283→            section.extend_from_slice(&encode_unsigned_leb128(self.elements.len() as u64));
  1284→  1284→            for elem in &self.elements {
  1285→  1285→                section.extend_from_slice(&encode_unsigned_leb128(elem.table_index as u64));
  1286→  1286→                section.extend_from_slice(&elem.offset_expr);
  1287→  1287→                section.extend_from_slice(&encode_unsigned_leb128(elem.func_indices.len() as u64));
  1288→  1288→                for &idx in &elem.func_indices {
  1289→  1289→                    section.extend_from_slice(&encode_unsigned_leb128(idx as u64));
  1290→  1290→                }
  1291→  1291→            }
  1292→  1292→            emit_section(&mut module, SECTION_ELEMENT, &section);
  1293→  1293→        }
  1294→  1294→
  1295→  1295→        // Code section
  1296→  1296→        if !self.code.is_empty() {
  1297→  1297→            let mut section = Vec::new();
  1298→  1298→            section.extend_from_slice(&encode_unsigned_leb128(self.code.len() as u64));
  1299→  1299→            for func_body in &self.code {
  1300→  1300→                // Encode the function body: locals + code + end byte
  1301→  1301→                let mut body_bytes = Vec::new();
  1302→  1302→                body_bytes
  1303→  1303→                    .extend_from_slice(&encode_unsigned_leb128(func_body.locals.len() as u64));
  1304→  1304→                for (count, ty) in &func_body.locals {
  1305→  1305→                    body_bytes.extend_from_slice(&encode_unsigned_leb128(*count as u64));
  1306→  1306→                    body_bytes.push(ty.to_byte());
  1307→  1307→                }
  1308→  1308→                body_bytes.extend_from_slice(&func_body.body);
  1309→  1309→                // Function body is size-prefixed
  1310→  1310→                let body_size = body_bytes.len() as u64;
  1311→  1311→                section.extend_from_slice(&encode_unsigned_leb128(body_size));
  1312→  1312→                section.extend_from_slice(&body_bytes);
  1313→  1313→            }
  1314→  1314→            emit_section(&mut module, SECTION_CODE, &section);
  1315→  1315→        }
  1316→  1316→
  1317→  1317→        // Data section
  1318→  1318→        if !self.data.is_empty() {
  1319→  1319→            let mut section = Vec::new();
  1320→  1320→            section.extend_from_slice(&encode_unsigned_leb128(self.data.len() as u64));
  1321→  1321→            for seg in &self.data {
  1322→  1322→                section.extend_from_slice(&encode_unsigned_leb128(seg.memory_index as u64));
  1323→  1323→                section.extend_from_slice(&seg.offset_expr);
  1324→  1324→                section.extend_from_slice(&encode_unsigned_leb128(seg.data.len() as u64));
  1325→  1325→                section.extend_from_slice(&seg.data);
  1326→  1326→            }
  1327→  1327→            emit_section(&mut module, SECTION_DATA, &section);
  1328→  1328→        }
  1329→  1329→
  1330→  1330→        module
  1331→  1331→    }
  1332→  1332→}
  1333→  1333→
  1334→  1334→/// Emit a Wasm section: section ID + LEB128 size + content.
  1335→  1335→fn emit_section(module: &mut Vec<u8>, section_id: u8, content: &[u8]) {
  1336→  1336→    module.push(section_id);
  1337→  1337→    module.extend_from_slice(&encode_unsigned_leb128(content.len() as u64));
  1338→  1338→    module.extend_from_slice(content);
  1339→  1339→}
  1340→  1340→
  1341→  1341→/// Skip `count` unsigned LEB128 values in the byte slice, advancing `offset`.
  1342→  1342→fn skip_leb128(bytes: &[u8], offset: &mut usize, count: usize) {
  1343→  1343→    for _ in 0..count {
  1344→  1344→        while *offset < bytes.len() && (bytes[*offset] & 0x80) != 0 {
  1345→  1345→            *offset += 1;
  1346→  1346→        }
  1347→  1347→        if *offset < bytes.len() {
  1348→  1348→            *offset += 1;
  1349→  1349→        }
  1350→  1350→    }
  1351→  1351→}
  1352→  1352→
  1353→  1353→/// Skip `count` signed LEB128 values in the byte slice, advancing `offset`.
  1354→  1354→fn skip_signed_leb128(bytes: &[u8], offset: &mut usize, count: usize) {
  1355→  1355→    // Signed and unsigned LEB128 have the same encoding structure for skipping
  1356→  1356→    skip_leb128(bytes, offset, count);
  1357→  1357→}
  1358→  1358→
  1359→  1359→/// Convert a WasmType (from this module) to the serializable WasmValueType
  1360→  1360→/// (from the backend module).
  1361→  1361→fn wasm_type_to_backend(ty: WasmType) -> crate::backend::WasmValueType {
  1362→  1362→    match ty {
  1363→  1363→        WasmType::I32 => crate::backend::WasmValueType::I32,
  1364→  1364→        WasmType::I64 => crate::backend::WasmValueType::I64,
  1365→  1365→        WasmType::F32 => crate::backend::WasmValueType::F32,
  1366→  1366→        WasmType::F64 => crate::backend::WasmValueType::F64,
  1367→  1367→    }
  1368→  1368→}
  1369→  1369→
  1370→  1370→/// Convert a serializable WasmValueType (from the backend module) back to
  1371→  1371→/// the WasmType used by this module.
  1372→  1372→fn backend_to_wasm_type(ty: &crate::backend::WasmValueType) -> WasmType {
  1373→  1373→    match ty {
  1374→  1374→        crate::backend::WasmValueType::I32 => WasmType::I32,
  1375→  1375→        crate::backend::WasmValueType::I64 => WasmType::I64,
  1376→  1376→        crate::backend::WasmValueType::F32 => WasmType::F32,
  1377→  1377→        crate::backend::WasmValueType::F64 => WasmType::F64,
  1378→  1378→    }
  1379→  1379→}
  1380→  1380→
  1381→  1381→/// Skip a single Wasm instruction starting at `offset`, advancing past it.
  1382→  1382→///
  1383→  1383→/// This is used by `allocate_registers` to split the body bytecode into
  1384→  1384→/// per-instruction `AllocatedInstruction` entries with real opcode names.
  1385→  1385→fn skip_one_instruction(bytes: &[u8], offset: &mut usize) {
  1386→  1386→    if *offset >= bytes.len() {
  1387→  1387→        return;
  1388→  1388→    }
  1389→  1389→    let byte = bytes[*offset];
  1390→  1390→    *offset += 1;
  1391→  1391→
  1392→  1392→    match byte {
  1393→  1393→        // ── Multi-byte opcodes ────────────────────────────────────
  1394→  1394→        0xFC => {
  1395→  1395→            // Bulk memory / saturated arithmetic prefix
  1396→  1396→            skip_leb128(bytes, offset, 1); // sub-opcode
  1397→  1397→            let (subop, _) = decode_unsigned_leb128(&bytes[*offset - 1..]);
  1398→  1398→            match subop {
  1399→  1399→                0x08 => skip_leb128(bytes, offset, 2), // memory.init
  1400→  1400→                0x0A => skip_leb128(bytes, offset, 2), // memory.copy
  1401→  1401→                0x0B => skip_leb128(bytes, offset, 1), // memory.fill
  1402→  1402→                _ => {}                                // other sub-ops may have 0 or more operands
  1403→  1403→            }
  1404→  1404→        }
  1405→  1405→        0xFD => {
  1406→  1406→            // SIMD prefix
  1407→  1407→            skip_leb128(bytes, offset, 1); // sub-opcode
  1408→  1408→            let (subop, _) = decode_unsigned_leb128(&bytes[*offset - 1..]);
  1409→  1409→            if subop == 0x0C {
  1410→  1410→                // v128.const: skip 16 bytes of payload
  1411→  1411→                *offset += 16;
  1412→  1412→            }
  1413→  1413→        }
  1414→  1414→
  1415→  1415→        // ── Control flow ─────────────────────────────────────────
  1416→  1416→        0x00 => {} // unreachable
  1417→  1417→        0x01 => {} // nop
  1418→  1418→        0x02..=0x04 => {
  1419→  1419→            *offset += 1;
  1420→  1420→        } // block/loop/if + blocktype
  1421→  1421→        0x05 | 0x0B => {} // else, end
  1422→  1422→        0x0C | 0x0D => {
  1423→  1423→            skip_leb128(bytes, offset, 1);
  1424→  1424→        } // br, br_if
  1425→  1425→        0x0E => {
  1426→  1426→            // br_table: count + labels + default
  1427→  1427→            let (count, size) = decode_unsigned_leb128(&bytes[*offset..]);
  1428→  1428→            *offset += size;
  1429→  1429→            skip_leb128(bytes, offset, count as usize + 1);
  1430→  1430→        }
  1431→  1431→        0x0F => {} // return
  1432→  1432→        0x10 => {
  1433→  1433→            skip_leb128(bytes, offset, 1);
  1434→  1434→        } // call
  1435→  1435→        0x11 => {
  1436→  1436→            skip_leb128(bytes, offset, 2);
  1437→  1437→        } // call_indirect
  1438→  1438→
  1439→  1439→        // ── Parametric ───────────────────────────────────────────
  1440→  1440→        0x1A | 0x1B => {} // drop, select
  1441→  1441→
  1442→  1442→        // ── Variable ─────────────────────────────────────────────
  1443→  1443→        0x20..=0x24 => {
  1444→  1444→            skip_leb128(bytes, offset, 1);
  1445→  1445→        } // local.get/set/tee, global.get/set
  1446→  1446→
  1447→  1447→        // ── Memory ───────────────────────────────────────────────
  1448→  1448→        0x28..=0x3E => {
  1449→  1449→            skip_leb128(bytes, offset, 2);
  1450→  1450→        } // loads/stores: align + offset
  1451→  1451→        0x3F | 0x40 => {
  1452→  1452→            skip_leb128(bytes, offset, 1);
  1453→  1453→        } // memory.size, memory.grow
  1454→  1454→
  1455→  1455→        // ── Numeric ──────────────────────────────────────────────
  1456→  1456→        0x41 => {
  1457→  1457→            skip_signed_leb128(bytes, offset, 1);
  1458→  1458→        } // i32.const
  1459→  1459→        0x42 => {
  1460→  1460→            skip_signed_leb128(bytes, offset, 1);
  1461→  1461→        } // i64.const
  1462→  1462→        0x43 => {
  1463→  1463→            *offset += 4;
  1464→  1464→        } // f32.const
  1465→  1465→        0x44 => {
  1466→  1466→            *offset += 8;
  1467→  1467→        } // f64.const
  1468→  1468→
  1469→  1469→        // All other single-byte opcodes (comparisons, arithmetic, conversions)
  1470→  1470→        0x45..=0x4F
  1471→  1471→        | 0x50..=0x5A
  1472→  1472→        | 0x5B..=0x66
  1473→  1473→        | 0x67..=0x78
  1474→  1474→        | 0x79..=0x8A
  1475→  1475→        | 0x8C
  1476→  1476→        | 0x91..=0x95
  1477→  1477→        | 0x9A
  1478→  1478→        | 0x9F..=0xA3
  1479→  1479→        | 0xA7..=0xBF => {}
  1480→  1480→
  1481→  1481→        _ => {} // Unknown opcode; nothing more to skip
  1482→  1482→    }
  1483→  1483→}

  1488→  1488→
  1489→  1489→/// Context for lowering an IR function to Wasm bytecode.
  1490→  1490→struct LoweringContext {
  1491→  1491→    /// Map from virtual register ID to local index.
  1492→  1492→    vreg_to_local: HashMap<u32, u32>,
  1493→  1493→    /// Map from virtual register ID to its Wasm type.
  1494→  1494→    vreg_types: HashMap<u32, WasmType>,
  1495→  1495→    /// Number of locals (including parameters).
  1496→  1496→    num_locals: u32,
  1497→  1497→    /// Local declarations (for the function body).
  1498→  1498→    locals: Vec<(u32, WasmType)>,
  1499→  1499→    /// Map from block label to its label depth index for branch targets.
  1500→  1500→    block_labels: HashMap<String, u32>,
  1501→  1501→    /// Accumulated Wasm instructions.
  1502→  1502→    instrs: Vec<WasmInstr>,
  1503→  1503→    /// Expected result types for the function (used for return type coercion).
  1504→  1504→    result_types: Vec<WasmType>,
  1505→  1505→}
  1506→  1506→
  1507→  1507→impl LoweringContext {
  1508→  1508→    fn new(result_types: Vec<WasmType>) -> Self {
  1509→  1509→        Self {
  1510→  1510→            vreg_to_local: HashMap::new(),
  1511→  1511→            vreg_types: HashMap::new(),
  1512→  1512→            num_locals: 0,
  1513→  1513→            locals: Vec::new(),
  1514→  1514→            block_labels: HashMap::new(),
  1515→  1515→            instrs: Vec::new(),
  1516→  1516→            result_types,
  1517→  1517→        }
  1518→  1518→    }
  1519→  1519→
  1520→  1520→    /// Allocate a local for a virtual register, returning the local index.
  1521→  1521→    fn alloc_local(&mut self, vreg_id: u32, ty: WasmType) -> u32 {
  1522→  1522→        let idx = self.num_locals;
  1523→  1523→        self.vreg_to_local.insert(vreg_id, idx);
  1524→  1524→        self.vreg_types.insert(vreg_id, ty);
  1525→  1525→        self.num_locals += 1;
  1526→  1526→        // Try to merge with an existing (count, type) entry
  1527→  1527→        if let Some(last) = self.locals.last_mut() {
  1528→  1528→            if last.1 == ty {
  1529→  1529→                last.0 += 1;
  1530→  1530→                return idx;
  1531→  1531→            }
  1532→  1532→        }
  1533→  1533→        self.locals.push((1, ty));
  1534→  1534→        idx
  1535→  1535→    }
  1536→  1536→
  1537→  1537→    /// Get the local index for a virtual register.
  1538→  1538→    fn get_local(&self, vreg_id: u32) -> Option<u32> {
  1539→  1539→        self.vreg_to_local.get(&vreg_id).copied()
  1540→  1540→    }
  1541→  1541→
  1542→  1542→    /// Emit an instruction.
  1543→  1543→    fn emit(&mut self, instr: WasmInstr) {
  1544→  1544→        self.instrs.push(instr);
  1545→  1545→    }
  1546→  1546→
  1547→  1547→    /// Push a value onto the Wasm stack from an IRValue.
  1548→  1548→    /// On Wasm32, pointers and most integer values are pushed as i32.
  1549→  1549→    /// I64 is used only when the type hint explicitly requests it (e.g., for
  1550→  1550→    /// I64Load/I64Store of 64-bit values).  Addresses are always truncated
  1551→  1551→    /// to i32 since the Wasm32 address space is 32 bits.
  1552→  1552→    fn push_value(&mut self, val: &IRValue, type_hint: Option<&WasmType>) {
  1553→  1553→        match val {
  1554→  1554→            IRValue::Immediate(v) => {
  1555→  1555→                let ty = type_hint.copied().unwrap_or(WasmType::I32);
  1556→  1556→                match ty {
  1557→  1557→                    WasmType::I64 => self.emit(WasmInstr::I64Const(*v)),
  1558→  1558→                    WasmType::F32 => self.emit(WasmInstr::F32Const(*v as f32)),
  1559→  1559→                    WasmType::F64 => self.emit(WasmInstr::F64Const(*v as f64)),
  1560→  1560→                    _ => self.emit(WasmInstr::I32Const(*v as i32)),
  1561→  1561→                }
  1562→  1562→            }
  1563→  1563→            IRValue::Register(id) => {
  1564→  1564→                // If the register hasn't been allocated yet, allocate it as i32.
  1565→  1565→                // On Wasm32, all integer locals are i32.
  1566→  1566→                if self.get_local(*id).is_none() {
  1567→  1567→                    self.alloc_local(*id, WasmType::I32);
  1568→  1568→                }
  1569→  1569→                if let Some(local_idx) = self.get_local(*id) {
  1570→  1570→                    self.emit(WasmInstr::LocalGet(local_idx));
  1571→  1571→                }
  1572→  1572→            }
  1573→  1573→            IRValue::Address(addr) => {
  1574→  1574→                self.emit(WasmInstr::I32Const(*addr as i32)); // wasm32: pointers are i32
  1575→  1575→            }
  1576→  1576→            IRValue::Label(_) => {
  1577→  1577→                // Labels are handled via block structure; not pushed as values
  1578→  1578→            }
  1579→  1579→        }
  1580→  1580→    }
  1581→  1581→
  1582→  1582→    /// Store the top of the Wasm stack into a virtual register's local.
  1583→  1583→    fn pop_to_vreg(&mut self, vreg_id: u32, ty: WasmType) {
  1584→  1584→        if !self.vreg_to_local.contains_key(&vreg_id) {
  1585→  1585→            self.alloc_local(vreg_id, ty);
  1586→  1586→        }
  1587→  1587→        if let Some(local_idx) = self.get_local(vreg_id) {
  1588→  1588→            self.emit(WasmInstr::LocalSet(local_idx));
  1589→  1589→        }
  1590→  1590→    }
  1591→  1591→}
  1592→  1592→
  1593→  1593→/// Determine the Wasm type for dedicated arithmetic IR instructions (Add, Sub,
  1594→  1594→/// Mul, Div, Cmp).  These instructions carry an optional `ty` field that
  1595→  1595→/// indicates the operand width.  On the Wasm32 target, all integer types map
  1596→  1596→/// to I32 since pointers are 32 bits and the address space is 32 bits;
  1597→  1597→/// only float types retain their original width.  This is consistent with
  1598→  1598→/// `wasm_type_for_binop` and ensures pointer arithmetic always uses i32 ops.
  1599→  1599→fn wasm_type_for_dedicated_arith(ir_ty: Option<&IRType>) -> WasmType {
  1600→  1600→    match ir_ty {
  1601→  1601→        Some(IRType::F32) => WasmType::F32,
  1602→  1602→        Some(IRType::F64) => WasmType::F64,
  1603→  1603→        _ => WasmType::I32, // all integer types → i32 on wasm32 (pointers are i32)
  1604→  1604→    }
  1605→  1605→}
  1606→  1606→
  1607→  1607→/// Determine the Wasm type for an IR BinOp based on the operand types and
  1608→  1608→/// the optional IR type annotation.  On the Wasm32 target, all integer
  1609→  1609→/// operations use i32 since the address space is 32 bits; only float types
  1610→  1610→/// retain their original width.
  1611→  1611→fn wasm_type_for_binop(
  1612→  1612→    _op: &BinOpKind,
  1613→  1613→    _lhs: &IRValue,
  1614→  1614→    _rhs: &IRValue,
  1615→  1615→    ir_ty: Option<&IRType>,
  1616→  1616→    _vreg_types: &HashMap<u32, WasmType>,
  1617→  1617→) -> WasmType {
  1618→  1618→    // If the IR provides a type, use it (but map I64/U64 to I32 for Wasm32).
  1619→  1619→    if let Some(ty) = ir_ty {
  1620→  1620→        return match ty {
  1621→  1621→            IRType::F32 => WasmType::F32,
  1622→  1622→            IRType::F64 => WasmType::F64,
  1623→  1623→            _ => WasmType::I32, // all integer types → i32 on wasm32
  1624→  1624→        };
  1625→  1625→    }
  1626→  1626→    // Default to i32 for all integer ops on wasm32
  1627→  1627→    WasmType::I32
  1628→  1628→}
  1629→  1629→
  1630→  1630→/// Infer the Wasm type of an IR value based on its representation.
  1631→  1631→///
  1632→  1632→/// On the Wasm32 target, all integer values are i32.  Only float immediates
  1633→  1633→/// use the wider type.
  1634→  1634→fn infer_wasm_type(val: &IRValue) -> WasmType {
  1635→  1635→    match val {
  1636→  1636→        IRValue::Immediate(_)
  1637→  1637→        | IRValue::Register(_)
  1638→  1638→        | IRValue::Address(_)
  1639→  1639→        | IRValue::Label(_) => WasmType::I32,
  1640→  1640→    }
  1641→  1641→}
  1642→  1642→
  1643→  1643→/// Lower an IR function to Wasm bytecode, returning the function body bytes
  1644→  1644→/// and local declarations.
  1645→  1645→fn lower_function(func: &IRFunction) -> Result<(WasmFuncBody, WasmFuncType), BackendError> {
  1646→  1646→    // Compute result types.
  1647→  1647→    // In Wasm32, all integer results are i32 since it's a 32-bit target.
  1648→  1648→    let result_types: Vec<WasmType> = func
  1649→  1649→        .result_types
  1650→  1650→        .iter()
  1651→  1651→        .filter_map(WasmType::from_ir_type)
  1652→  1652→        .map(|t| if t.is_integer() { WasmType::I32 } else { t })
  1653→  1653→        .collect();
  1654→  1654→    let mut ctx = LoweringContext::new(result_types);
  1655→  1655→
  1656→  1656→    // Assign locals for parameters.
  1657→  1657→    // In Wasm32, all integer parameters are i32 regardless of IR type, since
  1658→  1658→    // pointers and all integer values fit in 32 bits on a 32-bit target.
  1659→  1659→    // Only float types retain their original width.
  1660→  1660→    for (i, param) in func.params.iter().enumerate() {
  1661→  1661→        let ty = func
  1662→  1662→            .param_types
  1663→  1663→            .get(i)
  1664→  1664→            .and_then(WasmType::from_ir_type)
  1665→  1665→            .map(|t| if t.is_integer() { WasmType::I32 } else { t })
  1666→  1666→            .unwrap_or(WasmType::I32);
  1667→  1667→        if let IRValue::Register(id) = param {
  1668→  1668→            let idx = ctx.num_locals;
  1669→  1669→            ctx.vreg_to_local.insert(*id, idx);
  1670→  1670→            ctx.vreg_types.insert(*id, ty);
  1671→  1671→            ctx.num_locals += 1;
  1672→  1672→            if let Some(last) = ctx.locals.last_mut() {
  1673→  1673→                if last.1 == ty {
  1674→  1674→                    last.0 += 1;
  1675→  1675→                } else {
  1676→  1676→                    ctx.locals.push((1, ty));
  1677→  1677→                }
  1678→  1678→            } else {
  1679→  1679→                ctx.locals.push((1, ty));
  1680→  1680→            }
  1681→  1681→        }
  1682→  1682→    }
  1683→  1683→
  1684→  1684→    // Lower each block
  1685→  1685→    for (block_idx, block) in func.blocks.iter().enumerate() {
  1686→  1686→        // Emit a block for structured control flow (except the entry block)
  1687→  1687→        if block_idx > 0 {
  1688→  1688→            ctx.emit(WasmInstr::Block(None)); // void block (no result value)
  1689→  1689→            ctx.block_labels
  1690→  1690→                .insert(block.label.clone(), block_idx as u32);
  1691→  1691→        } else {
  1692→  1692→            ctx.block_labels.insert(block.label.clone(), 0);
  1693→  1693→        }
  1694→  1694→
  1695→  1695→        // Lower instructions
  1696→  1696→        for instr in &block.instructions {
  1697→  1697→            lower_instruction(instr, &mut ctx)?;
  1698→  1698→        }
  1699→  1699→
  1700→  1700→        // Lower terminator
