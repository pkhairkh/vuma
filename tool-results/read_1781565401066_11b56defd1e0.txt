     1→     1→//! Structured JSON diagnostics output for LLM integration.
     2→     2→//!
     3→     3→//! This module provides machine-readable diagnostic messages that can be
     4→     4→//! serialized to JSON, enabling LLM-based tools and IDE integrations to
     5→     5→//! programmatically consume compiler errors, warnings, and hints.
     6→     6→//!
     7→     7→//! # Architecture
     8→     8→//!
     9→     9→//! Each diagnostic carries a structured error code (`"E001"`, `"W001"`, etc.),
    10→    10→//! a severity level, a human-readable message, a source identifier (which
    11→    11→//! compiler stage produced it), a precise source location, optional related
    12→    12→//! information, quick-fix suggestions with edit ranges, and an optional
    13→    13→//! causal chain linking to parent diagnostics.
    14→    14→//!
    15→    15→//! # Error Catalog
    16→    16→//!
    17→    17→//! The diagnostic codes are organized into the following ranges:
    18→    18→//!
    19→    19→//! | Range    | Category       | Description                              |
    20→    20→//! |----------|----------------|------------------------------------------|
    21→    21→//! | E001–E030| Compilation    | Syntax, type, name resolution errors     |
    22→    22→//! | E031–E040| Codegen        | Register allocation, encoding, relocation|
    23→    23→//! | E041–E050| Verification   | Invariant violations, proof failures     |
    24→    24→//! | W001–W010| Warnings       | Unused vars, performance hints            |
    25→    25→//! | I001–I005| Informational  | General compiler information messages     |
    26→    26→//!
    27→    27→//! # Example JSON output
    28→    28→//!
    29→    29→//! ```json
    30→    30→//! [
    31→    31→//!   {
    32→    32→//!     "code": "E002",
    33→    33→//!     "severity": "error",
    34→    34→//!     "message": "undefined variable `foo`",
    35→    35→//!     "source": "parser",
    36→    36→//!     "location": {
    37→    37→//!       "file": "main.vu",
    38→    38→//!       "start_line": 10,
    39→    39→//!       "start_col": 4,
    40→    40→//!       "end_line": 10,
    41→    41→//!       "end_col": 7
    42→    42→//!     },
    43→    43→//!     "related": [],
    44→    44→//!     "suggestions": ["did you mean `for`?"],
    45→    45→//!     "chain": []
    46→    46→//!   }
    47→    47→//! ]
    48→    48→//! ```
    49→    49→//!
    50→    50→//! # Integration
    51→    51→//!
    52→    52→//! Use [`from_parse_error`] to convert parser errors, [`from_vuma_error`] for
    53→    53→//! pipeline-level errors, and [`from_codegen_error`] for code-generation
    54→    54→//! failures.
    55→    55→

    57→    57→use std::collections::HashMap;

    61→    61→use vuma_parser::{offset_to_location, ParseError, ParseErrorKind, Severity as ParserSeverity};
    62→    62→

    66→    66→// DiagnosticSeverity
    67→    67→// ═══════════════════════════════════════════════════════════════════════════
    68→    68→
    69→    69→/// Severity level for a machine-readable diagnostic.
    70→    70→///
    71→    71→/// Unlike the parser's `Severity` (which has `Note`), this uses `Hint` for
    72→    72→/// LLM-consumable suggestions and `Info` for informational notes.
    73→    73→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
    74→    74→#[serde(rename_all = "lowercase")]
    75→    75→pub enum DiagnosticSeverity {
    76→    76→    /// A hard error — the program cannot be compiled.
    77→    77→    Error,
    78→    78→    /// A warning — suspicious code that is technically valid.
    79→    79→    Warning,
    80→    80→    /// An informational note, typically providing context.
    81→    81→    Info,
    82→    82→    /// A hint or quick-fix suggestion.
    83→    83→    Hint,
    84→    84→}
    85→    85→
    86→    86→impl fmt::Display for DiagnosticSeverity {
    87→    87→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
    88→    88→        match self {
    89→    89→            DiagnosticSeverity::Error => write!(f, "error"),
    90→    90→            DiagnosticSeverity::Warning => write!(f, "warning"),
    91→    91→            DiagnosticSeverity::Info => write!(f, "info"),
    92→    92→            DiagnosticSeverity::Hint => write!(f, "hint"),
    93→    93→        }
    94→    94→    }
    95→    95→}
    96→    96→
    97→    97→impl From<ParserSeverity> for DiagnosticSeverity {
    98→    98→    fn from(s: ParserSeverity) -> Self {
    99→    99→        match s {
   100→   100→            ParserSeverity::Error => DiagnosticSeverity::Error,
   101→   101→            ParserSeverity::Warning => DiagnosticSeverity::Warning,
   102→   102→            ParserSeverity::Note => DiagnosticSeverity::Info,
   103→   103→        }
   104→   104→    }
   105→   105→}
   106→   106→
   107→   107→impl DiagnosticSeverity {
   108→   108→    /// Returns the LSP numeric severity value.
   109→   109→    pub fn to_lsp_severity(&self) -> u32 {
   110→   110→        match self {
   111→   111→            DiagnosticSeverity::Error => 1,
   112→   112→            DiagnosticSeverity::Warning => 2,
   113→   113→            DiagnosticSeverity::Info => 3,
   114→   114→            DiagnosticSeverity::Hint => 4,
   115→   115→        }
   116→   116→    }
   117→   117→}

   120→   120→// DiagnosticSourceLocation
   121→   121→// ═══════════════════════════════════════════════════════════════════════════
   122→   122→
   123→   123→/// Machine-readable source location with line/column ranges.
   124→   124→///
   125→   125→/// All positions are 1-based (as is conventional for compiler diagnostics)
   126→   126→/// rather than the 0-based positions used internally by the parser.
   127→   127→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   128→   128→pub struct DiagnosticSourceLocation {
   129→   129→    /// File path or module name.
   130→   130→    pub file: String,
   131→   131→    /// Start line (1-based).
   132→   132→    pub start_line: u32,
   133→   133→    /// Start column (1-based).
   134→   134→    pub start_col: u32,
   135→   135→    /// End line (1-based).
   136→   136→    pub end_line: u32,
   137→   137→    /// End column (1-based, exclusive).
   138→   138→    pub end_col: u32,
   139→   139→}
   140→   140→
   141→   141→impl DiagnosticSourceLocation {
   142→   142→    /// Create a location spanning a single point (zero-width).
   143→   143→    pub fn point(file: impl Into<String>, line: u32, col: u32) -> Self {
   144→   144→        Self {
   145→   145→            file: file.into(),
   146→   146→            start_line: line,
   147→   147→            start_col: col,
   148→   148→            end_line: line,
   149→   149→            end_col: col,
   150→   150→        }
   151→   151→    }
   152→   152→
   153→   153→    /// Create a location spanning a range within a single line.
   154→   154→    pub fn range(
   155→   155→        file: impl Into<String>,
   156→   156→        line: u32,
   157→   157→        start_col: u32,
   158→   158→        end_col: u32,
   159→   159→    ) -> Self {
   160→   160→        Self {
   161→   161→            file: file.into(),
   162→   162→            start_line: line,
   163→   163→            start_col,
   164→   164→            end_line: line,
   165→   165→            end_col,
   166→   166→        }
   167→   167→    }
   168→   168→
   169→   169→    /// Create a multi-line location.
   170→   170→    pub fn multi_line(
   171→   171→        file: impl Into<String>,
   172→   172→        start_line: u32,
   173→   173→        start_col: u32,
   174→   174→        end_line: u32,
   175→   175→        end_col: u32,
   176→   176→    ) -> Self {
   177→   177→        Self {
   178→   178→            file: file.into(),
   179→   179→            start_line,
   180→   180→            start_col,
   181→   181→            end_line,
   182→   182→            end_col,
   183→   183→        }
   184→   184→    }
   185→   185→
   186→   186→    /// Create a synthetic "unknown" location for diagnostics that cannot be
   187→   187→    /// tied to a specific source position.
   188→   188→    pub fn unknown() -> Self {
   189→   189→        Self {
   190→   190→            file: String::new(),
   191→   191→            start_line: 0,
   192→   192→            start_col: 0,
   193→   193→            end_line: 0,
   194→   194→            end_col: 0,
   195→   195→        }
   196→   196→    }
   197→   197→
   198→   198→    /// Returns `true` if this location represents an unknown / synthetic
   199→   199→    /// position (all fields zero, no file).
   200→   200→    pub fn is_unknown(&self) -> bool {
   201→   201→        self.file.is_empty() && self.start_line == 0 && self.start_col == 0
   202→   202→    }
   203→   203→}
   204→   204→
   205→   205→impl fmt::Display for DiagnosticSourceLocation {
   206→   206→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   207→   207→        if self.is_unknown() {
   208→   208→            write!(f, "<unknown>")
   209→   209→        } else if self.start_line == self.end_line && self.start_col == self.end_col {
   210→   210→            write!(f, "{}:{}:{}", self.file, self.start_line, self.start_col)
   211→   211→        } else {
   212→   212→            write!(
   213→   213→                f,
   214→   214→                "{}:{}:{}-{}:{}",
   215→   215→                self.file, self.start_line, self.start_col, self.end_line, self.end_col
   216→   216→            )
   217→   217→        }
   218→   218→    }
   219→   219→}

   246→   246→// Suggestion (structured)
   247→   247→// ═══════════════════════════════════════════════════════════════════════════
   248→   248→
   249→   249→/// A structured suggestion with an optional edit range for automatic fixing.
   250→   250→///
   251→   251→/// Suggestions can be simple text hints (no edit range) or precise code
   252→   252→/// edits that an IDE or tool can apply automatically.
   253→   253→#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
   254→   254→pub struct Suggestion {
   255→   255→    /// Human-readable description of the suggestion.
   256→   256→    pub message: String,
   257→   257→    /// Optional source location that this suggestion replaces.
   258→   258→    #[serde(skip_serializing_if = "Option::is_none")]
   259→   259→    pub edit_range: Option<DiagnosticSourceLocation>,
   260→   260→    /// Optional replacement text for the edit range.
   261→   261→    #[serde(skip_serializing_if = "Option::is_none")]
   262→   262→    pub replacement: Option<String>,
   263→   263→    /// The applicability of this suggestion.
   264→   264→    #[serde(default = "default_applicability")]
   265→   265→    pub applicability: SuggestionApplicability,
   266→   266→}
   267→   267→
   268→   268→/// How likely a suggestion is to be the right fix.
   269→   269→#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
   270→   270→#[serde(rename_all = "snake_case")]
   271→   271→pub enum SuggestionApplicability {
   272→   272→    /// The suggestion is definitely what the user intended (machine-applicable).
   273→   273→    MachineApplicable,
   274→   274→    /// The suggestion has a very high likelihood of being correct.
   275→   275→    MaybeIncorrect,
   276→   276→    /// The suggestion may or may not be correct — requires human judgment.
   277→   277→    HasPlaceholders,
   278→   278→    /// The suggestion is a rough hint, not a precise fix.
   279→   279→    Unspecified,
   280→   280→}
   281→   281→
   282→   282→fn default_applicability() -> SuggestionApplicability {
   283→   283→    SuggestionApplicability::Unspecified
   284→   284→}
   285→   285→
   286→   286→impl Suggestion {
   287→   287→    /// Create a simple text-only suggestion (no automatic edit).
   288→   288→    pub fn text(message: impl Into<String>) -> Self {
   289→   289→        Self {
   290→   290→            message: message.into(),
   291→   291→            edit_range: None,
   292→   292→            replacement: None,
   293→   293→            applicability: SuggestionApplicability::Unspecified,
   294→   294→        }
   295→   295→    }
   296→   296→
   297→   297→    /// Create a suggestion with an exact edit range and replacement text.
   298→   298→    pub fn edit(
   299→   299→        message: impl Into<String>,
   300→   300→        range: DiagnosticSourceLocation,
   301→   301→        replacement: impl Into<String>,
   302→   302→    ) -> Self {
   303→   303→        Self {
   304→   304→            message: message.into(),
   305→   305→            edit_range: Some(range),
   306→   306→            replacement: Some(replacement.into()),
   307→   307→            applicability: SuggestionApplicability::MachineApplicable,
   308→   308→        }
   309→   309→    }
   310→   310→
   311→   311→    /// Create a machine-applicable suggestion with an edit range.
   312→   312→    pub fn machine_applicable(
   313→   313→        message: impl Into<String>,
   314→   314→        range: DiagnosticSourceLocation,
   315→   315→        replacement: impl Into<String>,
   316→   316→    ) -> Self {
   317→   317→        Self {
   318→   318→            message: message.into(),
   319→   319→            edit_range: Some(range),
   320→   320→            replacement: Some(replacement.into()),
   321→   321→            applicability: SuggestionApplicability::MachineApplicable,
   322→   322→        }
   323→   323→    }
   324→   324→
   325→   325→    /// Create a suggestion with placeholders (needs human review).
   326→   326→    pub fn with_placeholders(message: impl Into<String>) -> Self {
   327→   327→        Self {
   328→   328→            message: message.into(),
   329→   329→            edit_range: None,
   330→   330→            replacement: None,
   331→   331→            applicability: SuggestionApplicability::HasPlaceholders,
   332→   332→        }
   333→   333→    }
   334→   334→}

   472→   472→        "W002" => "Implicit type conversion",
   473→   473→        "W003" => "Large constant (performance hint)",
   474→   474→        "W004" => "Dead code",
   475→   475→        "W005" => "Redundant cast",
   476→   476→        "W006" => "Shadowed variable",
   477→   477→        "W007" => "Unnecessary mut keyword",
   478→   478→        "W008" => "Deprecated feature",
   479→   479→        "W009" => "Unused import",
   480→   480→        "W010" => "Reachable panic",
   481→   481→
   482→   482→        // ── Informational (I001–I005) ──
   483→   483→        "I001" => "Compilation started",
   484→   484→        "I002" => "Compilation stage completed",
   485→   485→        "I003" => "Optimization applied",
   486→   486→        "I004" => "Verification passed",
   487→   487→        "I005" => "Build artifact produced",
   488→   488→
   489→   489→        _ => "Unknown diagnostic code",
   490→   490→    }
   491→   491→}
   492→   492→
   493→   493→/// Returns the category prefix for a diagnostic code.
   494→   494→///
   495→   495→/// - `"E"` for errors, `"W"` for warnings, `"I"` for informational.
   496→   496→pub fn code_category(code: &str) -> &'static str {
   497→   497→    if code.starts_with('E') {
   498→   498→        "error"
   499→   499→    } else if code.starts_with('W') {
   500→   500→        "warning"
   501→   501→    } else if code.starts_with('I') {
   502→   502→        "info"
   503→   503→    } else {
   504→   504→        "unknown"
   505→   505→    }
   506→   506→}
   507→   507→
   508→   508→/// Returns the sub-category range for a diagnostic code.
   509→   509→///
   510→   510→/// - `"compilation"` for E001–E030
   511→   511→/// - `"codegen"` for E031–E040
   512→   512→/// - `"verification"` for E041–E050
   513→   513→/// - `"warning"` for W001–W010
   514→   514→/// - `"informational"` for I001–I005
   515→   515→pub fn code_subcategory(code: &str) -> &'static str {
   516→   516→    if let Ok(num) = code[1..].parse::<u32>() {
   517→   517→        match code.chars().next() {
   518→   518→            Some('E') => {
   519→   519→                if num <= 30 {
   520→   520→                    "compilation"
   521→   521→                } else if num <= 40 {
   522→   522→                    "codegen"
   523→   523→                } else if num <= 50 {
   524→   524→                    "verification"
   525→   525→                } else {
   526→   526→                    "unknown"
   527→   527→                }
   528→   528→            }
   529→   529→            Some('W') => "warning",
   530→   530→            Some('I') => "informational",
   531→   531→            _ => "unknown",
   532→   532→        }
   533→   533→    } else {
   534→   534→        "unknown"
   535→   535→    }
   536→   536→}

   539→   539→// VumaDiagnostic
   540→   540→// ═══════════════════════════════════════════════════════════════════════════
   541→   541→
   542→   542→/// A machine-readable diagnostic with structured fields for LLM consumption.
   543→   543→///
   544→   544→/// This is the primary output type for JSON diagnostics. Every field is
   545→   545→/// serializable and carries enough information for an LLM or IDE to
   546→   546→/// understand, categorize, and potentially auto-fix the issue.
   547→   547→///
   548→   548→/// # Error Chaining
   549→   549→///
   550→   550→/// Diagnostics support causal chains via the [`chain`](VumaDiagnostic::chain)
   551→   551→/// method. A chain represents the cause of a diagnostic: for example,
   552→   552→/// "E003: Type mismatch" may be caused by "E023: Unknown type 'int'".
   553→   553→///
   554→   554→/// # Suggestions
   555→   555→///
   556→   556→/// Each diagnostic can carry zero or more [`Suggestion`] values, which may
   557→   557→/// include precise edit ranges for automatic code fixes.
   558→   558→#[derive(Debug, Clone, Serialize, Deserialize)]
   559→   559→pub struct VumaDiagnostic {
   560→   560→    /// Structured diagnostic code, e.g. `"E001"`, `"W001"`.
   561→   561→    pub code: String,
   562→   562→    /// Severity level.
   563→   563→    pub severity: DiagnosticSeverity,
   564→   564→    /// Human-readable message (suitable for display).
   565→   565→    pub message: String,
   566→   566→    /// Compiler stage that produced this diagnostic
   567→   567→    /// (`"parser"`, `"scg"`, `"ir"`, `"codegen"`, etc.).
   568→   568→    pub source: String,
   569→   569→    /// Precise source location.
   570→   570→    pub location: DiagnosticSourceLocation,
   571→   571→    /// Related source locations with explanatory messages.
   572→   572→    pub related: Vec<RelatedInfo>,
   573→   573→    /// Quick-fix suggestions (structured with optional edit ranges).
   574→   574→    #[serde(default)]
   575→   575→    pub suggestions: Vec<Suggestion>,
   576→   576→    /// Causal chain: list of diagnostics that caused this one.
   577→   577→    /// The first element is the immediate cause, the last is the root cause.
   578→   578→    #[serde(default)]
   579→   579→    pub chain: Vec<Box<VumaDiagnostic>>,
   580→   580→    /// Backward-compatible plain-text suggestions (stored alongside
   581→   581→    /// structured suggestions for JSON serialization compatibility).
   582→   582→    #[serde(default, skip_serializing_if = "Vec::is_empty")]
   583→   583→    pub legacy_suggestions: Vec<String>,
   584→   584→}
   585→   585→
   586→   586→impl VumaDiagnostic {
   587→   587→    /// Create a new diagnostic with all fields.
   588→   588→    pub fn new(
   589→   589→        code: impl Into<String>,
   590→   590→        severity: DiagnosticSeverity,
   591→   591→        message: impl Into<String>,
   592→   592→        source: impl Into<String>,
   593→   593→        location: DiagnosticSourceLocation,
   594→   594→    ) -> Self {
   595→   595→        Self {
   596→   596→            code: code.into(),
   597→   597→            severity,
   598→   598→            message: message.into(),
   599→   599→            source: source.into(),
   600→   600→            location,
   601→   601→            related: Vec::new(),
   602→   602→            suggestions: Vec::new(),
   603→   603→            chain: Vec::new(),
   604→   604→            legacy_suggestions: Vec::new(),
   605→   605→        }
   606→   606→    }
   607→   607→
   608→   608→    /// Add a related-info entry.
   609→   609→    pub fn with_related(mut self, info: RelatedInfo) -> Self {
   610→   610→        self.related.push(info);
   611→   611→        self
   612→   612→    }
   613→   613→
   614→   614→    /// Add a quick-fix suggestion (plain text, backward-compatible).
   615→   615→    pub fn with_suggestion(mut self, suggestion: impl Into<String>) -> Self {
   616→   616→        self.legacy_suggestions.push(suggestion.into());
   617→   617→        self
   618→   618→    }
   619→   619→
   620→   620→    /// Add a structured suggestion with optional edit range.
   621→   621→    pub fn with_structured_suggestion(mut self, suggestion: Suggestion) -> Self {
   622→   622→        self.suggestions.push(suggestion);
   623→   623→        self
   624→   624→    }
   625→   625→
   626→   626→    /// Chain another diagnostic as the cause of this one.
   627→   627→    ///
   628→   628→    /// The chained diagnostic becomes the immediate cause. Subsequent
   629→   629→    /// calls to `chain()` push further causes, building a causal chain
   630→   630→    /// from immediate cause to root cause.
   631→   631→    ///
   632→   632→    /// # Example
   633→   633→    ///
   634→   634→    /// ```
   635→   635→    /// use vuma::diagnostics::*;
   636→   636→    ///
   637→   637→    /// let root = VumaDiagnostic::new(
   638→   638→    ///     "E023", DiagnosticSeverity::Error,
   639→   639→    ///     "Unknown type 'int' — did you mean 'i32'?",
   640→   640→    ///     "parser", DiagnosticSourceLocation::unknown()
   641→   641→    /// );
   642→   642→    /// let diag = VumaDiagnostic::new(
   643→   643→    ///     "E003", DiagnosticSeverity::Error,
   644→   644→    ///     "Type mismatch", "parser", DiagnosticSourceLocation::unknown()
   645→   645→    /// ).chain(root);
   646→   646→    ///
   647→   647→    /// assert_eq!(diag.chain.len(), 1);
   648→   648→    /// assert_eq!(diag.chain[0].code, "E023");
   649→   649→    /// ```
   650→   650→    pub fn chain(mut self, cause: VumaDiagnostic) -> Self {
   651→   651→        self.chain.push(Box::new(cause));
   652→   652→        self
   653→   653→    }
   654→   654→
   655→   655→    /// Returns the root cause of this diagnostic (the last element in the
   656→   656→    /// causal chain), or `None` if there is no chain.
   657→   657→    pub fn root_cause(&self) -> Option<&VumaDiagnostic> {
   658→   658→        self.chain.last().map(|b| b.as_ref())
   659→   659→    }
   660→   660→
   661→   661→    /// Returns the immediate cause of this diagnostic (the first element
   662→   662→    /// in the causal chain), or `None` if there is no chain.
   663→   663→    pub fn immediate_cause(&self) -> Option<&VumaDiagnostic> {
   664→   664→        self.chain.first().map(|b| b.as_ref())
   665→   665→    }
   666→   666→
   667→   667→    /// Returns the full causal chain as a slice of diagnostics.
   668→   668→    pub fn causal_chain(&self) -> &[Box<VumaDiagnostic>] {
   669→   669→        &self.chain
   670→   670→    }
   671→   671→
   672→   672→    /// Returns `true` if this diagnostic has a causal chain.
   673→   673→    pub fn has_chain(&self) -> bool {
   674→   674→        !self.chain.is_empty()
   675→   675→    }
   676→   676→
   677→   677→    /// Returns all suggestions, combining both structured and legacy
   678→   678→    /// plain-text suggestions.
   679→   679→    pub fn all_suggestions(&self) -> Vec<&Suggestion> {
   680→   680→        self.suggestions.iter().collect()
   681→   681→    }
   682→   682→
   683→   683→    /// Returns `true` if this diagnostic has any machine-applicable
   684→   684→    /// suggestions (i.e. suggestions with edit ranges).
   685→   685→    pub fn has_machine_applicable_fixes(&self) -> bool {
   686→   686→        self.suggestions.iter().any(|s| {
   687→   687→            s.edit_range.is_some()
   688→   688→                && s.replacement.is_some()
   689→   689→                && s.applicability == SuggestionApplicability::MachineApplicable
   690→   690→        })
   691→   691→    }
   692→   692→
   693→   693→    // ── Output formats ────────────────────────────────────────────────
   694→   694→
   695→   695→    /// Serialize this diagnostic as a JSON string.
   696→   696→    pub fn to_json(&self) -> String {
   697→   697→        serde_json::to_string(self).unwrap_or_else(|_| {
   698→   698→            r#"{"code":"E000","severity":"error","message":"JSON serialization failed","source":"diagnostics","location":{"file":"","start_line":0,"start_col":0,"end_line":0,"end_col":0},"related":[],"suggestions":[],"chain":[]}"#.to_string()
   699→   699→        })
   700→   700→    }
   701→   701→
   702→   702→    /// Serialize this diagnostic as a pretty-printed JSON string.
   703→   703→    pub fn to_json_pretty(&self) -> String {
   704→   704→        serde_json::to_string_pretty(self).unwrap_or_else(|_| self.to_json())
   705→   705→    }
   706→   706→
   707→   707→    /// Format this diagnostic as plain text (for logs, no ANSI colors).
   708→   708→    ///
   709→   709→    /// Format: `severity[code]: message (source at location)`
   710→   710→    pub fn to_plain_text(&self) -> String {
   711→   711→        let mut out = format!(
   712→   712→            "{}[{}]: {} ({} at {})",
   713→   713→            self.severity, self.code, self.message, self.source, self.location
   714→   714→        );
   715→   715→        for related in &self.related {
   716→   716→            out.push_str(&format!("\n  note: {} at {}", related.message, related.location));
   717→   717→        }
   718→   718→        for suggestion in &self.suggestions {
   719→   719→            out.push_str(&format!("\n  help: {}", suggestion.message));
   720→   720→            if let Some(ref repl) = suggestion.replacement {
   721→   721→                out.push_str(&format!(" → `{}`", repl));
   722→   722→            }
   723→   723→        }
   724→   724→        for suggestion in &self.legacy_suggestions {
   725→   725→            out.push_str(&format!("\n  help: {}", suggestion));
   726→   726→        }
   727→   727→        for cause in &self.chain {
   728→   728→            out.push_str(&format!(
   729→   729→                "\n  caused by: {}[{}]: {}",
   730→   730→                cause.severity, cause.code, cause.message
   731→   731→            ));
   732→   732→        }
   733→   733→        out
   734→   734→    }
   735→   735→
   736→   736→    /// Format this diagnostic as ANSI-colored terminal output (rich text).
   737→   737→    ///
   738→   738→    /// Uses standard compiler diagnostic coloring:
   739→   739→    /// - Errors in red, warnings in yellow, info in cyan, hints in blue.
   740→   740→    /// - Code and source in bold.
   741→   741→    /// - Suggestions in green.
   742→   742→    /// - Causal chain indented with "caused by" prefix.
   743→   743→    pub fn to_rich_text(&self) -> String {
   744→   744→        let mut out = String::new();
   745→   745→
   746→   746→        // Severity prefix with color
   747→   747→        let (sev_label, sev_code) = match self.severity {
   748→   748→            DiagnosticSeverity::Error => ("\x1b[31merror\x1b[0m", "\x1b[31;1m"),
   749→   749→            DiagnosticSeverity::Warning => ("\x1b[33mwarning\x1b[0m", "\x1b[33;1m"),
   750→   750→            DiagnosticSeverity::Info => ("\x1b[36minfo\x1b[0m", "\x1b[36;1m"),
   751→   751→            DiagnosticSeverity::Hint => ("\x1b[34mhint\x1b[0m", "\x1b[34;1m"),
   752→   752→        };
   753→   753→
   754→   754→        // Main line: severity[code]: message
   755→   755→        out.push_str(&format!(
   756→   756→            "{}[\x1b[1m{}{}\x1b[0m]: {}\n",
   757→   757→            sev_label, sev_code, self.code, self.message
   758→   758→        ));
   759→   759→
   760→   760→        // Location line
   761→   761→        if !self.location.is_unknown() {
   762→   762→            out.push_str(&format!(
   763→   763→                "  \x1b[2m--> {}:{}:{}\x1b[0m\n",
   764→   764→                self.location.file, self.location.start_line, self.location.start_col
   765→   765→            ));
   766→   766→        }
   767→   767→
   768→   768→        // Source indicator
   769→   769→        out.push_str(&format!("  \x1b[2m[{}]\x1b[0m\n", self.source));
   770→   770→
   771→   771→        // Related info
   772→   772→        for related in &self.related {
   773→   773→            out.push_str(&format!(
   774→   774→                "  \x1b[36mnote\x1b[0m: {} \x1b[2m(at {})\x1b[0m\n",
   775→   775→                related.message, related.location
   776→   776→            ));
   777→   777→        }
   778→   778→
   779→   779→        // Structured suggestions
   780→   780→        for suggestion in &self.suggestions {
   781→   781→            out.push_str(&format!(
   782→   782→                "  \x1b[32mhelp\x1b[0m: {}",
   783→   783→                suggestion.message
   784→   784→            ));
   785→   785→            if let Some(ref repl) = suggestion.replacement {
   786→   786→                out.push_str(&format!(" → \x1b[32m`{}`\x1b[0m", repl));
   787→   787→            }
   788→   788→            if suggestion.applicability == SuggestionApplicability::MachineApplicable {
   789→   789→                out.push_str(" \x1b[2m[machine-applicable]\x1b[0m");
   790→   790→            }
   791→   791→            out.push('\n');
   792→   792→        }
   793→   793→
   794→   794→        // Legacy suggestions
   795→   795→        for suggestion in &self.legacy_suggestions {
   796→   796→            out.push_str(&format!("  \x1b[32mhelp\x1b[0m: {}\n", suggestion));
   797→   797→        }
   798→   798→
   799→   799→        // Causal chain (recursively format)
   800→   800→        for (i, cause) in self.chain.iter().enumerate() {
   801→   801→            let indent = "  ".repeat(i + 1);
   802→   802→            let (cause_sev, cause_code) = match cause.severity {
   803→   803→                DiagnosticSeverity::Error => ("\x1b[31merror\x1b[0m", "\x1b[31;1m"),
   804→   804→                DiagnosticSeverity::Warning => ("\x1b[33mwarning\x1b[0m", "\x1b[33;1m"),
   805→   805→                DiagnosticSeverity::Info => ("\x1b[36minfo\x1b[0m", "\x1b[36;1m"),
   806→   806→                DiagnosticSeverity::Hint => ("\x1b[34mhint\x1b[0m", "\x1b[34;1m"),
   807→   807→            };
   808→   808→            out.push_str(&format!(
   809→   809→                "{}\x1b[2mcaused by:\x1b[0m {}[\x1b[1m{}{}\x1b[0m]: {}\n",
   810→   810→                indent, cause_sev, cause_code, cause.code, cause.message
   811→   811→            ));
   812→   812→            if !cause.location.is_unknown() {
   813→   813→                out.push_str(&format!(
   814→   814→                    "{}  \x1b[2m--> {}:{}:{}\x1b[0m\n",
   815→   815→                    indent, cause.location.file, cause.location.start_line, cause.location.start_col
   816→   816→                ));
   817→   817→            }
   818→   818→        }
   819→   819→
   820→   820→        out
   821→   821→    }
   822→   822→
   823→   823→    /// Format this diagnostic as an LSP Diagnostic.
   824→   824→    ///
   825→   825→    /// Returns a JSON `Value` conforming to the LSP Diagnostic structure:
   826→   826→    /// - `range`: 0-based Position range
   827→   827→    /// - `severity`: LSP numeric severity (1=Error, 2=Warning, 3=Info, 4=Hint)
   828→   828→    /// - `code`: the diagnostic code string
   829→   829→    /// - `source`: the compiler stage
   830→   830→    /// - `message`: the human-readable message
   831→   831→    /// - `relatedInformation`: optional related locations
   832→   832→    pub fn to_lsp(&self) -> serde_json::Value {
   833→   833→        use serde_json::json;
   834→   834→
   835→   835→        let start_line = if self.location.start_line > 0 {
   836→   836→            self.location.start_line - 1
   837→   837→        } else {
   838→   838→            0
   839→   839→        };
   840→   840→        let start_col = if self.location.start_col > 0 {
   841→   841→            self.location.start_col - 1
   842→   842→        } else {
   843→   843→            0
   844→   844→        };
   845→   845→        let end_line = if self.location.end_line > 0 {
   846→   846→            self.location.end_line - 1
   847→   847→        } else {
   848→   848→            0
   849→   849→        };
   850→   850→        let end_col = if self.location.end_col > 0 {
   851→   851→            self.location.end_col - 1
   852→   852→        } else {
   853→   853→            0
   854→   854→        };
   855→   855→
   856→   856→        let mut diagnostic = json!({
   857→   857→            "range": {
   858→   858→                "start": { "line": start_line, "character": start_col },
   859→   859→                "end": { "line": end_line, "character": end_col }
   860→   860→            },
   861→   861→            "severity": self.severity.to_lsp_severity(),
   862→   862→            "code": self.code,
   863→   863→            "source": self.source,
   864→   864→            "message": self.message,
   865→   865→        });
   866→   866→
   867→   867→        // Add related information if present
   868→   868→        if !self.related.is_empty() {
   869→   869→            let related: Vec<serde_json::Value> = self.related.iter().map(|r| {
   870→   870→                let r_start_line = if r.location.start_line > 0 { r.location.start_line - 1 } else { 0 };
   871→   871→                let r_start_col = if r.location.start_col > 0 { r.location.start_col - 1 } else { 0 };
   872→   872→                let r_end_line = if r.location.end_line > 0 { r.location.end_line - 1 } else { 0 };
   873→   873→                let r_end_col = if r.location.end_col > 0 { r.location.end_col - 1 } else { 0 };
   874→   874→                json!({
   875→   875→                    "location": {
   876→   876→                        "uri": format!("file://{}", r.location.file),
   877→   877→                        "range": {
   878→   878→                            "start": { "line": r_start_line, "character": r_start_col },
   879→   879→                            "end": { "line": r_end_line, "character": r_end_col }
   880→   880→                        }
   881→   881→                    },
   882→   882→                    "message": r.message
   883→   883→                })
   884→   884→            }).collect();
   885→   885→            diagnostic["relatedInformation"] = json!(related);
   886→   886→        }
   887→   887→
   888→   888→        // Add LSP CodeDescription with href if it's a known code
   889→   889→        let desc = code_description(&self.code);
   890→   890→        if desc != "Unknown diagnostic code" {
   891→   891→            diagnostic["codeDescription"] = json!({
   892→   892→                "href": format!("https://vuma.dev/docs/diagnostics/{}", self.code)
   893→   893→            });
   894→   894→        }
   895→   895→
   896→   896→        // Add tags for warnings (Unnecessary = 1, Deprecated = 2)
   897→   897→        let mut tags: Vec<u32> = Vec::new();
   898→   898→        match self.code.as_str() {
   899→   899→            "W001" | "W004" | "W009" => tags.push(1), // Unnecessary
   900→   900→            "W008" => tags.push(2),                     // Deprecated
   901→   901→            _ => {}
   902→   902→        }
   903→   903→        if !tags.is_empty() {
   904→   904→            diagnostic["tags"] = json!(tags);
   905→   905→        }
   906→   906→
   907→   907→        diagnostic
   908→   908→    }
   909→   909→}
   910→   910→
   911→   911→impl fmt::Display for VumaDiagnostic {
   912→   912→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   913→   913→        write!(
   914→   914→            f,
   915→   915→            "{}[{}]: {} ({} at {})",
   916→   916→            self.severity, self.code, self.message, self.source, self.location
   917→   917→        )?;
   918→   918→        for related in &self.related {
   919→   919→            write!(f, "\n  note: {} at {}", related.message, related.location)?;
   920→   920→        }
   921→   921→        for suggestion in &self.suggestions {
   922→   922→            write!(f, "\n  help: {}", suggestion.message)?;
   923→   923→            if let Some(ref repl) = suggestion.replacement {
   924→   924→                write!(f, " → `{}`", repl)?;
   925→   925→            }
   926→   926→        }
   927→   927→        for suggestion in &self.legacy_suggestions {
   928→   928→            write!(f, "\n  help: {}", suggestion)?;
   929→   929→        }
   930→   930→        for cause in &self.chain {
   931→   931→            write!(
   932→   932→                f,
   933→   933→                "\n  caused by: {}[{}]: {}",
   934→   934→                cause.severity, cause.code, cause.message
   935→   935→            )?;
   936→   936→        }
   937→   937→        Ok(())
   938→   938→    }
   939→   939→}

   959→   959→/// Summary statistics for a collection of diagnostics.
   960→   960→///
   961→   961→/// Counts diagnostics by severity and by individual code, providing
   962→   962→/// a quick overview of the health of a compilation.
   963→   963→#[derive(Debug, Clone, Default, Serialize, Deserialize)]
   964→   964→pub struct DiagnosticSummary {
   965→   965→    /// Total number of diagnostics.
   966→   966→    pub total: usize,
   967→   967→    /// Number of errors.
   968→   968→    pub errors: usize,
   969→   969→    /// Number of warnings.
   970→   970→    pub warnings: usize,
   971→   971→    /// Number of informational messages.
   972→   972→    pub infos: usize,
   973→   973→    /// Number of hints.
   974→   974→    pub hints: usize,
   975→   975→    /// Count per diagnostic code (e.g. `"E001"` → 3).
   976→   976→    pub by_code: HashMap<String, usize>,
   977→   977→    /// Count per source (e.g. `"parser"` → 5).
   978→   978→    pub by_source: HashMap<String, usize>,
   979→   979→    /// Count per subcategory (e.g. `"compilation"` → 3).
   980→   980→    pub by_subcategory: HashMap<String, usize>,
   981→   981→}
   982→   982→
   983→   983→impl DiagnosticSummary {
   984→   984→    /// Create a new empty summary.
   985→   985→    pub fn new() -> Self {
   986→   986→        Self::default()
   987→   987→    }
   988→   988→
   989→   989→    /// Build a summary from a slice of diagnostics.
   990→   990→    pub fn from_diagnostics(diagnostics: &[VumaDiagnostic]) -> Self {
   991→   991→        let mut summary = Self::new();
   992→   992→        for diag in diagnostics {
   993→   993→            summary.add(diag);
   994→   994→        }
   995→   995→        summary
   996→   996→    }
   997→   997→
   998→   998→    /// Add a single diagnostic to the summary.
   999→   999→    pub fn add(&mut self, diag: &VumaDiagnostic) {
  1000→  1000→        self.total += 1;
  1001→  1001→        match diag.severity {
  1002→  1002→            DiagnosticSeverity::Error => self.errors += 1,
  1003→  1003→            DiagnosticSeverity::Warning => self.warnings += 1,
  1004→  1004→            DiagnosticSeverity::Info => self.infos += 1,
  1005→  1005→            DiagnosticSeverity::Hint => self.hints += 1,
  1006→  1006→        }
  1007→  1007→        *self.by_code.entry(diag.code.clone()).or_insert(0) += 1;
  1008→  1008→        *self.by_source.entry(diag.source.clone()).or_insert(0) += 1;
  1009→  1009→        let subcat = code_subcategory(&diag.code).to_string();
  1010→  1010→        *self.by_subcategory.entry(subcat).or_insert(0) += 1;
  1011→  1011→    }
  1012→  1012→
  1013→  1013→    /// Returns `true` if there are any errors.
  1014→  1014→    pub fn has_errors(&self) -> bool {
  1015→  1015→        self.errors > 0
  1016→  1016→    }
  1017→  1017→
  1018→  1018→    /// Returns `true` if there are any warnings.
  1019→  1019→    pub fn has_warnings(&self) -> bool {
  1020→  1020→        self.warnings > 0
  1021→  1021→    }
  1022→  1022→
  1023→  1023→    /// Returns the count for a specific diagnostic code.
  1024→  1024→    pub fn count_for_code(&self, code: &str) -> usize {
  1025→  1025→        self.by_code.get(code).copied().unwrap_or(0)
  1026→  1026→    }
  1027→  1027→
  1028→  1028→    /// Returns the count for a specific source.
  1029→  1029→    pub fn count_for_source(&self, source: &str) -> usize {
  1030→  1030→        self.by_source.get(source).copied().unwrap_or(0)
  1031→  1031→    }
  1032→  1032→
  1033→  1033→    /// Serialize this summary as JSON.
  1034→  1034→    pub fn to_json(&self) -> String {
  1035→  1035→        serde_json::to_string(self).unwrap_or_else(|_| "{}".to_string())
  1036→  1036→    }
  1037→  1037→
  1038→  1038→    /// Serialize this summary as pretty-printed JSON.
  1039→  1039→    pub fn to_json_pretty(&self) -> String {
  1040→  1040→        serde_json::to_string_pretty(self).unwrap_or_else(|_| "{}".to_string())
  1041→  1041→    }
  1042→  1042→}

  1075→  1075→/// Convert a [`ParseError`] to a [`VumaDiagnostic`].
  1076→  1076→///
  1077→  1077→/// `source` is the full source text (used to compute line/column from byte
  1078→  1078→/// offsets). `file` is the optional file path.
  1079→  1079→pub fn from_parse_error(
  1080→  1080→    err: &ParseError,
  1081→  1081→    source: &str,
  1082→  1082→    file: Option<&str>,
  1083→  1083→) -> VumaDiagnostic {
  1084→  1084→    let code = code_for_parse_error_kind(&err.kind).to_string();
  1085→  1085→    let loc = offset_to_location(source, err.span.start, file);
  1086→  1086→
  1087→  1087→    // Compute end position from span end.
  1088→  1088→    let end_loc = if err.span.end > err.span.start && err.span.end <= source.len() {
  1089→  1089→        offset_to_location(source, err.span.end, file)
  1090→  1090→    } else {
  1091→  1091→        loc.clone()
  1092→  1092→    };
  1093→  1093→
  1094→  1094→    let diag_loc = DiagnosticSourceLocation {
  1095→  1095→        file: loc.file.clone().unwrap_or_default(),
  1096→  1096→        start_line: loc.line as u32 + 1, // convert 0-based to 1-based
  1097→  1097→        start_col: loc.column as u32 + 1,
  1098→  1098→        end_line: end_loc.line as u32 + 1,
  1099→  1099→        end_col: end_loc.column as u32 + 1,
  1100→  1100→    };
  1101→  1101→
  1102→  1102→    let mut diag = VumaDiagnostic::new(
  1103→  1103→        code,
  1104→  1104→        DiagnosticSeverity::Error,
  1105→  1105→        &err.message,
  1106→  1106→        "parser",
  1107→  1107→        diag_loc,
  1108→  1108→    );
  1109→  1109→
  1110→  1110→    if let Some(ref suggestion) = err.suggestion {
  1111→  1111→        diag = diag.with_suggestion(format!("did you mean '{}'?", suggestion));
  1112→  1112→    }
  1113→  1113→
  1114→  1114→    // Store the line text as a hint if available.
  1115→  1115→    if let Some(ref line_text) = loc.line_text {
  1116→  1116→        diag = diag.with_suggestion(format!("source: {}", line_text.trim()));
  1117→  1117→    }
  1118→  1118→
  1119→  1119→    diag
  1120→  1120→}

  1135→  1135→/// Convert a [`VumaError`] to one or more [`VumaDiagnostic`]s.

  1300→  1300→/// Convert a [`CodegenError`] to a [`VumaDiagnostic`].
  1301→  1301→pub fn from_codegen_error(err: &CodegenError) -> VumaDiagnostic {
  1302→  1302→    let code = code_for_codegen_error(err).to_string();
  1303→  1303→    let message = err.to_string();
  1304→  1304→
  1305→  1305→    let (source, suggestion) = match err {
  1306→  1306→        CodegenError::UnknownVariable { name } => (
  1307→  1307→            "codegen",
  1308→  1308→            Some(Suggestion::edit(
  1309→  1309→                format!("declare '{}' before use", name),
  1310→  1310→                DiagnosticSourceLocation::unknown(),
  1311→  1311→                format!("let {} = …;", name),
  1312→  1312→            )),
  1313→  1313→        ),
  1314→  1314→        CodegenError::RegisterAllocFailed(_) => (
  1315→  1315→            "register-alloc",
  1316→  1316→            Some(Suggestion::text(
  1317→  1317→                "reduce register pressure by splitting live ranges".to_string(),
  1318→  1318→            )),
  1319→  1319→        ),
  1320→  1320→        CodegenError::TranslationError(_) => (
  1321→  1321→            "ir",
  1322→  1322→            None,
  1323→  1323→        ),
  1324→  1324→        CodegenError::EncodingError(_) => (
  1325→  1325→            "codegen",
  1326→  1326→            None,
  1327→  1327→        ),
  1328→  1328→        CodegenError::ElfError(_) => (
  1329→  1329→            "emit",
  1330→  1330→            None,
  1331→  1331→        ),
  1332→  1332→        CodegenError::InvalidInstruction(_) => (
  1333→  1333→            "codegen",
  1334→  1334→            None,
  1335→  1335→        ),
  1336→  1336→        CodegenError::WasmSectionNotFound { section } => (
  1337→  1337→            "codegen",
  1338→  1338→            Some(Suggestion::text(
  1339→  1339→                format!("ensure '{}' section is generated", section),
  1340→  1340→            )),
  1341→  1341→        ),
  1342→  1342→        CodegenError::UnresolvedRelocation { symbol, function, .. } => (
  1343→  1343→            "codegen",
  1344→  1344→            Some(Suggestion::text(
  1345→  1345→                format!("define function '{}' or provide an external definition for '{}' referenced in '{}'", symbol, symbol, function),
  1346→  1346→            )),
  1347→  1347→        ),
  1348→  1348→    };
  1349→  1349→
  1350→  1350→    let mut diag = VumaDiagnostic::new(
  1351→  1351→        code,
  1352→  1352→        DiagnosticSeverity::Error,
  1353→  1353→        message,
  1354→  1354→        source,
  1355→  1355→        DiagnosticSourceLocation::unknown(),
  1356→  1356→    );
  1357→  1357→

  1367→  1367→// ═══════════════════════════════════════════════════════════════════════════
  1368→  1368→
  1369→  1369→/// Convert a [`MemorySafetyViolation`] into a [`VumaDiagnostic`].
  1370→  1370→///
  1371→  1371→/// Maps each violation to its E041–E050 code and creates a structured
  1372→  1372→/// diagnostic with source location and quick-fix suggestions where
  1373→  1373→/// applicable.
  1374→  1374→pub fn from_memory_safety_violation(violation: &vuma_codegen::MemorySafetyViolation) -> VumaDiagnostic {
  1375→  1375→    use vuma_codegen::MemorySafetyViolation;
  1376→  1376→
  1377→  1377→    let code = violation.code().to_string();
  1378→  1378→    let message = violation.description();
  1379→  1379→    let severity = DiagnosticSeverity::Error;
  1380→  1380→
  1381→  1381→    let (source, suggestion): (&str, Option<Suggestion>) = match violation {
  1382→  1382→        MemorySafetyViolation::UseAfterFree { allocation_name, .. } => (
  1383→  1383→            "memory-safety",
  1384→  1384→            Some(Suggestion::text(format!(
  1385→  1385→                "ensure '{}' is not accessed after it is freed", allocation_name
  1386→  1386→            ))),
  1387→  1387→        ),
  1388→  1388→        MemorySafetyViolation::DoubleFree { allocation_name, .. } => (
  1389→  1389→            "memory-safety",
  1390→  1390→            Some(Suggestion::text(format!(
  1391→  1391→                "'{}' is freed more than once — remove the duplicate free", allocation_name
  1392→  1392→            ))),
  1393→  1393→        ),
  1394→  1394→        MemorySafetyViolation::MemoryLeak { allocation_name, .. } => (
  1395→  1395→            "memory-safety",
  1396→  1396→            Some(Suggestion::text(format!(
  1397→  1397→                "add a 'free({})' before the function returns", allocation_name
  1398→  1398→            ))),
  1399→  1399→        ),
  1400→  1400→        MemorySafetyViolation::BoundsCheckFailure { array_name, .. } => (
  1401→  1401→            "runtime-safety",
  1402→  1402→            Some(Suggestion::text(format!(
  1403→  1403→                "check the index before accessing '{}'", array_name
  1404→  1404→            ))),
  1405→  1405→        ),
  1406→  1406→        MemorySafetyViolation::NullDereference { pointer_name } => (
  1407→  1407→            "memory-safety",
  1408→  1408→            Some(Suggestion::text(format!(
  1409→  1409→                "add a null check before dereferencing '{}'", pointer_name
  1410→  1410→            ))),
  1411→  1411→        ),
  1412→  1412→        MemorySafetyViolation::DanglingPointer { pointer_name, .. } => (
  1413→  1413→            "memory-safety",
  1414→  1414→            Some(Suggestion::text(format!(
  1415→  1415→                "'{}' references a stack allocation that escapes its scope — allocate on the heap instead",
  1416→  1416→                pointer_name
  1417→  1417→            ))),
  1418→  1418→        ),
  1419→  1419→        MemorySafetyViolation::UninitializedRead { variable_name } => (
  1420→  1420→            "memory-safety",
  1421→  1421→            Some(Suggestion::text(format!(
  1422→  1422→                "initialize '{}' before reading it", variable_name
  1423→  1423→            ))),
  1424→  1424→        ),
  1425→  1425→        MemorySafetyViolation::BufferOverflow { buffer_name, .. } => (
  1426→  1426→            "runtime-safety",
  1427→  1427→            Some(Suggestion::text(format!(
  1428→  1428→                "check the offset before writing to '{}'", buffer_name
  1429→  1429→            ))),
  1430→  1430→        ),
  1431→  1431→        MemorySafetyViolation::UseAfterScope { variable_name, .. } => (
  1432→  1432→            "memory-safety",
  1433→  1433→            Some(Suggestion::text(format!(
  1434→  1434→                "'{}' is used after its scope ends — extend the scope or copy the value",
  1435→  1435→                variable_name
  1436→  1436→            ))),
  1437→  1437→        ),
  1438→  1438→        MemorySafetyViolation::InvalidFree { pointer_name, reason } => (
  1439→  1439→            "memory-safety",
  1440→  1440→            Some(Suggestion::text(format!(
  1441→  1441→                "'{}' cannot be freed: {} — check the pointer origin", pointer_name, reason
  1442→  1442→            ))),
  1443→  1443→        ),
  1444→  1444→    };
  1445→  1445→
  1446→  1446→    let mut diag = VumaDiagnostic::new(
  1447→  1447→        code,
  1448→  1448→        severity,
  1449→  1449→        message,
  1450→  1450→        source,
  1451→  1451→        DiagnosticSourceLocation::unknown(),
  1452→  1452→    );
  1453→  1453→
  1454→  1454→    if let Some(s) = suggestion {
  1455→  1455→        diag = diag.with_structured_suggestion(s);
  1456→  1456→    }
  1457→  1457→
  1458→  1458→    diag
  1459→  1459→}

  1461→  1461→/// Convert all violations from a [`MemorySafetyReport`] into diagnostics.
  1462→  1462→pub fn from_memory_safety_report(report: &vuma_codegen::MemorySafetyReport) -> Vec<VumaDiagnostic> {
  1463→  1463→    report

  1465→  1465→        .iter()
  1466→  1466→        .map(from_memory_safety_violation)
  1467→  1467→        .collect()
  1468→  1468→}
