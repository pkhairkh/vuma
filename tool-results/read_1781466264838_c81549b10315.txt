     1→//! VUMA CLI — Command-line interface for the VUMA compiler.
     2→//!
     3→//! Subcommands:
     4→//! - `vuma build <file>` — Parse + compile to ARM64 ELF (default), save to output file
     5→//! - `vuma run <file>`   — Build + execute (via QEMU aarch64 or native)
     6→//! - `vuma check <file>` — Parse + SCG + BD inference + IVE verification only
     7→//! - `vuma emit <isa> <file>` — Compile to specific ISA
     8→//! - `vuma disasm <file>` — Read binary and disassemble
     9→//! - `vuma verify <file>` — Run IVE 5-invariant verification
    10→//! - `vuma repl` — Interactive REPL (parse expr, print AST)

   230→/// Read source file content with a human-readable error on failure.
   231→fn read_source(path: &PathBuf) -> Result<String, String> {
   232→    fs::read_to_string(path)
   233→        .map_err(|e| format!("error: cannot read source file '{}': {}", path.display(), e))

   247→/// Print compilation errors to stderr in a human-readable format.
   248→fn print_errors(errors: &[VumaError]) {
   249→    for err in errors {
   250→        eprintln!("error[{}]: {}", err.stage(), err);
   251→    }

   254→/// Determine the default output path for a given input file.
   255→fn default_output_path(input: &Path) -> PathBuf {
   256→    let stem = input.file_stem().unwrap_or_default().to_string_lossy();
   257→    let dir = input.parent().unwrap_or(std::path::Path::new("."));
   258→    dir.join(format!("{}.o", stem))
   259→}

   265→/// `vuma build <file>` — Parse + compile to ELF, save to output file.
   266→fn cmd_build(
   267→    cli: &Cli,
   268→    file: &PathBuf,
   269→    output: &Option<PathBuf>,
   270→    target: TargetArg,
   271→) -> Result<(), String> {
   272→    let source = read_source(file)?;
   273→    let config = make_config(cli, CompileTarget::from(target));
   274→    let result = compile(&source, &config).map_err(|errors| {
   275→        print_errors(&errors);
   276→        format!("compilation failed with {} error(s)", errors.len())
   277→    })?;
   278→
   279→    let out_path = output
   280→        .as_ref()
   281→        .cloned()
   282→        .unwrap_or_else(|| default_output_path(file));
   283→    fs::write(&out_path, &result.binary).map_err(|e| {
   284→        format!(
   285→            "error: cannot write output file '{}': {}",
   286→            out_path.display(),
   287→            e
   288→        )
   289→    })?;
   290→    // Make the output file executable on Unix.
   291→    #[cfg(unix)]
   292→    {
   293→        use std::os::unix::fs::PermissionsExt;
   294→        let mut perms = fs::metadata(&out_path)
   295→            .map_err(|e| format!("error: cannot stat '{}': {}", out_path.display(), e))?
   296→            .permissions();
   297→        perms.set_mode(0o755);
   298→        fs::set_permissions(&out_path, perms)
   299→            .map_err(|e| format!("error: cannot chmod '{}': {}", out_path.display(), e))?;
   300→    }
   301→
   302→    println!(
   303→        "Compiled {} -> {} ({} bytes, {} SCG nodes, {} IR instructions)",
   304→        file.display(),
   305→        out_path.display(),
   306→        result.binary.len(),
   307→        result.scg.node_count(),
   308→        result.ir_instruction_count,
   309→    );
   310→
   311→    // Print stage timings.
   312→    for (stage, ms) in &result.stage_timings {
   313→        println!("  {:20} {}ms", stage, ms);
   314→    }
   315→
   316→    Ok(())
   317→}

   319→/// `vuma run <file>` — Build + execute.
   320→fn cmd_run(cli: &Cli, file: &PathBuf, args: &[String]) -> Result<(), String> {
   321→    let source = read_source(file)?;
   322→    let config = make_config(cli, CompileTarget::Pi5Linux);
   323→    let result = compile(&source, &config).map_err(|errors| {
   324→        print_errors(&errors);
   325→        format!("compilation failed with {} error(s)", errors.len())
   326→    })?;
   327→
   328→    // Write binary to a temp file.
   329→    let tmp_dir = std::env::temp_dir();
   330→    let exe_path = tmp_dir.join(format!("vuma_run_{}", std::process::id()));
   331→    fs::write(&exe_path, &result.binary).map_err(|e| {
   332→        format!(
   333→            "error: cannot write temporary executable '{}': {}",
   334→            exe_path.display(),
   335→            e
   336→        )
   337→    })?;
   338→
   339→    // Make the file executable on Unix.
   340→    #[cfg(unix)]
   341→    {
   342→        use std::os::unix::fs::PermissionsExt;
   343→        let mut perms = fs::metadata(&exe_path)
   344→            .map_err(|e| format!("error: cannot stat '{}': {}", exe_path.display(), e))?
   345→            .permissions();
   346→        perms.set_mode(0o755);
   347→        fs::set_permissions(&exe_path, perms)
   348→            .map_err(|e| format!("error: cannot chmod '{}': {}", exe_path.display(), e))?;
   349→    }
   350→
   351→    // Try to execute: first natively (if on aarch64), then via qemu-aarch64.
   352→    let native_output = Command::new(&exe_path).args(args).output();
   353→
   354→    let output = match native_output {
   355→        Ok(out) => out,
   356→        Err(_) => {
   357→            // Native execution failed; try qemu-aarch64.
   358→            let qemu_output = Command::new("qemu-aarch64")
   359→                .arg(&exe_path)
   360→                .args(args)
   361→                .output()
   362→                .map_err(|e| {
   363→                    format!(
   364→                        "error: failed to execute binary (neither native nor qemu-aarch64): {}",
   365→                        e
   366→                    )
   367→                })?;
   368→            qemu_output
   369→        }
   370→    };
   371→
   372→    io::stdout()
   373→        .write_all(&output.stdout)
   374→        .map_err(|e| format!("error: failed to write program output: {}", e))?;
   375→    io::stderr()
   376→        .write_all(&output.stderr)
   377→        .map_err(|e| format!("error: failed to write program stderr: {}", e))?;
   378→
   379→    // Clean up temp file.
   380→    let _ = fs::remove_file(&exe_path);
   381→
   382→    let code = output.status.code().unwrap_or(1);
   383→    if code != 0 {
   384→        return Err(format!("program exited with code {}", code));
   385→    }
   386→
   387→    Ok(())
   388→}

   390→/// `vuma check <file>` — Parse + SCG + BD inference + IVE verification only.
   391→fn cmd_check(cli: &Cli, file: &PathBuf) -> Result<(), String> {
   392→    let source = read_source(file)?;
   393→    let mut config = make_config(cli, CompileTarget::Pi5Linux);
   394→    // check mode: always run verification, don't skip
   395→    if config.verification_level == VerificationLevel::None {
   396→        config.verification_level = VerificationLevel::Normal;
   397→    }
   398→
   399→    // Run the full compile — the check command doesn't save the binary,
   400→    // it just verifies the program compiles and passes verification.
   401→    let result = compile(&source, &config).map_err(|errors| {
   402→        print_errors(&errors);
   403→        format!("check failed with {} error(s)", errors.len())
   404→    })?;
   405→
   406→    println!("Check passed for {}", file.display());
   407→    println!("  SCG nodes:      {}", result.scg.node_count());
   408→    println!("  IR functions:   {}", result.ir_function_count);
   409→    println!("  IR instructions: {}", result.ir_instruction_count);
   410→
   411→    if let Some(ref verification) = result.verification {
   412→        println!("  Verification:   {}", verification.overall);
   413→    }
   414→
   415→    // Print stage timings.
   416→    for (stage, ms) in &result.stage_timings {
   417→        println!("  {:20} {}ms", stage, ms);
   418→    }
   419→
   420→    Ok(())
   421→}

   423→/// `vuma emit <isa> <file>` — Compile to a specific ISA target.
   424→///
   425→/// Uses the direct AST → codegen SCG → IR path for better code quality
   426→/// than the main pipeline (which goes through the semantic SCG and loses
   427→/// most program semantics in the SCG→IR bridge).
   428→fn cmd_emit(
   429→    cli: &Cli,
   430→    isa: IsaArg,
   431→    file: &PathBuf,
   432→    output: &Option<PathBuf>,
   433→) -> Result<(), String> {
   434→    let source = read_source(file)?;
   435→    let backend_kind = BackendKind::from(isa);
   436→
   437→    // Step 1: Parse source → AST.
   438→    let mut parser = vuma_parser::Parser::new(&source);
   439→    let parse_result = parser.parse_program();
   440→    if parse_result.is_err() {
   441→        return Err(format!("parse error: {:?}", parse_result.errors));
   442→    }
   443→    if !parse_result.errors.is_empty() {
   444→        eprintln!("[emit] WARNING: {} non-fatal parse errors:", parse_result.errors.len());
   445→        for err in &parse_result.errors {
   446→            eprintln!("[emit]   {:?}", err);
   447→        }
   448→    }
   449→    let program = parse_result.value.unwrap();
   450→
   451→    // Step 2: Bridge parser AST → codegen SCG.
   452→    let codegen_scg = bridge_ast_to_codegen_scg(&program);
   453→
   454→    // Step 3: Lower codegen SCG → IR.
   455→    let mut ir_builder = ScgToIr::new();
   456→    let ir_program = ir_builder.convert(&codegen_scg).map_err(|e| {
   457→        format!("IR conversion error: {}", e)
   458→    })?;
   459→
   460→    eprintln!("[emit] IR program has {} functions", ir_program.functions.len());
   461→    for func in &ir_program.functions {
   462→        eprintln!("[emit] Function: {} ({} params, {} vregs)", func.name, func.params.len(), func.vregs.len());
   463→        for block in &func.blocks {
   464→            eprintln!("[emit]   Block: {}", block.label);
   465→            for instr in &block.instructions {
   466→                eprintln!("[emit]     {:?}", instr);
   467→            }
   468→        }
   469→    }
   470→
   471→    // Step 4: Create backend and allocate registers.
   472→    let backend = create_backend(backend_kind).map_err(|e| {
   473→        format!("error: cannot create {} backend: {}", backend_kind.isa_name(), e)
   474→    })?;
   475→
   476→    let mut allocated_functions = Vec::new();
   477→    for func in &ir_program.functions {
   478→        match backend.allocate_registers(func) {
   479→            Ok(allocated) => allocated_functions.push(allocated),
   480→            Err(e) => {
   481→                eprintln!("warning: register allocation failed for '{}': {}", func.name, e);
   482→            }
   483→        }
   484→    }
   485→
   486→    let out_path = output
   487→        .as_ref()
   488→        .cloned()
   489→        .unwrap_or_else(|| default_output_path(file));
   490→
   491→    // Step 5: Encode and write output.
   492→    if !allocated_functions.is_empty() {
   493→        let allocated_program = vuma_codegen::backend::AllocatedProgram {
   494→            functions: allocated_functions,
   495→            total_code_size: 0,
   496→            total_data_size: 0,
   497→        };
   498→        match backend.encode_program(&allocated_program) {
   499→            Ok(bytes) => {
   500→                fs::write(&out_path, &bytes).map_err(|e| {
   501→                    format!("error: cannot write output file '{}': {}", out_path.display(), e)
   502→                })?;
   503→                // Make the output file executable on Unix.
   504→                #[cfg(unix)]
   505→                {
   506→                    use std::os::unix::fs::PermissionsExt;
   507→                    let mut perms = fs::metadata(&out_path)
   508→                        .map_err(|e| format!("error: cannot stat '{}': {}", out_path.display(), e))?
   509→                        .permissions();
   510→                    perms.set_mode(0o755);
   511→                    fs::set_permissions(&out_path, perms)
   512→                        .map_err(|e| format!("error: cannot chmod '{}': {}", out_path.display(), e))?;
   513→                }
   514→                println!(
   515→                    "Emitted {} -> {} ({} bytes, ISA: {})",
   516→                    file.display(),
   517→                    out_path.display(),
   518→                    bytes.len(),
   519→                    backend.name(),
   520→                );
   521→            }
   522→            Err(e) => {
   523→                return Err(format!("{} encoding failed: {}", backend.name(), e));
   524→            }
   525→        }
   526→    } else {
   527→        return Err("no functions were successfully allocated".to_string());
   528→    }
   529→
   530→    Ok(())
   531→}
   532→
   533→/// Bridge parser AST → codegen SCG.
   534→///
   535→/// This converts the parser's AST into the codegen's SCG representation,
   536→/// which can then be lowered to IR. This bypasses the main pipeline's
   537→/// semantic SCG which loses most program semantics.
   538→///
   539→/// Key design: expressions are flattened into three-address code (TAC) by
   540→/// introducing temporary variables for sub-expressions. This preserves the
   541→/// full semantics of nested binary operations, function calls, dereferences,
   542→/// and casts.
   543→fn bridge_ast_to_codegen_scg(program: &vuma_parser::ast::Program) -> Scg {
   544→    use vuma_parser::ast::Item;
   545→
   546→    let mut nodes: Vec<ScgNode> = Vec::new();
   547→
   548→    for item in &program.items {
   549→        if let Item::FnDef(fn_def) = item {
   550→            let params: Vec<ScgParam> = fn_def
   551→                .params
   552→                .iter()
   553→                .map(|p| ScgParam {
   554→                    name: p.name.clone(),
   555→                    ty: bridge_type_to_codegen_scg(&p.ty),
   556→                })
   557→                .collect();
   558→
   559→            let results = if let Some(ref ret_ty) = fn_def.return_type {
   560→                vec![bridge_type_to_codegen_scg(&Some(ret_ty.clone()))]
   561→            } else {
   562→                vec![]
   563→            };
   564→
   565→            let mut ctx = BridgeCtx::new();
   566→            let mut body = bridge_block_to_scg_stmts(&fn_def.body, &mut ctx);
   567→
   568→            // Ensure every function ends with a Return statement.
   569→            // If the body doesn't end with a Return, add an implicit one.
   570→            // When the function has a return type and the last statement was an
   571→            // expression, use ctx.last_expr_result as the return value.
   572→            let has_return = body.last().map_or(false, |s| matches!(s, ScgStatement::Return(_)));
   573→            if !has_return {
   574→                let ret_val = if !results.is_empty() {
   575→                    // First check if the last expression was tracked.
   576→                    if let Some(ref expr) = ctx.last_expr_result {
   577→                        Some(expr.clone())
   578→                    } else {
   579→                        // Otherwise, look for the last computation/call result.
   580→                        body.iter().rev().find_map(|s| match s {
   581→                            ScgStatement::Computation(comp) => Some(ScgExpr::Var(comp.dst.clone())),
   582→                            ScgStatement::Call(call) => call.dst.as_ref().map(|d| ScgExpr::Var(d.clone())),
   583→                            _ => None,
   584→                        })
   585→                    }
   586→                } else {
   587→                    None
   588→                };
   589→                body.push(ScgStatement::Return(ret_val.into_iter().collect()));
   590→            }
   591→
   592→            nodes.push(ScgNode::Function(ScgFunction {
   593→                name: fn_def.name.clone(),
   594→                params,
   595→                results,
   596→                body,
   597→            }));
   598→        }
   599→    }
   600→
   601→    Scg { nodes }
   602→}
   603→
   604→/// Context for the AST → codegen SCG bridge, tracking a monotonic temp counter
   605→/// and the last expression result for implicit returns.
   606→struct BridgeCtx {
   607→    temp_counter: u32,
   608→    /// The result of the last expression statement, if any.
   609→    /// Used for implicit return when a function body ends with an expression.
   610→    last_expr_result: Option<ScgExpr>,
   611→}
   612→
   613→impl BridgeCtx {
   614→    fn new() -> Self {
   615→        Self { temp_counter: 0, last_expr_result: None }
   616→    }
   617→
   618→    /// Allocate a unique temporary variable name.
   619→    fn alloc_temp(&mut self) -> String {
   620→        let name = format!("__t{}", self.temp_counter);
   621→        self.temp_counter += 1;
   622→        name
   623→    }
   624→}

   626→/// Convert a parser type annotation to a codegen SCG type.
   627→fn bridge_type_to_codegen_scg(ty: &Option<vuma_parser::ast::Type>) -> ScgType {
   628→    match ty {
   629→        Some(vuma_parser::ast::Type::BDBase(name)) => match name.as_str() {
   630→            "i8" => ScgType::I8,
   631→            "i16" => ScgType::I16,
   632→            "i32" => ScgType::I32,
   633→            "i64" => ScgType::I64,
   634→            "u8" => ScgType::U8,
   635→            "u16" => ScgType::U16,
   636→            "u32" => ScgType::U32,
   637→            "u64" => ScgType::U64,
   638→            _ => ScgType::I64,
   639→        },
   640→        Some(vuma_parser::ast::Type::Ptr(_)) => ScgType::Ptr,
   641→        Some(vuma_parser::ast::Type::RegionPtr { .. }) => ScgType::Ptr,
   642→        _ => ScgType::Void,
   643→    }
   644→}

   647→/// Convert a parser block into codegen SCG statements, flattening expressions
   648→/// into three-address code with temporaries.
   649→fn bridge_block_to_scg_stmts(block: &vuma_parser::ast::Block, ctx: &mut BridgeCtx) -> Vec<ScgStatement> {
   650→    block
   651→        .statements
   652→        .iter()
   653→        .flat_map(|s| bridge_stmt_to_scg(s, ctx))
   654→        .collect()
   655→}

   657→/// Flatten an expression into three-address code. Returns the ScgExpr that
   658→/// holds the result, and appends any intermediate computation statements
   659→/// to `stmts`.
   660→///
   661→/// This is the core of the bridge: it recursively decomposes nested
   662→/// expressions into a sequence of simple computation nodes, each operating
   663→/// on at most two operands and producing one result. This preserves the
   664→/// full semantics of the original expression tree.
   665→fn flatten_expr(
   666→    expr: &vuma_parser::ast::Expr,
   667→    stmts: &mut Vec<ScgStatement>,
   668→    ctx: &mut BridgeCtx,
   669→) -> ScgExpr {
   670→    use vuma_parser::ast::{Expr, Lit, UnOp};
   671→
   672→    match expr {
   673→        // ── Leaf expressions: return directly, no flattening needed ──
   674→        Expr::Var { name, .. } => ScgExpr::Var(name.clone()),
   675→        Expr::Lit { value, .. } => match value {
   676→            Lit::Int(n) => ScgExpr::Int(*n),
   677→            Lit::Float(f) => ScgExpr::Float(*f),
   678→            Lit::Bool(b) => ScgExpr::Int(if *b { 1 } else { 0 }),
   679→            Lit::Address(a) => ScgExpr::Int(*a as i64),
   680→            Lit::String(_) => ScgExpr::Int(0),
   681→        },
   682→
   683→        // ── Binary operations: flatten lhs and rhs, then emit one Computation ──
   684→        Expr::BinOp { op, lhs, rhs, .. } => {
   685→            let lhs_expr = flatten_expr(lhs, stmts, ctx);
   686→            let rhs_expr = flatten_expr(rhs, stmts, ctx);
   687→            let dst = ctx.alloc_temp();
   688→            let binop_kind = map_ast_binop(op);
   689→            stmts.push(ScgStatement::Computation(ComputationNode {
   690→                dst: dst.clone(),
   691→                op: binop_kind,
   692→                lhs: lhs_expr,
   693→                rhs: rhs_expr,
   694→                tail_call: false,
   695→            }));
   696→            ScgExpr::Var(dst)
   697→        }
   698→
   699→        // ── Unary operations: flatten operand, then emit one Computation ──
   700→        Expr::UnOp { op, expr: operand, .. } => {
   701→            let operand_expr = flatten_expr(operand, stmts, ctx);
   702→            let dst = ctx.alloc_temp();
   703→            match op {
   704→                UnOp::BitNot => {
   705→                    stmts.push(ScgStatement::Computation(ComputationNode {
   706→                        dst: dst.clone(),
   707→                        op: BinOpKind::Xor,
   708→                        lhs: operand_expr,
   709→                        rhs: ScgExpr::Int(-1),
   710→                        tail_call: false,
   711→                    }));
   712→                }
   713→                UnOp::Neg => {
   714→                    stmts.push(ScgStatement::Computation(ComputationNode {
   715→                        dst: dst.clone(),
   716→                        op: BinOpKind::Sub,
   717→                        lhs: ScgExpr::Int(0),
   718→                        rhs: operand_expr,
   719→                        tail_call: false,
   720→                    }));
   721→                }
   722→                UnOp::Not => {
   723→                    stmts.push(ScgStatement::Computation(ComputationNode {
   724→                        dst: dst.clone(),
   725→                        op: BinOpKind::Eq,
   726→                        lhs: operand_expr,
   727→                        rhs: ScgExpr::Int(0),
   728→                        tail_call: false,
   729→                    }));
   730→                }
   731→                UnOp::Deref => {
   732→                    stmts.push(ScgStatement::Access(AccessNode::Load {
   733→                        dst: dst.clone(),
   734→                        ptr: operand_expr,
   735→                        offset: None,
   736→                    }));
   737→                }
   738→            }
   739→            ScgExpr::Var(dst)
   740→        }
   741→
   742→        // ── Function call: flatten args, emit CallNode ──
   743→        Expr::Call { callee, args, .. } => {
   744→            let func_name = match callee.as_ref() {
   745→                Expr::Var { name, .. } => name.clone(),
   746→                _ => "_unknown".into(),
   747→            };
   748→            let flat_args: Vec<ScgExpr> = args.iter()
   749→                .map(|a| flatten_expr(a, stmts, ctx))
   750→                .collect();
   751→            let dst = ctx.alloc_temp();
   752→            stmts.push(ScgStatement::Call(CallNode {
   753→                dst: Some(dst.clone()),
   754→                func: func_name,
   755→                args: flat_args,
   756→            }));
   757→            ScgExpr::Var(dst)
   758→        }
   759→
   760→        // ── Dereference: flatten the address, emit Load ──
   761→        Expr::Deref { expr, .. } => {
   762→            let addr = flatten_expr(expr, stmts, ctx);
   763→            let dst = ctx.alloc_temp();
   764→            stmts.push(ScgStatement::Access(AccessNode::Load {
   765→                dst: dst.clone(),
   766→                ptr: addr,
   767→                offset: None,
   768→            }));
   769→            ScgExpr::Var(dst)
   770→        }
   771→
   772→        // ── Offset (pointer arithmetic): flatten base and offset, emit Add ──
   773→        Expr::Offset { base, offset, .. } => {
   774→            let base_expr = flatten_expr(base, stmts, ctx);
   775→            let off_expr = flatten_expr(offset, stmts, ctx);
   776→            let dst = ctx.alloc_temp();
   777→            stmts.push(ScgStatement::Computation(ComputationNode {
   778→                dst: dst.clone(),
   779→                op: BinOpKind::Add,
   780→                lhs: base_expr,
   781→                rhs: off_expr,
   782→                tail_call: false,
   783→            }));
   784→            ScgExpr::Var(dst)
   785→        }
   786→
   787→        // ── Cast: flatten operand, pass through ──
   788→        Expr::Cast { expr, .. } => flatten_expr(expr, stmts, ctx),
   789→
   790→        // ── TypeAscription: flatten inner expression ──
   791→        Expr::TypeAscription { expr, .. } => flatten_expr(expr, stmts, ctx),
   792→
   793→        // ── Index: flatten base and index, compute addr, emit Load ──
   794→        Expr::Index { expr, index, .. } => {
   795→            let base_expr = flatten_expr(expr, stmts, ctx);
   796→            let idx_expr = flatten_expr(index, stmts, ctx);
   797→            let addr = ctx.alloc_temp();
   798→            stmts.push(ScgStatement::Computation(ComputationNode {
   799→                dst: addr.clone(),
   800→                op: BinOpKind::Add,
   801→                lhs: base_expr,
   802→                rhs: idx_expr,
   803→                tail_call: false,
   804→            }));
   805→            let dst = ctx.alloc_temp();
   806→            stmts.push(ScgStatement::Access(AccessNode::Load {
   807→                dst: dst.clone(),
   808→                ptr: ScgExpr::Var(addr),
   809→                offset: None,
   810→            }));
   811→            ScgExpr::Var(dst)
   812→        }
   813→
   814→        // ── Range: just flatten start (range is handled by For) ──
   815→        Expr::Range { start, .. } => flatten_expr(start, stmts, ctx),
   816→
   817→        // ── Allocate: emit as a heap allocation call ──
   818→        Expr::Allocate { size, .. } => {
   819→            let size_expr = flatten_expr(size, stmts, ctx);
   820→            let dst = ctx.alloc_temp();
   821→            stmts.push(ScgStatement::Call(CallNode {
   822→                dst: Some(dst.clone()),
   823→                func: "__vuma_alloc".to_string(),
   824→                args: vec![size_expr],
   825→            }));
   826→            ScgExpr::Var(dst)
   827→        }
   828→
   829→        // ── Null → 0 ──
   830→        Expr::Null { .. } => ScgExpr::Int(0),
   831→
   832→        // ── Uninitialized → 0 ──
   833→        Expr::Uninitialized { .. } => ScgExpr::Int(0),
   834→
   835→        // ── Fallback for unsupported expression types ──
   836→        _ => ScgExpr::Int(0),
   837→    }
   838→}
   839→
   840→/// Map a VUMA AST BinOp to a codegen BinOpKind.
   841→fn map_ast_binop(op: &vuma_parser::ast::BinOp) -> BinOpKind {
   842→    use vuma_parser::ast::BinOp;
   843→    match op {
   844→        BinOp::Add => BinOpKind::Add,
   845→        BinOp::Sub => BinOpKind::Sub,
   846→        BinOp::Mul => BinOpKind::Mul,
   847→        BinOp::Div => BinOpKind::SDiv,
   848→        BinOp::Mod => BinOpKind::SRem,
   849→        BinOp::And => BinOpKind::And,
   850→        BinOp::Or => BinOpKind::Or,
   851→        BinOp::BitAnd => BinOpKind::And,
   852→        BinOp::BitOr => BinOpKind::Or,
   853→        BinOp::BitXor => BinOpKind::Xor,
   854→        BinOp::Shl => BinOpKind::Shl,
   855→        BinOp::Shr => BinOpKind::ShrL,
   856→        BinOp::Eq => BinOpKind::Eq,
   857→        BinOp::Ne => BinOpKind::Ne,
   858→        BinOp::Lt => BinOpKind::SLt,
   859→        BinOp::Le => BinOpKind::SLe,
   860→        BinOp::Gt => BinOpKind::SGt,
   861→        BinOp::Ge => BinOpKind::SGe,
   862→    }
   863→}

   865→/// Convert a single parser statement into zero or more codegen SCG statements.
   866→/// Uses `flatten_expr` to decompose nested expressions into three-address code.
   867→fn bridge_stmt_to_scg(stmt: &vuma_parser::ast::Stmt, ctx: &mut BridgeCtx) -> Vec<ScgStatement> {
   868→    use vuma_parser::ast::Stmt as PStmt;
   869→
   870→    match stmt {
   871→        // ── let x [: T] = expr ──
   872→        PStmt::Let(let_stmt) => {
   873→            let mut stmts = Vec::new();
   874→
   875→            // Check if the RHS is an allocate() call → AllocationNode::Stack
   876→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &let_stmt.value {
   877→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
   878→                    if name == "allocate" {
   879→                        let size: u32 = args.first()
   880→                            .and_then(|a| {
   881→                                if let vuma_parser::ast::Expr::Lit { value, .. } = a {
   882→                                    if let vuma_parser::ast::Lit::Int(n) = value {
   883→                                        return Some(*n as u32);
   884→                                    }
   885→                                }
   886→                                None
   887→                            })
   888→                            .unwrap_or(8);
   889→                        return vec![ScgStatement::Allocation(AllocationNode::Stack {
   890→                            name: let_stmt.name.clone(),
   891→                            size,
   892→                            ty: ScgType::Ptr,
   893→                        })];
   894→                    }
   895→                    // Other function calls → CallNode (flatten args)
   896→                    let flat_args: Vec<ScgExpr> = args.iter()
   897→                        .map(|a| flatten_expr(a, &mut stmts, ctx))
   898→                        .collect();
   899→                    stmts.push(ScgStatement::Call(CallNode {
   900→                        dst: Some(let_stmt.name.clone()),
   901→                        func: name.clone(),
   902→                        args: flat_args,
   903→                    }));
   904→                    return stmts;
   905→                }
   906→            }
   907→
   908→            // Check if the RHS is an Allocate expression → AllocationNode::Stack
   909→            if let vuma_parser::ast::Expr::Allocate { size, .. } = &let_stmt.value {
   910→                let size_val: u32 = match size.as_ref() {
   911→                    vuma_parser::ast::Expr::Lit { value, .. } => {
   912→                        if let vuma_parser::ast::Lit::Int(n) = value {
   913→                            *n as u32
   914→                        } else {
   915→                            8
   916→                        }
   917→                    }
   918→                    _ => 8,
   919→                };
   920→                return vec![ScgStatement::Allocation(AllocationNode::Stack {
   921→                    name: let_stmt.name.clone(),
   922→                    size: size_val,
   923→                    ty: ScgType::Ptr,
   924→                })];
   925→            }
   926→
   927→            // General case: flatten the expression and assign to dst
   928→            let result = flatten_expr(&let_stmt.value, &mut stmts, ctx);
   929→            match &result {
   930→                ScgExpr::Var(name) if name == &let_stmt.name => {}
   931→                _ => {
   932→                    stmts.push(ScgStatement::Computation(ComputationNode {
   933→                        dst: let_stmt.name.clone(),
   934→                        op: BinOpKind::Add,
   935→                        lhs: result,
   936→                        rhs: ScgExpr::Int(0),
   937→                        tail_call: false,
   938→                    }));
   939→                }
   940→            }
   941→            stmts
   942→        }
   943→
   944→        // ── target = value ──
   945→        PStmt::Assign(assign_stmt) => {
   946→            let mut stmts = Vec::new();
   947→
   948→            // Detect dereference writes: `*expr = val` → Access::Store
   949→            if let vuma_parser::ast::AssignTarget::Deref { expr, .. } = &assign_stmt.target {
   950→                let ptr = flatten_expr(expr, &mut stmts, ctx);
   951→                let value = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
   952→                stmts.push(ScgStatement::Access(AccessNode::Store {
   953→                    ptr,
   954→                    offset: None,
   955→                    value,
   956→                }));
   957→                return stmts;
   958→            }
   959→
   960→            // Handle Index target: ptr[index] = value
   961→            if let vuma_parser::ast::AssignTarget::Index { expr, index, .. } = &assign_stmt.target {
   962→                let base = flatten_expr(expr, &mut stmts, ctx);
   963→                let idx = flatten_expr(index, &mut stmts, ctx);
   964→                let addr = ctx.alloc_temp();
   965→                stmts.push(ScgStatement::Computation(ComputationNode {
   966→                    dst: addr.clone(),
   967→                    op: BinOpKind::Add,
   968→                    lhs: base,
   969→                    rhs: idx,
   970→                    tail_call: false,
   971→                }));
   972→                let value = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
   973→                stmts.push(ScgStatement::Access(AccessNode::Store {
   974→                    ptr: ScgExpr::Var(addr),
   975→                    offset: None,
   976→                    value,
   977→                }));
   978→                return stmts;
   979→            }
   980→
   981→            let dst = match &assign_stmt.target {
   982→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
   983→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
   984→                vuma_parser::ast::AssignTarget::Deref { .. } => "_deref".into(),
   985→                vuma_parser::ast::AssignTarget::Index { .. } => "_index".into(),
   986→            };
   987→
   988→            // Detect allocate() expression → AllocationNode::Stack
   989→            if let vuma_parser::ast::Expr::Allocate { size, .. } = &assign_stmt.value {
   990→                let size_val: u32 = match size.as_ref() {
   991→                    vuma_parser::ast::Expr::Lit { value, .. } => {
   992→                        if let vuma_parser::ast::Lit::Int(n) = value {
   993→                            *n as u32
   994→                        } else {
   995→                            8
   996→                        }
   997→                    }
   998→                    _ => 8,
   999→                };
  1000→                return vec![ScgStatement::Allocation(AllocationNode::Stack {
  1001→                    name: dst,
  1002→                    size: size_val,
  1003→                    ty: ScgType::Ptr,
  1004→                })];
  1005→            }
  1006→
  1007→            // Detect function calls in assign: `x = foo(args)` → CallNode (flatten args)
  1008→            if let vuma_parser::ast::Expr::Call { callee, args, .. } = &assign_stmt.value {
  1009→                if let vuma_parser::ast::Expr::Var { name, .. } = callee.as_ref() {
  1010→                    let flat_args: Vec<ScgExpr> = args.iter()
  1011→                        .map(|a| flatten_expr(a, &mut stmts, ctx))
  1012→                        .collect();
  1013→                    stmts.push(ScgStatement::Call(CallNode {
  1014→                        dst: Some(dst),
  1015→                        func: name.clone(),
  1016→                        args: flat_args,
  1017→                    }));
  1018→                    return stmts;
  1019→                }
  1020→            }
  1021→
  1022→            // General case: flatten the value expression and assign to dst
  1023→            let result = flatten_expr(&assign_stmt.value, &mut stmts, ctx);
  1024→            match &result {
  1025→                ScgExpr::Var(name) if name == &dst => {}
  1026→                _ => {
  1027→                    stmts.push(ScgStatement::Computation(ComputationNode {
  1028→                        dst,
  1029→                        op: BinOpKind::Add,
  1030→                        lhs: result,
  1031→                        rhs: ScgExpr::Int(0),
  1032→                        tail_call: false,
  1033→                    }));
  1034→                }
  1035→            }
  1036→            stmts
  1037→        }
  1038→
  1039→        // ── target op= value ──
  1040→        PStmt::CompoundAssign(ca_stmt) => {
  1041→            let mut stmts = Vec::new();
  1042→            let dst = match &ca_stmt.target {
  1043→                vuma_parser::ast::AssignTarget::Var { name, .. } => name.clone(),
  1044→                vuma_parser::ast::AssignTarget::DerefField { field, .. } => field.clone(),
  1045→                _ => "_".into(),
  1046→            };
  1047→            let binop = match ca_stmt.op {
  1048→                vuma_parser::ast::CompoundOp::Add => BinOpKind::Add,
  1049→                vuma_parser::ast::CompoundOp::Sub => BinOpKind::Sub,
  1050→                vuma_parser::ast::CompoundOp::Mul => BinOpKind::Mul,
  1051→                vuma_parser::ast::CompoundOp::Div => BinOpKind::SDiv,
  1052→                vuma_parser::ast::CompoundOp::Mod => BinOpKind::SRem,
  1053→                vuma_parser::ast::CompoundOp::BitAnd => BinOpKind::And,
  1054→                vuma_parser::ast::CompoundOp::BitOr => BinOpKind::Or,
  1055→                vuma_parser::ast::CompoundOp::BitXor => BinOpKind::Xor,
  1056→                vuma_parser::ast::CompoundOp::Shl => BinOpKind::Shl,
  1057→                vuma_parser::ast::CompoundOp::Shr => BinOpKind::ShrL,
  1058→            };
  1059→            let rhs = flatten_expr(&ca_stmt.value, &mut stmts, ctx);
  1060→            stmts.push(ScgStatement::Computation(ComputationNode {
  1061→                dst: dst.clone(),
  1062→                op: binop,
  1063→                lhs: ScgExpr::Var(dst),
  1064→                rhs,
  1065→                tail_call: false,
  1066→            }));
  1067→            stmts
  1068→        }
  1069→
  1070→        PStmt::Allocate(_) => vec![],
  1071→        PStmt::Free(_) => vec![],
  1072→
  1073→        PStmt::Cast(cast_stmt) => {
  1074→            let mut stmts = Vec::new();
  1075→            let _ = flatten_expr(&cast_stmt.expr, &mut stmts, ctx);
  1076→            stmts
  1077→        }
  1078→
  1079→        PStmt::Return(ret_stmt) => {
  1080→            let mut stmts = Vec::new();
  1081→            let values = match &ret_stmt.value {
  1082→                Some(expr) => vec![flatten_expr(expr, &mut stmts, ctx)],
  1083→                None => vec![],
  1084→            };
  1085→            stmts.push(ScgStatement::Return(values));
  1086→            stmts
  1087→        }
  1088→
  1089→        PStmt::Expr(expr_stmt) => {
  1090→            let mut stmts = Vec::new();
  1091→            let result = flatten_expr(&expr_stmt.expr, &mut stmts, ctx);
  1092→            ctx.last_expr_result = Some(result);
  1093→            stmts
  1094→        }
  1095→
  1096→        PStmt::If(if_stmt) => {
  1097→            let mut pre_stmts = Vec::new();
  1098→            let cond = flatten_expr(&if_stmt.condition, &mut pre_stmts, ctx);
  1099→            let then_body = bridge_block_to_scg_stmts(&if_stmt.then_block, ctx);
  1100→            let else_body = if_stmt
  1101→                .else_block
  1102→                .as_ref()
  1103→                .map(|b| bridge_block_to_scg_stmts(b, ctx));
  1104→            let mut result = pre_stmts;
  1105→            result.push(ScgStatement::Control(ControlNode::If {
  1106→                cond,
  1107→                then_body,
  1108→                else_body,
  1109→            }));
  1110→            result
  1111→        }
  1112→
  1113→        // while condition { body }
  1114→        // Lowered as: Loop { <compute cond>; if cond { body } else { break } }
  1115→        PStmt::While(while_stmt) => {
  1116→            let then_body = bridge_block_to_scg_stmts(&while_stmt.body, ctx);
  1117→            let mut loop_body = Vec::new();
  1118→            let cond = flatten_expr(&while_stmt.condition, &mut loop_body, ctx);
  1119→            loop_body.push(ScgStatement::Control(ControlNode::If {
  1120→                cond,
  1121→                then_body,
  1122→                else_body: Some(vec![ScgStatement::Control(ControlNode::Break)]),
  1123→            }));
  1124→            vec![ScgStatement::Control(ControlNode::Loop { body: loop_body })]
  1125→        }
  1126→
  1127→        // for name in start..end { body }
  1128→        // Lowered as:
  1129→        //   name = start
  1130→        //   loop { _cond = name < end; if _cond { body; name = name + 1 } else { break } }
  1131→        PStmt::For(for_stmt) => {
  1132→            let mut pre_stmts = Vec::new();
  1133→            let (start_expr, end_expr) = match &for_stmt.iter {
  1134→                vuma_parser::ast::Expr::Range { start, end, .. } => {
  1135→                    let s = flatten_expr(start, &mut pre_stmts, ctx);
  1136→                    let e = flatten_expr(end, &mut pre_stmts, ctx);
  1137→                    (s, e)
  1138→                }
  1139→                other => {
  1140→                    (ScgExpr::Int(0), ScgExpr::Int(0))
  1141→                }
  1142→            };
  1143→
  1144→            let init_stmt = ScgStatement::Computation(ComputationNode {
  1145→                dst: for_stmt.name.clone(),
  1146→                op: BinOpKind::Add,
  1147→                lhs: start_expr,
  1148→                rhs: ScgExpr::Int(0),
  1149→                tail_call: false,
  1150→            });
  1151→
  1152→            let mut loop_body = Vec::new();
  1153→            let cond_temp = format!("{}_cond", for_stmt.name);
  1154→            loop_body.push(ScgStatement::Computation(ComputationNode {
  1155→                dst: cond_temp.clone(),
  1156→                op: BinOpKind::SLt,
  1157→                lhs: ScgExpr::Var(for_stmt.name.clone()),
  1158→                rhs: end_expr.clone(),
  1159→                tail_call: false,
  1160→            }));
  1161→
  1162→            let inner_body = bridge_block_to_scg_stmts(&for_stmt.body, ctx);
  1163→            let mut full_then = inner_body;
  1164→            full_then.push(ScgStatement::Computation(ComputationNode {
  1165→                dst: for_stmt.name.clone(),
  1166→                op: BinOpKind::Add,
  1167→                lhs: ScgExpr::Var(for_stmt.name.clone()),
  1168→                rhs: ScgExpr::Int(1),
  1169→                tail_call: false,
  1170→            }));
  1171→
  1172→            loop_body.push(ScgStatement::Control(ControlNode::If {
  1173→                cond: ScgExpr::Var(cond_temp),
  1174→                then_body: full_then,
  1175→                else_body: Some(vec![ScgStatement::Control(ControlNode::Break)]),
  1176→            }));
  1177→
  1178→            let mut result = pre_stmts;
  1179→            result.push(init_stmt);
  1180→            result.push(ScgStatement::Control(ControlNode::Loop { body: loop_body }));
  1181→            result
  1182→        }
  1183→
  1184→        PStmt::Loop(loop_stmt) => {
  1185→            let body = bridge_block_to_scg_stmts(&loop_stmt.body, ctx);
  1186→            vec![ScgStatement::Control(ControlNode::Loop { body })]
  1187→        }
  1188→
  1189→        PStmt::Break(_) => vec![ScgStatement::Control(ControlNode::Break)],
  1190→        PStmt::Continue(_) => vec![ScgStatement::Control(ControlNode::Continue)],
  1191→        PStmt::BdDirective(_)
  1192→        | PStmt::Sync(_)
  1193→        | PStmt::UnsafeBlock { .. }
  1194→        | PStmt::Match(_)
  1195→        | PStmt::Access(_) => vec![],
  1196→    }
  1197→}
  1198→fn cmd_disasm(file: &PathBuf, isa: IsaArg, base_addr_str: &str) -> Result<(), String> {
  1199→    let bytes = fs::read(file)
  1200→        .map_err(|e| format!("error: cannot read binary file '{}': {}", file.display(), e))?;
  1201→
  1202→    let base_addr = u64::from_str_radix(base_addr_str.trim_start_matches("0x"), 16)
  1203→        .map_err(|e| format!("error: invalid base address '{}': {}", base_addr_str, e))?;
  1204→
  1205→    let backend_kind = BackendKind::from(isa);
  1206→    let backend = create_backend(backend_kind).map_err(|e| {
  1207→        format!(
  1208→            "error: cannot create {} backend: {}",
  1209→            backend_kind.isa_name(),
  1210→            e
  1211→        )
  1212→    })?;
  1213→
  1214→    let instructions = backend.disassemble(&bytes, base_addr);
  1215→
  1216→    println!(
  1217→        "Disassembly of {} ({} bytes, ISA: {}):",
  1218→        file.display(),
  1219→        bytes.len(),
  1220→        backend.name()
  1221→    );
  1222→    for line in &instructions {
  1223→        println!("{}", line);
  1224→    }
  1225→
  1226→    Ok(())
  1227→}

  1229→/// `vuma verify <file>` — Run IVE 5-invariant verification.
  1230→fn cmd_verify(cli: &Cli, file: &PathBuf) -> Result<(), String> {
  1231→    let source = read_source(file)?;
  1232→    let mut config = make_config(cli, CompileTarget::Pi5Linux);
  1233→    // Force exhaustive verification for the verify subcommand.
  1234→    config.verification_level = VerificationLevel::Exhaustive;
  1235→
  1236→    let result = compile(&source, &config).map_err(|errors| {
  1237→        print_errors(&errors);
  1238→        format!(
  1239→            "compilation/verification failed with {} error(s)",
  1240→            errors.len()
  1241→        )
  1242→    })?;
  1243→
  1244→    match result.verification {
  1245→        Some(ref verification) => {
  1246→            println!("IVE Verification Results for {}", file.display());
  1247→            println!("  Overall verdict: {}", verification.overall);
  1248→            println!(
  1249→                "  Summary: {} invariant(s) checked",
  1250→                verification.summary.total_checked
  1251→            );
  1252→            println!("           {} passed", verification.summary.passed);
  1253→            println!("           {} failed", verification.summary.failed);
  1254→            println!("           {} unverified", verification.summary.unverified);
  1255→
  1256→            // Print individual invariant results.
  1257→            for per_inv in &verification.per_invariant {
  1258→                let status_str = match &per_inv.result.status {
  1259→                    vuma_ive::VerificationStatus::Proven => "PROVEN",
  1260→                    vuma_ive::VerificationStatus::ProbablySafe { .. } => "PROBABLY_SAFE",
  1261→                    vuma_ive::VerificationStatus::Unverified { .. } => "UNVERIFIED",
  1262→                    vuma_ive::VerificationStatus::Violated { .. } => "VIOLATED",
  1263→                };
  1264→                println!(
  1265→                    "  [{}] {:?} — {}",
  1266→                    status_str, per_inv.kind, per_inv.result.message,
  1267→                );
  1268→                if let Some(ref evidence) = per_inv.result.evidence {
  1269→                    println!("    evidence: {}", evidence);
  1270→                }
  1271→            }
  1272→        }
  1273→        None => {
  1274→            println!("Verification was skipped (verification level set to None).");
  1275→        }
  1276→    }
  1277→
  1278→    Ok(())
  1279→}

  1281→/// `vuma repl` — Interactive REPL: parse expressions and print AST.
  1282→fn cmd_repl() -> Result<(), String> {
  1283→    println!("VUMA REPL v0.1.0");
  1284→    println!("Type VUMA expressions or statements. Enter ':quit' to exit, ':help' for help.");
  1285→
  1286→    let stdin = io::stdin();
  1287→    let mut input = String::new();
  1288→
  1289→    loop {
  1290→        print!("vuma> ");
  1291→        io::stdout()
  1292→            .flush()
  1293→            .map_err(|e| format!("flush error: {}", e))?;
  1294→
  1295→        input.clear();
  1296→        match stdin.read_line(&mut input) {
  1297→            Ok(0) => break, // EOF
  1298→            Ok(_) => {}
  1299→            Err(e) => return Err(format!("error reading input: {}", e)),
  1300→        }
  1301→
  1302→        let trimmed = input.trim();
  1303→        if trimmed.is_empty() {
  1304→            continue;
  1305→        }
  1306→        if trimmed == ":quit" || trimmed == ":q" {
  1307→            break;
  1308→        }
  1309→        if trimmed == ":help" || trimmed == ":h" {
  1310→            println!("  :quit, :q   — Exit the REPL");
  1311→            println!("  :help, :h   — Show this help");
  1312→            println!("  <expr>      — Parse and display the AST");
  1313→            println!("  <stmt>      — Parse and display the AST");
  1314→            println!("  <program>   — Parse and display the full program AST");
  1315→            continue;
  1316→        }
  1317→
  1318→        // Try to parse as a full program first.
  1319→        let mut parser = vuma_parser::Parser::new(trimmed);
  1320→        let result = parser.parse_program();
  1321→        if !result.has_errors() {
  1322→            // Successfully parsed; display the AST.
  1323→            let program = result.unwrap();
  1324→            println!("{:#?}", program);
  1325→            continue;
  1326→        }
  1327→        // Program parse had errors; try to parse as an expression via
  1328→        // wrapping it in a dummy program context.
  1329→        let program_errors = result.errors.clone();
  1330→        // Since parse_expr is private, we re-parse as a minimal program.
  1331→        let wrapped = format!("fn _repl_expr() {{ {} }}", trimmed);
  1332→        let mut expr_parser = vuma_parser::Parser::new(&wrapped);
  1333→        let expr_result = expr_parser.parse_program();
  1334→        if !expr_result.has_errors() {
  1335→            // Print the function body AST.
  1336→            let program = expr_result.unwrap();
  1337→            if let Some(item) = program.items.first() {
  1338→                println!("{:#?}", item);
  1339→            } else {
  1340→                println!("{:#?}", program);
  1341→            }
  1342→        } else {
  1343→            // Show the original parse errors.
  1344→            for err in &program_errors {
  1345→                eprintln!("parse error: {}", err);
  1346→            }
  1347→        }
  1348→    }
  1349→
  1350→    Ok(())
  1351→}
