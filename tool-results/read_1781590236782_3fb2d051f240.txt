     1→// Signal-Safe Reentrant Hash — SHA256d in a SIGALRM Handler
     2→//
     3→// This program installs a SIGALRM signal handler that computes SHA256d
     4→// of a fixed message, stores the result atomically, and verifies in main
     5→// that the handler's hash matches the reference computation. It proves
     6→// that SHA256d is safe to call from a signal context — no registers are
     7→// corrupted, no shared state is clobbered, and atomic handoff works
     8→// correctly across all 8 VUMA backends.
     9→//
    10→// VUMA Key Concepts:
    11→//   - Signal handler installation via sigaction() FFI
    12→//   - struct sigaction built byte-by-byte (no multi-byte store)
    13→//   - Function pointer taken with @handler_name and stored in memory
    14→//   - AtomicStore / AtomicLoad for handler-to-main communication
    15→//   - Reentrancy: each SHA256d call allocates its own working memory,
    16→//     so concurrent calls from main and handler never interfere
    17→//   - u32 arithmetic masked with & 4294967295 per VUMA rules
    18→//   - No bitwise NOT: use x ^ 4294967295 instead of ~x
    19→//   - Every allocate() paired with free()
    20→//
    21→// Signal Handler Safety:
    22→//   - Handler only calls allocate/free (signal-safe in VUMA, which
    23→//     uses brk/mmap directly — no libc malloc)
    24→//   - Handler uses AtomicStore to publish results (release semantics
    25→//     ensure hash bytes are visible before the flag)
    26→//   - Handler does NOT call write, printf, or any non-async-signal-
    27→//     safe function
    28→//   - Handler's SHA256d uses entirely separate working memory from
    29→//     main's SHA256d, so there is no shared mutable state
    30→//
    31→// struct sigaction layout (Linux, 64-bit, little-endian):
    32→//   Offset   0: sa_handler  (8 bytes, function pointer)
    33→//   Offset   8: sa_mask     (128 bytes, sigset_t — all zeros)
    34→//   Offset 136: sa_flags    (4 bytes, int — SA_RESTART = 0x10000000)
    35→//   Offset 140: sa_restorer (8 bytes, function pointer — 0)
    36→//   Total: ~152 bytes (allocate 160 for alignment padding)
    37→//
    38→// Expected exit code: 0 (success — handler hash matches reference)
    39→
    40→// ===========================================================================
    41→// FFI Declarations
    42→// ===========================================================================
    43→// sigaction: install a signal handler
    44→// alarm: schedule SIGALRM after N seconds
    45→// write/read/exit: standard POSIX I/O and process control
    46→
    47→extern "C" {
    48→    fn sigaction(signum: i64, act: Address, oldact: Address) -> i64;
    49→    fn alarm(seconds: i64) -> i64;
    50→    fn write(fd: i64, buf: Address, count: i64) -> i64;
    51→    fn read(fd: i64, buf: Address, count: i64) -> i64;
    52→    fn exit(code: i64);
    53→}
    54→
    55→// ===========================================================================
    56→// Global Shared State for Signal Handler Communication
    57→// ===========================================================================
    58→// These module-level variables are set by main() before the alarm is
    59→// armed, and read by the signal handler when SIGALRM fires. Since main
    60→// does not modify them after initialization and the kernel's signal
    61→// delivery provides an implicit memory barrier, there is no data race.
    62→//
    63→// g_msg_addr:   pointer to the message bytes ("signal-test")
    64→// g_msg_len:    length of the message in bytes (11)
    65→// g_result_addr: pointer to 32-byte buffer for handler's hash result
    66→// g_flag_addr:  pointer to 8-byte atomic flag (u64: 0=not ready, 1=done)
    67→
    68→g_msg_addr: Address = 0;
    69→g_msg_len: u64 = 0;
    70→g_result_addr: Address = 0;
    71→g_flag_addr: Address = 0;
    72→
    73→// ===========================================================================
    74→// SHA-256 Helper: u32 right-rotate
    75→// ===========================================================================
    76→// VUMA does not have a built-in rotate, so we compose from shifts.
    77→// rotate_right(x, n) = (x >> n) | (x << (32 - n))
    78→// The result is masked to 32 bits because left shifts in 64-bit
    79→// registers produce bits above bit 31 that must be cleared.
    80→
    81→fn rotr32(x: u32, n: u32) -> u32 {
    82→    return ((x >> n) | (x << (32 - n))) & 4294967295;
    83→}
    84→
    85→// ===========================================================================
    86→// SHA-256 logical functions (FIPS 180-4 Section 4.1.2)
    87→// ===========================================================================
    88→
    89→fn ch(x: u32, y: u32, z: u32) -> u32 {
    90→    // Constant-time Ch: (x & y) ^ ((x ^ 0xFFFFFFFF) & z)
    91→    // No branches, naturally constant-time.
    92→    return (x & y) ^ ((x ^ 4294967295) & z);
    93→}
    94→
    95→fn maj(a: u32, b: u32, c: u32) -> u32 {
    96→    return (a & b) ^ (a & c) ^ (b & c);
    97→}
    98→
    99→fn big_sigma0(x: u32) -> u32 {
   100→    return rotr32(x, 2) ^ rotr32(x, 13) ^ rotr32(x, 22);
   101→}
   102→
   103→fn big_sigma1(x: u32) -> u32 {
   104→    return rotr32(x, 6) ^ rotr32(x, 11) ^ rotr32(x, 25);
   105→}
   106→
   107→fn small_sigma0(x: u32) -> u32 {
   108→    return (rotr32(x, 7) ^ rotr32(x, 18) ^ (x >> 3)) & 4294967295;
   109→}
   110→
   111→fn small_sigma1(x: u32) -> u32 {
   112→    return (rotr32(x, 17) ^ rotr32(x, 19) ^ (x >> 10)) & 4294967295;
   113→}

   115→// ===========================================================================
   116→// Read/write u32 from byte buffer (big-endian, per FIPS 180-4)
   117→// ===========================================================================
   118→// VUMA has no multi-byte load/store — must compose from byte access.
   119→// Each byte load is zero-extended; each byte store masks to low 8 bits.
   120→
   121→fn read_u32_be(buf: Address, offset: u64) -> u32 {
   122→    b0: u32 = *(buf + offset);
   123→    b1: u32 = *(buf + offset + 1);
   124→    b2: u32 = *(buf + offset + 2);
   125→    b3: u32 = *(buf + offset + 3);
   126→    return ((b0 << 24) | (b1 << 16) | (b2 << 8) | b3) & 4294967295;
   127→}
   128→
   129→fn write_u32_be(buf: Address, offset: u64, val: u32) {
   130→    *(buf + offset)     = (val >> 24) & 255;
   131→    *(buf + offset + 1) = (val >> 16) & 255;
   132→    *(buf + offset + 2) = (val >> 8) & 255;
   133→    *(buf + offset + 3) = val & 255;
   134→}
   135→
   136→// ===========================================================================
   137→// Store/load a u32 from the W schedule array (stride = 4 bytes)
   138→// ===========================================================================
   139→
   140→fn w_store(w_base: Address, idx: u64, val: u32) {
   141→    write_u32_be(w_base, idx * 4, val);
   142→}
   143→
   144→fn w_load(w_base: Address, idx: u64) -> u32 {
   145→    return read_u32_be(w_base, idx * 4);
   146→}

   238→// ===========================================================================
   239→// SHA-256 compression: process one 512-bit block
   240→// ===========================================================================
   241→// Per FIPS 180-4 Section 6.2.2. All u32 additions are masked to 32 bits
   242→// because VUMA uses 64-bit registers — carry bits above bit 31 would
   243→// corrupt subsequent right-shift operations in rotr32.
   244→
   245→fn sha256_transform(state: Address, k: Address, w: Address, block: Address) {
   246→    // Step 1: Build message schedule W[0..63]
   247→    i: u64 = 0;
   248→    for i in 0..16 {
   249→        val: u32 = read_u32_be(block, i * 4);
   250→        w_store(w, i, val);
   251→    }
   252→    for i in 16..64 {
   253→        w15: u32 = w_load(w, i - 15);
   254→        w2: u32 = w_load(w, i - 2);
   255→        w7: u32 = w_load(w, i - 7);
   256→        w16: u32 = w_load(w, i - 16);
   257→        val: u32 = (small_sigma1(w2) + w7 + small_sigma0(w15) + w16) & 4294967295;
   258→        w_store(w, i, val);
   259→    }
   260→
   261→    // Step 2: Initialize working variables from current hash state
   262→    a: u32 = read_u32_be(state, 0);
   263→    b: u32 = read_u32_be(state, 4);
   264→    c: u32 = read_u32_be(state, 8);
   265→    d: u32 = read_u32_be(state, 12);
   266→    e: u32 = read_u32_be(state, 16);
   267→    f: u32 = read_u32_be(state, 20);
   268→    g: u32 = read_u32_be(state, 24);
   269→    h: u32 = read_u32_be(state, 28);
   270→
   271→    // Step 3: 64-round compression
   272→    // ch() and maj() are inherently constant-time (pure bitwise, no
   273→    // branches). Working variable rotations (h=g, g=f, ...) are also
   274→    // branch-free — resistant to timing side-channel attacks.
   275→    for i in 0..64 {
   276→        ki: u32 = read_u32_be(k, i * 4);
   277→        wi: u32 = w_load(w, i);
   278→        t1: u32 = (h + big_sigma1(e) + ch(e, f, g) + ki + wi) & 4294967295;
   279→        t2: u32 = (big_sigma0(a) + maj(a, b, c)) & 4294967295;
   280→        h = g;
   281→        g = f;
   282→        f = e;
   283→        e = (d + t1) & 4294967295;
   284→        d = c;
   285→        c = b;
   286→        b = a;
   287→        a = (t1 + t2) & 4294967295;
   288→    }
   289→
   290→    // Step 4: Add compressed chunk to current hash value
   291→    write_u32_be(state, 0,  (read_u32_be(state, 0)  + a) & 4294967295);
   292→    write_u32_be(state, 4,  (read_u32_be(state, 4)  + b) & 4294967295);
   293→    write_u32_be(state, 8,  (read_u32_be(state, 8)  + c) & 4294967295);
   294→    write_u32_be(state, 12, (read_u32_be(state, 12) + d) & 4294967295);
   295→    write_u32_be(state, 16, (read_u32_be(state, 16) + e) & 4294967295);
   296→    write_u32_be(state, 20, (read_u32_be(state, 20) + f) & 4294967295);
   297→    write_u32_be(state, 24, (read_u32_be(state, 24) + g) & 4294967295);
   298→    write_u32_be(state, 28, (read_u32_be(state, 28) + h) & 4294967295);
   299→}

   301→// ===========================================================================
   302→// Pad a message into a 64-byte block (for messages ≤ 55 bytes)
   303→// ===========================================================================
   304→// FIPS 180-4 padding: append 0x80, zero-fill to offset 56, then
   305→// append 64-bit big-endian bit-length at offset 56..63.
   306→
   307→fn sha256_pad_block(block: Address, msg: Address, msg_len: u64) {
   308→    // Copy message bytes
   309→    i: u64 = 0;
   310→    for i in 0..msg_len {
   311→        *(block + i) = *(msg + i);
   312→    }
   313→
   314→    // Append the '1' bit (0x80 byte)
   315→    *(block + msg_len) = 128;
   316→
   317→    // Zero-fill between padding and length field
   318→    for i in msg_len + 1..56 {
   319→        *(block + i) = 0;
   320→    }
   321→
   322→    // Append 64-bit big-endian message length in bits
   323→    bit_len: u64 = msg_len * 8;
   324→    *(block + 56) = (bit_len >> 56) & 255;
   325→    *(block + 57) = (bit_len >> 48) & 255;
   326→    *(block + 58) = (bit_len >> 40) & 255;
   327→    *(block + 59) = (bit_len >> 32) & 255;
   328→    *(block + 60) = (bit_len >> 24) & 255;
   329→    *(block + 61) = (bit_len >> 16) & 255;
   330→    *(block + 62) = (bit_len >> 8) & 255;
   331→    *(block + 63) = bit_len & 255;
   332→}

   334→// ===========================================================================
   335→// Copy 32 bytes from source to destination
   336→// ===========================================================================
   337→
   338→fn copy32(dst: Address, src: Address) {
   339→    i: u64 = 0;
   340→    for i in 0..32 {
   341→        *(dst + i) = *(src + i);
   342→    }
   343→}

   345→// ===========================================================================
   346→// SHA256d: SHA-256(SHA-256(message))
   347→// ===========================================================================
   348→// Double-SHA-256 construction. Each call allocates its own working
   349→// memory (state, K, W, block, inner) making it inherently reentrant —
   350→// safe to call from both main and a signal handler concurrently.
   351→
   352→fn sha256d(msg: Address, msg_len: u64, out: Address) {
   353→    // Allocate working memory — each call gets its own, no shared state
   354→    state = allocate(32);
   355→    k = allocate(256);
   356→    w = allocate(256);
   357→    block = allocate(64);
   358→    inner = allocate(32);
   359→
   360→    // Initialize round constants (same for both passes)
   361→    sha256_init_k(k);
   362→
   363→    // ---- Inner hash: SHA-256(message) ----
   364→    sha256_init_state(state);
   365→    sha256_pad_block(block, msg, msg_len);
   366→    sha256_transform(state, k, w, block);
   367→    copy32(inner, state);
   368→
   369→    // ---- Outer hash: SHA-256(inner_hash) ----
   370→    sha256_init_state(state);
   371→    sha256_pad_block(block, inner, 32);
   372→    sha256_transform(state, k, w, block);
   373→    copy32(out, state);
   374→
   375→    // Clean up all allocations
   376→    free(state);
   377→    free(k);
   378→    free(w);
   379→    free(block);
   380→    free(inner);
   381→}

   383→// ===========================================================================
   384→// Memory Helpers for struct sigaction Construction
   385→// ===========================================================================
   386→// VUMA has no multi-byte store, so we compose 32-bit and 64-bit writes
   387→// from byte-level operations. These helpers write in little-endian byte
   388→// order, which is the native byte order for x86_64, AArch64, RISC-V,
   389→// LoongArch64, and ppc64le Linux targets.
   390→//
   391→// NOTE: For big-endian targets (ppc64be, s390x), the byte order in the
   392→// sigaction struct would need to be reversed. This example targets LE.
   393→
   394→fn write_u64_le(buf: Address, offset: u64, val: u64) {
   395→    *(buf + offset)     = val & 255;
   396→    *(buf + offset + 1) = (val >> 8) & 255;
   397→    *(buf + offset + 2) = (val >> 16) & 255;
   398→    *(buf + offset + 3) = (val >> 24) & 255;
   399→    *(buf + offset + 4) = (val >> 32) & 255;
   400→    *(buf + offset + 5) = (val >> 40) & 255;
   401→    *(buf + offset + 6) = (val >> 48) & 255;
   402→    *(buf + offset + 7) = (val >> 56) & 255;
   403→}
   404→
   405→fn write_u32_le(buf: Address, offset: u64, val: u32) {
   406→    *(buf + offset)     = val & 255;
   407→    *(buf + offset + 1) = (val >> 8) & 255;
   408→    *(buf + offset + 2) = (val >> 16) & 255;
   409→    *(buf + offset + 3) = (val >> 24) & 255;
   410→}

   412→fn memzero(buf: Address, len: u64) {
   413→    i: u64 = 0;
   414→    for i in 0..len {
   415→        *(buf + i) = 0;
   416→    }
   417→}

   419→// ===========================================================================
   420→// SIGALRM Handler — Async-Signal-Safe SHA256d Computation
   421→// ===========================================================================
   422→// This function is installed as the SIGALRM handler via sigaction().
   423→// The OS calls it with the signal number as the sole argument.
   424→//
   425→// Async-signal-safety guarantees:
   426→//   1. No calls to non-async-signal-safe functions (no write, no printf)
   427→//   2. Only uses allocate/free (signal-safe in VUMA — uses brk/mmap)
   428→//   3. Uses AtomicStore to publish results (release semantics ensure
   429→//      all prior writes to the hash buffer are visible before the flag)
   430→//   4. SHA256d is reentrant — each call allocates separate working memory
   431→//
   432→// The handler reads the message pointer and length from module-level
   433→// globals (set by main before the alarm was armed), computes SHA256d,
   434→// copies the result into the shared buffer, then sets the atomic flag.
   435→
   436→fn signal_handler(signum: i64) {
   437→    // Compute SHA256d of the shared message
   438→    // This allocation is signal-safe in VUMA (brk/mmap, no libc malloc)
   439→    result = allocate(32);
   440→    sha256d(g_msg_addr, g_msg_len, result);
   441→
   442→    // Copy the 32-byte hash into the shared result buffer
   443→    // This byte-by-byte copy is safe because main will not read
   444→    // these bytes until it observes the atomic flag set to 1.
   445→    i: u64 = 0;
   446→    for i in 0..32 {
   447→        *(g_result_addr + i) = *(result + i);
   448→    }
   449→
   450→    // Free the handler's local hash buffer
   451→    free(result);
   452→
   453→    // Publish: set the atomic flag to 1 with release semantics.
   454→    // AtomicStore ensures that all prior writes (the hash bytes above)
   455→    // are visible to any thread that subsequently loads the flag with
   456→    // acquire semantics. This is the crucial memory ordering that
   457→    // makes handler-to-main communication safe.
   458→    //
   459→    // This also exercises the AtomicStore code path on all 8 VUMA
   460→    // backends, which is important for testing atomic correctness
   461→    // on LoongArch64, PPC64, and RISC-V where atomic instruction
   462→    // sequences are non-trivial.
   463→    AtomicStore(1, g_flag_addr);
   464→}

   466→// ===========================================================================
   467→// Main Entry Point
   468→// ===========================================================================
   469→// 1. Allocate shared state (message, result buffer, atomic flag)
   470→// 2. Build struct sigaction and install signal handler
   471→// 3. Compute reference SHA256d in main context
   472→// 4. Arm alarm(1) — SIGALRM fires in 1 second
   473→// 5. Busy loop: repeatedly compute SHA256d, checking atomic flag
   474→// 6. When handler sets flag, compare handler's hash with reference
   475→// 7. Exit 0 if match (success), exit 1 if mismatch (bug)
   476→
   477→fn main() -> i32 {
   478→    // ----------------------------------------------------------------
   479→    // Step 1: Prepare the fixed message "signal-test" (11 bytes)
   480→    // ----------------------------------------------------------------
   481→    msg = allocate(11);
   482→    *(msg + 0)  = 115;   // 's'
   483→    *(msg + 1)  = 105;   // 'i'
   484→    *(msg + 2)  = 103;   // 'g'
   485→    *(msg + 3)  = 110;   // 'n'
   486→    *(msg + 4)  = 97;    // 'a'
   487→    *(msg + 5)  = 108;   // 'l'
   488→    *(msg + 6)  = 45;    // '-'
   489→    *(msg + 7)  = 116;   // 't'
   490→    *(msg + 8)  = 101;   // 'e'
   491→    *(msg + 9)  = 115;   // 's'
   492→    *(msg + 10) = 116;   // 't'
   493→
   494→    // ----------------------------------------------------------------
   495→    // Step 2: Allocate shared state for handler communication
   496→    // ----------------------------------------------------------------
   497→    // handler_result: 32-byte buffer where the handler writes its hash
   498→    // handler_flag:   8-byte atomic u64 — 0 = not ready, 1 = done
   499→    handler_result = allocate(32);
   500→    handler_flag = allocate(8);
   501→
   502→    // Initialize the atomic flag to 0 (not ready)
   503→    AtomicStore(0, handler_flag);
   504→
   505→    // ----------------------------------------------------------------
   506→    // Step 3: Set global pointers for the signal handler
   507→    // ----------------------------------------------------------------
   508→    // These are written before the alarm is armed, so the handler
   509→    // will observe consistent values when it fires.
   510→    g_msg_addr = msg;
   511→    g_msg_len = 11;
   512→    g_result_addr = handler_result;
   513→    g_flag_addr = handler_flag;
   514→
   515→    // ----------------------------------------------------------------
   516→    // Step 4: Build struct sigaction in memory (byte-by-byte)
   517→    // ----------------------------------------------------------------
   518→    // Layout (Linux 64-bit):
   519→    //   Offset   0: sa_handler  — 8 bytes (function pointer)
   520→    //   Offset   8: sa_mask     — 128 bytes (sigset_t, all zeros)
   521→    //   Offset 136: sa_flags    — 4 bytes (SA_RESTART = 0x10000000)
   522→    //   Offset 140: sa_restorer — 8 bytes (0 = no restorer)
   523→    //   Total: 148 bytes — allocate 160 for alignment padding
   524→    act = allocate(160);
   525→    memzero(act, 160);
   526→
   527→    // sa_handler = address of our signal_handler function
   528→    // @signal_handler takes the function's address (u64 / Address)
   529→    handler_addr: u64 = @signal_handler;
   530→    write_u64_le(act, 0, handler_addr);
   531→
   532→    // sa_mask = all zeros (no signals blocked during handler)
   533→    // Already zeroed by memzero — nothing to write.
   534→
   535→    // sa_flags = SA_RESTART (0x10000000 = 268435456)
   536→    // SA_RESTART causes interrupted syscalls to be automatically
   537→    // restarted, which prevents spurious EINTR errors in the
   538→    // busy loop below.
   539→    write_u32_le(act, 136, 268435456);
   540→
   541→    // sa_restorer = 0 (no custom restorer)
   542→    // Already zeroed by memzero — nothing to write.
   543→
   544→    // ----------------------------------------------------------------
   545→    // Step 5: Install the signal handler via sigaction()
   546→    // ----------------------------------------------------------------
   547→    // SIGALRM = 14 on all Linux architectures.
   548→    // sigaction returns 0 on success, -1 on error.
   549→    ret: i64 = sigaction(14, act, 0);
   550→    if ret != 0 {
   551→        // sigaction failed — clean up and exit with error
   552→        free(msg);
   553→        free(handler_result);
   554→        free(handler_flag);
   555→        free(act);
   556→        return 1;
   557→    }
   558→
   559→    // ----------------------------------------------------------------
   560→    // Step 6: Compute reference SHA256d in main context
   561→    // ----------------------------------------------------------------
   562→    // This is the "ground truth" hash that the handler's result
   563→    // must match. If the signal frame corrupts registers or the
   564→    // SHA256d implementation is not reentrant, the hashes will
   565→    // differ and we exit 1.
   566→    ref_hash = allocate(32);
   567→    sha256d(msg, 11, ref_hash);
   568→
   569→    // ----------------------------------------------------------------
   570→    // Step 7: Arm the alarm — SIGALRM fires in 1 second
   571→    // ----------------------------------------------------------------
   572→    // After this call, the kernel will deliver SIGALRM to this
   573→    // process after approximately 1 second. The signal_handler
   574→    // function will be invoked asynchronously in this thread.
   575→    alarm(1);
   576→
   577→    // ----------------------------------------------------------------
   578→    // Step 8: Busy loop — compute SHA256d while waiting for signal
   579→    // ----------------------------------------------------------------
   580→    // We repeatedly compute SHA256d to:
   581→    //   a) Keep the CPU busy so the signal actually interrupts us
   582→    //   b) Test that SHA256d is truly reentrant — if the handler's
   583→    //      SHA256d call corrupts any state used by main's SHA256d,
   584→    //      subsequent iterations would produce wrong results
   585→    //   c) Test that AtomicLoad correctly observes the handler's
   586→    //      AtomicStore across all 8 backends
   587→    //
   588→    // The loop exits when the atomic flag transitions from 0 to 1,
   589→    // which happens when the signal handler completes.
   590→    loop {
   591→        // Check the atomic flag with acquire semantics.
   592→        // AtomicLoad ensures we observe all writes that happened
   593→        // before the handler's AtomicStore(1, flag).
   594→        flag_val: u64 = AtomicLoad(handler_flag);
   595→        if flag_val == 1 {
   596→            break;
   597→        }
   598→
   599→        // Do a SHA256d computation to exercise the reentrant path.
   600→        // If the signal fires during this computation and the
   601→        // handler's SHA256d call interferes with this one, the
   602→        // result would be wrong — but since each call allocates
   603→        // its own working memory, they are independent.
   604→        scratch = allocate(32);
   605→        sha256d(msg, 11, scratch);
   606→        free(scratch);
   607→    }
   608→
   609→    // ----------------------------------------------------------------
   610→    // Step 9: Verify — compare handler's hash with reference
   611→    // ----------------------------------------------------------------
   612→    // The handler wrote its result into handler_result byte-by-byte,
   613→    // then set the atomic flag. Thanks to the release/acquire
   614→    // ordering, all 32 bytes are guaranteed visible here.
   615→    match: i32 = 1;    // assume match until proven otherwise
   616→    i: u64 = 0;
   617→    for i in 0..32 {
   618→        h_byte: u32 = *(handler_result + i);
   619→        r_byte: u32 = *(ref_hash + i);
   620→        if h_byte != r_byte {
   621→            match = 0;
   622→        }
   623→    }
   624→
   625→    // ----------------------------------------------------------------
   626→    // Step 10: Clean up all allocations
   627→    // ----------------------------------------------------------------
   628→    // Every allocate() must be paired with free() — VUMA's IVE
   629→    // verifies no memory leaks.
   630→    free(msg);
   631→    free(handler_result);
   632→    free(handler_flag);
   633→    free(act);
   634→    free(ref_hash);
   635→
   636→    // ----------------------------------------------------------------
   637→    // Step 11: Return result
   638→    // ----------------------------------------------------------------
   639→    // Exit 0: handler hash matches reference (signal-safe reentrant
   640→    //         hash verified — no register corruption, no state clobber)
   641→    // Exit 1: hashes differ (BUG — signal handler corrupted state or
   642→    //         atomic handoff failed)
   643→    if match == 1 {
   644→        return 0;
   645→    } else {
   646→        return 1;
   647→    }
   648→}
