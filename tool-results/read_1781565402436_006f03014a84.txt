     1→     1→//! # VUMA Compiler API for LLM Consumption
     2→     2→//!
     3→     3→//! This module provides a clean, programmatic interface designed for LLM agents
     4→     4→//! and automated tools that need to compile VUMA source code and receive
     5→     5→//! structured, machine-readable results.
     6→     6→//!
     7→     7→//! ## Design Principles
     8→     8→//!
     9→     9→//! - **Always succeeds**: Every method returns a structured result; errors are
    10→    10→//!   captured as diagnostics rather than panicking or returning `Err`.
    11→    11→//! - **Serializable**: All result types derive `Serialize` / `Deserialize` so
    12→    12→//!   they can be sent over JSON-based IPC channels.
    13→    13→//! - **Summaries, not internals**: The API returns enough information for an
    14→    14→//!   LLM to reason about the program (function signatures, call graph, node
    15→    15→//!   counts) without exposing the full graph representation.
    16→    16→//! - **Target-aware**: The API can compile for any supported backend
    17→    17→//!   (x86_64, AArch64, RISC-V, etc.) and returns target-specific outputs.
    18→    18→//!
    19→    19→//! ## Quick Start
    20→    20→//!
    21→    21→//! ```rust
    22→    22→//! use vuma::api::VumaCompiler;
    23→    23→//!
    24→    24→//! let compiler = VumaCompiler::new();
    25→    25→//! let source = r#"
    26→    26→//!     fn main() {
    27→    27→//!         x = 1 + 2;
    28→    28→//!     }
    29→    29→//! "#;
    30→    30→//!
    31→    31→//! let result = compiler.compile(source);
    32→    32→//! if result.success {
    33→    33→//!     println!("Compiled {} functions", result.scg.unwrap().function_count);
    34→    34→//! } else {
    35→    35→//!     for diag in &result.diagnostics {
    36→    36→//!         println!("[{}] {}", diag.severity, diag.message);
    37→    37→//!     }
    38→    38→//! }
    39→    39→//! ```

    66→    66→/// The primary compiler API for programmatic (LLM) consumption.
    67→    67→///
    68→    68→/// `VumaCompiler` is the main entry point for LLMs and automated tools.
    69→    69→/// It wraps the full VUMA compilation pipeline and returns structured results
    70→    70→/// that are easy to parse and reason about programmatically.

    89→    89→    /// Compile from source string. Returns a structured result.
    90→    90→    ///
    91→    91→    /// This runs the full compilation pipeline:
    92→    92→    /// Parse → AST → SCG → BD Inference → MSG → IVE Verification
    93→    93→    /// → SCG Transforms → IR Lowering → Register Allocation → Code Emission
    94→    94→    ///
    95→    95→    /// The result always contains a value — check `result.success` to
    96→    96→    /// determine if compilation succeeded, and inspect `result.diagnostics`
    97→    97→    /// for any warnings or errors.
    98→    98→    pub fn compile(&self, source: &str) -> CompileResult {
    99→    99→        let start = Instant::now();
   100→   100→        let source_lines = source.lines().count();
   101→   101→        let source_bytes = source.len();
   102→   102→
   103→   103→        match pipeline::compile(source, &self.config) {
   104→   104→            Ok(output) => {
   105→   105→                let scg_summary = Some(build_scg_summary(&output.scg));
   106→   106→
   107→   107→                // Get disassembly from the binary using the default backend
   108→   108→                let disasm = disassemble_default(&output.binary);
   109→   109→
   110→   110→                let target_output = Some(TargetOutput {
   111→   111→                    backend: "aarch64".to_string(),
   112→   112→                    binary: output.binary.clone(),
   113→   113→                    binary_size: output.binary.len(),
   114→   114→                    disassembly: disasm,
   115→   115→                });
   116→   116→
   117→   117→                // Collect any non-fatal diagnostics
   118→   118→                let diagnostics = Vec::new();
   119→   119→
   120→   120→                CompileResult {
   121→   121→                    success: true,
   122→   122→                    diagnostics,
   123→   123→                    scg: scg_summary,
   124→   124→                    target: target_output,
   125→   125→                    metadata: CompileMetadata {
   126→   126→                        compile_time_ms: start.elapsed().as_millis() as u64,
   127→   127→                        source_lines,
   128→   128→                        source_bytes,
   129→   129→                    },
   130→   130→                }
   131→   131→            }
   132→   132→            Err(errors) => {
   133→   133→                let diagnostics = errors
   134→   134→                    .iter()
   135→   135→                    .flat_map(|e| diagnostics::from_vuma_error(e))
   136→   136→                    .collect();
   137→   137→
   138→   138→                CompileResult {
   139→   139→                    success: false,
   140→   140→                    diagnostics,
   141→   141→                    scg: None,
   142→   142→                    target: None,
   143→   143→                    metadata: CompileMetadata {
   144→   144→                        compile_time_ms: start.elapsed().as_millis() as u64,
   145→   145→                        source_lines,
   146→   146→                        source_bytes,
   147→   147→                    },
   148→   148→                }
   149→   149→            }
   150→   150→        }
   151→   151→    }

   153→   153→    /// Compile for a specific target backend.
   154→   154→    ///
   155→   155→    /// Valid target strings: `"x86_64"`, `"aarch64"`, `"riscv64"`,
   156→   156→    /// `"wasm32"`, `"loongarch64"`, `"arm32"`, `"mips64"`, `"ppc64"`.
   157→   157→    ///
   158→   158→    /// If the target string is not recognised, the result will contain
   159→   159→    /// error diagnostics.
   160→   160→    pub fn compile_for_target(&self, source: &str, target: &str) -> CompileResult {
   161→   161→        let start = Instant::now();
   162→   162→        let source_lines = source.lines().count();
   163→   163→        let source_bytes = source.len();
   164→   164→
   165→   165→        // Parse the target string into a BackendKind
   166→   166→        let backend_kind = match parse_target(target) {
   167→   167→            Some(kind) => kind,
   168→   168→            None => {
   169→   169→                return CompileResult {
   170→   170→                    success: false,
   171→   171→                    diagnostics: vec![VumaDiagnostic::new(
   172→   172→                        "E021",
   173→   173→                        DiagnosticSeverity::Error,
   174→   174→                        format!(
   175→   175→                            "Unknown target '{}'. Available: x86_64, aarch64, riscv64, \
   176→   176→                             wasm32, loongarch64, arm32, mips64, ppc64",
   177→   177→                            target
   178→   178→                        ),
   179→   179→                        "target-selection",
   180→   180→                        DiagnosticSourceLocation::unknown(),
   181→   181→                    )],
   182→   182→                    scg: None,
   183→   183→                    target: None,
   184→   184→                    metadata: CompileMetadata {
   185→   185→                        compile_time_ms: start.elapsed().as_millis() as u64,
   186→   186→                        source_lines,
   187→   187→                        source_bytes,
   188→   188→                    },
   189→   189→                };
   190→   190→            }
   191→   191→        };
   192→   192→
   193→   193→        // Run the front-end pipeline (parse through SCG transforms)
   194→   194→        let front_result = run_frontend(source, &self.config);
   195→   195→
   196→   196→        let (scg, mut diagnostics) = match front_result {
   197→   197→            FrontendResult::Ok { scg } => (scg, Vec::new()),
   198→   198→            FrontendResult::Err { diagnostics } => {
   199→   199→                return CompileResult {
   200→   200→                    success: false,
   201→   201→                    diagnostics,
   202→   202→                    scg: None,
   203→   203→                    target: None,
   204→   204→                    metadata: CompileMetadata {
   205→   205→                        compile_time_ms: start.elapsed().as_millis() as u64,
   206→   206→                        source_lines,
   207→   207→                        source_bytes,
   208→   208→                    },
   209→   209→                };
   210→   210→            }
   211→   211→        };
   212→   212→
   213→   213→        // Build SCG summary from the validated SCG
   214→   214→        let scg_summary = Some(build_scg_summary(&scg));
   215→   215→
   216→   216→        // Run target-specific codegen
   217→   217→        let target_output = match run_backend_codegen(&scg, backend_kind) {
   218→   218→            Ok(output) => Some(output),
   219→   219→            Err(diags) => {
   220→   220→                diagnostics.extend(diags);
   221→   221→                return CompileResult {
   222→   222→                    success: false,
   223→   223→                    diagnostics,
   224→   224→                    scg: scg_summary,
   225→   225→                    target: None,
   226→   226→                    metadata: CompileMetadata {
   227→   227→                        compile_time_ms: start.elapsed().as_millis() as u64,
   228→   228→                        source_lines,
   229→   229→                        source_bytes,
   230→   230→                    },
   231→   231→                };
   232→   232→            }
   233→   233→        };
   234→   234→
   235→   235→        CompileResult {
   236→   236→            success: true,
   237→   237→            diagnostics,
   238→   238→            scg: scg_summary,
   239→   239→            target: target_output,
   240→   240→            metadata: CompileMetadata {
   241→   241→                compile_time_ms: start.elapsed().as_millis() as u64,
   242→   242→                source_lines,
   243→   243→                source_bytes,
   244→   244→            },
   245→   245→        }
   246→   246→    }

   248→   248→    /// Just parse — return AST/SCG without codegen.
   249→   249→    ///
   250→   250→    /// Useful for LLMs that want to understand the program structure
   251→   251→    /// without incurring the cost of full code generation.
   252→   252→    pub fn parse(&self, source: &str) -> ParseResult {
   253→   253→        let start = Instant::now();
   254→   254→
   255→   255→        use vuma_parser::{AstToScg, Parser};
   256→   256→
   257→   257→        // Parse source to AST
   258→   258→        let mut parser = Parser::new(source);
   259→   259→        let parse_output = parser.parse_program();
   260→   260→
   261→   261→        if parse_output.has_errors() {
   262→   262→            let diagnostics = diagnostics::from_parse_errors(&parse_output.errors, source, None);
   263→   263→
   264→   264→            return ParseResult {
   265→   265→                success: false,
   266→   266→                diagnostics,
   267→   267→                ast_summary: None,
   268→   268→                scg: None,
   269→   269→                metadata: CompileMetadata {
   270→   270→                    compile_time_ms: start.elapsed().as_millis() as u64,
   271→   271→                    source_lines: source.lines().count(),
   272→   272→                    source_bytes: source.len(),
   273→   273→                },
   274→   274→            };
   275→   275→        }
   276→   276→
   277→   277→        let ast = parse_output.unwrap();
   278→   278→        let ast_summary = Some(build_ast_summary(&ast));
   279→   279→
   280→   280→        // Convert AST to SCG
   281→   281→        let mut converter = AstToScg::new();
   282→   282→        match converter.convert(&ast) {
   283→   283→            Ok(scg) => {
   284→   284→                let scg_summary = Some(build_scg_summary(&scg));
   285→   285→                ParseResult {
   286→   286→                    success: true,
   287→   287→                    diagnostics: Vec::new(),
   288→   288→                    ast_summary,
   289→   289→                    scg: scg_summary,
   290→   290→                    metadata: CompileMetadata {
   291→   291→                        compile_time_ms: start.elapsed().as_millis() as u64,
   292→   292→                        source_lines: source.lines().count(),
   293→   293→                        source_bytes: source.len(),
   294→   294→                    },
   295→   295→                }
   296→   296→            }
   297→   297→            Err(e) => {
   298→   298→                let diagnostics = vec![VumaDiagnostic::new(
   299→   299→                    "E019",
   300→   300→                    DiagnosticSeverity::Error,
   301→   301→                    format!("{}", e),
   302→   302→                    "ast-to-scg",
   303→   303→                    DiagnosticSourceLocation::unknown(),
   304→   304→                )];
   305→   305→                ParseResult {
   306→   306→                    success: false,
   307→   307→                    diagnostics,
   308→   308→                    ast_summary,
   309→   309→                    scg: None,
   310→   310→                    metadata: CompileMetadata {
   311→   311→                        compile_time_ms: start.elapsed().as_millis() as u64,
   312→   312→                        source_lines: source.lines().count(),
   313→   313→                        source_bytes: source.len(),
   314→   314→                    },
   315→   315→                }
   316→   316→            }
   317→   317→        }
   318→   318→    }

   381→   381→    /// Validate source without full compilation.
   382→   382→    ///
   383→   383→    /// Runs parsing and SCG validation, returning a list of diagnostics.
   384→   384→    /// This is the fastest way to check if a program is well-formed.
   385→   385→    pub fn validate(&self, source: &str) -> Vec<VumaDiagnostic> {
   386→   386→        use vuma_parser::{AstToScg, Parser};
   387→   387→
   388→   388→        let mut all_diagnostics = Vec::new();
   389→   389→
   390→   390→        // Parse
   391→   391→        let mut parser = Parser::new(source);
   392→   392→        let parse_output = parser.parse_program();
   393→   393→
   394→   394→        if parse_output.has_errors() {
   395→   395→            all_diagnostics.extend(diagnostics::from_parse_errors(
   396→   396→                &parse_output.errors,
   397→   397→                source,
   398→   398→                None,
   399→   399→            ));
   400→   400→            return all_diagnostics;
   401→   401→        }
   402→   402→
   403→   403→        let ast = parse_output.unwrap();
   404→   404→
   405→   405→        // AST → SCG
   406→   406→        let mut converter = AstToScg::new();
   407→   407→        let scg = match converter.convert(&ast) {
   408→   408→            Ok(scg) => scg,
   409→   409→            Err(e) => {
   410→   410→                all_diagnostics.push(VumaDiagnostic::new(
   411→   411→                    "E019",
   412→   412→                    DiagnosticSeverity::Error,
   413→   413→                    format!("{}", e),
   414→   414→                    "ast-to-scg",
   415→   415→                    DiagnosticSourceLocation::unknown(),
   416→   416→                ));
   417→   417→                return all_diagnostics;
   418→   418→            }
   419→   419→        };
   420→   420→
   421→   421→        // Validate SCG
   422→   422→        let validation = scg.validate();
   423→   423→        if !validation.is_valid {
   424→   424→            for err in &validation.errors {
   425→   425→                all_diagnostics.push(VumaDiagnostic::new(
   426→   426→                    "E022",
   427→   427→                    DiagnosticSeverity::Error,
   428→   428→                    err.clone(),
   429→   429→                    "scg-validation",
   430→   430→                    DiagnosticSourceLocation::unknown(),
   431→   431→                ));
   432→   432→            }
   433→   433→        }
   434→   434→
   435→   435→        all_diagnostics
   436→   436→    }

   438→   438→    /// Verify a VUMA program by running all five IVE invariant checkers
   439→   439→    /// on the SCG and producing a structured verification report.
   440→   440→    ///
   441→   441→    /// This method runs the full front-end pipeline (parse → SCG),
   442→   442→    /// then invokes the IVE `InvariantAggregator` to check all five
   443→   443→    /// core invariants (liveness, exclusivity, interpretation, origin,
   444→   444→    /// cleanup) and the proof system to produce pass/fail per invariant
   445→   445→    /// with counterexamples for any violations.
   446→   446→    ///
   447→   447→    /// # Returns
   448→   448→    ///
   449→   449→    /// A [`VerificationReport`] containing:
   450→   450→    /// - Per-invariant pass/fail status
   451→   451→    /// - Counterexamples for each violation
   452→   452→    /// - An overall pass/fail verdict
   453→   453→    /// - Timing metadata
   454→   454→    ///
   455→   455→    /// # Example
   456→   456→    ///
   457→   457→    /// ```rust
   458→   458→    /// use vuma::api::VumaCompiler;
   459→   459→    ///
   460→   460→    /// let compiler = VumaCompiler::new();
   461→   461→    /// let source = "fn main() {}";
   462→   462→    /// let report = compiler.verify(source);
   463→   463→    /// println!("Overall verdict: {}", report.overall_verdict);
   464→   464→    /// for inv in &report.invariants {
   465→   465→    ///     println!("  {} — {}", inv.kind, inv.status);
   466→   466→    /// }
   467→   467→    /// ```
   468→   468→    pub fn verify(&self, source: &str) -> VerificationReport {
   469→   469→        let start = Instant::now();
   470→   470→
   471→   471→        // Run the front-end pipeline to get the SCG.
   472→   472→        let front_result = run_frontend(source, &self.config);
   473→   473→
   474→   474→        let scg = match front_result {
   475→   475→            FrontendResult::Ok { scg } => scg,
   476→   476→            FrontendResult::Err { diagnostics } => {
   477→   477→                let messages: Vec<String> =
   478→   478→                    diagnostics.iter().map(|d| d.message.clone()).collect();
   479→   479→                return VerificationReport {
   480→   480→                    overall_verdict: VerificationVerdict::Error,
   481→   481→                    invariants: Vec::new(),
   482→   482→                    diagnostics: messages,
   483→   483→                    metadata: VerificationMetadata {
   484→   484→                        total_elapsed_ms: start.elapsed().as_millis() as u64,
   485→   485→                        source_lines: source.lines().count(),
   486→   486→                        source_bytes: source.len(),
   487→   487→                    },
   488→   488→                };
   489→   489→            }
   490→   490→        };
   491→   491→
   492→   492→        // Run the IVE invariant aggregator at Normal level (all 5 checks).
   493→   493→        let aggregator = InvariantAggregator::new().with_level(IveVerificationLevel::Normal);
   494→   494→        let input = VerificationInput::from_scg(scg.clone());
   495→   495→        let aggregated = aggregator.verify_all(&input);
   496→   496→
   497→   497→        // Convert the aggregated result into per-invariant API results,
   498→   498→        // building counterexamples from the proof system for any violations.
   499→   499→        let mut invariants = Vec::with_capacity(aggregated.per_invariant.len());
   500→   500→        for pir in &aggregated.per_invariant {
   501→   501→            let kind_str = pir.kind.label().to_string();
   502→   502→
   503→   503→            let (status, counterexample) = if pir.is_pass() {
   504→   504→                (InvariantVerificationStatus::Pass, None)
   505→   505→            } else if pir.is_fail() {
   506→   506→                // Build a proof-system counterexample from the IVE violation.
   507→   507→                let proof_ce = build_proof_counterexample(&pir.result);
   508→   508→                (InvariantVerificationStatus::Fail, Some(proof_ce))
   509→   509→            } else {
   510→   510→                (InvariantVerificationStatus::Unverified, None)
   511→   511→            };
   512→   512→
   513→   513→            invariants.push(InvariantVerification {
   514→   514→                kind: kind_str,
   515→   515→                status,
   516→   516→                message: pir.result.message.clone(),
   517→   517→                elapsed_ms: pir.elapsed_ms,
   518→   518→                counterexample,
   519→   519→            });
   520→   520→        }
   521→   521→
   522→   522→        // Determine overall verdict.
   523→   523→        let overall_verdict = match aggregated.overall {
   524→   524→            vuma_ive::OverallVerdict::Pass => VerificationVerdict::Pass,
   525→   525→            vuma_ive::OverallVerdict::Fail => VerificationVerdict::Fail,
   526→   526→            vuma_ive::OverallVerdict::Inconclusive => VerificationVerdict::Inconclusive,
   527→   527→            vuma_ive::OverallVerdict::NoChecks => VerificationVerdict::Error,
   528→   528→        };
   529→   529→
   530→   530→        // Also attempt proof-system verification for a cross-check.
   531→   531→        let proof_bundle = build_proof_bundle(&scg);
   532→   532→        let proof_statuses = proof_bundle.status();
   533→   533→
   534→   534→        // If the proof system found failures that the IVE missed,
   535→   535→        // upgrade unverified results to fail.
   536→   536→        for (i, (_inv_name, proof_status)) in proof_statuses.iter().enumerate() {
   537→   537→            if i < invariants.len() {
   538→   538→                if let InvariantStatus::Failed(reason) = proof_status {
   539→   539→                    if invariants[i].status == InvariantVerificationStatus::Unverified {
   540→   540→                        invariants[i].status = InvariantVerificationStatus::Fail;
   541→   541→                        invariants[i].counterexample = Some(CounterexampleInfo {
   542→   542→                            description: reason.clone(),
   543→   543→                            execution_trace: Vec::new(),
   544→   544→                        });
   545→   545→                    }
   546→   546→                }
   547→   547→            }
   548→   548→        }
   549→   549→
   550→   550→        let diagnostics = Vec::new();
   551→   551→        let total_elapsed_ms = start.elapsed().as_millis() as u64;
   552→   552→
   553→   553→        VerificationReport {
   554→   554→            overall_verdict,
   555→   555→            invariants,
   556→   556→            diagnostics,
   557→   557→            metadata: VerificationMetadata {
   558→   558→                total_elapsed_ms,
   559→   559→                source_lines: source.lines().count(),
   560→   560→                source_bytes: source.len(),
   561→   561→            },
   562→   562→        }
   563→   563→    }
   564→   564→}
   565→   565→
   566→   566→impl Default for VumaCompiler {
   567→   567→    fn default() -> Self {
   568→   568→        Self::new()
   569→   569→    }
   570→   570→}

   726→   726→/// Serialize `Vec<u8>` as a hex string for compact JSON representation.
   727→   727→fn serialize_binary_hex<S: serde::Serializer>(data: &Vec<u8>, s: S) -> Result<S::Ok, S::Error> {
   728→   728→    let hex: String = data.iter().map(|b| format!("{:02x}", b)).collect();
   729→   729→    s.serialize_str(&hex)
   730→   730→}
   731→   731→
   732→   732→/// Deserialize `Vec<u8>` from a hex string.
   733→   733→fn deserialize_binary_hex<'de, D: serde::Deserializer<'de>>(d: D) -> Result<Vec<u8>, D::Error> {
   734→   734→    use serde::de::Error;
   735→   735→
   736→   736→    struct HexVisitor;
   737→   737→
   738→   738→    impl<'de> serde::de::Visitor<'de> for HexVisitor {
   739→   739→        type Value = Vec<u8>;
   740→   740→
   741→   741→        fn expecting(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   742→   742→            write!(f, "a hex-encoded string")
   743→   743→        }
   744→   744→
   745→   745→        fn visit_str<E: serde::de::Error>(self, v: &str) -> Result<Self::Value, E> {
   746→   746→            if v.len() % 2 != 0 {
   747→   747→                return Err(E::custom("hex string has odd length"));
   748→   748→            }
   749→   749→            (0..v.len())
   750→   750→                .step_by(2)
   751→   751→                .map(|i| {
   752→   752→                    u8::from_str_radix(&v[i..i + 2], 16)
   753→   753→                        .map_err(|e| E::custom(format!("hex decode error: {}", e)))
   754→   754→                })
   755→   755→                .collect()
   756→   756→        }
   757→   757→    }
   758→   758→
   759→   759→    d.deserialize_str(HexVisitor)
   760→   760→}

   915→   915→/// Front-end pipeline result (everything up to and including SCG transforms).
   916→   916→enum FrontendResult {
   917→   917→    Ok {
   918→   918→        scg: vuma_scg::SCG,
   919→   919→    },
   920→   920→    Err {
   921→   921→        diagnostics: Vec<VumaDiagnostic>,
   922→   922→    },
   923→   923→}
   924→   924→
   925→   925→/// Run the front-end pipeline: Parse → AST → SCG → Validate → BD Inference
   926→   926→/// → IVE Verification → SCG Transforms.
   927→   927→fn run_frontend(source: &str, config: &CompileConfig) -> FrontendResult {
   928→   928→    use vuma_ive::{InvariantAggregator, VerificationLevel as IveVerificationLevel};
   929→   929→    use vuma_parser::{AstToScg, Parser};
   930→   930→
   931→   931→    // Stage 1: Parse
   932→   932→    let mut parser = Parser::new(source);
   933→   933→    let parse_output = parser.parse_program();
   934→   934→    if parse_output.has_errors() {
   935→   935→        return FrontendResult::Err {
   936→   936→            diagnostics: diagnostics::from_parse_errors(&parse_output.errors, source, None),
   937→   937→        };
   938→   938→    }
   939→   939→    let ast = parse_output.unwrap();
   940→   940→
   941→   941→    // Stage 2: AST → SCG
   942→   942→    let mut converter = AstToScg::new();
   943→   943→    let mut scg = match converter.convert(&ast) {
   944→   944→        Ok(scg) => scg,
   945→   945→        Err(e) => {
   946→   946→            return FrontendResult::Err {
   947→   947→                diagnostics: vec![VumaDiagnostic::new(
   948→   948→                    "E019",
   949→   949→                    DiagnosticSeverity::Error,
   950→   950→                    format!("{}", e),
   951→   951→                    "ast-to-scg",
   952→   952→                    DiagnosticSourceLocation::unknown(),
   953→   953→                )],
   954→   954→            };
   955→   955→        }
   956→   956→    };
   957→   957→
   958→   958→    // Stage 3: SCG Validation (non-fatal — warn but continue)
   959→   959→    let _validation = scg.validate();
   960→   960→
   961→   961→    // Stage 4: BD Inference (refine types)
   962→   962→    let inference_engine = vuma_ive::InferenceEngine::new();
   963→   963→    let bd_results = inference_engine.infer_types(&scg);
   964→   964→    pipeline::refine_scg_types_with_bd(&mut scg, &bd_results);
   965→   965→
   966→   966→    // Stage 5: IVE Verification (non-fatal)
   967→   967→    if config.verification_level != VerificationLevel::None {
   968→   968→        let ive_level = match config.verification_level {
   969→   969→            VerificationLevel::Quick => IveVerificationLevel::Quick,
   970→   970→            VerificationLevel::Normal => IveVerificationLevel::Normal,
   971→   971→            VerificationLevel::Exhaustive => IveVerificationLevel::Exhaustive,
   972→   972→            VerificationLevel::None => unreachable!(),
   973→   973→        };
   974→   974→        let aggregator = InvariantAggregator::new().with_level(ive_level);
   975→   975→        let input = vuma_ive::verification::VerificationInput::from_scg(scg.clone());
   976→   976→        let _ = aggregator.verify_all(&input);
   977→   977→    }
   978→   978→
   979→   979→    // Stage 6: SCG Transforms
   980→   980→    pipeline::run_scg_transforms(&mut scg, config);
   981→   981→
   982→   982→    FrontendResult::Ok { scg }
   983→   983→}

   985→   985→/// Run target-specific codegen using the Backend trait.
   986→   986→fn run_backend_codegen(
   987→   987→    scg: &vuma_scg::SCG,
   988→   988→    backend_kind: vuma_codegen::backend::BackendKind,
   989→   989→) -> Result<TargetOutput, Vec<VumaDiagnostic>> {
   990→   990→    use vuma_codegen::backend::{create_backend, AllocatedProgram};
   991→   991→    use vuma_codegen::regalloc::RegAllocator;
   992→   992→    use vuma_codegen::scg_to_ir::IRBuilder;
   993→   993→
   994→   994→    // Bridge SCG to codegen SCG
   995→   995→    let codegen_scg = pipeline::bridge_scg_to_codegen(scg);
   996→   996→
   997→   997→    // IR Lowering
   998→   998→    let mut ir_builder = IRBuilder::new();
   999→   999→    let ir_program = match ir_builder.build(&codegen_scg) {
  1000→  1000→        Ok(ir) => ir,
  1001→  1001→        Err(e) => {
  1002→  1002→            return Err(vec![diagnostics::from_codegen_error(&e)]);
  1003→  1003→        }
  1004→  1004→    };
  1005→  1005→
  1006→  1006→    // Create backend
  1007→  1007→    let backend = match create_backend(backend_kind) {
  1008→  1008→        Ok(b) => b,
  1009→  1009→        Err(e) => {
  1010→  1010→            return Err(vec![VumaDiagnostic::new(
  1011→  1011→                "E023",
  1012→  1012→                DiagnosticSeverity::Error,
  1013→  1013→                format!("{}", e),
  1014→  1014→                "backend-creation",
  1015→  1015→                DiagnosticSourceLocation::unknown(),
  1016→  1016→            )]);
  1017→  1017→        }
  1018→  1018→    };
  1019→  1019→
  1020→  1020→    // Register allocation — delegate to the backend
  1021→  1021→    let mut allocated_functions = Vec::new();
  1022→  1022→    for func in &ir_program.functions {
  1023→  1023→        match backend.allocate_registers(func) {
  1024→  1024→            Ok(allocd) => allocated_functions.push(allocd),
  1025→  1025→            Err(e) => {
  1026→  1026→                return Err(vec![VumaDiagnostic::new(
  1027→  1027→                    "E017",
  1028→  1028→                    DiagnosticSeverity::Error,
  1029→  1029→                    format!("{}: {}", func.name, e),
  1030→  1030→                    "register-alloc",
  1031→  1031→                    DiagnosticSourceLocation::unknown(),
  1032→  1032→                )]);
  1033→  1033→            }
  1034→  1034→        }
  1035→  1035→    }
  1036→  1036→
  1037→  1037→    let allocated_program = AllocatedProgram {
  1038→  1038→        functions: allocated_functions,
  1039→  1039→        total_code_size: 0,
  1040→  1040→        total_data_size: 0,
  1041→  1041→    };
  1042→  1042→
  1043→  1043→    // Encode the program
  1044→  1044→    let binary = match backend.encode_program(&allocated_program) {
  1045→  1045→        Ok(binary) => binary,
  1046→  1046→        Err(e) => {
  1047→  1047→            use vuma_codegen::backend::BackendError;
  1048→  1048→            use crate::diagnostics::RelatedInfo;
  1049→  1049→            match &e {
  1050→  1050→                BackendError::UnresolvedRelocation { symbol, function, offset, reloc_type, .. } => {
  1051→  1051→                    let mut diag = VumaDiagnostic::new(
  1052→  1052→                        "E037",
  1053→  1053→                        DiagnosticSeverity::Error,
  1054→  1054→                        format!("unresolved relocation: symbol '{}' not found", symbol),
  1055→  1055→                        "codegen",
  1056→  1056→                        DiagnosticSourceLocation::unknown(),
  1057→  1057→                    );
  1058→  1058→                    diag = diag.with_related(
  1059→  1059→                        RelatedInfo::new(
  1060→  1060→                            DiagnosticSourceLocation::unknown(),
  1061→  1061→                            format!("referenced in function '{}' at offset 0x{:X} (relocation type: {})", function, offset, reloc_type),
  1062→  1062→                        ),
  1063→  1063→                    );
  1064→  1064→                    return Err(vec![diag]);
  1065→  1065→                }
  1066→  1066→                _ => {
  1067→  1067→                    return Err(vec![VumaDiagnostic::new(
  1068→  1068→                        "E020",
  1069→  1069→                        DiagnosticSeverity::Error,
  1070→  1070→                        format!("{}", e),
  1071→  1071→                        "code-emission",
  1072→  1072→                        DiagnosticSourceLocation::unknown(),
  1073→  1073→                    )]);
  1074→  1074→                }
  1075→  1075→            }
  1076→  1076→        }
  1077→  1077→    };
  1078→  1078→
  1079→  1079→    // Disassemble
  1080→  1080→    let base_addr = backend.target_info().default_base_address();
  1081→  1081→    let disasm_lines = backend.disassemble(&binary, base_addr);
  1082→  1082→    let disassembly = disasm_lines.join("\n");
  1083→  1083→
  1084→  1084→    Ok(TargetOutput {
  1085→  1085→        backend: backend_kind.isa_name().to_string(),
  1086→  1086→        binary_size: binary.len(),
  1087→  1087→        binary,
  1088→  1088→        disassembly,
  1089→  1089→    })
  1090→  1090→}

  1092→  1092→/// Build an edge index from the SCG for efficient traversal.
  1093→  1093→struct ScgEdgeIndex {
  1094→  1094→    outgoing: HashMap<vuma_scg::NodeId, Vec<vuma_scg::EdgeData>>,

  1096→  1096→}

  1098→  1098→impl ScgEdgeIndex {
  1099→  1099→    fn build(scg: &vuma_scg::SCG) -> Self {
  1100→  1100→        let mut outgoing: HashMap<vuma_scg::NodeId, Vec<vuma_scg::EdgeData>> = HashMap::new();
  1101→  1101→        let mut incoming: HashMap<vuma_scg::NodeId, Vec<vuma_scg::EdgeData>> = HashMap::new();
  1102→  1102→        for edge in scg.edges() {
  1103→  1103→            outgoing.entry(edge.source).or_default().push(edge.clone());
  1104→  1104→            incoming.entry(edge.target).or_default().push(edge.clone());
  1105→  1105→        }

  1108→  1108→

  1111→  1111→    }
  1112→  1112→
  1113→  1113→    fn incoming(&self, id: vuma_scg::NodeId) -> &[vuma_scg::EdgeData] {
  1114→  1114→        self.incoming.get(&id).map(|v| v.as_slice()).unwrap_or(&[])
  1115→  1115→    }
  1116→  1116→}

  1118→  1118→/// Build an SCG summary from the full SCG graph.
  1119→  1119→fn build_scg_summary(scg: &vuma_scg::SCG) -> ScgSummary {
  1120→  1120→    use vuma_scg::{ControlKind, EdgeKind, NodePayload};
  1121→  1121→
  1122→  1122→    let total_nodes = scg.node_count();
  1123→  1123→    let total_edges = scg.edge_count();
  1124→  1124→    let edge_idx = ScgEdgeIndex::build(scg);
  1125→  1125→
  1126→  1126→    // Identify function entry nodes and build per-function summaries
  1127→  1127→    let mut functions = Vec::new();
  1128→  1128→
  1129→  1129→    // Collect function entries
  1130→  1130→    let func_entries: Vec<(vuma_scg::NodeId, String)> = scg
  1131→  1131→        .nodes()
  1132→  1132→        .filter_map(|n| {
  1133→  1133→            if let NodePayload::Control(c) = &n.payload {
  1134→  1134→                if c.kind == ControlKind::FunctionEntry {
  1135→  1135→                    let name = c.label.clone().unwrap_or_else(|| "unknown".to_string());
  1136→  1136→                    return Some((n.id, name));
  1137→  1137→                }
  1138→  1138→            }
  1139→  1139→            None
  1140→  1140→        })
  1141→  1141→        .collect();
  1142→  1142→
  1143→  1143→    // For each function entry, count nodes reachable via ControlFlow
  1144→  1144→    for (entry_id, func_name) in &func_entries {
  1145→  1145→        let mut reachable = std::collections::HashSet::new();
  1146→  1146→        let mut queue = std::collections::VecDeque::new();
  1147→  1147→        queue.push_back(*entry_id);
  1148→  1148→        reachable.insert(*entry_id);
  1149→  1149→
  1150→  1150→        while let Some(current) = queue.pop_front() {
  1151→  1151→            for edge in edge_idx.outgoing(current) {
  1152→  1152→                if edge.kind == EdgeKind::ControlFlow && !reachable.contains(&edge.target) {
  1153→  1153→                    reachable.insert(edge.target);
  1154→  1154→                    queue.push_back(edge.target);
  1155→  1155→                }
  1156→  1156→            }
  1157→  1157→        }
  1158→  1158→
  1159→  1159→        let node_count = reachable.len();
  1160→  1160→
  1161→  1161→        // Extract parameters from outgoing DataFlow edges from the entry
  1162→  1162→        let mut params = Vec::new();
  1163→  1163→        for edge in edge_idx.outgoing(*entry_id) {
  1164→  1164→            if edge.kind == EdgeKind::DataFlow {
  1165→  1165→                if let Some(target_node) = scg.get_node(edge.target) {
  1166→  1166→                    let name = match &target_node.payload {
  1167→  1167→                        NodePayload::Allocation(a) => {
  1168→  1168→                            a.type_name.clone().unwrap_or_else(|| "param".to_string())
  1169→  1169→                        }
  1170→  1170→                        NodePayload::Computation(c) => {
  1171→  1171→                            c.result_type.clone().unwrap_or_else(|| "param".to_string())
  1172→  1172→                        }
  1173→  1173→                        _ => "param".to_string(),
  1174→  1174→                    };
  1175→  1175→                    let ty = match &target_node.payload {
  1176→  1176→                        NodePayload::Allocation(a) => {
  1177→  1177→                            a.type_name.as_deref().unwrap_or("i64").to_string()
  1178→  1178→                        }
  1179→  1179→                        NodePayload::Computation(c) => {
  1180→  1180→                            c.result_type.as_deref().unwrap_or("i64").to_string()
  1181→  1181→                        }
  1182→  1182→                        NodePayload::Cast(c) => c.to_type.clone(),
  1183→  1183→                        _ => "i64".to_string(),
  1184→  1184→                    };
  1185→  1185→                    params.push((name, ty));
  1186→  1186→                }
  1187→  1187→            }
  1188→  1188→        }
  1189→  1189→
  1190→  1190→        // Find calls within this function
  1191→  1191→        let mut calls = Vec::new();
  1192→  1192→        for node_id in &reachable {
  1193→  1193→            if let Some(node) = scg.get_node(*node_id) {
  1194→  1194→                if let NodePayload::Computation(comp) = &node.payload {
  1195→  1195→                    let op_label = comp.kind.label();
  1196→  1196→                    if !is_known_binop(&op_label) && !op_label.starts_with('_') {
  1197→  1197→                        if !calls.contains(&op_label) {
  1198→  1198→                            calls.push(op_label);
  1199→  1199→                        }
  1200→  1200→                    }
  1201→  1201→                }
  1202→  1202→            }
  1203→  1203→        }
  1204→  1204→
  1205→  1205→        // Determine return type
  1206→  1206→        let return_type = find_return_type(scg, *entry_id, &edge_idx);
  1207→  1207→
  1208→  1208→        functions.push(FunctionSummary {
  1209→  1209→            name: func_name.clone(),
  1210→  1210→            params,
  1211→  1211→            return_type,
  1212→  1212→            node_count,
  1213→  1213→            calls,
  1214→  1214→        });
  1215→  1215→    }
  1216→  1216→
  1217→  1217→    // If no function entries were found, create a single "main" summary
  1218→  1218→    if functions.is_empty() {
  1219→  1219→        let mut calls = Vec::new();
  1220→  1220→        for node in scg.nodes() {
  1221→  1221→            if let NodePayload::Computation(comp) = &node.payload {
  1222→  1222→                let op_label = comp.kind.label();
  1223→  1223→                if !is_known_binop(&op_label) && !op_label.starts_with('_') {
  1224→  1224→                    if !calls.contains(&op_label) {
  1225→  1225→                        calls.push(op_label);
  1226→  1226→                    }
  1227→  1227→                }
  1228→  1228→            }
  1229→  1229→        }
  1230→  1230→
  1231→  1231→        functions.push(FunctionSummary {
  1232→  1232→            name: "main".to_string(),
  1233→  1233→            params: Vec::new(),
  1234→  1234→            return_type: "void".to_string(),
  1235→  1235→            node_count: total_nodes,
  1236→  1236→            calls,
  1237→  1237→        });
  1238→  1238→    }
  1239→  1239→
  1240→  1240→    ScgSummary {
  1241→  1241→        function_count: functions.len(),
  1242→  1242→        functions,
  1243→  1243→        total_nodes,
  1244→  1244→        total_edges,
  1245→  1245→    }
  1246→  1246→}

  1248→  1248→/// Check if an operation string is a known binary operation.
  1249→  1249→fn is_known_binop(op: &str) -> bool {
  1250→  1250→    matches!(
  1251→  1251→        op,
  1252→  1252→        "add" | "sub" | "mul" | "sdiv" | "udiv" | "srem" | "urem" | "and" | "or" | "xor"
  1253→  1253→            | "shl" | "shr.l" | "shr.a" | "slt" | "sle" | "sgt" | "sge" | "ult" | "ule"
  1254→  1254→            | "ugt" | "uge" | "eq" | "ne" | "+" | "-" | "*" | "/" | "%" | "&" | "|"
  1255→  1255→            | "^" | "<<" | ">>" | "<" | "<=" | ">" | ">=" | "=="
  1256→  1256→    )
  1257→  1257→}

  1259→  1259→/// Find the return type of a function by tracing to its FunctionReturn node.
  1260→  1260→fn find_return_type(
  1261→  1261→    scg: &vuma_scg::SCG,
  1262→  1262→    entry_id: vuma_scg::NodeId,
  1263→  1263→    edge_idx: &ScgEdgeIndex,
  1264→  1264→) -> String {
  1265→  1265→    use vuma_scg::{ControlKind, EdgeKind, NodePayload};
  1266→  1266→
  1267→  1267→    let mut visited = std::collections::HashSet::new();
  1268→  1268→    let mut queue = std::collections::VecDeque::new();
  1269→  1269→    queue.push_back(entry_id);
  1270→  1270→    visited.insert(entry_id);
  1271→  1271→
  1272→  1272→    while let Some(current) = queue.pop_front() {
  1273→  1273→        for edge in edge_idx.outgoing(current) {
  1274→  1274→            if edge.kind == EdgeKind::ControlFlow && !visited.contains(&edge.target) {
  1275→  1275→                visited.insert(edge.target);
  1276→  1276→                if let Some(node) = scg.get_node(edge.target) {
  1277→  1277→                    if let NodePayload::Control(c) = &node.payload {
  1278→  1278→                        if c.kind == ControlKind::FunctionReturn {
  1279→  1279→                            // Check incoming DataFlow edges for return value type
  1280→  1280→                            for ret_edge in edge_idx.incoming(edge.target) {
  1281→  1281→                                if ret_edge.kind == EdgeKind::DataFlow {
  1282→  1282→                                    if let Some(src) = scg.get_node(ret_edge.source) {
  1283→  1283→                                        if let NodePayload::Computation(comp) = &src.payload {
  1284→  1284→                                            if let Some(rt) = &comp.result_type {
  1285→  1285→                                                return rt.clone();
  1286→  1286→                                            }
  1287→  1287→                                        }
  1288→  1288→                                    }
  1289→  1289→                                }
  1290→  1290→                            }
  1291→  1291→                            return "void".to_string();
  1292→  1292→                        }
  1293→  1293→                    }
  1294→  1294→                }
  1295→  1295→                queue.push_back(edge.target);
  1296→  1296→            }
  1297→  1297→        }
  1298→  1298→    }
  1299→  1299→
  1300→  1300→    "void".to_string()
  1301→  1301→}

  1368→  1368→/// Build a proof-system counterexample from an IVE verification result.
  1369→  1369→///
  1370→  1370→/// Takes the IVE `VerificationResult` (which uses its own counterexample
  1371→  1371→/// format) and converts it into a proof-system `CounterExample`, then
  1372→  1372→/// extracts the relevant information into the API's `CounterexampleInfo`.
  1373→  1373→fn build_proof_counterexample(
  1374→  1374→    result: &vuma_ive::result::VerificationResult,
  1375→  1375→) -> CounterexampleInfo {
  1376→  1376→    use vuma_ive::result::VerificationStatus;
  1377→  1377→
  1378→  1378→    match &result.status {
  1379→  1379→        VerificationStatus::Violated { counterexample } => {
  1380→  1380→            // Convert the IVE counterexample into a proof-system
  1381→  1381→            // counterexample for structural consistency.
  1382→  1382→            let proof_inv = match result.invariant.as_str() {
  1383→  1383→                "liveness" => vuma_proof::proof::InvariantName::Liveness,
  1384→  1384→                "exclusivity" => vuma_proof::proof::InvariantName::Exclusivity,
  1385→  1385→                "cleanup" => vuma_proof::proof::InvariantName::Cleanup,
  1386→  1386→                "origin" => vuma_proof::proof::InvariantName::Origin,

  1390→  1390→
  1391→  1391→            let violation_point = ViolationPoint::new(
  1392→  1392→                proof_inv,

  1394→  1394→                0, // program offset
  1395→  1395→            );

  1397→  1397→            let minimal_ce = proof_ce.minimal();

  1400→  1400→            let trace: Vec<String> = minimal_ce
  1401→  1401→                .execution
  1402→  1402→                .iter()
  1403→  1403→                .map(|step| step.to_string())
  1404→  1404→                .collect();
  1405→  1405→
  1406→  1406→            CounterexampleInfo {
  1407→  1407→                description: counterexample.description.clone(),
  1408→  1408→                execution_trace: trace,
  1409→  1409→            }
  1410→  1410→        }
  1411→  1411→        _ => CounterexampleInfo {
  1412→  1412→            description: result.message.clone(),
  1413→  1413→            execution_trace: Vec::new(),
  1414→  1414→        },
  1415→  1415→    }
