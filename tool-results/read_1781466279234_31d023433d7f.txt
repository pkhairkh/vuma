     1→     1→//! Profile-guided optimization data collection and analysis.
     2→     2→//!
     3→     3→//! The profiling subsystem continuously collects runtime observations — branch
     4→     4→//! directions, call frequencies, edge traversal counts, hardware performance
     5→     5→//! counter values — and exposes methods to identify hot paths, cold spots, and
     6→     6→//! generate optimization recommendations.
     7→     7→//!
     8→     8→//! # Design notes
     9→     9→//!
    10→    10→//! * **Always-on profiling**: COR continuously collects profile data as the
    11→    11→//!   program executes. There is no separate "profiled run"; the data
    12→    12→//!   converges incrementally.
    13→    13→//! * **Low overhead**: Counters are simple `u64` bumps; no heap allocation
    14→    14→//!   occurs on the fast path.
    15→    15→//! * **Region-class granularity**: Allocation statistics are partitioned by
    16→    16→//!   *region class* (a coarse grouping of compiled regions) so the optimizer
    17→    17→//!   can make decisions without examining every individual region.
    18→    18→//! * **Pi 5 PMU support**: Hardware performance counters (cycles, instructions,
    19→    19→//!   cache misses, branch misses) are collected per sample when running on
    20→    20→//!   Raspberry Pi 5, enabling micro-architectural optimisation.
    21→    21→//!
    22→    22→//! # Core types
    23→    23→//!
    24→    24→//! * [`ProfileCollector`] — thread-safe collector that accumulates profiling
    25→    25→//!   samples and produces [`ProfileData`].
    26→    26→//! * [`ProfileData`] — execution counts per SCG node, edge traversal
    27→    27→//!   frequencies, and hot-path information.
    28→    28→//! * [`ProfileSample`] — a single profiling sample: timestamp, node id,
    29→    29→//!   execution time in nanoseconds, and optional PMU counters.
    30→    30→//! * [`HotPath`] — a sequence of nodes that collectively account for >80 %
    31→    31→//!   of total execution time.
    32→    32→//! * [`ProfileReport`] — a full analysis of profiling data including hot
    33→    33→//!   spots, cold spots, and optimisation recommendations.
    34→    34→

    43→    43→// ---------------------------------------------------------------------------
    44→    44→// Mutex poisoning recovery helper
    45→    45→// ---------------------------------------------------------------------------
    46→    46→
    47→    47→/// Locks a `Mutex<T>`, recovering from poisoning if necessary.
    48→    48→///
    49→    49→/// If a thread panicked while holding the lock, the Mutex becomes
    50→    50→/// "poisoned" and normal `lock().unwrap()` would panic. This helper
    51→    51→/// instead recovers the guard by extracting the inner data from the
    52→    52→/// `PoisonError`, allowing the runtime to continue operating with
    53→    53→/// the last consistent state.
    54→    54→///
    55→    55→/// This is appropriate for profiling data where a stale but valid
    56→    56→/// snapshot is preferable to a runtime crash.
    57→    57→fn lock_profile<T>(mutex: &Mutex<T>) -> MutexGuard<'_, T> {
    58→    58→    mutex.lock().unwrap_or_else(|e| e.into_inner())
    59→    59→}

    61→    61→// ---------------------------------------------------------------------------
    62→    62→// Pi 5 Hardware Performance Counters (PMU)
    63→    63→// ---------------------------------------------------------------------------
    64→    64→
    65→    65→/// Raspberry Pi 5 hardware performance counter snapshot.
    66→    66→///
    67→    67→/// The Pi 5's Cortex-A76 cores expose PMU counters via the `cntvct_el0`
    68→    68→/// virtual counter and the ARMv8.2-A PMU event system. This struct captures
    69→    69→/// the four most useful events for optimisation guidance:
    70→    70→///
    71→    71→/// * **Cycle count** — total CPU cycles elapsed (via `PMCCNTR_EL0`).
    72→    72→/// * **Instruction count** — instructions architecturally executed.
    73→    73→/// * **Cache misses** — L1 data cache refills from main memory.
    74→    74→/// * **Branch misses** — branches mispredicted at the branch-prediction unit.
    75→    75→///
    76→    76→/// On targets where PMU counters are unavailable (e.g. user-space without
    77→    77→/// `perf_event_open` access), all fields default to zero and are simply
    78→    78→/// ignored by the analysis pipeline.
    79→    79→#[derive(Debug, Clone, Default, Serialize, Deserialize)]
    80→    80→pub struct Pi5PmuCounters {
    81→    81→    /// Total CPU cycles elapsed during the sampled region.
    82→    82→    pub cycle_count: u64,
    83→    83→    /// Instructions architecturally executed.
    84→    84→    pub instruction_count: u64,
    85→    85→    /// L1 data-cache misses (refills from DRAM).
    86→    86→    pub cache_misses: u64,
    87→    87→    /// Branch mispredictions.
    88→    88→    pub branch_misses: u64,
    89→    89→}
    90→    90→
    91→    91→impl Pi5PmuCounters {
    92→    92→    /// Creates a zeroed PMU counter snapshot.
    93→    93→    pub fn new() -> Self {
    94→    94→        Self::default()
    95→    95→    }
    96→    96→
    97→    97→    /// Computes instructions-per-cycle (IPC). Returns 0.0 if cycle_count is 0.
    98→    98→    pub fn ipc(&self) -> f64 {
    99→    99→        if self.cycle_count == 0 {
   100→   100→            0.0
   101→   101→        } else {
   102→   102→            self.instruction_count as f64 / self.cycle_count as f64
   103→   103→        }
   104→   104→    }
   105→   105→
   106→   106→    /// Computes cache miss rate (misses / instructions). Returns 0.0 if
   107→   107→    /// instruction_count is 0.
   108→   108→    pub fn cache_miss_rate(&self) -> f64 {
   109→   109→        if self.instruction_count == 0 {
   110→   110→            0.0
   111→   111→        } else {
   112→   112→            self.cache_misses as f64 / self.instruction_count as f64
   113→   113→        }
   114→   114→    }
   115→   115→
   116→   116→    /// Computes branch misprediction rate. Returns 0.0 if instruction_count is 0.
   117→   117→    pub fn branch_miss_rate(&self) -> f64 {
   118→   118→        if self.instruction_count == 0 {
   119→   119→            0.0
   120→   120→        } else {
   121→   121→            self.branch_misses as f64 / self.instruction_count as f64
   122→   122→        }
   123→   123→    }
   124→   124→}

   127→   127→// AllocStats
   128→   128→// ---------------------------------------------------------------------------
   129→   129→
   130→   130→/// Statistics about memory allocation behaviour within the runtime.
   131→   131→///
   132→   132→/// `AllocStats` tracks aggregate allocation counts as well as per-region-class
   133→   133→/// breakdowns, enabling the optimizer to identify allocation-heavy regions and
   134→   134→/// suggest strategies such as arena allocation or object pooling.
   135→   135→#[derive(Debug, Clone, Default, Serialize, Deserialize)]
   136→   136→pub struct AllocStats {
   137→   137→    /// Total number of heap allocations observed.
   138→   138→    pub total_allocations: u64,
   139→   139→    /// Total number of heap frees observed.
   140→   140→    pub total_frees: u64,
   141→   141→    /// Peak observed live heap usage (in bytes).
   142→   142→    pub peak_usage: u64,
   143→   143→    /// Allocation statistics partitioned by region class.
   144→   144→    ///
   145→   145→    /// The key is a human-readable region class label (e.g. `"hot_loop"`,
   146→   146→    /// `"cold_path"`); the value is the number of allocations attributed to
   147→   147→    /// that class.
   148→   148→    pub by_region_class: HashMap<String, u64>,
   149→   149→}
   150→   150→
   151→   151→impl AllocStats {
   152→   152→    /// Creates an empty `AllocStats`.
   153→   153→    pub fn new() -> Self {
   154→   154→        Self::default()
   155→   155→    }
   156→   156→
   157→   157→    /// Records a single allocation of `size` bytes in the given region class.
   158→   158→    pub fn record_alloc(&mut self, size: u64, region_class: &str) {
   159→   159→        self.total_allocations += 1;
   160→   160→        *self
   161→   161→            .by_region_class
   162→   162→            .entry(region_class.to_owned())
   163→   163→            .or_insert(0) += 1;
   164→   164→        // A simple peak-tracking heuristic: each alloc adds to live set.
   165→   165→        // In a real implementation we'd track the current live set precisely.
   166→   166→        let live = self.total_allocations.saturating_sub(self.total_frees);
   167→   167→        let estimated_usage = live * size; // rough estimate
   168→   168→        if estimated_usage > self.peak_usage {
   169→   169→            self.peak_usage = estimated_usage;
   170→   170→        }
   171→   171→    }
   172→   172→
   173→   173→    /// Records a single free in the given region class.
   174→   174→    pub fn record_free(&mut self, region_class: &str) {
   175→   175→        self.total_frees += 1;
   176→   176→        // We don't decrement by_region_class since that tracks cumulative
   177→   177→        // allocations per class, not live objects.
   178→   178→        let _ = region_class; // acknowledged
   179→   179→    }
   180→   180→
   181→   181→    /// Returns the current number of live allocations (approximate).
   182→   182→    pub fn live_allocations(&self) -> u64 {
   183→   183→        self.total_allocations.saturating_sub(self.total_frees)
   184→   184→    }
   185→   185→}

   236→   236→// HotPath
   237→   237→// ---------------------------------------------------------------------------
   238→   238→
   239→   239→/// A hot path: a sequence of SCG nodes that collectively account for more
   240→   240→/// than 80 % of total observed execution time.
   241→   241→///
   242→   242→/// The nodes are ordered by descending execution time (hottest first). The
   243→   243→/// `total_time_ns` field records the sum of execution times across all nodes
   244→   244→/// in the path, while `cumulative_fraction` records what fraction of the
   245→   245→/// *global* execution time this path represents.
   246→   246→#[derive(Debug, Clone, Serialize, Deserialize)]
   247→   247→pub struct HotPath {
   248→   248→    /// Nodes on the hot path, sorted hottest first.
   249→   249→    pub nodes: Vec<NodeId>,
   250→   250→    /// Total execution time (ns) of all nodes on this path.
   251→   251→    pub total_time_ns: u64,
   252→   252→    /// Fraction of global execution time accounted for (0.0 – 1.0).
   253→   253→    pub cumulative_fraction: f64,
   254→   254→}
   255→   255→
   256→   256→impl Default for HotPath {
   257→   257→    fn default() -> Self {
   258→   258→        Self::new()
   259→   259→    }
   260→   260→}
   261→   261→
   262→   262→impl HotPath {
   263→   263→    /// Creates an empty hot path.
   264→   264→    pub fn new() -> Self {
   265→   265→        Self {
   266→   266→            nodes: Vec::new(),
   267→   267→            total_time_ns: 0,
   268→   268→            cumulative_fraction: 0.0,
   269→   269→        }
   270→   270→    }
   271→   271→
   272→   272→    /// Returns `true` if the hot path accounts for >80 % of execution time.
   273→   273→    pub fn is_dominant(&self) -> bool {
   274→   274→        self.cumulative_fraction > 0.80
   275→   275→    }
   276→   276→}

   278→   278→// ---------------------------------------------------------------------------
   279→   279→// ProfileData
   280→   280→// ---------------------------------------------------------------------------
   281→   281→
   282→   282→/// Collected profile data used to guide runtime optimizations.
   283→   283→///
   284→   284→/// The struct is updated continuously as the program executes. The optimizer
   285→   285→/// periodically reads it to decide which regions to recompile at higher
   286→   286→/// optimization levels or speculatively specialize.
   287→   287→#[derive(Debug, Clone, Default, Serialize, Deserialize)]
   288→   288→pub struct ProfileData {
   289→   289→    /// Per-node execution counts (how many times each SCG node was entered).
   290→   290→    pub call_counts: HashMap<NodeId, u64>,
   291→   291→
   292→   292→    /// Per-edge traversal frequencies (how many times each SCG edge was taken).
   293→   293→    pub edge_frequencies: HashMap<EdgeId, u64>,
   294→   294→
   295→   295→    /// Per-node cumulative execution time in nanoseconds.
   296→   296→    pub node_time_ns: HashMap<NodeId, u64>,
   297→   297→
   298→   298→    /// Aggregated Pi 5 PMU counters per node.
   299→   299→    pub node_pmu: HashMap<NodeId, Pi5PmuCounters>,
   300→   300→
   301→   301→    /// Allocation statistics.
   302→   302→    pub allocation_stats: AllocStats,
   303→   303→
   304→   304→    /// Hot paths cache (recomputed on demand).
   305→   305→    hot_paths_cache: Vec<HotPath>,
   306→   306→}
   307→   307→
   308→   308→impl ProfileData {
   309→   309→    /// Creates empty profile data.
   310→   310→    pub fn new() -> Self {
   311→   311→        Self::default()
   312→   312→    }
   313→   313→
   314→   314→    /// Records an access (execution) of the given node with optional execution
   315→   315→    /// time.
   316→   316→    pub fn record_access(&mut self, node_id: NodeId) {
   317→   317→        let count = self.call_counts.entry(node_id).or_insert(0);
   318→   318→        *count += 1;
   319→   319→    }
   320→   320→
   321→   321→    /// Records an execution of the given node with a known execution time.
   322→   322→    pub fn record_access_timed(&mut self, node_id: NodeId, time_ns: u64) {
   323→   323→        let count = self.call_counts.entry(node_id).or_insert(0);
   324→   324→        *count += 1;
   325→   325→        let time = self.node_time_ns.entry(node_id).or_insert(0);
   326→   326→        *time += time_ns;
   327→   327→    }
   328→   328→
   329→   329→    /// Records a call (invocation) of the given node, equivalent to
   330→   330→    /// `record_access` but with a distinct semantic to support future
   331→   331→    /// call-graph construction.
   332→   332→    pub fn record_call(&mut self, node_id: NodeId) {
   333→   333→        let count = self.call_counts.entry(node_id).or_insert(0);
   334→   334→        *count += 1;
   335→   335→    }
   336→   336→
   337→   337→    /// Records a traversal of the given SCG edge.
   338→   338→    pub fn record_edge(&mut self, edge_id: EdgeId) {
   339→   339→        let freq = self.edge_frequencies.entry(edge_id).or_insert(0);
   340→   340→        *freq += 1;
   341→   341→    }
   342→   342→
   343→   343→    /// Records PMU counter values for a node (accumulates into existing values).
   344→   344→    pub fn record_pmu(&mut self, node_id: NodeId, pmu: &Pi5PmuCounters) {
   345→   345→        let entry = self.node_pmu.entry(node_id).or_default();
   346→   346→        entry.cycle_count += pmu.cycle_count;
   347→   347→        entry.instruction_count += pmu.instruction_count;
   348→   348→        entry.cache_misses += pmu.cache_misses;
   349→   349→        entry.branch_misses += pmu.branch_misses;
   350→   350→    }
   351→   351→
   352→   352→    /// Ingests a batch of [`ProfileSample`] values, updating all internal
   353→   353→    /// counters.
   354→   354→    pub fn ingest_samples(&mut self, samples: &[ProfileSample]) {
   355→   355→        for sample in samples {
   356→   356→            self.record_access_timed(sample.node_id, sample.execution_time_ns);
   357→   357→            if let Some(ref pmu) = sample.pmu {
   358→   358→                self.record_pmu(sample.node_id, pmu);
   359→   359→            }
   360→   360→        }
   361→   361→    }
   362→   362→
   363→   363→    /// Computes the total execution time across all nodes (ns).
   364→   364→    pub fn total_execution_time_ns(&self) -> u64 {
   365→   365→        self.node_time_ns.values().sum()
   366→   366→    }
   367→   367→
   368→   368→    /// Returns hot paths whose nodes collectively account for >80 % of total
   369→   369→    /// execution time.
   370→   370→    ///
   371→   371→    /// The algorithm sorts all nodes by descending cumulative execution time
   372→   372→    /// and greedily adds nodes to the hot path until the 80 % threshold is
   373→   373→    /// crossed.
   374→   374→    pub fn compute_hot_paths(&mut self) -> Vec<HotPath> {
   375→   375→        let total = self.total_execution_time_ns();
   376→   376→        if total == 0 {
   377→   377→            self.hot_paths_cache = Vec::new();
   378→   378→            return Vec::new();
   379→   379→        }
   380→   380→
   381→   381→        // Sort nodes by cumulative time, descending.
   382→   382→        let mut nodes: Vec<(NodeId, u64)> =
   383→   383→            self.node_time_ns.iter().map(|(&n, &t)| (n, t)).collect();
   384→   384→        nodes.sort_by_key(|b| std::cmp::Reverse(b.1));
   385→   385→
   386→   386→        let mut accumulated: u64 = 0;
   387→   387→        let mut path_nodes: Vec<NodeId> = Vec::new();
   388→   388→
   389→   389→        for (node_id, time_ns) in &nodes {
   390→   390→            path_nodes.push(*node_id);
   391→   391→            accumulated += time_ns;
   392→   392→            let fraction = accumulated as f64 / total as f64;
   393→   393→            if fraction > 0.80 {
   394→   394→                let hot_path = HotPath {
   395→   395→                    nodes: path_nodes.clone(),
   396→   396→                    total_time_ns: accumulated,
   397→   397→                    cumulative_fraction: fraction,
   398→   398→                };
   399→   399→                self.hot_paths_cache = vec![hot_path];
   400→   400→                return self.hot_paths_cache.clone();
   401→   401→            }
   402→   402→        }
   403→   403→
   404→   404→        // If we get here, even all nodes together don't exceed 80 % (edge
   405→   405→        // case with very few samples). Return whatever we have.
   406→   406→        let fraction = accumulated as f64 / total as f64;
   407→   407→        let hot_path = HotPath {
   408→   408→            nodes: path_nodes,
   409→   409→            total_time_ns: accumulated,
   410→   410→            cumulative_fraction: fraction,
   411→   411→        };
   412→   412→        self.hot_paths_cache = vec![hot_path];
   413→   413→        self.hot_paths_cache.clone()
   414→   414→    }
   415→   415→
   416→   416→    /// Returns the top `k` hottest nodes by access count, sorted descending.
   417→   417→    pub fn get_hot_paths(&mut self, k: usize) -> Vec<(NodeId, u64)> {
   418→   418→        let mut pairs: Vec<(NodeId, u64)> =
   419→   419→            self.call_counts.iter().map(|(&n, &c)| (n, c)).collect();
   420→   420→        pairs.sort_by_key(|b| std::cmp::Reverse(b.1));
   421→   421→        pairs.truncate(k);
   422→   422→        pairs
   423→   423→    }
   424→   424→
   425→   425→    /// Returns nodes with zero or negligible execution (cold spots).
   426→   426→    ///
   427→   427→    /// A node is considered "cold" if it has a call count of zero or its
   428→   428→    /// cumulative execution time is less than `cold_threshold_ns`.
   429→   429→    pub fn cold_spots(&self, all_nodes: &[NodeId], cold_threshold_ns: u64) -> Vec<NodeId> {
   430→   430→        all_nodes
   431→   431→            .iter()
   432→   432→            .copied()
   433→   433→            .filter(|&n| {
   434→   434→                let count = self.call_counts.get(&n).copied().unwrap_or(0);
   435→   435→                let time = self.node_time_ns.get(&n).copied().unwrap_or(0);
   436→   436→                count == 0 || time < cold_threshold_ns
   437→   437→            })
   438→   438→            .collect()
   439→   439→    }
   440→   440→
   441→   441→    /// Returns a list of optimization suggestions based on the collected
   442→   442→    /// profile data.
   443→   443→    pub fn suggest_optimizations(&self) -> Vec<OptimizationSuggestion> {
   444→   444→        let mut suggestions = Vec::new();
   445→   445→
   446→   446→        // Suggest inlining for the top 3 hottest nodes.
   447→   447→        let mut nodes: Vec<(NodeId, u64)> =
   448→   448→            self.call_counts.iter().map(|(&n, &c)| (n, c)).collect();
   449→   449→        nodes.sort_by_key(|b| std::cmp::Reverse(b.1));
   450→   450→        for (node_id, count) in nodes.iter().take(3) {
   451→   451→            if *count > 100 {
   452→   452→                suggestions.push(OptimizationSuggestion {
   453→   453→                    kind: SuggestionKind::Inline,
   454→   454→                    target_node: *node_id,
   455→   455→                    reason: format!("Node called {} times — candidate for inlining", count),
   456→   456→                });
   457→   457→            }
   458→   458→        }
   459→   459→
   460→   460→        // Suggest arena allocation for high-allocation region classes.
   461→   461→        for (class, allocs) in &self.allocation_stats.by_region_class {
   462→   462→            if *allocs > 50 {
   463→   463→                suggestions.push(OptimizationSuggestion {
   464→   464→                    kind: SuggestionKind::ArenaAlloc,
   465→   465→                    target_node: 0, // no specific node; applies to class
   466→   466→                    reason: format!(
   467→   467→                        "Region class '{}' has {} allocations — consider arena allocation",
   468→   468→                        class, allocs
   469→   469→                    ),
   470→   470→                });
   471→   471→            }
   472→   472→        }
   473→   473→
   474→   474→        // Suggest cache optimisation for nodes with high cache miss rates.
   475→   475→        for (&node_id, pmu) in &self.node_pmu {
   476→   476→            if pmu.cache_miss_rate() > 0.05 {
   477→   477→                suggestions.push(OptimizationSuggestion {
   478→   478→                    kind: SuggestionKind::CacheOptimize,
   479→   479→                    target_node: node_id,
   480→   480→                    reason: format!(
   481→   481→                        "Node {} has {:.1}% cache miss rate — consider data layout optimisation",
   482→   482→                        node_id,
   483→   483→                        pmu.cache_miss_rate() * 100.0
   484→   484→                    ),
   485→   485→                });
   486→   486→            }
   487→   487→        }
   488→   488→
   489→   489→        // Suggest branch prediction layout for high branch miss rates.
   490→   490→        for (&node_id, pmu) in &self.node_pmu {
   491→   491→            if pmu.branch_miss_rate() > 0.03 {
   492→   492→                suggestions.push(OptimizationSuggestion {
   493→   493→                    kind: SuggestionKind::BranchLayout,
   494→   494→                    target_node: node_id,
   495→   495→                    reason: format!(
   496→   496→                        "Node {} has {:.1}% branch misprediction rate — consider branch layout hints",
   497→   497→                        node_id,
   498→   498→                        pmu.branch_miss_rate() * 100.0
   499→   499→                    ),
   500→   500→                });
   501→   501→            }
   502→   502→        }
   503→   503→
   504→   504→        suggestions
   505→   505→    }
   506→   506→}

   509→   509→// ProfileCollector
   510→   510→// ---------------------------------------------------------------------------
   511→   511→
   512→   512→/// Thread-safe runtime profiling data collector.
   513→   513→///
   514→   514→/// `ProfileCollector` is the central object through which the COR records
   515→   515→/// profiling observations. It wraps a [`ProfileData`] instance in a
   516→   516→/// [`Mutex`] so it can be shared across threads, and provides convenience
   517→   517→/// methods for recording samples and edges.
   518→   518→///
   519→   519→/// # Usage
   520→   520→///
   521→   521→/// ```no_run
   522→   522→/// use vuma_cor::profile::ProfileCollector;
   523→   523→/// use vuma_cor::types::NodeId;
   524→   524→///
   525→   525→/// let collector = ProfileCollector::new();
   526→   526→/// collector.record_access(42);
   527→   527→/// collector.record_edge(7);
   528→   528→/// let data = collector.snapshot();
   529→   529→/// ```
   530→   530→pub struct ProfileCollector {
   531→   531→    /// Inner profile data, protected by a mutex.
   532→   532→    data: Mutex<ProfileData>,
   533→   533→    /// Nanosecond origin used for computing relative timestamps.
   534→   534→    epoch: Instant,
   535→   535→    /// Atomic counter for total number of samples collected (fast-path read).
   536→   536→    sample_count: AtomicU64,
   537→   537→}
   538→   538→
   539→   539→impl ProfileCollector {
   540→   540→    /// Creates a new, empty profile collector.
   541→   541→    pub fn new() -> Self {
   542→   542→        Self {
   543→   543→            data: Mutex::new(ProfileData::new()),
   544→   544→            epoch: Instant::now(),
   545→   545→            sample_count: AtomicU64::new(0),
   546→   546→        }
   547→   547→    }
   548→   548→
   549→   549→    /// Returns the number of samples collected so far.
   550→   550→    pub fn sample_count(&self) -> u64 {
   551→   551→        self.sample_count.load(Ordering::Relaxed)
   552→   552→    }
   553→   553→
   554→   554→    /// Records an execution of the given SCG node.
   555→   555→    pub fn record_access(&self, node_id: NodeId) {
   556→   556→        let mut data = lock_profile(&self.data);
   557→   557→        data.record_access(node_id);
   558→   558→        self.sample_count.fetch_add(1, Ordering::Relaxed);
   559→   559→    }
   560→   560→
   561→   561→    /// Records an execution of the given node with timing information.
   562→   562→    pub fn record_access_timed(&self, node_id: NodeId, time_ns: u64) {
   563→   563→        let mut data = lock_profile(&self.data);
   564→   564→        data.record_access_timed(node_id, time_ns);
   565→   565→        self.sample_count.fetch_add(1, Ordering::Relaxed);
   566→   566→    }
   567→   567→
   568→   568→    /// Records a traversal of the given SCG edge.
   569→   569→    pub fn record_edge(&self, edge_id: EdgeId) {
   570→   570→        let mut data = lock_profile(&self.data);
   571→   571→        data.record_edge(edge_id);
   572→   572→    }
   573→   573→
   574→   574→    /// Records a full [`ProfileSample`].
   575→   575→    pub fn record_sample(&self, sample: &ProfileSample) {
   576→   576→        let mut data = lock_profile(&self.data);
   577→   577→        data.record_access_timed(sample.node_id, sample.execution_time_ns);
   578→   578→        if let Some(ref pmu) = sample.pmu {
   579→   579→            data.record_pmu(sample.node_id, pmu);
   580→   580→        }
   581→   581→        self.sample_count.fetch_add(1, Ordering::Relaxed);
   582→   582→    }
   583→   583→
   584→   584→    /// Records PMU counter values for a node.
   585→   585→    pub fn record_pmu(&self, node_id: NodeId, pmu: &Pi5PmuCounters) {
   586→   586→        let mut data = lock_profile(&self.data);
   587→   587→        data.record_pmu(node_id, pmu);
   588→   588→    }
   589→   589→
   590→   590→    /// Creates a [`ProfileSample`] using the collector's internal clock as
   591→   591→    /// the timestamp.
   592→   592→    pub fn make_sample(&self, node_id: NodeId, execution_time_ns: u64) -> ProfileSample {
   593→   593→        let timestamp_ns = self.epoch.elapsed().as_nanos() as u64;
   594→   594→        ProfileSample::new(timestamp_ns, node_id, execution_time_ns)
   595→   595→    }
   596→   596→
   597→   597→    /// Creates a [`ProfileSample`] with PMU counters using the collector's
   598→   598→    /// internal clock.
   599→   599→    pub fn make_sample_with_pmu(
   600→   600→        &self,
   601→   601→        node_id: NodeId,
   602→   602→        execution_time_ns: u64,
   603→   603→        pmu: Pi5PmuCounters,
   604→   604→    ) -> ProfileSample {
   605→   605→        let timestamp_ns = self.epoch.elapsed().as_nanos() as u64;
   606→   606→        ProfileSample::with_pmu(timestamp_ns, node_id, execution_time_ns, pmu)
   607→   607→    }
   608→   608→
   609→   609→    /// Takes a snapshot of the current profile data.
   610→   610→    ///
   611→   611→    /// This clones the internal `ProfileData`, so the caller gets a
   612→   612→    /// consistent view while the collector continues to accumulate.
   613→   613→    pub fn snapshot(&self) -> ProfileData {
   614→   614→        lock_profile(&self.data).clone()
   615→   615→    }
   616→   616→
   617→   617→    /// Resets all collected profile data.
   618→   618→    pub fn reset(&self) {
   619→   619→        let mut data = lock_profile(&self.data);
   620→   620→        *data = ProfileData::new();
   621→   621→        self.sample_count.store(0, Ordering::Relaxed);
   622→   622→        let _ = self.epoch; // epoch stays the same — timestamps remain monotonic
   623→   623→    }
   624→   624→}
   625→   625→
   626→   626→impl Default for ProfileCollector {
   627→   627→    fn default() -> Self {
   628→   628→        Self::new()
   629→   629→    }
   630→   630→}

   633→   633→// ProfileReport
   634→   634→// ---------------------------------------------------------------------------
   635→   635→
   636→   636→/// A comprehensive analysis report produced from profiling data.
   637→   637→///
   638→   638→/// `ProfileReport` is the output of [`collect_profile`] and contains hot
   639→   639→/// spots, cold spots, hot paths, and optimisation recommendations derived
   640→   640→/// from the raw profile samples and SCG structure.
   641→   641→#[derive(Debug, Clone, Serialize, Deserialize)]
   642→   642→pub struct ProfileReport {
   643→   643→    /// Total number of samples analysed.
   644→   644→    pub total_samples: usize,
   645→   645→    /// Total execution time across all sampled nodes (ns).
   646→   646→    pub total_execution_time_ns: u64,
   647→   647→    /// Nodes ranked by execution time (hottest first), with their cumulative
   648→   648→    /// time and call count.
   649→   649→    pub hot_spots: Vec<NodeHotSpot>,
   650→   650→    /// Nodes with zero or negligible execution.
   651→   651→    pub cold_spots: Vec<NodeId>,
   652→   652→    /// Hot paths (node sequences accounting for >80 % of execution time).
   653→   653→    pub hot_paths: Vec<HotPath>,
   654→   654→    /// Aggregate Pi 5 PMU counters across all samples.
   655→   655→    pub aggregate_pmu: Pi5PmuCounters,
   656→   656→    /// Per-node PMU counter summaries.
   657→   657→    pub node_pmu: HashMap<NodeId, Pi5PmuCounters>,
   658→   658→    /// Optimisation suggestions derived from the analysis.
   659→   659→    pub recommendations: Vec<OptimizationSuggestion>,
   660→   660→}
   661→   661→
   662→   662→/// A hot spot: a single node with its execution statistics.
   663→   663→#[derive(Debug, Clone, Serialize, Deserialize)]
   664→   664→pub struct NodeHotSpot {
   665→   665→    /// The SCG node identifier.
   666→   666→    pub node_id: NodeId,
   667→   667→    /// Number of times this node was executed.
   668→   668→    pub call_count: u64,
   669→   669→    /// Cumulative execution time (ns) for this node.
   670→   670→    pub total_time_ns: u64,
   671→   671→    /// Fraction of global execution time (0.0 – 1.0).
   672→   672→    pub time_fraction: f64,
   673→   673→}

   675→   675→// ---------------------------------------------------------------------------
   676→   676→// collect_profile — main analysis entry point
   677→   677→// ---------------------------------------------------------------------------
   678→   678→
   679→   679→/// Analyses a batch of profile samples against the SCG and produces a
   680→   680→/// [`ProfileReport`].
   681→   681→///
   682→   682→/// This is the primary entry point for profile-guided analysis. It:
   683→   683→///
   684→   684→/// 1. Ingests all samples into a [`ProfileData`] instance.
   685→   685→/// 2. Computes hot spots (nodes ranked by execution time).
   686→   686→/// 3. Identifies cold spots (nodes with zero or negligible execution).
   687→   687→/// 4. Computes hot paths (node sequences accounting for >80 % of time).
   688→   688→/// 5. Aggregates PMU counter data.
   689→   689→/// 6. Generates optimisation recommendations.
   690→   690→pub fn collect_profile(scg: &SCG, samples: &[ProfileSample]) -> ProfileReport {
   691→   691→    let mut profile_data = ProfileData::new();
   692→   692→    profile_data.ingest_samples(samples);
   693→   693→
   694→   694→    let total_time = profile_data.total_execution_time_ns();
   695→   695→
   696→   696→    // --- Hot spots ---
   697→   697→    let mut hot_spots: Vec<NodeHotSpot> = profile_data
   698→   698→        .node_time_ns
   699→   699→        .iter()
   700→   700→        .map(|(&node_id, &time_ns)| {
   701→   701→            let call_count = profile_data.call_counts.get(&node_id).copied().unwrap_or(0);
   702→   702→            let time_fraction = if total_time > 0 {
   703→   703→                time_ns as f64 / total_time as f64
   704→   704→            } else {
   705→   705→                0.0
   706→   706→            };
   707→   707→            NodeHotSpot {
   708→   708→                node_id,
   709→   709→                call_count,
   710→   710→                total_time_ns: time_ns,
   711→   711→                time_fraction,
   712→   712→            }
   713→   713→        })
   714→   714→        .collect();
   715→   715→    hot_spots.sort_by_key(|b| std::cmp::Reverse(b.total_time_ns));
   716→   716→
   717→   717→    // --- Cold spots ---
   718→   718→    // Build a list of all node IDs from 0..scg.node_count (in the full
   719→   719→    // implementation the SCG would provide an iterator).
   720→   720→    let all_nodes: Vec<NodeId> = (0..scg.node_count as u64).collect();
   721→   721→    let cold_threshold_ns = if total_time > 0 { total_time / 1000 } else { 1 };
   722→   722→    let cold_spots = profile_data.cold_spots(&all_nodes, cold_threshold_ns);
   723→   723→
   724→   724→    // --- Hot paths ---
   725→   725→    let hot_paths = profile_data.compute_hot_paths();
   726→   726→
   727→   727→    // --- Aggregate PMU ---
   728→   728→    let mut aggregate_pmu = Pi5PmuCounters::new();
   729→   729→    for pmu in profile_data.node_pmu.values() {
   730→   730→        aggregate_pmu.cycle_count += pmu.cycle_count;
   731→   731→        aggregate_pmu.instruction_count += pmu.instruction_count;
   732→   732→        aggregate_pmu.cache_misses += pmu.cache_misses;
   733→   733→        aggregate_pmu.branch_misses += pmu.branch_misses;
   734→   734→    }
   735→   735→
   736→   736→    // --- Recommendations ---
   737→   737→    let recommendations = profile_data.suggest_optimizations();
   738→   738→
   739→   739→    ProfileReport {
   740→   740→        total_samples: samples.len(),
   741→   741→        total_execution_time_ns: total_time,
   742→   742→        hot_spots,
   743→   743→        cold_spots,
   744→   744→        hot_paths,
   745→   745→        aggregate_pmu,
   746→   746→        node_pmu: profile_data.node_pmu.clone(),
   747→   747→        recommendations,
   748→   748→    }
   749→   749→}

   751→   751→// ---------------------------------------------------------------------------
   752→   752→// Optimization suggestion types
   753→   753→// ---------------------------------------------------------------------------
   754→   754→
   755→   755→/// The kind of optimization being suggested.
   756→   756→#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
   757→   757→pub enum SuggestionKind {
   758→   758→    /// Inline the node's body at its call sites.
   759→   759→    Inline,
   760→   760→    /// Replace per-call heap allocations with an arena.
   761→   761→    ArenaAlloc,
   762→   762→    /// Speculatively specialize generic code.
   763→   763→    Specialize,
   764→   764→    /// Optimise data layout for better cache utilisation.
   765→   765→    CacheOptimize,
   766→   766→    /// Reorder branches to favour the common path.
   767→   767→    BranchLayout,
   768→   768→}
   769→   769→
   770→   770→/// A single optimization suggestion produced by profile analysis.
   771→   771→#[derive(Debug, Clone, Serialize, Deserialize)]
   772→   772→pub struct OptimizationSuggestion {
   773→   773→    /// The kind of optimization.
   774→   774→    pub kind: SuggestionKind,
   775→   775→    /// The node that would be affected (0 if the suggestion is region-wide).
   776→   776→    pub target_node: NodeId,
   777→   777→    /// Human-readable explanation.
   778→   778→    pub reason: String,
   779→   779→}

   996→   996→    #[test]
   997→   997→    fn test_profile_mutex_recovery() {
   998→   998→        // Simulate Mutex poisoning and verify that lock_profile recovers.
   999→   999→        //
  1000→  1000→        // When a thread panics while holding a Mutex, the Mutex becomes
  1001→  1001→        // "poisoned". The lock_profile helper recovers from this by
  1002→  1002→        // calling into_inner() on the PoisonError, which gives access
  1003→  1003→        // to the data even after a panic.
  1004→  1004→
  1005→  1005→        use std::sync::Arc;
  1006→  1006→        use std::thread;
  1007→  1007→
  1008→  1008→        let collector = Arc::new(ProfileCollector::new());
  1009→  1009→
  1010→  1010→        // Record some data before the panic.
  1011→  1011→        collector.record_access(1);
  1012→  1012→        collector.record_access(2);
  1013→  1013→
  1014→  1014→        // Spawn a thread that panics while holding the lock.
  1015→  1015→        // We do this by manually poisoning a separate Mutex that shares
  1016→  1016→        // the same pattern, since we can't directly poison the
  1017→  1017→        // ProfileCollector's internal Mutex from outside.
  1018→  1018→        //
  1019→  1019→        // Instead, we test the lock_profile helper directly with a
  1020→  1020→        // controlled Mutex.
  1021→  1021→        let test_mutex: Arc<Mutex<ProfileData>> = Arc::new(Mutex::new(ProfileData::new()));
  1022→  1022→        {
  1023→  1023→            let m = Arc::clone(&test_mutex);
  1024→  1024→            let handle = thread::spawn(move || {
  1025→  1025→                let _guard = m.lock().unwrap();
  1026→  1026→                // Panic while holding the lock to poison the Mutex.
  1027→  1027→                panic!("intentional panic to poison mutex");
  1028→  1028→            });
  1029→  1029→            // The thread panicked, so the Mutex should be poisoned.
  1030→  1030→            let _ = handle.join();
  1031→  1031→        }
  1032→  1032→
  1033→  1033→        // Verify the Mutex is poisoned — normal lock().unwrap() would panic.
  1034→  1034→        assert!(
  1035→  1035→            test_mutex.is_poisoned(),
  1036→  1036→            "Mutex should be poisoned after thread panic"
  1037→  1037→        );
  1038→  1038→
  1039→  1039→        // Now verify that lock_profile recovers from the poisoning.
  1040→  1040→        let mut guard = lock_profile(&test_mutex);
  1041→  1041→        // We should be able to read the data.
  1042→  1042→        assert!(guard.call_counts.is_empty());
  1043→  1043→
  1044→  1044→        // We should also be able to mutate the data through the recovered guard.
  1045→  1045→        guard.record_access(42);
  1046→  1046→        drop(guard);
  1047→  1047→
  1048→  1048→        // Verify the mutation was applied.
  1049→  1049→        let guard = lock_profile(&test_mutex);
  1050→  1050→        assert_eq!(guard.call_counts.get(&42).copied().unwrap_or(0), 1);
  1051→  1051→
  1052→  1052→        // Also verify that ProfileCollector itself would recover:
  1053→  1053→        // The collector's internal lock_profile calls will recover
  1054→  1054→        // from poisoning if it ever occurs.
  1055→  1055→        // We can't easily poison the collector's internal Mutex from
  1056→  1056→        // outside, but we verify the helper function works correctly
  1057→  1057→        // above.
  1058→  1058→    }
