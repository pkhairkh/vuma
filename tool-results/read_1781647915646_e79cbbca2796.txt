     1→//! Recursive-descent parser for the VUMA language frontend.
     2→//!
     3→//! Consumes the token stream produced by [`crate::lexer::Lexer`] and builds
     4→//! an [`crate::ast::Program`]. The parser uses precedence climbing for
     5→//! expression sub-parsing and implements error recovery by skipping
     6→//! to the next statement boundary (`;` or `}`) or item boundary on failure.
     7→//!
     8→//! # Supported constructs
     9→//!
    10→//! **Items** (top-level declarations):
    11→//! - `fn`, `struct`, `enum`, `region`, `import`, `export`, `const`, `static`, `mod`
    12→//!
    13→//! **Statements**:
    14→//! - `let`, assignment, compound assignment (`+=`, `-=`, etc.), `if`, `while`, `for`,
    15→//!   `loop`, `match`, `return`, `break`, `continue`, `sync`, `free`, `allocate`,
    16→//!   `bd`/`repd`/`capd`/`reld` directives, expression statements
    17→//!
    18→//! **Expressions**:
    19→//! - Binary/unary ops, calls, field/index access, deref, addr-of, cast,
    20→//!   `sizeof`, `alignof`, `derive`, `async`, `spawn`, `allocate` (as expr),
    21→//!   literals (int, float, bool, string, address, null), struct init, namespace access
    22→
    23→use crate::ast::*;
    24→use crate::error::{
    25→    check_llm_construct, suggest_keyword, suggest_vuma_type, ErrorRecovery, ParseError,
    26→    ParseErrorKind, ParseResult, Span,
    27→};
    28→use crate::lexer::{Lexer, Token, TokenKind};
    29→

    31→// Parser

    33→
    34→/// Default maximum recursion depth for expression parsing.
    35→const MAX_EXPR_DEPTH: u32 = 256;
    36→
    37→/// A recursive-descent parser that turns a token stream into an AST.
    38→pub struct Parser<'src> {
    39→    /// Underlying lexer.
    40→    lexer: Lexer<'src>,
    41→    /// The most recently consumed token.
    42→    current: Token,
    43→    /// Accumulated parse errors (for error recovery).
    44→    errors: Vec<ParseError>,
    45→    /// Pushback buffer for token rewinding (used in struct literal disambiguation).
    46→    pushback: std::collections::VecDeque<Token>,
    47→    /// Current recursion depth in expression parsing.
    48→    expr_depth: u32,
    49→    /// Maximum allowed recursion depth for expression parsing.
    50→    max_depth: u32,
    51→    /// When true, struct literal parsing is suppressed in `parse_postfix()`.
    52→    /// Used when parsing the end expression of a range (`..`) so that `{`
    53→    /// is not consumed as part of a struct literal and remains available
    54→    /// for the enclosing construct (e.g. a for-loop body).
    55→    no_struct_literal: bool,
    56→    /// Name of the function currently being parsed (for context-aware errors).
    57→    current_fn_name: Option<String>,
    58→}
    59→
    60→/// Token kinds that begin a top-level item declaration.
    61→const ITEM_STARTERS: &[TokenKind] = &[
    62→    TokenKind::Fn,
    63→    TokenKind::Struct,
    64→    TokenKind::Enum,
    65→    TokenKind::Region,
    66→    TokenKind::Import,
    67→    TokenKind::Export,
    68→    TokenKind::Const,
    69→    TokenKind::Static,
    70→    TokenKind::Pub,
    71→    TokenKind::Hash,
    72→    // `mod` is TokenKind::Mod — handled directly
    73→];
    74→
    75→/// Compound-assignment token kinds mapped to their operators.
    76→///
    77→/// Returns `Err(ParseError)` for invalid tokens instead of panicking,
    78→/// so that callers can propagate the error cleanly with `?`.
    79→fn compound_op_from_token(kind: TokenKind, span: Span) -> Result<CompoundOp, ParseError> {
    80→    match kind {
    81→        TokenKind::PlusEq => Ok(CompoundOp::Add),
    82→        TokenKind::MinusEq => Ok(CompoundOp::Sub),
    83→        TokenKind::StarEq => Ok(CompoundOp::Mul),
    84→        TokenKind::SlashEq => Ok(CompoundOp::Div),
    85→        TokenKind::PercentEq => Ok(CompoundOp::Mod),
    86→        TokenKind::AmpEq => Ok(CompoundOp::BitAnd),
    87→        TokenKind::PipeEq => Ok(CompoundOp::BitOr),
    88→        TokenKind::CaretEq => Ok(CompoundOp::BitXor),
    89→        TokenKind::ShlEq => Ok(CompoundOp::Shl),
    90→        TokenKind::ShrEq => Ok(CompoundOp::Shr),
    91→        _ => Err(ParseError::invalid_compound_op(
    92→            format!("expected compound assignment operator, found {:?}", kind),
    93→            span,
    94→        )),
    95→    }
    96→}
    97→
    98→/// Check if a token kind is a compound assignment operator.
    99→fn is_compound_assign(kind: TokenKind) -> bool {
   100→    matches!(
   101→        kind,
   102→        TokenKind::PlusEq
   103→            | TokenKind::MinusEq
   104→            | TokenKind::StarEq
   105→            | TokenKind::SlashEq
   106→            | TokenKind::PercentEq
   107→            | TokenKind::AmpEq
   108→            | TokenKind::PipeEq
   109→            | TokenKind::CaretEq
   110→            | TokenKind::ShlEq
   111→            | TokenKind::ShrEq
   112→    )
   113→}
   114→
   115→impl<'src> Parser<'src> {
   116→    /// Create a new parser over the given source text.
   117→    pub fn new(source: &'src str) -> Self {
   118→        let mut lexer = Lexer::new(source);
   119→        let current = lexer.next_token();
   120→        Self {
   121→            lexer,
   122→            current,
   123→            errors: Vec::new(),
   124→            pushback: std::collections::VecDeque::new(),
   125→            expr_depth: 0,
   126→            max_depth: MAX_EXPR_DEPTH,
   127→            no_struct_literal: false,
   128→            current_fn_name: None,
   129→        }
   130→    }
   131→
   132→    /// Create a new parser with a custom maximum expression recursion depth.
   133→    pub fn with_max_depth(source: &'src str, max_depth: u32) -> Self {
   134→        let mut parser = Self::new(source);
   135→        parser.max_depth = max_depth;
   136→        parser
   137→    }
   138→
   139→    // -- public entry point --------------------------------------------------
   140→
   141→    /// Parse the full source into a [`Program`].
   142→    ///
   143→    /// If errors are encountered the parser attempts recovery and continues;
   144→    /// the returned [`ParseResult`] carries both the partial AST and any
   145→    /// accumulated non-fatal errors, enabling multi-error reporting in a
   146→    /// single pass. All errors are collected — the parser never stops at
   147→    /// the first error.
   148→    pub fn parse_program(&mut self) -> ParseResult<Program> {
   149→        let start = self.current.span.start;
   150→        let mut items = Vec::new();
   151→
   152→        // Also collect any lexer-level errors
   153→        let lexer_errors = self.lexer.take_errors();
   154→        self.errors.extend(lexer_errors);
   155→
   156→        while !self.at(TokenKind::Eof) {
   157→            match self.parse_item() {
   158→                Ok(item) => items.push(item),
   159→                Err(err) => {
   160→                    self.errors.push(err);
   161→                    self.recover_to_item_boundary();
   162→                }
   163→            }
   164→        }
   165→
   166→        // Collect any remaining lexer errors that appeared during parsing
   167→        let lexer_errors = self.lexer.take_errors();
   168→        self.errors.extend(lexer_errors);
   169→
   170→        let end = self.current.span.end;
   171→        let program = Program {
   172→            items,
   173→            span: Span::new(start, end),
   174→        };
   175→
   176→        // Resolve line/column for all accumulated errors
   177→        let source = self.lexer.source();
   178→        for err in &mut self.errors {
   179→            err.resolve_location(source, None);
   180→        }
   181→
   182→        if self.errors.is_empty() {
   183→            ParseResult::ok(program)
   184→        } else {
   185→            ParseResult::ok_with_errors(program, std::mem::take(&mut self.errors))
   186→        }
   187→    }
   188→
   189→    /// Return all accumulated parse errors.
   190→    pub fn errors(&self) -> &[ParseError] {
   191→        &self.errors
   192→    }

   194→    // -- lookahead -----------------------------------------------------------
   195→
   196→    /// Peek at the token AFTER `self.current` without consuming anything.
   197→    /// Returns a clone of the peeked token.
   198→    fn peek_next(&mut self) -> Token {
   199→        self.lexer.peek().clone()
   200→    }
   201→
   202→    // -- item parsing --------------------------------------------------------
   203→
   204→    /// Parse a single top-level item or statement.
   205→    fn parse_item(&mut self) -> Result<Item, ParseError> {
   206→        // Parse outer attributes (#[...]) that precede the item.
   207→        let attrs = self.parse_outer_attributes()?;
   208→
   209→        // Parse visibility modifier.
   210→        let visibility = self.parse_visibility()?;
   211→
   212→        match self.current.kind {
   213→            TokenKind::Fn => self.parse_fn_def(false, visibility, attrs).map(Item::FnDef),
   214→            TokenKind::Async => {
   215→                // Could be `async fn` or `async { block }` as expression
   216→                let next = self.peek_next();
   217→                if next.kind == TokenKind::Fn {
   218→                    self.parse_fn_def(true, visibility, attrs).map(Item::FnDef)
   219→                } else {
   220→                    self.parse_stmt().map(Item::Stmt)
   221→                }
   222→            }
   223→            TokenKind::Struct => self
   224→                .parse_struct_def(visibility, attrs)
   225→                .map(Item::StructDef),
   226→            TokenKind::Enum => self.parse_enum_def(visibility, attrs).map(Item::EnumDef),
   227→            TokenKind::Region => {
   228→                // Distinguish: `region name = allocate(...)` vs `region` used as
   229→                // a variable name in an expression/assignment (e.g. `region = allocate(8);`)
   230→                let next = self.peek_next();
   231→                if next.kind == TokenKind::Ident {
   232→                    self.parse_region_def().map(Item::RegionDef)
   233→                } else {
   234→                    self.parse_stmt().map(Item::Stmt)
   235→                }
   236→            }
   237→            TokenKind::Import => self.parse_import().map(Item::Import),
   238→            TokenKind::Export => self.parse_export().map(Item::Export),
   239→            TokenKind::Const => self.parse_const_item(visibility, attrs).map(Item::Const),
   240→            TokenKind::Static => self.parse_static_item(visibility, attrs).map(Item::Static),
   241→            TokenKind::Mod => self.parse_module_def().map(Item::ModuleDef),
   242→            TokenKind::Trait => self.parse_trait_def(visibility, attrs).map(Item::TraitDef),
   243→            TokenKind::Impl => self.parse_impl_block(attrs).map(Item::ImplBlock),
   244→            TokenKind::Extern => self.parse_extern_block().map(Item::ExternBlock),
   245→            TokenKind::Ident => {
   246→                let lexeme = self.current.lexeme.as_str();
   247→                match lexeme {
   248→                    "static" => self.parse_static_item(visibility, attrs).map(Item::Static),
   249→                    _ => self.parse_stmt().map(Item::Stmt),
   250→                }
   251→            }
   252→            _ => self.parse_stmt().map(Item::Stmt),
   253→        }
   254→    }
   255→
   256→    /// [`async`] `fn` <ident> [`<` type_params `>`] `(` <params>? `)` [`->` <type>] [`;` | `{` <block> `}`]
   257→    fn parse_fn_def(
   258→        &mut self,
   259→        is_async: bool,
   260→        _visibility: Visibility,
   261→        _attrs: Vec<Attribute>,
   262→    ) -> Result<FnDef, ParseError> {
   263→        let start = self.current.span.start;
   264→
   265→        // Consume 'async' if present
   266→        if self.at(TokenKind::Async) {
   267→            self.advance();
   268→        }
   269→
   270→        self.expect(TokenKind::Fn)?;
   271→
   272→        let name = self.expect_name()?;
   273→
   274→        // Track current function name for context-aware error messages
   275→        self.current_fn_name = Some(name.clone());
   276→
   277→        // Optional generic type parameters: `<T, U>`
   278→        let type_params = if self.at(TokenKind::Lt) {
   279→            self.parse_generic_params()?
   280→        } else {
   281→            Vec::new()
   282→        };
   283→
   284→        self.expect(TokenKind::LParen)?;
   285→
   286→        let params = self.parse_params()?;
   287→
   288→        self.expect(TokenKind::RParen)?;
   289→
   290→        let return_type = if self.at(TokenKind::Arrow) {
   291→            self.advance();
   292→            Some(self.parse_type()?)
   293→        } else {
   294→            None
   295→        };
   296→
   297→        // Optional where clause
   298→        let where_clause = if self.at(TokenKind::Where) {
   299→            Some(self.parse_where_clause()?)
   300→        } else {
   301→            None
   302→        };
   303→
   304→        // For trait methods, we may have `;` instead of a body.
   305→        // We create a synthetic empty block for required method signatures.
   306→        let body = if self.at(TokenKind::LBrace) {
   307→            self.parse_block()?
   308→        } else if self.at(TokenKind::Semicolon) {
   309→            self.advance(); // consume ';'
   310→            Block {
   311→                statements: Vec::new(),
   312→                span: Span::new(self.current.span.start, self.current.span.end),
   313→            }
   314→        } else {
   315→            Block {
   316→                statements: Vec::new(),
   317→                span: Span::synthetic(),
   318→            }
   319→        };
   320→        let end = body.span.end;
   321→
   322→        // Clear the current function name after we're done parsing it
   323→        self.current_fn_name = None;
   324→
   325→        Ok(FnDef {
   326→            visibility: _visibility,
   327→            attrs: _attrs,
   328→            name,
   329→            type_params,
   330→            params,
   331→            return_type,
   332→            body,
   333→            is_async,
   334→            where_clause,
   335→            span: Span::new(start, end),
   336→        })
   337→    }

   339→    /// `struct` <ident> [`<` type_params `>`] `{` <fields> `}`
   340→    fn parse_struct_def(
   341→        &mut self,
   342→        _visibility: Visibility,
   343→        _attrs: Vec<Attribute>,
   344→    ) -> Result<StructDef, ParseError> {
   345→        let start = self.current.span.start;
   346→        self.expect(TokenKind::Struct)?;
   347→
   348→        let name = self.expect_name()?;
   349→
   350→        // Optional generic type parameters with bounds
   351→        let type_params = if self.at(TokenKind::Lt) {
   352→            self.parse_type_params_with_bounds()?
   353→        } else {
   354→            Vec::new()
   355→        };
   356→
   357→        // Where clause comes before '{'
   358→        let where_clause = if self.at(TokenKind::Where) {
   359→            Some(self.parse_where_clause()?)
   360→        } else {
   361→            None
   362→        };
   363→
   364→        self.expect(TokenKind::LBrace)?;
   365→
   366→        let mut fields = Vec::new();
   367→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   368→            let f_start = self.current.span.start;
   369→            let fname = self.expect_name()?;
   370→            self.expect(TokenKind::Colon)?;
   371→            let ftype = self.parse_type()?;
   372→            let f_end = self.current.span.end;
   373→            fields.push(StructField {
   374→                name: fname,
   375→                ty: ftype,
   376→                span: Span::new(f_start, f_end),
   377→            });
   378→            if self.at(TokenKind::Comma) {
   379→                self.advance();
   380→            } else {
   381→                break;
   382→            }
   383→        }
   384→
   385→        self.expect(TokenKind::RBrace)?;
   386→
   387→        Ok(StructDef {
   388→            visibility: _visibility,
   389→            attrs: _attrs,
   390→            name,
   391→            type_params,
   392→            fields,
   393→            where_clause,
   394→            span: Span::new(start, self.current.span.end),
   395→        })
   396→    }

   398→    /// `enum` <ident> [`<` type_params `>`] `{` <variants> `}`
   399→    fn parse_enum_def(
   400→        &mut self,
   401→        _visibility: Visibility,
   402→        _attrs: Vec<Attribute>,
   403→    ) -> Result<EnumDef, ParseError> {
   404→        let start = self.current.span.start;
   405→        self.expect(TokenKind::Enum)?;
   406→
   407→        let name = self.expect_name()?;
   408→
   409→        let type_params = if self.at(TokenKind::Lt) {
   410→            self.parse_type_params_with_bounds()?
   411→        } else {
   412→            Vec::new()
   413→        };
   414→
   415→        // Where clause comes before '{'
   416→        let where_clause = if self.at(TokenKind::Where) {
   417→            Some(self.parse_where_clause()?)
   418→        } else {
   419→            None
   420→        };
   421→
   422→        self.expect(TokenKind::LBrace)?;
   423→
   424→        let mut variants = Vec::new();
   425→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   426→            let v_start = self.current.span.start;
   427→            let vname = self.expect_name()?;
   428→
   429→            let payload = if self.at(TokenKind::LParen) {
   430→                self.advance();
   431→                let ty = self.parse_type()?;
   432→                self.expect(TokenKind::RParen)?;
   433→                Some(ty)
   434→            } else {
   435→                None
   436→            };
   437→
   438→            let v_end = self.current.span.end;
   439→            variants.push(EnumVariant {
   440→                name: vname,
   441→                payload,
   442→                span: Span::new(v_start, v_end),
   443→            });
   444→            if self.at(TokenKind::Comma) {
   445→                self.advance();
   446→            } else {
   447→                break;
   448→            }
   449→        }
   450→
   451→        self.expect(TokenKind::RBrace)?;
   452→
   453→        Ok(EnumDef {
   454→            visibility: _visibility,
   455→            attrs: _attrs,
   456→            name,
   457→            type_params,
   458→            variants,
   459→            where_clause,
   460→            span: Span::new(start, self.current.span.end),
   461→        })
   462→    }

   464→    /// Parse comma-separated type parameter names inside `< … >`.
   465→    #[allow(dead_code)]
   466→    fn parse_type_params(&mut self) -> Result<Vec<String>, ParseError> {
   467→        self.expect(TokenKind::Lt)?;
   468→        let mut params = Vec::new();
   469→        while !self.at_closing_gt() && !self.at(TokenKind::Eof) {
   470→            params.push(self.expect_name()?);
   471→            if self.at(TokenKind::Comma) {
   472→                self.advance();
   473→            } else {
   474→                break;
   475→            }
   476→        }
   477→        self.expect_gt_closing_generic()?;
   478→        Ok(params)
   479→    }
   480→
   481→    /// Parse type parameters with optional bounds: `<T: Display + Clone, U>`
   482→    fn parse_type_params_with_bounds(&mut self) -> Result<Vec<TypeParam>, ParseError> {
   483→        self.expect(TokenKind::Lt)?;
   484→        let mut params = Vec::new();
   485→        while !self.at_closing_gt() && !self.at(TokenKind::Eof) {
   486→            let name = self.expect_name()?;
   487→            let bounds = if self.at(TokenKind::Colon) {
   488→                self.advance();
   489→                self.parse_trait_bounds()?
   490→            } else {
   491→                Vec::new()
   492→            };
   493→            params.push(TypeParam { name, bounds });
   494→            if self.at(TokenKind::Comma) {
   495→                self.advance();
   496→            } else {
   497→                break;
   498→            }
   499→        }
   500→        self.expect_gt_closing_generic()?;
   501→        Ok(params)
   502→    }
   503→
   504→    /// Parse trait bounds: `Trait + AnotherTrait + ...`
   505→    fn parse_trait_bounds(&mut self) -> Result<Vec<Type>, ParseError> {
   506→        let mut bounds = Vec::new();
   507→        bounds.push(self.parse_type()?);
   508→        while self.at(TokenKind::Plus) {
   509→            self.advance();
   510→            bounds.push(self.parse_type()?);
   511→        }
   512→        Ok(bounds)
   513→    }
   514→
   515→    /// Parse a where clause: `where T: Trait + AnotherTrait, U: Trait`
   516→    fn parse_where_clause(&mut self) -> Result<WhereClause, ParseError> {
   517→        self.expect(TokenKind::Where)?;
   518→        let mut predicates = Vec::new();
   519→        loop {
   520→            let type_name = self.expect_name()?;
   521→            self.expect(TokenKind::Colon)?;
   522→            let bounds = self.parse_trait_bounds()?;
   523→            predicates.push(WherePredicate { type_name, bounds });
   524→            if self.at(TokenKind::Comma) {
   525→                self.advance();
   526→            } else {
   527→                break;
   528→            }
   529→        }
   530→        Ok(WhereClause { predicates })
   531→    }
   532→
   533→    /// Parse generic parameter list `<T, U, ...>` with optional bounds.
   534→    /// Returns parsed type parameters for recording in the AST.
   535→    fn parse_generic_params(&mut self) -> Result<Vec<TypeParam>, ParseError> {
   536→        self.parse_type_params_with_bounds()
   537→    }
   538→
   539→    /// Parse a comma-separated parameter list (may be empty).
   540→    fn parse_params(&mut self) -> Result<Vec<Param>, ParseError> {
   541→        let mut params = Vec::new();
   542→        if self.at(TokenKind::RParen) {
   543→            return Ok(params);
   544→        }
   545→        loop {
   546→            let span = self.current.span;
   547→            let name = self.expect_name()?;
   548→            let ty = if self.at(TokenKind::Colon) {
   549→                self.advance();
   550→                Some(self.parse_type()?)
   551→            } else {
   552→                None
   553→            };
   554→            params.push(Param { name, ty, span });
   555→            if self.at(TokenKind::Comma) {
   556→                self.advance();
   557→            } else {
   558→                break;
   559→            }
   560→        }
   561→        Ok(params)
   562→    }

   564→    /// `region` <ident> `=` `allocate` `(` <expr> `)` `;`
   565→    fn parse_region_def(&mut self) -> Result<RegionDef, ParseError> {
   566→        let start = self.current.span.start;
   567→        self.expect(TokenKind::Region)?;
   568→
   569→        let name = self.expect_name()?;
   570→
   571→        self.expect(TokenKind::Assign)?;
   572→        self.expect(TokenKind::Allocate)?;
   573→        self.expect(TokenKind::LParen)?;
   574→
   575→        let size_expr = self.parse_expr()?;
   576→
   577→        self.expect(TokenKind::RParen)?;
   578→        self.expect(TokenKind::Semicolon)?;
   579→
   580→        Ok(RegionDef {
   581→            name,
   582→            size_expr,
   583→            span: Span::new(start, self.current.span.end),
   584→        })
   585→    }

   587→    /// `import` <string> [`::` `{` <names> `}`] `;`
   588→    ///
   589→    /// Supported forms:
   590→    /// - `import "crypto.vuma";` — import all functions from file
   591→    /// - `import "crypto.vuma"::{sha256, sha256d};` — import specific functions
   592→    /// - `import "crypto.vuma" {sha256, sha256d};` — legacy form (also accepted)
   593→    fn parse_import(&mut self) -> Result<Import, ParseError> {
   594→        let start = self.current.span.start;
   595→        self.expect(TokenKind::Import)?;
   596→
   597→        let path = self.expect_string()?;
   598→        let mut symbols = Vec::new();
   599→
   600→        // Accept optional `::` before `{` for the new syntax
   601→        if self.at(TokenKind::PathSep) {
   602→            self.advance();
   603→        }
   604→
   605→        if self.at(TokenKind::LBrace) {
   606→            self.advance();
   607→            while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   608→                symbols.push(self.expect_name()?);
   609→                if self.at(TokenKind::Comma) {
   610→                    self.advance();
   611→                } else {
   612→                    break;
   613→                }
   614→            }
   615→            self.expect(TokenKind::RBrace)?;
   616→        }
   617→
   618→        self.expect(TokenKind::Semicolon)?;
   619→
   620→        Ok(Import {
   621→            path,
   622→            symbols,
   623→            span: Span::new(start, self.current.span.end),
   624→        })
   625→    }

   737→    /// `trait` <name> [`<` type_params `>`] `{` <members> `}`
   738→    fn parse_trait_def(
   739→        &mut self,
   740→        _visibility: Visibility,
   741→        _attrs: Vec<Attribute>,
   742→    ) -> Result<TraitDef, ParseError> {
   743→        let start = self.current.span.start;
   744→        self.expect(TokenKind::Trait)?;
   745→
   746→        let name = self.expect_name()?;
   747→
   748→        let type_params = if self.at(TokenKind::Lt) {
   749→            self.parse_type_params_with_bounds()?
   750→        } else {
   751→            Vec::new()
   752→        };
   753→
   754→        self.expect(TokenKind::LBrace)?;
   755→
   756→        let mut associated_types = Vec::new();
   757→        let mut associated_consts = Vec::new();
   758→        let mut required_methods = Vec::new();
   759→        let mut provided_methods = Vec::new();
   760→
   761→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   762→            // Associated type: `type Name;`
   763→            if self.at(TokenKind::Type) {
   764→                let next = self.peek_next();
   765→                if next.kind == TokenKind::Ident {
   766→                    self.advance(); // consume 'type'
   767→                    let ty_name = self.expect_name()?;
   768→                    self.expect(TokenKind::Semicolon)?;
   769→                    associated_types.push(ty_name);
   770→                    continue;
   771→                }
   772→            }
   773→            // Associated const: `const NAME: Type [= expr];`
   774→            if self.at(TokenKind::Const) {
   775→                let ac_start = self.current.span.start;
   776→                let next = self.peek_next();
   777→                if next.kind == TokenKind::Ident {
   778→                    self.advance(); // consume 'const'
   779→                    let ac_name = self.expect_name()?;
   780→                    self.expect(TokenKind::Colon)?;
   781→                    let ac_ty = self.parse_type()?;
   782→                    let ac_value = if self.at(TokenKind::Assign) {
   783→                        self.advance();
   784→                        Some(self.parse_expr()?)
   785→                    } else {
   786→                        None
   787→                    };
   788→                    self.expect(TokenKind::Semicolon)?;
   789→                    associated_consts.push(AssocConst {
   790→                        name: ac_name,
   791→                        ty: ac_ty,
   792→                        value: ac_value,
   793→                        span: Span::new(ac_start, self.current.span.end),
   794→                    });
   795→                    continue;
   796→                }
   797→            }
   798→            // Method: `fn name(...) -> T;` or `fn name(...) -> T { body }`
   799→            if self.at(TokenKind::Fn) {
   800→                let method = self.parse_fn_def(false, Visibility::default(), Vec::new())?;
   801→                if method.body.statements.is_empty()
   802→                    && !method.is_async
   803→                    && method.where_clause.is_none()
   804→                {
   805→                    // Empty body suggests a required method signature
   806→                    // But we can't easily tell — if the body is just {} it's provided
   807→                    // We'll check: if body span is just `{}` with nothing inside, treat as required
   808→                    // Actually, let's use a simpler heuristic: if the method has a non-empty
   809→                    // body (more than 0 statements), it's provided.
   810→                    required_methods.push(method);
   811→                } else {
   812→                    provided_methods.push(method);
   813→                }
   814→                continue;
   815→            }
   816→            // Skip unexpected tokens in trait body
   817→            self.advance();
   818→        }
   819→
   820→        self.expect(TokenKind::RBrace)?;
   821→
   822→        let where_clause = if self.at(TokenKind::Where) {
   823→            Some(self.parse_where_clause()?)
   824→        } else {
   825→            None
   826→        };
   827→
   828→        Ok(TraitDef {
   829→            visibility: _visibility,
   830→            attrs: _attrs,
   831→            name,
   832→            type_params,
   833→            associated_types,
   834→            associated_consts,
   835→            required_methods,
   836→            provided_methods,
   837→            where_clause,
   838→            span: Span::new(start, self.current.span.end),
   839→        })
   840→    }

   842→    /// `impl` [<trait_name> `for`] <type> `{` <methods> `}`
   843→    fn parse_impl_block(&mut self, _attrs: Vec<Attribute>) -> Result<ImplBlock, ParseError> {
   844→        let start = self.current.span.start;
   845→        self.expect(TokenKind::Impl)?;
   846→
   847→        // Parse optional generic params on the impl itself: `impl<T> ...`
   848→        // This must come before the trait/type name.
   849→        let type_params = if self.at(TokenKind::Lt) {
   850→            self.parse_generic_params()?
   851→        } else {
   852→            Vec::new()
   853→        };
   854→
   855→        // Parse the first name — could be a trait name or the target type
   856→        let first_name = self.expect_name()?;
   857→
   858→        // Check if `for` follows — if so, first_name is the trait name
   859→        let (trait_name, target_type) = if self.at(TokenKind::For) {
   860→            self.advance(); // consume 'for'
   861→            let target = self.parse_type()?;
   862→            (Some(first_name), target)
   863→        } else {
   864→            // No 'for' — first_name is the target type
   865→            // Check for generic args on the target type
   866→            let target = if self.at(TokenKind::Lt) {
   867→                // Generic type: `Name<Args>`
   868→                self.advance(); // consume '<'
   869→                let mut args = Vec::new();
   870→                if !self.at_closing_gt() {
   871→                    args.push(self.parse_type()?);
   872→                    while self.at(TokenKind::Comma) {
   873→                        self.advance();
   874→                        args.push(self.parse_type()?);
   875→                    }
   876→                }
   877→                self.expect_gt_closing_generic()?;
   878→                Type::Generic {
   879→                    name: first_name,
   880→                    args,
   881→                }
   882→            } else {
   883→                Type::BDBase(first_name)
   884→            };
   885→            (None, target)
   886→        };
   887→
   888→        self.expect(TokenKind::LBrace)?;
   889→
   890→        let mut methods = Vec::new();
   891→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   892→            if self.at(TokenKind::Fn) {
   893→                match self.parse_fn_def(false, Visibility::default(), Vec::new()) {
   894→                    Ok(method) => methods.push(method),
   895→                    Err(err) => {
   896→                        self.errors.push(err);
   897→                        self.recover_to_statement_boundary();
   898→                    }
   899→                }
   900→            } else {
   901→                self.advance(); // skip unexpected tokens
   902→            }
   903→        }
   904→
   905→        self.expect(TokenKind::RBrace)?;
   906→
   907→        let where_clause = if self.at(TokenKind::Where) {
   908→            Some(self.parse_where_clause()?)
   909→        } else {
   910→            None
   911→        };
   912→
   913→        Ok(ImplBlock {
   914→            attrs: _attrs,
   915→            type_params,
   916→            trait_name,
   917→            target_type,
   918→            methods,
   919→            where_clause,
   920→            span: Span::new(start, self.current.span.end),
   921→        })
   922→    }

   924→    /// `extern` `"C"` `{` <fn_decl>* `}`
   925→    ///
   926→    /// Parses an extern block declaring external functions.
   927→    /// Example: `extern "C" { fn write(fd: i64, buf: ptr, count: i64) -> i64; }`
   928→    fn parse_extern_block(&mut self) -> Result<ExternBlockDef, ParseError> {
   929→        let start = self.current.span.start;
   930→        self.expect(TokenKind::Extern)?;
   931→
   932→        // Parse the calling convention string: "C", "system", etc.
   933→        let convention = if self.at(TokenKind::String) {
   934→            let conv = self.current.lexeme.clone();
   935→            self.advance();
   936→            conv
   937→        } else {
   938→            "C".to_string() // default to C calling convention
   939→        };
   940→
   941→        self.expect(TokenKind::LBrace)?;
   942→
   943→        let mut functions = Vec::new();
   944→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
   945→            if self.at(TokenKind::Fn) {
   946→                match self.parse_extern_fn_decl() {
   947→                    Ok(fn_decl) => functions.push(fn_decl),
   948→                    Err(err) => {
   949→                        self.errors.push(err);
   950→                        self.recover_to_statement_boundary();
   951→                    }
   952→                }
   953→            } else {
   954→                self.advance(); // skip unexpected tokens
   955→            }
   956→        }
   957→
   958→        self.expect(TokenKind::RBrace)?;
   959→
   960→        Ok(ExternBlockDef {
   961→            convention,
   962→            functions,
   963→            span: Span::new(start, self.current.span.end),
   964→        })
   965→    }

   967→    /// Parse a single function declaration inside an extern block.

   969→    fn parse_extern_fn_decl(&mut self) -> Result<ExternFnDecl, ParseError> {

   976→        let params = self.parse_params()?;
   977→        self.expect(TokenKind::RParen)?;

   979→        let return_type = if self.at(TokenKind::Arrow) {

  1000→
  1001→    /// `{` <stmt>* `}`
  1002→    ///
  1003→    /// On error within a statement, the parser pushes the error to
  1004→    /// [`self.errors`] and recovers to the next statement boundary, allowing
  1005→    /// the rest of the block to be parsed and all errors to be collected.
  1006→    fn parse_block(&mut self) -> Result<Block, ParseError> {
  1007→        let start = self.current.span.start;
  1008→        self.expect(TokenKind::LBrace)?;
  1009→
  1010→        let mut statements = Vec::new();
  1011→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
  1012→            match self.parse_stmt() {
  1013→                Ok(stmt) => statements.push(stmt),
  1014→                Err(err) => {
  1015→                    self.errors.push(err);
  1016→                    self.recover_to_statement_boundary();
  1017→                }
  1018→            }
  1019→        }
  1020→
  1021→        // If we hit EOF without finding '}', push an error but don't fail
  1022→        if self.at(TokenKind::Eof) {
  1023→            self.errors.push(ParseError::expected(
  1024→                "'}'",
  1025→                "end of file",
  1026→                self.current.span,
  1027→            ));
  1028→            return Ok(Block {
  1029→                statements,
  1030→                span: Span::new(start, self.current.span.end),
  1031→            });
  1032→        }
  1033→
  1034→        self.expect(TokenKind::RBrace)?;
  1035→        let end = self.current.span.end;
  1036→
  1037→        Ok(Block {
  1038→            statements,
  1039→            span: Span::new(start, end),
  1040→        })
  1041→    }
  1042→
  1043→    /// Parse a single statement.
  1044→    fn parse_stmt(&mut self) -> Result<Stmt, ParseError> {
  1045→        match self.current.kind {
  1046→            TokenKind::Let => self.parse_let_stmt(),
  1047→            TokenKind::Mut => {
  1048→                // LLM mistake: `mut` is not a VUMA keyword for declarations
  1049→                let span = self.current.span;
  1050→                self.errors.push(ParseError::llm_mistake(
  1051→                    "`mut` is not used in VUMA — variables are mutable by default; just use assignment without `mut`",
  1052→                    span,
  1053→                    "remove `mut`",
  1054→                ));
  1055→                self.advance(); // consume `mut`
  1056→                // Try to continue parsing what comes after `mut`
  1057→                self.parse_stmt()
  1058→            }
  1059→            TokenKind::MacroIdent => {
  1060→                // LLM mistake: Rust-style macro invocation (println!, etc.)
  1061→                let span = self.current.span;
  1062→                let lexeme = self.current.lexeme.clone();
  1063→                let name_without_bang = lexeme.strip_suffix('!').unwrap_or(&lexeme);
  1064→                let hint = check_llm_construct(name_without_bang)
  1065→                    .unwrap_or("VUMA does not support macro syntax with `!`");
  1066→                self.errors.push(ParseError::llm_mistake(
  1067→                    format!("`{}` is not valid VUMA syntax — {}", lexeme, hint),
  1068→                    span,
  1069→                    "use `write` or format strings instead",
  1070→                ));
  1071→                self.advance(); // consume the macro identifier
  1072→                // Try to skip the macro arguments and continue
  1073→                if self.at(TokenKind::LParen) {
  1074→                    self.skip_balanced_parens();
  1075→                }
  1076→                if self.at(TokenKind::Semicolon) {
  1077→                    self.advance();
  1078→                }
  1079→                // Return a synthetic expression statement to allow recovery
  1080→                Ok(Stmt::Expr(ExprStmt {
  1081→                    expr: Expr::Uninitialized {
  1082→                        span: Span::new(span.start, span.end),
  1083→                    },
  1084→                    span,
  1085→                }))
  1086→            }
  1087→            TokenKind::If => self.parse_if_stmt(),
  1088→            TokenKind::While => self.parse_while_stmt(),
  1089→            TokenKind::For => self.parse_for_stmt(),
  1090→            TokenKind::Match => self.parse_match_stmt(),
  1091→            TokenKind::Return => self.parse_return_stmt(),
  1092→            TokenKind::Sync => self.parse_sync_stmt(),
  1093→            TokenKind::Free => self.parse_free_stmt(),
  1094→            TokenKind::Allocate => self.parse_allocate_stmt(),
  1095→            // BD directives
  1096→            TokenKind::Bd => self.parse_bd_directive(BdDirectiveKind::Bd),
  1097→            TokenKind::Repd => self.parse_bd_directive(BdDirectiveKind::Repd),
  1098→            TokenKind::Capd => self.parse_bd_directive(BdDirectiveKind::Capd),
  1099→            TokenKind::Reld => self.parse_bd_directive(BdDirectiveKind::Reld),
  1100→            // Handle `region` used as a variable name in assignments
  1101→            TokenKind::Region => self.parse_assign_or_expr_stmt(),
  1102→            // Break, Continue, and Loop are now proper keywords
  1103→            TokenKind::Break => self.parse_break_stmt(),
  1104→            TokenKind::Continue => self.parse_continue_stmt(),
  1105→            TokenKind::Loop => self.parse_loop_stmt(),
  1106→            // Unsafe block
  1107→            TokenKind::Unsafe => self.parse_unsafe_block(),
  1108→            // Handle Ident-based type-ascription declarations
  1109→            TokenKind::Ident => {
  1110→                // Check for type-ascription declaration: `name: type = expr;`
  1111→                let next = self.peek_next();
  1112→                if next.kind == TokenKind::Colon {
  1113→                    self.parse_type_ascription_decl()
  1114→                } else {
  1115→                    self.parse_assign_or_expr_stmt()
  1116→                }
  1117→            }
  1118→            _ => {
  1119→                // Could be an assignment or an expression statement.
  1120→                self.parse_assign_or_expr_stmt()
  1121→            }
  1122→        }
  1123→    }

  1125→    /// `let` <name> [`:` <type>] [`=` <expr>] `;`
  1126→    fn parse_let_stmt(&mut self) -> Result<Stmt, ParseError> {
  1127→        let start = self.current.span.start;
  1128→        self.expect(TokenKind::Let)?;
  1129→
  1130→        let name = self.expect_name()?;
  1131→
  1132→        let ty = if self.at(TokenKind::Colon) {
  1133→            self.advance();
  1134→            Some(self.parse_type()?)
  1135→        } else {
  1136→            None
  1137→        };
  1138→
  1139→        let value = if self.at(TokenKind::Assign) {
  1140→            self.advance();
  1141→            self.parse_expr()?
  1142→        } else {
  1143→            // `let x;` without initializer — properly represented as Uninitialized
  1144→            Expr::Uninitialized {
  1145→                span: Span::new(self.current.span.start, self.current.span.start),
  1146→            }
  1147→        };
  1148→
  1149→        self.expect(TokenKind::Semicolon)?;
  1150→
  1151→        Ok(Stmt::Let(LetStmt {
  1152→            name,
  1153→            ty,
  1154→            value,
  1155→            span: Span::new(start, self.current.span.end),
  1156→        }))
  1157→    }

  1159→    /// <name> `:` <type> `=` <expr> `;`  (type-ascription declaration without `let`)
  1160→    fn parse_type_ascription_decl(&mut self) -> Result<Stmt, ParseError> {
  1161→        let start = self.current.span.start;
  1162→
  1163→        let name = self.expect_name()?;
  1164→        self.expect(TokenKind::Colon)?;
  1165→        let ty = self.parse_type()?;
  1166→        self.expect(TokenKind::Assign)?;
  1167→        let value = self.parse_expr()?;
  1168→        self.expect(TokenKind::Semicolon)?;
  1169→
  1170→        Ok(Stmt::Let(LetStmt {
  1171→            name,
  1172→            ty: Some(ty),
  1173→            value,

  1178→    /// Try to parse an assignment, compound assignment, or expression statement.
  1179→    ///
  1180→    /// Handles: `x = expr;`, `*x = expr;`, `(*x).field = expr;`, `x[i] = expr;`,
  1181→    /// `x += expr;`, `x -= expr;`, etc., or plain expression statement `expr;`.
  1182→    fn parse_assign_or_expr_stmt(&mut self) -> Result<Stmt, ParseError> {
  1183→        let start = self.current.span.start;
  1184→
  1185→        // Parse the full expression first (including prefix ops like `*`),
  1186→        // then check for `=` or compound assignment.
  1187→        let expr = self.parse_expr()?;
  1188→
  1189→        // Compound assignment: `expr += value;`, `expr -= value;`, etc.
  1190→        if is_compound_assign(self.current.kind) {
  1191→            let op = compound_op_from_token(self.current.kind, self.current.span)?;
  1192→            self.advance(); // consume compound op
  1193→            let value = self.parse_expr()?;
  1194→            self.expect(TokenKind::Semicolon)?;
  1195→
  1196→            let target = self.expr_to_assign_target(expr, start)?;
  1197→            return Ok(Stmt::CompoundAssign(CompoundAssignStmt {
  1198→                target,
  1199→                op,
  1200→                value,
  1201→                span: Span::new(start, self.current.span.end),
  1202→            }));
  1203→        }
  1204→
  1205→        if self.at(TokenKind::Assign) {
  1206→            self.advance(); // consume '='
  1207→            let value = self.parse_expr()?;
  1208→            self.expect(TokenKind::Semicolon)?;
  1209→
  1210→            let target = self.expr_to_assign_target(expr, start)?;
  1211→            return Ok(Stmt::Assign(AssignStmt {
  1212→                target,
  1213→                value,
  1214→                span: Span::new(start, self.current.span.end),
  1215→            }));
  1216→        }
  1217→
  1218→        // Plain expression statement.
  1219→        // Allow omitting semicolon before `}` (tail expression / block return value)
  1220→        // or before `)` (last argument in macro-like call) or at EOF.
  1221→        if !self.at(TokenKind::RBrace) && !self.at(TokenKind::RParen) && !self.at(TokenKind::Eof) {
  1222→            self.expect(TokenKind::Semicolon)?;
  1223→        }
  1224→        Ok(Stmt::Expr(ExprStmt {
  1225→            expr,
  1226→            span: Span::new(start, self.current.span.end),
  1227→        }))
  1228→    }

  1388→    /// `match` <expr> `{` <arms> `}`
  1389→    fn parse_match_stmt(&mut self) -> Result<Stmt, ParseError> {
  1390→        let start = self.current.span.start;
  1391→        self.expect(TokenKind::Match)?;
  1392→        let subject = self.parse_expr()?;
  1393→        self.expect(TokenKind::LBrace)?;
  1394→
  1395→        let mut arms = Vec::new();
  1396→        while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
  1397→            let arm_start = self.current.span.start;
  1398→            let pattern = self.parse_match_pattern()?;
  1399→
  1400→            // Optional guard: `if expr`
  1401→            let guard = if self.at(TokenKind::If) {
  1402→                self.advance(); // consume 'if'
  1403→                Some(self.parse_expr()?)
  1404→            } else {
  1405→                None
  1406→            };
  1407→
  1408→            self.expect(TokenKind::FatArrow)?;
  1409→            let body = self.parse_expr()?;
  1410→            let arm_end = body.span().end;
  1411→            arms.push(MatchArm {
  1412→                pattern,
  1413→                guard,
  1414→                body,
  1415→                span: Span::new(arm_start, arm_end),
  1416→            });
  1417→            if self.at(TokenKind::Comma) {
  1418→                self.advance();
  1419→            }
  1420→        }
  1421→
  1422→        self.expect(TokenKind::RBrace)?;
  1423→        Ok(Stmt::Match(MatchStmt {
  1424→            subject,
  1425→            arms,
  1426→            span: Span::new(start, self.current.span.end),
  1427→        }))
  1428→    }

  1430→    /// Parse a match pattern.
  1431→    fn parse_match_pattern(&mut self) -> Result<MatchPattern, ParseError> {
  1432→        let span = self.current.span;
  1433→        match self.current.kind {
  1434→            // `_` wildcard — tokenized as either Ident("_") or TokenKind::Underscore
  1435→            TokenKind::Ident if self.current.lexeme == "_" => {
  1436→                self.advance();
  1437→                Ok(MatchPattern::Wildcard(span))
  1438→            }
  1439→            TokenKind::Underscore => {
  1440→                self.advance();
  1441→                Ok(MatchPattern::Wildcard(span))
  1442→            }
  1443→            TokenKind::Number => {
  1444→                let lexeme = self.current.lexeme.clone();
  1445→                self.advance();
  1446→                let value: i64 = lexeme.parse().map_err(|_| {
  1447→                    ParseError::new(
  1448→                        format!("invalid integer literal: {}", lexeme),
  1449→                        span,
  1450→                        ParseErrorKind::UnexpectedToken,
  1451→                    )
  1452→                })?;
  1453→                Ok(MatchPattern::Lit {
  1454→                    value: Lit::Int(value),
  1455→                    span,
  1456→                })
  1457→            }
  1458→            TokenKind::Float => {
  1459→                let lexeme = self.current.lexeme.clone();
  1460→                self.advance();
  1461→                let value: f64 = lexeme.parse().map_err(|_| {
  1462→                    ParseError::new(
  1463→                        format!("invalid float literal: {}", lexeme),
  1464→                        span,
  1465→                        ParseErrorKind::UnexpectedToken,
  1466→                    )
  1467→                })?;
  1468→                Ok(MatchPattern::Lit {
  1469→                    value: Lit::Float(value),
  1470→                    span,
  1471→                })
  1472→            }
  1473→            TokenKind::String => {
  1474→                let value = self.current.lexeme.clone();
  1475→                self.advance();
  1476→                Ok(MatchPattern::Lit {
  1477→                    value: Lit::String(value),
  1478→                    span,
  1479→                })
  1480→            }
  1481→            TokenKind::Address => {
  1482→                let lexeme = self.current.lexeme.clone();
  1483→                self.advance();
  1484→                let hex_str = lexeme.trim_start_matches("0x").trim_start_matches("0X");
  1485→                let value: u64 = u64::from_str_radix(hex_str, 16).map_err(|_| {
  1486→                    ParseError::invalid_address(format!("invalid hex address: {}", lexeme), span)
  1487→                })?;
  1488→                Ok(MatchPattern::Lit {
  1489→                    value: Lit::Address(value),
  1490→                    span,
  1491→                })
  1492→            }
  1493→            TokenKind::True => {
  1494→                self.advance();
  1495→                Ok(MatchPattern::Lit {
  1496→                    value: Lit::Bool(true),
  1497→                    span,
  1498→                })
  1499→            }
  1500→            TokenKind::False => {
  1501→                self.advance();
  1502→                Ok(MatchPattern::Lit {
  1503→                    value: Lit::Bool(false),
  1504→                    span,
  1505→                })
  1506→            }
  1507→            TokenKind::Ident => {
  1508→                let name = self.expect_name()?;
  1509→                // Check for struct pattern: `Name { field, ... }`
  1510→                if self.at(TokenKind::LBrace) {
  1511→                    self.advance();
  1512→                    let mut fields = Vec::new();
  1513→                    while !self.at(TokenKind::RBrace) && !self.at(TokenKind::Eof) {
  1514→                        fields.push(self.expect_name()?);
  1515→                        if self.at(TokenKind::Comma) {
  1516→                            self.advance();
  1517→                        } else {
  1518→                            break;
  1519→                        }
  1520→                    }
  1521→                    self.expect(TokenKind::RBrace)?;
  1522→                    Ok(MatchPattern::Struct { name, fields, span })
  1523→                } else if self.at(TokenKind::LParen) {
  1524→                    // Enum variant pattern: `Some(v)` or `None()`
  1525→                    self.advance();
  1526→                    let binding = if self.at(TokenKind::RParen) {
  1527→                        None
  1528→                    } else {
  1529→                        let b = self.expect_name()?;
  1530→                        Some(b)
  1531→                    };
  1532→                    self.expect(TokenKind::RParen)?;
  1533→                    Ok(MatchPattern::Enum {
  1534→                        name,
  1535→                        binding,
  1536→                        span,
  1537→                    })
  1538→                } else {
  1539→                    Ok(MatchPattern::Ident { name, span })
  1540→                }
  1541→            }
  1542→            // Handle Option/Result keywords as enum variant patterns
  1543→            TokenKind::SomeKw | TokenKind::NoneKw | TokenKind::OkKw | TokenKind::ErrKw => {
  1544→                let name = self.current.lexeme.clone();
  1545→                self.advance();
  1546→                if self.at(TokenKind::LParen) {
  1547→                    self.advance();
  1548→                    let binding = if self.at(TokenKind::RParen) {
  1549→                        None
  1550→                    } else {
  1551→                        let b = self.expect_name()?;
  1552→                        Some(b)
  1553→                    };
  1554→                    self.expect(TokenKind::RParen)?;
  1555→                    Ok(MatchPattern::Enum {
  1556→                        name,
  1557→                        binding,
  1558→                        span,
  1559→                    })
  1560→                } else {
  1561→                    Ok(MatchPattern::Ident { name, span })
  1562→                }
  1563→            }
  1564→            _ => Err(ParseError::unexpected(
  1565→                format!("expected match pattern, found {}", self.current.kind),
  1566→                self.current.span,
  1567→            )),
  1568→        }
  1569→    }

  1647→
  1648→    /// Parse an expression with full operator precedence.
  1649→    pub fn parse_expr(&mut self) -> Result<Expr, ParseError> {
  1650→        self.parse_expr_with_precedence(0)
  1651→    }
  1652→
  1653→    /// Precedence-climbing expression parser.
  1654→    ///
  1655→    /// Precedence levels (higher = binds tighter):
  1656→    ///   0  =>  ||
  1657→    ///   1  =>  &&
  1658→    ///   2  =>  ==  !=
  1659→    ///   3  =>  <  <=  >  >=
  1660→    ///   4  =>  |  (bitwise OR)
  1661→    ///   5  =>  ^  (bitwise XOR)
  1662→    ///   6  =>  &  (bitwise AND)
  1663→    ///   7  =>  <<  >>
  1664→    ///   8  =>  +  -
  1665→    ///   9  =>  *  /  %
  1666→    fn parse_expr_with_precedence(&mut self, min_prec: u8) -> Result<Expr, ParseError> {
  1667→        self.expr_depth += 1;
  1668→        if self.expr_depth > self.max_depth {
  1669→            self.expr_depth -= 1;
  1670→            return Err(ParseError::new(
  1671→                format!(
  1672→                    "expression nesting depth exceeds maximum ({})",
  1673→                    self.max_depth
  1674→                ),
  1675→                self.current.span,
  1676→                ParseErrorKind::InvalidSyntax,
  1677→            ));
  1678→        }
  1679→        let result = self.parse_expr_with_precedence_inner(min_prec);
  1680→        self.expr_depth -= 1;
  1681→        result
  1682→    }
  1683→
  1684→    /// Inner implementation of precedence climbing (called after depth check).
  1685→    fn parse_expr_with_precedence_inner(&mut self, min_prec: u8) -> Result<Expr, ParseError> {
  1686→        let mut left = self.parse_unary()?;
  1687→
  1688→        loop {
  1689→            // Range expression: `start..end` — handled as a very low-precedence
  1690→            // binary-like construct that produces Expr::Range instead of BinOp.
  1691→            if self.current.kind == TokenKind::DotDot && min_prec == 0 {
  1692→                let start = left.span().start;
  1693→                self.advance(); // consume '..'
  1694→                // Suppress struct literal parsing for the range end so that
  1695→                // `0..n { … }` does not interpret `n {` as a struct literal.
  1696→                let prev = self.no_struct_literal;
  1697→                self.no_struct_literal = true;
  1698→                let end_expr = self.parse_expr_with_precedence(1)?;
  1699→                self.no_struct_literal = prev;
  1700→                let end = end_expr.span().end;
  1701→                left = Expr::Range {
  1702→                    start: Box::new(left),
  1703→                    end: Box::new(end_expr),
  1704→                    span: Span::new(start, end),
  1705→                };
  1706→                continue;
  1707→            }
  1708→
  1709→            let (op, prec) = match self.current.kind {
  1710→                // Logical
  1711→                TokenKind::OrOr => (BinOp::Or, 0),
  1712→                TokenKind::AndAnd => (BinOp::And, 1),
  1713→                // Comparison
  1714→                TokenKind::EqEq => (BinOp::Eq, 2),
  1715→                TokenKind::Ne => (BinOp::Ne, 2),
  1716→                TokenKind::Lt => (BinOp::Lt, 3),
  1717→                TokenKind::Le => (BinOp::Le, 3),
  1718→                TokenKind::Gt => (BinOp::Gt, 3),
  1719→                TokenKind::Ge => (BinOp::Ge, 3),
  1720→                // Bitwise
  1721→                TokenKind::Pipe => (BinOp::BitOr, 4),
  1722→                TokenKind::Caret => (BinOp::BitXor, 5),
  1723→                TokenKind::Ampersand => (BinOp::BitAnd, 6),
  1724→                TokenKind::Shl => (BinOp::Shl, 7),
  1725→                TokenKind::Shr => (BinOp::Shr, 7),
  1726→                // Additive
  1727→                TokenKind::Plus => (BinOp::Add, 8),
  1728→                TokenKind::Minus => (BinOp::Sub, 8),
  1729→                // Multiplicative
  1730→                TokenKind::Star => (BinOp::Mul, 9),
  1731→                TokenKind::Slash => (BinOp::Div, 9),
  1732→                TokenKind::Percent => (BinOp::Mod, 9),
  1733→                _ => break,
  1734→            };
  1735→
  1736→            if prec < min_prec {
  1737→                break;
  1738→            }
  1739→
  1740→            let start = left.span().start;
  1741→            self.advance(); // consume the operator
  1742→            let right = self.parse_expr_with_precedence(prec + 1)?;
  1743→            let end = right.span().end;
  1744→
  1745→            left = Expr::BinOp {
  1746→                op,
  1747→                lhs: Box::new(left),
  1748→                rhs: Box::new(right),
  1749→                span: Span::new(start, end),
  1750→            };
  1751→        }
  1752→
  1753→        Ok(left)
  1754→    }

  1823→    /// Parse postfix operators: calls, field access, indexing, `as` casts,
  1824→    /// namespace access `::`.
  1825→    fn parse_postfix(&mut self) -> Result<Expr, ParseError> {
  1826→        let mut expr = self.parse_primary()?;
  1827→
  1828→        loop {
  1829→            match self.current.kind {
  1830→                TokenKind::LParen => {
  1831→                    // Function call.
  1832→                    let start = expr.span().start;
  1833→                    self.advance(); // consume '('
  1834→                    let mut args = Vec::new();
  1835→                    if !self.at(TokenKind::RParen) {
  1836→                        args.push(self.parse_expr()?);
  1837→                        while self.at(TokenKind::Comma) {
  1838→                            self.advance();
  1839→                            args.push(self.parse_expr()?);
  1840→                        }
  1841→                    }
  1842→                    self.expect(TokenKind::RParen)?;
  1843→                    let end = self.current.span.end;
  1844→                    expr = Expr::Call {
  1845→                        callee: Box::new(expr),
  1846→                        args,
  1847→                        span: Span::new(start, end),
  1848→                    };
  1849→                }
  1850→                TokenKind::Dot => {
  1851→                    // Field access or .await
  1852→                    let start = expr.span().start;
  1853→                    self.advance(); // consume '.'
  1854→
  1855→                    // Check for .await
  1856→                    if self.current.kind == TokenKind::Await {
  1857→                        self.advance(); // consume 'await'
  1858→                        let end = self.current.span.end;
  1859→                        expr = Expr::Await {
  1860→                            expr: Box::new(expr),
  1861→                            span: Span::new(start, end),
  1862→                        };
  1863→                        continue;
  1864→                    }
  1865→
  1866→                    let field = self.expect_name()?;
  1867→                    let end = self.current.span.end;
  1868→                    expr = Expr::FieldAccess {
  1869→                        expr: Box::new(expr),
  1870→                        field,
  1871→                        span: Span::new(start, end),
  1872→                    };
  1873→                }
  1874→                TokenKind::LBracket => {
  1875→                    // Index access: expr[index]
  1876→                    let start = expr.span().start;
  1877→                    self.advance(); // consume '['
  1878→                    let index = self.parse_expr()?;
  1879→                    self.expect(TokenKind::RBracket)?;
  1880→                    let end = self.current.span.end;
  1881→                    expr = Expr::Index {
  1882→                        expr: Box::new(expr),
  1883→                        index: Box::new(index),
  1884→                        span: Span::new(start, end),
  1885→                    };
  1886→                }
  1887→                TokenKind::LBrace => {
  1888→                    // Struct literal (only if `expr` is an identifier-like name
  1889→                    // and the brace is followed by `ident :` — otherwise the `{`
  1890→                    // belongs to a block statement like `if cond { … }`).
  1891→                    //
  1892→                    // When `no_struct_literal` is set (e.g. parsing the end of a
  1893→                    // range expression like `0..n`), skip this branch entirely so
  1894→                    // that the `{` remains for the enclosing construct.
  1895→                    if self.no_struct_literal {
  1896→                        break;
  1897→                    }
  1898→                    let start = expr.span().start;
  1899→                    if let Expr::Var { name, .. } = &expr {
  1900→                        let name = name.clone();
  1901→                        let saved_lbrace = self.current.clone();
  1902→                        self.advance(); // consume '{'
  1903→
  1904→                        // Disambiguation: if the first token inside the braces
  1905→                        // is NOT `ident :`, this is not a struct literal.
  1906→                        // Rewind and let the caller (if/while/for/match) handle the block.
  1907→                        let is_struct_literal = if self.at(TokenKind::RBrace) {
  1908→                            // Empty braces: `Foo {}` is a valid struct literal
  1909→                            true
  1910→                        } else if self.current.kind == TokenKind::Ident {
  1911→                            // Peek at the token after the field name
  1912→                            let after_field = self.peek_next();
  1913→                            after_field.kind == TokenKind::Colon
  1914→                        } else {
  1915→                            false
  1916→                        };
  1917→
  1918→                        if !is_struct_literal {
  1919→                            // Rewind: put back the current token and the `{`
  1920→                            self.push_back_current(saved_lbrace);
  1921→                            break;
  1922→                        }
  1923→
  1924→                        let mut fields = Vec::new();
  1925→                        if !self.at(TokenKind::RBrace) {
  1926→                            let fname = self.expect_name()?;
  1927→                            self.expect(TokenKind::Colon)?;
  1928→                            let fval = self.parse_expr()?;
  1929→                            fields.push((fname, fval));
  1930→                            while self.at(TokenKind::Comma) {
  1931→                                self.advance();
  1932→                                if self.at(TokenKind::RBrace) {
  1933→                                    break; // trailing comma
  1934→                                }
  1935→                                let fname = self.expect_name()?;
  1936→                                self.expect(TokenKind::Colon)?;
  1937→                                let fval = self.parse_expr()?;
  1938→                                fields.push((fname, fval));
  1939→                            }
  1940→                        }
  1941→                        self.expect(TokenKind::RBrace)?;
  1942→                        let end = self.current.span.end;
  1943→                        expr = Expr::StructInit {
  1944→                            name,
  1945→                            fields,
  1946→                            span: Span::new(start, end),
  1947→                        };
  1948→                    } else {
  1949→                        break;
  1950→                    }
  1951→                }
  1952→                TokenKind::As => {
  1953→                    // Type cast.
  1954→                    let start = expr.span().start;
  1955→                    self.advance(); // consume 'as'
  1956→                    let target_type = self.parse_type()?;
  1957→                    let end = self.current.span.end;
  1958→                    expr = Expr::Cast {
  1959→                        expr: Box::new(expr),
  1960→                        target_type,
  1961→                        span: Span::new(start, end),
  1962→                    };
  1963→                }
  1964→                TokenKind::PathSep => {
  1965→                    // Namespace / associated function access: `expr::name`
  1966→                    let start = expr.span().start;
  1967→                    self.advance(); // consume '::'
  1968→                    let name = self.expect_name()?;
  1969→                    let end = self.current.span.end;
  1970→                    expr = Expr::NamespaceAccess {
  1971→                        expr: Box::new(expr),
  1972→                        name,
  1973→                        span: Span::new(start, end),
  1974→                    };
  1975→                }
  1976→                _ => break,
  1977→            }
  1978→        }
  1979→
  1980→        Ok(expr)
  1981→    }
