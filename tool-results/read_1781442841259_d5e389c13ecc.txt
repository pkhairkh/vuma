     1→//! # LoongArch 64-bit Backend
     2→//!
     3→//! Implements the `Backend` trait for the LoongArch 64-bit target (LP64 ABI).
     4→//! This module provides:
     5→//!
     6→//! - `Gpr` — General-purpose register enum (r0–r31)
     7→//! - `Fpr` — Floating-point register enum (f0–f31)
     8→//! - `Instruction` — LoongArch64 instruction enum with correct encoding
     9→//! - Encoding helpers for all 9 instruction formats (2R, 3R, 4R, 2RI8, 2RI12,
    10→//!   2RI14, 2RI16, 1RI21, I26)
    11→//! - `LoongArch64Backend` — `Backend` implementation that lowers IR to LoongArch64
    12→//!   machine code and emits ELF64 binaries
    13→//!
    14→//! ## LoongArch64 Register Convention (LP64 ABI)
    15→//!
    16→//! | Register(s) | ABI Name | Role                              |
    17→//! |-------------|----------|-----------------------------------|
    18→//! | r0          | zero     | Hardwired zero                    |
    19→//! | r1          | ra       | Return address                    |
    20→//! | r2          | tp       | Thread pointer                    |
    21→//! | r3          | sp       | Stack pointer                     |
    22→//! | r4–r11      | a0–a7    | Argument / return registers       |
    23→//! | r12–r20     | t0–t8    | Caller-saved temporaries          |
    24→//! | r21         | —        | Reserved                          |
    25→//! | r22         | fp       | Frame pointer (callee-saved)      |
    26→//! | r23–r31     | s0–s8    | Callee-saved                      |
    27→//!
    28→//! ## LoongArch64 FP Register Convention (LP64 ABI)
    29→//!
    30→//! | Register(s) | ABI Name | Role                              |
    31→//! |-------------|----------|-----------------------------------|
    32→//! | f0–f7       | fa0–fa7  | FP argument / return registers    |
    33→//! | f8–f23      | ft0–ft15 | Caller-saved temporaries          |
    34→//! | f24–f31     | fs0–fs7  | Callee-saved                      |
    35→//!
    36→//! ## Instruction Formats
    37→//!
    38→//! All instructions are 32 bits, little-endian, with no branch delay slots.
    39→//! Nine formats: 2R, 3R, 4R, 2RI8, 2RI12, 2RI14, 2RI16, 1RI21, I26.
    40→//!
    41→//! ## References
    42→//!
    43→//! - LoongArch Reference Manual, Volume 1: Basic Architecture
    44→//! - <https://loongson.github.io/LoongArch-Documentation/>
    45→

    59→#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]

    61→    R0 = 0,
    62→    Ra = 1,
    63→    Tp = 2,
    64→    Sp = 3,
    65→    A0 = 4,
    66→    A1 = 5,

    75→    T2 = 14,
    76→    T3 = 15,
    77→    T4 = 16,
    78→    T5 = 17,
    79→    T6 = 18,
    80→    T7 = 19,

    95→impl Gpr {
    96→    /// Returns the 5-bit encoding index for this register.
    97→    pub fn encoding(&self) -> u32 {
    98→        *self as u32
    99→    }
   100→
   101→    /// Returns `true` if this register is available for register allocation.
   102→    ///
   103→    /// R0 (zero), Ra, Tp, and Sp are reserved.
   104→    pub fn is_allocatable(&self) -> bool {
   105→        !matches!(self, Gpr::R0 | Gpr::Ra | Gpr::Tp | Gpr::Sp)
   106→    }
   107→
   108→    /// Returns `true` if this register is callee-saved (fp, s0–s8).
   109→    pub fn is_callee_saved(&self) -> bool {
   110→        matches!(
   111→            self,
   112→            Gpr::Fp
   113→                | Gpr::S0
   114→                | Gpr::S1
   115→                | Gpr::S2
   116→                | Gpr::S3
   117→                | Gpr::S4
   118→                | Gpr::S5
   119→                | Gpr::S6
   120→                | Gpr::S7
   121→                | Gpr::S8
   122→        )
   123→    }
   124→
   125→    /// Returns `true` if this register is an argument register (a0–a7).
   126→    pub fn is_arg_reg(&self) -> bool {
   127→        matches!(
   128→            self,
   129→            Gpr::A0 | Gpr::A1 | Gpr::A2 | Gpr::A3 | Gpr::A4 | Gpr::A5 | Gpr::A6 | Gpr::A7
   130→        )
   131→    }
   132→

   142→            Gpr::A2 => "$a2",
   143→            Gpr::A3 => "$a3",
   144→            Gpr::A4 => "$a4",
   145→            Gpr::A5 => "$a5",
   146→            Gpr::A6 => "$a6",
   147→            Gpr::A7 => "$a7",
   148→            Gpr::T0 => "$t0",
   149→            Gpr::T1 => "$t1",
   150→            Gpr::T2 => "$t2",
   151→            Gpr::T3 => "$t3",
   152→            Gpr::T4 => "$t4",
   153→            Gpr::T5 => "$t5",
   154→            Gpr::T6 => "$t6",
   155→            Gpr::T7 => "$t7",
   156→            Gpr::T8 => "$t8",
   157→            Gpr::R21 => "$r21",
   158→            Gpr::Fp => "$fp",
   159→            Gpr::S0 => "$s0",
   160→            Gpr::S1 => "$s1",
   161→            Gpr::S2 => "$s2",
   162→            Gpr::S3 => "$s3",
   163→            Gpr::S4 => "$s4",
   164→            Gpr::S5 => "$s5",
   165→            Gpr::S6 => "$s6",
   166→            Gpr::S7 => "$s7",
   167→            Gpr::S8 => "$s8",

   169→    }

   171→    /// Returns the Gpr for a given argument index (0–7). Returns `None` for
   172→    /// indices >= 8.
   173→    pub fn arg_register(index: usize) -> Option<Gpr> {
   174→        match index {
   175→            0 => Some(Gpr::A0),
   176→            1 => Some(Gpr::A1),
   177→            2 => Some(Gpr::A2),
   178→            3 => Some(Gpr::A3),
   179→            4 => Some(Gpr::A4),
   180→            5 => Some(Gpr::A5),

   182→            7 => Some(Gpr::A7),
   183→            _ => None,
   184→        }
   185→    }
   186→}

   323→// ===========================================================================
   324→// Instruction Encoding Helpers
   325→// ===========================================================================
   326→
   327→/// Encode a 2R format instruction.
   328→///
   329→/// Format: `opcode[31:10] | rj[9:5] | rd[4:0]`
   330→fn encode_2r(opcode: u32, rj: u32, rd: u32) -> [u8; 4] {
   331→    let word = ((opcode & 0x3FF_FFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   332→    word.to_le_bytes()
   333→}
   334→
   335→/// Encode a 3R format instruction.
   336→///
   337→/// Format: `opcode[31:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   338→fn encode_3r(opcode: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   339→    let word = ((opcode & 0x1FFFF) << 15) | ((rk & 0x1F) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   340→    word.to_le_bytes()
   341→}
   342→
   343→/// Encode a 4R format instruction.
   344→///
   345→/// Format: `opcode[31:20] | ra[19:15] | rk[14:10] | rj[9:5] | rd[4:0]`
   346→#[allow(dead_code)]
   347→fn encode_4r(opcode: u32, ra: u32, rk: u32, rj: u32, rd: u32) -> [u8; 4] {
   348→    let word = ((opcode & 0xFFF) << 20)
   349→        | ((ra & 0x1F) << 15)
   350→        | ((rk & 0x1F) << 10)
   351→        | ((rj & 0x1F) << 5)
   352→        | (rd & 0x1F);
   353→    word.to_le_bytes()
   354→}
   355→
   356→/// Encode a 2RI8 format instruction.
   357→///
   358→/// Format: `opcode[31:22] | I8[21:14] | rj[9:5] | rd[4:0]`
   359→fn encode_2ri8(opcode: u32, imm8: u32, rj: u32, rd: u32) -> [u8; 4] {
   360→    let word = ((opcode & 0x3FF) << 22) | ((imm8 & 0xFF) << 14) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   361→    word.to_le_bytes()
   362→}
   363→
   364→/// Encode a 2RI12 format instruction.
   365→///
   366→/// Format: `opcode[31:22] | I12[21:10] | rj[9:5] | rd[4:0]`
   367→fn encode_2ri12(opcode: u32, imm12: u32, rj: u32, rd: u32) -> [u8; 4] {
   368→    let word =
   369→        ((opcode & 0x3FF) << 22) | ((imm12 & 0xFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   370→    word.to_le_bytes()
   371→}
   372→
   373→/// Encode a 2RI14 format instruction.
   374→///
   375→/// Format: `opcode[31:24] | I14[23:10] | rj[9:5] | rd[4:0]`
   376→fn encode_2ri14(opcode: u32, imm14: u32, rj: u32, rd: u32) -> [u8; 4] {
   377→    let word =
   378→        ((opcode & 0xFF) << 24) | ((imm14 & 0x3FFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   379→    word.to_le_bytes()
   380→}
   381→
   382→/// Encode a 2RI16 format instruction.
   383→///
   384→/// Format: `opcode[31:26] | I16[25:10] | rj[9:5] | rd[4:0]`
   385→fn encode_2ri16(opcode: u32, imm16: u32, rj: u32, rd: u32) -> [u8; 4] {
   386→    let word =
   387→        ((opcode & 0x3F) << 26) | ((imm16 & 0xFFFF) << 10) | ((rj & 0x1F) << 5) | (rd & 0x1F);
   388→    word.to_le_bytes()
   389→}
   390→
   391→/// Encode a 1RI21 format instruction (used for BEQZ, BNEZ).
   392→///
   393→/// Format: `opcode[31:26] | offs21[15:0] in bits[25:10] | offs21[20:16] in bits[9:5] | rj[4:0]`
   394→///
   395→/// Note: the offset bits are SWAPPED compared to a linear layout.
   396→/// The lower 16 bits of the offset go in the higher position (bits[25:10]),
   397→/// and the upper 5 bits go in the lower position (bits[9:5]).
   398→fn encode_1ri21(opcode: u32, imm21: u32, rj: u32) -> [u8; 4] {
   399→    let word = ((opcode & 0x3F) << 26)
   400→        | ((imm21 & 0xFFFF) << 10)
   401→        | ((imm21 >> 16) & 0x1F) << 5
   402→        | (rj & 0x1F);
   403→    word.to_le_bytes()
   404→}
   405→
   406→/// Encode an I26 format instruction (used for B, BL).
   407→///
   408→/// Format: `opcode[31:26] | offs26[15:0] in bits[25:10] | offs26[25:16] in bits[9:0]`
   409→///
   410→/// Note: the offset bits are SWAPPED compared to a linear layout.
   411→/// The lower 16 bits of the offset go in the higher position (bits[25:10]),
   412→/// and the upper 10 bits go in the lower position (bits[9:0]).
   413→fn encode_i26(opcode: u32, imm26: u32) -> [u8; 4] {
   414→    let word = ((opcode & 0x3F) << 26)
   415→        | ((imm26 & 0xFFFF) << 10)
   416→        | ((imm26 >> 16) & 0x3FF);
   417→    word.to_le_bytes()
   418→}

   850→impl Instruction {
   851→    /// Encode this instruction into a 4-byte little-endian machine code word.
   852→    ///
   853→    /// Encoding follows the LoongArch ISA Specification.
   854→    pub fn encode(&self) -> [u8; 4] {
   855→        match self {
   856→            // ── Arithmetic (3R) ────────────────────────────────────
   857→            Instruction::AddW { rd, rj, rk } => {
   858→                encode_3r(OPC_ADD_W, rk.encoding(), rj.encoding(), rd.encoding())
   859→            }
   860→            Instruction::AddD { rd, rj, rk } => {
   861→                encode_3r(OPC_ADD_D, rk.encoding(), rj.encoding(), rd.encoding())
   862→            }
   863→            Instruction::SubW { rd, rj, rk } => {
   864→                encode_3r(OPC_SUB_W, rk.encoding(), rj.encoding(), rd.encoding())
   865→            }
   866→            Instruction::SubD { rd, rj, rk } => {
   867→                encode_3r(OPC_SUB_D, rk.encoding(), rj.encoding(), rd.encoding())
   868→            }
   869→            Instruction::Slt { rd, rj, rk } => {
   870→                encode_3r(OPC_SLT, rk.encoding(), rj.encoding(), rd.encoding())
   871→            }
   872→            Instruction::Sltu { rd, rj, rk } => {
   873→                encode_3r(OPC_SLTU, rk.encoding(), rj.encoding(), rd.encoding())
   874→            }
   875→            Instruction::MulW { rd, rj, rk } => {
   876→                encode_3r(OPC_MUL_W, rk.encoding(), rj.encoding(), rd.encoding())
   877→            }
   878→            Instruction::MulD { rd, rj, rk } => {
   879→                encode_3r(OPC_MUL_D, rk.encoding(), rj.encoding(), rd.encoding())
   880→            }
   881→            Instruction::DivW { rd, rj, rk } => {
   882→                encode_3r(OPC_DIV_W, rk.encoding(), rj.encoding(), rd.encoding())
   883→            }
   884→            Instruction::ModW { rd, rj, rk } => {
   885→                encode_3r(OPC_MOD_W, rk.encoding(), rj.encoding(), rd.encoding())
   886→            }
   887→            Instruction::DivD { rd, rj, rk } => {
   888→                encode_3r(OPC_DIV_D, rk.encoding(), rj.encoding(), rd.encoding())
   889→            }
   890→            Instruction::ModD { rd, rj, rk } => {
   891→                encode_3r(OPC_MOD_D, rk.encoding(), rj.encoding(), rd.encoding())
   892→            }
   893→
   894→            // ── Logical (3R) ──────────────────────────────────────
   895→            Instruction::And { rd, rj, rk } => {
   896→                encode_3r(OPC_AND, rk.encoding(), rj.encoding(), rd.encoding())
   897→            }
   898→            Instruction::Or { rd, rj, rk } => {
   899→                encode_3r(OPC_OR, rk.encoding(), rj.encoding(), rd.encoding())
   900→            }
   901→            Instruction::Xor { rd, rj, rk } => {
   902→                encode_3r(OPC_XOR, rk.encoding(), rj.encoding(), rd.encoding())
   903→            }
   904→            Instruction::Nor { rd, rj, rk } => {
   905→                encode_3r(OPC_NOR, rk.encoding(), rj.encoding(), rd.encoding())
   906→            }
   907→            Instruction::Andn { rd, rj, rk } => {
   908→                encode_3r(OPC_ANDN, rk.encoding(), rj.encoding(), rd.encoding())
   909→            }
   910→            Instruction::Orn { rd, rj, rk } => {
   911→                encode_3r(OPC_ORN, rk.encoding(), rj.encoding(), rd.encoding())

  1538→/// Build a minimal ELF64 binary for LoongArch64 from raw code bytes.
  1539→///
  1540→/// Produces a static executable with two LOAD segments:
  1541→///   1. PF_R | PF_X — .text (code)
  1542→///   2. PF_R | PF_W — .data / stack (writable memory)
  1543→fn build_loongarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1544→    const PAGE_SIZE: u64 = 0x10000; // 64 KB (LoongArch64 typical page size)
  1545→
  1546→    let elf_header_size: u64 = 64;
  1547→    let phdr_size: u64 = 56;
  1548→    let num_phdrs: u64 = 2;
  1549→    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1550→    // Page-align the text segment start in the file.
  1551→    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1552→    let text_size = code.len() as u64;
  1553→
  1554→    // The data segment starts on the next page after the text.
  1555→    let text_file_end = text_offset + text_size;
  1556→    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1557→    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1558→    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1559→    let entry_point = base_addr + text_offset;
  1560→
  1561→    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1562→
  1563→    // --- e_ident ---
  1564→    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1565→    elf.push(2); // ELFCLASS64
  1566→    elf.push(1); // ELFDATA2LSB
  1567→    elf.push(1); // EV_CURRENT
  1568→    elf.push(3); // ELFOSABI_LINUX
  1569→    elf.push(0); // padding
  1570→    elf.extend_from_slice(&[0u8; 7]); // padding
  1571→
  1572→    // --- ELF header fields ---
  1573→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1574→    elf.extend_from_slice(&258u16.to_le_bytes()); // e_machine = EM_LOONGARCH
  1575→    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1576→    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1577→    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1578→    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1579→    elf.extend_from_slice(&0x43u32.to_le_bytes()); // e_flags = 0x43 (LP64D ABI double-float)
  1580→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1581→    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1582→    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1583→    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1584→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1585→    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1586→
  1587→    // --- Program Header 1: LOAD (PF_R | PF_X) — .text ---
  1588→    // p_filesz = actual code size, p_memsz = page-aligned (QEMU needs this for mapping)
  1589→    let text_memsz = ((text_size + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1590→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1591→    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1592→    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1593→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1594→    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1595→    elf.extend_from_slice(&text_memsz.to_le_bytes()); // p_filesz (cover full page for QEMU)
  1596→    elf.extend_from_slice(&text_memsz.to_le_bytes()); // p_memsz
  1597→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1598→
  1599→    // --- Program Header 2: LOAD (PF_R | PF_W) — .data / stack ---
  1600→    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1601→    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1602→    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1603→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1604→    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1605→    elf.extend_from_slice(&data_size.to_le_bytes()); // p_filesz (write zeros so QEMU can mmap)
  1606→    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1607→    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1608→
  1609→    // --- .text section ---
  1610→    // Pad to page-aligned text_offset
  1611→    while (elf.len() as u64) < text_offset {
  1612→        elf.push(0);
  1613→    }
  1614→    elf.extend_from_slice(code);
  1615→
  1616→    // --- Pad to data segment offset and write data segment ---
  1617→    while (elf.len() as u64) < data_offset {
  1618→        elf.push(0);
  1619→    }
  1620→    // Write data segment (all zeros, but must exist in file for QEMU mmap)
  1621→    elf.extend_from_slice(&vec![0u8; data_size as usize]);
  1622→
  1623→    elf
  1624→}

  1630→/// Decode a single LoongArch64 32-bit instruction word into a mnemonic string.
  1631→fn decode_loongarch64_instruction(word: u32) -> String {
  1632→    let opcode = word & 0x7f;
  1633→    match opcode {
  1634→        0x00 => {
  1635→            let rd = (word >> 7) & 0x1f;
  1636→            let rj = (word >> 15) & 0x1f;
  1637→            let rk = (word >> 10) & 0x1f;
  1638→            format!("add.w $r{}, $r{}, $r{}", rd, rj, rk)
  1639→        }
  1640→        0x01 => {
  1641→            let rd = (word >> 7) & 0x1f;
  1642→            let rj = (word >> 15) & 0x1f;
  1643→            let rk = (word >> 10) & 0x1f;
  1644→            format!("sub.w $r{}, $r{}, $r{}", rd, rj, rk)
  1645→        }
  1646→        0x02 => {
  1647→            let rd = (word >> 7) & 0x1f;
  1648→            let rj = (word >> 15) & 0x1f;
  1649→            let rk = (word >> 10) & 0x1f;
  1650→            format!("mul.w $r{}, $r{}, $r{}", rd, rj, rk)
  1651→        }
  1652→        0x05 => {
  1653→            let rd = (word >> 7) & 0x1f;
  1654→            let _rj = (word >> 15) & 0x1f; // lu12i.w does not use rj in the disassembly
  1655→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1656→            format!("lu12i.w $r{}, {}({})", rd, si12, si12)
  1657→        }
  1658→        0x08 => {
  1659→            let rd = (word >> 7) & 0x1f;
  1660→            let rj = (word >> 15) & 0x1f;
  1661→            let si12 = ((word >> 10) as i32) << 20 >> 20;
  1662→            format!("ld.w $r{}, $r{}, {}({})", rd, rj, si12, si12)
  1663→        }
  1664→        0x0a => {
  1665→            let rd = (word >> 7) & 0x1f;
  1666→            let rj = (word >> 15) & 0x1f;

  1721→/// Compute the stack frame size for an IR function on LoongArch64.
  1722→///
  1723→/// Sums `Alloc` instruction sizes, adds 16 bytes for the ra/fp save pair,
  1724→/// and rounds up to 16-byte alignment.
  1725→fn loongarch64_compute_frame_size(func: &IRFunction) -> usize {
  1726→    let mut total: u32 = 16; // ra + fp save pair
  1727→    for block in &func.blocks {
  1728→        for instr in &block.instructions {
  1729→            if let IRInstr::Alloc { size, .. } = instr {
  1730→                let aligned = (*size).div_ceil(16) * 16;
  1731→                total += aligned;
  1732→            }
  1733→        }
  1734→    }
  1735→    // Round up to 16-byte alignment
  1736→    total = (total + 15) & !15;
  1737→    total as usize
  1738→}

  1777→/// Map from virtual register ID to a physical GPR using a simple linear scan.
  1778→fn map_vreg_to_gpr(
  1779→    vreg_id: u32,
  1780→    arg_index: Option<usize>,
  1781→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1782→) -> Gpr {
  1783→    if let Some(gpr) = vreg_map.get(&vreg_id) {
  1784→        return *gpr;
  1785→    }
  1786→
  1787→    // If this is an argument, map it to the corresponding argument register.
  1788→    if let Some(idx) = arg_index {
  1789→        if let Some(reg) = Gpr::arg_register(idx) {
  1790→            vreg_map.insert(vreg_id, reg);
  1791→            return reg;
  1792→        }
  1793→    }
  1794→
  1795→    // Otherwise, pick the first available allocatable register.
  1796→    let used: std::collections::HashSet<Gpr> = vreg_map.values().copied().collect();
  1797→    for &reg in ALLOCATABLE_GPRS {
  1798→        if !used.contains(&reg) {
  1799→            vreg_map.insert(vreg_id, reg);
  1800→            return reg;
  1801→        }
  1802→    }
  1803→
  1804→    // Fallback: use t8 as a scratch register.
  1805→    Gpr::T8
  1806→}

  1816→/// Load a 64-bit immediate into a register using the lu12i.w/ori/lu32i.d/lu52i.d
  1817→/// instruction sequence.
  1818→///
  1819→/// Returns the list of AllocatedInstructions emitted to perform the load.
  1820→fn load_imm_la64(rd: Gpr, imm: i64) -> Vec<AllocatedInstruction> {
  1821→    let mut result = Vec::new();
  1822→    let val = imm as u64;
  1823→
  1824→    // lu12i.w rd, bits[31:12]
  1825→    let hi20 = ((val >> 12) & 0xFFFFF) as i32;
  1826→    result.push(emit_alloc_instr(
  1827→        Instruction::Lu12iW { rd, imm20: hi20 },
  1828→        vec![],
  1829→        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1830→    ));
  1831→
  1832→    // ori rd, rd, bits[11:0]
  1833→    let lo12 = (val & 0xFFF) as u32;
  1834→    result.push(emit_alloc_instr(
  1835→        Instruction::Ori {
  1836→            rd,
  1837→            rj: rd,
  1838→            imm12: lo12,
  1839→        },
  1840→        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1841→        vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1842→    ));
  1843→
  1844→    // lu32i.d is actually pcaddi in QEMU — skip it
  1845→    // Instead, handle upper bits differently
  1846→
  1847→    let lower32 = val & 0xFFFFFFFF;
  1848→    let sign_ext = if lower32 & 0x80000000 != 0 {
  1849→        0xFFFFFFFF00000000u64
  1850→    } else {
  1851→        0u64
  1852→    };
  1853→
  1854→    if val >> 32 != sign_ext >> 32 {
  1855→        // Upper bits don't match sign extension
  1856→        if val >> 32 == 0 && lower32 & 0x80000000 != 0 {
  1857→            // Zero-extend: slli.d + srli.d
  1858→            result.push(emit_alloc_instr(
  1859→                Instruction::SlliD { rd, rj: rd, imm8: 32 },
  1860→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1861→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1862→            ));
  1863→            result.push(emit_alloc_instr(
  1864→                Instruction::SrliD { rd, rj: rd, imm8: 32 },
  1865→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1866→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1867→            ));
  1868→        } else {
  1869→            // Full 64-bit: use lu52i.d for bits[63:52] then slli+srli+or for bits[51:32]
  1870→            let hi52 = ((val >> 52) & 0xFFF) as i32;
  1871→            result.push(emit_alloc_instr(
  1872→                Instruction::Lu52iD {
  1873→                    rd,
  1874→                    rj: rd,
  1875→                    imm12: hi52,
  1876→                },
  1877→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1878→                vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1879→            ));
  1880→            let bits_51_32 = ((val >> 32) & 0xFFFFF) as u32;
  1881→            if bits_51_32 != 0 {
  1882→                let temp = Gpr::T0;
  1883→                let hi_51_32 = ((val >> 32) & 0xFFFFF) as i32;
  1884→                result.push(emit_alloc_instr(
  1885→                    Instruction::Lu12iW { rd: temp, imm20: hi_51_32 },
  1886→                    vec![],
  1887→                    vec![PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1888→                ));
  1889→                result.push(emit_alloc_instr(
  1890→                    Instruction::SlliD { rd: temp, rj: temp, imm8: 32 },
  1891→                    vec![PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1892→                    vec![PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1893→                ));
  1894→                result.push(emit_alloc_instr(
  1895→                    Instruction::SlliD { rd, rj: rd, imm8: 32 },
  1896→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1897→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1898→                ));
  1899→                result.push(emit_alloc_instr(
  1900→                    Instruction::SrliD { rd, rj: rd, imm8: 32 },
  1901→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1902→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1903→                ));
  1904→                result.push(emit_alloc_instr(
  1905→                    Instruction::Or { rd, rj: rd, rk: temp },
  1906→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding()), PhysicalReg::new(RegClass::Gpr, temp.encoding())],
  1907→                    vec![PhysicalReg::new(RegClass::Gpr, rd.encoding())],
  1908→                ));
  1909→            }
  1910→        }
  1911→    }

  1916→/// Resolve an IRValue to a physical GPR.
  1917→///
  1918→/// - For `IRValue::Register`: looks up in `vreg_map`.
  1919→/// - For `IRValue::Immediate`: loads the value into `scratch` and returns
  1920→///   `(scratch, instructions)`.
  1921→/// - For `IRValue::Address`: loads the address into `scratch`.
  1922→/// - For `IRValue::Label`: loads a placeholder into `scratch`.
  1923→fn resolve_gpr_la64(
  1924→    val: &IRValue,
  1925→    vreg_map: &mut std::collections::HashMap<u32, Gpr>,
  1926→    scratch: Gpr,
  1927→) -> (Gpr, Vec<AllocatedInstruction>) {

  1929→        IRValue::Register(id) => {
  1930→            let reg = map_vreg_to_gpr(*id, None, vreg_map);
  1931→            (reg, Vec::new())
  1932→        }
  1933→        IRValue::Immediate(imm) => {
  1934→            let code = load_imm_la64(scratch, *imm);
  1935→            (scratch, code)

  1937→        IRValue::Address(addr) => {
  1938→            let code = load_imm_la64(scratch, *addr as i64);

  1943→            let code = load_imm_la64(scratch, 0);
  1944→            (scratch, code)
  1945→        }
  1946→    }
  1947→}
