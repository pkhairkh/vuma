     1ŌåÆ//! # x86_64 Backend
     2ŌåÆ//!
     3ŌåÆ//! Implements the `Backend` trait for the x86_64 target (SystemV ABI).
     4ŌåÆ//! This module provides:

    13ŌåÆ//! ## x86_64 Register Convention (SystemV ABI)
    14ŌåÆ//!
    15ŌåÆ//! | Register | Role                    | Callee-saved |
    16ŌåÆ//! |----------|-------------------------|-------------|
    17ŌåÆ//! | RAX      | Return value / scratch  | No          |
    18ŌåÆ//! | RCX      | 4th int arg / scratch   | No          |
    19ŌåÆ//! | RDX      | 3rd int arg / scratch   | No          |
    20ŌåÆ//! | RBX      | Callee-saved            | Yes         |
    21ŌåÆ//! | RSP      | Stack pointer           | (special)   |
    22ŌåÆ//! | RBP      | Frame pointer           | Yes         |
    23ŌåÆ//! | RSI      | 2nd int arg / scratch   | No          |
    24ŌåÆ//! | RDI      | 1st int arg / scratch   | No          |
    25ŌåÆ//! | R8ŌĆōR9    | 5th/6th int arg         | No          |
    26ŌåÆ//! | R10ŌĆōR11  | Scratch                 | No          |
    27ŌåÆ//! | R12ŌĆōR15  | Callee-saved            | Yes         |
    28ŌåÆ//!
    29ŌåÆ//! ## References

   380ŌåÆ/// - `b`: Extension of the ModR/M `rm` field or SIB `base` field
   381ŌåÆ///
   382ŌåÆ/// Returns `None` if no REX byte is needed (all bits are 0).
   383ŌåÆfn rex_prefix(w: bool, r: bool, x: bool, b: bool) -> Option<u8> {
   384ŌåÆ    let byte = 0x40 | (w as u8) << 3 | (r as u8) << 2 | (x as u8) << 1 | b as u8;
   385ŌåÆ    if byte > 0x40 {
   386ŌåÆ        Some(byte)
   387ŌåÆ    } else {
   388ŌåÆ        None
   389ŌåÆ    }
   390ŌåÆ}

   396ŌåÆ/// Encode a ModR/M byte.
   397ŌåÆ///
   398ŌåÆ/// - `mod_bits`: 2-bit mod field (0=mem, 1=mem+disp8, 2=mem+disp32, 3=reg)
   399ŌåÆ/// - `reg`: 3-bit reg field (register or opcode extension)
   400ŌåÆ/// - `rm`: 3-bit r/m field (register or memory operand)
   401ŌåÆfn modrm(mod_bits: u8, reg: u8, rm: u8) -> u8 {
   402ŌåÆ    (mod_bits & 3) << 6 | (reg & 7) << 3 | (rm & 7)
   403ŌåÆ}

   405ŌåÆ/// Encode a SIB byte.
   406ŌåÆ///

   409ŌåÆ/// - `base`: 3-bit base register
   410ŌåÆfn sib(scale: u8, index: u8, base: u8) -> u8 {
   411ŌåÆ    (scale & 3) << 6 | (index & 7) << 3 | (base & 7)
   412ŌåÆ}

   418ŌåÆ/// Emit a REX.W prefix plus opcode, then a ModR/M byte for reg-reg operations.
   419ŌåÆ///
   420ŌåÆ/// This is the common pattern for 64-bit ALU instructions: REX.W + opcode + ModR/M(mod=3, reg, rm).
   421ŌåÆfn emit_rexw_reg_reg(code: &mut Vec<u8>, opcode: u8, reg: Gpr, rm: Gpr) {
   422ŌåÆ    let r = reg.needs_rex();
   423ŌåÆ    let b = rm.needs_rex();
   424ŌåÆ    if let Some(rex) = rex_prefix(true, r, false, b) {
   425ŌåÆ        code.push(rex);
   426ŌåÆ    } else {
   427ŌåÆ        // REX.W is still needed for 64-bit operations even if r and b are 0
   428ŌåÆ        code.push(0x48);
   429ŌåÆ    }
   430ŌåÆ    code.push(opcode);
   431ŌåÆ    code.push(modrm(3, reg.encoding() & 7, rm.encoding() & 7));
   432ŌåÆ}

   481ŌåÆ/// Encode MOV r64, [r64+offset] (REX.W + 8B /r + displacement)
   482ŌåÆ///

   484ŌåÆ/// - RSP/R12 as base: SIB byte required
   485ŌåÆ/// - RBP/R13 as base: mod=1 with disp8=0 even for zero offset
   486ŌåÆ/// - Offset fits in i8: disp8; otherwise: disp32
   487ŌåÆpub fn encode_mov_reg_mem(dst: Gpr, base: Gpr, offset: i32) -> Vec<u8> {
   488ŌåÆ    let mut code = Vec::with_capacity(8);
   489ŌåÆ    let r = dst.needs_rex();
   490ŌåÆ    let b = base.needs_rex();

   493ŌåÆ        code.push(rex);

   498ŌåÆ

   500ŌåÆ    let needs_disp8_for_zero = base == Gpr::Rbp || base == Gpr::R13;

   514ŌåÆ        } else {
   515ŌåÆ            code.push(modrm(2, dst.encoding() & 7, 4));
   516ŌåÆ            code.push(sib(0, 4, base.encoding() & 7));
   517ŌåÆ            code.extend_from_slice(&offset.to_le_bytes());
   518ŌåÆ        }
   519ŌåÆ    } else if (-128..=127).contains(&offset) {
   520ŌåÆ        // mod=01, disp8
   521ŌåÆ        code.push(modrm(1, dst.encoding() & 7, base.encoding() & 7));
   522ŌåÆ        code.push(offset as u8);
   523ŌåÆ    } else {

   525ŌåÆ        code.push(modrm(2, dst.encoding() & 7, base.encoding() & 7));

   530ŌåÆ}

  1739ŌåÆ/// Handles register-register (mod=3) only for simplicity.

  1741ŌåÆ    if pos >= bytes.len() {

  1743ŌåÆ    }

  1746ŌåÆ    let mod_bits = (modrm >> 6) & 3;
  1747ŌåÆ    let reg = ((modrm >> 3) & 7) | (if rex_r { 8 } else { 0 });
  1748ŌåÆ    let rm = (modrm & 7) | (if rex_b { 8 } else { 0 });
  1749ŌåÆ
  1750ŌåÆ    if mod_bits == 3 {
  1751ŌåÆ        // Register-register
  1752ŌåÆ        (reg, rm, new_pos)
  1753ŌåÆ    } else {
  1754ŌåÆ        // For memory operands, just return the rm as-is (simplified)
  1755ŌåÆ        (reg, rm, new_pos)
  1756ŌåÆ    }
  1757ŌåÆ}

  1763ŌåÆ/// Build a minimal ELF64 binary for x86_64 from raw code bytes.
  1764ŌåÆ///
  1765ŌåÆ/// Produces a static executable with a single LOAD segment containing the
  1766ŌåÆ/// `.text` section. Entry point is at `base_addr` + header offset.
  1767ŌåÆfn build_minimal_x86_64_elf(code: &[u8], base_addr: u64) -> Vec<u8> {
  1768ŌåÆ    const PAGE_SIZE: u64 = 0x1000; // 4 KB
  1769ŌåÆ
  1770ŌåÆ    let elf_header_size: u64 = 64;
  1771ŌåÆ    let phdr_size: u64 = 56;
  1772ŌåÆ    let phdr_end = elf_header_size + phdr_size;
  1773ŌåÆ    // Page-align the text segment start for mmap compatibility (required by QEMU).
  1774ŌåÆ    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1775ŌåÆ    let text_size = code.len() as u64;
  1776ŌåÆ    let entry_point = base_addr + text_offset;
  1777ŌåÆ
  1778ŌåÆ    let mut elf = Vec::with_capacity(text_offset as usize + code.len());
  1779ŌåÆ
  1780ŌåÆ    // --- e_ident ---
  1781ŌåÆ    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1782ŌåÆ    elf.push(2); // ELFCLASS64
  1783ŌåÆ    elf.push(1); // ELFDATA2LSB
  1784ŌåÆ    elf.push(1); // EV_CURRENT
  1785ŌåÆ    elf.push(3); // ELFOSABI_LINUX
  1786ŌåÆ    elf.push(0); // padding
  1787ŌåÆ    elf.extend_from_slice(&[0u8; 7]); // padding
  1788ŌåÆ
  1789ŌåÆ    // --- ELF header fields ---
  1790ŌåÆ    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1791ŌåÆ    elf.extend_from_slice(&62u16.to_le_bytes()); // e_machine = EM_X86_64
  1792ŌåÆ    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1793ŌåÆ    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1794ŌåÆ    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1795ŌåÆ    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1796ŌåÆ    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1797ŌåÆ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1798ŌåÆ    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1799ŌåÆ    elf.extend_from_slice(&1u16.to_le_bytes()); // e_phnum
  1800ŌåÆ    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1801ŌåÆ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1802ŌåÆ    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1803ŌåÆ
  1804ŌåÆ    // --- Program Header (single LOAD segment: PF_R | PF_X) ---
  1805ŌåÆ    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1806ŌåÆ    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1807ŌåÆ    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1808ŌåÆ    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1809ŌåÆ    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1810ŌåÆ    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1811ŌåÆ    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1812ŌåÆ    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1813ŌåÆ
  1814ŌåÆ    // --- Padding + Code section ---
  1815ŌåÆ    // Pad to page-aligned text_offset
  1816ŌåÆ    while (elf.len() as u64) < text_offset {
  1817ŌåÆ        elf.push(0);
  1818ŌåÆ    }
  1819ŌåÆ    elf.extend_from_slice(code);
  1820ŌåÆ
  1821ŌåÆ    elf
  1822ŌåÆ}

  1848ŌåÆ/// Compute the stack frame size for an IR function on x86_64.
  1849ŌåÆ///
  1850ŌåÆ/// Sums `Alloc` instruction sizes, adds 8 bytes for the RBP save,
  1851ŌåÆ/// and rounds up to 16-byte alignment.
  1852ŌåÆfn x86_64_compute_frame_size(func: &IRFunction) -> usize {
  1853ŌåÆ    let mut total: usize = 8; // Saved RBP
  1854ŌåÆ    for block in &func.blocks {
  1855ŌåÆ        for instr in &block.instructions {
  1856ŌåÆ            if let crate::ir::IRInstr::Alloc { size, .. } = instr {
  1857ŌåÆ                let aligned = (*size as usize).div_ceil(16) * 16;
  1858ŌåÆ                total += aligned;
  1859ŌåÆ            }
  1860ŌåÆ        }
  1861ŌåÆ    }
  1862ŌåÆ    // Round up to 16-byte alignment
  1863ŌåÆ    (total + 15) & !15
  1864ŌåÆ}
