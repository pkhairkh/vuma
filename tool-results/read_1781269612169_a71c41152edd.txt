   273‚Üí
   274‚Üí/// Target-specific information needed during code generation.
   275‚Üí///
   276‚Üí/// This trait provides a data-driven interface for query target properties.
   277‚Üí/// It must be implementable for ALL supported ISAs, including fundamentally
   278‚Üí/// different architectures like Wasm (stack machine, no registers) and
   279‚Üí/// MIPS (branch delay slots).
   280‚Üí///
   281‚Üí/// # Object Safety
   282‚Üí///
   283‚Üí/// This trait is object-safe: all methods take `&self` and return only
   284‚Üí/// owned types or references with `'static` lifetime.
   285‚Üípub trait TargetInfo: Send + Sync + 'static {
   286‚Üí    // === Identity ===
   287‚Üí
   288‚Üí    /// ISA name in lowercase (e.g., "aarch64", "riscv64", "wasm32").
   289‚Üí    fn isa_name(&self) -> &'static str;
   290‚Üí
   291‚Üí    /// LLVM-style target triple (e.g., "aarch64-unknown-linux-gnu").
   292‚Üí    fn target_triple(&self) -> &'static str;
   293‚Üí
   294‚Üí    /// ELF `e_machine` value.  Returns 0 for non-ELF targets (Wasm).
   295‚Üí    fn elf_machine_type(&self) -> u16;
   296‚Üí
   297‚Üí    /// Default base address for the .text section.
   298‚Üí    fn default_base_address(&self) -> u64;
   299‚Üí
   300‚Üí    // === Data model ===
   301‚Üí
   302‚Üí    /// Pointer width in bytes (4 for 32-bit, 8 for 64-bit).
   303‚Üí    fn pointer_width(&self) -> usize;
   304‚Üí
   305‚Üí    /// Size in bytes of `ty` on this target.
   306‚Üí    fn size_of(&self, ty: &IRType) -> usize;
   307‚Üí
   308‚Üí    /// Natural alignment in bytes of `ty` on this target.
   309‚Üí    fn alignment_of(&self, ty: &IRType) -> usize;
   310‚Üí
   311‚Üí    /// Byte order of this target.
   312‚Üí    fn endianness(&self) -> Endianness;
   313‚Üí
   314‚Üí    // === Register architecture ===
   315‚Üí
   316‚Üí    /// Whether this target has registers at all.  `false` for Wasm (stack machine).
   317‚Üí    fn has_registers(&self) -> bool;
   318‚Üí
   319‚Üí    /// Number of general-purpose registers.  0 for Wasm.
   320‚Üí    fn num_gp_regs(&self) -> usize;
   321‚Üí
   322‚Üí    /// Number of SIMD/FP registers.  0 for Wasm.
   323‚Üí    fn num_simd_fp_regs(&self) -> usize;
   324‚Üí
   325‚Üí    /// Whether the ISA has a hardwired-zero register (RISC-V x0, LoongArch r0).
   326‚Üí    fn has_hardwired_zero(&self) -> bool;
   327‚Üí
   328‚Üí    /// Whether the ISA uses a link register (ARM, RISC-V, MIPS, PPC) rather than
   329‚Üí    /// pushing the return address on the stack (x86_64).
   330‚Üí    fn has_link_register(&self) -> bool;
   331‚Üí
   332‚Üí    /// Whether branches have delay slots (MIPS only).
   333‚Üí    fn has_branch_delay_slots(&self) -> bool;
   334‚Üí
   335‚Üí    /// Whether this ISA uses a TOC (Table of Contents) pointer (PPC64 r2).
   336‚Üí    fn has_toc_pointer(&self) -> bool;
   337‚Üí
   338‚Üí    /// Whether this ISA has dedicated condition register fields (PPC64 CR0-CR7).
   339‚Üí    fn has_condition_registers(&self) -> bool;
   340‚Üí
   341‚Üí    // === Calling convention ===
   342‚Üí
   343‚Üí    /// Name of the calling convention (e.g., "aapcs64", "lp64d", "systemv").
   344‚Üí    fn calling_convention_name(&self) -> &'static str;
   345‚Üí
   346‚Üí    /// Number of integer argument registers.
   347‚Üí    fn num_int_arg_regs(&self) -> usize;
   348‚Üí
   349‚Üí    /// Number of FP/SIMD argument registers.
   350‚Üí    fn num_fp_arg_regs(&self) -> usize;
   351‚Üí
   352‚Üí    /// Required stack alignment in bytes.
   353‚Üí    fn stack_alignment(&self) -> usize;
   354‚Üí
   355‚Üí    // === Instruction encoding ===
   356‚Üí
   357‚Üí    /// Alignment requirement for instructions in bytes (4 for fixed-width RISCs,
   358‚Üí    /// 1 for x86_64 and Wasm).
   359‚Üí    fn instruction_alignment(&self) -> usize;
   360‚Üí
   361‚Üí    /// Minimum and maximum instruction width in bytes.
   362‚Üí    /// - Fixed-width 32-bit ISAs: (4, 4)
   363‚Üí    /// - x86_64: (1, 15)
   364‚Üí    /// - RISC-V with RVC: (2, 4)
   365‚Üí    /// - Wasm: (1, ~) but typically (1, 15)
   366‚Üí    fn instruction_width_range(&self) -> (usize, usize);
   367‚Üí
   368‚Üí    // === Output format ===
   369‚Üí
   370‚Üí    /// Binary format produced by this backend.
   371‚Üí    fn output_format(&self) -> OutputFormat;
   372‚Üí}

   378‚Üí/// A code generation backend for a specific target architecture.
   379‚Üí///
   380‚Üí/// Each supported ISA implements this trait, providing register allocation,
   381‚Üí/// instruction encoding, program emission, and disassembly.
   382‚Üí///
   383‚Üí/// # Object Safety
   384‚Üí///
   385‚Üí/// This trait is object-safe.
   386‚Üípub trait Backend: Send + Sync + 'static {
   387‚Üí    /// Returns a reference to this backend's target info.
   388‚Üí    fn target_info(&self) -> &dyn TargetInfo;
   389‚Üí
   390‚Üí    /// Allocate physical registers for an IR function.
   391‚Üí    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError>;
   392‚Üí
   393‚Üí    /// Encode a single allocated function into machine code bytes.
   394‚Üí    fn encode_function(&self, func: &AllocatedFunction) -> Result<Vec<u8>, BackendError>;
   395‚Üí
   396‚Üí    /// Encode an entire allocated program into its final binary form
   397‚Üí    /// (ELF, .wasm, raw binary, etc.).
   398‚Üí    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError>;
   399‚Üí
   400‚Üí    /// Returns the bytes for a minimal return stub (e.g., `RET` on ARM64,
   401‚Üí    /// `mov eax, 0; ret` on x86_64, `end` on Wasm).
   402‚Üí    fn return_stub(&self) -> Vec<u8>;
   403‚Üí
   404‚Üí    /// Returns a trampoline that jumps to `entry_addr`.
   405‚Üí    fn trampoline(&self, entry_addr: u64) -> Vec<u8>;
   406‚Üí
   407‚Üí    /// Disassemble `bytes` starting at virtual address `addr`.
   408‚Üí    fn disassemble(&self, bytes: &[u8], addr: u64) -> Vec<String>;
   409‚Üí
   410‚Üí    /// Human-readable name of this backend.
   411‚Üí    fn name(&self) -> &'static str;
   412‚Üí}

  1118‚Üí///

  1126‚Üí    let rm = (word >> 16) & 0x1F;

  1135‚Üí    // RET: d65f03c0
  1136‚Üí    if word == 0xD65F03C0 {

  1140‚Üí    let _top8 = word >> 24;
  1141‚Üí    let _top10 = word >> 22;

  1144‚Üí    if (word >> 23) & 0x1FF == 0b1_0010_0010 {
  1145‚Üí        // ADD Xd, Xn, #imm12

  1147‚Üí    }

  1149‚Üí        // SUB Xd, Xn, #imm12
  1150‚Üí        return format!("sub x{}, x{}, #{}", rd, rn, imm12);
  1151‚Üí    }
  1152‚Üí

  1154‚Üí    if (word >> 24) & 0xFF == 0b1000_1011 {
  1155‚Üí        // ADD Xd, Xn, Xm
  1156‚Üí        return format!("add x{}, x{}, x{}", rd, rn, rm);
  1157‚Üí    }
  1158‚Üí

  1160‚Üí    if (word >> 24) & 0xFF == 0b1101_0110 {
  1161‚Üí        // SUB Xd, Xn, Xm (bit 30 set = sub)
  1162‚Üí        return format!("sub x{}, x{}, x{}", rd, rn, rm);
  1163‚Üí    }
  1164‚Üí
  1165‚Üí    // --- AND (shifted register): 1_00_0101_0_00_xxx
  1166‚Üí    if (word >> 24) & 0xFE == 0b1000_1010 {
  1167‚Üí        // Check bit 21-15: opcode[31:21] = 10001010_000
  1168‚Üí        if (word >> 21) & 0x7FF == 0b10001010000 {
  1169‚Üí            return format!("and x{}, x{}, x{}", rd, rn, rm);
  1170‚Üí        }
  1171‚Üí    }
  1172‚Üí
  1173‚Üí    // --- ORR (shifted register): 1_01_0101_0_00_xxx
  1174‚Üí    if (word >> 21) & 0x7FF == 0b10101010000 {
  1175‚Üí        return format!("orr x{}, x{}, x{}", rd, rn, rm);
  1176‚Üí    }
  1177‚Üí
  1178‚Üí    // --- EOR (shifted register): 1_10_0101_0_00_xxx
  1179‚Üí    if (word >> 21) & 0x7FF == 0b11001010000 {
  1180‚Üí        return format!("eor x{}, x{}, x{}", rd, rn, rm);
  1181‚Üí    }
  1182‚Üí
  1183‚Üí    // --- MOV (register): alias for ORR Xd, XZR, Xm
  1184‚Üí    // ORR Xd, XZR, Xm: 10101010_000_rm_000000_xzr_rd

  1186‚Üí    if (word >> 21) & 0x7FF == 0b10101010000 && rn == 31 {

  1188‚Üí    }
  1189‚Üí
  1190‚Üí    // --- MUL: MADD Xd, Xn, Xm, XZR

  1192‚Üí    if (word >> 21) & 0x7FF == 0b10011011000 && ((word >> 10) & 0x1F) == 0b01111 {

  1194‚Üí    }
  1195‚Üí

  1200‚Üí

  1203‚Üí        return format!("udiv x{}, x{}, x{}", rd, rn, rm);
  1204‚Üí    }
  1205‚Üí

  1207‚Üí    // 11100001_00_xxx_xxx_xxx_xxx_xxx_11111_xxx
  1208‚Üí    if (word >> 23) & 0x1FF == 0b1_1110_0010 && rd == 31 {
  1209‚Üí        return format!("cmp x{}, #{}", rn, imm12);
  1210‚Üí    }
  1211‚Üí
  1212‚Üí    // --- CMP (register): SUBS XZR, Xn, Xm

  1214‚Üí        return format!("cmp x{}, x{}", rn, rm);
  1215‚Üí    }
  1216‚Üí

  1218‚Üí    if (word >> 24) & 0xFF == 0x54 {

  1221‚Üí            1 => "ne",
  1222‚Üí            2 => "cs",
  1223‚Üí            3 => "cc",
  1224‚Üí            4 => "mi",
  1225‚Üí            5 => "pl",
  1226‚Üí            6 => "vs",
  1227‚Üí            7 => "vc",
  1228‚Üí            8 => "hi",
  1229‚Üí            9 => "ls",

  1236‚Üí        };

  1248‚Üí

  1250‚Üí    if (word >> 26) & 0x3F == 0b100101 {
  1251‚Üí        let imm26 = word & 0x3FFFFFF;

  1367‚Üí    elf.push(2); // ELFCLASS64
  1368‚Üí    elf.push(1); // ELFDATA2LSB
  1369‚Üí    elf.push(1); // EV_CURRENT
  1370‚Üí    elf.push(3); // ELFOSABI_LINUX
  1371‚Üí    elf.push(0); // padding
  1372‚Üí    elf.extend_from_slice(&[0u8; 7]); // padding
  1373‚Üí
  1374‚Üí    // --- ELF header fields ---
  1375‚Üí    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1376‚Üí    elf.extend_from_slice(&183u16.to_le_bytes()); // e_machine = EM_AARCH64
  1377‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1378‚Üí    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1379‚Üí    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1380‚Üí    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1381‚Üí    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1382‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1383‚Üí    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1384‚Üí    elf.extend_from_slice(&1u16.to_le_bytes()); // e_phnum
  1385‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1386‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1387‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1388‚Üí
  1389‚Üí    // --- Program Header (single LOAD segment: PF_R | PF_X) ---
  1390‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1391‚Üí    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1392‚Üí    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1393‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1394‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1395‚Üí    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1396‚Üí    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1397‚Üí    elf.extend_from_slice(&16u64.to_le_bytes()); // p_align
  1398‚Üí
  1399‚Üí    // --- Code section ---
  1400‚Üí    elf.extend_from_slice(code);
  1401‚Üí
  1402‚Üí    elf
  1403‚Üí}

  1410‚Üí    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError> {
  1411‚Üí        // Use the existing Emitter to emit the function, which internally
  1412‚Üí        // performs register allocation and instruction encoding.
  1413‚Üí        let mut emitter = crate::emit::Emitter::new();
  1414‚Üí        let code = emitter
  1415‚Üí            .emit_function(func)
  1416‚Üí            .map_err(|e| BackendError::RegisterAllocFailed {
  1417‚Üí                isa: "aarch64",
  1418‚Üí                reason: e.to_string(),
  1419‚Üí            })?;
  1420‚Üí
  1421‚Üí        let func_name = func.name.clone();
  1422‚Üí        let frame_size = aarch64_compute_frame_size(func);
  1423‚Üí
  1424‚Üí        // Collect relocations from the emitter and adjust offsets from
  1425‚Üí        // absolute text-section offsets to function-relative byte offsets.
  1426‚Üí        let relocations: Vec<RelocationEntry> = emitter
  1427‚Üí            .relocations()
  1428‚Üí            .iter()
  1429‚Üí            .map(|r| RelocationEntry {
  1430‚Üí                offset: r.offset,
  1431‚Üí                symbol: r.symbol.clone(),
  1432‚Üí                reloc_type: r.reloc_type.clone(),
  1433‚Üí            })
  1434‚Üí            .collect();
  1435‚Üí
  1436‚Üí        // Convert each 32-bit ARM64 instruction word into an AllocatedInstruction
  1437‚Üí        // with its little-endian encoded bytes.
  1438‚Üí        let instructions: Vec<AllocatedInstruction> = code
  1439‚Üí            .iter()
  1440‚Üí            .enumerate()
  1441‚Üí            .map(|(i, &word)| AllocatedInstruction {
  1442‚Üí                opcode: format!("arm64_{}", i),
  1443‚Üí                reads: vec![],
  1444‚Üí                writes: vec![],
  1445‚Üí                encoded: word.to_le_bytes().to_vec(),
  1446‚Üí            })
  1447‚Üí            .collect();
  1448‚Üí
  1449‚Üí        let code_size = instructions.len() * 4;
  1450‚Üí
  1451‚Üí        Ok(AllocatedFunction {
  1452‚Üí            name: func_name,
  1453‚Üí            blocks: vec![AllocatedBlock {
  1454‚Üí                label: "entry".to_string(),
  1455‚Üí                instructions,
  1456‚Üí                code_offset: 0,
  1457‚Üí            }],
  1458‚Üí            frame_size,
  1459‚Üí            callee_saved: vec![],
  1460‚Üí            spill_slots: 0,
  1461‚Üí            code_size,
  1462‚Üí            relocations,
  1463‚Üí        })
  1464‚Üí    }

  1466‚Üí    fn encode_function(&self, func: &AllocatedFunction) -> Result<Vec<u8>, BackendError> {
  1467‚Üí        let mut bytes = Vec::new();
  1468‚Üí        for block in &func.blocks {
  1469‚Üí            for instr in &block.instructions {
  1470‚Üí                bytes.extend_from_slice(&instr.encoded);
  1471‚Üí            }
  1472‚Üí        }
  1473‚Üí        Ok(bytes)
  1474‚Üí    }
  1475‚Üí
  1476‚Üí    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError> {
  1477‚Üí        // ‚îÄ‚îÄ Phase 1: Concatenate all function code, recording byte offsets ‚îÄ‚îÄ
  1478‚Üí        let mut all_code: Vec<u8> = Vec::new();
  1479‚Üí        let mut func_offsets: Vec<(String, usize)> = Vec::new(); // (name, start_offset)
  1480‚Üí        // Collect all relocations from all functions, adjusting their offsets
  1481‚Üí        // to be relative to the combined code buffer.
  1482‚Üí        let mut all_relocations: Vec<(usize, String, String)> = Vec::new(); // (abs_offset, symbol, reloc_type)
  1483‚Üí
  1484‚Üí        for func in &program.functions {
  1485‚Üí            let func_start = all_code.len();
  1486‚Üí            func_offsets.push((func.name.clone(), func_start));
  1487‚Üí
  1488‚Üí            for block in &func.blocks {
  1489‚Üí                for instr in &block.instructions {
  1490‚Üí                    all_code.extend_from_slice(&instr.encoded);
  1491‚Üí                }
  1492‚Üí            }
  1493‚Üí
  1494‚Üí            // Collect relocations, adjusting offsets to be absolute in the combined code
  1495‚Üí            for reloc in &func.relocations {
  1496‚Üí                all_relocations.push((
  1497‚Üí                    func_start + reloc.offset as usize,
  1498‚Üí                    reloc.symbol.clone(),
  1499‚Üí                    reloc.reloc_type.clone(),
  1500‚Üí                ));
  1501‚Üí            }
  1502‚Üí        }
  1503‚Üí
  1504‚Üí        // ‚îÄ‚îÄ Phase 2: Patch relocations for inter-function calls ‚îÄ‚îÄ
  1505‚Üí        // Build a map from function name ‚Üí byte offset in the combined code
  1506‚Üí        let func_offset_map: HashMap<String, usize> = func_offsets.into_iter().collect();
  1507‚Üí
  1508‚Üí        for (abs_offset, symbol, reloc_type) in &all_relocations {
  1509‚Üí            if reloc_type == "R_AARCH64_CALL26" {
  1510‚Üí                if let Some(&target_offset) = func_offset_map.get(symbol) {
  1511‚Üí                    // R_AARCH64_CALL26: patch = (S + A - P) >> 2, encoded in bits [25:0]
  1512‚Üí                    // S = target offset in code buffer, A = addend (0), P = relocation position
  1513‚Üí                    // The offset is in units of 4 bytes (instructions).
  1514‚Üí                    let text_offset: u64 = 64 + 56; // ELF header + PHDR
  1515‚Üí                    let s = text_offset + target_offset as u64;
  1516‚Üí                    let p = text_offset + *abs_offset as u64;
  1517‚Üí                    let offset_words = ((s as i64) - (p as i64)) >> 2;
  1518‚Üí                    // Read the existing 4-byte instruction, clear bits [25:0],
  1519‚Üí                    // and insert the new 26-bit offset.
  1520‚Üí                    let instr_start = *abs_offset;
  1521‚Üí                    let mut word = u32::from_le_bytes([
  1522‚Üí                        all_code[instr_start],
  1523‚Üí                        all_code[instr_start + 1],
  1524‚Üí                        all_code[instr_start + 2],
  1525‚Üí                        all_code[instr_start + 3],
  1526‚Üí                    ]);
  1527‚Üí                    word = (word & !0x03FF_FFFF) | ((offset_words as u32) & 0x03FF_FFFF);
  1528‚Üí                    all_code[instr_start..instr_start + 4].copy_from_slice(&word.to_le_bytes());
  1529‚Üí                }
  1530‚Üí                // If the symbol is not found (e.g., external like __vuma_free),
  1531‚Üí                // leave the placeholder ‚Äî would need a real linker for those.
  1532‚Üí            }
  1533‚Üí        }
  1534‚Üí
  1535‚Üí        // ‚îÄ‚îÄ Phase 3: Replace RET in the entry function with exit syscall ‚îÄ‚îÄ
  1536‚Üí        // The first function is the ELF entry point, which is jumped to by the
  1537‚Üí        // kernel ‚Äî there is no return address on the stack, so `RET` would
  1538‚Üí        // jump to garbage.  Replace the final `RET` with:
  1539‚Üí        //   MOV X0, X0       ; nop (keep return value in X0)
  1540‚Üí        //   MOV X8, #93      ; sys_exit on AArch64
  1541‚Üí        //   SVC #0           ; supervisor call
  1542‚Üí        if !program.functions.is_empty() {
  1543‚Üí            let first_func = &program.functions[0];
  1544‚Üí            // Find the last RET instruction (0xD65F03C0, little-endian: C0 03 5F D6)
  1545‚Üí            // in the first function's code by walking all instructions.
  1546‚Üí            let ret_bytes: [u8; 4] = [0xC0, 0x03, 0x5F, 0xD6];
  1547‚Üí            let mut instr_offset = 0usize;
  1548‚Üí            let mut found_offset: Option<usize> = None;
  1549‚Üí            for block in &first_func.blocks {
  1550‚Üí                for instr in &block.instructions {
  1551‚Üí                    if instr.encoded.len() == 4 && instr.encoded[..] == ret_bytes[..] {
  1552‚Üí                        found_offset = Some(instr_offset);
  1553‚Üí                    }
  1554‚Üí                    instr_offset += instr.encoded.len();
  1555‚Üí                }
  1556‚Üí            }
  1557‚Üí            if let Some(offset) = found_offset {
  1558‚Üí                // Replace the 4-byte RET with the exit syscall sequence:
  1559‚Üí                //   MOV X8, #93      ‚Üí 0xD2800E88 (little-endian: 88 0E 80 D2)
  1560‚Üí                //   SVC #0           ‚Üí 0xD4000001 (little-endian: 01 00 00 D4)
  1561‚Üí                // That's 8 bytes replacing 4 bytes ‚Äî we need to splice.
  1562‚Üí                let exit_syscall: [u8; 8] = [
  1563‚Üí                    0x88, 0x0E, 0x80, 0xD2, // MOV X8, #93
  1564‚Üí                    0x01, 0x00, 0x00, 0xD4, // SVC #0
  1565‚Üí                ];
  1566‚Üí                let mut new_code = Vec::with_capacity(all_code.len() + 4);
  1567‚Üí                new_code.extend_from_slice(&all_code[..offset]);
  1568‚Üí                new_code.extend_from_slice(&exit_syscall);
  1569‚Üí                new_code.extend_from_slice(&all_code[offset + 4..]);
  1570‚Üí                all_code = new_code;
  1571‚Üí            }
  1572‚Üí        }
  1573‚Üí
  1574‚Üí        // Wrap in a minimal ELF64 binary for AArch64.
  1575‚Üí        Ok(build_minimal_aarch64_elf(&all_code, 0x400000))
  1576‚Üí    }

  1595‚Üí    fn disassemble(&self, bytes: &[u8], addr: u64) -> Vec<String> {
  1596‚Üí        // Mnemonic decoder for AArch64 (4-byte fixed-width instructions).
  1597‚Üí        let mut lines = Vec::new();
  1598‚Üí        let mut offset = 0usize;
  1599‚Üí        let mut pc = addr;
  1600‚Üí        while offset + 4 <= bytes.len() {
  1601‚Üí            let word = u32::from_le_bytes([
  1602‚Üí                bytes[offset],
  1603‚Üí                bytes[offset + 1],
  1604‚Üí                bytes[offset + 2],
  1605‚Üí                bytes[offset + 3],
  1606‚Üí            ]);
  1607‚Üí            let mnemonic = if let Some(instr) = crate::arm64::Instruction::decode(word) {
  1608‚Üí                format!("{}", instr)
  1609‚Üí            } else {
  1610‚Üí                decode_aarch64(word)
  1611‚Üí            };
  1612‚Üí            lines.push(format!("{:#010x}:  {:08x}  {}", pc, word, mnemonic));
  1613‚Üí            offset += 4;
  1614‚Üí            pc += 4;
  1615‚Üí        }
  1616‚Üí        if offset < bytes.len() {
  1617‚Üí            let remaining = &bytes[offset..];
  1618‚Üí            lines.push(format!("{:#010x}:  {:02x?}", pc, remaining));
  1619‚Üí        }
  1620‚Üí        lines
  1621‚Üí    }
