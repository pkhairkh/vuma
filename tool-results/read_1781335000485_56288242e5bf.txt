    58→/// SHA-256 logical functions (FIPS 180-4 Section 4.1.2).
    59→fn ch(x: u32, y: u32, z: u32) -> u32 {
    60→    (x & y) ^ (!x & z)
    61→}
    62→
    63→fn maj(a: u32, b: u32, c: u32) -> u32 {
    64→    (a & b) ^ (a & c) ^ (b & c)
    65→}
    66→
    67→fn big_sigma0(x: u32) -> u32 {
    68→    x.rotate_right(2) ^ x.rotate_right(13) ^ x.rotate_right(22)
    69→}
    70→
    71→fn big_sigma1(x: u32) -> u32 {
    72→    x.rotate_right(6) ^ x.rotate_right(11) ^ x.rotate_right(25)
    73→}
    74→
    75→fn small_sigma0(x: u32) -> u32 {
    76→    x.rotate_right(7) ^ x.rotate_right(18) ^ (x >> 3)
    77→}
    78→
    79→fn small_sigma1(x: u32) -> u32 {
    80→    x.rotate_right(17) ^ x.rotate_right(19) ^ (x >> 10)
    81→}

    83→/// SHA-256 compression function: process one 512-bit (64-byte) block.
    84→fn sha256_transform(state: &mut [u32; 8], block: &[u8; 64]) {
    85→    // Build message schedule W[0..63].
    86→    let mut w = [0u32; 64];
    87→    for i in 0..16 {
    88→        w[i] = u32::from_be_bytes([
    89→            block[i * 4],
    90→            block[i * 4 + 1],
    91→            block[i * 4 + 2],
    92→            block[i * 4 + 3],
    93→        ]);
    94→    }
    95→    for i in 16..64 {
    96→        w[i] = small_sigma1(w[i - 2])
    97→            .wrapping_add(w[i - 7])
    98→            .wrapping_add(small_sigma0(w[i - 15]))
    99→            .wrapping_add(w[i - 16]);
   100→    }
   101→
   102→    // Initialize working variables.
   103→    let [mut a, mut b, mut c, mut d, mut e, mut f, mut g, mut h] = *state;
   104→
   105→    // 64-round compression.
   106→    for i in 0..64 {
   107→        let t1 = h
   108→            .wrapping_add(big_sigma1(e))
   109→            .wrapping_add(ch(e, f, g))
   110→            .wrapping_add(K[i])
   111→            .wrapping_add(w[i]);
   112→        let t2 = big_sigma0(a).wrapping_add(maj(a, b, c));
   113→
   114→        h = g;
   115→        g = f;
   116→        f = e;
   117→        e = d.wrapping_add(t1);
   118→        d = c;
   119→        c = b;
   120→        b = a;
   121→        a = t1.wrapping_add(t2);
   122→    }
   123→
   124→    // Add compressed chunk to hash state.
   125→    state[0] = state[0].wrapping_add(a);
   126→    state[1] = state[1].wrapping_add(b);
   127→    state[2] = state[2].wrapping_add(c);
   128→    state[3] = state[3].wrapping_add(d);
   129→    state[4] = state[4].wrapping_add(e);
   130→    state[5] = state[5].wrapping_add(f);
   131→    state[6] = state[6].wrapping_add(g);
   132→    state[7] = state[7].wrapping_add(h);
   133→}

   135→/// Compute SHA-256 of an arbitrary byte message, returning 32-byte digest.
   136→fn sha256(message: &[u8]) -> [u8; 32] {
   137→    let mut state = H_INIT;
   138→
   139→    // Padding: append 0x80, then zeros, then 64-bit big-endian bit length.
   140→    // Total padded length must be a multiple of 64 bytes.
   141→    let msg_len = message.len();
   142→    let bit_len = (msg_len as u64) * 8;
   143→
   144→    // Determine padded length: next multiple of 64 that can fit the padding.
   145→    // We need at least 1 byte (0x80) + 8 bytes (length) = 9 bytes after message.
   146→    let padded_len = if msg_len % 64 < 56 {
   147→        (msg_len / 64 + 1) * 64
   148→    } else {
   149→        (msg_len / 64 + 2) * 64
   150→    };
   151→
   152→    let mut padded = vec![0u8; padded_len];
   153→    padded[..msg_len].copy_from_slice(message);
   154→    padded[msg_len] = 0x80;
   155→    let len_bytes = bit_len.to_be_bytes();
   156→    padded[padded_len - 8..].copy_from_slice(&len_bytes);
   157→
   158→    // Process each 64-byte block.
   159→    for chunk_start in (0..padded_len).step_by(64) {
   160→        let mut block = [0u8; 64];
   161→        block.copy_from_slice(&padded[chunk_start..chunk_start + 64]);
   162→        sha256_transform(&mut state, &block);
   163→    }
   164→
   165→    // Produce digest (8 x u32 in big-endian).
   166→    let mut digest = [0u8; 32];
   167→    for i in 0..8 {
   168→        digest[i * 4..i * 4 + 4].copy_from_slice(&state[i].to_be_bytes());
   169→    }
   170→    digest
   171→}
   172→
   173→/// Compute SHA256d: SHA-256(SHA-256(message)).
   174→/// The inner SHA-256 produces a 32-byte digest, which is then hashed
   175→/// again by the outer SHA-256.
   176→fn sha256d(message: &[u8]) -> [u8; 32] {
   177→    let inner = sha256(message);
   178→    sha256(&inner)
   179→}
   180→
   181→/// Convert a 32-byte digest to a lowercase hex string for comparison.
   182→fn digest_to_hex(digest: &[u8; 32]) -> String {
   183→    digest.iter().map(|b| format!("{:02x}", b)).collect()
   184→}

   330→    // Bitcoin uses SHA256d for block headers. The genesis block header
   331→    // (little-endian serialized) produces a well-known SHA256d hash.

   333→    // Bitcoin genesis block SHA256d (byte-reversed for display):

   625→#[test]
   626→fn test_sha256_message_schedule() {
   627→    // Verify the message schedule expansion (W[0..63]) for a known block.
   628→    // Use the padded "abc" block:
   629→    //   "abc" = 0x61 0x62 0x63, then 0x80, then zeros, then length 24 bits = 0x18.
   630→    let mut block = [0u8; 64];
   631→    block[0] = 0x61; // 'a'
   632→    block[1] = 0x62; // 'b'
   633→    block[2] = 0x63; // 'c'
   634→    block[3] = 0x80; // padding
   635→    // ... zeros ...
   636→    block[63] = 0x18; // 24 bits = 3 bytes * 8
   637→
   638→    // Manually compute W[0..3] (the first 4 words of the message).
   639→    let w0 = u32::from_be_bytes([0x61, 0x62, 0x63, 0x80]);
   640→    let _w1 = 0u32;
   641→    assert_eq!(w0, 0x6162_6380, "W[0] for padded 'abc' block");
   642→
   643→    // Verify W[16] calculation uses the sigma functions.
   644→    let mut w = [0u32; 64];
   645→    for i in 0..16 {
   646→        w[i] = u32::from_be_bytes([
   647→            block[i * 4],
   648→            block[i * 4 + 1],
   649→            block[i * 4 + 2],
   650→            block[i * 4 + 3],
   651→        ]);
   652→    }
   653→    assert_eq!(w[0], 0x6162_6380, "W[0] must be padded 'abc'");
   654→    assert_eq!(w[15], 0x0000_0018, "W[15] must be the bit length");
   655→
   656→    // Compute W[16] through the schedule.
   657→    for i in 16..64 {
   658→        w[i] = small_sigma1(w[i - 2])
   659→            .wrapping_add(w[i - 7])
   660→            .wrapping_add(small_sigma0(w[i - 15]))
   661→            .wrapping_add(w[i - 16]);
   662→    }
   663→
   664→    // W[16] should be non-zero (combines sigma functions with earlier W values).
   665→    assert_ne!(w[16], 0, "W[16] must be non-zero after schedule expansion");
   666→
   667→    // All W values must be valid u32 (implicitly true, but let's verify range).
   668→    for i in 0..64 {
   669→        assert!(
   670→            w[i] <= u32::MAX,
   671→            "W[{}] must be valid u32: {}",
   672→            i,
   673→            w[i]
   674→        );
   675→    }
   676→}
