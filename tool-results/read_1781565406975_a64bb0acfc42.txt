   330‚Üí
   331‚Üí/// Target-specific information needed during code generation.
   332‚Üí///
   333‚Üí/// This trait provides a data-driven interface for query target properties.
   334‚Üí/// It must be implementable for ALL supported ISAs, including fundamentally
   335‚Üí/// different architectures like Wasm (stack machine, no registers) and
   336‚Üí/// MIPS (branch delay slots).
   337‚Üí///
   338‚Üí/// # Object Safety
   339‚Üí///
   340‚Üí/// This trait is object-safe: all methods take `&self` and return only
   341‚Üí/// owned types or references with `'static` lifetime.
   342‚Üípub trait TargetInfo: Send + Sync + 'static {
   343‚Üí    // === Identity ===
   344‚Üí
   345‚Üí    /// ISA name in lowercase (e.g., "aarch64", "riscv64", "wasm32").
   346‚Üí    fn isa_name(&self) -> &'static str;
   347‚Üí
   348‚Üí    /// LLVM-style target triple (e.g., "aarch64-unknown-linux-gnu").
   349‚Üí    fn target_triple(&self) -> &'static str;
   350‚Üí
   351‚Üí    /// ELF `e_machine` value.  Returns 0 for non-ELF targets (Wasm).
   352‚Üí    fn elf_machine_type(&self) -> u16;
   353‚Üí
   354‚Üí    /// Default base address for the .text section.
   355‚Üí    fn default_base_address(&self) -> u64;
   356‚Üí
   357‚Üí    // === Data model ===
   358‚Üí
   359‚Üí    /// Pointer width in bytes (4 for 32-bit, 8 for 64-bit).
   360‚Üí    fn pointer_width(&self) -> usize;
   361‚Üí
   362‚Üí    /// Size in bytes of `ty` on this target.
   363‚Üí    fn size_of(&self, ty: &IRType) -> usize;
   364‚Üí
   365‚Üí    /// Natural alignment in bytes of `ty` on this target.
   366‚Üí    fn alignment_of(&self, ty: &IRType) -> usize;
   367‚Üí
   368‚Üí    /// Byte order of this target.
   369‚Üí    fn endianness(&self) -> Endianness;
   370‚Üí
   371‚Üí    // === Register architecture ===
   372‚Üí
   373‚Üí    /// Whether this target has registers at all.  `false` for Wasm (stack machine).
   374‚Üí    fn has_registers(&self) -> bool;
   375‚Üí
   376‚Üí    /// Number of general-purpose registers.  0 for Wasm.
   377‚Üí    fn num_gp_regs(&self) -> usize;
   378‚Üí
   379‚Üí    /// Number of SIMD/FP registers.  0 for Wasm.
   380‚Üí    fn num_simd_fp_regs(&self) -> usize;
   381‚Üí
   382‚Üí    /// Whether the ISA has a hardwired-zero register (RISC-V x0, LoongArch r0).
   383‚Üí    fn has_hardwired_zero(&self) -> bool;
   384‚Üí
   385‚Üí    /// Whether the ISA uses a link register (ARM, RISC-V, MIPS, PPC) rather than
   386‚Üí    /// pushing the return address on the stack (x86_64).
   387‚Üí    fn has_link_register(&self) -> bool;
   388‚Üí
   389‚Üí    /// Whether branches have delay slots (MIPS only).
   390‚Üí    fn has_branch_delay_slots(&self) -> bool;
   391‚Üí
   392‚Üí    /// Whether this ISA uses a TOC (Table of Contents) pointer (PPC64 r2).
   393‚Üí    fn has_toc_pointer(&self) -> bool;
   394‚Üí
   395‚Üí    /// Whether this ISA has dedicated condition register fields (PPC64 CR0-CR7).
   396‚Üí    fn has_condition_registers(&self) -> bool;
   397‚Üí
   398‚Üí    // === Calling convention ===
   399‚Üí
   400‚Üí    /// Name of the calling convention (e.g., "aapcs64", "lp64d", "systemv").
   401‚Üí    fn calling_convention_name(&self) -> &'static str;
   402‚Üí
   403‚Üí    /// Number of integer argument registers.
   404‚Üí    fn num_int_arg_regs(&self) -> usize;
   405‚Üí
   406‚Üí    /// Number of FP/SIMD argument registers.
   407‚Üí    fn num_fp_arg_regs(&self) -> usize;
   408‚Üí
   409‚Üí    /// Required stack alignment in bytes.
   410‚Üí    fn stack_alignment(&self) -> usize;
   411‚Üí
   412‚Üí    // === Instruction encoding ===
   413‚Üí
   414‚Üí    /// Alignment requirement for instructions in bytes (4 for fixed-width RISCs,
   415‚Üí    /// 1 for x86_64 and Wasm).
   416‚Üí    fn instruction_alignment(&self) -> usize;
   417‚Üí
   418‚Üí    /// Minimum and maximum instruction width in bytes.
   419‚Üí    /// - Fixed-width 32-bit ISAs: (4, 4)
   420‚Üí    /// - x86_64: (1, 15)
   421‚Üí    /// - RISC-V with RVC: (2, 4)
   422‚Üí    /// - Wasm: (1, ~) but typically (1, 15)
   423‚Üí    fn instruction_width_range(&self) -> (usize, usize);
   424‚Üí
   425‚Üí    // === Output format ===
   426‚Üí
   427‚Üí    /// Binary format produced by this backend.
   428‚Üí    fn output_format(&self) -> OutputFormat;
   429‚Üí}

   434‚Üí
   435‚Üí/// A code generation backend for a specific target architecture.
   436‚Üí///
   437‚Üí/// Each supported ISA implements this trait, providing register allocation,
   438‚Üí/// instruction encoding, program emission, and disassembly.
   439‚Üí///
   440‚Üí/// # Object Safety
   441‚Üí///
   442‚Üí/// This trait is object-safe.
   443‚Üípub trait Backend: Send + Sync + 'static {
   444‚Üí    /// Returns a reference to this backend's target info.
   445‚Üí    fn target_info(&self) -> &dyn TargetInfo;
   446‚Üí
   447‚Üí    /// Allocate physical registers for an IR function.
   448‚Üí    fn allocate_registers(&self, func: &IRFunction) -> Result<AllocatedFunction, BackendError>;
   449‚Üí
   450‚Üí    /// Encode a single allocated function into machine code bytes.
   451‚Üí    fn encode_function(&self, func: &AllocatedFunction) -> Result<Vec<u8>, BackendError>;
   452‚Üí
   453‚Üí    /// Encode an entire allocated program into its final binary form
   454‚Üí    /// (ELF, .wasm, raw binary, etc.).
   455‚Üí    fn encode_program(&self, program: &AllocatedProgram) -> Result<Vec<u8>, BackendError>;
   456‚Üí

  1164‚Üí// ---------------------------------------------------------------------------
  1165‚Üí
  1166‚Üí/// Decode a 32-bit AArch64 instruction word into a human-readable mnemonic.
  1167‚Üí///
  1168‚Üí/// Covers the most common AArch64 instructions: ADD, SUB, MOV, LDR, STR, B,
  1169‚Üí/// BL, RET, CMP, B.cond, STP, LDP, NOP, MUL, SDIV, UDIV, AND, ORR, EOR,
  1170‚Üí/// plus several additional frequently-encountered encodings.
  1171‚Üífn decode_aarch64(word: u32) -> String {
  1172‚Üí    let rd = word & 0x1F;
  1173‚Üí    let rn = (word >> 5) & 0x1F;
  1174‚Üí    let rt = rd; // alias for load/store destination
  1175‚Üí    let rm = (word >> 16) & 0x1F;
  1176‚Üí    let imm12 = (word >> 10) & 0xFFF;
  1177‚Üí    let cond = word & 0xF;
  1178‚Üí
  1179‚Üí    // NOP: d503201f
  1180‚Üí    if word == 0xD503201F {
  1181‚Üí        return "nop".to_string();
  1182‚Üí    }
  1183‚Üí
  1184‚Üí    // RET: d65f03c0
  1185‚Üí    if word == 0xD65F03C0 {
  1186‚Üí        return "ret".to_string();
  1187‚Üí    }
  1188‚Üí
  1189‚Üí    let _top8 = word >> 24;
  1190‚Üí    let _top10 = word >> 22;
  1191‚Üí
  1192‚Üí    // --- ADD/SUB (immediate): 100100xx ...
  1193‚Üí    if (word >> 23) & 0x1FF == 0b1_0010_0010 {
  1194‚Üí        // ADD Xd, Xn, #imm12
  1195‚Üí        return format!("add x{}, x{}, #{}", rd, rn, imm12);
  1196‚Üí    }
  1197‚Üí    if (word >> 23) & 0x1FF == 0b1_1010_0010 {
  1198‚Üí        // SUB Xd, Xn, #imm12
  1199‚Üí        return format!("sub x{}, x{}, #{}", rd, rn, imm12);
  1200‚Üí    }
  1201‚Üí
  1202‚Üí    // --- ADD (shifted register): 1_00_0101_1_xxx ...
  1203‚Üí    if (word >> 24) & 0xFF == 0b1000_1011 {
  1204‚Üí        // ADD Xd, Xn, Xm
  1205‚Üí        return format!("add x{}, x{}, x{}", rd, rn, rm);
  1206‚Üí    }
  1207‚Üí
  1208‚Üí    // --- SUB (shifted register): 1_00_0101_1_xxx ... with S=1 at bit30
  1209‚Üí    if (word >> 24) & 0xFF == 0b1101_0110 {
  1210‚Üí        // SUB Xd, Xn, Xm (bit 30 set = sub)
  1211‚Üí        return format!("sub x{}, x{}, x{}", rd, rn, rm);
  1212‚Üí    }
  1213‚Üí
  1214‚Üí    // --- AND (shifted register): 1_00_0101_0_00_xxx
  1215‚Üí    if (word >> 24) & 0xFE == 0b1000_1010 {
  1216‚Üí        // Check bit 21-15: opcode[31:21] = 10001010_000
  1217‚Üí        if (word >> 21) & 0x7FF == 0b10001010000 {
  1218‚Üí            return format!("and x{}, x{}, x{}", rd, rn, rm);
  1219‚Üí        }
  1220‚Üí    }
  1221‚Üí
  1222‚Üí    // --- ORR (shifted register): 1_01_0101_0_00_xxx
  1223‚Üí    if (word >> 21) & 0x7FF == 0b10101010000 {
  1224‚Üí        return format!("orr x{}, x{}, x{}", rd, rn, rm);
  1225‚Üí    }
  1226‚Üí
  1227‚Üí    // --- EOR (shifted register): 1_10_0101_0_00_xxx
  1228‚Üí    if (word >> 21) & 0x7FF == 0b11001010000 {
  1229‚Üí        return format!("eor x{}, x{}, x{}", rd, rn, rm);
  1230‚Üí    }
  1231‚Üí
  1232‚Üí    // --- MOV (register): alias for ORR Xd, XZR, Xm
  1233‚Üí    // ORR Xd, XZR, Xm: 10101010_000_rm_000000_xzr_rd
  1234‚Üí    // More general: ORR with Rn=XZR(31)
  1235‚Üí    if (word >> 21) & 0x7FF == 0b10101010000 && rn == 31 {
  1236‚Üí        return format!("mov x{}, x{}", rd, rm);
  1237‚Üí    }
  1238‚Üí
  1239‚Üí    // --- MUL: MADD Xd, Xn, Xm, XZR
  1240‚Üí    // Encoding: 1_00_1101_1_000_Rm_0_01111_Rn_Rd
  1241‚Üí    if (word >> 21) & 0x7FF == 0b10011011000 && ((word >> 10) & 0x1F) == 0b01111 {
  1242‚Üí        return format!("mul x{}, x{}, x{}", rd, rn, rm);
  1243‚Üí    }
  1244‚Üí
  1245‚Üí    // --- SDIV: 1_00_1101_1_0100_00000_00001_Rn_Rd  (actually 1_00_1101_0100_Rm_00001_Rn_Rd)
  1246‚Üí    if (word >> 21) & 0x7FF == 0b10011010100 && (word >> 10) & 0x1F == 0b00001 {
  1247‚Üí        return format!("sdiv x{}, x{}, x{}", rd, rn, rm);
  1248‚Üí    }
  1249‚Üí
  1250‚Üí    // --- UDIV: 1_00_1101_0000_Rm_00001_Rn_Rd
  1251‚Üí    if (word >> 21) & 0x7FF == 0b10011010000 && (word >> 10) & 0x1F == 0b00001 {
  1252‚Üí        return format!("udiv x{}, x{}, x{}", rd, rn, rm);
  1253‚Üí    }
  1254‚Üí
  1255‚Üí    // --- CMP (immediate): SUBS XZR, Xn, #imm12
  1256‚Üí    // 11100001_00_xxx_xxx_xxx_xxx_xxx_11111_xxx
  1257‚Üí    if (word >> 23) & 0x1FF == 0b1_1110_0010 && rd == 31 {
  1258‚Üí        return format!("cmp x{}, #{}", rn, imm12);
  1259‚Üí    }
  1260‚Üí
  1261‚Üí    // --- CMP (register): SUBS XZR, Xn, Xm
  1262‚Üí    if (word >> 21) & 0x7FF == 0b11101011000 && rd == 31 {
  1263‚Üí        return format!("cmp x{}, x{}", rn, rm);
  1264‚Üí    }
  1265‚Üí
  1266‚Üí    // --- B.cond: 0101010x xxxxxxxxxx xxxxxx cond
  1267‚Üí    if (word >> 24) & 0xFF == 0x54 {
  1268‚Üí        let cond_name = match cond {
  1269‚Üí            0 => "eq",
  1270‚Üí            1 => "ne",
  1271‚Üí            2 => "cs",
  1272‚Üí            3 => "cc",
  1273‚Üí            4 => "mi",
  1274‚Üí            5 => "pl",
  1275‚Üí            6 => "vs",
  1276‚Üí            7 => "vc",
  1277‚Üí            8 => "hi",
  1278‚Üí            9 => "ls",
  1279‚Üí            10 => "ge",
  1280‚Üí            11 => "lt",
  1281‚Üí            12 => "gt",
  1282‚Üí            13 => "le",
  1283‚Üí            14 => "al",
  1284‚Üí            _ => "??",
  1285‚Üí        };
  1286‚Üí        let imm19 = (word >> 5) & 0x7FFFF;
  1287‚Üí        let offset = ((imm19 as i32) << 13) >> 11; // sign-extend and *4
  1288‚Üí        return format!("b.{} {:+}", cond_name, offset);
  1289‚Üí    }
  1290‚Üí
  1291‚Üí    // --- B (unconditional): 000101xx xxxxxxxxxxxxxxxxxxxx
  1292‚Üí    if (word >> 26) & 0x3F == 0b000101 {
  1293‚Üí        let imm26 = word & 0x3FFFFFF;
  1294‚Üí        let offset = ((imm26 as i32) << 6) >> 4; // sign-extend and *4
  1295‚Üí        return format!("b {:+}", offset);
  1296‚Üí    }
  1297‚Üí
  1298‚Üí    // --- BL: 100101xx xxxxxxxxxxxxxxxxxxxx
  1299‚Üí    if (word >> 26) & 0x3F == 0b100101 {
  1300‚Üí        let imm26 = word & 0x3FFFFFF;
  1301‚Üí        let offset = ((imm26 as i32) << 6) >> 4;
  1302‚Üí        return format!("bl {:+}", offset);
  1303‚Üí    }
  1304‚Üí
  1305‚Üí    // --- LDR (unsigned offset): 11111001_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1306‚Üí    if (word >> 22) & 0x3FF == 0b1111100101 {
  1307‚Üí        let imm12_raw = (word >> 10) & 0xFFF;
  1308‚Üí        let offset = imm12_raw * 8; // scale by 8 for 64-bit
  1309‚Üí        return format!("ldr x{}, [x{}, #{}]", rt, rn, offset);
  1310‚Üí    }
  1311‚Üí
  1312‚Üí    // --- STR (unsigned offset): 11111000_01_xxx_xxx_xxx_xxx_xxx_xn_rt
  1313‚Üí    if (word >> 22) & 0x3FF == 0b1111100001 {
  1314‚Üí        let imm12_raw = (word >> 10) & 0xFFF;
  1315‚Üí        let offset = imm12_raw * 8;
  1316‚Üí        return format!("str x{}, [x{}, #{}]", rt, rn, offset);
  1317‚Üí    }
  1318‚Üí
  1319‚Üí    // --- LDP (signed offset, 64-bit): 101_0100_110_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1320‚Üí    if (word >> 22) & 0x3FF == 0b1010100110 {
  1321‚Üí        let rt2 = (word >> 10) & 0x1F;
  1322‚Üí        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1323‚Üí        let offset = imm7 * 8;
  1324‚Üí        return format!("ldp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1325‚Üí    }
  1326‚Üí
  1327‚Üí    // --- STP (signed offset, 64-bit): 101_0100_010_xxx_xxx_xxx_xxx_xxx_xn_rt2
  1328‚Üí    if (word >> 22) & 0x3FF == 0b1010100010 {
  1329‚Üí        let rt2 = (word >> 10) & 0x1F;
  1330‚Üí        let imm7 = ((word >> 15) & 0x7F) as i8 as i32;
  1331‚Üí        let offset = imm7 * 8;
  1332‚Üí        return format!("stp x{}, x{}, [x{}, #{}]", rt, rt2, rn, offset);
  1333‚Üí    }
  1334‚Üí
  1335‚Üí    // --- MOVZ: 110100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1336‚Üí    if (word >> 23) & 0x1FF == 0b110100101 {
  1337‚Üí        let hw = (word >> 21) & 0x3;
  1338‚Üí        let imm16 = (word >> 5) & 0xFFFF;
  1339‚Üí        return format!("movz x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1340‚Üí    }
  1341‚Üí
  1342‚Üí    // --- MOVK: 111100101_ww_xxx_xxx_xxx_xxx_xxx_xn_rd
  1343‚Üí    if (word >> 23) & 0x1FF == 0b111100101 {
  1344‚Üí        let hw = (word >> 21) & 0x3;
  1345‚Üí        let imm16 = (word >> 5) & 0xFFFF;
  1346‚Üí        return format!("movk x{}, #{}{}, LSL #{}", rd, imm16, "", hw * 16);
  1347‚Üí    }
  1348‚Üí
  1349‚Üí    format!(".word {:08x}", word)
  1350‚Üí}

  1379‚Üí/// Compute the stack frame size for an IR function.
  1380‚Üí///
  1381‚Üí/// Replicates the private `compute_frame_size` function in `emit.rs`:
  1382‚Üí/// sums `Alloc` instruction sizes and rounds up to 16-byte alignment.
  1383‚Üí/// NOTE: Does NOT include 16 bytes for FP/LR pair (handled by prologue separately).
  1384‚Üífn aarch64_compute_frame_size(func: &IRFunction) -> usize {
  1385‚Üí    let mut total: u32 = 0; // Alloc sizes only; FP/LR handled separately
  1386‚Üí    for block in &func.blocks {
  1387‚Üí        for instr in &block.instructions {
  1388‚Üí            if let IRInstr::Alloc { size, .. } = instr {
  1389‚Üí                let aligned = (*size).div_ceil(16) * 16;
  1390‚Üí                total += aligned;
  1391‚Üí            }
  1392‚Üí        }
  1393‚Üí    }
  1394‚Üí    // Round up to 16-byte alignment
  1395‚Üí    total = (total + 15) & !15;
  1396‚Üí    total as usize
  1397‚Üí}

  1399‚Üí/// Build an ELF64 binary for AArch64 Linux with 2 LOAD segments.
  1400‚Üí///
  1401‚Üí/// Segment 1: PF_R | PF_X ‚Äî contains .text (code)
  1402‚Üí/// Segment 2: PF_R | PF_W ‚Äî contains .data + stack space (writable)
  1403‚Üí///
  1404‚Üí/// The two segments are separated by page alignment (4KB) to ensure
  1405‚Üí/// the kernel maps them with different permissions. Without this,
  1406‚Üí/// a single PF_R|PF_W|PF_X segment is insecure and may cause
  1407‚Üí/// QEMU/Linux to reject the executable.
  1408‚Üífn build_aarch64_elf_2seg(code: &[u8], base_addr: u64) -> Vec<u8> {
  1409‚Üí    const PAGE_SIZE: u64 = 0x1000; // 4 KB
  1410‚Üí
  1411‚Üí    let elf_header_size: u64 = 64;
  1412‚Üí    let phdr_size: u64 = 56;
  1413‚Üí    let num_phdrs: u64 = 2;
  1414‚Üí    let phdr_end = elf_header_size + num_phdrs * phdr_size;
  1415‚Üí    // Page-align the text segment start in the file.  The kernel's ELF
  1416‚Üí    // loader mmap()s each LOAD segment, and the file offset must be
  1417‚Üí    // congruent with the virtual address modulo the page size.  The
  1418‚Üí    // simplest way to guarantee this is to place the text at a
  1419‚Üí    // page-aligned file offset, with vaddr = base_addr.
  1420‚Üí    let text_offset = ((phdr_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1421‚Üí    let text_size = code.len() as u64;
  1422‚Üí
  1423‚Üí    // The data segment starts on the next page after the text.
  1424‚Üí    let text_file_end = text_offset + text_size;
  1425‚Üí    let data_vaddr = ((base_addr + text_file_end + PAGE_SIZE - 1) / PAGE_SIZE) * PAGE_SIZE;
  1426‚Üí    let data_offset = data_vaddr - base_addr; // file offset for data segment
  1427‚Üí    let data_size: u64 = PAGE_SIZE; // 1 page of writable memory for stack/data
  1428‚Üí    let entry_point = base_addr + text_offset;
  1429‚Üí
  1430‚Üí    let mut elf = Vec::with_capacity((data_offset + data_size) as usize);
  1431‚Üí
  1432‚Üí    // --- e_ident ---
  1433‚Üí    elf.extend_from_slice(&[0x7f, b'E', b'L', b'F']); // magic
  1434‚Üí    elf.push(2); // ELFCLASS64
  1435‚Üí    elf.push(1); // ELFDATA2LSB
  1436‚Üí    elf.push(1); // EV_CURRENT
  1437‚Üí    elf.push(3); // ELFOSABI_LINUX
  1438‚Üí    elf.push(0); // padding
  1439‚Üí    elf.extend_from_slice(&[0u8; 7]); // padding
  1440‚Üí
  1441‚Üí    // --- ELF header fields ---
  1442‚Üí    elf.extend_from_slice(&2u16.to_le_bytes()); // e_type = ET_EXEC
  1443‚Üí    elf.extend_from_slice(&183u16.to_le_bytes()); // e_machine = EM_AARCH64
  1444‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // e_version
  1445‚Üí    elf.extend_from_slice(&entry_point.to_le_bytes()); // e_entry
  1446‚Üí    elf.extend_from_slice(&elf_header_size.to_le_bytes()); // e_phoff
  1447‚Üí    elf.extend_from_slice(&0u64.to_le_bytes()); // e_shoff (no section headers)
  1448‚Üí    elf.extend_from_slice(&0u32.to_le_bytes()); // e_flags
  1449‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_ehsize
  1450‚Üí    elf.extend_from_slice(&56u16.to_le_bytes()); // e_phentsize
  1451‚Üí    elf.extend_from_slice(&2u16.to_le_bytes()); // e_phnum = 2
  1452‚Üí    elf.extend_from_slice(&64u16.to_le_bytes()); // e_shentsize
  1453‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shnum
  1454‚Üí    elf.extend_from_slice(&0u16.to_le_bytes()); // e_shstrndx
  1455‚Üí
  1456‚Üí    // --- Program Header 1: LOAD (PF_R | PF_X) ‚Äî .text ---
  1457‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1458‚Üí    elf.extend_from_slice(&5u32.to_le_bytes()); // p_flags = PF_R | PF_X
  1459‚Üí    elf.extend_from_slice(&text_offset.to_le_bytes()); // p_offset
  1460‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_vaddr
  1461‚Üí    elf.extend_from_slice(&(base_addr + text_offset).to_le_bytes()); // p_paddr
  1462‚Üí    elf.extend_from_slice(&text_size.to_le_bytes()); // p_filesz
  1463‚Üí    elf.extend_from_slice(&text_size.to_le_bytes()); // p_memsz
  1464‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1465‚Üí
  1466‚Üí    // --- Program Header 2: LOAD (PF_R | PF_W) ‚Äî .data / stack ---
  1467‚Üí    elf.extend_from_slice(&1u32.to_le_bytes()); // p_type = PT_LOAD
  1468‚Üí    elf.extend_from_slice(&6u32.to_le_bytes()); // p_flags = PF_R | PF_W
  1469‚Üí    elf.extend_from_slice(&data_offset.to_le_bytes()); // p_offset
  1470‚Üí    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_vaddr
  1471‚Üí    elf.extend_from_slice(&data_vaddr.to_le_bytes()); // p_paddr
  1472‚Üí    elf.extend_from_slice(&0u64.to_le_bytes()); // p_filesz (no initialized data)
  1473‚Üí    elf.extend_from_slice(&data_size.to_le_bytes()); // p_memsz (writable pages)
  1474‚Üí    elf.extend_from_slice(&PAGE_SIZE.to_le_bytes()); // p_align
  1475‚Üí
  1476‚Üí    // --- .text section ---
  1477‚Üí    // Pad to page-aligned text_offset
  1478‚Üí    while (elf.len() as u64) < text_offset {
  1479‚Üí        elf.push(0);
  1480‚Üí    }
  1481‚Üí    elf.extend_from_slice(code);
  1482‚Üí
  1483‚Üí    // --- Pad to data segment offset (if needed) ---
  1484‚Üí    while (elf.len() as u64) < data_offset {
  1485‚Üí        elf.push(0);
  1486‚Üí    }
  1487‚Üí
  1488‚Üí    // No file data for the .data segment (it's BSS-like, zero-initialized)
  1489‚Üí
  1490‚Üí    elf
  1491‚Üí}

  1493‚Üí/// Build ARM64 runtime I/O functions using Linux SVC syscalls.
  1494‚Üí///
  1495‚Üí/// Provides:
  1496‚Üí/// - `__vuma_print_hex`: Print X0 as 8 hex digits to stdout (FD=1)
  1497‚Üí///   Uses sys_write (X8=64) via SVC #0.
  1498‚Üí///   Saves/restores X1,X2,X3,X8,X9,X10 on stack.
  1499‚Üí///
  1500‚Üí/// - `__vuma_print_int`: Print X0 as a decimal integer to stdout (FD=1)
  1501‚Üí///   Converts digit-by-digit into a stack buffer, then sys_write.
  1502‚Üí///
  1503‚Üí/// - `__vuma_print_newline`: Print a newline character to stdout.
  1504‚Üí///
  1505‚Üí/// All functions follow AAPCS64: X0 is the argument, X1-X7 are caller-saved,
  1506‚Üí/// X8 is the indirect result register / syscall number, X19-X28 are callee-saved.
  1507‚Üífn build_aarch64_runtime() -> Vec<u8> {
  1508‚Üí    let mut code = Vec::new();
  1509‚Üí
  1510‚Üí    // ‚îÄ‚îÄ __vuma_print_hex ‚îÄ‚îÄ
  1511‚Üí    // Input: X0 = 64-bit value to print as 8 hex digits
  1512‚Üí    // Clobbers: X1, X2, X3, X8, X9, X10
  1513‚Üí    // Stack frame: 48 bytes (8 for pair save + 8 for hex chars + 32 padding)
  1514‚Üí
  1515‚Üí    // Prologue: save callee-saved and link register
  1516‚Üí    // STP X29, X30, [SP, #-48]!  (pre-indexed, 48-byte frame)
  1517‚Üí    code.extend_from_slice(&0xA9BF7BFDu32.to_le_bytes());
  1518‚Üí    // ADD X29, SP, #0
  1519‚Üí    code.extend_from_slice(&0x910003FDu32.to_le_bytes());
  1520‚Üí
  1521‚Üí    // Save X9, X10 on stack at [SP, #16]
  1522‚Üí    // STP X9, X10, [SP, #16]
  1523‚Üí    code.extend_from_slice(&0xA90127E9u32.to_le_bytes());
  1524‚Üí
  1525‚Üí    // Save X1, X2 on stack at [SP, #32]
  1526‚Üí    // STP X1, X2, [SP, #32]
  1527‚Üí    code.extend_from_slice(&0xA9020441u32.to_le_bytes());
  1528‚Üí
  1529‚Üí    // Save X3, X8 on stack at [SP, #40]  (offset 40 = 5*8)
  1530‚Üí    // Wait, 40 = 5*8 but STP needs offset/8 for signed-offset. 40/8 = 5.
  1531‚Üí    // STP X3, X8, [SP, #40]
  1532‚Üí    code.extend_from_slice(&0xA9052C63u32.to_le_bytes());
  1533‚Üí
  1534‚Üí    // Hex lookup table: we'll use ADD + LDRB pattern.
  1535‚Üí    // Instead, compute hex digit directly:
  1536‚Üí    // For each nibble: digit = nibble; if digit < 10 then char = digit + '0' else char = digit - 10 + 'a'
  1537‚Üí
  1538‚Üí    // We'll write 8 hex chars starting at SP+16 (reuse save area after we load saves)
  1539‚Üí    // Actually, let's use SP+16 as the char buffer (8 bytes).
  1540‚Üí    // First, save the important regs, then overwrite the buffer area.
  1541‚Üí
  1542‚Üí    // Process 8 nibbles from most significant to least significant
  1543‚Üí    // X0 = value, iterate 8 times shifting right by 28, 24, 20, ..., 4, 0
  1544‚Üí
  1545‚Üí    // X9 = char buffer pointer = SP
  1546‚Üí    // X10 = loop counter (0..8)
  1547‚Üí
  1548‚Üí    // ADD X9, SP, #16  (buffer starts at SP+16)
  1549‚Üí    code.extend_from_slice(&0x910043E9u32.to_le_bytes());
  1550‚Üí    // MOV X10, #0
  1551‚Üí    code.extend_from_slice(&0xD280001Au32.to_le_bytes());
  1552‚Üí    // MOV X3, #0  (bit shift amount, starts at 28)
  1553‚Üí    code.extend_from_slice(&0xD280003Du32.to_le_bytes()); // MOVZ X3, #28>>4... wait
  1554‚Üí    // MOV X3, #28
  1555‚Üí    code.extend_from_slice(&0xD2800383u32.to_le_bytes()); // MOVZ X3, #28
  1556‚Üí
  1557‚Üí    // Loop label:
  1558‚Üí    let loop_offset = code.len();
  1559‚Üí
  1560‚Üí    // Extract nibble: X2 = (X0 >> X3) & 0xF
  1561‚Üí    // LSR X2, X0, X3
  1562‚Üí    code.extend_from_slice(&0x9AC02482u32.to_le_bytes()); // LSR X2, X0, X3
  1563‚Üí    // AND X2, X2, #0xF
  1564‚Üí    code.extend_from_slice(&0x92400C42u32.to_le_bytes()); // AND X2, X2, #15
  1565‚Üí
  1566‚Üí    // Convert nibble to hex char:
  1567‚Üí    // if X2 < 10: char = X2 + 48 ('0')
  1568‚Üí    // else: char = X2 - 10 + 97 ('a')
  1569‚Üí    // CMP X2, #10
  1570‚Üí    code.extend_from_slice(&0xF100141Fu32.to_le_bytes()); // CMP X2, #10? No...
  1571‚Üí    // CMP X2, #10: use ADD X1, X2, #0 then CMP... Actually:
  1572‚Üí    // SUB X1, X2, #10 ‚Üí if < 10, result is negative (borrow set)
  1573‚Üí    // Let's use: CMP X2, #9; B.GT hex_alpha
  1574‚Üí    // CMP X2, #9
  1575‚Üí    code.extend_from_slice(&0xF100129Fu32.to_le_bytes()); // CMP X2, #9
  1576‚Üí
  1577‚Üí    // ADD X1, X2, #48  (default: '0' + digit)
  1578‚Üí    code.extend_from_slice(&0x91000C41u32.to_le_bytes()); // ADD X1, X2, #48
  1579‚Üí
  1580‚Üí    // B.GT hex_alpha (placeholder, will be +1 instruction)
  1581‚Üí    code.extend_from_slice(&0x54000069u32.to_le_bytes()); // B.GT +4*2 = +8 bytes
  1582‚Üí
  1583‚Üí    // B store_char
  1584‚Üí    code.extend_from_slice(&0x14000002u32.to_le_bytes()); // B +2 instructions
  1585‚Üí
  1586‚Üí    // hex_alpha: ADD X1, X2, #87  (87 = 97-10, so digit-10+'a')
  1587‚Üí    code.extend_from_slice(&0x91002AC1u32.to_le_bytes()); // ADD X1, X2, #87
  1588‚Üí
  1589‚Üí    // store_char: STRB W1, [X9, X10]
  1590‚Üí    // STRB W1, [X9, X10] ‚Äî register offset
  1591‚Üí    code.extend_from_slice(&0x382A6821u32.to_le_bytes()); // STRB W1, [X9, X10]
  1592‚Üí
  1593‚Üí    // Increment: SUB X3, X3, #4; ADD X10, X10, #1; CMP X10, #8; B.LT loop
  1594‚Üí    code.extend_from_slice(&0xD1000863u32.to_le_bytes()); // SUB X3, X3, #4
  1595‚Üí    code.extend_from_slice(&0x9100054Au32.to_le_bytes()); // ADD X10, X10, #1
  1596‚Üí    code.extend_from_slice(&0xF100151Fu32.to_le_bytes()); // CMP X10, #8
  1597‚Üí
  1598‚Üí    // Calculate branch back to loop start
  1599‚Üí    let current_pos = code.len() as i64;
  1600‚Üí    let loop_start = loop_offset as i64;
  1601‚Üí    let back_offset = (loop_start - current_pos) / 4 - 1; // B.LT is relative to PC+4
  1602‚Üí    let bcond = 0x5400000Bu32 | ((back_offset as u32) & 0x7FFFF); // B.LT
  1603‚Üí    code.extend_from_slice(&bcond.to_le_bytes());
  1604‚Üí
  1605‚Üí    // ‚îÄ‚îÄ sys_write(1, buf, 8) ‚îÄ‚îÄ
  1606‚Üí    // MOV X0, #1  (stdout fd)
  1607‚Üí    code.extend_from_slice(&0xD2800020u32.to_le_bytes()); // MOVZ X0, #1
  1608‚Üí    // ADD X1, SP, #16  (buffer pointer)
  1609‚Üí    code.extend_from_slice(&0x910043E1u32.to_le_bytes());
  1610‚Üí    // MOV X2, #8  (length)
  1611‚Üí    code.extend_from_slice(&0xD2800102u32.to_le_bytes()); // MOVZ X2, #8
  1612‚Üí    // MOV X8, #64  (sys_write)
  1613‚Üí    code.extend_from_slice(&0xD2800808u32.to_le_bytes()); // MOVZ X8, #64
  1614‚Üí    // SVC #0
  1615‚Üí    code.extend_from_slice(&0xD4000001u32.to_le_bytes());
  1616‚Üí
  1617‚Üí    // Restore registers
  1618‚Üí    // LDP X9, X10, [SP, #16]
  1619‚Üí    code.extend_from_slice(&0xA94127E9u32.to_le_bytes());
  1620‚Üí    // LDP X1, X2, [SP, #32]
  1621‚Üí    code.extend_from_slice(&0xA9420441u32.to_le_bytes());
  1622‚Üí    // LDP X3, X8, [SP, #40]
  1623‚Üí    code.extend_from_slice(&0xA9452C63u32.to_le_bytes());
  1624‚Üí    // LDP X29, X30, [SP], #48  (post-indexed load)
  1625‚Üí    code.extend_from_slice(&0xA8C37BFDu32.to_le_bytes());
  1626‚Üí
  1627‚Üí    // RET
  1628‚Üí    code.extend_from_slice(&0xD65F03C0u32.to_le_bytes());
  1629‚Üí
  1630‚Üí    // ‚îÄ‚îÄ __vuma_print_int ‚îÄ‚îÄ
  1631‚Üí    // Input: X0 = 64-bit signed integer to print as decimal
  1632‚Üí    // Strategy: Divide by 10 repeatedly, push digits onto stack, then write.
  1633‚Üí    // X9 = digit count, X10 = buffer pointer (SP-based)
  1634‚Üí    // Stack frame: 64 bytes (16 for save pair + 48 for digit buffer)
  1635‚Üí
  1636‚Üí    // STP X29, X30, [SP, #-64]!
  1637‚Üí    code.extend_from_slice(&0xA9C27BFDu32.to_le_bytes());
  1638‚Üí    // ADD X29, SP, #0
  1639‚Üí    code.extend_from_slice(&0x910003FDu32.to_le_bytes());
  1640‚Üí    // STP X9, X10, [SP, #16]
  1641‚Üí    code.extend_from_slice(&0xA90127E9u32.to_le_bytes());
  1642‚Üí    // STP X1, X2, [SP, #32]
  1643‚Üí    code.extend_from_slice(&0xA9020441u32.to_le_bytes());
  1644‚Üí    // STP X3, X8, [SP, #48]
  1645‚Üí    code.extend_from_slice(&0xA9030463u32.to_le_bytes());
  1646‚Üí
  1647‚Üí    // Handle negative numbers
  1648‚Üí    // CMP X0, #0
  1649‚Üí    code.extend_from_slice(&0xB100001Fu32.to_le_bytes());
  1650‚Üí    // B.GE positive
  1651‚Üí    code.extend_from_slice(&0x5400002Au32.to_le_bytes()); // B.GE +4
  1652‚Üí
  1653‚Üí    // Print '-' for negative
  1654‚Üí    // MOV X1, #45  ('-')
  1655‚Üí    code.extend_from_slice(&0xD28005A1u32.to_le_bytes()); // MOVZ X1, #45
  1656‚Üí    // STRB W1, [SP, #16]
  1657‚Üí    code.extend_from_slice(&0x390013E1u32.to_le_bytes());
  1658‚Üí    // sys_write(1, SP+16, 1)
  1659‚Üí    code.extend_from_slice(&0xD2800020u32.to_le_bytes()); // MOV X0, #1
  1660‚Üí    code.extend_from_slice(&0x910043E1u32.to_le_bytes()); // ADD X1, SP, #16
  1661‚Üí    code.extend_from_slice(&0xD2800022u32.to_le_bytes()); // MOV X2, #1
  1662‚Üí    code.extend_from_slice(&0xD2800808u32.to_le_bytes()); // MOV X8, #64
  1663‚Üí    code.extend_from_slice(&0xD4000001u32.to_le_bytes()); // SVC #0
  1664‚Üí

  1666‚Üí    // NEG X0, X0 = SUB X0, XZR, X0
  1667‚Üí    code.extend_from_slice(&0xCB0003E0u32.to_le_bytes()); // SUB X0, XZR, X0
  1668‚Üí
  1669‚Üí    // positive: convert digits
  1670‚Üí    // X9 = digit count = 0
  1671‚Üí    code.extend_from_slice(&0xD2800019u32.to_le_bytes()); // MOVZ X9, #0
  1672‚Üí    // X10 = buffer pointer = SP + 17 (leave room for potential '-')
