     1→//! # VUMA Compiler API for LLM Consumption
     2→//!
     3→//! This module provides a clean, programmatic interface designed for LLM agents
     4→//! and automated tools that need to compile VUMA source code and receive
     5→//! structured, machine-readable results.
     6→//!
     7→//! ## Design Principles
     8→//!
     9→//! - **Always succeeds**: Every method returns a structured result; errors are
    10→//!   captured as diagnostics rather than panicking or returning `Err`.
    11→//! - **Serializable**: All result types derive `Serialize` / `Deserialize` so
    12→//!   they can be sent over JSON-based IPC channels.
    13→//! - **Summaries, not internals**: The API returns enough information for an
    14→//!   LLM to reason about the program (function signatures, call graph, node
    15→//!   counts) without exposing the full graph representation.
    16→//! - **Target-aware**: The API can compile for any supported backend
    17→//!   (x86_64, AArch64, RISC-V, etc.) and returns target-specific outputs.
    18→//!
    19→//! ## Quick Start
    20→//!
    21→//! ```rust
    22→//! use vuma::api::VumaCompiler;
    23→//!
    24→//! let compiler = VumaCompiler::new();
    25→//! let source = r#"
    26→//!     fn main() {
    27→//!         x = 1 + 2;
    28→//!     }
    29→//! "#;
    30→//!
    31→//! let result = compiler.compile(source);
    32→//! if result.success {
    33→//!     println!("Compiled {} functions", result.scg.unwrap().function_count);
    34→//! } else {
    35→//!     for diag in &result.diagnostics {
    36→//!         println!("[{}] {}", diag.severity, diag.message);
    37→//!     }
    38→//! }
    39→//! ```

    66→/// The primary compiler API for programmatic (LLM) consumption.
    67→///
    68→/// `VumaCompiler` is the main entry point for LLMs and automated tools.
    69→/// It wraps the full VUMA compilation pipeline and returns structured results
    70→/// that are easy to parse and reason about programmatically.
    71→#[derive(Debug, Clone)]
    72→pub struct VumaCompiler {
    73→    config: CompileConfig,
    74→}
    75→
    76→impl VumaCompiler {
    77→    /// Create a new compiler with default configuration.
    78→    pub fn new() -> Self {
    79→        Self {
    80→            config: CompileConfig::default(),
    81→        }
    82→    }
    83→
    84→    /// Create a compiler with a specific configuration.
    85→    pub fn with_config(config: CompileConfig) -> Self {
    86→        Self { config }
    87→    }

    89→    /// Compile from source string. Returns a structured result.
    90→    ///
    91→    /// This runs the full compilation pipeline:
    92→    /// Parse → AST → SCG → BD Inference → MSG → IVE Verification
    93→    /// → SCG Transforms → IR Lowering → Register Allocation → Code Emission
    94→    ///
    95→    /// The result always contains a value — check `result.success` to
    96→    /// determine if compilation succeeded, and inspect `result.diagnostics`
    97→    /// for any warnings or errors.
    98→    pub fn compile(&self, source: &str) -> CompileResult {
    99→        let start = Instant::now();
   100→        let source_lines = source.lines().count();
   101→        let source_bytes = source.len();
   102→
   103→        match pipeline::compile(source, &self.config) {
   104→            Ok(output) => {
   105→                let scg_summary = Some(build_scg_summary(&output.scg));
   106→
   107→                // Get disassembly from the binary using the default backend
   108→                let disasm = disassemble_default(&output.binary);
   109→
   110→                let target_output = Some(TargetOutput {
   111→                    backend: "aarch64".to_string(),
   112→                    binary: output.binary.clone(),
   113→                    binary_size: output.binary.len(),
   114→                    disassembly: disasm,
   115→                });
   116→
   117→                // Collect any non-fatal diagnostics
   118→                let diagnostics = Vec::new();
   119→
   120→                CompileResult {
   121→                    success: true,
   122→                    diagnostics,
   123→                    scg: scg_summary,
   124→                    target: target_output,
   125→                    metadata: CompileMetadata {
   126→                        compile_time_ms: start.elapsed().as_millis() as u64,
   127→                        source_lines,
   128→                        source_bytes,
   129→                    },
   130→                }
   131→            }
   132→            Err(errors) => {
   133→                let diagnostics = errors
   134→                    .iter()
   135→                    .flat_map(|e| diagnostics::from_vuma_error(e))
   136→                    .collect();
   137→
   138→                CompileResult {
   139→                    success: false,
   140→                    diagnostics,
   141→                    scg: None,
   142→                    target: None,
   143→                    metadata: CompileMetadata {
   144→                        compile_time_ms: start.elapsed().as_millis() as u64,
   145→                        source_lines,
   146→                        source_bytes,
   147→                    },
   148→                }
   149→            }
   150→        }
   151→    }

   153→    /// Compile for a specific target backend.
   154→    ///
   155→    /// Valid target strings: `"x86_64"`, `"aarch64"`, `"riscv64"`,
   156→    /// `"wasm32"`, `"loongarch64"`, `"arm32"`, `"mips64"`, `"ppc64"`.
   157→    ///
   158→    /// If the target string is not recognised, the result will contain
   159→    /// error diagnostics.
   160→    pub fn compile_for_target(&self, source: &str, target: &str) -> CompileResult {
   161→        let start = Instant::now();
   162→        let source_lines = source.lines().count();
   163→        let source_bytes = source.len();
   164→
   165→        // Parse the target string into a BackendKind
   166→        let backend_kind = match parse_target(target) {
   167→            Some(kind) => kind,
   168→            None => {
   169→                return CompileResult {
   170→                    success: false,
   171→                    diagnostics: vec![VumaDiagnostic::new(
   172→                        "E021",
   173→                        DiagnosticSeverity::Error,
   174→                        format!(
   175→                            "Unknown target '{}'. Available: x86_64, aarch64, riscv64, \
   176→                             wasm32, loongarch64, arm32, mips64, ppc64",
   177→                            target
   178→                        ),
   179→                        "target-selection",
   180→                        DiagnosticSourceLocation::unknown(),
   181→                    )],
   182→                    scg: None,
   183→                    target: None,
   184→                    metadata: CompileMetadata {
   185→                        compile_time_ms: start.elapsed().as_millis() as u64,
   186→                        source_lines,
   187→                        source_bytes,
   188→                    },
   189→                };
   190→            }
   191→        };
   192→
   193→        // Run the front-end pipeline (parse through SCG transforms)
   194→        let front_result = run_frontend(source, &self.config);
   195→
   196→        let (scg, mut diagnostics) = match front_result {
   197→            FrontendResult::Ok { scg } => (scg, Vec::new()),
   198→            FrontendResult::Err { diagnostics } => {
   199→                return CompileResult {
   200→                    success: false,
   201→                    diagnostics,
   202→                    scg: None,
   203→                    target: None,
   204→                    metadata: CompileMetadata {
   205→                        compile_time_ms: start.elapsed().as_millis() as u64,
   206→                        source_lines,
   207→                        source_bytes,
   208→                    },
   209→                };
   210→            }
   211→        };
   212→
   213→        // Build SCG summary from the validated SCG
   214→        let scg_summary = Some(build_scg_summary(&scg));
   215→
   216→        // Run target-specific codegen
   217→        let target_output = match run_backend_codegen(&scg, backend_kind) {
   218→            Ok(output) => Some(output),
   219→            Err(diags) => {
   220→                diagnostics.extend(diags);
   221→                return CompileResult {
   222→                    success: false,
   223→                    diagnostics,
   224→                    scg: scg_summary,
   225→                    target: None,
   226→                    metadata: CompileMetadata {
   227→                        compile_time_ms: start.elapsed().as_millis() as u64,
   228→                        source_lines,
   229→                        source_bytes,
   230→                    },
   231→                };
   232→            }
   233→        };
   234→
   235→        CompileResult {
   236→            success: true,
   237→            diagnostics,
   238→            scg: scg_summary,
   239→            target: target_output,
   240→            metadata: CompileMetadata {
   241→                compile_time_ms: start.elapsed().as_millis() as u64,
   242→                source_lines,
   243→                source_bytes,
   244→            },
   245→        }
   246→    }

   248→    /// Just parse — return AST/SCG without codegen.
   249→    ///
   250→    /// Useful for LLMs that want to understand the program structure
   251→    /// without incurring the cost of full code generation.
   252→    pub fn parse(&self, source: &str) -> ParseResult {
   253→        let start = Instant::now();
   254→
   255→        use vuma_parser::{AstToScg, Parser};
   256→
   257→        // Parse source to AST
   258→        let mut parser = Parser::new(source);
   259→        let parse_output = parser.parse_program();
   260→
   261→        if parse_output.has_errors() {
   262→            let diagnostics = diagnostics::from_parse_errors(&parse_output.errors, source, None);
   263→
   264→            return ParseResult {
   265→                success: false,
   266→                diagnostics,
   267→                ast_summary: None,
   268→                scg: None,
   269→                metadata: CompileMetadata {
   270→                    compile_time_ms: start.elapsed().as_millis() as u64,
   271→                    source_lines: source.lines().count(),
   272→                    source_bytes: source.len(),
   273→                },
   274→            };
   275→        }
   276→
   277→        let ast = parse_output.unwrap();
   278→        let ast_summary = Some(build_ast_summary(&ast));
   279→
   280→        // Convert AST to SCG
   281→        let mut converter = AstToScg::new();
   282→        match converter.convert(&ast) {
   283→            Ok(scg) => {
   284→                let scg_summary = Some(build_scg_summary(&scg));
   285→                ParseResult {
   286→                    success: true,
   287→                    diagnostics: Vec::new(),
   288→                    ast_summary,
   289→                    scg: scg_summary,
   290→                    metadata: CompileMetadata {
   291→                        compile_time_ms: start.elapsed().as_millis() as u64,
   292→                        source_lines: source.lines().count(),
   293→                        source_bytes: source.len(),
   294→                    },
   295→                }
   296→            }
   297→            Err(e) => {
   298→                let diagnostics = vec![VumaDiagnostic::new(
   299→                    "E019",
   300→                    DiagnosticSeverity::Error,
   301→                    format!("{}", e),
   302→                    "ast-to-scg",
   303→                    DiagnosticSourceLocation::unknown(),
   304→                )];
   305→                ParseResult {
   306→                    success: false,
   307→                    diagnostics,
   308→                    ast_summary,
   309→                    scg: None,
   310→                    metadata: CompileMetadata {
   311→                        compile_time_ms: start.elapsed().as_millis() as u64,
   312→                        source_lines: source.lines().count(),
   313→                        source_bytes: source.len(),
   314→                    },
   315→                }
   316→            }
   317→        }
   318→    }

   320→    /// Get SCG summary for a source string.
   321→    ///
   322→    /// This runs the front-end pipeline (parse + SCG construction +
   323→    /// validation + BD inference + SCG transforms) but skips codegen.
   324→    /// It is faster than `compile()` and is ideal for program analysis.
   325→    pub fn analyze(&self, source: &str) -> ScgSummary {
   326→        let front_result = run_frontend(source, &self.config);
   327→        match front_result {
   328→            FrontendResult::Ok { scg, .. } => build_scg_summary(&scg),
   329→            FrontendResult::Err { .. } => ScgSummary {
   330→                function_count: 0,
   331→                functions: Vec::new(),
   332→                total_nodes: 0,
   333→                total_edges: 0,
   334→            },
   335→        }
   336→    }

   338→    /// List available targets.
   339→    ///
   340→    /// Returns information about every backend the compiler supports.
   341→    pub fn available_targets(&self) -> Vec<ApiTargetInfo> {
   342→        use vuma_codegen::backend::{create_backend, BackendKind};
   343→
   344→        let all_kinds = [
   345→            BackendKind::AArch64,
   346→            BackendKind::X86_64,
   347→            BackendKind::RiscV64,
   348→            BackendKind::Wasm32,
   349→            BackendKind::LoongArch64,
   350→            BackendKind::Arm32,
   351→            BackendKind::Mips64,
   352→            BackendKind::PowerPC64,
   353→        ];
   354→
   355→        all_kinds
   356→            .iter()
   357→            .filter_map(|&kind| {
   358→                create_backend(kind).ok().map(|backend| {
   359→                    let info = backend.target_info();
   360→                    ApiTargetInfo {
   361→                        name: kind.isa_name().to_string(),
   362→                        triple: info.target_triple().to_string(),
   363→                        pointer_width: info.pointer_width() * 8, // bytes → bits
   364→                        endianness: match info.endianness() {
   365→                            vuma_codegen::backend::Endianness::Little => "little".to_string(),
   366→                            vuma_codegen::backend::Endianness::Big => "big".to_string(),
   367→                            vuma_codegen::backend::Endianness::Bi => "bi".to_string(),
   368→                        },
   369→                        output_format: match info.output_format() {
   370→                            vuma_codegen::backend::OutputFormat::Elf64 => "elf64".to_string(),
   371→                            vuma_codegen::backend::OutputFormat::Elf32 => "elf32".to_string(),
   372→                            vuma_codegen::backend::OutputFormat::WasmBinary => "wasm".to_string(),
   373→                            vuma_codegen::backend::OutputFormat::RawBinary => "raw".to_string(),
   374→                        },
   375→                    }
   376→                })
   377→            })
   378→            .collect()
   379→    }

   381→    /// Validate source without full compilation.
   382→    ///
   383→    /// Runs parsing and SCG validation, returning a list of diagnostics.
   384→    /// This is the fastest way to check if a program is well-formed.
   385→    pub fn validate(&self, source: &str) -> Vec<VumaDiagnostic> {
   386→        use vuma_parser::{AstToScg, Parser};
   387→
   388→        let mut all_diagnostics = Vec::new();
   389→
   390→        // Parse
   391→        let mut parser = Parser::new(source);
   392→        let parse_output = parser.parse_program();
   393→
   394→        if parse_output.has_errors() {
   395→            all_diagnostics.extend(diagnostics::from_parse_errors(
   396→                &parse_output.errors,
   397→                source,
   398→                None,
   399→            ));
   400→            return all_diagnostics;
   401→        }
   402→
   403→        let ast = parse_output.unwrap();
   404→
   405→        // AST → SCG
   406→        let mut converter = AstToScg::new();
   407→        let scg = match converter.convert(&ast) {
   408→            Ok(scg) => scg,
   409→            Err(e) => {
   410→                all_diagnostics.push(VumaDiagnostic::new(
   411→                    "E019",
   412→                    DiagnosticSeverity::Error,
   413→                    format!("{}", e),
   414→                    "ast-to-scg",
   415→                    DiagnosticSourceLocation::unknown(),
   416→                ));
   417→                return all_diagnostics;
   418→            }
   419→        };
   420→
   421→        // Validate SCG
   422→        let validation = scg.validate();
   423→        if !validation.is_valid {
   424→            for err in &validation.errors {
   425→                all_diagnostics.push(VumaDiagnostic::new(
   426→                    "E022",
   427→                    DiagnosticSeverity::Error,
   428→                    err.clone(),
   429→                    "scg-validation",
   430→                    DiagnosticSourceLocation::unknown(),
   431→                ));
   432→            }
   433→        }
   434→
   435→        all_diagnostics
   436→    }

   438→    /// Verify a VUMA program by running all five IVE invariant checkers
   439→    /// on the SCG and producing a structured verification report.
   440→    ///
   441→    /// This method runs the full front-end pipeline (parse → SCG),
   442→    /// then invokes the IVE `InvariantAggregator` to check all five
   443→    /// core invariants (liveness, exclusivity, interpretation, origin,
   444→    /// cleanup) and the proof system to produce pass/fail per invariant
   445→    /// with counterexamples for any violations.
   446→    ///
   447→    /// # Returns
   448→    ///
   449→    /// A [`VerificationReport`] containing:
   450→    /// - Per-invariant pass/fail status
   451→    /// - Counterexamples for each violation
   452→    /// - An overall pass/fail verdict
   453→    /// - Timing metadata
   454→    ///
   455→    /// # Example
   456→    ///
   457→    /// ```rust
   458→    /// use vuma::api::VumaCompiler;
   459→    ///
   460→    /// let compiler = VumaCompiler::new();
   461→    /// let source = "fn main() {}";
   462→    /// let report = compiler.verify(source);
   463→    /// println!("Overall verdict: {}", report.overall_verdict);
   464→    /// for inv in &report.invariants {
   465→    ///     println!("  {} — {}", inv.kind, inv.status);
   466→    /// }
   467→    /// ```
   468→    pub fn verify(&self, source: &str) -> VerificationReport {
   469→        let start = Instant::now();
   470→
   471→        // Run the front-end pipeline to get the SCG.
   472→        let front_result = run_frontend(source, &self.config);
   473→
   474→        let scg = match front_result {
   475→            FrontendResult::Ok { scg } => scg,
   476→            FrontendResult::Err { diagnostics } => {
   477→                let messages: Vec<String> =
   478→                    diagnostics.iter().map(|d| d.message.clone()).collect();
   479→                return VerificationReport {
   480→                    overall_verdict: VerificationVerdict::Error,
   481→                    invariants: Vec::new(),
   482→                    diagnostics: messages,
   483→                    metadata: VerificationMetadata {
   484→                        total_elapsed_ms: start.elapsed().as_millis() as u64,
   485→                        source_lines: source.lines().count(),
   486→                        source_bytes: source.len(),
   487→                    },
   488→                };
   489→            }
   490→        };
   491→
   492→        // Run the IVE invariant aggregator at Normal level (all 5 checks).
   493→        let aggregator = InvariantAggregator::new().with_level(IveVerificationLevel::Normal);
   494→        let input = VerificationInput::from_scg(scg.clone());
   495→        let aggregated = aggregator.verify_all(&input);
   496→
   497→        // Convert the aggregated result into per-invariant API results,
   498→        // building counterexamples from the proof system for any violations.
   499→        let mut invariants = Vec::with_capacity(aggregated.per_invariant.len());
   500→        for pir in &aggregated.per_invariant {
   501→            let kind_str = pir.kind.label().to_string();
   502→
   503→            let (status, counterexample) = if pir.is_pass() {
   504→                (InvariantVerificationStatus::Pass, None)
   505→            } else if pir.is_fail() {
   506→                // Build a proof-system counterexample from the IVE violation.
   507→                let proof_ce = build_proof_counterexample(&pir.result);
   508→                (InvariantVerificationStatus::Fail, Some(proof_ce))
   509→            } else {
   510→                (InvariantVerificationStatus::Unverified, None)
   511→            };
   512→
   513→            invariants.push(InvariantVerification {
   514→                kind: kind_str,
   515→                status,
   516→                message: pir.result.message.clone(),
   517→                elapsed_ms: pir.elapsed_ms,
   518→                counterexample,
   519→            });
   520→        }
   521→
   522→        // Determine overall verdict.
   523→        let overall_verdict = match aggregated.overall {
   524→            vuma_ive::OverallVerdict::Pass => VerificationVerdict::Pass,
   525→            vuma_ive::OverallVerdict::Fail => VerificationVerdict::Fail,
   526→            vuma_ive::OverallVerdict::Inconclusive => VerificationVerdict::Inconclusive,
   527→            vuma_ive::OverallVerdict::NoChecks => VerificationVerdict::Error,
   528→        };
   529→
   530→        // Also attempt proof-system verification for a cross-check.
   531→        let proof_bundle = build_proof_bundle(&scg);
   532→        let proof_statuses = proof_bundle.status();
   533→
   534→        // If the proof system found failures that the IVE missed,
   535→        // upgrade unverified results to fail.
   536→        for (i, (_inv_name, proof_status)) in proof_statuses.iter().enumerate() {
   537→            if i < invariants.len() {
   538→                if let InvariantStatus::Failed(reason) = proof_status {
   539→                    if invariants[i].status == InvariantVerificationStatus::Unverified {
   540→                        invariants[i].status = InvariantVerificationStatus::Fail;
   541→                        invariants[i].counterexample = Some(CounterexampleInfo {
   542→                            description: reason.clone(),
   543→                            execution_trace: Vec::new(),
   544→                        });
   545→                    }
   546→                }
   547→            }
   548→        }
   549→
   550→        let diagnostics = Vec::new();
   551→        let total_elapsed_ms = start.elapsed().as_millis() as u64;
   552→
   553→        VerificationReport {
   554→            overall_verdict,
   555→            invariants,
   556→            diagnostics,
   557→            metadata: VerificationMetadata {
   558→                total_elapsed_ms,
   559→                source_lines: source.lines().count(),
   560→                source_bytes: source.len(),
   561→            },
   562→        }
   563→    }
   564→}

   569→    }

   575→
   576→/// Result of compiling a VUMA program.
   577→///
   578→/// This is the primary return type for `VumaCompiler::compile()` and
   579→/// `VumaCompiler::compile_for_target()`. It always contains a value —
   580→/// check `success` to determine the outcome.
   581→#[derive(Debug, Clone, Serialize, Deserialize)]
   582→pub struct CompileResult {
   583→    /// Whether compilation succeeded (produced a binary).
   584→    pub success: bool,
   585→    /// Any diagnostics (errors, warnings, notes) produced during compilation.
   586→    pub diagnostics: Vec<VumaDiagnostic>,
   587→    /// Summary of the SCG (Semantic Computation Graph).
   588→    ///
   589→    /// Present when parsing succeeds, even if codegen fails.
   590→    pub scg: Option<ScgSummary>,
   591→    /// Compiled output for the target.
   592→    ///
   593→    /// Present only when compilation succeeds.
   594→    pub target: Option<TargetOutput>,
   595→    /// Metadata about the compilation process.
   596→    pub metadata: CompileMetadata,
   597→}
   598→
   599→/// Result of parsing (without codegen).
   600→#[derive(Debug, Clone, Serialize, Deserialize)]
   601→pub struct ParseResult {
   602→    /// Whether parsing succeeded.
   603→    pub success: bool,
   604→    /// Any diagnostics produced during parsing.
   605→    pub diagnostics: Vec<VumaDiagnostic>,
   606→    /// Summary of the AST structure.
   607→    pub ast_summary: Option<AstSummary>,
   608→    /// Summary of the SCG (if AST → SCG conversion succeeded).
   609→    pub scg: Option<ScgSummary>,
   610→    /// Metadata about the parse process.
   611→    pub metadata: CompileMetadata,
   612→}

   726→/// Serialize `Vec<u8>` as a hex string for compact JSON representation.
   727→fn serialize_binary_hex<S: serde::Serializer>(data: &Vec<u8>, s: S) -> Result<S::Ok, S::Error> {
   728→    let hex: String = data.iter().map(|b| format!("{:02x}", b)).collect();
   729→    s.serialize_str(&hex)
   730→}
   731→
   732→/// Deserialize `Vec<u8>` from a hex string.
   733→fn deserialize_binary_hex<'de, D: serde::Deserializer<'de>>(d: D) -> Result<Vec<u8>, D::Error> {
   734→    use serde::de::Error;
   735→
   736→    struct HexVisitor;
   737→
   738→    impl<'de> serde::de::Visitor<'de> for HexVisitor {
   739→        type Value = Vec<u8>;
   740→
   741→        fn expecting(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
   742→            write!(f, "a hex-encoded string")
   743→        }
   744→
   745→        fn visit_str<E: serde::de::Error>(self, v: &str) -> Result<Self::Value, E> {
   746→            if v.len() % 2 != 0 {
   747→                return Err(E::custom("hex string has odd length"));
   748→            }
   749→            (0..v.len())
   750→                .step_by(2)
   751→                .map(|i| {
   752→                    u8::from_str_radix(&v[i..i + 2], 16)
   753→                        .map_err(|e| E::custom(format!("hex decode error: {}", e)))
   754→                })
   755→                .collect()
   756→        }
   757→    }
   758→
   759→    d.deserialize_str(HexVisitor)
   760→}

   908→    }

   915→/// Front-end pipeline result (everything up to and including SCG transforms).
   916→enum FrontendResult {
   917→    Ok {
   918→        scg: vuma_scg::SCG,
   919→    },
   920→    Err {
   921→        diagnostics: Vec<VumaDiagnostic>,
   922→    },
   923→}
   924→
   925→/// Run the front-end pipeline: Parse → AST → SCG → Validate → BD Inference
   926→/// → IVE Verification → SCG Transforms.
   927→fn run_frontend(source: &str, config: &CompileConfig) -> FrontendResult {
   928→    use vuma_ive::{InvariantAggregator, VerificationLevel as IveVerificationLevel};
   929→    use vuma_parser::{AstToScg, Parser};
   930→
   931→    // Stage 1: Parse
   932→    let mut parser = Parser::new(source);
   933→    let parse_output = parser.parse_program();
   934→    if parse_output.has_errors() {
   935→        return FrontendResult::Err {
   936→            diagnostics: diagnostics::from_parse_errors(&parse_output.errors, source, None),
   937→        };
   938→    }
   939→    let ast = parse_output.unwrap();
   940→
   941→    // Stage 2: AST → SCG
   942→    let mut converter = AstToScg::new();
   943→    let mut scg = match converter.convert(&ast) {
   944→        Ok(scg) => scg,
   945→        Err(e) => {
   946→            return FrontendResult::Err {
   947→                diagnostics: vec![VumaDiagnostic::new(
   948→                    "E019",
   949→                    DiagnosticSeverity::Error,
   950→                    format!("{}", e),
   951→                    "ast-to-scg",
   952→                    DiagnosticSourceLocation::unknown(),
   953→                )],
   954→            };
   955→        }
   956→    };
   957→
   958→    // Stage 3: SCG Validation (non-fatal — warn but continue)
   959→    let _validation = scg.validate();
   960→
   961→    // Stage 4: BD Inference (refine types)
   962→    let inference_engine = vuma_ive::InferenceEngine::new();
   963→    let bd_results = inference_engine.infer_types(&scg);
   964→    pipeline::refine_scg_types_with_bd(&mut scg, &bd_results);
   965→
   966→    // Stage 5: IVE Verification (non-fatal)
   967→    if config.verification_level != VerificationLevel::None {
   968→        let ive_level = match config.verification_level {
   969→            VerificationLevel::Quick => IveVerificationLevel::Quick,
   970→            VerificationLevel::Normal => IveVerificationLevel::Normal,
   971→            VerificationLevel::Exhaustive => IveVerificationLevel::Exhaustive,
   972→            VerificationLevel::None => unreachable!(),
   973→        };
   974→        let aggregator = InvariantAggregator::new().with_level(ive_level);
   975→        let input = vuma_ive::verification::VerificationInput::from_scg(scg.clone());
   976→        let _ = aggregator.verify_all(&input);
   977→    }
   978→
   979→    // Stage 6: SCG Transforms
   980→    pipeline::run_scg_transforms(&mut scg, config);
   981→
   982→    FrontendResult::Ok { scg }
   983→}

   985→/// Run target-specific codegen using the Backend trait.
   986→fn run_backend_codegen(
   987→    scg: &vuma_scg::SCG,
   988→    backend_kind: vuma_codegen::backend::BackendKind,
   989→) -> Result<TargetOutput, Vec<VumaDiagnostic>> {
   990→    use vuma_codegen::backend::{create_backend, AllocatedProgram};
   991→    use vuma_codegen::regalloc::RegAllocator;
   992→    use vuma_codegen::scg_to_ir::IRBuilder;
   993→
   994→    // Bridge SCG to codegen SCG
   995→    let codegen_scg = pipeline::bridge_scg_to_codegen(scg);
   996→
   997→    // IR Lowering
   998→    let mut ir_builder = IRBuilder::new();
   999→    let ir_program = match ir_builder.build(&codegen_scg) {
  1000→        Ok(ir) => ir,
  1001→        Err(e) => {
  1002→            return Err(vec![diagnostics::from_codegen_error(&e)]);
  1003→        }
  1004→    };
  1005→
  1006→    // Create backend
  1007→    let backend = match create_backend(backend_kind) {
  1008→        Ok(b) => b,
  1009→        Err(e) => {
  1010→            return Err(vec![VumaDiagnostic::new(
  1011→                "E023",
  1012→                DiagnosticSeverity::Error,
  1013→                format!("{}", e),
  1014→                "backend-creation",
  1015→                DiagnosticSourceLocation::unknown(),
  1016→            )]);
  1017→        }
  1018→    };
  1019→
  1020→    // Register allocation — delegate to the backend
  1021→    let mut allocated_functions = Vec::new();
  1022→    for func in &ir_program.functions {
  1023→        match backend.allocate_registers(func) {
  1024→            Ok(allocd) => allocated_functions.push(allocd),
  1025→            Err(e) => {
  1026→                return Err(vec![VumaDiagnostic::new(
  1027→                    "E017",
  1028→                    DiagnosticSeverity::Error,
  1029→                    format!("{}: {}", func.name, e),
  1030→                    "register-alloc",
  1031→                    DiagnosticSourceLocation::unknown(),
  1032→                )]);
  1033→            }
  1034→        }
  1035→    }
  1036→
  1037→    let allocated_program = AllocatedProgram {
  1038→        functions: allocated_functions,
  1039→        total_code_size: 0,
  1040→        total_data_size: 0,
  1041→    };
  1042→
  1043→    // Encode the program
  1044→    let binary = match backend.encode_program(&allocated_program) {
  1045→        Ok(binary) => binary,
  1046→        Err(e) => {
  1047→            return Err(vec![VumaDiagnostic::new(
  1048→                "E020",
  1049→                DiagnosticSeverity::Error,
  1050→                format!("{}", e),
  1051→                "code-emission",
  1052→                DiagnosticSourceLocation::unknown(),
  1053→            )]);
  1054→        }
  1055→    };
  1056→
  1057→    // Disassemble
  1058→    let base_addr = backend.target_info().default_base_address();
  1059→    let disasm_lines = backend.disassemble(&binary, base_addr);
  1060→    let disassembly = disasm_lines.join("\n");
  1061→
  1062→    Ok(TargetOutput {
  1063→        backend: backend_kind.isa_name().to_string(),
  1064→        binary_size: binary.len(),
  1065→        binary,
  1066→        disassembly,
  1067→    })
  1068→}

  1070→/// Build an edge index from the SCG for efficient traversal.
  1071→struct ScgEdgeIndex {
  1072→    outgoing: HashMap<vuma_scg::NodeId, Vec<vuma_scg::EdgeData>>,
  1073→    incoming: HashMap<vuma_scg::NodeId, Vec<vuma_scg::EdgeData>>,
  1074→}
  1075→
  1076→impl ScgEdgeIndex {
  1077→    fn build(scg: &vuma_scg::SCG) -> Self {
  1078→        let mut outgoing: HashMap<vuma_scg::NodeId, Vec<vuma_scg::EdgeData>> = HashMap::new();
  1079→        let mut incoming: HashMap<vuma_scg::NodeId, Vec<vuma_scg::EdgeData>> = HashMap::new();
  1080→        for edge in scg.edges() {
  1081→            outgoing.entry(edge.source).or_default().push(edge.clone());
  1082→            incoming.entry(edge.target).or_default().push(edge.clone());
  1083→        }
  1084→        Self { outgoing, incoming }
  1085→    }
  1086→
  1087→    fn outgoing(&self, id: vuma_scg::NodeId) -> &[vuma_scg::EdgeData] {
  1088→        self.outgoing.get(&id).map(|v| v.as_slice()).unwrap_or(&[])
  1089→    }
  1090→
  1091→    fn incoming(&self, id: vuma_scg::NodeId) -> &[vuma_scg::EdgeData] {
  1092→        self.incoming.get(&id).map(|v| v.as_slice()).unwrap_or(&[])
  1093→    }
  1094→}
  1095→
  1096→/// Build an SCG summary from the full SCG graph.
  1097→fn build_scg_summary(scg: &vuma_scg::SCG) -> ScgSummary {
  1098→    use vuma_scg::{ControlKind, EdgeKind, NodePayload};
  1099→
  1100→    let total_nodes = scg.node_count();
  1101→    let total_edges = scg.edge_count();
  1102→    let edge_idx = ScgEdgeIndex::build(scg);
  1103→
  1104→    // Identify function entry nodes and build per-function summaries
  1105→    let mut functions = Vec::new();
  1106→
  1107→    // Collect function entries
  1108→    let func_entries: Vec<(vuma_scg::NodeId, String)> = scg
  1109→        .nodes()
  1110→        .filter_map(|n| {
  1111→            if let NodePayload::Control(c) = &n.payload {
  1112→                if c.kind == ControlKind::FunctionEntry {
  1113→                    let name = c.label.clone().unwrap_or_else(|| "unknown".to_string());
  1114→                    return Some((n.id, name));
  1115→                }
  1116→            }
  1117→            None
  1118→        })
  1119→        .collect();
  1120→
  1121→    // For each function entry, count nodes reachable via ControlFlow
  1122→    for (entry_id, func_name) in &func_entries {
  1123→        let mut reachable = std::collections::HashSet::new();
  1124→        let mut queue = std::collections::VecDeque::new();
  1125→        queue.push_back(*entry_id);
  1126→        reachable.insert(*entry_id);
  1127→
  1128→        while let Some(current) = queue.pop_front() {
  1129→            for edge in edge_idx.outgoing(current) {
  1130→                if edge.kind == EdgeKind::ControlFlow && !reachable.contains(&edge.target) {
  1131→                    reachable.insert(edge.target);
  1132→                    queue.push_back(edge.target);
  1133→                }
  1134→            }
  1135→        }
  1136→
  1137→        let node_count = reachable.len();
  1138→
  1139→        // Extract parameters from outgoing DataFlow edges from the entry
  1140→        let mut params = Vec::new();
  1141→        for edge in edge_idx.outgoing(*entry_id) {
  1142→            if edge.kind == EdgeKind::DataFlow {
  1143→                if let Some(target_node) = scg.get_node(edge.target) {
  1144→                    let name = match &target_node.payload {
  1145→                        NodePayload::Allocation(a) => {
  1146→                            a.type_name.clone().unwrap_or_else(|| "param".to_string())
  1147→                        }
  1148→                        NodePayload::Computation(c) => {
  1149→                            c.result_type.clone().unwrap_or_else(|| "param".to_string())
  1150→                        }
  1151→                        _ => "param".to_string(),
  1152→                    };
  1153→                    let ty = match &target_node.payload {
  1154→                        NodePayload::Allocation(a) => {
  1155→                            a.type_name.as_deref().unwrap_or("i64").to_string()
  1156→                        }
  1157→                        NodePayload::Computation(c) => {
  1158→                            c.result_type.as_deref().unwrap_or("i64").to_string()
  1159→                        }
  1160→                        NodePayload::Cast(c) => c.to_type.clone(),
  1161→                        _ => "i64".to_string(),
  1162→                    };
  1163→                    params.push((name, ty));
  1164→                }
  1165→            }
  1166→        }
  1167→
  1168→        // Find calls within this function
  1169→        let mut calls = Vec::new();
  1170→        for node_id in &reachable {
  1171→            if let Some(node) = scg.get_node(*node_id) {
  1172→                if let NodePayload::Computation(comp) = &node.payload {
  1173→                    if !is_known_binop(&comp.operation) && !comp.operation.starts_with('_') {
  1174→                        if !calls.contains(&comp.operation) {
  1175→                            calls.push(comp.operation.clone());
  1176→                        }
  1177→                    }
  1178→                }
  1179→            }
  1180→        }
  1181→
  1182→        // Determine return type
  1183→        let return_type = find_return_type(scg, *entry_id, &edge_idx);
  1184→
  1185→        functions.push(FunctionSummary {
  1186→            name: func_name.clone(),
  1187→            params,
  1188→            return_type,
  1189→            node_count,
  1190→            calls,
  1191→        });
  1192→    }
  1193→
  1194→    // If no function entries were found, create a single "main" summary
  1195→    if functions.is_empty() {
  1196→        let mut calls = Vec::new();
  1197→        for node in scg.nodes() {
  1198→            if let NodePayload::Computation(comp) = &node.payload {
  1199→                if !is_known_binop(&comp.operation) && !comp.operation.starts_with('_') {
  1200→                    if !calls.contains(&comp.operation) {
  1201→                        calls.push(comp.operation.clone());
  1202→                    }
  1203→                }
  1204→            }
  1205→        }
  1206→
  1207→        functions.push(FunctionSummary {
  1208→            name: "main".to_string(),
  1209→            params: Vec::new(),
  1210→            return_type: "void".to_string(),
  1211→            node_count: total_nodes,
  1212→            calls,
  1213→        });
  1214→    }
  1215→
  1216→    ScgSummary {
  1217→        function_count: functions.len(),
  1218→        functions,
  1219→        total_nodes,
  1220→        total_edges,
  1221→    }
  1222→}
  1223→
  1224→/// Check if an operation string is a known binary operation.
  1225→fn is_known_binop(op: &str) -> bool {
  1226→    matches!(
  1227→        op,
  1228→        "add" | "sub" | "mul" | "sdiv" | "udiv" | "srem" | "urem" | "and" | "or" | "xor"
  1229→            | "shl" | "shr.l" | "shr.a" | "slt" | "sle" | "sgt" | "sge" | "ult" | "ule"
  1230→            | "ugt" | "uge" | "eq" | "ne" | "+" | "-" | "*" | "/" | "%" | "&" | "|"
  1231→            | "^" | "<<" | ">>" | "<" | "<=" | ">" | ">=" | "=="
  1232→    )
  1233→}

  1235→/// Find the return type of a function by tracing to its FunctionReturn node.
  1236→fn find_return_type(
  1237→    scg: &vuma_scg::SCG,
  1238→    entry_id: vuma_scg::NodeId,
  1239→    edge_idx: &ScgEdgeIndex,
  1240→) -> String {
  1241→    use vuma_scg::{ControlKind, EdgeKind, NodePayload};
  1242→
  1243→    let mut visited = std::collections::HashSet::new();
  1244→    let mut queue = std::collections::VecDeque::new();
  1245→    queue.push_back(entry_id);
  1246→    visited.insert(entry_id);
  1247→
  1248→    while let Some(current) = queue.pop_front() {
  1249→        for edge in edge_idx.outgoing(current) {
  1250→            if edge.kind == EdgeKind::ControlFlow && !visited.contains(&edge.target) {
  1251→                visited.insert(edge.target);
  1252→                if let Some(node) = scg.get_node(edge.target) {
  1253→                    if let NodePayload::Control(c) = &node.payload {
  1254→                        if c.kind == ControlKind::FunctionReturn {
  1255→                            // Check incoming DataFlow edges for return value type
  1256→                            for ret_edge in edge_idx.incoming(edge.target) {
  1257→                                if ret_edge.kind == EdgeKind::DataFlow {
  1258→                                    if let Some(src) = scg.get_node(ret_edge.source) {
  1259→                                        if let NodePayload::Computation(comp) = &src.payload {
  1260→                                            if let Some(rt) = &comp.result_type {
  1261→                                                return rt.clone();
  1262→                                            }
  1263→                                        }
  1264→                                    }
  1265→                                }
  1266→                            }
  1267→                            return "void".to_string();
  1268→                        }
  1269→                    }
  1270→                }
  1271→                queue.push_back(edge.target);
  1272→            }
  1273→        }
  1274→    }
  1275→
  1276→    "void".to_string()
  1277→}

  1279→/// Build an AST summary from the parsed program.
  1280→fn build_ast_summary(ast: &vuma_parser::Program) -> AstSummary {
  1281→    use vuma_parser::Item;
  1282→
  1283→    let mut function_names = Vec::new();
  1284→    let mut region_names = Vec::new();
  1285→    let mut import_count = 0;
  1286→
  1287→    for item in &ast.items {
  1288→        match item {
  1289→            Item::FnDef(fn_decl) => {
  1290→                function_names.push(fn_decl.name.clone());
  1291→            }
  1292→            Item::RegionDef(region_decl) => {
  1293→                region_names.push(region_decl.name.clone());
  1294→            }
  1295→            Item::Import(_) => {
  1296→                import_count += 1;
  1297→            }
  1298→            _ => {}
  1299→        }
  1300→    }
  1301→
  1302→    AstSummary {
  1303→        item_count: ast.items.len(),
  1304→        function_names,
  1305→        region_names,
  1306→        import_count,
  1307→    }
  1308→}

  1310→/// Disassemble the default (AArch64) binary output.
  1311→fn disassemble_default(binary: &[u8]) -> String {
  1312→    use vuma_codegen::backend::{create_backend, BackendKind};
  1313→
  1314→    match create_backend(BackendKind::AArch64) {
  1315→        Ok(backend) => {
  1316→            let base = backend.target_info().default_base_address();
  1317→            backend.disassemble(binary, base).join("\n")
  1318→        }
  1319→        Err(_) => format!("({} bytes of binary output)", binary.len()),
  1320→    }
  1321→}

  1344→/// Build a proof-system counterexample from an IVE verification result.
  1345→///
  1346→/// Takes the IVE `VerificationResult` (which uses its own counterexample
  1347→/// format) and converts it into a proof-system `CounterExample`, then
  1348→/// extracts the relevant information into the API's `CounterexampleInfo`.
  1349→fn build_proof_counterexample(
  1350→    result: &vuma_ive::result::VerificationResult,
  1351→) -> CounterexampleInfo {
  1352→    use vuma_ive::result::VerificationStatus;
  1353→
  1354→    match &result.status {
  1355→        VerificationStatus::Violated { counterexample } => {
  1356→            // Convert the IVE counterexample into a proof-system
  1357→            // counterexample for structural consistency.
  1358→            let proof_inv = match result.invariant.as_str() {
  1359→                "liveness" => vuma_proof::proof::InvariantName::Liveness,
  1360→                "exclusivity" => vuma_proof::proof::InvariantName::Exclusivity,
  1361→                "cleanup" => vuma_proof::proof::InvariantName::Cleanup,
  1362→                "origin" => vuma_proof::proof::InvariantName::Origin,
  1363→                "interpretation" => vuma_proof::proof::InvariantName::Interpretation,
  1364→                _ => vuma_proof::proof::InvariantName::Liveness,
  1365→            };
  1366→
  1367→            let violation_point = ViolationPoint::new(
  1368→                proof_inv,
  1369→                &counterexample.description,
  1370→                0, // program offset
  1371→            );
  1372→            let proof_ce = ProofCounterExample::from_violation(&result.message, violation_point);
  1373→            let minimal_ce = proof_ce.minimal();
  1374→
  1375→            // Convert trace steps to human-readable strings.
  1376→            let trace: Vec<String> = minimal_ce
  1377→                .execution
  1378→                .iter()
  1379→                .map(|step| step.to_string())
  1380→                .collect();
  1381→
  1382→            CounterexampleInfo {
  1383→                description: counterexample.description.clone(),
  1384→                execution_trace: trace,
  1385→            }
  1386→        }
  1387→        _ => CounterexampleInfo {
  1388→            description: result.message.clone(),
  1389→            execution_trace: Vec::new(),
  1390→        },
  1391→    }
  1392→}
