     1→     1→//! # ABI Conformance Testing
     2→     2→//!
     3→     3→//! Verifies that each backend follows its platform's calling convention by:
     4→     4→//! 1. Creating IR functions with varying argument counts
     5→     5→//! 2. Running `allocate_registers` on each backend
     6→     6→//! 3. Checking the generated code uses correct registers via:
     7→     7→//!    - TargetInfo trait (calling convention metadata)
     8→     8→//!    - AllocatedFunction register assignments
     9→     9→//!    - Disassembled output verification
    10→    10→//!
    11→    11→//! ## Calling Conventions Tested
    12→    12→//!
    13→    13→//! | Backend      | ABI          | Integer Args        | Return  | Address |
    14→    14→//! |--------------|--------------|---------------------|---------|---------|
    15→    15→//! | x86_64       | System V     | RDI,RSI,RDX,RCX,R8,R9 | RAX   | 8 bytes |
    16→    16→//! | AArch64      | AAPCS64      | X0-X7               | X0      | 8 bytes |
    17→    17→//! | RISC-V 64    | RV64G LP64D  | A0-A7 (x10-x17)    | A0      | 8 bytes |
    18→    18→//! | ARM32        | AAPCS        | R0-R3               | R0      | 4 bytes |
    19→    19→//! | MIPS64       | N64          | $a0-$a7 ($4-$11)    | $v0     | 8 bytes |
    20→    20→//! | PPC64        | ELFv2        | R3-R10              | R3      | 8 bytes |
    21→    21→//! | LoongArch64  | LP64         | $a0-$a7 ($r4-$r11)  | $a0     | 8 bytes |
    22→    22→//! | Wasm32       | Stack machine | (stack params)     | (stack) | 4 bytes |
    23→    23→

    34→    34→
    35→    35→/// Build a simple IR function named `name` with `n` i64 parameters that
    36→    36→/// returns the first parameter (or 0 if no params).
    37→    37→///
    38→    38→/// The function body simply returns `params[0]` if available, otherwise
    39→    39→/// an immediate 0.  This is sufficient to exercise the calling convention
    40→    40→/// for argument passing and return-value placement.
    41→    41→fn make_func_with_n_args(name: &str, n: usize) -> IRFunction {
    42→    42→    let mut func = IRFunction::new(name);
    43→    43→    // Register parameters as vregs 0..n
    44→    44→    for i in 0..n {
    45→    45→        func.param_types.push(IRType::I64);
    46→    46→        func.params.push(IRValue::Register(i as u32));
    47→    47→        func.vregs.insert(i as u32, vuma_codegen::ir::VirtualRegister::named(i as u32, format!("a{}", i)));
    48→    48→    }
    49→    49→    func.result_types.push(IRType::I64);
    50→    50→    func.results.push(IRValue::Register(n as u32));
    51→    51→
    52→    52→    // Return the first parameter (or 0)
    53→    53→    let ret_val = if n > 0 {
    54→    54→        IRValue::Register(0)
    55→    55→    } else {
    56→    56→        IRValue::Immediate(0)
    57→    57→    };
    58→    58→    func.current_block().terminator = IRTerminator::Return(vec![ret_val]);
    59→    59→    func
    60→    60→}
    61→    61→
    62→    62→/// Build a function that calls another function with `n` i64 arguments.
    63→    63→/// This exercises argument *passing* in the calling convention.
    64→    64→fn make_func_with_call_n_args(n: usize) -> IRFunction {
    65→    65→    let mut func = IRFunction::new("caller");
    66→    66→    // vreg 0 = the call result
    67→    67→    func.vregs.insert(0, vuma_codegen::ir::VirtualRegister::anonymous(0));
    68→    68→
    69→    69→    // Build argument values: all immediates 1..=n
    70→    70→    let args: Vec<IRValue> = (1..=n as i64).map(IRValue::Immediate).collect();
    71→    71→
    72→    72→    func.current_block().instructions.push(IRInstr::Call {
    73→    73→        dst: Some(IRValue::Register(0)),
    74→    74→        func: "callee".to_string(),
    75→    75→        args,
    76→    76→        is_extern: false,
    77→    77→    });
    78→    78→    func.current_block().terminator = IRTerminator::Return(vec![IRValue::Register(0)]);
    79→    79→    func
    80→    80→}

    83→    83→// TargetInfo-based calling convention validation
    84→    84→// ===========================================================================
    85→    85→
    86→    86→/// Validate that every TargetInfo reports internally-consistent calling
    87→    87→/// convention properties.
    88→    88→fn validate_cc_info(info: &dyn TargetInfo) {
    89→    89→    let isa = info.isa_name();
    90→    90→
    91→    91→    // num_int_arg_regs must be consistent with has_registers
    92→    92→    if !info.has_registers() {
    93→    93→        assert_eq!(
    94→    94→            info.num_int_arg_regs(), 0,
    95→    95→            "{}: has_registers=false but num_int_arg_regs={}",
    96→    96→            isa, info.num_int_arg_regs()
    97→    97→        );
    98→    98→    } else {
    99→    99→        assert!(
   100→   100→            info.num_int_arg_regs() > 0,
   101→   101→            "{}: register-based ISA must have at least 1 integer arg register",
   102→   102→            isa
   103→   103→        );
   104→   104→    }
   105→   105→
   106→   106→    // calling_convention_name must be non-empty
   107→   107→    assert!(
   108→   108→        !info.calling_convention_name().is_empty(),
   109→   109→        "{}: calling convention name must not be empty",
   110→   110→        isa
   111→   111→    );
   112→   112→
   113→   113→    // stack_alignment must be a power of 2
   114→   114→    let sa = info.stack_alignment();
   115→   115→    assert!(sa > 0 && sa.is_power_of_two(),
   116→   116→        "{}: stack alignment {} must be a positive power of 2", isa, sa);
   117→   117→
   118→   118→    // pointer_width must match output format
   119→   119→    match info.output_format() {
   120→   120→        OutputFormat::Elf32 => assert_eq!(info.pointer_width(), 4,
   121→   121→            "{}: Elf32 output must have 4-byte pointers", isa),
   122→   122→        OutputFormat::Elf64 => assert_eq!(info.pointer_width(), 8,
   123→   123→            "{}: Elf64 output must have 8-byte pointers", isa),
   124→   124→        OutputFormat::WasmBinary => assert_eq!(info.pointer_width(), 4,
   125→   125→            "{}: Wasm32 must have 4-byte pointers", isa),
   126→   126→        OutputFormat::RawBinary => {} // no constraint
   127→   127→    }
   128→   128→}

   385→   385→/// Verify that every backend can allocate a function with arguments
   386→   386→/// exceeding the register count (stack args needed).
   387→   387→#[test]
   388→   388→fn test_all_backends_stack_args() {
   389→   389→    for kind in [
   390→   390→        BackendKind::AArch64,
   391→   391→        BackendKind::X86_64,
   392→   392→        BackendKind::RiscV64,
   393→   393→        BackendKind::Arm32,
   394→   394→        BackendKind::Mips64,
   395→   395→        BackendKind::PowerPC64,

   512→   512→/// Check that the return value from a function uses the correct register class.
   513→   513→/// For all register-based backends, the return value should be in a GPR.
   514→   514→///
   515→   515→/// Some backends populate the `reads`/`writes` fields of AllocatedInstruction;
   516→   516→/// others may leave them empty. This test validates that backends which DO
   517→   517→/// populate these fields use GPRs for integer return values.
   518→   518→#[test]
   519→   519→fn test_all_backends_return_in_gpr() {
   520→   520→    for kind in [
   521→   521→        BackendKind::AArch64,
   522→   522→        BackendKind::X86_64,
   523→   523→        BackendKind::RiscV64,
   524→   524→        BackendKind::Arm32,
   525→   525→        BackendKind::Mips64,
   526→   526→        BackendKind::PowerPC64,
   527→   527→        BackendKind::LoongArch64,
   528→   528→    ] {
   529→   529→        let backend = create_backend(kind).unwrap();
   530→   530→        let func = make_func_with_n_args("ret_test", 1);
   531→   531→        if let Ok(af) = backend.allocate_registers(&func) {
   532→   532→            // Collect all register writes across all instructions
   533→   533→            let all_writes: Vec<_> = af.blocks.iter()
   534→   534→                .flat_map(|b| b.instructions.iter())
   535→   535→                .flat_map(|i| i.writes.iter())
   536→   536→                .collect();
   537→   537→
   538→   538→            // If the backend populates writes, verify that at least one GPR is used
   539→   539→            // for a function returning i64. If writes are empty, the backend uses
   540→   540→            // a different encoding strategy (direct machine code), which is fine.
   541→   541→            if !all_writes.is_empty() {
   542→   542→                let has_gpr_writes: Vec<_> = all_writes.iter()
   543→   543→                    .filter(|r| r.class == RegClass::Gpr)
   544→   544→                    .collect();
   545→   545→                assert!(
   546→   546→                    !has_gpr_writes.is_empty(),
   547→   547→                    "{}: function returning i64 must write to at least one GPR",
   548→   548→                    kind.isa_name()
   549→   549→                );
   550→   550→            }
   551→   551→            // Regardless, the encoded output should be non-empty
   552→   552→            if let Ok(bytes) = backend.encode_function(&af) {
   553→   553→                assert!(!bytes.is_empty(), "{}: encoded output must not be empty", kind.isa_name());
   554→   554→            }
   555→   555→        }
   556→   556→    }
   557→   557→}
