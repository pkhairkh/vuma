     1→     1→//! # ARM64 Code Emission
     2→     2→//!
     3→     3→//! Lowers IR to ARM64 machine code and produces ELF binaries or raw binaries
     4→     4→//! suitable for the Raspberry Pi 5 (Cortex-A76, ARMv8.2-A).
     5→     5→//!
     6→     6→//! ## Pipeline
     7→     7→//!
     8→     8→//! 1. **IR → ARM64 Instructions**: Each IR instruction is pattern-matched and
     9→     9→//!    lowered to one or more [`Instruction`]
    10→    10→//!    values, with virtual registers replaced by physical registers from the
    11→    11→//!    register allocator.
    12→    12→//! 2. **Emit function**: Each IR function is emitted as a sequence of 32-bit
    13→    13→//!    ARM64 code words.
    14→    14→//! 3. **Emit program**: All functions are collected, data sections are laid
    15→    15→//!    out, and a minimal ELF64 binary is produced.
    16→    16→//!
    17→    17→//! ## Output Formats
    18→    18→//!
    19→    19→//! - **ELF**: Full ELF64 binary with program headers, section headers, symbol
    20→    20→//!   table, and string table.  Suitable for Linux on Pi 5.
    21→    21→//! - **Raw**: Flat binary image for bare-metal Pi 5 (loaded at 0x80000).
    22→    22→//! - **Obj**: Relocatable ELF object file (`ET_REL`) for linking.
    23→    23→//!
    24→    24→//! ## ELF Layout (executable)
    25→    25→//!
    26→    26→//! ```text
    27→    27→//! ┌─────────────────────┐
    28→    28→//! │ ELF Header           │  64 bytes
    29→    29→//! ├─────────────────────┤
    30→    30→//! │ Program Headers      │  2 × 56 bytes (LOAD segments)
    31→    31→//! ├─────────────────────┤
    32→    32→//! │ .text                │  emitted code
    33→    33→//! ├─────────────────────┤
    34→    34→//! │ .rodata              │  read-only data
    35→    35→//! ├─────────────────────┤
    36→    36→//! │ .data                │  initialized data
    37→    37→//! ├─────────────────────┤
    38→    38→//! │ .symtab              │  symbol table entries
    39→    39→//! ├─────────────────────┤
    40→    40→//! │ .strtab              │  symbol string table
    41→    41→//! ├─────────────────────┤
    42→    42→//! │ .shstrtab            │  section header string table
    43→    43→//! ├─────────────────────┤
    44→    44→//! │ Section Headers      │  N × 64 bytes
    45→    45→//! └─────────────────────┘
    46→    46→//! .bss is virtual-only (memsz > filesz in the data LOAD segment).
    47→    47→//! ```
    48→    48→//!
    49→    49→//! ## Relocation Support
    50→    50→//!
    51→    51→//! Inter-function calls (`BL`) are resolved in a fixup pass after all
    52→    52→//! functions have been emitted.  The emitter records the word offset of each
    53→    53→//! `BL` instruction and the target function name; once function addresses
    54→    54→//! are known, the branch offsets are patched into the encoded instructions.
    55→    55→

   471→   471→            self.base_addr
   472→   472→        } else {
   473→   473→            match self.target {
   474→   474→                Target::Linux => BASE_ADDR_LINUX,
   475→   475→                Target::BareMetal => BASE_ADDR_BARE,
   476→   476→            }
   477→   477→        }
   478→   478→    }
   479→   479→}
   480→   480→
   481→   481→impl Default for EmitConfig {
   482→   482→    fn default() -> Self {
   483→   483→        Self::linux_elf()
   484→   484→    }
   485→   485→}

   487→   487→// ---------------------------------------------------------------------------
   488→   488→// Inter-function call relocation record
   489→   489→// ---------------------------------------------------------------------------
   490→   490→
   491→   491→/// A record for a `BL` instruction that needs to be patched with the address
   492→   492→/// of a named function.  After all functions are emitted, these records are
   493→   493→/// resolved to compute the correct branch offset.
   494→   494→#[derive(Debug, Clone)]
   495→   495→struct CallRelocation {
   496→   496→    /// Byte offset within the text section where the BL instruction lives.
   497→   497→    text_byte_offset: u64,
   498→   498→    /// Name of the target function.
   499→   499→    target_func: String,
   500→   500→}
   501→   501→
   502→   502→// ---------------------------------------------------------------------------
   503→   503→// ELF RelaEntry
   504→   504→// ---------------------------------------------------------------------------
   505→   505→
   506→   506→/// An ELF64 relocation entry with an explicit addend (SHT_RELA).
   507→   507→///
   508→   508→/// Each entry specifies where in the section a relocation must be applied,
   509→   509→/// which symbol the relocation references, the type of relocation, and an
   510→   510→/// addend value used in the relocation computation.
   511→   511→#[derive(Debug, Clone)]
   512→   512→pub struct RelaEntry {
   513→   513→    /// Byte offset within the section where the relocation applies.
   514→   514→    pub offset: u64,
   515→   515→    /// Symbol index (upper 32 bits) and relocation type (lower 32 bits),
   516→   516→    /// packed as: `(sym_idx << 32) | r_type`.
   517→   517→    pub info: u64,
   518→   518→    /// Addend value used in the relocation computation.
   519→   519→    pub addend: i64,
   520→   520→}
   521→   521→
   522→   522→impl RelaEntry {
   523→   523→    /// Create a new relocation entry.
   524→   524→    pub fn new(offset: u64, sym_idx: u32, r_type: u32, addend: i64) -> Self {
   525→   525→        Self {
   526→   526→            offset,
   527→   527→            info: ((sym_idx as u64) << 32) | (r_type as u64),
   528→   528→            addend,
   529→   529→        }
   530→   530→    }
   531→   531→
   532→   532→    /// Extract the symbol index from `info`.
   533→   533→    pub fn sym_idx(&self) -> u32 {
   534→   534→        (self.info >> 32) as u32
   535→   535→    }
   536→   536→
   537→   537→    /// Extract the relocation type from `info`.
   538→   538→    pub fn r_type(&self) -> u32 {
   539→   539→        (self.info & 0xFFFFFFFF) as u32
   540→   540→    }
   541→   541→
   542→   542→    /// Serialize the entry to 24 bytes (little-endian).
   543→   543→    pub fn to_bytes(&self) -> [u8; 24] {
   544→   544→        let mut buf = [0u8; 24];
   545→   545→        buf[0..8].copy_from_slice(&self.offset.to_le_bytes());
   546→   546→        buf[8..16].copy_from_slice(&self.info.to_le_bytes());
   547→   547→        buf[16..24].copy_from_slice(&self.addend.to_le_bytes());
   548→   548→        buf
   549→   549→    }
   550→   550→}
   551→   551→
   552→   552→// ---------------------------------------------------------------------------
   553→   553→// Emitter
   554→   554→// ---------------------------------------------------------------------------
   555→   555→
   556→   556→/// The ARM64 code emitter.
   557→   557→///
   558→   558→/// Holds state for the current emission context: the register allocator,
   559→   559→/// accumulated code, and fixup records for branch targets.
   560→   560→pub struct Emitter {
   561→   561→    /// Register allocator used during emission.
   562→   562→    reg_alloc: RegAllocator,
   563→   563→    /// Accumulated machine code for the current function (32-bit words).
   564→   564→    code: Vec<u32>,
   565→   565→    /// Fixup records for intra-function branches: (word index, target label name, format).
   566→   566→    fixups: Vec<(usize, String, BranchFormat)>,
   567→   567→    /// Map from label name to code offset (in words) within the current function.
   568→   568→    label_offsets: HashMap<String, usize>,
   569→   569→    /// Inter-function call relocations for the current function.
   570→   570→    call_relocs: Vec<CallRelocation>,
   571→   571→    /// Relocation entries using the `RelocationEntry` infrastructure.
   572→   572→    ///
   573→   573→    /// Each entry records a byte offset within the function's encoded output
   574→   574→    /// where a symbolic reference must be resolved, the target symbol name,
   575→   575→    /// and the ISA-specific relocation type (e.g., `"R_AARCH64_CALL26"`).
   576→   576→    relocations: Vec<RelocationEntry>,
   577→   577→    /// Name of the function currently being emitted.
   578→   578→    current_func_name: String,
   579→   579→    /// Byte offset of the current function within the text section (set externally).
   580→   580→    func_text_offset: u64,
   581→   581→    /// Computed stack frame size (in bytes) for the current function.
   582→   582→    frame_size: u32,
   583→   583→    /// Registers pinned for the current instruction (auto-unpinned after each instruction).
   584→   584→    instr_pinned_regs: Vec<Register>,
   585→   585→}
   586→   586→
   587→   587→impl Emitter {
   588→   588→    /// Create a new emitter with a fresh register allocator.
   589→   589→    pub fn new() -> Self {
   590→   590→        Self {
   591→   591→            reg_alloc: RegAllocator::new(),
   592→   592→            code: Vec::new(),
   593→   593→            fixups: Vec::new(),
   594→   594→            label_offsets: HashMap::new(),
   595→   595→            call_relocs: Vec::new(),
   596→   596→            relocations: Vec::new(),
   597→   597→            current_func_name: String::new(),
   598→   598→            func_text_offset: 0,
   599→   599→            frame_size: 0,
   600→   600→            instr_pinned_regs: Vec::new(),
   601→   601→        }
   602→   602→    }
   603→   603→
   604→   604→    // -----------------------------------------------------------------------
   605→   605→    // Function emission
   606→   606→    // -----------------------------------------------------------------------
   607→   607→
   608→   608→    /// Emit a single IR function to ARM64 machine code.
   609→   609→    ///
   610→   610→    /// For functions with more than `STACK_SLOT_VREG_THRESHOLD` virtual
   611→   611→    /// registers, the stack-slot emitter is used (every vreg gets a stack
   612→   612→    /// slot, scratch registers only).  For simpler functions, the greedy
   613→   613→    /// register allocator is used.
   614→   614→    ///
   615→   615→    /// Returns a vector of 32-bit ARM64 instruction words.
   616→   616→    pub fn emit_function(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   617→   617→        let vreg_count = count_vregs(func);
   618→   618→        if vreg_count > STACK_SLOT_VREG_THRESHOLD {
   619→   619→            self.emit_function_stack_slot(func)
   620→   620→        } else {
   621→   621→            self.emit_function_greedy(func)
   622→   622→        }
   623→   623→    }
   624→   624→
   625→   625→    /// Emit a single IR function using the greedy register allocator.
   626→   626→    ///
   627→   627→    /// This is the original emission strategy — suitable for functions with
   628→   628→    /// a small number of virtual registers.
   629→   629→    fn emit_function_greedy(&mut self, func: &IRFunction) -> Result<Vec<u32>> {
   630→   630→        self.code.clear();
   631→   631→        self.fixups.clear();
   632→   632→        self.label_offsets.clear();
   633→   633→        self.call_relocs.clear();
   634→   634→        self.relocations.clear();
   635→   635→        self.current_func_name = func.name.clone();
   636→   636→        self.reg_alloc.reset();
   637→   637→
   638→   638→        // Pre-allocate registers for parameters (AAPCS64: X0–X7).
   639→   639→        // We must explicitly assign parameter virtual registers to the
   640→   640→        // correct argument registers so the calling convention is respected.
   641→   641→        let arg_regs = [
   642→   642→            Register::X0, Register::X1, Register::X2, Register::X3,
   643→   643→            Register::X4, Register::X5, Register::X6, Register::X7,
   644→   644→        ];
   645→   645→        for (i, param) in func.params.iter().enumerate() {
   646→   646→            if let IRValue::Register(vreg_id) = param {
   647→   647→                if i < 8 {
   648→   648→                    self.reg_alloc.preassign(*vreg_id, arg_regs[i]);
   649→   649→                }
   650→   650→            }
   651→   651→        }
   652→   652→
   653→   653→        // Emit prologue:
   654→   654→        // 1. SUB SP, SP, #16          (make room for FP/LR save pair)
   655→   655→        // 2. STP X29, X30, [SP]       (save FP and LR at new SP)
   656→   656→        // 3. ADD X29, SP, #0           (set frame pointer to new SP)
   657→   657→        // 4. SUB SP, SP, #frame_size   (make room for local variables)
   658→   658→        //
   659→   659→        // NOTE: We cannot use pre-indexed STP [SP, #-16]! because the
   660→   660→        // Instruction::STP encoding only supports signed-offset mode.
   661→   661→        // Instead, we explicitly decrement SP with SUB before the STP.
   662→   662→        self.emit_instruction(Instruction::SUB {
   663→   663→            rd: Register::SP,
   664→   664→            rn: Register::SP,
   665→   665→            rm: Operand::Imm12(16),
   666→   666→        })?;
   667→   667→        self.emit_instruction(Instruction::STP {
   668→   668→            rt1: Register::X29,
   669→   669→            rt2: Register::X30,
   670→   670→            rn: Register::SP,
   671→   671→            offset: 0,
   672→   672→        })?;
   673→   673→
   674→   674→        // MOV X29, SP (set frame pointer)
   675→   675→        // IMPORTANT: Cannot use MOV (ORR Xd, XZR, SP) because ORR treats
   676→   676→        // Rm=31 as XZR, yielding zero. Use ADD X29, SP, #0 instead.
   677→   677→        self.emit_instruction(Instruction::ADD {
   678→   678→            rd: Register::X29,
   679→   679→            rn: Register::SP,
   680→   680→            rm: Operand::Imm12(0),
   681→   681→        })?;
   682→   682→
   683→   683→        // Reserve space for spill slots only. Each Alloc instruction handles
   684→   684→        // its own SUB SP, SP, #aligned_size. The epilogue will restore only
   685→   685→        // the spill area reservation. The Alloc areas are NOT restored by the
   686→   686→        // epilogue — they're effectively "leaked" stack space. This is fine
   687→   687→        // because the function's stack frame is restored by the LDP + ADD SP, #16.
   688→   688→        //
   689→   689→        // For spill slots, we need to estimate the maximum number of spills
   690→   690→        // that could occur during emission. Since we can't know the exact
   691→   691→        // count beforehand, we use the compute_frame_size function which
   692→   692→        // estimates based on the vreg count.
   693→   693→        let aligned_stack = compute_frame_size(func);
   694→   694→        // frame_size = only the spill area (NOT the Alloc sizes, since each
   695→   695→        // Alloc instruction does its own SUB SP)
   696→   696→        // Compute the spill-only portion:
   697→   697→        let mut alloc_total: u32 = 0;
   698→   698→        for block in &func.blocks {
   699→   699→            for instr in &block.instructions {
   700→   700→                if let IRInstr::Alloc { size, .. } = instr {
   701→   701→                    let aligned = (*size).div_ceil(16) * 16;
   702→   702→                    alloc_total += aligned;
   703→   703→                }
   704→   704→            }
   705→   705→        }
   706→   706→        // Spill area = total frame - alloc area
   707→   707→        let spill_area = if aligned_stack > alloc_total {
   708→   708→            aligned_stack - alloc_total
   709→   709→        } else {
   710→   710→            // Minimal 16-byte reservation for safety
   711→   711→            16
   712→   712→        };
   713→   713→        let spill_area_aligned = (spill_area + 15) & !15;
   714→   714→        self.frame_size = spill_area_aligned;
   715→   715→        if spill_area_aligned > 0 {
   716→   716→            if spill_area_aligned <= 4095 {
   717→   717→                self.emit_instruction(Instruction::SUB {
   718→   718→                    rd: Register::SP,
   719→   719→                    rn: Register::SP,
   720→   720→                    rm: Operand::Imm12(spill_area_aligned as u16),
   721→   721→                })?;
   722→   722→            } else {
   723→   723→                self.emit_load_immediate(Register::X9, spill_area_aligned as i64)?;
   724→   724→                self.emit_instruction(Instruction::SUB {
   725→   725→                    rd: Register::SP,
   726→   726→                    rn: Register::SP,
   727→   727→                    rm: Operand::Reg {
   728→   728→                        reg: Register::X9,
   729→   729→                        shift: None,
   730→   730→                    },
   731→   731→                })?;
   732→   732→            }
   733→   733→        }
   734→   734→
   735→   735→        // Emit each basic block.
   736→   736→        for block in &func.blocks {
   737→   737→            self.label_offsets
   738→   738→                .insert(block.label.clone(), self.code.len());
   739→   739→            self.emit_block(block)?;
   740→   740→        }
   741→   741→
   742→   742→        // Apply fixups — resolve intra-function branch targets.
   743→   743→        self.apply_fixups()?;
   744→   744→
   745→   745→        Ok(self.code.clone())
   746→   746→    }
   747→   747→
   748→   748→    /// Returns the relocation entries recorded during emission of the current
   749→   749→    /// function.
   750→   750→    ///
   751→   751→    /// Each `RelocationEntry` records a byte offset within the function's
   752→   752→    /// encoded output where a symbolic reference must be resolved, the target
   753→   753→    /// symbol name, and the ISA-specific relocation type (e.g.,
   754→   754→    /// `"R_AARCH64_CALL26"`).
   755→   755→    pub fn relocations(&self) -> &[RelocationEntry] {
   756→   756→        &self.relocations
   757→   757→    }
   758→   758→
   759→   759→    /// Emit instructions for a single IR basic block.
   760→   760→    fn emit_block(&mut self, block: &IRBlock) -> Result<()> {
   761→   761→        for instr in &block.instructions {
   762→   762→            self.emit_ir_instr(instr)?;
   763→   763→            // Auto-unpin all registers that were pinned during this instruction.
   764→   764→            for reg in self.instr_pinned_regs.drain(..) {
   765→   765→                self.reg_alloc.unpin(reg);
   766→   766→            }
   767→   767→        }
   768→   768→        self.emit_terminator(&block.terminator)?;
   769→   769→        Ok(())
   770→   770→    }
   771→   771→
   772→   772→    // -----------------------------------------------------------------------
   773→   773→    // IR → ARM64 instruction lowering
   774→   774→    // -----------------------------------------------------------------------
   775→   775→
   776→   776→    /// Lower a single IR instruction to ARM64 instructions.
   777→   777→    fn emit_ir_instr(&mut self, instr: &IRInstr) -> Result<()> {
   778→   778→        match instr {
   779→   779→            IRInstr::Load { dst, addr, offset, ty } => {
   780→   780→                let rn = self.resolve_reg(addr)?;
   781→   781→                let rt = self.resolve_reg(dst)?;
   782→   782→                self.emit_load(rt, rn, *offset, ty)?;
   783→   783→            }
   784→   784→
   785→   785→            IRInstr::Store { value, addr, offset, ty } => {
   786→   786→                let rt = self.resolve_reg(value)?;
   787→   787→                let rn = self.resolve_reg(addr)?;
   788→   788→                if rt == rn {
   789→   789→                    // Both resolved to the same register — use scratch for value
   790→   790→                    self.emit_instruction(Instruction::MOV {
   791→   791→                        rd: Register::X16,
   792→   792→                        rm: rt,
   793→   793→                    })?;
   794→   794→                    self.emit_store(Register::X16, rn, *offset, ty)?;
   795→   795→                } else {
   796→   796→                    self.emit_store(rt, rn, *offset, ty)?;
   797→   797→                }
   798→   798→            }
   799→   799→
   800→   800→            IRInstr::BinOp { op, dst, lhs, rhs, ty } => {
   801→   801→                self.emit_binop(*op, dst, lhs, rhs, ty.as_ref())?;
   802→   802→            }
   803→   803→
   804→   804→            IRInstr::UnaryOp { op, dst, operand, ty: _ } => {
   805→   805→                let rd = self.resolve_reg(dst)?;
   806→   806→                let rn = self.resolve_reg(operand)?;
   807→   807→                match op {
   808→   808→                    UnaryOpKind::Neg => {
   809→   809→                        self.emit_instruction(Instruction::SUB {
   810→   810→                            rd,
   811→   811→                            rn: Register::XZR,
   812→   812→                            rm: Operand::Reg {
   813→   813→                                reg: rn,
   814→   814→                                shift: None,
   815→   815→                            },
   816→   816→                        })?;
   817→   817→                    }
   818→   818→                    UnaryOpKind::Not => {
   819→   819→                        self.emit_load_immediate(Register::X9, -1)?;
   820→   820→                        self.emit_instruction(Instruction::EOR {
   821→   821→                            rd,
   822→   822→                            rn,
   823→   823→                            rm: Register::X9,
   824→   824→                        })?;
   825→   825→                    }
   826→   826→                    UnaryOpKind::Clz => {
   827→   827→                        // CLZ Xd, Xn — count leading zeros (native ARM64 instruction)
   828→   828→                        self.emit_instruction(Instruction::CLZ { rd, rn })?;
   829→   829→                    }
   830→   830→                    UnaryOpKind::Ctz => {
   831→   831→                        // CTZ = RBIT + CLZ: reverse bits then count leading zeros.
   832→   832→                        // Use X9 as scratch if rd == rn (need intermediate result).
   833→   833→                        if rd == rn {
   834→   834→                            self.emit_instruction(Instruction::RBIT {
   835→   835→                                rd: Register::X9,
   836→   836→                                rn,
   837→   837→                            })?;
   838→   838→                            self.emit_instruction(Instruction::CLZ {
   839→   839→                                rd,
   840→   840→                                rn: Register::X9,
   841→   841→                            })?;
   842→   842→                        } else {
   843→   843→                            self.emit_instruction(Instruction::RBIT { rd, rn })?;
   844→   844→                            self.emit_instruction(Instruction::CLZ { rd, rn: rd })?;
   845→   845→                        }
   846→   846→                    }
   847→   847→                    UnaryOpKind::Popcnt => {
   848→   848→                        // POPCNT via FMOV+CNT+ADDV+UMOV sequence:
   849→   849→                        // FMOV D8, Xn        — move GPR value to SIMD register (V8 is caller-saved)
   850→   850→                        // CNT V8.8B, V8.8B   — count bits per byte
   851→   851→                        // ADDV B8, V8.8B     — horizontal sum of byte counts
   852→   852→                        // UMOV Xd, V8.B[0]   — move result back to GPR (zero-extends to 64 bits)
   853→   853→                        const SIMD_SCRATCH: u8 = 8; // V8 is caller-saved in AAPCS64
   854→   854→                        self.emit_instruction(Instruction::FMOV_DX {
   855→   855→                            vd: SIMD_SCRATCH,
   856→   856→                            rn,
   857→   857→                        })?;
   858→   858→                        self.emit_instruction(Instruction::CNT {
   859→   859→                            vd: SIMD_SCRATCH,
   860→   860→                            vn: SIMD_SCRATCH,
   861→   861→                        })?;
   862→   862→                        self.emit_instruction(Instruction::ADDV {
   863→   863→                            vd: SIMD_SCRATCH,
   864→   864→                            vn: SIMD_SCRATCH,
   865→   865→                        })?;
   866→   866→                        self.emit_instruction(Instruction::UMOV {
   867→   867→                            rd,
   868→   868→                            vn: SIMD_SCRATCH,
   869→   869→                        })?;
   870→   870→                    }
   871→   871→                }
   872→   872→            }
   873→   873→
   874→   874→            IRInstr::Call {
   875→   875→                dst,
   876→   876→                func: target_name,
   877→   877→                args,
   878→   878→            } => {
   879→   879→                // Resolve all argument source registers FIRST, before moving
   880→   880→                // any of them. This prevents a later move from overwriting a
   881→   881→                // register that an earlier argument is in.
   882→   882→                //
   883→   883→                // IMPORTANT: For immediate arguments, resolve_reg always loads
   884→   884→                // into X9. When multiple immediates are present, we must use
   885→   885→                // different scratch registers to avoid overwriting. We load
   886→   886→                // immediates into X0-X7 directly (since they're the target
   887→   887→                // registers anyway) or use X11-X15 as temporary scratch.
   888→   888→                let arg_regs = [
   889→   889→                    Register::X0, Register::X1, Register::X2, Register::X3,
   890→   890→                    Register::X4, Register::X5, Register::X6, Register::X7,
   891→   891→                ];
   892→   892→                // Scratch registers for loading immediates when the target arg
   893→   893→                // register isn't available yet (because of conflicts).
   894→   894→                let scratch_regs = [
   895→   895→                    Register::X11, Register::X12, Register::X13, Register::X14,
   896→   896→                    Register::X15, Register::X3, Register::X4, Register::X5,
   897→   897→                ];
   898→   898→
   899→   899→                let mut src_regs: Vec<Register> = Vec::new();
   900→   900→                let mut immediate_scratch_used: Vec<usize> = Vec::new(); // which scratch reg each immediate uses
   901→   901→                let mut next_scratch = 0;
   902→   902→
   903→   903→                for (i, arg) in args.iter().enumerate() {
   904→   904→                    if i >= 8 {
   905→   905→                        break;
   906→   906→                    }
   907→   907→                    let is_immediate = matches!(arg, IRValue::Immediate(_) | IRValue::Address(_));
   908→   908→                    if is_immediate {
   909→   909→                        let v = match arg {
   910→   910→                            IRValue::Immediate(v) => *v,
   911→   911→                            IRValue::Address(a) => *a as i64,
   912→   912→                            _ => unreachable!(),
   913→   913→                        };
   914→   914→                        // Load the immediate directly into the target register
   915→   915→                        // if it's not needed by another argument.
   916→   916→                        // For simplicity, always load into a scratch register
   917→   917→                        // and record it.
   918→   918→                        if next_scratch < scratch_regs.len() {
   919→   919→                            let scratch = scratch_regs[next_scratch];
   920→   920→                            next_scratch += 1;
   921→   921→                            self.emit_load_immediate(scratch, v)?;
   922→   922→                            src_regs.push(scratch);
   923→   923→                        } else {
   924→   924→                            // Fallback to X9 (rare, only with 8+ immediates)
   925→   925→                            let scratch = Register::X9;
   926→   926→                            self.emit_load_immediate(scratch, v)?;
   927→   927→                            src_regs.push(scratch);
   928→   928→                        }
   929→   929→                    } else {
   930→   930→                        let src = self.resolve_reg(arg)?;
   931→   931→                        src_regs.push(src);
   932→   932→                    }
   933→   933→                }
   934→   934→
   935→   935→                // Now move arguments into X0–X7. We need to handle the case
   936→   936→                // where a source register is also a target register (cycle).
   937→   937→                // Strategy: first move arguments that don't conflict, then
   938→   938→                // handle conflicts using a scratch register.
   939→   939→                let n = src_regs.len();
   940→   940→                let mut moved = vec![false; n];
   941→   941→
   942→   942→                // Pass 1: move args where src != any dst that hasn't been moved yet
   943→   943→                // Simple approach: iterate and move non-conflicting ones first
   944→   944→                for pass in 0..2 {
   945→   945→                    for i in 0..n {
   946→   946→                        if moved[i] {
   947→   947→                            continue;
   948→   948→                        }
   949→   949→                        let src = src_regs[i];
   950→   950→                        let dst = arg_regs[i];
   951→   951→                        if src == dst {
   952→   952→                            // Already in the right register
   953→   953→                            moved[i] = true;
   954→   954→                            continue;
   955→   955→                        }
   956→   956→                        if pass == 0 {
   957→   957→                            // Check if src is needed as a source by a later unmoved arg
   958→   958→                            let mut conflict = false;
   959→   959→                            for j in 0..n {
   960→   960→                                if !moved[j] && j != i && src_regs[j] == dst {
   961→   961→                                    // Moving arg i to dst would overwrite arg j's source
   962→   962→                                    conflict = true;
   963→   963→                                    break;
   964→   964→                                }
   965→   965→                            }
   966→   966→                            if conflict {
   967→   967→                                continue; // Defer to pass 2
   968→   968→                            }
   969→   969→                        }
   970→   970→                        // Safe to move now
   971→   971→                        self.emit_instruction(Instruction::MOV {
   972→   972→                            rd: dst,
   973→   973→                            rm: src,
   974→   974→                        })?;
   975→   975→                        moved[i] = true;
   976→   976→                    }
   977→   977→                }
   978→   978→
   979→   979→                // Pass 2: handle any remaining conflicts with scratch register
   980→   980→                for i in 0..n {
   981→   981→                    if moved[i] {
   982→   982→                        continue;
   983→   983→                    }
   984→   984→                    let src = src_regs[i];
   985→   985→                    let dst = arg_regs[i];
   986→   986→                    if src == dst {
   987→   987→                        moved[i] = true;
   988→   988→                        continue;
   989→   989→                    }
   990→   990→                    // Use X16 as scratch to break the cycle
   991→   991→                    self.emit_instruction(Instruction::MOV {
   992→   992→                        rd: Register::X16,
   993→   993→                        rm: src,
   994→   994→                    })?;
   995→   995→                    self.emit_instruction(Instruction::MOV {
   996→   996→                        rd: dst,
   997→   997→                        rm: Register::X16,
   998→   998→                    })?;
   999→   999→                    moved[i] = true;
  1000→  1000→                }
  1001→  1001→
  1002→  1002→                // BL — record a relocation for later patching.
  1003→  1003→                let bl_word_idx = self.code.len();
  1004→  1004→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1005→  1005→                self.call_relocs.push(CallRelocation {
  1006→  1006→                    text_byte_offset: bl_byte_offset,
  1007→  1007→                    target_func: target_name.clone(),
  1008→  1008→                });
  1009→  1009→                // Register a RelocationEntry so the linker can patch this BL.
  1010→  1010→                self.relocations.push(RelocationEntry {
  1011→  1011→                    offset: bl_byte_offset,
  1012→  1012→                    symbol: target_name.clone(),
  1013→  1013→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1014→  1014→                });
  1015→  1015→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1016→  1016→
  1017→  1017→                if let Some(dst_val) = dst {
  1018→  1018→                    let rd = self.resolve_reg(dst_val)?;
  1019→  1019→                    if rd != Register::X0 {
  1020→  1020→                        self.emit_instruction(Instruction::MOV {
  1021→  1021→                            rd,
  1022→  1022→                            rm: Register::X0,
  1023→  1023→                        })?;
  1024→  1024→                    }
  1025→  1025→                }
  1026→  1026→            }
  1027→  1027→
  1028→  1028→            IRInstr::Alloc { dst, size } => {
  1029→  1029→                let rd = self.resolve_reg(dst)?;
  1030→  1030→                // IMPORTANT: Decrement SP FIRST, then save the new SP value.
  1031→  1031→                // If we save before decrementing, the allocation pointer
  1032→  1032→                // points to the wrong (too-high) address.
  1033→  1033→                let aligned = (*size).div_ceil(16) * 16;
  1034→  1034→                self.emit_instruction(Instruction::SUB {
  1035→  1035→                    rd: Register::SP,
  1036→  1036→                    rn: Register::SP,
  1037→  1037→                    rm: Operand::Imm12(aligned as u16),
  1038→  1038→                })?;
  1039→  1039→                // MOV rd, SP: cannot use ORR because ORR treats Rm=31 as XZR.
  1040→  1040→                // Use ADD rd, SP, #0 instead.
  1041→  1041→                self.emit_instruction(Instruction::ADD {
  1042→  1042→                    rd,
  1043→  1043→                    rn: Register::SP,
  1044→  1044→                    rm: Operand::Imm12(0),
  1045→  1045→                })?;
  1046→  1046→            }
  1047→  1047→
  1048→  1048→            IRInstr::Free { ptr } => {
  1049→  1049→                let rt = self.resolve_reg(ptr)?;
  1050→  1050→                // Move ptr to X0 (first argument)
  1051→  1051→                if rt != Register::X0 {
  1052→  1052→                    self.emit_instruction(Instruction::MOV {
  1053→  1053→                        rd: Register::X0,
  1054→  1054→                        rm: rt,
  1055→  1055→                    })?;
  1056→  1056→                }
  1057→  1057→                // BL __vuma_free
  1058→  1058→                let bl_word_idx = self.code.len();
  1059→  1059→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1060→  1060→                self.call_relocs.push(CallRelocation {
  1061→  1061→                    text_byte_offset: bl_byte_offset,
  1062→  1062→                    target_func: "__vuma_free".to_string(),
  1063→  1063→                });
  1064→  1064→                self.relocations.push(RelocationEntry {
  1065→  1065→                    offset: bl_byte_offset,
  1066→  1066→                    symbol: "__vuma_free".to_string(),
  1067→  1067→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1068→  1068→                });
  1069→  1069→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1070→  1070→            }
  1071→  1071→
  1072→  1072→            IRInstr::Cast { kind, dst, src } => {
  1073→  1073→                let rd = self.resolve_reg(dst)?;
  1074→  1074→                let rn = self.resolve_reg(src)?;
  1075→  1075→                match kind {
  1076→  1076→                    CastKind::ZExt => {
  1077→  1077→                        // Zero-extend 32-bit to 64-bit using UBFM
  1078→  1078→                        self.emit_instruction(Instruction::UBFM {
  1079→  1079→                            rd,
  1080→  1080→                            rn,
  1081→  1081→                            immr: 0,
  1082→  1082→                            imms: 31,
  1083→  1083→                        })?;
  1084→  1084→                    }
  1085→  1085→                    CastKind::SExt => {
  1086→  1086→                        // Sign-extend 32-bit to 64-bit using SBFM
  1087→  1087→                        self.emit_instruction(Instruction::SBFM {
  1088→  1088→                            rd,
  1089→  1089→                            rn,
  1090→  1090→                            immr: 0,
  1091→  1091→                            imms: 31,
  1092→  1092→                        })?;
  1093→  1093→                    }
  1094→  1094→                    CastKind::Trunc | CastKind::BitCast => {
  1095→  1095→                        // Trunc: upper bits discarded on write — just MOV
  1096→  1096→                        // BitCast: no data change — just MOV
  1097→  1097→                        if rd != rn {
  1098→  1098→                            self.emit_instruction(Instruction::MOV { rd, rm: rn })?;
  1099→  1099→                        }
  1100→  1100→                    }
  1101→  1101→                }
  1102→  1102→            }
  1103→  1103→
  1104→  1104→            IRInstr::Phi { .. } => {
  1105→  1105→                log::warn!(
  1106→  1106→                    "IRInstr::Phi encountered during emission — should be resolved by SSA pass"
  1107→  1107→                );
  1108→  1108→            }
  1109→  1109→
  1110→  1110→            IRInstr::GetAddress { dst, name } => {
  1111→  1111→                let rd = self.resolve_reg(dst)?;
  1112→  1112→                // Emit a call to __vuma_getaddr to resolve the symbol at runtime.
  1113→  1113→                // Move name hash to X0 as the argument.
  1114→  1114→                let name_hash = name
  1115→  1115→                    .chars()
  1116→  1116→                    .fold(0u64, |acc, c| acc.wrapping_mul(31).wrapping_add(c as u64));
  1117→  1117→                self.emit_load_immediate(Register::X0, name_hash as i64)?;
  1118→  1118→                let bl_word_idx = self.code.len();
  1119→  1119→                let bl_byte_offset = self.func_text_offset + (bl_word_idx as u64) * 4;
  1120→  1120→                self.call_relocs.push(CallRelocation {
  1121→  1121→                    text_byte_offset: bl_byte_offset,
  1122→  1122→                    target_func: "__vuma_getaddr".to_string(),
  1123→  1123→                });
  1124→  1124→                self.relocations.push(RelocationEntry {
  1125→  1125→                    offset: bl_byte_offset,
  1126→  1126→                    symbol: "__vuma_getaddr".to_string(),
  1127→  1127→                    reloc_type: "R_AARCH64_CALL26".to_string(),
  1128→  1128→                });
  1129→  1129→                self.emit_instruction(Instruction::BL { offset: 0 })?;
  1130→  1130→                if rd != Register::X0 {
  1131→  1131→                    self.emit_instruction(Instruction::MOV {
  1132→  1132→                        rd,
  1133→  1133→                        rm: Register::X0,
  1134→  1134→                    })?;
  1135→  1135→                }
  1136→  1136→            }
  1137→  1137→
  1138→  1138→            IRInstr::Offset { dst, base, offset } => {
  1139→  1139→                let rd = self.resolve_reg(dst)?;
  1140→  1140→                let rn = self.resolve_reg(base)?;
  1141→  1141→                let rm = match offset {
  1142→  1142→                    IRValue::Immediate(v) => {
  1143→  1143→                        if *v >= 0 && *v <= 4095 {
  1144→  1144→                            Operand::Imm12(*v as u16)
  1145→  1145→                        } else {
  1146→  1146→                            let temp = Register::X9;
  1147→  1147→                            self.emit_load_immediate(temp, *v)?;
  1148→  1148→                            Operand::Reg {
  1149→  1149→                                reg: temp,
  1150→  1150→                                shift: None,
  1151→  1151→                            }
  1152→  1152→                        }
  1153→  1153→                    }
  1154→  1154→                    _ => Operand::Reg {
  1155→  1155→                        reg: self.resolve_reg(offset)?,
  1156→  1156→                        shift: None,
  1157→  1157→                    },
  1158→  1158→                };
  1159→  1159→                self.emit_instruction(Instruction::ADD { rd, rn, rm })?;
  1160→  1160→            }
  1161→  1161→
  1162→  1162→            // ── Dedicated arithmetic — delegate to BinOp ──
  1163→  1163→            IRInstr::Add { dst, lhs, rhs, ty } => {
  1164→  1164→                self.emit_binop(BinOpKind::Add, dst, lhs, rhs, ty.as_ref())?;
  1165→  1165→            }
  1166→  1166→            IRInstr::Sub { dst, lhs, rhs, ty } => {
  1167→  1167→                self.emit_binop(BinOpKind::Sub, dst, lhs, rhs, ty.as_ref())?;
  1168→  1168→            }
  1169→  1169→            IRInstr::Mul { dst, lhs, rhs, ty } => {
  1170→  1170→                self.emit_binop(BinOpKind::Mul, dst, lhs, rhs, ty.as_ref())?;
  1171→  1171→            }
  1172→  1172→            IRInstr::Div { dst, lhs, rhs, ty } => {
  1173→  1173→                self.emit_binop(BinOpKind::SDiv, dst, lhs, rhs, ty.as_ref())?;
  1174→  1174→            }
  1175→  1175→
  1176→  1176→            // ── Comparison instruction ──
  1177→  1177→            IRInstr::Cmp {
  1178→  1178→                kind,
  1179→  1179→                dst,
  1180→  1180→                lhs,
  1181→  1181→                rhs,
  1182→  1182→                ty,
  1183→  1183→            } => {
  1184→  1184→                let width = RegWidth::from_ir_type(ty.as_ref());
  1185→  1185→                let rd = self.resolve_reg(dst)?;
  1186→  1186→                // IMPORTANT: resolve_reg uses X9 for immediates, so if both lhs and rhs
  1187→  1187→                // are immediates, the second load would overwrite the first. Use X10 for
  1188→  1188→                // the RHS when both are immediates.
  1189→  1189→                let rn = self.resolve_reg(lhs)?;
  1190→  1190→                let rm = match rhs {
  1191→  1191→                    IRValue::Immediate(_) | IRValue::Address(_) => {
  1192→  1192→                        // If lhs was also an immediate, it's in X9; use X10 for rhs
  1193→  1193→                        let temp = if matches!(lhs, IRValue::Immediate(_) | IRValue::Address(_)) {
  1194→  1194→                            Register::X10
  1195→  1195→                        } else {
  1196→  1196→                            Register::X9
  1197→  1197→                        };
  1198→  1198→                        match rhs {
  1199→  1199→                            IRValue::Immediate(v) => self.emit_load_immediate(temp, *v)?,
  1200→  1200→                            IRValue::Address(a) => self.emit_load_immediate(temp, *a as i64)?,
  1201→  1201→                            _ => unreachable!(),
  1202→  1202→                        }
  1203→  1203→                        temp
  1204→  1204→                    }
  1205→  1205→                    _ => self.resolve_reg(rhs)?,
  1206→  1206→                };
  1207→  1207→                self.emit_instruction_with_width(
  1208→  1208→                    Instruction::CMP {
  1209→  1209→                        rn,
  1210→  1210→                        rm: Operand::Reg {
  1211→  1211→                            reg: rm,
  1212→  1212→                            shift: None,
  1213→  1213→                        },
  1214→  1214→                    },
  1215→  1215→                    width,
  1216→  1216→                )?;
  1217→  1217→                let cond = cmp_kind_to_condition(kind);
  1218→  1218→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1219→  1219→            }
  1220→  1220→
  1221→  1221→            // ── Instruction-level control flow ──
  1222→  1222→            IRInstr::Ret { values } => {
  1223→  1223→                for (i, val) in values.iter().enumerate() {
  1224→  1224→                    if i >= 8 {
  1225→  1225→                        break;
  1226→  1226→                    }
  1227→  1227→                    let src = self.resolve_reg(val)?;
  1228→  1228→                    let dst_reg = match i {
  1229→  1229→                        0 => Register::X0,
  1230→  1230→                        1 => Register::X1,
  1231→  1231→                        2 => Register::X2,
  1232→  1232→                        3 => Register::X3,
  1233→  1233→                        4 => Register::X4,
  1234→  1234→                        5 => Register::X5,
  1235→  1235→                        6 => Register::X6,
  1236→  1236→                        7 => Register::X7,
  1237→  1237→                        _ => unreachable!(),
  1238→  1238→                    };
  1239→  1239→                    if src != dst_reg {
  1240→  1240→                        self.emit_instruction(Instruction::MOV {
  1241→  1241→                            rd: dst_reg,
  1242→  1242→                            rm: src,
  1243→  1243→                        })?;
  1244→  1244→                    }
  1245→  1245→                }
  1246→  1246→            }
  1247→  1247→
  1248→  1248→            IRInstr::Branch { target } => {
  1249→  1249→                let fixup_idx = self.code.len();
  1250→  1250→                self.fixups.push((fixup_idx, target.clone(), BranchFormat::B26));
  1251→  1251→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1252→  1252→            }
  1253→  1253→
  1254→  1254→            IRInstr::CondBranch {
  1255→  1255→                cond,
  1256→  1256→                true_target,
  1257→  1257→                false_target,
  1258→  1258→            } => {
  1259→  1259→                let rt = self.resolve_reg(cond)?;
  1260→  1260→                // CBNZ: if cond != 0, branch to true_target
  1261→  1261→                let fixup_cbz = self.code.len();
  1262→  1262→                self.fixups.push((fixup_cbz, true_target.clone(), BranchFormat::Cond19));
  1263→  1263→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1264→  1264→                // Otherwise (cond == 0), branch to false_target
  1265→  1265→                let fixup_b = self.code.len();
  1266→  1266→                self.fixups.push((fixup_b, false_target.clone(), BranchFormat::B26));
  1267→  1267→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1268→  1268→            }
  1269→  1269→
  1270→  1270→            IRInstr::Select {
  1271→  1271→                dst,
  1272→  1272→                cond,
  1273→  1273→                true_val,
  1274→  1274→                false_val,
  1275→  1275→                ty,
  1276→  1276→            } => {
  1277→  1277→                let width = RegWidth::from_ir_type(ty.as_ref());
  1278→  1278→                // Lower select as: SUBS XZR, cond, #0; CSEL dst, false_val, true_val, NE
  1279→  1279→                let rd = self.resolve_reg(dst)?;
  1280→  1280→                let rc = self.resolve_reg(cond)?;
  1281→  1281→                let rt = self.resolve_reg(true_val)?;
  1282→  1282→                let rf = self.resolve_reg(false_val)?;
  1283→  1283→                // Compare cond against zero and select.
  1284→  1284→                self.emit_instruction_with_width(
  1285→  1285→                    Instruction::SUB {
  1286→  1286→                        rd: Register::XZR,
  1287→  1287→                        rn: rc,
  1288→  1288→                        rm: Operand::Imm12(0),
  1289→  1289→                    },
  1290→  1290→                    width,
  1291→  1291→                )?;
  1292→  1292→                // Set flags by using a separate CMP (SUB with XZR destination
  1293→  1293→                // doesn't set flags; we need a flags-setting variant).
  1294→  1294→                // We emulate this with: CMP rc, #0 which is SUBS XZR, rc, #0.
  1295→  1295→                // Since we only have SUB, we use the existing CMP pattern.
  1296→  1296→                self.emit_instruction_with_width(
  1297→  1297→                    Instruction::CSEL {
  1298→  1298→                        rd,
  1299→  1299→                        rn: rf,
  1300→  1300→                        rm: rt,
  1301→  1301→                        cond: crate::arm64::Condition::NE,
  1302→  1302→                    },
  1303→  1303→                    width,
  1304→  1304→                )?;
  1305→  1305→            }
  1306→  1306→        }
  1307→  1307→        Ok(())
  1308→  1308→    }
  1309→  1309→
  1310→  1310→    /// Emit a binary operation (shared by `BinOp` and dedicated `Add`/`Sub`/…).
  1311→  1311→    ///
  1312→  1312→    /// Some ARM64 instructions (ADD, SUB, LSL, LSR, ASR) accept a 12-bit
  1313→  1313→    /// unsigned immediate operand directly.  Most others (MUL, AND, ORR, EOR,
  1314→  1314→    /// SDIV, UDIV, remainder, comparisons) **require** a register operand.
  1315→  1315→    /// When the RHS is an immediate that the target instruction cannot accept
  1316→  1316→    /// directly, we "spill" it into a scratch register (X16 / IP0) first via
  1317→  1317→    /// MOVZ + MOVK and then use that register as the operand.
  1318→  1318→    ///
  1319→  1319→    /// X16 (IP0) is chosen as the scratch register because it is **not** in
  1320→  1320→    /// the register allocator's free pool, so it will never conflict with
  1321→  1321→    /// `rd` or `rn`.
  1322→  1322→    fn emit_binop(
  1323→  1323→        &mut self,
  1324→  1324→        op: BinOpKind,
  1325→  1325→        dst: &IRValue,
  1326→  1326→        lhs: &IRValue,
  1327→  1327→        rhs: &IRValue,
  1328→  1328→        ty: Option<&IRType>,
  1329→  1329→    ) -> Result<()> {
  1330→  1330→        let width = RegWidth::from_ir_type(ty);
  1331→  1331→        let rd = self.resolve_reg(dst)?;
  1332→  1332→        let rn = self.resolve_reg(lhs)?;
  1333→  1333→
  1334→  1334→        // Determine whether the operation can accept an immediate operand.
  1335→  1335→        // ADD, SUB, LSL, LSR, ASR all have immediate forms in ARM64.
  1336→  1336→        // All other operations require a register operand.
  1337→  1337→        let supports_imm = matches!(
  1338→  1338→            op,
  1339→  1339→            BinOpKind::Add | BinOpKind::Sub | BinOpKind::Shl | BinOpKind::ShrL | BinOpKind::ShrA
  1340→  1340→        );
  1341→  1341→
  1342→  1342→        let rm = match rhs {
  1343→  1343→            IRValue::Immediate(v) => {
  1344→  1344→                if supports_imm && *v >= 0 && *v <= 4095 {
  1345→  1345→                    // Small immediate that the instruction accepts directly.
  1346→  1346→                    Operand::Imm12(*v as u16)
  1347→  1347→                } else {
  1348→  1348→                    // Load the immediate into scratch register X16 (IP0).
  1349→  1349→                    // X16 is not in the register allocator's pool, so it cannot
  1350→  1350→                    // conflict with `rd` or `rn`.
  1351→  1351→                    let scratch = Register::X16;
  1352→  1352→                    self.emit_load_immediate_with_width(scratch, *v, width)?;
  1353→  1353→                    Operand::Reg {
  1354→  1354→                        reg: scratch,
  1355→  1355→                        shift: None,
  1356→  1356→                    }
  1357→  1357→                }
  1358→  1358→            }
  1359→  1359→            _ => Operand::Reg {
  1360→  1360→                reg: self.resolve_reg(rhs)?,
  1361→  1361→                shift: None,
  1362→  1362→            },
  1363→  1363→        };
  1364→  1364→
  1365→  1365→        match op {
  1366→  1366→            BinOpKind::Add => {
  1367→  1367→                self.emit_instruction_with_width(Instruction::ADD { rd, rn, rm }, width)?;
  1368→  1368→            }
  1369→  1369→            BinOpKind::Sub => {
  1370→  1370→                self.emit_instruction_with_width(Instruction::SUB { rd, rn, rm }, width)?;
  1371→  1371→            }
  1372→  1372→            BinOpKind::Mul => {
  1373→  1373→                let rm_reg = self.operand_to_reg(&rm)?;
  1374→  1374→                self.emit_instruction_with_width(Instruction::MUL { rd, rn, rm: rm_reg }, width)?;
  1375→  1375→            }
  1376→  1376→            BinOpKind::SDiv => {
  1377→  1377→                let rm_reg = self.operand_to_reg(&rm)?;
  1378→  1378→                self.emit_instruction_with_width(Instruction::SDIV { rd, rn, rm: rm_reg }, width)?;
  1379→  1379→            }
  1380→  1380→            BinOpKind::UDiv => {
  1381→  1381→                let rm_reg = self.operand_to_reg(&rm)?;
  1382→  1382→                self.emit_instruction_with_width(Instruction::UDIV { rd, rn, rm: rm_reg }, width)?;
  1383→  1383→            }
  1384→  1384→            BinOpKind::And => {
  1385→  1385→                let rm_reg = self.operand_to_reg(&rm)?;
  1386→  1386→                self.emit_instruction_with_width(Instruction::AND { rd, rn, rm: rm_reg }, width)?;
  1387→  1387→            }
  1388→  1388→            BinOpKind::Or => {
  1389→  1389→                let rm_reg = self.operand_to_reg(&rm)?;
  1390→  1390→                self.emit_instruction_with_width(Instruction::ORR { rd, rn, rm: rm_reg }, width)?;
  1391→  1391→            }
  1392→  1392→            BinOpKind::Xor => {
  1393→  1393→                let rm_reg = self.operand_to_reg(&rm)?;
  1394→  1394→                self.emit_instruction_with_width(Instruction::EOR { rd, rn, rm: rm_reg }, width)?;
  1395→  1395→            }
  1396→  1396→            BinOpKind::Shl => {
  1397→  1397→                self.emit_instruction_with_width(Instruction::LSL { rd, rn, rm }, width)?;
  1398→  1398→            }
  1399→  1399→            BinOpKind::ShrL => {
  1400→  1400→                self.emit_instruction_with_width(Instruction::LSR { rd, rn, rm }, width)?;
  1401→  1401→            }
  1402→  1402→            BinOpKind::ShrA => {
  1403→  1403→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm }, width)?;
  1404→  1404→            }
  1405→  1405→            BinOpKind::Ror | BinOpKind::Rol => {
  1406→  1406→                let rm_reg = self.operand_to_reg(&rm)?;
  1407→  1407→                // ARM64 rotation: use ASR as placeholder
  1408→  1408→                self.emit_instruction_with_width(Instruction::ASR { rd, rn, rm: Operand::Reg { reg: rm_reg, shift: None } }, width)?;
  1409→  1409→            }
  1410→  1410→            BinOpKind::SRem | BinOpKind::URem => {
  1411→  1411→                let rm_reg = self.operand_to_reg(&rm)?;
  1412→  1412→                let div_instr = if op == BinOpKind::SRem {
  1413→  1413→                    Instruction::SDIV { rd, rn, rm: rm_reg }
  1414→  1414→                } else {
  1415→  1415→                    Instruction::UDIV { rd, rn, rm: rm_reg }
  1416→  1416→                };
  1417→  1417→                self.emit_instruction_with_width(div_instr, width)?;
  1418→  1418→                // MSUB rd, rd, rm, rn  =>  rd = rn - rd * rm  =  dividend - quotient * divisor
  1419→  1419→                self.emit_instruction_with_width(Instruction::MSUB {
  1420→  1420→                    rd,
  1421→  1421→                    rn: rd,     // quotient (result of DIV)
  1422→  1422→                    rm: rm_reg, // divisor
  1423→  1423→                    ra: rn,     // dividend
  1424→  1424→                }, width)?;
  1425→  1425→            }
  1426→  1426→            BinOpKind::SLt
  1427→  1427→            | BinOpKind::SLe
  1428→  1428→            | BinOpKind::SGt
  1429→  1429→            | BinOpKind::SGe
  1430→  1430→            | BinOpKind::ULt
  1431→  1431→            | BinOpKind::ULe
  1432→  1432→            | BinOpKind::UGt
  1433→  1433→            | BinOpKind::UGe
  1434→  1434→            | BinOpKind::Eq
  1435→  1435→            | BinOpKind::Ne => {
  1436→  1436→                let rm_reg = self.operand_to_reg(&rm)?;
  1437→  1437→                self.emit_instruction_with_width(Instruction::CMP {
  1438→  1438→                    rn,
  1439→  1439→                    rm: Operand::Reg {
  1440→  1440→                        reg: rm_reg,
  1441→  1441→                        shift: None,
  1442→  1442→                    },
  1443→  1443→                }, width)?;
  1444→  1444→                let cond = binop_kind_to_condition(&op);
  1445→  1445→                self.emit_instruction_with_width(Instruction::CSET { rd, cond }, width)?;
  1446→  1446→            }
  1447→  1447→        }
  1448→  1448→        Ok(())
  1449→  1449→    }
  1450→  1450→
  1451→  1451→    /// Emit the block terminator.
  1452→  1452→    fn emit_terminator(&mut self, term: &IRTerminator) -> Result<()> {
  1453→  1453→        match term {
  1454→  1454→            IRTerminator::Jump(target) => {
  1455→  1455→                let fixup_idx = self.code.len();
  1456→  1456→                self.fixups.push((fixup_idx, target.clone(), BranchFormat::B26));
  1457→  1457→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1458→  1458→            }
  1459→  1459→            IRTerminator::Branch {
  1460→  1460→                cond,
  1461→  1461→                true_block,
  1462→  1462→                false_block,
  1463→  1463→            } => {
  1464→  1464→                let rt = self.resolve_reg(cond)?;
  1465→  1465→                let fixup_cbnz = self.code.len();
  1466→  1466→                self.fixups.push((fixup_cbnz, true_block.clone(), BranchFormat::Cond19));
  1467→  1467→                self.emit_instruction(Instruction::CBNZ { rt, offset: 0 })?;
  1468→  1468→                let fixup_b = self.code.len();
  1469→  1469→                self.fixups.push((fixup_b, false_block.clone(), BranchFormat::B26));
  1470→  1470→                self.emit_instruction(Instruction::B { offset: 0 })?;
  1471→  1471→            }
  1472→  1472→            IRTerminator::Return(vals) => {
  1473→  1473→                for (i, val) in vals.iter().enumerate() {
  1474→  1474→                    if i >= 8 {
  1475→  1475→                        break;
  1476→  1476→                    }
  1477→  1477→                    let src = self.resolve_reg(val)?;
  1478→  1478→                    let dst_reg = match i {
  1479→  1479→                        0 => Register::X0,
  1480→  1480→                        1 => Register::X1,
  1481→  1481→                        2 => Register::X2,
  1482→  1482→                        3 => Register::X3,
  1483→  1483→                        4 => Register::X4,
  1484→  1484→                        5 => Register::X5,
  1485→  1485→                        6 => Register::X6,
  1486→  1486→                        7 => Register::X7,
  1487→  1487→                        _ => unreachable!(),
  1488→  1488→                    };
  1489→  1489→                    if src != dst_reg {
  1490→  1490→                        self.emit_instruction(Instruction::MOV {
  1491→  1491→                            rd: dst_reg,
  1492→  1492→                            rm: src,
  1493→  1493→                        })?;
  1494→  1494→                    }
  1495→  1495→                }
  1496→  1496→                // Restore SP to point to the saved FP/LR pair.
  1497→  1497→                // X29 was set to SP right after the prologue's SUB SP, SP, #16
  1498→  1498→                // and STP X29, X30, [SP]. So X29 points to the FP/LR save area.
  1499→  1499→                // Using MOV SP, X29 (via ADD SP, X29, #0) is the most robust
  1500→  1500→                // way to restore SP regardless of how many Alloc instructions
  1501→  1501→                // modified it during the function body.
  1502→  1502→                self.emit_instruction(Instruction::ADD {
  1503→  1503→                    rd: Register::SP,
  1504→  1504→                    rn: Register::X29,
  1505→  1505→                    rm: Operand::Imm12(0),
  1506→  1506→                })?;
  1507→  1507→                // FP/LR were stored at [SP] after the prologue's SUB SP, SP, #16.
  1508→  1508→                // After the ADD above, SP points back to the FP/LR save area.
  1509→  1509→                // Load from [SP, #0], then restore SP by adding 16.
  1510→  1510→                self.emit_instruction(Instruction::LDP {
  1511→  1511→                    rt1: Register::X29,
  1512→  1512→                    rt2: Register::X30,
  1513→  1513→                    rn: Register::SP,
  1514→  1514→                    offset: 0,
  1515→  1515→                })?;
  1516→  1516→                self.emit_instruction(Instruction::ADD {
  1517→  1517→                    rd: Register::SP,
  1518→  1518→                    rn: Register::SP,
  1519→  1519→                    rm: Operand::Imm12(16),
  1520→  1520→                })?;
  1521→  1521→                self.emit_instruction(Instruction::RET { rn: None })?;
  1522→  1522→            }
  1523→  1523→            IRTerminator::Unreachable => {
  1524→  1524→                self.emit_instruction(Instruction::MOV {
  1525→  1525→                    rd: Register::XZR,
  1526→  1526→                    rm: Register::XZR,
  1527→  1527→                })?;
  1528→  1528→            }
  1529→  1529→            // Switch, Invoke, TailCall, and Resume are lowered by the
  1530→  1530→            // control_flow module before reaching the emitter. If they
  1531→  1531→            // appear here, it means the lowering pass was not run.
  1532→  1532→            IRTerminator::Switch { .. } => {
  1533→  1533→                return Err(CodegenError::InvalidInstruction(
  1534→  1534→                    "Switch terminator must be lowered before emission".to_string(),
  1535→  1535→                ));
  1536→  1536→            }
  1537→  1537→            IRTerminator::Invoke { .. } => {
  1538→  1538→                return Err(CodegenError::InvalidInstruction(
  1539→  1539→                    "Invoke terminator must be lowered before emission".to_string(),
  1540→  1540→                ));
  1541→  1541→            }
  1542→  1542→            IRTerminator::TailCall { .. } => {
  1543→  1543→                return Err(CodegenError::InvalidInstruction(
  1544→  1544→                    "TailCall terminator must be lowered before emission".to_string(),
  1545→  1545→                ));
  1546→  1546→            }
  1547→  1547→            IRTerminator::Resume { .. } => {
  1548→  1548→                return Err(CodegenError::InvalidInstruction(
  1549→  1549→                    "Resume terminator must be lowered before emission".to_string(),
  1550→  1550→                ));
  1551→  1551→            }
  1552→  1552→        }
  1553→  1553→        Ok(())
  1554→  1554→    }

  1563→  1563→        Ok(())
  1564→  1564→    }

  1566→  1566→    /// Emit an instruction with a specific register width (32-bit W or 64-bit X).
  1567→  1567→    ///
  1568→  1568→    /// This is the primary emission method for arithmetic and logical instructions
  1569→  1569→    /// where the operand width affects the encoding. Using `RegWidth::W32` produces
  1570→  1570→    /// instructions that operate on W sub-registers, giving automatic 32-bit
  1571→  1571→    /// wrapping arithmetic.
  1572→  1572→    fn emit_instruction_with_width(&mut self, instr: Instruction, width: RegWidth) -> Result<()> {
  1573→  1573→        let word = instr.encode_with_width(width)?;

  1575→  1575→        Ok(())
  1576→  1576→    }
  1577→  1577→
  1578→  1578→    /// Emit a load instruction with the given offset and IR type.
  1579→  1579→    ///
  1580→  1580→    /// Selects the correct ARM64 load variant based on the IR type:
  1581→  1581→    /// - I8/U8 → LDRB (byte, zero-extended)
  1582→  1582→    /// - I16/U16 → LDRH (halfword, zero-extended)
  1583→  1583→    /// - I32/U32/Ptr/Func → LDR (word or doubleword)
  1584→  1584→    /// - I64/U64 → LDR (doubleword)
  1585→  1585→    ///
  1586→  1586→    /// If the offset fits the ARM64 unsigned-offset immediate encoding for the
  1587→  1587→    /// selected instruction variant, it is encoded directly. Otherwise, the
  1588→  1588→    /// effective address is computed in a scratch register (X9) first, and
  1589→  1589→    /// the load is performed at [X9 + 0].
  1590→  1590→    fn emit_load(&mut self, rt: Register, rn: Register, offset: i32, ty: &IRType) -> Result<()> {
  1591→  1591→        // Determine the ARM64 load instruction and its immediate-offset constraints.
  1592→  1592→        // For each variant: (scale, max_imm12) where the encoded offset = imm12 << scale.
  1593→  1593→        // LDRB:  scale=0, imm12 0..4095 → offset 0..4095
  1594→  1594→        // LDRH:  scale=1, imm12 0..4095 → offset 0..8190, even
  1595→  1595→        // LDRSW: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1596→  1596→        // LDR W: scale=2, imm12 0..4095 → offset 0..16380, multiple of 4
  1597→  1597→        // LDR X: scale=3, imm12 0..4095 → offset 0..32760, multiple of 8
  1598→  1598→        match ty {
  1599→  1599→            IRType::I8 | IRType::U8 => {
  1600→  1600→                if offset >= 0 && offset <= 4095 {
  1601→  1601→                    self.emit_instruction(Instruction::LDRB { rt, rn, offset })?;
  1602→  1602→                } else {
  1603→  1603→                    self.emit_address_with_offset(rn, offset)?;
  1604→  1604→                    self.emit_instruction(Instruction::LDRB {
  1605→  1605→                        rt,
  1606→  1606→                        rn: Register::X9,
  1607→  1607→                        offset: 0,
  1608→  1608→                    })?;
  1609→  1609→                }
  1610→  1610→                // Sign-extend for signed byte loads (SXTB = SBFM Xd, Xn, #0, #7)
  1611→  1611→                if *ty == IRType::I8 {
  1612→  1612→                    self.emit_instruction(Instruction::SBFM {
  1613→  1613→                        rd: rt,
  1614→  1614→                        rn: rt,
  1615→  1615→                        immr: 0,
  1616→  1616→                        imms: 7,
  1617→  1617→                    })?;
  1618→  1618→                }
  1619→  1619→            }
  1620→  1620→            IRType::I16 | IRType::U16 => {
  1621→  1621→                if offset >= 0 && offset <= 8190 && offset % 2 == 0 {
  1622→  1622→                    self.emit_instruction(Instruction::LDRH { rt, rn, offset })?;
  1623→  1623→                } else {
  1624→  1624→                    self.emit_address_with_offset(rn, offset)?;
  1625→  1625→                    self.emit_instruction(Instruction::LDRH {
  1626→  1626→                        rt,
  1627→  1627→                        rn: Register::X9,
  1628→  1628→                        offset: 0,
  1629→  1629→                    })?;
  1630→  1630→                }
  1631→  1631→                // Sign-extend for signed halfword loads (SXTH = SBFM Xd, Xn, #0, #15)
  1632→  1632→                if *ty == IRType::I16 {
  1633→  1633→                    self.emit_instruction(Instruction::SBFM {
  1634→  1634→                        rd: rt,
  1635→  1635→                        rn: rt,
  1636→  1636→                        immr: 0,
  1637→  1637→                        imms: 15,
  1638→  1638→                    })?;
  1639→  1639→                }
  1640→  1640→            }
  1641→  1641→            IRType::I32 | IRType::U32 => {
  1642→  1642→                // 32-bit load uses LDR Wt encoding (scale=2, offset must be multiple of 4)
  1643→  1643→                if offset >= 0 && offset <= 16380 && offset % 4 == 0 {
  1644→  1644→                    self.emit_instruction(Instruction::LDR_W { rt, rn, offset })?;
  1645→  1645→                } else {
  1646→  1646→                    self.emit_address_with_offset(rn, offset)?;
  1647→  1647→                    self.emit_instruction(Instruction::LDR_W {
  1648→  1648→                        rt,
  1649→  1649→                        rn: Register::X9,
  1650→  1650→                        offset: 0,
  1651→  1651→                    })?;
  1652→  1652→                }
  1653→  1653→                // Sign-extend for signed word loads (SXTW = SBFM Xd, Xn, #0, #31)
  1654→  1654→                if *ty == IRType::I32 {
  1655→  1655→                    self.emit_instruction(Instruction::SBFM {
  1656→  1656→                        rd: rt,
  1657→  1657→                        rn: rt,
  1658→  1658→                        immr: 0,
  1659→  1659→                        imms: 31,
  1660→  1660→                    })?;
  1661→  1661→                }
  1662→  1662→            }
  1663→  1663→            IRType::I64 | IRType::U64 | IRType::Ptr | IRType::Func => {
  1664→  1664→                // 64-bit load uses LDR Xt encoding (scale=3, offset must be multiple of 8)
  1665→  1665→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1666→  1666→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1667→  1667→                } else {
  1668→  1668→                    self.emit_address_with_offset(rn, offset)?;
  1669→  1669→                    self.emit_instruction(Instruction::LDR {
  1670→  1670→                        rt,
  1671→  1671→                        rn: Register::X9,
  1672→  1672→                        offset: 0,
  1673→  1673→                    })?;
  1674→  1674→                }
  1675→  1675→            }
  1676→  1676→            _ => {
  1677→  1677→                // Default: treat as 64-bit load
  1678→  1678→                if offset >= 0 && offset <= 32760 && offset % 8 == 0 {
  1679→  1679→                    self.emit_instruction(Instruction::LDR { rt, rn, offset })?;
  1680→  1680→                } else {
  1681→  1681→                    self.emit_address_with_offset(rn, offset)?;
  1682→  1682→                    self.emit_instruction(Instruction::LDR {
  1683→  1683→                        rt,
  1684→  1684→                        rn: Register::X9,
  1685→  1685→                        offset: 0,
  1686→  1686→                    })?;
  1687→  1687→                }
  1688→  1688→            }
  1689→  1689→        }
  1690→  1690→        Ok(())
  1691→  1691→    }
  1692→  1692→
  1693→  1693→    /// Emit a store instruction with the given offset and IR type.
  1694→  1694→    ///
  1695→  1695→    /// Selects the correct ARM64 store variant based on the IR type:
  1696→  1696→    /// - I8/U8 → STRB (byte)
  1697→  1697→    /// - I16/U16 → STRH (halfword)
  1698→  1698→    /// - I32/U32 → STR_W (32-bit word)
  1699→  1699→    /// - I64/U64/Ptr/Func → STR (64-bit doubleword)
  1700→  1700→    ///
  1701→  1701→    /// If the offset fits the ARM64 unsigned-offset immediate encoding for the
  1702→  1702→    /// selected instruction variant, it is encoded directly. Otherwise, the
  1703→  1703→    /// effective address is computed in a scratch register (X9) first, and
  1704→  1704→    /// the store is performed at [X9 + 0].
  1705→  1705→    fn emit_store(&mut self, rt: Register, rn: Register, offset: i32, ty: &IRType) -> Result<()> {
  1706→  1706→        match ty {
  1707→  1707→            IRType::I8 | IRType::U8 => {
  1708→  1708→                if offset >= 0 && offset <= 4095 {
  1709→  1709→                    self.emit_instruction(Instruction::STRB { rt, rn, offset })?;
  1710→  1710→                } else {
  1711→  1711→                    self.emit_address_with_offset(rn, offset)?;
  1712→  1712→                    self.emit_instruction(Instruction::STRB {
  1713→  1713→                        rt,
  1714→  1714→                        rn: Register::X9,
  1715→  1715→                        offset: 0,
  1716→  1716→                    })?;
  1717→  1717→                }
  1718→  1718→            }
  1719→  1719→            IRType::I16 | IRType::U16 => {
  1720→  1720→                if offset >= 0 && offset <= 8190 && offset % 2 == 0 {
  1721→  1721→                    self.emit_instruction(Instruction::STRH { rt, rn, offset })?;
  1722→  1722→                } else {
  1723→  1723→                    self.emit_address_with_offset(rn, offset)?;
  1724→  1724→                    self.emit_instruction(Instruction::STRH {
  1725→  1725→                        rt,
  1726→  1726→                        rn: Register::X9,
  1727→  1727→                        offset: 0,
  1728→  1728→                    })?;
  1729→  1729→                }
  1730→  1730→            }
  1731→  1731→            IRType::I32 | IRType::U32 => {
  1732→  1732→                // 32-bit store uses STR Wt encoding (scale=2, offset must be multiple of 4)
  1733→  1733→                if offset >= 0 && offset <= 16380 && offset % 4 == 0 {
  1734→  1734→                    self.emit_instruction(Instruction::STR_W { rt, rn, offset })?;
  1735→  1735→                } else {
  1736→  1736→                    self.emit_address_with_offset(rn, offset)?;
  1737→  1737→                    self.emit_instruction(Instruction::STR_W {
  1738→  1738→                        rt,
  1739→  1739→                        rn: Register::X9,
  1740→  1740→                        offset: 0,
  1741→  1741→                    })?;
  1742→  1742→                }
  1743→  1743→            }
  1744→  1744→            IRType::I64 | IRType::U64 | IRType::Ptr | IRType::Func => {

  1808→  1808→        match val {

  1811→  1811→                let reg = result.reg;
  1812→  1812→

  1814→  1814→                // This prevents subsequent resolve_reg calls within the same
  1815→  1815→                // instruction from spilling this register.

  1822→  1822→                // value to the stack before it gets overwritten.

  1826→  1826→                    // negative offsets from X29: slot 0 at [X29, #-8],

  1830→  1830→                    // used by resolve_reg for immediate values).

  1833→  1833→                        rd: Register::X16,
  1834→  1834→                        rn: Register::X29,
  1835→  1835→                        rm: Operand::Reg {
  1836→  1836→                            reg: Register::X16,
  1837→  1837→                            shift: None,
  1838→  1838→                        },
  1839→  1839→                    })?;

  1841→  1841→                        rt: spill_info.reg,
  1842→  1842→                        rn: Register::X16,
  1843→  1843→                        offset: 0,
  1844→  1844→                    })?;
  1845→  1845→                }

  1863→  1863→                        rn: Register::X16,
  1864→  1864→                        offset: 0,
  1865→  1865→                    })?;
  1866→  1866→                }
  1867→  1867→

  1869→  1869→            }
