     1‚Üí     1‚Üí//! Profile-guided optimization engine for the Continuous Optimization Runtime.
     2‚Üí     2‚Üí//!
     3‚Üí     3‚Üí//! This module defines the [`OptimizationEngine`] and the [`OptimizationPass`]
     4‚Üí     4‚Üí//! trait, along with four concrete passes that apply profile-guided
     5‚Üí     5‚Üí//! transformations to the Semantic Computation Graph (SCG):
     6‚Üí     6‚Üí//!
     7‚Üí     7‚Üí//! - [`HotPathInlining`] ‚Äî inlines frequently-called function nodes.
     8‚Üí     8‚Üí//! - [`ColdPathOutline`] ‚Äî moves rarely-executed code to separate out-of-line
     9‚Üí     9‚Üí//!   functions, improving instruction-cache utilisation on hot paths.
    10‚Üí    10‚Üí//! - [`LoopOptimization`] ‚Äî unrolls hot loops and, where possible, emits
    11‚Üí    11‚Üí//!   SIMD/vectorized instructions.
    12‚Üí    12‚Üí//! - [`MemoryOptimization`] ‚Äî inserts prefetch hints and aligns data to
    13‚Üí    13‚Üí//!   cache-line boundaries, targeting the Raspberry Pi 5 L1/L2 hierarchy.
    14‚Üí    14‚Üí//!
    15‚Üí    15‚Üí//! The top-level function [`apply_optimizations`] runs all registered passes
    16‚Üí    16‚Üí//! over the SCG and returns an [`OptimizationResult`] summarising what was
    17‚Üí    17‚Üí//! done.
    18‚Üí    18‚Üí
    19‚Üí    19‚Üíuse crate::config::{Config, TargetArch};

    22‚Üí    22‚Üíuse std::collections::HashMap;

    26‚Üí    26‚Üí// ---------------------------------------------------------------------------
    27‚Üí    27‚Üí
    28‚Üí    28‚Üí/// A summarised profile report consumed by optimisation passes.
    29‚Üí    29‚Üí///
    30‚Üí    30‚Üí/// `ProfileReport` is constructed from the raw [`ProfileData`] and provides
    31‚Üí    31‚Üí/// pre-classified hot/cold node lists, loop back-edge frequencies, and
    32‚Üí    32‚Üí/// allocation hotspots so that each pass does not have to recompute these
    33‚Üí    33‚Üí/// from scratch.
    34‚Üí    34‚Üí#[derive(Debug, Clone)]
    35‚Üí    35‚Üípub struct ProfileReport {
    36‚Üí    36‚Üí    /// Nodes sorted by call count, descending (hottest first).
    37‚Üí    37‚Üí    pub hot_nodes: Vec<(NodeId, u64)>,
    38‚Üí    38‚Üí    /// Nodes whose call count is below the cold threshold.
    39‚Üí    39‚Üí    pub cold_nodes: Vec<(NodeId, u64)>,
    40‚Üí    40‚Üí    /// Loop back-edges and their traversal counts.
    41‚Üí    41‚Üí    /// Each entry is `(edge_id, source_node, target_node, weight)`.
    42‚Üí    42‚Üí    pub loop_back_edges: Vec<(EdgeId, NodeId, NodeId, u64)>,
    43‚Üí    43‚Üí    /// Allocation hotspots by region class.
    44‚Üí    44‚Üí    pub allocation_hotspots: Vec<(String, u64)>,
    45‚Üí    45‚Üí    /// Total number of profile samples recorded.
    46‚Üí    46‚Üí    pub total_samples: u64,
    47‚Üí    47‚Üí}
    48‚Üí    48‚Üí
    49‚Üí    49‚Üíimpl ProfileReport {
    50‚Üí    50‚Üí    /// Hotness threshold: nodes called more than this are "hot".
    51‚Üí    51‚Üí    const HOT_THRESHOLD: u64 = 100;
    52‚Üí    52‚Üí    /// Coldness threshold: nodes called fewer than this are "cold".
    53‚Üí    53‚Üí    const COLD_THRESHOLD: u64 = 5;
    54‚Üí    54‚Üí
    55‚Üí    55‚Üí    /// Builds a `ProfileReport` from raw [`ProfileData`].
    56‚Üí    56‚Üí    ///
    57‚Üí    57‚Üí    /// The constructor sorts nodes into hot/cold buckets and identifies loop
    58‚Üí    58‚Üí    /// back-edges (edges whose `weight` significantly exceeds the average
    59‚Üí    59‚Üí    /// edge weight in the SCG).
    60‚Üí    60‚Üí    pub fn from_profile_data(profile: &ProfileData, scg: &SCG) -> Self {
    61‚Üí    61‚Üí        let mut all_nodes: Vec<(NodeId, u64)> =
    62‚Üí    62‚Üí            profile.call_counts.iter().map(|(&n, &c)| (n, c)).collect();
    63‚Üí    63‚Üí        all_nodes.sort_by_key(|b| std::cmp::Reverse(b.1));
    64‚Üí    64‚Üí
    65‚Üí    65‚Üí        let hot_nodes: Vec<(NodeId, u64)> = all_nodes
    66‚Üí    66‚Üí            .iter()
    67‚Üí    67‚Üí            .filter(|(_, c)| *c > Self::HOT_THRESHOLD)
    68‚Üí    68‚Üí            .cloned()
    69‚Üí    69‚Üí            .collect();
    70‚Üí    70‚Üí
    71‚Üí    71‚Üí        let cold_nodes: Vec<(NodeId, u64)> = all_nodes
    72‚Üí    72‚Üí            .iter()
    73‚Üí    73‚Üí            .filter(|(_, c)| *c < Self::COLD_THRESHOLD)
    74‚Üí    74‚Üí            .cloned()
    75‚Üí    75‚Üí            .collect();
    76‚Üí    76‚Üí
    77‚Üí    77‚Üí        // Identify loop back-edges: edges where source > target (simple
    78‚Üí    78‚Üí        // heuristic) or where the weight is at least 10√ó the median.
    79‚Üí    79‚Üí        let avg_weight = if scg.edges.is_empty() {
    80‚Üí    80‚Üí            1
    81‚Üí    81‚Üí        } else {
    82‚Üí    82‚Üí            let total: u64 = scg.edges.values().map(|e| e.weight).sum();
    83‚Üí    83‚Üí            total / scg.edges.len().max(1) as u64
    84‚Üí    84‚Üí        };
    85‚Üí    85‚Üí
    86‚Üí    86‚Üí        let loop_back_edges: Vec<(EdgeId, NodeId, NodeId, u64)> = scg
    87‚Üí    87‚Üí            .edges
    88‚Üí    88‚Üí            .values()
    89‚Üí    89‚Üí            .filter(|e| e.weight > avg_weight * 10 || e.target <= e.source)
    90‚Üí    90‚Üí            .map(|e| (e.id, e.source, e.target, e.weight))
    91‚Üí    91‚Üí            .collect();
    92‚Üí    92‚Üí
    93‚Üí    93‚Üí        let allocation_hotspots: Vec<(String, u64)> = profile
    94‚Üí    94‚Üí            .allocation_stats
    95‚Üí    95‚Üí            .by_region_class
    96‚Üí    96‚Üí            .iter()
    97‚Üí    97‚Üí            .filter(|(_, count)| **count > 50)
    98‚Üí    98‚Üí            .map(|(k, &v)| (k.clone(), v))
    99‚Üí    99‚Üí            .collect();
   100‚Üí   100‚Üí
   101‚Üí   101‚Üí        let total_samples: u64 = profile.call_counts.values().sum();
   102‚Üí   102‚Üí
   103‚Üí   103‚Üí        ProfileReport {
   104‚Üí   104‚Üí            hot_nodes,
   105‚Üí   105‚Üí            cold_nodes,
   106‚Üí   106‚Üí            loop_back_edges,
   107‚Üí   107‚Üí            allocation_hotspots,
   108‚Üí   108‚Üí            total_samples,
   109‚Üí   109‚Üí        }
   110‚Üí   110‚Üí    }
   111‚Üí   111‚Üí
   112‚Üí   112‚Üí    /// Returns `true` if the given node is classified as hot.
   113‚Üí   113‚Üí    pub fn is_hot(&self, node_id: NodeId) -> bool {
   114‚Üí   114‚Üí        self.hot_nodes.iter().any(|(n, _)| *n == node_id)
   115‚Üí   115‚Üí    }
   116‚Üí   116‚Üí
   117‚Üí   117‚Üí    /// Returns `true` if the given node is classified as cold.
   118‚Üí   118‚Üí    pub fn is_cold(&self, node_id: NodeId) -> bool {
   119‚Üí   119‚Üí        self.cold_nodes.iter().any(|(n, _)| *n == node_id)
   120‚Üí   120‚Üí    }
   121‚Üí   121‚Üí
   122‚Üí   122‚Üí    /// Returns the call count for the given node, or 0 if not profiled.
   123‚Üí   123‚Üí    pub fn call_count(&self, node_id: NodeId) -> u64 {
   124‚Üí   124‚Üí        self.hot_nodes
   125‚Üí   125‚Üí            .iter()
   126‚Üí   126‚Üí            .chain(self.cold_nodes.iter())
   127‚Üí   127‚Üí            .find(|(n, _)| *n == node_id)
   128‚Üí   128‚Üí            .map(|(_, c)| *c)
   129‚Üí   129‚Üí            .unwrap_or(0)
   130‚Üí   130‚Üí    }
   131‚Üí   131‚Üí}

   242‚Üí   242‚Üí
   243‚Üí   243‚Üí        OptimizationResult {
   244‚Üí   244‚Üí            pass_results,

   320‚Üí   320‚Üíimpl OptimizationPass for HotPathInlining {
   321‚Üí   321‚Üí    fn name(&self) -> &str {
   322‚Üí   322‚Üí        "HotPathInlining"
   323‚Üí   323‚Üí    }
   324‚Üí   324‚Üí
   325‚Üí   325‚Üí    fn apply(&self, scg: &mut SCG, profile: &ProfileReport) -> PassResult {
   326‚Üí   326‚Üí        let mut result = PassResult::empty(self.name());
   327‚Üí   327‚Üí
   328‚Üí   328‚Üí        for &(node_id, call_count) in &profile.hot_nodes {
   329‚Üí   329‚Üí            let node = match scg.get_node_mut(node_id) {
   330‚Üí   330‚Üí                Some(n) => n,
   331‚Üí   331‚Üí                None => continue,
   332‚Üí   332‚Üí            };
   333‚Üí   333‚Üí
   334‚Üí   334‚Üí            // Only inline Call nodes that haven't already been inlined.
   335‚Üí   335‚Üí            if node.kind != NodeKind::Call || node.is_inlined {
   336‚Üí   336‚Üí                continue;
   337‚Üí   337‚Üí            }
   338‚Üí   338‚Üí
   339‚Üí   339‚Üí            // Only inline if the callee is small enough.
   340‚Üí   340‚Üí            if node.code_size > self.max_inline_size {
   341‚Üí   341‚Üí                log::debug!(
   342‚Üí   342‚Üí                    "HotPathInlining: skipping node {} ‚Äî code_size {} exceeds limit {}",
   343‚Üí   343‚Üí                    node_id,
   344‚Üí   344‚Üí                    node.code_size,
   345‚Üí   345‚Üí                    self.max_inline_size,
   346‚Üí   346‚Üí                );
   347‚Üí   347‚Üí                continue;
   348‚Üí   348‚Üí            }
   349‚Üí   349‚Üí
   350‚Üí   350‚Üí            let saved_call_overhead = call_count as f64 * 5.0; // ~5 cycles per call
   351‚Üí   351‚Üí            result.estimated_speedup += saved_call_overhead / profile.total_samples.max(1) as f64;
   352‚Üí   352‚Üí
   353‚Üí   353‚Üí            node.is_inlined = true;
   354‚Üí   354‚Üí            result.transformations.push(Transformation {
   355‚Üí   355‚Üí                kind: TransformationKind::Inlined,
   356‚Üí   356‚Üí                target_node: node_id,
   357‚Üí   357‚Üí                description: format!(
   358‚Üí   358‚Üí                    "Inlined hot call node (called {}√ó, code_size={}B)",
   359‚Üí   359‚Üí                    call_count, node.code_size,
   360‚Üí   360‚Üí                ),
   361‚Üí   361‚Üí            });
   362‚Üí   362‚Üí        }
   363‚Üí   363‚Üí
   364‚Üí   364‚Üí        // Ensure speedup is at least 1.0 (no regression).
   365‚Üí   365‚Üí        result.estimated_speedup = result.estimated_speedup.max(1.0);
   366‚Üí   366‚Üí
   367‚Üí   367‚Üí        log::info!(
   368‚Üí   368‚Üí            "HotPathInlining: {} transformations, estimated speedup {:.3}√ó",
   369‚Üí   369‚Üí            result.count(),
   370‚Üí   370‚Üí            result.estimated_speedup,
   371‚Üí   371‚Üí        );
   372‚Üí   372‚Üí
   373‚Üí   373‚Üí        result
   374‚Üí   374‚Üí    }
   375‚Üí   375‚Üí}

   381‚Üí   381‚Üí/// Moves cold code to separate out-of-line functions.
   382‚Üí   382‚Üí///
   383‚Üí   383‚Üí/// When a node is classified as cold by the profile report and resides on

   393‚Üí   393‚Üí    /// Creates a new `ColdPathOutline` pass with default settings.

   412‚Üí   412‚Üí    fn apply(&self, scg: &mut SCG, profile: &ProfileReport) -> PassResult {
   413‚Üí   413‚Üí        let mut result = PassResult::empty(self.name());
   414‚Üí   414‚Üí
   415‚Üí   415‚Üí        // Collect node IDs to outline (we cannot borrow scg mutably while
   416‚Üí   416‚Üí        // iterating over profile.cold_nodes which may reference nodes not
   417‚Üí   417‚Üí        // in the graph, so we collect first and mutate second).
   418‚Üí   418‚Üí        let mut to_outline: Vec<(NodeId, String)> = Vec::new();
   419‚Üí   419‚Üí
   420‚Üí   420‚Üí        for &(node_id, call_count) in &profile.cold_nodes {
   421‚Üí   421‚Üí            let node = match scg.get_node(node_id) {
   422‚Üí   422‚Üí                Some(n) => n,
   423‚Üí   423‚Üí                None => continue,
   424‚Üí   424‚Üí            };
   425‚Üí   425‚Üí
   426‚Üí   426‚Üí            // Only outline nodes that are on the same path as hot nodes.
   427‚Üí   427‚Üí            let adjacent_to_hot = if self.outline_adjacent_to_hot {
   428‚Üí   428‚Üí                node.incoming_edges
   429‚Üí   429‚Üí                    .iter()
   430‚Üí   430‚Üí                    .chain(node.outgoing_edges.iter())
   431‚Üí   431‚Üí                    .any(|&eid| {
   432‚Üí   432‚Üí                        scg.get_edge(eid)
   433‚Üí   433‚Üí                            .is_some_and(|e| profile.is_hot(e.source) || profile.is_hot(e.target))
   434‚Üí   434‚Üí                    })
   435‚Üí   435‚Üí            } else {
   436‚Üí   436‚Üí                false
   437‚Üí   437‚Üí            };
   438‚Üí   438‚Üí
   439‚Üí   439‚Üí            if !adjacent_to_hot && !profile.is_hot(node_id) {
   440‚Üí   440‚Üí                // Cold node not adjacent to any hot node ‚Äî less benefit.
   441‚Üí   441‚Üí                continue;
   442‚Üí   442‚Üí            }
   443‚Üí   443‚Üí
   444‚Üí   444‚Üí            to_outline.push((
   445‚Üí   445‚Üí                node_id,
   446‚Üí   446‚Üí                format!(
   447‚Üí   447‚Üí                    "Outlined cold node (called {}√ó, adjacent to hot path)",
   448‚Üí   448‚Üí                    call_count,
   449‚Üí   449‚Üí                ),
   450‚Üí   450‚Üí            ));
   451‚Üí   451‚Üí        }
   452‚Üí   452‚Üí
   453‚Üí   453‚Üí        // Also outline Branch nodes on cold paths adjacent to hot nodes.
   454‚Üí   454‚Üí        if self.outline_adjacent_to_hot {
   455‚Üí   455‚Üí            for &node_id in scg.nodes.keys() {
   456‚Üí   456‚Üí                let node = match scg.get_node(node_id) {
   457‚Üí   457‚Üí                    Some(n) => n,
   458‚Üí   458‚Üí                    None => continue,
   459‚Üí   459‚Üí                };
   460‚Üí   460‚Üí                if node.kind != NodeKind::Branch || node.is_outlined {
   461‚Üí   461‚Üí                    continue;
   462‚Üí   462‚Üí                }
   463‚Üí   463‚Üí                // Check if this branch is adjacent to a hot node but is itself cold.
   464‚Üí   464‚Üí                let adjacent_hot = node
   465‚Üí   465‚Üí                    .incoming_edges
   466‚Üí   466‚Üí                    .iter()
   467‚Üí   467‚Üí                    .chain(node.outgoing_edges.iter())
   468‚Üí   468‚Üí                    .any(|&eid| {
   469‚Üí   469‚Üí                        scg.get_edge(eid)
   470‚Üí   470‚Üí                            .is_some_and(|e| profile.is_hot(e.source) || profile.is_hot(e.target))
   471‚Üí   471‚Üí                    });
   472‚Üí   472‚Üí                if adjacent_hot && profile.is_cold(node_id) {
   473‚Üí   473‚Üí                    to_outline.push((
   474‚Üí   474‚Üí                        node_id,
   475‚Üí   475‚Üí                        "Outlined cold branch adjacent to hot path".to_owned(),
   476‚Üí   476‚Üí                    ));
   477‚Üí   477‚Üí                }
   478‚Üí   478‚Üí            }
   479‚Üí   479‚Üí        }
   480‚Üí   480‚Üí
   481‚Üí   481‚Üí        for (node_id, description) in to_outline {
   482‚Üí   482‚Üí            if let Some(node) = scg.get_node_mut(node_id) {
   483‚Üí   483‚Üí                node.is_outlined = true;
   484‚Üí   484‚Üí                result.transformations.push(Transformation {
   485‚Üí   485‚Üí                    kind: TransformationKind::Outlined,
   486‚Üí   486‚Üí                    target_node: node_id,
   487‚Üí   487‚Üí                    description,
   488‚Üí   488‚Üí                });
   489‚Üí   489‚Üí            }
   490‚Üí   490‚Üí        }
   491‚Üí   491‚Üí
   492‚Üí   492‚Üí        // Estimated speedup: each outlined node frees icache space.
   493‚Üí   493‚Üí        // Rough model: 2% per outlined node, capped at 20%.
   494‚Üí   494‚Üí        let outline_benefit = (result.count() as f64 * 0.02).min(0.20);
   495‚Üí   495‚Üí        result.estimated_speedup = 1.0 + outline_benefit;
   496‚Üí   496‚Üí
   497‚Üí   497‚Üí        log::info!(
   498‚Üí   498‚Üí            "ColdPathOutline: {} transformations, estimated speedup {:.3}√ó",
   499‚Üí   499‚Üí            result.count(),
   500‚Üí   500‚Üí            result.estimated_speedup,
   501‚Üí   501‚Üí        );
   502‚Üí   502‚Üí
   503‚Üí   503‚Üí        result
   504‚Üí   504‚Üí    }
   505‚Üí   505‚Üí}

   507‚Üí   507‚Üí// ---------------------------------------------------------------------------
   508‚Üí   508‚Üí// LoopOptimization
   509‚Üí   509‚Üí// ---------------------------------------------------------------------------
   510‚Üí   510‚Üí
   511‚Üí   511‚Üí/// Unrolls hot loops and, where profitable, vectorises them.
   512‚Üí   512‚Üí///
   513‚Üí   513‚Üí/// Loop unrolling reduces branch overhead and exposes instruction-level
   514‚Üí   514‚Üí/// parallelism. Vectorization replaces scalar loop bodies with SIMD
   515‚Üí   515‚Üí/// operations, which is especially effective on the Cortex-A76 (Pi 5)
   516‚Üí   516‚Üí/// with its 128-bit NEON unit.
   517‚Üí   517‚Üípub struct LoopOptimization {
   518‚Üí   518‚Üí    /// Minimum trip count (from profile) to consider a loop "hot" enough
   519‚Üí   519‚Üí    /// for unrolling.
   520‚Üí   520‚Üí    pub min_trip_count: u64,
   521‚Üí   521‚Üí    /// Maximum unroll factor to apply.
   522‚Üí   522‚Üí    pub max_unroll_factor: u32,
   523‚Üí   523‚Üí    /// Whether to attempt vectorisation.
   524‚Üí   524‚Üí    pub enable_vectorization: bool,
   525‚Üí   525‚Üí}
   526‚Üí   526‚Üí
   527‚Üí   527‚Üíimpl LoopOptimization {
   528‚Üí   528‚Üí    /// Creates a new `LoopOptimization` pass with default settings.
   529‚Üí   529‚Üí    pub fn new() -> Self {
   530‚Üí   530‚Üí        LoopOptimization {
   531‚Üí   531‚Üí            min_trip_count: 100,
   532‚Üí   532‚Üí            max_unroll_factor: 8,
   533‚Üí   533‚Üí            enable_vectorization: true,
   534‚Üí   534‚Üí        }
   535‚Üí   535‚Üí    }
   536‚Üí   536‚Üí
   537‚Üí   537‚Üí    /// Creates a pass with a custom minimum trip count.
   538‚Üí   538‚Üí    pub fn with_min_trip_count(mut self, count: u64) -> Self {
   539‚Üí   539‚Üí        self.min_trip_count = count;
   540‚Üí   540‚Üí        self
   541‚Üí   541‚Üí    }
   542‚Üí   542‚Üí
   543‚Üí   543‚Üí    /// Determines the best unroll factor for a loop with the given trip
   544‚Üí   544‚Üí    /// count. Chooses the largest power-of-two ‚â§ `max_unroll_factor` that
   545‚Üí   545‚Üí    /// does not exceed the trip count.
   546‚Üí   546‚Üí    fn best_unroll_factor(&self, trip_count: u64) -> u32 {
   547‚Üí   547‚Üí        let mut factor: u32 = 1;
   548‚Üí   548‚Üí        let mut candidate = 2;
   549‚Üí   549‚Üí        while candidate <= self.max_unroll_factor && (candidate as u64) <= trip_count {
   550‚Üí   550‚Üí            factor = candidate;
   551‚Üí   551‚Üí            candidate *= 2;
   552‚Üí   552‚Üí        }
   553‚Üí   553‚Üí        factor
   554‚Üí   554‚Üí    }
   555‚Üí   555‚Üí
   556‚Üí   556‚Üí    /// Heuristic: determines whether a loop body is vectorizable.
   557‚Üí   557‚Üí    ///
   558‚Üí   558‚Üí    /// In the full implementation this would analyse data dependencies,
   559‚Üí   559‚Üí    /// stride patterns, and type widths. For now we use a simple heuristic:
   560‚Üí   560‚Üí    /// a loop is vectorizable if it has a Memory node among its successors
   561‚Üí   561‚Üí    /// and is not already vectorized.
   562‚Üí   562‚Üí    fn is_vectorizable(&self, scg: &SCG, loop_node_id: NodeId) -> bool {
   563‚Üí   563‚Üí        let loop_node = match scg.get_node(loop_node_id) {
   564‚Üí   564‚Üí            Some(n) => n,
   565‚Üí   565‚Üí            None => return false,
   566‚Üí   566‚Üí        };
   567‚Üí   567‚Üí        if loop_node.is_vectorized {
   568‚Üí   568‚Üí            return false;
   569‚Üí   569‚Üí        }
   570‚Üí   570‚Üí        // Check if any successor reachable through outgoing edges is a
   571‚Üí   571‚Üí        // Memory node (indicating a load/store pattern).
   572‚Üí   572‚Üí        let mut has_memory = false;
   573‚Üí   573‚Üí        for &eid in &loop_node.outgoing_edges {
   574‚Üí   574‚Üí            if let Some(edge) = scg.get_edge(eid) {
   575‚Üí   575‚Üí                if let Some(target) = scg.get_node(edge.target) {
   576‚Üí   576‚Üí                    if target.kind == NodeKind::Memory {
   577‚Üí   577‚Üí                        has_memory = true;
   578‚Üí   578‚Üí                        break;
   579‚Üí   579‚Üí                    }
   580‚Üí   580‚Üí                }
   581‚Üí   581‚Üí            }
   582‚Üí   582‚Üí        }
   583‚Üí   583‚Üí        has_memory
   584‚Üí   584‚Üí    }
   585‚Üí   585‚Üí}

   590‚Üí   590‚Üí    }

   592‚Üí   592‚Üí
   593‚Üí   593‚Üíimpl OptimizationPass for LoopOptimization {
   594‚Üí   594‚Üí    fn name(&self) -> &str {
   595‚Üí   595‚Üí        "LoopOptimization"
   596‚Üí   596‚Üí    }
   597‚Üí   597‚Üí
   598‚Üí   598‚Üí    fn apply(&self, scg: &mut SCG, profile: &ProfileReport) -> PassResult {
   599‚Üí   599‚Üí        let mut result = PassResult::empty(self.name());
   600‚Üí   600‚Üí
   601‚Üí   601‚Üí        // Build a map from loop-header node ID to estimated trip count
   602‚Üí   602‚Üí        // derived from back-edge weights.
   603‚Üí   603‚Üí        let mut loop_trip_counts: HashMap<NodeId, u64> = HashMap::new();
   604‚Üí   604‚Üí        for &(_eid, _src, target, weight) in &profile.loop_back_edges {
   605‚Üí   605‚Üí            let entry = loop_trip_counts.entry(target).or_insert(0);
   606‚Üí   606‚Üí            *entry = (*entry).max(weight);
   607‚Üí   607‚Üí        }
   608‚Üí   608‚Üí
   609‚Üí   609‚Üí        // Also consider Loop-kind nodes that are hot in the profile.
   610‚Üí   610‚Üí        for &(node_id, call_count) in &profile.hot_nodes {
   611‚Üí   611‚Üí            let node = match scg.get_node(node_id) {
   612‚Üí   612‚Üí                Some(n) => n,
   613‚Üí   613‚Üí                None => continue,
   614‚Üí   614‚Üí            };
   615‚Üí   615‚Üí            if node.kind == NodeKind::Loop || node.kind == NodeKind::LoopHeader {
   616‚Üí   616‚Üí                let entry = loop_trip_counts.entry(node_id).or_insert(0);
   617‚Üí   617‚Üí                *entry = (*entry).max(call_count);
   618‚Üí   618‚Üí            }
   619‚Üí   619‚Üí        }
   620‚Üí   620‚Üí
   621‚Üí   621‚Üí        // Apply unrolling and vectorisation.
   622‚Üí   622‚Üí        let loop_nodes: Vec<NodeId> = loop_trip_counts.keys().cloned().collect();
   623‚Üí   623‚Üí        for loop_node_id in loop_nodes {
   624‚Üí   624‚Üí            let trip_count = loop_trip_counts[&loop_node_id];
   625‚Üí   625‚Üí            if trip_count < self.min_trip_count {
   626‚Üí   626‚Üí                continue;
   627‚Üí   627‚Üí            }
   628‚Üí   628‚Üí
   629‚Üí   629‚Üí            let node = match scg.get_node_mut(loop_node_id) {
   630‚Üí   630‚Üí                Some(n) => n,
   631‚Üí   631‚Üí                None => continue,
   632‚Üí   632‚Üí            };
   633‚Üí   633‚Üí
   634‚Üí   634‚Üí            if node.kind != NodeKind::Loop && node.kind != NodeKind::LoopHeader {
   635‚Üí   635‚Üí                continue;
   636‚Üí   636‚Üí            }
   637‚Üí   637‚Üí
   638‚Üí   638‚Üí            // Unroll.
   639‚Üí   639‚Üí            let factor = self.best_unroll_factor(trip_count);
   640‚Üí   640‚Üí            if factor > 1 && node.unroll_factor < factor {
   641‚Üí   641‚Üí                let old_factor = node.unroll_factor;
   642‚Üí   642‚Üí                node.unroll_factor = factor;
   643‚Üí   643‚Üí                result.transformations.push(Transformation {
   644‚Üí   644‚Üí                    kind: TransformationKind::LoopUnrolled { factor },
   645‚Üí   645‚Üí                    target_node: loop_node_id,
   646‚Üí   646‚Üí                    description: format!(
   647‚Üí   647‚Üí                        "Unrolled loop (trip={}, factor {}‚Üí{})",
   648‚Üí   648‚Üí                        trip_count, old_factor, factor,
   649‚Üí   649‚Üí                    ),
   650‚Üí   650‚Üí                });
   651‚Üí   651‚Üí            }
   652‚Üí   652‚Üí
   653‚Üí   653‚Üí            // Vectorise (requires mutable borrow again, but we already
   654‚Üí   654‚Üí            // released the first one ‚Äî we need to reborrow).
   655‚Üí   655‚Üí            if self.enable_vectorization && self.is_vectorizable(scg, loop_node_id) {
   656‚Üí   656‚Üí                let node = scg.get_node_mut(loop_node_id).unwrap();
   657‚Üí   657‚Üí                if !node.is_vectorized {
   658‚Üí   658‚Üí                    node.is_vectorized = true;
   659‚Üí   659‚Üí                    result.transformations.push(Transformation {
   660‚Üí   660‚Üí                        kind: TransformationKind::LoopVectorized,
   661‚Üí   661‚Üí                        target_node: loop_node_id,
   662‚Üí   662‚Üí                        description: format!("Vectorized loop (trip={}, NEON/SIMD)", trip_count,),
   663‚Üí   663‚Üí                    });
   664‚Üí   664‚Üí                }
   665‚Üí   665‚Üí            }
   666‚Üí   666‚Üí        }
   667‚Üí   667‚Üí
   668‚Üí   668‚Üí        // Speedup model: unrolling gives ~factor/2 improvement up to 2√ó;
   669‚Üí   669‚Üí        // vectorisation gives ~4√ó for memory-bound loops.
   670‚Üí   670‚Üí        let unroll_speedup: f64 = result
   671‚Üí   671‚Üí            .transformations
   672‚Üí   672‚Üí            .iter()
   673‚Üí   673‚Üí            .filter(|t| matches!(t.kind, TransformationKind::LoopUnrolled { .. }))
   674‚Üí   674‚Üí            .map(|t| {
   675‚Üí   675‚Üí                if let TransformationKind::LoopUnrolled { factor } = t.kind {
   676‚Üí   676‚Üí                    (factor as f64 / 2.0).min(2.0)
   677‚Üí   677‚Üí                } else {
   678‚Üí   678‚Üí                    1.0
   679‚Üí   679‚Üí                }
   680‚Üí   680‚Üí            })
   681‚Üí   681‚Üí            .product::<f64>()
   682‚Üí   682‚Üí            .max(1.0);
   683‚Üí   683‚Üí
   684‚Üí   684‚Üí        let vec_count = result
   685‚Üí   685‚Üí            .transformations
   686‚Üí   686‚Üí            .iter()
   687‚Üí   687‚Üí            .filter(|t| matches!(t.kind, TransformationKind::LoopVectorized))
   688‚Üí   688‚Üí            .count();
   689‚Üí   689‚Üí
   690‚Üí   690‚Üí        // Each vectorised loop adds ~4√ó speedup for that loop body.
   691‚Üí   691‚Üí        let vec_speedup = if vec_count > 0 {
   692‚Üí   692‚Üí            1.0 + vec_count as f64 * 0.3
   693‚Üí   693‚Üí        } else {
   694‚Üí   694‚Üí            1.0
   695‚Üí   695‚Üí        };
   696‚Üí   696‚Üí
   697‚Üí   697‚Üí        result.estimated_speedup = (unroll_speedup * vec_speedup).max(1.0);
   698‚Üí   698‚Üí
   699‚Üí   699‚Üí        log::info!(
   700‚Üí   700‚Üí            "LoopOptimization: {} transformations, estimated speedup {:.3}√ó",
   701‚Üí   701‚Üí            result.count(),
   702‚Üí   702‚Üí            result.estimated_speedup,
   703‚Üí   703‚Üí        );
   704‚Üí   704‚Üí
   705‚Üí   705‚Üí        result
   706‚Üí   706‚Üí    }
   707‚Üí   707‚Üí}

   712‚Üí   712‚Üí
   713‚Üí   713‚Üí/// Inserts prefetch hints and aligns data to cache-line boundaries.
   714‚Üí   714‚Üí///
   715‚Üí   715‚Üí/// On the Raspberry Pi 5 (BCM2712, Cortex-A76):
   716‚Üí   716‚Üí///
   717‚Üí   717‚Üí/// - **L1 data cache**: 64 KB per core, 64-byte cache lines, 4-way set
   718‚Üí   718‚Üí///   associative.
   719‚Üí   719‚Üí/// - **L2 cache**: 512 KB shared (per core pair), 64-byte cache lines.
   720‚Üí   720‚Üí///
   721‚Üí   721‚Üí/// This pass:
   722‚Üí   722‚Üí///
   723‚Üí   723‚Üí/// 1. Marks [`Memory`](NodeKind::Memory) nodes on hot paths with prefetch
   724‚Üí   724‚Üí///    instructions so data is loaded into L1 before it is needed.
   725‚Üí   725‚Üí/// 2. Aligns memory accesses to 64-byte boundaries to avoid cross-cache-line
   726‚Üí   726‚Üí///    loads that waste bandwidth.
   727‚Üí   727‚Üípub struct MemoryOptimization {
   728‚Üí   728‚Üí    /// Cache-line size in bytes. Default: 64 (Pi 5 / Cortex-A76).
   729‚Üí   729‚Üí    pub cache_line_size: usize,
   730‚Üí   730‚Üí    /// L1 data cache size in bytes. Default: 65_536 (64 KB).
   731‚Üí   731‚Üí    pub l1d_size: usize,
   732‚Üí   732‚Üí    /// L2 cache size in bytes. Default: 524_288 (512 KB).
   733‚Üí   733‚Üí    pub l2_size: usize,
   734‚Üí   734‚Üí    /// Whether to target the Pi 5 specifically.
   735‚Üí   735‚Üí    pub target_pi5: bool,
   736‚Üí   736‚Üí}
   737‚Üí   737‚Üí
   738‚Üí   738‚Üíimpl MemoryOptimization {
   739‚Üí   739‚Üí    /// Creates a new `MemoryOptimization` pass targeting the given architecture.
   740‚Üí   740‚Üí    pub fn new(target_arch: TargetArch) -> Self {

   743‚Üí   743‚Üí            cache_line_size: 64,

   745‚Üí   745‚Üí            l2_size: 524_288,

   763‚Üí   763‚Üíimpl OptimizationPass for MemoryOptimization {
   764‚Üí   764‚Üí    fn name(&self) -> &str {
   765‚Üí   765‚Üí        "MemoryOptimization"
   766‚Üí   766‚Üí    }
   767‚Üí   767‚Üí
   768‚Üí   768‚Üí    fn apply(&self, scg: &mut SCG, profile: &ProfileReport) -> PassResult {
   769‚Üí   769‚Üí        let mut result = PassResult::empty(self.name());
   770‚Üí   770‚Üí
   771‚Üí   771‚Üí        // Collect (node_id, is_hot) for Memory nodes.
   772‚Üí   772‚Üí        let memory_nodes: Vec<(NodeId, bool)> = scg
   773‚Üí   773‚Üí            .nodes
   774‚Üí   774‚Üí            .iter()
   775‚Üí   775‚Üí            .filter(|(_, n)| n.kind == NodeKind::Memory)
   776‚Üí   776‚Üí            .map(|(&id, _)| (id, profile.is_hot(id)))
   777‚Üí   777‚Üí            .collect();
   778‚Üí   778‚Üí
   779‚Üí   779‚Üí        for (node_id, is_hot) in memory_nodes {
   780‚Üí   780‚Üí            let node = match scg.get_node_mut(node_id) {
   781‚Üí   781‚Üí                Some(n) => n,
   782‚Üí   782‚Üí                None => continue,
   783‚Üí   783‚Üí            };
   784‚Üí   784‚Üí
   785‚Üí   785‚Üí            // Insert prefetch for hot memory accesses.
   786‚Üí   786‚Üí            if is_hot && !node.has_prefetch {
   787‚Üí   787‚Üí                node.has_prefetch = true;
   788‚Üí   788‚Üí                result.transformations.push(Transformation {
   789‚Üí   789‚Üí                    kind: TransformationKind::PrefetchInserted,
   790‚Üí   790‚Üí                    target_node: node_id,
   791‚Üí   791‚Üí                    description: format!(
   792‚Üí   792‚Üí                        "Prefetch inserted for hot memory node (cache-line {}B, L1D {}KB)",
   793‚Üí   793‚Üí                        self.cache_line_size,
   794‚Üí   794‚Üí                        self.l1d_size / 1024,
   795‚Üí   795‚Üí                    ),
   796‚Üí   796‚Üí                });
   797‚Üí   797‚Üí            }
   798‚Üí   798‚Üí
   799‚Üí   799‚Üí            // Align to cache-line boundary on Pi 5 or for hot paths.
   800‚Üí   800‚Üí            if (self.target_pi5 || is_hot) && node.alignment < self.cache_line_size {
   801‚Üí   801‚Üí                let old_alignment = node.alignment;
   802‚Üí   802‚Üí                node.alignment = self.cache_line_size;
   803‚Üí   803‚Üí                result.transformations.push(Transformation {
   804‚Üí   804‚Üí                    kind: TransformationKind::CacheLineAligned {
   805‚Üí   805‚Üí                        alignment: self.cache_line_size,
   806‚Üí   806‚Üí                    },
   807‚Üí   807‚Üí                    target_node: node_id,
   808‚Üí   808‚Üí                    description: format!(
   809‚Üí   809‚Üí                        "Aligned memory node {}‚Üí{}B (Pi 5 L1/L2 cache-line)",
   810‚Üí   810‚Üí                        old_alignment, self.cache_line_size,
   811‚Üí   811‚Üí                    ),
   812‚Üí   812‚Üí                });
   813‚Üí   813‚Üí            }
   814‚Üí   814‚Üí        }
   815‚Üí   815‚Üí
   816‚Üí   816‚Üí        // Speedup model: prefetch reduces latency ~20% per access;
   817‚Üí   817‚Üí        // alignment avoids cross-line penalties ~10%.
   818‚Üí   818‚Üí        let prefetch_count = result
   819‚Üí   819‚Üí            .transformations
   820‚Üí   820‚Üí            .iter()
   821‚Üí   821‚Üí            .filter(|t| matches!(t.kind, TransformationKind::PrefetchInserted))
   822‚Üí   822‚Üí            .count();
   823‚Üí   823‚Üí        let align_count = result
   824‚Üí   824‚Üí            .transformations
   825‚Üí   825‚Üí            .iter()
   826‚Üí   826‚Üí            .filter(|t| matches!(t.kind, TransformationKind::CacheLineAligned { .. }))
   827‚Üí   827‚Üí            .count();
   828‚Üí   828‚Üí
   829‚Üí   829‚Üí        let prefetch_benefit = (prefetch_count as f64 * 0.05).min(0.30);
   830‚Üí   830‚Üí        let align_benefit = (align_count as f64 * 0.02).min(0.15);
   831‚Üí   831‚Üí        result.estimated_speedup = 1.0 + prefetch_benefit + align_benefit;
   832‚Üí   832‚Üí
   833‚Üí   833‚Üí        log::info!(
   834‚Üí   834‚Üí            "MemoryOptimization: {} transformations, estimated speedup {:.3}√ó",
   835‚Üí   835‚Üí            result.count(),
   836‚Üí   836‚Üí            result.estimated_speedup,
   837‚Üí   837‚Üí        );
   838‚Üí   838‚Üí
   839‚Üí   839‚Üí        result
   840‚Üí   840‚Üí    }
   841‚Üí   841‚Üí}
